171 bool DFTEngine::Evaluate(Orbitals& orbitals) {
172  // set the parallelization
173 #ifdef _OPENMP
174  omp_set_num_threads(_openmp_threads);
175 #endif
176 
177  Eigen::VectorXd& MOEnergies = orbitals.MOEnergies();
178  Eigen::MatrixXd& MOCoeff = orbitals.MOCoefficients();
179  if (MOEnergies.size() != _dftbasis.AOBasisSize()) {
180  MOEnergies.resize(_dftbasis.AOBasisSize());
181  }
182  if (MOCoeff.rows() != _dftbasis.AOBasisSize() ||
183  MOCoeff.cols() != _dftbasis.AOBasisSize()) {
184  MOCoeff.conservativeResize(_dftbasis.AOBasisSize(),
185  _dftbasis.AOBasisSize());
186  }
187 
188  Eigen::MatrixXd H0 = _dftAOkinetic.Matrix() + _dftAOESP.getNuclearpotential();
189  CTP_LOG(ctp::logDEBUG, *_pLog)
190  << ctp::TimeStamp() << " Constructed independent particle hamiltonian "
191  << flush;
192  NuclearRepulsion();
193  if (_with_ecp) {
194  H0 += _dftAOECP.Matrix();
195  }
196  if (_addexternalsites) {
197  H0 += _dftAOESP.getExternalpotential();
198  H0 += _dftAODipole_Potential.getExternalpotential();
199  H0 += _dftAOQuadrupole_Potential.getExternalpotential();
200  double estat = ExternalRepulsion();
201  CTP_LOG(ctp::logDEBUG, *_pLog)
202  << ctp::TimeStamp() << " E_electrostatic " << estat << flush;
203  _E_nucnuc += estat;
204  }
205 
206  if (_integrate_ext_density) {
207  Orbitals extdensity;
208  extdensity.ReadFromCpt(_orbfilename);
209  H0 += IntegrateExternalDensity(extdensity);
210  }
211 
212  if (_do_externalfield) {
213  CTP_LOG(ctp::logDEBUG, *_pLog)
214  << ctp::TimeStamp() << " Integrated external potential on grid "
215  << flush;
216  double externalgrid_nucint = ExternalGridRepulsion(_externalgrid_nuc);
217  CTP_LOG(ctp::logDEBUG, *_pLog)
218  << ctp::TimeStamp() << " Nuclei external potential interaction "
219  << externalgrid_nucint << " Hartree" << flush;
220  H0 += _gridIntegration_ext.IntegrateExternalPotential(_externalgrid);
221  _E_nucnuc += externalgrid_nucint;
222  }
223 
224  CTP_LOG(ctp::logDEBUG, *_pLog)
225  << ctp::TimeStamp() << " Nuclear Repulsion Energy is " << _E_nucnuc
226  << flush;
227 
228  if (_with_guess) {
229  CTP_LOG(ctp::logDEBUG, *_pLog)
230  << ctp::TimeStamp() << " Reading guess from orbitals object/file"
231  << flush;
232  orbitals.MOCoefficients() = OrthogonalizeGuess(orbitals.MOCoefficients());
233  _dftAOdmat = _conv_accelerator.DensityMatrix(MOCoeff, MOEnergies);
234  } else {
235  CTP_LOG(ctp::logDEBUG, *_pLog)
236  << ctp::TimeStamp() << " Setup Initial Guess using: " << _initial_guess
237  << flush;
238  if (_initial_guess == "independent") {
239  _conv_accelerator.SolveFockmatrix(MOEnergies, MOCoeff, H0);
240  _dftAOdmat = _conv_accelerator.DensityMatrix(MOCoeff, MOEnergies);
241 
242  } else if (_initial_guess == "atom") {
243  _dftAOdmat = AtomicGuess(orbitals);
244  CalculateERIs(_dftbasis, _dftAOdmat);
245 
246  if (_use_small_grid) {
247  orbitals.AOVxc() = _gridIntegration_small.IntegrateVXC(_dftAOdmat);
248  CTP_LOG(ctp::logDEBUG, *_pLog)
249  << ctp::TimeStamp() << " Filled approximate DFT Vxc matrix "
250  << flush;
251  } else {
252  orbitals.AOVxc() = _gridIntegration.IntegrateVXC(_dftAOdmat);
253  CTP_LOG(ctp::logDEBUG, *_pLog)
254  << ctp::TimeStamp() << " Filled DFT Vxc matrix " << flush;
255  }
256  Eigen::MatrixXd H = H0 + _ERIs.getERIs() + orbitals.AOVxc();
257  if (_ScaHFX > 0) {
258  if (_with_RI) {
259  _ERIs.CalculateEXX(_dftAOdmat);
260  } else {
261  _ERIs.CalculateEXX_4c_small_molecule(_dftAOdmat);
262  }
263  H -= 0.5 * _ScaHFX * _ERIs.getEXX();
264  }
265  _conv_accelerator.SolveFockmatrix(MOEnergies, MOCoeff, H);
266  _dftAOdmat = _conv_accelerator.DensityMatrix(MOCoeff, MOEnergies);
267 
268  CTP_LOG(ctp::logDEBUG, *_pLog)
269  << ctp::TimeStamp()
270  << " Full atomic density Matrix gives N=" << std::setprecision(9)
271  << _dftAOdmat.cwiseProduct(_dftAOoverlap.Matrix()).sum()
272  << " electrons." << flush;
273  } else {
274  throw std::runtime_error("Initial guess method not known/implemented");
275  }
276  }
277 
278  CTP_LOG(ctp::logDEBUG, *_pLog)
279  << ctp::TimeStamp() << " STARTING SCF cycle" << flush;
280  CTP_LOG(ctp::logDEBUG, *_pLog) << " -----------------------------------------"
281  "---------------------------------"
282  << flush;
283 
284  for (int this_iter = 0; this_iter < _max_iter; this_iter++) {
285  CTP_LOG(ctp::logDEBUG, *_pLog) << flush;
286  CTP_LOG(ctp::logDEBUG, *_pLog)
287  << ctp::TimeStamp() << " Iteration " << this_iter + 1 << " of "
288  << _max_iter << flush;
289 
290  CTP_LOG(ctp::logDEBUG, *_pLog)
291  << ctp::TimeStamp() << " Filled DFT Electron repulsion matrix" << flush;
292  double vxcenergy = 0.0;
293  if (_use_small_grid && _conv_accelerator.getDIIsError() > 1e-3) {
294  orbitals.AOVxc() = _gridIntegration_small.IntegrateVXC(_dftAOdmat);
295  vxcenergy = _gridIntegration_small.getTotEcontribution();
296  CTP_LOG(ctp::logDEBUG, *_pLog)
297  << ctp::TimeStamp() << " Filled approximate DFT Vxc matrix " << flush;
298  } else {
299  orbitals.AOVxc() = _gridIntegration.IntegrateVXC(_dftAOdmat);
300  vxcenergy = _gridIntegration.getTotEcontribution();
301  CTP_LOG(ctp::logDEBUG, *_pLog)
302  << ctp::TimeStamp() << " Filled DFT Vxc matrix " << flush;
303  }
304  CalculateERIs(_dftbasis, _dftAOdmat);
305  Eigen::MatrixXd H = H0 + _ERIs.getERIs() + orbitals.AOVxc();
306  if (_ScaHFX > 0) {
307  if (_with_RI) {
308  if (_conv_accelerator.getUseMixing()) {
309  _ERIs.CalculateEXX(_dftAOdmat);
310  } else {
311  Eigen::Block<Eigen::MatrixXd> occblock =
312  MOCoeff.block(0, 0, MOEnergies.rows(), _numofelectrons / 2);
313  _ERIs.CalculateEXX(occblock, _dftAOdmat);
314  }
315  } else {
316  _ERIs.CalculateEXX_4c_small_molecule(_dftAOdmat);
317  }
318  CTP_LOG(ctp::logDEBUG, *_pLog)
319  << ctp::TimeStamp() << " Filled DFT Electron exchange matrix"
320  << flush;
321  H -= 0.5 * _ScaHFX * _ERIs.getEXX();
322  }
323 
324  double Eone = _dftAOdmat.cwiseProduct(H0).sum();
325  double Etwo = 0.5 * _ERIs.getERIsenergy() + vxcenergy;
326  if (_ScaHFX > 0) {
327  Etwo -= _ScaHFX / 4 * _ERIs.getEXXsenergy();
328  }
329  double totenergy = Eone + _E_nucnuc + Etwo;
330  CTP_LOG(ctp::logDEBUG, *_pLog)
331  << ctp::TimeStamp() << " Single particle energy "
332  << std::setprecision(12) << Eone << flush;
333  CTP_LOG(ctp::logDEBUG, *_pLog)
334  << ctp::TimeStamp() << " Two particle energy " << std::setprecision(12)
335  << Etwo << flush;
336  CTP_LOG(ctp::logDEBUG, *_pLog)
337  << ctp::TimeStamp() << std::setprecision(12)
338  << " Local Exc contribution " << vxcenergy << flush;
339  if (_ScaHFX > 0) {
340  CTP_LOG(ctp::logDEBUG, *_pLog)
341  << ctp::TimeStamp() << std::setprecision(12)
342  << " Non local Ex contribution "
343  << -_ScaHFX / 4 * _ERIs.getEXXsenergy() << flush;
344  }
345  CTP_LOG(ctp::logDEBUG, *_pLog)
346  << ctp::TimeStamp() << " Total Energy " << std::setprecision(12)
347  << totenergy << flush;
348 
349  _dftAOdmat = _conv_accelerator.Iterate(_dftAOdmat, H, MOEnergies, MOCoeff,
350  totenergy);
351 
352  if (tools::globals::verbose) {
353  PrintMOs(MOEnergies);
354  }
355 
356  CTP_LOG(ctp::logDEBUG, *_pLog)
357  << "\t\tGAP "
358  << MOEnergies(_numofelectrons / 2) - MOEnergies(_numofelectrons / 2 - 1)
359  << flush;
360 
361  if (_conv_accelerator.isConverged()) {
362  CTP_LOG(ctp::logDEBUG, *_pLog)
363  << ctp::TimeStamp() << " Total Energy has converged to "
364  << std::setprecision(9) << _conv_accelerator.getDeltaE()
365  << "[Ha] after " << this_iter + 1
366  << " iterations. DIIS error is converged up to "
367  << _conv_accelerator.getDIIsError() << flush;
368  CTP_LOG(ctp::logDEBUG, *_pLog)
369  << ctp::TimeStamp() << " Final Single Point Energy "
370  << std::setprecision(12) << totenergy << " Ha" << flush;
371  PrintMOs(MOEnergies);
372  orbitals.setQMEnergy(totenergy);
373  CalcElDipole(orbitals);
374  break;
375  }
376 
377  CTP_LOG(ctp::logDEBUG, *_pLog)
378  << ctp::TimeStamp() << " Density Matrix gives N="
379  << _dftAOdmat.cwiseProduct(_dftAOoverlap.Matrix()).sum()
380  << " electrons." << flush;
381  }
382  return true;
383 }

449  }
450  }

462  _conv_accelerator.PrintConfigOptions();
463

470  CTP_LOG(ctp::logDEBUG, *_pLog)
471  << ctp::TimeStamp()
472  << " Setup invariant parts of Electron Repulsion integrals " << flush;
473  } else {
474 
475  if (_four_center_method == "cache") {
476 
477  CTP_LOG(ctp::logDEBUG, *_pLog)
478  << ctp::TimeStamp() << " Calculating 4c integrals. " << flush;
479  _ERIs.Initialize_4c_small_molecule(_dftbasis);
480  CTP_LOG(ctp::logDEBUG, *_pLog)
481  << ctp::TimeStamp() << " Calculated 4c integrals. " << flush;
482  }
483 
484  if (_with_screening && _four_center_method == "direct") {
485  CTP_LOG(ctp::logDEBUG, *_pLog)
486  << ctp::TimeStamp() << " Calculating 4c diagonals. " << flush;
487  _ERIs.Initialize_4c_screening(_dftbasis, _screening_eps);
488  CTP_LOG(ctp::logDEBUG, *_pLog)
489  << ctp::TimeStamp() << " Calculated 4c diagonals. " << flush;
490  }
491  }
492 
493  return;

496 Eigen::MatrixXd DFTEngine::RunAtomicDFT_unrestricted(QMAtom* uniqueAtom) {
497  bool with_ecp = _with_ecp;
498  if (uniqueAtom->getType() == "H" || uniqueAtom->getType() == "He") {
499  with_ecp = false;
500  }
501 
502  std::vector<QMAtom*> atom;
503  atom.push_back(uniqueAtom);
504 
505  AOBasis dftbasis;
506  NumericalIntegration gridIntegration;
507  dftbasis.AOBasisFill(_dftbasisset, atom);
508  AOBasis ecp;
509  if (with_ecp) {
510  ecp.ECPFill(_ecpbasisset, atom);
511  }
512  gridIntegration.GridSetup(_grid_name, atom, dftbasis);
513  gridIntegration.setXCfunctional(_xc_functional_name);
514  int numofelectrons = uniqueAtom->getNuccharge();
515  int alpha_e = 0;
516  int beta_e = 0;
517 
518  if ((numofelectrons % 2) != 0) {
519  alpha_e = numofelectrons / 2 + numofelectrons % 2;
520  beta_e = numofelectrons / 2;
521  } else {
522  alpha_e = numofelectrons / 2;
523  beta_e = alpha_e;
524  }
525 
526  AOOverlap dftAOoverlap;
527  AOKinetic dftAOkinetic;
528  AOESP dftAOESP;
529  AOECP dftAOECP;
530  ERIs ERIs_atom;
531 
532  // DFT AOOverlap matrix
533 
534  dftAOoverlap.Fill(dftbasis);
535  dftAOkinetic.Fill(dftbasis);
536 
537  dftAOESP.Fillnucpotential(dftbasis, atom);
538  ERIs_atom.Initialize_4c_small_molecule(dftbasis);
539 
540  Eigen::VectorXd MOEnergies_alpha;
541  Eigen::MatrixXd MOCoeff_alpha;
542  Eigen::VectorXd MOEnergies_beta;
543  Eigen::MatrixXd MOCoeff_beta;
544  ConvergenceAcc Convergence_alpha;
545 
546  ConvergenceAcc Convergence_beta;
547  ConvergenceAcc::options opt_alpha = _conv_opt;
548  opt_alpha.mode = ConvergenceAcc::KSmode::open;
549  opt_alpha.histlength = 20;
550  opt_alpha.levelshift = 0.1;
551  opt_alpha.levelshiftend = 0.0;
552  opt_alpha.usediis = true;
553  opt_alpha.adiis_start = 0.0;
554  opt_alpha.diis_start = 0.0;
555  opt_alpha.numberofelectrons = alpha_e;
556 
557  ConvergenceAcc::options opt_beta = opt_alpha;
558  opt_beta.numberofelectrons = beta_e;
559 
560  ctp::Logger log;
561  Convergence_alpha.Configure(opt_alpha);
562  Convergence_alpha.setLogger(&log);
563  Convergence_alpha.setOverlap(dftAOoverlap, 1e-8);
564  Convergence_beta.Configure(opt_beta);
565  Convergence_beta.setLogger(&log);
566  Convergence_beta.setOverlap(dftAOoverlap, 1e-8);
567  /**** Construct initial density ****/
568 
569  Eigen::MatrixXd H0 = dftAOkinetic.Matrix() + dftAOESP.getNuclearpotential();
570  if (with_ecp) {
571  dftAOECP.setECP(&ecp);
572  dftAOECP.Fill(dftbasis);
573  H0 += dftAOECP.Matrix();
574  }
575  Convergence_alpha.SolveFockmatrix(MOEnergies_alpha, MOCoeff_alpha, H0);
576 
577  Eigen::MatrixXd dftAOdmat_alpha =
578  Convergence_alpha.DensityMatrix(MOCoeff_alpha, MOEnergies_alpha);
579  if (uniqueAtom->getType() == "H") {
580  return dftAOdmat_alpha;
581  }
582  Convergence_beta.SolveFockmatrix(MOEnergies_beta, MOCoeff_beta, H0);
583  Eigen::MatrixXd dftAOdmat_beta =
584  Convergence_beta.DensityMatrix(MOCoeff_beta, MOEnergies_beta);
585 
586  int maxiter = 80;
587  for (int this_iter = 0; this_iter < maxiter; this_iter++) {
588  ERIs_atom.CalculateERIs_4c_small_molecule(dftAOdmat_alpha + dftAOdmat_beta);
589  double E_two_alpha =
590  ERIs_atom.getERIs().cwiseProduct(dftAOdmat_alpha).sum();
591  double E_two_beta = ERIs_atom.getERIs().cwiseProduct(dftAOdmat_beta).sum();
592  Eigen::MatrixXd H_alpha = H0 + ERIs_atom.getERIs();
593  Eigen::MatrixXd H_beta = H0 + ERIs_atom.getERIs();
594 
595  Eigen::MatrixXd AOVxc_alpha = gridIntegration.IntegrateVXC(dftAOdmat_alpha);
596  double E_vxc_alpha = gridIntegration.getTotEcontribution();
597 
598  Eigen::MatrixXd AOVxc_beta = gridIntegration.IntegrateVXC(dftAOdmat_beta);
599  double E_vxc_beta = gridIntegration.getTotEcontribution();
600  H_alpha += AOVxc_alpha;
601  H_beta += AOVxc_beta;
602  E_two_alpha += E_vxc_alpha;
603  E_two_beta += E_vxc_beta;
604 
605  if (_ScaHFX > 0) {
606  ERIs_atom.CalculateEXX_4c_small_molecule(dftAOdmat_alpha);
607  double E_exx_alpha =
608  -0.5 * _ScaHFX *
609  ERIs_atom.getEXX().cwiseProduct(dftAOdmat_alpha).sum();
610  H_alpha -= _ScaHFX * ERIs_atom.getEXX();
611  E_two_alpha += E_exx_alpha;
612  ERIs_atom.CalculateEXX_4c_small_molecule(dftAOdmat_beta);
613  double E_exx_beta = -0.5 * _ScaHFX *
614  ERIs_atom.getEXX().cwiseProduct(dftAOdmat_beta).sum();
615  H_beta -= _ScaHFX * ERIs_atom.getEXX();
616  E_two_beta += E_exx_beta;
617  }
618 
619  double E_one_alpha = dftAOdmat_alpha.cwiseProduct(H0).sum();
620  double E_one_beta = dftAOdmat_beta.cwiseProduct(H0).sum();
621  double E_alpha = E_one_alpha + E_two_alpha;
622  double E_beta = E_one_beta + E_two_beta;
623  double totenergy = E_alpha + E_beta;
624  // evolve alpha
625  dftAOdmat_alpha = Convergence_alpha.Iterate(
626  dftAOdmat_alpha, H_alpha, MOEnergies_alpha, MOCoeff_alpha, E_alpha);
627  // evolve beta
628  dftAOdmat_beta = Convergence_beta.Iterate(
629  dftAOdmat_beta, H_beta, MOEnergies_beta, MOCoeff_beta, E_beta);
630 
631  if (tools::globals::verbose) {
632  CTP_LOG(ctp::logDEBUG, *_pLog)
633  << ctp::TimeStamp() << " Iter " << this_iter << " of " << maxiter
634  << " Etot " << totenergy << " diise_a "
635  << Convergence_alpha.getDIIsError() << " diise_b "
636  << Convergence_beta.getDIIsError() << "\n\t\t a_gap "
637  << MOEnergies_alpha(alpha_e) - MOEnergies_alpha(alpha_e - 1)
638  << " b_gap " << MOEnergies_beta(beta_e) - MOEnergies_beta(beta_e - 1)
639  << " Nalpha="
640  << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_alpha).sum()
641  << " Nbeta="
642  << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_beta).sum() << flush;
643  }
644  bool converged =
645  Convergence_alpha.isConverged() && Convergence_beta.isConverged();
646  if (converged || this_iter == maxiter - 1) {
647 
648  if (converged) {
649  CTP_LOG(ctp::logDEBUG, *_pLog)
650  << ctp::TimeStamp() << " Converged after " << this_iter + 1
651  << " iterations" << flush;
652  } else {
653  CTP_LOG(ctp::logDEBUG, *_pLog)
654  << ctp::TimeStamp() << " Not converged after " << this_iter + 1
655  << " iterations. Unconverged density.\n\t\t\t"
656  << " DIIsError_alpha=" << Convergence_alpha.getDIIsError()
657  << " DIIsError_beta=" << Convergence_beta.getDIIsError() << flush;
658  }
659  break;
660  }
661  }
662  Eigen::MatrixXd avgmatrix =
663  SphericalAverageShells(dftAOdmat_alpha + dftAOdmat_beta, dftbasis);
664  CTP_LOG(ctp::logDEBUG, *_pLog)
665  << ctp::TimeStamp() << " Atomic density Matrix for "
666  << uniqueAtom->getType() << " gives N=" << std::setprecision(9)
667  << avgmatrix.cwiseProduct(dftAOoverlap.Matrix()).sum() << " electrons."
668  << flush;
669  return avgmatrix;

993 // average atom densities matrices, for SP and other combined shells average
994 // each subshell separately.
995 Eigen::MatrixXd DFTEngine::SphericalAverageShells(const Eigen::MatrixXd& dmat,
996  AOBasis& dftbasis) {
997  Eigen::MatrixXd avdmat = Eigen::MatrixXd::Zero(dmat.rows(), dmat.cols());
998  int start = 0.0;
999  std::vector<int> starts;
1000  std::vector<int> ends;
1001  for (AOShell* shell : dftbasis) {
1002  int end = shell->getNumFunc() + start;
1003 
1004  if (shell->isCombined()) {
1005  std::vector<int> temp = NumFuncSubShell(shell->getType());
1006  int numfunc = start;
1007  for (int& SubshellFunc : temp) {
1008  starts.push_back(numfunc);
1009  numfunc += SubshellFunc;
1010  ends.push_back(numfunc);
1011  }
1012  } else {
1013  starts.push_back(start);
1014  ends.push_back(end);
1015  }
1016  start = end;
1017  }
1018  for (unsigned k = 0; k < starts.size(); k++) {
1019  int s1 = starts[k];
1020  int e1 = ends[k];
1021  int len1 = e1 - s1;
1022  for (unsigned l = 0; l < starts.size(); l++) {
1023  int s2 = starts[l];
1024  int e2 = ends[l];
1025  int len2 = e2 - s2;
1026  double diag = 0.0;
1027  double offdiag = 0.0;
1028  for (int i = 0; i < len1; ++i) {
1029  for (int j = 0; j < len2; ++j) {
1030  if (i == j) {
1031  diag += dmat(s1 + i, s2 + j);
1032  } else {
1033  offdiag += dmat(s1 + i, s2 + j);
1034  }
1035  }
1036  }
1037  if (len1 == len2) {
1038  diag = diag / double(len1);
1039  offdiag = offdiag / double(len1 * (len1 - 1));
1040  } else {
1041  double avg = (diag + offdiag) / double(len1 * len2);
1042  diag = avg;
1043  offdiag = avg;
1044  }
1045  for (int i = 0; i < len1; ++i) {
1046  for (int j = 0; j < len2; ++j) {
1047  if (i == j) {
1048  avdmat(s1 + i, s2 + j) = diag;
1049  } else {
1050  avdmat(s1 + i, s2 + j) = offdiag;
1051  }
1052  }
1053  }
1054  }
1055  }
1056 
1057  return avdmat;
1058 }

1060 Eigen::MatrixXd DFTEngine::IntegrateExternalDensity(Orbitals& extdensity) {
1061  BasisSet basis;
1062  basis.LoadBasisSet(extdensity.getDFTbasisName());
1063  AOBasis aobasis;
1064  aobasis.AOBasisFill(basis, extdensity.QMAtoms());
1065  NumericalIntegration numint;
1066  numint.GridSetup(_gridquality, extdensity.QMAtoms(), aobasis);
1067  Eigen::MatrixXd dmat = extdensity.DensityMatrixGroundState();
1068 
1069  numint.IntegrateDensity(dmat);
1070  CTP_LOG(ctp::logDEBUG, *_pLog)
1071  << ctp::TimeStamp() << " Calculated external density" << flush;
1072  Eigen::MatrixXd e_contrib = numint.IntegratePotential(_dftbasis);
1073  CTP_LOG(ctp::logDEBUG, *_pLog)
1074  << ctp::TimeStamp() << " Calculated potential from electron density"
1075  << flush;
1076  AOESP esp;
1077  esp.Fillnucpotential(_dftbasis, extdensity.QMAtoms());
1078 
1079  double nuc_energy = 0.0;
1080  for (const QMAtom* atom : _atoms) {
1081  nuc_energy +=
1082  numint.IntegratePotential(atom->getPos()) * atom->getNuccharge();
1083  for (const QMAtom* extatom : extdensity.QMAtoms()) {
1084  const double dist = tools::abs(atom->getPos() - extatom->getPos());
1085  nuc_energy += atom->getNuccharge() * extatom->getNuccharge() / dist;
1086  }
1087  }
