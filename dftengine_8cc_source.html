43 void DFTEngine::Initialize(tools::Property& options) {
44 
45  const std::string key_xtpdft = "xtpdft";
46  dftbasis_name_ = options.get(".basisset").as<std::string>();
47 
48  if (options.exists(".auxbasisset")) {
49  auxbasis_name_ = options.get(".auxbasisset").as<std::string>();
50  }
51 
52  if (!auxbasis_name_.empty()) {
53  screening_eps_ = options.get(key_xtpdft + ".screening_eps").as<double>();
54  fock_matrix_reset_ =
55  options.get(key_xtpdft + ".fock_matrix_reset").as<Index>();
56  }
57  if (options.exists(".ecp")) {
58  ecp_name_ = options.get(".ecp").as<std::string>();
59  }
60 
61  initial_guess_ = options.get(".initial_guess").as<std::string>();
62 
63  grid_name_ = options.get(key_xtpdft + ".integration_grid").as<std::string>();
64  xc_functional_name_ = options.get(".functional").as<std::string>();
65

67  integrate_ext_density_ = true;
68  orbfilename_ =
69  options.get(key_xtpdft + ".externaldensity.orbfile").as<std::string>();
70  gridquality_ = options.get(key_xtpdft + ".externaldensity.gridquality")
71  .as<std::string>();
72  state_ =
73  options.get(key_xtpdft + ".externaldensity.state").as<std::string>();
74  }
75 
76  if (options.exists(".externalfield")) {
77  integrate_ext_field_ = true;
78  extfield_ = options.get(".externalfield").as<Eigen::Vector3d>();
79  }
80 
81  conv_opt_.Econverged =
82  options.get(key_xtpdft + ".convergence.energy").as<double>();
83  conv_opt_.error_converged =
84  options.get(key_xtpdft + ".convergence.error").as<double>();
85  max_iter_ =
86  options.get(key_xtpdft + ".convergence.max_iterations").as<Index>();
87

93  conv_opt_.usediis = false;

96  conv_opt_.histlength = 1;
97  conv_opt_.maxout = false;
98  }

100  options.get(key_xtpdft + ".convergence.mixing").as<double>();
101  conv_opt_.levelshift =
102  options.get(key_xtpdft + ".convergence.levelshift").as<double>();
103  conv_opt_.levelshiftend =
104  options.get(key_xtpdft + ".convergence.levelshift_end").as<double>();

109  conv_opt_.diis_start =

169 tools::EigenSystem DFTEngine::ModelPotentialGuess(
170  const Mat_p_Energy& H0, const QMMolecule& mol,
171  const Vxc_Potential<Vxc_Grid>& vxcpotential) const {
172  Eigen::MatrixXd Dmat = AtomicGuess(mol);
173  Mat_p_Energy e_vxc = vxcpotential.IntegrateVXC(Dmat);
174  XTP_LOG(Log::info, *pLog_)
175  << TimeStamp() << " Filled DFT Vxc matrix " << std::flush;
176 
177  Eigen::MatrixXd H = H0.matrix() + e_vxc.matrix();
178 
179  if (ScaHFX_ > 0) {
180  std::array<Eigen::MatrixXd, 2> both =
181  CalcERIs_EXX(Eigen::MatrixXd::Zero(0, 0), Dmat, 1e-12);
182  H += both[0];
183  H += ScaHFX_ * both[1];
184  } else {
185  H += CalcERIs(Dmat, 1e-12);
186  }
187  return conv_accelerator_.SolveFockmatrix(H);

190 bool DFTEngine::Evaluate(Orbitals& orb) {
191  Prepare(orb);
192  Mat_p_Energy H0 = SetupH0(orb.QMAtoms());
193  tools::EigenSystem MOs;
194  MOs.eigenvalues() = Eigen::VectorXd::Zero(H0.cols());
195  MOs.eigenvectors() = Eigen::MatrixXd::Zero(H0.rows(), H0.cols());
196  Vxc_Potential<Vxc_Grid> vxcpotential = SetupVxc(orb.QMAtoms());
197  ConfigOrbfile(orb);
198 
199  if (initial_guess_ == "orbfile") {
200  XTP_LOG(Log::error, *pLog_)
201  << TimeStamp() << " Reading guess from orbitals object/file"
202  << std::flush;
203  MOs = orb.MOs();
204  MOs.eigenvectors() = OrthogonalizeGuess(MOs.eigenvectors());
205  } else {
206  XTP_LOG(Log::error, *pLog_)
207  << TimeStamp() << " Setup Initial Guess using: " << initial_guess_
208  << std::flush;
209  if (initial_guess_ == "independent") {
210  MOs = IndependentElectronGuess(H0);
211  } else if (initial_guess_ == "atom") {
212  MOs = ModelPotentialGuess(H0, orb.QMAtoms(), vxcpotential);
213  } else {
214  throw std::runtime_error("Initial guess method not known/implemented");
215  }
216  }
217 
218  Eigen::MatrixXd Dmat = conv_accelerator_.DensityMatrix(MOs);
219  XTP_LOG(Log::info, *pLog_)
220  << TimeStamp() << " Guess Matrix gives N=" << std::setprecision(9)
221  << Dmat.cwiseProduct(dftAOoverlap_.Matrix()).sum() << " electrons."
222  << std::flush;
223 
224  XTP_LOG(Log::error, *pLog_)
225  << TimeStamp() << " STARTING SCF cycle" << std::flush;
226  XTP_LOG(Log::error, *pLog_)
227  << " ----------------------------------------------"
228  "----------------------------"
229  << std::flush;
230 
231  Eigen::MatrixXd J = Eigen::MatrixXd::Zero(Dmat.rows(), Dmat.cols());
232  Eigen::MatrixXd K;
233  if (ScaHFX_ > 0) {
234  K = Eigen::MatrixXd::Zero(Dmat.rows(), Dmat.cols());
235  }
236 
237  double start_incremental_F_threshold = 1e-4;
238  if (!auxbasis_name_.empty()) {
239  start_incremental_F_threshold = 0.0; // Disable if RI is used
240  }
241  IncrementalFockBuilder incremental_fock(*pLog_, start_incremental_F_threshold,
242  fock_matrix_reset_);
243  incremental_fock.Configure(Dmat);
244 
245  for (Index this_iter = 0; this_iter < max_iter_; this_iter++) {
246  XTP_LOG(Log::error, *pLog_) << std::flush;
247  XTP_LOG(Log::error, *pLog_) << TimeStamp() << " Iteration " << this_iter + 1
248  << " of " << max_iter_ << std::flush;
249 
250  Mat_p_Energy e_vxc = vxcpotential.IntegrateVXC(Dmat);
251  XTP_LOG(Log::info, *pLog_)
252  << TimeStamp() << " Filled DFT Vxc matrix " << std::flush;
253 
254  Eigen::MatrixXd H = H0.matrix() + e_vxc.matrix();
255  double Eone = Dmat.cwiseProduct(H0.matrix()).sum();
256  double Etwo = e_vxc.energy();
257  double exx = 0.0;
258 
259  incremental_fock.Start(this_iter, conv_accelerator_.getDIIsError());
260  incremental_fock.resetMatrices(J, K, Dmat);
261  incremental_fock.UpdateCriteria(conv_accelerator_.getDIIsError(),
262  this_iter);
263 
264  double integral_error =
265  std::min(conv_accelerator_.getDIIsError() * 1e-5, 1e-5);
266  if (ScaHFX_ > 0) {
267  std::array<Eigen::MatrixXd, 2> both = CalcERIs_EXX(
268  MOs.eigenvectors(), incremental_fock.getDmat_diff(), integral_error);
269  J += both[0];
270  H += J;
271  Etwo += 0.5 * Dmat.cwiseProduct(J).sum();
272  K += both[1];
273  H += 0.5 * ScaHFX_ * K;
274  exx = 0.25 * ScaHFX_ * Dmat.cwiseProduct(K).sum();
275  XTP_LOG(Log::info, *pLog_)
276  << TimeStamp() << " Filled F+K matrix " << std::flush;
277  } else {
278  J += CalcERIs(incremental_fock.getDmat_diff(), integral_error);
279  XTP_LOG(Log::info, *pLog_)
280  << TimeStamp() << " Filled F matrix " << std::flush;
281  H += J;
282  Etwo += 0.5 * Dmat.cwiseProduct(J).sum();
283  }
284 
285  Etwo += exx;
286  double totenergy = Eone + H0.energy() + Etwo;
287  XTP_LOG(Log::info, *pLog_) << TimeStamp() << " Single particle energy "
288  << std::setprecision(12) << Eone << std::flush;
289  XTP_LOG(Log::info, *pLog_) << TimeStamp() << " Two particle energy "
290  << std::setprecision(12) << Etwo << std::flush;
291  XTP_LOG(Log::info, *pLog_)
292  << TimeStamp() << std::setprecision(12) << " Local Exc contribution "
293  << e_vxc.energy() << std::flush;
294  if (ScaHFX_ > 0) {
295  XTP_LOG(Log::info, *pLog_)
296  << TimeStamp() << std::setprecision(12)
297  << " Non local Ex contribution " << exx << std::flush;
298  }
299  XTP_LOG(Log::error, *pLog_)
300  << TimeStamp() << " Total Energy " << std::setprecision(12) << totenergy
301  << std::flush;
302 
303  Dmat = conv_accelerator_.Iterate(Dmat, H, MOs, totenergy);
304  incremental_fock.UpdateDmats(Dmat, conv_accelerator_.getDIIsError(),
305  this_iter);
306 
307  PrintMOs(MOs.eigenvalues(), Log::info);
308 
309  XTP_LOG(Log::info, *pLog_) << "\t\tGAP "
310  << MOs.eigenvalues()(numofelectrons_ / 2) -
311  MOs.eigenvalues()(numofelectrons_ / 2 - 1)
312  << std::flush;
313 
314  if (conv_accelerator_.isConverged()) {
315  XTP_LOG(Log::error, *pLog_)
316  << TimeStamp() << " Total Energy has converged to "
317  << std::setprecision(9) << conv_accelerator_.getDeltaE()
318  << "[Ha] after " << this_iter + 1
319  << " iterations. DIIS error is converged up to "
320  << conv_accelerator_.getDIIsError() << std::flush;
321  XTP_LOG(Log::error, *pLog_)
322  << TimeStamp() << " Final Single Point Energy "
323  << std::setprecision(12) << totenergy << " Ha" << std::flush;
324  XTP_LOG(Log::error, *pLog_) << TimeStamp() << std::setprecision(12)
325  << " Final Local Exc contribution "
326  << e_vxc.energy() << " Ha" << std::flush;
327  if (ScaHFX_ > 0) {
328  XTP_LOG(Log::error, *pLog_) << TimeStamp() << std::setprecision(12)
329  << " Final Non Local Ex contribution "
330  << exx << " Ha" << std::flush;
331  }
332 
333  PrintMOs(MOs.eigenvalues(), Log::error);
334  orb.setQMEnergy(totenergy);
335  orb.MOs() = MOs;
336  CalcElDipole(orb);
337  break;
338  } else if (this_iter == max_iter_ - 1) {
339  XTP_LOG(Log::error, *pLog_)
340  << TimeStamp() << " DFT calculation has not converged after "
341  << max_iter_
342  << " iterations. Use more iterations or another convergence "
343  "acceleration scheme."
344  << std::flush;
345  return false;
346  }
347  }
348  return true;
349 }

351 Mat_p_Energy DFTEngine::SetupH0(const QMMolecule& mol) const {
352 
353  AOKinetic dftAOkinetic;
354 
355  dftAOkinetic.Fill(dftbasis_);
356  XTP_LOG(Log::info, *pLog_)
357  << TimeStamp() << " Filled DFT Kinetic energy matrix ." << std::flush;
358 
359  AOMultipole dftAOESP;
360  dftAOESP.FillPotential(dftbasis_, mol);
361  XTP_LOG(Log::info, *pLog_)
362  << TimeStamp() << " Filled DFT nuclear potential matrix." << std::flush;
363 
364  Eigen::MatrixXd H0 = dftAOkinetic.Matrix() + dftAOESP.Matrix();
365  XTP_LOG(Log::error, *pLog_)
366  << TimeStamp() << " Constructed independent particle hamiltonian "
367  << std::flush;
368  double E0 = NuclearRepulsion(mol);
369  XTP_LOG(Log::error, *pLog_) << TimeStamp() << " Nuclear Repulsion Energy is "
370  << std::setprecision(9) << E0 << std::flush;
371 
372  if (!ecp_name_.empty()) {
373  AOECP dftAOECP;
374  dftAOECP.FillPotential(dftbasis_, ecp_);
375  H0 += dftAOECP.Matrix();
376  XTP_LOG(Log::info, *pLog_)
377  << TimeStamp() << " Filled DFT ECP matrix" << std::flush;
378  }
379 
380  if (externalsites_ != nullptr) {
381  XTP_LOG(Log::error, *pLog_) << TimeStamp() << " " << externalsites_->size()
382  << " External sites" << std::flush;
383  bool has_quadrupoles = std::any_of(
384  externalsites_->begin(), externalsites_->end(),
385  [](const std::unique_ptr<StaticSite>& s) { return s->getRank() == 2; });
386  std::string header =
387  " Name Coordinates[a0] charge[e] dipole[e*a0] ";
388  if (has_quadrupoles) {
389  header += " quadrupole[e*a0^2]";
390  }
391  XTP_LOG(Log::error, *pLog_) << header << std::flush;
392  Index limit = 50;
393  Index counter = 0;
394  for (const std::unique_ptr<StaticSite>& site : *externalsites_) {
395  if (counter == limit) {
396  break;
397  }
398  std::string output =
399  (boost::format(" %1$s"
400  " %2$+1.4f %3$+1.4f %4$+1.4f"
401  " %5$+1.4f") %
402  site->getElement() % site->getPos()[0] % site->getPos()[1] %
403  site->getPos()[2] % site->getCharge())
404  .str();
405  const Eigen::Vector3d& dipole = site->getDipole();
406  output += (boost::format(" %1$+1.4f %2$+1.4f %3$+1.4f") % dipole[0] %
407  dipole[1] % dipole[2])
408  .str();
409  if (site->getRank() > 1) {
410  Eigen::VectorXd quadrupole = site->Q().tail<5>();
411  output +=
412  (boost::format(" %1$+1.4f %2$+1.4f %3$+1.4f %4$+1.4f %5$+1.4f") %
413  quadrupole[0] % quadrupole[1] % quadrupole[2] % quadrupole[3] %
414  quadrupole[4])
415  .str();
416  }
417  XTP_LOG(Log::error, *pLog_) << output << std::flush;
418  counter++;
419  }
420  if (counter == limit) {
421  XTP_LOG(Log::error, *pLog_)
422  << " ... (" << externalsites_->size() - limit
423  << " sites not displayed)\n"
424  << std::flush;
425  }
426 
427  Mat_p_Energy ext_multipoles =
428  IntegrateExternalMultipoles(mol, *externalsites_);
429  XTP_LOG(Log::error, *pLog_)
430  << TimeStamp() << " Nuclei-external site interaction energy "
431  << std::setprecision(9) << ext_multipoles.energy() << std::flush;
432  E0 += ext_multipoles.energy();
433  H0 += ext_multipoles.matrix();
434  }
435 
436  if (integrate_ext_density_) {
437  Orbitals extdensity;
438  extdensity.ReadFromCpt(orbfilename_);
439  Mat_p_Energy extdensity_result = IntegrateExternalDensity(mol, extdensity);
440  E0 += extdensity_result.energy();
441  XTP_LOG(Log::error, *pLog_)
442  << TimeStamp() << " Nuclei-external density interaction energy "
443  << std::setprecision(9) << extdensity_result.energy() << std::flush;
444  H0 += extdensity_result.matrix();
445  }
446 
447  if (integrate_ext_field_) {
448 
449  XTP_LOG(Log::error, *pLog_)
450  << TimeStamp() << " Integrating external electric field with F[Hrt]="
451  << extfield_.transpose() << std::flush;
452  H0 += IntegrateExternalField(mol);
453  }
454 
455  return Mat_p_Energy(E0, H0);
456 }

493 Eigen::MatrixXd DFTEngine::RunAtomicDFT_unrestricted(
494  const QMAtom& uniqueAtom) const {
495  bool with_ecp = !ecp_name_.empty();
496  if (uniqueAtom.getElement() == "H" || uniqueAtom.getElement() == "He") {
497  with_ecp = false;
498  }
499 
500  QMMolecule atom = QMMolecule("individual_atom", 0);
501  atom.push_back(uniqueAtom);
502 
503  BasisSet basisset;
504  basisset.Load(dftbasis_name_);
505  AOBasis dftbasis;
506  dftbasis.Fill(basisset, atom);
507  Vxc_Grid grid;
508  grid.GridSetup(grid_name_, atom, dftbasis);
509  Vxc_Potential<Vxc_Grid> gridIntegration(grid);
510  gridIntegration.setXCfunctional(xc_functional_name_);
511 
512  ECPAOBasis ecp;
513  if (with_ecp) {
514  ECPBasisSet ecps;
515  ecps.Load(ecp_name_);
516  ecp.Fill(ecps, atom);
517  }
518 
519  Index numofelectrons = uniqueAtom.getNuccharge();
520  Index alpha_e = 0;
521  Index beta_e = 0;
522 
523  if ((numofelectrons % 2) != 0) {
524  alpha_e = numofelectrons / 2 + numofelectrons % 2;
525  beta_e = numofelectrons / 2;
526  } else {
527  alpha_e = numofelectrons / 2;
528  beta_e = alpha_e;
529  }
530 
531  AOOverlap dftAOoverlap;
532  AOKinetic dftAOkinetic;
533  AOMultipole dftAOESP;
534  AOECP dftAOECP;
535  ERIs ERIs_atom;
536 
537  // DFT AOOverlap matrix
538 
539  dftAOoverlap.Fill(dftbasis);
540  dftAOkinetic.Fill(dftbasis);
541 
542  dftAOESP.FillPotential(dftbasis, atom);
543  ERIs_atom.Initialize_4c(dftbasis);
544 
545  ConvergenceAcc Convergence_alpha;
546  ConvergenceAcc Convergence_beta;
547  ConvergenceAcc::options opt_alpha = conv_opt_;
548  opt_alpha.mode = ConvergenceAcc::KSmode::open;
549  opt_alpha.histlength = 20;
550  opt_alpha.levelshift = 0.1;
551  opt_alpha.levelshiftend = 0.0;
552  opt_alpha.usediis = true;
553  opt_alpha.adiis_start = 0.0;
554  opt_alpha.diis_start = 0.0;
555  opt_alpha.numberofelectrons = alpha_e;
556 
557  ConvergenceAcc::options opt_beta = opt_alpha;
558  opt_beta.numberofelectrons = beta_e;
559 
560  Logger log;
561  Convergence_alpha.Configure(opt_alpha);
562  Convergence_alpha.setLogger(&log);
563  Convergence_alpha.setOverlap(dftAOoverlap, 1e-8);
564  Convergence_beta.Configure(opt_beta);
565  Convergence_beta.setLogger(&log);
566  Convergence_beta.setOverlap(dftAOoverlap, 1e-8);
567  /**** Construct initial density ****/
568 
569  Eigen::MatrixXd H0 = dftAOkinetic.Matrix() + dftAOESP.Matrix();
570  if (with_ecp) {
571  dftAOECP.FillPotential(dftbasis, ecp);
572  H0 += dftAOECP.Matrix();
573  }
574  tools::EigenSystem MOs_alpha = Convergence_alpha.SolveFockmatrix(H0);
575 
576  Eigen::MatrixXd dftAOdmat_alpha = Convergence_alpha.DensityMatrix(MOs_alpha);
577  if (uniqueAtom.getElement() == "H") {
578  return dftAOdmat_alpha;
579  }
580  tools::EigenSystem MOs_beta = Convergence_beta.SolveFockmatrix(H0);
581  Eigen::MatrixXd dftAOdmat_beta = Convergence_beta.DensityMatrix(MOs_beta);
582

584  for (Index this_iter = 0; this_iter < maxiter; this_iter++) {
585 
586  Eigen::MatrixXd H_alpha = H0;
587  Eigen::MatrixXd H_beta = H_alpha;
588 
589  double E_two_alpha = 0.0;
590  double E_two_beta = 0.0;
591 
592  double integral_error = std::min(1e-5 * 0.5 *
593  (Convergence_alpha.getDIIsError() +
594  Convergence_beta.getDIIsError()),
595  1e-5);
596  if (ScaHFX_ > 0) {
597  std::array<Eigen::MatrixXd, 2> both_alpha =
598  ERIs_atom.CalculateERIs_EXX_4c(dftAOdmat_alpha, integral_error);
599 
600  std::array<Eigen::MatrixXd, 2> both_beta =
601  ERIs_atom.CalculateERIs_EXX_4c(dftAOdmat_beta, integral_error);
602  Eigen::MatrixXd Hartree = both_alpha[0] + both_beta[0];
603  E_two_alpha += Hartree.cwiseProduct(dftAOdmat_alpha).sum();
604  E_two_beta += Hartree.cwiseProduct(dftAOdmat_beta).sum();
605  E_two_alpha += 0.5 * both_alpha[1].cwiseProduct(dftAOdmat_alpha).sum();
606  E_two_beta += 0.5 * both_beta[1].cwiseProduct(dftAOdmat_beta).sum();
607  H_alpha += Hartree + ScaHFX_ * both_alpha[1];
608  H_beta += Hartree + ScaHFX_ * both_beta[1];
609 
610  } else {
611  Eigen::MatrixXd Hartree = ERIs_atom.CalculateERIs_4c(
612  dftAOdmat_alpha + dftAOdmat_beta, integral_error);
613  E_two_alpha += Hartree.cwiseProduct(dftAOdmat_alpha).sum();
614  E_two_beta += Hartree.cwiseProduct(dftAOdmat_beta).sum();
615  H_alpha += Hartree;
616  H_beta += Hartree;
617  }
618 
619  Mat_p_Energy e_vxc_alpha = gridIntegration.IntegrateVXC(dftAOdmat_alpha);
620  H_alpha += e_vxc_alpha.matrix();
621  E_two_alpha += e_vxc_alpha.energy();
622 
623  Mat_p_Energy e_vxc_beta = gridIntegration.IntegrateVXC(dftAOdmat_beta);
624  H_beta += e_vxc_beta.matrix();
625  E_two_beta += e_vxc_beta.energy();
626

628  double E_one_beta = dftAOdmat_beta.cwiseProduct(H0).sum();
629  double E_alpha = E_one_alpha + E_two_alpha;
630  double E_beta = E_one_beta + E_two_beta;
631  double totenergy = E_alpha + E_beta;
632  // evolve alpha
633  dftAOdmat_alpha =
634  Convergence_alpha.Iterate(dftAOdmat_alpha, H_alpha, MOs_alpha, E_alpha);
635  // evolve beta
636  dftAOdmat_beta =
637  Convergence_beta.Iterate(dftAOdmat_beta, H_beta, MOs_beta, E_beta);
638 
639  XTP_LOG(Log::debug, *pLog_)
640  << TimeStamp() << " Iter " << this_iter << " of " << maxiter << " Etot "
641  << totenergy << " diise_a " << Convergence_alpha.getDIIsError()
642  << " diise_b " << Convergence_beta.getDIIsError() << "\n\t\t a_gap "
643  << MOs_alpha.eigenvalues()(alpha_e) -
644  MOs_alpha.eigenvalues()(alpha_e - 1)
645  << " b_gap "
646  << MOs_beta.eigenvalues()(beta_e) - MOs_beta.eigenvalues()(beta_e - 1)
647  << " Nalpha="
648  << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_alpha).sum()
649  << " Nbeta=" << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_beta).sum()
650  << std::flush;
651 
652  bool converged =
653  Convergence_alpha.isConverged() && Convergence_beta.isConverged();
654  if (converged || this_iter == maxiter - 1) {
655 
656  if (converged) {
657  XTP_LOG(Log::info, *pLog_)
658  << TimeStamp() << " Converged after " << this_iter + 1
659  << " iterations" << std::flush;
660  } else {

662  << TimeStamp() << " Not converged after " << this_iter + 1
663  << " iterations. Unconverged density.\n\t\t\t"
664  << " DIIsError_alpha=" << Convergence_alpha.getDIIsError()
665  << " DIIsError_beta=" << Convergence_beta.getDIIsError()
666  << std::flush;
667  }
668  break;
669  }
670  }

672  SphericalAverageShells(dftAOdmat_alpha + dftAOdmat_beta, dftbasis);

675  << " gives N=" << std::setprecision(9)
676  << avgmatrix.cwiseProduct(dftAOoverlap.Matrix()).sum() << " electrons."
677  << std::flush;

894 // spherically average the density matrix belonging to two shells
895 Eigen::MatrixXd DFTEngine::SphericalAverageShells(
896  const Eigen::MatrixXd& dmat, const AOBasis& dftbasis) const {
897  Eigen::MatrixXd avdmat = Eigen::MatrixXd::Zero(dmat.rows(), dmat.cols());
898  for (const AOShell& shellrow : dftbasis) {
899  Index size_row = shellrow.getNumFunc();
900  Index start_row = shellrow.getStartIndex();
901  for (const AOShell& shellcol : dftbasis) {
902  Index size_col = shellcol.getNumFunc();
903  Index start_col = shellcol.getStartIndex();
904  Eigen::MatrixXd shelldmat =
905  dmat.block(start_row, start_col, size_row, size_col);
906  if (shellrow.getL() == shellcol.getL()) {
907  double diagavg = shelldmat.diagonal().sum() / double(shelldmat.rows());
908  Index offdiagelements =
909  shelldmat.rows() * shelldmat.cols() - shelldmat.cols();
910  double offdiagavg = (shelldmat.sum() - shelldmat.diagonal().sum()) /
911  double(offdiagelements);
912  avdmat.block(start_row, start_col, size_row, size_col).array() =
913  offdiagavg;
914  avdmat.block(start_row, start_col, size_row, size_col)
915  .diagonal()
916  .array() = diagavg;
917  } else {
918  double avg = shelldmat.sum() / double(shelldmat.size());
919  avdmat.block(start_row, start_col, size_row, size_col).array() = avg;
920  }
921  }
922  }
923  return avdmat;
924 }
