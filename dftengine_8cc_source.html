46 void DFTEngine::Initialize(Property& options) {
47 
48  string key = "package";
49  const string key_xtpdft = "package.xtpdft";
50  _dftbasis_name = options.get(key + ".basisset").as<string>();
51 
52  if (options.exists(key + ".auxbasisset")) {
53  _auxbasis_name = options.get(key + ".auxbasisset").as<string>();
54  }
55 
56  _four_center_method =
57  options.get(key_xtpdft + ".four_center_method").as<std::string>();
58 
59  if (_four_center_method != "RI") {
60  _with_screening = options.get(key_xtpdft + ".with_screening").as<bool>();
61  _screening_eps = options.get(key_xtpdft + ".screening_eps").as<double>();
62  }
63 
64  if (options.exists(key + ".ecp")) {
65  _ecp_name = options.get(key + ".ecp").as<string>();
66  _with_ecp = true;
67  } else {
68  _with_ecp = false;
69  }
70  _with_guess = options.get(key + ".read_guess").as<bool>();
71  _initial_guess = options.get(key_xtpdft + ".initial_guess").as<string>();
72 
73  _grid_name = options.get(key_xtpdft + ".integration_grid").as<string>();
74  _xc_functional_name = options.get(key + ".functional").as<string>();
75 
76  if (options.exists(key_xtpdft + ".externaldensity")) {
77  _integrate_ext_density = true;
78  _orbfilename = options.ifExistsReturnElseThrowRuntimeError<string>(
79  key_xtpdft + ".externaldensity.orbfile");
80  _gridquality = options.ifExistsReturnElseThrowRuntimeError<string>(
81  key_xtpdft + ".externaldensity.gridquality");
82  _state = options.ifExistsReturnElseThrowRuntimeError<string>(
83  key_xtpdft + ".externaldensity.state");
84  }
85 
86  if (options.exists(key_xtpdft + ".externalfield")) {
87  _integrate_ext_field = true;
88 
89  _extfield = options.ifExistsReturnElseThrowRuntimeError<Eigen::Vector3d>(
90  key_xtpdft + ".externalfield");
91  }
92 
93  if (options.exists(key_xtpdft + ".convergence")) {
94  _conv_opt.Econverged =
95  options.get(key_xtpdft + ".convergence.energy").as<double>();
96  _conv_opt.error_converged =
97  options.get(key_xtpdft + ".convergence.error").as<double>();
98  _max_iter =
99  options.get(key_xtpdft + ".convergence.max_iterations").as<Index>();
100 
101  string method =
102  options.get(key_xtpdft + ".convergence.method").as<string>();
103  if (method == "DIIS") {
104  _conv_opt.usediis = true;
105  } else if (method == "mixing") {
106  _conv_opt.usediis = false;
107  }
108  if (!_conv_opt.usediis) {
109  _conv_opt.histlength = 1;
110  _conv_opt.maxout = false;
111  }
112  _conv_opt.mixingparameter =
113  options.get(key_xtpdft + ".convergence.mixing").as<double>();
114  _conv_opt.levelshift =
115  options.get(key_xtpdft + ".convergence.levelshift").as<double>();
116  _conv_opt.levelshiftend =
117  options.get(key_xtpdft + ".convergence.levelshift_end").as<double>();
118  _conv_opt.maxout =
119  options.get(key_xtpdft + ".convergence.DIIS_maxout").as<bool>();
120  _conv_opt.histlength =
121  options.get(key_xtpdft + ".convergence.DIIS_length").as<Index>();
122  _conv_opt.diis_start =
123  options.get(key_xtpdft + ".convergence.DIIS_start").as<double>();
124  _conv_opt.adiis_start =
125  options.get(key_xtpdft + ".convergence.ADIIS_start").as<double>();
126  }
127 
128  return;
129 }

156 Mat_p_Energy DFTEngine::CalcEXXs(const Eigen::MatrixXd& MOCoeff,
157  const Eigen::MatrixXd& Dmat) const {
158  if (_four_center_method == "RI") {
159  if (_conv_accelerator.getUseMixing() || MOCoeff.rows() == 0) {
160  return _ERIs.CalculateEXX(Dmat);
161  } else {
162  Eigen::MatrixXd occblock =
163  MOCoeff.block(0, 0, MOCoeff.rows(), _numofelectrons / 2);
164  return _ERIs.CalculateEXX(occblock, Dmat);
165  }
166  } else {
167  if (_four_center_method == "direct") {
168  throw std::runtime_error(
169  "direct 4c method only works with LDA and GGA functionals.");
170  }
171  return _ERIs.CalculateEXX_4c_small_molecule(Dmat);
172  }

180 tools::EigenSystem DFTEngine::ModelPotentialGuess(
181  const Mat_p_Energy& H0, const QMMolecule& mol,
182  const Vxc_Potential<Vxc_Grid>& vxcpotential) const {
183  Eigen::MatrixXd Dmat = AtomicGuess(mol);
184  Mat_p_Energy ERIs = CalculateERIs(Dmat);
185  Mat_p_Energy e_vxc = vxcpotential.IntegrateVXC(Dmat);
186  XTP_LOG(Log::info, *_pLog)
187  << TimeStamp() << " Filled DFT Vxc matrix " << flush;
188 
189  Eigen::MatrixXd H = H0.matrix() + ERIs.matrix() + e_vxc.matrix();
190  if (_ScaHFX > 0) {
191  Mat_p_Energy EXXs = CalcEXXs(Eigen::MatrixXd::Zero(0, 0), Dmat);
192  H -= 0.5 * _ScaHFX * EXXs.matrix();
193  }
194  return _conv_accelerator.SolveFockmatrix(H);
195 }

197 bool DFTEngine::Evaluate(Orbitals& orb) {
198  Prepare(orb.QMAtoms());
199  Mat_p_Energy H0 = SetupH0(orb.QMAtoms());
200  tools::EigenSystem MOs;
201  MOs.eigenvalues() = Eigen::VectorXd::Zero(H0.cols());
202  MOs.eigenvectors() = Eigen::MatrixXd::Zero(H0.rows(), H0.cols());
203  Vxc_Potential<Vxc_Grid> vxcpotential = SetupVxc(orb.QMAtoms());
204  ConfigOrbfile(orb);
205 
206  if (_with_guess) {
207  XTP_LOG(Log::error, *_pLog)
208  << TimeStamp() << " Reading guess from orbitals object/file" << flush;
209  MOs = orb.MOs();
210  MOs.eigenvectors() = OrthogonalizeGuess(MOs.eigenvectors());
211  } else {
212  XTP_LOG(Log::error, *_pLog)
213  << TimeStamp() << " Setup Initial Guess using: " << _initial_guess
214  << flush;
215  if (_initial_guess == "independent") {
216  MOs = IndependentElectronGuess(H0);
217  } else if (_initial_guess == "atom") {
218  MOs = ModelPotentialGuess(H0, orb.QMAtoms(), vxcpotential);
219  } else {
220  throw std::runtime_error("Initial guess method not known/implemented");
221  }
222  }
223 
224  Eigen::MatrixXd Dmat = _conv_accelerator.DensityMatrix(MOs);
225  XTP_LOG(Log::info, *_pLog)
226  << TimeStamp() << " Guess Matrix gives N=" << std::setprecision(9)
227  << Dmat.cwiseProduct(_dftAOoverlap.Matrix()).sum() << " electrons."
228  << flush;
229 
230  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " STARTING SCF cycle" << flush;
231  XTP_LOG(Log::error, *_pLog)
232  << " ----------------------------------------------"
233  "----------------------------"
234  << flush;
235 
236  for (Index this_iter = 0; this_iter < _max_iter; this_iter++) {
237  XTP_LOG(Log::error, *_pLog) << flush;
238  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Iteration " << this_iter + 1
239  << " of " << _max_iter << flush;
240 
241  Mat_p_Energy e_vxc = vxcpotential.IntegrateVXC(Dmat);
242  XTP_LOG(Log::info, *_pLog)
243  << TimeStamp() << " Filled DFT Vxc matrix " << flush;
244 
245  Mat_p_Energy ERIs = CalculateERIs(Dmat);
246  Eigen::MatrixXd H = H0.matrix() + ERIs.matrix() + e_vxc.matrix();
247  double Eone = Dmat.cwiseProduct(H0.matrix()).sum();
248  double Etwo = 0.5 * ERIs.energy() + e_vxc.energy();
249  double exx = 0.0;
250  if (_ScaHFX > 0) {
251  Mat_p_Energy EXXs = CalcEXXs(MOs.eigenvectors(), Dmat);
252  XTP_LOG(Log::info, *_pLog)
253  << TimeStamp() << " Filled DFT Electron exchange matrix" << flush;
254  H -= 0.5 * _ScaHFX * EXXs.matrix();
255  exx = -_ScaHFX / 4 * EXXs.energy();
256  }
257  Etwo += exx;
258  double totenergy = Eone + H0.energy() + Etwo;
259  XTP_LOG(Log::info, *_pLog) << TimeStamp() << " Single particle energy "
260  << std::setprecision(12) << Eone << flush;
261  XTP_LOG(Log::info, *_pLog) << TimeStamp() << " Two particle energy "
262  << std::setprecision(12) << Etwo << flush;
263  XTP_LOG(Log::info, *_pLog)
264  << TimeStamp() << std::setprecision(12) << " Local Exc contribution "
265  << e_vxc.energy() << flush;
266  if (_ScaHFX > 0) {
267  XTP_LOG(Log::info, *_pLog)
268  << TimeStamp() << std::setprecision(12)
269  << " Non local Ex contribution " << exx << flush;
270  }
271  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Total Energy "
272  << std::setprecision(12) << totenergy << flush;
273 
274  Dmat = _conv_accelerator.Iterate(Dmat, H, MOs, totenergy);
275 
276  PrintMOs(MOs.eigenvalues(), Log::info);
277 
278  XTP_LOG(Log::info, *_pLog) << "\t\tGAP "
279  << MOs.eigenvalues()(_numofelectrons / 2) -
280  MOs.eigenvalues()(_numofelectrons / 2 - 1)
281  << flush;
282 
283  if (_conv_accelerator.isConverged()) {
284  XTP_LOG(Log::error, *_pLog)
285  << TimeStamp() << " Total Energy has converged to "
286  << std::setprecision(9) << _conv_accelerator.getDeltaE()
287  << "[Ha] after " << this_iter + 1
288  << " iterations. DIIS error is converged up to "
289  << _conv_accelerator.getDIIsError() << flush;
290  XTP_LOG(Log::error, *_pLog)
291  << TimeStamp() << " Final Single Point Energy "
292  << std::setprecision(12) << totenergy << " Ha" << flush;
293  PrintMOs(MOs.eigenvalues(), Log::error);
294  orb.setQMEnergy(totenergy);
295  orb.MOs() = MOs;
296  CalcElDipole(orb);
297  break;
298  } else if (this_iter == _max_iter - 1) {
299  XTP_LOG(Log::error, *_pLog)
300  << TimeStamp() << " DFT calculation has not converged after "
301  << _max_iter
302  << " iterations. Use more iterations or another convergence "
303  "acceleration scheme."
304  << std::flush;
305  return false;
306  }
307  }
308  return true;
309 }

311 Mat_p_Energy DFTEngine::SetupH0(const QMMolecule& mol) const {
312 
313  AOKinetic dftAOkinetic;
314 
315  dftAOkinetic.Fill(_dftbasis);
316  XTP_LOG(Log::info, *_pLog)
317  << TimeStamp() << " Filled DFT Kinetic energy matrix ." << flush;
318 
319  AOMultipole dftAOESP;
320  dftAOESP.FillPotential(_dftbasis, mol);
321  XTP_LOG(Log::info, *_pLog)
322  << TimeStamp() << " Filled DFT nuclear potential matrix." << flush;
323 
324  Eigen::MatrixXd H0 = dftAOkinetic.Matrix() + dftAOESP.Matrix();
325  XTP_LOG(Log::error, *_pLog)
326  << TimeStamp() << " Constructed independent particle hamiltonian "
327  << flush;
328  double E0 = NuclearRepulsion(mol);
329  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Nuclear Repulsion Energy is "
330  << std::setprecision(9) << E0 << flush;
331 
332  if (_with_ecp) {
333  AOECP dftAOECP;
334  dftAOECP.FillPotential(_dftbasis, _ecp);
335  H0 += dftAOECP.Matrix();
336  XTP_LOG(Log::info, *_pLog)
337  << TimeStamp() << " Filled DFT ECP matrix" << flush;
338  }
339 
340  if (_addexternalsites) {
341  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " " << _externalsites->size()
342  << " External sites" << flush;
343  if (_externalsites->size() < 200) {
344  XTP_LOG(Log::error, *_pLog)
345  << " Name Coordinates[a0] charge[e] dipole[e*a0] "
346  " quadrupole[e*a0^2] "
347  << flush;
348 
349  for (const std::unique_ptr<StaticSite>& site : *_externalsites) {
350  std::string output =
351  (boost::format(" %1$s"
352  " %2$+1.4f %3$+1.4f %4$+1.4f"
353  " %5$+1.4f") %
354  site->getElement() % site->getPos()[0] % site->getPos()[1] %
355  site->getPos()[2] % site->getCharge())
356  .str();
357  const Eigen::Vector3d& dipole = site->getDipole();
358  output += (boost::format(" %1$+1.4f %2$+1.4f %3$+1.4f") % dipole[0] %
359  dipole[1] % dipole[2])
360  .str();
361  if (site->getRank() > 1) {
362  Eigen::VectorXd quadrupole = site->Q().tail<5>();
363  output += (boost::format(
364  " %1$+1.4f %2$+1.4f %3$+1.4f %4$+1.4f %5$+1.4f") %
365  quadrupole[0] % quadrupole[1] % quadrupole[2] %
366  quadrupole[3] % quadrupole[4])
367  .str();
368  }
369  XTP_LOG(Log::error, *_pLog) << output << flush;
370  }
371  }
372 
373  Mat_p_Energy ext_multipoles =
374  IntegrateExternalMultipoles(mol, *_externalsites);
375  XTP_LOG(Log::error, *_pLog)
376  << TimeStamp() << " Nuclei-external site interaction energy "
377  << std::setprecision(9) << ext_multipoles.energy() << flush;
378  E0 += ext_multipoles.energy();
379  H0 += ext_multipoles.matrix();
380  }
381 
382  if (_integrate_ext_density) {
383  Orbitals extdensity;
384  extdensity.ReadFromCpt(_orbfilename);
385  Mat_p_Energy extdensity_result = IntegrateExternalDensity(mol, extdensity);
386  E0 += extdensity_result.energy();
387  XTP_LOG(Log::error, *_pLog)
388  << TimeStamp() << " Nuclei- external density interaction energy "
389  << std::setprecision(9) << extdensity_result.energy() << flush;
390  H0 += extdensity_result.matrix();
391  }
392 
393  if (_integrate_ext_field) {
394 
395  XTP_LOG(Log::error, *_pLog)
396  << TimeStamp() << " Integrating external electric field with F[Hrt]="
397  << _extfield.transpose() << flush;
398  H0 += IntegrateExternalField(mol);
399  }
400 
401  return Mat_p_Energy(E0, H0);
402 }

405

427

429 
430  XTP_LOG(Log::info, *_pLog)

442  << TimeStamp() << " Calculated 4c diagonals. " << flush;
443  }
444  }
445 
446  return;

449 Eigen::MatrixXd DFTEngine::RunAtomicDFT_unrestricted(
450  const QMAtom& uniqueAtom) const {
451  bool with_ecp = _with_ecp;
452  if (uniqueAtom.getElement() == "H" || uniqueAtom.getElement() == "He") {
453  with_ecp = false;
454  }
455 
456  QMMolecule atom = QMMolecule("individual_atom", 0);
457  atom.push_back(uniqueAtom);
458 
459  BasisSet basisset;
460  basisset.Load(_dftbasis_name);
461  AOBasis dftbasis;
462  dftbasis.Fill(basisset, atom);
463  Vxc_Grid grid;
464  grid.GridSetup(_grid_name, atom, dftbasis);
465  Vxc_Potential<Vxc_Grid> gridIntegration(grid);
466  gridIntegration.setXCfunctional(_xc_functional_name);
467 
468  ECPAOBasis ecp;
469  if (with_ecp) {
470  ECPBasisSet ecps;
471  ecps.Load(_ecp_name);
472  ecp.Fill(ecps, atom);
473  }
474 
475  Index numofelectrons = uniqueAtom.getNuccharge();
476  Index alpha_e = 0;
477  Index beta_e = 0;
478 
479  if ((numofelectrons % 2) != 0) {
480  alpha_e = numofelectrons / 2 + numofelectrons % 2;
481  beta_e = numofelectrons / 2;
482  } else {
483  alpha_e = numofelectrons / 2;
484  beta_e = alpha_e;
485  }
486 
487  AOOverlap dftAOoverlap;
488  AOKinetic dftAOkinetic;
489  AOMultipole dftAOESP;
490  AOECP dftAOECP;
491  ERIs ERIs_atom;
492 
493  // DFT AOOverlap matrix
494 
495  dftAOoverlap.Fill(dftbasis);
496  dftAOkinetic.Fill(dftbasis);
497 
498  dftAOESP.FillPotential(dftbasis, atom);
499  ERIs_atom.Initialize_4c_small_molecule(dftbasis);
500 
501  ConvergenceAcc Convergence_alpha;
502  ConvergenceAcc Convergence_beta;
503  ConvergenceAcc::options opt_alpha = _conv_opt;
504  opt_alpha.mode = ConvergenceAcc::KSmode::open;
505  opt_alpha.histlength = 20;
506  opt_alpha.levelshift = 0.1;
507  opt_alpha.levelshiftend = 0.0;
508  opt_alpha.usediis = true;
509  opt_alpha.adiis_start = 0.0;
510  opt_alpha.diis_start = 0.0;
511  opt_alpha.numberofelectrons = alpha_e;
512 
513  ConvergenceAcc::options opt_beta = opt_alpha;
514  opt_beta.numberofelectrons = beta_e;
515 
516  Logger log;
517  Convergence_alpha.Configure(opt_alpha);
518  Convergence_alpha.setLogger(&log);
519  Convergence_alpha.setOverlap(dftAOoverlap, 1e-8);
520  Convergence_beta.Configure(opt_beta);
521  Convergence_beta.setLogger(&log);
522  Convergence_beta.setOverlap(dftAOoverlap, 1e-8);
523  /**** Construct initial density ****/
524 
525  Eigen::MatrixXd H0 = dftAOkinetic.Matrix() + dftAOESP.Matrix();
526  if (with_ecp) {
527  dftAOECP.FillPotential(dftbasis, ecp);
528  H0 += dftAOECP.Matrix();
529  }
530  tools::EigenSystem MOs_alpha = Convergence_alpha.SolveFockmatrix(H0);
531 
532  Eigen::MatrixXd dftAOdmat_alpha = Convergence_alpha.DensityMatrix(MOs_alpha);
533  if (uniqueAtom.getElement() == "H") {
534  return dftAOdmat_alpha;
535  }
536  tools::EigenSystem MOs_beta = Convergence_beta.SolveFockmatrix(H0);
537  Eigen::MatrixXd dftAOdmat_beta = Convergence_beta.DensityMatrix(MOs_beta);
538 
539  Index maxiter = 80;
540  for (Index this_iter = 0; this_iter < maxiter; this_iter++) {
541  Mat_p_Energy ERIs = ERIs_atom.CalculateERIs_4c_small_molecule(
542  dftAOdmat_alpha + dftAOdmat_beta);
543  double E_two_alpha = ERIs.matrix().cwiseProduct(dftAOdmat_alpha).sum();
544  double E_two_beta = ERIs.matrix().cwiseProduct(dftAOdmat_beta).sum();
545  Eigen::MatrixXd H_alpha = H0 + ERIs.matrix();
546  Eigen::MatrixXd H_beta = H0 + ERIs.matrix();
547 
548  Mat_p_Energy e_vxc = gridIntegration.IntegrateVXC(dftAOdmat_alpha);
549  Eigen::MatrixXd AOVxc_alpha = e_vxc.matrix();
550  double E_vxc_alpha = e_vxc.energy();
551  H_alpha += AOVxc_alpha;
552  E_two_alpha += E_vxc_alpha;
553 
554  e_vxc = gridIntegration.IntegrateVXC(dftAOdmat_beta);
555  Eigen::MatrixXd AOVxc_beta = e_vxc.matrix();
556  double E_vxc_beta = e_vxc.energy();
557  H_beta += AOVxc_beta;
558  E_two_beta += E_vxc_beta;
559 
560  if (_ScaHFX > 0) {
561  Mat_p_Energy EXXs =
562  ERIs_atom.CalculateEXX_4c_small_molecule(dftAOdmat_alpha);
563  double E_exx_alpha =
564  -0.5 * _ScaHFX * EXXs.matrix().cwiseProduct(dftAOdmat_alpha).sum();
565  H_alpha -= _ScaHFX * EXXs.matrix();
566  E_two_alpha += E_exx_alpha;
567  ERIs_atom.CalculateEXX_4c_small_molecule(dftAOdmat_beta);
568  double E_exx_beta =
569  -0.5 * _ScaHFX * EXXs.matrix().cwiseProduct(dftAOdmat_beta).sum();
570  H_beta -= _ScaHFX * EXXs.matrix();
571  E_two_beta += E_exx_beta;
572  }
573 
574  double E_one_alpha = dftAOdmat_alpha.cwiseProduct(H0).sum();
575  double E_one_beta = dftAOdmat_beta.cwiseProduct(H0).sum();
576  double E_alpha = E_one_alpha + E_two_alpha;
577  double E_beta = E_one_beta + E_two_beta;
578  double totenergy = E_alpha + E_beta;
579  // evolve alpha
580  dftAOdmat_alpha =
581  Convergence_alpha.Iterate(dftAOdmat_alpha, H_alpha, MOs_alpha, E_alpha);
582  // evolve beta
583  dftAOdmat_beta =
584  Convergence_beta.Iterate(dftAOdmat_beta, H_beta, MOs_beta, E_beta);
585 
586  XTP_LOG(Log::debug, *_pLog)
587  << TimeStamp() << " Iter " << this_iter << " of " << maxiter << " Etot "
588  << totenergy << " diise_a " << Convergence_alpha.getDIIsError()
589  << " diise_b " << Convergence_beta.getDIIsError() << "\n\t\t a_gap "
590  << MOs_alpha.eigenvalues()(alpha_e) -
591  MOs_alpha.eigenvalues()(alpha_e - 1)
592  << " b_gap "
593  << MOs_beta.eigenvalues()(beta_e) - MOs_beta.eigenvalues()(beta_e - 1)
594  << " Nalpha="
595  << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_alpha).sum()
596  << " Nbeta=" << dftAOoverlap.Matrix().cwiseProduct(dftAOdmat_beta).sum()
597  << flush;
598 
599  bool converged =
600  Convergence_alpha.isConverged() && Convergence_beta.isConverged();
601  if (converged || this_iter == maxiter - 1) {
602 
603  if (converged) {
604  XTP_LOG(Log::info, *_pLog) << TimeStamp() << " Converged after "
605  << this_iter + 1 << " iterations" << flush;
606  } else {
607  XTP_LOG(Log::info, *_pLog)
608  << TimeStamp() << " Not converged after " << this_iter + 1
609  << " iterations. Unconverged density.\n\t\t\t"
610  << " DIIsError_alpha=" << Convergence_alpha.getDIIsError()
611  << " DIIsError_beta=" << Convergence_beta.getDIIsError() << flush;
612  }
613  break;
614  }
615  }
616  Eigen::MatrixXd avgmatrix =
617  SphericalAverageShells(dftAOdmat_alpha + dftAOdmat_beta, dftbasis);
618  XTP_LOG(Log::info, *_pLog)
619  << TimeStamp() << " Atomic density Matrix for " << uniqueAtom.getElement()
620  << " gives N=" << std::setprecision(9)
621  << avgmatrix.cwiseProduct(dftAOoverlap.Matrix()).sum() << " electrons."
622  << flush;
623  return avgmatrix;

834 // average atom densities matrices, for SP and other combined shells average
835 // each subshell separately.
836 Eigen::MatrixXd DFTEngine::SphericalAverageShells(
837  const Eigen::MatrixXd& dmat, const AOBasis& dftbasis) const {
838  Eigen::MatrixXd avdmat = Eigen::MatrixXd::Zero(dmat.rows(), dmat.cols());
839  Index start = 0.0;
840  std::vector<Index> starts;
841  std::vector<Index> ends;
842  for (const AOShell& shell : dftbasis) {
843  Index end = shell.getNumFunc() + start;
844 
845  if (shell.isCombined()) {
846  std::vector<Index> temp = NumFuncSubShell(shell.getType());
847  Index numfunc = start;
848  for (Index& SubshellFunc : temp) {
849  starts.push_back(numfunc);
850  numfunc += SubshellFunc;
851  ends.push_back(numfunc);
852  }
853  } else {
854  starts.push_back(start);
855  ends.push_back(end);
856  }
857  start = end;
858  }
859  for (Index k = 0; k < Index(starts.size()); k++) {
860  Index s1 = starts[k];
861  Index e1 = ends[k];
862  Index len1 = e1 - s1;
863  for (Index l = 0; l < Index(starts.size()); l++) {
864  Index s2 = starts[l];
865  Index e2 = ends[l];
866  Index len2 = e2 - s2;
867  double diag = 0.0;
868  double offdiag = 0.0;
869  for (Index i = 0; i < len1; ++i) {
870  for (Index j = 0; j < len2; ++j) {
871  if (i == j) {
872  diag += dmat(s1 + i, s2 + j);
873  } else {
874  offdiag += dmat(s1 + i, s2 + j);
875  }
876  }
877  }
878  if (len1 == len2) {
879  diag = diag / double(len1);
880  offdiag = offdiag / double(len1 * (len1 - 1));
881  } else {
882  double avg = (diag + offdiag) / double(len1 * len2);
883  diag = avg;
884  offdiag = avg;
885  }
886  for (Index i = 0; i < len1; ++i) {
887  for (Index j = 0; j < len2; ++j) {
888  if (i == j) {
889  avdmat(s1 + i, s2 + j) = diag;
890  } else {
891  avdmat(s1 + i, s2 + j) = offdiag;
892  }
893  }
894  }
895  }
896  }
897  return avdmat;
898 }
