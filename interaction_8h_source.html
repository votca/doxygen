210 inline vec IAngle::Grad(const Topology &top, int bead) {
211  vec v1(top.getDist(_beads[1], _beads[0]));
212  vec v2(top.getDist(_beads[1], _beads[2]));
213 
214  double acos_prime =
215  1.0 /
216  (sqrt(1 -
217  (v1 * v2) * (v1 * v2) / (abs(v1) * abs(v2) * abs(v1) * abs(v2))));
218  switch (bead) {
219  case (0):
220  return acos_prime *
221  (-v2 / (abs(v1) * abs(v2)) +
222  (v1 * v2) * v1 / (abs(v2) * abs(v1) * abs(v1) * abs(v1)));
223  break;
224  case (1):
225  return acos_prime *
226  ((v1 + v2) / (abs(v1) * abs(v2)) -
227  (v1 * v2) * ((v2 * v2) * v1 + (v1 * v1) * v2) /
228  (abs(v1) * abs(v1) * abs(v1) * abs(v2) * abs(v2) * abs(v2)));
229  break;
230  case (2):
231  return acos_prime *
232  (-v1 / (abs(v1) * abs(v2)) +
233  (v1 * v2) * v2 / (abs(v1) * abs(v2) * abs(v2) * abs(v2)));
234  break;
235  }
236  // should never reach this
237  assert(false);
238  return vec(0, 0, 0);
239 }

241 inline double IDihedral::EvaluateVar(const Topology &top) {
242  vec v1(top.getDist(_beads[0], _beads[1]));
243  vec v2(top.getDist(_beads[1], _beads[2]));
244  vec v3(top.getDist(_beads[2], _beads[3]));
245  vec n1, n2;
246  n1 = v1 ^ v2; // calculate the normal vector
247  n2 = v2 ^ v3; // calculate the normal vector
248  double sign = (v1 * n2 < 0) ? -1 : 1;
249  return sign * acos(n1 * n2 / sqrt((n1 * n1) * (n2 * n2)));
250 }

252 inline vec IDihedral::Grad(const Topology &top, int bead) {
253  vec v1(top.getDist(_beads[0], _beads[1]));
254  vec v2(top.getDist(_beads[1], _beads[2]));
255  vec v3(top.getDist(_beads[2], _beads[3]));
256  vec n1, n2;
257  n1 = v1 ^ v2; // calculate the normal vector
258  n2 = v2 ^ v3; // calculate the normal vector
259  double sign = (v1 * n2 < 0) ? -1 : 1;
260  vec returnvec; // vector to return
261  double returnvec0, returnvec1, returnvec2; // components of the return vector
262  vec e0(1, 0, 0); // unit vector pointing in x-direction
263  vec e1(0, 1, 0); // unit vector pointing in y-direction
264  vec e2(0, 0, 1); // unit vector pointing in z-direction
265 
266  double acos_prime =
267  (-1.0 / (sqrt(1 -
268  (n1 * n2) * (n1 * n2) /
269  (abs(n1) * abs(n2) * abs(n1) * abs(n2))))) *
270  sign;
271  switch (bead) {
272  case (0): { //
273  returnvec0 = acos_prime * ((n2 * (v2 ^ e0)) / (abs(n1) * abs(n2)) -
274  ((n1 * n2) * (n1 * (v2 ^ e0))) /
275  (abs(n1) * abs(n1) * abs(n1) * abs(n2)));
276  returnvec1 = acos_prime * ((n2 * (v2 ^ e1)) / (abs(n1) * abs(n2)) -
277  ((n1 * n2) * (n1 * (v2 ^ e1))) /
278  (abs(n1) * abs(n1) * abs(n1) * abs(n2)));
279  returnvec2 = acos_prime * ((n2 * (v2 ^ e2)) / (abs(n1) * abs(n2)) -
280  ((n1 * n2) * (n1 * (v2 ^ e2))) /
281  (abs(n1) * abs(n1) * abs(n1) * abs(n2)));
282  returnvec.setX(returnvec0);
283  returnvec.setY(returnvec1);
284  returnvec.setZ(returnvec2);
285  return returnvec;
286  break;
287  }
288  case (1): { //
289  returnvec0 =
290  acos_prime *
291  ((n1 * (v3 ^ e0) + n2 * ((e0 ^ v1) + (e0 ^ v2))) / (abs(n1) * abs(n2)) -
292  ((n1 * n2) *
293  ((n1 * ((e0 ^ v1) + (e0 ^ v2))) /
294  (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
295  (n2 * (v3 ^ e0)) / (abs(n1) * abs(n2) * abs(n2) * abs(n2)))));
296  returnvec1 =
297  acos_prime *
298  ((n1 * (v3 ^ e1) + n2 * ((e1 ^ v1) + (e1 ^ v2))) / (abs(n1) * abs(n2)) -
299  ((n1 * n2) *
300  ((n1 * ((e1 ^ v1) + (e1 ^ v2))) /
301  (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
302  (n2 * (v3 ^ e1)) / (abs(n1) * abs(n2) * abs(n2) * abs(n2)))));
303  returnvec2 =
304  acos_prime *
305  ((n1 * (v3 ^ e2) + n2 * ((e2 ^ v1) + (e2 ^ v2))) / (abs(n1) * abs(n2)) -
306  ((n1 * n2) *
307  ((n1 * ((e2 ^ v1) + (e2 ^ v2))) /
308  (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
309  (n2 * (v3 ^ e2)) / (abs(n1) * abs(n2) * abs(n2) * abs(n2)))));
310  returnvec.setX(returnvec0);
311  returnvec.setY(returnvec1);
312  returnvec.setZ(returnvec2);
313  return returnvec;
314  break;
315  };
316  case (2): { //
317  returnvec0 =
318  acos_prime *
319  ((n1 * ((e0 ^ v2) + (e0 ^ v3)) + n2 * (v1 ^ e0)) / (abs(n1) * abs(n2)) -
320  ((n1 * n2) *
321  ((n1 * (v1 ^ e0)) / (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
322  (n2 * ((e0 ^ v2) + (e0 ^ v3))) /
323  (abs(n1) * abs(n2) * abs(n2) * abs(n2)))));
324  returnvec1 =
325  acos_prime *
326  ((n1 * ((e1 ^ v2) + (e1 ^ v3)) + n2 * (v1 ^ e1)) / (abs(n1) * abs(n2)) -
327  ((n1 * n2) *
328  ((n1 * (v1 ^ e1)) / (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
329  (n2 * ((e1 ^ v2) + (e1 ^ v3))) /
330  (abs(n1) * abs(n2) * abs(n2) * abs(n2)))));
331  returnvec2 =
332  acos_prime *
333  ((n1 * ((e2 ^ v2) + (e2 ^ v3)) + n2 * (v1 ^ e2)) / (abs(n1) * abs(n2)) -
334  ((n1 * n2) *
335  ((n1 * (v1 ^ e2)) / (abs(n1) * abs(n1) * abs(n1) * abs(n2)) +
336  (n2 * ((e2 ^ v2) + (e2 ^ v3))) /
