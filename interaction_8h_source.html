208 inline Eigen::Vector3d IAngle::Grad(const Topology &top, int bead) {
209  Eigen::Vector3d v1(top.getDist(_beads[1], _beads[0]));
210  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
211 
212  double acos_prime =
213  1.0 / (sqrt(1 - std::pow(v1.dot(v2), 2) /
214  (v1.squaredNorm() * v2.squaredNorm())));
215  switch (bead) {
216  case (0):
217  return acos_prime *
218  (-v2 / (v1.norm() * v2.norm()) +
219  (v1.dot(v2) * v1) / (v1.squaredNorm() * v2.squaredNorm()));
220  break;
221  case (1):
222  return acos_prime *
223  ((v1 + v2) / (v1.norm() * v2.norm()) -
224  (v1.dot(v2)) * (v2.squaredNorm() * v1 + v1.squaredNorm() * v2) /
225  (std::pow(v1.norm(), 3) * std::pow(v2.norm(), 3)));
226  break;
227  case (2):
228  return acos_prime * (-v1 / (v1.norm() * v2.norm())) +
229  (v1.dot(v2) * v2 / (v1.norm() * std::pow(v2.norm(), 3)));
230  break;
231  }
232  // should never reach this
233  assert(false);
234  return Eigen::Vector3d::Zero();
235 }

237 inline double IDihedral::EvaluateVar(const Topology &top) {
238  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
239  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
240  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
241  Eigen::Vector3d n1 = v1.cross(v2); // calculate the normal vector
242  Eigen::Vector3d n2 = v2.cross(v3); // calculate the normal vector
243  double sign = (v1.dot(n2) < 0) ? -1 : 1;
244  return sign *
245  std::acos(n1.dot(n2) / sqrt(n1.squaredNorm() * n2.squaredNorm()));
246 }

248 inline Eigen::Vector3d IDihedral::Grad(const Topology &top, int bead) {
249  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
250  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
251  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
252  Eigen::Vector3d n1, n2;
253  n1 = v1.cross(v2); // calculate the normal vector
254  n2 = v2.cross(v3); // calculate the normal vector
255  double sign = (v1.dot(n2) < 0) ? -1 : 1;
256  Eigen::Vector3d returnvec;
257 
258  Eigen::Matrix3d e = Eigen::Matrix3d::Identity();
259 
260  double acos_prime =
261  sign * (-1.0 / (sqrt(1 - std::pow(n1.dot(n2), 2) /
262  (n1.squaredNorm() * n2.squaredNorm()))));
263  switch (bead) {
264  case (0): {
265  for (int i = 0; i < 3; i++) {
266  returnvec[i] = n2.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
267  n1.dot(n2) * n1.dot(v2.cross(e.col(i))) /
268  (n2.norm() * std::pow(n1.norm(), 3));
269  }
270  return acos_prime * returnvec;
271  break;
272  }
273  case (1): {
274  for (int i = 0; i < 3; i++) {
275  returnvec[i] =
276  (n1.dot(v3.cross(e.col(i))) +
277  n2.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
278  (n1.norm() * n2.norm()) -
279  n1.dot(n2) * ((n1.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
280  (n2.norm() * std::pow(n1.norm(), 3)) +
281  n2.dot(v3.cross(e.col(i))) /
282  (n1.norm() * std::pow(n2.norm(), 3)));
283  }
284  return acos_prime * returnvec;
285  break;
286  };
287  case (2): {
288  for (int i = 0; i < 3; i++) {
289  returnvec[i] =
290  (n1.dot(e.col(i).cross(v2) + e.col(i).cross(v3)) +
291  n2.dot(v1.cross(e.col(i)))) /
292  (n1.norm() * n2.norm()) -
293  n1.dot(n2) * (n1.dot(v1.cross(e.col(i))) /
294  (n2.norm() * std::pow(n1.norm(), 3)) +
295  (n2.dot(e.col(i).cross(v2) + e.col(i).cross(v3))) /
296  (n1.norm() * std::pow(n2.norm(), 3)));
297  }
298  return acos_prime * returnvec;
299  break;
300  };
301  case (3): { //
302  for (int i = 0; i < 3; i++) {
303  returnvec[i] = n1.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
304  n1.dot(n2) * n2.dot(v2.cross(e.col(i))) /
305  (n1.norm() * std::pow(n2.norm(), 3));
306  }
307  return acos_prime * returnvec;
308  break;
309  };
310  }
311  // should never reach this
312  assert(false);
313  return Eigen::Vector3d::Zero();
