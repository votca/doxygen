215 inline Eigen::Vector3d IAngle::Grad(const Topology &top, Index bead) {
216  Eigen::Vector3d v1(top.getDist(_beads[1], _beads[0]));
217  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
218 
219  double acos_prime =
220  1.0 / (sqrt(1 - std::pow(v1.dot(v2), 2) /
221  (v1.squaredNorm() * v2.squaredNorm())));
222  switch (bead) {
223  case (0):
224  return acos_prime *
225  (-v2 / (v1.norm() * v2.norm()) +
226  (v1.dot(v2) * v1) / (v1.squaredNorm() * v2.squaredNorm()));
227  break;
228  case (1):
229  return acos_prime *
230  ((v1 + v2) / (v1.norm() * v2.norm()) -
231  (v1.dot(v2)) * (v2.squaredNorm() * v1 + v1.squaredNorm() * v2) /
232  (std::pow(v1.norm(), 3) * std::pow(v2.norm(), 3)));
233  break;
234  case (2):
235  return acos_prime * (-v1 / (v1.norm() * v2.norm())) +
236  (v1.dot(v2) * v2 / (v1.norm() * std::pow(v2.norm(), 3)));
237  break;
238  }
239  // should never reach this
240  assert(false);
241  return Eigen::Vector3d::Zero();
242 }

244 inline double IDihedral::EvaluateVar(const Topology &top) {
245  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
246  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
247  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
248  Eigen::Vector3d n1 = v1.cross(v2); // calculate the normal vector
249  Eigen::Vector3d n2 = v2.cross(v3); // calculate the normal vector
250  double sign = (v1.dot(n2) < 0) ? -1 : 1;
251  return sign *
252  std::acos(n1.dot(n2) / sqrt(n1.squaredNorm() * n2.squaredNorm()));
253 }

255 inline Eigen::Vector3d IDihedral::Grad(const Topology &top, Index bead) {
256  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
257  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
258  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
259  Eigen::Vector3d n1, n2;
260  n1 = v1.cross(v2); // calculate the normal vector
261  n2 = v2.cross(v3); // calculate the normal vector
262  double sign = (v1.dot(n2) < 0) ? -1 : 1;
263  Eigen::Vector3d returnvec;
264 
265  Eigen::Matrix3d e = Eigen::Matrix3d::Identity();
266 
267  double acos_prime =
268  sign * (-1.0 / (sqrt(1 - std::pow(n1.dot(n2), 2) /
269  (n1.squaredNorm() * n2.squaredNorm()))));
270  switch (bead) {
271  case (0): {
272  for (Index i = 0; i < 3; i++) {
273  returnvec[i] = n2.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
274  n1.dot(n2) * n1.dot(v2.cross(e.col(i))) /
275  (n2.norm() * std::pow(n1.norm(), 3));
276  }
277  return acos_prime * returnvec;
278  break;
279  }
280  case (1): {
281  for (Index i = 0; i < 3; i++) {
282  returnvec[i] =
283  (n1.dot(v3.cross(e.col(i))) +
284  n2.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
285  (n1.norm() * n2.norm()) -
286  n1.dot(n2) * ((n1.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
287  (n2.norm() * std::pow(n1.norm(), 3)) +
288  n2.dot(v3.cross(e.col(i))) /
289  (n1.norm() * std::pow(n2.norm(), 3)));
290  }
291  return acos_prime * returnvec;
292  break;
293  };
294  case (2): {
295  for (Index i = 0; i < 3; i++) {
296  returnvec[i] =
297  (n1.dot(e.col(i).cross(v2) + e.col(i).cross(v3)) +
298  n2.dot(v1.cross(e.col(i)))) /
299  (n1.norm() * n2.norm()) -
300  n1.dot(n2) * (n1.dot(v1.cross(e.col(i))) /
301  (n2.norm() * std::pow(n1.norm(), 3)) +
302  (n2.dot(e.col(i).cross(v2) + e.col(i).cross(v3))) /
303  (n1.norm() * std::pow(n2.norm(), 3)));
304  }
305  return acos_prime * returnvec;
306  break;
307  };
308  case (3): { //
309  for (Index i = 0; i < 3; i++) {
310  returnvec[i] = n1.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
311  n1.dot(n2) * n2.dot(v2.cross(e.col(i))) /
312  (n1.norm() * std::pow(n2.norm(), 3));
313  }
314  return acos_prime * returnvec;
315  break;
316  };
317  }
318  // should never reach this
319  assert(false);
320  return Eigen::Vector3d::Zero();
321 }
