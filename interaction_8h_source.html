214 inline Eigen::Vector3d IAngle::Grad(const Topology &top, long int bead) {
215  Eigen::Vector3d v1(top.getDist(_beads[1], _beads[0]));
216  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
217 
218  double acos_prime =
219  1.0 / (sqrt(1 - std::pow(v1.dot(v2), 2) /
220  (v1.squaredNorm() * v2.squaredNorm())));
221  switch (bead) {
222  case (0):
223  return acos_prime *
224  (-v2 / (v1.norm() * v2.norm()) +
225  (v1.dot(v2) * v1) / (v1.squaredNorm() * v2.squaredNorm()));
226  break;
227  case (1):
228  return acos_prime *
229  ((v1 + v2) / (v1.norm() * v2.norm()) -
230  (v1.dot(v2)) * (v2.squaredNorm() * v1 + v1.squaredNorm() * v2) /
231  (std::pow(v1.norm(), 3) * std::pow(v2.norm(), 3)));
232  break;
233  case (2):
234  return acos_prime * (-v1 / (v1.norm() * v2.norm())) +
235  (v1.dot(v2) * v2 / (v1.norm() * std::pow(v2.norm(), 3)));
236  break;
237  }
238  // should never reach this
239  assert(false);
240  return Eigen::Vector3d::Zero();
241 }

243 inline double IDihedral::EvaluateVar(const Topology &top) {
244  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
245  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
246  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
247  Eigen::Vector3d n1 = v1.cross(v2); // calculate the normal vector
248  Eigen::Vector3d n2 = v2.cross(v3); // calculate the normal vector
249  double sign = (v1.dot(n2) < 0) ? -1 : 1;
250  return sign *
251  std::acos(n1.dot(n2) / sqrt(n1.squaredNorm() * n2.squaredNorm()));
252 }

254 inline Eigen::Vector3d IDihedral::Grad(const Topology &top, long int bead) {
255  Eigen::Vector3d v1(top.getDist(_beads[0], _beads[1]));
256  Eigen::Vector3d v2(top.getDist(_beads[1], _beads[2]));
257  Eigen::Vector3d v3(top.getDist(_beads[2], _beads[3]));
258  Eigen::Vector3d n1, n2;
259  n1 = v1.cross(v2); // calculate the normal vector
260  n2 = v2.cross(v3); // calculate the normal vector
261  double sign = (v1.dot(n2) < 0) ? -1 : 1;
262  Eigen::Vector3d returnvec;
263 
264  Eigen::Matrix3d e = Eigen::Matrix3d::Identity();
265 
266  double acos_prime =
267  sign * (-1.0 / (sqrt(1 - std::pow(n1.dot(n2), 2) /
268  (n1.squaredNorm() * n2.squaredNorm()))));
269  switch (bead) {
270  case (0): {
271  for (long int i = 0; i < 3; i++) {
272  returnvec[i] = n2.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
273  n1.dot(n2) * n1.dot(v2.cross(e.col(i))) /
274  (n2.norm() * std::pow(n1.norm(), 3));
275  }
276  return acos_prime * returnvec;
277  break;
278  }
279  case (1): {
280  for (long int i = 0; i < 3; i++) {
281  returnvec[i] =
282  (n1.dot(v3.cross(e.col(i))) +
283  n2.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
284  (n1.norm() * n2.norm()) -
285  n1.dot(n2) * ((n1.dot(e.col(i).cross(v1) + e.col(i).cross(v2))) /
286  (n2.norm() * std::pow(n1.norm(), 3)) +
287  n2.dot(v3.cross(e.col(i))) /
288  (n1.norm() * std::pow(n2.norm(), 3)));
289  }
290  return acos_prime * returnvec;
291  break;
292  };
293  case (2): {
294  for (long int i = 0; i < 3; i++) {
295  returnvec[i] =
296  (n1.dot(e.col(i).cross(v2) + e.col(i).cross(v3)) +
297  n2.dot(v1.cross(e.col(i)))) /
298  (n1.norm() * n2.norm()) -
299  n1.dot(n2) * (n1.dot(v1.cross(e.col(i))) /
300  (n2.norm() * std::pow(n1.norm(), 3)) +
301  (n2.dot(e.col(i).cross(v2) + e.col(i).cross(v3))) /
302  (n1.norm() * std::pow(n2.norm(), 3)));
303  }
304  return acos_prime * returnvec;
305  break;
306  };
307  case (3): { //
308  for (long int i = 0; i < 3; i++) {
309  returnvec[i] = n1.dot(v2.cross(e.col(i))) / (n1.norm() * n2.norm()) -
310  n1.dot(n2) * n2.dot(v2.cross(e.col(i))) /
311  (n1.norm() * std::pow(n2.norm(), 3));
312  }
313  return acos_prime * returnvec;
314  break;
315  };
316  }
317  // should never reach this
318  assert(false);
319  return Eigen::Vector3d::Zero();
320 }
