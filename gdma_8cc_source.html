72  void GDMA::RunExternal() {
73 
74 
75  // check if the input file exists
76  string fullInput = _runFolder + "/gdma.in";
77  if (!boost::filesystem::exists(fullInput)) {
78  CTP_LOG(ctp::logINFO, *_log) << "GDMA input file has not been found!" << flush;
79  throw runtime_error(" GDMA cannot be run! ");
80  }
81 
82  // check if fchk exists
83  string fullFChk = _runFolder + "/system.fchk";
84  if (!boost::filesystem::exists(fullFChk)) {
85  // try converting Chk to FChk
86  string fullChk = _runFolder + "/" + _chkFile;
87  if (boost::filesystem::exists(fullChk)) {
88  // use formchk
89  string command;
90  command = "cd " + _runFolder + "; formchk " + _chkFile + " system.fchk > /dev/null";
91  if (std::system(command.c_str())){
92  throw runtime_error("Command "+ command + "failed");
93  }
94  // check again for fchk
95  if (!boost::filesystem::exists(fullFChk)) {
96  CTP_LOG(ctp::logINFO, *_log) << "Formatted Checkpoint file has not been found and cannot be created!" << flush;
97  throw runtime_error(" GDMA cannot be run! ");
98 
99  }
100  } else {
101  CTP_LOG(ctp::logINFO, *_log) << "Formatted Checkpoint file has not been found and cannot be created!" << flush;
102  throw runtime_error(" GDMA cannot be run! ");
103  }
104  }
105 
106 
107  // now we seem ready to go
108  string command;
109  command = "cd " + _runFolder + "; " + _executable + " < gdma.in > " + _outFile;
110  if (std::system(command.c_str())){
111  throw runtime_error("Command "+ command + "failed");
112  }

116  void GDMA::ParseOutputFile() {
117 
118  string gdma_output_name_full = _runFolder + "/gdma.out";
119  std::ifstream gdma_output(gdma_output_name_full.c_str());
120  std::string line;
121  while (gdma_output) {
122 
123  getline(gdma_output, line);
124  // if a line has an equality sign, must be energy
125  std::string::size_type atom_pos = line.find("x =");
126  if (atom_pos != std::string::npos) {
127  // found an atom, read one more line
128  getline(gdma_output, line);
129  // determine rank
130  std::vector<string> results;
131  boost::trim(line);
132  std::vector<double> Qs; // temp vector for reading in
133 
134  boost::algorithm::split(results, line, boost::is_any_of("\t "),
135  boost::algorithm::token_compress_on);
136 
137  int rank = boost::lexical_cast<int>(results[3]);
138  if (rank < 0 || rank > 2) {
139  throw runtime_error((boost::format(" Invalid GDMA rank %s!") %rank).str());
140  }
141 
142  // atomic charge
143  if (rank >= 0) {
144  // getting charge
145  getline(gdma_output, line);
146  std::vector<string> results;
147  boost::trim(line);
148  boost::algorithm::split(results, line, boost::is_any_of("\t "),
149  boost::algorithm::token_compress_on);
150  double Q00 = boost::lexical_cast<double>(results.back());
151  Qs.push_back(Q00);
152  }
153 
154  // atomic dipole components
155  if (rank >= 1) {
156 
157  // getting dipoles
158  getline(gdma_output, line);
159  std::vector<string> results;
160  boost::trim(line);
161  boost::algorithm::split(results, line, boost::is_any_of("\t "),
162  boost::algorithm::token_compress_on);
163  double Q10 = boost::lexical_cast<double>(results[5]);
164  double Q11c = boost::lexical_cast<double>(results[8]);
165  double Q11s = boost::lexical_cast<double>(results.back());
166  Qs.push_back(Q10);
167  Qs.push_back(Q11c);
168  Qs.push_back(Q11s);
169  }
170 
171  // atomic quadrupole components
172  if (rank == 2) {
173 
174  getline(gdma_output, line);
175  std::vector<string> results;
176  boost::trim(line);
177  boost::algorithm::split(results, line, boost::is_any_of("\t "),
178  boost::algorithm::token_compress_on);
179  double Q20 = boost::lexical_cast<double>(results[5]);
180  double Q21c = boost::lexical_cast<double>(results[8]);
181  double Q21s = boost::lexical_cast<double>(results.back());
182  getline(gdma_output, line);
183  boost::trim(line);
184  boost::algorithm::split(results, line, boost::is_any_of("\t "),
185  boost::algorithm::token_compress_on);
186  double Q22c = boost::lexical_cast<double>(results[2]);
187  double Q22s = boost::lexical_cast<double>(results[5]);
188 
189  Qs.push_back(Q20);
190  Qs.push_back(Q21c);
191  Qs.push_back(Q21s);
192  Qs.push_back(Q22c);
193  Qs.push_back(Q22s);
194 
195  }
196  _multipoles.push_back( Qs );
197 
198  } // atom
199  std::string::size_type break_pos = line.find("Total multipoles referred to origin at");
200  if (break_pos != std::string::npos) break;
201  } // gdma_output
202  }
