40  void GDMA::Initialize(Property* options) {
41 
42  string key = "gdma";
43  if (options->exists(key + ".chk")) {
44  _chkFile = options->get(key + ".chk").as<string> ();
45  } else {
46  _chkFile = "system.chk";
47  }
48 
49  if (options->exists(key + ".executable")) {
50  _executable = options->get(key + ".executable").as<string> ();
51  } else {
52  _executable = "gdma";
53  }
54 
55  if (options->exists(key + ".density")) {
56  _density = options->get(key + ".density").as<string> ();
57  } else {
58  _density = "SCF"; // defaults to ground state SCF density
59  }
60 
61  if (options->exists(key + ".limit")) {
62  _limit = options->get(key + ".multipoles.limit").as<int> ();
63  } else {
64  _limit = 2; // default to quadrupoles
65  }
66 
67  if (_limit > 2) {
68  cerr << "Tried to use GDMA with Rank > 2 ";
69  throw std::runtime_error("Not supported!");
70  }
71 
72  // specifying this as a single double seems stupid because it's
73  // element-based...
74  // _radius = options->get(key + ".multipoles.radius").as<double> ();
75 
76  if (options->exists(key + ".switch")) {
77  _switch = options->get(key + ".multipoles.switch").as<double> ();
78  } else {
79  _switch = 4.0; // corresponds to GDMA default
80  }
81 
82  if (options->exists(key + ".output")) {
83  _outFile = options->get(key + ".output").as<string> ();
84  } else {
85  _outFile = "gdma.out";
86  }
87 
88  }

111  void GDMA::RunExternal() {
112 
113 
114  // check if the input file exists
115  string fullInput = _runFolder + "/gdma.in";
116  if (!boost::filesystem::exists(fullInput)) {
117  CTP_LOG(ctp::logINFO, *_log) << "GDMA input file has not been found!" << flush;
118  throw runtime_error(" GDMA cannot be run! ");
119  }
120 
121  // check if fchk exists
122  string fullFChk = _runFolder + "/system.fchk";
123  if (!boost::filesystem::exists(fullFChk)) {
124  // try converting Chk to FChk
125  string fullChk = _runFolder + "/" + _chkFile;
126  if (boost::filesystem::exists(fullChk)) {
127  // use formchk
128  string _command;
129  _command = "cd " + _runFolder + "; formchk " + _chkFile + " system.fchk > /dev/null";
130  if (std::system(_command.c_str())){
131  throw runtime_error("Command "+ _command + "failed");
132  }
133  // check again for fchk
134  if (!boost::filesystem::exists(fullFChk)) {
135  CTP_LOG(ctp::logINFO, *_log) << "Formatted Checkpoint file has not been found and cannot be created!" << flush;
136  throw runtime_error(" GDMA cannot be run! ");
137 
138  }
139  } else {
140  CTP_LOG(ctp::logINFO, *_log) << "Formatted Checkpoint file has not been found and cannot be created!" << flush;
141  throw runtime_error(" GDMA cannot be run! ");
142  }
143  }
144 
145 
146  // now we seem ready to go
147  string _command;
148  _command = "cd " + _runFolder + "; " + _executable + " < gdma.in > " + _outFile;
149  if (std::system(_command.c_str())){
150  throw runtime_error("Command "+ _command + "failed");
151  }
152 
153  }

155  void GDMA::ParseOutputFile() {
156 
157  string _gdma_output_name_full = _runFolder + "/gdma.out";
158  std::ifstream _gdma_output(_gdma_output_name_full.c_str());
159  std::string _line;
160  while (_gdma_output) {
161 
162  getline(_gdma_output, _line);
163  // if a line has an equality sign, must be energy
164  std::string::size_type atom_pos = _line.find("x =");
165  if (atom_pos != std::string::npos) {
166  // found an atom, read one more line
167  getline(_gdma_output, _line);
168  // determine rank
169  std::vector<string> results;
170  boost::trim(_line);
171  std::vector<double> Qs; // temp vector for reading in
172 
173  boost::algorithm::split(results, _line, boost::is_any_of("\t "),
174  boost::algorithm::token_compress_on);
175  //cout << results[1] << ":" << results[2] << ":" << results[3] << ":" << results[4] << endl;
176 
177  int rank = boost::lexical_cast<int>(results[3]);
178  if (rank < 0 || rank > 2) {
179  throw runtime_error((boost::format(" Invalid GDMA rank %s!") %rank).str());
180  }
181 
182  // atomic charge
183  if (rank >= 0) {
184  // getting charge
185  getline(_gdma_output, _line);
186  std::vector<string> results;
187  boost::trim(_line);
188  boost::algorithm::split(results, _line, boost::is_any_of("\t "),
189  boost::algorithm::token_compress_on);
190  double Q00 = boost::lexical_cast<double>(results.back());
191  Qs.push_back(Q00);
192 
193  // CTP_LOG(logINFO, *_log) << "New Q00 " << Q00 << flush;
194 
195  }
196 
197  // atomic dipole components
198  if (rank >= 1) {
199 
200  // getting dipoles
201  getline(_gdma_output, _line);
202  std::vector<string> results;
203  boost::trim(_line);
204  boost::algorithm::split(results, _line, boost::is_any_of("\t "),
205  boost::algorithm::token_compress_on);
206  double Q10 = boost::lexical_cast<double>(results[5]);
207  double Q11c = boost::lexical_cast<double>(results[8]);
208  double Q11s = boost::lexical_cast<double>(results.back());
209  Qs.push_back(Q10);
210  Qs.push_back(Q11c);
211  Qs.push_back(Q11s);
212  /*
213  CTP_LOG(logINFO, *_log) << "New Q10 " << Q10 << flush;
214  CTP_LOG(logINFO, *_log) << "New Q11c " << Q11c << flush;
215  CTP_LOG(logINFO, *_log) << "New Q11s " << Q11s << flush;
216  */
217 
218  }
219 
220  // atomic quadrupole components
221  if (rank == 2) {
222 
223  getline(_gdma_output, _line);
224  std::vector<string> results;
225  boost::trim(_line);
226  boost::algorithm::split(results, _line, boost::is_any_of("\t "),
227  boost::algorithm::token_compress_on);
228  double Q20 = boost::lexical_cast<double>(results[5]);
229  double Q21c = boost::lexical_cast<double>(results[8]);
230  double Q21s = boost::lexical_cast<double>(results.back());
231  getline(_gdma_output, _line);
232  boost::trim(_line);
233  boost::algorithm::split(results, _line, boost::is_any_of("\t "),
234  boost::algorithm::token_compress_on);
235  double Q22c = boost::lexical_cast<double>(results[2]);
236  double Q22s = boost::lexical_cast<double>(results[5]);
237 
238  Qs.push_back(Q20);
239  Qs.push_back(Q21c);
240  Qs.push_back(Q21s);
241  Qs.push_back(Q22c);
242  Qs.push_back(Q22s);
243 
244  /*
245  CTP_LOG(logINFO, *_log) << "New Q20 " << Q20 << flush;
246  CTP_LOG(logINFO, *_log) << "New Q21c " << Q21c << flush;
247  CTP_LOG(logINFO, *_log) << "New Q21s " << Q21s << flush;
248  CTP_LOG(logINFO, *_log) << "New Q22c " << Q22c << flush;
249  CTP_LOG(logINFO, *_log) << "New Q22s " << Q22s << flush;
250  */
251 
252  }
253 
254  _multipoles.push_back( Qs );
255 
256  } // atom
257 
258  std::string::size_type break_pos = _line.find("Total multipoles referred to origin at");
259  if (break_pos != std::string::npos) break;
260 
261 
262  } // gdma_output
263  // CTP_LOG(logINFO, *_log) << "Done with GDMA" << flush;
264 
265 
266  }
