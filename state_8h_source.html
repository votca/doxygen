94 template <class TGraph, class TCarrier>
95 void State<TGraph,TCarrier>::Load(const char* filename, TGraph* graph){
96 
97  ifstream statestore;
98  statestore.open(filename);
99  string line;
100 
101  int car_nodeID; int car_type;
102  double car_distancex; double car_distancey; double car_distancez;
103 
104  int carrier_ID = 0;
105 
106  if(statestore.is_open()) {
107  while (getline(statestore,line)) {
108 
109  //read and parse lines
110  std::istringstream linestream(line);
111  linestream >> car_nodeID >> car_type >> car_distancex >> car_distancey >> car_distancez;
112 
113  //add carrier to node
114  TCarrier* newcarrier = AddCarrier(carrier_ID);
115  Node* carrier_node = graph->GetNode(car_nodeID);
116  newcarrier->SetCarrierNode(carrier_node);
117  carrier_node->AddCarrier(carrier_ID);
118 
119  //set carrier type
120  newcarrier->SetCarrierType(car_type);
121 
122  // Set distance travelled
123  votca::tools::vec carrier_distance(car_distancex,car_distancey,car_distancez);
124  newcarrier->SetDistance(carrier_distance);
125 
126  // New carrier is in the simulation box
127  newcarrier->SetInBox(true);
128  carrier_ID++;
129  }
130  }
131  else {
132  std::cout << "WARNING: Can't open state store file" << "\n";
133  }
134 
135  statestore.close();
136 }
