81 void State::Initialize_inject_trees(Graph* graph, Inject_Type injecttype, Globaleventinfo* globevent) {
82  electron_inject->initialize(graph->nodes.size());
83  hole_inject->initialize(graph->nodes.size());
84 
85  for (int inode = 0; inode < graph->nodes.size(); inode++) {
86  if (injecttype == Equal) {
87  electron_inject->setrate(inode, 1.0);
88  hole_inject->setrate(inode, 1.0);
89  }
90  else if(injecttype == Fermi) {
91  electron_inject->setrate(inode, exp(-1.0*globevent->beta*graph->nodes[inode]->static_electron_node_energy));
92  hole_inject->setrate(inode, exp(-1.0*globevent->beta*graph->nodes[inode]->static_hole_node_energy));
93  }
94  }

104 void State::Init_coulomb_mesh(Graph* graph, Globaleventinfo* globevent){
105  meshsizeX = ceil(graph->sim_box_size.x()/globevent->coulcut);
106  meshsizeY = ceil(graph->sim_box_size.y()/globevent->coulcut);
107  meshsizeZ = ceil(graph->sim_box_size.z()/globevent->coulcut);
108 
109  coulomb_mesh.resize(meshsizeX);
110  for(int i = 0;i<meshsizeX;i++) {
111  coulomb_mesh[i].resize(meshsizeY);
112  for(int j = 0;j<meshsizeY;j++) {
113  coulomb_mesh[i][j].resize(meshsizeZ);
114  for(int k=0;k<meshsizeZ;k++) {
115  coulomb_mesh[i][j][k].resize(2);
116  }
117  }
118  }
119 
120  for(int ic=0;ic<electrons.size();ic++){
121  Add_to_coulomb_mesh(graph, electrons[ic], globevent);
122  }
123 
124  for(int ic=0;ic<holes.size();ic++){
125  Add_to_coulomb_mesh(graph, holes[ic], globevent);
126  }
127 }

172 void State::Save(string SQL_state_filename){
173 
174  votca::tools::Database db;
175  db.Open( SQL_state_filename );
176  db.BeginTransaction();
177 
178  votca::tools::Statement *stmt;
179  stmt = db.Prepare( "INSERT INTO carriers ("
180  "node_id, carrier_type, distanceX, "
181  "distanceY, distanceZ)"
182  "VALUES ("
183  "?, ?, ?,"
184  "?, ?)");
185 
186  for(int electron_nr = 0;electron_nr<electrons.size();electron_nr++) {
187  if (El_in_sim_box(electron_nr)) {
188  stmt->Bind(1, electrons[electron_nr]->carrier_node_ID);
189  stmt->Bind(2, 0);
190  myvec carrier_distance = electrons[electron_nr]->carrier_distance;
191  stmt->Bind(3, carrier_distance.x());
192  stmt->Bind(4, carrier_distance.y());
193  stmt->Bind(5, carrier_distance.z());
194  stmt->InsertStep();
195  stmt->Reset();
196  }
197  }
198 
199  for(int hole_nr = 0;hole_nr<holes.size();hole_nr++) {
200  if (Ho_in_sim_box(hole_nr)) {
201  stmt->Bind(1, holes[hole_nr]->carrier_node_ID);
202  stmt->Bind(2, 1);
203  myvec carrier_distance = holes[hole_nr]->carrier_distance;
204  stmt->Bind(3, carrier_distance.x());
205  stmt->Bind(4, carrier_distance.y());
206  stmt->Bind(5, carrier_distance.z());
207  stmt->InsertStep();
208  stmt->Reset();
209  }
210  }

212  delete stmt;
213  stmt = NULL;

218 void State::Load(string SQL_state_filename, Graph* graph, Globaleventinfo* globevent){
219 
220  votca::tools::Database db;
221  db.Open( SQL_state_filename );
222 
223  votca::tools::Statement *stmt;
224  stmt = db.Prepare("SELECT node_id, carrier_type, distanceX, distanceY, distanceZ FROM carriers;");
225 
226  while (stmt->Step() != SQLITE_DONE)
227  {
228  int cartype = stmt->Column<int>(1);
229  if(cartype == 0) { // electron
230  if(electron_reservoir.empty()) {Grow(electrons,electron_reservoir,globevent->state_grow_size, graph->max_pair_degree);}
231  int electron_nr = Buy(electrons, electron_reservoir);
232  int carnode_ID = stmt->Column<int>(0);
233  electrons[electron_nr]->carrier_node_ID = carnode_ID;
234  electrons[electron_nr]->carrier_type = Electron;
235  graph->nodes[carnode_ID]->carriers_on_node.push_back(electrons[electron_nr]);
236  double distancex = stmt->Column<double>(2);
237  double distancey = stmt->Column<double>(3);
238  double distancez = stmt->Column<double>(4);
239  electrons[electron_nr]->carrier_distance = myvec(distancex,distancey,distancez);
240  }
241  else if(cartype == 1) { // hole
242  if(hole_reservoir.empty()) {Grow(holes,hole_reservoir,globevent->state_grow_size, graph->max_pair_degree);}
243  int hole_nr = Buy(holes, hole_reservoir);
244  int carnode_ID = stmt->Column<int>(0);
245  holes[hole_nr]->carrier_node_ID = carnode_ID;
246  holes[hole_nr]->carrier_type = Hole;
247  graph->nodes[carnode_ID]->carriers_on_node.push_back(holes[hole_nr]);
248  double distancex = stmt->Column<double>(2);
249  double distancey = stmt->Column<double>(3);
250  double distancez = stmt->Column<double>(4);
251  holes[hole_nr]->carrier_distance = myvec(distancex,distancey,distancez);
252  }
253  }
254  delete stmt;
255  stmt = NULL;
256 }
