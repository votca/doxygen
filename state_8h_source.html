81 void State::Initialize_inject_trees(Graph* graph, Inject_Type injecttype, Globaleventinfo* globevent) {
82  electron_inject->initialize(graph->nodes.size());
83  hole_inject->initialize(graph->nodes.size());
84 
85  for (unsigned int inode = 0; inode < graph->nodes.size(); inode++) {
86  if (injecttype == Equal) {
87  electron_inject->setrate(inode, 1.0);
88  hole_inject->setrate(inode, 1.0);
89  }
90  else if(injecttype == Fermi) {
91  electron_inject->setrate(inode, exp(-1.0*globevent->beta*graph->nodes[inode]->static_electron_node_energy));
92  hole_inject->setrate(inode, exp(-1.0*globevent->beta*graph->nodes[inode]->static_hole_node_energy));
93  }
94  }

104 void State::Init_coulomb_mesh(Graph* graph, Globaleventinfo* globevent){
105  meshsizeX = ceil(graph->sim_box_size.x()/globevent->coulcut);
106  meshsizeY = ceil(graph->sim_box_size.y()/globevent->coulcut);
107  meshsizeZ = ceil(graph->sim_box_size.z()/globevent->coulcut);
108 
109  coulomb_mesh.resize(meshsizeX);
110  for(int i = 0;i<meshsizeX;i++) {
111  coulomb_mesh[i].resize(meshsizeY);
112  for(int j = 0;j<meshsizeY;j++) {
113  coulomb_mesh[i][j].resize(meshsizeZ);
114  for(int k=0;k<meshsizeZ;k++) {
115  coulomb_mesh[i][j][k].resize(2);
116  }
117  }
118  }
119 
120  for(unsigned int ic=0;ic<electrons.size();ic++){
121  Add_to_coulomb_mesh(graph, electrons[ic], globevent);
122  }
123 
124  for(unsigned int ic=0;ic<holes.size();ic++){
125  Add_to_coulomb_mesh(graph, holes[ic], globevent);
126  }
127 }

129 void State::Add_to_coulomb_mesh(Graph* graph, Carrier* carrier, Globaleventinfo* globevent){
130 
131  int charge;
132  if(carrier->carrier_type == Electron) {
133  charge = 0;
134  }
135  else if(carrier->carrier_type == Hole) {
136  charge = 1;
137  } else {
138  throw runtime_error("carrier->carrier_type should be Hole or Electron");
139  }
140 
141  double posx = graph->nodes[carrier->carrier_node_ID]->node_position.x();
142  double posy = graph->nodes[carrier->carrier_node_ID]->node_position.y();
143  double posz = graph->nodes[carrier->carrier_node_ID]->node_position.z();
144 
145  int iposx = floor(posx/globevent->coulcut);
146  int iposy = floor(posy/globevent->coulcut);
147  int iposz = floor(posz/globevent->coulcut);
148 
149  coulomb_mesh[iposx][iposy][iposz][charge].push_back(carrier->carrier_ID);
150 }

176 void State::Save(string SQL_state_filename){
177 
178  votca::tools::Database db;
179  db.Open( SQL_state_filename );
180  db.BeginTransaction();
181 
182  votca::tools::Statement *stmt;
183  stmt = db.Prepare( "INSERT INTO carriers ("
184  "node_id, carrier_type, distanceX, "
185  "distanceY, distanceZ)"
186  "VALUES ("
187  "?, ?, ?,"
188  "?, ?)");
189 
190  for(unsigned int electron_nr = 0;electron_nr<electrons.size();electron_nr++) {
191  if (El_in_sim_box(electron_nr)) {
192  stmt->Bind(1, electrons[electron_nr]->carrier_node_ID);
193  stmt->Bind(2, 0);
194  myvec carrier_distance = electrons[electron_nr]->carrier_distance;
195  stmt->Bind(3, carrier_distance.x());
196  stmt->Bind(4, carrier_distance.y());
197  stmt->Bind(5, carrier_distance.z());
198  stmt->InsertStep();
199  stmt->Reset();
200  }
201  }
202 
203  for(unsigned int hole_nr = 0;hole_nr<holes.size();hole_nr++) {
204  if (Ho_in_sim_box(hole_nr)) {
205  stmt->Bind(1, holes[hole_nr]->carrier_node_ID);
206  stmt->Bind(2, 1);
207  myvec carrier_distance = holes[hole_nr]->carrier_distance;
208  stmt->Bind(3, carrier_distance.x());
209  stmt->Bind(4, carrier_distance.y());
210  stmt->Bind(5, carrier_distance.z());
211  stmt->InsertStep();
212  stmt->Reset();
213  }
214  }

216  delete stmt;
217  stmt = NULL;

222 void State::Load(string SQL_state_filename, Graph* graph, Globaleventinfo* globevent){
223 
224  votca::tools::Database db;
225  db.Open( SQL_state_filename );
226 
227  votca::tools::Statement *stmt;
228  stmt = db.Prepare("SELECT node_id, carrier_type, distanceX, distanceY, distanceZ FROM carriers;");
229 
230  while (stmt->Step() != SQLITE_DONE)
231  {
232  int cartype = stmt->Column<int>(1);
233  if(cartype == 0) { // electron
234  if(electron_reservoir.empty()) {Grow(electrons,electron_reservoir,globevent->state_grow_size, graph->max_pair_degree);}
235  int electron_nr = Buy(electrons, electron_reservoir);
236  int carnode_ID = stmt->Column<int>(0);
237  electrons[electron_nr]->carrier_node_ID = carnode_ID;
238  electrons[electron_nr]->carrier_type = Electron;
239  graph->nodes[carnode_ID]->carriers_on_node.push_back(electrons[electron_nr]);
240  double distancex = stmt->Column<double>(2);
241  double distancey = stmt->Column<double>(3);
242  double distancez = stmt->Column<double>(4);
243  electrons[electron_nr]->carrier_distance = myvec(distancex,distancey,distancez);
244  }
245  else if(cartype == 1) { // hole
246  if(hole_reservoir.empty()) {Grow(holes,hole_reservoir,globevent->state_grow_size, graph->max_pair_degree);}
247  int hole_nr = Buy(holes, hole_reservoir);
248  int carnode_ID = stmt->Column<int>(0);
249  holes[hole_nr]->carrier_node_ID = carnode_ID;
250  holes[hole_nr]->carrier_type = Hole;
251  graph->nodes[carnode_ID]->carriers_on_node.push_back(holes[hole_nr]);
252  double distancex = stmt->Column<double>(2);
253  double distancey = stmt->Column<double>(3);
254  double distancez = stmt->Column<double>(4);
255  holes[hole_nr]->carrier_distance = myvec(distancex,distancey,distancez);
256  }
257  }
258  delete stmt;
259  stmt = NULL;
260 }
