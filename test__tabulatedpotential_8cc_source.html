39 vector<double> getColumnFromFile(string file_name, int column) {
40  vector<double> data;
41  ifstream file;
42  file.open(file_name);
43  string line;
44  if (file.is_open()) {
45  while (getline(file, line)) {
46  string word;
47  istringstream ss(line);
48  for (int i = 0; i < column; ++i) {
49  ss >> word;
50  }
51  data.push_back(stod(word));
52  }
53  file.close();
54  }
55  return data;
56 }
