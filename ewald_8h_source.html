163  next->add("type", (format("%1$s") % name1).str());
164  next->add("mps", (format("MP_FILES/%1$s_%2$s.mps")

178 template<class EwaldMethod>
179 XJob Ewald<EwaldMethod>::ProcessInputString(Job *job, Topology *top,
180  QMThread *thread) {
181 
182  // Input string looks like this:
183  // <id1>:<name1>:<mpsfile1> <id2>:<name2>: ... ... ...
184 
185  string input = job->getInput().as<string>();
186  vector<Segment*> qmSegs;
187  vector<string> qmSegMps;
188  vector<string> split;
189  Tokenizer toker(input, " \t\n");
190  toker.ToVector(split);
191 
192  for (unsigned int i = 0; i < split.size(); ++i) {
193 
194  string id_seg_mps = split[i];
195  vector<string> split_id_seg_mps;
196  Tokenizer toker(id_seg_mps, ":");
197  toker.ToVector(split_id_seg_mps);
198 
199  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
200  string segName = split_id_seg_mps[1];
201  string mpsFile = split_id_seg_mps[2];
202 
203  Segment *seg = top->getSegment(segId);
204  if (seg->getName() != segName) {
205  CTP_LOG(logERROR,*(thread->getLogger()))
206  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
207  << " maltagged as " << segName << ". Skip job ..." << flush;
208  throw std::runtime_error("Input does not match topology.");
209  }
210 
211  qmSegs.push_back(seg);
212  qmSegMps.push_back(mpsFile);
213  }
214 
215  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
216 }

220 Job::JobResult Ewald<EwaldMethod>::EvalJob(Topology *top, Job *job,
221  QMThread *thread) {
222 
223  boost::timer::cpu_timer cpu_t;
224  cpu_t.start();
225  boost::timer::cpu_times t_in = cpu_t.elapsed();
226 
227  Logger *log = thread->getLogger();
228  CTP_LOG(logINFO,*log)
229  << "Job input = " << job->getInput().as<string>() << flush;
230 
231  // CREATE XJOB FROM JOB INPUT STRING
232  XJob xjob = this->ProcessInputString(job, top, thread);
233 
234  // GENERATE POLAR TOPOLOGY (GENERATE VS LOAD IF PREPOLARIZED)
235  if (_polar_bg_arch == "") {
236  CTP_LOG(logINFO,*log) << "Mps-Mapper: Generate FGC FGN BGN" << flush;
237  _mps_mapper.Gen_FGC_FGN_BGN(top, &xjob, thread);
238  }
239  else {
240  CTP_LOG(logINFO,*log) << "Mps-Mapper: Generate FGC, load FGN BGN from '"
241  << _polar_bg_arch << "'" << flush;
242  _mps_mapper.Gen_FGC_Load_FGN_BGN(top, &xjob, _polar_bg_arch, thread);
243  }
244 
245  // CALL THOLEWALD MAGIC
246  EwaldMethod ewaldnd = EwaldMethod(top, xjob.getPolarTop(), _options,
247  thread->getLogger());
248  if (_pdb_check)
249  ewaldnd.WriteDensitiesPDB(xjob.getTag()+".densities.pdb");
250  ewaldnd.Evaluate();
251  if (_ptop_check)
252  ewaldnd.WriteDensitiesPtop(xjob.getTag()+".fg.ptop",
253  xjob.getTag()+".mg.ptop", xjob.getTag()+".bg.ptop");
254 
255  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
256  Property output = ewaldnd.GenerateOutputString();
257  Job::JobResult jres = Job::JobResult();
258  jres.setOutput(output);
259  jres.setStatus(Job::COMPLETE);
260 
261  if (!ewaldnd.Converged()) {
262  jres.setStatus(Job::FAILED);
263  jres.setError(ewaldnd.GenerateErrorString());
264  CTP_LOG(logERROR,*log) << ewaldnd.GenerateErrorString() << flush;
265  }
266 
267  boost::timer::cpu_times t_out = cpu_t.elapsed();
268  double t_run = (t_out.wall-t_in.wall)/1e9/60.;
269  CTP_LOG(logINFO,*log)
270  << "Job runtime was " << t_run << " min" << flush;
271 
272  return jres;
273 }
