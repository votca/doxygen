176  // Input string looks like this:

178 
179  string input = job->getInput().as<string>();

183  Tokenizer toker(input, " \t\n");
184  toker.ToVector(split);
185 
186  for (int i = 0; i < split.size(); ++i) {
187 
188  string id_seg_mps = split[i];
189  vector<string> split_id_seg_mps;
190  Tokenizer toker(id_seg_mps, ":");
191  toker.ToVector(split_id_seg_mps);
192 
193  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
194  string segName = split_id_seg_mps[1];
195  string mpsFile = split_id_seg_mps[2];
196 
197  Segment *seg = top->getSegment(segId);
198  if (seg->getName() != segName) {
199  LOG(logERROR,*(thread->getLogger()))
200  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
201  << " maltagged as " << segName << ". Skip job ..." << flush;
202  throw std::runtime_error("Input does not match topology.");
203  }
204 
205  qmSegs.push_back(seg);
206  qmSegMps.push_back(mpsFile);
207  }
208 
209  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
210 }
