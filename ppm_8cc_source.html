26 void PPM::PPM_construct_parameters(const RPA& rpa) {
27 
28  // Solve Eigensystem
29  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(
30  rpa.calculate_epsilon_r(screening_r));
31  _ppm_phi = es.eigenvectors();
32 
33  // store PPM weights from eigenvalues
34  _ppm_weight = 1 - es.eigenvalues().array().inverse();
35 
36  // a) phi^t * epsilon(1) * phi e.g. transform epsilon(1) to the same space as
37  // epsilon(0)
38  Eigen::MatrixXd ortho =
39  _ppm_phi.transpose() * rpa.calculate_epsilon_i(screening_i) * _ppm_phi;
40  Eigen::MatrixXd epsilon_1_inv = ortho.inverse();
41  // determine PPM frequencies
42  _ppm_freq.resize(es.eigenvalues().size());
43 #pragma omp parallel for
44  for (Index i = 0; i < es.eigenvalues().size(); i++) {
45  if (_ppm_weight(i) < 1.e-5) {
46  _ppm_weight(i) = 0.0;
47  _ppm_freq(i) = 0.5; // Hartree
48  continue;
49  } else {
50  double nom = epsilon_1_inv(i, i) - 1.0;
51  double frac =
52  -1.0 * nom / (nom + _ppm_weight(i)) * screening_i * screening_i;
53  _ppm_freq(i) = std::sqrt(std::abs(frac));
54  }
55  }
56  return;
57 }
