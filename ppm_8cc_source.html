28  void PPM::PPM_construct_parameters(const RPA& rpa) {
29 
30  //Solve Eigensystem
31  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(rpa.GetEpsilon_r()[0]);
32  //we store _ppm_phi_T instead of _ppm_phi because we need it for later transformations
33  _ppm_phi=es.eigenvectors();
34 
35  // store PPM weights from eigenvalues
36  _ppm_weight=Eigen::VectorXd::Ones(es.eigenvalues().size())-es.eigenvalues().cwiseInverse();
37 
38  // a) phi^t * epsilon(1) * phi e.g. transform epsilon(1) to the same space as epsilon(0)
39  Eigen::MatrixXd ortho=_ppm_phi.transpose()*rpa.GetEpsilon_i()[0]*_ppm_phi;
40  Eigen::MatrixXd epsilon_1_inv=ortho.inverse();
41  // determine PPM frequencies
42  _ppm_freq.resize(es.eigenvalues().size());
43  #pragma omp parallel for
44  for (unsigned _i = 0; _i < es.eigenvalues().size(); _i++) {
45  if (_ppm_weight(_i) < 1.e-5) {
46  _ppm_weight(_i) = 0.0;
47  _ppm_freq(_i) = 0.5;//Hartree
48  continue;
49  } else {
50  double _nom = epsilon_1_inv(_i, _i) - 1.0;
51  double _frac = -1.0 * _nom / (_nom + _ppm_weight(_i)) * screening_i *screening_i;
52  _ppm_freq(_i) = sqrt(std::abs(_frac));
53  }
54  }
55  return;
56  }
