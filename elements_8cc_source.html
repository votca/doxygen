92 pair<string,double> Elements::findShortNameOfElementClosestInMass_(double mass){
93  string eleShort = "H";
94  double diff = abs(mass-_Mass[eleShort]);
95  for(const auto & ele_pr : _Mass ){
96  if(abs(ele_pr.second-mass)<diff){
97  eleShort = ele_pr.first;
98  diff = abs(ele_pr.second-mass);
99  }
100  }
101  return pair<string,double>(eleShort,diff);
102 }
