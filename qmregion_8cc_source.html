35 void QMRegion::Initialize(const tools::Property& prop) {
36  if (this->id_ != 0) {
37  throw std::runtime_error(
38  this->identify() +
39  " must always be region 0. Currently only one qm region is possible.");
40  }
41 
42  initstate_ = prop.get("state").as<QMState>();
43  if (initstate_.Type() == QMStateType::Hole ||
44  initstate_.Type() == QMStateType::Electron) {
45  throw std::runtime_error(
46  "Charged QM Regions are not implemented currently");
47  }
48  if (initstate_.Type().isExciton() || initstate_.Type().isGWState()) {
49 
50  do_gwbse_ = true;
51  gwbseoptions_ = prop.get("gwbse");
52  if (prop.exists("statetracker")) {
53  tools::Property filter = prop.get("statetracker");
54  statetracker_.setLogger(&log_);
55  statetracker_.Initialize(filter);
56  statetracker_.setInitialState(initstate_);
57  statetracker_.PrintInfo();
58  } else {
59  throw std::runtime_error(
60  "No statetracker options for excited states found");
61  }
62  }
63 
64  grid_accuracy_for_ext_interaction_ =
65  prop.get("grid_for_potential").as<std::string>();
66  DeltaE_ = prop.get("tolerance_energy").as<double>();
67  DeltaD_ = prop.get("tolerance_density").as<double>();
68 
69  dftoptions_ = prop.get("dftpackage");
70 }

72 bool QMRegion::Converged() const {
73  if (!E_hist_.filled()) {
74  return false;
75  }
76 
77  double Echange = E_hist_.getDiff();
78  double Dchange =
79  Dmat_hist_.getDiff().norm() / double(Dmat_hist_.back().cols());
80  double Dmax = Dmat_hist_.getDiff().cwiseAbs().maxCoeff();
81  std::string info = "not converged";
82  bool converged = false;
83  if (Dchange < DeltaD_ && Dmax < DeltaD_ && std::abs(Echange) < DeltaE_) {
84  info = "converged";
85  converged = true;
86  }
87  XTP_LOG(Log::error, log_)
88  << " Region:" << this->identify() << " " << this->getId() << " is "
89  << info << " deltaE=" << Echange << " RMS Dmat=" << Dchange
90  << " MaxDmat=" << Dmax << std::flush;
91  return converged;
92 }

94 void QMRegion::Evaluate(std::vector<std::unique_ptr<Region> >& regions) {
95 
96  std::vector<double> interact_energies = ApplyInfluenceOfOtherRegions(regions);
97  double e_ext =
98  std::accumulate(interact_energies.begin(), interact_energies.end(), 0.0);
99  XTP_LOG(Log::info, log_)
100  << TimeStamp()
101  << " Calculated interaction potentials with other regions. E[hrt]= "
102  << e_ext << std::flush;
103  XTP_LOG(Log::info, log_) << "Writing inputs" << std::flush;
104  qmpackage_->setRunDir(workdir_);
105  qmpackage_->WriteInputFile(orb_);
106  XTP_LOG(Log::error, log_) << "Running DFT calculation" << std::flush;
107  bool run_success = qmpackage_->Run();
108  if (!run_success) {
109  info_ = false;
110  errormsg_ = "DFT-run failed";
111  return;
112  }
113 
114  bool Logfile_parse = qmpackage_->ParseLogFile(orb_);
115  if (!Logfile_parse) {
116  info_ = false;
117  errormsg_ = "Parsing DFT logfile failed.";
118  return;
119  }
120  bool Orbfile_parse = qmpackage_->ParseMOsFile(orb_);
121  if (!Orbfile_parse) {
122  info_ = false;
123  errormsg_ = "Parsing DFT orbfile failed.";
124  return;
125  }
126  QMState state = QMState("groundstate");
127  double energy = orb_.getDFTTotalEnergy();
128  if (do_gwbse_) {
129  GWBSE gwbse(orb_);
130  gwbse.setLogger(&log_);
131  gwbse.Initialize(gwbseoptions_);
132  gwbse.Evaluate();
133  state = statetracker_.CalcStateAndUpdate(orb_);
134  if (state.Type().isExciton()) {
135  energy += orb_.getExcitedStateEnergy(state);
136  } else {
137  // if unoccupied, add QP level energy
138  if (state.StateIdx() > orb_.getHomo()) {
139  energy += orb_.getExcitedStateEnergy(state);
140  } else {
141  // if unoccupied, subtract QP level energy
142  energy -= orb_.getExcitedStateEnergy(state);
143  }
144  }
145  }
146  E_hist_.push_back(energy);
147  Dmat_hist_.push_back(orb_.DensityMatrixFull(state));
148  return;
149 }

160 double QMRegion::charge() const {
161  double charge = 0.0;
162  if (!do_gwbse_) {
163  Index nuccharge = 0;
164  for (const QMAtom& a : orb_.QMAtoms()) {
165  nuccharge += a.getNuccharge();
166  }
167 
168  Index electrons = orb_.getNumberOfAlphaElectrons() * 2;
169  charge = double(nuccharge - electrons);
170  } else {
171  QMState state = statetracker_.InitialState();
172  if (state.Type().isExciton()) {
173  charge = 0.0;
174  } else if (state.Type().isSingleParticleState()) {
175  if (state.StateIdx() <= orb_.getHomo()) {
176  charge = +1.0;
177  } else {
178  charge = -1.0;
179  }
180  }
181  }
182  return charge;
183 }

236 void QMRegion::ApplyQMFieldToPolarSegments(
237  std::vector<PolarSegment>& segments) const {
238 
239  Vxc_Grid grid;
240  AOBasis basis =
241  orb_.SetupDftBasis(); // grid needs a basis in scope all the time
242  grid.GridSetup(grid_accuracy_for_ext_interaction_, orb_.QMAtoms(), basis);
243  DensityIntegration<Vxc_Grid> numint(grid);
244 
245  QMState state = QMState("groundstate");
246  if (do_gwbse_) {
247  state = statetracker_.CalcState(orb_);
248  }
249  Eigen::MatrixXd dmat = orb_.DensityMatrixFull(state);
250  double Ngrid = numint.IntegrateDensity(dmat);
251  AOOverlap overlap;
252  overlap.Fill(basis);
253  double N_comp = dmat.cwiseProduct(overlap.Matrix()).sum();
254  if (std::abs(Ngrid - N_comp) > 0.001) {
255  XTP_LOG(Log::error, log_) << "=======================" << std::flush;
256  XTP_LOG(Log::error, log_)
257  << "WARNING: Calculated Densities at Numerical Grid, Number of "
258  "electrons "
259  << Ngrid << " is far away from the the real value " << N_comp
260  << ", you should increase the accuracy of the integration grid."
261  << std::flush;
262  }
263 #pragma omp parallel for
264  for (Index i = 0; i < Index(segments.size()); ++i) {
265  PolarSegment& seg = segments[i];
266  for (PolarSite& site : seg) {
267  site.V_noE() += numint.IntegrateField(site.getPos());
268  }
269  }
270 
271  StaticSegment seg(orb_.QMAtoms().getType(), orb_.QMAtoms().getId());
272  for (const QMAtom& atom : orb_.QMAtoms()) {
273  seg.push_back(StaticSite(atom, double(atom.getNuccharge())));
274  }
275  AddNucleiFields(segments, seg);
276 }
