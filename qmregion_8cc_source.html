36void QMRegion::Initialize(const tools::Property& prop) {
37 if (this->id_ != 0) {
38 throw std::runtime_error(
39 this->identify() +
40 " must always be region 0. Currently only one qm region is possible.");
41 }
42
43 initstate_ = prop.get("state").as<QMState>();
44 if (initstate_.Type() == QMStateType::Hole ||
45 initstate_.Type() == QMStateType::Electron) {
46 throw std::runtime_error(
47 "Charged QM Regions are not implemented currently");
48 }
49 if (initstate_.Type().isExciton() || initstate_.Type().isGWState()) {
50
51 do_gwbse_ = true;
52 gwbseoptions_ = prop.get("gwbse");
53 if (prop.exists("statetracker")) {
54 tools::Property filter = prop.get("statetracker");
55 statetracker_.setLogger(&log_);
56 statetracker_.Initialize(filter);
57 statetracker_.setInitialState(initstate_);
58 statetracker_.PrintInfo();
59 } else {
60 throw std::runtime_error(
61 "No statetracker options for excited states found");
62 }
63 }
64
65 if (initstate_.Type().isKSState()) {
66
67 if (prop.exists("statetracker")) {
68 tools::Property filter = prop.get("statetracker");
69 statetracker_.setLogger(&log_);
70 statetracker_.Initialize(filter);
71 statetracker_.setInitialState(initstate_);
72 statetracker_.PrintInfo();
73 } else {
74 throw std::runtime_error("No statetracker options for KS states found");
75 }
76 }
77
78 grid_accuracy_for_ext_interaction_ =
79 prop.get("grid_for_potential").as<std::string>();
80 DeltaE_ = prop.get("tolerance_energy").as<double>();
81 DeltaD_ = prop.get("tolerance_density").as<double>();
82 DeltaDmax_ = prop.get("tolerance_density_max").as<double>();
83
84 dftoptions_ = prop.get("dftpackage");
85 localize_options_ = prop.get("localize");
86
87 if (prop.get("dftpackage.xtpdft.dft_in_dft.activeatoms")
88 .as<std::string>()
89 .size() != 0) {
90 do_localize_ = true;
91 do_dft_in_dft_ = true;
92 }
93}

95bool QMRegion::Converged() const {
96 if (!E_hist_.filled()) {
97 return false;
98 }
99
100 double Echange = E_hist_.getDiff();
101 double Dchange =
102 Dmat_hist_.getDiff().norm() / double(Dmat_hist_.back().cols());
103 double Dmax = Dmat_hist_.getDiff().cwiseAbs().maxCoeff();
104 std::string info = "not converged";
105 bool converged = false;
106 if (Dchange < DeltaD_ && Dmax < DeltaDmax_ && std::abs(Echange) < DeltaE_) {
107 info = "converged";
108 converged = true;
109 }
110 XTP_LOG(Log::error, log_)
111 << " Region:" << this->identify() << " " << this->getId() << " is "
112 << info << " deltaE=" << Echange << " RMS Dmat=" << Dchange
113 << " MaxDmat=" << Dmax << std::flush;
114 return converged;
115}

117void QMRegion::Evaluate(std::vector<std::unique_ptr<Region> >& regions) {
118
119 std::vector<double> interact_energies = ApplyInfluenceOfOtherRegions(regions);
120 double e_ext =
121 std::accumulate(interact_energies.begin(), interact_energies.end(), 0.0);
122 XTP_LOG(Log::info, log_)
123 << TimeStamp()
124 << " Calculated interaction potentials with other regions. E[hrt]= "
125 << e_ext << std::flush;
126 XTP_LOG(Log::info, log_) << "Writing inputs" << std::flush;
127 qmpackage_->setRunDir(workdir_);
128 qmpackage_->WriteInputFile(orb_);
129 XTP_LOG(Log::error, log_) << "Running DFT calculation" << std::flush;
130 bool run_success = qmpackage_->Run();
131 if (!run_success) {
132 info_ = false;
133 errormsg_ = "DFT-run failed";
134 return;
135 }
136
137 bool Logfile_parse = qmpackage_->ParseLogFile(orb_);
138 if (!Logfile_parse) {
139 info_ = false;
140 errormsg_ = "Parsing DFT logfile failed.";
141 return;
142 }
143 bool Orbfile_parse = qmpackage_->ParseMOsFile(orb_);
144 if (!Orbfile_parse) {
145 info_ = false;
146 errormsg_ = "Parsing DFT orbfile failed.";
147 return;
148 }
149 if (do_localize_) {
150 PMLocalization pml(log_, localize_options_);
151 pml.computePML(orb_);
152 }
153 QMMolecule originalmol = orb_.QMAtoms();
154 if (do_dft_in_dft_) {
155
156 // this only works with XTPDFT, so locally override global qmpackage_
157 std::unique_ptr<QMPackage> xtpdft =
158 std::unique_ptr<QMPackage>(QMPackageFactory().Create("xtp"));
159 xtpdft->setLog(&log_);
160 xtpdft->Initialize(dftoptions_);
161 xtpdft->setRunDir(workdir_);
162 Index charge = 0;
163 if (initstate_.Type() == QMStateType::Electron) {
164 charge = -1;
165 } else if (initstate_.Type() == QMStateType::Hole) {
166 charge = +1;
167 }
168 xtpdft->setCharge(charge);
169
170 std::vector<double> energies = std::vector<double>(regions.size(), 0.0);
171 for (std::unique_ptr<Region>& reg : regions) {
172 Index id = reg->getId();
173 if (id == this->getId()) {
174 continue;
175 }
176
177 StaticRegion Staticdummy(0, log_);
178 PolarRegion Polardummy(0, log_);
179 XTP_LOG(Log::error, log_)
180 << TimeStamp() << " Evaluating interaction between "
181 << this->identify() << " " << this->getId() << " and "
182 << reg->identify() << " " << reg->getId() << std::flush;
183 if (reg->identify() == Staticdummy.identify()) {
184 StaticRegion* staticregion = dynamic_cast<StaticRegion*>(reg.get());
185 xtpdft->AddRegion(*staticregion);
186 } else if (reg->identify() == Polardummy.identify()) {
187 PolarRegion* polarregion = dynamic_cast<PolarRegion*>(reg.get());
188 xtpdft->AddRegion(*polarregion);
189 } else {
190 throw std::runtime_error(
191 "Interaction of regions with types:" + this->identify() + " and " +
192 reg->identify() + " not implemented");
193 }
194 }
195
196 e_ext = std::accumulate(interact_energies.begin(), interact_energies.end(),
197 0.0);
198 XTP_LOG(Log::info, log_)
199 << TimeStamp()
200 << " Calculated interaction potentials with other regions. E[hrt]= "
201 << e_ext << std::flush;
202 XTP_LOG(Log::info, log_) << "Writing inputs" << std::flush;
203 xtpdft->WriteInputFile(orb_);
204 bool active_run = xtpdft->RunActiveRegion();
205 if (!active_run) {
206 throw std::runtime_error("\n DFT in DFT embedding failed. Stopping!");
207 }
208 Logfile_parse = xtpdft->ParseLogFile(orb_);
209 if (!Logfile_parse) {
210 throw std::runtime_error("\n Parsing DFT logfile failed. Stopping!");
211 }
212 }
213
214 QMState state = QMState("groundstate");
215 double energy = orb_.getDFTTotalEnergy();
216
217 if (initstate_.Type().isKSState()) {
218 state = statetracker_.CalcStateAndUpdate(orb_);
219 // if unoccupied, add QP level energy
220 if (state.StateIdx() > orb_.getHomo()) {
221 energy += orb_.getExcitedStateEnergy(state);
222 } else {
223 // if unoccupied, subtract QP level energy
224 energy -= orb_.getExcitedStateEnergy(state);
225 }
226 }
227
228 if (do_gwbse_) {
229 if (do_dft_in_dft_) {
230 Index active_electrons = orb_.getNumOfActiveElectrons();
231 orb_.MOs() = orb_.getEmbeddedMOs();
232 orb_.setNumberOfAlphaElectrons(active_electrons);
233 orb_.setNumberOfOccupiedLevels(active_electrons / 2);
234 }
235 GWBSE gwbse(orb_);
236 gwbse.setLogger(&log_);
237 gwbse.Initialize(gwbseoptions_);
238 gwbse.Evaluate();
239 state = statetracker_.CalcStateAndUpdate(orb_);
240 if (state.Type().isExciton()) {
241 energy += orb_.getExcitedStateEnergy(state);
242 } else {
243 // if unoccupied, add QP level energy
244 if (state.StateIdx() > orb_.getHomo()) {
245 energy += orb_.getExcitedStateEnergy(state);
246 } else {
247 // if unoccupied, subtract QP level energy
248 energy -= orb_.getExcitedStateEnergy(state);
249 }
250 }
251 }
252 // If truncation was enabled then rewrite full basis/aux-basis, MOs in full
253 // basis and full QMAtoms
254 if (orb_.getCalculationType() == "Truncated") {
255 orb_.MOs().eigenvectors() = orb_.getTruncMOsFullBasis();
256 orb_.QMAtoms().clearAtoms();
257 orb_.QMAtoms() = originalmol;
258 orb_.SetupDftBasis(orb_.getDftBasis().Name());
259 orb_.SetupAuxBasis(orb_.getAuxBasis().Name());
260 }
261 E_hist_.push_back(energy);
262
263 Dmat_hist_.push_back(orb_.DensityMatrixFull(state));
264 return;
265}

276double QMRegion::charge() const {
277 double charge = 0.0;
278 if (!do_gwbse_) {
279 Index nuccharge = 0;
280 for (const QMAtom& a : orb_.QMAtoms()) {
281 nuccharge += a.getNuccharge();
282 }
283
284 Index electrons = orb_.getNumberOfAlphaElectrons() * 2;
285 charge = double(nuccharge - electrons);
286 } else {
287 QMState state = statetracker_.InitialState();
288 if (state.Type().isExciton()) {
289 charge = 0.0;
290 } else if (state.Type().isSingleParticleState() ||
291 state.Type().isKSState()) {
292 if (state.StateIdx() <= orb_.getHomo()) {
293 charge = +1.0;
294 } else {
295 charge = -1.0;
296 }
297 }
298 }
299 return charge;
300}

353void QMRegion::ApplyQMFieldToPolarSegments(
354 std::vector<PolarSegment>& segments) const {
355
356 Vxc_Grid grid;
357 AOBasis basis =
358 orb_.getDftBasis(); // grid needs a basis in scope all the time
359 grid.GridSetup(grid_accuracy_for_ext_interaction_, orb_.QMAtoms(), basis);
360 DensityIntegration<Vxc_Grid> numint(grid);
361
362 QMState state = QMState("groundstate");
363 if (do_gwbse_ || initstate_.Type().isKSState()) {
364 state = statetracker_.CalcState(orb_);
365 }
366 Eigen::MatrixXd dmat = orb_.DensityMatrixFull(state);
367 double Ngrid = numint.IntegrateDensity(dmat);
368 AOOverlap overlap;
369 overlap.Fill(basis);
370 double N_comp = dmat.cwiseProduct(overlap.Matrix()).sum();
371 if (std::abs(Ngrid - N_comp) > 0.001) {
372 XTP_LOG(Log::error, log_) << "=======================" << std::flush;
373 XTP_LOG(Log::error, log_)
374 << "WARNING: Calculated Densities at Numerical Grid, Number of "
375 "electrons "
376 << Ngrid << " is far away from the the real value " << N_comp
377 << ", you should increase the accuracy of the integration grid."
378 << std::flush;
379 }
380#pragma omp parallel for
381 for (Index i = 0; i < Index(segments.size()); ++i) {
382 PolarSegment& seg = segments[i];
383 for (PolarSite& site : seg) {
384 site.V_noE() += numint.IntegrateField(site.getPos());
385 }
386 }
387
388 StaticSegment seg(orb_.QMAtoms().getType(), orb_.QMAtoms().getId());
389 for (const QMAtom& atom : orb_.QMAtoms()) {
390 seg.push_back(StaticSite(atom, double(atom.getNuccharge())));
391 }
392 AddNucleiFields(segments, seg);
393}
