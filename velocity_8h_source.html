71 bool Velocity::EvaluateFrame(Topology *top) {
72 
73  // ============================================== //
74  // READ IN ALL SITES WHICH TO INCLUDE IN ANALYSIS //
75 
76  std::string line;
77  std::ifstream intt;
78  intt.open(_includeSegsFile.c_str());
79 
80  if (intt.is_open() ) {
81  while ( intt.good() ) {
82  std::getline(intt, line);
83 
84  vector< string > split;
85  Tokenizer toker(line, " ");
86  toker.ToVector(split);
87  if ( !split.size() ||
88  split[0] == "#" ||
89  split[0].substr(0,1) == "#" ) { continue; }
90 
91  int segId = boost::lexical_cast<int>(split[0]);
92  _sites.push_back(top->getSegment(segId));
93  }
94  }
95  else {
96  throw std::runtime_error("No such file: '"+_includeSegsFile+"'.");
97  }
98 
99  cout << endl << "... ... Including " << _sites.size() << " sites." << flush;
100 
101 
102  for (unsigned int i = 0; i < _states.size(); ++i) {
103 
104  int state = _states[i];
105 
106  // ================================= //
107  // CALCULATE SITE OCC. PROBABILITIES //
108 
109  double Z = 0;
110  vector<Segment*> ::iterator sit;
111  for (sit = _sites.begin(); sit < _sites.end(); ++sit) {
112 
113  Z += exp( - (*sit)->getSiteEnergy(state) / _kT );
114  }
115 
116  double check_sum = 0.0;
117 
118  for (sit = _sites.begin(); sit < _sites.end(); ++sit) {
119 
120  double p = exp( - (*sit)->getSiteEnergy(state) / _kT ) / Z;
121  check_sum += p;
122 
123  (*sit)->setOcc(p, state);
124  }
125 
126  cout << endl << "1 = " << check_sum << flush;
127 
128 
129  // ================================= //
130  // CALCULATE AVG VELOCITY //
131 
132  QMNBList &nblist = top->NBList();
133 
134  vector<Segment*> ::iterator sit1;
135  vector<Segment*> ::iterator sit2;
136 
137  vec avg_v = vec(0,0,0);
138 
139  for (sit1 = _sites.begin(); sit1 < _sites.end(); ++sit1) {
140  for (sit2 = sit1 + 1; sit2 < _sites.end(); ++sit2) {
141 
142  QMPair *qmp = nblist.FindPair((*sit1),(*sit2));
143 
144  if (qmp == NULL) {
145  continue;
146  }
147 
148  avg_v += (*sit1)->getOcc(state) * qmp->getRate12(state) * qmp->R();
149  avg_v += - (*sit2)->getOcc(state) * qmp->getRate21(state) * qmp->R();
150  }}
151 
152 
153  avg_v *= 1e-9;
154 
155  cout << endl
156  << "... ... State " << state << ": v = " << avg_v
157  << flush;
158 
159  }
160 
161  _sites.clear();
162 
163  return true;
164 }
