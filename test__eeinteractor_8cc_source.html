61 double CalcDipoleEnergy(StaticSegment& seg1, StaticSegment& seg2) {
62  Eigen::Vector3d R = (seg2[0].getPos() - seg1[0].getPos());
63  Eigen::Vector3d mu1 = seg1[0].getDipole();
64  Eigen::Vector3d mu2 = seg2[0].getDipole();
65  double r = R.norm();
66  double e_ref =
67  (r * r * mu1.dot(mu2) - 3 * mu1.dot(R) * mu2.dot(R)) / std::pow(r, 5);
68  return e_ref;
69 }

109 double CalcQuadrupoleEnergy(StaticSegment& seg1, StaticSegment& seg2) {
110  Eigen::Vector3d R = (seg2[0].getPos() - seg1[0].getPos());
111  double q20_1 = seg1[0].Q()[4];
112  double q20_2 = seg2[0].Q()[4];
113  double r = R.norm();
114  Eigen::Vector3d Rhat = R / r;
115  Eigen::Vector3d na = Eigen::Vector3d::UnitZ();
116  Eigen::Vector3d nb = Eigen::Vector3d::UnitZ();
117  double naR = Rhat.dot(na);
118  double nbR = Rhat.dot(nb);
119  // stone book p51 eqn 3.3.17
120  double e_ref = q20_1 * q20_2 / std::pow(r, 5) * 0.75 *
121  (35 * naR * naR * nbR * nbR - 5 * naR - 5 * nbR -
122  20 * naR * nbR * na.dot(nb) + 2 * na.dot(nb) + 1);
123  return e_ref;
124 }
