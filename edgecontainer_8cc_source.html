52 int EdgeContainer::getDegree(const int vertex) const {
53  if (!adj_list_.count(vertex)) throw invalid_argument("vertex is not defined");
54  int degree_count = 0;
55  if (adj_list_.at(vertex).size() == 0) {
56  return degree_count;
57  }
58  for (const pair<int, int>& neighbor_and_count : adj_list_.at(vertex)) {
59  if (neighbor_and_count.first == vertex) {
60  degree_count += neighbor_and_count.second * 2;
61  } else {
62  degree_count += neighbor_and_count.second;
63  }
64  }

107 void EdgeContainer::addEdge(Edge edge) {
108 
109  int point1 = edge.getEndPoint1();
110  int point2 = edge.getEndPoint2();
111  if (adj_list_[point1].count(point2)) {
112  ++adj_list_[point1][point2];
113  } else {
114  adj_list_[point1][point2] = 1;
115  }
116 
117  // Do not add the same edge a second time if the points are the same
118  if (point1 != point2) {
119  if (adj_list_[point2].count(point1)) {
120  ++adj_list_[point2][point1];
121  } else {
122  adj_list_[point2][point1] = 1;
123  }
124  }
125  return;
126 }

166 vector<Edge> EdgeContainer::getEdges() const {
167  unordered_map<Edge, int> extra_edge_count;
168  for (const auto& vertex_and_neigh_and_count : adj_list_) {
169  for (const pair<int, int>& neigh_and_count :
170  vertex_and_neigh_and_count.second) {
171  extra_edge_count[Edge(vertex_and_neigh_and_count.first,
172  neigh_and_count.first)] = neigh_and_count.second;
173  }
174  }
175  vector<Edge> edges;
176  for (pair<const Edge, int>& edge_count : extra_edge_count) {
177  for (int count = 0; count < edge_count.second; ++count) {
178  edges.push_back(edge_count.first);
179  }
180  }
181  return edges;
182 }
