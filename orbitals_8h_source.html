115 Index NumberofStates(QMStateType type) const {
116 switch (type.Type()) {
117 case QMStateType::Singlet:
118 return Index(BSE_singlet_.eigenvalues().size());
119 break;
120 case QMStateType::Triplet:
121 return Index(BSE_triplet_.eigenvalues().size());
122 break;
123 case QMStateType::KSstate:
124 return Index(mos_.eigenvalues().size());
125 break;
126 case QMStateType::PQPstate:
127 return Index(QPpert_energies_.size());
128 break;
129 case QMStateType::DQPstate:
130 return Index(QPdiag_.eigenvalues().size());
131 break;
132 default:
133 return 1;
134 break;
135 }
