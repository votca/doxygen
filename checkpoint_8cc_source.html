61 CheckpointFile::CheckpointFile(std::string fN, CheckpointAccessLevel access)
62  : _fileName(fN), _accessLevel(access) {
63 
64  try {
65  H5::Exception::dontPrint();
66  hid_t fcpl_id = H5Pcreate(H5P_FILE_CREATE);
67  H5::FileCreatPropList fcpList(fcpl_id);
68  switch (_accessLevel) {
69  case CheckpointAccessLevel::READ:
70  _fileHandle = H5::H5File(_fileName, H5F_ACC_RDONLY);
71  break;
72  case CheckpointAccessLevel::CREATE:
73  _fileHandle = H5::H5File(_fileName, H5F_ACC_TRUNC, fcpList);
74  break;
75  case CheckpointAccessLevel::MODIFY:
76  if (!FileExists(_fileName)) {
77  _fileHandle = H5::H5File(_fileName, H5F_ACC_TRUNC, fcpList);
78  } else {
79  _fileHandle = H5::H5File(_fileName, H5F_ACC_RDWR, fcpList);
80  }
81  }
82 
83  } catch (H5::Exception&) {
84  std::stringstream message;
85  message << "Could not access file " << _fileName;
86  message << " with permission to " << _accessLevel << "." << std::endl;
87 
88  throw std::runtime_error(message.str());
89  }
90 }
