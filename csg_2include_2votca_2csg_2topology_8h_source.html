50 /* Boost deque has been chosen and contents have been replaced with objects
51  * as opposed to heap allocated types:
52  * 1. To get rid of indirection
53  * 2. Clarify ownership
54  * 3. To ensure pointers are not invalidated if the container changes size
55  * that is not a guarantee with a vector
56  * 4. To provide better contiguous memory access, not possible with std::deque
57  * or list
58  */

63  block_molecule_4x_t;

271  void setBox(const Eigen::Matrix3d &box, BoundaryCondition::eBoxtype boxtype =
272  BoundaryCondition::typeAuto) {
273  // determine box type automatically in case boxtype==typeAuto
274  if (boxtype == BoundaryCondition::typeAuto) {
275  boxtype = autoDetectBoxType(box);
276  }
277 
278  switch (boxtype) {
279  case BoundaryCondition::typeTriclinic:
280  bc_ = std::make_unique<TriclinicBox>();
281  break;
282  case BoundaryCondition::typeOrthorhombic:
283  bc_ = std::make_unique<OrthorhombicBox>();
284  break;
285  default:
286  bc_ = std::make_unique<OpenBox>();
287  break;
288  }
289
