267  void setBox(const matrix &box, BoundaryCondition::eBoxtype boxtype =
268  BoundaryCondition::typeAuto) {
269  // determine box type automatically in case boxtype==typeAuto
270  if (boxtype == BoundaryCondition::typeAuto) {
271  boxtype = autoDetectBoxType(box);
272  }
273 
274  if (_bc) {
275  delete (_bc);
276  }
277 
278  switch (boxtype) {
279  case BoundaryCondition::typeTriclinic:
280  _bc = new TriclinicBox();
281  break;
282  case BoundaryCondition::typeOrthorhombic:
283  _bc = new OrthorhombicBox();
284  break;
285  default:
286  _bc = new OpenBox();
287  break;
288  }
289
