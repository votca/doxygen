244  void setBox(const matrix &box, BoundaryCondition::eBoxtype boxtype=BoundaryCondition::typeAuto) {
245  // determine box type automatically in case boxtype==typeAuto
246  if(boxtype==BoundaryCondition::typeAuto) {
247  boxtype = autoDetectBoxType(box);
248  }
249 
250  if(_bc) {
251  delete (_bc);
252  }
253 
254  switch(boxtype) {
255  case BoundaryCondition::typeTriclinic:
256  _bc = new TriclinicBox();
257  break;
258  case BoundaryCondition::typeOrthorhombic:
259  _bc = new OrthorhombicBox();
260  break;
261  default:
262  _bc = new OpenBox();
263  break;
264  }
265
