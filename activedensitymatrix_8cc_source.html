40 Eigen::MatrixXd ActiveDensityMatrix::activedensitymatrix(
41  const Eigen::MatrixXd &localized_mo_coeff) {
42  AOBasis aobasis = orbitals_.getDftBasis();
43  AOOverlap overlap;
44  overlap.Fill(aobasis);
45  Index numOfActiveOrbs = 0;
46  std::vector<Index> numfuncpatom = aobasis.getFuncPerAtom();
47  Eigen::MatrixXd active_mo_coeff;
48 
49  for (Index LocMoCoeff_col_i = 0; LocMoCoeff_col_i < localized_mo_coeff.cols();
50  LocMoCoeff_col_i++) {
51  // Calculate orbital wise Mulliken population
52  const Eigen::MatrixXd orbital_wise_population =
53  localized_mo_coeff.col(LocMoCoeff_col_i).transpose() *
54  overlap.Matrix() *
55  localized_mo_coeff.col(LocMoCoeff_col_i).asDiagonal();
56  const Eigen::RowVectorXd MullikenPop_per_basisset =
57  orbital_wise_population.colwise().sum();
58  Index start = 0;
59  for (Index atom_id = 0; atom_id < Index(numfuncpatom.size()); atom_id++) {
60  const double MullikenPop_per_atom =
61  MullikenPop_per_basisset.segment(start, numfuncpatom[atom_id]).sum();
62  if ((std::find(activeatoms_.begin(), activeatoms_.end(), atom_id) !=
63  activeatoms_.end()) &&
64  MullikenPop_per_atom > threshold_) {
65  active_mo_coeff.conservativeResize(localized_mo_coeff.rows(),
66  numOfActiveOrbs + 1);
67  active_mo_coeff.col(numOfActiveOrbs) =
68  localized_mo_coeff.col(LocMoCoeff_col_i);
69  numOfActiveOrbs++;
70  break;
71  }
72  start += numfuncpatom[atom_id];
73  }
74  }
75  const Eigen::MatrixXd dmat_active =
76  2 * active_mo_coeff * active_mo_coeff.transpose();
77  return dmat_active;
78 }
