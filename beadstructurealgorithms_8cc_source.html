27 vector<BeadStructure> breakIntoStructures(BeadStructure& beadstructure) {
28  vector<BeadStructure> structures;
29  if (beadstructure.isSingleStructure()) {
30  structures.push_back(beadstructure);
31  } else {
32 
33  vector<Graph> sub_graphs =
34  decoupleIsolatedSubGraphs(beadstructure.getGraph());
35  for (vector<Graph>::iterator sub_graph_it = sub_graphs.begin();
36  sub_graph_it != sub_graphs.end(); ++sub_graph_it) {
37  vector<Edge> sub_graph_edges = sub_graph_it->getEdges();
38  vector<int> sub_graph_vertices = sub_graph_it->getVertices();
39  BeadStructure beadstructure_temp;
40  for (const int& vertex : sub_graph_vertices) {
41  beadstructure_temp.AddBead(beadstructure.getBead(vertex));
42  }
43  for (const Edge& edge : sub_graph_edges) {
44  beadstructure_temp.ConnectBeads(edge.getEndPoint1(),
45  edge.getEndPoint2());
46  }
47  structures.push_back(beadstructure_temp);
48  }
49  }

51 }
