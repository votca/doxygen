104 void StateReservoir::Random_injection_one_type(int nr_charges, int carrier_type, Bsumtree* site_inject_probs, GraphKMC* graph, Eventinfo* eventinfo, votca::tools::Random2 *RandomVariable){
105 
106  for (int it = 0; it < graph->Numberofnodes() ; it++) {
107  if(graph->GetNode(it)->occ() == -1) {
108  site_inject_probs->setrate(it,1.0); //
109  }
110  else {
111  site_inject_probs->setrate(it,0.0); // no double occup
112  }
113  }
114 
115  double tot_probsum;
116  int carcounter = 0;
117 
118  while(carcounter!=nr_charges){
119 
120  tot_probsum = site_inject_probs->compute_sum();
121 
122  double randn = tot_probsum*RandomVariable->rand_uniform();
123  long node_ID;
124  NodeDevice* chosen_node;
125 
126  node_ID = site_inject_probs->search(randn);
127  chosen_node = graph->GetNode(node_ID);
128 
129  if(this->ReservoirEmpty()) this->Grow(eventinfo->growsize, eventinfo->maxpairdegree);
130  this->Add_carrier_to_chosen_node(chosen_node, carrier_type, eventinfo);
131 
132  site_inject_probs->setrate(node_ID,0.0); // no double occupation
133  carcounter++;
134  }
135 }

137 void StateReservoir::Add_carrier_to_chosen_node(NodeDevice* chosen_node, int carrier_type, Eventinfo* eventinfo){
138 
139  //add carrier to node
140  int carrier_ID = this->Buy();
141  CarrierBulk* newcarrier = this->GetCarrier(carrier_ID);
142  newcarrier->SetCarrierNode(chosen_node);
143  chosen_node->AddCarrier(carrier_ID);
144 
145  //set carrier type
146  newcarrier->SetCarrierType(carrier_type);
147 
148  //initialize distance
149  newcarrier->SetDistance(votca::tools::vec(0.0,0.0,0.0));
150 
151  // New carrier is in the simulation box
152  newcarrier->SetInBox(true);
153 
154  //initialize coulomb interactions
155  newcarrier->Init_to_Coulomb(eventinfo->maxpairdegree);
156  newcarrier->Set_from_Coulomb(0.0);
