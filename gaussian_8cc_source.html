34 void Gaussian::Initialize(tools::Property& options) {
35 
36  // GAUSSIAN file names
37  std::string fileName = "system";
38  _input_file_name = fileName + ".com";
39  _log_file_name = fileName + ".log";
40  _shell_file_name = fileName + ".sh";
41  _mo_file_name = "fort.7";
42  std::string key = "package";
43  ParseCommonOptions(options);
44 
45  if (options.exists(key + ".vdWRadii")) {
46  _vdWfooter = options.get(key + ".vdWRadii").as<std::string>();
47  }
48 
49  /* G09 by default deletes functions from the basisset according to some
50  * criterion based on, a.o., the contraction coefficients. This can lead
51  * to inconsistencies when MOs are later used in VOTCA's GWBSE modules
52  * (and other post-processing routines). G09's default can be modified
53  * by the keywork int=nobasistransform. This will add this keyword
54  * automatically to the _options string for runs with G09.
55  */
56  if (_executable == "g09") {
57  std::string::size_type basistransform_pos =
58  (boost::algorithm::to_lower_copy(_options)).find("nobasistransform");
59  if (basistransform_pos == std::string::npos) {
60  _options = _options + " int=nobasistransform ";
61  }
62  }
63  std::string::size_type iop_pos;
64  // check if the guess keyword is present, if yes, append the guess later
65  if (_write_guess) {
66  iop_pos = _options.find("cards");
67  if (iop_pos != std::string::npos) {
68  _options = _options + " cards ";
69  }
70  }
71 
72  // check if the basis set is available ("/gen")
73  iop_pos = _options.find("gen");
74  if (iop_pos != std::string::npos) {
75  _write_basis_set = true;
76  } else {
77  _write_basis_set = false;
78  }
79 
80  // check if pseudopotentials are required ("pseudo")
81  iop_pos = _options.find("pseudo");
82  if (iop_pos != std::string::npos) {
83  _write_pseudopotentials = true;
84  } else {
85  _write_pseudopotentials = false;
86  }
87 }

101 /* Custom basis sets are written on a per-element basis to
102  * 'elementname'.gbs files, which are then included in the
103  * Gaussian input file using @'elementname'.gbs
104  */
105 void Gaussian::WriteBasisset(std::ofstream& com_file,
106  const QMMolecule& qmatoms) {
107 
108  std::vector<std::string> UniqueElements = qmatoms.FindUniqueElements();
109  BasisSet bs;
110  bs.Load(_basisset_name);
111  XTP_LOG(logDEBUG, *_pLog) << "Loaded Basis Set " << _basisset_name << flush;
112 
113  for (const std::string& element_name : UniqueElements) {
114 
115  const Element& element = bs.getElement(element_name);
116  /* Write each basis set to a element_name.gbs file
117  * and include the gbs file in the com-file via Gaussian's @ function
118  */
119  std::ofstream el_file;
120  std::string el_file_name = _run_dir + "/" + element_name + ".gbs";
121 
122  el_file.open(el_file_name);
123  // element name, [possibly indeces of centers], zero to indicate the end
124  com_file << "@" << element_name << ".gbs" << endl;
125  el_file << element_name << " 0" << endl;
126  for (const Shell& shell : element) {
127  // gaussian can only use S,P,SP,D,F,G shells so we split them up if not SP
128  if (shell.getType() == "SP" || !shell.isCombined()) {
129  // shell type, number primitives, scale factor
130  el_file << shell.getType() << " " << shell.getSize() << " "
131  << FortranFormat(shell.getScale()) << endl;
132  for (const GaussianPrimitive& gaussian : shell) {
133  el_file << FortranFormat(gaussian._decay);
134  for (const double& contraction : gaussian._contraction) {
135  if (contraction != 0.0) {
136  el_file << " " << FortranFormat(contraction);
137  }
138  }
139  el_file << endl;
140  }
141  } else {
142  string type = shell.getType();
143  for (unsigned i = 0; i < type.size(); ++i) {
144  string subtype = string(type, i, 1);
145  el_file << subtype << " " << shell.getSize() << " "
146  << FortranFormat(shell.getScale()) << endl;
147 
148  for (const GaussianPrimitive& gaussian : shell) {
149  el_file << FortranFormat(gaussian._decay);
150  el_file << " "
151  << FortranFormat(gaussian._contraction[FindLmax(subtype)]);
152  }
153  el_file << endl;
154  }
155  }
156  }
157 
158  el_file << "****\n";
159  el_file.close();
160  }
161 
162  com_file << endl;
163  return;
164 }

166 /* If custom ECPs are used, they need to be specified in the input file
167  * in a section following the basis set includes.
168  */
169 void Gaussian::WriteECP(std::ofstream& com_file, const QMMolecule& qmatoms) {
170  std::vector<std::string> UniqueElements = qmatoms.FindUniqueElements();
171 
172  ECPBasisSet ecp;
173  ecp.Load(_ecp_name);
174 
175  XTP_LOG(logDEBUG, *_pLog) << "Loaded Pseudopotentials " << _ecp_name << flush;
176 
177  for (const std::string& element_name : UniqueElements) {
178  try {
179  ecp.getElement(element_name);
180  } catch (std::runtime_error& error) {
181  XTP_LOG(logDEBUG, *_pLog)
182  << "No pseudopotential for " << element_name << " available" << flush;
183  continue;
184  }
185  const ECPElement& element = ecp.getElement(element_name);
186  // element name, [possibly indeces of centers], zero to indicate the end
187  com_file << element_name << " 0\n"
188  << _ecp_name << " " << element.getLmax() << " "
189  << element.getNcore() << endl;
190 
191  for (const ECPShell& shell : element) {
192  // shell type, number primitives, scale factor
193  com_file << shell.getType() << endl;
194  com_file << shell.getSize() << endl;
195 
196  for (const ECPGaussianPrimitive& gaussian : shell) {
197  com_file << gaussian._power << " " << FortranFormat(gaussian._decay)
198  << " " << FortranFormat(gaussian._contraction) << endl;
199  }
200  }
201  }
202  com_file << endl;
203  return;
204 }

212 void Gaussian::WriteBackgroundCharges(std::ofstream& com_file) {
213 
214  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
215  for (const std::unique_ptr<StaticSite>& site : _externalsites) {
216  Eigen::Vector3d pos = site->getPos() * tools::conv::bohr2ang;
217  string sitestring =
218  boost::str(fmt % pos.x() % pos.y() % pos.z() % site->getCharge());
219  if (site->getCharge() != 0.0) com_file << sitestring << endl;
220 
221  std::vector<MinimalMMCharge> split_multipoles = SplitMultipoles(*site);
222  for (const auto& mpoles : split_multipoles) {
223  Eigen::Vector3d pos = mpoles._pos * tools::conv::bohr2ang;
224  string multipole =
225  boost::str(fmt % pos.x() % pos.y() % pos.z() % mpoles._q);
226  com_file << multipole << endl;
227  }
228  }
229 
230  com_file << endl;
231  return;
232 }

234 /* An initial guess for the electron density can be provided by
235  * a set of molecular orbital coefficients in the input file,
236  * triggered by the 'guess=cards' keyword. This MUST be done in
237  * Fortran fixed format 5D15.8. The information about the guess
238  * itself is taken from a prepared orbitals object.
239  */
240 void Gaussian::WriteGuess(const Orbitals& orbitals_guess,
241  std::ofstream& com_file) {
242  Eigen::MatrixXd MOs = ReorderMOsBack(orbitals_guess);
243  com_file << "(5D15.8)" << endl;
244  int level = 1;
245  int ncolumns = 5;
246  for (int i = 0; i < MOs.cols(); ++i) {
247  com_file << setw(5) << level << endl;
248  Eigen::VectorXd mr = MOs.col(i);
249  int column = 1;
250  for (unsigned j = 0; j < mr.size(); ++j) {
251  com_file << FortranFormat(mr[j]);
252  if (column == ncolumns) {
253  com_file << std::endl;
254  column = 0;
255  }
256  column++;
257  }
258  level++;
259  if (column != 1) com_file << endl;
260  }
261  com_file << 0 << endl;
262  return;
263 }

305 bool Gaussian::WriteInputFile(const Orbitals& orbitals) {
306 
307  std::string temp_suffix = "/id";
308  std::string scratch_dir_backup = _scratch_dir;
309 
310  std::ofstream com_file;
311  std::string com_file_name_full = _run_dir + "/" + _input_file_name;
312  com_file.open(com_file_name_full);
313 
314  // header
315  WriteHeader(com_file);
316 
317  const QMMolecule& qmatoms = orbitals.QMAtoms();
318 
319  WriteCoordinates(com_file, qmatoms);
320 
321  /* The order in which the following information has to appear
322  * in the Gaussian input file is different from version g03 to
323  * version g09. The newest version (g2016) is not supported.
324  */
325  if (_executable == "g03") {
326 
327  // if we need to write basis sets, do it now
328  if (_write_basis_set) WriteBasisset(com_file, qmatoms);
329 
330  // write ECPs
331  if (_write_pseudopotentials) WriteECP(com_file, qmatoms);
332 
333  // write the background charges
334  if (_write_charges) WriteBackgroundCharges(com_file);
335 
336  // write inital guess
337  if (_write_guess) {
338  WriteGuess(orbitals, com_file);
339  }
340 
341  } else if (_executable == "g09") {
342 
343  // write the background charges
344  // if (_write_charges) WriteBackgroundCharges(_com_file, qmatoms);
345  if (_write_charges) WriteBackgroundCharges(com_file);
346  // if we need to write basis sets, do it now
347  if (_write_basis_set) WriteBasisset(com_file, qmatoms);
348 
349  // write ECPs
350  if (_write_pseudopotentials) WriteECP(com_file, qmatoms);
351 
352  // write inital guess
353  if (_write_guess) {
354  WriteGuess(orbitals, com_file);
355  }
356 
357  } else {
358  throw std::runtime_error(
359  "Gaussian executable unknown. Must be either g03 or g09.");
360  }
361 
362  com_file << _vdWfooter << endl;
363 
364  com_file << endl;
365  com_file.close();
366  // and now generate a shell script to run both jobs
367  XTP_LOG(logDEBUG, *_pLog)
368  << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
369 
370  _scratch_dir = scratch_dir_backup + temp_suffix;
371  WriteShellScript();
372  _scratch_dir = scratch_dir_backup;
373 
374  return true;
375 }

401 bool Gaussian::Run() {
402  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: running [" << _executable << " "
403  << _input_file_name << "]" << flush;
404 
405  if (std::system(NULL)) {
406  // if scratch is provided, run the shell script;
407  // otherwise run gaussian directly and rely on global variables
408  std::string command;
409  if (_scratch_dir.size() != 0) {
410  command = "cd " + _run_dir + "; tcsh " + _shell_file_name;
411  // _command = "cd " + _run_dir + "; mkdir -p " + _scratch_dir
412  // +"; " + _executable + " " + _input_file_name;
413  } else {
414  command = "cd " + _run_dir + "; mkdir -p $GAUSS_SCRDIR; " + _executable +
415  " " + _input_file_name;
416  }
417  int check = std::system(command.c_str());
418  if (check == -1) {
419  XTP_LOG(logERROR, *_pLog)
420  << _input_file_name << " failed to start" << flush;
421  return false;
422  }
423  if (CheckLogFile()) {
424  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: finished job" << flush;
425  return true;
426  } else {
427  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: job failed" << flush;
428  }
429  } else {
430  XTP_LOG(logERROR, *_pLog)
431  << _input_file_name << " failed to start" << flush;
432  return false;
433  }
434  return true;
435 }

440 void Gaussian::CleanUp() {
441 
442  // cleaning up the generated files
443  if (_cleanup.size() != 0) {
444 
445  XTP_LOG(logDEBUG, *_pLog) << "Removing " << _cleanup << " files" << flush;
446  tools::Tokenizer tok_cleanup(_cleanup, ", ");
447  std::vector<std::string> cleanup_info;
448  tok_cleanup.ToVector(cleanup_info);
449  for (const std::string& substring : cleanup_info) {
450 
451  if (substring == "com") {
452  std::string file_name = _run_dir + "/" + _input_file_name;
453  remove(file_name.c_str());
454  }
455 
456  if (substring == "sh") {
457  std::string file_name = _run_dir + "/" + _shell_file_name;
458  remove(file_name.c_str());
459  }
460 
461  if (substring == "log") {
462  std::string file_name = _run_dir + "/" + _log_file_name;
463  remove(file_name.c_str());
464  }
465 
466  if (substring == "fort.7") {
467  std::string file_name = _run_dir + "/" + substring;
468  remove(file_name.c_str());
469  }
470 
471  if (substring == "gbs" && _write_basis_set) {
472  std::vector<std::string> fileswithfileending;
473  boost::filesystem::recursive_directory_iterator fit(_run_dir);
474  boost::filesystem::recursive_directory_iterator endit;
475  while (fit != endit) {
476  if (boost::filesystem::is_regular_file(*fit) &&
477  fit->path().extension() == substring)
478  fileswithfileending.push_back(fit->path().filename().string());
479  ++fit;
480  }
481  for (const auto filename : fileswithfileending) {
482  std::string file_name = _run_dir + "/" + filename;
483  remove(file_name.c_str());
484  }
485  }
486  }
487  }
488  return;
489 }

494 bool Gaussian::ParseMOsFile(Orbitals& orbitals) {
495  std::map<int, std::vector<double> > coefficients;
496  std::map<int, double> energies;
497 
498  std::string line;
499  unsigned levels = 0;
500  unsigned level = 0;
501  unsigned basis_size = 0;
502 
503  std::string orb_file_name_full = _mo_file_name;
504  if (_run_dir != "") orb_file_name_full = _run_dir + "/" + _mo_file_name;
505  std::ifstream input_file(orb_file_name_full);
506 
507  if (input_file.fail()) {
508  XTP_LOG(logERROR, *_pLog)
509  << "File " << _mo_file_name << " with molecular orbitals is not found "
510  << flush;
511  return false;
512  } else {
513  XTP_LOG(logDEBUG, *_pLog) << "Reading MOs from " << _mo_file_name << flush;
514  }
515 
516  // number of coefficients per line is in the first line of the file (5D15.8)
517  getline(input_file, line);
518  std::vector<std::string> strs;
519  boost::algorithm::split(strs, line, boost::is_any_of("(D)"));
520  std::string format = strs.at(2);
521 
522  while (input_file) {
523 
524  getline(input_file, line);
525  // if a line has an equality sign, must be energy
526  std::string::size_type energy_pos = line.find("=");
527 
528  if (energy_pos != std::string::npos) {
529 
530  boost::trim(line);
531  tools::Tokenizer tok(line, "\t =");
532  std::vector<std::string> results = tok.ToVector();
533  level = boost::lexical_cast<int>(results.front());
534  boost::replace_first(results.back(), "D", "e");
535  energies[level] = boost::lexical_cast<double>(results.back());
536  levels++;
537 
538  } else {
539 
540  while (line.size() > 1) {
541  std::string coefficient;
542  coefficient.assign(line, 0, 15);
543  boost::trim(coefficient);
544  boost::replace_first(coefficient, "D", "e");
545  double coef = boost::lexical_cast<double>(coefficient);
546  coefficients[level].push_back(coef);
547  line.erase(0, 15);
548  }
549  }
550  }
551 
552  // some sanity checks
553  XTP_LOG(logDEBUG, *_pLog) << "Energy levels: " << levels << flush;
554  std::map<int, std::vector<double> >::iterator iter = coefficients.begin();
555  basis_size = iter->second.size();
556 
557  for (iter = coefficients.begin()++; iter != coefficients.end(); iter++) {
558  if (iter->second.size() != basis_size) {
559  XTP_LOG(logERROR, *_pLog)
560  << "Error reading " << _mo_file_name
561  << ". Basis set size change from level to level." << flush;
562  return false;
563  }
564  }
565 
566  XTP_LOG(logDEBUG, *_pLog) << "Basis set size: " << basis_size << flush;
567 
568  // copying information to the orbitals object
569  orbitals.setBasisSetSize(basis_size); // = _basis_size;
570 
571  // copying energies to the orbitals object
572  Eigen::VectorXd& mo_energies = orbitals.MOs().eigenvalues();
573  mo_energies.resize(levels);
574  for (int i = 0; i < mo_energies.size(); i++) mo_energies[i] = energies[i + 1];
575 
576  // copying mo coefficients to the orbitals object
577  Eigen::MatrixXd& mo_coefficients = orbitals.MOs().eigenvectors();
578  mo_coefficients.resize(levels, basis_size);
579  for (int i = 0; i < mo_coefficients.rows(); i++) {
580  for (int j = 0; j < mo_coefficients.cols(); j++) {
581  mo_coefficients(j, i) = coefficients[i + 1][j];
582  }
583  }
584 
585  ReorderOutput(orbitals);
586  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: done reading MOs" << flush;
587 
588  return true;
589 }

591 bool Gaussian::CheckLogFile() const {
592 
593  // check if the log file exists
594  boost::filesystem::path arg_path;
595  char ch;
596 
597  std::string full_name = (arg_path / _run_dir / _log_file_name).c_str();
598  ifstream input_file(full_name);
599 
600  if (input_file.fail()) {
601  XTP_LOG(logERROR, *_pLog)
602  << "GAUSSIAN: " << full_name << " is not found" << flush;
603  return false;
604  };
605 
606  input_file.seekg(0, ios_base::end); // go to the EOF
607 
608  // get empty lines and end of lines out of the way
609  do {
610  input_file.seekg(-2, ios_base::cur);
611  input_file.get(ch);
612  } while (ch == '\n' || ch == ' ' || ch == '\t' ||
613  (int)input_file.tellg() == -1);
614 
615  // get the beginning of the line or the file
616  do {
617  input_file.seekg(-2, ios_base::cur);
618  input_file.get(ch);
619  } while (ch != '\n' && (int)input_file.tellg() != -1);
620 
621  std::string line;
622  getline(input_file, line);
623  input_file.close();
624 
625  std::string::size_type success = line.find("Normal termination of Gaussian");
626  if (success == std::string::npos) {
627  XTP_LOG(logERROR, *_pLog)
628  << "GAUSSIAN: " << full_name << " is incomplete" << flush;
629  return false;
630  } else {
631  return true;
632  }
633 }

635 StaticSegment Gaussian::GetCharges() const {
636  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: parsing " << _log_file_name << flush;
637 
638  StaticSegment result("charges", 0);
639  std::string log_file_name_full = _log_file_name;

780 void Gaussian::GetCoordinates(T& mol,
781  const std::vector<std::string>& archive) const {
782  typedef typename std::iterator_traits<typename T::iterator>::value_type Atom;
783  XTP_LOG(logDEBUG, *_pLog) << "Getting the coordinates" << flush;
784  bool has_atoms = mol.size() > 0;
785  tools::Tokenizer tok(archive[3], "\\");
786  std::vector<std::string> atom_block = tok.ToVector();
787  for (unsigned i = 0; i < atom_block.size() - 1; i++) {
788  tools::Tokenizer tok2(atom_block[i + 1], ",");
789  std::vector<std::string> atom = tok2.ToVector();
790  std::string atom_type = atom[0];
791  int endindex = atom.size() - 1;

793  double y = boost::lexical_cast<double>(atom[endindex - 1]);
794  double z = boost::lexical_cast<double>(atom[endindex]);
795  Eigen::Vector3d pos(x, y, z);
796  pos *= tools::conv::ang2bohr;
797  if (has_atoms == false) {
798  mol.push_back(Atom(i, atom_type, pos));
799  } else {
800  Atom& pAtom = mol.at(i);
801  pAtom.setPos(pos);
802  }
803  }

809 bool Gaussian::ParseLogFile(Orbitals& orbitals) {
810  std::string line;
811  std::vector<std::string> results;
812  bool has_occupied_levels = false;
813  bool has_unoccupied_levels = false;
814 
815  int occupied_levels = 0;
816  int unoccupied_levels = 0;
817  int number_of_electrons = 0;
818  int basis_set_size = 0;
819 
820  XTP_LOG(logDEBUG, *_pLog) << "GAUSSIAN: parsing " << _log_file_name << flush;
821 
822  std::string log_file_name_full = _log_file_name;
823  if (_run_dir != "") log_file_name_full = _run_dir + "/" + _log_file_name;
824 
825  // check if LOG file is complete
826  if (!CheckLogFile()) return false;
827 
828  // save qmpackage name
829  orbitals.setQMpackage(getPackageName());
830  orbitals.setDFTbasisName(_basisset_name);
831 
832  if (_write_pseudopotentials) {
833  orbitals.setECPName(_ecp_name);
834  }
835 
836  double self_energy = 0.0;
837  bool ScaHFX_found = false;
838  // Start parsing the file line by line
839  ifstream input_file(log_file_name_full);
840  std::vector<std::string> archive;
841 
842  while (input_file) {
843 
844  getline(input_file, line);
845  boost::trim(line);
846 
847  /* Check for ScaHFX = factor of HF exchange included in functional */
848  std::string::size_type HFX_pos = line.find("ScaHFX=");
849  if (HFX_pos != std::string::npos) {
850  tools::Tokenizer tok(line, "\t ");
851  std::vector<std::string> line_split = tok.ToVector();
852  double ScaHFX = boost::lexical_cast<double>(line_split.back());
853  orbitals.setScaHFX(ScaHFX);
854  ScaHFX_found = true;
855  XTP_LOG(logDEBUG, *_pLog)
856  << "DFT with " << ScaHFX << " of HF exchange!" << flush;
857  }
858 
859  /*
860  * number of occupied and virtual orbitals
861  * N alpha electrons M beta electrons
862  */
863  std::string::size_type electrons_pos = line.find("alpha electrons");
864  if (electrons_pos != std::string::npos) {
865  tools::Tokenizer tok(line, "\t ");
866  std::vector<std::string> line_split = tok.ToVector();
867  number_of_electrons = boost::lexical_cast<int>(line_split.front());
868  orbitals.setNumberOfAlphaElectrons(number_of_electrons);
869  XTP_LOG(logDEBUG, *_pLog)
870  << "Alpha electrons: " << number_of_electrons << flush;
871  }
872 
873  /*
874  * basis set size
875  * N basis functions, M primitive gaussians, K cartesian basis functions
876  */
877  std::string::size_type basis_pos = line.find("basis functions,");
878  if (basis_pos != std::string::npos) {
879  tools::Tokenizer tok(line, "\t ");
880  std::vector<std::string> line_split = tok.ToVector();
881  basis_set_size = boost::lexical_cast<int>(line_split.front());
882  orbitals.setBasisSetSize(basis_set_size);
883  XTP_LOG(logDEBUG, *_pLog)
884  << "Basis functions: " << basis_set_size << flush;
885  }
886 
887  /*
888  * energies of occupied/unoccupied levels
889  * Alpha occ.(virt.) eigenvalues -- e1 e2 e3 e4 e5
890  */
891  std::string::size_type eigenvalues_pos = line.find("Alpha");
892  if (eigenvalues_pos != std::string::npos) {
893  std::list<std::string> stringList;
894  while (eigenvalues_pos != std::string::npos && !has_occupied_levels &&
895  !has_unoccupied_levels) {
896 
897  boost::iter_split(stringList, line, boost::first_finder("--"));
898  boost::trim(stringList.back());
899  tools::Tokenizer tok(stringList.back(), "\t ");
900  std::vector<std::string> energies = tok.ToVector();
901 
902  if (stringList.front().find("virt.") != std::string::npos) {
903  unoccupied_levels += energies.size();
904  energies.clear();
905  }
906  if (stringList.front().find("occ.") != std::string::npos) {
907  occupied_levels += energies.size();
908  energies.clear();
909  }
910  getline(input_file, line);
911  eigenvalues_pos = line.find("Alpha");
912  boost::trim(line);
913 
914  if (eigenvalues_pos == std::string::npos) {
915  has_occupied_levels = true;
916  has_unoccupied_levels = true;
917  orbitals.setNumberOfOccupiedLevels(occupied_levels);
918  XTP_LOG(logDEBUG, *_pLog)
919  << "Occupied levels: " << occupied_levels << flush;
920  XTP_LOG(logDEBUG, *_pLog)
921  << "Unoccupied levels: " << unoccupied_levels << flush;
922  }
923  } // end of the while loop
924  } // end of the eigenvalue parsing
925 
926  /*
927  * Coordinates of the final configuration
928  * stored in the archive at the end of the file
929  */
930  GetArchive(archive, line, input_file);
931 
932  std::string::size_type self_energy_pos =
933  line.find("Self energy of the charges");
934 
935  if (self_energy_pos != std::string::npos) {
936  XTP_LOG(logDEBUG, *_pLog) << "Getting the self energy\n";
937  tools::Tokenizer tok(line, "=");
938  std::vector<std::string> block = tok.ToVector();
939  tools::Tokenizer tok2(block[1], "\t ");
940  std::vector<std::string> energy = tok2.ToVector();
941  self_energy = boost::lexical_cast<double>(energy[0]);
942  XTP_LOG(logDEBUG, *_pLog) << "Self energy " << self_energy << flush;
943  }
944 
945  } // end of reading the file line-by-line
946 
947  if (archive.empty()) {
948  throw std::runtime_error(
949  "Gaussian log file is missing the archive at the end.");
950  }
951 
952  QMMolecule& mol = orbitals.QMAtoms();
953  GetCoordinates(mol, archive);
954  double qm_energy = GetQMEnergy(archive);
955  orbitals.setQMEnergy(qm_energy - self_energy);
956  XTP_LOG(logDEBUG, *_pLog) << "Done parsing" << flush;
957  input_file.close();
958 
959  if (!ScaHFX_found) {
960  XTP_LOG(logDEBUG, *_pLog)
961  << "WARNING === WARNING \n, could not find ScaHFX= entry in log."
962  "\n probably you forgt #P in the beginning of the input file.\n"
963  " If you are running a hybrid functional calculation redo it! Now! "
964  "Please!\n ===WARNING=== \n"
965  << flush;
966  orbitals.setScaHFX(0.0);
967  }
968 
969  return true;
970 }
