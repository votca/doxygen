94 inline CGMoleculeDef *CGEngine::getMoleculeDef(string name)
95 {
96  map<string, CGMoleculeDef*>::iterator iter;
97 
98  // if there is only 1 molecule definition, don't care about the name
99  if(_molecule_defs.size()==1 && name == "unnamed") {
100  return (*(_molecule_defs.begin())).second;
101  }
102 
103  iter = _molecule_defs.find(name);
104  if(iter == _molecule_defs.end()) return NULL;
105  return (*iter).second;
106 }
