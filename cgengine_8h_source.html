92 inline CGMoleculeDef *CGEngine::getMoleculeDef(std::string name) {
93  std::map<std::string, CGMoleculeDef *>::iterator iter;
94 
95  // if there is only 1 molecule definition, don't care about the name
96  if (_molecule_defs.size() == 1 && name == "unnamed") {
97  return (*(_molecule_defs.begin())).second;
98  }
99 
100  iter = _molecule_defs.find(name);
101  if (iter == _molecule_defs.end()) {
102  return nullptr;
103  }
104  return (*iter).second;
105 }
