25 int FindLmax(const std::string& type) {
26  int lmax = std::numeric_limits<int>::min();
27  // single type shells
28  if (type.length() == 1) {
29  if (type == "S") {
30  lmax = 0;
31  } else if (type == "P") {
32  lmax = 1;
33  } else if (type == "D") {
34  lmax = 2;
35  } else if (type == "F") {
36  lmax = 3;
37  } else if (type == "G") {
38  lmax = 4;
39  } else if (type == "H") {
40  lmax = 5;
41  } else if (type == "I") {
42  lmax = 6;
43  } else {
44  throw std::runtime_error("FindLmax: Shelltype not known");
45  }
46  } else {
47  for (unsigned i = 0; i < type.length(); ++i) {
48  std::string local_shell = std::string(type, i, 1);
49  int test = FindLmax(local_shell);
50  if (test > lmax) {
51  lmax = test;
52  }
53  }
54  }
55  return lmax;
56 }

243 void BasisSet::Load(const std::string& name) {
244 
245  _name = name;
246  // if name contains .xml, assume a basisset .xml file is located in the
247  // working directory
248  std::size_t found_xml = name.find(".xml");
249  std::string xmlFile;
250  if (found_xml != std::string::npos) {
251  xmlFile = name;
252  } else {
253  // get the path to the shared folders with xml files
254  char* votca_share = getenv("VOTCASHARE");
255  if (votca_share == NULL)
256  throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
257  xmlFile = std::string(getenv("VOTCASHARE")) +
258  std::string("/xtp/basis_sets/") + name + std::string(".xml");
259  }
260  tools::Property basis_property;
261  basis_property.LoadFromXML(xmlFile);
262  std::vector<tools::Property*> elementProps =
263  basis_property.Select("basis.element");
264 
265  for (tools::Property* elementProp : elementProps) {
266  std::string elementName = elementProp->getAttribute<std::string>("name");
267  Element& element = addElement(elementName);
268  std::vector<tools::Property*> shellProps = elementProp->Select("shell");
269  for (tools::Property* shellProp : shellProps) {
270  std::string shellType = shellProp->getAttribute<std::string>("type");
271  double shellScale = shellProp->getAttribute<double>("scale");
272 
273  Shell& shell = element.addShell(shellType, shellScale);
274  std::vector<tools::Property*> constProps = shellProp->Select("constant");
275  for (tools::Property* constProp : constProps) {
276  double decay = constProp->getAttribute<double>("decay");
277  std::vector<double> contraction =
278  std::vector<double>(shell.getLmax() + 1, 0.0);
279  std::vector<tools::Property*> contrProps =
280  constProp->Select("contractions");
281  for (tools::Property* contrProp : contrProps) {
282  std::string contrType = contrProp->getAttribute<std::string>("type");
283  double contrFactor = contrProp->getAttribute<double>("factor");
284  if (contrType == "S")
285  contraction[0] = contrFactor;
286  else if (contrType == "P")
287  contraction[1] = contrFactor;
288  else if (contrType == "D")
289  contraction[2] = contrFactor;
290  else if (contrType == "F")
291  contraction[3] = contrFactor;
292  else if (contrType == "G")
293  contraction[4] = contrFactor;
294  else if (contrType == "H")
295  contraction[5] = contrFactor;
296  else if (contrType == "I")
297  contraction[6] = contrFactor;
298  else {
299  throw std::runtime_error("LoadBasiset:Contractiontype not known");
300  }
301  }
302  shell.addGaussian(decay, contraction);
303  }
304  }
305  }
306  return;
307 }
