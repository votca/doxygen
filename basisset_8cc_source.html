211 void BasisSet::LoadBasisSet ( std::string name )
212 {
213  Property basis_property;
214 
215  // if name contains .xml, assume a basisset .xml file is located in the working directory
216  std::size_t found_xml = name.find(".xml");
217  std::string xmlFile;
218  if (found_xml!=std::string::npos) {
219  xmlFile = name;
220  } else {
221  // get the path to the shared folders with xml files
222  char *votca_share = getenv("VOTCASHARE");
223  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
224  xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/xtp/basis_sets/") + name + std::string(".xml");
225  }
226  bool success = load_property_from_xml(basis_property, xmlFile);
227 
228  if ( !success ) {; }
229 
230  list<Property*> elementProps = basis_property.Select("basis.element");
231 
232  for (list<Property*> ::iterator ite = elementProps.begin(); ite != elementProps.end(); ++ite)
233  {
234  std::string elementName = (*ite)->getAttribute<std::string>("name");
235  Element *element = addElement( elementName );
236  //cout << "\nElement " << elementName;
237 
238  list<Property*> shellProps = (*ite)->Select("shell");
239  for (list<Property*> ::iterator its = shellProps.begin(); its != shellProps.end(); ++its)
240  {
241  std::string shellType = (*its)->getAttribute<std::string>("type");
242  double shellScale = (*its)->getAttribute<double>("scale");
243 
244  Shell* shell = element->addShell( shellType, shellScale );
245  //cout << "\n\tShell " << shellType;
246 
247  list<Property*> constProps = (*its)->Select("constant");
248  for (list<Property*> ::iterator itc = constProps.begin(); itc != constProps.end(); ++itc) {
249  double decay = (*itc)->getAttribute<double>("decay");
250  // << " decay "<<decay<<endl;
251  std::vector<double> contraction;
252  contraction.resize(shell->getLmax()+1);
253  list<Property*> contrProps = (*itc)->Select("contractions");
254  for (list<Property*> ::iterator itcont = contrProps.begin(); itcont != contrProps.end(); ++itcont){
255  std::string contrType = (*itcont)->getAttribute<std::string>("type");
256  double contrFactor = (*itcont)->getAttribute<double>("factor");
257  //cout << " factor " << contrFactor << endl;
258  if ( contrType == "S" ) contraction[0] = contrFactor;
259  else if ( contrType == "P" ) contraction[1] = contrFactor;
260  else if ( contrType == "D" ) contraction[2] = contrFactor;
261  else if ( contrType == "F" ) contraction[3] = contrFactor;
262  else if ( contrType == "G" ) contraction[4] = contrFactor;
263  else if ( contrType == "H" ) contraction[5] = contrFactor;
264  else if ( contrType == "I" ) contraction[6] = contrFactor;
265  else{
266  throw runtime_error("LoadBasiset:Contractiontype not known");
267  }
268  }
269  //cout<<"contraction ";
270  // for( int i=0;i<contraction.size();i++){
271  // cout<<contraction[i];
272  // }
273  // cout<<std::endl;
274  shell->addGaussian(decay, contraction);
275  }
276 
277  }
