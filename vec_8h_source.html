266 inline std::istream &operator>>(std::istream &in, vec& v)
267 {
268  char c;
269  in.get(c);
270  if(c != '[') {
271  throw std::runtime_error("error, invalid character in vector string");
272  }
273 
274  string str;
275  while (in.good()) {
276  in.get(c);
277  if(c==']') { // found end of vector
278  Tokenizer tok(str, ",");
279  vector<double> d;
280  tok.ConvertToVector(d);
281  if(d.size() != 3)
282  throw std::runtime_error("error, invalid number of entries in vector");
283  v.setX(d[0]);
284  v.setY(d[1]);
285  v.setZ(d[2]);
286  return in;
287  }
288  str += c;
289  }
290  throw std::runtime_error("did not find closing bracket in string to vec conversion");
291 
292  return in;
293 }
