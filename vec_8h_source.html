163 inline vec::vec(const std::string &str)
164 {
165  // usage: vec(" 1 2.5 17 "); separator = spaces
166  Tokenizer tok(str, " ");
167  std::vector< std::string > values;
168  tok.ToVector(values);
169  if (values.size()!=3)
170  {
171  throw std::runtime_error("\n\n\t error, string to vec, size!=3\n\n");

173  try {
174  _x = boost::lexical_cast<double>(values[0]);
175  _y = boost::lexical_cast<double>(values[1]);
176  _z = boost::lexical_cast<double>(values[2]);

286 inline std::istream &operator>>(std::istream &in, vec& v)
287 {
288  char c;
289  in.get(c);
290  if(c != '[') {
291  throw std::runtime_error("error, invalid character in vector string");
292  }
293 
294  std::string str;
295  while (in.good()) {
296  in.get(c);
297  if(c==']') { // found end of vector
298  Tokenizer tok(str, ",");
299  std::vector<double> d;
300  tok.ConvertToVector(d);
301  if(d.size() != 3)
302  throw std::runtime_error("error, invalid number of entries in vector");
303  v.setX(d[0]);
304  v.setY(d[1]);
305  v.setZ(d[2]);
306  return in;
307  }
308  str += c;
309  }
310  throw std::runtime_error("did not find closing bracket in string to vec conversion");
311 
312  return in;
313 }
