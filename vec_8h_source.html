136 inline vec::vec(const string &str)
137 {
138  // usage: vec(" 1 2.5 17 "); separator = spaces
139  Tokenizer tok(str, " ");
140  vector< string > values;
141  tok.ToVector(values);
142  if (values.size()!=3)
143  {
144  throw std::runtime_error("\n\n\t error, string to vec, size!=3\n\n");
145  }
146  try {
147  _x = boost::lexical_cast<double>(values[0]);
148  _y = boost::lexical_cast<double>(values[1]);
149  _z = boost::lexical_cast<double>(values[2]);
150  }
151  catch(const boost::bad_lexical_cast& e)
152  {
153  throw std::runtime_error("\n\n\t error, string to vec, can't convert string to double\n\n");
154  }
155 }

235 inline std::istream &operator>>(std::istream &in, vec& v)
236 {
237  char c;
238  in.get(c);
239  if(c != '[') {
240  throw std::runtime_error("error, invalid character in vector string");
241  }
242 
243  string str;
244  while (in.good()) {
245  in.get(c);
246  if(c==']') { // found end of vector
247  Tokenizer tok(str, ",");
248  vector<double> d;
249  tok.ConvertToVector(d);
250  if(d.size() != 3)
251  throw std::runtime_error("error, invalid number of entries in vector");
252  v.setX(d[0]);
253  v.setY(d[1]);
254  v.setZ(d[2]);
255  return in;
256  }
257  str += c;
258  }
259  throw std::runtime_error("did not find closing bracket in string to vec conversion");
260 
261  return in;
262 }
