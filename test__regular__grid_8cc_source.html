155  bool check_larger = (point.array() > max).any();
156  BOOST_CHECK_EQUAL(check_larger, false);
157  if (check_larger) {
158  std::cout << point.transpose() << std::endl;
159  std::cout << max.transpose() << std::endl;
160  }
161  bool check_smaller = (point.array() < min).any();
162  BOOST_CHECK_EQUAL(check_smaller, false);
163  if (check_smaller) {
164  std::cout << point.transpose() << std::endl;
165  std::cout << min.transpose() << std::endl;
166  }
167  }
168  }
