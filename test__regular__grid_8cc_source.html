70BOOST_AUTO_TEST_CASE(regular_grid_build_large) {
71 libint2::initialize();
72 QMMolecule mol("none", 0);
73
74 mol.LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
75 "/regular_grid/molecule.xyz");
76 auto extend = mol.CalcSpatialMinMax();
77 AOBasis aobasis = CreateBasis(mol);
78
79 Regular_Grid grid;
80 Eigen::Array<votca::Index, 3, 1> steps(30, 30, 30);
81 Eigen::Array3d padding(1.0, 1.0, 1.0);
82 grid.GridSetup(steps, padding, mol, aobasis);
83 auto max = extend.second.array() + padding + 1e-9;
84 auto min = extend.first.array() - padding - 1e-9;
85 for (auto& a : grid) {
86 for (auto& point : a.getGridPoints()) {
87 bool check_larger = (point.array() > max).any();
88 BOOST_CHECK_EQUAL(check_larger, false);
89 if (check_larger) {
90 std::cout << point.transpose() << std::endl;
91 std::cout << max.transpose() << std::endl;
92 }
93 bool check_smaller = (point.array() < min).any();
94 BOOST_CHECK_EQUAL(check_smaller, false);
95 if (check_smaller) {
96 std::cout << point.transpose() << std::endl;
97 std::cout << min.transpose() << std::endl;
98 }
99 }
100 }
101
102 BOOST_CHECK_EQUAL(grid.getGridSize(), 27000);
103 BOOST_CHECK_EQUAL(grid.getBoxesSize(), 54);
104
105 BOOST_CHECK_CLOSE(grid[0].getGridPoints()[0].x(), min.x(), 1e-5);
106 BOOST_CHECK_CLOSE(grid[0].getGridPoints()[0].y(), min.y(), 1e-5);
107 BOOST_CHECK_CLOSE(grid[0].getGridPoints()[0].z(), min.z(), 1e-5);
108 BOOST_CHECK_CLOSE(grid[0].getGridPoints()[30].y(), -2.0379047286687406, 1e-5);
109 BOOST_CHECK_CLOSE(grid[0].getGridPoints()[45].z(), 0.075477952913657109,
110 1e-5);
111
112 libint2::finalize();
113}
