25 void Neighborlist::Initialize(tools::Property& options) {
26 
27  // update options with the VOTCASHARE defaults
28  UpdateWithDefaults(options, "xtp");
29  std::string key = "options." + Identify();
30 
31  std::vector<tools::Property*> segs = options.Select(key + ".segments");
32 
33  for (tools::Property* segprop : segs) {
34  std::string types = segprop->get("type").as<std::string>();
35  double cutoff = segprop->get("cutoff").as<double>() * tools::conv::nm2bohr;
36 
37  tools::Tokenizer tok(types, " ");
38  std::vector<std::string> names;
39  tok.ToVector(names);
40 
41  if (names.size() != 2) {
42  throw std::runtime_error(
43  "ERROR: Faulty pair definition for cut-off's: Need two segment names "
44  "separated by a space");
45  }
46  _cutoffs[names[0]][names[1]] = cutoff * tools::conv::nm2bohr;
47  _cutoffs[names[1]][names[0]] = cutoff * tools::conv::nm2bohr;
48  if (std::find(_included_segments.begin(), _included_segments.end(),
49  names[0]) == _included_segments.end()) {
50  _included_segments.push_back(names[0]);
51  }
52  if (std::find(_included_segments.begin(), _included_segments.end(),
53  names[1]) == _included_segments.end()) {
54  _included_segments.push_back(names[1]);
55  }
56  }
57 
58  if (options.exists(key + ".constant")) {
59  _useConstantCutoff = true;
60  _constantCutoff =
61  options.get(key + ".constant").as<double>() * tools::conv::nm2bohr;
62  } else {
63  _useConstantCutoff = false;
64  }
65  if (options.exists(key + ".exciton_cutoff")) {
66  _useExcitonCutoff = true;
67  _excitonqmCutoff = options.get(key + ".exciton_cutoff").as<double>() *
68  tools::conv::nm2bohr;
69  } else {
70  _useExcitonCutoff = false;
71  }
72 }

74 int Neighborlist::DetClassicalPairs(Topology& top) {
75  int classical_pairs = 0;
76 #pragma omp parallel for
77  for (int i = 0; i < top.NBList().size(); i++) {
78  const Segment* seg1 = top.NBList()[i]->Seg1();
79  const Segment* seg2 = top.NBList()[i]->Seg2();
80  if (top.GetShortestDist(*seg1, *seg2) > _excitonqmCutoff) {
81  top.NBList()[i]->setType(QMPair::Excitoncl);
82 #pragma omp critical
83  { classical_pairs++; }
84  } else {
85  top.NBList()[i]->setType(QMPair::Hopping);
86  }
87  } // Type 3 Exciton_classical approx
88  return classical_pairs;
89 }

91 bool Neighborlist::EvaluateFrame(Topology& top) {
92  OPENMP::setMaxThreads(_nThreads);
93  std::cout << " Using " << OPENMP::getMaxThreads() << " threads" << std::flush;
94 
95  if (tools::globals::verbose) {
96  std::cout << std::endl << "... ..." << std::flush;
97  }
98 
99  double min = top.getBox().diagonal().minCoeff();
100 
101  std::vector<Segment*> segs;
102  for (Segment& seg : top.Segments()) {
103  if (_useConstantCutoff ||
104  std::find(_included_segments.begin(), _included_segments.end(),
105  seg.getType()) != _included_segments.end()) {
106  segs.push_back(&seg);
107  seg.getApproxSize();
108  }
109  }
110  std::cout << std::endl;
111  std::cout << "Evaluating " << segs.size() << " segments for neighborlist. ";
112  if ((top.Segments().size() - segs.size()) != 0) {
113  std::cout << top.Segments().size() - segs.size()
114  << " segments are not taken into account as specified"
115  << std::endl;
116  } else {
117  std::cout << std::endl;
118  }
119  if (!_useConstantCutoff) {
120  std::cout << "The following segments are used in the neigborlist creation"
121  << std::endl;
122  std::cout << "\t" << std::flush;
123  for (const std::string& st : _included_segments) {
124  std::cout << " " << st << std::flush;
125  }
126  std::cout << std::endl;
127  }
128 
129  std::cout << "\r ... ... Evaluating " << std::flush;
130  std::vector<std::string> skippedpairs;
131 
132  top.NBList().Cleanup();
133 
134  boost::progress_display progress(segs.size());
135  // cache approx sizes
136  std::vector<double> approxsize = std::vector<double>(segs.size(), 0.0);
137 #pragma omp parallel for
138  for (unsigned i = 0; i < segs.size(); i++) {
139  approxsize[i] = segs[i]->getApproxSize();
140  }
141 #pragma omp parallel for schedule(guided)
142  for (unsigned i = 0; i < segs.size(); i++) {
143  Segment* seg1 = segs[i];
144  double cutoff = _constantCutoff;
145  for (unsigned j = i + 1; j < segs.size(); j++) {
146  Segment* seg2 = segs[j];
147  if (!_useConstantCutoff) {
148  try {
149  cutoff = _cutoffs.at(seg1->getType()).at(seg2->getType());
150  } catch (const std::exception&) {
151  std::string pairstring = seg1->getType() + "/" + seg2->getType();
152  if (std::find(skippedpairs.begin(), skippedpairs.end(), pairstring) ==
153  skippedpairs.end()) {
154 #pragma omp critical
155  { skippedpairs.push_back(pairstring); }
156  }
157  continue;
158  }
159  }
160 
161  if (cutoff > 0.5 * min) {
162  throw std::runtime_error(
163  (boost::format("Cutoff is larger than half the box size. Maximum "
164  "allowed cutoff is %1$1.1f") %
165  (tools::conv::nm2bohr * 0.5 * min))
166  .str());
167  }
168  double cutoff2 = cutoff * cutoff;
169  Eigen::Vector3d segdistance =
170  top.PbShortestConnect(seg1->getPos(), seg2->getPos());
171  double segdistance2 = segdistance.squaredNorm();
172  double outside = cutoff + approxsize[i] + approxsize[j];
173 
174  if (segdistance2 < cutoff2) {
175 #pragma omp critical
176  { top.NBList().Add(*seg1, *seg2, segdistance); }
177 
178  } else if (segdistance2 > (outside * outside)) {
179  continue;
180  } else {
181  double R = top.GetShortestDist(*seg1, *seg2);
182  if ((R * R) < cutoff2) {
183 #pragma omp critical
184  { top.NBList().Add(*seg1, *seg2, segdistance); }
185  }
186  }
187  } /* exit loop seg2 */
188 #pragma omp critical
189  { ++progress; }
190  } /* exit loop seg1 */
191 
192  if (skippedpairs.size() > 0) {
193  std::cout << "WARNING: No cut-off specified for segment pairs of type "
194  << std::endl;
195  for (const std::string& st : skippedpairs) {
196  std::cout << st << std::endl;
197  }
198  std::cout << "pairs were skipped" << std::endl;
199  }
200 
201  std::cout << std::endl
202  << " ... ... Created " << top.NBList().size() << " direct pairs.";
203  if (_useExcitonCutoff) {
204  std::cout << std::endl
205  << " ... ... Determining classical pairs " << std::endl;
206  int classical_pairs = DetClassicalPairs(top);
207  std::cout << " ... ... Found " << classical_pairs << " classical pairs "
208  << std::endl;
209  }
210 
211  // sort qmpairs by seg1id and then by seg2id then reindex the pair id
212  // according to that.
213  top.NBList().sortAndReindex([](QMPair* a, QMPair* b) {
214  if (a->Seg1()->getId() != b->Seg1()->getId()) {
215  return a->Seg1()->getId() < b->Seg1()->getId();
216  }
217  return a->Seg2()->getId() < b->Seg2()->getId();
218  });
219 
220  return true;
221 }
