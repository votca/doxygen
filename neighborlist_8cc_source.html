25 void Neighborlist::Initialize(const tools::Property& user_options) {
26 
27  tools::Property options =
28  LoadDefaultsAndUpdateWithUserOptions("xtp", user_options);
29 
30  std::vector<tools::Property*> segs = options.Select(".segments");
31 
32  for (tools::Property* segprop : segs) {
33  std::string types = segprop->get("type").as<std::string>();
34  double cutoff = segprop->get("cutoff").as<double>() * tools::conv::nm2bohr;
35 
36  tools::Tokenizer tok(types, " ");
37  std::vector<std::string> names;
38  tok.ToVector(names);
39 
40  if (names.size() != 2) {
41  throw std::runtime_error(
42  "ERROR: Faulty pair definition for cut-off's: Need two segment names "
43  "separated by a space");
44  }
45  _cutoffs[names[0]][names[1]] = cutoff * tools::conv::nm2bohr;
46  _cutoffs[names[1]][names[0]] = cutoff * tools::conv::nm2bohr;
47  if (std::find(_included_segments.begin(), _included_segments.end(),
48  names[0]) == _included_segments.end()) {
49  _included_segments.push_back(names[0]);
50  }
51  if (std::find(_included_segments.begin(), _included_segments.end(),
52  names[1]) == _included_segments.end()) {
53  _included_segments.push_back(names[1]);
54  }
55  }
56 
57  if (options.exists(".constant")) {
58  _useConstantCutoff = true;
59  _constantCutoff =
60  options.get(".constant").as<double>() * tools::conv::nm2bohr;
61  } else {
62  _useConstantCutoff = false;
63  }
64  if (options.exists(".exciton_cutoff")) {
65  _useExcitonCutoff = true;
66  _excitonqmCutoff =
67  options.get(".exciton_cutoff").as<double>() * tools::conv::nm2bohr;
68  } else {
69  _useExcitonCutoff = false;
70  }
71 }

73 Index Neighborlist::DetClassicalPairs(Topology& top) {
74  Index classical_pairs = 0;
75 #pragma omp parallel for
76  for (Index i = 0; i < top.NBList().size(); i++) {
77  const Segment* seg1 = top.NBList()[i]->Seg1();
78  const Segment* seg2 = top.NBList()[i]->Seg2();
79  if (top.GetShortestDist(*seg1, *seg2) > _excitonqmCutoff) {
80  top.NBList()[i]->setType(QMPair::Excitoncl);
81 #pragma omp critical
82  { classical_pairs++; }
83  } else {
84  top.NBList()[i]->setType(QMPair::Hopping);
85  }
86  } // Type 3 Exciton_classical approx
87  return classical_pairs;
88 }

90 bool Neighborlist::EvaluateFrame(Topology& top) {
91  OPENMP::setMaxThreads(_nThreads);
92  std::cout << " Using " << OPENMP::getMaxThreads() << " threads" << std::flush;
93 
94  double min = top.getBox().diagonal().minCoeff();
95 
96  std::vector<Segment*> segs;
97  for (Segment& seg : top.Segments()) {
98  if (_useConstantCutoff ||
99  std::find(_included_segments.begin(), _included_segments.end(),
100  seg.getType()) != _included_segments.end()) {
101  segs.push_back(&seg);
102  seg.getApproxSize();
103  }
104  }
105  std::cout << std::endl;
106  std::cout << "Evaluating " << segs.size() << " segments for neighborlist. ";
107  if ((top.Segments().size() - segs.size()) != 0) {
108  std::cout << top.Segments().size() - segs.size()
109  << " segments are not taken into account as specified"
110  << std::endl;
111  } else {
112  std::cout << std::endl;
113  }
114  if (!_useConstantCutoff) {
115  std::cout << "The following segments are used in the neigborlist creation"
116  << std::endl;
117  std::cout << "\t" << std::flush;
118  for (const std::string& st : _included_segments) {
119  std::cout << " " << st << std::flush;
120  }
121  std::cout << std::endl;
122  }
123 
124  std::cout << "\r ... ... Evaluating " << std::flush;
125  std::vector<std::string> skippedpairs;
126 
127  top.NBList().Cleanup();
128 
129  boost::progress_display progress(segs.size());
130  // cache approx sizes
131  std::vector<double> approxsize = std::vector<double>(segs.size(), 0.0);
132 #pragma omp parallel for
133  for (Index i = 0; i < Index(segs.size()); i++) {
134  approxsize[i] = segs[i]->getApproxSize();
135  }
136 #pragma omp parallel for schedule(guided)
137  for (Index i = 0; i < Index(segs.size()); i++) {
138  Segment* seg1 = segs[i];
139  double cutoff = _constantCutoff;
140  for (Index j = i + 1; j < Index(segs.size()); j++) {
141  Segment* seg2 = segs[j];
142  if (!_useConstantCutoff) {
143  try {
144  cutoff = _cutoffs.at(seg1->getType()).at(seg2->getType());
145  } catch (const std::exception&) {
146  std::string pairstring = seg1->getType() + "/" + seg2->getType();
147  if (std::find(skippedpairs.begin(), skippedpairs.end(), pairstring) ==
148  skippedpairs.end()) {
149 #pragma omp critical
150  { skippedpairs.push_back(pairstring); }
151  }
152  continue;
153  }
154  }
155 
156  if (cutoff > 0.5 * min) {
157  throw std::runtime_error(
158  (boost::format("Cutoff is larger than half the box size. Maximum "
159  "allowed cutoff is %1$1.1f") %
160  (tools::conv::nm2bohr * 0.5 * min))
161  .str());
162  }
163  double cutoff2 = cutoff * cutoff;
164  Eigen::Vector3d segdistance =
165  top.PbShortestConnect(seg1->getPos(), seg2->getPos());
166  double segdistance2 = segdistance.squaredNorm();
167  double outside = cutoff + approxsize[i] + approxsize[j];
168 
169  if (segdistance2 < cutoff2) {
170 #pragma omp critical
171  { top.NBList().Add(*seg1, *seg2, segdistance); }
172 
173  } else if (segdistance2 > (outside * outside)) {
174  continue;
175  } else {
176  double R = top.GetShortestDist(*seg1, *seg2);
177  if ((R * R) < cutoff2) {
178 #pragma omp critical
179  { top.NBList().Add(*seg1, *seg2, segdistance); }
180  }
181  }
182  } /* exit loop seg2 */
183 #pragma omp critical
184  { ++progress; }
185  } /* exit loop seg1 */
186 
187  if (skippedpairs.size() > 0) {
188  std::cout << "WARNING: No cut-off specified for segment pairs of type "
189  << std::endl;
190  for (const std::string& st : skippedpairs) {
191  std::cout << st << std::endl;
192  }
193  std::cout << "pairs were skipped" << std::endl;
194  }
195 
196  std::cout << std::endl
197  << " ... ... Created " << top.NBList().size() << " direct pairs.";
198  if (_useExcitonCutoff) {
199  std::cout << std::endl
200  << " ... ... Determining classical pairs " << std::endl;
201  Index classical_pairs = DetClassicalPairs(top);
202  std::cout << " ... ... Found " << classical_pairs << " classical pairs "
203  << std::endl;
204  }
205 
206  // sort qmpairs by seg1id and then by seg2id then reindex the pair id
207  // according to that.
208  top.NBList().sortAndReindex([](QMPair* a, QMPair* b) {
209  if (a->Seg1()->getId() != b->Seg1()->getId()) {
210  return a->Seg1()->getId() < b->Seg1()->getId();
211  }
212  return a->Seg2()->getId() < b->Seg2()->getId();
213  });
214 
215  return true;
216 }
