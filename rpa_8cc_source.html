27 void RPA::UpdateRPAInputEnergies(const Eigen::VectorXd& dftenergies,
28  const Eigen::VectorXd& gwaenergies,
29  int qpmin) {
30  int rpatotal = _rpamax - _rpamin + 1;
31  _energies = dftenergies.segment(_rpamin, rpatotal);
32  int gwsize = gwaenergies.size();
33  int lumo = _homo + 1;
34 
35  int qpmax = qpmin + gwsize - 1;
36  _energies.segment(qpmin - _rpamin, gwsize) = gwaenergies;
37  double DFTgap = dftenergies(lumo) - dftenergies(_homo);
38  double QPgap = gwaenergies(lumo - qpmin) - gwaenergies(_homo - qpmin);
39  double shift = QPgap - DFTgap;
40  int levelaboveqpmax = _rpamax - qpmax;
41  _energies.segment(qpmax + 1 - _rpamin, levelaboveqpmax).array() += shift;
42 }

44 template <bool imag>
45 Eigen::MatrixXd RPA::calculate_epsilon(double frequency) const {
46  const int size = _Mmn.auxsize();
47  std::vector<Eigen::MatrixXd> thread_result = std::vector<Eigen::MatrixXd>(
48  OPENMP::getMaxThreads(), Eigen::MatrixXd::Zero(size, size));
49  const int lumo = _homo + 1;
50  const int n_occ = lumo - _rpamin;
51  const int n_unocc = _rpamax - lumo + 1;
52  const double freq2 = frequency * frequency;
53  const double eta2 = _eta * _eta;
54 #pragma omp parallel for
55  for (int m_level = 0; m_level < n_occ; m_level++) {
56  const double qp_energy_m = _energies(m_level);
57 
58  const Eigen::MatrixXd Mmn_RPA =
59  _Mmn[m_level].block(n_occ, 0, n_unocc, size);
60 
61  const Eigen::ArrayXd deltaE =
62  _energies.segment(n_occ, n_unocc).array() - qp_energy_m;
63  Eigen::VectorXd denom;
64  if (imag) {
65  denom = 4 * deltaE / (deltaE.square() + freq2);
66  } else {
67  Eigen::ArrayXd deltEf = deltaE - frequency;
68  Eigen::ArrayXd sum = deltEf / (deltEf.square() + eta2);
69  deltEf = deltaE + frequency;
70  sum += deltEf / (deltEf.square() + eta2);
71  denom = 2 * sum;
72  }
73  auto temp = Mmn_RPA.transpose() * denom.asDiagonal();
74  Eigen::MatrixXd tempresult = temp * Mmn_RPA;
75  thread_result[OPENMP::getThreadId()] += tempresult;
76  }
77  Eigen::MatrixXd result = Eigen::MatrixXd::Identity(size, size);
78  for (const auto& mat : thread_result) {
79  result += mat;
80  }
81  return result;
82 }
