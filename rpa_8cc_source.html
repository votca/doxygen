42  void GWBSE::PPM_construct_parameters(const ub::matrix<double>& _overlap_cholesky_inverse) {
43 
44  // multiply with L-1^t from the right
45 
46  ub::matrix<double> _temp = ub::prod(_epsilon[0], ub::trans(_overlap_cholesky_inverse));
47  // multiply with L-1 from the left
48 
49  _temp = ub::prod(_overlap_cholesky_inverse, _temp);
50 
51  // get eigenvalues and eigenvectors of this matrix
52  ub::vector<double> _eigenvalues;
53  ub::matrix<double> _eigenvectors;
54 
55  linalg_eigenvalues(_temp, _eigenvalues, _eigenvectors);
56  _eigenvectors=ub::trans(_eigenvectors);
57  // multiply eigenvectors with overlap_cholesky_inverse and store as eigenvalues of epsilon
58  _ppm_phi = ub::prod(_eigenvectors, _overlap_cholesky_inverse);
59 
60  // store PPM weights from eigenvalues
61  _ppm_weight.resize(_eigenvalues.size());
62  for (unsigned _i = 0; _i < _eigenvalues.size(); _i++) {
63  _ppm_weight(_i) = 1.0 - 1.0 / _eigenvalues(_i);
64  }
65 
66  // determine PPM frequencies
67  _ppm_freq.resize(_eigenvalues.size());
68  // a) phi^t * epsilon(1) * phi
69 
70  _temp = ub::prod(_ppm_phi, _epsilon[1]);
71 
72  _eigenvectors = ub::prod(_temp, ub::trans(_ppm_phi));
73  // b) invert
74 
75  linalg_invert(_eigenvectors, _temp);
76  // c) PPM parameters -> diagonal elements
77 
78  #pragma omp parallel for
79  for (unsigned _i = 0; _i < _eigenvalues.size(); _i++) {
80 
81  if (_screening_freq(1, 0) == 0.0) {
82  if (_ppm_weight(_i) < 1.e-5) {
83  _ppm_weight(_i) = 0.0;
84  _ppm_freq(_i) = 0.5;//Hartree
85  continue;
86  } else {
87  double _nom = _temp(_i, _i) - 1.0;
88  double _frac = -1.0 * _nom / (_nom + _ppm_weight(_i)) * _screening_freq(1, 1) * _screening_freq(1, 1);
89  _ppm_freq(_i) = sqrt(std::abs(_frac));
90  }
91 
92  } else {
93  // only purely imaginary frequency assumed
94  cerr << " mixed frequency! real part: " << _screening_freq(1, 0) << " imaginary part: " << _screening_freq(1, 1) << flush;
95  exit(1);
96  }
97 
98  }
99 
100  // epsilon can be deleted
101  _epsilon[0].resize(0, 0);
102  _epsilon[1].resize(0, 0);
103 
104 
105  return;
106  }

110  ub::matrix<double> GWBSE::RPA_imaginary(const TCMatrix& _Mmn_RPA,const double screening_freq) {
111  const int _size = _Mmn_RPA.get_beta(); // size of gwbasis
112  const int index_n = _Mmn_RPA.get_nmin();
113  const int index_m = _Mmn_RPA.get_mmin();
114  const double screenf2=screening_freq * screening_freq;
115  ub::matrix<double> result=ub::zero_matrix<double>(_size);
116 
117  const ub::vector<double>& qp_energies= _qp_energies;
118 
119 
120 
121  #pragma omp parallel for
122  for (int _m_level = 0; _m_level < _Mmn_RPA.get_mtot(); _m_level++) {
123  const double _qp_energy_m=qp_energies(_m_level + index_m);
124 #if (GWBSE_DOUBLE)
125  const ub::matrix<double>& Mmn_RPA = _Mmn_RPA[ _m_level ];
126 #else
127  const ub::matrix<double> Mmn_RPA = _Mmn_RPA[ _m_level ];
128 #endif
129  // a temporary matrix, that will get filled in empty levels loop
130  ub::matrix<double> _temp = ub::matrix<double>(_Mmn_RPA.get_ntot(), _size);
131 
132  // loop over empty levels
133  for (int _n_level = 0; _n_level < _Mmn_RPA.get_ntot(); _n_level++) {
134 
135 
136  const double _deltaE = qp_energies(_n_level + index_n) -_qp_energy_m ; // get indices and units right!!!
137 
138  // this only works, if we have either purely real or purely imaginary frequencies
139 
140  // purely imaginary
141  const double _energy_factor = 4.0 * _deltaE / (_deltaE * _deltaE + screenf2);//hartree
142  for (int _i_gw = 0; _i_gw < _size; _i_gw++) {
143  _temp(_n_level, _i_gw) = _energy_factor * Mmn_RPA(_i_gw, _n_level);
144  } // matrix size
145 
146  } // empty levels
147  _temp=ub::prod(Mmn_RPA, _temp);
148  // now multiply and add to epsilon
149  #pragma omp critical
150  {
151  result+=_temp;
152  }
153  } // occupied levels
154 
155  return result;
156  }

211 
212  // loop over frequencies
213  for ( unsigned _i_freq = 0 ; _i_freq < _screening_freq.size1() ; _i_freq++ ){
214 
215  if ( _screening_freq( _i_freq, 0) == 0.0 ) {
216  _epsilon[ _i_freq ]+=RPA_imaginary(_Mmn_RPA, _screening_freq( _i_freq, 1));
217  }
218 
219  else if ( _screening_freq( _i_freq, 1) == 0.0 ) {
220  // purely real

229  } // loop over frequencies
