42  void GWBSE::PPM_construct_parameters(const ub::matrix<double>& _overlap_cholesky_inverse,const ub::matrix<double>& transcholeskyinverse) {
43 
44  // multiply with L-1^t from the right
45 
46  ub::matrix<double> _temp = ub::prod(_epsilon[0], ub::trans(_overlap_cholesky_inverse));
47  // multiply with L-1 from the left
48 
49  _temp = ub::prod(_overlap_cholesky_inverse, _temp);
50 
51  // get eigenvalues and eigenvectors of this matrix
52  ub::vector<double> _eigenvalues;
53  ub::matrix<double> _eigenvectors;
54 
55  linalg_eigenvalues(_temp, _eigenvalues, _eigenvectors);
56 
57  // multiply eigenvectors with overlap_cholesky_inverse and store as eigenvalues of epsilon
58  _ppm_phi = ub::prod(transcholeskyinverse, _eigenvectors);
59 
60  // store PPM weights from eigenvalues
61  _ppm_weight.resize(_eigenvalues.size());
62  for (unsigned _i = 0; _i < _eigenvalues.size(); _i++) {
63  _ppm_weight(_i) = 1.0 - 1.0 / _eigenvalues(_i);
64  }
65 
66  // determine PPM frequencies
67  _ppm_freq.resize(_eigenvalues.size());
68  // a) phi^t * epsilon(1) * phi
69  _temp=ub::trans(_ppm_phi);
70  _temp = ub::prod(_temp, _epsilon[1]);
71 
72  _eigenvectors = ub::prod(_temp, _ppm_phi);
73  // b) invert
74 
75  _temp = ub::zero_matrix<double>(_eigenvalues.size(), _eigenvalues.size());
76 
77  linalg_invert(_eigenvectors, _temp); //eigenvectors is destroyed after!
78  // c) PPM parameters -> diagonal elements
79 
80  #pragma omp parallel for
81  for (unsigned _i = 0; _i < _eigenvalues.size(); _i++) {
82 
83  if (_screening_freq(1, 0) == 0.0) {
84  if (_ppm_weight(_i) < 1.e-5) {
85  _ppm_weight(_i) = 0.0;
86  _ppm_freq(_i) = 0.5;//Hartree
87  continue;
88  } else {
89  double _nom = _temp(_i, _i) - 1.0;
90  double _frac = -1.0 * _nom / (_nom + _ppm_weight(_i)) * _screening_freq(1, 1) * _screening_freq(1, 1);
91  _ppm_freq(_i) = sqrt(std::abs(_frac));
92  }
93 
94  } else {
95  // only purely imaginary frequency assumed
96  cerr << " mixed frequency! real part: " << _screening_freq(1, 0) << " imaginary part: " << _screening_freq(1, 1) << flush;
97  exit(1);
98  }
99 
100  }
101 
102 
103  // will be needed transposed later
104  _ppm_phi = ub::trans(_ppm_phi);
105 
106  // epsilon can be deleted
107  _epsilon[0].resize(0, 0);
108  _epsilon[1].resize(0, 0);
109 
110 
111  return;
112  }

116  void GWBSE::RPA_imaginary(ub::matrix<double>& result, const TCMatrix& _Mmn_RPA,const double screening_freq) {
117  int _size = _Mmn_RPA[0].size1(); // size of gwbasis
118 
119  #pragma omp parallel for
120  for (int _m_level = 0; _m_level < _Mmn_RPA.get_mtot(); _m_level++) {
121  int index_m = _Mmn_RPA.get_mmin();
122 #if (GWBSE_DOUBLE)
123  const ub::matrix<double>& Mmn_RPA = _Mmn_RPA[ _m_level ];
124 #else
125  const ub::matrix<double> Mmn_RPA = _Mmn_RPA[ _m_level ];
126 #endif
127  // a temporary matrix, that will get filled in empty levels loop
128  ub::matrix<double> _temp = ub::zero_matrix<double>(_Mmn_RPA.get_ntot(), _size);
129 
130  // loop over empty levels
131  for (int _n_level = 0; _n_level < _Mmn_RPA.get_ntot(); _n_level++) {
132  int index_n = _Mmn_RPA.get_nmin();
133 
134  double _deltaE = _qp_energies(_n_level + index_n) - _qp_energies(_m_level + index_m); // get indices and units right!!!
135 
136  // this only works, if we have either purely real or purely imaginary frequencies
137 
138  // purely imaginary
139  double _energy_factor = 4.0 * _deltaE / (_deltaE * _deltaE + screening_freq * screening_freq);//hartree
140  for (int _i_gw = 0; _i_gw < _size; _i_gw++) {
141  _temp(_n_level, _i_gw) = _energy_factor * Mmn_RPA(_i_gw, _n_level);
142  } // matrix size
143 
144  } // empty levels
145 
146  // now multiply and add to epsilon
147  ub::matrix<double> _add = ub::prod(Mmn_RPA, _temp);
148  #pragma omp critical
149  {
150  result += _add;
151  }
152  } // occupied levels
153  return;
154  }

164

204  void GWBSE::RPA_calculate_epsilon(const TCMatrix& _Mmn_RPA){
205 
206  // loop over frequencies
207  for ( unsigned _i_freq = 0 ; _i_freq < _screening_freq.size1() ; _i_freq++ ){
208 
209  if ( _screening_freq( _i_freq, 0) == 0.0 ) {
210  RPA_imaginary(_epsilon[ _i_freq ],_Mmn_RPA, _screening_freq( _i_freq, 1));
211  }
212 
213  else if ( _screening_freq( _i_freq, 1) == 0.0 ) {
214  // purely real

216  }
217  else {
218  // mixed -> FAIL

221  }
222 
223  } // loop over frequencies

231  void GWBSE::RPA_prepare_threecenters(TCMatrix& _Mmn_RPA,const TCMatrix& _Mmn_full, AOBasis& gwbasis,
232  const AOMatrix& gwoverlap,const ub::matrix<double>& gwoverlap_inverse ){
233 
234  const double pi = boost::math::constants::pi<double>();
235  ub::range full=ub::range(0, gwbasis.AOBasisSize());
236  ub::range RPA_cut=ub::range(_Mmn_RPA.get_nmin() - _Mmn_full.get_nmin(), _Mmn_RPA.get_nmax() - _Mmn_full.get_nmin() + 1);
237  // loop over m-levels in _Mmn_RPA
238  #pragma omp parallel for
239  for (int _m_level = 0; _m_level < _Mmn_RPA.size(); _m_level++) {
240 
241 
242 
243  #if (GWBSE_DOUBLE)
244  const ub::matrix<double>& _Mmn_double = _Mmn_full[ _m_level ];
245 #else
246  const ub::matrix<double> _Mmn_double = _Mmn_full[ _m_level ];
247 #endif
248 
249  ub::matrix<double> _temp = ub::prod(gwoverlap_inverse, _Mmn_double);
250 
251 
252  // loop over n-levels in _Mmn_full
253  for (int _n_level = 0; _n_level < _Mmn_full.get_ntot(); _n_level++) {
254 
255  double sc_plus = 0.0;
256  double sc_minus = 0.0;
257 
258  // loop over gwbasis shells
259  for (AOBasis::AOShellIterator _is = gwbasis.firstShell(); _is != gwbasis.lastShell(); ++_is) {
260  const AOShell* _shell = gwbasis.getShell(_is);
261 
262 
263  if (_shell->getLmin() == 0) {
264  double decay = (*_shell->firstGaussian())->getDecay();
265 
266 
267  int _start = _shell->getStartIndex();
268 
269  double _factor = pow((2.0 * pi / decay), 0.75);
270 
271  //look at s function only
272 
273  double _test = _temp(_start, _n_level);
274  if (_test > 0.0) {
275  sc_plus += _factor* _test;
276  } else if (_test < 0.0) {
277  sc_minus -= _factor* _test;
278  }
279  }
280  }
281 
282  if (_m_level <= _Mmn_RPA.get_mmax() && _n_level >= _Mmn_RPA.get_nmin()) {
283 
284  double target = std::sqrt(sc_plus * sc_minus);
285  //cout << "s+: " << sc_plus << " s-: " << sc_minus << " target " << target << endl;
286  sc_plus = target / sc_plus;
287  sc_minus = target / sc_minus;
288 
289  // loop over gwbasis shells
290  for (AOBasis::AOShellIterator _is = gwbasis.firstShell(); _is != gwbasis.lastShell(); _is++) {
291  const AOShell* _shell = gwbasis.getShell(_is);
292 
293 
294  if (_shell->getLmin() == 0) {
295  double decay = (*_shell->firstGaussian())->getDecay();
296 
297  int _start = _shell->getStartIndex();
298  double _factor = pow((2.0 * pi / decay), 0.75);
299 
300  // only do something for s- shells
301 
302  // loop over all functions in shell
303 
304  if (std::abs(_factor) > 1.e-10) {
305  double _test = _temp(_start, _n_level);
306  if (_test > 0.0) {
307  _temp(_start, _n_level) = _temp(_start, _n_level) * sc_plus;
308  } else {
309  _temp(_start, _n_level) = _temp(_start, _n_level) * sc_minus;
310  }
311  }
312 
313  }
314  }// end loop over all shells
315  }
316 
317  }// loop n-levels
318 
319  // multiply _temp with overlap
320  ub::matrix<real_gwbse> _temp2 = ub::prod(gwoverlap.Matrix(), _temp);
321 
322  // copy to _Mmn_RPA
323  _Mmn_RPA[ _m_level ] = ub::project(_temp2, full, RPA_cut);
324 
325 
326  }// loop m-levels
327  return;
328  }
