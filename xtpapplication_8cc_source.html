72 bool XtpApplication::EvaluateOptions() {
73 
74  if (OptionsMap().count("list")) {
75  std::cout << "Available XTP" + CalculatorType() + "s: \n";
76  for (const auto& name : CalculatorNames()) {
77  PrintDescription(std::cout, name, "xtp/xml", Application::HelpShort);
78  }
79  StopExecution();
80  return true;
81  }
82 #ifdef USE_CUDA
83  OpenMP_CUDA::SetNoGPUs(OptionsMap()["gpus"].as<Index>());
84 #endif
85 
86  if (OptionsMap().count("description")) {
87  CheckRequired("description", "no " + CalculatorType() + " is given");
88  tools::Tokenizer tok(OptionsMap()["description"].as<std::string>(),
89  " ,\n\t");
90  for (const std::string& n : tok) {
91  if (CalcExists(n)) {
92  PrintDescription(std::cout, n, "xtp/xml", Application::HelpLong);
93  } else {
94  std::cout << CalculatorType() << " " << n << " does not exist\n";
95  }
96  }
97  StopExecution();
98  return true;
99  }
100  CheckRequired("execute", "Nothing to do here: Abort.");
101 
102  tools::Tokenizer calcs(OptionsMap()["execute"].as<std::string>(), " ,\n\t");
103  std::vector<std::string> calc_string = calcs.ToVector();
104  if (calc_string.size() != 1) {
105  throw std::runtime_error("You can only run one " + CalculatorType() +
106  " at the same time.");
107  }
108 
109  if (CalcExists(calc_string[0])) {
110  CreateCalculator(calc_string[0]);
111  } else {
112  std::cout << CalculatorType() << " " << calc_string[0]
113  << " does not exist\n";
114  StopExecution();
115  }
116 
117  EvaluateSpecificOptions();
118 
119  return true;
120 }
