27 void CrossCorrelate::AutoCorrelate(DataCollection<double>::selection& data) {
28  Index N = data[0].size();
29  Eigen::Map<Eigen::VectorXd> input(data[0].data(), N);
30  Eigen::FFT<double> fft;
31  Eigen::VectorXcd frequency = fft.fwd(input);
32  Eigen::VectorXcd magnitude = frequency.cwiseAbs2();
33 
34  _corrfunc.resize(N);
35  Eigen::Map<Eigen::VectorXd> corr_map(_corrfunc.data(), N);
36  corr_map = fft.inv(magnitude);
37  double d = corr_map(0);
38  corr_map.array() /= d;
39 }
