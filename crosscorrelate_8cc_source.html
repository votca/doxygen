31 void CrossCorrelate::AutoCorrelate(DataCollection<double>::selection* data) {
32 #ifdef NOFFTW
33  throw std::runtime_error(
34  "CrossCorrelate::AutoCorrelate is not compiled-in due to disabling of "
35  "FFTW -recompile Votca Tools with FFTW3 support ");
36 #else
37  size_t N = (*data)[0].size();
38  if (N > (size_t)std::numeric_limits<int>::max()) {
39  throw std::runtime_error("CrossCorrelate::AutoCorrelate: size is too big");
40  }
41 
42  _corrfunc.resize(N);
43 
44  fftw_complex* tmp;
45  fftw_plan fft, ifft;
46 
47  tmp = (fftw_complex*)fftw_malloc(sizeof(fftw_complex) * (N / 2 + 1));
48 
49  fft = fftw_plan_dft_r2c_1d((int)N, &(*data)[0][0], tmp, FFTW_ESTIMATE);
50  ifft = fftw_plan_dft_c2r_1d((int)N, tmp, &_corrfunc[0], FFTW_ESTIMATE);
51  fftw_execute(fft);
52 
53  tmp[0][0] = tmp[0][1] = 0;
54  for (size_t i = 1; i < N / 2 + 1; i++) {
55  tmp[i][0] = tmp[i][0] * tmp[i][0] + tmp[i][1] * tmp[i][1];
56  tmp[i][1] = 0;
57  }
58  fftw_execute(ifft);
59 
60  double d = _corrfunc[0];
61  for (size_t i = 0; i < N; i++) {
62  _corrfunc[i] = _corrfunc[i] / d;
63  }
64  fftw_destroy_plan(fft);
65  fftw_destroy_plan(ifft);
66  fftw_free(tmp);
67 #endif
68 }
