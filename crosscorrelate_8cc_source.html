40 void CrossCorrelate::AutoCorrelate(DataCollection<double>::selection& data) {
41  size_t N = data[0].size();
42  if (N > (size_t)std::numeric_limits<int>::max()) {
43  throw std::runtime_error("CrossCorrelate::AutoCorrelate: size is too big");
44  }
45 
46  _corrfunc.resize(N);
47 
48  fftw_complex* tmp;
49  fftw_plan fft, ifft;
50 
51  tmp = (fftw_complex*)fftw_malloc(sizeof(fftw_complex) * (N / 2 + 1));
52 
53  fft = fftw_plan_dft_r2c_1d((int)N, &data[0][0], tmp, FFTW_ESTIMATE);
54  ifft = fftw_plan_dft_c2r_1d((int)N, tmp, &_corrfunc[0], FFTW_ESTIMATE);
55  fftw_execute(fft);
56 
57  tmp[0][0] = tmp[0][1] = 0;
58  for (size_t i = 1; i < N / 2 + 1; i++) {
59  tmp[i][0] = tmp[i][0] * tmp[i][0] + tmp[i][1] * tmp[i][1];
60  tmp[i][1] = 0;
61  }
62  fftw_execute(ifft);
63 
64  double d = _corrfunc[0];
65  for (size_t i = 0; i < N; i++) {
66  _corrfunc[i] = _corrfunc[i] / d;
67  }
68  fftw_destroy_plan(fft);
69  fftw_destroy_plan(ifft);
70  fftw_free(tmp);
71 }
