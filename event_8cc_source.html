110 void Event::Determine_rate(StateReservoir* state, Longrange* longrange, Eventinfo* eventinfo) {
111 
112  Node* node1 = _link->node1();
113  //votca::tools::vec node1vec = node1->position();
114  //double leftnode1pos = node1vec.z();
115  //double rightnode1pos = eventinfo->simboxsize.z() - leftnode1pos;
116  Node* node2 = _link->node2();
117  //votca::tools::vec node2vec = node2->position();
118  //double leftnode2pos = node2vec.z();
119  //double rightnode2pos = eventinfo->simboxsize.z() - leftnode2pos;
120 
121  double prefactor = 1.0; // total prefactor
122  double charge=0;
123  double static_node_energy_from=0;
124  double static_node_energy_to=0;
125 
126  const double hbar = 6.58211928E-16; // eV*s
127  const double Pi = 3.14159265358979323846264338327950288419716939937510;
128  const double kB = 8.617332478E-5; // ev/K
129 
130  if(_carrier_type == (int) Electron) {
131  charge = -1.0;
132  prefactor = prefactor*(eventinfo->electron_transport_prefactor);
133  static_node_energy_from = dynamic_cast<NodeSQL*>(node1)->eAnion() + dynamic_cast<NodeSQL*>(node1)->UcCnNe();
134  static_node_energy_to = dynamic_cast<NodeSQL*>(node2)->eAnion() + dynamic_cast<NodeSQL*>(node2)->UcCnNe();
135  if(_init_type == Injection) static_node_energy_from = eventinfo->avelectronenergy;
136  if(_final_type == Collection) static_node_energy_to = eventinfo->avelectronenergy;
137  }
138  else if(_carrier_type == (int) Hole) {
139  charge = 1.0;
140  double gamma = eventinfo->gamma;
141  prefactor = prefactor*(eventinfo->hole_transport_prefactor);
142  double temp_static_node_energy_from = dynamic_cast<NodeSQL*>(node1)->eCation() + dynamic_cast<NodeSQL*>(node1)->UcCnNh();
143  static_node_energy_from = temp_static_node_energy_from;
144  double temp_static_node_energy_to = dynamic_cast<NodeSQL*>(node2)->eCation() + dynamic_cast<NodeSQL*>(node2)->UcCnNh();
145  static_node_energy_to = temp_static_node_energy_to;
146  if(_init_type == Injection) {
147  static_node_energy_from = eventinfo->avholeenergy;
148  static_node_energy_to = gamma*(temp_static_node_energy_to - eventinfo->avholeenergy) + eventinfo->avholeenergy;
149  }
150  if(_final_type == Collection) {
151  static_node_energy_from = gamma*(temp_static_node_energy_from - eventinfo->avholeenergy) + eventinfo->avholeenergy;
152  static_node_energy_to = eventinfo->avholeenergy;
153  }
154  }
155 
156  //first transfer integrals
157 
158  double _transferfactor = 1.0;
159  votca::tools::vec distancevector = _link->r12();
160  double Reorg=0;
161  double Jeff2=0;
162  double distance;
163 
164  if (eventinfo->formalism == "Miller") {
165  distance = abs(distancevector);
166  _transferfactor = exp(-2.0*eventinfo->alpha*distance);
167  }
168  else if((eventinfo->formalism == "Marcus")&&(_init_type==Injection||_final_type==Collection)) {
169  distance = abs(distancevector);
170  Jeff2 = exp(-2.0*eventinfo->alpha*distance);
171 
172  // take Reorg equal to inside organic material (obviously not true))
173  if(_carrier_type == (int) Electron) {
174  Reorg = eventinfo->electron_injection_reorg;
175  }
176  if(_carrier_type == (int) Hole) {
177  Reorg = eventinfo->hole_injection_reorg;
178  }
179  _transferfactor = (2*Pi/hbar)*(Jeff2/sqrt(4*Pi*Reorg*kB*eventinfo->temperature));
180  }
181  else if(eventinfo->formalism == "Marcus") {
182  if(_carrier_type == (int) Electron) {
183  Jeff2 = dynamic_cast<LinkSQL*>(_link)->Jeff2e();
184  Reorg = dynamic_cast<NodeSQL*>(node1)->UnCnNe() + dynamic_cast<NodeSQL*>(node2)->UcNcCe() + dynamic_cast<LinkSQL*>(_link)->lOe();
185  }
186  if(_carrier_type == (int) Hole) {
187  Jeff2 = dynamic_cast<LinkSQL*>(_link)->Jeff2h();
188  Reorg = dynamic_cast<NodeSQL*>(node1)->UnCnNh() + dynamic_cast<NodeSQL*>(node2)->UcNcCh() + dynamic_cast<LinkSQL*>(_link)->lOh();
189  }
190  _transferfactor = (2*Pi/hbar)*(Jeff2/sqrt(4*Pi*Reorg*kB*eventinfo->temperature));
191  }
192 
193  //second, calculate boltzmann factor
194 
195  double init_energy;
196  double final_energy;
197  double from_event_energy = 0.0;
198  double to_event_energy = 0.0;
199 
200  if(_init_type == Injection) {
201  if(node1->type() == (int) LeftElectrodeNode) {from_event_energy -= eventinfo->left_injection_barrier;}
202  else if(node1->type() == (int) RightElectrodeNode) {from_event_energy -= eventinfo->right_injection_barrier;}
203  prefactor *= eventinfo->injection_prefactor;
204  } // injection
205  else if(_init_type == TransferFrom) { } // transfer
206 
207  if(_final_type == TransferTo) { } // transfer
208  else if(_final_type == Collection) {
209  if(node2->type() == (int) LeftElectrodeNode) {to_event_energy -= eventinfo->left_injection_barrier;}
210  else if(node2->type() == (int) RightElectrodeNode) {to_event_energy -= eventinfo->right_injection_barrier;}
211  prefactor *= eventinfo->collection_prefactor;
212  } // collection
213  else if(_final_type == Recombination) { to_event_energy -= eventinfo->binding_energy; prefactor *= eventinfo->recombination_prefactor;} // recombination
214 
215  double sr_coulomb_from;
216  double sr_coulomb_to;
217  if(!eventinfo->norc) {
218  sr_coulomb_from = Determine_from_sr_coulomb(node1, state, eventinfo);
219  sr_coulomb_to = Determine_to_sr_coulomb(node1, state, eventinfo);
220  }
221  else {
222  sr_coulomb_from = 0.0;
223  sr_coulomb_to = 0.0;
224  }
225  double selfimpot_from;
226  double selfimpot_to;
227  double lr_coulomb_from;
228  double lr_coulomb_to;
229 
230  if(eventinfo->device) {
231  selfimpot_from = Determine_self_coulomb(node1, eventinfo);
232  selfimpot_to = Determine_self_coulomb(node2, eventinfo);
233 
234  lr_coulomb_from = charge*Determine_lr_coulomb(node1, longrange, eventinfo);
235  lr_coulomb_to = charge*Determine_lr_coulomb(node2, longrange, eventinfo);
236 
237  init_energy = static_node_energy_from + selfimpot_from + from_event_energy + sr_coulomb_from + lr_coulomb_from;
238  final_energy = static_node_energy_to + selfimpot_to + to_event_energy + sr_coulomb_to + lr_coulomb_to;
239  }
240  else {
241  init_energy = static_node_energy_from + from_event_energy + sr_coulomb_from;
242  final_energy = static_node_energy_to + to_event_energy + sr_coulomb_to;
243  }
244 
245  double energycontrib = 0.0;
246 
247  if (eventinfo->formalism == "Miller") {
248  if(_final_type == Blocking) {
249  _energyfactor = 0.0; // Keep this here for eventual simulation of bipolaron formation for example
250  }
251  else {
252  energycontrib = final_energy - init_energy - charge*(eventinfo->efield_x*distancevector.x()+eventinfo->efield_y*distancevector.y()+eventinfo->efield_z*distancevector.z());
253  if (energycontrib>0.0) {
254  _energyfactor = exp(-1.0*energycontrib/(kB*eventinfo->temperature));
255  }
256  else {
257  _energyfactor = 1.0;
258  }
259  }
260  }
261  else if (eventinfo->formalism == "Marcus") {
262  if(_final_type == Blocking) {
263  _energyfactor = 0.0; // Keep this here for eventual simulation of bipolaron formation for example
264  }
265  else {
266  energycontrib = final_energy - init_energy - charge*(eventinfo->efield_x*distancevector.x()+eventinfo->efield_y*distancevector.y()+eventinfo->efield_z*distancevector.z());
267  _energyfactor = exp(-1.0*(energycontrib+Reorg)*(energycontrib+Reorg)/(4*Reorg*kB*eventinfo->temperature));
268  }
269  }
270  _rate = prefactor*_transferfactor*_energyfactor;
271 }
