51  // Initializes the object from a [Donor Bridge1 Bridge2 ... Acceptor] string
52  SuperExchangeType(string initString) {
53 
54  Tokenizer tok(initString, " ");
55  std::vector< std::string > names;
56  tok.ToVector(names);
57 
58  if (names.size() < 3) {
59  std::cout << "ERROR: Faulty superexchange definition: "
60  << "Need at least three segment names (DONOR BRIDGES ACCEPTOR separated by a space" << std::endl;
61  throw std::runtime_error("Error in options file.");
62  }
63 
64  // fill up the donor-bride-acceptor structure
65 
66  donor = names.front();
67  acceptor = names.back();
68 
69  for ( std::vector<std::string>::iterator it = ++names.begin() ; it != --names.end(); it++ ) {
70  bridges.push_back(*it);
71  }
72  }

76  bool isOfBridge(string segment_type ) {
77  std::list<string>::iterator findIter = std::find(bridges.begin(), bridges.end(), segment_type);
78  return findIter != bridges.end();
79  };

81  bool isOfDonorAcceptor ( string segment_type ) {
82  return segment_type == donor || segment_type == acceptor ;

85  std::string asString() {
86  std::string ts;
87  ts += donor;
88  for( std::list<std::string>::iterator si = bridges.begin(); si != bridges.end(); si++ ) ts = ts + " " + *si;
89  ts += " " + acceptor;
90  return ts;
91  }
