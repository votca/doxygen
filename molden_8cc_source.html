10void Molden::writeAtoms(const Orbitals& orbitals,

12 for (const auto& atom : orbitals.QMAtoms()) {

20void Molden::writeMOs(const Orbitals& orbitals, std::ofstream& outFile) const {
21 Eigen::VectorXd energies = orbitals.MOs().eigenvalues();
22 bool fromVotcaToExternal = true;
23 OrbReorder reorder(reorderList_, multipliers_, fromVotcaToExternal);
24 Eigen::MatrixXd moCoefficients = orbitals.MOs().eigenvectors();
25 reorder.reorderOrbitals(moCoefficients, orbitals.getDftBasis());
26
27 for (Index i = 0; i < orbitals.getBasisSetSize(); i++) { // over columns
28 outFile << "Sym= \n";
29 outFile << boost::format("Ene= %-20.12e\n") % energies[i];
30 outFile << "Spin= Alpha\n";
31 outFile << boost::format("Occup= %-5.2f\n") %
32 (2 * (i < orbitals.getLumo()));
33 for (Index j = 0; j < orbitals.getBasisSetSize(); j++) {
34 outFile << boost::format("%5d %22.12e\n") % (j + 1) %
35 moCoefficients(j, i);
36 }
37 }

40void Molden::writeBasisSet(const Orbitals& orbitals,
41 std::ofstream& outFile) const {
42 AOBasis basis = orbitals.getDftBasis();
43 for (const auto& atom : orbitals.QMAtoms()) {
44 // The 0 in the format string of the next line is meaningless it
45 // is included for backwards compatibility of molden files
46 outFile << boost::format("%4d 0 \n") % (atom.getId() + 1);
47 const std::vector<const AOShell*> shells =
48 basis.getShellsofAtom(atom.getId());
49 for (const AOShell* shell : shells) {
50 // The 1.0 at the end of the next line is meaningless it
51 // is included for backwards compatibility of molden files
52 outFile << boost::format("%-3s %4d %3.1f \n") %
53 boost::to_lower_copy(EnumToString(shell->getL())) %
54 shell->getSize() % 1.0;
55 for (const AOGaussianPrimitive& gaussian : *shell) {
56 outFile << boost::format("%22.10e %22.10e\n") % gaussian.getDecay() %
57 gaussian.getContraction();
58 }
59 }
60 outFile << " \n";
61 }

97std::string Molden::readAtoms(QMMolecule& mol, const std::string& units,
98 std::ifstream& input_file) const {
99 std::string line;
100 std::istringstream iss(" ");
101 while (std::getline(input_file, line)) {
102 boost::trim(line);
103 if (line == "" || line[0] == '[') {
104 return line;
105 }
106 iss.str(line);
107 iss.clear();
108
109 // extract data
110 double x, y, z;
111 Index atom_id;
112 std::string junk;
113 std::string atom_type;
114 iss >> atom_type >> atom_id >> junk >> x >> y >> z;
115 atom_id =
116 atom_id - 1; // molden uses indexing from 1, we use indexing from 0
117
118 // Add data to orbitals object
119 Eigen::Vector3d pos(x, y, z);
120 if (units == "Angs") {
121 pos = tools::conv::ang2bohr * pos;
122 }
123 mol.push_back(QMAtom(atom_id, atom_type, pos));
124 }
125 return "";
126}

128// The returned string contains the line with the next section header
129// or, if it is the last section, an empty string.
130std::string Molden::readMOs(Orbitals& orbitals,
131 std::ifstream& input_file) const {
132
133 // setup space to store everything
134 Index basis_size = orbitals.getBasisSetSize();
135 if (basis_size == 0) {
136 throw std::runtime_error(
137 "Basis size not set, atoms were not parsed first.");
138 }
139 orbitals.MOs().eigenvalues().resize(basis_size);
140 orbitals.MOs().eigenvectors().resize(basis_size, basis_size);
141
142 // actual parsing
143 std::string line;
144 std::string tempStr;
145 double tempDouble;
146 Index tempIndex;
147 std::istringstream iss(" ");
148 for (Index i = 0; i < basis_size; i++) { // loop over mo's
149 std::getline(input_file, line); // skip symmetry label
150 // energy line
151 std::getline(input_file, line);
152 iss.str(line);
153 iss.clear();
154 iss >> tempStr >> tempDouble;
155 orbitals.MOs().eigenvalues()[i] = tempDouble;
156 // spin channel line
157 std::getline(input_file, line);
158 iss.str(line);
159 iss.clear();
160 iss >> tempStr >> tempStr;
161 // occupation line
162 std::getline(input_file, line);
163 iss.str(line);
164 iss.clear();
165 iss >> tempStr >> tempDouble;
166 if ((int)tempDouble == 1) {
167 XTP_LOG(Log::error, log_)
168 << "WARNING: you are reading a molden file with an openshell "
169 "system.\n Openshell systems are currently not supported in "
170 "VOTCA! \n There are, however, use cases for reading an "
171 "openshell system, hence we keep on running."
172 << std::endl;
173 }
174
175 // MO coefficients
176 for (int j = 0; j < basis_size; j++) { // loop over ao's
177 std::getline(input_file, line);
178 iss.str(line);
179 iss.clear();
180 iss >> tempIndex >> tempDouble;
181 orbitals.MOs().eigenvectors()(j, i) = tempDouble;
182 }
183 }
184
185 OrbReorder reorder(reorderList_, multipliers_);
186 reorder.reorderOrbitals(orbitals.MOs().eigenvectors(),
187 orbitals.getDftBasis());
188
189 getline(input_file, line);
190 return line;
191}

200void Molden::parseMoldenFile(const std::string& filename,
201 Orbitals& orbitals) const {
202
203 if (basisset_name_ == "") {
204 throw std::runtime_error(
205 "Basisset names should be set before reading the molden file.");
206 }
207
208 std::ifstream input_file(filename);
209 // Check if succesfull
210 if (input_file.fail()) {
211 throw std::runtime_error("Could not open molden file with name: " +
212 filename);
213 }
214
215 std::string line;
216 std::getline(input_file, line);
217 while (input_file) {
218 boost::trim(line);
219 if (line[0] != '[') { // ignore non-relevant lines
220 std::getline(input_file, line);
221 continue;
222 }
223
224 // Extract the part between square brackets
225 long unsigned close = line.find("]");
226 std::string sectionType = line.substr(1, close - 1);
227
228 // Import data from relevant sections
229 if (sectionType == "Atoms") {
230 std::string units = line.substr(close + 1);
231 boost::trim(units);
232 XTP_LOG(Log::error, log_)
233 << "Reading atoms using " << units << " units." << std::flush;
234 orbitals.QMAtoms().clearAtoms();
235 if (orbitals.QMAtoms().size() == 0) {
236 line = readAtoms(orbitals.QMAtoms(), units, input_file);
237 } else {
238 throw std::runtime_error(
239 "Could not clear QMAtoms while reading Molden file.");
240 }
241 addBasissetInfo(orbitals);
242 } else if (sectionType == "GTO") {
243 XTP_LOG(Log::error, log_)
244 << "Basisset specification is ignored." << std::flush;
245 XTP_LOG(Log::error, log_)
246 << "Basissets are specified via the mol2orb.xml options file."
247 << std::flush;
248 std::getline(input_file, line);
249 } else if (sectionType == "MO") {
250 if (orbitals.QMAtoms().size() == 0) {
251 throw std::runtime_error(
252 "Atoms should be specified before MO coefficients.");
253 } else {
254 XTP_LOG(Log::error, log_)
255 << "Reading molecular orbital coefficients" << std::flush;
256 line = readMOs(orbitals, input_file);
257 }
258 } else if (sectionType == "STO") {
259 throw std::runtime_error(
260 "Slater Type Orbitals (STOs) are not supported in VOTCA-XTP.");
261 } else {
262 std::getline(input_file, line);
263 }
264 }
265
266 XTP_LOG(Log::error, log_) << "Done parsing molden file" << std::flush;
267}
