59 void SqlApplication::Run() {
60 
61  // load_property_from_xml(_options, _op_vm["options"].as<string>());
62 
63  // EVALUATE OPTIONS
64  int nThreads = OptionsMap()["nthreads"].as<int>();
65  int nframes = OptionsMap()["nframes"].as<int>();
66  int fframe = OptionsMap()["first-frame"].as<int>();
67  if (fframe-- == 0) throw runtime_error("ERROR: First frame is 0, counting "
68  "in VOTCA::CTP starts from 1.");
69  int save = OptionsMap()["save"].as<int>();
70 
71  // STATESAVER & PROGRESS OBSERVER
72  string statefile = OptionsMap()["file"].as<string>();
73  StateSaverSQLite statsav;
74  statsav.Open(_top, statefile);
75 
76  // INITIALIZE & RUN CALCULATORS
77  cout << "Initializing calculators " << endl;
78  BeginEvaluate(nThreads);
79 
80  int frameId = -1;
81  int framesDone = 0;
82  while (statsav.NextFrame() && framesDone < nframes) {
83  frameId += 1;
84  if (frameId < fframe) continue;
85  cout << "Evaluating frame " << _top.getDatabaseId() << endl;
86  EvaluateFrame();
87  if (save == 1) { statsav.WriteFrame(); }
88  else { cout << "Changes have not been written to state file." << endl; }
89  framesDone += 1;
90  }
91 
92  if (framesDone == 0)
93  cout << "Input requires first frame = " << fframe+1 << ", # frames = "
94  << nframes << " => No frames processed.";
95 
96  statsav.Close();
97  EndEvaluate();
98 
99 }
