246  void setBox(const matrix &box, BoundaryCondition::eBoxtype boxtype=BoundaryCondition::typeAuto) {
247  // determine box type automatically in case boxtype==typeAuto
248  if(boxtype==BoundaryCondition::typeAuto) {
249  boxtype = autoDetectBoxType(box);
250  }
251 
252  if(_bc) {
253  delete (_bc);
254  }
255 
256  switch(boxtype) {
