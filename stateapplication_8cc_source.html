56 void StateApplication::Run() {
57 
58  std::string name = ProgramName();
59  if (VersionString() != "") name = name + ", version " + VersionString();
60  xtp::HelpTextHeader(name);
61 
62  load_property_from_xml(_options, _op_vm["options"].as<std::string>());
63  int nThreads = OptionsMap()["nthreads"].as<int>();
64  int nframes = OptionsMap()["nframes"].as<int>();
65  int fframe = OptionsMap()["first-frame"].as<int>();
66  bool save = OptionsMap()["save"].as<bool>();
67 
68  // STATESAVER & PROGRESS OBSERVER
69  std::string statefile = OptionsMap()["file"].as<std::string>();
70  StateSaver statsav(statefile);
71  std::vector<int> frames = statsav.getFrames();
72  if (frames.empty()) {
73  throw std::runtime_error("Statefile " + statefile + " not found.");
74  }
75  // INITIALIZE & RUN CALCULATORS
76  std::cout << "Initializing calculator" << std::endl;
77  BeginEvaluate(nThreads);
78  std::cout << frames.size() << " frames in statefile, Ids are: ";
79  for (int frame : frames) {
80  std::cout << frame << " ";
81  }
82  std::cout << std::endl;
83  if (fframe < int(frames.size())) {
84  std::cout << "Starting at frame " << frames[fframe] << std::endl;
85  } else {
86  std::cout << "First frame:" << fframe
87  << " is larger than number of frames:" << int(frames.size())
88  << std::endl;
89  return;
90  }
91 
92  if ((fframe + nframes) > int(frames.size())) {
93  nframes = frames.size() - fframe;
94  }
95 
96  for (int i = fframe; i < nframes; i++) {
97  std::cout << "Evaluating frame " << frames[i] << std::endl;
98  Topology top = statsav.ReadFrame(frames[i]);
99  EvaluateFrame(top);
100  if (save && _calculator->WriteToStateFile()) {
101  statsav.WriteFrame(top);
102  } else {
103  std::cout << "Changes have not been written to state file." << std::endl;
104  }
105  }
106 }
