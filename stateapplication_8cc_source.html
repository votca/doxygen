55 void StateApplication::execute() {
56 
57  _options.LoadFromXML(_op_vm["options"].as<std::string>());
58  Index nframes = OptionsMap()["nframes"].as<Index>();
59  Index fframe = OptionsMap()["first-frame"].as<Index>();
60  bool save = OptionsMap()["save"].as<bool>();
61 
62  // STATESAVER & PROGRESS OBSERVER
63  std::string statefile = OptionsMap()["file"].as<std::string>();
64  StateSaver statsav(statefile);
65  std::vector<Index> frames = statsav.getFrames();
66  if (frames.empty()) {
67  throw std::runtime_error("Statefile " + statefile + " not found.");
68  }
69  // INITIALIZE & RUN CALCULATORS
70  std::cout << "Initializing calculator" << std::endl;
71  ConfigCalculator();
72  std::cout << frames.size() << " frames in statefile, Ids are: ";
73  for (Index frame : frames) {
74  std::cout << frame << " ";
75  }
76  std::cout << std::endl;
77  if (fframe < Index(frames.size())) {
78  std::cout << "Starting at frame " << frames[fframe] << std::endl;
79  } else {
80  std::cout << "First frame:" << fframe
81  << " is larger than number of frames:" << Index(frames.size())
82  << std::endl;
83  return;
84  }
85 
86  if ((fframe + nframes) > Index(frames.size())) {
87  nframes = Index(frames.size()) - fframe;
88  }
89 
90  for (Index i = fframe; i < nframes; i++) {
91  std::cout << "Evaluating frame " << frames[i] << std::endl;
92  Topology top = statsav.ReadFrame(frames[i]);
93  EvaluateFrame(top);
94  if (save && savetoStateFile()) {
95  statsav.WriteFrame(top);
96  } else {
97  std::cout << "Changes have not been written to state file." << std::endl;
98  }
99  }
100 }
