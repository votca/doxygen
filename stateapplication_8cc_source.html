56 void StateApplication::Run() {
57 
58  std::string name = ProgramName();
59  if (VersionString() != "") {
60  name = name + ", version " + VersionString();
61  }
62  xtp::HelpTextHeader(name);
63 
64  _options.LoadFromXML(_op_vm["options"].as<std::string>());
65  Index nThreads = OptionsMap()["nthreads"].as<Index>();
66  Index nframes = OptionsMap()["nframes"].as<Index>();
67  Index fframe = OptionsMap()["first-frame"].as<Index>();
68  bool save = OptionsMap()["save"].as<bool>();
69 
70  // STATESAVER & PROGRESS OBSERVER
71  std::string statefile = OptionsMap()["file"].as<std::string>();
72  StateSaver statsav(statefile);
73  std::vector<Index> frames = statsav.getFrames();
74  if (frames.empty()) {
75  throw std::runtime_error("Statefile " + statefile + " not found.");
76  }
77  // INITIALIZE & RUN CALCULATORS
78  std::cout << "Initializing calculator" << std::endl;
79  BeginEvaluate(nThreads);
80  std::cout << frames.size() << " frames in statefile, Ids are: ";
81  for (Index frame : frames) {
82  std::cout << frame << " ";
83  }
84  std::cout << std::endl;
85  if (fframe < Index(frames.size())) {
86  std::cout << "Starting at frame " << frames[fframe] << std::endl;
87  } else {
88  std::cout << "First frame:" << fframe
89  << " is larger than number of frames:" << Index(frames.size())
90  << std::endl;
91  return;
92  }
93 
94  if ((fframe + nframes) > Index(frames.size())) {
95  nframes = Index(frames.size()) - fframe;
96  }
97 
98  for (Index i = fframe; i < nframes; i++) {
99  std::cout << "Evaluating frame " << frames[i] << std::endl;
100  Topology top = statsav.ReadFrame(frames[i]);
101  EvaluateFrame(top);
102  if (save && _calculator->WriteToStateFile()) {
103  statsav.WriteFrame(top);
104  } else {
105  std::cout << "Changes have not been written to state file." << std::endl;
106  }
107  }
108 }
