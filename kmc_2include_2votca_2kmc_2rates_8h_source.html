64 void Rates::search(int& charge, int& jump, double searchkey) {
65  long i = sumtree.search(searchkey);
66  if (i<injection_events) { // Injection occurred
67  charge = -1; // Encoding for an injection event
68  jump = i; // Encoding for which injection event occurred
69  }
70  else {
71  i -= injection_events;
72  // Reconstruct charge and jump numbers
73  ldiv_t mydiv = ldiv(i,(long)nbtot);
74  charge = mydiv.quot;
75  jump = mydiv.rem;
76  }
77 }
