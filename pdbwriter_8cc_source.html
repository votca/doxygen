57 void PDBWriter::WriteBox(const Eigen::Matrix3d &box) {
58  boost::format boxfrmt("CRYST1%1$9.3f%2$9.3f%3$9.3f%4$7.2f%5$7.2f%6$7.2f\n");
59  double a = box.col(0).norm();
60  double b = box.col(1).norm();
61  double c = box.col(2).norm();
62  double alpha =
63  180 / tools::conv::Pi * std::acos(box.col(1).dot(box.col(2)) / (b * c));
64  double beta =
65  180 / tools::conv::Pi * std::acos(box.col(0).dot(box.col(2)) / (a * c));
66  double gamma =
67  180 / tools::conv::Pi * std::acos(box.col(0).dot(box.col(1)) / (a * b));
68  _out << boxfrmt % a % b % c % alpha % beta % gamma;
69 }

71 void PDBWriter::writeSymmetry(Bead *bead) {
72  if (bead->getSymmetry() > 1) {
73  Eigen::Vector3d r = 10 * bead->getPos();
74  boost::format beadfrmt(
75  "HETATM%1$5d %2$4s %3$3s %4$1s%5$4d %6$8.3f%7$8.3f%8$8.3f\n");
76  Eigen::Vector3d ru = 0.1 * bead->getU() + r;
77 
78  _out << beadfrmt % (bead->getId() + 1) % 100000 // atom serial number
79  % bead->getName() // atom name
80  % "REU" // residue name
81  % " " // chain identifier 1 char
82  % (bead->getResnr() + 1) // residue sequence number
83  % ru.x() % ru.y() % ru.z(); // we skip the charge
84 
85  if (bead->getSymmetry() > 2) {
86  Eigen::Vector3d rv = 0.1 * bead->getV() + r;
87  _out << beadfrmt % (bead->getId() + 1) % 100000 // atom serial number
88  % bead->getName() // atom name
89  % "REV" // residue name
90  % " " // chain identifier 1 char
91  % (bead->getResnr() + 1) // residue sequence number
92  % rv.x() % rv.y() % rv.z(); // we skip the charge
93  }
94  }
95  return;
