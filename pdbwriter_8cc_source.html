37 void PDBWriter::Write(Topology *conf)
38 {
39  Topology *top = conf;
40  fprintf(_out, "MODEL %4d\n", conf->getStep());
41  for(BeadContainer::iterator iter=conf->Beads().begin();
42  iter!=conf->Beads().end(); ++iter) {
43  Bead *bi = *iter;
44  vec r = bi->getPos();
45  //truncate strings if necessary
46  string resname="";
47  if(top->getResidue(bi->getResnr()))
48  resname = top->getResidue(bi->getResnr())->getName();
49  string atomname = bi->getName();
50  if (resname.size() > 3) {
51  resname = resname.substr(0,3);
52  }
53  if (atomname.size() > 4) {
54  atomname = atomname.substr(0,4);
55  }
56 
57  fprintf(_out,
58  "ATOM %5d %4s %3s %1s%4d %8.3f%8.3f%8.3f\n",
59  (bi->getId()+1)%100000, // atom serial number
60  atomname.c_str(), // atom name
61  resname.c_str(), // residue name
62  " ", // chain identifier 1 char
63  bi->getResnr()+1, // residue sequence number
64  10*r.x(), 10*r.y(), 10*r.z()); //nm -> Angs
65  //we skip the charge
66 
67  if(bi->getSymmetry()>=2) {
68  vec ru = 0.1*bi->getU() + r;
69 
70  fprintf(_out,
71  "HETATM%5d %4s %3s %1s%4d %8.3f%8.3f%8.4f\n",
72  bi->getId()+1, // atom serial number
73  bi->getName().c_str(), // atom name
74  "REU", // residue name
75  " ", // chain identifier 1 char
76  bi->getResnr()+1, // residue sequence number
77  ru.x(), ru.y(), ru.z()); //we skip the charge
78  }
79  if(bi->getSymmetry()>=3) {
80  vec rv = 0.1*bi->getV() + r;
81  fprintf(_out,
82  "HETATM%5d %4s %3s %1s%4d %8.3f%8.3f%8.4f\n",
83  bi->getId()+1, // atom serial number
84  bi->getName().c_str(), // atom name
85  "REV", // residue name
86  " ", // chain identifier 1 char
87  bi->getResnr()+1, // residue sequence number
88  rv.x(), rv.y(), rv.z()); //we skip the charge
89  }
90  }
91  fprintf(_out, "ENDMDL\n");

93 }
