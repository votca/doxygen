51 void Forces::Calculate(const Orbitals& orbitals) {
52 
53  int natoms = orbitals.QMAtoms().size();
54  _forces = Eigen::MatrixX3d::Zero(natoms, 3);
55 
56  ctp::TLogLevel ReportLevel = _pLog->getReportLevel(); // backup report level
57  if (!tools::globals::verbose) {
58  _pLog->setReportLevel(ctp::logERROR); // go silent for force calculations
59  }
60  for (int atom_index = 0; atom_index < natoms; atom_index++) {
61  if (tools::globals::verbose) {
62  CTP_LOG(ctp::logINFO, *_pLog)
63  << "FORCES--DEBUG working on atom " << atom_index << flush;
64  }
65  Eigen::Vector3d atom_force = Eigen::Vector3d::Zero();
66  // Calculate Force on this atom
67  if (_force_method == "forward")
68  atom_force = NumForceForward(orbitals, atom_index);
69  if (_force_method == "central")
70  atom_force = NumForceCentral(orbitals, atom_index);
71  _forces.row(atom_index) = atom_force.transpose();
72  }
73  _pLog->setReportLevel(ReportLevel); //
74  if (_remove_total_force) RemoveTotalForce();
75  return;
76 }

103 Eigen::Vector3d Forces::NumForceForward(Orbitals orbitals, int atom_index) {
104  Eigen::Vector3d force = Eigen::Vector3d::Zero();
105  // get this atoms's current coordinates
106  double energy_center =
107  orbitals.getTotalStateEnergy(_filter.CalcState(orbitals));
108  const tools::vec current_pos = orbitals.QMAtoms()[atom_index]->getPos();
109 
110  for (unsigned i_cart = 0; i_cart < 3; i_cart++) {
111  tools::vec displacement_vec(0, 0, 0);
112  displacement_vec[i_cart] = _displacement;
113  // update the coordinate
114  tools::vec pos_displaced = current_pos + displacement_vec;
115  orbitals.QMAtoms()[atom_index]->setPos(pos_displaced);
116  _gwbse_engine.ExcitationEnergies(orbitals);
117  double energy_displaced =
118  orbitals.getTotalStateEnergy(_filter.CalcState(orbitals));
119  force(i_cart) = (energy_center - energy_displaced) / _displacement;
120  orbitals.QMAtoms()[atom_index]->setPos(
121  current_pos); // restore original coordinate into segment
122  } // Cartesian directions
123  return force;
124 }

126 Eigen::Vector3d Forces::NumForceCentral(Orbitals orbitals, int atom_index) {
127 
128  const tools::vec current_pos = orbitals.QMAtoms()[atom_index]->getPos();
129  Eigen::Vector3d force = Eigen::Vector3d::Zero();
130  for (unsigned i_cart = 0; i_cart < 3; i_cart++) {
131  if (tools::globals::verbose) {
132  CTP_LOG(ctp::logINFO, *_pLog)
133  << "FORCES--DEBUG Cartesian component " << i_cart << flush;
134  }
135  tools::vec displacement_vec(0, 0, 0);
136  displacement_vec[i_cart] = _displacement;
137  // update the coordinate
138  tools::vec pos_displaced = current_pos + displacement_vec;
139  orbitals.QMAtoms()[atom_index]->setPos(pos_displaced);
140  _gwbse_engine.ExcitationEnergies(orbitals);
141  double energy_displaced_plus =
142  orbitals.getTotalStateEnergy(_filter.CalcState(orbitals));
143  // update the coordinate
144  pos_displaced = current_pos - displacement_vec;
145  orbitals.QMAtoms()[atom_index]->setPos(pos_displaced);
146  _gwbse_engine.ExcitationEnergies(orbitals);
147  double energy_displaced_minus =
148  orbitals.getTotalStateEnergy(_filter.CalcState(orbitals));
149  force(i_cart) =
150  0.5 * (energy_displaced_minus - energy_displaced_plus) / _displacement;
151  orbitals.QMAtoms()[atom_index]->setPos(
152  current_pos); // restore original coordinate into orbital
153  }
154  return force;
155 }

157 void Forces::RemoveTotalForce() {
158  Eigen::Vector3d avgtotal_force =
159  _forces.colwise().sum() / double(_forces.rows());
160  for (unsigned i_atom = 0; i_atom < _forces.rows(); i_atom++) {
161  _forces.row(i_atom) -= avgtotal_force;
162  }
163  return;
164 }
