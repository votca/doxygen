51 void Forces::Calculate(const Orbitals& orbitals) {
52 
53  Index natoms = orbitals.QMAtoms().size();
54  _forces = Eigen::MatrixX3d::Zero(natoms, 3);
55 
56  for (Index atom_index = 0; atom_index < natoms; atom_index++) {
57 
58  XTP_LOG(Log::debug, *_pLog)
59  << "FORCES--DEBUG working on atom " << atom_index << flush;
60  Eigen::Vector3d atom_force = Eigen::Vector3d::Zero();
61  // Calculate Force on this atom
62  if (_force_method == "forward") {
63  atom_force = NumForceForward(orbitals, atom_index);
64  }
65  if (_force_method == "central") {
66  atom_force = NumForceCentral(orbitals, atom_index);
67  }
68  _forces.row(atom_index) = atom_force.transpose();
69  }
70  if (_remove_total_force) {
71  RemoveTotalForce();
72  }
73  return;
74 }

101 Eigen::Vector3d Forces::NumForceForward(Orbitals orbitals, Index atom_index) {
102  Eigen::Vector3d force = Eigen::Vector3d::Zero();
103  // get this atoms's current coordinates
104  double energy_center =
105  orbitals.getTotalStateEnergy(_tracker.CalcState(orbitals));
106  const Eigen::Vector3d current_pos = orbitals.QMAtoms()[atom_index].getPos();
107  for (Index i_cart = 0; i_cart < 3; i_cart++) {
108  Eigen::Vector3d displacement_vec = Eigen::Vector3d::Zero();
109  displacement_vec[i_cart] = _displacement;
110  // update the coordinate
111  Eigen::Vector3d pos_displaced = current_pos + displacement_vec;
112  orbitals.QMAtoms()[atom_index].setPos(pos_displaced);
113  _gwbse_engine.ExcitationEnergies(orbitals);
114  double energy_displaced =
115  orbitals.getTotalStateEnergy(_tracker.CalcState(orbitals));
116  force(i_cart) = (energy_center - energy_displaced) / _displacement;
117  orbitals.QMAtoms()[atom_index].setPos(
118  current_pos); // restore original coordinate into segment
119  } // Cartesian directions
120  return force;
121 }

123 Eigen::Vector3d Forces::NumForceCentral(Orbitals orbitals, Index atom_index) {
124  Eigen::Vector3d force = Eigen::Vector3d::Zero();
125  const Eigen::Vector3d current_pos = orbitals.QMAtoms()[atom_index].getPos();
126  for (Index i_cart = 0; i_cart < 3; i_cart++) {
127  XTP_LOG(Log::debug, *_pLog)
128  << "FORCES--DEBUG Cartesian component " << i_cart << flush;
129  Eigen::Vector3d displacement_vec = Eigen::Vector3d::Zero();
130  displacement_vec[i_cart] = _displacement;
131  // update the coordinate
132  Eigen::Vector3d pos_displaced = current_pos + displacement_vec;
133  orbitals.QMAtoms()[atom_index].setPos(pos_displaced);
134  _gwbse_engine.ExcitationEnergies(orbitals);
135  double energy_displaced_plus =
136  orbitals.getTotalStateEnergy(_tracker.CalcState(orbitals));
137  // update the coordinate
138  pos_displaced = current_pos - displacement_vec;
139  orbitals.QMAtoms()[atom_index].setPos(pos_displaced);
140  _gwbse_engine.ExcitationEnergies(orbitals);
141  double energy_displaced_minus =
142  orbitals.getTotalStateEnergy(_tracker.CalcState(orbitals));
143  force(i_cart) =
144  0.5 * (energy_displaced_minus - energy_displaced_plus) / _displacement;
145  orbitals.QMAtoms()[atom_index].setPos(
146  current_pos); // restore original coordinate into orbital
147  }
148  return force;
149 }

151 void Forces::RemoveTotalForce() {
152  Eigen::Vector3d avgtotal_force =
153  _forces.colwise().sum() / double(_forces.rows());
154  for (Index i_atom = 0; i_atom < _forces.rows(); i_atom++) {
155  _forces.row(i_atom) -= avgtotal_force;
156  }
157  return;
158 }
