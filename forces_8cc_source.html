27  void Forces::Initialize(Property *options) {
28 
29  // checking if there is only one segment
30  _nsegments = _segments.size();
31  if (_nsegments > 1) throw runtime_error(string("\n Force calculation for more than 1 conjugated segment not supported. Stopping!"));
32 
33  // pre-check forces method
34  std::vector<string> choices = {"forward", "central"};
35  _force_method = options->ifExistsAndinListReturnElseThrowRuntimeError<string>(".method", choices);
36 
37  // output level
38  _noisy_output = options->ifExistsReturnElseReturnDefault<bool>(".noisy", false);
39 
40 
41  // precaution in case we implement approx. analytic forces in the future
42  if ((_force_method == "forward") || (_force_method == "central")) {
43  _displacement = options->ifExistsReturnElseReturnDefault<double>(".displacement", 0.001); // Angstrom
44  }
45 
46  // check for force removal options
47  choices = {"total", "CoM", "none"};
48  string _force_removal = options->ifExistsAndinListReturnElseThrowRuntimeError<string>(".removal", choices);
49  if (_force_removal == "total") _remove_total_force = true;
50  if (_force_removal == "CoM") _remove_CoM_force = true;
51 
52  _natoms = _segments[0]->Atoms().size();
53  _forces = ub::zero_matrix<double>(_natoms, 3);
54 
55 
56  return;
57

60  void Forces::Calculate(const double& energy) {
61 
62  ctp::TLogLevel _ReportLevel = _pLog->getReportLevel(); // backup report level
63  if ( ! _noisy_output ){
64  _pLog->setReportLevel(ctp::logERROR); // go silent for force calculations
65  }
66 
67  //backup current coordinates (WHY?)
68  std::vector <ctp::Segment* > _molecule;
69  ctp::Segment _current_coordinates(0, "mol");
70  _qminterface.Orbitals2Segment(&_current_coordinates, _orbitals);
71  _molecule.push_back(&_current_coordinates);
72 
73  // displace all atoms in each Cartesian coordinate and get new energy
74  std::vector< ctp::Atom* > _atoms;
75  std::vector< ctp::Atom* > ::iterator ait;
76  _atoms = _current_coordinates.Atoms();
77 
78  int _i_atom = 0;
79  for (ait = _atoms.begin(); ait < _atoms.end(); ++ait) {
80 
81  if ( _noisy_output ){
82  CTP_LOG(ctp::logINFO, *_pLog) << "FORCES--DEBUG working on atom " << _i_atom << flush;
83  }
84 
85  // Calculate Force on this atom
86  ub::matrix_range< ub::matrix<double> > _force = ub::subrange(_forces, _i_atom, _i_atom + 1, 0, 3);
87  if (_force_method == "forward") NumForceForward(energy, ait, _force, _molecule);
88  if (_force_method == "central") NumForceCentral(energy, ait, _force, _molecule);
89  _i_atom++;
90  }
91 
92  _pLog->setReportLevel(_ReportLevel); //
93 
94  // Remove Total Force, if requested
95  if (_remove_total_force) RemoveTotalForce();
96  //Report();
97 
98  return;

117  /* Calculate forces on an atom numerically by forward differences */
118  void Forces::NumForceForward(double energy, std::vector< ctp::Atom* > ::iterator ait, ub::matrix_range< ub::matrix<double> >& _force,
119  std::vector<ctp::Segment*> _molecule) {
120 
121  // get this atoms's current coordinates
122  vec _current_pos = (*ait)->getQMPos(); // in nm
123 
124  for (unsigned _i_cart = 0; _i_cart < 3; _i_cart++) {
125 
126  // get displacement std::vector
127  vec _displaced(0, 0, 0);
128  if (_i_cart == 0) {
129  _displaced.setX(_displacement * tools::conv::ang2nm); // x, _displacement in Angstrom, now in nm
130  }
131  if (_i_cart == 1) {
132  _displaced.setY(_displacement * tools::conv::ang2nm); // y, _displacement in in Angstrom, now in nm
133  }
134  if (_i_cart == 2) {
135  _displaced.setZ(_displacement * tools::conv::ang2nm); // z, _displacement in in Angstrom, now in nm
136  }
137 
138  // update the coordinate
139  vec _pos_displaced = _current_pos + _displaced;
140 
141  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
142 
143  // run DFT and GW-BSE for this geometry
144  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
145 
146  // get total energy for this excited state
147  double energy_displaced = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
148 
149  // calculate force and put into matrix
150  _force(0, _i_cart) = (energy - energy_displaced) / (_displacement * votca::tools::conv::ang2bohr); // force a.u./a.u.
151  (*ait)->setQMPos(_current_pos); // restore original coordinate into segment
152  } // Cartesian directions
153  return;
154  }

156  /* Calculate forces on atoms numerically by central differences */
157  void Forces::NumForceCentral(double energy, std::vector< ctp::Atom* > ::iterator ait, ub::matrix_range< ub::matrix<double> >& _force,
158  std::vector<ctp::Segment*> _molecule) {
159 
160 
161  vec _current_pos = (*ait)->getQMPos(); // in nm
162 
163  // go through all cartesian components
164  for (unsigned _i_cart = 0; _i_cart < 3; _i_cart++) {
165 
166  if ( _noisy_output ){
167  CTP_LOG(ctp::logINFO, *_pLog) << "FORCES--DEBUG Cartesian component " << _i_cart << flush;
168  }
169 
170  // get displacement vector in positive direction
171  vec _displaced(0, 0, 0);
172  if (_i_cart == 0) {
173  _displaced.setX(_displacement * tools::conv::ang2nm); // x, _displacement in Bohr
174  }
175  if (_i_cart == 1) {
176  _displaced.setY(_displacement * tools::conv::ang2nm); // y, _displacement in in Angstrom
177  }
178  if (_i_cart == 2) {
179  _displaced.setZ(_displacement * tools::conv::ang2nm); // z, _displacement in in Angstrom
180  }
181 
182  // update the coordinate
183  vec _pos_displaced = _current_pos + _displaced;
184  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
185 
186  // run DFT and GW-BSE for this geometry
187  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
188 
189  // get total energy for this excited state
190  double energy_displaced_plus = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
191 
192  // get displacement vector in negative direction
193 
194  // update the coordinate
195  _pos_displaced = _current_pos - 2.0 * _displaced;
196  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
197 
198  // run DFT and GW-BSE for this geometry
199  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
200 
201  // get total energy for this excited state
202  double energy_displaced_minus = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
203 
204  // calculate force and put into matrix
205  _force(0, _i_cart) = 0.5 * (energy_displaced_minus - energy_displaced_plus) / (_displacement * votca::tools::conv::ang2bohr); // force a.u./a.u.
206 
207  (*ait)->setQMPos(_current_pos); // restore original coordinate into segment
208  }
209 
210  return;
211  }

213  /* Adjust forces so that sum of forces is zero */
214  void Forces::RemoveTotalForce() {
215 
216  // total force on all atoms
217  ub::vector<double> _total_force = TotalForce();
218 
219  // zero total force
220  for (unsigned _i_atom = 0; _i_atom < _natoms; _i_atom++) {
221  for (unsigned _i_cart = 0; _i_cart < 3; _i_cart++) {
222  _forces(_i_atom, _i_cart) -= _total_force(_i_cart) / _natoms;
223  }
224  }
225  return;
226  }
