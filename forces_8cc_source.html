44 void Forces::Calculate(const Orbitals& orbitals) {
45 
46  Index natoms = orbitals.QMAtoms().size();
47  forces_ = Eigen::MatrixX3d::Zero(natoms, 3);
48 
49  for (Index atom_index = 0; atom_index < natoms; atom_index++) {
50 
51  XTP_LOG(Log::debug, *pLog_)
52  << "FORCES--DEBUG working on atom " << atom_index << flush;
53  Eigen::Vector3d atom_force = Eigen::Vector3d::Zero();
54  // Calculate Force on this atom
55  if (force_method_ == "forward") {
56  atom_force = NumForceForward(orbitals, atom_index);
57  }
58  if (force_method_ == "central") {
59  atom_force = NumForceCentral(orbitals, atom_index);
60  }
61  forces_.row(atom_index) = atom_force.transpose();
62  }
63  if (remove_total_force_) {
64  RemoveTotalForce();
65  }
66  return;
67 }

94 Eigen::Vector3d Forces::NumForceForward(Orbitals orbitals, Index atom_index) {
95  Eigen::Vector3d force = Eigen::Vector3d::Zero();
96  // get this atoms's current coordinates
97  double energy_center =
98  orbitals.getTotalStateEnergy(tracker_.CalcState(orbitals));
99  const Eigen::Vector3d current_pos = orbitals.QMAtoms()[atom_index].getPos();
100  for (Index i_cart = 0; i_cart < 3; i_cart++) {
101  Eigen::Vector3d displacement_vec = Eigen::Vector3d::Zero();
102  displacement_vec[i_cart] = displacement_;
103  // update the coordinate
104  Eigen::Vector3d pos_displaced = current_pos + displacement_vec;
105  orbitals.updateAtomPostion(atom_index, pos_displaced);
106  gwbse_engine_.ExcitationEnergies(orbitals);
107  double energy_displaced =
108  orbitals.getTotalStateEnergy(tracker_.CalcState(orbitals));
109  force(i_cart) = (energy_center - energy_displaced) / displacement_;
110  orbitals.updateAtomPostion(
111  atom_index, current_pos); // restore original coordinate into segment
112  } // Cartesian directions
113  return force;
114 }

116 Eigen::Vector3d Forces::NumForceCentral(Orbitals orbitals, Index atom_index) {
117  Eigen::Vector3d force = Eigen::Vector3d::Zero();
118  const Eigen::Vector3d current_pos = orbitals.QMAtoms()[atom_index].getPos();
119  for (Index i_cart = 0; i_cart < 3; i_cart++) {
120  XTP_LOG(Log::debug, *pLog_)
121  << "FORCES--DEBUG Cartesian component " << i_cart << flush;
122  Eigen::Vector3d displacement_vec = Eigen::Vector3d::Zero();
123  displacement_vec[i_cart] = displacement_;
124  // update the coordinate
125  Eigen::Vector3d pos_displaced = current_pos + displacement_vec;
126  orbitals.updateAtomPostion(atom_index, pos_displaced);
127  gwbse_engine_.ExcitationEnergies(orbitals);
128  double energy_displaced_plus =
129  orbitals.getTotalStateEnergy(tracker_.CalcState(orbitals));
130  // update the coordinate
131  pos_displaced = current_pos - displacement_vec;
132  orbitals.updateAtomPostion(atom_index, pos_displaced);
133  gwbse_engine_.ExcitationEnergies(orbitals);
134  double energy_displaced_minus =
135  orbitals.getTotalStateEnergy(tracker_.CalcState(orbitals));
136  force(i_cart) =
137  0.5 * (energy_displaced_minus - energy_displaced_plus) / displacement_;
138  orbitals.updateAtomPostion(
139  atom_index, current_pos); // restore original coordinate into orbital
140  }
141  return force;
142 }

144 void Forces::RemoveTotalForce() {
145  Eigen::Vector3d avgtotal_force =
146  forces_.colwise().sum() / double(forces_.rows());
147  for (Index i_atom = 0; i_atom < forces_.rows(); i_atom++) {
148  forces_.row(i_atom) -= avgtotal_force;
149  }
150  return;
151 }
