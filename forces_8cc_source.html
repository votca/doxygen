118  /* Calculate forces on an atom numerically by forward differences */
119  Eigen::Vector3d Forces::NumForceForward(double energy, std::vector< ctp::Atom* > ::iterator ait, std::vector<ctp::Segment*> _molecule) {
120  Eigen::Vector3d force=Eigen::Vector3d::Zero();
121  // get this atoms's current coordinates
122  tools::vec _current_pos = (*ait)->getQMPos(); // in nm
123 
124  for (unsigned _i_cart = 0; _i_cart < 3; _i_cart++) {
125 
126  // get displacement std::vector
127  tools::vec _displaced(0, 0, 0);
128  if (_i_cart == 0) {
129  _displaced.setX(_displacement * tools::conv::ang2nm); // x, _displacement in Angstrom, now in nm
130  }
131  if (_i_cart == 1) {
132  _displaced.setY(_displacement * tools::conv::ang2nm); // y, _displacement in in Angstrom, now in nm
133  }
134  if (_i_cart == 2) {
135  _displaced.setZ(_displacement * tools::conv::ang2nm); // z, _displacement in in Angstrom, now in nm
136  }
137 
138  // update the coordinate
139  tools::vec _pos_displaced = _current_pos + _displaced;
140 
141  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
142 
143  // run DFT and GW-BSE for this geometry
144  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
145 
146  // get total energy for this excited state
147  double energy_displaced = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
148 
149  // calculate force and put into matrix
150  force(_i_cart) = (energy - energy_displaced) / (_displacement * votca::tools::conv::ang2bohr); // force a.u./a.u.
151  (*ait)->setQMPos(_current_pos); // restore original coordinate into segment
152  } // Cartesian directions
153  return force;
154  }

156  /* Calculate forces on atoms numerically by central differences */
157  Eigen::Vector3d Forces::NumForceCentral(double energy, std::vector< ctp::Atom* > ::iterator ait, std::vector<ctp::Segment*> _molecule) {
158 
159 
160  tools::vec _current_pos = (*ait)->getQMPos(); // in nm
161  Eigen::Vector3d force=Eigen::Vector3d::Zero();
162  // go through all cartesian components
163  for (unsigned _i_cart = 0; _i_cart < 3; _i_cart++) {
164 
165  if ( _noisy_output ){
166  CTP_LOG(ctp::logINFO, *_pLog) << "FORCES--DEBUG Cartesian component " << _i_cart << flush;
167  }
168 
169  // get displacement vector in positive direction
170  tools::vec _displaced(0, 0, 0);
171  if (_i_cart == 0) {
172  _displaced.setX(_displacement * tools::conv::ang2nm); // x, _displacement in Bohr
173  }
174  if (_i_cart == 1) {
175  _displaced.setY(_displacement * tools::conv::ang2nm); // y, _displacement in in Angstrom
176  }
177  if (_i_cart == 2) {
178  _displaced.setZ(_displacement * tools::conv::ang2nm); // z, _displacement in in Angstrom
179  }
180 
181  // update the coordinate
182  tools::vec _pos_displaced = _current_pos + _displaced;
183  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
184 
185  // run DFT and GW-BSE for this geometry
186  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
187 
188  // get total energy for this excited state
189  double energy_displaced_plus = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
190 
191  // get displacement vector in negative direction
192 
193  // update the coordinate
194  _pos_displaced = _current_pos - 2.0 * _displaced;
195  (*ait)->setQMPos(_pos_displaced); // put updated coordinate into segment
196 
197  // run DFT and GW-BSE for this geometry
198  _gwbse_engine.ExcitationEnergies(_qmpackage, _molecule, _orbitals);
199 
200  // get total energy for this excited state
201  double energy_displaced_minus = _orbitals->GetTotalEnergy(_spin_type, _opt_state);
202 
203  // calculate force and put into matrix
204  force(_i_cart) = 0.5 * (energy_displaced_minus - energy_displaced_plus) / (_displacement * votca::tools::conv::ang2bohr); // force a.u./a.u.
205 
206  (*ait)->setQMPos(_current_pos); // restore original coordinate into segment
207  }
208 
209  return force;
210  }

213  void Forces::RemoveTotalForce() {
214  // total force on all atoms
215  Eigen::Vector3d _total_force = TotalForce();
216  // zero total force
217  for (unsigned _i_atom = 0; _i_atom < _natoms; _i_atom++) {

219  }
220  return;
