118 bool Calculator::IsValidOption(const Property &prop,
119  const std::vector<std::string> &choices) {
120  const std::string &head = choices.front();
121  std::ostringstream oss;
122  bool is_valid = true;
123  if (head == "bool") {
124  is_valid = Calculator::IsValidCast<bool>(prop);
125  } else if (head == "float") {
126  is_valid = Calculator::IsValidCast<double>(prop);
127  } else if (head == "float+") {
128  is_valid =
129  Calculator::IsValidCast<double>(prop) && (prop.as<double>() >= 0.0);
130  } else if (head == "int") {
131  is_valid = Calculator::IsValidCast<Index>(prop);
132  } else if (head == "int+") {
133  is_valid = Calculator::IsValidCast<Index>(prop) && (prop.as<Index>() >= 0);
134  } else {
135  std::string value = prop.as<std::string>();
136  std::string att = prop.getAttribute<std::string>("choices");
137  std::size_t start_bracket = att.find('[');
138  if (start_bracket == std::string::npos) {
139  // There is a single choice out of multiple default valid choices
140  auto it = std::find(std::cbegin(choices), std::cend(choices), value);
141  is_valid = (it != std::cend(choices));
142  } else {
143  // there are multiple valid choices
144  Tokenizer tok{value, " ,"};
145  for (const std::string &word : tok) {
146  auto it = std::find(std::cbegin(choices), std::cend(choices), word);
147  if (it == std::cend(choices)) {
148  is_valid = false;
149  break;
150  }
151  }
152  }
153  }
154  return is_valid;
155 }
