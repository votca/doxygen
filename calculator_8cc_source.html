23 void Calculator::UpdateWithDefaults(Property &options, std::string package) {
24 
25  // copy options from the object supplied by the Application
26  std::string id = Identify();
27  Property options_id = options.get("options." + id);
28 
29  // add default values if specified in VOTCASHARE
30  char *votca_share = getenv("VOTCASHARE");
31  if (votca_share == nullptr) {
32  throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
33  }
34  // load the xml description of the calculator (with defaults and test values)
35  std::string xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/") +
36  package + std::string("/xml/") + id +
37  std::string(".xml");
38 
39  Property defaults_all;
40  defaults_all.LoadFromXML(xmlFile);
41  Property defaults = defaults_all.get("options." + id);
42 
43  // if a value not given or a tag not present, provide default values
44  AddDefaults(options_id, defaults);
45 
46  // output calculator options
47  std::string indent(" ");
48  Index level = 1;
49  votca::tools::PropertyIOManipulator IndentedText(PropertyIOManipulator::TXT,
50  level, indent);
51  if (Log::verbose()) {
52  std::cout << "\n... ... options\n"
53  << IndentedText << options_id << "... ... options\n"
54  << std::flush;
55  }
56 }

58 void Calculator::AddDefaults(Property &p, const Property &defaults) {
59 
60  for (const Property &prop : defaults) {
61  std::string name = prop.path() + "." + prop.name();
62 
63  Property rootp = *p.begin();
64  if (prop.hasAttribute("default")) {
65  if (rootp.exists(name)) {
66  if (rootp.HasChildren()) {
67  rootp.value() = prop.value();
68  }
69  } else {
70  rootp.add(prop.name(), prop.value());
71  }
72  }
73  AddDefaults(p, prop);
74  }
75 }
