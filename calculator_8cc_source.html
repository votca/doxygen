59 void Calculator::OverwriteDefaultsWithUserInput(const Property &p,
60  Property &defaults) {
61 
62  // Go through everything that is defined in user option
63  for (const Property &prop : p) {
64  if (prop.HasChildren()) {
65  if (defaults.exists(prop.name())) {
66  OverwriteDefaultsWithUserInput(prop, defaults.get(prop.name()));
67  } else {
68  Property &new_prop = defaults.add(prop.name(), "");
69  new_prop = prop;
70  }
71  } else if (defaults.exists(prop.name())) {
72  defaults.set(prop.name(), prop.value());
73  } else {
74  defaults.add(prop.name(), prop.value());
75  }
76  }

89 void Calculator::RecursivelyCheckOptions(const Property &p) {
90  for (const Property &prop : p) {
91  if (prop.HasChildren()) {
92  RecursivelyCheckOptions(prop);
93  } else {
94  std::vector<std::string> choices = Calculator::GetPropertyChoices(prop);
95  const std::string &head = choices.front();
96  if (head != "") {
97  if (!Calculator::IsValidOption(prop, choices)) {
98  std::ostringstream oss;
99  oss << "\nThe input value for \"" << prop.name() << "\"";
100  if (choices.size() == 1) {
101  oss << " should be a \"" << head << "\"";
102  } else {
103  oss << " should be one of the following values: ";
104  for (const std::string &c : choices) {
105  oss << "\"" << c << "\""

119 bool Calculator::IsValidOption(const Property &prop,
120  const std::vector<std::string> &choices) {
121  const std::string &head = choices.front();
122  std::ostringstream oss;
123  bool is_valid;
124  if (head == "bool") {
125  is_valid = Calculator::IsValidCast<bool>(prop);
126  } else if (head == "float") {
127  is_valid = Calculator::IsValidCast<double>(prop);
128  } else if (head == "float+") {
129  is_valid =
130  Calculator::IsValidCast<double>(prop) && (prop.as<double>() >= 0.0);
131  } else if (head == "int") {
132  is_valid = Calculator::IsValidCast<Index>(prop);
133  } else if (head == "int+") {
134  is_valid = Calculator::IsValidCast<Index>(prop) && (prop.as<Index>() >= 0);
135  } else {
136  std::string value = prop.as<std::string>();
137  auto it = std::find(std::cbegin(choices), std::cend(choices), value);
138  is_valid = (it != std::cend(choices));
139  }
140  return is_valid;
141 }
