33 std::vector<Index> StateSaver::getFrames() const {
34  if (!tools::filesystem::FileExists(_hdf5file)) {
35  return std::vector<Index>();
36  }
37 
38  boost::interprocess::file_lock flock(_hdf5file.c_str());
39  flock.lock();
40  CheckpointFile cpf(_hdf5file, CheckpointAccessLevel::READ);
41  CheckpointReader r = cpf.getReader();
