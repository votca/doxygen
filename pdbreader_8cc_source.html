39bool PDBReader::ReadTopology(string file, Topology &top) {
40 topology_ = true;
41 top.Cleanup();
42
43 fl_.open(file);
44 if (!fl_.is_open()) {
45 throw std::ios_base::failure("Error on open topology file: " + file);
46 }
47
48 NextFrame(top);
49 fl_.close();
50
51 return true;
52}

71bool PDBReader::NextFrame(Topology &top) {
72
73 cout << "The VOTCA pdb reader attempts to follow the official format: "
74 << endl;
75 cout << "http://www.wwpdb.org/documentation/file-format" << endl;
76 cout << endl;
77 cout << "NOTE the element symbol is important for associating a mass with "
78 << endl;
79 cout << "each atom, if no element symbol is specified the atom name will be"
80 << endl;
81 cout << "used and assumed to be an element symbol." << endl;
82 cout << "The pdb file format does not support pseudo atoms such as a CH3 "
83 << endl;
84 cout << "and thus VOTCA cannot read such atoms in when using a .pdb file."
85 << endl;
86 cout << "If you need to use pseudo atoms consider using a different file "
87 << endl;
88 cout << "format such as the lammps data format." << endl;
89
90 string line;
91 tools::Elements elements;
92 // Two column vector for storing all bonds
93 // 1 - id of first atom
94 // 2 - id of second atom
95 vector<vector<Index>> bond_pairs;
96 // Store pointers to every bead
97 // WARNING we are assuming in the bead_Eigen::Vector3d that the indices of the
98 // beads
99 // correspond to the order in which they are read in. As in the first
100 // bead read in will be at index 0, etc...
101 vector<Bead *> bead_vec;
103 // Read in information from .pdb file
105 Index bead_count = 0;
106 while (tools::getline(fl_, line)) {
107 if (tools::wildcmp("CRYST1*", line)) {
108 string a, b, c, alpha, beta, gamma;
109 try {
110 // 1 - 6 Record name "CRYST1"
111 a = string(line, (7 - 1), 9);
112 // 7 - 15 Real(9.3) a (Angstroms)
113 b = string(line, (16 - 1), 9);
114 // 16 - 24 Real(9.3) b (Angstroms)
115 c = string(line, (25 - 1), 9);
116 // 25 - 33 Real(9.3) c (Angstroms)
117 alpha = string(line, (34 - 1), 7);
118 // 34 - 40 Real(7.2) alpha (degrees)
119 beta = string(line, (41 - 1), 7);
120 // 41 - 47 Real(7.2) beta (degrees)
121 gamma = string(line, (48 - 1), 7);
122 // 48 - 54 Real(7.2) gamma (degrees)
123 // 56 - 66 LString Space group
124 // 67 - 70 Integer Z value
125 } catch (std::out_of_range &) {
126 throw std::runtime_error("Misformated pdb file in CRYST1 line");
127 }
128 boost::algorithm::trim(a);
129 boost::algorithm::trim(b);
130 boost::algorithm::trim(c);
131 boost::algorithm::trim(alpha);
132 boost::algorithm::trim(beta);
133 boost::algorithm::trim(gamma);
134 if ((!tools::wildcmp("90*", alpha)) || (!tools::wildcmp("90*", beta)) ||
135 (!tools::wildcmp("90*", gamma))) {
136 throw std::runtime_error(
137 "Non cubical box in pdb file not implemented, yet!");
138 }
139 double aa = stod(a) / 10.0;
140 double bb = stod(b) / 10.0;
141 double cc = stod(c) / 10.0;
142 Eigen::Matrix3d box = Eigen::Matrix3d::Zero();
143 box.diagonal() = Eigen::Vector3d(aa, bb, cc);
144 top.setBox(box);
145 }
146 // Only read the CONECT keyword if the topology is set too true
147 if (topology_ && tools::wildcmp("CONECT*", line)) {
148 vector<string> bonded_atms;
149 string atm1;
150 try {
151 // If the CONECT keyword is found then there must be at least
152 // two atom identifiers, more than that is optional.
153
154 stringstream ss;
155 ss << string(line.substr(6));
156 // 1 - 6 Record name "CONECT"
157 // 11 - 7 Real(5) atm1 (ID)
158 ss >> atm1;
159 string temp_atm;
160 ss >> temp_atm;
161 bonded_atms.push_back(temp_atm);
162 ss >> temp_atm;
163 // Here we have taken a less rigorous approach to the .pdb files
164 // we do not care at this point how large the ids of the atoms are
165 // they can be greater than 99,999 with this approach.
166 while (ss) {
167 bonded_atms.push_back(temp_atm);
168 ss >> temp_atm;
169 }
170 } catch (std::out_of_range &) {
171 throw std::runtime_error("Misformated pdb file in CONECT line\n" +
172 line);
173 }
174
175 vector<Index> row(2);
176 boost::algorithm::trim(atm1);
177 Index at1 = boost::lexical_cast<Index>(atm1);
178 row.at(0) = at1;
179
180 for (auto &bonded_atm : bonded_atms) {
181 Index at2 = boost::lexical_cast<Index>(bonded_atm);
182 row.at(1) = at2;
183 // Because every bond will be counted twice in a .pdb file
184 // we will only add bonds where the id (atm1) is less than the
185 // bonded_atm
186 if (at1 < at2) {
187 bond_pairs.push_back(row);
188 }
189 }
190 }
191
192 if (tools::wildcmp("ATOM*", line) || tools::wildcmp("HETATM*", line)) {
193
194 // according to PDB format
195 string x, y, z, resNum, resName, atName;
196 string charge, elem_sym;
197 // string atNum;
198 try {
199 /* Some pdb don't include all this, read only what we really need*/
200 /* leave this here in case we need more later*/
201
202 // str , "ATOM", "HETATM"
203 // string recType (line,( 1-1),6);
204 // Index , Atom serial number
205 // atNum = string(line,( 7-1),6);
206 // str , Atom name
207 atName = string(line, (13 - 1), 4);
208 // char , Alternate location indicator
209 // string atAltLoc (line,(17-1),1);
210 // str , Residue name
211 resName = string(line, (18 - 1), 3);
212 // char , Chain identifier
213 // string chainID (line,(22-1),1);
214 // Index , Residue sequence number
215 resNum = string(line, (23 - 1), 4);
216 // char , Code for insertion of res
217 // string atICode (line,(27-1),1);
218 // float 8.3 , x
219 x = string(line, (31 - 1), 8);
220 // float 8.3 , y
221 y = string(line, (39 - 1), 8);
222 // float 8.3 , z
223 z = string(line, (47 - 1), 8);
224 // float 6.2 , Occupancy
225 // string atOccup (line,(55-1),6);
226 // float 6.2 , Temperature factor
227 // string atTFactor (line,(61-1),6);
228 // str , Segment identifier
229 // string segID (line,(73-1),4);
230 // str , Element symbol
231 elem_sym = string(line, (77 - 1), 2);
232 // str , Charge on the atom
233 charge = string(line, (79 - 1), 2);
234 } catch (std::out_of_range &) {
235 string err_msg = "Misformated pdb file in atom line # " +
236 boost::lexical_cast<string>(bead_count) +
237 "\n the correct pdb file format requires 80 "
238 "characters in width (spaces matter). Furthermore, " +
239 "\n to read the topology in from a .pdb file the "
240 "following attributes must be " +
241 "\n specified: "
242 " " +
243 "\n Atom Name, Residue Name, Residue Number, x, y, z, "
244 "element symbol"
245 "charge (optional) \n";
246 throw std::runtime_error(err_msg);
247 }
248 boost::algorithm::trim(atName);
249 boost::algorithm::trim(resName);
250 boost::algorithm::trim(resNum);
251 boost::algorithm::trim(x);
252 boost::algorithm::trim(y);
253 boost::algorithm::trim(z);
254 boost::algorithm::trim(elem_sym);
255 boost::algorithm::trim(charge);
256
257 bead_count++;
258
259 Bead *b;
260 // Only read the CONECT keyword if the topology is set too true
261 if (topology_) {
262 Index resnr;
263 try {
264 resnr = boost::lexical_cast<Index>(resNum);
265 } catch (bad_lexical_cast &) {
266 throw std::runtime_error(
267 "Cannot convert resNum='" + resNum +
268 "' to int, that usallly means: misformated pdb file");
269 }
270
271 if (resName == "") {
272 cout << "WARNING no resname specified, assigning name to: UNK"
273 << endl;
274 resName = "UNK";
275 }
276
277 if (resnr < 1) {
278 throw std::runtime_error("Misformated pdb file, resnr has to be > 0");
279 }
280 // TODO: fix the case that resnr is not in ascending order
281 if (resnr > top.ResidueCount()) {
282 while ((resnr - 1) > top.ResidueCount()) { // pdb resnr should start
283 // with 1 but accept
284 // sloppy files
285
286 // create dummy residue, hopefully it will never show
287
288 top.CreateResidue(resName);
289 cout << "Warning: residue numbers not continuous, create dummy "
290 "residue with residue number "
291 << top.ResidueCount() << endl;
292 }
293 top.CreateResidue(resName);
294 }
295 // This is not correct, but still better than no type at all!
296 if (!top.BeadTypeExist(atName)) {
297 top.RegisterBeadType(atName);
298 }
299
300 // Determine if the charge has been provided in the .pdb file or if we
301 // will be assuming it is 0
302 double ch = 0;
303 if (charge != "") {
304 ch = stod(charge);
305 } else {
306 cout << "WARNING no charge was specified for " << endl;
307 cout << line << endl;
308 cout << "Assuming a charge of 0" << endl;
309 }
310
311 if (elem_sym == "") {
312 cout << "WARNING no element was specified, assuming atom name is "
313 << endl;
314 cout << "an element symbol: " << atName << endl;
315 if (elements.isElement(atName)) {
316 elem_sym = atName;
317 } else {
318 throw std::runtime_error(
319 "Atom name is not an element symbol, so substitution fails, "
320 "the element is needed in order to resolve the mass.");
321 }
322 }
323
324 // CreateBead takes 6 parameters in the following order
325 // 1 - symmetry of the bead
326 // 2 - name of the bead (string)
327 // 3 - bead type (BeadType *)
328 // 4 - residue number (Index)
329 // 5 - mass (double)
330 // 6 - charge (double)
331 //
332 // res -1 as internal number starts with 0
333 b = top.CreateBead(Bead::spherical, atName, atName, resnr - 1,
334 elements.getMass(elem_sym), ch);
335 } else {
336 b = top.getBead(bead_count - 1);
337 }
338 // convert to nm from A
339 b->setPos(
340 Eigen::Vector3d(stod(x) / 10.0, stod(y) / 10.0, stod(z) / 10.0));
341
342 bead_vec.push_back(b);
343 }
344
345 if ((line == "ENDMDL") || (line == "END") || (fl_.eof())) {
346 break;
347 }
348 }
349
350 if (!topology_ && (bead_count > 0) && bead_count != top.BeadCount()) {
351 throw std::runtime_error(
352 "number of beads in topology and trajectory differ");
353 }
354
356 // Sort data and determine atom structure, connect with top (molecules, bonds)
358
359 // Extra processing is done if the file is a topology file, in which case the
360 // atoms must be sorted into molecules and the bond interactions recorded
361 if (topology_) {
362 // Now we need to add the bond pairs
363 // WARNING We are assuming the atom ids are contiguous with no gaps
364
365 // First Index - is the index of the atom
366 // Second Index - is the index of the molecule
367 map<Index, Index> atm_molecule;
368
369 // First Index - is the index of the molecule
370 // list<Index> - is a list of the atoms in the molecule
371 unordered_map<Index, list<Index>> molecule_atms;
372
373 // Keep track of the number of molecules we have created through an index
374 Index mol_index = 0;
375
376 // Cycle through all bonds
377 for (auto &bond_pair : bond_pairs) {
378
379 Index atm_id1 = bond_pair.at(0);
380 Index atm_id2 = bond_pair.at(1);
381 // Check to see if either atm referred to in the bond is already
382 // attached to a molecule
383 auto mol_iter1 = atm_molecule.find(atm_id1);
384 auto mol_iter2 = atm_molecule.find(atm_id2);
385
386 // This means neither atom is attached to a molecule
387 if (mol_iter1 == atm_molecule.end() && mol_iter2 == atm_molecule.end()) {
388 // We are going to create a new row for a new molecule
389 list<Index> atms_in_mol;
390 atms_in_mol.push_back(atm_id1);
391 atms_in_mol.push_back(atm_id2);
392 molecule_atms[mol_index] = atms_in_mol;
393 // Associate atm1 and atm2 with the molecule index
394 atm_molecule[atm_id1] = mol_index;
395 atm_molecule[atm_id2] = mol_index;
396 // Increment the molecule index
397 mol_index++;
398
399 // This means only atm2 is attached to a molecule
400 } else if (mol_iter1 == atm_molecule.end()) {
401 // Add atm1 to the molecule that contains atm2
402 molecule_atms[mol_iter2->second].push_back(atm_id1);
403 // Associate atm1 with the molecule it is now part of
404 atm_molecule[atm_id1] = mol_iter2->second;
405
406 // This means only atm1 is attached to a molecule
407 } else if (mol_iter2 == atm_molecule.end()) {
408 // Add atm2 to the molecule that contains atm1
409 molecule_atms[mol_iter1->second].push_back(atm_id2);
410 // Associate atm1 with the molecule it is now part of
411 atm_molecule[atm_id2] = mol_iter1->second;
412
413 } else if (mol_iter1 != mol_iter2) {
414 // This means both atm1 and atm2 are attached to a molecule
415 // But if they are already attached to the same molecule there is
416 // nothing else to be done.
417 Index chosen_mol;
418 Index obsolete_mol;
419 // We will merge the atms to the molecule with the smallest index
420 if (mol_iter1->second < mol_iter2->second) {
421 chosen_mol = mol_iter1->second;
422 obsolete_mol = mol_iter2->second;
423 } else {
424 chosen_mol = mol_iter2->second;
425 obsolete_mol = mol_iter1->second;
426 }
427
428 // Now we will proceed to cycle through the atms that were in the now
429 // obsolete molecule and make sure they are pointing to the new molecule
430 for (Index &atm_temp : molecule_atms[obsolete_mol]) {
431
432 atm_molecule[atm_temp] = chosen_mol;
433 }
434
435 // Splicing will remove atoms from the now obsolete molecule and place
436 // them on the chosen molecule.
437 molecule_atms[chosen_mol].splice(molecule_atms[chosen_mol].end(),
438 molecule_atms[obsolete_mol]);
439
440 // Finally we will clear out the record of the obsolete molecule
441 molecule_atms.erase(obsolete_mol);
442 }
443 }
444#ifndef NDEBUG
445 cerr << "Consistency check for pdbreader" << endl;
446 Index i = 0;
447 for (auto lis = molecule_atms.begin(); lis != molecule_atms.end(); lis++) {
448 cerr << "Molecule " << i << endl;
449 cerr << "Atoms: ";
450 for (auto atm_ind = lis->second.begin(); atm_ind != lis->second.end();
451 atm_ind++) {
452 cerr << *atm_ind << " ";
453 }
454 cerr << endl;
455 i++;
456 }
457 cerr << endl;
458#endif
459 // Now that we know which interactions belong to which molecules we can:
460 // 1 Add the molecules
461 // 2 Add the bond interactions
462
463 // Molecule map
464 // First Index - is the index of the molecule
465 // Molecule* - is a pointer to the Molecule object
466 map<Index, Molecule *> mol_map;
467
468 // Used to reindex the molecules so that they start at 0 and progress
469 // with out gaps in their ids.
470 // First Index - is the index of the old molecule
471 // Second Index - is the new index
472 map<Index, Index> mol_reInd_map;
473
474 Index ind = 0;
475 for (auto mol = molecule_atms.begin(); mol != molecule_atms.end(); mol++) {
476
477 string mol_name = "PDB Molecule " + boost::lexical_cast<string>(ind);
478
479 Molecule *mi = top.CreateMolecule(mol_name);
480 mol_map[mol->first] = mi;
481 mol_reInd_map[mol->first] = ind;
482
483 // Add all the atoms to the appropriate molecule object
484 list<Index> atm_list = molecule_atms[mol->first];
485 for (Index &atm_temp : atm_list) {
486
487 string residuename = "DUM";
488 mi->AddBead(bead_vec.at(atm_temp - 1), residuename);
489 }
490 ind++;
491 }
492
493 Index bond_indx = 0;
494 // Cyle through the bonds and add them to the appropriate molecule
495 for (auto &bond_pair : bond_pairs) {
496
497 Index atm_id1 = bond_pair.at(0);
498 Index atm_id2 = bond_pair.at(1);
499 // Should be able to just look at one of the atoms the bond is attached
500 // too because the other will also be attached to the same molecule.
501 Index mol_ind = atm_molecule[atm_id1];
502 Molecule *mi = mol_map[mol_ind];
503 // Grab the id of the bead associated with the atom
504 // It may be the case that the atom id's and bead id's are different
505 Index bead_id1 = bead_vec.at(atm_id1 - 1)->getId();
506 Index bead_id2 = bead_vec.at(atm_id2 - 1)->getId();
507 Interaction *ic = new IBond(bead_id1, bead_id2);
508 ic->setGroup("BONDS");
509 ic->setIndex(bond_indx);
510 bond_indx++;
511 ic->setMolecule(mol_reInd_map[mol_ind]);
512 top.AddBondedInteraction(ic);
513 mi->AddInteraction(ic);
514 }
515
516 // Finally we want to build an exclusion matrix
517 top.RebuildExclusions();
518 }
519
520 return !fl_.eof();
521}
