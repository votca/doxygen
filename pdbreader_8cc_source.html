64 bool PDBReader::NextFrame(Topology &top)
65 {
66  string line;
67  int i = 0 ;
68  while ( std::getline(_fl, line) ){
69  if( wildcmp("CRYST1*",line.c_str())){
70  string a, b, c, alpha, beta, gamma;
71  try {
72  //1 - 6 Record name "CRYST1"
73  a=string(line,(7-1),9); //7 - 15 Real(9.3) a (Angstroms)
74  b=string(line,(16-1),9); //16 - 24 Real(9.3) b (Angstroms)
75  c=string(line,(25-1),9); //25 - 33 Real(9.3) c (Angstroms)
76  alpha=string(line,(34-1),7); //34 - 40 Real(7.2) alpha (degrees)
77  beta=string(line,(41-1),7); //41 - 47 Real(7.2) beta (degrees)
78  gamma=string(line,(48-1),7); //48 - 54 Real(7.2) gamma (degrees)
79  //56 - 66 LString Space group
80  //67 - 70 Integer Z value
81  } catch (std::out_of_range& err) {
82  throw std::runtime_error("Misformated pdb file in CRYST1 line");
83  }
84  boost::algorithm::trim(a);
85  boost::algorithm::trim(b);
86  boost::algorithm::trim(c);
87  boost::algorithm::trim(alpha);
88  boost::algorithm::trim(beta);
89  boost::algorithm::trim(gamma);
90  if ((!wildcmp("90*",alpha.c_str()))||(!wildcmp("90*",alpha.c_str()))||(!wildcmp("90*",alpha.c_str()))){
91  throw std::runtime_error("Non cubical box in pdb file not implemented, yet!");
92  }
93  top.setBox(matrix(vec(boost::lexical_cast<double>(a)/10.0, 0, 0),
94  vec(0, boost::lexical_cast<double>(b)/10.0, 0),
95  vec(0, 0, boost::lexical_cast<double>(c)/10.0)));
96 
97  }
98  if( wildcmp("ATOM*",line.c_str()) || wildcmp("HETATM*",line.c_str())){
99 
100  // according to PDB format
101  string x,y,z, resNum, resName, atName;
102  try {
103  /* Some pdb don't include all this, read only what we realy need*/
104  /* leave this here in case we need more later*/
105  //string recType (line,( 1-1),6); // str, "ATOM", "HETATM"
106  //string atNum (line,( 7-1),6); // int, Atom serial number
107  atName=string(line,(13-1),4); // str, Atom name
108  //string atAltLoc (line,(17-1),1); // char, Alternate location indicator
109  resName=string(line,(18-1),3); // str, Residue name
110  //string chainID (line,(22-1),1); // char, Chain identifier
111  resNum=string(line,(23-1),4); // int, Residue sequence number
112  //string atICode (line,(27-1),1); // char, Code for insertion of res
113  x=string(line,(31-1),8); // float 8.3 ,x
114  y=string(line,(39-1),8); // float 8.3 ,y
115  z=string(line,(47-1),8); // float 8.3 ,z
116  //string atOccup (line,(55-1),6); // float 6.2, Occupancy
117  //string atTFactor (line,(61-1),6); // float 6.2, Temperature factor
118  //string segID (line,(73-1),4); // str, Segment identifier
119  //string atElement (line,(77-1),2); // str, Element symbol
120  //string atCharge (line,(79-1),2); // str, Charge on the atom
121 
122  } catch (std::out_of_range& err) {
123  throw std::runtime_error("Misformated pdb file in atom line # "+ boost::lexical_cast<string>(i));
124  }
125  boost::algorithm::trim(atName);
126  boost::algorithm::trim(resName);
127  boost::algorithm::trim(resNum);
128  boost::algorithm::trim(x);
129  boost::algorithm::trim(y);
130  boost::algorithm::trim(z);
131 
132  i++;
133  if(!_topology && i > top.BeadCount())
134  throw std::runtime_error("number of beads in topology and trajectory differ");
135 
136  Bead *b;
137  if(_topology){
138  int resnr;
139  try {
140  resnr = boost::lexical_cast<int>(resNum);
141  } catch(bad_lexical_cast &) {
142  throw std::runtime_error("Cannot convert resNum='"+ resNum+"' to int, that usallly means: misformated pdb file");
143  }
144  if (resnr < 1)
145  throw std::runtime_error("Misformated pdb file, resnr has to be > 0");
146  //TODO: fix the case that resnr is not in ascending order
147  if(resnr > top.ResidueCount()) {
148  while ((resnr-1)>top.ResidueCount()){ //pdb resnr should start with 1 but accept sloppy files
149  top.CreateResidue("DUMMY"); // create dummy residue, hopefully it will never show
150  cout << "Warning: residue numbers not continous, create DUMMY residue with nr " << top.ResidueCount() << endl;
151  }
152  top.CreateResidue(resName);
153  }
154  //this is not correct, but still better than no type at all!
155  BeadType *type = top.GetOrCreateBeadType(atName);
156 
157  // res -1 as internal number starts with 0
158  b = top.CreateBead(1, atName, type, resnr-1, 1., 0.);
159  } else {
160  b = top.getBead(i-1);
161  }
162  // convert to nm from A
163  b->setPos(vec(
164  boost::lexical_cast<double>(x)/10.0,
165  boost::lexical_cast<double>(y)/10.0,
166  boost::lexical_cast<double>(z)/10.0
167  ));
168 
169  }
170 
171  if (( line == "ENDMDL" ) || ( line == "END" ) || ( _fl.eof())){
172  break;
173  }
174  }
175 
176  if(!_topology && (i>0) && i != top.BeadCount())
177  throw std::runtime_error("number of beads in topology and trajectory differ");
178 
179  if (_topology)
180  cout << "WARNING: topology created from .pdb file, charges and masses are wrong!\n";
181 
182  return !_fl.eof();
183 }
