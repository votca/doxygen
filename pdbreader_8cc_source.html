64 bool PDBReader::NextFrame(Topology &top) {
65  string line;
66  // Two column vector for storing all bonds
67  // 1 - id of first atom
68  // 2 - id of second atom
69  vector<vector<int>> bond_pairs;
70  // Store pointers to every bead
71  // WARNING we are assuming in the bead_vec that the indices of the beads
72  // correspond to the order in which they are read in. As in the first
73  // bead read in will be at index 0, etc...
74  vector<Bead *> bead_vec;
76  // Read in information from .pdb file
78  int bead_count = 0;
79  while (std::getline(_fl, line)) {
80  if (wildcmp("CRYST1*", line.c_str())) {
81  string a, b, c, alpha, beta, gamma;
82  try {
83  // 1 - 6 Record name "CRYST1"
84  a = string(line, (7 - 1), 9);
85  // 7 - 15 Real(9.3) a (Angstroms)
86  b = string(line, (16 - 1), 9);
87  // 16 - 24 Real(9.3) b (Angstroms)
88  c = string(line, (25 - 1), 9);
89  // 25 - 33 Real(9.3) c (Angstroms)
90  alpha = string(line, (34 - 1), 7);
91  // 34 - 40 Real(7.2) alpha (degrees)
92  beta = string(line, (41 - 1), 7);
93  // 41 - 47 Real(7.2) beta (degrees)
94  gamma = string(line, (48 - 1), 7);
95  // 48 - 54 Real(7.2) gamma (degrees)
96  // 56 - 66 LString Space group
97  // 67 - 70 Integer Z value
98  }
99  catch (std::out_of_range &err) {
100  throw std::runtime_error("Misformated pdb file in CRYST1 line");
101  }
102  boost::algorithm::trim(a);
103  boost::algorithm::trim(b);
104  boost::algorithm::trim(c);
105  boost::algorithm::trim(alpha);
106  boost::algorithm::trim(beta);
107  boost::algorithm::trim(gamma);
108  if ((!wildcmp("90*", alpha.c_str())) ||
109  (!wildcmp("90*", alpha.c_str())) ||
110  (!wildcmp("90*", alpha.c_str()))) {
111  throw std::runtime_error(
112  "Non cubical box in pdb file not implemented, yet!");
113  }
114  double aa = boost::lexical_cast<double>(a) / 10.0;
115  double bb = boost::lexical_cast<double>(b) / 10.0;
116  double cc = boost::lexical_cast<double>(c) / 10.0;
117  top.setBox(matrix(vec(aa, 0, 0), vec(0, bb, 0), vec(0, 0, cc)));
118  }
119  // Only read the CONECT keyword if the topology is set too true
120  if (_topology && wildcmp("CONECT*", line.c_str())) {
121  vector<string> bonded_atms;
122  string atm1;
123  // Keep track of the number of bonds
124  int num_bonds = 0;
125  try {
126  // If the CONECT keyword is found then there must be at least
127  // two atom identifiers, more than that is optional.
128 
129  stringstream ss;
130  ss << string(line.substr(6));
131  // 1 - 6 Record name "CONECT"
132  // 11 - 7 Real(5) atm1 (ID)
133  ss >> atm1;
134  string temp_atm;
135  ss >> temp_atm;
136  bonded_atms.push_back(temp_atm);
137  num_bonds = 1;
138  ss >> temp_atm;
139  // Here we have taken a less rigorous approach to the .pdb files
140  // we do not care at this point how large the ids of the atoms are
141  // they can be greater than 99,999 with this approach.
142  while (ss) {
143  bonded_atms.push_back(temp_atm);
144  num_bonds++;
145  ss >> temp_atm;
146  }
147  }
148  catch (std::out_of_range &err) {
149  throw std::runtime_error("Misformated pdb file in CONECT line\n" +
150  line);
151  }
152 
153  vector<int> row(2);
154  boost::algorithm::trim(atm1);
155  int at1 = boost::lexical_cast<int>(atm1);
156  row.at(0) = at1;
157 
158  for (auto bonded_atm = bonded_atms.begin();
159  bonded_atm != bonded_atms.end(); bonded_atm++) {
160  int at2 = boost::lexical_cast<int>(*bonded_atm);
161  row.at(1) = at2;
162  // Because every bond will be counted twice in a .pdb file
163  // we will only add bonds where the id (atm1) is less than the
164  // bonded_atm
165  if (at1 < at2) {
166  bond_pairs.push_back(row);
167  }
168  }
169  }
170 
171  if (wildcmp("ATOM*", line.c_str()) || wildcmp("HETATM*", line.c_str())) {
172 
173  // according to PDB format
174  string x, y, z, resNum, resName, atName;
175  string charge;
176  // string atNum;
177  try {
178  /* Some pdb don't include all this, read only what we really need*/
179  /* leave this here in case we need more later*/
180 
181  // str , "ATOM", "HETATM"
182  // string recType (line,( 1-1),6);
183  // int , Atom serial number
184  // atNum = string(line,( 7-1),6);
185  // str , Atom name
186  atName = string(line, (13 - 1), 4);
187  // char , Alternate location indicator
188  // string atAltLoc (line,(17-1),1);
189  // str , Residue name
190  resName = string(line, (18 - 1), 3);
191  // char , Chain identifier
192  // string chainID (line,(22-1),1);
193  // int , Residue sequence number
194  resNum = string(line, (23 - 1), 4);
195  // char , Code for insertion of res
196  // string atICode (line,(27-1),1);
197  // float 8.3 , x
198  x = string(line, (31 - 1), 8);
199  // float 8.3 , y
200  y = string(line, (39 - 1), 8);
201  // float 8.3 , z
202  z = string(line, (47 - 1), 8);
203  // float 6.2 , Occupancy
204  // string atOccup (line,(55-1),6);
205  // float 6.2 , Temperature factor
206  // string atTFactor (line,(61-1),6);
207  // str , Segment identifier
208  // string segID (line,(73-1),4);
209  // str , Element symbol
210  // elem_sym = string(line,(77-1),2);
211  // str , Charge on the atom
212  charge = string(line, (79 - 1), 2);
213  }
214  catch (std::out_of_range &err) {
215  string err_msg = "Misformated pdb file in atom line # " +
216  boost::lexical_cast<string>(bead_count) +
217  "\n the correct pdb file format requires 80 "
218  "characters in width. Furthermore, " +
219  "\n to read the topology in from a .pdb file the "
220  "following attributes must be " +
221  "\n specified: "
222  " " +
223  "\n Atom Name, Residue Name, Residue Number, x, y, z, "
224  "charge (optional) \n";
225  throw std::runtime_error(err_msg);
226  }
227  boost::algorithm::trim(atName);
228  boost::algorithm::trim(resName);
229  boost::algorithm::trim(resNum);
230  boost::algorithm::trim(x);
231  boost::algorithm::trim(y);
232  boost::algorithm::trim(z);
233  boost::algorithm::trim(charge);
234 
235  bead_count++;
236 
237  Bead *b;
238  // Only read the CONECT keyword if the topology is set too true
239  if (_topology) {
240  int resnr;
241  try {
242  resnr = boost::lexical_cast<int>(resNum);
243  }
244  catch (bad_lexical_cast &) {
245  throw std::runtime_error(
246  "Cannot convert resNum='" + resNum +
247  "' to int, that usallly means: misformated pdb file");
248  }
249  if (resnr < 1)
250  throw std::runtime_error("Misformated pdb file, resnr has to be > 0");
251  // TODO: fix the case that resnr is not in ascending order
252  if (resnr > top.ResidueCount()) {
253  while ((resnr - 1) > top.ResidueCount()) { // pdb resnr should start
254  // with 1 but accept
255  // sloppy files
256 
257  // create dummy residue, hopefully it will never show
258  top.CreateResidue("DUMMY");
259  cout << "Warning: residue numbers not continous, create DUMMY "
260  "residue with nr " << top.ResidueCount() << endl;
261  }
262  top.CreateResidue(resName);
263  }
264  // This is not correct, but still better than no type at all!
265  BeadType *type = top.GetOrCreateBeadType(atName);
266 
267  // Determine if the charge has been provided in the .pdb file or if we
268  // will be assuming it is 0
269  double ch = 0;
270  if (charge != "") {
271  ch = boost::lexical_cast<double>(charge);
272  }
273  // CreateBead takes 6 parameters in the following order
274  // 1 - symmetry of the bead (1-indicates sphere, 3-indicates
275  // ellipsoidal)
276  // 2 - name of the bead (string)
277  // 3 - bead type (BeadType *)
278  // 4 - residue number (int)
279  // 5 - mass (double)
280  // 6 - charge (double)
281  //
282  // res -1 as internal number starts with 0
283  b = top.CreateBead(1, atName, type, resnr - 1, this->getMass(atName),
284  ch);
285  } else {
286  b = top.getBead(bead_count - 1);
287  }
288  // convert to nm from A
289  b->setPos(vec(boost::lexical_cast<double>(x) / 10.0,
290  boost::lexical_cast<double>(y) / 10.0,
291  boost::lexical_cast<double>(z) / 10.0));
292 
293  bead_vec.push_back(b);
294  }
295 
296  if ((line == "ENDMDL") || (line == "END") || (_fl.eof())) {
297  break;
298  }
299  }
300 
301  if (!_topology && (bead_count > 0) && bead_count != top.BeadCount())
302  throw std::runtime_error(
303  "number of beads in topology and trajectory differ");
304 
306  // Sort data and determine atom structure, connect with top (molecules, bonds)
308 
309  // Extra processing is done if the file is a topology file, in which case the
310  // atoms must be sorted into molecules and the bond interactions recorded
311  if (_topology) {
312  // Now we need to add the bond pairs
313  // WARNING We are assuming the atom ids are contiguous with no gaps
314 
315  // First int - is the index of the atom
316  // Second int - is the index of the molecule
317  map<int, int> atm_molecule;
318 
319  // First int - is the index of the molecule
320  // list<int> - is a list of the atoms in the molecule
321  unordered_map<int, list<int>> molecule_atms;
322 
323  // Keep track of the number of molecules we have created through an index
324  int mol_index = 0;
325 
326  // Cycle through all bonds
327  for (auto row = bond_pairs.begin(); row != bond_pairs.end(); row++) {
328 
329  int atm_id1 = row->at(0);
330  int atm_id2 = row->at(1);
331  // Check to see if either atm referred to in the bond is already
332  // attached to a molecule
333  auto mol_iter1 = atm_molecule.find(atm_id1);
334  auto mol_iter2 = atm_molecule.find(atm_id2);
335 
336  // This means neither atom is attached to a molecule
337  if (mol_iter1 == atm_molecule.end() && mol_iter2 == atm_molecule.end()) {
338  // We are going to create a new row for a new molecule
339  list<int> atms_in_mol;
340  atms_in_mol.push_back(atm_id1);
341  atms_in_mol.push_back(atm_id2);
342  molecule_atms[mol_index] = atms_in_mol;
343  // Associate atm1 and atm2 with the molecule index
344  atm_molecule[atm_id1] = mol_index;
345  atm_molecule[atm_id2] = mol_index;
346  // Increment the molecule index
347  mol_index++;
348 
349  // This means only atm2 is attached to a molecule
350  } else if (mol_iter1 == atm_molecule.end()) {
351  // Add atm1 to the molecule that contains atm2
352  molecule_atms[mol_iter2->second].push_back(atm_id1);
353  // Associate atm1 with the molecule it is now part of
354  atm_molecule[atm_id1] = mol_iter2->second;
355 
356  // This means only atm1 is attached to a molecule
357  } else if (mol_iter2 == atm_molecule.end()) {
358  // Add atm2 to the molecule that contains atm1
359  molecule_atms[mol_iter1->second].push_back(atm_id2);
360  // Associate atm1 with the molecule it is now part of
361  atm_molecule[atm_id2] = mol_iter1->second;
362 
363  } else if (mol_iter1 != mol_iter2) {
364  // This means both atm1 and atm2 are attached to a molecule
365  // But if they are already attached to the same molecule there is
366  // nothing else to be done.
367  int chosen_mol;
368  int obsolete_mol;
369  // We will merge the atms to the molecule with the smallest index
370  if (mol_iter1->second < mol_iter2->second) {
371  chosen_mol = mol_iter1->second;
372  obsolete_mol = mol_iter2->second;
373  } else {
374  chosen_mol = mol_iter2->second;
375  obsolete_mol = mol_iter1->second;
376  }
377 
378  // Now we will proceed to cycle through the atms that were in the now
379  // obsolete molecule and make sure they are pointing to the new molecule
380  for (auto atm_temp = molecule_atms[obsolete_mol].begin();
381  atm_temp != molecule_atms[obsolete_mol].end(); atm_temp++) {
382 
383  atm_molecule[*atm_temp] = chosen_mol;
384  }
385 
386  // Splicing will remove atoms from the now obsolete molecule and place
387  // them on the chosen molecule.
388  molecule_atms[chosen_mol].splice(molecule_atms[chosen_mol].end(),
389  molecule_atms[obsolete_mol]);
390 
391  // Finally we will clear out the record of the obsolete molecule
392  molecule_atms.erase(obsolete_mol);
393  }
394  }
395 #ifndef NDEBUG
396  cerr << "Consistency check for pdbreader" << endl;
397  int i = 0;
398  for (auto lis = molecule_atms.begin(); lis != molecule_atms.end(); lis++) {
399  cerr << "Molecule " << i << endl;
400  cerr << "Atoms: ";
401  for (auto atm_ind = lis->second.begin(); atm_ind != lis->second.end();
402  atm_ind++) {
403  cerr << *atm_ind << " ";
404  }
405  cerr << endl;
406  i++;
407  }
408  cerr << endl;
409 #endif
410  // Now that we know which interactions belong to which molecules we can:
411  // 1 Add the molecules
412  // 2 Add the bond interactions
413 
414  // Molecule map
415  // First int - is the index of the molecule
416  // Molecule* - is a pointer to the Molecule object
417  map<int, Molecule *> mol_map;
418 
419  // Used to reindex the molecules so that they start at 0 and progress
420  // with out gaps in their ids.
421  // First int - is the index of the old molecule
422  // Second int - is the new index
423  map<int, int> mol_reInd_map;
424 
425  int ind = 0;
426  for (auto mol = molecule_atms.begin(); mol != molecule_atms.end(); mol++) {
427 
428  string mol_name = "PDB Molecule " + boost::lexical_cast<string>(ind);
429 
430  Molecule *mi = top.CreateMolecule(mol_name);
431  mol_map[mol->first] = mi;
432  mol_reInd_map[mol->first] = ind;
433 
434  // Add all the atoms to the appropriate molecule object
435  list<int> atm_list = molecule_atms[mol->first];
436  for (auto atm_temp = atm_list.begin(); atm_temp != atm_list.end();
437  atm_temp++) {
438 
439  string residuename = "DUM";
440  mi->AddBead(bead_vec.at(*atm_temp - 1), residuename);
441  }
442  ind++;
443  }
444 
445  int bond_indx = 0;
446  // Cyle through the bonds and add them to the appropriate molecule
447  for (auto bond_pair = bond_pairs.begin(); bond_pair != bond_pairs.end();
448  bond_pair++) {
449 
450  int atm_id1 = bond_pair->at(0);
451  int atm_id2 = bond_pair->at(1);
452  // Should be able to just look at one of the atoms the bond is attached
453  // too because the other will also be attached to the same molecule.
454  int mol_ind = atm_molecule[atm_id1];
455  Molecule *mi = mol_map[mol_ind];
456  // Grab the id of the bead associated with the atom
457  // It may be the case that the atom id's and bead id's are different
458  int bead_id1 = bead_vec.at(atm_id1 - 1)->getId();
459  int bead_id2 = bead_vec.at(atm_id2 - 1)->getId();
460  Interaction *ic = new IBond(bead_id1, bead_id2);
461  ic->setGroup("BONDS");
462  ic->setIndex(bond_indx);
463  bond_indx++;
464  ic->setMolecule(mol_reInd_map[mol_ind]);
465  top.AddBondedInteraction(ic);
466  mi->AddInteraction(ic);
467  }
468 
469  // Finally we want to build an exclusion matrix
470  top.RebuildExclusions();
471  }
472 
473  return !_fl.eof();
474 }
