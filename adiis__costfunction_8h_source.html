42  Eigen::Map<const Eigen::VectorXd> x(parameters, _DiF.size());
43  Eigen::VectorXd c = x.cwiseAbs2();
44  double xnorm = c.sum();
45  c /= xnorm;
46 
47  cost[0] = (2 * c.transpose() * _DiF + c.transpose() * _DiFj * c).value();
48  if (gradient != NULL) {
49  Eigen::VectorXd dEdc = 2.0 * _DiF + _DiFj * c + _DiFj.transpose() * c;
50  Eigen::MatrixXd jac = Eigen::MatrixXd::Zero(c.size(), c.size());
51  for (int i = 0; i < jac.rows(); i++) {
52  for (int j = 0; j < jac.cols(); j++) {
53  jac(i, j) = -c(i)*2.0 * x(j) / xnorm;
54  }
55  // Extra term on diagonal
56  jac(i, i) += 2.0 * x(i) / xnorm;
57  }
58  Eigen::VectorXd dEdxv = jac.transpose() * dEdc;
59  for (int i = 0; i < dEdxv.size(); ++i) {
60  gradient[i] = dEdxv(i);

62  }
