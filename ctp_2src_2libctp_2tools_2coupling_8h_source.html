99 bool Coupling::Evaluate() {
100 
101  _log.setReportLevel( logDEBUG );
102  _log.setMultithreading( true );
103 
104  _log.setPreface(logINFO, "\n... ...");
105  _log.setPreface(logERROR, "\n... ...");
106  _log.setPreface(logWARNING, "\n... ...");
107  _log.setPreface(logDEBUG, "\n... ...");
108 
109  // get the corresponding object from the QMPackageFactory
110  QMPackage *_qmpackage = QMPackages().Create( _package );
111  _qmpackage->setLog( &_log );
112  _qmpackage->Initialize( &_package_options );
113 
114  Orbitals _orbitalsA, _orbitalsB, _orbitalsAB;
115 
116  _qmpackage->setOrbitalsFileName( _orbA );
117  bool _parse_orbitalsA_status = _qmpackage->ParseOrbitalsFile( &_orbitalsA );
118  if ( !_parse_orbitalsA_status ) { CTP_LOG(logERROR,_log) << "Failed to read orbitals of molecule A" << std::flush; }
119 
120  _qmpackage->setOrbitalsFileName( _orbB );
121  bool _parse_orbitalsB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsB );
122  if ( !_parse_orbitalsB_status ) { CTP_LOG(logERROR,_log) << "Failed to read orbitals of molecule B" << std::flush; }
123 
124  _qmpackage->setOrbitalsFileName( _orbAB );
125  bool _parse_orbitalsAB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
126  if ( !_parse_orbitalsAB_status ) { CTP_LOG(logERROR,_log) << "Failed to read orbitals of dimer AB" << std::flush; }
127 
128  _qmpackage->setLogFileName( _logA );
129  bool _parse_logA_status = _qmpackage->ParseLogFile( &_orbitalsA );
130  if ( !_parse_logA_status ) { CTP_LOG(logERROR,_log) << "Failed to read log of molecule A" << std::flush; }
131 
132  _qmpackage->setLogFileName( _logB );
133  bool _parse_logB_status = _qmpackage->ParseLogFile( &_orbitalsB );
134  if ( !_parse_logB_status ) { CTP_LOG(logERROR,_log) << "Failed to read log of molecule B" << std::flush; }
135 
136  _qmpackage->setLogFileName( _logAB );
137  bool _parse_logAB_status = _qmpackage->ParseLogFile( &_orbitalsAB );
138  if ( !_parse_logAB_status ) { CTP_LOG(logERROR,_log) << "Failed to read log of molecule AB" << std::flush; }
139 
140  int _degAH = 1;
141  int _degAL = 1;
142  int _degBH = 1;
143  int _degBL = 1;
144 
145  // trim monomers A and B to one level
146  if ((_trimA == -1) || (_trimB == -1) ) { // any -1 overrides the specification of the other
147 
148  // find degeneracy of HOMOs and LUMOs
149  std::vector<int> list_levelsAH = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons()-1, _degeneracy ));
150  _degAH = list_levelsAH.size();
151  std::vector<int> list_levelsAL = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons(), _degeneracy ));
152  _degAL = list_levelsAL.size();
153 
154  std::vector<int> list_levelsBH = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons()-1, _degeneracy ));
155  _degBH = list_levelsBH.size();
156  std::vector<int> list_levelsBL = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons(), _degeneracy ));
157  _degBL = list_levelsBL.size();
158 
159  _orbitalsA.Trim(_degAH,_degAL);
160  _orbitalsB.Trim(_degBH,_degBL);
161 
162  // trim by the factors (_trimA-1) and (_trimB-1)
163  } else {
164 
165  if ( _orbitalsA.getNumberOfElectrons()*(_trimA-1) < _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() ) {
166  CTP_LOG(logDEBUG,_log) << "Trimming virtual orbitals A:"
167  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
168  << _orbitalsA.getNumberOfElectrons()*(_trimA-1) << std::flush;
169  _orbitalsA.Trim(_trimA);
170  }
171 
172  if ( _orbitalsB.getNumberOfElectrons()*(_trimB-1) < _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() ) {
173  CTP_LOG(logDEBUG,_log) << "Trimming virtual orbitals B:"
174  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
175  << _orbitalsB.getNumberOfElectrons()*(_trimB-1) << std::flush;
176  _orbitalsB.Trim(_trimB);
177  }
178 
179  }
180 
181  Overlap _overlap;
182  _overlap.setLogger(&_log);
183 
184  ub::matrix<double> _JAB;
185  bool _calculate_integrals = _overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
186  if ( !_calculate_integrals ) { CTP_LOG(logERROR,_log) << "Failed to evaluate integrals" << std::flush; }
187 
188  std::cout << _log;
189 
190 
191  // output the results
192  Property _summary;
193  Property *_job_output = &_summary.add("output","");
194  Property *_pair_summary = &_job_output->add("pair","");
195  int HOMO_A = _orbitalsA.getNumberOfElectrons();
196  int HOMO_B = _orbitalsB.getNumberOfElectrons();
197  int LUMO_A = HOMO_A + 1;
198  int LUMO_B = HOMO_B + 1;
199  _pair_summary->setAttribute("homoA", HOMO_A);
200  _pair_summary->setAttribute("homoB", HOMO_B);
201 
202  if ( (_trimA == -1) || (_trimB == -1) ) {
203 
204  // HOMO-HOMO coupling
205  double JAB = _overlap.getCouplingElement(_degAH, _degBH , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
206  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
207  double energyA = _orbitalsA.getEnergy(_degAH);
208  double energyB = _orbitalsB.getEnergy(_degBH);
209  _overlap_summary->setAttribute("orbA", HOMO_A);
210  _overlap_summary->setAttribute("orbB", HOMO_B);
211  //_overlap_summary->setAttribute("jAB", JAB);
212  _overlap_summary->setAttribute("eA", energyA);
213  _overlap_summary->setAttribute("eB", energyB);
214 
215  // LUMO-LUMO coupling
216  JAB = _overlap.getCouplingElement(_degAH+1, _degBH+1 , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
217  _overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
218  energyA = _orbitalsA.getEnergy(_degAH +1);
219  energyB = _orbitalsB.getEnergy(_degBH +1);
220  _overlap_summary->setAttribute("orbA", LUMO_A);
221  _overlap_summary->setAttribute("orbB", LUMO_B);
222  //_overlap_summary->setAttribute("jAB", JAB);
223  _overlap_summary->setAttribute("eA", energyA);
224  _overlap_summary->setAttribute("eB", energyB);
225 
226  } else {
227 
228  for (int levelA = HOMO_A - _levA +1; levelA <= LUMO_A + _levA - 1; ++levelA ) {
229  for (int levelB = HOMO_B - _levB + 1; levelB <= LUMO_B + _levB -1 ; ++levelB ) {
230  double JAB = _overlap.getCouplingElement( levelA , levelB, &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy );
231  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
232  double energyA = _orbitalsA.getEnergy( levelA );
233  double energyB = _orbitalsB.getEnergy( levelB );
234  _overlap_summary->setAttribute("orbA", levelA);
235  _overlap_summary->setAttribute("orbB", levelB);
236  //_overlap_summary->setAttribute("jAB", JAB);
237  _overlap_summary->setAttribute("eA", energyA);
238  _overlap_summary->setAttribute("eB", energyB);
239  }
240  }
241  }
242 
243  votca::tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 1, "");
244 
245  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
246  ofs << *_job_output;
247  ofs.close();
248 
249  return true;
250 }
