147  /* Custom basis sets are written on a per-element basis to
148  * 'elementname'.gbs files, which are then included in the
149  * Gaussian input file using @'elementname'.gbs
150  */
151  void Gaussian::WriteBasisset(std::ofstream& com_file, std::vector<QMAtom*>& qmatoms) {
152 
153 
154  std::vector<std::string> UniqueElements= FindUniqueElements(qmatoms);
155  BasisSet bs;
156  bs.LoadBasisSet(_basisset_name);
157  CTP_LOG(ctp::logDEBUG, *_pLog) << "Loaded Basis Set " << _basisset_name << flush;
158 
159  for (const std::string& element_name:UniqueElements) {
160 
161  const Element& element = bs.getElement(element_name);
162  /* Write each basis set to a element_name.gbs file
163  * and include the gbs file in the com-file via Gaussian's @ function
164  */
165  std::ofstream el_file;
166  std::string el_file_name = _run_dir + "/" + element_name + ".gbs";
167 
168  el_file.open(el_file_name.c_str());
169  // element name, [possibly indeces of centers], zero to indicate the end
170  com_file << "@" << element_name << ".gbs" << endl;
171  el_file << element_name << " 0" << endl;
172  for (const Shell& shell:element) {
173  //gaussian can only use S,P,SP,D,F,G shells so we split them up if not SP
174  if (shell.getType() == "SP" || !shell.isCombined()) {
175  // shell type, number primitives, scale factor
176  el_file << shell.getType() << " " << shell.getSize() << " " << FortranFormat(shell.getScale()) << endl;
177  for (const GaussianPrimitive& gaussian:shell) {
178  el_file << FortranFormat(gaussian._decay);
179  for (const double& contraction:gaussian._contraction) {
180  if (contraction!= 0.0) {
181  el_file << " " << FortranFormat(contraction);
182  }
183  }
184  el_file << endl;
185  }
186  } else {
187  string type = shell.getType();
188  for (unsigned i = 0; i < type.size(); ++i) {
189  string subtype = string(type, i, 1);
190  el_file << subtype << " " << shell.getSize() << " " << FortranFormat(shell.getScale()) << endl;
191 
192  for (const GaussianPrimitive& gaussian:shell) {
193  el_file << FortranFormat(gaussian._decay);
194  el_file << " " << FortranFormat(gaussian._contraction[FindLmax(subtype)]);
195  }
196  el_file << endl;
197  }
198  }
199  }
200 
201  el_file << "****\n";
202  el_file.close();
203 
204  }
205 
206  com_file << endl;
207  return;
208  }

213  void Gaussian::WriteECP(std::ofstream& com_file, std::vector<QMAtom*>& qmatoms) {
214  std::vector<std::string> UniqueElements= FindUniqueElements(qmatoms);
215 
216  BasisSet ecp;
217  ecp.LoadPseudopotentialSet(_ecp_name);
218 
219  CTP_LOG(ctp::logDEBUG, *_pLog) << "Loaded Pseudopotentials " << _ecp_name << flush;
220 
221  for (const std::string& element_name:UniqueElements) {
222  try{
223  ecp.getElement(element_name);
224  }catch(std::runtime_error& error){
225  CTP_LOG(ctp::logDEBUG, *_pLog) << "No pseudopotential for " << element_name<<" available" << flush;
226  continue;
227  }
228  const Element& element = ecp.getElement(element_name);
229  // element name, [possibly indeces of centers], zero to indicate the end
230  com_file << element_name << " 0\n"
231  << _ecp_name << " "
232  << element.getLmax() << " " << element.getNcore() << endl;
233 
234  for (const Shell& shell:element) {
235  // shell type, number primitives, scale factor
236  com_file << shell.getType() << endl;
237  com_file << shell.getSize() << endl;
238 
239  for (const GaussianPrimitive& gaussian:shell) {
240  com_file << gaussian._power << " " << FortranFormat(gaussian._decay) << " " << FortranFormat(gaussian._contraction[0]) << endl;
241  }
242  }
243  }
244  com_file << endl;
245  return;
246  }

254  void Gaussian::WriteBackgroundCharges(std::ofstream& com_file) {
255 
256  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
257  for (std::shared_ptr<ctp::PolarSeg> seg:_PolarSegments) {
258  for (ctp::APolarSite* site:*seg) {
259 
260  string sitestring=boost::str(fmt % ((site->getPos().getX())*votca::tools::conv::nm2ang)
261  % (site->getPos().getY()*votca::tools::conv::nm2ang)
262  % (site->getPos().getZ()*votca::tools::conv::nm2ang)
263  % site->getQ00());
264  if (site->getQ00() != 0.0) com_file << sitestring << endl;
265 
266  if (site->getRank() > 0 || _with_polarization ) {
267 
268  std::vector< std::vector<double> > _split_multipoles = SplitMultipoles(site);
269  for (const auto& mpoles:_split_multipoles){
270  string multipole=boost::str( fmt % mpoles[0] % mpoles[1] % mpoles[2] % mpoles[3]);
271  com_file << multipole << endl;
272 
273  }
274  }
275  }
276  }
277  com_file << endl;
278  return;
279  }

281  /* An initial guess for the electron density can be provided by
282  * a set of molecular orbital coefficients in the input file,
283  * triggered by the 'guess=cards' keyword. This MUST be done in
284  * Fortran fixed format 5D15.8. The information about the guess
285  * itself is taken from a prepared orbitals object.
286  */
287  void Gaussian::WriteGuess(Orbitals& orbitals_guess, std::ofstream& com_file) {
288  ReorderMOsBack(orbitals_guess);
289  com_file << "(5D15.8)" << endl;
290  int level = 1;
291  int ncolumns = 5;
292  for (int i=0;i<orbitals_guess.MOCoefficients().cols();++i) {
293  com_file << setw(5) << level << endl;
294  Eigen::VectorXd mr = orbitals_guess.MOCoefficients().col(i);
295  int column = 1;
296  for (unsigned j = 0; j < mr.size(); ++j) {
297  com_file << FortranFormat(mr[j]);
298  if (column == ncolumns) {
299  com_file << std::endl;
300  column = 0;
301  }
302  column++;
303  }
304  level++;
305  if (column != 1) com_file << endl;
306  }
307  com_file << 0 << endl;
308  return;
309  }

384  bool Gaussian::WriteInputFile(Orbitals& orbitals) {
385 
386  std::string temp_suffix = "/id";
387  std::string scratch_dir_backup = _scratch_dir;
388 
389  std::ofstream com_file;
390  std::string _com_file_name_full = _run_dir + "/" + _input_file_name;
391  com_file.open(_com_file_name_full.c_str());
392 
393  // header
394  WriteHeader(com_file);
395 
396  std::vector< QMAtom* > qmatoms = orbitals.QMAtoms();
397 
398  WriteCoordinates(com_file, qmatoms);
399 
400  /* The order in which the following information has to appear
401  * in the Gaussian input file is different from version g03 to
402  * version g09. The newest version (g2016) is not supported.
403  */
404  if (_executable == "g03") {
405 
406  // if we need to write basis sets, do it now
407  if (_write_basis_set) WriteBasisset(com_file, qmatoms);
408 
409  // write ECPs
410  if (_write_pseudopotentials) WriteECP(com_file, qmatoms);
411 
412  // write the background charges
413  //if (_write_charges) WriteBackgroundCharges(_com_file, qmatoms);
414  if (_write_charges) WriteBackgroundCharges(com_file);
415 
416  // write inital guess
417  if (_write_guess){
418  WriteGuess(orbitals, com_file);
419  }
420 
421  } else if (_executable == "g09") {
422 
423  // write the background charges
424  //if (_write_charges) WriteBackgroundCharges(_com_file, qmatoms);
425  if (_write_charges) WriteBackgroundCharges(com_file);
426  // if we need to write basis sets, do it now
427  if (_write_basis_set) WriteBasisset(com_file, qmatoms);
428 
429  // write ECPs
430  if (_write_pseudopotentials) WriteECP(com_file, qmatoms);
431 
432  // write inital guess
433  if (_write_guess){
434  WriteGuess(orbitals, com_file);
435  }
436 
437  } else {
438  throw std::runtime_error("Gaussian executable unknown. Must be either g03 or g09.");
439  }
440 
441  // for Vxc AO matrix output only with pre-compiled G03
442  if (_output_Vxc) WriteVXCRunInputFile();
443 
444 
445  com_file << _vdWfooter << endl;
446 
447  com_file << endl;
448  com_file.close();
449  // and now generate a shell script to run both jobs
450  CTP_LOG(ctp::logDEBUG, *_pLog) << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
451 
452  _scratch_dir = scratch_dir_backup + temp_suffix;
453  WriteShellScript();
454  _scratch_dir = scratch_dir_backup;
455 
456  return true;
457  }

490  bool Gaussian::Run() {
491  CTP_LOG(ctp::logDEBUG, *_pLog) << "GAUSSIAN: running [" << _executable << " " << _input_file_name << "]" << flush;
492 
493  if (std::system(NULL)) {
494  // if scratch is provided, run the shell script;
495  // otherwise run gaussian directly and rely on global variables
496  std::string command;
497  if (_scratch_dir.size() != 0 || _output_Vxc) {
498  command = "cd " + _run_dir + "; tcsh " + _shell_file_name;
499  // _command = "cd " + _run_dir + "; mkdir -p " + _scratch_dir +"; " + _executable + " " + _input_file_name;
500  } else {
501  command = "cd " + _run_dir + "; mkdir -p $GAUSS_SCRDIR; " + _executable + " " + _input_file_name;
502  }
503  int check = std::system(command.c_str());
504  if (check == -1) {
505  CTP_LOG(ctp::logERROR, *_pLog) << _input_file_name << " failed to start" << flush;
506  return false;
507  }
508  if (CheckLogFile()) {
509  CTP_LOG(ctp::logDEBUG, *_pLog) << "GAUSSIAN: finished job" << flush;
510  return true;
511  } else {
512  CTP_LOG(ctp::logDEBUG, *_pLog) << "GAUSSIAN: job failed" << flush;
513  }
514  } else {
515  CTP_LOG(ctp::logERROR, *_pLog) << _input_file_name << " failed to start" << flush;
516  return false;
517  }
518  return true;
519  }

524  void Gaussian::CleanUp() {
525 
526  // cleaning up the generated files
527  if (_cleanup.size() != 0) {
528 
529  CTP_LOG(ctp::logDEBUG, *_pLog) << "Removing " << _cleanup << " files" << flush;
530  tools::Tokenizer tok_cleanup(_cleanup, ", ");
531  std::vector <std::string> cleanup_info;
532  tok_cleanup.ToVector(cleanup_info);
533  for (const std::string& substring:cleanup_info) {
534 
535  if (substring == "com") {
536  std::string file_name = _run_dir + "/" + _input_file_name;
537  remove(file_name.c_str());
538  if (_output_Vxc) {
539  std::string file_name = _run_dir + "/" + _input_vxc_file_name;
540  remove(file_name.c_str());
541  }
542  }
543 
544  if (substring == "sh") {
545  std::string file_name = _run_dir + "/" + _shell_file_name;
546  remove(file_name.c_str());
547  }
548 
549  if (substring == "log") {
550  std::string file_name = _run_dir + "/" + _log_file_name;
551  remove(file_name.c_str());
552  if (_output_Vxc) {
553  size_t lastdot = _log_file_name.find_last_of(".");
554  if (lastdot == std::string::npos) {
555  cerr << endl;
556  cerr << "Could not remove Vxc log file" << flush;
557  }
558  std::string file_name2 = file_name.substr(0, lastdot) + "-2.log";
559  remove(file_name2.c_str());
560  }
561  }
562 
563  if (substring == "chk") {
564  std::string file_name = _run_dir + "/" + _chk_file_name;
565  remove(file_name.c_str());
566  }
567 
568  if (substring == "fort.7") {
569  std::string file_name = _run_dir + "/" + substring;
570  remove(file_name.c_str());
571  if (_output_Vxc) {
572  std::string file_name = _run_dir + "/" + "fort.24";
573  remove(file_name.c_str());
574  }
575  }
576 
577  if (substring == "gbs" && _write_basis_set) {
578  std::vector<std::string> fileswithfileending;
579  boost::filesystem::recursive_directory_iterator fit(_run_dir);
580  boost::filesystem::recursive_directory_iterator endit;
581  while (fit != endit) {
582  if (boost::filesystem::is_regular_file(* fit) && fit->path().extension() == substring) fileswithfileending.push_back(fit->path().filename().string());
583  ++fit;
584  }
585  for (const auto filename : fileswithfileending) {
586  std::string file_name = _run_dir + "/" + filename;
587  remove(file_name.c_str());
588  }
589  }
590 
591  }
592  }
593  return;
594  }
