34BOOST_AUTO_TEST_CASE(esp_charges) {
35 libint2::initialize();
36 Orbitals orbitals;
37 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
38 "/espfit/molecule.xyz");
39 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) +
40 "/espfit/3-21G.xml");
41 orbitals.setNumberOfOccupiedLevels(5);
42
43 Eigen::MatrixXd MOs = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
44 std::string(XTP_TEST_DATA_FOLDER) + "/espfit/MOs.mm");
45 orbitals.MOs().eigenvectors() = MOs;
46 orbitals.MOs().eigenvalues() = Eigen::VectorXd::Ones(13);
47 QMState gs = QMState("n");
48 Logger log;
49 Espfit esp = Espfit(log);
50 esp.setUseSVD(1e-8);
51 StaticSegment result = esp.Fit2Density(orbitals, gs, "xcoarse");
52 Eigen::VectorXd pcharges = Eigen::VectorXd::Zero(orbitals.QMAtoms().size());
53 Index index = 0;
54 for (const auto& site : result) {
55 pcharges(index) = site.getCharge();
56 index++;
57 }
58 Eigen::VectorXd p_ref = Eigen::VectorXd::Zero(3);
59 p_ref << -0.827774, 0.413985, 0.41379;
60
61 bool check_esp_num = p_ref.isApprox(pcharges, 0.01);
62 if (!check_esp_num) {
63 std::cout << "ref" << std::endl;
64 std::cout << p_ref << std::endl;
65 std::cout << "calc" << std::endl;
66 std::cout << pcharges << std::endl;
67 }
68 BOOST_CHECK_EQUAL(check_esp_num, 1);
69
70 std::vector<std::pair<Index, Index> > pairconstraint;
71 std::pair<Index, Index> p1;
72 p1.first = 1;
73 p1.second = 2;
74 pairconstraint.push_back(p1);
75 Espfit esp2 = Espfit(log);
76 esp2.setUseSVD(1e-8);
77 esp2.setPairConstraint(pairconstraint);
78 StaticSegment result2 = esp2.Fit2Density(orbitals, gs, "xcoarse");
79 Eigen::VectorXd pcharges_equal = Eigen::VectorXd::Zero(result2.size());
80 index = 0;
81 for (const auto& site : result2) {
82 pcharges_equal(index) = site.getCharge();
83 index++;
84 }
85
86 bool check_p1 = (std::abs(pcharges_equal(1) - pcharges_equal(2)) < 1e-6);
87 BOOST_CHECK_EQUAL(check_p1, 1);
88
89 std::vector<QMFragment<double> > regionconstraint;
90
91 std::string indeces = "1:2";
92 QMFragment<double> reg = QMFragment<double>(0, indeces);
93 reg.value() = 1.0;
94 regionconstraint.push_back(reg);
95 Espfit esp3 = Espfit(log);
96 esp3.setRegionConstraint(regionconstraint);
97 esp3.setUseSVD(1e-8);
98 StaticSegment result3 = esp3.Fit2Density(orbitals, gs, "xcoarse");
99 Eigen::VectorXd pcharges_reg =
100 Eigen::VectorXd::Zero(orbitals.QMAtoms().size());
101 index = 0;
102
103 for (const auto& site : result3) {
104 pcharges_reg(index) = site.getCharge();
105 index++;
106 }
107
108 bool check_reg = (std::abs(pcharges_reg.segment(1, 2).sum() - 1.0) < 1e-6);
109 if (!check_reg) {
110 std::cout << "All charges " << pcharges_reg << std::endl;
111 std::cout << "Sum of charges 1,2,3 should equal 1:"
112 << pcharges_reg.segment(1, 2).sum() << std::endl;
113 }
114 BOOST_CHECK_EQUAL(check_reg, 1);
115
116 libint2::finalize();
117}
