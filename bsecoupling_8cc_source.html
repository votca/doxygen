32 void BSECoupling::Initialize(Property& options){
33 
34  #if (GWBSE_DOUBLE)
35  CTP_LOG(ctp::logDEBUG, *_pLog) << " Compiled with full double support" << flush;
36  #else
37  CTP_LOG(ctp::logDEBUG, *_pLog) << " Compiled with float/double mixture (standard)" << flush;
38  #endif
39 
40  std::string key = Identify();
41  _doSinglets=false;
42  _doTriplets=false;
43  _output_perturbation=false;
44 
45 
46  _openmp_threads = 0;
47 
48  if ( options.exists( key + ".openmp") ) {
49  _openmp_threads = options.get(key + ".openmp").as<int> ();
50  }
51 
52  string spintype = options.get(key + ".spin").as<string> ();
53  if(spintype=="all"){
54  _doSinglets=true;
55  _doTriplets=true;
56  }
57  else if(spintype=="triplet"){
58  _doTriplets=true;
59  }
60  else if(spintype=="singlet"){
61  _doSinglets=true;
62  }
63  else{
64  throw std::runtime_error((boost::format("Choice % for type not known. Available singlet,triplet,all") % spintype).str());
65  }
66 
67 
68  if ( options.exists( key + ".algorithm") ) {
69  string algorithm = options.get(key + ".algorithm").as<string> ();
70  if(algorithm=="perturbation"){
71  _output_perturbation=true;
72  }
73  }
74 
75  _levA = options.get(key + ".moleculeA.states").as<int> ();
76  _levB = options.get(key + ".moleculeB.states").as<int> ();
77  _occA = options.get(key + ".moleculeA.occLevels").as<int> ();
78  _occB = options.get(key + ".moleculeB.occLevels").as<int> ();
79  _unoccA = options.get(key + ".moleculeA.unoccLevels").as<int> ();
80  _unoccB = options.get(key + ".moleculeB.unoccLevels").as<int> ();
81 
82 }

84 void BSECoupling::WriteToProperty(const Orbitals& orbitalsA, const Orbitals& orbitalsB,
85  Property& summary, const QMState& stateA, const QMState& stateB){
86  Property &coupling_summary = summary.add("coupling","");
87  double energyA =0;
88  double energyB =0;
89  double JAB_pert=0;
90  double JAB_diag=0;
91  if(stateA.Type()==QMStateType::Singlet){
92  energyA=orbitalsA.BSESingletEnergies()(stateA.Index())*conv::hrt2ev;
93  energyB=orbitalsB.BSESingletEnergies()(stateB.Index())*conv::hrt2ev;
94  JAB_pert=getSingletCouplingElement(stateA.Index(),stateB.Index(),0);
95  JAB_diag=getSingletCouplingElement(stateA.Index(),stateB.Index(),1);
96  }else if(stateA.Type()==QMStateType::Triplet){
97  energyA=orbitalsA.BSETripletEnergies()(stateA.Index())*conv::hrt2ev;
98  energyB=orbitalsB.BSETripletEnergies()(stateB.Index())*conv::hrt2ev;
99  JAB_pert=getTripletCouplingElement(stateA.Index(),stateB.Index(),0);
100  JAB_diag=getTripletCouplingElement(stateA.Index(),stateB.Index(),1);
101  }
102  coupling_summary.setAttribute("stateA", stateA.ToString());
103  coupling_summary.setAttribute("stateB", stateB.ToString());
104  coupling_summary.setAttribute("eA", (format("%1$1.6e") % energyA).str());
105  coupling_summary.setAttribute("eB", (format("%1$1.6e") % energyB).str());
106  coupling_summary.setAttribute("j_pert", (format("%1$1.6e") % JAB_pert).str());
107  coupling_summary.setAttribute("j_diag", (format("%1$1.6e") % JAB_diag).str());
108 }

110 void BSECoupling::Addoutput(Property &type_summary,const Orbitals& orbitalsA,
111  const Orbitals& orbitalsB){
112  tools::Property& bsecoupling= type_summary.add(Identify(),"");
113  string algorithm="j_diag";
114  if (_output_perturbation){
115  algorithm="j_pert";
116  }
117  if (_doSinglets){
118  QMStateType singlet=QMStateType(QMStateType::Singlet);
119  Property &singlet_summary = bsecoupling.add(singlet.ToLongString(),"");
120  singlet_summary.setAttribute("algorithm",algorithm);
121  for (int stateA = 0; stateA < _levA ; ++stateA ) {
122  QMState qmstateA=QMState(singlet,stateA,false);
123  for (int stateB = 0; stateB <_levB ; ++stateB ) {
124  QMState qmstateB=QMState(singlet,stateB,false);
125  WriteToProperty(orbitalsA, orbitalsB, singlet_summary, qmstateA, qmstateB);
126  }
127  }
128  }
129 
130  if ( _doTriplets){
131  QMStateType triplet=QMStateType(QMStateType::Triplet);
132  Property &triplet_summary = bsecoupling.add(triplet.ToLongString(),"");
133  triplet_summary.setAttribute("algorithm",algorithm);
134  for (int stateA = 0; stateA < _levA ; ++stateA ) {
135  QMState qmstateA=QMState(triplet,stateA,false);
136  for (int stateB = 0; stateB <_levB ; ++stateB ) {
137  QMState qmstateB=QMState(triplet,stateB,false);
138  WriteToProperty(orbitalsA, orbitalsB, triplet_summary, qmstateA, qmstateB);
139  }
140  }
141  }
142 }

165 void BSECoupling::CalculateCouplings(const Orbitals& orbitalsA, const Orbitals& orbitalsB, Orbitals& orbitalsAB) {
166  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculating exciton couplings" << flush;
167  #ifdef _OPENMP
168 
169  if ( _openmp_threads > 0 ) omp_set_num_threads(_openmp_threads);
170  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using "<< omp_get_max_threads()<<" threads" << flush;
171  #endif
172 
173  CheckAtomCoordinates(orbitalsA, orbitalsB, orbitalsAB);
174 
175  // constructing the direct product orbA x orbB
176  int basisA = orbitalsA.getBasisSetSize();
177  int basisB = orbitalsB.getBasisSetSize();
178 
179  if ( ( basisA == 0 ) || ( basisB == 0 ) ) {
180  throw std::runtime_error("Basis set size is not stored in monomers");
181  }
182 
183  // number of levels stored in monomers
184  int levelsA = orbitalsA.getNumberOfLevels();
185  int levelsB = orbitalsB.getNumberOfLevels();
186 
187  // get exciton information of molecule A
188  int _bseA_cmax = orbitalsA.getBSEcmax();
189  int _bseA_cmin = orbitalsA.getBSEcmin();
190  int _bseA_vmax = orbitalsA.getBSEvmax();
191  int _bseA_vmin = orbitalsA.getBSEvmin();
192  int _bseA_vtotal = _bseA_vmax - _bseA_vmin +1 ;
193  int _bseA_ctotal = _bseA_cmax - _bseA_cmin +1 ;
194  int _bseA_size = _bseA_vtotal * _bseA_ctotal;
195  int _bseA_singlet_exc = orbitalsA.BSESingletCoefficients().cols();
196  int _bseA_triplet_exc = orbitalsA.BSETripletCoefficients().cols();
197 
198  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule A has " << _bseA_singlet_exc << " singlet excitons with dimension " << _bseA_size << flush;
199  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule A has " << _bseA_triplet_exc << " triplet excitons with dimension " << _bseA_size << flush;
200 
201  // now, two storage assignment matrices for two-particle functions
202  Eigen::MatrixXi combA;
203  combA.resize(_bseA_size,2);
204  int cnt = 0;
205  for ( int _v = 0; _v < _bseA_vtotal; _v++){
206  for ( int _c = 0; _c < _bseA_ctotal; _c++){
207  combA(cnt,0) = _v;
208  combA(cnt,1) = _bseA_vtotal + _c;
209  cnt++;
210  }
211  }
212 
213  // get exciton information of molecule B
214  int _bseB_cmax = orbitalsB.getBSEcmax();
215  int _bseB_cmin = orbitalsB.getBSEcmin();
216  int _bseB_vmax = orbitalsB.getBSEvmax();
217  int _bseB_vmin = orbitalsB.getBSEvmin();
218  int _bseB_vtotal = _bseB_vmax - _bseB_vmin +1 ;
219  int _bseB_ctotal = _bseB_cmax - _bseB_cmin +1 ;
220  int _bseB_size = _bseB_vtotal * _bseB_ctotal;
221  int _bseB_singlet_exc = orbitalsB.BSESingletCoefficients().cols();
222  int _bseB_triplet_exc = orbitalsB.BSETripletCoefficients().cols();
223 
224  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule B has " << _bseB_singlet_exc << " singlet excitons with dimension " << _bseB_size << flush;
225  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule B has " << _bseB_triplet_exc << " triplet excitons with dimension " << _bseB_size << flush;
226 
227  // now, two storage assignment matrices for two-particle functions
228  Eigen::MatrixXi combB;
229  combB.resize(_bseB_size,2);
230  cnt = 0;
231  for ( int _v = 0; _v < _bseB_vtotal; _v++){
232  for ( int _c = 0; _c < _bseB_ctotal; _c++){
233  combB(cnt,0) = _bseA_vtotal + _bseA_ctotal + _v;
234  combB(cnt,1) = _bseA_vtotal + _bseA_ctotal + _bseB_vtotal + _c;
235  cnt++;
236  }
237  }
238 
239  if(_levA>_bseA_singlet_exc){
240  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of excitons you want is greater than stored for molecule A. Setting to max number available" << flush;
241  _levA=_bseA_singlet_exc;
242  }
243  if(_levB>_bseB_singlet_exc){
244  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of excitons you want is greater than stored for molecule B. Setting to max number available" << flush;
245  _levB=_bseB_singlet_exc;
246  }
247 
248  if(_levA>_bseA_singlet_exc){
249  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of Frenkel states you want is greater than stored for molecule A. Setting to max number available" << flush;
250  _levA=_bseA_singlet_exc;
251  }
252  if(_levB>_bseB_singlet_exc){
253  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of Frenkel states you want is greater than stored for molecule B. Setting to max number available" << flush;
254  _levB=_bseB_singlet_exc;
255  }
256 
257  if(_unoccA>_bseA_ctotal){
258  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of occupied orbitals in molecule A for CT creation exceeds number of KS-orbitals in BSE" << flush;
259  _unoccA=_bseA_ctotal;
260  }
261  else if (_unoccA<0){
262  _unoccA=_bseA_ctotal;
263  }
264  if(_unoccB>_bseB_ctotal){
265  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of occupied orbitals in molecule B for CT creation exceeds number of KS-orbitals in BSE" << flush;
266  _unoccB=_bseB_ctotal;
267  }
268  else if (_unoccB<0){
269  _unoccB=_bseB_ctotal;
270  }
271 
272  if(_occA>_bseA_vtotal){
273  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of unoccupied orbitals in molecule A for CT creation exceeds number of KS-orbitals in BSE" << flush;
274  _occA=_bseA_vtotal;
275  }
276  else if (_occA<0){
277  _occA=_bseA_vtotal;
278  }
279  if(_occB>_bseB_vtotal){
280  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of unoccupied orbitals in molecule B for CT creation exceeds number of KS-orbitals in BSE" << flush;
281  _occB=_bseB_vtotal;
282  }else if (_occB<0){
283  _occB=_bseB_vtotal;
284  }
285 
286  // get exciton information of pair AB
287  int _bseAB_cmax = orbitalsAB.getBSEcmax();
288  int _bseAB_cmin = orbitalsAB.getBSEcmin();
289  int _bseAB_vmax = orbitalsAB.getBSEvmax();
290  int _bseAB_vmin = orbitalsAB.getBSEvmin();
291  int _bseAB_vtotal = _bseAB_vmax - _bseAB_vmin +1 ;
292  int _bseAB_ctotal = _bseAB_cmax - _bseAB_cmin +1 ;
293  int _bseAB_size = _bseAB_vtotal * _bseAB_ctotal;
294  // check if electron-hole interaction matrices are stored
295  if ( ! orbitalsAB.hasEHinteraction_triplet() && _doTriplets){
296  throw std::runtime_error( "BSE EH for triplets not stored " );
297  }
298  if ( ! orbitalsAB.hasEHinteraction_singlet() && _doSinglets){
299  throw std::runtime_error( "BSE EH for singlets not stored " );
300  }
301  const MatrixXfd& eh_t = orbitalsAB.eh_t();
302  const MatrixXfd& eh_s = orbitalsAB.eh_s();
303  if(_doTriplets){
304  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " dimer AB has BSE EH interaction triplet with dimension " << eh_t.rows() << " x " << eh_t.cols() << flush;
305  }
306  if(_doSinglets){
307  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " dimer AB has BSE EH interaction singlet with dimension " << eh_s.rows() << " x " << eh_s.cols() << flush;
308  }
309  // now, two storage assignment matrices for two-particle functions
310  Eigen::MatrixXi combAB;
311  combAB.resize(_bseAB_size,2);
312  cnt = 0;
313  for ( int _v = 0; _v < _bseAB_vtotal; _v++){
314  for ( int _c = 0; _c < _bseAB_ctotal; _c++){
315  combAB(cnt,0) = _bseAB_vmin + _v;
316  combAB(cnt,1) = _bseAB_vmin + _bseAB_vtotal + _c;
317  cnt++;
318  }
319  }
320 
321  // DFT levels of monomers can be reduced to those used in BSE
322  levelsA = _bseA_vtotal + _bseA_ctotal;
323  levelsB = _bseB_vtotal + _bseB_ctotal;
324  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " levels used in BSE of molA: " << _bseA_vmin << " to " << _bseA_cmax << " total: " << _bseA_vtotal + _bseA_ctotal << flush;
325  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " levels used in BSE of molB: " << _bseB_vmin << " to " << _bseB_cmax << " total: " << _bseB_vtotal + _bseB_ctotal << flush;
326 
327  if ( ( levelsA == 0 ) || (levelsB == 0) ) {
328  throw std::runtime_error("No information about number of occupied/unoccupied levels is stored" );
329  }
330 
331  // | Orbitals_A 0 | | Overlap_A |
332  // | 0 Orbitals_B |.T X | Overlap_B | X ( Orbitals_AB )
333 
334  Eigen::MatrixXd psi_AxB =Eigen::MatrixXd::Zero( levelsA + levelsB, basisA + basisB );
335  // constructing merged orbitals
336  psi_AxB.block(0,0,levelsA , basisA) = orbitalsA.MOCoefficients().block(_bseA_vmin,0, _bseA_cmax+1-_bseA_vmin, basisA );
337  psi_AxB.block(levelsA, basisA,levelsB,basisB) =orbitalsB.MOCoefficients().block(_bseB_vmin,0,_bseB_cmax+1-_bseA_vmin,basisB);
338 
339  // psi_AxB * S_AB * psi_AB
340  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " projecting monomer onto dimer orbitals" << flush;
341 
342  Eigen::MatrixXd overlapAB;
343  if ( orbitalsAB.hasAOOverlap() ) {
344  CTP_LOG(ctp::logDEBUG,*_pLog) << "Reading overlap matrix from orbitals" << flush;
345  overlapAB= orbitalsAB.AOOverlap();
346  }else{
347  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating overlap matrix for basisset: "<< orbitalsAB.getDFTbasis()<< flush;
348  overlapAB=CalculateOverlapMatrix(orbitalsAB);
349  }
350 
351  Eigen::MatrixXd psi_AxB_dimer_basis = psi_AxB.transpose()*overlapAB*orbitalsAB.MOCoefficients();
352  overlapAB.resize(0,0);
353  int LevelsA = levelsA;
354  for (int i=0;i<psi_AxB_dimer_basis.rows();i++){
355  double mag=psi_AxB_dimer_basis.row(i).squaredNorm();
356  if (mag<0.95){
357  int monomer = 0;
358  int level = 0;
359  if ( i < LevelsA ) {
360  monomer = 1;
361  level = _bseA_vmin + i;
362  } else {
363  monomer = 2;
364  level = _bseB_vmin + i -levelsA;
365  }
366  CTP_LOG(ctp::logERROR,*_pLog) << "\nERROR: " << i << " Projection of orbital "
367  << level << " of monomer " << monomer << " on dimer is insufficient,mag="
368  <<mag<<" maybe the orbital order is screwed up, otherwise increase dimer basis.\n"<<flush;
369  }
370  }
371 
372  //notation AB is CT states with A+B-, BA is the counterpart
373  //Setting up CT-states:
374  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setting up CT-states" << flush;
375  //Number of A+B- states
376  int noAB=_occA*_unoccB;
377  //Number of A-B+ states
378  int noBA=_unoccA*_occB;
379 
380  Eigen::MatrixXi comb_CTAB=Eigen::MatrixXi::Zero(noAB,2);
381  cnt = 0;
382  // iterate A over occupied, B over unoccupied
383  int v_start=_bseA_vtotal-_occA;
384  for ( int _v = v_start; _v < _bseA_vtotal; _v++){
385  for ( int _c = 0; _c <_unoccB; _c++){
386  comb_CTAB(cnt,0) =_v;
387  comb_CTAB(cnt,1) = _bseA_vtotal+_bseA_ctotal+_bseB_vtotal + _c;
388 
389  cnt++;
390  }
391  }
392  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() <<" "<<noAB <<" CT states A+B- created" << flush;
393 
394  Eigen::MatrixXi comb_CTBA=Eigen::MatrixXi::Zero(noBA,2);
395  cnt = 0;
396  // iterate A over unoccupied, B over occupied
397  v_start=_bseB_vtotal-_occB;
398  for ( int _v = v_start; _v < _bseB_vtotal; _v++){
399  for ( int _c = 0; _c <_unoccA; _c++){
400  comb_CTBA(cnt,0) =_bseA_vtotal+_bseA_ctotal+_v;
401  comb_CTBA(cnt,1) = _bseA_vtotal+ _c;
402 
403  cnt++;
404  }
405  }
406  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() <<" "<<noBA <<" CT states B+A- created" << flush;
407 
408  // these 4 matrixes, matrix(i,j) contains the j-th dimer MO component of the i-th excitation
409 
410  ctAB.resize(noAB,_bseAB_size);
411  #pragma omp parallel for
412  for ( int _i_CT = 0 ; _i_CT < noAB ; _i_CT++){
413  for ( int _i_bseAB = 0 ; _i_bseAB < _bseAB_size ; _i_bseAB++){
414  ctAB(_i_CT,_i_bseAB)=psi_AxB_dimer_basis( comb_CTAB(_i_CT,0), combAB( _i_bseAB,0) ) * psi_AxB_dimer_basis( comb_CTAB(_i_CT,1), combAB( _i_bseAB,1) );
415  }
416  }
417 
418  ctBA.resize(noBA,_bseAB_size);
419  #pragma omp parallel for
420  for ( int _i_CT = 0 ; _i_CT < noBA ; _i_CT++){
421  for ( int _i_bseAB = 0 ; _i_bseAB < _bseAB_size ; _i_bseAB++){
422  ctBA(_i_CT,_i_bseAB)=psi_AxB_dimer_basis( comb_CTBA(_i_CT,0), combAB( _i_bseAB,0) ) * psi_AxB_dimer_basis( comb_CTBA(_i_CT,1), combAB( _i_bseAB,1) );
423  }
424  }
425 
426  _kap.resize(_bseA_size,_bseAB_size);
427  #pragma omp parallel for
428  for ( int _i_bseA = 0 ; _i_bseA < _bseA_size ; _i_bseA++){
429  for ( int _i_bseAB = 0 ; _i_bseAB < _bseAB_size ; _i_bseAB++){
430  _kap(_i_bseA,_i_bseAB) = psi_AxB_dimer_basis( combA(_i_bseA,0), combAB( _i_bseAB,0) ) * psi_AxB_dimer_basis( combA(_i_bseA,1), combAB( _i_bseAB,1) );
431 
432  }
433  }
434 
435  _kbp.resize(_bseB_size,_bseAB_size);
436  #pragma omp parallel for
437  for ( int _i_bseB = 0 ; _i_bseB < _bseB_size ; _i_bseB++){
438  for ( int _i_bseAB = 0 ; _i_bseAB < _bseAB_size ; _i_bseAB++){
439  _kbp(_i_bseB,_i_bseAB) = psi_AxB_dimer_basis( combB(_i_bseB,0), combAB( _i_bseAB,0) ) * psi_AxB_dimer_basis( combB(_i_bseB,1), combAB( _i_bseAB,1) );
440  }
441  }
442 
443  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " construct projection of product functions " << flush;
444 
445  psi_AxB_dimer_basis.resize(0,0);
446  combAB.resize(0,0);
447  combA.resize(0,0);
448  combB.resize(0,0);
449  // now the different spin types
450  if (_doSinglets) {
451  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Evaluating singlets" << flush;
452  Eigen::MatrixXd Hamiltonian_AB = eh_s.cast<double>();
453  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setup Hamiltonian" << flush;
454  const Eigen::MatrixXd bseA_T = orbitalsA.BSESingletCoefficients().block(0,0,orbitalsA.BSESingletCoefficients().rows(),_levA).transpose().cast<double>();

478 std::vector< Eigen::MatrixXd > BSECoupling::ProjectExcitons(const Eigen::MatrixXd& bseA_T, const Eigen::MatrixXd& bseB_T,
479  Eigen::MatrixXd& H){
480 
481  // get projection of monomer excitons on dimer product functions
482  Eigen::MatrixXd _proj_excA = bseA_T* _kap;
483  Eigen::MatrixXd _proj_excB = bseB_T* _kbp;
484 
485  _bse_exc=_levA+_levB;
486  int ctABsize=ctAB.rows();
487  int ctBAsize=ctBA.rows();
488  _ct=ctABsize+ctBAsize;
489  int nobasisfunc=H.rows();
490 
491  Eigen::MatrixXd fe_states=Eigen::MatrixXd::Zero(_bse_exc,nobasisfunc);
492  fe_states.block(0,0, _levA, nobasisfunc )=_proj_excA;
493  fe_states.block( _levA,0,_levB,nobasisfunc )=_proj_excB;
494 
495  Eigen::MatrixXd ct_states=Eigen::MatrixXd::Zero(_ct,nobasisfunc);
496 
497  if (_ct > 0) {
498  //orthogonalize ct-states with respect to the FE states.
499  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Orthogonalizing CT-states with respect to FE-states" << flush;
500 
501  if (ctABsize > 0) {
502  ct_states.block(0, 0, ctABsize, nobasisfunc) = ctAB;
503  }
504  if (ctBAsize > 0) {
505  ct_states.block(ctABsize, 0, ctBAsize, nobasisfunc) = ctBA;
506  }
507 
508  //orthogonalize ct-states with respect to FE states
509  Eigen::MatrixXd correction = ct_states * fe_states.transpose() * fe_states;
510  ct_states -= correction;
511  correction.resize(0, 0);
512  //normalize
513  Eigen::VectorXd norm=ct_states.rowwise().norm();
514  for (int i = 0; i < _ct; i++) {
515  ct_states.row(i) /= norm(i);
516  }
517  int minstateindex=0;
518  double minnorm=norm.minCoeff(&minstateindex);
519  if(minnorm<0.95){
520  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " WARNING: CT-state " << minstateindex << " norm is only " << minnorm << flush;
521  }
522  }
523  Eigen::MatrixXd projection(_bse_exc+_ct,nobasisfunc);
524  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " merging projections into one vector " << flush;
525  projection.block(0 ,0, _bse_exc,nobasisfunc)=fe_states;
526 
527  if(_ct>0){
528  projection.block(_bse_exc,0,_ct ,nobasisfunc )=ct_states;
529  }
530  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setting up coupling matrix size "<< _bse_exc +_ct<<"x"<<_bse_exc +_ct << flush;
531  // matrix _J
532  // E_A J_AB J_A_ABCT J_A_BACT
533  // J_BA E_B J_B_ABCT J_B_BACT
534  // J_ABCT_A J_ABCT_B E_ABCT J_ABCT_BACT
535  // J_BACT_A J_BACT_B J_BACT_ABCT E_BACT
536 
537  // this only works for hermitian/symmetric H so only in TDA
538 
539  Eigen::MatrixXd J_dimer=projection*H*projection.transpose();
540  H.resize(0,0);
541 
542  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setting up overlap matrix size "<< _bse_exc +_ct<<"x"<<_bse_exc +_ct << flush;
543  Eigen::MatrixXd S_dimer=projection*projection.transpose();
544 
545  projection.resize(0,0);
546  if(tools::globals::verbose && _bse_exc+_ct<100){
547  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------"<<flush;
548  CTP_LOG(ctp::logDEBUG, *_pLog) << "_J_dimer[Ryd]"<<flush;
549 
550  CTP_LOG(ctp::logDEBUG, *_pLog) << J_dimer<<flush;
551  CTP_LOG(ctp::logDEBUG, *_pLog) << "_S_dimer"<<flush;
552 
553  CTP_LOG(ctp::logDEBUG, *_pLog) << S_dimer<<flush;
554  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------"<<flush;
555  }
556 
557  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(S_dimer);
558  Eigen::MatrixXd Sm1=es.operatorInverseSqrt();
559  J_dimer=Sm1*J_dimer*Sm1;
560 
561 
562  if(tools::globals::verbose && _bse_exc+_ct<100){
563  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------"<<flush;
564  CTP_LOG(ctp::logDEBUG, *_pLog) << "_J_ortho[Ryd]"<<flush;
565  CTP_LOG(ctp::logDEBUG, *_pLog) << J_dimer<<flush;
566  CTP_LOG(ctp::logDEBUG, *_pLog) << "_S-1/2"<<flush;
567  CTP_LOG(ctp::logDEBUG, *_pLog) << Sm1<<flush;
568  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------"<<flush;
569  }
570  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() <<
571  " Smallest value of dimer overlapmatrix is "<<es.eigenvalues()(0)<< flush;
572 
573  std::vector< Eigen::MatrixXd >J;
574 
575  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Running Perturbation algorithm"<< flush;
576  J.push_back( Perturbation(J_dimer));
577  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Running Projection algorithm"<< flush;
578  J.push_back( Fulldiag(J_dimer));
579 
580

583  CTP_LOG(ctp::logDEBUG, *_pLog) << "Jeff_pert[Hrt]"<<flush;
584  CTP_LOG(ctp::logDEBUG, *_pLog) << J[0]<<flush;
585  CTP_LOG(ctp::logDEBUG, *_pLog) << "Jeff_diag[Hrt]"<<flush;
586  CTP_LOG(ctp::logDEBUG, *_pLog) << J[1]<<flush;
587  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------"<<flush;
588  }

593 Eigen::MatrixXd BSECoupling::Perturbation(const Eigen::MatrixXd& J_dimer){
594 
595  Eigen::MatrixXd Jmatrix =Eigen::MatrixXd::Zero(_bse_exc, _bse_exc);
596  bool diag_ct = true;
597  Eigen::MatrixXd J_result=J_dimer;
598  if (_ct > 0 && diag_ct) {
599  Eigen::MatrixXd transformation = Eigen::MatrixXd::Identity(_bse_exc + _ct, _bse_exc + _ct);
600  Eigen::MatrixXd Ct = J_dimer.block(_bse_exc,_bse_exc,_ct, _ct);
601  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(Ct);
602  transformation.block(_bse_exc, _bse_exc ,_ct,_ct) = es.eigenvectors();
603  Ct.resize(0, 0);
604 
605  if (tools::globals::verbose) {
606  CTP_LOG(ctp::logDEBUG, *_pLog) << "FE state hamiltonian" << flush;
607  CTP_LOG(ctp::logDEBUG, *_pLog) << J_dimer.block(0,0, _bse_exc,_bse_exc) << flush;
608  if (_ct > 0) {
609  CTP_LOG(ctp::logDEBUG, *_pLog) << "eigenvalues of CT states" << flush;
610  CTP_LOG(ctp::logDEBUG, *_pLog) << es.eigenvalues() << flush;
611  }
612  }
613 
614  J_result = transformation.transpose()*J_dimer*transformation;
615  if (tools::globals::verbose && _bse_exc + _ct < 100) {
616  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
617  CTP_LOG(ctp::logDEBUG, *_pLog) << "_J_ortho[Hrt] CT-state diag" << flush;
618  CTP_LOG(ctp::logDEBUG, *_pLog) << J_result << flush;
619  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
620  }
621  }
622  for (int stateA = 0; stateA < _levA; stateA++) {
623  double Ea = J_result(stateA, stateA);
624  for (int stateB = 0; stateB < _levB; stateB++) {
625  int stateBd = stateB + _levA;
626  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculating coupling between exciton A"
627  << stateA + 1 << " and exciton B" << stateB + 1 << flush;
628  double J = J_result(stateA, stateBd);
629 
630  double Eb = J_result(stateBd, stateBd);
631  for (int k = _bse_exc; k < (_bse_exc + _ct); k++) {
632  double Eab = J_result(k, k);
633  if (std::abs(Eab - Ea) < 0.001) {
634  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << "Energydifference between state A "
635  << stateA + 1 << "and CT state " << k + 1 << " is " << Eab - Ea << "[Hrt]" << flush;
636  }
637  if (std::abs(Eab - Eb) < 0.001) {
638  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << "Energydifference between state B "
639  << stateB + 1 << "and CT state " << k + 1 << " is " << Eab - Eb << "[Hrt]" << flush;
640 
641  }
642  J += 0.5 * J_result(k, stateA) * J_result(k, stateBd)*(1 / (Ea - Eab) + 1 / (Eb - Eab)); // Have no clue why 0.5
643  }
644  Jmatrix(stateA, stateBd) = J;
645  Jmatrix(stateBd, stateA) = J;
646 
647 
648  }
649  }
650  return Jmatrix;

654 Eigen::MatrixXd BSECoupling::Fulldiag(const Eigen::MatrixXd& J_dimer){
655  Eigen::MatrixXd Jmat = Eigen::MatrixXd::Zero(_bse_exc, _bse_exc);
656 
657  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(J_dimer);
658  if (tools::globals::verbose && _bse_exc + _ct < 10) {
659  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
660  CTP_LOG(ctp::logDEBUG, *_pLog) << "Eigenvectors of J" << flush;
661  CTP_LOG(ctp::logDEBUG, *_pLog) << es.eigenvectors() << flush;
662  CTP_LOG(ctp::logDEBUG, *_pLog) << "J_eigenvalues[Hrt]" << flush;
663  CTP_LOG(ctp::logDEBUG, *_pLog) << es.eigenvalues() << flush;
664  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
665  }
666  //Calculate projection on subspace for every pair of excitons separately
667  for (int stateA = 0; stateA < _levA; stateA++) {
668  for (int stateB = 0; stateB < _levB; stateB++) {
669  int stateBd = stateB + _levA;
670  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculating coupling between exciton A" << stateA + 1 << " and exciton B" << stateB + 1 << flush;
671  std::vector<int> index;
672  std::vector<int> signvec;
673  for (int i = 0; i < _bse_exc + _ct; i++) {
674  if (i == int(stateA) || i == int(stateBd)) {
675  double close = 0.0;
676  int ind = 0;
677  int sign = 0;
678  //row
679  for (int j = 0; j < _bse_exc + _ct; j++) {
680  bool check = true;
681  // if index i is already in index
682  // should not happen but if one vector was similar to two others.
683  for (unsigned l = 0; l < index.size(); l++) {
684  if (j == index[l]) {
685  check = false;
686  break;
687  }
688  }
689  if (check && std::abs(es.eigenvalues()(i, j)) > close) {
690  ind = j;
691  close = std::abs(es.eigenvalues()(i, j));
692  if (es.eigenvalues()(i, j) >= 0) {
693  sign = 1;
694  } else {
695  sign = -1;
696  }
697  }
698  }
699  index.push_back(ind);
700  signvec.push_back(sign);
701  }
702  }
703 
704  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Order is: [Initial state n->nth eigenvalue]" << flush;
705  CTP_LOG(ctp::logDEBUG, *_pLog) << " A" << stateA + 1 << ":" << stateA + 1 << "->" << index[0] + 1 << " ";
706  CTP_LOG(ctp::logDEBUG, *_pLog) << " B" << stateB + 1 << ":" << stateBd + 1 << "->" << index[1] + 1 << " " << flush;
707 
708  //setting up transformation matrix Tmat and diagonal matrix Emat for the eigenvalues;
709  Eigen::MatrixXd Emat = Eigen::MatrixXd::Zero(2, 2);
710  Eigen::MatrixXd Tmat = Eigen::MatrixXd::Zero(2, 2);
711  //find the eigenvectors which are most similar to the initial states
712  //row
713  for (int i = 0; i < 2; i++) {
714  int k = index[i];
715  double sign = signvec[i];
716  double normr = 1 / std::sqrt(es.eigenvectors()(stateA, k) * es.eigenvectors()(stateA, k) + es.eigenvectors()(stateBd, k) * es.eigenvectors()(stateBd, k));
717  Tmat(0, i) = sign * es.eigenvectors()(stateA, k) * normr;
718  Tmat(1, i) = sign * es.eigenvectors()(stateBd, k) * normr;
719  Emat(i, i) = es.eigenvectors()(k);
720  }
721 
722  if ((Tmat(1, 1) * Tmat(0, 0) - Tmat(1, 0) * Tmat(0, 1)) < 0) {
723  CTP_LOG(ctp::logDEBUG, *_pLog) << " Reduced state matrix is not in a right handed basis, multiplying second eigenvector by -1 " << flush;
724  Tmat(0, 1) = -Tmat(0, 1);
725  Tmat(1, 1) = -Tmat(1, 1);
726  }
727 
728  if (tools::globals::verbose) {
729  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
730  CTP_LOG(ctp::logDEBUG, *_pLog) << "_T" << flush;
731  CTP_LOG(ctp::logDEBUG, *_pLog) << Tmat << flush;
732 
733  }
734 
735  Eigen::MatrixXd S_small = Tmat*Tmat.transpose();
736  if (tools::globals::verbose) {
737 
738  CTP_LOG(ctp::logDEBUG, *_pLog) << "S_small" << flush;
739  CTP_LOG(ctp::logDEBUG, *_pLog) << S_small << flush;
740 
741  }
742  //orthogonalize that matrix
743 
744  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> ss(S_small);
745  Eigen::MatrixXd sm1=ss.operatorInverseSqrt();
746  Emat=sm1*Emat*sm1;
747 
748  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Smallest value of dimer overlapmatrix is " << ss.eigenvalues()(0) << flush;
749  if (tools::globals::verbose) {
750 
751  CTP_LOG(ctp::logDEBUG, *_pLog) << "S-1/2" << flush;
752  CTP_LOG(ctp::logDEBUG, *_pLog) << sm1 << flush;
753  CTP_LOG(ctp::logDEBUG, *_pLog) << "E_ortho" << flush;
754  CTP_LOG(ctp::logDEBUG, *_pLog) << Emat << flush;
755  }
756  Tmat = Tmat*sm1;
757 
758  if (tools::globals::verbose) {
759 
760  CTP_LOG(ctp::logDEBUG, *_pLog) << "T_ortho" << flush;
761  CTP_LOG(ctp::logDEBUG, *_pLog) << Tmat << flush;
762  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
763  }
764 
765 
766  Eigen::MatrixXd J_small = Tmat*Emat*Tmat.transpose();
767  if (tools::globals::verbose) {
768  CTP_LOG(ctp::logDEBUG, *_pLog) << "T_ortho*E_ortho*T_ortho^T" << flush;
769  CTP_LOG(ctp::logDEBUG, *_pLog) << J_small << flush;
770  }
771 
772  Jmat(stateA, stateBd) = J_small(0, 1);
773  Jmat(stateBd, stateA) = J_small(1, 0);
774 
775  }
776  }
777 
778  return Jmat;
779 }
