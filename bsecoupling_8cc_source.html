91 void BSECoupling::addoutput(Property *_type_summary,Orbitals* _orbitalsA,
92  Orbitals* _orbitalsB){
93 
94  string algorithm="full_diag";
95  int methodindex=1;
96  if (_output_perturbation){
97  algorithm="perturbation";
98  methodindex=0;
99  }
100  _type_summary->setAttribute("algorithm",algorithm);
101  if (_doSinglets){
102  Property *_singlet_summary = &_type_summary->add("singlets","");
103  for (int stateA = 0; stateA < _levA ; ++stateA ) {
104  for (int stateB = 0; stateB <_levB ; ++stateB ) {
105  double JAB = getSingletCouplingElement( stateA , stateB, methodindex);
106 
107  Property *_coupling_summary = &_singlet_summary->add("coupling", (format("%1$1.6e") % JAB).str());
108  double energyA = _orbitalsA->BSESingletEnergies()(stateA)*conv::hrt2ev;
109  double energyB = _orbitalsB->BSESingletEnergies()(stateB)*conv::hrt2ev;
110  _coupling_summary->setAttribute("excitonA", stateA);
111  _coupling_summary->setAttribute("excitonB", stateB);
112  _coupling_summary->setAttribute("energyA", (format("%1$1.6e") % energyA).str());
113  _coupling_summary->setAttribute("energyB", (format("%1$1.6e") % energyB).str());
114  _coupling_summary->setAttribute("pert", (format("%1$1.6e") % getSingletCouplingElement( stateA , stateB, 0)).str());
115  _coupling_summary->setAttribute("diag", (format("%1$1.6e") % getSingletCouplingElement( stateA , stateB, 1)).str());
116 
117  }
118  }
119  }
120 
121 
122  if ( _doTriplets){
123  Property *_triplet_summary = &_type_summary->add("triplets","");
124  for (int stateA = 0; stateA < _levA ; ++stateA ) {
125  for (int stateB = 0; stateB < _levA ; ++stateB ) {
126  double JAB = getTripletCouplingElement( stateA , stateB,methodindex );
127  //real_gwbse energyAD = getTripletDimerEnergy( stateA );
128  //real_gwbse energyBD = getTripletDimerEnergy( stateB );
129  Property *_coupling_summary = &_triplet_summary->add("coupling", (format("%1$1.6e") % JAB).str());
130  double energyA = _orbitalsA->BSETripletEnergies()(stateA)*conv::hrt2ev;
131  double energyB = _orbitalsB->BSETripletEnergies()(stateB)*conv::hrt2ev;
132  _coupling_summary->setAttribute("excitonA", stateA);
133  _coupling_summary->setAttribute("excitonB", stateB);
134  _coupling_summary->setAttribute("energyA", (format("%1$1.6e") % energyA).str());
135  _coupling_summary->setAttribute("energyB", (format("%1$1.6e") % energyB).str());
136  _coupling_summary->setAttribute("pert", (format("%1$1.6e") % getTripletCouplingElement( stateA , stateB, 0)).str());
137  _coupling_summary->setAttribute("diag", (format("%1$1.6e") % getTripletCouplingElement( stateA , stateB, 1)).str());
138 
139  }
140  }
141  }

166 bool BSECoupling::CalculateCouplings(Orbitals* _orbitalsA, Orbitals* _orbitalsB, Orbitals* _orbitalsAB) {
167  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculating exciton couplings" << flush;
168  // set the parallelization
169  #ifdef _OPENMP
170 
171  if ( _openmp_threads > 0 ) omp_set_num_threads(_openmp_threads);
172  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using "<< omp_get_max_threads()<<" threads" << flush;
173  #endif
174 
175 
176 
177  _orbitalsAB->setCoupledExcitonsA(_levA);
178  _orbitalsAB->setCoupledExcitonsB(_levB);
179  //check to see if ordering of atoms agrees
180  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
181  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
182  const std::vector<QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
183 
184  for (unsigned i=0;i<atomsAB.size();i++){
185  QMAtom* dimer=atomsAB[i];
186  QMAtom* monomer=NULL;
187  if (i<atomsA.size()){
188  monomer=atomsA[i];
189  }
190  else if (i<atomsB.size()+atomsA.size() ){
191  monomer=atomsB[i-atomsA.size()];
192  }
193  else{
194  throw runtime_error((boost::format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
195  }
196 
197  if(monomer->getType() != dimer->getType()){
198  throw runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
199  }
200  if(tools::abs(monomer->getPos()-dimer->getPos())>0.001){
201  CTP_LOG(ctp::logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << flush;
202  break;
203  }
204 
205  }
206 
207 
208  // constructing the direct product orbA x orbB
209  int _basisA = _orbitalsA->getBasisSetSize();
210  int _basisB = _orbitalsB->getBasisSetSize();
211 
212  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
213  CTP_LOG(ctp::logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
214  return false;
215  }
216 
217  // number of levels stored in monomers
218  int _levelsA = _orbitalsA->getNumberOfLevels();
219  int _levelsB = _orbitalsB->getNumberOfLevels();
220 
221 
222  // get exciton information of molecule A
223  int _bseA_cmax = _orbitalsA->getBSEcmax();
224  int _bseA_cmin = _orbitalsA->getBSEcmin();
225  int _bseA_vmax = _orbitalsA->getBSEvmax();
226  int _bseA_vmin = _orbitalsA->getBSEvmin();
227  int _bseA_vtotal = _bseA_vmax - _bseA_vmin +1 ;
228  int _bseA_ctotal = _bseA_cmax - _bseA_cmin +1 ;
229  int _bseA_size = _bseA_vtotal * _bseA_ctotal;
230  int _bseA_singlet_exc = _orbitalsA->BSESingletCoefficients().cols();
231  int _bseA_triplet_exc = _orbitalsA->BSETripletCoefficients().cols();
232 
233  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule A has " << _bseA_singlet_exc << " singlet excitons with dimension " << _bseA_size << flush;
234  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule A has " << _bseA_triplet_exc << " triplet excitons with dimension " << _bseA_size << flush;
235 
236  // now, two storage assignment matrices for two-particle functions
237  Eigen::MatrixXi _combA;
238  _combA.resize(_bseA_size,2);
239  int _cnt = 0;
240  for ( int _v = 0; _v < _bseA_vtotal; _v++){
241  for ( int _c = 0; _c < _bseA_ctotal; _c++){
242  _combA(_cnt,0) = _v;
243  _combA(_cnt,1) = _bseA_vtotal + _c;
244  _cnt++;
245  }
246  }
247 
248  // get exciton information of molecule B
249  int _bseB_cmax = _orbitalsB->getBSEcmax();
250  int _bseB_cmin = _orbitalsB->getBSEcmin();
251  int _bseB_vmax = _orbitalsB->getBSEvmax();
252  int _bseB_vmin = _orbitalsB->getBSEvmin();
253  int _bseB_vtotal = _bseB_vmax - _bseB_vmin +1 ;
254  int _bseB_ctotal = _bseB_cmax - _bseB_cmin +1 ;
255  int _bseB_size = _bseB_vtotal * _bseB_ctotal;
256  int _bseB_singlet_exc = _orbitalsB->BSESingletCoefficients().cols();
257  int _bseB_triplet_exc = _orbitalsB->BSETripletCoefficients().cols();
258 
259 
260 
261  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule B has " << _bseB_singlet_exc << " singlet excitons with dimension " << _bseB_size << flush;
262  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " molecule B has " << _bseB_triplet_exc << " triplet excitons with dimension " << _bseB_size << flush;
263 
264  // now, two storage assignment matrices for two-particle functions
265  Eigen::MatrixXi _combB;
266  _combB.resize(_bseB_size,2);
267  _cnt = 0;
268  for ( int _v = 0; _v < _bseB_vtotal; _v++){
269  for ( int _c = 0; _c < _bseB_ctotal; _c++){
270  _combB(_cnt,0) = _bseA_vtotal + _bseA_ctotal + _v;
271  _combB(_cnt,1) = _bseA_vtotal + _bseA_ctotal + _bseB_vtotal + _c;
272  _cnt++;
273  }
274  }
275 
276  if(_levA>_bseA_singlet_exc){
277  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of excitons you want is greater than stored for molecule A. Setting to max number available" << flush;
278  _levA=_bseA_singlet_exc;
279  }
280  if(_levB>_bseB_singlet_exc){
281  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of excitons you want is greater than stored for molecule B. Setting to max number available" << flush;
282  _levB=_bseB_singlet_exc;
283  }
284 
285 
286  if(_levA>_bseA_singlet_exc){
287  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of Frenkel states you want is greater than stored for molecule A. Setting to max number available" << flush;
288  _levA=_bseA_singlet_exc;
289  }
290  if(_levB>_bseB_singlet_exc){
291  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of Frenkel states you want is greater than stored for molecule B. Setting to max number available" << flush;
292  _levB=_bseB_singlet_exc;
293  }
294 
295  if(_unoccA>_bseA_ctotal){
296  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of occupied orbitals in molecule A for CT creation exceeds number of KS-orbitals in BSE" << flush;
297  _unoccA=_bseA_ctotal;
298  }
299  else if (_unoccA<0){
300  _unoccA=_bseA_ctotal;
301  }
302  if(_unoccB>_bseB_ctotal){
303  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of occupied orbitals in molecule B for CT creation exceeds number of KS-orbitals in BSE" << flush;
304  _unoccB=_bseB_ctotal;
305  }
306  else if (_unoccB<0){
307  _unoccB=_bseB_ctotal;
308  }
309 
310  if(_occA>_bseA_vtotal){
311  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of unoccupied orbitals in molecule A for CT creation exceeds number of KS-orbitals in BSE" << flush;
312  _occA=_bseA_vtotal;
313  }
314  else if (_occA<0){
315  _occA=_bseA_vtotal;
316  }
317  if(_occB>_bseB_vtotal){
318  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Number of unoccupied orbitals in molecule B for CT creation exceeds number of KS-orbitals in BSE" << flush;
319  _occB=_bseB_vtotal;
320  }else if (_occB<0){
321  _occB=_bseB_vtotal;
322  }
323 
324 
325 
326 
327  // get exciton information of pair AB
328  int _bseAB_cmax = _orbitalsAB->getBSEcmax();
329  int _bseAB_cmin = _orbitalsAB->getBSEcmin();
330  int _bseAB_vmax = _orbitalsAB->getBSEvmax();
331  int _bseAB_vmin = _orbitalsAB->getBSEvmin();
332  int _bseAB_vtotal = _bseAB_vmax - _bseAB_vmin +1 ;
333  int _bseAB_ctotal = _bseAB_cmax - _bseAB_cmin +1 ;
334  int _bseAB_size = _bseAB_vtotal * _bseAB_ctotal;
335  // check if electron-hole interaction matrices are stored
336  if ( ! _orbitalsAB->hasEHinteraction_triplet() && _doTriplets){
337  CTP_LOG(ctp::logERROR,*_pLog) << "BSE EH for triplets not stored " << flush;
338  return false;
339  }
340  if ( ! _orbitalsAB->hasEHinteraction_singlet() && _doSinglets){
341  CTP_LOG(ctp::logERROR,*_pLog) << "BSE EH for singlets not stored " << flush;
342  return false;
343  }
344  const MatrixXfd& _eh_t = _orbitalsAB->eh_t();
345  const MatrixXfd& _eh_s = _orbitalsAB->eh_s();
346  if(_doTriplets){
347  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " dimer AB has BSE EH interaction triplet with dimension " << _eh_t.rows() << " x " << _eh_t.cols() << flush;
348  }
349  if(_doSinglets){
350  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " dimer AB has BSE EH interaction singlet with dimension " << _eh_s.rows() << " x " << _eh_s.cols() << flush;
351  }
352  // now, two storage assignment matrices for two-particle functions
353  Eigen::MatrixXi _combAB;
354  _combAB.resize(_bseAB_size,2);
355  _cnt = 0;
356  for ( int _v = 0; _v < _bseAB_vtotal; _v++){
357  for ( int _c = 0; _c < _bseAB_ctotal; _c++){
358  //_combAB(_cnt,0) = _v;
359  //_combAB(_cnt,1) = _bseAB_vtotal + _c;
360 
361  _combAB(_cnt,0) = _bseAB_vmin + _v;
362  _combAB(_cnt,1) = _bseAB_vmin + _bseAB_vtotal + _c;
363 
364  _cnt++;
365  }
366  }
367 
368 
369 
370 
371  // DFT levels of monomers can be reduced to those used in BSE
372  _levelsA = _bseA_vtotal + _bseA_ctotal;
373  _levelsB = _bseB_vtotal + _bseB_ctotal;
374  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " levels used in BSE of molA: " << _bseA_vmin << " to " << _bseA_cmax << " total: " << _bseA_vtotal + _bseA_ctotal << flush;
375  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " levels used in BSE of molB: " << _bseB_vmin << " to " << _bseB_cmax << " total: " << _bseB_vtotal + _bseB_ctotal << flush;
376 
377 
378  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
379  CTP_LOG(ctp::logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
380  return false;
381  }
382 
383  // | Orbitals_A 0 | | Overlap_A |
384  // | 0 Orbitals_B |.T X | Overlap_B | X ( Orbitals_AB )
385 
386  Eigen::MatrixXd _psi_AxB =Eigen::MatrixXd::Zero( _levelsA + _levelsB, _basisA + _basisB );
387 
388 
389 
390  // constructing merged orbitals
391  _psi_AxB.block(0,0,_levelsA , _basisA) = _orbitalsA->MOCoefficients().block(_bseA_vmin,0, _bseA_cmax+1-_bseA_vmin, _basisA );
392  _psi_AxB.block(_levelsA, _basisA,_levelsB,_basisB) =_orbitalsB->MOCoefficients().block(_bseB_vmin,0,_bseB_cmax+1-_bseA_vmin,_basisB);
393 
394  // psi_AxB * S_AB * psi_AB
395  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " projecting monomer onto dimer orbitals" << flush;
396 
397  Eigen::MatrixXd _overlapAB;
398  if ( _orbitalsAB->hasAOOverlap() ) {
399  CTP_LOG(ctp::logDEBUG,*_pLog) << "Reading overlap matrix from orbitals" << flush;
400  _overlapAB= _orbitalsAB->AOOverlap();
401  }else{
402  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating overlap matrix for basisset: "<< _orbitalsAB->getDFTbasis()<< flush;
403  BasisSet _dftbasisset;
404  AOBasis _dftbasis;
405  _dftbasisset.LoadBasisSet(_orbitalsAB->getDFTbasis());
406 
407  _dftbasis.AOBasisFill(&_dftbasisset, _orbitalsAB->QMAtoms());
408  AOOverlap _dftAOoverlap;
409  _dftAOoverlap.Fill(_dftbasis);
410  _overlapAB=_dftAOoverlap.Matrix();
411  }
412 
413 
414  Eigen::MatrixXd _psi_AxB_dimer_basis = _psi_AxB.transpose()*_overlapAB*_orbitalsAB->MOCoefficients();
415  _overlapAB.resize(0,0);
416 
417 
418  //cout<< "_psi_AxB_dimer"<<endl;
419  unsigned int LevelsA = _levelsA;
420  for (unsigned i=0;i<_psi_AxB_dimer_basis.rows();i++){
421  double mag=_psi_AxB_dimer_basis.row(i).squaredNorm();
422  if (mag<0.95){
423  int monomer = 0;
424  int level = 0;
425  if ( i < LevelsA ) {
426  monomer = 1;
427  level = _bseA_vmin + i;
428  } else {
429  monomer = 2;
430  level = _bseB_vmin + i -_levelsA;
431 
432  }
433  CTP_LOG(ctp::logERROR,*_pLog) << "\nERROR: " << i << " Projection of orbital " << level << " of monomer " << monomer << " on dimer is insufficient,mag="<<mag<<" maybe the orbital order is screwed up, otherwise increase dimer basis.\n"<<flush;
434  }
435  }
436 
437 
438  //notation AB is CT states with A+B-, BA is the counterpart
439  //Setting up CT-states:
440  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setting up CT-states" << flush;
441  //Number of A+B- states
442  int noAB=_occA*_unoccB;
443  //Number of A-B+ states
444  int noBA=_unoccA*_occB;
445 
446 
447  Eigen::MatrixXi comb_CTAB=Eigen::MatrixXi::Zero(noAB,2);
448 
449  _cnt = 0;
450  // iterate A over occupied, B over unoccupied
451  int v_start=_bseA_vtotal-_occA;
452  for ( int _v = v_start; _v < _bseA_vtotal; _v++){
453  for ( int _c = 0; _c <_unoccB; _c++){
454  comb_CTAB(_cnt,0) =_v;
455  comb_CTAB(_cnt,1) = _bseA_vtotal+_bseA_ctotal+_bseB_vtotal + _c;
456 
457  _cnt++;
458  }
459  }
460  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() <<" "<<noAB <<" CT states A+B- created" << flush;
461 
462  Eigen::MatrixXi comb_CTBA=Eigen::MatrixXi::Zero(noBA,2);
463  _cnt = 0;
464  // iterate A over unoccupied, B over occupied
465  v_start=_bseB_vtotal-_occB;
466  for ( int _v = v_start; _v < _bseB_vtotal; _v++){
467  for ( int _c = 0; _c <_unoccA; _c++){
468  comb_CTBA(_cnt,0) =_bseA_vtotal+_bseA_ctotal+_v;
469  comb_CTBA(_cnt,1) = _bseA_vtotal+ _c;
470 
471  _cnt++;
472  }
473  }
474  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() <<" "<<noBA <<" CT states B+A- created" << flush;
475 
476 
477 
478  // these 4 matrixes, matrix(i,j) contains the j-th dimer MO component of the i-th excitation
479 
480  ctAB.resize(noAB,_bseAB_size);
481  #pragma omp parallel for
482  for ( int _i_CT = 0 ; _i_CT < noAB ; _i_CT++){
483  for ( int _i_bseAB = 0 ; _i_bseAB < _bseAB_size ; _i_bseAB++){
484  ctAB(_i_CT,_i_bseAB)=_psi_AxB_dimer_basis( comb_CTAB(_i_CT,0), _combAB( _i_bseAB,0) ) * _psi_AxB_dimer_basis( comb_CTAB(_i_CT,1), _combAB( _i_bseAB,1) );
485  }
486  }
487 
488 
489  ctBA.resize(noBA,_bseAB_size);
490  #pragma omp parallel for

724 Eigen::MatrixXd BSECoupling::Perturbation(const Eigen::MatrixXd& _J_dimer){
725 
726  Eigen::MatrixXd _J =Eigen::MatrixXd::Zero(_bse_exc, _bse_exc);
727  bool _diag_ct = true;
728  Eigen::MatrixXd _J_result=_J_dimer;
729  if (_ct > 0 && _diag_ct) {

732

734  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(Ct);
735

738 
739  Ct.resize(0, 0);
740 
741  if (tools::globals::verbose) {
742 
743  CTP_LOG(ctp::logDEBUG, *_pLog) << "FE state hamiltonian" << flush;
744  CTP_LOG(ctp::logDEBUG, *_pLog) << _J_dimer.block(0,0, _bse_exc,_bse_exc) << flush;

748  }

758  }
759  }
760  for (int stateA = 0; stateA < _levA; stateA++) {
761  double Ea = _J_result(stateA, stateA);
762  for (int stateB = 0; stateB < _levB; stateB++) {
763  int stateBd = stateB + _bseA_exc;

774  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << "Energydifference between state B " << stateB + 1 << "and CT state " << k + 1 << " is " << Eab - Eb << "[Hrt]" << flush;
775 
776  }
777  J += 0.5 * _J_result(k, stateA) * _J_result(k, stateBd)*(1 / (Ea - Eab) + 1 / (Eb - Eab)); // Have no clue why 0.5
778  }
779  _J(stateA, stateBd) = J;

783  }

791 Eigen::MatrixXd BSECoupling::Fulldiag(const Eigen::MatrixXd& _J_dimer){
792  Eigen::MatrixXd _J = Eigen::MatrixXd::Zero(_bse_exc, _bse_exc);
793 
794  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(_J_dimer);
795 
796  if (tools::globals::verbose && _bse_exc + _ct < 10) {
797  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
798  CTP_LOG(ctp::logDEBUG, *_pLog) << "Eigenvectors of J" << flush;
799 
800  CTP_LOG(ctp::logDEBUG, *_pLog) << es.eigenvectors() << flush;
801  CTP_LOG(ctp::logDEBUG, *_pLog) << "J_eigenvalues[Hrt]" << flush;
802  CTP_LOG(ctp::logDEBUG, *_pLog) << es.eigenvalues() << flush;
803  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
804  }
805  //Calculate projection on subspace for every pair of excitons separately
806  for (int stateA = 0; stateA < _levA; stateA++) {
807  for (int stateB = 0; stateB < _levB; stateB++) {
808  int stateBd = stateB + _bseA_exc;
809  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculating coupling between exciton A" << stateA + 1 << " and exciton B" << stateB + 1 << flush;
810  std::vector<unsigned> index;
811  std::vector<int> signvec;
812  for (unsigned i = 0; i < _bse_exc + _ct; i++) {
813  if (i == unsigned(stateA) || i == unsigned(stateBd)) {
814 
815  double close = 0.0;
816  unsigned ind = 0;
817  int sign = 0;
818  //row
819  for (unsigned j = 0; j < _bse_exc + _ct; j++) {
820  bool check = true;
821  // if index i is already in index
822  // should not happen but if one vector was similar to two others.
823  for (unsigned l = 0; l < index.size(); l++) {
824  if (j == index[l]) {
825  check = false;
826  break;
827  }
828  }
829 
830  if (check && std::abs(es.eigenvalues()(i, j)) > close) {
831  ind = j;
832  close = std::abs(es.eigenvalues()(i, j));
833  if (es.eigenvalues()(i, j) >= 0) {
834  sign = 1;
835  } else {
836  sign = -1;
837  }
838  }
839  }
840  index.push_back(ind);
841  signvec.push_back(sign);
842  }
843  }
844 
845  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Order is: [Initial state n->nth eigenvalue]" << flush;
846  CTP_LOG(ctp::logDEBUG, *_pLog) << " A" << stateA + 1 << ":" << stateA + 1 << "->" << index[0] + 1 << " ";
847  CTP_LOG(ctp::logDEBUG, *_pLog) << " B" << stateB + 1 << ":" << stateBd + 1 << "->" << index[1] + 1 << " " << flush;
848 
849  //setting up transformation matrix _T and diagonal matrix _E for the eigenvalues;
850 
851  Eigen::MatrixXd _E = Eigen::MatrixXd::Zero(2, 2);
852  Eigen::MatrixXd _T = Eigen::MatrixXd::Zero(2, 2);
853  //find the eigenvectors which are most similar to the initial states
854 
855  //row
856  for (unsigned i = 0; i < 2; i++) {
857  unsigned k = index[i];
858  double sign = signvec[i];
859  double normr = 1 / std::sqrt(es.eigenvectors()(stateA, k) * es.eigenvectors()(stateA, k) + es.eigenvectors()(stateBd, k) * es.eigenvectors()(stateBd, k));
860  _T(0, i) = sign * es.eigenvectors()(stateA, k) * normr;
861  _T(1, i) = sign * es.eigenvectors()(stateBd, k) * normr;
862  _E(i, i) = es.eigenvectors()(k);
863  }
864 
865 
866  if ((_T(1, 1) * _T(0, 0) - _T(1, 0) * _T(0, 1)) < 0) {
867  CTP_LOG(ctp::logDEBUG, *_pLog) << " Reduced state matrix is not in a right handed basis, multiplying second eigenvector by -1 " << flush;
868  _T(0, 1) = -_T(0, 1);
869  _T(1, 1) = -_T(1, 1);
870  }
871 
872  if (tools::globals::verbose) {
873  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
874  CTP_LOG(ctp::logDEBUG, *_pLog) << "_T" << flush;
875  CTP_LOG(ctp::logDEBUG, *_pLog) << _T << flush;
876 
877  }
878 
879  Eigen::MatrixXd S_small = _T*_T.transpose();
880  if (tools::globals::verbose) {
881 
882  CTP_LOG(ctp::logDEBUG, *_pLog) << "S_small" << flush;
883  CTP_LOG(ctp::logDEBUG, *_pLog) << S_small << flush;
884 
885  }
886  //orthogonalize that matrix
887 
888  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> ss(S_small);
889  Eigen::MatrixXd sm1=ss.operatorInverseSqrt();
890  _E=sm1*_E*sm1;
891 
892  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Smallest value of dimer overlapmatrix is " << ss.eigenvalues()(0) << flush;
893  if (tools::globals::verbose) {
894 
895  CTP_LOG(ctp::logDEBUG, *_pLog) << "S-1/2" << flush;
896  CTP_LOG(ctp::logDEBUG, *_pLog) << sm1 << flush;
897  CTP_LOG(ctp::logDEBUG, *_pLog) << "E_ortho" << flush;
898  CTP_LOG(ctp::logDEBUG, *_pLog) << _E << flush;
899  }
900  _T = _T*sm1;
901 
902  if (tools::globals::verbose) {
903 
904  CTP_LOG(ctp::logDEBUG, *_pLog) << "T_ortho" << flush;
905  CTP_LOG(ctp::logDEBUG, *_pLog) << _T << flush;
906  CTP_LOG(ctp::logDEBUG, *_pLog) << "---------------------------------------" << flush;
907  }
908

912  CTP_LOG(ctp::logDEBUG, *_pLog) << "T_ortho*E_ortho*T_ortho^T" << flush;
913  CTP_LOG(ctp::logDEBUG, *_pLog) << _J_small << flush;
914  }
915 
916  _J(stateA, stateBd) = _J_small(0, 1);
917  _J(stateBd, stateA) = _J_small(1, 0);
918 
919  }
920  }
921 
922  return _J;
923 }
