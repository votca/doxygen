38 void BSECoupling::Initialize(Property& options) {
39 
40  std::string key = Identify();
41 
42  string spintype =
43  options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".spin");
44  if (spintype == "all") {
45  _doSinglets = true;
46  _doTriplets = true;
47  } else if (spintype == "triplet") {
48  _doTriplets = true;
49  } else if (spintype == "singlet") {
50  _doSinglets = true;
51  } else {
52  throw std::runtime_error(
53  (boost::format(
54  "Choice % for type not known. Available singlet,triplet,all") %
55  spintype)
56  .str());
57  }
58 
59  _output_perturbation = options.ifExistsReturnElseReturnDefault<bool>(
60  key + ".use_perturbation", _output_perturbation);
61 
62  _levA = options.get(key + ".moleculeA.states").as<Index>();
63  _levB = options.get(key + ".moleculeB.states").as<Index>();
64  _occA = options.get(key + ".moleculeA.occLevels").as<Index>();
65  _occB = options.get(key + ".moleculeB.occLevels").as<Index>();
66  _unoccA = options.get(key + ".moleculeA.unoccLevels").as<Index>();
67  _unoccB = options.get(key + ".moleculeB.unoccLevels").as<Index>();
68 }

70 void BSECoupling::WriteToProperty(Property& summary, const QMState& stateA,
71  const QMState& stateB) const {
72  Property& coupling_summary = summary.add("coupling", "");
73  double JAB_pert = 0;
74  double JAB_diag = 0;
75  if (stateA.Type() == QMStateType::Singlet) {
76  JAB_pert =
77  getSingletCouplingElement(stateA.StateIdx(), stateB.StateIdx(), 0);
78  JAB_diag =
79  getSingletCouplingElement(stateA.StateIdx(), stateB.StateIdx(), 1);
80  } else if (stateA.Type() == QMStateType::Triplet) {
81  JAB_pert =
82  getTripletCouplingElement(stateA.StateIdx(), stateB.StateIdx(), 0);
83  JAB_diag =
84  getTripletCouplingElement(stateA.StateIdx(), stateB.StateIdx(), 1);
85  }
86  coupling_summary.setAttribute("stateA", stateA.ToString());
87  coupling_summary.setAttribute("stateB", stateB.ToString());
88  coupling_summary.setAttribute("j_pert", (format("%1$1.6e") % JAB_pert).str());
89  coupling_summary.setAttribute("j_diag", (format("%1$1.6e") % JAB_diag).str());
90 }

92 void BSECoupling::Addoutput(Property& type_summary, const Orbitals&,
93  const Orbitals&) const {
94  tools::Property& bsecoupling = type_summary.add(Identify(), "");
95  string algorithm = "j_diag";
96  if (_output_perturbation) {
97  algorithm = "j_pert";
98  }
99  if (_doSinglets) {
100  QMStateType singlet = QMStateType(QMStateType::Singlet);
101  Property& singlet_summary = bsecoupling.add(singlet.ToLongString(), "");
102  singlet_summary.setAttribute("algorithm", algorithm);
103  for (Index stateA = 0; stateA < _levA; ++stateA) {
104  QMState qmstateA = QMState(singlet, stateA, false);
105  for (Index stateB = 0; stateB < _levB; ++stateB) {
106  QMState qmstateB = QMState(singlet, stateB, false);
107  WriteToProperty(singlet_summary, qmstateA, qmstateB);
108  }
109  }
110  }
111 
112  if (_doTriplets) {
113  QMStateType triplet = QMStateType(QMStateType::Triplet);
114  Property& triplet_summary = bsecoupling.add(triplet.ToLongString(), "");
115  triplet_summary.setAttribute("algorithm", algorithm);
116  for (Index stateA = 0; stateA < _levA; ++stateA) {
117  QMState qmstateA = QMState(triplet, stateA, false);
118  for (Index stateB = 0; stateB < _levB; ++stateB) {
119  QMState qmstateB = QMState(triplet, stateB, false);
120  WriteToProperty(triplet_summary, qmstateA, qmstateB);
121  }
122  }
123  }
124 }

138 Eigen::MatrixXd BSECoupling::SetupCTStates(Index bseA_vtotal, Index bseB_vtotal,
139  Index bseAB_vtotal,
140  Index bseAB_ctotal,
141  const Eigen::MatrixXd& A_AB,
142  const Eigen::MatrixXd& B_AB) const {
143 
144  Index noAB = _occA * _unoccB;
145  Index noBA = _unoccA * _occB;
146  Index bseAB_total = bseAB_vtotal + bseAB_ctotal;
147  Index bseAB_size = bseAB_vtotal * bseAB_ctotal;
148  Eigen::MatrixXd CTstates = Eigen::MatrixXd::Zero(bseAB_size, noAB + noBA);
149 
150  auto A_occ = A_AB.block(0, bseA_vtotal - _occA, bseAB_total, _occA);
151  auto A_unocc = A_AB.block(0, bseA_vtotal, bseAB_total, _unoccA);
152  auto B_occ = B_AB.block(0, bseB_vtotal - _occB, bseAB_total, _occB);
153  auto B_unocc = B_AB.block(0, bseB_vtotal, bseAB_total, _unoccB);
154 
155  const Eigen::MatrixXd A_occ_occ = A_occ.topRows(bseAB_vtotal);
156  const Eigen::MatrixXd B_unocc_unocc = B_unocc.bottomRows(bseAB_ctotal);
157 
158  // notation AB is CT states with A+B-, BA is the counterpart
159  // Setting up CT-states:
160  XTP_LOG(Log::error, *_pLog)
161  << TimeStamp() << " Setting up CT-states" << flush;
162 
163  // Number of A+B- states
164 #pragma omp parallel for
165  for (Index a_occ = 0; a_occ < _occA; a_occ++) {
166  for (Index b_unocc = 0; b_unocc < _unoccB; b_unocc++) {
167  Index index = a_occ * _unoccB + b_unocc;
168  Eigen::MatrixXd Coeff =
169  B_unocc_unocc.col(b_unocc) * A_occ_occ.col(a_occ).transpose();
170  CTstates.col(index) =
171  Eigen::Map<Eigen::VectorXd>(Coeff.data(), bseAB_size);
172  }
173  }
174  XTP_LOG(Log::error, *_pLog)
175  << TimeStamp() << " " << noBA << " CT states A+B- created" << flush;
176 
177  const Eigen::MatrixXd A_unocc_unocc = A_unocc.bottomRows(bseAB_ctotal);
178  const Eigen::MatrixXd B_occ_occ = B_occ.topRows(bseAB_vtotal);
179 
180 #pragma omp parallel for
181  for (Index b_occ = 0; b_occ < _occB; b_occ++) {
182  for (Index a_unocc = 0; a_unocc < _unoccA; a_unocc++) {
183  Index index = b_occ * _unoccA + a_unocc + noAB;
184  Eigen::MatrixXd Coeff =
185  A_unocc_unocc.col(a_unocc) * B_occ_occ.col(b_occ).transpose();
186  CTstates.col(index) =
187  Eigen::Map<Eigen::VectorXd>(Coeff.data(), bseAB_size);
188  }
189  }
190  XTP_LOG(Log::error, *_pLog)
191  << TimeStamp() << " " << noBA << " CT states A-B+ created" << flush;
192  return CTstates;
193 }

195 Eigen::MatrixXd BSECoupling::ProjectFrenkelExcitons(
196  const Eigen::MatrixXd& BSE_Coeffs, const Eigen::MatrixXd& X_AB,
197  Index bseX_vtotal, Index bseX_ctotal, Index bseAB_vtotal,
198  Index bseAB_ctotal) const {
199  Index bseAB_size = bseAB_vtotal * bseAB_ctotal;
200  auto X_occ = X_AB.leftCols(bseX_vtotal);
201  auto X_unocc = X_AB.rightCols(bseX_ctotal);
202  const Eigen::MatrixXd X_occ_occ = X_occ.topRows(bseAB_vtotal);
203  const Eigen::MatrixXd X_unocc_unocc = X_unocc.bottomRows(bseAB_ctotal);
204  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(bseAB_size, BSE_Coeffs.cols());
205  // no pragma here because often we will only have one Coeff
206  for (Index i = 0; i < BSE_Coeffs.cols(); i++) {
207  Eigen::VectorXd coeff = BSE_Coeffs.col(i);
208  Eigen::Map<Eigen::MatrixXd> coeffmatrix =
209  Eigen::Map<Eigen::MatrixXd>(coeff.data(), bseX_ctotal, bseX_vtotal);
210  Eigen::MatrixXd proj = X_unocc_unocc * coeffmatrix * X_occ_occ.transpose();
211  result.col(i) = Eigen::Map<Eigen::VectorXd>(proj.data(), proj.size());
212  }
213  return result;
214 }

225 void BSECoupling::CalculateCouplings(const Orbitals& orbitalsA,
226  const Orbitals& orbitalsB,
227  const Orbitals& orbitalsAB) {
228  XTP_LOG(Log::error, *_pLog)
229  << TimeStamp() << " Calculating exciton couplings" << flush;
230  // set the parallelization
231  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Using "
232  << OPENMP::getMaxThreads() << " threads" << flush;
233 
234  CheckAtomCoordinates(orbitalsA, orbitalsB, orbitalsAB);
235 
236  // constructing the direct product orbA x orbB
237  Index basisB = orbitalsB.getBasisSetSize();
238  Index basisA = orbitalsA.getBasisSetSize();
239  if ((basisA == 0) || (basisB == 0)) {
240  throw std::runtime_error("Basis set size is not stored in monomers");
241  }
242  // get exciton information of molecule A
243  Index bseA_cmax = orbitalsA.getBSEcmax();
244  Index bseA_cmin = orbitalsA.getBSEcmin();
245  Index bseA_vmax = orbitalsA.getBSEvmax();
246  Index bseA_vmin = orbitalsA.getBSEvmin();
247  Index bseA_vtotal = bseA_vmax - bseA_vmin + 1;
248  Index bseA_ctotal = bseA_cmax - bseA_cmin + 1;
249  Index bseA_total = bseA_vtotal + bseA_ctotal;
250  Index bseA_size = bseA_vtotal * bseA_ctotal;
251  Index bseA_singlet_exc = orbitalsA.BSESinglets().eigenvectors().cols();
252  Index bseA_triplet_exc = orbitalsA.BSETriplets().eigenvectors().cols();
253 
254  XTP_LOG(Log::error, *_pLog)
255  << TimeStamp() << " molecule A has " << bseA_singlet_exc
256  << " singlet excitons with dimension " << bseA_size << flush;
257  XTP_LOG(Log::error, *_pLog)
258  << TimeStamp() << " molecule A has " << bseA_triplet_exc
259  << " triplet excitons with dimension " << bseA_size << flush;
260 
261  // get exciton information of molecule B
262  Index bseB_cmax = orbitalsB.getBSEcmax();
263  Index bseB_cmin = orbitalsB.getBSEcmin();
264  Index bseB_vmax = orbitalsB.getBSEvmax();
265  Index bseB_vmin = orbitalsB.getBSEvmin();
266  Index bseB_vtotal = bseB_vmax - bseB_vmin + 1;
267  Index bseB_ctotal = bseB_cmax - bseB_cmin + 1;
268  Index bseB_total = bseB_vtotal + bseB_ctotal;
269  Index bseB_size = bseB_vtotal * bseB_ctotal;
270  Index bseB_singlet_exc = orbitalsB.BSESinglets().eigenvectors().cols();
271  Index bseB_triplet_exc = orbitalsB.BSETriplets().eigenvectors().cols();
272 
273  XTP_LOG(Log::error, *_pLog)
274  << TimeStamp() << " molecule B has " << bseB_singlet_exc
275  << " singlet excitons with dimension " << bseB_size << flush;
276  XTP_LOG(Log::error, *_pLog)
277  << TimeStamp() << " molecule B has " << bseB_triplet_exc
278  << " triplet excitons with dimension " << bseB_size << flush;
279 
280  if (_levA > bseA_singlet_exc) {
281  XTP_LOG(Log::error, *_pLog)
282  << TimeStamp()
283  << " Number of excitons you want is greater than stored for molecule "
284  "A. Setting to max number available"
285  << flush;
286  _levA = bseA_singlet_exc;
287  }
288  if (_levB > bseB_singlet_exc) {
289  XTP_LOG(Log::error, *_pLog)
290  << TimeStamp()
291  << " Number of excitons you want is greater than stored for molecule "
292  "B. Setting to max number available"
293  << flush;
294  _levB = bseB_singlet_exc;
295  }
296 
297  if (_levA > bseA_singlet_exc) {
298  XTP_LOG(Log::error, *_pLog)
299  << TimeStamp()
300  << " Number of Frenkel states you want is greater than stored for "
301  "molecule A. Setting to max number available"
302  << flush;
303  _levA = bseA_singlet_exc;
304  }
305  if (_levB > bseB_singlet_exc) {
306  XTP_LOG(Log::error, *_pLog)
307  << TimeStamp()
308  << " Number of Frenkel states you want is greater than stored for "
309  "molecule B. Setting to max number available"
310  << flush;
311  _levB = bseB_singlet_exc;
312  }
313 
314  if (_unoccA > bseA_ctotal) {
315  XTP_LOG(Log::error, *_pLog)
316  << TimeStamp()
317  << " Number of occupied orbitals in molecule A for CT creation "
318  "exceeds number of KS-orbitals in BSE"
319  << flush;
320  _unoccA = bseA_ctotal;
321  } else if (_unoccA < 0) {
322  _unoccA = bseA_ctotal;
323  XTP_LOG(Log::error, *_pLog)
324  << TimeStamp()
325  << " Number of occupied orbitals in molecule B for CT creation "
326  "exceeds number of KS-orbitals in BSE"
327  << flush;
328  }
329  if (_unoccB > bseB_ctotal) {
330  XTP_LOG(Log::error, *_pLog)
331  << TimeStamp()
332  << " Number of occupied orbitals in molecule B for CT creation "
333  "exceeds number of KS-orbitals in BSE"
334  << flush;
335  _unoccB = bseB_ctotal;
336  } else if (_unoccB < 0) {
337  _unoccB = bseB_ctotal;
338  }
339 
340  if (_occA > bseA_vtotal) {
341  XTP_LOG(Log::error, *_pLog)
342  << TimeStamp()
343  << " Number of unoccupied orbitals in molecule A for CT creation "
344  "exceeds number of KS-orbitals in BSE"
345  << flush;
346  _occA = bseA_vtotal;
347  } else if (_occA < 0) {
348  _occA = bseA_vtotal;
349  }
350  if (_occB > bseB_vtotal) {
351  XTP_LOG(Log::error, *_pLog)
352  << TimeStamp()
353  << " Number of unoccupied orbitals in molecule B for CT creation "
354  "exceeds number of KS-orbitals in BSE"
355  << flush;
356  _occB = bseB_vtotal;
357  } else if (_occB < 0) {
358  _occB = bseB_vtotal;
359  }
360 
361  // get exciton information of pair AB
362  Index bseAB_cmax = orbitalsAB.getBSEcmax();
363  Index bseAB_cmin = orbitalsAB.getBSEcmin();
364  Index bseAB_vmax = orbitalsAB.getBSEvmax();
365  Index bseAB_vmin = orbitalsAB.getBSEvmin();
366  Index basisAB = orbitalsAB.getBasisSetSize();
367  Index bseAB_vtotal = bseAB_vmax - bseAB_vmin + 1;
368  Index bseAB_ctotal = bseAB_cmax - bseAB_cmin + 1;
369  Index bseAB_total = bseAB_vtotal + bseAB_ctotal;
370  Index bseAB_size = bseAB_vtotal * bseAB_ctotal;
371 
372  // DFT levels of monomers can be reduced to those used in BSE
373  XTP_LOG(Log::error, *_pLog)
374  << TimeStamp() << " levels used for BSE of molA: " << bseA_vmin
375  << " to " << bseA_cmax << " total: " << bseA_total << flush;
376  XTP_LOG(Log::error, *_pLog)
377  << TimeStamp() << " levels used for BSE of molB: " << bseB_vmin
378  << " to " << bseB_cmax << " total: " << bseB_total << flush;
379  XTP_LOG(Log::error, *_pLog)
380  << TimeStamp() << " levels used for BSE of dimer AB: " << bseAB_vmin
381  << " to " << bseAB_cmax << " total: " << bseAB_total << flush;
382 
383  Eigen::MatrixXd MOsA =
384  orbitalsA.MOs().eigenvectors().block(0, bseA_vmin, basisA, bseA_total);
385  Eigen::MatrixXd MOsB =
386  orbitalsB.MOs().eigenvectors().block(0, bseB_vmin, basisB, bseB_total);
387  Eigen::MatrixXd MOsAB = orbitalsAB.MOs().eigenvectors().block(
388  0, bseAB_vmin, basisAB, bseAB_total);
389 
390  XTP_LOG(Log::info, *_pLog)
391  << TimeStamp() << " Calculating overlap matrix for basisset: "
392  << orbitalsAB.getDFTbasisName() << flush;
393 
394  Eigen::MatrixXd overlap = CalculateOverlapMatrix(orbitalsAB) * MOsAB;
395  XTP_LOG(Log::info, *_pLog)
396  << TimeStamp() << " Projecting monomers onto dimer orbitals" << flush;
397 
398  Eigen::MatrixXd A_AB = overlap.topRows(basisA).transpose() * MOsA;
399  Eigen::MatrixXd B_AB = overlap.bottomRows(basisB).transpose() * MOsB;
400  Eigen::VectorXd mag_A = A_AB.colwise().squaredNorm();
401  if (mag_A.any() < 0.95) {
402  XTP_LOG(Log::error, *_pLog)
403  << "\nWarning: "
404  << "Projection of orbitals of monomer A on dimer is insufficient,mag="
405  << mag_A.minCoeff() << flush;
406  }
407  Eigen::VectorXd mag_B = B_AB.colwise().squaredNorm();
408  if (mag_B.any() < 0.95) {
409  XTP_LOG(Log::error, *_pLog)
410  << "\nWarning: "
411  << "Projection of orbitals of monomer B on dimer is insufficient,mag="
412  << mag_B.minCoeff() << flush;
413  }
414 
415  AOBasis dftbasis = orbitalsAB.SetupDftBasis();
416  AOBasis auxbasis = orbitalsAB.SetupAuxBasis();
417  TCMatrix_gwbse Mmn;
418  // rpamin here, because RPA needs till rpamin
419  Mmn.Initialize(auxbasis.AOBasisSize(), orbitalsAB.getRPAmin(),
420  orbitalsAB.getGWAmax(), orbitalsAB.getRPAmin(),
421  orbitalsAB.getRPAmax());
422  Mmn.Fill(auxbasis, dftbasis, orbitalsAB.MOs().eigenvectors());
423 
424  const Eigen::MatrixXd& qpcoeff = orbitalsAB.QPdiag().eigenvectors();
425  Eigen::MatrixXd Hqp = qpcoeff *
426  orbitalsAB.QPdiag().eigenvalues().asDiagonal() *
427  qpcoeff.transpose();
428 
429  BSE::options opt;
430  opt.cmax = orbitalsAB.getBSEcmax();
431  opt.homo = orbitalsAB.getHomo();
432  opt.qpmin = orbitalsAB.getGWAmin();
433  opt.qpmax = orbitalsAB.getGWAmax();
434  opt.rpamax = orbitalsAB.getRPAmax();
435  opt.rpamin = orbitalsAB.getRPAmin();
436  opt.useTDA = true;
437  opt.vmin = orbitalsAB.getBSEvmin();
438  opt.use_Hqp_offdiag = orbitalsAB.GetFlagUseHqpOffdiag();
439  BSE bse(*_pLog, Mmn);
440  bse.configure(opt, orbitalsAB.RPAInputEnergies(), Hqp);
441  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Setup BSE operator" << flush;
442 
443  // now the different spin types
444  if (_doSinglets) {
445  XTP_LOG(Log::error, *_pLog)
446  << TimeStamp() << " Evaluating singlets" << flush;
447  XTP_LOG(Log::error, *_pLog)
448  << TimeStamp() << " Setup Hamiltonian" << flush;
449  Eigen::MatrixXd FE_AB = Eigen::MatrixXd::Zero(bseAB_size, _levA + _levB);
450  const Eigen::MatrixXd bseA =
451  orbitalsA.BSESinglets().eigenvectors().leftCols(_levA);
452  FE_AB.leftCols(_levA) = ProjectFrenkelExcitons(
453  bseA, A_AB, bseA_vtotal, bseA_ctotal, bseAB_vtotal, bseAB_ctotal);
454  const Eigen::MatrixXd bseB =
455  orbitalsB.BSESinglets().eigenvectors().leftCols(_levB);
456  FE_AB.rightCols(_levB) = ProjectFrenkelExcitons(
457  bseB, B_AB, bseB_vtotal, bseB_ctotal, bseAB_vtotal, bseAB_ctotal);
458  Eigen::MatrixXd CTStates = SetupCTStates(
459  bseA_vtotal, bseB_vtotal, bseAB_vtotal, bseAB_ctotal, A_AB, B_AB);
460  JAB_singlet =
461  ProjectExcitons(FE_AB, CTStates, bse.getSingletOperator_TDA());
462  XTP_LOG(Log::error, *_pLog)
463  << TimeStamp() << " calculated singlet couplings " << flush;
464  }
465  if (_doTriplets) {
466  XTP_LOG(Log::error, *_pLog)
467  << TimeStamp() << " Evaluating triplets" << flush;
468  Eigen::MatrixXd FE_AB = Eigen::MatrixXd::Zero(bseAB_size, _levA + _levB);
469  const Eigen::MatrixXd bseA =
470  orbitalsA.BSETriplets().eigenvectors().leftCols(_levA);
471  FE_AB.leftCols(_levA) = ProjectFrenkelExcitons(
472  bseA, A_AB, bseA_vtotal, bseA_ctotal, bseAB_vtotal, bseAB_ctotal);
473  const Eigen::MatrixXd bseB =
474  orbitalsB.BSETriplets().eigenvectors().leftCols(_levB);
475  FE_AB.rightCols(_levB) = ProjectFrenkelExcitons(
476  bseB, B_AB, bseB_vtotal, bseB_ctotal, bseAB_vtotal, bseAB_ctotal);
477  Eigen::MatrixXd CTStates = SetupCTStates(
478  bseA_vtotal, bseB_vtotal, bseAB_vtotal, bseAB_ctotal, A_AB, B_AB);
479  JAB_triplet =
480  ProjectExcitons(FE_AB, CTStates, bse.getTripletOperator_TDA());
481  XTP_LOG(Log::error, *_pLog)
482  << TimeStamp() << " calculated triplet couplings " << flush;
483  }
484  XTP_LOG(Log::error, *_pLog)
485  << TimeStamp() << " Done with exciton couplings" << flush;
486 }

488 Eigen::MatrixXd BSECoupling::OrthogonalizeCTs(Eigen::MatrixXd& FE_AB,
489  Eigen::MatrixXd& CTStates) const {
490  Index ct = CTStates.cols();
491 
492  if (ct > 0) {
493  XTP_LOG(Log::error, *_pLog)
494  << TimeStamp() << " Orthogonalizing CT-states with respect to FE-states"
495  << flush;
496  Eigen::MatrixXd correction = FE_AB * (FE_AB.transpose() * CTStates);
497  CTStates -= correction;
498 
499  // normalize
500  Eigen::VectorXd norm = CTStates.colwise().norm();
501  for (Index i = 0; i < CTStates.cols(); i++) {
502  CTStates.col(i) /= norm(i);
503  }
504  Index minstateindex = 0;
505  double minnorm = norm.minCoeff(&minstateindex);
506  if (minnorm < 0.95) {
507  XTP_LOG(Log::error, *_pLog)
508  << TimeStamp() << " WARNING: CT-state " << minstateindex
509  << " norm is only " << minnorm << flush;
510  }
511  }
512  Index bse_exc = _levA + _levB;
513 
514  Index bseAB_size = CTStates.rows();
515  Eigen::MatrixXd projection(bseAB_size, bse_exc + ct);
516  XTP_LOG(Log::info, *_pLog)
517  << TimeStamp() << " merging projections into one vector " << flush;
518  projection.leftCols(bse_exc) = FE_AB;
519  FE_AB.resize(0, 0);
520  if (ct > 0) {
521  projection.rightCols(ct) = CTStates;
522  }
523  CTStates.resize(0, 0);
524  return projection;
525 }

527 template <class BSE_OPERATOR>
528 Eigen::MatrixXd BSECoupling::CalcJ_dimer(BSE_OPERATOR& H,
529  Eigen::MatrixXd& projection) const {
530 
531  XTP_LOG(Log::info, *_pLog)
532  << TimeStamp() << " Setting up coupling matrix size "
533  << projection.cols() << flush;
534  // matrix _J
535  // E_A J_AB J_A_ABCT J_A_BACT
536  // J_BA E_B J_B_ABCT J_B_BACT
537  // J_ABCT_A J_ABCT_B E_ABCT J_ABCT_BACT
538  // J_BACT_A J_BACT_B J_BACT_ABCT E_BACT
539 
540  // this only works for hermitian/symmetric H so only in TDA
541 
542  Eigen::MatrixXd J_dimer = projection.transpose() * (H * projection).eval();
543 
544  XTP_LOG(Log::info, *_pLog)
545  << TimeStamp() << " Setting up overlap matrix size "
546  << projection.cols() << flush;
547  Eigen::MatrixXd S_dimer = projection.transpose() * projection;
548 
549  projection.resize(0, 0);
550  if (projection.cols()) {
551  XTP_LOG(Log::debug, *_pLog)
552  << "---------------------------------------" << flush;
553  XTP_LOG(Log::debug, *_pLog) << "_J_dimer[Ryd]" << flush;
554 
555  XTP_LOG(Log::debug, *_pLog) << J_dimer << flush;
556  XTP_LOG(Log::debug, *_pLog) << "_S_dimer" << flush;
557 
558  XTP_LOG(Log::debug, *_pLog) << S_dimer << flush;
559  XTP_LOG(Log::debug, *_pLog)
560  << "---------------------------------------" << flush;
561  }
562 
563  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(S_dimer);
564  Eigen::MatrixXd Sm1 = es.operatorInverseSqrt();
565  Eigen::MatrixXd J_ortho = Sm1 * J_dimer * Sm1;
566 
567  if (projection.cols()) {
568  XTP_LOG(Log::debug, *_pLog)
569  << "---------------------------------------" << flush;
570  XTP_LOG(Log::debug, *_pLog) << "_J_ortho[Ryd]" << flush;
571  XTP_LOG(Log::debug, *_pLog) << J_ortho << flush;
572  XTP_LOG(Log::debug, *_pLog) << "_S-1/2" << flush;
573  XTP_LOG(Log::debug, *_pLog) << Sm1 << flush;
574  XTP_LOG(Log::debug, *_pLog)
575  << "---------------------------------------" << flush;
576  }
577  XTP_LOG(Log::debug, *_pLog)
578  << TimeStamp() << " Smallest value of dimer overlapmatrix is "
579  << es.eigenvalues()(0) << flush;
580  return J_ortho;
581 }

611 Eigen::MatrixXd BSECoupling::Perturbation(
612  const Eigen::MatrixXd& J_dimer) const {
613  Index bse_exc = _levA + _levB;
614  Index ct = J_dimer.rows() - bse_exc;
615  Eigen::MatrixXd J_result = J_dimer;
616  if (ct > 0) {
617  Eigen::MatrixXd transformation =
618  Eigen::MatrixXd::Identity(J_dimer.rows(), J_dimer.cols());
619  Eigen::MatrixXd Ct = J_dimer.bottomRightCorner(ct, ct);
620  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(Ct);
621  transformation.bottomRightCorner(ct, ct) = es.eigenvectors();
622  Ct.resize(0, 0);
623 
624  XTP_LOG(Log::debug, *_pLog) << "FE state hamiltonian" << flush;
625  XTP_LOG(Log::debug, *_pLog)
626  << J_dimer.topLeftCorner(bse_exc, bse_exc) << flush;
627  if (ct > 0) {
628  XTP_LOG(Log::debug, *_pLog) << "eigenvalues of CT states" << flush;
629  XTP_LOG(Log::debug, *_pLog) << es.eigenvalues().transpose() << flush;
630  }
631 
632  J_result = transformation.transpose() * J_dimer * transformation;
633  XTP_LOG(Log::debug, *_pLog)
634  << "---------------------------------------" << flush;
635  XTP_LOG(Log::debug, *_pLog) << "_J_ortho[Hrt] CT-state diag" << flush;
636  XTP_LOG(Log::debug, *_pLog) << J_result << flush;
637  XTP_LOG(Log::debug, *_pLog)
638  << "---------------------------------------" << flush;
639  }
640 
641  Eigen::MatrixXd Jmatrix = Eigen::MatrixXd::Zero(bse_exc, bse_exc);
642  for (Index stateA = 0; stateA < _levA; stateA++) {
643  double Ea = J_result(stateA, stateA);
644  for (Index stateB = 0; stateB < _levB; stateB++) {
645  Index stateBd = stateB + _levA;
646  XTP_LOG(Log::info, *_pLog)
647  << TimeStamp() << " Calculating coupling between exciton A"
648  << stateA + 1 << " and exciton B" << stateB + 1 << flush;
649  double J = J_result(stateA, stateBd);
650 
651  double Eb = J_result(stateBd, stateBd);
652  for (Index k = bse_exc; k < (bse_exc + ct); k++) {
653  double Eab = J_result(k, k);
654  if (std::abs(Eab - Ea) < 0.001) {
655  XTP_LOG(Log::error, *_pLog)
656  << TimeStamp() << "Energydifference between state A "
657  << stateA + 1 << "and CT state " << k + 1 << " is " << Eab - Ea
658  << "[Hrt]" << flush;
659  }
660  if (std::abs(Eab - Eb) < 0.001) {
661  XTP_LOG(Log::error, *_pLog)
662  << TimeStamp() << "Energydifference between state B "
663  << stateB + 1 << "and CT state " << k + 1 << " is " << Eab - Eb
664  << "[Hrt]" << flush;
665  }
666  J += 0.5 * J_result(k, stateA) * J_result(k, stateBd) *
667  (1 / (Ea - Eab) + 1 / (Eb - Eab)); // Have no clue why 0.5
668  }
669  Jmatrix(stateA, stateBd) = J;
670  Jmatrix(stateBd, stateA) = J;
671  }
672  }
673  return Jmatrix;
674 }

676 Eigen::MatrixXd BSECoupling::Fulldiag(const Eigen::MatrixXd& J_dimer) const {
677  Index bse_exc = _levA + _levB;
678  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(J_dimer);
679  XTP_LOG(Log::debug, *_pLog)
680  << "---------------------------------------" << flush;
681  XTP_LOG(Log::debug, *_pLog) << "Eigenvectors of J" << flush;
682  XTP_LOG(Log::debug, *_pLog) << es.eigenvectors() << flush;
683  XTP_LOG(Log::debug, *_pLog) << "J_eigenvalues[Hrt]" << flush;
684  XTP_LOG(Log::debug, *_pLog) << es.eigenvalues() << flush;
685  XTP_LOG(Log::debug, *_pLog)
686  << "---------------------------------------" << flush;
687  Eigen::MatrixXd Jmat = Eigen::MatrixXd::Zero(bse_exc, bse_exc);
688  // Calculate projection on subspace for every pair of excitons separately
689  for (Index stateA = 0; stateA < _levA; stateA++) {
690  for (Index stateB = 0; stateB < _levB; stateB++) {
691 
692  Index stateBd = stateB + _levA;
693  XTP_LOG(Log::info, *_pLog)
694  << TimeStamp() << " Calculating coupling between exciton A"
695  << stateA + 1 << " and exciton B" << stateB + 1 << flush;
696 
697  std::array<int, 2> indexes;
698  std::array<int, 2> signs;
699 
700  // Find the eigenstate state, which in an L2 is closed to A or B
701  // respectively
702  es.eigenvectors().row(stateA).cwiseAbs().maxCoeff(&indexes[0]);
703  es.eigenvectors().row(stateBd).cwiseAbs().maxCoeff(&indexes[1]);
704  if (indexes[0] == indexes[1]) {
705  Eigen::RowVectorXd stateamplitudes =
706  es.eigenvectors().row(stateBd).cwiseAbs();
707  stateamplitudes[indexes[1]] = 0.0;
708  stateamplitudes.maxCoeff(&indexes[1]);
709  }
710 
711  signs[0] = GetSign(es.eigenvectors()(stateA, indexes[0]));
712  signs[1] = GetSign(es.eigenvectors()(stateBd, indexes[1]));
713 
714  XTP_LOG(Log::info, *_pLog)
715  << TimeStamp() << " Order is: [Initial state n->nth eigenvalue]"
716  << flush;
717  XTP_LOG(Log::info, *_pLog) << " A" << stateA + 1 << ":" << stateA + 1
718  << "->" << indexes[0] + 1 << " ";
719  XTP_LOG(Log::info, *_pLog) << " B" << stateB + 1 << ":" << stateBd + 1
720  << "->" << indexes[1] + 1 << " " << flush;
721 
722  // setting up transformation matrix Tmat and diagonal matrix Emat for the
723  // eigenvalues;
724  Eigen::Matrix2d Emat = Eigen::Matrix2d::Zero();
725  Eigen::Matrix2d Tmat = Eigen::Matrix2d::Zero();
726  // find the eigenvectors which are most similar to the initial states
727  // row
728  for (Index i = 0; i < 2; i++) {
729  Index k = indexes[i];
730  double sign = signs[i];
731  Tmat(0, i) = sign * es.eigenvectors()(stateA, k);
732  Tmat(1, i) = sign * es.eigenvectors()(stateBd, k);
733  Emat(i, i) = es.eigenvalues()(k);
734  }
735  Tmat.colwise().normalize();
736 
737  if (Tmat.determinant() < 0) {
738  XTP_LOG(Log::info, *_pLog)
739  << " Reduced state matrix is not in a right handed basis, "
740  "multiplying second eigenvector by -1 "
741  << flush;
742  Tmat.col(1) *= -1;
743  }
744 
745  XTP_LOG(Log::debug, *_pLog)
746  << "---------------------------------------" << flush;
747  XTP_LOG(Log::debug, *_pLog) << "_T" << flush;
748  XTP_LOG(Log::debug, *_pLog) << Tmat << flush;
749 
750  Eigen::Matrix2d S_small = Tmat * Tmat.transpose();
751 
752  XTP_LOG(Log::debug, *_pLog) << "S_small" << flush;
753  XTP_LOG(Log::debug, *_pLog) << S_small << flush;
754  // orthogonalize that matrix
755 
756  Eigen::SelfAdjointEigenSolver<Eigen::Matrix2d> ss(S_small);
757  Eigen::Matrix2d sm1 = ss.operatorInverseSqrt();
758  Emat = sm1 * Emat * sm1;
759 
760  XTP_LOG(Log::info, *_pLog)
761  << TimeStamp() << " Smallest value of dimer overlapmatrix is "
762  << ss.eigenvalues()(0) << flush;
763 
764  XTP_LOG(Log::debug, *_pLog) << "S-1/2" << flush;
765  XTP_LOG(Log::debug, *_pLog) << sm1 << flush;
766  XTP_LOG(Log::debug, *_pLog) << "E_ortho" << flush;
767  XTP_LOG(Log::debug, *_pLog) << Emat << flush;
768 
769  Tmat = Tmat * sm1;
770 
771  XTP_LOG(Log::debug, *_pLog) << "T_ortho" << flush;
772  XTP_LOG(Log::debug, *_pLog) << Tmat << flush;
773  XTP_LOG(Log::debug, *_pLog)
774  << "---------------------------------------" << flush;
775 
776  Eigen::Matrix2d J_small = Tmat * Emat * Tmat.transpose();
777  XTP_LOG(Log::debug, *_pLog) << "T_ortho*E_ortho*T_ortho^T" << flush;
778  XTP_LOG(Log::debug, *_pLog) << J_small << flush;
779 
780  Jmat(stateA, stateBd) = J_small(0, 1);
781  Jmat(stateBd, stateA) = J_small(1, 0);
782  }
783  }
784 
785  return Jmat;
786 }
