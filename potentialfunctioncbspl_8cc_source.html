129 void PotentialFunctionCBSPL::extrapolExclParam(){
130 
131  double u0 = _lam(_nexcl);
132  double m = (_lam(_nexcl + 1) - _lam(_nexcl)) /
133  (_rbreak(_nexcl + 1) - _rbreak(_nexcl));
134  double r0 = _rbreak(_nexcl);
135 
136  // using linear extrapolation
137  // u(r) = ar + b
138  // a = m
139  // b = - m*r0 + u0
140  // m = (u1-u0)/(r1-r0)
141

143  double b = -1.0*m*r0 + u0;
144  for (int i = 0; i < _nexcl; i++)
145  _lam(i) = a*_rbreak(i) + b;
