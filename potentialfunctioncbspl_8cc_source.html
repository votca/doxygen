20 PotentialFunctionCBSPL::PotentialFunctionCBSPL(const string& name_,const int nlam_,
21  const double min_, const double max_) :
22  PotentialFunction(name_,nlam_,min_,max_) {
23 
24  /* Here nlam_ is the total number of coeff values that are to be optimized
25  * To ensure that potential and force go to zero smoothly near cut-off,
26  * as suggested in Ref. PCCP, 11, 1901, 2009, coeff values leading up to
27  * cut-off and beyond take a value of zero.
28  *
29  * Since region less than rmin is not sampled sufficiently for stability
30  * first _nexcl coefficients are not optimized instead their values are
31  * extrapolated from first statistically significant knot values near rmin
32  */
33 
34  int nknots;
35 
36  nknots = _lam.size();
37 
38  _nbreak = nknots - 2;
39 
40  _dr = ( _cut_off )/( double (_nbreak - 1) );
41 
42  // break point locations
43  // since ncoeff = nbreak +2 , r values for last two coefficients are also
44  // computed
45  _rbreak.resize(nknots,false);
46  _rbreak.clear();
47 
48  for( int i = 0; i < nknots; i++)
49  _rbreak(i) = i*_dr;
50 
51  // exclude knots corresponding to r <= _min
52  _nexcl = min( int( ( _min )/_dr ), _nbreak - 2 ) + 1;
53 
54  // account for finite numerical division of _min/_dr
55  // e.g. 0.24/0.02 may result in 11.99999999999999
56  if( _rbreak(_nexcl) == _min ) _nexcl++;
57 
58  // fixing last 4 knots to zeros is reasonable
59  _ncutcoeff = 4;
60 
61  // check if we have enough parameters to optimize
62  if((int(_lam.size()) - _nexcl - _ncutcoeff) < 1)
63  {
64  throw std::runtime_error("In potential "+_name+": no parameters to optimize!\n"
65  "All the knot values fall in the range of either excluded (due to high repulsive region) or cut-off region.\n"
66  "This issue can be resolved by one or combination of following steps:\n"
67  "1. Make sure you are using large-enough cut-off for this CG potential.\n"
68  "2. Make sure the CG-MD runs are sufficiently long and CG-MD RDF are statistically reliable.\n"
69  "3. Use more knot values.\n");
70  }
71 
72  _M.resize(4,4,false);
73  _M.clear();
74  _M(0,0) = 1.0; _M(0,1) = 4.0; _M(0,2) = 1.0; _M(0,3) = 0.0;
75  _M(1,0) = -3.0; _M(1,1) = 0.0; _M(1,2) = 3.0; _M(1,3) = 0.0;
76  _M(2,0) = 3.0; _M(2,1) = -6.0; _M(2,2) = 3.0; _M(2,3) = 0.0;
77  _M(3,0) = -1.0; _M(3,1) = 3.0; _M(3,2) = -3.0; _M(3,3) = 1.0;
78  _M /= 6.0;
79 
80 }

147 void PotentialFunctionCBSPL::extrapolExclParam(){
148 
149  double u0 = _lam(_nexcl);
150  double m = (_lam(_nexcl + 1) - _lam(_nexcl)) /
151  (_rbreak(_nexcl + 1) - _rbreak(_nexcl));
152  double r0 = _rbreak(_nexcl);
153 
154  /* If the slope m is positive then the potential core
155  * will be attractive. So, artificially forcing core to be
156  * repulsive by setting m = -m
157  */
158  if( m > 0)
159  {
160  cout << _name << " potential's extrapolated core is attractive!" << endl;
161  cout << "Artifically enforcing repulsive core.\n" << endl;
162  m *= -1.0;
163  }
164  // using linear extrapolation
165  // u(r) = ar + b
166  // a = m
167  // b = - m*r0 + u0
168  // m = (u1-u0)/(r1-r0)
169 
170  double a = m;
171  double b = -1.0*m*r0 + u0;
172  for (int i = 0; i < _nexcl; i++)
173  _lam(i) = a*_rbreak(i) + b;
174 
175 }

189 double PotentialFunctionCBSPL::CalculateF (const double r) const {
190 
191  if( r <= _cut_off){
192 
193  double u = 0.0;
194 
195  ub::vector<double> R;
196  ub::vector<double> B;
197  R.resize(4,false); R.clear();
198  B.resize(4,false); B.clear();
199 
200  int indx = min( int( r /_dr ) , _nbreak-2 );;
201  double rk = indx*_dr;;
202  double t = (r - rk)/_dr;
203 
204  R(0) = 1.0; R(1) = t; R(2) = t*t; R(3) = t*t*t;
205  ub::vector<double> RM = ub::prod(R,_M);
206  B(0) = _lam(indx); B(1) = _lam(indx+1); B(2) = _lam(indx+2);
207  B(3) = _lam(indx+3);
208 
209  u += ub::inner_prod(B,RM);
210 
211  return u;
212 
213  } else
214  return 0.0;
215 
216 }

218 // calculate first derivative w.r.t. ith parameter
219 double PotentialFunctionCBSPL::CalculateDF(const int i, const double r) const{
220 
221  // since first _nexcl parameters are not optimized for stability reasons
222 
223 
224  if ( r <= _cut_off ) {
225 
226  unsigned int i_opt = i + _nexcl;
227  unsigned int indx;
228  double rk;
229 
230  indx = min( int( ( r )/_dr ), _nbreak-2 );
231  rk = indx*_dr;
232 
233  if ( i_opt >= indx && i_opt <= indx+3 ){
234 
235  ub::vector<double> R;
236  R.resize(4,false); R.clear();
237 
238  double t = ( r - rk)/_dr;
239 
240  R(0) = 1.0; R(1) = t; R(2) = t*t; R(3) = t*t*t;
241 
242  ub::vector<double> RM = ub::prod(R,_M);
243 
244  return RM(i_opt-indx);
245 
246  }else
247  return 0.0;
248 
249  } else
250  return 0.0;
251 
252 }
