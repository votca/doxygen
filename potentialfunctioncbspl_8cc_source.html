11  * cut-off and beyond take a value of zero.
12  *
13  * Since region less than rmin is not sampled sufficiently for stability
14  * first _nexcl coefficients are not optimized instead their values are
15  * extrapolated from first statistically significant knot values near rmin
16  */

23 
24  _dr = ( _cut_off )/( double (_nbreak - 1) );
25

27  // since ncoeff = nbreak +2 , r values for last two coefficients are also
28  // computed
29  _rbreak.resize(nknots,false);
30  _rbreak.clear();
31 
32  for( int i = 0; i < nknots; i++)
33  _rbreak(i) = i*_dr;
34 
35  // exclude knots corresponding to r <= _min
36  _nexcl = min( int( ( _min )/_dr ), _nbreak - 2 ) + 1;

40  if( _rbreak(_nexcl) == _min ) _nexcl++;
41

43  _ncutcoeff = 4;

50  "This issue can be resolved by one or combination of following steps:\n"

131 void PotentialFunctionCBSPL::extrapolExclParam(){
132 
133  double u0 = _lam(_nexcl);
134  double m = (_lam(_nexcl + 1) - _lam(_nexcl)) /
135  (_rbreak(_nexcl + 1) - _rbreak(_nexcl));
136  double r0 = _rbreak(_nexcl);
137 
138  /* If the slope m is positive then the potential core
139  * will be attractive. So, artificially forcing core to be
140  * repulsive by setting m = -m
141  */
142  if( m > 0)
143  {
144  cout << _name << " potential's extrapolated core is attractive!" << endl;
145  cout << "Artifically enforcing repulsive core.\n" << endl;
146  m *= -1.0;
147  }
148  // using linear extrapolation
149  // u(r) = ar + b
150  // a = m
151  // b = - m*r0 + u0
152  // m = (u1-u0)/(r1-r0)
153 
154  double a = m;
155  double b = -1.0*m*r0 + u0;
156  for (int i = 0; i < _nexcl; i++)
157  _lam(i) = a*_rbreak(i) + b;
158 
159 }
