131 void PotentialFunctionCBSPL::extrapolExclParam(){
132 
133  double u0 = _lam(_nexcl);
134  double m = (_lam(_nexcl + 1) - _lam(_nexcl)) /
135  (_rbreak(_nexcl + 1) - _rbreak(_nexcl));
136  double r0 = _rbreak(_nexcl);
137 
138  // using linear extrapolation
139  // u(r) = ar + b
140  // a = m
141  // b = - m*r0 + u0
142  // m = (u1-u0)/(r1-r0)
143 
144  double a = m;
145  double b = -1.0*m*r0 + u0;
146  for (int i = 0; i < _nexcl; i++)
147  _lam(i) = a*_rbreak(i) + b;
