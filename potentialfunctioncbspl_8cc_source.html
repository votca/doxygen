29 PotentialFunctionCBSPL::PotentialFunctionCBSPL(const string &name_,
30  const int nlam_,
31  const double min_,
32  const double max_)
33  : PotentialFunction(name_, nlam_, min_, max_) {
34 
35  /* Here nlam_ is the total number of coeff values that are to be optimized
36  * To ensure that potential and force go to zero smoothly near cut-off,
37  * as suggested in Ref. PCCP, 11, 1901, 2009, coeff values leading up to
38  * cut-off and beyond take a value of zero.
39  *
40  * Since region less than rmin is not sampled sufficiently for stability
41  * first _nexcl coefficients are not optimized instead their values are
42  * extrapolated from first statistically significant knot values near rmin
43  */
44 
45  int nknots;
46 
47  nknots = _lam.size();
48 
49  _nbreak = nknots - 2;
50 
51  _dr = (_cut_off) / (double(_nbreak - 1));
52 
53  // break point locations
54  // since ncoeff = nbreak +2 , r values for last two coefficients are also
55  // computed
56  _rbreak = Eigen::VectorXd::Zero(nknots);
57 
58  for (int i = 0; i < nknots; i++) _rbreak(i) = i * _dr;
59 
60  // exclude knots corresponding to r <= _min
61  _nexcl = min(int((_min) / _dr), _nbreak - 2) + 1;
62 
63  // account for finite numerical division of _min/_dr
64  // e.g. 0.24/0.02 may result in 11.99999999999999
65  if (_rbreak(_nexcl) == _min) _nexcl++;
66 
67  // fixing last 4 knots to zeros is reasonable
68  _ncutcoeff = 4;
69 
70  // check if we have enough parameters to optimize
71  if ((int(_lam.size()) - _nexcl - _ncutcoeff) < 1) {
72  throw std::runtime_error(
73  "In potential " + _name +
74  ": no parameters to optimize!\n"
75  "All the knot values fall in the range of either excluded (due to high "
76  "repulsive region) or cut-off region.\n"
77  "This issue can be resolved by one or combination of following steps:\n"
78  "1. Make sure you are using large-enough cut-off for this CG "
79  "potential.\n"
80  "2. Make sure the CG-MD runs are sufficiently long and CG-MD RDF are "
81  "statistically reliable.\n"
82  "3. Use more knot values.\n");
83  }
84 
85  _M = Eigen::MatrixXd::Zero(4, 4);
86  _M(0, 0) = 1.0;
87  _M(0, 1) = 4.0;
88  _M(0, 2) = 1.0;
89  _M(0, 3) = 0.0;
90  _M(1, 0) = -3.0;
91  _M(1, 1) = 0.0;
92  _M(1, 2) = 3.0;
93  _M(1, 3) = 0.0;
94  _M(2, 0) = 3.0;
95  _M(2, 1) = -6.0;
96  _M(2, 2) = 3.0;
97  _M(2, 3) = 0.0;
98  _M(3, 0) = -1.0;
99  _M(3, 1) = 3.0;
100  _M(3, 2) = -3.0;
101  _M(3, 3) = 1.0;
102  _M /= 6.0;
103 }

164 void PotentialFunctionCBSPL::extrapolExclParam() {
165 
166  double u0 = _lam(_nexcl);
167  double m = (_lam(_nexcl + 1) - _lam(_nexcl)) /
168  (_rbreak(_nexcl + 1) - _rbreak(_nexcl));
169  double r0 = _rbreak(_nexcl);
170 
171  /* If the slope m is positive then the potential core
172  * will be attractive. So, artificially forcing core to be
173  * repulsive by setting m = -m
174  */
175  if (m > 0) {
176  cout << _name << " potential's extrapolated core is attractive!" << endl;
177  cout << "Artifically enforcing repulsive core.\n" << endl;
178  m *= -1.0;
179  }
180  // using linear extrapolation
181  // u(r) = ar + b
182  // a = m
183  // b = - m*r0 + u0
184  // m = (u1-u0)/(r1-r0)
185 
186  double a = m;
187  double b = -1.0 * m * r0 + u0;
188  for (int i = 0; i < _nexcl; i++) _lam(i) = a * _rbreak(i) + b;
189 }

201 double PotentialFunctionCBSPL::CalculateF(const double r) const {
202 
203  if (r <= _cut_off) {
204 
205  double u = 0.0;
206  int indx = min(int(r / _dr), _nbreak - 2);
207  ;
208  double rk = indx * _dr;
209  ;
210  double t = (r - rk) / _dr;
211 
212  Eigen::VectorXd R = Eigen::VectorXd::Zero(4);
213  R(0) = 1.0;
214  R(1) = t;
215  R(2) = t * t;
216  R(3) = t * t * t;
217  Eigen::VectorXd B = _lam.segment(indx, 4);
218  u += ((R.transpose() * _M) * B).value();
219  return u;
220 
221  } else
222  return 0.0;
223 }

225 // calculate first derivative w.r.t. ith parameter
226 double PotentialFunctionCBSPL::CalculateDF(const int i, const double r) const {
227 
228  // since first _nexcl parameters are not optimized for stability reasons
229 
230  if (r <= _cut_off) {
231 
232  unsigned int i_opt = i + _nexcl;
233  unsigned int indx;
234  double rk;
235 
236  indx = min(int((r) / _dr), _nbreak - 2);
237  rk = indx * _dr;
238 
239  if (i_opt >= indx && i_opt <= indx + 3) {
240 
241  Eigen::VectorXd R = Eigen::VectorXd::Zero(4);
242 
243  double t = (r - rk) / _dr;
244 
245  R(0) = 1.0;
246  R(1) = t;
247  R(2) = t * t;
248  R(3) = t * t * t;
249 
250  Eigen::VectorXd RM = R.transpose() * _M;
251 
252  return RM(i_opt - indx);
253 
254  } else
255  return 0.0;
256 
257  } else
258  return 0.0;
259 }
