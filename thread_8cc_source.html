39 void Thread::Start() {
40 
41  pthread_attr_init(&_attr);
42  /*
43  * according to the POSIX standard, threads are created in the joinable state
44  * by default
45  * however, some platforms do not obey
46  *
47  * explicitly create the thread in the joinable state
48  * the main program can then wait for the thread by calling
49  * pthread_join(thread)
50  *
51  */
52  pthread_attr_setdetachstate(&_attr, PTHREAD_CREATE_JOINABLE);
53  _finished = false;
54 
55  Index rc =
56  pthread_create(&_thread, &_attr, runwrapper, static_cast<void *>(this));
57  if (rc) {

59  boost::lexical_cast<std::string>(rc));
