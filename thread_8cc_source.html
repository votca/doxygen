40  void Thread::Start() {
41  pthread_attr_t attr;
42 
43  pthread_attr_init(&attr);
44  /*
45  * according to the POSIX standard, threads are created in the joinable state
46  * by default
47  * however, some platforms do not obey
48  *
49  * explicitly create the thread in the joinable state
50  * the main program can then wait for the thread by calling pthread_join(thread)
51  *
52  */
53  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
54  _finished = false;
55 
56  int rc = pthread_create(&_thread, &attr, runwrapper, (void *) this);
57  if (rc) {
58  throw std::runtime_error("ERROR; return code from pthread_create() is "
59  + boost::lexical_cast<std::string>(rc));
60  }
61 
62  }
