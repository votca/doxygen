00040         void Thread::Start() {
00041             pthread_attr_t attr;
00042 
00043             pthread_attr_init(&attr);
00044             /*
00045              * according to the POSIX standard, threads are created in the joinable state
00046              * by default
00047              * however, some platforms do not obey
00048              *
00049              * explicitly create the thread in the joinable state
00050              * the main program can then wait for the thread by calling pthread_join(thread)
00051              *
00052              */
00053             pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
00054             _finished = false;
00055 
00056             int rc = pthread_create(&_thread, &attr, runwrapper, (void *) this);
00057             if (rc) {
00058                 throw std::runtime_error("ERROR; return code from pthread_create() is "
00059                         + boost::lexical_cast<std::string>(rc));
00060             }
00061 
00062         }
