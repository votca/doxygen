35BOOST_AUTO_TEST_CASE(interpolate_test) {
36
37 int size = 80;
38 Eigen::VectorXd x = Eigen::VectorXd::Zero(size);
39 Eigen::VectorXd y = Eigen::VectorXd::Zero(size);
40 for (int i = 0; i < size; ++i) {
41 x(i) = 0.25 * i;
42 y(i) = std::sin(x(i));
43 }
44 AkimaSpline cspline;
45 cspline.setBCInt(0);
46 cspline.Interpolate(x, y);
47
48 Eigen::VectorXd rs = Eigen::VectorXd::Zero(10);
49 rs << 0.45, 0.47, 0.8, 0.75, 0.6, 0.4, 0.9, 0.55, 0, 0;
50 Eigen::VectorXd values_ref = Eigen::VectorXd::Zero(10);
51 values_ref << 0.434362, 0.452449, 0.717761, 0.681639, 0.564937, 0.388734,
52 0.783279, 0.52316, 0, 0;
53 Eigen::VectorXd derivatives_ref = Eigen::VectorXd::Zero(10);
54 derivatives_ref << 0.906562, 0.902216, 0.698382, 0.747323, 0.818045, 0.918909,
55 0.615268, 0.854075, 1.02038, 1.02038;
56 Eigen::VectorXd values = cspline.Calculate(rs);
57 Eigen::VectorXd derivatives = cspline.CalculateDerivative(rs);
58
59 bool equal_val = values_ref.isApprox(values, 1e-5);
60
61 if (!equal_val) {
62 std::cout << "result value" << std::endl;
63 std::cout << values.transpose() << std::endl;
64 std::cout << "ref value" << std::endl;
65 std::cout << values_ref.transpose() << std::endl;
66 }
67 BOOST_CHECK_EQUAL(equal_val, true);
68
69 bool equal_derivative = derivatives_ref.isApprox(derivatives, 1e-5);
70
71 if (!equal_derivative) {
72 std::cout << "result value" << std::endl;
73 std::cout << derivatives.transpose() << std::endl;
74 std::cout << "ref value" << std::endl;
75 std::cout << derivatives_ref.transpose() << std::endl;
76 }
77 BOOST_CHECK_EQUAL(equal_derivative, true);
78}
