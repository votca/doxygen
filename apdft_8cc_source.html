41 bool APDFT::Evaluate() {
42 
43  Orbitals orb;
44  orb.ReadFromCpt(_orbfile);
45  AOBasis basis = orb.SetupDftBasis();
46  Vxc_Grid grid;
47  grid.GridSetup(_grid_accuracy, orb.QMAtoms(), basis);
48 
49  DensityIntegration<Vxc_Grid> integration(grid);
50 
51  integration.IntegrateDensity(orb.DensityMatrixFull(_state));
52  std::vector<double> potential_values;
53  potential_values.reserve(orb.QMAtoms().size());
54  for (const auto &atom : orb.QMAtoms()) {
55  potential_values.push_back(integration.IntegratePotential(atom.getPos()));
56  }
57 
58  std::fstream outfile;
59  outfile.open(_outputfile, std::fstream::out);
60  outfile << "AtomId, Element, Potential[Hartree]" << std::endl;
61  for (Index i = 0; i < orb.QMAtoms().size(); i++) {
62  outfile << orb.QMAtoms()[i].getId() << " " << orb.QMAtoms()[i].getElement()
63  << " " << std::setprecision(14) << potential_values[i] << std::endl;
64  }
65  outfile.close();
66  return true;
