46 bool APDFT::Evaluate() {
47 
48  Orbitals orb;
49  orb.ReadFromCpt(_orbfile);
50  AOBasis basis = orb.SetupDftBasis();
51  Vxc_Grid grid;
52  grid.GridSetup(_grid_accuracy, orb.QMAtoms(), basis);
53 
54  DensityIntegration<Vxc_Grid> integration(grid);
55 
56  integration.IntegrateDensity(orb.DensityMatrixFull(_state));
57  std::vector<double> potential_values;
58  potential_values.reserve(orb.QMAtoms().size());
59  for (const auto &atom : orb.QMAtoms()) {
60  potential_values.push_back(integration.IntegratePotential(atom.getPos()));
61  }
62 
63  std::fstream outfile;
64  outfile.open(_outputfile, std::fstream::out);
65  outfile << "AtomId, Element, Potential[Hartree]" << std::endl;
66  for (Index i = 0; i < orb.QMAtoms().size(); i++) {
67  outfile << orb.QMAtoms()[i].getId() << " " << orb.QMAtoms()[i].getElement()
68  << " " << std::setprecision(14) << potential_values[i] << std::endl;
69  }
70  outfile.close();
71  return true;
72 }
