32  {

34  return make_pair(S,-1);

40  getline(in, word);
41  if ( word == " Two-electron integral symmetry is turned off."

44  in >> word;
45  sscanf(word.c_str(), "%i", &Nbasis);
46  cerr << "NBasis = " << Nbasis << endl;
47  }
48 
49  if (word.find("*** Overlap")!=string::npos)
50  {
51  S = new double * [Nbasis];
52 
53  for (int m=0;m<Nbasis;m++)
54  {
55  S[m] = new double[Nbasis]; //get mem for over lap matrix
56  }
57 
58  for (int j=0; j<Nbasis; j+=5) //go through the basis set in steps of 5
59  {
60  getline(in,word); //Get labels
61  int line=0;
62  for (int i=j; i<Nbasis; i++)
63  {
64  in >> word; //Get label

66  { //Don't have all fields
67  for (int l=0; l<=line; l++)
68  {
69  in >> word;

71  word=word.substr(0,pos)+"e"+word.substr(pos+1);
72  sscanf(word.c_str(), "%lf" , &S[j+l][i]);
73  S[i][j+l]=S[j+l][i];
74  }
75  }
76  else

79  {
