37BOOST_AUTO_TEST_CASE(bse_operator) {
38 libint2::initialize();
39 Orbitals orbitals;
40 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
41 "/bse/molecule.xyz");
42 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) + "/bse/3-21G.xml");
43 AOBasis aobasis = orbitals.getDftBasis();
44
45 orbitals.setNumberOfOccupiedLevels(4);
46 Eigen::MatrixXd& MOs = orbitals.MOs().eigenvectors();
47 MOs = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
48 std::string(XTP_TEST_DATA_FOLDER) + "/bse_operator/MOs.mm");
49
50 Eigen::MatrixXd Hqp = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
51 std::string(XTP_TEST_DATA_FOLDER) + "/bse_operator/Hqp.mm");
52
53 Eigen::VectorXd& mo_energy = orbitals.MOs().eigenvalues();
54 mo_energy = Eigen::VectorXd::Zero(17);
55 mo_energy << -0.612601, -0.341755, -0.341755, -0.341755, 0.137304, 0.16678,
56 0.16678, 0.16678, 0.671592, 0.671592, 0.671592, 0.974255, 1.01205,
57 1.01205, 1.01205, 1.64823, 19.4429;
58 Logger log;
59 TCMatrix_gwbse Mmn;
60 Mmn.Initialize(aobasis.AOBasisSize(), 0, 16, 0, 16);
61 Mmn.Fill(aobasis, aobasis, MOs);
62
63 Eigen::MatrixXd rpa_op = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
64 std::string(XTP_TEST_DATA_FOLDER) + "/bse_operator/rpa_op.mm");
65
66 Eigen::VectorXd epsilon_inv = Eigen::VectorXd::Zero(aobasis.AOBasisSize());
67 Mmn.MultiplyRightWithAuxMatrix(rpa_op);
68 epsilon_inv << 0.999807798016267, 0.994206065211371, 0.917916768047073,
69 0.902913813951883, 0.902913745974602, 0.902913584797742,
70 0.853352878674581, 0.853352727016914, 0.853352541699637, 0.79703468058566,
71 0.797034577207669, 0.797034400395582, 0.787701833916331,
72 0.518976361745313, 0.518975064844033, 0.518973712898761,
73 0.459286057710524;
74
75 BSEOperator_Options opt;
76 opt.cmax = 8;
77 opt.homo = 4;
78 opt.qpmin = 0;
79 opt.rpamin = 0;
80 opt.vmin = 0;
81
82 orbitals.setBSEindices(0, 16);
83 HqpOperator Hqp_op(epsilon_inv, Mmn, Hqp);
84 Hqp_op.configure(opt);
85 const Eigen::MatrixXd identity =
86 Eigen::MatrixXd::Identity(Hqp_op.rows(), Hqp_op.cols());
87 Eigen::MatrixXd hqp_mat = Hqp_op * identity;
88
89 Eigen::MatrixXd hqp_ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(

119
