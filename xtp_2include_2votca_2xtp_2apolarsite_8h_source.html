404  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
405  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
406  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
407  e12 = pol2.getPos() - pol1.getPos();
408  R = 1 / abs(e12);
409  R2 = R * R;
410  R3 = R2 * R;
411  R4 = R3 * R;
412  R5 = R4 * R;
413  e12 *= R;
414 
415  // Thole damping init.
416  // double P_proj_1 = fabs(e12 * pol1.pax) * pol1.eigenpxx
417  // + fabs(e12 * pol1.pay) * pol1.eigenpyy
418  // + fabs(e12 * pol1.paz) * pol1.eigenpzz;
419  // double P_proj_2 = fabs(e12 * pol2.pax) * pol2.eigenpxx
420  // + fabs(e12 * pol2.pay) * pol2.eigenpyy
421  // + fabs(e12 * pol2.paz) * pol2.eigenpzz;
422 
423  // u3 = 1 / (R3 * sqrt(pol1.getIsoP() * pol2.getIsoP()));
424  u3 = 1 / (R3 * sqrt(1. / 3. *
425  (pol1.Pxx * pol2.Pxx + pol1.Pyy * pol2.Pyy +
426  pol1.Pzz * pol2.Pzz)));
427 
428  // rax = pol1._locX * e12;
429  // ray = pol1._locY * e12;
430  // raz = pol1._locZ * e12;
431  // rbx = - pol2._locX * e12;
432  // rby = - pol2._locY * e12;
433  // rbz = - pol2._locZ * e12;
434 
435  rax = e12.getX();
436  ray = e12.getY();
437  raz = e12.getZ();
438  rbx = -rax;
439  rby = -ray;
440  rbz = -raz;
441 
442  // cxx = pol1._locX * pol2._locX;
443  // cxy = pol1._locX * pol2._locY;
444  // cxz = pol1._locX * pol2._locZ;
445  // cyx = pol1._locY * pol2._locX;
446  // cyy = pol1._locY * pol2._locY;
447  // cyz = pol1._locY * pol2._locZ;

449  // czy = pol1._locZ * pol2._locY;

453  cxy = 0;
