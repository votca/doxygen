34BOOST_AUTO_TEST_CASE(aomatrices_dipole_test) {
35 libint2::initialize();
36 Orbitals orbitals;
37 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
38 "/aomatrix3d/molecule.xyz");
39 BasisSet basis;
40 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/aomatrix3d/3-21G.xml");
41 AOBasis aobasis;
42 aobasis.Fill(basis, orbitals.QMAtoms());
43
44 AODipole dip;
45 dip.Fill(aobasis);
46
47 std::array<Eigen::MatrixXd, 3> dip_ref;
48 for (unsigned i = 0; i < dip_ref.size(); i++) {
49 dip_ref[i] = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
50 std::string(XTP_TEST_DATA_FOLDER) + "/aomatrix3d/dip_ref" +
51 std::to_string(i) + ".mm");
52 }
53
54 for (unsigned i = 0; i < dip_ref.size(); i++) {
55 bool check_dip = dip.Matrix()[i].isApprox(dip_ref[i], 0.0001);
56 BOOST_CHECK_EQUAL(check_dip, true);
57 if (!check_dip) {
58 cout << "ref" << i << endl;
59 cout << dip_ref[i] << endl;
60 cout << "result" << i << endl;
61 cout << dip.Matrix()[i] << endl;
62 }
63 }
64
65 // Now the same computation with a non zero center
66 Eigen::Vector3d center;
67 center << 1.71, 2.34, 3.54;
68 dip.setCenter(center);
69 dip.Fill(aobasis);
70
71 for (unsigned i = 0; i < dip_ref.size(); i++) {
72 dip_ref[i] = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
73 std::string(XTP_TEST_DATA_FOLDER) + "/aomatrix3d/dip_ref" +
74 std::to_string(i) + "_nonzero.mm");
75 }
76
77 for (unsigned i = 0; i < dip_ref.size(); i++) {
78 bool check_nonzero_center = dip.Matrix()[i].isApprox(dip_ref[i], 0.0001);
79 BOOST_CHECK_EQUAL(check_nonzero_center, true);
80 if (!check_nonzero_center) {
81 cout << "ref" << i << endl;
82 cout << dip_ref[i] << endl;
83 cout << "result" << i << endl;
84 cout << dip.Matrix()[i] << endl;
85 }
86 }
87
88 libint2::finalize();
89}
