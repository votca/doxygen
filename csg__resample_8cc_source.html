51 int main(int argc, char** argv)
52 {
53 
54  string in_file, out_file, grid, fitgrid, comment, type, boundaries;
55  Spline *spline=NULL;
56  Table in, out, der;
57  // program options
58  po::options_description desc("Allowed options");
59 
60  desc.add_options()
61  ("help", "produce this help message")
62  ("in", po::value<string>(&in_file), "table to read")
63  ("out", po::value<string>(&out_file), "table to write")
64  ("derivative", po::value<string>(), "table to write")
65  ("grid", po::value<string>(&grid), "new grid spacing (min:step:max). If 'grid' is specified only, interpolation is performed.")
66  ("type", po::value<string>(&type)->default_value("akima"), "[cubic|akima|linear]. If option is not specified, the default type 'akima' is assumed.")
67  ("fitgrid", po::value<string>(&fitgrid), "specify fit grid (min:step:max). If 'grid' and 'fitgrid' are specified, a fit is performed.")
68  ("nocut", "Option for fitgrid: Normally, values out of fitgrid boundaries are cut off. If they shouldn't, choose --nocut.")
69  ("comment", po::value<string>(&comment), "store a comment in the output table")
70  ("boundaries", po::value<string>(&boundaries), "(natural|periodic|derivativezero) sets boundary conditions");
71 
72  po::variables_map vm;
73  try {
74  po::store(po::parse_command_line(argc, argv, desc), vm);
75  po::notify(vm);
76  }
77  catch(po::error err) {
78  cout << "error parsing command line: " << err.what() << endl;
79  return -1;
80  }
81 
82  try {
83  // does the user want help?
84  if (vm.count("help")) {
85  help_text();
86  cout << desc << endl;
87  return 0;
88  }
89 
90  check_option(desc, vm, "in");
91  check_option(desc, vm, "out");
92 
93  if(!(vm.count("grid") || vm.count("fitgrid"))) {
94  cout << "Need grid for interpolation or fitgrid for fit.\n";
95  return 1;
96  }
97 
98  if((!vm.count("grid")) && vm.count("fitgrid")) {
99  cout << "Need a grid for fitting as well.\n";
100  return 1;
101  }
102 
103 
104  double min, max, step;
105  {
106  Tokenizer tok(grid, ":");
107  vector<string> toks;
108  tok.ToVector(toks);
109  if(toks.size()!=3) {
110  cout << "wrong range format, use min:step:max\n";
111  return 1;
112  }
113  min = boost::lexical_cast<double>(toks[0]);
114  step = boost::lexical_cast<double>(toks[1]);
115  max = boost::lexical_cast<double>(toks[2]);
116  }
117 
118 
119  in.Load(in_file);
120 
121  if (vm.count("type")) {
122  if(type=="cubic") {
123  spline = new CubicSpline();
124  }
125  else if(type=="akima") {
126  spline = new AkimaSpline();
127  }
128  else if(type=="linear") {
129  spline = new LinSpline();
130  }
131  else {
132  throw std::runtime_error("unknown type");
133  }
134  }
135  spline->setBC(Spline::splineNormal);
136 
137 
138  if (vm.count("boundaries")) {
139  if(boundaries=="periodic") {
140  spline->setBC(Spline::splinePeriodic);
141  }
142  if(boundaries=="derivativezero") {
143  spline->setBC(Spline::splineDerivativeZero);
144  }
145  //default: normal
146  }
147 
148 
149  // in case fit is specified
150  if (vm.count("fitgrid")) {
151  Tokenizer tok(fitgrid, ":");
152  vector<string> toks;
153  tok.ToVector(toks);
154  if(toks.size()!=3) {
155  cout << "wrong range format in fitgrid, use min:step:max\n";
156  return 1;
157  }
158  double sp_min, sp_max, sp_step;
159  sp_min = boost::lexical_cast<double>(toks[0]);
160  sp_step = boost::lexical_cast<double>(toks[1]);
161  sp_max = boost::lexical_cast<double>(toks[2]);
162  cout << "doing " << type << " fit " << sp_min << ":" << sp_step << ":" << sp_max << endl;
163 
164  // cut off any values out of fitgrid boundaries (exception: do nothing in case of --nocut)
165  ub::vector<double> x_copy;
166  ub::vector<double> y_copy;
167  if (!vm.count("nocut")) {
168  // determine vector size
169  int minindex=-1, maxindex=-1;
170  for (size_t i=0; i<in.x().size(); i++) {
171  if(in.x(i)<sp_min) {
172  minindex = i;
173  }
174  if(in.x(i)<sp_max) {
175  maxindex = i;
176  }
177  }
178  // copy data values in [sp_min,sp_max] into new vectors
179  minindex++;
180  x_copy = ub::zero_vector<double>(maxindex-minindex+1);
181  y_copy = ub::zero_vector<double>(maxindex-minindex+1);
182  for (int i=minindex; i<=maxindex; i++) {
183  x_copy(i-minindex) = in.x(i);
184  y_copy(i-minindex) = in.y(i);
185  }
186  }
187 
188  // fitting
189  spline->GenerateGrid(sp_min, sp_max, sp_step);
190  try {
191  if (vm.count("nocut")) {
192  spline->Fit(in.x(), in.y());
193  } else {
194  spline->Fit(x_copy, y_copy);
195  }
196  } catch (const char* message) {
197  if(strcmp("qrsolve_zero_column_in_matrix",message)) {
198  throw std::runtime_error("error in Linalg::linalg_qrsolve : Not enough data for fit, please adjust grid (zero row in fit matrix)");
199  }
200  else if(strcmp("constrained_qrsolve_zero_column_in_matrix",message)) {
201  throw std::runtime_error("error in Linalg::linalg_constrained_qrsolve : Not enough data for fit, please adjust grid (zero row in fit matrix)");
202  }
203  else throw std::runtime_error("Unknown error in csg_resample while fitting.");
204  }
205  } else {
206  // otherwise do interpolation (default = cubic)
207  try {
208  spline->Interpolate(in.x(), in.y());
209  } catch (const char* message) {
210  if(strcmp("qrsolve_zero_column_in_matrix",message)) {
211  throw std::runtime_error("error in Linalg::linalg_qrsolve : Not enough data, please adjust grid (zero row in fit matrix)");
212  }
213  else if(strcmp("constrained_qrsolve_zero_column_in_matrix",message)) {
214  throw std::runtime_error("error in Linalg::linalg_constrained_qrsolve : Not enough data, please adjust grid (zero row in fit matrix)");
215  }
216  else throw std::runtime_error("Unknown error in csg_resample while interpolating.");
217  }
218  }
219 
220 
221  out.GenerateGridSpacing(min, max, step);
222  spline->Calculate(out.x(), out.y());
223 
224 
225  //store a comment line
226  if (vm.count("comment")){
227  out.set_comment(comment);
228  }
229  out.y() = out.y();
230  out.flags() = ub::scalar_vector<double>(out.flags().size(), 'o');
231 
232  der.GenerateGridSpacing(min, max, step);
233  der.flags() = ub::scalar_vector<double>(der.flags().size(), 'o');
234 
235  unsigned int i=0;
236  for(i=0; out.x(i) < in.x(0) && i<out.size(); ++i);
237 
238  unsigned int j=0;
239  for(;i < out.size(); ++i) {
240  for(; j < in.size(); ++j)
241  if(in.x(j) >= out.x(i) || fabs(in.x(j)-out.x(i) ) < 1e-12) // fix for precison errors
242  break;
243  if(in.size() == j) break;
244  out.flags(i) = in.flags(j);
245  der.flags(i) = in.flags(j);
246  }
247 
248  out.Save(out_file);
249 
250  if (vm.count("derivative")) {
251  spline->CalculateDerivative(der.x(), der.y());
252 
253  der.Save(vm["derivative"].as<string>());
254  }
255 
256  delete spline;
257  }
258  catch(std::exception &error) {
259  cerr << "an error occurred:\n" << error.what() << endl;
260  return -1;
261  }
262  return 0;
263 }
