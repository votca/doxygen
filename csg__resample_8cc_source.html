50 int main(int argc, char **argv) {
51 
52  string in_file, out_file, grid, fitgrid, comment, type, boundaries;
53 
54  Table in, out, der;
55  // program options
56  po::options_description desc("Allowed options");
57 
58  desc.add_options()("help", "produce this help message")(
59  "in", po::value<string>(&in_file), "table to read")(
60  "out", po::value<string>(&out_file), "table to write")(
61  "derivative", po::value<string>(), "table to write")(
62  "grid", po::value<string>(&grid),
63  "new grid spacing (min:step:max). If 'grid' is specified only, "
64  "interpolation is performed.")(
65  "type", po::value<string>(&type)->default_value("akima"),
66  "[cubic|akima|linear]. If option is not specified, the default type "
67  "'akima' is assumed.")("fitgrid", po::value<string>(&fitgrid),
68  "specify fit grid (min:step:max). If 'grid' and "
69  "'fitgrid' are specified, a fit is performed.")(
70  "nocut",
71  "Option for fitgrid: Normally, values out of fitgrid boundaries are cut "
72  "off. If they shouldn't, choose --nocut.")(
73  "comment", po::value<string>(&comment),
74  "store a comment in the output table")(
75  "boundaries", po::value<string>(&boundaries),
76  "(natural|periodic|derivativezero) sets boundary conditions");
77 
78  po::variables_map vm;
79  try {
80  po::store(po::parse_command_line(argc, argv, desc), vm);
81  po::notify(vm);
82  } catch (po::error &err) {
83  cout << "error parsing command line: " << err.what() << endl;
84  return -1;
85  }
86 
87  try {
88  // does the user want help?
89  if (vm.count("help")) {
90  help_text();
91  cout << desc << endl;
92  return 0;
93  }
94 
95  check_option(desc, vm, "in");
96  check_option(desc, vm, "out");
97 
98  if (!(vm.count("grid") || vm.count("fitgrid"))) {
99  cout << "Need grid for interpolation or fitgrid for fit.\n";
100  return 1;
101  }
102 
103  if ((!vm.count("grid")) && vm.count("fitgrid")) {
104  cout << "Need a grid for fitting as well.\n";
105  return 1;
106  }
107 
108  double min, max, step;
109  {
110  Tokenizer tok(grid, ":");
111  vector<string> toks;
112  tok.ToVector(toks);
113  if (toks.size() != 3) {
114  cout << "wrong range format, use min:step:max\n";
115  return 1;
116  }
117  min = stod(toks[0]);
118  step = stod(toks[1]);
119  max = stod(toks[2]);
120  }
121 
122  in.Load(in_file);
123  std::unique_ptr<Spline> spline;
124  if (vm.count("type")) {
125  if (type == "cubic") {
126  spline = std::make_unique<CubicSpline>(CubicSpline());
127  } else if (type == "akima") {
128  spline = std::make_unique<AkimaSpline>(AkimaSpline());
129  } else if (type == "linear") {
130  spline = std::make_unique<LinSpline>(LinSpline());
131  } else {
132  throw std::runtime_error("unknown spline type:" + type);
133  }
134  }
135  spline->setBC(Spline::splineNormal);
136 
137  if (vm.count("boundaries")) {
138  if (boundaries == "periodic") {
139  spline->setBC(Spline::splinePeriodic);
140  }
141  if (boundaries == "derivativezero") {
142  spline->setBC(Spline::splineDerivativeZero);
143  }
144  // default: normal
145  }
146 
147  // in case fit is specified
148  if (vm.count("fitgrid")) {
149  Tokenizer tok(fitgrid, ":");
150  vector<string> toks;
151  tok.ToVector(toks);
152  if (toks.size() != 3) {
153  cout << "wrong range format in fitgrid, use min:step:max\n";
154  return 1;
155  }
156  double sp_min, sp_max, sp_step;
157  sp_min = stod(toks[0]);
158  sp_step = stod(toks[1]);
159  sp_max = stod(toks[2]);
160  cout << "doing " << type << " fit " << sp_min << ":" << sp_step << ":"
161  << sp_max << endl;
162 
163  // cut off any values out of fitgrid boundaries (exception: do nothing in
164  // case of --nocut)
165  Eigen::VectorXd x_copy;
166  Eigen::VectorXd y_copy;
167  if (!vm.count("nocut")) {
168  // determine vector size
169  votca::Index minindex = -1, maxindex = -1;
170  for (votca::Index i = 0; i < in.x().size(); i++) {
171  if (in.x(i) < sp_min) {
172  minindex = i;
173  }
174  if (in.x(i) < sp_max) {
175  maxindex = i;
176  }
177  }
178  // copy data values in [sp_min,sp_max] into new vectors
179  minindex++;
180  x_copy = Eigen::VectorXd::Zero(maxindex - minindex + 1);
181  y_copy = Eigen::VectorXd::Zero(maxindex - minindex + 1);
182  for (votca::Index i = minindex; i <= maxindex; i++) {
183  x_copy(i - minindex) = in.x(i);
184  y_copy(i - minindex) = in.y(i);
185  }
186  }
187 
188  // fitting
189  spline->GenerateGrid(sp_min, sp_max, sp_step);
190  try {
191  if (vm.count("nocut")) {
192  spline->Fit(in.x(), in.y());
193  } else {
194  spline->Fit(x_copy, y_copy);
195  }
196  } catch (const char *message) {
197  if (strcmp("qrsolve_zero_column_in_matrix", message)) {
198  throw std::runtime_error(
199  "error in Linalg::linalg_qrsolve : Not enough data for fit, "
200  "please adjust grid (zero row in fit matrix)");
201  } else if (strcmp("constrained_qrsolve_zero_column_in_matrix",
202  message)) {
203  throw std::runtime_error(
204  "error in Linalg::linalg_constrained_qrsolve : Not enough data "
205  "for fit, please adjust grid (zero row in fit matrix)");
206  } else {
207  throw std::runtime_error(
208  "Unknown error in csg_resample while fitting.");
209  }
210  }
211  } else {
212  // otherwise do interpolation (default = cubic)
213  try {
214  spline->Interpolate(in.x(), in.y());
215  } catch (const char *message) {
216  if (strcmp("qrsolve_zero_column_in_matrix", message)) {
217  throw std::runtime_error(
218  "error in Linalg::linalg_qrsolve : Not enough data, please "
219  "adjust grid (zero row in fit matrix)");
220  } else if (strcmp("constrained_qrsolve_zero_column_in_matrix",
221  message)) {
222  throw std::runtime_error(
223  "error in Linalg::linalg_constrained_qrsolve : Not enough data, "
224  "please adjust grid (zero row in fit matrix)");
225  } else {
226  throw std::runtime_error(
227  "Unknown error in csg_resample while interpolating.");
228  }
229  }
230  }
231 
232  out.GenerateGridSpacing(min, max, step);
233  out.y() = spline->Calculate(out.x());
234 
235  // store a comment line
236  if (vm.count("comment")) {
237  out.set_comment(comment);
238  }
239  out.flags() = std::vector<char>(out.flags().size(), 'o');
240 
241  der.GenerateGridSpacing(min, max, step);
242  der.flags() = std::vector<char>(der.flags().size(), 'o');
243 
244  votca::Index i = 0;
245  for (i = 0; out.x(i) < in.x(0) && i < out.size(); ++i) {
246  ;
247  }
248 
249  votca::Index j = 0;
250  for (; i < out.size(); ++i) {
251  for (; j < in.size(); ++j) {
252  if (in.x(j) >= out.x(i) ||
253  std::abs(in.x(j) - out.x(i)) < 1e-12) { // fix for precision errors
254  break;
255  }
256  }
257  if (in.size() == j) {
258  break;
259  }
260  out.flags(i) = in.flags(j);
261  der.flags(i) = in.flags(j);
262  }
263 
264  out.Save(out_file);
265 
266  if (vm.count("derivative")) {
267  der.y() = spline->CalculateDerivative(der.x());
268 
269  der.Save(vm["derivative"].as<string>());
270  }
271 
272  } catch (std::exception &error) {
273  cerr << "an error occurred:\n" << error.what() << endl;
274  return -1;
275  }
276  return 0;
277 }
