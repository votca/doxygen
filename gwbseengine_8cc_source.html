56  _do_dft_input = false;

60 
61  if (_tasks_string.find("guess") != std::string::npos) _do_guess = true;
62  if (_tasks_string.find("input") != std::string::npos) _do_dft_input = true;
63  if (_tasks_string.find("dft") != std::string::npos) _do_dft_run = true;
64  if (_tasks_string.find("parse") != std::string::npos) _do_dft_parse = true;
65  if (_tasks_string.find("gwbse") != std::string::npos) _do_gwbse = true;
66

69  load_property_from_xml(_gwbse_options, _gwbse_xml.c_str());
70 
71  // DFT log and MO file names
72  _MO_file = options->get(".mofile").as<string> ();
73  _dftlog_file = options->get(".dftlog").as<string> ();
74 
75  // Logger redirection
76  _redirect_logger = options->ifExistsReturnElseReturnDefault<bool>(".redirect_logger", false);
77  _logger_file = "gwbse.log";
78 
79  // for requested merged guess, two archived orbitals objects are needed
80  if ( _do_guess ){
81  _guess_archiveA = options->ifExistsReturnElseThrowRuntimeError<string>(".archiveA");

83  }

94  void GWBSEENGINE::ExcitationEnergies(QMPackage* _qmpackage, vector<ctp::Segment*> _segments, Orbitals* _orbitals) {
95 
96 
97  //redirect log, if required
98  // define own logger for GW-BSE that is written into a runFolder logfile
99  ctp::Logger _gwbse_engine_logger(_pLog->getReportLevel());
100  if (_redirect_logger) {
101  _gwbse_engine_logger.setMultithreading(false);
102  _gwbse_engine_logger.setPreface(ctp::logINFO, (format("\n ...")).str());
103  _gwbse_engine_logger.setPreface(ctp::logERROR, (format("\n ...")).str());
104  _gwbse_engine_logger.setPreface(ctp::logWARNING, (format("\n ...")).str());
105  _gwbse_engine_logger.setPreface(ctp::logDEBUG, (format("\n ...")).str());
106  _qmpackage->setLog(&_gwbse_engine_logger);
107  }
108 
109  if (_do_dft_input) {
110 
111  // required for merged guess
112  Orbitals *_orbitalsAB = NULL;
113  if (_qmpackage->GuessRequested() && _do_guess) { // do not want to do an SCF loop for a dimer
114  if (_redirect_logger) {
115  CTP_LOG_SAVE(ctp::logINFO, _gwbse_engine_logger) << "Guess requested, reading molecular orbitals" << flush;
116  } else {
117  CTP_LOG_SAVE(ctp::logINFO, *_pLog) << "Guess requested, reading molecular orbitals" << flush;
118  }
119  Orbitals _orbitalsA, _orbitalsB;
120  _orbitalsAB = new Orbitals();
121  // load the corresponding monomer orbitals and prepare the dimer guess
122 
123  // failed to load; wrap-up and finish current job
124  if (!_orbitalsA.Load(_guess_archiveA)) {
125  throw runtime_error(string("Do input: failed loading orbitals from ") + _guess_archiveA);
126  }
127 
128  if (!_orbitalsB.Load(_guess_archiveB)) {
129  throw runtime_error(string("Do input: failed loading orbitals from ") + _guess_archiveB);
130 
131  }
132 
133  _orbitals->PrepareGuess(&_orbitalsA, &_orbitalsB, _orbitalsAB);
134 
135  }
136 
137  _qmpackage->WriteInputFile(_segments, _orbitalsAB);
138  }
139 
140  if (_do_dft_run) {
141 
142  bool run_success = _qmpackage->Run( _orbitals );
143  if (!run_success) {
144  throw runtime_error(string("\n GW-BSE without DFT is difficult. Stopping!"));
145  }
146  }
147 
148  // parse DFT data, if required
149  if (_do_dft_parse) {
150  if ( _qmpackage->getPackageName() != "xtp" ) {
151  if (_redirect_logger) {
152  CTP_LOG_SAVE(ctp::logINFO, _gwbse_engine_logger) << "Parsing DFT data from " << _dftlog_file << " and " << _MO_file << flush;
153  } else {
154  CTP_LOG_SAVE(ctp::logINFO, *_pLog) << "Parsing DFT data from " << _dftlog_file << " and " << _MO_file << flush;
155  }
156  _qmpackage->setLogFileName(_dftlog_file);
157  _qmpackage->setOrbitalsFileName(_MO_file);
158  _qmpackage->ParseLogFile(_orbitals);
159 
160  _qmpackage->ParseOrbitalsFile(_orbitals);
161  }
162  _orbitals->setDFTbasis(_qmpackage->getBasisSetName());
163  }
164 
165  // if no parsing of DFT data is requested, reload serialized orbitals object
166  if (!_do_dft_parse && _do_gwbse) {
167  if (_redirect_logger) {
168  CTP_LOG_SAVE(ctp::logINFO, _gwbse_engine_logger) << "Loading serialized data from " << _archive_file << flush;
169  } else {
170  CTP_LOG_SAVE(ctp::logINFO, *_pLog) << "Loading serialized data from " << _archive_file << flush;
171  }
172  _orbitals->Load(_archive_file);
173  }
174 
175  if (_do_gwbse) {
176  GWBSE _gwbse = GWBSE(_orbitals);
177  _gwbse.setLogger(_pLog);
178  if (_redirect_logger) _gwbse.setLogger(&_gwbse_engine_logger);
179  _gwbse.Initialize(&_gwbse_options);
180  _gwbse.Evaluate();
181  if (_redirect_logger) SaveRedirectedLogger(&_gwbse_engine_logger);
182  Property *_output_summary = &(_summary.add("output", ""));
183  _gwbse.addoutput(_output_summary);
184  }
185  return;
186  }
