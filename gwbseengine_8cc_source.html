42void GWBSEEngine::Initialize(tools::Property& options,
43 std::string archive_filename) {
44
45 archive_file_ = archive_filename;
46
47 std::string tasks_string = options.get(".tasks").as<std::string>();
48
49 // We split either on a space or a comma
50 tools::Tokenizer tokenizedTasks(tasks_string, " ,");
51 std::vector<std::string> tasks = tokenizedTasks.ToVector();
52 // Check which tasks exist and set them to true
53 do_guess_ = std::find(tasks.begin(), tasks.end(), "guess") != tasks.end();
54 do_dft_input_ = std::find(tasks.begin(), tasks.end(), "input") != tasks.end();
55 do_dft_run_ = std::find(tasks.begin(), tasks.end(), "dft") != tasks.end();
56 do_dft_parse_ = std::find(tasks.begin(), tasks.end(), "parse") != tasks.end();
57 do_gwbse_ = std::find(tasks.begin(), tasks.end(), "gwbse") != tasks.end();
58 do_localize_ =
59 std::find(tasks.begin(), tasks.end(), "localize") != tasks.end();
60 do_dft_in_dft_ =
61 std::find(tasks.begin(), tasks.end(), "dft_in_dft") != tasks.end();
62
63 // XML option file for GWBSE
64 if (do_gwbse_) {
65 gwbse_options_ = options.get(".gwbse");
66 }
67 if (do_localize_) {
68 localize_options_ = options.get(".localize");
69 }
70 if (do_dft_in_dft_ && !do_localize_) {
71 throw std::runtime_error(
72 "Can't do DFT in DFT embedding without localization");
73 }
74 // DFT log and MO file names
75 MO_file_ = qmpackage_->getMOFile();
76 dftlog_file_ = qmpackage_->getLogFile();
77
78 // Logger redirection
79 if (options.exists(".logging_file")) {
80 logger_file_ = options.get(".logging_file").as<std::string>();
81 }
82
83 // for requested merged guess, two archived orbitals objects are needed
84 if (do_guess_) {
85 guess_archiveA_ = options.get(".archiveA").as<std::string>();
86 guess_archiveB_ = options.get(".archiveB").as<std::string>();
87 }
88
89 return;
90}

97void GWBSEEngine::ExcitationEnergies(Orbitals& orbitals) {
98
99 // redirect log, if required
100 // define own logger for GW-BSE that is written into a runFolder logfile
101 Logger gwbse_engine_logger(pLog_->getReportLevel());
102 Logger* logger = pLog_;
103 if (!logger_file_.empty()) {
104 gwbse_engine_logger.setMultithreading(false);
105 gwbse_engine_logger.setPreface(Log::info, "\n ...");
106 gwbse_engine_logger.setPreface(Log::error, "\n ...");
107 gwbse_engine_logger.setPreface(Log::warning, "\n ...");
108 gwbse_engine_logger.setPreface(Log::debug, "\n ...");
109 logger = &gwbse_engine_logger;
110 }
111 qmpackage_->setLog(logger);
112 if (do_dft_input_) {
113 // required for merged guess
114 if (qmpackage_->GuessRequested() && do_guess_) { // do not want to do an
115 // SCF loop for a dimer
116 XTP_LOG(Log::error, *logger)
117 << "Guess requested, reading molecular orbitals" << flush;
118 Orbitals orbitalsA, orbitalsB;
119 orbitalsA.ReadFromCpt(guess_archiveA_);
120 orbitalsB.ReadFromCpt(guess_archiveB_);
121 orbitals.PrepareDimerGuess(orbitalsA, orbitalsB);
122 }
123 qmpackage_->WriteInputFile(orbitals);
124 }
125 if (do_dft_run_) {
126 bool run_success = qmpackage_->Run();
127 if (!run_success) {
128 throw std::runtime_error("\n DFT-run failed. Stopping!");
129 }
130 }
131
132 // parse DFT data, if required
133 if (do_dft_parse_) {
134 XTP_LOG(Log::error, *logger) << "Parsing DFT data from " << dftlog_file_
135 << " and " << MO_file_ << flush;
136 qmpackage_->setLogFileName(dftlog_file_);
137 qmpackage_->setMOsFileName(MO_file_);
138
139 bool Logfile_parse = qmpackage_->ParseLogFile(orbitals);
140 if (!Logfile_parse) {
141 throw std::runtime_error("\n Parsing DFT logfile " + dftlog_file_ +
142 " failed. Stopping!");
143 }
144 bool Orbfile_parse = qmpackage_->ParseMOsFile(orbitals);
145 if (!Orbfile_parse) {
146 throw std::runtime_error("\n Parsing DFT orbfile " + MO_file_ +
147 " failed. Stopping!");
148 }
149 qmpackage_->CleanUp();
150 }
151
152 if (do_localize_) {
153 PMLocalization pml(*logger, localize_options_);
154 pml.computePML(orbitals);
155 }
156
157 if (do_dft_in_dft_) {
158 qmpackage_->WriteInputFile(orbitals);
159 bool run_success = qmpackage_->RunActiveRegion();
160 if (!run_success) {
161 throw std::runtime_error("\n DFT in DFT embedding failed. Stopping!");
162 }
163 bool Logfile_parse = qmpackage_->ParseLogFile(orbitals);
164 if (!Logfile_parse) {
165 throw std::runtime_error("\n Parsing DFT logfile " + dftlog_file_ +
166 " failed. Stopping!");
167 }
168 bool Orbfile_parse = qmpackage_->ParseMOsFile(orbitals);
169 if (!Orbfile_parse) {
170 throw std::runtime_error("\n Parsing DFT orbfile " + MO_file_ +
171 " failed. Stopping!");
172 }
173 }
174
175 // if no parsing of DFT data is requested, reload serialized orbitals object
176 if (!do_dft_parse_ && do_gwbse_) {
177 XTP_LOG(Log::error, *logger)
178 << "Loading serialized data from " << archive_file_ << flush;
179 orbitals.ReadFromCpt(archive_file_);
180 }
181 tools::Property& output_summary = summary_.add("output", "");
182
183 if (do_dft_in_dft_ && do_gwbse_) {
184 Orbitals orb_embedded = orbitals;
185 orb_embedded.MOs() = orb_embedded.getEmbeddedMOs();
186 Index active_electrons = orb_embedded.getNumOfActiveElectrons();
187 orb_embedded.setNumberOfAlphaElectrons(active_electrons);
188 orb_embedded.setNumberOfOccupiedLevels(active_electrons / 2);
189 GWBSE gwbse = GWBSE(orb_embedded);
190 gwbse.setLogger(logger);
191 gwbse.Initialize(gwbse_options_);
192 gwbse.Evaluate();
193 gwbse.addoutput(output_summary);
194 }
195
196 if (do_gwbse_ && !do_dft_in_dft_) {
197 GWBSE gwbse = GWBSE(orbitals);
198 gwbse.setLogger(logger);
199 gwbse.Initialize(gwbse_options_);
200 gwbse.Evaluate();
201 gwbse.addoutput(output_summary);
202 }
203
204 // if do truncatedgwbse
205 if (!logger_file_.empty()) {
206 WriteLoggerToFile(logger);
207 }
208 return;
209}

211void GWBSEEngine::WriteLoggerToFile(Logger* pLog) {
212 std::ofstream ofs;
213 ofs.open(logger_file_, std::ofstream::out);
214 if (!ofs.is_open()) {
215 throw std::runtime_error("Bad file handle: " + logger_file_);
216 }
217 ofs << (*pLog) << std::endl;
218 ofs.close();
219 return;
220}
