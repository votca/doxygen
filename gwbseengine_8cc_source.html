37 void GWBSEEngine::Initialize(tools::Property& options,
38  std::string archive_filename) {
39 
40  _archive_file = archive_filename;
41 
42  std::string tasks_string = options.get(".tasks").as<std::string>();
43 
44  if (tasks_string.find("guess") != std::string::npos) {
45  _do_guess = true;
46  }
47  if (tasks_string.find("input") != std::string::npos) {
48  _do_dft_input = true;
49  }
50  if (tasks_string.find("dft") != std::string::npos) {
51  _do_dft_run = true;
52  }
53  if (tasks_string.find("parse") != std::string::npos) {
54  _do_dft_parse = true;
55  }
56  if (tasks_string.find("gwbse") != std::string::npos) {
57  _do_gwbse = true;
58  }
59 
60  // XML option file for GWBSE
61  if (_do_gwbse) {
62  std::string _gwbse_xml = options.get(".gwbse_options").as<std::string>();
63  _gwbse_options.LoadFromXML(_gwbse_xml);
64  }
65  // DFT log and MO file names
66  _MO_file = options.get(".mofile").as<std::string>();
67  _dftlog_file = options.get(".dftlog").as<std::string>();
68 
69  // Logger redirection
70  _redirect_logger = options.ifExistsReturnElseReturnDefault<bool>(
71  ".redirect_logger", _redirect_logger);
72  _logger_file = "gwbse.log";
73 
74  // for requested merged guess, two archived orbitals objects are needed
75  if (_do_guess) {
76  _guess_archiveA =
77  options.ifExistsReturnElseThrowRuntimeError<std::string>(".archiveA");
78  _guess_archiveB =
79  options.ifExistsReturnElseThrowRuntimeError<std::string>(".archiveB");
80  }
81 
82  return;
83 }

90 void GWBSEEngine::ExcitationEnergies(Orbitals& orbitals) {
91 
92  // redirect log, if required
93  // define own logger for GW-BSE that is written into a runFolder logfile
94  Logger gwbse_engine_logger(_pLog->getReportLevel());
95  Logger* logger = _pLog;
96  if (_redirect_logger) {
97  gwbse_engine_logger.setMultithreading(false);
98  gwbse_engine_logger.setPreface(Log::info, "\n ...");
99  gwbse_engine_logger.setPreface(Log::error, "\n ...");
100  gwbse_engine_logger.setPreface(Log::warning, "\n ...");
101  gwbse_engine_logger.setPreface(Log::debug, "\n ...");
102  logger = &gwbse_engine_logger;
103  }
104  _qmpackage->setLog(logger);
105  if (_do_dft_input) {
106  // required for merged guess
107  if (_qmpackage->GuessRequested() && _do_guess) { // do not want to do an
108  // SCF loop for a dimer
109  XTP_LOG(Log::error, *logger)
110  << "Guess requested, reading molecular orbitals" << flush;
111  Orbitals orbitalsA, orbitalsB;
112  orbitalsA.ReadFromCpt(_guess_archiveA);
113  orbitalsB.ReadFromCpt(_guess_archiveB);
114  orbitals.PrepareDimerGuess(orbitalsA, orbitalsB);
115  }
116  _qmpackage->WriteInputFile(orbitals);
117  }
118  if (_do_dft_run) {
119  bool run_success = _qmpackage->Run();
120  if (!run_success) {
121  throw std::runtime_error("\n DFT-run failed. Stopping!");
122  }
123  }
124 
125  // parse DFT data, if required
126  if (_do_dft_parse) {
127  XTP_LOG(Log::error, *logger) << "Parsing DFT data from " << _dftlog_file
128  << " and " << _MO_file << flush;
129  _qmpackage->setLogFileName(_dftlog_file);
130  _qmpackage->setMOsFileName(_MO_file);
131 
132  bool Logfile_parse = _qmpackage->ParseLogFile(orbitals);
133  if (!Logfile_parse) {
134  throw std::runtime_error("\n Parsing DFT logfile " + _dftlog_file +
135  " failed. Stopping!");
136  }
137  bool Orbfile_parse = _qmpackage->ParseMOsFile(orbitals);
138  if (!Orbfile_parse) {
139  throw std::runtime_error("\n Parsing DFT orbfile " + _MO_file +
140  " failed. Stopping!");
141  }
142  _qmpackage->CleanUp();
143  }
144 
145  // if no parsing of DFT data is requested, reload serialized orbitals object
146  if (!_do_dft_parse && _do_gwbse) {
147  XTP_LOG(Log::error, *logger)
148  << "Loading serialized data from " << _archive_file << flush;
149  orbitals.ReadFromCpt(_archive_file);
150  }
151  tools::Property& output_summary = _summary.add("output", "");
152  if (_do_gwbse) {
153  GWBSE gwbse = GWBSE(orbitals);
154  gwbse.setLogger(logger);
155  gwbse.Initialize(_gwbse_options);
156  gwbse.Evaluate();
157  gwbse.addoutput(output_summary);
158  }
159  if (_redirect_logger) {
160  WriteLoggerToFile(logger);
161  }
162  return;
163 }

165 void GWBSEEngine::WriteLoggerToFile(Logger* pLog) {
166  std::ofstream ofs;
167  ofs.open(_logger_file, std::ofstream::out);
168  if (!ofs.is_open()) {
169  throw std::runtime_error("Bad file handle: " + _logger_file);
170  }
171  ofs << (*pLog) << std::endl;
172  ofs.close();
173  return;
174 }
