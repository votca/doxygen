37 void GWBSEEngine::Initialize(tools::Property& options,
38  std::string archive_filename) {
39 
40  _archive_file = archive_filename;
41  std::string key = Identify();
42 
43  // get the tasks
44  std::string _tasks_string = options.get(".tasks").as<std::string>();
45  _do_guess = false;
46  _do_dft_input = false;
47  _do_dft_run = false;
48  _do_dft_parse = false;
49  _do_gwbse = false;
50 
51  if (_tasks_string.find("guess") != std::string::npos) _do_guess = true;
52  if (_tasks_string.find("input") != std::string::npos) _do_dft_input = true;
53  if (_tasks_string.find("dft") != std::string::npos) _do_dft_run = true;
54  if (_tasks_string.find("parse") != std::string::npos) _do_dft_parse = true;
55  if (_tasks_string.find("gwbse") != std::string::npos) _do_gwbse = true;
56 
57  // XML option file for GWBSE
58  if (_do_gwbse) {
59  std::string _gwbse_xml = options.get(".gwbse_options").as<std::string>();
60  load_property_from_xml(_gwbse_options, _gwbse_xml.c_str());
61  }
62  // DFT log and MO file names
63  _MO_file = options.get(".mofile").as<std::string>();
64  _dftlog_file = options.get(".dftlog").as<std::string>();
65 
66  // Logger redirection
67  _redirect_logger =
68  options.ifExistsReturnElseReturnDefault<bool>(".redirect_logger", false);
69  _logger_file = "gwbse.log";
70 
71  // for requested merged guess, two archived orbitals objects are needed
72  if (_do_guess) {
73  _guess_archiveA =
74  options.ifExistsReturnElseThrowRuntimeError<std::string>(".archiveA");
75  _guess_archiveB =
76  options.ifExistsReturnElseThrowRuntimeError<std::string>(".archiveB");
77  }
78 
79  return;
80 }

87 void GWBSEEngine::ExcitationEnergies(Orbitals& orbitals) {
88 
89  // redirect log, if required
90  // define own logger for GW-BSE that is written into a runFolder logfile
91  ctp::Logger gwbse_engine_logger(_pLog->getReportLevel());
92  ctp::Logger* logger = _pLog;
93  if (_redirect_logger) {
94  gwbse_engine_logger.setMultithreading(false);
95  gwbse_engine_logger.setPreface(ctp::logINFO, (format("\n ...")).str());
96  gwbse_engine_logger.setPreface(ctp::logERROR, (format("\n ...")).str());
97  gwbse_engine_logger.setPreface(ctp::logWARNING, (format("\n ...")).str());
98  gwbse_engine_logger.setPreface(ctp::logDEBUG, (format("\n ...")).str());
99  logger = &gwbse_engine_logger;
100  }
101  _qmpackage->setLog(logger);
102  if (_do_dft_input) {
103  // required for merged guess
104  if (_qmpackage->GuessRequested() && _do_guess) { // do not want to do an
105  // SCF loop for a dimer
106  CTP_LOG_SAVE(ctp::logINFO, *logger)
107  << "Guess requested, reading molecular orbitals" << flush;
108  Orbitals orbitalsA, orbitalsB;
109  orbitalsA.ReadFromCpt(_guess_archiveA);
110  orbitalsB.ReadFromCpt(_guess_archiveB);
111  orbitals.PrepareDimerGuess(orbitalsA, orbitalsB);
112  }
113  _qmpackage->WriteInputFile(orbitals);
114  }
115  if (_do_dft_run) {
116  bool run_success = _qmpackage->Run();
117  if (!run_success) {
118  throw std::runtime_error("\n DFT-run failed. Stopping!");
119  }
120  }
121 
122  // parse DFT data, if required
123  if (_do_dft_parse) {
124  CTP_LOG_SAVE(ctp::logINFO, *logger)
125  << "Parsing DFT data from " << _dftlog_file << " and " << _MO_file
126  << flush;
127  _qmpackage->setLogFileName(_dftlog_file);
128  _qmpackage->setOrbitalsFileName(_MO_file);
129 
130  bool Logfile_parse = _qmpackage->ParseLogFile(orbitals);
131  if (!Logfile_parse) {
132  throw std::runtime_error("\n Parsing DFT logfile " + _dftlog_file +
133  " failed. Stopping!");
134  }
135  bool Orbfile_parse = _qmpackage->ParseOrbitalsFile(orbitals);
136  if (!Orbfile_parse) {
137  throw std::runtime_error("\n Parsing DFT orbfile " + _MO_file +
138  " failed. Stopping!");
139  }
140  _qmpackage->CleanUp();
141  }
142 
143  // if no parsing of DFT data is requested, reload serialized orbitals object
144  if (!_do_dft_parse && _do_gwbse) {
145  CTP_LOG_SAVE(ctp::logINFO, *logger)
146  << "Loading serialized data from " << _archive_file << flush;
147  orbitals.ReadFromCpt(_archive_file);
148  }
149  tools::Property& output_summary = _summary.add("output", "");
150  if (_do_gwbse) {
151  GWBSE gwbse = GWBSE(orbitals);
152  gwbse.setLogger(logger);
153  gwbse.Initialize(_gwbse_options);
154  gwbse.Evaluate();
155  gwbse.addoutput(output_summary);
156  }
157  if (_redirect_logger) WriteLoggerToFile(logger);
158  return;
159 }
