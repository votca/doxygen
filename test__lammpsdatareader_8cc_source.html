58BOOST_AUTO_TEST_CASE(test_topologyreader) {
59
60 string lammpsdatafilename =
61 std::string(CSG_TEST_DATA_FOLDER) + "/lammpsdatareader/test_polymer.data";
62
63 Topology top;
64
65 TopologyReader::RegisterPlugins();
66 std::unique_ptr<TopologyReader> lammpsDataReader =
67 TopReaderFactory().Create(lammpsdatafilename);
68 lammpsDataReader->ReadTopology(lammpsdatafilename, top);
69
70 BOOST_CHECK_EQUAL(top.BeadCount(), 100);
71 Eigen::Vector3d first_bead_correct_pos(6.2806, 5.25127, 4.98873);
72 Bead *firstBead = top.getBead(0);
73 auto first_bead_pos = firstBead->getPos();
74 BOOST_CHECK(first_bead_correct_pos.isApprox(first_bead_pos, 1e-3));
75
76 Eigen::Vector3d last_bead_correct_pos(10.278495, 7.80388, 5.99629);
77 Bead *lastBead = top.getBead(99);
78 auto last_bead_pos = lastBead->getPos();
79 BOOST_CHECK(last_bead_correct_pos.isApprox(last_bead_pos, 1e-3));
80
81 auto mol = top.getMolecule(0);
82
83 BOOST_CHECK_EQUAL(mol->getName(), "N100");
84 BOOST_CHECK_EQUAL(mol->BeadCount(), 100);
85
86 auto interaction_cont = top.BondedInteractions();
87 votca::Index numBondInter = 99;
88 votca::Index numAngleInter = 98;
89 votca::Index numDihedralInter = 97;
90 votca::Index totalInter = numBondInter + numAngleInter + numDihedralInter;
91 BOOST_CHECK_EQUAL(interaction_cont.size(), totalInter);
92}
