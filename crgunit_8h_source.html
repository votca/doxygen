182 inline vec CrgUnit::GetCom() {
183  vector <vec>::iterator itp = _positions.begin();
184  vec com(0.,0.,0.);
185  for ( ; itp!=_positions.end() ; ++itp )
186  com += *itp;
187  return unit<bohr,nm>::to( com/double(_positions.size()) );

190 inline void CrgUnit::shift(const vec & displ) {
191  vec displ_bohr = unit<nm,bohr>::to(displ);
192  vector <vec> ::iterator it_vec;
193  for (it_vec = _positions.begin();
194  it_vec != _positions.end();
195  ++it_vec)
196  *it_vec = *it_vec + displ_bohr;
197 }

199 inline vector <vec> CrgUnit::shift_pos(const vec & a) {
200  vector <vec>::iterator it_pos;
201  vector <vec> res;
202  for (it_pos = _positions.begin();
203  it_pos != _positions.end();
204  ++it_pos) {
205  vec b = (*it_pos) + a;
206  res.push_back(b);
207  }
208  return res;
209 }
