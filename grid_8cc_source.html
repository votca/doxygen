43 void Grid::setupgrid(const QMMolecule& Atomlist) {
44 
45  tools::Elements elements;
46  std::pair<Eigen::Vector3d, Eigen::Vector3d> extension =
47  Atomlist.CalcSpatialMinMax();
48  Eigen::Array3d min = extension.first.array();
49  Eigen::Array3d max = extension.second.array();
50  Eigen::Array3d doublesteps = (max - min + 2 * _padding) / _gridspacing;
51  Eigen::Array<votca::Index, 3, 1> steps = (doublesteps.ceil()).cast<Index>();
52 
53  // needed to symmetrize grid around molecule
54  Eigen::Array3d padding =
55  (doublesteps - steps.cast<double>()) * _gridspacing * 0.5 + _padding;
56  Eigen::Array3d minpos = min - padding;
57  for (Index i = 0; i <= steps.x(); i++) {
58  double x = minpos.x() + double(i) * _gridspacing;
59  for (Index j = 0; j <= steps.y(); j++) {
60  double y = minpos.y() + double(j) * _gridspacing;
61  for (Index k = 0; k <= steps.z(); k++) {
62  double z = minpos.z() + double(k) * _gridspacing;
63  bool is_valid = false;
64  Eigen::Vector3d gridpos(x, y, z);
65  for (const QMAtom& atom : Atomlist) {
66  const Eigen::Vector3d& atompos = atom.getPos();
67  double distance2 = (gridpos - atompos).squaredNorm();
68  double atomcutoff =
69  elements.getVdWChelpG(atom.getElement()) * tools::conv::ang2bohr;
70  if (distance2 < (atomcutoff * atomcutoff)) {
71  is_valid = false;
72  break;
73  } else if (distance2 < (_cutoff * _cutoff)) {
74  is_valid = true;
75  }
76  }
77  if (is_valid) {
78  _gridpoints.push_back(gridpos);
79  }
80  }
81  }
82  }
83 
84  _gridvalues = Eigen::VectorXd::Zero(_gridpoints.size());
85  return;
86 }
