252 void Grid::subdivide(const vec &v1, const vec &v2, const vec &v3, std::vector<vec> &spherepoints, const int depth) {
253  if(depth == 0) {
254  spherepoints.push_back(v1);
255  spherepoints.push_back(v2);
256  spherepoints.push_back(v3);
257  return;
258  }
259  const vec v12 = (v1 + v2).normalize();
260  const vec v23 = (v2 + v3).normalize();
261  const vec v31 = (v3 + v1).normalize();
262  subdivide(v1, v12, v31, spherepoints, depth - 1);
263  subdivide(v2, v23, v12, spherepoints, depth - 1);
264  subdivide(v3, v31, v23, spherepoints, depth - 1);
265  subdivide(v12, v23, v31,spherepoints, depth - 1);
266 }

296 void Grid::setupradialgrid(const int depth) {
297  _cubegrid = false;
298  std::vector<vec> spherepoints;
299  initialize_sphere(spherepoints, depth);
300  double x = 0.0;
301  double y = 0.0;
302  double z = 0.0;
303  std::vector<ctp::QMAtom* >::const_iterator ait;
304  for (ait = _atomlist->begin(); ait != _atomlist->end(); ++ait) {
305  x += (*ait)->x;
306  y += (*ait)->y;
307  z += (*ait)->z;
308  }
309 
310  double numofatoms = double(_atomlist->size());
311  vec centerofmolecule = vec(x / numofatoms, y / numofatoms, z / numofatoms);
312  //cout <<endl;
313  //cout << "Center of molecule is "<< centerofmolecule<<"."<<endl;
314 
315  bool cutoff_smaller_molecule=false;
316  double temp_cutoff=_cutoff;
317  for (ait = _atomlist->begin(); ait != _atomlist->end(); ++ait) {
318  x = (*ait)->x;
319  y = (*ait)->y;
320  z = (*ait)->z;
321 
322  vec temppos=vec(x,y,z);
323  double dist=abs(centerofmolecule-temppos)+2.0; // 2.0 because that is approximately the VdW radius
324  if (dist>temp_cutoff){
325  cutoff_smaller_molecule=true;
326  temp_cutoff=dist;
327  }
328  }
329 
330  if (cutoff_smaller_molecule){
331  cout <<endl;
332  cout << "Specified cutoff of "<< _cutoff<<" to small. Cutoff is set to "<< temp_cutoff <<"Angstroem."<<endl;
333  _cutoff=temp_cutoff;
334  }
335 
336  //till here everything was Angstroem
337 
338  std::vector<vec>::const_iterator git;
339  for (git = spherepoints.begin(); git != spherepoints.end(); ++git) {
340  vec position=((*git)*_cutoff+centerofmolecule)*conv::ang2nm;
341  _gridpoints.push_back(position.converttoub());
342  if(_createpolarsites){
343  string name="H";
344  ctp::APolarSite *apolarsite= new ctp::APolarSite(0,name);
345  apolarsite->setRank(0);
346  apolarsite->setQ00(0,0); // <- charge state 0 <> 'neutral'
347  apolarsite->setIsoP(0.0);
348  apolarsite->setPos(position);
349  _gridsites.push_back(apolarsite);
350  _all_gridsites.push_back(apolarsite);
351  }
352  }
353  if (_sites_seg != NULL) delete _sites_seg;
354  _sites_seg = new ctp::PolarSeg(0, _gridsites);
355

358 void Grid::setupgrid(){
359 
360 
361  Elements _elements;
362  double xmin=std::numeric_limits<double>::max();
363  double ymin=xmin;
364  double zmin=xmin;
365 
366  double xmax=std::numeric_limits<double>::min();
367  double ymax=xmax;
368  double zmax=xmax;
369  double xtemp,ytemp,ztemp;
370 
371  if(_useVdWcutoff){
372  _padding=0.0;
373  for (std::vector<ctp::QMAtom* >::const_iterator atom = _atomlist->begin(); atom != _atomlist->end(); ++atom ){
374  if(_elements.getVdWChelpG((*atom)->type)+_shift_cutoff>_padding) _padding=_elements.getVdWChelpG((*atom)->type)+_shift_cutoff;
375  }
376  }
377 
378 
379  for (std::vector<ctp::QMAtom* >::const_iterator atom = _atomlist->begin(); atom != _atomlist->end(); ++atom ) {
380  xtemp=(*atom)->x;
381  ytemp=(*atom)->y;
382  ztemp=(*atom)->z;
383  if (xtemp<xmin) xmin=xtemp;
384  if (xtemp>xmax) xmax=xtemp;
385  if (ytemp<ymin) ymin=ytemp;
386  if (ytemp>ymax) ymax=ytemp;
387  if (ztemp<zmin) zmin=ztemp;
388  if (ztemp>zmax) zmax=ztemp;
389  }
390 
391  _lowerbound=vec(xmin-_padding,ymin-_padding,zmin-_padding);
392  vec _upperbound=vec(xmax+_padding,ymax+_padding,zmax+_padding);
393  vec steps=(_upperbound-_lowerbound)/_gridspacing;
394  _xsteps=int(ceil(steps.getX()));
395  _ysteps=int(ceil(steps.getY()));
396  _zsteps=int(ceil(steps.getZ()));
397 
398  // needed to symmetrize grid around molecule
399  double padding_x=(steps.getX()-_xsteps)*_gridspacing*0.5+_padding;
400  double padding_y=(steps.getY()-_ysteps)*_gridspacing*0.5+_padding;
401  double padding_z=(steps.getZ()-_zsteps)*_gridspacing*0.5+_padding;
402 
403 
404  for(int i=0;i<=_xsteps;i++){
405  double x=xmin-padding_x+i*_gridspacing;
406  for(int j=0;j<=_ysteps;j++){
407  double y=ymin-padding_y+j*_gridspacing;
408  for(int k=0;k<=_zsteps;k++){
409  double z=zmin-padding_z+k*_gridspacing;
410  bool _is_valid = false;
411  for (std::vector<ctp::QMAtom* >::const_iterator atom = _atomlist->begin(); atom != _atomlist->end(); ++atom ) {
412  //cout << "Punkt " << x <<":"<< y << ":"<<z << endl;
413  xtemp=(*atom)->x;
414  ytemp=(*atom)->y;
415  ztemp=(*atom)->z;
416  double distance2=pow((x-xtemp),2)+pow((y-ytemp),2)+pow((z-ztemp),2);
417  if(_useVdWcutoff) _cutoff=_elements.getVdWChelpG((*atom)->type)+_shift_cutoff;
418  if(_useVdWcutoff_inside)_cutoff_inside=_elements.getVdWChelpG((*atom)->type)+_shift_cutoff_inside;
419  //cout << "Punkt " << x <<":"<< y << ":"<<z << ":"<< distance2 << ":"<< (*atom)->type <<":"<<pow(VdW,2)<< endl;
420  if ( distance2<pow(_cutoff_inside,2)){
421  _is_valid = false;
422  break;
423  }
424  else if ( distance2<pow(_cutoff,2)) _is_valid = true;
425  }
426  if (_is_valid || _cubegrid){
427  vec temppos=vec(x,y,z);
428  temppos=conv::ang2nm*temppos;
429 
430  if(_createpolarsites){
431 
432 
433  string name="H";
434  ctp::APolarSite *apolarsite= new ctp::APolarSite(0,name);
435  apolarsite->setRank(0);
436  apolarsite->setQ00(0,0); // <- charge state 0 <> 'neutral'
437  apolarsite->setIsoP(0.0);
438  apolarsite->setPos(temppos);
439  if(_is_valid){
440  _gridsites.push_back(apolarsite);
441  _gridpoints.push_back(temppos);
442  }
443  else {apolarsite->setIsVirtual(true);}
444  _all_gridsites.push_back(apolarsite);
445  }
446  else if(!_createpolarsites){_gridpoints.push_back(temppos);}
447  }
448  }
449  }
450  }
451  if (_sites_seg != NULL) delete _sites_seg;
452  _sites_seg = new ctp::PolarSeg(0, _gridsites);
453 }
