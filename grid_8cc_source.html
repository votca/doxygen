99 void Grid::readgridfromCubeFile(std::string filename, bool ignore_zeros){
100  Elements _elements;
101  _cubegrid=true;
102 
103  if(_gridpoints.size()>0) throw std::runtime_error("Grid object already has points.");
104  std::ifstream in1;
105  std::string s;
106  in1.open(filename.c_str(), std::ios::in);
107  getline(in1, s);
108  getline(in1, s);
109  int natoms;
110  double xstart,ystart,zstart;
111  double xincr,yincr,zincr;
112 
113  double tempdouble;
114  std::string name="H";
115  in1 >> natoms;
116  in1 >> xstart;
117  in1 >> ystart;
118  in1 >> zstart;
119  in1 >> _xsteps;
120  in1 >> xincr;
121  in1 >> tempdouble;
122  in1 >> tempdouble;
123  in1 >> _ysteps;
124  in1 >> tempdouble;
125  in1 >> yincr;
126  in1 >> tempdouble;
127  in1 >> _zsteps;
128  in1 >> tempdouble;
129  in1 >> tempdouble;
130  in1 >> zincr;
131 
132 
133  if(xincr==yincr && yincr==zincr && zincr==xincr){
134  _gridspacing=xincr*conv::bohr2ang;
135  }
136  else{
137  throw std::runtime_error("Gridspacing in x,y,z is different, currently not implemented, loading aborted");
138  }
139  _lowerbound=vec(xstart*conv::bohr2ang,ystart*conv::bohr2ang,zstart*conv::bohr2ang);
140 
141  _atomlist= new std::vector< QMAtom* >;
142  for (int iatom =0; iatom < std::abs(natoms); iatom++) {
143  // get center coordinates in Bohr
144  double x ;
145  double y ;
146  double z ;
147  int atnum ;
148  double crg ;
149 
150  // get from first cube
151  in1 >> atnum;
152  in1 >> crg;
153  in1 >> x;
154  in1 >> y;
155  in1 >> z;
156 
157  QMAtom *qmatom=new QMAtom(iatom,_elements.getEleName(atnum),x,y,z);
158  _atomlist->push_back(qmatom);
159  }
160  double potential=0.0;
161  //cout << "File has " << xsteps << " : "<< ysteps << " : "<<zsteps << " gridpoints"<<endl;
162  for (int _ix = 0; _ix < _xsteps; _ix++) {
163  double posx=(xstart+_ix*xincr);
164 
165  for (int _iy = 0; _iy < _ysteps; _iy++) {
166  double posy=(ystart+_iy*yincr);
167 
168 
169  for (int _iz = 0; _iz < _zsteps; _iz++) {
170  double posz=(zstart+_iz*zincr);
171  in1 >> potential;
172  vec pos=vec(posx,posy,posz);
173  ctp::APolarSite *apolarsite= new ctp::APolarSite(0,name);
174  apolarsite->setRank(0);
175  apolarsite->setQ00(0,0); // <- charge state 0 <> 'neutral'
176  apolarsite->setIsoP(0.0);
177  vec temp=pos*tools::conv::bohr2nm;
178  apolarsite->setPos(temp);
179  apolarsite->setPhi(potential,0);
180  if(potential!=0.0 || !ignore_zeros){
181  _gridsites.push_back(apolarsite);
182  _gridpoints.push_back(pos);
183  }
184  else {apolarsite->setIsVirtual(true);}
185  _all_gridsites.push_back(apolarsite);
186 
187 
188  }}}
189 
190  return;
191  }

246 void Grid::setupgrid(){
247 
248 
249  Elements _elements;
250  std::vector<QMAtom* >::const_iterator atom;
251  if (_useVdWcutoff) {
252  _padding = 0.0;
253  for (atom = _atomlist->begin(); atom != _atomlist->end(); ++atom) {
254  double VdWcutoff=_elements.getVdWChelpG((*atom)->getType())*tools::conv::ang2bohr;
255  if ( VdWcutoff+ _shift_cutoff > _padding) {
256  _padding = VdWcutoff + _shift_cutoff;
257  }
258  }
259  }
260 
261  double xmin=std::numeric_limits<double>::max();
262  double ymin=xmin;
263  double zmin=xmin;
264 
265  double xmax=std::numeric_limits<double>::min();
266  double ymax=xmax;
267  double zmax=xmax;
268  double xtemp,ytemp,ztemp;
269 
270  for (atom = _atomlist->begin(); atom != _atomlist->end(); ++atom ) {
271  const tools::vec & pos=(*atom)->getPos();
272  xtemp=pos.getX();
273  ytemp=pos.getY();
274  ztemp=pos.getZ();
275  if (xtemp<xmin) xmin=xtemp;
276  if (xtemp>xmax) xmax=xtemp;
277  if (ytemp<ymin) ymin=ytemp;
278  if (ytemp>ymax) ymax=ytemp;
279  if (ztemp<zmin) zmin=ztemp;
280  if (ztemp>zmax) zmax=ztemp;
281  }
282 
283  _lowerbound=vec(xmin-_padding,ymin-_padding,zmin-_padding);
284  vec _upperbound=vec(xmax+_padding,ymax+_padding,zmax+_padding);
285  vec steps=(_upperbound-_lowerbound)/_gridspacing;
286  _xsteps=int(ceil(steps.getX()));
287  _ysteps=int(ceil(steps.getY()));
288  _zsteps=int(ceil(steps.getZ()));
289 
290  // needed to symmetrize grid around molecule
291  double padding_x=(steps.getX()-_xsteps)*_gridspacing*0.5+_padding;
292  double padding_y=(steps.getY()-_ysteps)*_gridspacing*0.5+_padding;
293  double padding_z=(steps.getZ()-_zsteps)*_gridspacing*0.5+_padding;
294 
295 
296  for(int i=0;i<=_xsteps;i++){
297  double x=xmin-padding_x+i*_gridspacing;
298  for(int j=0;j<=_ysteps;j++){
299  double y=ymin-padding_y+j*_gridspacing;
300  for(int k=0;k<=_zsteps;k++){
301  double z=zmin-padding_z+k*_gridspacing;
302  bool _is_valid = false;
303  vec gridpos=vec(x,y,z);
304  for (atom = _atomlist->begin(); atom != _atomlist->end(); ++atom ) {
305  vec atompos=(*atom)->getPos();
306  double distance2=(gridpos-atompos)*(gridpos-atompos);
307  if(_useVdWcutoff){
308  _cutoff=_elements.getVdWChelpG((*atom)->getType())*tools::conv::ang2bohr+_shift_cutoff;
309  }
310  if(_useVdWcutoff_inside){
311  _cutoff_inside=_elements.getVdWChelpG((*atom)->getType())*tools::conv::ang2bohr+_shift_cutoff_inside;
312  }
313  if ( distance2<(_cutoff_inside*_cutoff_inside)){
314  _is_valid = false;
315  break;
316  }
317  else if ( distance2<(_cutoff*_cutoff)) _is_valid = true;
318  }
319  if (_is_valid || _cubegrid){
320 
321  if(_createpolarsites){
322 
323  string name="X";
324  ctp::APolarSite *apolarsite= new ctp::APolarSite(0,name);
325  apolarsite->setRank(0);
326  apolarsite->setQ00(0,0); // <- charge state 0 <> 'neutral'
327  apolarsite->setIsoP(0.0);
328  vec temp=gridpos*conv::bohr2nm;
329  apolarsite->setPos(temp);
330  if(_is_valid){
331  _gridsites.push_back(apolarsite);
332  _gridpoints.push_back(gridpos);
333  }
334  else {apolarsite->setIsVirtual(true);}
335  _all_gridsites.push_back(apolarsite);
336  }
337  else if(!_createpolarsites){_gridpoints.push_back(gridpos);}
338  }
339  }
340  }
341  }
342  return;
343 }
