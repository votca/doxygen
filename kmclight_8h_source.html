263 vector<NodeLight*> KMCLight::LoadGraph()
264 {
265  string injectionpattern = "*";
266 
267  vector<NodeLight*> node;
268 
269  // Load nodes
270  votca::tools::Database db;
271  db.Open( _filename );
272  if(verbose >= 1) {cout << "LOADING GRAPH" << endl << "database file: " << _filename << endl; }
273  votca::tools::Statement *stmt = db.Prepare("SELECT id-1, name FROM segments;");
274 
275  int i=0;
276  while (stmt->Step() != SQLITE_DONE)
277  {
278  NodeLight *newNodeLight = new NodeLight();
279  node.push_back(newNodeLight);
280 
281  int newid = stmt->Column<int>(0);
282  string name = stmt->Column<string>(1);
283  node[i]->id = newid;
284  if (votca::tools::wildcmp(injectionpattern.c_str(), name.c_str()))
285  {
286  node[i]->injectable = 1;
287  }
288  else
289  {
290  node[i]->injectable = 0;
291  }
292  i++;
293  }
294  delete stmt;
295  if(verbose >= 1) { cout << "segments: " << node.size() << endl; }
296 
297  // Load pairs and rates
298  int numberofpairs = 0;
299  stmt = db.Prepare("SELECT seg1-1 AS 'segment1', seg2-1 AS 'segment2', rate12e AS 'rate', drX, drY, drZ FROM pairs UNION SELECT seg2-1 AS 'segment1', seg1-1 AS 'segment2', rate21e AS 'rate', -drX AS 'drX', -drY AS 'drY', -drZ AS 'drZ' FROM pairs ORDER BY segment1;");
300  while (stmt->Step() != SQLITE_DONE)
301  {
302  int seg1 = stmt->Column<int>(0);
303  int seg2 = stmt->Column<int>(1);
304  double rate12 = stmt->Column<double>(2);
305  myvec dr = myvec(stmt->Column<double>(3), stmt->Column<double>(4), stmt->Column<double>(5));
306  node[seg1]->AddEventLight(seg2,rate12,dr);
307  numberofpairs ++;
308  }
309  delete stmt;
310 
311  if(verbose >= 1) { cout << "pairs: " << numberofpairs/2 << endl; }
312 
313  // Calculate initial escape rates
314  for(unsigned int i=0; i<node.size(); i++)
315  {
316  node[i]->InitEscapeRate();
317  }
318  return node;
319 }

399 vector<double> KMCLight::RunVSSM(vector<NodeLight*> node, double runtime, unsigned int numberofcharges, votca::tools::Random2 *RandomVariable)
400 {
401  int tid = 0;
402  # ifdef _OPENMP
403  tid = omp_get_thread_num();
404  # endif
405 
406  int realtime_start = time(NULL);
407  if(tid == 0)
408  {
409  cout << endl << "Algorithm: VSSM for Multiple Charges" << endl;
410  cout << "number of charges: " << numberofcharges << endl;
411  cout << "number of nodes: " << node.size() << endl;
412  }
413 
414  if(numberofcharges > node.size())
415  {
416  throw runtime_error("ERROR in kmcmultiple: specified number of charges is greater than the number of nodes. This conflicts with single occupation.");
417  }
418 
419  double outputfrequency = runtime/100;
420  vector<double> occP(node.size(),0.);
421 
422  // Injection
423  vector< ChargecarrierLight* > carrier;
424  for (unsigned int i = 0; i < numberofcharges; i++)
425  {
426  ChargecarrierLight *newCharge = new ChargecarrierLight;
427  newCharge->id = i;
428  newCharge->node = node[RandomVariable->rand_uniform_int(node.size())];
429  while(newCharge->node->occupied == 1 || newCharge->node->injectable != 1)
430  { // maybe already occupied? or maybe not injectable?
431  newCharge->node = node[RandomVariable->rand_uniform_int(node.size())];
432  }
433  newCharge->node->occupied = 1;
434  newCharge->dr_travelled = myvec(0,0,0);
435  if(tid == 0) {cout << "starting position for charge " << i+1 << ": segment " << newCharge->node->id+1 << endl; }
436  carrier.push_back(newCharge);
437  }
438 
439  double simtime = 0.;
440  int step = 0;
441  double nextoutput = outputfrequency;
442 
443  progressbarLight(0.);
444  vector<int> forbiddennodes;
445  vector<int> forbiddendests;
446  while(simtime < runtime)
447  {
448  double cumulated_rate = 0;
449  for(unsigned int i=0; i<numberofcharges; i++)
450  {
451  cumulated_rate += carrier[i]->node->EscapeRate();
452  }
453  if(cumulated_rate == 0)
454  { // this should not happen: no possible jumps defined for a node
455  throw runtime_error("ERROR in kmcmultiple: Incorrect rates in the database file. All the escape rates for the current setting are 0.");
456  }
457  // go forward in time
458  double dt = 0;
459  double rand_u = 1-RandomVariable->rand_uniform();
460  while(rand_u == 0)
461  {
462  cout << "WARNING: encountered 0 as a random variable! New try." << endl;
463  rand_u = 1-RandomVariable->rand_uniform();
464  }
465  dt = -1 / cumulated_rate * log(rand_u);
466  simtime += dt;
467  if(verbose >= 1 && tid == 0) {cout << "simtime += " << dt << endl << endl;}
468  step += 1;
469 
470  for(unsigned int i=0; i<numberofcharges; i++)
471  {
472  carrier[i]->node->occupationtime += dt;
473  }
474 
475  ResetForbiddenLight(forbiddennodes);
476  int level1step = 0;
477  while(level1step == 0)
478  // LEVEL 1
479  {
480  // determine which electron will escape
481  NodeLight* do_oldnode;
482  NodeLight* do_newnode;
483  ChargecarrierLight* do_affectedcarrier;
484 
485  double u = 1 - RandomVariable->rand_uniform();
486  for(unsigned int i=0; i<numberofcharges; i++)
487  {
488  u -= carrier[i]->node->EscapeRate()/cumulated_rate;
489  if(u <= 0)
490  {
491  do_oldnode = carrier[i]->node;
492  do_affectedcarrier = carrier[i];
493  break;
494  }
495  do_oldnode = carrier[i]->node;
496  do_affectedcarrier = carrier[i];
497  }
498 
499  double maxprob = 0.;
500  double newprob = 0.;
501  myvec dr;
502  if(verbose >= 1 && tid == 0) {cout << "Charge number " << do_affectedcarrier->id+1 << " which is sitting on segment " << do_oldnode->id+1 << " will escape!" << endl ;}
503  if(ForbiddenLight(do_oldnode->id, forbiddennodes) == 1) {continue;}
504 
505  // determine where it will jump to
506  ResetForbiddenLight(forbiddendests);
507  while(true)
508  {
509  // LEVEL 2
510  if(verbose >= 1 && tid == 0) {cout << "There are " << do_oldnode->event.size() << " possible jumps for this charge:"; }
511 
512  do_newnode = NULL;
513  u = 1 - RandomVariable->rand_uniform();
514  for(unsigned int j=0; j<do_oldnode->event.size(); j++)
515  {
516  if(ForbiddenLight(do_oldnode->event[j].destination, forbiddendests) == 1)
517  { // directly skip forbidden events
518  if(verbose >= 1 && tid == 0) { cout << " [" << do_oldnode->event[j].destination+1 << " FORBIDDEN]" ; }
519  continue;
520  }
521  if(verbose >= 1 && tid == 0) { cout << " " << do_oldnode->event[j].destination+1 ; }
522  u -= do_oldnode->event[j].rate/do_oldnode->EscapeRate();
523  if(u <= 0)
524  {
525  do_newnode = node[do_oldnode->event[j].destination];
526  dr = do_oldnode->event[j].dr;
527  break;
528  }
529  do_newnode = node[do_oldnode->event[j].destination];
530  dr = do_oldnode->event[j].dr;
531  }
532 
533  if(do_newnode == NULL)
534  {
535  if(verbose >= 1 && tid == 0) {cout << endl << "NodeLight " << do_oldnode->id+1 << " is SURROUNDED by forbidden destinations and zero rates. Adding it to the list of forbidden nodes. After that: selection of a new escape node." << endl; }
536  AddtForbiddenLight(do_oldnode->id, forbiddennodes);
537  int nothing=0;
538  cin >> nothing;
539  break; // select new escape node (ends level 2 but without setting level1step to 1)
540  }
541  if(verbose >= 1 && tid == 0) {cout << endl << "Selected jump: " << do_newnode->id+1 << endl; }
542 
543  // if the new segment is unoccupied: jump; if not: nothing??
544  if(do_newnode->occupied == 1)
545  {
546  if(Surrounded(do_oldnode, forbiddendests) == 1)
547  {
548  if(verbose >= 1 && tid == 0) {cout << "NodeLight " << do_oldnode->id+1 << " is SURROUNDED by forbidden destinations. Adding it to the list of forbidden nodes. After that: selection of a new escape node." << endl; }
549  AddtForbiddenLight(do_oldnode->id, forbiddennodes);
550  break; // select new escape node (ends level 2 but without setting level1step to 1)
551  }
552  if(verbose >= 1 && tid == 0) {cout << "Selected segment: " << do_newnode->id+1 << " is already OCCUPIED. Added to forbidden list." << endl << endl;}
553  AddtForbiddenLight(do_newnode->id, forbiddendests);
554  if(verbose >= 1 && tid == 0) {cout << "Now choosing different hopping destination." << endl; }
555  continue; // select new destination
556  }
557  else
558  {
559  do_newnode->occupied = 1;
560  do_oldnode->occupied = 0;
561  do_affectedcarrier->node = do_newnode;
562  do_affectedcarrier->dr_travelled += dr;
563  level1step = 1;
564  if(verbose >= 1 && tid == 0) {cout << "Charge has jumped to segment: " << do_newnode->id+1 << "." << endl;}
565  // cout << "old node: " << do_oldnode->id+1 << ", occupation: " << do_oldnode->occupied << endl;
566  // cout << "new node: " << do_newnode->id+1 << ", occupation: " << do_newnode->occupied << endl;
567  break; // this ends LEVEL 2 , so that the time is updated and the next MC step started
568  }
569 
570  if(verbose >= 1 && tid == 0) {cout << "." << endl;}
571  // END LEVEL 2
572  }
573  // END LEVEL 1
574  }
575 
576 
577  if(simtime > nextoutput && tid == 0)
578  {
579  nextoutput = simtime + outputfrequency;
580  progressbarLight(simtime/runtime);
581  cout << " remaining: ";
582  printtimeLight(int((runtime/simtime-1) * (int(time(NULL)) - realtime_start)));
583  }
584  // cout << "step " << step << endl;
585  }
586  progressbarLight(1.);
587 
588 
589  // calculate occupation probabilities from occupation times
590  for(unsigned int j=0; j<node.size(); j++)
591  {
592  occP[j] = node[j]->occupationtime / simtime;
593  }
594 
595 
596  if (tid == 0)
597  {
598  cout << endl << "finished KMC simulation after " << step << " steps." << endl;
599  cout << "runtime: ";
600  printtime(time(NULL) - realtime_start);
601  myvec dr_travelled = myvec (0,0,0);
602  cout << endl << "Average velocities (m/s): " << endl;
603  for(unsigned int i=0; i<numberofcharges; i++)
604  {
605  //cout << std::scientific << " charge " << i+1 << ": " << carrier[i]->dr_travelled/simtime*1e-9 << endl;
606  cout << std::scientific << " charge " << i+1 << ": " << carrier[i]->dr_travelled/simtime << endl;
607  dr_travelled += carrier[i]->dr_travelled;
608  }
609  dr_travelled /= numberofcharges;
610  //cout << std::scientific << " Overall average velocity (m/s): " << dr_travelled/simtime*1e-9 << endl;
611  cout << std::scientific << " Overall average velocity (m/s): " << dr_travelled/simtime << endl;
612 
613  cout << endl << "Distances travelled (m): " << endl;
614  for(unsigned int i=0; i<numberofcharges; i++)
615  {
616  cout << std::scientific << " charge " << i+1 << ": " << carrier[i]->dr_travelled << endl;
617  }
618 
619  // calculate mobilities
620  double absolute_field = sqrt(_fieldX*_fieldX + _fieldY*_fieldY + _fieldZ*_fieldZ);
621  double average_mobilityZ = 0;
622  if (absolute_field != 0)
623  {
624  cout << endl << "Mobilities (cm^2/Vs): " << endl;
625  for(unsigned int i=0; i<numberofcharges; i++)
626  {
627  //myvec velocity = carrier[i]->dr_travelled/simtime*1e-9;
628  myvec velocity = carrier[i]->dr_travelled/simtime;
629  double absolute_velocity = sqrt(velocity.x()*velocity.x() + velocity.y()*velocity.y() + velocity.z()*velocity.z());
630  //cout << std::scientific << " charge " << i+1 << ": mu=" << absolute_velocity/absolute_field*1E4 << endl;
631  cout << std::scientific << " charge " << i+1 << ": muZ=" << velocity.z()/_fieldZ*1E4 << endl;
632  average_mobilityZ += velocity.z()/_fieldZ*1E4;
633  }
634  average_mobilityZ /= numberofcharges;
635  cout << std::scientific << " Overall average z-mobility <muZ>=" << average_mobilityZ << endl;
636  }
637  cout << endl;
638 
639 
640  }
641  return occP;
642 }

645 void KMCLight::WriteOcc(vector<double> occP, vector<NodeLight*> node)
646 {
647  votca::tools::Database db;
648  cout << "Opening for writing " << _filename << endl;
649  db.Open(_filename);
650  db.Exec("BEGIN;");
651  votca::tools::Statement *stmt = db.Prepare("UPDATE segments SET occPe = ? WHERE id = ?;"); // electron occ. prob., check (think about) this
652  for(unsigned int i=0; i<node.size(); ++i)
653  {
654  stmt->Reset();
655  stmt->Bind(1, occP[i]);
656  stmt->Bind(2, node[i]->id+1);
