38  ctp::APolarSite *QMMInterface::Convert(ctp::QMAtom *atm, int id) {
39  double A_to_nm = 0.1;
40  vec pos = A_to_nm * vec(atm->x, atm->y, atm->z);
41  double q = atm->charge;
42  std::string elem = atm->type;
43  double pol = 0.0;
44  try {
45  pol = _polar_table.at(elem);
46  } catch (out_of_range) {
47  std::cout << std::endl << "QMMInterface - no default polarizability given "
48  << "for element type '" << elem << "'. Defaulting to 1A**3" << std::flush;
49  pol = 1e-3;
50  }
51 
52  ctp::APolarSite *new_aps = new ctp::APolarSite(id, elem);
53  new_aps->setRank(0);
54  new_aps->setPos(pos);
55  new_aps->setQ00(q, 0); // <- charge state 0 <> 'neutral'
56  new_aps->setIsoP(pol);
57 
58  return new_aps;
59  }

128  void QMMInterface::addMMAtomtoOrb(ctp::APolarSite * aps, Orbitals &orb, bool with_polarisation) {
129  vec pos = aps->getPos() * tools::conv::nm2ang;
130  double Q = aps->getQ00();
131  orb.AddAtom(aps->getName(), pos, Q, true);
132 
133  if (_split_dpl) {
134  vec tot_dpl = vec(0, 0, 0);
135  if (with_polarisation) {
136  tot_dpl += aps->getU1();
137  }
138  if (aps->getRank() > 0) {
139  tot_dpl += aps->getQ1();
140  }
141  // Calculate virtual charge positions
142  double a = _dpl_spacing; // this is in nm
143  double mag_d = abs(tot_dpl); // this is in e * nm
144  if (mag_d > 1e-9) {
145  vec dir_d_0 = tot_dpl.normalize();
146  vec dir_d = dir_d_0.normalize();
147  vec A = pos + 0.5 * a * dir_d * tools::conv::nm2ang; // converted to AA
148  vec B = pos - 0.5 * a * dir_d * tools::conv::nm2ang;
149  double qA = mag_d / a;
150  double qB = -qA;
151  orb.AddAtom("A", A, qA, true);
152  orb.AddAtom("B", B, qB, true);
153  }
154  }
155 
156  if (aps->getRank() > 1 && _split_dpl) {
157  tools::matrix components = aps->getQ2cartesian();
158  tools::matrix::eigensystem_t system;
159  components.SolveEigensystem(system);
160  double a = _dpl_spacing;
161  string Atomnameplus[] = {"Xp", "Yp", "Zp"};
162  string Atomnameminus[] = {"Xm", "Ym", "Zm"};
163  for (unsigned i = 0; i < 3; i++) {
164 
165  double q = system.eigenvalues[i] / (a * a);
166  if (q < 1e-9) {
167  continue;
168  }
169  tools::vec vec1 = pos + 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
170  tools::vec vec2 = pos - 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
171  orb.AddAtom(Atomnameplus[i], vec1, q, true);
172  orb.AddAtom(Atomnameminus[i], vec2, q, true);
173 
174  }
175  }
176 
177  return;
178  }
