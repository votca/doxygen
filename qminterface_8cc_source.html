38  ctp::APolarSite *QMMInterface::Convert(ctp::QMAtom *atm, int id) {
39  double A_to_nm = 0.1;
40  vec pos = A_to_nm * vec(atm->x, atm->y, atm->z);
41  double q = atm->charge;
42  std::string elem = atm->type;
43  double pol = 0.0;
44  try {
45  pol = _polar_table.at(elem);
46  } catch (out_of_range) {
47  std::cout << std::endl << "QMMInterface - no default polarizability given "
48  << "for element type '" << elem << "'. Defaulting to 1A**3" << std::flush;
49  pol = 1e-3;
50  }
51 
52  ctp::APolarSite *new_aps = new ctp::APolarSite(id, elem);
53  new_aps->setRank(0);
54  new_aps->setPos(pos);
55  new_aps->setQ00(q, 0); // <- charge state 0 <> 'neutral'
56  new_aps->setIsoP(pol);
57 
58  return new_aps;
59  }

153  void QMMInterface::addMMAtomtoOrb(ctp::APolarSite * aps, Orbitals &orb, bool with_polarisation) {
154  const tools::vec pos = aps->getPos() * tools::conv::nm2ang;
155  double Q = aps->getQ00();
156  orb.AddAtom(aps->getName(), pos, Q, true);
157 
158  if (_split_dpl) {
159  tools::vec tot_dpl = tools::vec(0.0);
160  if (with_polarisation) {
161  tot_dpl += aps->getU1();
162  }
163  if (aps->getRank() > 0) {
164  tot_dpl += aps->getQ1();
165  }
166  // Calculate virtual charge positions
167  double a = _dpl_spacing; // this is in nm
168  double mag_d = abs(tot_dpl); // this is in e * nm
169  if (mag_d > 1e-9) {
170  tools::vec dir_d = tot_dpl.normalize();
171  tools::vec A = pos + 0.5 * a * dir_d * tools::conv::nm2ang; // converted to AA
172  tools::vec B = pos - 0.5 * a * dir_d * tools::conv::nm2ang;
173  double qA = mag_d / a;
174  double qB = -qA;
175  orb.AddAtom("A", A, qA, true);
176  orb.AddAtom("B", B, qB, true);
177  }
178  }
179 
180  if (aps->getRank() > 1 && _split_dpl) {
181  tools::matrix components = aps->getQ2cartesian();
182  tools::matrix::eigensystem_t system;
183  components.SolveEigensystem(system);
184  double a = 2*_dpl_spacing;
185  string Atomnameplus[] = {"X", "Y", "Z"};
186  string Atomnameminus[] = {"X", "Y", "Z"};
187  for (unsigned i = 0; i < 3; i++) {
188 
189  double q = system.eigenvalues[i] / (a * a);
190  if (std::abs(q) < 1e-9) {
191  continue;
192  }
193  tools::vec vec1 = pos + 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
194  tools::vec vec2 = pos - 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
195  orb.AddAtom(Atomnameplus[i], vec1, q, true);
196  orb.AddAtom(Atomnameminus[i], vec2, q, true);
197 
198  }
199 
200  }
201 
202  return;
203  }
