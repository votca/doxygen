27 ctp::APolarSite *QMInterface::Convert(QMAtom *atm, int id) {
28 
29  std::string elem = atm->getType();
30  ctp::APolarSite *new_aps = new ctp::APolarSite(id, elem);
31 
32  double pol = 0.0;
33  try {
34  pol = _element.getPolarizability(elem);
35  } catch (const std::invalid_argument &out_of_range) {
36  std::cout << std::endl
37  << "QMMInterface - no default polarizability given "
38  << "for element type '" << elem << "'. Defaulting to 1A**3"
39  << std::flush;
40  pol = 1e-3;
41  }
42  new_aps->setIsoP(pol);
43 
44  tools::vec pos = atm->getPos() * tools::conv::bohr2nm;
45  double q = atm->getPartialcharge();
46  new_aps->setRank(0);
47  new_aps->setPos(pos);
48  new_aps->setQ00(q, 0); // <- charge state 0 <> 'neutral'
49 
50  return new_aps;
51 }

63 std::vector<QMAtom *> QMInterface::Convert(
64  std::vector<ctp::Segment *> segments) {
65  std::vector<QMAtom *> qmatoms;
66  int AtomId = 0;
67  for (ctp::Segment *segment : segments) {
68  std::vector<ctp::Atom *> &atoms = segment->Atoms();
69  for (ctp::Atom *atom : atoms) {
70  if (!atom->HasQMPart()) {
71  continue;
72  }
73  tools::vec pos = atom->getQMPos() * tools::conv::nm2bohr;
74  std::string name = atom->getElement();
75  QMAtom *qmatom = new QMAtom(AtomId, name, pos);
76  AtomId++;
77  qmatoms.push_back(qmatom);
78  }
79  }
80  return qmatoms;
81 }

83 void QMInterface::GenerateQMAtomsFromPolarSegs(ctp::PolarTop *ptop,
84  Orbitals &orb) {
85  int AtomID = 0;
86  // INNER SHELL QM0
87  for (unsigned int i = 0; i < ptop->QM0().size(); ++i) {
88  ctp::PolarSeg *pseg = ptop->QM0()[i];
89  for (unsigned int j = 0; j < pseg->size(); ++j) {
90  ctp::APolarSite *aps = (*pseg)[j];
91  tools::vec pos = aps->getPos() * tools::conv::nm2bohr;
92  orb.AddAtom(AtomID, aps->getName(), pos);
93  AtomID++;
94  }
95  }
96 
97  return;
98 }

120 void QMInterface::Orbitals2Segment(ctp::Segment &segment,
121  const Orbitals &orbitals) {
122 
123  std::vector<QMAtom *> _atoms = orbitals.QMAtoms();
124  std::string type;

129  tools::vec position = atom->getPos() * votca::tools::conv::bohr2nm;
130  pAtom->setPos(position);
131  pAtom->setQMPart(id, position);
132  pAtom->setElement(type);
133  segment.AddAtom(pAtom);
134  }
