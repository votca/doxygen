56  // sets preface strings for Log::error, Log::warning, ...
57  void setPreface(Log::Level level, std::string preface) {
58  switch (level) {
59 
60  case Log::Level::error:
61  _errorPreface = preface;
62  break;
63  case Log::Level::warning:
64  _warnPreface = preface;
65  break;
66  case Log::Level::info:
67  _infoPreface = preface;
68  break;
69  case Log::Level::debug:
70  _dbgPreface = preface;
71  break;
72  }
73  }

75  void EnablePreface() { _writePreface = true; }
76  void DisablePreface() { _writePreface = false; }
77 
78  // flushes all collected messages
79  void FlushBuffer() {
80  std::cout << _stringStream.str();
81  _stringStream.str("");
82  }
83 
84  // returns the pointer to the collected messages
85  std::string Messages() {
86  std::string _messages = _stringStream.str();
87  _stringStream.str("");
88  return _messages;
89  }
90 
91  private:
92  // Log Level (WARNING, INFO, etc)
93  Log::Level _LogLevel = Log::Level::error;
94 
95  // temporary buffer to store messages
96  std::ostringstream _stringStream;
97 
98  // Multithreading
99  bool _maverick = true;
100 
101  std::string _errorPreface = "\n ERROR ";
102  std::string _warnPreface = "\n WARNING ";
103  std::string _infoPreface = "\n ";
104  std::string _dbgPreface = "\n DEBUG ";
105  bool _writePreface = true;
106 
107  protected:
108  int sync() override {
109 
110  std::ostringstream _message;
111 
112  if (_writePreface) {
113  switch (_LogLevel) {
114  case Log::Level::error:
115  _message << _errorPreface;
116  break;
117  case Log::Level::warning:
118  _message << _warnPreface;
119  break;
120  case Log::Level::info:
121  _message << _infoPreface;
122  break;
123  case Log::Level::debug:
124  _message << _dbgPreface;
125  break;
126  }
127  }
128 
129  if (!_maverick) {
130  // collect all messages of one thread
131  _stringStream << _message.str() << " " << str();
132  } else {
133  // if only one thread outputs, flush immediately
134  std::cout << _message.str() << " " << str() << std::flush;
135  }
136  _message.str("");
137  str("");
138  return 0;
139  }
