51 inline void invert_symm( const ub::matrix<double> &A, ub::matrix<double> &V){
52  // matrix inversion using gsl
53 
54  #ifdef DEBUG_LINALG
55  auto start = std::chrono::system_clock::now();
56  #endif
57 
58  gsl_error_handler_t *handler = gsl_set_error_handler_off();
59  const size_t N = A.size1();
60  // signum s (for LU decomposition)
61  int s;
62  //make copy of A as A is destroyed by GSL
63  ub::matrix<double> work=A;
64  V.resize(N, N, false);
65 
66  // Define all the used matrices
67  gsl_matrix_view A_view = gsl_matrix_view_array(&work(0,0), N, N);
68  gsl_matrix_view V_view = gsl_matrix_view_array(&V(0,0), N, N);
69  gsl_permutation * perm = gsl_permutation_alloc (N);
70 
71  // Make LU decomposition of matrix A_view
72  gsl_linalg_LU_decomp (&A_view.matrix, perm, &s);
73 
74  // Invert the matrix A_view
75  (void)gsl_linalg_LU_invert (&A_view.matrix, perm, &V_view.matrix);
76 
77  gsl_set_error_handler(handler);
78 
79  #ifdef DEBUG_LINALG
80  auto end = std::chrono::system_clock::now();
81 
82  std::chrono::duration<double> elapsed_seconds = end - start;
83 
84  std::cout << "\n... ... ... \x1b[0;34mGSL Invert"
85  << "[" << A.size1() << "x" << A.size2() << "]"
86  << " time: " << elapsed_seconds.count()
87  << "s\x1b[0;39m";
88  #endif

148 /*
149  * Since ublas has very slow prod() function, we overwrite it here
150  * by the gsl (blas) product function. GSL has separate implementations
151  * for floats and doubles hence we have two specializations for these types
152  * Separate templates are provided for diagonal and symmetric matreces
153  * and their combinations. To check whether a specific routine is called,
154  * define the GSLDEBUG flag
155  */

158 // specialization for double precision (dgemm)
159 template<class F, class A>
160 inline matrix<double,F,A> // prod( m1, m2 )
161 prod(const matrix<double,F,A> &m1, const matrix<double,F,A> &m2)
162 {
163 
164  #ifdef DEBUG_LINALG
165  auto start = std::chrono::system_clock::now();
166  #endif
167 
168  gsl_matrix_const_view mA = gsl_matrix_const_view_array (&m1(0,0), m1.size1(), m1.size2());
169  gsl_matrix_const_view mB = gsl_matrix_const_view_array (&m2(0,0), m2.size1(), m2.size2());
170 
171  boost::numeric::ublas::matrix<double,F,A> AxB( m1.size1(), m2.size2() );
172  gsl_matrix_view mC = gsl_matrix_view_array (&AxB(0,0), AxB.size1(), AxB.size2());
173 
174 #ifdef NDEBUG
175  (void)gsl_blas_dgemm (CblasNoTrans, CblasNoTrans,
176  1.0, &mA.matrix, &mB.matrix,
177  0.0, &mC.matrix);
178 #else
179  int status = gsl_blas_dgemm (CblasNoTrans, CblasNoTrans,
180  1.0, &mA.matrix, &mB.matrix,
181  0.0, &mC.matrix);
182  assert(status == 0);
183 #endif
184 
185  #ifdef DEBUG_LINALG
186  auto end = std::chrono::system_clock::now();
187 
188  std::chrono::duration<double> elapsed_seconds = end - start;
189 
190  std::cout << "\n... ... ... \x1b[0;34mGSL "
191  << "[" << m1.size1() << "x" << m1.size2() << "]"
192  << "*[" << m2.size1() << "x" << m2.size2() << "]"
193  << " time: " << elapsed_seconds.count()
194  << "s\x1b[0;39m";
195  #endif
196 
197  return AxB;
198 
199 }
