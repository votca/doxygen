205  int L = 2;
206  vector<double> QCG(L*L+2*L+1, 0.0); // permanent
207  vector<double> uQCG(L*L+2*L+1, 0.0); // induced
208 
209  for (PolarSeg::iterator pit = begin();
210  pit < end(); ++pit) {
211  // PERMANENT MOMENTS
212  // Convert real to complex moments

282  vector<APolarSite*> cg_sites;
283  for (vector<PolarFrag*>::iterator fit = _pfrags.begin();
284  fit < _pfrags.end(); ++fit) {

286  vec target_pos = (*fit)->CalcPosPolarWeights();
287  int state = 0;
288  int L = 2;
289  vector<double> QCG(L*L+2*L+1, 0.0); // permanent
290  vector<double> uQCG(L*L+2*L+1, 0.0); // induced
291  vec u1_cg_sum = vec(0,0,0);
292  for (PolarFrag::iterator pit = (*fit)->begin();
293  pit < (*fit)->end(); ++pit) {
294  // PERMANENT MOMENTS
295  // Convert real to complex moments
296  vector<double> Qlm = (*pit)->getQs(0);
297  DMA::ComplexSphericalMoments Xlm(Qlm);
298  // Shift moments
299  DMA::MomentShift mshift;
300  vec shift = target_pos - (*pit)->getPos();
301  DMA::RegularSphericalHarmonics Clm(-shift);
302  vector<DMA::cmplx> Xlm_shifted = mshift.Shift(Xlm, Clm);
303  // Convert complex to real moments & add to base
304  DMA::RealSphericalMoments Qlm_shifted(Xlm_shifted);
305  Qlm_shifted.AddToVector(QCG);
306

312 // DMA::RealSphericalMoments Qlm_back(Xlm_shifted_back);
313 // cout << endl << "restored";
314 // Qlm_back.PrintReal();
315 
316  // INDUCED MOMENTS
317  u1_cg_sum += (*pit)->getU1();
318  // Convert real to complex moments
319  vec u1 = (*pit)->getU1();
320  vector<double> uQlm(L*L+2*L+1, 0.0);
321  uQlm[1] = u1.getZ(); // NOTE order is z-x-y == 10-11c-11s
322  uQlm[2] = u1.getX();
323  uQlm[3] = u1.getY();
324  DMA::ComplexSphericalMoments uXlm(uQlm);
325  // Shift moments
326  DMA::RegularSphericalHarmonics uClm(-shift);
327  vector<DMA::cmplx> uXlm_shifted = mshift.Shift(uXlm, uClm);
328  // Convert complex to real moments & add to base
329  DMA::RealSphericalMoments uQlm_shifted(uXlm_shifted);
330  uQlm_shifted.AddToVector(uQCG);
331  }
332 
333  // Induced moments: discard moments of rank > 1
334  vec u1_cg_red = vec(uQCG[2], uQCG[3], uQCG[1]); // order z-x-y to x-y-z
335  assert(votca::tools::abs(u1_cg_sum-u1_cg_red) < 1e-9
336  && "<PolarSeg::CoarseGrain> INDUCED MOMENTS TRAFO - ERROR");
337  // Collapse polarizabilities
338  MolPolEngine engine = MolPolEngine();
339  votca::tools::matrix PCG = engine.CalculateMolPol(*(*fit), tools::globals::verbose && false);
340  if (cg_anisotropic == false) {
341  // Reduce to isotropic tensor
342  votca::tools::matrix::eigensystem_t pcg_eigen;
343  PCG.SolveEigensystem(pcg_eigen);
344  double p11 = pcg_eigen.eigenvalues[0];
345  double p22 = pcg_eigen.eigenvalues[1];
346  double p33 = pcg_eigen.eigenvalues[2];
347  double piso = pow(p11*p22*p33, 1./3.);
348  PCG = votca::tools::matrix(
349  vec(piso,0,0), vec(0,piso,0), vec(0,0,piso));
350  }
351 
352  // Generate new coarse-grained site from the above
353  APolarSite *cg_site = new APolarSite((*fit)->getId(), (*fit)->getName());
354  cg_site->setResolution(APolarSite::coarsegrained);
355  cg_site->setPos(target_pos);
356  cg_site->setRank(L);
357  cg_site->setQs(QCG, state);
358  cg_site->setPs(PCG, state);
359  cg_site->setU1(u1_cg_red);
360  cg_site->Charge(state);
361  cg_sites.push_back(cg_site);
362  // Clear fragment & reload
363  (*fit)->clear();
364  (*fit)->push_back(cg_site);
365  }

370  for (pit = begin(); pit < end(); ++pit) delete *pit;
371  clear();
