76 bool DLPTopolApp::EvaluateTopology(Topology *top, Topology *top_ref)
77 {
78  // check the file names from options
79 
80  string fname=OptionsMap()["top"].as<string>();
81 
82  if( fname == ".dlpf" ) {
83  fname = "FIELD";
84  }
85 
86  cout << "input file-name: " << fname << endl;
87 
88  fname=OptionsMap()["out"].as<string>();
89 
90 #ifdef DEBUG
91  cout << "output file-name given: " << fname << endl;
92 #endif
93 
94  if( fname == ".dlpf" ) {
95  fname = "FIELD_CGV";
96  }
97 
98 #ifdef DEBUG
99  cout << "output file-name actual: " << fname << endl;
100 #else
101  cout << "output file-name: " << fname << endl;
102 #endif
103 
104  // do CG mapping
105 
106  MoleculeContainer &mols = top->Molecules();
107  MoleculeContainer MolecularTypes;
108  MoleculeContainer::iterator iter;
109 
110  int prv_mol_number = 1;
111  string prv_mol_name;
112  vector<int> nummols;
113 
114  vector<string> vdw_pairs;
115 
116  // find all unique molecular types
117 
118  for(iter=mols.begin(); iter!=mols.end(); ++iter) {
119  Molecule *mol = *iter;
120 
121  // molecules are ignored during the mapping stage
122  // i.e. the ignored ones do not enter the CG topology (*top) - ?
123  //if( IsIgnored(mol->getName()) ) continue;
124 
125  if( mol->getName()==prv_mol_name ) {
126  prv_mol_number++;
127  continue;
128  }
129 
130  nummols.push_back(prv_mol_number);
131  prv_mol_number = 1;
132  prv_mol_name = mol->getName();
133 
134  //#ifdef DEBUG
135  cout << "'" << mol->getName() << "' added to CG molecular types" << endl;
136  //#endif
137 
138  MolecularTypes.push_back(mol);
139 
140  // collect unique bead pairs over all molecular types found
141 
142  for(int ib1=0; ib1<mol->BeadCount(); ib1++) {
143  string bead_name1 = mol->getBead(ib1)->getType()->getName();
144  bead_name1 = bead_name1.substr(0,bead_name1.find_first_of("#")); // skip #index of atom from its name
145 
146  for(unsigned int imt=0; imt<MolecularTypes.size(); imt++) {
147 
148  for(int ib2=0; ib2<MolecularTypes[imt]->BeadCount(); ib2++) {
149 
150  string bead_name2 = MolecularTypes[imt]->getBead(ib2)->getType()->getName();
151  bead_name2 = bead_name2.substr(0,bead_name2.find_first_of("#")); // skip #index of atom from its name
152 
153  stringstream ss_bp1,ss_bp2;
154 
155  ss_bp1 << format("%8s%8s" ) % bead_name1 % bead_name2;
156  ss_bp2 << format("%8s%8s" ) % bead_name2 % bead_name1;
157 
158  bool is_new_pair=true;
159 
160  for(unsigned int ibp=0; ibp<vdw_pairs.size(); ibp++) {
161  if( ss_bp1.str()==vdw_pairs[ibp] || ss_bp2.str()==vdw_pairs[ibp] ) {
162  is_new_pair=false;
163  break;
164  }
165  }
166  if( is_new_pair ) {
167  vdw_pairs.push_back(ss_bp1.str());
168 #ifdef DEBUG
169  cout << "'" << ss_bp1.str() << "' added to CG vdw pairs" << endl;
170 #endif
171  }
172  }
173  }
174  }
175  }
176  nummols.push_back(prv_mol_number);
177 
178  if(MolecularTypes.size() > 1)
179  cout << "WARNING: creation of topology for multiple molecular types "
180  "is experimental at this stage\n";
181 
182  ofstream fl;
183  fl.open(fname.c_str());
184 
185  fl << "From VOTCA with love" << " # check/amend this file if needed!\n";
186  fl << "units kJ\n";
187  fl << "molecular types " << MolecularTypes.size() << endl;
188 
189  for(unsigned int i=0; i<MolecularTypes.size(); i++) {
190  WriteMolecularType(fl, *(MolecularTypes[i]), nummols[i+1]);
191  }
192 
193  // vdw seciton (pairwise vdw/CG interactions)
194 
195  if( vdw_pairs.size()>0 ) {
196 
197  fl << "vdw "<< vdw_pairs.size() << endl;
198 
199  for(unsigned int ibp=0; ibp<vdw_pairs.size(); ibp++) {
200  fl << vdw_pairs[ibp] << " tab 1.00000 0.00000\n";
201  }
202  }
203 
204  fl << "close" << endl;
205 
206  cout << "Created template for dlpoly topology - please, check & amend if needed!" << endl;
207 
208  fl.close();
209  return true;
210 }

213 void DLPTopolApp::WriteMoleculeAtoms(ostream &out, Molecule &cg)
214 {
215  out << "atoms " << cg.BeadCount() << endl;
216  out << "# name mass charge nrept ifrozen (optional: ngroup, index, name/type, type/residue, index/res-ID) \n";
217  for(int i=0; i<cg.BeadCount(); ++i) {
218  Bead *b=cg.getBead(i);
219 
220  string bname=b->getName();
221  string btype=b->getType()->getName();
222 
223  bname = bname.substr(0,bname.find_first_of("#")); // skip #index of atom from its name
224  btype = btype.substr(0,btype.find_first_of("#")); // skip #index of atom from its type
225 
226  out << format("%8s %10f %10f 1 0 1 %10d %8s %8s %10d \n")
227  % btype % b->getM() % b->getQ() % (i+1) % btype % bname % (i+1);
228  //% b->getType()->getName() % b->getM() % b->getQ() % (i+1) % b->getType()->getName() % b->getName() % (i+1);
229  }
230 }

232 void DLPTopolApp::WriteMoleculeInteractions(ostream &out, Molecule &cg)
233 {
234  InteractionContainer ics=cg.Interactions();
235  vector<Interaction *>::iterator iter;
236 
237  stringstream sout;
238 
239  int n_entries = 0;
240  int nb=-1;
241 
242  for(iter=ics.begin(); iter!=ics.end(); ++iter) {
243  Interaction *ic = *iter;
244  if(nb != ic->BeadCount()) {
245 
246  if(sout.str()!="")
247  out << n_entries << endl << sout.str();
248 
249  sout.str("");
250  n_entries = 0;
251 
252  nb=ic->BeadCount();
253  switch(nb) {
254  case 2:
255  out << "bonds ";
256  break;
257  case 3:
258  out << "angles ";
259  break;
260  case 4:
261  out << "dihedrals ";
262  break;
263  default:
264  throw runtime_error(string("cannot handle number of beads in interaction:") +
265  ic->getName());
266  }
267  }
268  n_entries++;
269  // to do: is it possible to use bond/angle/dihedral function types for 1:1 mapping? (CG map overwrites ic->Group anyway)
270  //sout << ic->getInteractionFunc(); // something like that (only for 1:1 mapping!)
271  sout << " tab ";
272  for(int i=0; i<nb; ++i)
273  sout << ic->getBeadId(i)+1 << " ";
274  sout << " 1.00000 0.00000" << " # " << ic->getName() << endl;
275  //sout << " 1.00000 0.00000" << " # " << ic->getGroup() << endl;
276  }
277  if(sout.str()!="") out << n_entries << endl << sout.str();
278 }
