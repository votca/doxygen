102 void DLPTopolApp::WriteAtoms(ostream &out, Molecule &cg)
103 {
104  out << "ATOMS " << cg.BeadCount() << "\n";
105  out << "# name mass charge nrept ifrozen (optional: ngroup, index, type, residue) \n";
106  for(int i=0; i<cg.BeadCount(); ++i) {
107  Bead *b=cg.getBead(i);
108 
109  out << format("%4s %f %f 1 0 1 %d %s \n")
110  % b->getType()->getName() % b->getM() % b->getQ() % (i+1) % b->getName();
111  }
112 }

114 void DLPTopolApp::WriteInteractions(ostream &out, Molecule &cg)
115 {
116  int nb=-1;
117 
118  Interaction *ic;
119  vector<Interaction *>::iterator iter;
120 
121  InteractionContainer &ics=cg.getParent()->BondedInteractions();
122 
123  int n_entries = ics.end()-ics.begin()+1;
124 
125  for(iter=ics.begin(); iter!=ics.end(); ++iter) {
126  ic = *iter;
127  if(ic->getMolecule() != cg.getId()) continue;
128  if(nb != ic->BeadCount()) {
129  nb=ic->BeadCount();
130  switch(nb) {
131  case 2:
132  out << "bonds ";
133  break;
134  case 3:
135  out << "angles ";
136  break;
137  case 4:
138  out << "dihedrals ";
139  break;
140  default:
141  throw runtime_error(string("cannot handle number of beads in interaction:") +
142  ic->getName());
143  }
144  out << " -1 # check/amend the number of interactions!\n";
145  //out << n_entries << endl;
146  }
147  out << " tab ";
148  for(int i=0; i<nb; ++i)
149  out << ic->getBeadId(i)+1 << " ";
150  out << " # " << ic->getName() << endl;
151  }
152 }
