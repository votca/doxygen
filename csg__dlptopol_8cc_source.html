78 bool DLPTopolApp::EvaluateTopology(Topology *top, Topology *) {
79  // check the file names from options
80 
81  string fname = OptionsMap()["top"].as<string>();
82 
83  if (fname == ".dlpf") {
84  fname = "FIELD";
85  }
86 
87  cout << "input file-name: " << fname << endl;
88 
89  fname = OptionsMap()["out"].as<string>();
90 
91 #ifdef DEBUG
92  cout << "output file-name given: " << fname << endl;
93 #endif
94 
95  if (fname == ".dlpf") {
96  fname = "FIELD_CGV";
97  }
98 
99 #ifdef DEBUG
100  cout << "output file-name actual: " << fname << endl;
101 #else
102  cout << "output file-name: " << fname << endl;
103 #endif
104 
105  // do CG mapping
106 
107  MoleculeContainer &mols = top->Molecules();
108  MoleculeContainer MolecularTypes;
109 
110  votca::Index prv_mol_number = 1;
111  string prv_mol_name;
112  vector<votca::Index> nummols;
113 
114  vector<string> vdw_pairs;
115 
116  // find all unique molecular types
117 
118  for (Molecule *mol : mols) {
119  // molecules are ignored during the mapping stage
120  // i.e. the ignored ones do not enter the CG topology (*top) - ?
121  // if( IsIgnored(mol->getName()) ) continue;
122 
123  if (mol->getName() == prv_mol_name) {
124  prv_mol_number++;
125  continue;
126  }
127 
128  nummols.push_back(prv_mol_number);
129  prv_mol_number = 1;
130  prv_mol_name = mol->getName();
131 
132  //#ifdef DEBUG
133  cout << "'" << mol->getName() << "' added to CG molecular types" << endl;
134  //#endif
135 
136  MolecularTypes.push_back(mol);
137 
138  // collect unique bead pairs over all molecular types found
139 
140  for (votca::Index ib1 = 0; ib1 < mol->BeadCount(); ib1++) {
141  string bead_name1 = mol->getBead(ib1)->getType();
142  bead_name1 = bead_name1.substr(
143  0,
144  bead_name1.find_first_of("#")); // skip #index of atom from its name
145 
146  for (auto &MolecularType : MolecularTypes) {
147 
148  for (votca::Index ib2 = 0; ib2 < MolecularType->BeadCount(); ib2++) {
149 
150  string bead_name2 = MolecularType->getBead(ib2)->getType();
151  bead_name2 = bead_name2.substr(
152  0, bead_name2.find_first_of("#")); // skip #index of atom from
153  // its name
154 
155  stringstream ss_bp1, ss_bp2;
156 
157  ss_bp1 << format("%8s%8s") % bead_name1 % bead_name2;
158  ss_bp2 << format("%8s%8s") % bead_name2 % bead_name1;
159 
160  bool is_new_pair = true;
161 
162  for (const auto &vdw_pair : vdw_pairs) {
163  if (ss_bp1.str() == vdw_pair || ss_bp2.str() == vdw_pair) {
164  is_new_pair = false;
165  break;
166  }
167  }
168  if (is_new_pair) {
169  vdw_pairs.push_back(ss_bp1.str());
170 #ifdef DEBUG
171  cout << "'" << ss_bp1.str() << "' added to CG vdw pairs" << endl;
172 #endif
173  }
174  }
175  }
176  }
177  }
178  nummols.push_back(prv_mol_number);
179 
180  if (MolecularTypes.size() > 1) {
181  cout << "WARNING: creation of topology for multiple molecular types "
182  "is experimental at this stage\n";
183  }
184 
185  ofstream fl;
186  fl.open(fname);
187 
188  fl << "From VOTCA with love"
189  << " # check/amend this file if needed!\n";
190  fl << "units kJ\n";
191  fl << "molecular types " << MolecularTypes.size() << endl;
192 
193  for (votca::Index i = 0; i < votca::Index(MolecularTypes.size()); i++) {
194  WriteMolecularType(fl, *(MolecularTypes[i]), nummols[i + 1]);
195  }
196 
197  // vdw seciton (pairwise vdw/CG interactions)
198 
199  if (vdw_pairs.size() > 0) {
200 
201  fl << "vdw " << vdw_pairs.size() << endl;
202 
203  for (const auto &vdw_pair : vdw_pairs) {
204  fl << vdw_pair << " tab 1.00000 0.00000\n";
205  }
206  }
207 
208  fl << "close" << endl;
209 
210  cout << "Created template for dlpoly topology - please, check & amend if "
211  "needed!"
212  << endl;
213 
214  fl.close();
215  return true;
216 }

218 void DLPTopolApp::WriteMoleculeAtoms(ostream &out, Molecule &cg) {
219  out << "atoms " << cg.BeadCount() << endl;
220  out << "# name mass charge nrept ifrozen (optional: ngroup, index, "
221  "name/type, type/residue, index/res-ID) \n";
222  for (votca::Index i = 0; i < cg.BeadCount(); ++i) {
223  Bead *b = cg.getBead(i);
224 
225  string bname = b->getName();
226  string btype = b->getType();
227 
228  bname = bname.substr(
229  0, bname.find_first_of("#")); // skip #index of atom from its name
230  btype = btype.substr(
231  0, btype.find_first_of("#")); // skip #index of atom from its type
232 
233  out << format("%8s %10f %10f 1 0 1 %10d %8s %8s %10d \n") %
234  btype % b->getMass() % b->getQ() % (i + 1) % btype % bname %
235  (i + 1);
236  //% b->getType()->getName() % b->getMass() % b->getQ() % (i+1) %
237  // b->getType()->getName() % b->getName() % (i+1);
238  }

241 void DLPTopolApp::WriteMoleculeInteractions(ostream &out, Molecule &cg) {
242 
243  stringstream sout;
244 
245  votca::Index n_entries = 0;
246  votca::Index nb = -1;
247 
248  for (Interaction *ic : cg.Interactions()) {
249  if (nb != ic->BeadCount()) {
250 
251  if (sout.str() != "") {
252  out << n_entries << endl << sout.str();
253  }
254 
255  sout.str("");
256  n_entries = 0;
257 
258  nb = ic->BeadCount();
259  switch (nb) {
260  case 2:
261  out << "bonds ";
262  break;
263  case 3:
264  out << "angles ";
265  break;
266  case 4:
267  out << "dihedrals ";
268  break;
269  default:
270  string err = "cannot handle number of beads in interaction:";
271  err += to_string(ic->getMolecule() + 1) + ":" + ic->getGroup();
272  err += ":" + to_string(ic->getIndex() + 1);
273  throw runtime_error(err);
274  }
275  }
276  n_entries++;
277  // to do: is it possible to use bond/angle/dihedral function types for 1:1
278  // mapping? (CG map overwrites ic->Group anyway)
279  // sout << ic->getInteractionFunc(); // something like that (only for 1:1
280  // mapping!)
281  sout << " tab ";
282  for (votca::Index i = 0; i < nb; ++i) {
283  sout << ic->getBeadId(i) + 1 << " ";
284  }
285  sout << " 1.00000 0.00000"
286  << " # ";
287  sout << to_string(ic->getMolecule() + 1);
288  sout << ":" + ic->getGroup();
289  sout << ":" + to_string(ic->getIndex() + 1) << endl;
290  }
291  if (sout.str() != "") {
292  out << n_entries << endl << sout.str();
293  }
294 }
