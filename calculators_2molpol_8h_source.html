359 int MolPol::SCF_Induce(vector<APolarSite*> &poles) {
360 
361  int maxIter = this->_maxIter;
362  double wSOR = this->_wSOR;
363  double eTOL = this->_epsTol;
364 
365  vector< APolarSite* >::iterator pit1;
366  vector< APolarSite* >::iterator pit2;
367 
368  // Induce to first order
369  for (pit1 = poles.begin(); pit1 < poles.end(); ++pit1) {
370  (*pit1)->InduceDirect();
371  }
372 
373  int iter = 0;
374  for ( ; iter < maxIter; ++iter) {
375 
376  // Reset induction field
377  for (pit1 = poles.begin(); pit1 < poles.end(); ++pit1) {
378  (*pit1)->ResetFieldU();
379  }
380 
381  // Calculate higher-order induction field
382  for (pit1 = poles.begin(); pit1 < poles.end(); ++pit1) {
383  for (pit2 = pit1 + 1; pit2 < poles.end(); ++pit2) {
384  _actor.FieldInduAlpha(*(*pit1),*(*pit2));
385  }
386  }
387 
388  // Induce dipoles
389  for (pit1 = poles.begin(); pit1 < poles.end(); ++pit1) {
390  (*pit1)->Induce(wSOR);
391  }
392 
393  // Check for convergence
394  bool converged = true;
395  double maxdU = -1;
396  double avgdU = 0.0;
397  int baseN = 0;
398 
399  for (pit1 = poles.begin(); pit1 < poles.end(); ++pit1) {
400  double dU = (*pit1)->HistdU();
401  avgdU += dU;
402  ++baseN;
403  if ( dU > maxdU ) { maxdU = dU; }
404  if ( dU > eTOL ) { converged = false; }
405  }
406 
407  avgdU /= baseN;
408  if (avgdU < eTOL/10.) { converged = true; }
409 
410  //cout << endl << "ITER" << iter << " | MAX dU " << maxdU
411  // << " | AVG dU " << avgdU;
412 
413  // Break if converged
414  if (converged) { break; }
415  else if (iter == maxIter - 1) {
416  cout << endl << "... ... ... "
417  << "WARNING Induced multipoles did not converge to precision: "
418  << " AVG dU:U " << avgdU << flush;
419  break;
420  }
421  }
422 
423  return iter;
424 }
