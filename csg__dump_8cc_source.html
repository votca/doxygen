48 bool CsgDumpApp::EvaluateTopology(Topology *top, Topology *) {
49  if (!OptionsMap().count("excl")) {
50  cout << "Boundary Condition: ";
51  if (top->getBoxType() == BoundaryCondition::typeAuto) {
52  cout << "auto";
53  } else if (top->getBoxType() == BoundaryCondition::typeTriclinic) {
54  cout << "triclinic";
55  } else if (top->getBoxType() == BoundaryCondition::typeOrthorhombic) {
56  cout << "orthorhombic";
57  } else if (top->getBoxType() == BoundaryCondition::typeOpen) {
58  cout << "open";
59  }
60  cout << endl;
61  if (top->getBoxType() != BoundaryCondition::typeOpen) {
62  cout << " Box matix:";
63  Eigen::Matrix3d box = top->getBox();
64  for (votca::Index i = 0; i < 3; i++) {
65  for (votca::Index j = 0; j < 3; j++) {
66  cout << " " << box(i, j);
67  }
68  cout << endl << " ";
69  }
70  }
71 
72  cout << "\nList of residues:\n";
73  for (votca::Index i = 0; i < top->ResidueCount(); i++) {
74  cout << i << " name: " << top->getResidue(i)->getName()
75  << " id: " << top->getResidue(i)->getId() << endl;
76  }
77 
78  cout << "\nList of molecules:\n";
79  for (Molecule *mol : top->Molecules()) {

81  << " beads: " << mol->BeadCount() << endl;

88  << top->getResidue(resnr)->getName() << " Charge "
89  << mol->getBead(i)->getQ() << endl;
90  }
91  }
92  } else {
