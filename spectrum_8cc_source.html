31 
32  // orbitals file or pure DFT output
33  _orbfile = options.ifExistsReturnElseThrowRuntimeError<std::string>(
34  key + ".orbitals");

44  _spectrum_type =

46  key + ".type", choices);

50  _maxexc =
51  options.ifExistsReturnElseReturnDefault<int>(key + ".maxexc", _maxexc);
52  _shiftby =
53  options.ifExistsReturnElseReturnDefault<double>(key + ".shift", _shiftby);
54

58 bool Spectrum::Evaluate() {
59  OPENMP::setMaxThreads(_nThreads);
60  _log.setReportLevel(logDEBUG);
61  _log.setMultithreading(true);
62 
63  _log.setPreface(logINFO, "\n... ...");
64  _log.setPreface(logERROR, "\n... ...");
65  _log.setPreface(logWARNING, "\n... ...");
66  _log.setPreface(logDEBUG, "\n... ...");
67 
68  XTP_LOG_SAVE(logDEBUG, _log)
69  << "Calculating absorption spectrum plot " << _orbfile << std::flush;
70 
71  Orbitals orbitals;
72  // load the QM data from serialized orbitals object
73  XTP_LOG_SAVE(logDEBUG, _log)
74  << " Loading QM data from " << _orbfile << std::flush;
75  orbitals.ReadFromCpt(_orbfile);
76 
77  // check if orbitals contains singlet energies and transition dipoles
78  if (!orbitals.hasBSESinglets()) {
79  throw std::runtime_error(
80  "BSE singlet energies not stored in QM data file!");
81  }
82 
83  if (!orbitals.hasTransitionDipoles()) {
84  throw std::runtime_error(
85  "BSE transition dipoles not stored in QM data file!");
86  }
87 
88  const Eigen::VectorXd BSESingletEnergies =
89  orbitals.BSESinglets().eigenvalues() * tools::conv::hrt2ev;
90  const std::vector<Eigen::Vector3d>& TransitionDipoles =
91  orbitals.TransitionDipoles();
92  Eigen::VectorXd osc = orbitals.Oscillatorstrengths();
93 
94  if (_maxexc > int(TransitionDipoles.size())) {
95  _maxexc = int(TransitionDipoles.size()) - 1;
96  }
97 
98  int n_exc = _maxexc - _minexc + 1;
99  XTP_LOG_SAVE(logDEBUG, _log)
100  << " Considering " << n_exc << " excitation with max energy "
101  << BSESingletEnergies(_maxexc) << " eV / min wave length "
102  << evtonm(BSESingletEnergies[_maxexc - 1]) << " nm" << std::flush;
103 
104  /*
105  *
106  * For a single excitation, broaden by Lineshape function L(v-W)
107  * eps(v) = f * L(v-W)
108  *
109  * where
110  * v: energy
111  * f: oscillator strength in dipole-length gauge
112  * W: excitation energy
113  *
114  * Lineshape function depend on FWHM and can be
115  *
116  * Gaussian
117  * L(v-W) = 1/(sqrt(2pi)sigma) * exp(-0.5 (v-W)^2/sigma^2
118  *
119  *
120  * with sigma: derived from FWHM (FWHM/2.3548)
121  *
122  * Lorentzian
123  * L(v-W) = 1/pi * 0.5 FWHM/( (v-w)^2 + 0.25*FWHM^2 )
124  *
125  * Full spectrum is superposition of individual spectra.
126  *
127  * Alternatively, one can calculate the imaginary part of the
128  * frequency-dependent dielectric function
129  *
130  * IM(eps(v)) ~ 1/v^2 * W^2 * |td|^2 * L(v-W)
131  * = 1/v^2 * W * f * L(v-W)
132  *
133  *
134  */
135 
136  std::ofstream ofs(_output_file, std::ofstream::out);
137 
138  if (_spectrum_type == "energy") {
139  ofs << "# E(eV) epsGaussian IM(eps)Gaussian epsLorentz "
140  "Im(esp)Lorentz\n";
141  for (int i_pt = 0; i_pt <= _n_pt; i_pt++) {
142 
143  double e = (_lower + i_pt * (_upper - _lower) / _n_pt);
144 
145  double eps_Gaussian = 0.0;
146  double imeps_Gaussian = 0.0;
147  double eps_Lorentzian = 0.0;
148  double imeps_Lorentzian = 0.0;
149 
150  for (int i_exc = _minexc; i_exc <= _maxexc; i_exc++) {
151  eps_Gaussian +=
152  osc[i_exc] *
153  Gaussian(e, BSESingletEnergies(i_exc) + _shiftby, _fwhm);
154  imeps_Gaussian += osc[i_exc] * BSESingletEnergies(i_exc) *
155  Gaussian(e, BSESingletEnergies(i_exc), _fwhm);
156  eps_Lorentzian +=
157  osc[i_exc] * Lorentzian(e, BSESingletEnergies(i_exc), _fwhm);
158  imeps_Lorentzian += osc[i_exc] * BSESingletEnergies(i_exc) *
159  Lorentzian(e, BSESingletEnergies(i_exc), _fwhm);
160  }
161 
162  ofs << e << " " << eps_Gaussian << " " << imeps_Gaussian << " "
163  << eps_Lorentzian << " " << imeps_Lorentzian << std::endl;
164  }
165 
166  XTP_LOG_SAVE(logDEBUG, _log)
167  << " Spectrum in energy range from " << _lower << " to " << _upper
168  << " eV and with broadening of FWHM " << _fwhm
169  << " eV written to file " << _output_file << std::flush;
170  }
171 
172  if (_spectrum_type == "wavelength") {
173 
174  ofs << "# lambda(nm) epsGaussian IM(eps)Gaussian epsLorentz "
175  "Im(esp)Lorentz\n";
176  for (int i_pt = 0; i_pt <= _n_pt; i_pt++) {
177 
178  double lambda = (_lower + i_pt * (_upper - _lower) / _n_pt);
179  double eps_Gaussian = 0.0;
180  double imeps_Gaussian = 0.0;
181  double eps_Lorentzian = 0.0;
182  double imeps_Lorentzian = 0.0;
183 
184  for (int i_exc = _minexc; i_exc <= _maxexc; i_exc++) {
185  double exc_lambda = nmtoev(BSESingletEnergies(i_exc) + _shiftby);
186  eps_Gaussian += osc[i_exc] * Gaussian(lambda, exc_lambda, _fwhm);
187  imeps_Gaussian +=
188  osc[i_exc] * exc_lambda * Gaussian(lambda, exc_lambda, _fwhm);
189  eps_Lorentzian += osc[i_exc] * Lorentzian(lambda, exc_lambda, _fwhm);
190  imeps_Lorentzian +=
191  osc[i_exc] * exc_lambda * Lorentzian(lambda, exc_lambda, _fwhm);
192  }
193 
194  ofs << lambda << " " << eps_Gaussian << " " << imeps_Gaussian
195  << " " << eps_Lorentzian << " " << imeps_Lorentzian << std::endl;
196  }
197  XTP_LOG_SAVE(logDEBUG, _log)
198  << " Spectrum in wavelength range from " << _lower << " to " << _upper
199  << " nm and with broadening of FWHM " << _fwhm
200  << " nm written to file " << _output_file << std::flush;
201  }
202 
203  ofs.close();
204  return true;
205 }

207 double Spectrum::Lorentzian(double x, double center, double fwhm) {
208  return 0.5 * fwhm / (std::pow(x - center, 2) + 0.25 * fwhm * fwhm) /
209  boost::math::constants::pi<double>();
210 }

212 double Spectrum::Gaussian(double x, double center, double fwhm) {
213  // FWHM = 2*sqrt(2 ln2) sigma = 2.3548 sigma
214  double sigma = fwhm / 2.3548;
215  return std::exp(-0.5 * std::pow((x - center) / sigma, 2)) / sigma /
216  sqrt(2.0 * boost::math::constants::pi<double>());
217 }
