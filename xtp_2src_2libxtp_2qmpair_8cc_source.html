37  : std::pair<Segment*, Segment*>(seg1, seg2), _id(id),
38  _hasGhost(0),
39  _rate12_e(0), _rate21_e(0),
40  _rate12_h(0), _rate21_h(0),
41  _has_e(false), _has_h(false),

46  _has_s(false), _has_t(false),

48  _Jeff2_s(0), _Jeff2_t(0),_pair_type( Hopping ) {
49 
50  _top = seg1->getTopology();
51 
52  vec r1 = seg1->getPos();
53  vec r2 = seg2->getPos();
54 
55  _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
56 
57  // Check whether pair formed across periodic boundary
58  if ( abs(r2 - r1 - _R) > 1e-8 ) {
59 
60  _ghost = new Segment(seg2);
61  _ghost->TranslateBy(r1 - r2 + _R);
62  _hasGhost = true;
63 
64  }
65  else {
66  _ghost = NULL;
67  }
