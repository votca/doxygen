64 bool KMCApplication::EvaluateOptions() {
65 
66  if(OptionsMap().count("list")) {
67  cout << "Available calculators: \n";
68  for(KMCCalculatorFactory::assoc_map::const_iterator iter=Calculators().getObjects().begin();
69  iter != Calculators().getObjects().end(); ++iter) {
70  PrintDescription( std::cout, (iter->first).c_str(), "kmc/xml", Application::HelpShort );
71  }
72  StopExecution();
73  return true;
74  }
75 
76 
77  if(OptionsMap().count("description")) {
78  CheckRequired("description", "no calculator is given");
79  Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
80  // loop over the names in the description string
81  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
82  // loop over calculators
83  bool printerror = true;
84  for(KMCCalculatorFactory::assoc_map::const_iterator iter=Calculators().getObjects().begin();
85  iter != Calculators().getObjects().end(); ++iter) {
86 
87  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
88  PrintDescription( std::cout, (iter->first).c_str(), "kmc/xml", Application::HelpLong );
89  printerror = false;
90  break;
91  }
92  }
93  if ( printerror ) cout << "Calculator " << *n << " does not exist\n";
94  }
95  StopExecution();
96  return true;
97  }
98 
99  Application::EvaluateOptions();
100 
101  CheckRequired("execute", "no calculator is given");
102  Tokenizer tok(OptionsMap()["execute"].as<string>(), " ,\n\t");
103  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n)
104  AddCalculator(Calculators().Create((*n).c_str()));
105 
106  CheckRequired("options", "please provide an xml file with program options");
107  CheckRequired("file", "no database file specified");
108 
109  string _outputfile = "";
110 
111  if(OptionsMap().count("textfile"))
112  {
113  _outputfile = OptionsMap()["textfile"].as<string > ();
114  }
115  if(_outputfile != "")
116  {
117  cout << "Output into file: " << _outputfile.c_str() << "." << endl;
118  //freopen(_outputfile.c_str(),"w",stdout);
119  }
120  else
121  {
122  cout << " Output to screen." << endl;
123  }
124 
125  _filename = OptionsMap()["file"].as<string > ();
126  cout << " Database file: " << _filename << endl;
127  return true;
128 }
