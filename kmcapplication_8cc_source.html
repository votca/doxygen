58 // check if required options are provided
59 bool KMCApplication::EvaluateOptions() {
60 
61  if(OptionsMap().count("list")) {
62  cout << "Available calculators: \n";
63  for(KMCCalculatorFactory::assoc_map::const_iterator iter=Calculators().getObjects().begin();
64  iter != Calculators().getObjects().end(); ++iter) {
65  PrintDescription( std::cout, (iter->first).c_str(), "kmc/xml", Application::HelpShort );
66  }
67  StopExecution();
68  return true;
69  }
70 
71 
72  if(OptionsMap().count("description")) {
73  CheckRequired("description", "no calculator is given");
74  Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
75  // loop over the names in the description string
76  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
77  // loop over calculators
78  bool printerror = true;
79  for(KMCCalculatorFactory::assoc_map::const_iterator iter=Calculators().getObjects().begin();
80  iter != Calculators().getObjects().end(); ++iter) {
81 
82  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
83  PrintDescription( std::cout, (iter->first).c_str(), "kmc/xml", Application::HelpLong );
84  printerror = false;
85  break;
86  }
87  }
88  if ( printerror ) cout << "Calculator " << *n << " does not exist\n";
89  }
90  StopExecution();
91  return true;
92  }
93 
94  Application::EvaluateOptions();
95 
96  CheckRequired("execute", "no calculator is given");
97  Tokenizer tok(OptionsMap()["execute"].as<string>(), " ,\n\t");
98  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n)
99  AddCalculator(Calculators().Create((*n).c_str()));
100 
101  CheckRequired("options", "please provide an xml file with program options");
102  CheckRequired("file", "no database file specified");
103 
104  string _outputfile = "";
105  //if(OptionsMap()["textfile"] == NULL)
106  //{cout << "Mist!";}
107  if(OptionsMap().count("textfile"))
108  {
109  _outputfile = OptionsMap()["textfile"].as<string > ();
110  }
111  if(_outputfile != "")
112  {
113  //char char_outputfile[1024] = {_outputfile}; // max. 1024 characters for filename
114  //char_outputfile = _outputfile;
115  cout << "Output into file: " << _outputfile.c_str() << "." << endl;
116  //freopen(_outputfile.c_str(),"w",stdout);
117  //cout << "hier ist output" << endl;
118  }
119  else
120  {
121  cout << " Output to screen." << endl;
122  }
123 
124  _filename = OptionsMap()["file"].as<string > ();
125  cout << " Database file: " << _filename << endl;
126  return true;
127 }
