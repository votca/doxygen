38void KMCMultiple::ParseSpecificOptions(const tools::Property& options) {
39
40 runtime_ = options.get(".runtime").as<double>();
41 field_ = options.get(".field").as<Eigen::Vector3d>();
42 double mtobohr = 1E9 * tools::conv::nm2bohr;
43 field_ *=
44 (tools::conv::ev2hrt / mtobohr); // Converting from V/m to Hartree/bohr
45
46 outputtime_ = options.get(".outputtime").as<double>();
47 timefile_ = options.ifExistsReturnElseReturnDefault<std::string>(".timefile",
48 timefile_);
49
50 std::string carriertype = options.get(".carriertype").as<std::string>();
51 carriertype_ = QMStateType(carriertype);
52 if (!carriertype_.isKMCState()) {
53 throw std::runtime_error("KMC cannot be run for state:" +
54 carriertype_.ToLongString());
55 }
56
57 log_.setReportLevel(Log::current_level);
58 log_.setCommonPreface("\n ...");
59}

61void KMCMultiple::PrintDiffandMu(const Eigen::Matrix3d& avgdiffusiontensor,
62 double simtime, unsigned long step) {
63 double absolute_field = field_.norm();
64
65 if (absolute_field == 0) {
66 unsigned long diffusionsteps = step / diffusionresolution_;
67 Eigen::Matrix3d result =
68 avgdiffusiontensor /
69 (double(diffusionsteps) * 2.0 * simtime * double(numberofcarriers_));
70 XTP_LOG(Log::error, log_)
71 << "\nStep: " << step
72 << " Diffusion tensor averaged over all carriers (nm^2/s):\n"
73 << result * tools::conv::bohr2nm * tools::conv::bohr2nm << std::flush;
74 } else {
75 double average_mobility = 0;
76 double bohr2Hrts_to_nm2Vs =
77 tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
78 XTP_LOG(Log::error, log_) << "\nMobilities (nm^2/Vs): " << std::flush;
79 for (Index i = 0; i < numberofcarriers_; i++) {
80 Eigen::Vector3d velocity = carriers_[i].get_dRtravelled() / simtime;
81 double mobility =
82 velocity.dot(field_) / (absolute_field * absolute_field);
83 XTP_LOG(Log::error, log_)
84 << std::scientific << " carrier " << i + 1
85 << ": mu=" << mobility * bohr2Hrts_to_nm2Vs << std::flush;
86 average_mobility +=
87 velocity.dot(field_) / (absolute_field * absolute_field);
88 }
89 average_mobility /= double(numberofcarriers_);
90 XTP_LOG(Log::error, log_)
91 << std::scientific
92 << " Overall average mobility in field direction <mu>="
93 << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << std::flush;
94 }
95}

115void KMCMultiple::WriteToEnergyFile(std::fstream& tfile, double simtime,
116 unsigned long step) const {
117 double absolute_field = field_.norm();
118 double currentenergy = 0;
119 double currentmobility = 0;
120 Eigen::Vector3d dr_travelled_current = Eigen::Vector3d::Zero();
121 double dr_travelled_field = 0.0;
122 Eigen::Vector3d avgvelocity_current = Eigen::Vector3d::Zero();
123 if (absolute_field != 0) {
124 for (const auto& carrier : carriers_) {
125 dr_travelled_current += carrier.get_dRtravelled();
126 currentenergy += carrier.getCurrentEnergy();
127 }
128 dr_travelled_current /= double(numberofcarriers_);
129 currentenergy /= double(numberofcarriers_);
130 avgvelocity_current = dr_travelled_current / simtime;
131 currentmobility =
132 avgvelocity_current.dot(field_) / (absolute_field * absolute_field);
133 dr_travelled_field = dr_travelled_current.dot(field_) / absolute_field;
134 }
135 double bohr2Hrts_to_nm2Vs =
136 tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
137 tfile << simtime << "\t" << step << "\t"
138 << currentenergy * tools::conv::hrt2ev << "\t"
139 << currentmobility * bohr2Hrts_to_nm2Vs << "\t"
140 << dr_travelled_field * tools::conv::bohr2nm << "\t"
141 << dr_travelled_current.norm() * tools::conv::bohr2nm << "\t"
142 << std::endl;
143}

145void KMCMultiple::PrintDiagDandMu(const Eigen::Matrix3d& avgdiffusiontensor,
146 double simtime, unsigned long step) {
147 unsigned long diffusionsteps = step / diffusionresolution_;
148 Eigen::Matrix3d result =
149 avgdiffusiontensor /
150 (double(diffusionsteps) * 2.0 * simtime * double(numberofcarriers_));
151
152 Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
153 es.computeDirect(result);
154 double bohr2_nm2 = tools::conv::bohr2nm * tools::conv::bohr2nm;
155 XTP_LOG(Log::error, log_) << "\nEigenvalues:\n " << std::flush;
156 for (Index i = 0; i < 3; i++) {
157 XTP_LOG(Log::error, log_)
158 << "Eigenvalue: " << es.eigenvalues()(i) * bohr2_nm2 << std::flush
159 << "Eigenvector: " << es.eigenvectors().col(i).x() << " "
160 << es.eigenvectors().col(i).y() << " " << es.eigenvectors().col(i).z()
161 << "\n"
162 << std::flush;
163 }
164 double bohr2Hrts_to_nm2Vs =
165 tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
166 // calculate average mobility from the Einstein relation
167 if (field_.norm() == 0) {
168 XTP_LOG(Log::error, log_)
169 << "The following value is calculated using the Einstein relation "
170 "and assuming an isotropic medium"
171 << std::flush;
172 double avgD = 1. / 3. * es.eigenvalues().sum();
173 double average_mobility = std::abs(avgD / temperature_);
174 XTP_LOG(Log::error, log_)
175 << std::scientific << " Overall average mobility <mu>="
176 << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << std::flush;
177 }
178}

180void KMCMultiple::PrintChargeVelocity(double simtime) {
181 Eigen::Vector3d avg_dr_travelled = Eigen::Vector3d::Zero();
182 for (Index i = 0; i < numberofcarriers_; i++) {
183 XTP_LOG(Log::error, log_)
184 << std::scientific << " carrier " << i + 1 << ": "
185 << carriers_[i].get_dRtravelled().transpose() / simtime *
186 tools::conv::bohr2nm
187 << std::flush;
188 avg_dr_travelled += carriers_[i].get_dRtravelled();
189 }
190 avg_dr_travelled /= double(numberofcarriers_);
191
192 Eigen::Vector3d avgvelocity = avg_dr_travelled / simtime;
193 XTP_LOG(Log::error, log_)
194 << std::scientific << " Overall average velocity (nm/s): "
195 << avgvelocity.transpose() * tools::conv::bohr2nm << std::flush;
196}

198void KMCMultiple::RunVSSM() {
199
200 std::chrono::time_point<std::chrono::system_clock> realtime_start =
201 std::chrono::system_clock::now();
202 XTP_LOG(Log::error, log_)
203 << "\nAlgorithm: VSSM for Multiple Charges" << std::flush;
204 XTP_LOG(Log::error, log_)
205 << "number of carriers: " << numberofcarriers_ << std::flush;
206 XTP_LOG(Log::error, log_)
207 << "number of nodes: " << nodes_.size() << std::flush;
208
209 bool checkifoutput = (outputtime_ != 0);
210 double nexttrajoutput = 0;
211 unsigned long maxsteps = boost::numeric_cast<unsigned long>(runtime_);
212 unsigned long outputstep = boost::numeric_cast<unsigned long>(outputtime_);
213 bool stopontime = false;
214
215 if (runtime_ > 100) {
216 XTP_LOG(Log::error, log_)
217 << "stop condition: " << maxsteps << " steps." << std::flush;
218
219 if (checkifoutput) {
220 XTP_LOG(Log::error, log_) << "output frequency: ";
221 XTP_LOG(Log::error, log_)
222 << "every " << outputstep << " steps." << std::flush;
223 }
224 } else {
225 stopontime = true;
226 XTP_LOG(Log::error, log_)
227 << "stop condition: " << runtime_ << " seconds runtime." << std::flush;
228
229 if (checkifoutput) {
230 XTP_LOG(Log::error, log_) << "output frequency:\n "
231 "every "
232 << outputtime_ << " seconds." << std::flush;
233 }
234 }
235 XTP_LOG(Log::error, log_)
236 << "(If you specify runtimes larger than 100 kmcmultiple assumes that "
237 "you are specifying the number of steps for both runtime and "
238 "outputtime.)"
239 << std::flush;
240
241 if (!stopontime && outputtime_ != 0 && floor(outputtime_) != outputtime_) {
242 throw std::runtime_error(
243 "ERROR in kmcmultiple: runtime was specified in steps (>100) and "
244 "outputtime in seconds (not an integer). Please use the same units for "
245 "both input parameters.");
246 }
247
248 if (numberofcarriers_ > Index(nodes_.size())) {
249 throw std::runtime_error(
250 "ERROR in kmcmultiple: specified number of carriers is greater than "
251 "the "
252 "number of nodes. This conflicts with single occupation.");
253 }
254
255 std::fstream traj;
256 std::fstream tfile;
257
258 if (checkifoutput) {
259
260 XTP_LOG(Log::error, log_)
261 << "Writing trajectory to " << trajectoryfile_ << "." << std::flush;
262 traj.open(trajectoryfile_, std::fstream::out);
263
264 traj << "time[s]\tsteps\t";
265 for (Index i = 0; i < numberofcarriers_; i++) {
266 traj << "carrier" << i + 1 << " x_\t";
267 traj << "carrier" << i + 1 << " y_\t";
268 traj << "carrier" << i + 1 << " z_";
269 if (i < numberofcarriers_ - 1) {
270 traj << "'\t";
271 }
272 }
273 traj << std::endl;
274 if (!timefile_.empty()) {
275 XTP_LOG(Log::error, log_)
276 << "Writing time dependence of energy and mobility to " << timefile_
277 << "." << std::flush;
278 tfile.open(timefile_, std::fstream::out);
279 tfile << "time[s]\t "
280 "steps\tenergy_per_carrier[eV]\tmobility[nm**2/"
281 "Vs]\tdistance_fielddirection[nm]\tdistance_absolute[nm]"
282 << std::endl;
283 }
284 }
285 RandomlyCreateCharges();
286 std::vector<Eigen::Vector3d> startposition(numberofcarriers_,
287 Eigen::Vector3d::Zero());
288 for (Index i = 0; i < numberofcarriers_; i++) {
289 startposition[i] = carriers_[i].getCurrentPosition();
290 }
291
292 traj << 0 << "\t";
293 traj << 0 << "\t";
294 for (Index i = 0; i < numberofcarriers_; i++) {
295 traj << startposition[i].x() * tools::conv::bohr2nm << "\t";
296 traj << startposition[i].y() * tools::conv::bohr2nm << "\t";
297 traj << startposition[i].z() * tools::conv::bohr2nm;
298 if (i < numberofcarriers_ - 1) {
299 traj << "\t";
300 } else {
301 traj << std::endl;
302 }
303 }
304
305 std::vector<GNode*> forbiddennodes;
306 std::vector<GNode*> forbiddendests;
307
308 Eigen::Matrix3d avgdiffusiontensor = Eigen::Matrix3d::Zero();
309
310 double simtime = 0.0;
311 unsigned long step = 0;
312
313 while (((stopontime && simtime < runtime_) ||
314 (!stopontime && step < maxsteps))) {
315
316 std::chrono::duration<double> elapsed_time =
317 std::chrono::system_clock::now() - realtime_start;
318 if (elapsed_time.count() > (maxrealtime_ * 60. * 60.)) {
319 XTP_LOG(Log::error, log_)
320 << "\nReal time limit of " << maxrealtime_ << " hours ("
321 << Index(maxrealtime_ * 60 * 60 + 0.5)
322 << " seconds) has been reached. Stopping here.\n"
323 << std::flush;
324 break;
325 }
326
327 double cumulated_rate = 0;
328 for (const auto& carrier : carriers_) {
329 cumulated_rate += carrier.getCurrentEscapeRate();
330 }
331 if (cumulated_rate <= 0) { // this should not happen: no possible jumps
332 // defined for a node
333 throw std::runtime_error(
334 "ERROR in kmcmultiple: Incorrect rates. All "
335 "the escape rates for the current setting are 0.");
336 }
337
338 double dt = Promotetime(cumulated_rate);
339
340 simtime += dt;
341 step++;
342
343 for (auto& carrier : carriers_) {
344 carrier.updateOccupationtime(dt);
345 }
346
347 ResetForbiddenlist(forbiddennodes);
348 bool level1step = true;
349 while (level1step) {
350
351 // determine which electron will escape
352 GNode* newnode = nullptr;
353 Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
354
355 if (CheckForbidden(affectedcarrier->getCurrentNode(), forbiddennodes)) {
356 continue;
357 }
358 ResetForbiddenlist(forbiddendests);
359 while (true) {
360 // LEVEL 2
361
362 const GLink& event =
363 ChooseHoppingDest(affectedcarrier->getCurrentNode());
364 newnode = event.getDestination();
365
366 if (newnode == nullptr) {
367 AddtoForbiddenlist(affectedcarrier->getCurrentNode(), forbiddennodes);
368 break; // select new escape node (ends level 2 but without setting
369 // level1step to 1)
370 }
371
372 // check after the event if this was allowed
373 if (CheckForbidden(*newnode, forbiddendests)) {
374 continue;
375 }
376
377 // if the new segment is unoccupied: jump; if not: add to forbidden
378 // list and choose new hopping destination
379 if (newnode->isOccupied()) {
380 if (CheckSurrounded(affectedcarrier->getCurrentNode(),
381 forbiddendests)) {
382 AddtoForbiddenlist(affectedcarrier->getCurrentNode(),
383 forbiddennodes);
384 break; // select new escape node (ends level 2 but without
385 // setting level1step to 1)
386 }
387 AddtoForbiddenlist(*newnode, forbiddendests);
388 continue; // select new destination
389 } else {
390 affectedcarrier->jumpAccordingEvent(event);
391 level1step = false;
392 break; // this ends LEVEL 2 , so that the time is updated and the
393 // next MC step started
394 }
395 // END LEVEL 2
396 }
397 // END LEVEL 1
398 }
399
400 if (step % diffusionresolution_ == 0) {
401 for (const auto& carrier : carriers_) {
402 avgdiffusiontensor += (carrier.get_dRtravelled()) *
403 (carrier.get_dRtravelled()).transpose();
404 }
405 }
406
407 if (step != 0 && step % intermediateoutput_frequency_ == 0) {
408 PrintDiffandMu(avgdiffusiontensor, simtime, step);
409 }
410
411 if (checkifoutput) {
412 bool outputsteps = (!stopontime && step % outputstep == 0);
413 bool outputtime = (stopontime && simtime > nexttrajoutput);
414 if (outputsteps || outputtime) {
415 // write to trajectory file
416 nexttrajoutput = simtime + outputtime_;
417 WriteToTrajectory(traj, startposition, simtime, step);
418 if (!timefile_.empty()) {
419 WriteToEnergyFile(tfile, simtime, step);
420 }
421 }
422 }
423 } // KMC
424
425 if (checkifoutput) {
426 traj.close();
427 if (!timefile_.empty()) {
428 tfile.close();
429 }
430 }
431
432 WriteOccupationtoFile(simtime, occfile_);
433
434 XTP_LOG(Log::error, log_) << "\nfinished KMC simulation after " << step
435 << " steps.\n"
436 "simulated time "
437 << simtime << " seconds.\n"
438 << std::flush;
439
440 PrintChargeVelocity(simtime);
441
442 XTP_LOG(Log::error, log_) << "\nDistances travelled (nm): " << std::flush;
443 for (Index i = 0; i < numberofcarriers_; i++) {
444 XTP_LOG(Log::error, log_)
445 << std::scientific << " carrier " << i + 1 << ": "
446 << carriers_[i].get_dRtravelled().transpose() * tools::conv::bohr2nm
447 << std::flush;
448 }
449
450 PrintDiffandMu(avgdiffusiontensor, simtime, step);
451 PrintDiagDandMu(avgdiffusiontensor, simtime, step);
452
453 return;
454}

456bool KMCMultiple::Evaluate(Topology& top) {
457
458 XTP_LOG(Log::error, log_) << "\n-----------------------------------"
459 "\n KMC FOR MULTIPLE CHARGES"
460 "\n-----------------------------------\n"
461 << std::flush;
462
463 XTP_LOG(Log::info, log_) << "\nInitialising random number generator"
464 << std::flush;
465 RandomVariable_.init(seed_);
466
467 LoadGraph(top);
468 RunVSSM();
469 std::cout << log_;
470 return true;
471}
