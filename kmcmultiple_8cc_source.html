35 void KMCMultiple::Initialize(tools::Property *options){

40  _numberofcharges=options->ifExistsReturnElseThrowRuntimeError<int>(key+".numberofcharges");
41  _injection_name=options->ifExistsReturnElseThrowRuntimeError<int>(key+".injectionpattern");
42 
43 
44  _maxrealtime=options->ifExistsReturnElseReturnDefault<double>(key+".maxrealtime",1E10);
45  _trajectoryfile=options->ifExistsReturnElseReturnDefault<std::string>(key+".trajectoryfile","trajectory.csv");
46  _temperature=options->ifExistsReturnElseReturnDefault<double>(key+".temperature",300);

55  _injectionmethod = "random";
56 
57  }
58  _field = options->ifExistsReturnElseReturnDefault<tools::vec>(key+".field",tools::vec(0,0,0));
59 
60 
61  _outputtime = options->ifExistsReturnElseReturnDefault<double>(key+".outputtime",0);
62  _timefile = options->ifExistsReturnElseReturnDefault<std::string>(key+".timefile","timedependence.csv");

66 
67

70

76 void KMCMultiple::RunVSSM(ctp::Topology *top)
77 {
78 
79  int realtime_start = time(NULL);
80  cout << endl << "Algorithm: VSSM for Multiple Charges" << endl;
81  cout << "number of charges: " << _numberofcharges << endl;
82  cout << "number of nodes: " << _nodes.size() << endl;
83 
84  bool checkifoutput=(_outputtime != 0);
85  double nexttrajoutput=_runtime;
86  unsigned long maxsteps=_runtime;
87  unsigned long outputstep=_outputtime;
88  bool stopontime=false;
89 
90  if(_runtime > 100){
91  cout << "stop condition: " << maxsteps << " steps." << endl;
92  }
93  else{
94  stopontime =true;
95  cout << "stop condition: " << _runtime << " seconds runtime." << endl;
96  cout << "(If you specify runtimes larger than 100 kmcmultiple assumes that you are specifying the number of steps.)" << endl;
97  }
98 
99 
100 
101  if(_numberofcharges > _nodes.size())
102  {
103  throw runtime_error("ERROR in kmcmultiple: specified number of charges is greater than the number of nodes. This conflicts with single occupation.");
104  }
105 
106  fstream traj;
107  fstream tfile;
108 
109  tfile.open (_timefile.c_str(), fstream::out);
110  if(checkifoutput){
111  cout << "Writing trajectory to " << _trajectoryfile << "." << endl;
112  traj.open (_trajectoryfile.c_str(), fstream::out);
113 
114 
115  traj << "'time[s]'\t";
116  traj << "'steps'\t";
117  for(unsigned int i=0; i<_numberofcharges; i++){
118  traj << "'carrier" << i+1 << "_x'\t";
119  traj << "'carrier" << i+1 << "_y'\t";
120  traj << "'carrier" << i+1 << "_z";
121  if(i<_numberofcharges-1){
122  traj << "'\t";
123  }
124  }
125  traj << endl;
126  cout << "Writing time dependence of energy and mobility to " << _timefile << "." << endl;
127  tfile << "time[s]\t steps\tenergy_per_carrier[eV]\tmobility[nm**2/Vs]\tdistance_fielddirection[nm]\tdistance_absolute[nm]" << endl;
128 
129  }

135  double absolute_field = tools::abs(_field);
136 
137  // Injection
138  cout << endl << "injection method: " << _injectionmethod << endl;
139 
140 
141  RandomlyCreateCharges();
142  vector<tools::vec> startposition(_numberofcharges,tools::vec(0,0,0));
143 
144  for(unsigned i=0;i<_carriers.size();i++){
145  startposition[i] = _carriers[i]->getCurrentPosition();
146  cout << "starting position for charge " << i+1 << ": segment " << _carriers[i]->getCurrentNodeId()+1 << endl;
147 
148  }
149 
150 
151  vector<int> forbiddennodes;
152  vector<int> forbiddendests;
153 
154  tools::matrix avgdiffusiontensor;
155  avgdiffusiontensor.ZeroMatrix();
156  const unsigned long diffusionresolution=1000;
157  double simtime = 0.0;
158  unsigned long step = 0;
159 
160  while(((stopontime && simtime < _runtime) || (!stopontime && step < maxsteps))){
161 
162  if((time(NULL) - realtime_start) > _maxrealtime*60.*60.)
163  {
164  cout << endl << "Real time limit of " << _maxrealtime << " hours (" << int(_maxrealtime*60*60 +0.5) <<" seconds) has been reached. Stopping here." << endl << endl;
165  break;
166  }
167 
168  double cumulated_rate = 0;
169  for(unsigned int i=0; i<_numberofcharges; i++)
170  {
171  cumulated_rate += _carriers[i]->getCurrentEscapeRate();
172  }
173  if(cumulated_rate == 0)
174  { // this should not happen: no possible jumps defined for a node
175  throw runtime_error("ERROR in kmcmultiple: Incorrect rates in the database file. All the escape rates for the current setting are 0.");
176  }
177 
178  double dt =Promotetime(cumulated_rate);
179 
180  simtime += dt;
181  step++;
182  if(tools::globals::verbose) {cout << "simtime += " << dt << endl << endl;}
183 
184 
185  for(unsigned int i=0; i<_numberofcharges; i++)
186  {
187  _carriers[i]->updateOccupationtime(dt);
188  }
189 
190 
191  ResetForbiddenlist(forbiddennodes);
192  bool level1step = true;
193  while(level1step){
194 
195 
196 
197  // determine which electron will escape
198 
199  GNode* newnode= NULL;
200  Chargecarrier* affectedcarrier=ChooseAffectedCarrier(cumulated_rate);
201 
202 
203 
204  if(CheckForbidden(affectedcarrier->getCurrentNodeId(), forbiddennodes)) {continue;}
205 
206  // determine where it will jump to
207  ResetForbiddenlist(forbiddendests);
208  while(true){
209  // LEVEL 2
210  if(tools::globals::verbose) {cout << "There are " <<affectedcarrier->getCurrentNode()->events.size() << " possible jumps for this charge:"; }
211 
212 
213  GLink* event=ChooseHoppingDest(affectedcarrier->getCurrentNode());
214  newnode = _nodes[event->destination];
215 
216 
217  if(newnode == NULL){
218  if(tools::globals::verbose) {
219  cout << endl << "Node " << affectedcarrier->getCurrentNodeId()+1 << " is SURROUNDED by forbidden destinations and zero rates. "
220  "Adding it to the list of forbidden nodes. After that: selection of a new escape node." << endl;
221  }
222  AddtoForbiddenlist(affectedcarrier->getCurrentNodeId(), forbiddennodes);
223  break; // select new escape node (ends level 2 but without setting level1step to 1)
224  }
225  if(tools::globals::verbose) {cout << endl << "Selected jump: " << newnode->id+1 << endl; }
226 
227  // check after the event if this was allowed
228  if(CheckForbidden(newnode->id, forbiddendests)){
229  if(tools::globals::verbose) {cout << "Node " << newnode->id+1 << " is FORBIDDEN. Now selection new hopping destination." << endl; }
230  continue;
231  }
232 
233  // if the new segment is unoccupied: jump; if not: add to forbidden list and choose new hopping destination
234  if(newnode->occupied){
235  if(CheckSurrounded(affectedcarrier->getCurrentNode(), forbiddendests)){
236  if(tools::globals::verbose) {
237  cout << "Node " << affectedcarrier->getCurrentNodeId()+1 << " is SURROUNDED by forbidden destinations. "
238  "Adding it to the list of forbidden nodes. After that: selection of a new escape node." << endl;
239  }
240  AddtoForbiddenlist(affectedcarrier->getCurrentNodeId(), forbiddennodes);
241  break; // select new escape node (ends level 2 but without setting level1step to 1)
242  }
243  if(tools::globals::verbose) {cout << "Selected segment: " << newnode->id+1 << " is already OCCUPIED. Added to forbidden list." << endl << endl;}
244  AddtoForbiddenlist(newnode->id, forbiddendests);
245  if(tools::globals::verbose) {cout << "Now choosing different hopping destination." << endl; }
246  continue; // select new destination
247  }
248  else{
249  affectedcarrier->jumpfromCurrentNodetoNode(newnode);
250  affectedcarrier->dr_travelled +=event->dr;
251  level1step = false;
252  if(tools::globals::verbose) {cout << "Charge has jumped to segment: " << newnode->id+1 << "." << endl;}
253 
254  break; // this ends LEVEL 2 , so that the time is updated and the next MC step started
255  }
256 
257  if(tools::globals::verbose) {cout << "." << endl;}
258  // END LEVEL 2
259  }
260  // END LEVEL 1
261  }
262 
263 
264 
265 
266  //outputstuff
267 
268  if(step%diffusionresolution==0){
269  for(unsigned int i=0; i<_numberofcharges; i++){
270  avgdiffusiontensor += (_carriers[i]->dr_travelled)|(_carriers[i]->dr_travelled);
271  }
272  }
273 
274 
275  bool outputsteps=(!stopontime && step%outputstep==0);
276  bool outputtime=(stopontime && simtime>nexttrajoutput);
277  if(checkifoutput && (outputsteps || outputtime)){
278  // write to trajectory file
279  nexttrajoutput = simtime + _outputtime;
280  traj << simtime << "\t";
281  for(unsigned int i=0; i<_numberofcharges; i++) {
282  traj << startposition[i].getX() + _carriers[i]->dr_travelled.getX() << "\t";
283  traj << startposition[i].getY() + _carriers[i]->dr_travelled.getY() << "\t";
284  traj << startposition[i].getZ() + _carriers[i]->dr_travelled.getZ();
285  if (i<_numberofcharges-1) {
286  traj << "\t";
287  }
288  else{
289  traj << endl;
290  }
291  }
292 
293 
294  double currentenergy = 0;
295  double currentmobility = 0;
296  tools::vec dr_travelled_current = tools::vec (0,0,0);
297  double dr_travelled_field=0.0;
298  tools::vec avgvelocity_current = tools::vec(0,0,0);
299  if(absolute_field != 0){
300  for(unsigned int i=0; i<_numberofcharges; i++){
301  dr_travelled_current += _carriers[i]->dr_travelled;
302  currentenergy += _carriers[i]->getCurrentEnergy();
303  }
304  dr_travelled_current /= _numberofcharges;
305  currentenergy /= _numberofcharges;
306  avgvelocity_current = dr_travelled_current/simtime;
307  currentmobility = (avgvelocity_current*_field) /absolute_field/absolute_field;
308  dr_travelled_field=(dr_travelled_current*_field)/absolute_field;
309  }
310 
311  tfile << simtime << "\t"<< step << "\t"<< currentenergy << "\t" << currentmobility << "\t" <<
312  dr_travelled_field << "\t" << tools::abs(dr_travelled_current)<<"\t"<< endl;
313 
314  }
315 
316  }//KMC
317 
318 
319 
320  if(checkifoutput)
321  {
322  traj.close();
323  tfile.close();
324  }
325 
326 
327  vector< ctp::Segment* >& seg = top->Segments();
328  for (unsigned i = 0; i < seg.size(); i++) {
329  double occupationprobability=_nodes[i]->occupationtime / simtime;
330  seg[i]->setOcc(occupationprobability,_carriertype);
331  }
332 
333  cout << endl << "finished KMC simulation after " << step << " steps." << endl;
334  cout << "simulated time " << simtime << " seconds." << endl;
335  cout << "runtime: ";
336  cout << endl << endl;
337 
338  tools::vec avg_dr_travelled = tools::vec (0,0,0);
339  for(unsigned int i=0; i<_numberofcharges; i++){
340  cout << std::scientific << " charge " << i+1 << ": " << _carriers[i]->dr_travelled/simtime << endl;
341  avg_dr_travelled += _carriers[i]->dr_travelled;
342  }
343  avg_dr_travelled /= _numberofcharges;
344 
345  tools::vec avgvelocity = avg_dr_travelled/simtime;
346  cout << std::scientific << " Overall average velocity (nm/s): " << avgvelocity << endl;
347 
348  cout << endl << "Distances travelled (nm): " << endl;
349  for(unsigned int i=0; i<_numberofcharges; i++){
350  cout << std::scientific << " charge " << i+1 << ": " << _carriers[i]->dr_travelled << endl;
351  }
352 
353  // calculate mobilities
354 
355  if (absolute_field != 0){
356  double average_mobility = 0;
357  cout << endl << "Mobilities (nm^2/Vs): " << endl;
358  for(unsigned int i=0; i<_numberofcharges; i++){
359  tools::vec velocity = _carriers[i]->dr_travelled/simtime;
360  cout << std::scientific << " charge " << i+1 << ": mu=" << (velocity*_field)/absolute_field/absolute_field << endl;
361  average_mobility += (velocity*_field) /absolute_field/absolute_field;
362  }
363  average_mobility /= _numberofcharges;
364  cout << std::scientific << " Overall average mobility in field direction <mu>=" << average_mobility << " nm^2/Vs " << endl;
365  }
366  cout << endl;
367 
368  // calculate diffusion tensor
369  unsigned long diffusionsteps=step/diffusionresolution;
370  avgdiffusiontensor /= (diffusionsteps*2*simtime*_numberofcharges);
371  cout<<endl<<"Diffusion tensor averaged over all carriers (nm^2/s):" << endl << avgdiffusiontensor << endl;
372 
373  tools::matrix::eigensystem_t diff_tensor_eigensystem;
374  cout<<endl<<"Eigenvalues: "<<endl<<endl;
375  avgdiffusiontensor.SolveEigensystem(diff_tensor_eigensystem);
376  for(int i=0; i<=2; i++)
377  {
378  cout<<"Eigenvalue: "<<diff_tensor_eigensystem.eigenvalues[i]<<endl<<"Eigenvector: ";
379 
380  cout<<diff_tensor_eigensystem.eigenvecs[i].x()<<" ";
381  cout<<diff_tensor_eigensystem.eigenvecs[i].y()<<" ";
382  cout<<diff_tensor_eigensystem.eigenvecs[i].z()<<endl<<endl;
383  }
384 
385  // calculate average mobility from the Einstein relation
386  if (absolute_field == 0){
387  cout << "The following value is calculated using the Einstein relation and assuming an isotropic medium" << endl;
388  double avgD = 1./3. * (diff_tensor_eigensystem.eigenvalues[0] + diff_tensor_eigensystem.eigenvalues[1] + diff_tensor_eigensystem.eigenvalues[2] );
389  double average_mobility = std::abs(avgD / tools::conv::kB / _temperature);
390  cout << std::scientific << " Overall average mobility <mu>=" << average_mobility << " nm^2/Vs (= " << endl;
391  }
