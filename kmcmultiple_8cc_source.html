38 void KMCMultiple::Initialize(const tools::Property& user_options) {
39 
40  tools::Property options =
41  LoadDefaultsAndUpdateWithUserOptions("xtp", user_options);
42  ParseCommonOptions(options);
43 
44  _runtime = options.get(".runtime").as<double>();
45  _field = options.get(".field").as<Eigen::Vector3d>();
46  double mtobohr = 1E9 * tools::conv::nm2bohr;
47  _field *=
48  (tools::conv::ev2hrt / mtobohr); // Converting from V/m to Hartree/bohr
49 
50  _outputtime = options.get(".outputtime").as<double>();
51  _timefile = options.ifExistsReturnElseReturnDefault<std::string>(".timefile",
52  _timefile);
53 
54  std::string carriertype = options.get(".carriertype").as<std::string>();
55  _carriertype = QMStateType(carriertype);
56  if (!_carriertype.isKMCState()) {
57  throw std::runtime_error("KMC cannot be run for state:" +
58  _carriertype.ToLongString());
59  }
60 
61  _log.setReportLevel(Log::current_level);
62  _log.setCommonPreface("\n ...");
63 }

65 void KMCMultiple::PrintDiffandMu(const Eigen::Matrix3d& avgdiffusiontensor,
66  double simtime, unsigned long step) {
67  double absolute_field = _field.norm();
68 
69  if (absolute_field == 0) {
70  unsigned long diffusionsteps = step / _diffusionresolution;
71  Eigen::Matrix3d result =
72  avgdiffusiontensor /
73  (double(diffusionsteps) * 2.0 * simtime * double(_numberofcarriers));
74  XTP_LOG(Log::error, _log)
75  << "\nStep: " << step
76  << " Diffusion tensor averaged over all carriers (nm^2/s):\n"
77  << result * tools::conv::bohr2nm * tools::conv::bohr2nm << std::flush;
78  } else {
79  double average_mobility = 0;
80  double bohr2Hrts_to_nm2Vs =
81  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
82  XTP_LOG(Log::error, _log) << "\nMobilities (nm^2/Vs): " << std::flush;
83  for (Index i = 0; i < _numberofcarriers; i++) {
84  Eigen::Vector3d velocity = _carriers[i].get_dRtravelled() / simtime;
85  double mobility =
86  velocity.dot(_field) / (absolute_field * absolute_field);
87  XTP_LOG(Log::error, _log)
88  << std::scientific << " carrier " << i + 1
89  << ": mu=" << mobility * bohr2Hrts_to_nm2Vs << std::flush;
90  average_mobility +=
91  velocity.dot(_field) / (absolute_field * absolute_field);
92  }
93  average_mobility /= double(_numberofcarriers);
94  XTP_LOG(Log::error, _log)
95  << std::scientific
96  << " Overall average mobility in field direction <mu>="
97  << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << std::flush;
98  }
99 }

119 void KMCMultiple::WriteToEnergyFile(std::fstream& tfile, double simtime,
120  unsigned long step) const {
121  double absolute_field = _field.norm();
122  double currentenergy = 0;
123  double currentmobility = 0;
124  Eigen::Vector3d dr_travelled_current = Eigen::Vector3d::Zero();
125  double dr_travelled_field = 0.0;
126  Eigen::Vector3d avgvelocity_current = Eigen::Vector3d::Zero();
127  if (absolute_field != 0) {
128  for (const auto& carrier : _carriers) {
129  dr_travelled_current += carrier.get_dRtravelled();
130  currentenergy += carrier.getCurrentEnergy();
131  }
132  dr_travelled_current /= double(_numberofcarriers);
133  currentenergy /= double(_numberofcarriers);
134  avgvelocity_current = dr_travelled_current / simtime;
135  currentmobility =
136  avgvelocity_current.dot(_field) / (absolute_field * absolute_field);
137  dr_travelled_field = dr_travelled_current.dot(_field) / absolute_field;
138  }
139  double bohr2Hrts_to_nm2Vs =
140  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
141  tfile << simtime << "\t" << step << "\t"
142  << currentenergy * tools::conv::hrt2ev << "\t"
143  << currentmobility * bohr2Hrts_to_nm2Vs << "\t"
144  << dr_travelled_field * tools::conv::bohr2nm << "\t"
145  << dr_travelled_current.norm() * tools::conv::bohr2nm << "\t"
146  << std::endl;
147 }

149 void KMCMultiple::PrintDiagDandMu(const Eigen::Matrix3d& avgdiffusiontensor,
150  double simtime, unsigned long step) {
151  unsigned long diffusionsteps = step / _diffusionresolution;
152  Eigen::Matrix3d result =
153  avgdiffusiontensor /
154  (double(diffusionsteps) * 2.0 * simtime * double(_numberofcarriers));
155 
156  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
157  es.computeDirect(result);
158  double bohr2_nm2 = tools::conv::bohr2nm * tools::conv::bohr2nm;
159  XTP_LOG(Log::error, _log) << "\nEigenvalues:\n " << std::flush;
160  for (Index i = 0; i < 3; i++) {
161  XTP_LOG(Log::error, _log)
162  << "Eigenvalue: " << es.eigenvalues()(i) * bohr2_nm2 << std::flush
163  << "Eigenvector: " << es.eigenvectors().col(i).x() << " "
164  << es.eigenvectors().col(i).y() << " " << es.eigenvectors().col(i).z()
165  << "\n"
166  << std::flush;
167  }
168  double bohr2Hrts_to_nm2Vs =
169  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
170  // calculate average mobility from the Einstein relation
171  if (_field.norm() == 0) {
172  XTP_LOG(Log::error, _log)
173  << "The following value is calculated using the Einstein relation "
174  "and assuming an isotropic medium"
175  << std::flush;
176  double avgD = 1. / 3. * es.eigenvalues().sum();
177  double average_mobility = std::abs(avgD / _temperature);
178  XTP_LOG(Log::error, _log)
179  << std::scientific << " Overall average mobility <mu>="
180  << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << std::flush;
181  }
182 }

202 void KMCMultiple::RunVSSM() {
203 
204  std::chrono::time_point<std::chrono::system_clock> realtime_start =
205  std::chrono::system_clock::now();
206  XTP_LOG(Log::error, _log)
207  << "\nAlgorithm: VSSM for Multiple Charges" << std::flush;
208  XTP_LOG(Log::error, _log)
209  << "number of carriers: " << _numberofcarriers << std::flush;
210  XTP_LOG(Log::error, _log)
211  << "number of nodes: " << _nodes.size() << std::flush;
212 
213  bool checkifoutput = (_outputtime != 0);
214  double nexttrajoutput = 0;
215  unsigned long maxsteps = boost::numeric_cast<unsigned long>(_runtime);
216  unsigned long outputstep = boost::numeric_cast<unsigned long>(_outputtime);
217  bool stopontime = false;
218 
219  if (_runtime > 100) {
220  XTP_LOG(Log::error, _log)
221  << "stop condition: " << maxsteps << " steps." << std::flush;
222 
223  if (checkifoutput) {
224  XTP_LOG(Log::error, _log) << "output frequency: ";
225  XTP_LOG(Log::error, _log)
226  << "every " << outputstep << " steps." << std::flush;
227  }
228  } else {
229  stopontime = true;
230  XTP_LOG(Log::error, _log)
231  << "stop condition: " << _runtime << " seconds runtime." << std::flush;
232 
233  if (checkifoutput) {
234  XTP_LOG(Log::error, _log) << "output frequency:\n "
235  "every "
236  << _outputtime << " seconds." << std::flush;
237  }
238  }
239  XTP_LOG(Log::error, _log)
240  << "(If you specify runtimes larger than 100 kmcmultiple assumes that "
241  "you are specifying the number of steps for both runtime and "
242  "outputtime.)"
243  << std::flush;
244 
245  if (!stopontime && _outputtime != 0 && floor(_outputtime) != _outputtime) {
246  throw std::runtime_error(
247  "ERROR in kmcmultiple: runtime was specified in steps (>100) and "
248  "outputtime in seconds (not an integer). Please use the same units for "
249  "both input parameters.");
250  }
251 
252  if (_numberofcarriers > Index(_nodes.size())) {
253  throw std::runtime_error(
254  "ERROR in kmcmultiple: specified number of carriers is greater than "
255  "the "
256  "number of nodes. This conflicts with single occupation.");
257  }
258 
259  std::fstream traj;
260  std::fstream tfile;
261 
262  if (checkifoutput) {
263 
264  XTP_LOG(Log::error, _log)
265  << "Writing trajectory to " << _trajectoryfile << "." << std::flush;
266  traj.open(_trajectoryfile, std::fstream::out);
267 
268  traj << "time[s]\tsteps\t";
269  for (Index i = 0; i < _numberofcarriers; i++) {
270  traj << "carrier" << i + 1 << "_x\t";
271  traj << "carrier" << i + 1 << "_y\t";
272  traj << "carrier" << i + 1 << "_z";
273  if (i < _numberofcarriers - 1) {
274  traj << "'\t";
275  }
276  }
277  traj << std::endl;
278  if (!_timefile.empty()) {
279  XTP_LOG(Log::error, _log)
280  << "Writing time dependence of energy and mobility to " << _timefile
281  << "." << std::flush;
282  tfile.open(_timefile, std::fstream::out);
283  tfile << "time[s]\t "
284  "steps\tenergy_per_carrier[eV]\tmobility[nm**2/"
285  "Vs]\tdistance_fielddirection[nm]\tdistance_absolute[nm]"
286  << std::endl;
287  }
288  }
289  RandomlyCreateCharges();
290  std::vector<Eigen::Vector3d> startposition(_numberofcarriers,
291  Eigen::Vector3d::Zero());
292  for (Index i = 0; i < _numberofcarriers; i++) {
293  startposition[i] = _carriers[i].getCurrentPosition();
294  }
295 
296  traj << 0 << "\t";
297  traj << 0 << "\t";
298  for (Index i = 0; i < _numberofcarriers; i++) {
299  traj << startposition[i].x() * tools::conv::bohr2nm << "\t";
300  traj << startposition[i].y() * tools::conv::bohr2nm << "\t";
301  traj << startposition[i].z() * tools::conv::bohr2nm;
302  if (i < _numberofcarriers - 1) {
303  traj << "\t";
304  } else {
305  traj << std::endl;
306  }
307  }
308 
309  std::vector<GNode*> forbiddennodes;
310  std::vector<GNode*> forbiddendests;
311 
312  Eigen::Matrix3d avgdiffusiontensor = Eigen::Matrix3d::Zero();
313 
314  double simtime = 0.0;
315  unsigned long step = 0;
316 
317  while (((stopontime && simtime < _runtime) ||
318  (!stopontime && step < maxsteps))) {
319 
320  std::chrono::duration<double> elapsed_time =
321  std::chrono::system_clock::now() - realtime_start;
322  if (elapsed_time.count() > (_maxrealtime * 60. * 60.)) {
323  XTP_LOG(Log::error, _log)
324  << "\nReal time limit of " << _maxrealtime << " hours ("
325  << Index(_maxrealtime * 60 * 60 + 0.5)
326  << " seconds) has been reached. Stopping here.\n"
327  << std::flush;
328  break;
329  }
330 
331  double cumulated_rate = 0;
332  for (const auto& carrier : _carriers) {
333  cumulated_rate += carrier.getCurrentEscapeRate();
334  }
335  if (cumulated_rate <= 0) { // this should not happen: no possible jumps
336  // defined for a node
337  throw std::runtime_error(
338  "ERROR in kmcmultiple: Incorrect rates. All "
339  "the escape rates for the current setting are 0.");
340  }
341 
342  double dt = Promotetime(cumulated_rate);
343 
344  simtime += dt;
345  step++;
346 
347  for (auto& carrier : _carriers) {
348  carrier.updateOccupationtime(dt);
349  }
350 
351  ResetForbiddenlist(forbiddennodes);
352  bool level1step = true;
353  while (level1step) {
354 
355  // determine which electron will escape
356  GNode* newnode = nullptr;
357  Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
358 
359  if (CheckForbidden(affectedcarrier->getCurrentNode(), forbiddennodes)) {
360  continue;
361  }
362  ResetForbiddenlist(forbiddendests);
363  while (true) {
364  // LEVEL 2
365 
366  const GLink& event =
367  ChooseHoppingDest(affectedcarrier->getCurrentNode());
368  newnode = event.getDestination();
369 
370  if (newnode == nullptr) {
371  AddtoForbiddenlist(affectedcarrier->getCurrentNode(), forbiddennodes);
372  break; // select new escape node (ends level 2 but without setting
373  // level1step to 1)
374  }
375 
376  // check after the event if this was allowed
377  if (CheckForbidden(*newnode, forbiddendests)) {
378  continue;
379  }
380 
381  // if the new segment is unoccupied: jump; if not: add to forbidden
382  // list and choose new hopping destination
383  if (newnode->isOccupied()) {
384  if (CheckSurrounded(affectedcarrier->getCurrentNode(),
385  forbiddendests)) {
386  AddtoForbiddenlist(affectedcarrier->getCurrentNode(),
387  forbiddennodes);
388  break; // select new escape node (ends level 2 but without
389  // setting level1step to 1)
390  }
391  AddtoForbiddenlist(*newnode, forbiddendests);
392  continue; // select new destination
393  } else {
394  affectedcarrier->jumpAccordingEvent(event);
395  level1step = false;
396  break; // this ends LEVEL 2 , so that the time is updated and the
397  // next MC step started
398  }
399  // END LEVEL 2
400  }
401  // END LEVEL 1
402  }
403 
404  if (step % _diffusionresolution == 0) {
405  for (const auto& carrier : _carriers) {
406  avgdiffusiontensor += (carrier.get_dRtravelled()) *
407  (carrier.get_dRtravelled()).transpose();
408  }
409  }
410 
411  if (step != 0 && step % _intermediateoutput_frequency == 0) {
412  PrintDiffandMu(avgdiffusiontensor, simtime, step);
413  }
414 
415  if (checkifoutput) {
416  bool outputsteps = (!stopontime && step % outputstep == 0);
417  bool outputtime = (stopontime && simtime > nexttrajoutput);
418  if (outputsteps || outputtime) {
419  // write to trajectory file
420  nexttrajoutput = simtime + _outputtime;
421  WriteToTrajectory(traj, startposition, simtime, step);
422  if (!_timefile.empty()) {
423  WriteToEnergyFile(tfile, simtime, step);
424  }
425  }
426  }
427  } // KMC
428 
429  if (checkifoutput) {
430  traj.close();
431  if (!_timefile.empty()) {
432  tfile.close();
433  }
434  }
435 
436  WriteOccupationtoFile(simtime, _occfile);
437 
438  XTP_LOG(Log::error, _log) << "\nfinished KMC simulation after " << step
439  << " steps.\n"
440  "simulated time "
441  << simtime << " seconds.\n"
442  << std::flush;
443 
444  PrintChargeVelocity(simtime);
445 
446  XTP_LOG(Log::error, _log) << "\nDistances travelled (nm): " << std::flush;
447  for (Index i = 0; i < _numberofcarriers; i++) {
448  XTP_LOG(Log::error, _log)
449  << std::scientific << " carrier " << i + 1 << ": "
450  << _carriers[i].get_dRtravelled().transpose() * tools::conv::bohr2nm
451  << std::flush;
452  }
453 
454  PrintDiffandMu(avgdiffusiontensor, simtime, step);
455  PrintDiagDandMu(avgdiffusiontensor, simtime, step);
456 
457  return;
458 }
