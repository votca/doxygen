31 void KMCMultiple::Initialize(tools::Property& options) {
32  std::string key = "options." + Identify();
33  ParseCommonOptions(options);
34  _runtime =
35  options.ifExistsReturnElseThrowRuntimeError<double>(key + ".runtime");
36  _field = options.ifExistsReturnElseReturnDefault<Eigen::Vector3d>(
37  key + ".field", Eigen::Vector3d::Zero());
38  double mtobohr = 1E9 * tools::conv::nm2bohr;
39  _field *=
40  (tools::conv::ev2hrt / mtobohr); // Converting from V/m to Hartree/bohr
41 
42  _outputtime =
43  options.ifExistsReturnElseReturnDefault<double>(key + ".outputtime", 0);
44  _timefile = options.ifExistsReturnElseReturnDefault<std::string>(
45  key + ".timefile", _timefile);
46 
47  std::string carriertype =
48  options.ifExistsReturnElseReturnDefault<std::string>(key + ".carriertype",
49  "e");
50  _carriertype = QMStateType(carriertype);
51  if (!_carriertype.isKMCState()) {
52  throw runtime_error("KMC cannot be run for state:" +
53  _carriertype.ToLongString());
54  }
55 
56  _log.setReportLevel(Log::current_level);
57  _log.setCommonPreface("\n ...");
58  return;
59 }

61 void KMCMultiple::PrintDiffandMu(const Eigen::Matrix3d& avgdiffusiontensor,
62  double simtime, unsigned long step) {
63  double absolute_field = _field.norm();
64 
65  if (absolute_field == 0) {
66  unsigned long diffusionsteps = step / _diffusionresolution;
67  Eigen::Matrix3d result =
68  avgdiffusiontensor /
69  (double(diffusionsteps) * 2.0 * simtime * double(_numberofcarriers));
70  XTP_LOG(Log::error, _log)
71  << "\nStep: " << step
72  << " Diffusion tensor averaged over all carriers (nm^2/s):\n"
73  << result * tools::conv::bohr2nm * tools::conv::bohr2nm << flush;
74  } else {
75  double average_mobility = 0;
76  double bohr2Hrts_to_nm2Vs =
77  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
78  XTP_LOG(Log::error, _log) << "\nMobilities (nm^2/Vs): " << flush;
79  for (Index i = 0; i < _numberofcarriers; i++) {
80  Eigen::Vector3d velocity = _carriers[i].get_dRtravelled() / simtime;
81  double mobility =
82  velocity.dot(_field) / (absolute_field * absolute_field);
83  XTP_LOG(Log::error, _log)
84  << std::scientific << " carrier " << i + 1
85  << ": mu=" << mobility * bohr2Hrts_to_nm2Vs << flush;
86  average_mobility +=
87  velocity.dot(_field) / (absolute_field * absolute_field);
88  }
89  average_mobility /= double(_numberofcarriers);
90  XTP_LOG(Log::error, _log)
91  << std::scientific
92  << " Overall average mobility in field direction <mu>="
93  << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << flush;
94  }
95 }

115 void KMCMultiple::WriteToEnergyFile(std::fstream& tfile, double simtime,
116  unsigned long step) const {
117  double absolute_field = _field.norm();
118  double currentenergy = 0;
119  double currentmobility = 0;
120  Eigen::Vector3d dr_travelled_current = Eigen::Vector3d::Zero();
121  double dr_travelled_field = 0.0;
122  Eigen::Vector3d avgvelocity_current = Eigen::Vector3d::Zero();
123  if (absolute_field != 0) {
124  for (const auto& carrier : _carriers) {
125  dr_travelled_current += carrier.get_dRtravelled();
126  currentenergy += carrier.getCurrentEnergy();
127  }
128  dr_travelled_current /= double(_numberofcarriers);
129  currentenergy /= double(_numberofcarriers);
130  avgvelocity_current = dr_travelled_current / simtime;
131  currentmobility =
132  avgvelocity_current.dot(_field) / (absolute_field * absolute_field);
133  dr_travelled_field = dr_travelled_current.dot(_field) / absolute_field;
134  }
135  double bohr2Hrts_to_nm2Vs =
136  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
137  tfile << simtime << "\t" << step << "\t"
138  << currentenergy * tools::conv::hrt2ev << "\t"
139  << currentmobility * bohr2Hrts_to_nm2Vs << "\t"
140  << dr_travelled_field * tools::conv::bohr2nm << "\t"
141  << dr_travelled_current.norm() * tools::conv::bohr2nm << "\t" << endl;
142 }

144 void KMCMultiple::PrintDiagDandMu(const Eigen::Matrix3d& avgdiffusiontensor,
145  double simtime, unsigned long step) {
146  unsigned long diffusionsteps = step / _diffusionresolution;
147  Eigen::Matrix3d result =
148  avgdiffusiontensor /
149  (double(diffusionsteps) * 2.0 * simtime * double(_numberofcarriers));
150 
151  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
152  es.computeDirect(result);
153  double bohr2_nm2 = tools::conv::bohr2nm * tools::conv::bohr2nm;
154  XTP_LOG(Log::error, _log) << "\nEigenvalues:\n " << flush;
155  for (Index i = 0; i < 3; i++) {
156  XTP_LOG(Log::error, _log)
157  << "Eigenvalue: " << es.eigenvalues()(i) * bohr2_nm2 << flush
158  << "Eigenvector: " << es.eigenvectors().col(i).x() << " "
159  << es.eigenvectors().col(i).y() << " " << es.eigenvectors().col(i).z()
160  << "\n"
161  << flush;
162  }
163  double bohr2Hrts_to_nm2Vs =
164  tools::conv::bohr2nm * tools::conv::bohr2nm / tools::conv::hrt2ev;
165  // calculate average mobility from the Einstein relation
166  if (_field.norm() == 0) {
167  XTP_LOG(Log::error, _log)
168  << "The following value is calculated using the Einstein relation "
169  "and assuming an isotropic medium"
170  << flush;
171  double avgD = 1. / 3. * es.eigenvalues().sum();
172  double average_mobility = std::abs(avgD / _temperature);
173  XTP_LOG(Log::error, _log)
174  << std::scientific << " Overall average mobility <mu>="
175  << average_mobility * bohr2Hrts_to_nm2Vs << " nm^2/Vs " << flush;
176  }
177 }

197 void KMCMultiple::RunVSSM() {
198 
199  Index realtime_start = time(nullptr);
200  XTP_LOG(Log::error, _log)
201  << "\nAlgorithm: VSSM for Multiple Charges" << flush;
202  XTP_LOG(Log::error, _log)
203  << "number of carriers: " << _numberofcarriers << flush;
204  XTP_LOG(Log::error, _log) << "number of nodes: " << _nodes.size() << flush;
205 
206  bool checkifoutput = (_outputtime != 0);
207  double nexttrajoutput = 0;
208  unsigned long maxsteps = boost::numeric_cast<unsigned long>(_runtime);
209  unsigned long outputstep = boost::numeric_cast<unsigned long>(_outputtime);
210  bool stopontime = false;
211 
212  if (_runtime > 100) {
213  XTP_LOG(Log::error, _log)
214  << "stop condition: " << maxsteps << " steps." << flush;
215 
216  if (checkifoutput) {
217  XTP_LOG(Log::error, _log) << "output frequency: ";
218  XTP_LOG(Log::error, _log) << "every " << outputstep << " steps." << flush;
219  }
220  } else {
221  stopontime = true;
222  XTP_LOG(Log::error, _log)
223  << "stop condition: " << _runtime << " seconds runtime." << flush;
224 
225  if (checkifoutput) {
226  XTP_LOG(Log::error, _log) << "output frequency:\n "
227  "every "
228  << _outputtime << " seconds." << flush;
229  }
230  }
231  XTP_LOG(Log::error, _log)
232  << "(If you specify runtimes larger than 100 kmcmultiple assumes that "
233  "you are specifying the number of steps for both runtime and "
234  "outputtime.)"
235  << flush;
236 
237  if (!stopontime && _outputtime != 0 && floor(_outputtime) != _outputtime) {
238  throw runtime_error(
239  "ERROR in kmcmultiple: runtime was specified in steps (>100) and "
240  "outputtime in seconds (not an integer). Please use the same units for "
241  "both input parameters.");
242  }
243 
244  if (_numberofcarriers > Index(_nodes.size())) {
245  throw runtime_error(
246  "ERROR in kmcmultiple: specified number of carriers is greater than "
247  "the "
248  "number of nodes. This conflicts with single occupation.");
249  }
250 
251  std::fstream traj;
252  std::fstream tfile;
253 
254  if (checkifoutput) {
255 
256  XTP_LOG(Log::error, _log)
257  << "Writing trajectory to " << _trajectoryfile << "." << flush;
258  traj.open(_trajectoryfile, std::fstream::out);
259 
260  traj << "time[s]\tsteps\t";
261  for (Index i = 0; i < _numberofcarriers; i++) {
262  traj << "carrier" << i + 1 << "_x\t";
263  traj << "carrier" << i + 1 << "_y\t";
264  traj << "carrier" << i + 1 << "_z";
265  if (i < _numberofcarriers - 1) {
266  traj << "'\t";
267  }
268  }
269  traj << endl;
270  if (!_timefile.empty()) {
271  XTP_LOG(Log::error, _log)
272  << "Writing time dependence of energy and mobility to " << _timefile
273  << "." << flush;
274  tfile.open(_timefile, std::fstream::out);
275  tfile << "time[s]\t "
276  "steps\tenergy_per_carrier[eV]\tmobility[nm**2/"
277  "Vs]\tdistance_fielddirection[nm]\tdistance_absolute[nm]"
278  << endl;
279  }
280  }
281  RandomlyCreateCharges();
282  vector<Eigen::Vector3d> startposition(_numberofcarriers,
283  Eigen::Vector3d::Zero());
284  for (Index i = 0; i < _numberofcarriers; i++) {
285  startposition[i] = _carriers[i].getCurrentPosition();
286  }
287 
288  traj << 0 << "\t";
289  traj << 0 << "\t";
290  for (Index i = 0; i < _numberofcarriers; i++) {
291  traj << startposition[i].x() * tools::conv::bohr2nm << "\t";
292  traj << startposition[i].y() * tools::conv::bohr2nm << "\t";
293  traj << startposition[i].z() * tools::conv::bohr2nm;
294  if (i < _numberofcarriers - 1) {
295  traj << "\t";
296  } else {
297  traj << endl;
298  }
299  }
300 
301  vector<GNode*> forbiddennodes;
302  vector<GNode*> forbiddendests;
303 
304  Eigen::Matrix3d avgdiffusiontensor = Eigen::Matrix3d::Zero();
305 
306  double simtime = 0.0;
307  unsigned long step = 0;
308 
309  while (((stopontime && simtime < _runtime) ||
310  (!stopontime && step < maxsteps))) {
311 
312  if ((time(nullptr) - realtime_start) > Index(_maxrealtime * 60. * 60.)) {
313  XTP_LOG(Log::error, _log)
314  << "\nReal time limit of " << _maxrealtime << " hours ("
315  << Index(_maxrealtime * 60 * 60 + 0.5)
316  << " seconds) has been reached. Stopping here.\n"
317  << flush;
318  break;
319  }
320 
321  double cumulated_rate = 0;
322  for (const auto& carrier : _carriers) {
323  cumulated_rate += carrier.getCurrentEscapeRate();
324  }
325  if (cumulated_rate <= 0) { // this should not happen: no possible jumps
326  // defined for a node
327  throw runtime_error(
328  "ERROR in kmcmultiple: Incorrect rates. All "
329  "the escape rates for the current setting are 0.");
330  }
331 
332  double dt = Promotetime(cumulated_rate);
333 
334  simtime += dt;
335  step++;
336 
337  for (auto& carrier : _carriers) {
338  carrier.updateOccupationtime(dt);
339  }
340 
341  ResetForbiddenlist(forbiddennodes);
342  bool level1step = true;
343  while (level1step) {
344 
345  // determine which electron will escape
346  GNode* newnode = nullptr;
347  Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
348 
349  if (CheckForbidden(affectedcarrier->getCurrentNode(), forbiddennodes)) {
350  continue;
351  }
352  ResetForbiddenlist(forbiddendests);
353  while (true) {
354  // LEVEL 2
355 
356  const GLink& event =
357  ChooseHoppingDest(affectedcarrier->getCurrentNode());
358  newnode = event.getDestination();
359 
360  if (newnode == nullptr) {
361  AddtoForbiddenlist(affectedcarrier->getCurrentNode(), forbiddennodes);
362  break; // select new escape node (ends level 2 but without setting
363  // level1step to 1)
364  }
365 
366  // check after the event if this was allowed
367  if (CheckForbidden(*newnode, forbiddendests)) {
368  continue;
369  }
370 
371  // if the new segment is unoccupied: jump; if not: add to forbidden
372  // list and choose new hopping destination
373  if (newnode->isOccupied()) {
374  if (CheckSurrounded(affectedcarrier->getCurrentNode(),
375  forbiddendests)) {
376  AddtoForbiddenlist(affectedcarrier->getCurrentNode(),
377  forbiddennodes);
378  break; // select new escape node (ends level 2 but without
379  // setting level1step to 1)
380  }
381  AddtoForbiddenlist(*newnode, forbiddendests);
382  continue; // select new destination
383  } else {
384  affectedcarrier->jumpAccordingEvent(event);
385  level1step = false;
386  break; // this ends LEVEL 2 , so that the time is updated and the
387  // next MC step started
388  }
389  // END LEVEL 2
390  }
391  // END LEVEL 1
392  }
393 
394  if (step % _diffusionresolution == 0) {
395  for (const auto& carrier : _carriers) {
396  avgdiffusiontensor += (carrier.get_dRtravelled()) *
397  (carrier.get_dRtravelled()).transpose();
398  }
399  }
400 
401  if (step != 0 && step % _intermediateoutput_frequency == 0) {
402  PrintDiffandMu(avgdiffusiontensor, simtime, step);
403  }
404 
405  if (checkifoutput) {
406  bool outputsteps = (!stopontime && step % outputstep == 0);
407  bool outputtime = (stopontime && simtime > nexttrajoutput);
408  if (outputsteps || outputtime) {
409  // write to trajectory file
410  nexttrajoutput = simtime + _outputtime;
411  WriteToTrajectory(traj, startposition, simtime, step);
412  if (!_timefile.empty()) {
413  WriteToEnergyFile(tfile, simtime, step);
414  }
415  }
416  }
417  } // KMC
418 
419  if (checkifoutput) {
420  traj.close();
421  if (!_timefile.empty()) {
422  tfile.close();
423  }
424  }
425 
426  WriteOccupationtoFile(simtime, _occfile);
427 
428  XTP_LOG(Log::error, _log) << "\nfinished KMC simulation after " << step
429  << " steps.\n"
430  "simulated time "
431  << simtime << " seconds.\n"
432  << flush;
433 
434  PrintChargeVelocity(simtime);
435 
436  XTP_LOG(Log::error, _log) << "\nDistances travelled (nm): " << flush;
437  for (Index i = 0; i < _numberofcarriers; i++) {
438  XTP_LOG(Log::error, _log)
439  << std::scientific << " carrier " << i + 1 << ": "
440  << _carriers[i].get_dRtravelled().transpose() * tools::conv::bohr2nm
441  << flush;
442  }
443 
444  PrintDiffandMu(avgdiffusiontensor, simtime, step);
445  PrintDiagDandMu(avgdiffusiontensor, simtime, step);
446 
447  return;
448 }
