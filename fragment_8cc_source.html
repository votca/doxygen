81 void Fragment::Rotate(matrix spin, vec refPos) {
82  std::vector <Atom*> ::iterator ait;
83  for (ait = _atoms.begin(); ait < _atoms.end(); ait++) {
84  vec dir = (*ait)->getQMPos() - refPos;
85  dir = spin * dir;
86  vec pos = refPos + dir;

111 void Fragment::calcPos(string tag) {
112  vec pos = vec(0,0,0);
113  double totWeight = 0.0;
114 
115  for (unsigned int i = 0; i< _atoms.size(); i++) {
116  if (tag == "MD") {
117  pos += _atoms[i]->getPos() * _atoms[i]->getWeight();
118  }
119  else if (tag == "QM") {
120  if (_atoms[i]->HasQMPart())
121  pos += _atoms[i]->getQMPos() * _atoms[i]->getWeight();
122  else
123  assert(_atoms[i]->getWeight() < 1e-6);
124  }
125  totWeight += _atoms[i]->getWeight();
126  }
127 
128  if (tag == "MD") {
129  _CoMD = pos / totWeight;
130  }
131  else if (tag == "QM") {
132  _CoQM = pos / totWeight;
133  }

140 void Fragment::Rigidify(bool Auto) {
141 
142  // +++++++++++++++++++++++++++++++++++++++++ //
143  // Establish reference atoms for local frame //
144  // +++++++++++++++++++++++++++++++++++++++++ //
145 
146  std::vector<Atom*> trihedron;
147 
148  if (Auto) {
149 
150  // Automated search for suitable atoms
151  bool enough = false;
152  std::vector< Atom* > ::iterator ait = this->Atoms().begin();
153 
154  while (!enough) {
155 
156  if (ait == this->Atoms().end()) { break; }
157  Atom *atm = *ait;
158  ait++;
159 
160  // When specifying mapping weights for rigid fragments,
161  // mobile atoms (whose position can vary significantly with
162  // respect to the rigid plane, for instance) should have a weight 0.
163  // We do not want to pick those to establish the local frame.
164  if (atm->getWeight() == 0.0) { continue; }
165  else {
166  trihedron.push_back(atm);
167  if (trihedron.size() == 3) { enough = true; }
168  }
169  }
170  }
171  else {
172 
173  // Take atoms specified in mapping file <= DEFAULT
174  std::vector<Atom*> ::iterator ait;
175  for (ait = _atoms.begin(); ait < _atoms.end(); ait++) {
176  if ( ! (*ait)->HasQMPart() ) { continue; }
177 
178  if ( (*ait)->getQMId() == _trihedron[0] ||
179  (*ait)->getQMId() == _trihedron[1] ||
180  (*ait)->getQMId() == _trihedron[2] ) {
181  Atom *atm = *ait;
182  trihedron.push_back(atm);
183  }
184  }
185  }
186 
187  _symmetry = trihedron.size();
188  if (!(trihedron.size() == _trihedron.size())) {
189  cout << endl << "ERROR Local frame ill-defined" << flush;
190  }
191 
192  // +++++++++++++++++++++++ //
193  // Construct trihedra axes //
194  // +++++++++++++++++++++++ //
195 
196  vec xMD, yMD, zMD;
197  vec xQM, yQM, zQM;
198 
199  if (_symmetry == 3) {
200 
201  vec r1MD = trihedron[0]->getPos();
202  vec r2MD = trihedron[1]->getPos();
203  vec r3MD = trihedron[2]->getPos();
204  vec r1QM = trihedron[0]->getQMPos();
205  vec r2QM = trihedron[1]->getQMPos();
206  vec r3QM = trihedron[2]->getQMPos();
207 
208  xMD = r2MD - r1MD;
209  yMD = r3MD - r1MD;
210  xQM = r2QM - r1QM;
211  yQM = r3QM - r1QM;
212 
213  zMD = xMD ^ yMD;
214  zQM = xQM ^ yQM;
215 
216  yMD = zMD ^ xMD;
217  yQM = zQM ^ xQM;
218 
219  xMD = xMD.normalize();
220  yMD = yMD.normalize();
221  zMD = zMD.normalize();
222  xQM = xQM.normalize();
223  yQM = yQM.normalize();
224  zQM = zQM.normalize();
225 
226  }
227 
228  else if (_symmetry == 2) {
229 
230  vec r1MD = trihedron[0]->getPos();
231  vec r2MD = trihedron[1]->getPos();
232  vec r1QM = trihedron[0]->getQMPos();
233  vec r2QM = trihedron[1]->getQMPos();
234 
235  xMD = r2MD - r1MD;
236  xQM = r2QM - r1QM;
237 
238  // Normalising not necessary, but recommendable, when doing...
239  xMD = xMD.normalize();
240  xQM = xQM.normalize();
241 
242  vec yMDtmp = vec(0,0,0);
243  vec yQMtmp = vec(0,0,0);
244 
245  // ... this: Check whether one of the components is equal or close to
246  // zero. If so, this easily gives a second leg for the trihedron.
247  if ( xMD.getX()*xMD.getX() < 1e-6 ) { yMDtmp = vec(1,0,0); }
248  else if ( xMD.getY()*xMD.getY() < 1e-6 ) { yMDtmp = vec(0,1,0); }
249  else if ( xMD.getZ()*xMD.getZ() < 1e-6 ) { yMDtmp = vec(0,0,1); }
250  if ( xQM.getX()*xQM.getX() < 1e-6 ) { yQMtmp = vec(1,0,0); }
251  else if ( xQM.getY()*xQM.getY() < 1e-6 ) { yQMtmp = vec(0,1,0); }
252  else if ( xQM.getZ()*xQM.getZ() < 1e-6 ) { yQMtmp = vec(0,0,1); }
253 
254  if ( abs(yMDtmp) < 0.5 ) {
255  // All components of xMD are unequal to zero => division is safe.
256  // Choose vector from plane with xMD * inPlaneVec = 0:
257  double tmp_x = 1.;
258  double tmp_y = 1.;
259  double tmp_z = 1/xMD.getZ() * (-xMD.getX()*tmp_x - xMD.getY()*tmp_y);
260  yMDtmp = vec(tmp_x, tmp_y, tmp_z);
261  yMDtmp.normalize();
262  }
263  if ( abs(yQMtmp) < 0.5 ) {
264  double tmp_x = 1.;
265  double tmp_y = 1.;
266  double tmp_z = 1/xQM.getZ() * (-xQM.getX()*tmp_x - xQM.getY()*tmp_y);
267  yQMtmp = vec(tmp_x, tmp_y, tmp_z);
268  yQMtmp.normalize();
269  }
270 
271  // Now proceed as for symmetry 3
272  zMD = xMD ^ yMDtmp;
273  yMD = zMD ^ xMD;
274  zQM = xQM ^ yQMtmp;
275  yQM = zQM ^ xQM;
276 
277  xMD.normalize();
278  yMD.normalize();
279  zMD.normalize();
280  xQM.normalize();
281  yQM.normalize();
282  zQM.normalize();
283 
284  }
285 
286  else if (_symmetry == 1) {
287 
288  xMD = vec(1,0,0);
289  yMD = vec(0,1,0);
290  zMD = vec(0,0,1);
291  xQM = vec(1,0,0);
292  yQM = vec(0,1,0);
293  zQM = vec(0,0,1);
294 
295  }
296 
297  else {
298  cout << endl
299  << "NOTE: Invalid definition of local frame in fragment "
300  << this->getName();
301  cout << ". Assuming point particle for mapping. "
302  << endl;
303 
304  xMD = vec(1,0,0);
305  yMD = vec(0,1,0);
306  zMD = vec(0,0,1);
307  xQM = vec(1,0,0);
308  yQM = vec(0,1,0);
309  zQM = vec(0,0,1);
310 
311  }
312 
313  // +++++++++++++++++ //
314  // Rotation matrices //
315  // +++++++++++++++++ //
316 
317  matrix rotMD = matrix(xMD, yMD, zMD);
318  matrix rotQM = matrix(xQM, yQM, zQM);
319 
320  matrix rotateQM2MD = rotMD * rotQM.Transpose();
321  _rotateQM2MD = rotateQM2MD;
322 
323 
324  // ++++++++++++++++++ //
325  // Transform fragment //
326  // ++++++++++++++++++ //
327 
328  this->calcPos("QM");
329  this->RotTransQM2MD();
330  _translateQM2MD = _CoMD - _CoQM;
331 
332 }
