51 void RangeParser::ParseBlock(string str)
52 {
53  Tokenizer tokenizer(str, ":");
54  vector<string> toks;
55 
56  block_t block;
57  block._stride = 1;
58 
59  tokenizer.ToVector(toks);
60  if(toks.size()>3 || toks.size() < 1) { throw runtime_error("invalid range"); }
61 
62  block._begin = block._end = ToNumber(toks[0]);
63 
64  if(toks.size() == 2)
65  block._end = ToNumber(toks[1]);
66 
67  if(toks.size() == 3) {
68  block._stride = ToNumber(toks[1]);
69  block._end = ToNumber(toks[2]);
70  }
71 
72  if(block._begin*block._stride > block._end*block._stride) {
73  throw runtime_error(string("invalid range " + str + ": begin, end and stride do not form a closed interval"));
74  }
75 
76  _blocks.push_back(block);
77 }
