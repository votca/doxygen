43 void RangeParser::ParseBlock(std::string str) {
44  Tokenizer tokenizer(str, ":");
45  std::vector<std::string> toks;
46 
47  block_t block;
48  block._stride = 1;
49 
50  tokenizer.ToVector(toks);
51  if (toks.size() > 3 || toks.size() < 1) {
52  throw std::runtime_error("invalid range");
53  }
54 
55  block._begin = block._end = std::stoi(toks[0]);
56 
57  if (toks.size() == 2) {
58  block._end = std::stoi(toks[1]);
59  }
60 
61  if (toks.size() == 3) {
62  block._stride = std::stoi(toks[1]);
63  block._end = std::stoi(toks[2]);
64  }
65 
66  if (block._begin * block._stride > block._end * block._stride) {
67  throw std::runtime_error(
68  std::string("invalid range " + str +
69  ": begin, end and stride do not form a closed interval"));
70  }
71 
72  _blocks.push_back(block);
73 }
