00051 void RangeParser::ParseBlock(string str)
00052 {
00053     Tokenizer tokenizer(str, ":");
00054     vector<string> toks;
00055     
00056     block_t block;
00057     block._stride = 1;
00058     
00059     tokenizer.ToVector(toks);
00060     if(toks.size()>3 || toks.size() < 1) { throw runtime_error("invalid range"); }
00061     
00062     block._begin = block._end = ToNumber(toks[0]);
00063     
00064     if(toks.size() == 2)
00065         block._end = ToNumber(toks[1]);
00066 
00067     if(toks.size() == 3) {
00068         block._stride = ToNumber(toks[1]);
00069         block._end = ToNumber(toks[2]);
00070     }
00071         
00072     if(block._begin*block._stride > block._end*block._stride) {
00073         throw runtime_error(string("invalid range " + str + ": begin, end and stride do not form a closed interval"));
00074     }
00075     
00076     _blocks.push_back(block);
00077 }
