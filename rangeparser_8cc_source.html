46 void RangeParser::ParseBlock(std::string str) {
47  Tokenizer tokenizer(str, ":");
48  std::vector<std::string> toks;
49 
50  block_t block;
51  block._stride = 1;
52 
53  tokenizer.ToVector(toks);
54  if (toks.size() > 3 || toks.size() < 1) {
55  throw std::runtime_error("invalid range");
56  }
57 
58  block._begin = block._end = std::stoi(toks[0]);
59 
60  if (toks.size() == 2) {
61  block._end = std::stoi(toks[1]);
62  }
63 
64  if (toks.size() == 3) {
65  block._stride = std::stoi(toks[1]);
66  block._end = std::stoi(toks[2]);
67  }
68 
69  if (block._begin * block._stride > block._end * block._stride) {
70  throw std::runtime_error(
71  std::string("invalid range " + str +
72  ": begin, end and stride do not form a closed interval"));
73  }
74 
75  _blocks.push_back(block);
76 }
