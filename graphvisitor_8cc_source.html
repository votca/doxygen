65 void GraphVisitor::exec(Graph& g, Edge ed){
66  auto unexp_vert = getUnexploredVertex_(ed);
67  // If no vertices are return than just ignore it means the same
68  // vertex was explored from a different direction
69  if(!unexp_vert.size()) return;
70  // If two values are returned this is a problem

72  throw runtime_error("More than one unexplored vertex in an edge,"
73  " did you set the starting node");
74  }
75  pair<int,GraphNode> pr(unexp_vert.at(0),g.getNode(unexp_vert.at(0)));
