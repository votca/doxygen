39 template <class Vector1, class Vector2>
40 std::array<typename Vector1::value_type, 3> cross(const Vector1& x,
41  const Vector2& y) {
42  std::array<typename Vector1::value_type, 3> result;
43  assert(size(x) == size(y) && "Dimensions of vectors do not match!");
44  assert(size(x) == 3 && "Only three dimensions allowed");
45  result[0] = x[1] * y[2] - x[2] * y[1];
46  result[1] = x[2] * y[0] - x[0] * y[2];
47  result[2] = x[0] * y[1] - x[1] * y[0];
48  return result;
49 }

112 template <class Vector>
113 typename std::array<std::array<typename Vector::value_type, 3>, 3> dualbase_3d(
114  const Vector& a, const Vector& b, const Vector& c) {
115  assert(size(a) == size(b) && "Dimensions of vector a and b do not match");
116  assert(size(b) == size(c) && "Dimensions of vector b and c do not match");
117  assert(size(a) == 3 && "Dimension must be 3");
118 
119  std::array<std::array<typename Vector::value_type, 3>, 3> result;
120 
121  typename Vector::value_type V = dot(a, cross(b, c));
122  typename Vector::value_type V_inv = 1.0 / V;
123 
124  result[0] = scale_3d(V_inv, cross(b, c));
125  result[1] = scale_3d(V_inv, cross(c, a));
126  result[2] = scale_3d(V_inv, cross(a, b));
127 
128  return result;
129 }
