33 QMPair::QMPair(int id, Segment *seg1, Segment *seg2)
34  : std::pair<Segment*, Segment*>(seg1, seg2), _id(id),
35  _hasGhost(0),
36  _rate12_e(0), _rate21_e(0),
37  _rate12_h(0), _rate21_h(0),
38  _has_e(false), _has_h(false),
39  _lambdaO_e(0), _lambdaO_h(0),
40  _Jeff2_e(0), _Jeff2_h(0),
41  _rate12_s(0), _rate21_s(0),
42  _rate12_t(0), _rate21_t(0),
43  _has_s(false), _has_t(false),
44  _lambdaO_s(0), _lambdaO_t(0),
45  _Jeff2_s(0), _Jeff2_t(0),_pair_type( Hopping ) {
46 
47  _top = seg1->getTopology();
48 
49  vec r1 = seg1->getPos();
50  vec r2 = seg2->getPos();
51 
52  _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
53 
54  // Check whether pair formed across periodic boundary
55  if ( abs(r2 - r1 - _R) > 1e-8 ) {
56 
57  _ghost = new Segment(seg2);
58  _ghost->TranslateBy(r1 - r2 + _R);
59  _hasGhost = true;
60 
61  }
62  else {
63  _ghost = NULL;
64  }
65 }

147 void QMPair::setJs(const std::vector<double> Js, int state) {
148  std::vector <double> ::const_iterator it;
149  double Jeff2 = 0.0;
150  if (state == -1) {
151  for (it = Js.begin(); it < Js.end(); ++it) {
152  Jeff2 += (*it)*(*it);
153  }
154  Jeff2 /= double(Js.size());
155  _Jeff2_e = Jeff2;
156  }
157  else if (state == +1) {
158  for (it = Js.begin(); it < Js.end(); ++it) {
159  Jeff2 += (*it)*(*it);
160  }
161  Jeff2 /= double(Js.size());
162  _Jeff2_h = Jeff2;
163  }
164  else {
165  throw std::runtime_error(" ERROR CODE whx__01x1u__");
166  }
167 }
