25 void ECPBasisSet::Load(const std::string& name) {
26  tools::Property basis_property;
27  _name = name;
28  // if name contains .xml, assume a ecp .xml file is located in the working
29  // directory
30  std::size_t found_xml = name.find(".xml");
31  std::string xmlFile;
32  if (found_xml != std::string::npos) {
33  xmlFile = name;
34  } else {
35  // get the path to the shared folders with xml files
36  char* votca_share = getenv("VOTCASHARE");
37  if (votca_share == nullptr) {
38  throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
39  }
40  xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/xtp/ecps/") +
41  name + std::string(".xml");
42  }
43  basis_property.LoadFromXML(xmlFile);
44 
45  std::vector<tools::Property*> elementProps =
46  basis_property.Select("pseudopotential.element");
47 
48  for (tools::Property* elementProp : elementProps) {
49  std::string elementName = elementProp->getAttribute<std::string>("name");
50  int lmax = elementProp->getAttribute<int>("lmax");
51  int ncore = elementProp->getAttribute<int>("ncore");
52 
53  ECPElement& element = addElement(elementName, lmax, ncore);
54 
55  std::vector<tools::Property*> shellProps = elementProp->Select("shell");
56  for (tools::Property* shellProp : shellProps) {
57  std::string shellType = shellProp->getAttribute<std::string>("type");
58  if (shellType.size() > 1) {
59  throw std::runtime_error(
60  "In ecps no combined shells e.g. SP are allowed");
61  }
62  ECPShell& shell = element.addShell(shellType);
63  std::vector<tools::Property*> constProps = shellProp->Select("constant");
64  for (tools::Property* constProp : constProps) {
65  int power = constProp->getAttribute<int>("power");
66  double decay = constProp->getAttribute<double>("decay");
67  double contraction = constProp->getAttribute<double>("contraction");
68  shell.addGaussian(power, decay, contraction);
69  }
70  }
71  }
72  return;
73 }

83 const ECPElement& ECPBasisSet::getElement(std::string element_type) const {
84  std::map<std::string, std::shared_ptr<ECPElement> >::const_iterator itm =
85  _elements.find(element_type);
86  if (itm == _elements.end()) {
87  throw std::runtime_error("Basis set " + _name +
88  " does not have element of type " + element_type);
89  }
90  const ECPElement& element = *((*itm).second);
91  return element;
92 }
