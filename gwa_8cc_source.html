46  void GWBSE::FullQPHamiltonian(){
47 
48  // constructing full QP Hamiltonian, storage in vxc
49  _vxc = -_vxc + _sigma_x + _sigma_c;
50  // diagonal elements are given by _qp_energies
51  for (unsigned _m = 0; _m < _vxc.size1(); _m++ ){
52  _vxc( _m,_m ) = _qp_energies( _m + _qpmin );
53  }
54 
55  // sigma matrices can be freed
56  _sigma_x.resize(0);
57  _sigma_c.resize(0);
58 
59 
60 
61  if ( _do_qp_diag ){
62  _qp_diag_energies.resize(_vxc.size1());
63  _qp_diag_coefficients.resize(_vxc.size1(), _vxc.size1());
64  linalg_eigenvalues(_vxc, _qp_diag_energies, _qp_diag_coefficients);
65  }
66  return;
67  }

70  void GWBSE::sigma_diag(const TCMatrix& _Mmn){
71 
72  unsigned _levelsum = _Mmn[0].size2(); // total number of bands
73  unsigned _gwsize = _Mmn[0].size1(); // size of the GW basis
74  const double pi = boost::math::constants::pi<double>();
75 
76 
77  #pragma omp parallel for
78  for (unsigned _gw_level = 0; _gw_level < _qptotal; _gw_level++) {
79  const ub::matrix<real_gwbse>& Mmn = _Mmn[ _gw_level + _qpmin ];
80  double sigma_x=0;
81  for ( unsigned _i_gw = 0 ; _i_gw < _gwsize ; _i_gw++ ){
82  // loop over all occupied bands used in screening
83  for ( unsigned _i_occ = 0 ; _i_occ <= _homo ; _i_occ++ ){
84  sigma_x-= Mmn( _i_gw , _i_occ ) * Mmn( _i_gw , _i_occ );
85  } // occupied bands
86  } // gwbasis functions
87  _sigma_x(_gw_level,_gw_level)=( 1.0 - _ScaHFX ) * sigma_x;
88  }
89 
90  if(_g_sc_max_iterations==0) {_g_sc_max_iterations=1;}
91  ub::vector<double>& dftenergies=_orbitals->MOEnergies();
92  // initial _qp_energies are dft energies
93  ub::vector<double>_qp_old=_qp_energies;
94 
95  bool energies_converged=false;
96 
97 
98  // only diagonal elements except for in final iteration
99  for (unsigned _g_iter = 0; _g_iter < _g_sc_max_iterations; _g_iter++) {
100  // loop over all GW levels
101 
102  #pragma omp parallel for
103  for (unsigned _gw_level = 0; _gw_level < _qptotal; _gw_level++) {
104  const ub::matrix<real_gwbse>& Mmn = _Mmn[ _gw_level + _qpmin ];
105  const double qpmin = _qp_old(_gw_level + _qpmin);
106 
107  double sigma_c=0.0;
108 
109  // loop over all functions in GW basis
110  for (unsigned _i_gw = 0; _i_gw < _gwsize; _i_gw++) {
111  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc PPM_construct_parameters
112  if (_ppm_weight(_i_gw) < 1.e-9) { continue;}
113  const double ppm_freq = _ppm_freq(_i_gw);
114  const double fac = _ppm_weight(_i_gw) * ppm_freq;
115  // loop over all bands
116  for (unsigned _i = 0; _i < _levelsum; _i++) {
117 
118  double occ = 1.0;
119  if (_i > _homo) occ = -1.0; // sign for empty levels
120 
121  // energy denominator
122  const double _denom = qpmin - _qp_old(_i) + occ*ppm_freq;
123 
124  double _stab = 1.0;
125  if (std::abs(_denom) < 0.25) {
126 
127  _stab = 0.5 * (1.0 - std::cos(4.0 * pi * std::abs(_denom)));
128 
129  }
130 
131  const double _factor =0.5* fac * _stab / _denom; //Hartree
132 
133  // sigma_c diagonal elements
134  sigma_c += _factor * Mmn(_i_gw, _i) * Mmn(_i_gw, _i);
135 
136  }// bands
137 
138  }// GW functions
139  _sigma_c(_gw_level, _gw_level)=sigma_c;
140  // update _qp_energies
141 
142  _qp_energies(_gw_level + _qpmin) = dftenergies(_gw_level + _qpmin) + sigma_c + _sigma_x(_gw_level, _gw_level) - _vxc(_gw_level, _gw_level);
143 
144  }// all bands
145  ub::vector<double> diff= _qp_old - _qp_energies;
146  energies_converged = true;
147  double diff_max=0;
148  unsigned state_max=0;
149  for (unsigned l = 0; l < diff.size(); l++) {
150  if(std::abs(diff(l))>std::abs(diff_max)){
151  diff_max=diff(l);
152  state_max=l;
153  }
154  if (std::abs(diff(l))>_g_sc_limit) {
155  energies_converged = false;
156  }
157  }
158  if(tools::globals::verbose){
159  double _DFTgap =dftenergies(_homo + 1) - dftenergies(_homo);
160  double _QPgap = _qp_energies( _homo +1 ) - _qp_energies( _homo );
161  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " QP_Iteration: " << _g_iter+1 <<" shift="<<_QPgap - _DFTgap <<" E_diff max="<<diff_max<<" StateNo:"<<state_max << flush;
162  }
163  double alpha=0.0;
164  _qp_energies=(1-alpha)*_qp_energies+alpha*_qp_old;
165 
166  if (energies_converged) {
167  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Converged after " << _g_iter+1 << " G iterations." << flush;
168  break;
169  }else if(_g_iter==_g_sc_max_iterations-1){
170  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " G-self-consistency cycle not converged after " << _g_sc_max_iterations << " iterations." << flush;
171  break;
172 
173  } else {
174  _qp_old = _qp_energies;
175  }
176 
177  } // iterations
178 
179  return;
180  }

184  void GWBSE::sigma_offdiag(const TCMatrix& _Mmn) {

186  unsigned _gwsize = _Mmn[0].size1(); // size of the GW basis
187  const double pi = boost::math::constants::pi<double>();
188 
189  #pragma omp parallel for
190  for (unsigned _gw_level1 = 0; _gw_level1 < _qptotal; _gw_level1++) {
191  const ub::matrix<real_gwbse>& Mmn1 = _Mmn[ _gw_level1 + _qpmin ];
192  for (unsigned _gw_level2 = 0; _gw_level2 < _gw_level1; _gw_level2++) {
193  const ub::matrix<real_gwbse>& Mmn2 = _Mmn[ _gw_level2 + _qpmin ];
194  double sigma_x=0;
195  for ( unsigned _i_gw = 0 ; _i_gw < _gwsize ; _i_gw++ ){
196  // loop over all occupied bands used in screening
197  for ( unsigned _i_occ = 0 ; _i_occ <= _homo ; _i_occ++ ){
198  sigma_x -= Mmn1( _i_gw , _i_occ ) * Mmn2( _i_gw , _i_occ );
199  } // occupied bands
200  } // gwbasis functions
201  _sigma_x(_gw_level1, _gw_level2)=( 1.0 - _ScaHFX ) * sigma_x;
202  }
203  }
204 
205  #pragma omp parallel for
206  for (unsigned _gw_level1 = 0; _gw_level1 < _qptotal; _gw_level1++) {
207  const double qpmin1 = _qp_energies(_gw_level1 + _qpmin);
208  const ub::matrix<real_gwbse>& Mmn1 = _Mmn[ _gw_level1 + _qpmin ];
209  for (unsigned _gw_level2 = 0; _gw_level2 < _gw_level1; _gw_level2++) {
210  const double qpmin2 = _qp_energies(_gw_level1 + _qpmin);
211  const ub::matrix<real_gwbse>& Mmn2 = _Mmn[ _gw_level2 + _qpmin ];
212  double sigma_c = 0;
213  for (unsigned _i_gw = 0; _i_gw < _gwsize; _i_gw++) {
214  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc PPM_construct_parameters
215  if (_ppm_weight(_i_gw) < 1.e-9) {
216  continue;
217  }
218  const double ppm_freq = _ppm_freq(_i_gw);
219  const double fac = _ppm_weight(_i_gw) * ppm_freq;
220  // loop over all screening levels
221  for (unsigned _i = 0; _i < _levelsum; _i++) {
222 
223  double occ = 1.0;
224  if (_i > _homo) occ = -1.0; // sign for empty levels
225 
226  // energy denominator
227  const double _denom1 = qpmin1 - _qp_energies(_i) + occ * ppm_freq;
228  const double _denom2 = qpmin2 - _qp_energies(_i) + occ * ppm_freq;
229 
230  double _stab1 = 1.0;
231  if (std::abs(_denom1) < 0.25) {
232  _stab1 = 0.5 * (1.0 - std::cos(4.0 * pi * std::abs(_denom1)));
233  }
234  const double factor1 = 0.5 * fac * _stab1 / _denom1; //Hartree
235  double _stab2 = 1.0;
236  if (std::abs(_denom2) < 0.25) {
237  _stab2 = 0.5 * (1.0 - std::cos(4.0 * pi * std::abs(_denom2)));
238  }
239  const double factor2 = 0.5 * fac * _stab2 / _denom2; //Hartree
240  sigma_c +=Mmn1(_i_gw, _i) * Mmn2(_i_gw, _i)*0.5*(factor1+factor2);
241  }// screening levels
242  }// GW functions
243 
244  _sigma_c(_gw_level1, _gw_level2) = sigma_c;
245 
246 
247  }// GW row
248  }//GW col
249 
250  return;
