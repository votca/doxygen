90

92  while(getline(in, line)) {
93  line_number++;
94  string conversion_error = t.getErrorDetails() + ", line " + boost::lexical_cast<string>(line_number);
95 
96  // remove comments and xmgrace stuff
97  line = line.substr(0, line.find("#"));
98  line = line.substr(0, line.find("@"));
99 
100  // tokenize string and put it to vector
101  Tokenizer tok(line, " \t");
102  vector<string> tokens;
103  tok.ToVector(tokens);
104 
105  // skip empty lines
106  if(tokens.size()==0) continue;
107 
108  // if first line is only 1 token, it's the size
109  if(tokens.size() == 1) {
110  N = lexical_cast<int>(tokens[0], conversion_error);
111  bHasN = true;
112  }
113  // it's the first data line with 2 or 3 entries
114  else if(tokens.size() == 2) {
115  t.push_back(lexical_cast<double>(tokens[0], conversion_error), lexical_cast<double>(tokens[1], conversion_error), 'i');
116  }
117  else if(tokens.size() > 2) {
118  char flag='i';
119  string sflag = tokens.back();
120  if(sflag == "i" || sflag == "o" || sflag == "u")
121  flag = sflag.c_str()[0];
122  t.push_back(lexical_cast<double>(tokens[0], conversion_error), lexical_cast<double>(tokens[1], conversion_error), flag);
123  }
124  else throw runtime_error("error, wrong table format");
125  }
126 
127  // read the rest
128  while(getline(in, line)) {
129  line_number++;
130  string conversion_error = t.getErrorDetails() + ", line " + boost::lexical_cast<string>(line_number);
131 
132  // remove comments and xmgrace stuff
133  line = line.substr(0, line.find("#"));
134  line = line.substr(0, line.find("@"));
135 
136  // tokenize string and put it to vector
137  Tokenizer tok(line, " \t");
138  vector<string> tokens;
139  tok.ToVector(tokens);
140 
141  // skip empty lines
142  if(tokens.size()==0) continue;
143 
144  // it's a data line
145  if(tokens.size() == 2) {
146  t.push_back(lexical_cast<double>(tokens[0], conversion_error), lexical_cast<double>(tokens[1], conversion_error), 'i');
147  }
148  else if(tokens.size() > 2) {
149  char flag='i';

151  flag = tokens[2].c_str()[0];
152  t.push_back(lexical_cast<double>(tokens[0], conversion_error), lexical_cast<double>(tokens[1], conversion_error), flag);
153  }
154  // otherwise error
155  else throw runtime_error("error, wrong table format");
156 
157  // was size given and did we read N values?
158  if(bHasN)

160  }
161
