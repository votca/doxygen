85 istream &operator>>(istream &in, Table &t) {
86  size_t N = 0;
87  bool bHasN = false;
88  string line;
89  int line_number = 0;
90  t.clear();
91 
92  // read till the first data line
93  while (getline(in, line)) {
94  line_number++;
95  string conversion_error = t.getErrorDetails() + ", line " +
96  boost::lexical_cast<string>(line_number);
97 
98  // remove comments and xmgrace stuff
99  line = line.substr(0, line.find("#"));
100  line = line.substr(0, line.find("@"));
101 
102  // tokenize string and put it to vector
103  Tokenizer tok(line, " \t");
104  vector<string> tokens;
105  tok.ToVector(tokens);
106 
107  // skip empty lines
108  if (tokens.size() == 0) continue;
109 
110  // if first line is only 1 token, it's the size
111  if (tokens.size() == 1) {
112  N = lexical_cast<int>(tokens[0], conversion_error);
113  bHasN = true;
114  } else if (tokens.size() == 2) {
115  // it's the first data line with 2 or 3 entries
116  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), 'i');
117  } else if (tokens.size() > 2) {
118  char flag = 'i';
119  string sflag = tokens.back();
120  if (sflag == "i" || sflag == "o" || sflag == "u") flag = sflag.c_str()[0];
121  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), flag);
122  } else {
123  throw runtime_error("error, wrong table format");
124  }
125  }
126 
127  // read the rest
128  while (getline(in, line)) {
129  line_number++;
130  string conversion_error = t.getErrorDetails() + ", line " +
131  boost::lexical_cast<string>(line_number);
132 
133  // remove comments and xmgrace stuff
134  line = line.substr(0, line.find("#"));
135  line = line.substr(0, line.find("@"));
136 
137  // tokenize string and put it to vector
138  Tokenizer tok(line, " \t");
139  vector<string> tokens;
140  tok.ToVector(tokens);
141 
142  // skip empty lines
143  if (tokens.size() == 0) continue;
144 
145  // it's a data line
146  if (tokens.size() == 2) {
147  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), 'i');
148  } else if (tokens.size() > 2) {
149  char flag = 'i';
150  if (tokens[2] == "i" || tokens[2] == "o" || tokens[2] == "u")
151  flag = tokens[2].c_str()[0];
152  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), flag);
153  } else {
154  // otherwise error
155  throw runtime_error("error, wrong table format");
156  }
157  // was size given and did we read N values?
158  if (bHasN) {
159  if (--N == 0) break;
160  }
161  }
162 
163  return in;
164 }
