94 istream &operator>>(istream &in, Table &t) {
95  size_t N = 0;
96  bool bHasN = false;
97  string line;
98  Index line_number = 0;
99  t.clear();
100 
101  // read till the first data line
102  while (getline(in, line)) {
103  line_number++;
104  string conversion_error = t.getErrorDetails() + ", line " +
105  boost::lexical_cast<string>(line_number);
106 
107  // remove comments and xmgrace stuff
108  line = line.substr(0, line.find("#"));
109  line = line.substr(0, line.find("@"));
110 
111  // tokenize string and put it to vector
112  Tokenizer tok(line, " \t");
113 
114  vector<string> tokens = tok.ToVector();
115 
116  // skip empty lines
117  if (tokens.size() == 0) {
118  continue;
119  }
120 
121  // if first line is only 1 token, it's the size
122  if (tokens.size() == 1) {
123  N = lexical_cast<Index>(tokens[0], conversion_error);
124  bHasN = true;
125  } else if (tokens.size() == 2) {
126  // it's the first data line with 2 or 3 entries
127  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), 'i');
128  } else if (tokens.size() > 2) {
129  char flag = 'i';
130  string sflag = tokens.back();
131  if (sflag == "i" || sflag == "o" || sflag == "u") {
132  flag = sflag[0];
133  }
134  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), flag);
135  } else {
136  throw runtime_error("error, wrong table format");
137  }
138  }
139 
140  // read the rest
141  while (getline(in, line)) {
142  line_number++;
143  string conversion_error = t.getErrorDetails() + ", line " +
144  boost::lexical_cast<string>(line_number);
145 
146  // remove comments and xmgrace stuff
147  line = line.substr(0, line.find("#"));
148  line = line.substr(0, line.find("@"));
149 
150  // tokenize string and put it to vector
151  Tokenizer tok(line, " \t");
152  vector<string> tokens;
153  tok.ToVector(tokens);
154 
155  // skip empty lines
156  if (tokens.size() == 0) {
157  continue;
158  }
159 
160  // it's a data line
161  if (tokens.size() == 2) {
162  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), 'i');
163  } else if (tokens.size() > 2) {
164  char flag = 'i';
165  if (tokens[2] == "i" || tokens[2] == "o" || tokens[2] == "u") {
166  flag = tokens[2][0];
167  }
168  t.push_back(std::stod(tokens[0]), std::stod(tokens[1]), flag);
169  } else {
170  // otherwise error
171  throw runtime_error("error, wrong table format");
172  }
173  // was size given and did we read N values?
174  if (bHasN) {
175  if (--N == 0) {
176  break;
177  }
178  }
179  }
180 
181  return in;
182 }

185  Index vec_size = (Index)((max - min) / spacing + 1.00000001);
186  resize(vec_size);

197 void Table::Smooth(Index Nsmooth) {
198 
199  Index n_2 = size() - 2;
200  if (n_2 < 0) {
201  throw std::runtime_error(
202  "Smoothing only works for arrays of size 3 and larger");
203  }
204  for (Index i = 0; i < Nsmooth; i++) {
205  _y.segment(1, n_2) =
206  (0.25 * (_y.head(n_2) + 2 * _y.segment(1, n_2) + _y.tail(n_2))).eval();
