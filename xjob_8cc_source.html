49 void XJob::CalcCenterPos() {
50 
51  vec center = vec(0,0,0);
52  double weight = 0;
53 
54  // Ref. point for PB correction
55  vec refPt = vec(0,0,0);
56  if (_qmSegs.size() == 0) { ; }
57  else { refPt = _qmSegs[0]->getPos(); }
58 
59  // Calc. center
60  for (unsigned int i = 0; i < _qmSegs.size(); ++i) {
61 
62  Segment *seg = _qmSegs[i];
63  vec pbc_com = refPt + _top->PbShortestConnect(refPt, seg->getPos());
64 
65  center += seg->Atoms().size() * pbc_com;
66  weight += seg->Atoms().size();
67 
68  this->_isSegInCenter[seg->getId()] = true;
69  }
70 
71  _center = center / weight;
72 }
