49 void Bsumtree::initialize(unsigned long nrelements) { // Must be called before use
50  // treesize is the smallest power of two above nrelements minus 1
51  treesize = (unsigned long) pow(2,ceil(log((double) nrelements)/log((double) 2)))-1; // number of nodes
52 
53  dirty_array.resize(treesize);
54  if (element_array.size()<nrelements) { element_array.resize(nrelements); }
55  partsum_array.resize(treesize);
56 
57  // Initialize arrays
58  for (unsigned long i=0;i<treesize;i++) {
59  dirty_array[i] = false;
60  partsum_array[i] = 0.0;
61  }
62  for (unsigned long i=0;i<nrelements;i++) {
63  element_array[i] = 0.0;
64  }
65 }

67 void Bsumtree::setrate(unsigned long i, double value) { // 0 <= i < nrelements
68  element_array[i] = value;
69  long j = i+treesize;
70  j = div(j-1,(long) 2).quot; // Parent node
71  while (!dirty_array[j]) { // Mark this node and all parents dirty if not already
72  dirty_array[j] = true;
73  if (j != 0) { // Make sure we stop at the root node
74  j = div(j-1,(long) 2).quot; // Parent node
75  }
76  }
77 }

83 double Bsumtree::compute_sum() { // Returns total sum of all elements
84  // recursively recompute all dirty nodes
85  long i = 0; // Start at root node
86  while (dirty(i)) {
87  if (dirty(2*i + 1)) { // Is left subtree dirty ?
88  i = 2*i + 1;
89  }
90  else {
91  if (dirty(2*i + 2)) { // Is right subtree dirty?
92  i = 2*i + 2;
93  }
94  else { // Both subtrees are clean, update this node
95  partsum_array[i] = partsum(2*i+1) + partsum(2*i+2);
96  dirty_array[i] = false;
97  if (i != 0) { // Make sure we stop at the root node
98  i = div(i-1,(long) 2).quot; // Parent node
99  }
100  }
101  }
102  }
103  return partsum_array[0];
104 }

106 // Search returns index to element i: sum(0..i) <= searchkey < sum(0..i+1),
107 // where the sum is taken over the succesive elements.
108 long Bsumtree::search(double searchkey) { // Returns index to element
109  long maxindex = treesize + nrelements;
110  long i = 0; // value must be located in subtree denoted by index i
111  while (2*i+2<maxindex) {
112  if (searchkey <= partsum(2*i+1)) { // value is located in left subtree
113  i = 2*i+1;
114  }
115  else { // value is located in right subtree
116  searchkey -= partsum(2*i+1); // values are relative
117  i = 2*i+2;
118  }
119  }
120  i -= treesize;
121  return i;
122 }
