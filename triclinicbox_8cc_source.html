23 /*
24  This way of determining the shortest distance is only working for a set of
25  triclinic boxes, in particular
26  a_y = a_z = b_z = 0
27  a_x > 0, b_y > 0, c_z > 0
28  b_x < 0.5 a_x, c_x < 0.5 a_x, c_y < 0.5 b_y
29  GROMACS checks if these conditions are satisfied.
30  If a simple search algorithm is used to determine if a particle
31  is a within cutoff r_c, make sure that r_c < 0.5 min(a_x, b_y, c_z)
32  */
33 Eigen::Vector3d TriclinicBox::BCShortestConnection(
34  const Eigen::Vector3d &r_i, const Eigen::Vector3d &r_j) const {
35  Eigen::Vector3d r_tp = r_j - r_i;
36  Eigen::Vector3d r_dp = r_tp - _box.col(2) * std::round(r_tp.z() / _box(2, 2));
37  Eigen::Vector3d r_sp = r_dp - _box.col(1) * std::round(r_dp.y() / _box(1, 1));
38  return r_sp - _box.col(0) * std::round(r_sp.x() / _box(0, 0));
39 }
