00022 /*
00023  This way of determining the shortest distance is only working for a set of 
00024  triclinic boxes, in particular
00025  a_y = a_z = b_z = 0
00026  a_x > 0, b_y > 0, c_z > 0
00027  b_x < 0.5 a_x, c_x < 0.5 a_x, c_y < 0.5 b_y
00028  GROMACS checks if these conditions are satisfied.
00029  If a simple search algorithm is used to determine if a particle
00030  is a within cutoff r_c, make sure that r_c < 0.5 min(a_x, b_y, c_z)
00031  */
00032 vec TriclinicBox::BCShortestConnection(const vec &r_i, const vec &r_j) const
00033 {
00034     vec r_tp, r_dp, r_sp, r_ij;
00035     vec a = _box.getCol(0); vec b = _box.getCol(1); vec c = _box.getCol(2);
00036     r_tp = r_j - r_i;
00037     r_dp = r_tp - c*round(r_tp.getZ()/c.getZ());
00038     r_sp = r_dp - b*round(r_dp.getY()/b.getY());
00039     r_ij = r_sp - a*round(r_sp.getX()/a.getX());
00040     return r_ij;
00041 }
