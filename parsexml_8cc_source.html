46 void ParseXML::Open(const std::string &filename) {
47  XML_Parser parser = XML_ParserCreate(nullptr);
48  if (!parser) {
49  throw std::runtime_error("Couldn't allocate memory for xml parser");
50  }
51 
52  XML_UseParserAsHandlerArg(parser);
53  XML_SetElementHandler(parser, start_hndl, end_hndl);
54  // XML_SetCharacterDataHandler(parser, char_hndl);
55 
56  std::ifstream fl;
57  fl.open(filename);
58  if (!fl.is_open()) {
59  throw std::ios_base::failure("Error on open xml file: " + filename);
60  }
61 
62  XML_SetUserData(parser, (void *)this);
63  while (!fl.eof()) {
64  std::string line;
65  getline(fl, line);
66  line = line + "\n";
67  if (line.length() > (size_t)std::numeric_limits<int>::max()) {
68  throw std::runtime_error("ParseXML::Open: line is too long");
69  }
70  if (!XML_Parse(parser, line.c_str(), (int)line.length(), fl.eof())) {
71  throw std::ios_base::failure(
72  filename + ": Parse error in " + filename + " at line " +
73  boost::lexical_cast<std::string>(XML_GetCurrentLineNumber(parser)) +
74  "\n" + XML_ErrorString(XML_GetErrorCode(parser)));
75  }
76  }
77  fl.close();
78 }
