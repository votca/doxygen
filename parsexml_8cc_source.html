00059         throw std::ios_base::failure("Error on open xml file: " + filename);
00060

00062     while (!fl.eof()) {
00063         string line;
00064         getline(fl, line);
00065         line = line + "\n";
00066         if (!XML_Parse(parser, line.c_str(), line.length(), fl.eof()))
00067             throw std::ios_base::failure(filename + ": Parse error in " + filename + " at line " +
00068                 boost::lexical_cast<string > (XML_GetCurrentLineNumber(parser)) + "\n" +
00069                 XML_ErrorString(XML_GetErrorCode(parser)));
00070     }
00071     fl.close();
00072 }
