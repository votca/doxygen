46 void ParseXML::Open(const string &filename)
47 {
48  XML_Parser parser = XML_ParserCreate(NULL);
49  if (!parser)
50  throw std::runtime_error("Couldn't allocate memory for xml parser");
51 
52  XML_UseParserAsHandlerArg(parser);
53  XML_SetElementHandler(parser, start_hndl, end_hndl);
54 // XML_SetCharacterDataHandler(parser, char_hndl);
55 
56  ifstream fl;
57  fl.open(filename.c_str());
58  if (!fl.is_open())
59  throw std::ios_base::failure("Error on open xml file: " + filename);
60 
61  XML_SetUserData(parser, (void*) this);
62  while (!fl.eof()) {
63  string line;
64  getline(fl, line);
65  line = line + "\n";
66  if (!XML_Parse(parser, line.c_str(), line.length(), fl.eof()))
67  throw std::ios_base::failure(filename + ": Parse error in " + filename + " at line " +
68  boost::lexical_cast<string > (XML_GetCurrentLineNumber(parser)) + "\n" +
69  XML_ErrorString(XML_GetErrorCode(parser)));
70  }
71  fl.close();
72 }
