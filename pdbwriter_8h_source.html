102 template <class T>
103 inline void PDBWriter::WriteContainer(T &container) {
104  boost::format atomfrmt(
105  "ATOM %1$5d %2$-4s %3$-3s %4$1s%5$4d %6$8.3f%7$8.3f%8$8.3f\n");
106 
107  for (auto &atom : getIterable(container)) {
108  Eigen::Vector3d r = getPos(atom);
109  std::string resname = getResname(container, atom);
110  std::string atomname = getName(atom);
111  if (resname.size() > 3) {
112  resname = resname.substr(0, 3);
113  }
114  if (atomname.size() > 4) {
115  atomname = atomname.substr(0, 4);
116  }
117 
118  _out << atomfrmt % (getId(atom) % 100000) // atom serial number
119  % atomname % resname % " " // chain identifier 1 char
120  % getResId(container, atom) // residue sequence number
121  % r.x() % r.y() % r.z();
122  // we skip the charge
123  writeSymmetry(atom);
124  }
125  _out << std::flush;
126 }
