43  * Calculates S^{-1/2}
44  */
45 void Overlap::SQRTOverlap(ub::symmetric_matrix<double> &S,
46  ub::matrix<double> &S2 ) {
47 
48  double (*_inv_sqrt)(double);
49  _inv_sqrt = &inv_sqrt;
50 
51  ub::vector<double> _eigenvalues;
52  ub::matrix<double> _eigenvectors;
53 
54  int _size = S.size1();
55 
56  _eigenvalues.resize( _size );
57  _eigenvectors.resize( _size, _size );
58 
59  votca::tools::linalg_eigenvalues_symmetric(S, _eigenvalues, _eigenvectors);
60 
61  // compute inverse sqrt of all eigenvalues
62  std::transform(_eigenvalues.begin(), _eigenvalues.end(), _eigenvalues.begin(), _inv_sqrt );
63 
64  // form a diagonal matrix S^{-1/2}
65  ub::diagonal_matrix<double> _diagS2( _eigenvalues.size(), _eigenvalues.data() );
66 
67  // multiply from the left on the U
68  ub::matrix<double> _temp = ub::prod( _eigenvectors, _diagS2 );
69 
70  // multiply from the right on the transpose U
71  S2 = ub::prod( _temp, ub::trans( _eigenvectors ) );
72 
73  }

75 double Overlap::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
76  Orbitals* _orbitalsB, ub::matrix<double>* _JAB, double _energy_difference ) {
77 
78 
79  int _levelsA = _orbitalsA->getNumberOfLevels();
80 
81  if ( _energy_difference != 0 ) {
82  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
83  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
84 
85  double _JAB_sq = 0; double _JAB_one_level;
86 
87  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
88  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
89  //cout << *iA << ':' << *iB << endl;
90  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
91  _JAB_sq += _JAB_one_level*_JAB_one_level ;
92  }
93  }
94 
95  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
96 
97  } else {
98 
99  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
100 
101  }
102  // the matrix should be symmetric, could also return this element
103  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );
104 }

117 bool Overlap::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
118  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
119 
120  LOG(logDEBUG,*_pLog) << "Calculating electronic couplings" << flush;
121 
122  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
123  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
124  const std::vector<QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
125 
126  for (unsigned i=0;i<atomsAB.size();i++){
127  QMAtom* dimer=atomsAB[i];
128  QMAtom* monomer=NULL;
129  if (i<atomsA.size()){
130  monomer=atomsA[i];
131  }
132  else if (i<atomsB.size()+atomsA.size() ){
133  monomer=atomsB[i-atomsA.size()];
134  }
135  else{
136  throw runtime_error((format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
137  }
138 
139  if(monomer->type != dimer->type){
140  throw runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
141  }
142  if(std::abs(monomer->x-dimer->x)>0.001 || std::abs(monomer->y-dimer->y)>0.001 || std::abs(monomer->z-dimer->z)>0.001){
143  LOG(logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << flush;
144  break;
145  }
146 
147  }
148 
149 
150 
151 
152  // constructing the direct product orbA x orbB
153  int _basisA = _orbitalsA->getBasisSetSize();
154  int _basisB = _orbitalsB->getBasisSetSize();
155 
156  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
157  LOG(logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
158  return false;
159  }
160 
161  int _levelsA = _orbitalsA->getNumberOfLevels();
162  int _levelsB = _orbitalsB->getNumberOfLevels();
163 
164  //boost::timer t; // start timing
165  //double _st = t.elapsed();
166 
167  LOG(logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
168  << " B[" << _levelsB << ":" << _basisB << "]" << flush;
169 
170  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
171  LOG(logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
172  return false;
173  }
174 
175  // | Orbitals_A 0 | | Overlap_A |
176  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
177  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
178  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
179  ub::matrix<double> _psi_AxB ( _levelsA + _levelsB, _basisA + _basisB );
180 
181 
182  LOG(logDEBUG,*_pLog) << "Constructing direct product AxB ["
183  << _psi_AxB.size1() << "x"
184  << _psi_AxB.size2() << "]" << flush;
185 
186  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
187  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
188  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
189  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
190 
191  // psi_AxB * S_AB * psi_AB
192  LOG(logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << flush;
193  ub::matrix<double> _orbitalsAB_Transposed = ub::trans( *_orbitalsAB->getOrbitals() );
194  if ( (*_orbitalsAB->getOverlap()).size1() == 0 ) {
195  LOG(logERROR,*_pLog) << "Overlap matrix is not stored";
196  return false;
197  }
198 
199  ub::matrix<double> _psi_AB = ub::prod( *_orbitalsAB->getOverlap(), _orbitalsAB_Transposed );
200  ub::matrix<double> _psi_AxB_dimer_basis = ub::prod( _psi_AxB, _psi_AB );
201  _psi_AB.clear();
202 
203  //check to see if projection quality is sufficient
204  for (unsigned i=0;i<_psi_AxB_dimer_basis.size1();i++){
205  double mag=0.0;
206  for (unsigned j=0;j<_psi_AxB_dimer_basis.size2();j++){
207  mag+=_psi_AxB_dimer_basis(i,j)*_psi_AxB_dimer_basis(i,j);
208 
209  }
210  if (mag<0.95){
211  throw runtime_error("\nERROR: Projection of monomer orbitals on dimer is insufficient, maybe the orbital order is screwed up, otherwise increase dimer basis.\n");
212  }
213  }
214 
215 
216  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
217  LOG(logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << flush;
218  ub::diagonal_matrix<double> _fock_AB( _orbitalsAB->getNumberOfLevels(), (*_orbitalsAB->getEnergies()).data() );
219  ub::matrix<double> _temp = ub::prod( _fock_AB, ub::trans( _psi_AxB_dimer_basis ) ) ;
220  ub::matrix<double> JAB_dimer = ub::prod( _psi_AxB_dimer_basis, _temp);
221 
222  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
223  LOG(logDEBUG,*_pLog) << "Constructing Overlap matrix" << flush;
224  ub::symmetric_matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
225  ub::matrix<double> _S_AxB_2(_S_AxB.size1(), _S_AxB.size1() );
226  ub::trans( _S_AxB );
227 
228  // Square root of the overlap matrix
229  LOG(logDEBUG,*_pLog) << "Calculating square root of the overlap matrix" << flush;
230  SQRTOverlap( _S_AxB , _S_AxB_2 );
231 
232 
233  LOG(logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
234  << JAB_dimer.size1() << "x"
235  << JAB_dimer.size2() << "]" << flush;
236 
237  ub::matrix<double> JAB_temp( _levelsA + _levelsB, _levelsA + _levelsB );
238  ub::noalias(JAB_temp) = ub::prod( JAB_dimer, _S_AxB_2 );
239  (*_JAB) = ub::prod( _S_AxB_2, JAB_temp );
240 
241 
242  LOG(logDEBUG,*_pLog) << "Done with electronic couplings" << flush;
243  return true;
244

258 bool Overlap::CalculateIntegralsOptimized(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
259  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
260 
261  LOG(logDEBUG,*_pLog) << "Calculating electronic couplings" << flush;
262 
263  // constructing the direct product orbA x orbB
264  int _basisA = _orbitalsA->getBasisSetSize();
265  int _basisB = _orbitalsB->getBasisSetSize();
266 
267  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
268  LOG(logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
269  return false;
270  }
271 
272  int _levelsA = _orbitalsA->getNumberOfLevels();
273  int _levelsB = _orbitalsB->getNumberOfLevels();
274 
275  boost::timer t; // start timing
276  double _st = t.elapsed();
277 
278  LOG(logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
279  << " B[" << _levelsB << ":" << _basisB << "]" << flush;
280 
281  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
282  LOG(logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
283  return false;
284  }
285 
286  // these flags should be set before any ublas header is called
287  // #define NDEBUG
288  // otherwise the code is very inefficient
289 
290  // | Orbitals_A 0 | | Overlap_A |
291  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
292  //constructing a slice of the Overlap matrix
293  ub::matrix_range< ub::symmetric_matrix<double> > Overlap_A = ub::project( *_orbitalsAB->getOverlap(), ub::range ( 0, _basisA), ub::range (0, _basisA +_basisB) );
294  ub::matrix_range< ub::symmetric_matrix<double> > Overlap_B = ub::project( *_orbitalsAB->getOverlap(), ub::range ( _basisA, _basisA +_basisB ), ub::range (0, _basisA +_basisB) );
295 
296  LOG(logDEBUG,*_pLog) << "Projecting the monomer onto dimer orbitals [" << _levelsA + _levelsB << "x" << _basisA + _basisB << "]";
297  ub::matrix<double> _psi_AB ( _levelsA + _levelsB, _basisA + _basisB );
298 
299  ub::matrix_range< ub::matrix<double> > _psi_AB_A = ub::project( _psi_AB, ub::range (0, _levelsA ), ub::range ( 0, _basisA +_basisB ) ) ;
300  ub::noalias(_psi_AB_A) = ub::prod(*_orbitalsA->getOrbitals(), Overlap_A);
301 
302  ub::matrix_range< ub::matrix<double> > _psi_AB_B = ub::project( _psi_AB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA +_basisB ) ) ;
303  ub::noalias(_psi_AB_B) = ub::prod(*_orbitalsB->getOrbitals(), Overlap_B );
304  LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s) " << flush; _st = t.elapsed();
305 
306  ub::matrix<double> _psi_AxB_dimer_basis (_levelsA + _levelsB, _basisA + _basisB );
307  ub::matrix<double> OrbAB_Transp = ub::trans( *_orbitalsAB->getOrbitals() );
308  LOG(logDEBUG,*_pLog) << "Transposing OrbitalsAB (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << flush; _st = t.elapsed();
309  ub::noalias(_psi_AxB_dimer_basis) = ub::prod( _psi_AB, OrbAB_Transp );
310 
311  LOG(logDEBUG,*_pLog) << "Multiplying PsiAB x OrbitalsAB (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << flush; _st = t.elapsed();
312 
313  _psi_AB.resize(0,0,false); OrbAB_Transp.resize(0,0,false);
314 
315  // _psi_AxB_dimer_basis * F * _psi_AxB_dimer_basis^T
316  LOG(logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis";
317  ub::zero_matrix<double> _zero ( _levelsA + _levelsB, _levelsA + _levelsB );
318  ub::matrix<double> JAB_dimer( _zero ) ;
319  ub::vector<double> energies = (*_orbitalsAB->getEnergies());
320 
321  for ( int i1 = 0; i1 < _levelsA + _levelsB ; i1++ ) {
322  for ( int i2 = i1; i2 < _levelsA + _levelsB; i2++ ) {
323  for ( int k = 0; k < _basisA + _basisB; k++ ) {
324  JAB_dimer(i1,i2) += _psi_AxB_dimer_basis.at_element(i1, k) * _psi_AxB_dimer_basis.at_element(i2, k) * energies(k);
325  }
326  JAB_dimer(i2,i1) = JAB_dimer(i1,i2);
327  }}
328  energies.clear();
329  LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << flush; _st = t.elapsed();
330 
331  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
332  ub::symmetric_matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
333  _psi_AxB_dimer_basis.resize(0,0,false);
334 
335  ub::matrix<double> _S_AxB_2(_S_AxB.size1(), _S_AxB.size1() );
336 
337  ub::trans( _S_AxB );
338  LOG(logDEBUG,*_pLog) << "Calculating square root of the overlap matrix ["
339  << _S_AxB.size1() << "x"
340  << _S_AxB.size2() << "]";
341  SQRTOverlap( _S_AxB , _S_AxB_2 );
342  _S_AxB.resize(0,0,false);
343  LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << flush; _st = t.elapsed();
344 
345 
346  LOG(logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
347  << JAB_dimer.size1() << "x"
348  << JAB_dimer.size2() << "]";
349 
350  ub::matrix<double> JAB_temp( _levelsA + _levelsB, _levelsA + _levelsB );
351 
352  ub::noalias(JAB_temp) = ub::prod( JAB_dimer, _S_AxB_2 );
353  (*_JAB) = ub::prod( _S_AxB_2, JAB_temp );
354 
355  // cleanup
356  JAB_dimer.resize(0,0,false); JAB_temp.resize(0,0,false); _S_AxB_2.resize(0,0,false);
357  LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << flush; _st = t.elapsed();
358 
359  LOG(logDEBUG,*_pLog) << "Done with electronic couplings" << flush;
360  return true;
