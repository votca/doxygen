40 void Overlap::SQRTOverlap(ub::symmetric_matrix<double> &S,
41  ub::matrix<double> &S2 ) {
42 
43  double (*_inv_sqrt)(double);
44  _inv_sqrt = &inv_sqrt;
45 
46  ub::vector<double> _eigenvalues;
47  ub::matrix<double> _eigenvectors;
48 
49  int _size = S.size1();
50 
51  _eigenvalues.resize( _size );
52  _eigenvectors.resize( _size, _size );
53 
54  //convert from symmetric to matrix for faster evaluation
55  ub::matrix<double> temp=S;
56 
57 
58  // TEST CASE FOR EIGEN
59  ub::matrix<double> m(3,3);
60  /*
61  m(0,0) = -2;
62  m(0,1) = -4;
63  m(0,2) = 2;
64  m(1,0) = -2;
65  m(1,1) = 1;
66  m(1,2) = 2;
67  m(2,0) = 4;
68  m(2,1) = 2;
69  m(2,2) = 5;
70  */
71 
72  /* 4 -2 -2
73  m(0,0) = 1;
74  m(0,1) = -3;
75  m(0,2) = 3;
76  m(1,0) = 3;
77  m(1,1) = -5;
78  m(1,2) = 3;
79  m(2,0) = 6;
80  m(2,1) = -6;
81  m(2,2) = 4;
82  */
83 
84  /*symmetric, -1, -1, 8
85  m(0,0) = 2;
86  m(0,1) = 1;
87  m(0,2) = -2;
88  m(1,0) = 1;
89  m(1,1) = 0;
90  m(1,2) = 1;
91  m(2,0) = -2;
92  m(2,1) = 1;
93  m(2,2) = 0;
94 
95 
96  _eigenvalues.resize( 3 );
97  _eigenvectors.resize( 3, 3 );
98 
99  linalg::eigenvalues_symm(m, _eigenvalues, _eigenvectors);

101  std::cout << "\nEigenvalues" << std::endl << _eigenvalues;
102  std::cout << "\nEigenvectors" << std::endl << _eigenvectors << std::endl;
103  exit(0);
104  */
105 
106 
107  linalg::eigenvalues_symm(temp, _eigenvalues, _eigenvectors);
108 
109  // compute inverse sqrt of all eigenvalues
110  std::transform(_eigenvalues.begin(), _eigenvalues.end(), _eigenvalues.begin(), _inv_sqrt );
111 
112  // form a diagonal matrix S^{-1/2}
113  ub::diagonal_matrix<double> _diagS2( _eigenvalues.size(), _eigenvalues.data() );
114 
115  // multiply from the left on the U
116  ub::matrix<double> _temp = ub::prod( _eigenvectors, _diagS2 );
117 
118  // multiply from the right on the transpose U
119  S2 = ub::prod( _temp, ub::trans( _eigenvectors ) );

123 double Overlap::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
124  Orbitals* _orbitalsB, ub::matrix<double>* _JAB, double _energy_difference ) {
125 
126 
127  int _levelsA = _orbitalsA->getNumberOfLevels();
128 
129  if ( _energy_difference != 0 ) {
130  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
131  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
132 
133  double _JAB_sq = 0; double _JAB_one_level;
134 
135  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
136  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
137  //cout << *iA << ':' << *iB << endl;
138  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
139  _JAB_sq += _JAB_one_level*_JAB_one_level ;
140  }
141  }
142 
143  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
144 
145  } else {
146 
147  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
148 
149  }
150  // the matrix should be symmetric, could also return this element
151  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );

165 bool Overlap::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
166  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
167 
168  CTP_LOG(logDEBUG,*_pLog) << "Calculating electronic couplings" << std::flush;
169 
170  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
171  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
172  const std::vector<QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
173 
174  for (unsigned i=0;i<atomsAB.size();i++){
175  QMAtom* dimer=atomsAB[i];
176  QMAtom* monomer=NULL;
177  if (i<atomsA.size()){
178  monomer=atomsA[i];
179  }
180  else if (i<atomsB.size()+atomsA.size() ){
181  monomer=atomsB[i-atomsA.size()];
182  }
183  else{
184  throw std::runtime_error((format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
185  }
186 
187  if(monomer->type != dimer->type){
188  throw std::runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
189  }
190  if(std::abs(monomer->x-dimer->x)>0.001 || std::abs(monomer->y-dimer->y)>0.001 || std::abs(monomer->z-dimer->z)>0.001){
191  CTP_LOG(logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << std::flush;
192  break;
193  }
194 
195  }
196 
197  // constructing the direct product orbA x orbB
198  int _basisA = _orbitalsA->getBasisSetSize();
199  int _basisB = _orbitalsB->getBasisSetSize();
200 
201  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
202  CTP_LOG(logERROR,*_pLog) << "Basis set size is not stored in monomers" << std::flush;
203  return false;
204  }
205 
206  int _levelsA = _orbitalsA->getNumberOfLevels();
207  int _levelsB = _orbitalsB->getNumberOfLevels();
208 
209  //boost::timer t; // start timing
210  //double _st = t.elapsed();
211 
212  CTP_LOG(logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
213  << " B[" << _levelsB << ":" << _basisB << "]" << std::flush;
214 
215  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
216  CTP_LOG(logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << std::flush;
217  return false;
218  }
219 
220  // | Orbitals_A 0 | | Overlap_A |
221  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
222  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
223  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
224  ub::matrix<double> _psi_AxB ( _levelsA + _levelsB, _basisA + _basisB );
225 
226  CTP_LOG(logDEBUG,*_pLog) << "BOOST: Constructing direct product AxB ["
227  << _psi_AxB.size1() << "x"
228  << _psi_AxB.size2() << "]" << std::flush;
229 
230 
231 /*
232  // TEST CASE FOR EIGEN
233  ub::matrix<double> m1_(2,3);
234  m1_(0,0) = 1;
235  m1_(0,1) = 2;
236  m1_(0,2) = 3;
237  m1_(1,0) = 4;
238  m1_(1,1) = 5;
239  m1_(1,2) = 6;
240  //std::cout << std::endl << m1_ << std::endl ;
241 
242  ub::matrix<double> m2_(3,2);
243  m2_(0,0) = 7;
244  m2_(0,1) = 8;
245  m2_(1,0) = 9;
246  m2_(1,1) = 10;
247  m2_(2,0) = 11;
248  m2_(2,1) = 12;
249  //std::cout << std::endl << m2_ << std::endl ;
250 
251  ub::prod(m1_, m2_);
252  */
253 
254  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
255  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
256  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
257  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
258 
259  // psi_AxB * S_AB * psi_AB
260  CTP_LOG(logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << std::flush;
261  ub::matrix<double> _orbitalsAB_Transposed = ub::trans( *_orbitalsAB->getOrbitals() );
262  if ( (*_orbitalsAB->getOverlap()).size1() == 0 ) {
263  CTP_LOG(logERROR,*_pLog) << "Overlap matrix is not stored";
264  return false;
265  }
266 
267  ub::matrix<double> _psi_AB = ub::prod( *_orbitalsAB->getOverlap(), _orbitalsAB_Transposed );
268  ub::matrix<double> _psi_AxB_dimer_basis = ub::prod( _psi_AxB, _psi_AB );
269  _psi_AB.clear();
270 
271  //check to see if projection quality is sufficient
272  for (unsigned i=0;i<_psi_AxB_dimer_basis.size1();i++){
273  double mag=0.0;
274  for (unsigned j=0;j<_psi_AxB_dimer_basis.size2();j++){
275  mag+=_psi_AxB_dimer_basis(i,j)*_psi_AxB_dimer_basis(i,j);
276 
277  }
278  if (mag<0.95){
279  throw std::runtime_error("\nERROR: Projection of monomer orbitals on dimer is insufficient, increase dimer basis.\n");
280  }
281  }
282 
283 
284  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
285  CTP_LOG(logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << std::flush;
286  ub::diagonal_matrix<double> _fock_AB( _orbitalsAB->getNumberOfLevels(), (*_orbitalsAB->getEnergies()).data() );
287  ub::matrix<double> _temp = ub::prod( _fock_AB, ub::trans( _psi_AxB_dimer_basis ) ) ;
288  ub::matrix<double> JAB_dimer = ub::prod( _psi_AxB_dimer_basis, _temp);
289 
290  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
291  CTP_LOG(logDEBUG,*_pLog) << "Constructing Overlap matrix" << std::flush;
292  ub::symmetric_matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
293  ub::matrix<double> _S_AxB_2(_S_AxB.size1(), _S_AxB.size1() );
294  ub::trans( _S_AxB );
295 
296  // Square root of the overlap matrix
297  CTP_LOG(logDEBUG,*_pLog) << "Calculating square root of the overlap matrix" << std::flush;
298  SQRTOverlap( _S_AxB , _S_AxB_2 );
299 
300 
301  CTP_LOG(logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
302  << JAB_dimer.size1() << "x"
303  << JAB_dimer.size2() << "]" << std::flush;
304 
305  ub::matrix<double> JAB_temp( _levelsA + _levelsB, _levelsA + _levelsB );
306  ub::noalias(JAB_temp) = ub::prod( JAB_dimer, _S_AxB_2 );
307  (*_JAB) = ub::prod( _S_AxB_2, JAB_temp );
308 
309 
310  CTP_LOG(logDEBUG,*_pLog) << "Done with electronic couplings" << std::flush;
311  return true;
312

326 bool Overlap::CalculateIntegralsOptimized(Orbitals* _orbitalsA, Orbitals* _orbitalsB,

373
