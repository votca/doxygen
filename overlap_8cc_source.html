39 void Overlap::SQRTOverlap(ub::symmetric_matrix<double> &S,
40  ub::matrix<double> &S2 ) {
41 
42  double (*_inv_sqrt)(double);
43  _inv_sqrt = &inv_sqrt;
44 
45  ub::vector<double> _eigenvalues;
46  ub::matrix<double> _eigenvectors;
47 
48  int _size = S.size1();
49 
50  _eigenvalues.resize( _size );
51  _eigenvectors.resize( _size, _size );
52  //convert from symmatrix to matrix for faster evaluation
53  ub::matrix<double> temp=S;
54  linalg_eigenvalues(temp, _eigenvalues, _eigenvectors);
55 
56  // compute inverse sqrt of all eigenvalues
57  std::transform(_eigenvalues.begin(), _eigenvalues.end(), _eigenvalues.begin(), _inv_sqrt );
58 
59  // form a diagonal matrix S^{-1/2}
60  ub::diagonal_matrix<double> _diagS2( _eigenvalues.size(), _eigenvalues.data() );
61 
62  // multiply from the left on the U
63  ub::matrix<double> _temp = ub::prod( _eigenvectors, _diagS2 );
64 
65  // multiply from the right on the transpose U
66  S2 = ub::prod( _temp, ub::trans( _eigenvectors ) );
67  return;
68  }

70 double Overlap::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
71  Orbitals* _orbitalsB, ub::matrix<double>* _JAB, double _energy_difference ) {
72 
73 
74  int _levelsA = _orbitalsA->getNumberOfLevels();
75 
76  if ( _energy_difference != 0 ) {
77  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
78  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
79 
80  double _JAB_sq = 0; double _JAB_one_level;
81 
82  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
83  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
84  //cout << *iA << ':' << *iB << endl;
85  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
86  _JAB_sq += _JAB_one_level*_JAB_one_level ;
87  }
88  }
89 
90  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
91 
92  } else {
93 
94  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
95 
96  }
97  // the matrix should be symmetric, could also return this element
98  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );

112 bool Overlap::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
113  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
114 
115  CTP_LOG(logDEBUG,*_pLog) << "Calculating electronic couplings" << std::flush;
116 
117  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
118  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
119  const std::vector<QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
120 
121  for (unsigned i=0;i<atomsAB.size();i++){
122  QMAtom* dimer=atomsAB[i];
123  QMAtom* monomer=NULL;
124  if (i<atomsA.size()){
125  monomer=atomsA[i];
126  }
127  else if (i<atomsB.size()+atomsA.size() ){
128  monomer=atomsB[i-atomsA.size()];
129  }
130  else{
131  throw std::runtime_error((format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
132  }
133 
134  if(monomer->type != dimer->type){
135  throw std::runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
136  }
137  if(std::abs(monomer->x-dimer->x)>0.001 || std::abs(monomer->y-dimer->y)>0.001 || std::abs(monomer->z-dimer->z)>0.001){
138  CTP_LOG(logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << std::flush;
139  break;
140  }
141 
142  }
143 
144 
145 
146 
147  // constructing the direct product orbA x orbB
148  int _basisA = _orbitalsA->getBasisSetSize();
149  int _basisB = _orbitalsB->getBasisSetSize();
150 
151  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
152  CTP_LOG(logERROR,*_pLog) << "Basis set size is not stored in monomers" << std::flush;
153  return false;
154  }
155 
156  int _levelsA = _orbitalsA->getNumberOfLevels();
157  int _levelsB = _orbitalsB->getNumberOfLevels();
158 
159  //boost::timer t; // start timing
160  //double _st = t.elapsed();
161 
162  CTP_LOG(logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
163  << " B[" << _levelsB << ":" << _basisB << "]" << std::flush;
164 
165  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
166  CTP_LOG(logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << std::flush;
167  return false;
168  }
169 
170  // | Orbitals_A 0 | | Overlap_A |
171  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
172  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
173  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
174  ub::matrix<double> _psi_AxB ( _levelsA + _levelsB, _basisA + _basisB );
175 
176 
177  CTP_LOG(logDEBUG,*_pLog) << "Constructing direct product AxB ["
178  << _psi_AxB.size1() << "x"
179  << _psi_AxB.size2() << "]" << std::flush;
180 
181  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
182  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
183  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
184  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
185 
186  // psi_AxB * S_AB * psi_AB
187  CTP_LOG(logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << std::flush;
188  ub::matrix<double> _orbitalsAB_Transposed = ub::trans( *_orbitalsAB->getOrbitals() );
189  if ( (*_orbitalsAB->getOverlap()).size1() == 0 ) {
190  CTP_LOG(logERROR,*_pLog) << "Overlap matrix is not stored";
191  return false;
192  }
193 
194  ub::matrix<double> _psi_AB = ub::prod( *_orbitalsAB->getOverlap(), _orbitalsAB_Transposed );
195  ub::matrix<double> _psi_AxB_dimer_basis = ub::prod( _psi_AxB, _psi_AB );
196  _psi_AB.clear();
197 
198  //check to see if projection quality is sufficient
199  for (unsigned i=0;i<_psi_AxB_dimer_basis.size1();i++){
200  double mag=0.0;
201  for (unsigned j=0;j<_psi_AxB_dimer_basis.size2();j++){
202  mag+=_psi_AxB_dimer_basis(i,j)*_psi_AxB_dimer_basis(i,j);
203 
204  }
205  if (mag<0.95){
206  throw std::runtime_error("\nERROR: Projection of monomer orbitals on dimer is insufficient, maybe the orbital order is screwed up, otherwise increase dimer basis.\n");
207  }
208  }
209 
210 
211  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
212  CTP_LOG(logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << std::flush;
213  ub::diagonal_matrix<double> _fock_AB( _orbitalsAB->getNumberOfLevels(), (*_orbitalsAB->getEnergies()).data() );
214  ub::matrix<double> _temp = ub::prod( _fock_AB, ub::trans( _psi_AxB_dimer_basis ) ) ;
215  ub::matrix<double> JAB_dimer = ub::prod( _psi_AxB_dimer_basis, _temp);
216 
217  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
218  CTP_LOG(logDEBUG,*_pLog) << "Constructing Overlap matrix" << std::flush;
219  ub::symmetric_matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
220  ub::matrix<double> _S_AxB_2(_S_AxB.size1(), _S_AxB.size1() );
221  ub::trans( _S_AxB );
222 
223  // Square root of the overlap matrix
224  CTP_LOG(logDEBUG,*_pLog) << "Calculating square root of the overlap matrix" << std::flush;
225  SQRTOverlap( _S_AxB , _S_AxB_2 );
226 
227 
228  CTP_LOG(logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
229  << JAB_dimer.size1() << "x"
230  << JAB_dimer.size2() << "]" << std::flush;
231 
232  ub::matrix<double> JAB_temp( _levelsA + _levelsB, _levelsA + _levelsB );
233  ub::noalias(JAB_temp) = ub::prod( JAB_dimer, _S_AxB_2 );
234  (*_JAB) = ub::prod( _S_AxB_2, JAB_temp );
235 
236 
237  CTP_LOG(logDEBUG,*_pLog) << "Done with electronic couplings" << std::flush;
238  return true;
239

253 bool Overlap::CalculateIntegralsOptimized(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
254  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
255 
256  CTP_LOG(logDEBUG,*_pLog) << "Calculating electronic couplings" << std::flush;
257 
258  // constructing the direct product orbA x orbB
259  int _basisA = _orbitalsA->getBasisSetSize();
260  int _basisB = _orbitalsB->getBasisSetSize();
261 
262  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
263  CTP_LOG(logERROR,*_pLog) << "Basis set size is not stored in monomers" << std::flush;
264  return false;
265  }
266 
267  int _levelsA = _orbitalsA->getNumberOfLevels();
268  int _levelsB = _orbitalsB->getNumberOfLevels();
269 
270  boost::timer t; // start timing
271  double _st = t.elapsed();
272 
273  CTP_LOG(logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
274  << " B[" << _levelsB << ":" << _basisB << "]" << std::flush;
275 
276  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
277  CTP_LOG(logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << std::flush;
278  return false;
279  }
280 
281  // these flags should be set before any ublas header is called
282  // #define NDEBUG
283  // otherwise the code is very inefficient
284 
285  // | Orbitals_A 0 | | Overlap_A |
286  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
287  //constructing a slice of the Overlap matrix
288  ub::matrix_range< ub::symmetric_matrix<double> > Overlap_A = ub::project( *_orbitalsAB->getOverlap(), ub::range ( 0, _basisA), ub::range (0, _basisA +_basisB) );
289  ub::matrix_range< ub::symmetric_matrix<double> > Overlap_B = ub::project( *_orbitalsAB->getOverlap(), ub::range ( _basisA, _basisA +_basisB ), ub::range (0, _basisA +_basisB) );
290 
291  CTP_LOG(logDEBUG,*_pLog) << "Projecting the monomer onto dimer orbitals [" << _levelsA + _levelsB << "x" << _basisA + _basisB << "]";
292  ub::matrix<double> _psi_AB ( _levelsA + _levelsB, _basisA + _basisB );
293 
294  ub::matrix_range< ub::matrix<double> > _psi_AB_A = ub::project( _psi_AB, ub::range (0, _levelsA ), ub::range ( 0, _basisA +_basisB ) ) ;
295  ub::noalias(_psi_AB_A) = ub::prod(*_orbitalsA->getOrbitals(), Overlap_A);
296 
297  ub::matrix_range< ub::matrix<double> > _psi_AB_B = ub::project( _psi_AB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA +_basisB ) ) ;
298  ub::noalias(_psi_AB_B) = ub::prod(*_orbitalsB->getOrbitals(), Overlap_B );
299  CTP_LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s) " << std::flush; _st = t.elapsed();
300 
301  ub::matrix<double> _psi_AxB_dimer_basis (_levelsA + _levelsB, _basisA + _basisB );
302  ub::matrix<double> OrbAB_Transp = ub::trans( *_orbitalsAB->getOrbitals() );
303  CTP_LOG(logDEBUG,*_pLog) << "Transposing OrbitalsAB (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << std::flush; _st = t.elapsed();
304  ub::noalias(_psi_AxB_dimer_basis) = ub::prod( _psi_AB, OrbAB_Transp );
305 
306  CTP_LOG(logDEBUG,*_pLog) << "Multiplying PsiAB x OrbitalsAB (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << std::flush; _st = t.elapsed();
307 
308  _psi_AB.resize(0,0,false); OrbAB_Transp.resize(0,0,false);
309 
310  // _psi_AxB_dimer_basis * F * _psi_AxB_dimer_basis^T
311  CTP_LOG(logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis";
312  ub::zero_matrix<double> _zero ( _levelsA + _levelsB, _levelsA + _levelsB );
313  ub::matrix<double> JAB_dimer( _zero ) ;
314  ub::vector<double> energies = (*_orbitalsAB->getEnergies());
315 
316  for ( int i1 = 0; i1 < _levelsA + _levelsB ; i1++ ) {
317  for ( int i2 = i1; i2 < _levelsA + _levelsB; i2++ ) {
318  for ( int k = 0; k < _basisA + _basisB; k++ ) {
319  JAB_dimer(i1,i2) += _psi_AxB_dimer_basis.at_element(i1, k) * _psi_AxB_dimer_basis.at_element(i2, k) * energies(k);
320  }
321  JAB_dimer(i2,i1) = JAB_dimer(i1,i2);
322  }}
323  energies.clear();
324  CTP_LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << std::flush; _st = t.elapsed();
325 
326  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
327  ub::symmetric_matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
328  _psi_AxB_dimer_basis.resize(0,0,false);
329 
330  ub::matrix<double> _S_AxB_2(_S_AxB.size1(), _S_AxB.size1() );
331 
332  ub::trans( _S_AxB );
333  CTP_LOG(logDEBUG,*_pLog) << "Calculating square root of the overlap matrix ["
334  << _S_AxB.size1() << "x"
335  << _S_AxB.size2() << "]";
336  SQRTOverlap( _S_AxB , _S_AxB_2 );
337  _S_AxB.resize(0,0,false);
338  CTP_LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << std::flush; _st = t.elapsed();
339 
340 
341  CTP_LOG(logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
342  << JAB_dimer.size1() << "x"
343  << JAB_dimer.size2() << "]";
344 
345  ub::matrix<double> JAB_temp( _levelsA + _levelsB, _levelsA + _levelsB );
346 
347  ub::noalias(JAB_temp) = ub::prod( JAB_dimer, _S_AxB_2 );
348  (*_JAB) = ub::prod( _S_AxB_2, JAB_temp );
349 
350  // cleanup
351  JAB_dimer.resize(0,0,false); JAB_temp.resize(0,0,false); _S_AxB_2.resize(0,0,false);
352  CTP_LOG(logDEBUG,*_pLog) << " (" << t.elapsed() - _st << "s)" << "\x1b[0;39m" << std::flush; _st = t.elapsed();
353 
354  CTP_LOG(logDEBUG,*_pLog) << "Done with electronic couplings" << std::flush;
355  return true;
