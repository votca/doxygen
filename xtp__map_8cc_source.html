99 void XtpMap::Run() {
100 
101  // +++++++++++++++++++++++++++++++++++++ //
102  // Initialize MD2QM Engine and SQLite Db //
103  // +++++++++++++++++++++++++++++++++++++ //
104 
105  bool abort = false;
106  _outdb = _op_vm["file"].as<string> ();
107  _statsav.Open(_qmtopol, _outdb, false);
108  int frames_in_db = _statsav.FramesInDatabase();
109  if (frames_in_db > 0) {
110  cout << endl << "ERROR <xtp_map> : state file '"
111  << _outdb << "' already in use. Abort." << endl;
112  abort = true;
113  }
114  _statsav.Close();
115  if (abort) return;
116 
117  string cgfile = _op_vm["segments"].as<string> ();
118  _md2qm.Initialize(cgfile);
119 
120 
121  // ++++++++++++++++++++++++++++ //
122  // Create MD topology from file //
123  // ++++++++++++++++++++++++++++ //
124 
125  // Create topology reader
126  string topfile = _op_vm["topology"].as<string> ();
127  CSG::TopologyReader *topread;
128  topread = CSG::TopReaderFactory().Create(topfile);
129 
130  if (topread == NULL) {
131  throw runtime_error( string("Input format not supported: ")
132  + _op_vm["topology"].as<string> () );
133  }
134 
135  topread->ReadTopology(topfile, this->_mdtopol);
136  if (TOOLS::globals::verbose) {
137  cout << "Read MD topology from " << topfile << ": Found "
138  << _mdtopol.BeadCount() << " atoms in "
139  << _mdtopol.MoleculeCount() << " molecules. "
140  << endl;
141  }
142 
143  // ++++++++++++++++++++++++++++++ //
144  // Create MD trajectory from file //
145  // ++++++++++++++++++++++++++++++ //
146 
147  // Create trajectory reader and initialize
148  string trjfile = _op_vm["coordinates"].as<string> ();
149  CSG::TrajectoryReader *trjread;
150  trjread = CSG::TrjReaderFactory().Create(trjfile);
151 
152  if (trjread == NULL) {
153  throw runtime_error( string("Input format not supported: ")
154  + _op_vm["coordinates"].as<string> () );
155  }
156  trjread->Open(trjfile);
157  trjread->FirstFrame(this->_mdtopol);
158 
159  int firstFrame = 1;
160  int nFrames = 1;
161  bool beginAt = 0;
162  double startTime = _mdtopol.getTime();
163 
164  if (_op_vm.count("nframes")) {
165  nFrames = _op_vm["nframes"].as<int> ();
166  }
167  if (_op_vm.count("first-frame")) {
168  firstFrame = _op_vm["first-frame"].as<int> ();
169  }
170  if (_op_vm.count("begin")) {
171  beginAt = true;
172  startTime = _op_vm["begin"].as<double> ();
173  }
174 
175  // Extract first frame specified
176  bool hasFrame;
177 
178  for (hasFrame = true; hasFrame == true;
179  hasFrame = trjread->NextFrame(this->_mdtopol)) {
180  if ( ((_mdtopol.getTime() < startTime) && beginAt )
181  || firstFrame > 1 ) {
182  firstFrame--;
183  continue;
184  }
185  break;
186  }
187  if ( ! hasFrame) {
188  trjread->Close();
189  delete trjread;
190 
191  throw runtime_error("Time or frame number exceeds trajectory length");
192  }
193 
194  // +++++++++++++++++++++++++ //
195  // Convert MD to QM Topology //
196  // +++++++++++++++++++++++++ //
197 
198  for (int saved = 0; hasFrame && saved < nFrames;
199  hasFrame = trjread->NextFrame(this->_mdtopol), saved++) {
200 
201  _md2qm.Md2Qm(&_mdtopol, &_qmtopol);
202 
203  // +++++++++++++++++++++++++ //
204  // Save to SQLite State File //
205  // +++++++++++++++++++++++++ //
206 
207  this->Save("");
208  }
209 
210  // trjread->Close();
211  // delete trjread;
212 
213 }

215 void XtpMap::Save(string mode) {
216 
217  _statsav.Open(_qmtopol, _outdb);
218 
219  _statsav.WriteFrame();
220 
221  if (TOOLS::globals::verbose) {
222  CTP::Topology *TopSQL = NULL;
223  TopSQL = _statsav.getTopology();
224  cout << endl << "Checking topology read from SQL file." << endl;
225  string pdbfile = "system.pdb";
226  _md2qm.CheckProduct(TopSQL, pdbfile);
227  }
228 
229  _statsav.Close();
230 
231 }
