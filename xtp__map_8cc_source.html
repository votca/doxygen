97 void XtpMap::Run() {
98 
99  std::string name = ProgramName();
100  if (VersionString() != "") {
101  name = name + ", version " + VersionString();
102  }
103  XTP::HelpTextHeader(name);
104 
105  // ++++++++++++++++++++++++++++ //
106  // Create MD topology from file //
107  // ++++++++++++++++++++++++++++ //
108 
109  // Create topology reader
110  string topfile = _op_vm["topology"].as<string>();
111  std::unique_ptr<CSG::TopologyReader> topread =
112  std::unique_ptr<CSG::TopologyReader>(
113  CSG::TopReaderFactory().Create(topfile));
114 
115  if (topread == nullptr) {
116  throw runtime_error(string("Input format not supported: ") +
117  _op_vm["topology"].as<string>());
118  }
119  CSG::Topology mdtopol;
120  topread->ReadTopology(topfile, mdtopol);
121  if (votca::Log::verbose()) {
122  cout << "Read MD topology from " << topfile << ": Found "
123  << mdtopol.BeadCount() << " atoms in " << mdtopol.MoleculeCount()
124  << " molecules. " << endl;
125  }
126 
127  // ++++++++++++++++++++++++++++++ //
128  // Create MD trajectory from file //
129  // ++++++++++++++++++++++++++++++ //
130 
131  // Create trajectory reader and initialize
132  string trjfile = _op_vm["coordinates"].as<string>();
133  std::unique_ptr<CSG::TrajectoryReader> trjread =
134  std::unique_ptr<CSG::TrajectoryReader>(
135  CSG::TrjReaderFactory().Create(trjfile));
136 
137  if (trjread == nullptr) {
138  throw runtime_error(string("Input format not supported: ") +
139  _op_vm["coordinates"].as<string>());
140  }
141  trjread->Open(trjfile);
142  trjread->FirstFrame(mdtopol);
143 
144  string mapfile = _op_vm["segments"].as<string>();
145  if (_op_vm.count("makesegments")) {
146  if (TOOLS::filesystem::FileExists(mapfile)) {
147  cout << endl
148  << "xtp_map : map file '" << mapfile
149  << "' already in use. Delete the current mapfile or specify a "
150  "different name."
151  << endl;
152  return;
153  }
154 
155  cout << " Writing template mapfile to " << mapfile << std::endl;
156 
157  TOOLS::Property mapfile_prop("topology", "", "");
158  TOOLS::Property& molecules = mapfile_prop.add("molecules", "");
159 
160  std::map<std::string, const CSG::Molecule*> firstmolecule;
161 
162  std::map<std::string, votca::Index> molecule_names;
163  for (const CSG::Molecule* mol : mdtopol.Molecules()) {
164  if (!molecule_names.count(mol->getName())) {
165  firstmolecule[mol->getName()] = mol;
166  }
167  molecule_names[mol->getName()]++;
168  }
169  for (const auto& mol : molecule_names) {
170  std::cout << "Found " << mol.second << " with name " << mol.first
171  << std::endl;
172  }
173  for (const auto& mol : molecule_names) {
174  TOOLS::Property& molecule = molecules.add("molecule", "");
175  molecule.add("mdname", mol.first);
176  TOOLS::Property& segments = molecule.add("segments", "");
177  TOOLS::Property& segment = segments.add("segment", "");
178  segment.add("name", "UPTOYOU_BUTUNIQUE");
179  segment.add("qmcoords_n", "XYZFILE_GROUNDSTATE");
180  segment.add("multipoles_n", "MPSFILE_GROUNDSTATE");
181  segment.add("map2md", "WANTTOMAPTOMDGEOMETRY");
182  segment.add("U_xX_nN_h", "REORG1_hole");
183  segment.add("U_nX_nN_h", "REORG2_hole");
184  segment.add("U_xN_xX_h", "REORG3_hole");
185  TOOLS::Property& fragments = segment.add("fragments", "");
186  TOOLS::Property& fragment = fragments.add("fragment", "");
187  std::string atomnames = "";
188  const CSG::Molecule* csgmol = firstmolecule[mol.first];
189  std::vector<const CSG::Bead*> sortedbeads;
190  sortedbeads.reserve(csgmol->BeadCount());
191  for (const CSG::Bead* bead : csgmol->Beads()) {
192  sortedbeads.push_back(bead);
193  }
194  std::sort(sortedbeads.begin(), sortedbeads.end(),
195  [&](const CSG::Bead* b1, const CSG::Bead* b2) {
196  return b1->getId() < b2->getId();
197  });
198 
199  for (const CSG::Bead* bead : sortedbeads) {
200  atomnames += " " + std::to_string(bead->getResnr()) + ":" +
201  bead->getName() + ":" + std::to_string(bead->getId());
202  }
203  fragment.add("name", "UPTOYOU_BUTUNIQUE");
204  fragment.add("mdatoms", atomnames);
205  fragment.add("qmatoms", "IDS of QMATOMS i.e 0:C 1:H 2:C");
206  fragment.add("mpoles", "IDS of MPOLES i.e 0:C 1:H 2:C");
207  fragment.add("weights",
208  "weights for mapping(often atomic mass) i.e. 12 1 12");
209  fragment.add("localframe", "IDs of up to 3 qmatoms or mpoles i.e. 0 1 2");
210  std::ofstream template_mapfile(mapfile);
211  template_mapfile << mapfile_prop << std::flush;
212  template_mapfile.close();
213 
214  std::cout << "MOLECULETYPE " << csgmol->getName() << std::endl;
215  std::cout << "SAMPLECOORDINATES" << std::endl;
216  std::cout << "ID NAME COORDINATES[Angstroem] " << std::endl;
217  for (const CSG::Bead* bead : sortedbeads) {
218  Eigen::Vector3d pos = bead->getPos() * votca::tools::conv::nm2ang;
219  std::string output =
220  (boost::format("%1$i %2$s %3$+1.4f %4$+1.4f %5$+1.4f\n") %
221  bead->getId() % bead->getName() % pos[0] % pos[1] % pos[2])
222  .str();
223  std::cout << output;
224  }
225  }
226  std::cout << std::flush;
227  return;
228  }
229 
230  if (!TOOLS::filesystem::FileExists(mapfile)) {
231  cout << endl
232  << "xtp_map : map file '" << mapfile << "' could not be found."
233  << endl;
234  return;
235  }
236  XTP::Md2QmEngine md2qm(mapfile);
237 
238  votca::Index firstFrame = _op_vm["first-frame"].as<votca::Index>();
239  votca::Index nFrames = _op_vm["nframes"].as<votca::Index>();
240  bool beginAt = false;
241  double time = _op_vm["begin"].as<double>();
242  double startTime = mdtopol.getTime();
243  if (time > 0.0) {
244  beginAt = true;
245  startTime = time;
246  }
247 
248  // Extract first frame specified
249  bool hasFrame;
250  votca::Index frames_found = 0;
251  votca::Index firstframecounter = firstFrame;
252  for (hasFrame = true; hasFrame == true;
253  hasFrame = trjread->NextFrame(mdtopol)) {
254  frames_found++;
255  if (((mdtopol.getTime() < startTime) && beginAt) || firstframecounter > 0) {
256  firstframecounter--;
257  continue;
258  }
259  break;
260  }
261  if (!hasFrame) {
262  trjread->Close();
263 
264  throw runtime_error("Time or frame number exceeds trajectory length");
265  }
266  if (votca::Log::verbose()) {
267  cout << "Read MD trajectory from " << trjfile << ": found " << frames_found
268  << " frames, starting from frame " << firstFrame << endl;
269  }
270  // +++++++++++++++++++++++++ //
271  // Convert MD to QM Topology //
272  // +++++++++++++++++++++++++ //
273 
274  string statefile = _op_vm["file"].as<string>();
275  if (TOOLS::filesystem::FileExists(statefile)) {
276  cout << endl
277  << "xtp_map : state file '" << statefile
278  << "' already in use. Delete the current statefile or specify a "
279  "different name."
280  << endl;
281  return;
282  }
283 
284  XTP::StateSaver statsav(statefile);
285  votca::Index laststep =
286  -1; // for some formats no step is given out so we check if the step
287  for (votca::Index saved = 0; hasFrame && saved < nFrames;
288  hasFrame = trjread->NextFrame(mdtopol), saved++) {
289  if (mdtopol.getStep() == laststep) {
290  mdtopol.setStep(laststep + 1);
291  }
292  laststep = mdtopol.getStep();
293  XTP::Topology qmtopol = md2qm.map(mdtopol);
294  statsav.WriteFrame(qmtopol);
295  }
296 }
