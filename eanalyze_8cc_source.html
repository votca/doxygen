25 void EAnalyze::Initialize(tools::Property &opt) {
26 
27  // update options with the VOTCASHARE defaults
28  UpdateWithDefaults(opt, "xtp");
29  std::string key = "options." + Identify();
30  if (opt.exists(key + ".resolution_pairs")) {
31  _resolution_pairs = opt.get(key + ".resolution_pairs").as<double>();
32  } else {
33  _skip_pairs = true;
34  }
35  if (opt.exists(key + ".resolution_sites")) {
36  _resolution_sites = opt.get(key + ".resolution_sites").as<double>();
37  } else {
38  _skip_sites = true;
39  }
40  if (opt.exists(key + ".resolution_space")) {
41  _resolution_space = opt.get(key + ".resolution_space").as<double>();
42  } else {
43  _skip_corr = true;
44  }
45 
46  if (opt.exists(key + ".pattern")) {
47  _seg_pattern = opt.get(key + ".pattern").as<std::string>();
48  } else
49  _seg_pattern = "*";
50 
51  if (opt.exists(key + ".states")) {
52  std::string statestrings = opt.get(key + ".states").as<std::string>();
53  tools::Tokenizer tok(statestrings, ",\n\t ");
54  std::vector<std::string> string_vec;
55  tok.ToVector(string_vec);
56  for (std::string &state : string_vec) {
57  _states.push_back(QMStateType(state));
58  }
59  } else {
60  _states.push_back(QMStateType::Electron);
61  _states.push_back(QMStateType::Hole);
62  }
63 
64  _doenergy_landscape = opt.ifExistsReturnElseReturnDefault<bool>(
65  key + ".energy_landscape", false);
66 
67  if (opt.exists(key + ".distancemode")) {
68  std::vector<std::string> choices = {"atoms", "centerofmass"};
69  std::string distancemode =
70  opt.ifExistsAndinListReturnElseThrowRuntimeError<std::string>(
71  key + ".distancemode", choices);
72  if (distancemode == "centerofmass") {
73  _atomdistances = false;
74  } else {
75  _atomdistances = true;
76  }
77  }

80 bool EAnalyze::EvaluateFrame(Topology &top) {
81 
82  // Short-list segments according to pattern
83  for (Segment &seg : top.Segments()) {
84  std::string seg_name = seg.getType();
85  if (votca::tools::wildcmp(_seg_pattern.c_str(), seg_name.c_str())) {
86  _seg_shortlist.push_back(&seg);
87  }
88  }
89  std::cout << std::endl
90  << "... ... Short-listed " << _seg_shortlist.size()
91  << " segments (pattern='" << _seg_pattern << "')" << std::flush;
92  std::cout
93  << std::endl
94  << "... ... ... NOTE Statistics of site energies and spatial"
95  << " correlations thereof are based on the short-listed segments only. "
96  << std::flush;
97  std::cout << std::endl
98  << "... ... ... "
99  << "Statistics of site-energy differences operate on the full list."
100  << std::flush;
101 
102  // Calculate
103  // ... Site-energy histogram, mean, width
104  // ... Pair-energy histogram, mean, width
105  // ... Site-energy correlation
106 
107  const QMNBList &nblist = top.NBList();
108 
109  for (QMStateType state : _states) {
110  std::cout << std::endl
111  << "... ... excited state " << state.ToString() << std::flush;
112 
113  if (!_seg_shortlist.size()) {
114  std::cout << std::endl
115  << "... ... ... No segments short-listed. Skip ... "
116  << std::flush;
117  } else {
118  if (_skip_sites) {
119  std::cout << std::endl
120  << "... ... ... Skip site-energy hist." << std::flush;
121  } else {
122  SiteHist(state);
123  }
124  if (_skip_corr) {
125  std::cout << std::endl
126  << "... ... ... Skip correlation ..." << std::flush;
127  } else {
128  SiteCorr(top, state);
129  }
130  }
131 
132  if (!nblist.size()) {
133  std::cout << std::endl
134  << "... ... ... No pairs in topology. Skip ... " << std::flush;
135  } else {
136  if (_skip_pairs) {
137  std::cout << std::endl
138  << "... ... ... Skip pair-energy hist." << std::flush;
139  } else {
140  PairHist(top, state);
141  }
142  }
143  }
144 
145  return true;
146 }

148 void EAnalyze::SiteHist(QMStateType state) const {
149 
150  std::vector<double> Es;
151  Es.reserve(_seg_shortlist.size());
152  for (Segment *seg : _seg_shortlist) {
153  Es.push_back(seg->getSiteEnergy(state) * tools::conv::hrt2ev);
154  }
155 
156  double MAX = *std::max_element(Es.begin(), Es.end());
157  double MIN = *std::min_element(Es.begin(), Es.end());
158  double sum = std::accumulate(Es.begin(), Es.end(), 0.0);
159  double AVG = sum / Es.size();
160  double sq_sum = std::inner_product(Es.begin(), Es.end(), Es.begin(), 0.0);
161  double STD = std::sqrt(sq_sum / Es.size() - AVG * AVG);
162 
163  // Prepare bins
164  int BIN = int((MAX - MIN) / _resolution_sites + 0.5) + 1;
165 
166  tools::HistogramNew hist;
167  hist.Initialize(MIN, MAX, BIN);
168  hist.ProcessRange<std::vector<double>::iterator>(Es.begin(), Es.end());
169  tools::Table &tab = hist.data();
170  tab.flags() = std::vector<char>(tab.size(), ' ');
171  std::string comment =
172  (boost::format("EANALYZE: SITE-ENERGY HISTOGRAM[eV] \n # AVG %1$4.7f STD "
173  "%2$4.7f MIN %3$4.7f MAX %4$4.7f") %
174  AVG % STD % MIN % MAX)
175  .str();
176  std::string filename = "eanalyze.sitehist_" + state.ToString() + ".out";
177  tab.set_comment(comment);
178  tab.Save(filename);
179 
180  // Write "seg x y z energy" with atomic {x,y,z}
181  if (_doenergy_landscape) {
182  std::string filename = "eanalyze.landscape_" + state.ToString() + ".out";
183  std::ofstream out;
184  out.open(filename);
185  if (!out) throw std::runtime_error("error, cannot open file " + filename);
186  for (Segment *seg : _seg_shortlist) {
187  if (seg->getId() < _first_seg) {
188  continue;
189  }
190  if (seg->getId() == _last_seg) {
191  break;
192  }
193  double E = seg->getSiteEnergy(state);
194  for (Atom &atm : *seg) {
195  out << boost::format("%1$3s %2$4.7f %3$4.7f %4$4.7f %5$4.7f\n") %
196  seg->getType() % atm.getPos().x() % atm.getPos().y() %
197  atm.getPos().z() % E;
198  }
199  }
200  out.close();
201  }
202 }

204 void EAnalyze::PairHist(const Topology &top, QMStateType state) const {
205 
206  const QMNBList &nblist = top.NBList();
207 
208  std::string filenamelist = "eanalyze.pairlist_" + state.ToString() + ".out";
209 
210  // Collect site-energy differences from neighbourlist
211  std::vector<double> dE;
212  dE.reserve(2 * nblist.size());
213  std::ofstream out;
214  out.open(filenamelist);
215  if (!out) throw std::runtime_error("error, cannot open file " + filenamelist);
216  for (QMPair *pair : nblist) {
217  double deltaE = pair->getdE12(state) * tools::conv::hrt2ev;
218  dE.push_back(deltaE);
219  dE.push_back(-deltaE);
220  out << boost::format("%1$5d %2$5d %3$4.7f \n") % pair->Seg1()->getId() %
221  pair->Seg2()->getId() % deltaE;
222  }
223  out.close();
224 
225  double MAX = *std::max_element(dE.begin(), dE.end());
226  double MIN = *std::min_element(dE.begin(), dE.end());
227  double sum = std::accumulate(dE.begin(), dE.end(), 0.0);
228  double AVG = sum / dE.size();
229  double sq_sum = std::inner_product(dE.begin(), dE.end(), dE.begin(), 0.0);
230  double STD = std::sqrt(sq_sum / dE.size() - AVG * AVG);

248 void EAnalyze::SiteCorr(const Topology &top, QMStateType state) const {
249 
250  std::vector<double> Es;
251  Es.reserve(_seg_shortlist.size());
252  for (Segment *seg : _seg_shortlist) {
253  Es.push_back(seg->getSiteEnergy(state) * tools::conv::hrt2ev);
254  }
255 
256  double sum = std::accumulate(Es.begin(), Es.end(), 0.0);
257  double AVG = sum / Es.size();
258  double sq_sum = std::inner_product(Es.begin(), Es.end(), Es.begin(), 0.0);
259  double VAR = sq_sum / Es.size() - AVG * AVG;
260  double STD = std::sqrt(VAR);
261 
262  // Collect inter-site distances, correlation product
263  tools::Table tabcorr;
264  int length = _seg_shortlist.size() * (_seg_shortlist.size() - 1) / 2;
265  tabcorr.resize(length);
266  int index = 0;
267  for (unsigned i = 0; i < _seg_shortlist.size(); i++) {
268  const Segment &segi = *_seg_shortlist[i];
269  for (unsigned j = i + 1; j < _seg_shortlist.size(); j++) {
270  const Segment &segj = *_seg_shortlist[j];
271  double R = 0;
272  if (_atomdistances) {
273  R = top.GetShortestDist(segi, segj);
274  } else {
275  R = (top.PbShortestConnect(segi.getPos(), segj.getPos())).norm();
276  }
277 
278  double C =
279  (segi.getSiteEnergy(state) - AVG) * (segj.getSiteEnergy(state) - AVG);
280  tabcorr.set(index, R * tools::conv::bohr2nm, C);
281  index++;
282  }
283  }
284 
285  double MIN = tabcorr.x().minCoeff();
286  double MAX = tabcorr.x().maxCoeff();
287 
288  // Prepare bins
289  int BIN = int((MAX - MIN) / _resolution_space + 0.5) + 1;
290  std::vector<std::vector<double> > histCs;
291  histCs.resize(BIN);
292 
293  for (int i = 0; i < tabcorr.size(); ++i) {
294  int bin = int((tabcorr.x()[i] - MIN) / _resolution_space + 0.5);
295  histCs[bin].push_back(tabcorr.y()[i]);
296  }
297 
298  tools::Table histC;
299  histC.SetHasYErr(true);
300  // Calculate spatial correlation
301  histC.resize(BIN);
302  for (int bin = 0; bin < BIN; ++bin) {
303  double corr = 0.0;
304  double dcorr2 = 0.0;
305  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
306  corr += histCs[bin][i] / VAR;
307  }
308  corr = corr / histCs[bin].size();
309  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
310  dcorr2 += (histCs[bin][i] / VAR / histCs[bin].size() - corr) *
311  (histCs[bin][i] / VAR / histCs[bin].size() - corr);
312  }
313  // error on mean value
314  dcorr2 = dcorr2 / histCs[bin].size() / (histCs[bin].size() - 1);
315  double R = MIN + bin * _resolution_space;
316  histC.set(bin, R, corr, ' ', std::sqrt(dcorr2));
317  }
318 
319  std::string filename = "eanalyze.sitecorr_" + state.ToString() + ".out";
320  std::string comment =
321  (boost::format("EANALYZE: DISTANCE[nm] SPATIAL SITE-ENERGY "
322  "CORRELATION[eV] \n # AVG "
323  "%1$4.7f STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") %
324  AVG % STD % MIN % MAX)
325  .str();
326  histC.set_comment(comment);
327  histC.Save(filename);
328 }
