29 void QMMM::Initialize(tools::Property& options) {
30 
31  std::string key = "options." + Identify();
32  ParseCommonOptions(options);
33  _jobfile = options.ifExistsReturnElseThrowRuntimeError<std::string>(
34  key + ".job_file");
35 
36  _print_regions_pdb = options.ifExistsReturnElseReturnDefault(
37  key + ".print_regions_pdb", _print_regions_pdb);
38 
39  _max_iterations = options.ifExistsReturnElseReturnDefault(
40  key + ".max_iterations", _max_iterations);
41 
42  if (options.exists(key + ".regions")) {
43  _regions_def = options.get(key + ".regions");
44  _regions_def.add("mapfile", _mapfile);
45  } else {
46  throw std::runtime_error("No region definitions found in optionsfile");
47  }
48 
49  if (options.exists(key + ".write_parse")) {
50  _write_parse = true;
51 
52  std::string states = options.ifExistsReturnElseReturnDefault<std::string>(
53  key + ".write_parse.states", "e h");
54  tools::Tokenizer tok(states, " ,;\n\t");
55  std::vector<std::string> statestrings = tok.ToVector();
56  _states.reserve(statestrings.size());
57  for (std::string s : statestrings) {
58  _states.push_back(QMState(s));
59  }
60  bool groundstate_found = false;
61  for (const QMState& state : _states) {
62  if (state.Type() == QMStateType::Gstate) groundstate_found = true;
63  }
64  if (!groundstate_found) {
65  _states.push_back(QMState("n"));
66  }
67  }
68 }

70 Job::JobResult QMMM::EvalJob(const Topology& top, Job& job, QMThread& Thread) {
71  std::chrono::time_point<std::chrono::system_clock> start =
72  std::chrono::system_clock::now();
73 
74  std::string qmmm_work_dir = "QMMM";
75  if (!this->hasQMRegion()) {
76  qmmm_work_dir = "MMMM";
77  }
78  std::string frame_dir =
79  "frame_" + boost::lexical_cast<std::string>(top.getStep());
80  std::string job_dir =
81  "job_" + std::to_string(job.getId()) + "_" + job.getTag();
82  boost::filesystem::path arg_path;
83  std::string workdir =
84  (arg_path / qmmm_work_dir / frame_dir / job_dir).generic_string();
85  boost::filesystem::create_directories(workdir);
86  Job::JobResult jres = Job::JobResult();
87  Logger& pLog = Thread.getLogger();
88  JobTopology jobtop = JobTopology(job, pLog, workdir);
89  jobtop.BuildRegions(top, _regions_def);
90 
91  if (_print_regions_pdb) {
92  std::string pdb_filename = "regions.pdb";
93  XTP_LOG_SAVE(logINFO, pLog) << TimeStamp() << " Writing jobtopology to "
94  << (workdir + "/" + pdb_filename) << std::flush;
95  jobtop.WriteToPdb(workdir + "/" + pdb_filename);
96  }
97 
98  int no_static_regions = 0;
99  for (std::unique_ptr<Region>& region : jobtop) {
100  no_static_regions += region->Converged();
101  }
102  bool no_top_scf = false;
103  if (jobtop.size() - no_static_regions < 2) {
104  XTP_LOG_SAVE(logINFO, pLog)
105  << TimeStamp() << " Only " << jobtop.size() - no_static_regions
106  << " scf region is used. The remaining regions are static. So no "
107  "inter regions scf is required. "
108  << std::flush;
109  no_top_scf = true;
110  _max_iterations = 1;
111  }
112  int iteration = 0;
113  for (; iteration < _max_iterations; iteration++) {
114 
115  XTP_LOG_SAVE(logINFO, pLog)
116  << TimeStamp() << " --Inter Region SCF Iteration " << iteration + 1
117  << " of " << _max_iterations << std::flush;
118 
119  for (std::unique_ptr<Region>& region : jobtop) {
120  XTP_LOG_SAVE(logINFO, pLog)
121  << TimeStamp() << " Evaluating " << region->identify() << " "
122  << region->getId() << std::flush;
123  region->Reset();
124  region->Evaluate(jobtop.Regions());
125  if (!region->Successful()) {
126  jres.setStatus(Job::JobStatus::FAILED);
127  jres.setError(region->ErrorMsg());
128  return jres;
129  }
130  }
131 
132  std::string checkpointfilename =
133  "checkpoint_iter_" + std::to_string(iteration + 1) + ".hdf5";
134  XTP_LOG_SAVE(logINFO, pLog) << TimeStamp() << " Writing checkpoint to "
135  << checkpointfilename << std::flush;
136  jobtop.WriteToHdf5(workdir + "/" + checkpointfilename);
137 
138  if (!no_top_scf) {
139  std::vector<bool> converged_regions;
140  for (std::unique_ptr<Region>& region : jobtop) {
141  converged_regions.push_back(region->Converged());
142  }
143 
144  double etot = 0.0;
145  for (const std::unique_ptr<Region>& reg : jobtop) {
146  etot += reg->Etotal();
147  }
148  XTP_LOG_SAVE(logINFO, pLog)
149  << TimeStamp() << " --Total Energy all regions " << etot
150  << std::flush;
151 
152  bool all_regions_converged =
153  std::all_of(converged_regions.begin(), converged_regions.end(),
154  [](bool i) { return i; });
155 
156  if (all_regions_converged) {
157  XTP_LOG_SAVE(logINFO, pLog)
158  << TimeStamp() << " Job converged after " << iteration + 1
159  << " iterations." << std::flush;
160  jres.setStatus(Job::JobStatus::COMPLETE);
161  break;
162  }
163  if (iteration == _max_iterations - 1) {
164  XTP_LOG_SAVE(logINFO, pLog)
165  << TimeStamp() << " Job did not converge after " << iteration + 1
166  << " iterations.\n Writing results to jobfile." << std::flush;
167  jres.setStatus(Job::JobStatus::FAILED);
168  jres.setError("Inter Region SCF did not converge in " +
169  std::to_string(_max_iterations) + " iterations.");
170  }
171  } else {
172  jres.setStatus(Job::JobStatus::COMPLETE);
173  }
174  }
175 
176  tools::Property results;
177  tools::Property& jobresult = results.add("output", "");
178  tools::Property& regionsresults = jobresult.add("regions", "");
179  double etot = 0.0;
180  double charge = 0.0;
181  for (const std::unique_ptr<Region>& reg : jobtop) {
182  reg->AddResults(regionsresults);
183  etot += reg->Etotal();
184  charge += reg->charge();
185  }
186  std::chrono::time_point<std::chrono::system_clock> end =
187  std::chrono::system_clock::now();
188  std::chrono::duration<double> elapsed_time = end - start;
189  jobresult.add("E_tot", std::to_string(etot * tools::conv::hrt2ev));
190  jobresult.add("Compute_Time", std::to_string(int(elapsed_time.count())));
191  jobresult.add("Total_Charge", std::to_string(charge));
192  if (!no_top_scf) {
193  jobresult.add("Iterations", std::to_string(iteration + 1));
194  }
195  jres.setOutput(results);
196  return jres;
197 }

199 bool QMMM::hasQMRegion() const {
200  Logger log;
201  QMRegion QMdummy(0, log, "");
202  bool found_qm = false;
203  std::vector<const tools::Property*> regions_def =
204  _regions_def.Select("region");
205  for (const tools::Property* reg : regions_def) {
206  std::string type =
207  reg->ifExistsReturnElseThrowRuntimeError<std::string>("type");

209  found_qm = true;
210  break;
211  }
212  }

216 void QMMM::WriteJobFile(const Topology& top) {
217 
218  if (!_write_parse) {
219  throw std::runtime_error(
220  "Cannot write jobfile, please add <write_parse><states>e "
221  "s1</states></write_parse> to your options.");
222  }
223 
224  std::cout << std::endl
225  << "... ... Writing job file " << _jobfile << std::flush;
226 
227  bool hasQMRegion = this->hasQMRegion();
228 
229  std::ofstream ofs;
230  ofs.open(_jobfile, std::ofstream::out);
231  if (!ofs.is_open())
232  throw std::runtime_error("\nERROR: bad file handle: " + _jobfile);
233 
234  ofs << "<jobs>" << std::endl;
235  int jobid = 0;
236  for (const Segment& seg : top.Segments()) {
237  for (const QMState& state : _states) {
238 
239  std::string marker = std::to_string(seg.getId()) + ":" + state.ToString();
240  std::string tag = seg.getType() + "_" + marker;
241 
242  tools::Property Input;
243  tools::Property& pInput = Input.add("input", "");
244  pInput.add("site_energies", marker);
245  tools::Property& regions = pInput.add("regions", "");
246  tools::Property& region = regions.add("region", "");
247  region.add("id", "0");
248  if (hasQMRegion) {
249  region.add("state", state.ToString());
250  }
251  region.add("segments", marker);
252  Job job(jobid, tag, Input, Job::AVAILABLE);
253  job.ToStream(ofs);
254  jobid++;
255  }
256  }
257 
258  ofs << "</jobs>" << std::endl;
259  ofs.close();
260  std::cout << std::endl
261  << "... ... In total " << jobid + 1 << " jobs" << std::flush;
262  return;
263 }
264 void QMMM::ReadJobFile(Topology& top) {
265 
266  if (!_write_parse) {
267  throw std::runtime_error(
268  "Cannot read jobfile, please add <write_parse><states>n e "
269  "h</states></write_parse> to your options.");
270  }
271 
272  int incomplete_jobs = 0;
273 
274  Eigen::Matrix<double, Eigen::Dynamic, 5> energies =
275  Eigen::Matrix<double, Eigen::Dynamic, 5>::Zero(top.Segments().size(), 5);
276  Eigen::Matrix<int, Eigen::Dynamic, 5> found =
277  Eigen::Matrix<int, Eigen::Dynamic, 5>::Zero(top.Segments().size(), 5);
278 
279  tools::Property xml;
280  xml.LoadFromXML(_jobfile);
281  std::vector<tools::Property*> jobProps = xml.Select("jobs.job");
282  for (tools::Property* job : jobProps) {
283 
284  int jobid = job->get("id").as<int>();
285  if (!job->exists("status")) {
286  throw std::runtime_error(
287  "Jobfile is malformed. <status> tag missing for job " +
288  std::to_string(jobid));
289  }
290  if (job->get("status").as<std::string>() != "COMPLETE" ||
291  !job->exists("output")) {
292  incomplete_jobs++;
293  continue;
294  }
295 
296  std::string marker = job->get("input.site_energies").as<std::string>();
297  tools::Tokenizer tok(marker, ":");
298  std::vector<std::string> split = tok.ToVector();
299  int segid = std::stoi(split[0]);
300  if (segid < 0 || segid >= int(top.Segments().size())) {
301  throw std::runtime_error("JobSegment id" + std::to_string(segid) +
302  " is not in topology for job " +
303  std::to_string(jobid));
304  }
305  QMState state;
306  try {
307  state.FromString(split[1]);
308  } catch (std::runtime_error& e) {
309  std::stringstream message;
310  message << e.what() << " for job " << jobid;
311  throw std::runtime_error(message.str());
312  }
313  double energy = job->get("output.E_tot").as<double>() * tools::conv::ev2hrt;
314  if (found(segid, state.Type().Type()) != 0) {
315  throw std::runtime_error("There are two entries in jobfile for segment " +
316  std::to_string(segid) +
317  " state:" + state.ToString());
318  }
319 
320  energies(segid, state.Type().Type()) = energy;
321  found(segid, state.Type().Type()) = 1;
322  }
323 
324  Eigen::Matrix<int, 1, 5> found_states = found.colwise().sum();
325  std::cout << std::endl;
326  for (int i = 0; i < 5; i++) {
327  if (found_states(i) > 0) {
328  QMStateType type(static_cast<QMStateType::statetype>(i));
329  std::cout << "Found " << found_states(i) << " states of type "
330  << type.ToString() << std::endl;
331  }
332  }
333  if (incomplete_jobs > 0) {
334  std::cout << incomplete_jobs << " incomplete jobs found." << std::endl;
335  }
336 
337  for (Segment& seg : top.Segments()) {
338  int segid = seg.getId();
339  for (int i = 0; i < 4; i++) {
340  QMStateType type(static_cast<QMStateType::statetype>(i));
341  if (found(segid, i) && found(segid, 4)) {
342  double energy = energies(segid, i) - energies(segid, 4);
343  seg.setEMpoles(type, energy);
344  }
345  }
346  }
347 }
