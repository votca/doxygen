29 void QMMM::Initialize(const tools::Property& user_options) {
30 
31  tools::Property options =
32  LoadDefaultsAndUpdateWithUserOptions("xtp", user_options);
33  ParseCommonOptions(options);
34 
35  _jobfile = options.get(".job_file").as<std::string>();
36  _print_regions_pdb = options.get(".print_regions_pdb").as<bool>();
37  _max_iterations = options.get(".max_iterations").as<Index>();
38  _regions_def = options.get(".regions");
39  _regions_def.add("mapfile", _mapfile);
40 
41  std::string states = options.get(".write_parse.states").as<std::string>();
42  tools::Tokenizer tok(states, " ,;\n\t");
43  std::vector<std::string> statestrings = tok.ToVector();
44  _states.reserve(statestrings.size());
45  for (std::string s : statestrings) {
46  _states.push_back(QMState(s));
47  }
48  bool groundstate_found = false;
49  for (const QMState& state : _states) {
50  if (state.Type() == QMStateType::Gstate) {
51  groundstate_found = true;
52  }
53  }
54  if (!groundstate_found) {
55  _states.push_back(QMState("n"));
56  }
57 }

59 Job::JobResult QMMM::EvalJob(const Topology& top, Job& job, QMThread& Thread) {
60  std::chrono::time_point<std::chrono::system_clock> start =
61  std::chrono::system_clock::now();
62 
63  std::string qmmm_work_dir = "QMMM";
64  if (!this->hasQMRegion()) {
65  qmmm_work_dir = "MMMM";
66  }
67  std::string frame_dir =
68  "frame_" + boost::lexical_cast<std::string>(top.getStep());
69  std::string job_dir =
70  "job_" + std::to_string(job.getId()) + "_" + job.getTag();
71  boost::filesystem::path arg_path;
72  std::string workdir =
73  (arg_path / qmmm_work_dir / frame_dir / job_dir).generic_string();
74  boost::filesystem::create_directories(workdir);
75  Job::JobResult jres = Job::JobResult();
76  Logger& pLog = Thread.getLogger();
77 
78  JobTopology jobtop = JobTopology(job, pLog, workdir);
79  jobtop.BuildRegions(top, _regions_def);
80 
81  if (_print_regions_pdb) {
82  std::string pdb_filename = "regions.pdb";
83  XTP_LOG(Log::error, pLog) << TimeStamp() << " Writing jobtopology to "
84  << (workdir + "/" + pdb_filename) << std::flush;
85  jobtop.WriteToPdb(workdir + "/" + pdb_filename);
86  }
87 
88  Index no_static_regions = 0;
89  for (std::unique_ptr<Region>& region : jobtop) {
90  no_static_regions += region->Converged();
91  }
92  bool no_top_scf = false;
93  if (jobtop.size() - no_static_regions < 2) {
94  XTP_LOG(Log::error, pLog)
95  << TimeStamp() << " Only " << jobtop.size() - no_static_regions
96  << " scf region is used. The remaining regions are static. So no "
97  "inter regions scf is required. "
98  << std::flush;
99  no_top_scf = true;
100  _max_iterations = 1;
101  }
102  Index iteration = 0;
103  for (; iteration < _max_iterations; iteration++) {
104 
105  XTP_LOG(Log::error, pLog)
106  << TimeStamp() << " --Inter Region SCF Iteration " << iteration + 1
107  << " of " << _max_iterations << std::flush;
108 
109  for (std::unique_ptr<Region>& region : jobtop) {
110  XTP_LOG(Log::error, pLog)
111  << TimeStamp() << " Evaluating " << region->identify() << " "
112  << region->getId() << std::flush;
113  region->Reset();
114  region->Evaluate(jobtop.Regions());
115  if (!region->Successful()) {
116  jres.setStatus(Job::JobStatus::FAILED);
117  jres.setError(region->ErrorMsg());
118  return jres;
119  }
120  }
121 
122  std::string checkpointfilename =
123  "checkpoint_iter_" + std::to_string(iteration + 1) + ".hdf5";
124  XTP_LOG(Log::error, pLog) << TimeStamp() << " Writing checkpoint to "
125  << checkpointfilename << std::flush;
126  jobtop.WriteToHdf5(workdir + "/" + checkpointfilename);
127 
128  if (!no_top_scf) {
129  std::vector<bool> converged_regions;
130  for (std::unique_ptr<Region>& region : jobtop) {
131  converged_regions.push_back(region->Converged());
132  }
133 
134  double etot = 0.0;
135  for (const std::unique_ptr<Region>& reg : jobtop) {
136  etot += reg->Etotal();
137  }
138  XTP_LOG(Log::error, pLog) << TimeStamp() << " --Total Energy all regions "
139  << etot << std::flush;
140 
141  bool all_regions_converged =
142  std::all_of(converged_regions.begin(), converged_regions.end(),
143  [](bool i) { return i; });
144 
145  if (all_regions_converged) {
146  XTP_LOG(Log::error, pLog)
147  << TimeStamp() << " Job converged after " << iteration + 1
148  << " iterations." << std::flush;
149  jres.setStatus(Job::JobStatus::COMPLETE);
150  break;
151  }
152  if (iteration == _max_iterations - 1) {
153  XTP_LOG(Log::error, pLog)
154  << TimeStamp() << " Job did not converge after " << iteration + 1
155  << " iterations.\n Writing results to jobfile." << std::flush;
156  jres.setStatus(Job::JobStatus::FAILED);
157  jres.setError("Inter Region SCF did not converge in " +
158  std::to_string(_max_iterations) + " iterations.");
159  }
160  } else {
161  jres.setStatus(Job::JobStatus::COMPLETE);
162  }
163  }
164 
165  tools::Property results;
166  tools::Property& jobresult = results.add("output", "");
167  tools::Property& regionsresults = jobresult.add("regions", "");
168  double etot = 0.0;
169  double charge = 0.0;
170  for (const std::unique_ptr<Region>& reg : jobtop) {
171  reg->AddResults(regionsresults);
172  etot += reg->Etotal();
173  charge += reg->charge();
174  }
175  std::chrono::time_point<std::chrono::system_clock> end =
176  std::chrono::system_clock::now();
177  std::chrono::duration<double> elapsed_time = end - start;
178  jobresult.add("E_tot", std::to_string(etot * tools::conv::hrt2ev));
179  jobresult.add("Compute_Time", std::to_string(Index(elapsed_time.count())));
180  jobresult.add("Total_Charge", std::to_string(charge));
181  if (!no_top_scf) {
182  jobresult.add("Iterations", std::to_string(iteration + 1));
183  }
184  jres.setOutput(results);
185  return jres;
186 }

188 bool QMMM::hasQMRegion() const {
189  Logger log;
190  QMRegion QMdummy(0, log, "");
191  bool found_qm = false;
192  std::vector<const tools::Property*> regions_def =
193  _regions_def.Select("region");
194  for (const tools::Property* reg : regions_def) {
195  std::string type =
196  reg->ifExistsReturnElseThrowRuntimeError<std::string>("type");
197  if (QMdummy.identify() == type) {
198  found_qm = true;
199  break;
200  }
201  }
202  return found_qm;

205 void QMMM::WriteJobFile(const Topology& top) {
206 
207  if (!_write_parse) {
208  throw std::runtime_error(
209  "Cannot write jobfile, please add <write_parse><states>e "
210  "s1</states></write_parse> to your options.");
211  }
212 
213  std::cout << std::endl
214  << "... ... Writing job file " << _jobfile << std::flush;
215 
216  bool hasQMRegion = this->hasQMRegion();
217 
218  std::ofstream ofs;
219  ofs.open(_jobfile, std::ofstream::out);
220  if (!ofs.is_open()) {
221  throw std::runtime_error("\nERROR: bad file handle: " + _jobfile);
222  }
223 
224  ofs << "<jobs>" << std::endl;
225  Index jobid = 0;
226  for (const Segment& seg : top.Segments()) {
227  for (const QMState& state : _states) {
228 
229  std::string marker = std::to_string(seg.getId()) + ":" + state.ToString();
230  std::string tag = seg.getType() + "_" + marker;
231 
232  tools::Property Input;
233  tools::Property& pInput = Input.add("input", "");
234  pInput.add("site_energies", marker);
235  tools::Property& regions = pInput.add("regions", "");
236  tools::Property& region = regions.add("region", "");
237  region.add("id", "0");
238  if (hasQMRegion) {
239  region.add("state", state.ToString());
240  }
241  region.add("segments", marker);
242  Job job(jobid, tag, Input, Job::AVAILABLE);
243  job.ToStream(ofs);
244  jobid++;
245  }
246  }
247 
248  ofs << "</jobs>" << std::endl;
249  ofs.close();
250  std::cout << std::endl
251  << "... ... In total " << jobid + 1 << " jobs" << std::flush;
252  return;

254 void QMMM::ReadJobFile(Topology& top) {
255 
256  if (!_write_parse) {
257  throw std::runtime_error(
258  "Cannot read jobfile, please add <write_parse><states>n e "
259  "h</states></write_parse> to your options.");
260  }
261 
262  Index incomplete_jobs = 0;
263 
264  Eigen::Matrix<double, Eigen::Dynamic, 5> energies =
265  Eigen::Matrix<double, Eigen::Dynamic, 5>::Zero(top.Segments().size(), 5);
266  Eigen::Matrix<int, Eigen::Dynamic, 5> found =
267  Eigen::Matrix<int, Eigen::Dynamic, 5>::Zero(top.Segments().size(), 5);
268 
269  tools::Property xml;
270  xml.LoadFromXML(_jobfile);
271  std::vector<tools::Property*> jobProps = xml.Select("jobs.job");
272  for (tools::Property* job : jobProps) {
273 
274  Index jobid = job->get("id").as<Index>();
275  if (!job->exists("status")) {
276  throw std::runtime_error(
277  "Jobfile is malformed. <status> tag missing for job " +
278  std::to_string(jobid));
279  }
280  if (job->get("status").as<std::string>() != "COMPLETE" ||
281  !job->exists("output")) {
282  incomplete_jobs++;
283  continue;
284  }
285 
286  std::string marker = job->get("input.site_energies").as<std::string>();
287  tools::Tokenizer tok(marker, ":");
288  std::vector<std::string> split = tok.ToVector();
289  Index segid = std::stoi(split[0]);
290  if (segid < 0 || segid >= Index(top.Segments().size())) {
291  throw std::runtime_error("JobSegment id" + std::to_string(segid) +
292  " is not in topology for job " +
293  std::to_string(jobid));
294  }
295  QMState state;
296  try {
297  state.FromString(split[1]);
298  } catch (std::runtime_error& e) {
299  std::stringstream message;
300  message << e.what() << " for job " << jobid;
301  throw std::runtime_error(message.str());
302  }
303  double energy = job->get("output.E_tot").as<double>() * tools::conv::ev2hrt;
304  if (found(segid, state.Type().Type()) != 0) {
305  throw std::runtime_error("There are two entries in jobfile for segment " +
306  std::to_string(segid) +
307  " state:" + state.ToString());
308  }
309 
310  energies(segid, state.Type().Type()) = energy;
311  found(segid, state.Type().Type()) = 1;
312  }
313 
314  Eigen::Matrix<int, 1, 5> found_states = found.colwise().sum();
315  std::cout << std::endl;
316  for (Index i = 0; i < 5; i++) {
317  if (found_states(i) > 0) {
318  QMStateType type(static_cast<QMStateType::statetype>(i));
319  std::cout << "Found " << found_states(i) << " states of type "
320  << type.ToString() << std::endl;
321  }
322  }
323  if (incomplete_jobs > 0) {
324  std::cout << incomplete_jobs << " incomplete jobs found." << std::endl;
325  }
326 
327  for (Segment& seg : top.Segments()) {
328  Index segid = seg.getId();
329  for (Index i = 0; i < 4; i++) {
330  QMStateType type(static_cast<QMStateType::statetype>(i));
331  if (found(segid, i) && found(segid, 4)) {
332  double energy = energies(segid, i) - energies(segid, 4);
333  seg.setEMpoles(type, energy);
334  }
335  }
336  }
337 }
