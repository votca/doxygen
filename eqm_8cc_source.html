47  void EQM::ParseOptionsXML(Property* options) {
48 
49  _maverick = (_nThreads == 1) ? true : false;
50  string key = "options." + Identify();
51  // job tasks
52  string _tasks_string = options->get(key + ".tasks").as<string> ();
53  if (_tasks_string.find("input") != std::string::npos) _do_dft_input = true;
54  if (_tasks_string.find("dft") != std::string::npos) _do_dft_run = true;
55  if (_tasks_string.find("parse") != std::string::npos) _do_dft_parse = true;
56  if (_tasks_string.find("gwbse") != std::string::npos) _do_gwbse = true;
57  if (_tasks_string.find("esp") != std::string::npos) _do_esp = true;
58 
59  key = "options." + Identify();
60 
61  if (options->exists(key + ".job_file")) {
62  _jobfile = options->get(key + ".job_file").as<string>();
63  } else {
64  throw std::runtime_error("Job-file not set. Abort.");
65  }
66 
67  // options for gwbse
68  key = "options." + Identify();
69  string _gwbse_xml = options->get(key + ".gwbse_options").as<string> ();
70  load_property_from_xml(_gwbse_options, _gwbse_xml.c_str());
71 
72  // options for dft package
73  string _package_xml = options->get(key + ".dftpackage").as<string> ();
74  load_property_from_xml(_package_options, _package_xml.c_str());
75  key = "package";
76  _package = _package_options.get(key + ".name").as<string> ();
77 
78  //options for esp/partialcharges
79  if (_do_esp) {
80  key = "options." + Identify();
81  string _esp_xml = options->get(key + ".esp_options").as<string> ();
82  load_property_from_xml(_esp_options, _esp_xml.c_str());
83  }
84 
85  }

87  void EQM::WriteJobFile(ctp::Topology *top) {
88 
89  cout << endl << "... ... Writing job file: " << flush;
90  std::ofstream ofs;
91  ofs.open(_jobfile.c_str(), std::ofstream::out);
92  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
93  ofs << "<jobs>" << endl;
94  int jobCount = 0;
95 
96  std::vector<ctp::Segment*> segments = top->Segments();
97  for (ctp::Segment* segment : segments) {
98  int id = ++jobCount;
99  string tag = "";
100  Property Input;
101  Property &pInput = Input.add("input", "");
102  Property &pSegment = pInput.add("segment", (format("%1$s") % segment->getId()).str());
103  pSegment.setAttribute<string>("type", segment->getName());
104  pSegment.setAttribute<int>("id", segment->getId());
105  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE);
106  job.ToStream(ofs, "xml");
107  }
108 
109  ofs << "</jobs>" << endl;
110  ofs.close();
111 
112  cout << jobCount << " jobs" << flush;
113 
114  }

134  ctp::Job::JobResult EQM::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *opThread) {
135 
136  Orbitals orbitals;
137  ctp::Job::JobResult jres = ctp::Job::JobResult();
138  Property _job_input = job->getInput();
139  list<Property*> lSegments = _job_input.Select("segment");
140  vector < ctp::Segment* > segments;
141  int segId = lSegments.front()->getAttribute<int>("id");
142  string segType = lSegments.front()->getAttribute<string>("type");
143  ctp::Segment *seg = top->getSegment(segId);
144  segments.push_back(seg);
145  QMInterface interface;
146  orbitals.QMAtoms() = interface.Convert(segments);
147 
148  ctp::Logger* pLog = opThread->getLogger();
149 
150  CTP_LOG(ctp::logINFO, *pLog) << ctp::TimeStamp() << " Evaluating site " << seg->getId() << flush;
151 
152  // directories and files
153  boost::filesystem::path arg_path;
154  string eqm_work_dir = "OR_FILES";
155  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
156  string orb_file = (format("%1%_%2%%3%") % "molecule" % segId % ".orb").str();
157  string mol_dir = (format("%1%%2%%3%") % "molecule" % "_" % segId).str();
158  string package_append = _package + "_"+Identify();
159  string work_dir = (arg_path / eqm_work_dir / package_append / frame_dir / mol_dir).c_str();
160 
161  Property job_summary;
162  Property &output_summary = job_summary.add("output", "");
163  Property &segment_summary = output_summary.add("segment", "");
164  string segName = seg->getName();
165  segId = seg->getId();
166  segment_summary.setAttribute("id", segId);
167  segment_summary.setAttribute("type", segName);
168  if (_do_dft_input || _do_dft_run || _do_dft_parse) {
169  CTP_LOG(ctp::logDEBUG, *pLog) << "Running DFT" << flush;
170  ctp::Logger dft_logger(ctp::logDEBUG);
171  dft_logger.setMultithreading(false);
172  dft_logger.setPreface(ctp::logINFO, (format("\nDFT INF ...")).str());
173  dft_logger.setPreface(ctp::logERROR, (format("\nDFT ERR ...")).str());
174  dft_logger.setPreface(ctp::logWARNING, (format("\nDFT WAR ...")).str());
175  dft_logger.setPreface(ctp::logDEBUG, (format("\nDFT DBG ...")).str());
176 
177  QMPackage *qmpackage = QMPackages().Create(_package);
178  qmpackage->setLog(&dft_logger);
179  qmpackage->setRunDir(work_dir);
180  qmpackage->Initialize(_package_options);
181 
182  // create input for DFT
183  if (_do_dft_input) {
184  boost::filesystem::create_directories(work_dir);
185  qmpackage->WriteInputFile(orbitals);
186  }
187 
188  bool run_dft_status = false;
189  if (_do_dft_run) {
190  run_dft_status = qmpackage->Run();
191  if (!run_dft_status) {
192  string output = "DFT run failed";
193  SetJobToFailed(jres, pLog, output);
194  delete qmpackage;
195  return jres;
196  }
197  }
198 
199  // parse the log/orbitals files
200  bool parse_log_status = false;
201  bool parse_orbitals_status = false;
202  if (_do_dft_parse) {
203  parse_log_status = qmpackage->ParseLogFile(orbitals);
204  if (!parse_log_status) {
205  string output = "log incomplete; ";
206  SetJobToFailed(jres, pLog, output);
207  delete qmpackage;
208  return jres;
209  }
210  parse_orbitals_status = qmpackage->ParseOrbitalsFile(orbitals);
211  if (!parse_orbitals_status) {
212  string output = "orbfile failed; ";
213  SetJobToFailed(jres, pLog, output);
214  delete qmpackage;
215  return jres;
216  }
217  }// end of the parse orbitals/log
218  qmpackage->CleanUp();
219  delete qmpackage;
220  WriteLoggerToFile(work_dir + "/dft.log",dft_logger);
221  }
222 
223  if (!_do_dft_parse) {
224  // load the DFT data from serialized orbitals object
225  string ORB_FILE =eqm_work_dir + "/molecules/" + frame_dir+ "/" + orb_file;
226  CTP_LOG(ctp::logDEBUG, *pLog) << ctp::TimeStamp() << " Loading DFT data from " << ORB_FILE << flush;
227  orbitals.ReadFromCpt(ORB_FILE);
228  }
229 
230  if (_do_gwbse) {
231  CTP_LOG(ctp::logDEBUG, *pLog) << "Running GWBSE" << flush;
232  try {
233  GWBSE gwbse = GWBSE(orbitals);
234  ctp::Logger gwbse_logger(ctp::logDEBUG);
235  gwbse_logger.setMultithreading(false);
236  gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
237  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
238  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
239  gwbse_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
240  gwbse.setLogger(&gwbse_logger);
241  gwbse.Initialize(_gwbse_options);
242  gwbse.Evaluate();
243  gwbse.addoutput(segment_summary);
244  WriteLoggerToFile(work_dir + "/gwbse.log", gwbse_logger);
245  }catch (std::runtime_error& error) {
246  std::string errormessage(error.what());
247  SetJobToFailed(jres, pLog, "GWBSE:"+errormessage);
248  return jres;
249  }
250  }
251 
252  if (_do_esp) {
253  CTP_LOG(ctp::logDEBUG, *pLog) << "Running ESPFIT" << flush;
254  try {
255  string mps_file = "";
256  Esp2multipole esp2multipole = Esp2multipole(pLog);
257  esp2multipole.Initialize(_esp_options);
258  string ESPDIR = "MP_FILES/" + frame_dir + "/" + esp2multipole.GetStateString();
259  esp2multipole.Extractingcharges(orbitals);
260  mps_file = (format("%1%_%2%_%3%.mps") % segType % segId % esp2multipole.GetStateString()).str();
261  boost::filesystem::create_directories(ESPDIR);
262  esp2multipole.WritetoFile(ESPDIR + "/" + mps_file);
263  CTP_LOG(ctp::logDEBUG, *pLog) << "Written charges to " << (ESPDIR + "/" + mps_file).c_str() << flush;
264  segment_summary.add("partialcharges", (ESPDIR + "/" + mps_file).c_str());
265  }catch (std::runtime_error& error) {
266  std::string errormessage(error.what());
267  SetJobToFailed(jres, pLog, "ESPFIT:"+errormessage);
268  return jres;
269  }
270  }
271  CTP_LOG(ctp::logINFO, *pLog) << ctp::TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
272 
273  if (_do_dft_parse || _do_gwbse) {
274  CTP_LOG(ctp::logDEBUG, *pLog) << "Saving data to " << orb_file << flush;
275  string DIR = eqm_work_dir + "/molecules/" + frame_dir;
276  boost::filesystem::create_directories(DIR);
277  string ORBFILE = DIR + "/" + orb_file;
278  orbitals.WriteToCpt(ORBFILE);
279  }
280 
281  // output of the JOB
282  jres.setOutput(job_summary);
283  jres.setStatus(ctp::Job::COMPLETE);
284 
285  // dump the LOG
286 
287  return jres;
288  }
289 
290  }
