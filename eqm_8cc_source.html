38 void EQM::ParseSpecificOptions(const tools::Property& options) {
39 
40  QMPackageFactory::RegisterAll();
41 
42  // job tasks
43  std::string tasks_string_ = options.get(".tasks").as<std::string>();
44 
45  if (tasks_string_.find("input") != std::string::npos) {
46  do_dft_input_ = true;
47  }
48  if (tasks_string_.find("dft") != std::string::npos) {
49  do_dft_run_ = true;
50  }
51  if (tasks_string_.find("parse") != std::string::npos) {
52  do_dft_parse_ = true;
53  }
54  if (tasks_string_.find("gwbse") != std::string::npos) {
55  do_gwbse_ = true;
56  }
57  if (tasks_string_.find("esp") != std::string::npos) {
58  do_esp_ = true;
59  }
60 
61  // set the basis sets and functional for DFT and GWBSE
62  gwbse_options_ = this->UpdateGWBSEOptions(options);
63  package_options_ = this->UpdateDFTOptions(options);
64  esp_options_ = options.get(".esp_options");
65 }

67 void EQM::WriteJobFile(const Topology& top) {
68 
69  std::cout << "\n... ... Writing job file: " << jobfile_ << std::flush;
70  std::ofstream ofs;
71  ofs.open(jobfile_, std::ofstream::out);
72  if (!ofs.is_open()) {
73  throw std::runtime_error("\nERROR: bad file handle: " + jobfile_);
74  }
75  ofs << "<jobs>" << std::endl;
76  Index jobCount = 0;
77 
78  const std::vector<Segment>& segments = top.Segments();
79  for (const Segment& segment : segments) {
80  Index id = segment.getId();
81  std::string tag = "";
82  tools::Property Input;
83  tools::Property& pInput = Input.add("input", "");
84  tools::Property& pSegment =
85  pInput.add("segment", (format("%1$s") % segment.getId()).str());
86  pSegment.setAttribute<std::string>("type", segment.getType());
87  pSegment.setAttribute<Index>("id", segment.getId());
88  Job job(id, tag, Input, Job::AVAILABLE);
89  job.ToStream(ofs);
90  jobCount++;
91  }
92 
93  ofs << "</jobs>" << std::endl;
94  ofs.close();
95 
96  std::cout << " with " << jobCount << " jobs" << std::flush;
97 }

117 Job::JobResult EQM::EvalJob(const Topology& top, Job& job, QMThread& opThread) {
118  Orbitals orbitals;
119  Job::JobResult jres = Job::JobResult();
120  tools::Property job_input_ = job.getInput();
121  std::vector<tools::Property*> lSegments = job_input_.Select("segment");
122  Index segId = lSegments.front()->getAttribute<Index>("id");
123  std::string segType = lSegments.front()->getAttribute<std::string>("type");
124  std::string qmgeo_state = "n";
125  if (lSegments.front()->exists("qm_geometry")) {
126  qmgeo_state = lSegments.front()->getAttribute<std::string>("qm_geometry");
127  }
128 
129  QMState state(qmgeo_state);
130  const Segment& seg = top.getSegment(segId);
131 
132  Logger& pLog = opThread.getLogger();
133  QMMapper mapper(pLog);
134  mapper.LoadMappingFile(mapfile_);
135  orbitals.QMAtoms() = mapper.map(seg, state);
136  XTP_LOG(Log::error, pLog)
137  << TimeStamp() << " Evaluating site " << seg.getId() << std::flush;
138 
139  // directories and files
140  boost::filesystem::path arg_path;
141  std::string eqm_work_dir = "OR_FILES";
142  std::string frame_dir =
143  "frame_" + boost::lexical_cast<std::string>(top.getStep());
144  std::string orb_file =
145  (format("%1%_%2%%3%") % "molecule" % segId % ".orb").str();
146  std::string mol_dir = (format("%1%%2%%3%") % "molecule" % "_" % segId).str();
147  std::string package_append = "workdir_" + Identify();
148  std::string work_dir =
149  (arg_path / eqm_work_dir / package_append / frame_dir / mol_dir)
150  .generic_string();
151 
152  tools::Property job_summary;
153  tools::Property& output_summary = job_summary.add("output", "");
154  tools::Property& segment_summary = output_summary.add("segment", "");
155  std::string segName = seg.getType();
156  segId = seg.getId();
157  segment_summary.setAttribute("id", segId);
158  segment_summary.setAttribute("type", segName);
159  if (do_dft_input_ || do_dft_run_ || do_dft_parse_) {
160  XTP_LOG(Log::error, pLog) << "Running DFT" << std::flush;
161  Logger dft_logger(votca::Log::current_level);
162  dft_logger.setMultithreading(false);
163  dft_logger.setPreface(Log::info, (format("\nDFT INF ...")).str());
164  dft_logger.setPreface(Log::error, (format("\nDFT ERR ...")).str());
165  dft_logger.setPreface(Log::warning, (format("\nDFT WAR ...")).str());
166  dft_logger.setPreface(Log::debug, (format("\nDFT DBG ...")).str());
167  std::string dft_key = "package";
168  std::string package =
169  package_options_.get(dft_key + ".name").as<std::string>();
170  std::unique_ptr<QMPackage> qmpackage = std::unique_ptr<QMPackage>(
171  QMPackageFactory::QMPackages().Create(package));
172  qmpackage->setLog(&dft_logger);
173  qmpackage->setRunDir(work_dir);
174  qmpackage->Initialize(package_options_);
175 
176  // create input for DFT
177  if (do_dft_input_) {
178  boost::filesystem::create_directories(work_dir);
179  qmpackage->WriteInputFile(orbitals);
180  }
181 
182  if (do_dft_run_) {
183  bool run_dft_status = qmpackage->Run();
184  if (!run_dft_status) {
185  std::string output = "DFT run failed";
186  SetJobToFailed(jres, pLog, output);
187  return jres;
188  }
189  }
190 
191  // parse the log/orbitals files
192  if (do_dft_parse_) {
193  bool parse_log_status = qmpackage->ParseLogFile(orbitals);
194  if (!parse_log_status) {
195  std::string output = "log incomplete; ";
196  SetJobToFailed(jres, pLog, output);
197  return jres;
198  }
199  bool parse_orbitals_status = qmpackage->ParseMOsFile(orbitals);
200  if (!parse_orbitals_status) {
201  std::string output = "orbfile failed; ";
202  SetJobToFailed(jres, pLog, output);
203  return jres;
204  }
205  // additionally copy *.gbw files for orca (-> iqm guess)
206  if (qmpackage->getPackageName() == "orca") {
207  std::string DIR = eqm_work_dir + "/molecules/" + frame_dir;
208  boost::filesystem::create_directories(DIR);
209  std::string gbw_file =
210  (format("%1%_%2%%3%") % "molecule" % segId % ".gbw").str();
211  std::string GBWFILE = DIR + "/" + gbw_file;
212  XTP_LOG(Log::error, pLog)
213  << "Copying MO data to " << gbw_file << std::flush;
214  std::string GBWFILE_workdir = work_dir + "/system.gbw";
215  boost::filesystem::copy_file(
216  GBWFILE_workdir, GBWFILE,
217  boost::filesystem::copy_option::overwrite_if_exists);
218  }
219 
220  } // end of the parse orbitals/log
221  qmpackage->CleanUp();
222  WriteLoggerToFile(work_dir + "/dft.log", dft_logger);
223  }
224 
225  if (!do_dft_parse_ && (do_gwbse_ || do_esp_)) {
226  // load the DFT data from serialized orbitals object
227  std::string ORB_FILE =
228  eqm_work_dir + "/molecules/" + frame_dir + "/" + orb_file;
229  XTP_LOG(Log::error, pLog)
230  << TimeStamp() << " Loading DFT data from " << ORB_FILE << std::flush;
231  orbitals.ReadFromCpt(ORB_FILE);
232  }
233 
234  if (do_gwbse_) {
235  XTP_LOG(Log::error, pLog) << "Running GWBSE" << std::flush;
236  try {
237  GWBSE gwbse = GWBSE(orbitals);
238  Logger gwbse_logger(votca::Log::current_level);
239  gwbse_logger.setMultithreading(false);
240  gwbse_logger.setPreface(Log::info, (format("\nGWBSE INF ...")).str());
241  gwbse_logger.setPreface(Log::error, (format("\nGWBSE ERR ...")).str());
242  gwbse_logger.setPreface(Log::warning, (format("\nGWBSE WAR ...")).str());
243  gwbse_logger.setPreface(Log::debug, (format("\nGWBSE DBG ...")).str());
244  gwbse.setLogger(&gwbse_logger);
245  gwbse.Initialize(gwbse_options_);
246  gwbse.Evaluate();
247  gwbse.addoutput(segment_summary);
248  WriteLoggerToFile(work_dir + "/gwbse.log", gwbse_logger);
249  } catch (std::runtime_error& error) {
250  std::string errormessage(error.what());
251  SetJobToFailed(jres, pLog, "GWBSE:" + errormessage);
252  return jres;
253  }
254  }
255 
256  if (do_esp_) {
257  XTP_LOG(Log::error, pLog) << "Running ESPFIT" << std::flush;
258  try {
259  Esp2multipole esp2multipole = Esp2multipole(pLog);
260  esp2multipole.Initialize(esp_options_);
261  std::string ESPDIR =
262  "MP_FILES/" + frame_dir + "/" + esp2multipole.GetStateString();
263  StaticSegment seg2 = esp2multipole.Extractingcharges(orbitals);
264  std::string mps_file = (format("%1%_%2%_%3%.mps") % segType % segId %
265  esp2multipole.GetStateString())
266  .str();
267  boost::filesystem::create_directories(ESPDIR);
268  seg2.WriteMPS(ESPDIR + "/" + mps_file,
269  "Generated by eqm:" + esp2multipole.GetStateString());
270  XTP_LOG(Log::error, pLog)
271  << "Written charges to " << (ESPDIR + "/" + mps_file) << std::flush;
272  segment_summary.add("partialcharges", (ESPDIR + "/" + mps_file));
273  } catch (std::runtime_error& error) {
274  std::string errormessage(error.what());
275  SetJobToFailed(jres, pLog, "ESPFIT:" + errormessage);
276  return jres;
277  }
278  }
279  XTP_LOG(Log::error, pLog) << TimeStamp() << " Finished evaluating site "
280  << seg.getId() << std::flush;
281 
282  if (do_dft_parse_ || do_gwbse_) {
283  XTP_LOG(Log::error, pLog) << "Saving data to " << orb_file << std::flush;
284  std::string DIR = eqm_work_dir + "/molecules/" + frame_dir;
285  boost::filesystem::create_directories(DIR);
286  std::string ORBFILE = DIR + "/" + orb_file;
287  orbitals.WriteToCpt(ORBFILE);
288  }
289 
290  // output of the JOB
291  jres.setOutput(job_summary);
292  jres.setStatus(Job::COMPLETE);
293 
294  return jres;
295 }
