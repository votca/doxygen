38  template<class QMPackage>
39  QMMachine<QMPackage>::QMMachine(ctp::XJob *job, ctp::XInductor *xind, QMPackage *qmpack,
40  Property *opt, string sfx, int nst, bool mav)
41  : _job(job), _xind(xind), _qmpack(qmpack), _subthreads(nst),
42  _isConverged(false) {
43 
44  string key = sfx + ".qmmmconvg";
45 
46  _crit_dR = opt->ifExistsReturnElseReturnDefault<double>(key + ".dR", 0.01); //nm
47  _crit_dQ = opt->ifExistsReturnElseReturnDefault<double>(key + ".dQ", 0.01); //e
48  _crit_dE_QM = opt->ifExistsReturnElseReturnDefault<double>(key + ".dQdE_QM", 0.001); //eV
49  _crit_dE_MM = opt->ifExistsReturnElseReturnDefault<double>(key + ".dE_MM", _crit_dE_QM); //eV
50  _maxIter = opt->ifExistsReturnElseReturnDefault<int>(key + ".max_iter", 32);
51 
52 
53  key = sfx;
54  bool split_dpl = opt->ifExistsReturnElseReturnDefault<bool>(key + ".split_dpl", true);
55  double dpl_spacing = opt->ifExistsReturnElseReturnDefault<double>(key + ".dpl_spacing", 1e-3);
56  qminterface.setMultipoleSplitting(split_dpl, dpl_spacing);
57 
58  // check for archiving
59  std::string _archiving_string = opt->ifExistsReturnElseReturnDefault<std::string>(key + ".archiving", "");
60  if (_archiving_string.find("iterations") != std::string::npos) {
61  _do_archive = true;
62  } else {
63  _do_archive = false;
64  }
65 
66  // check for static or polarized qmmm
67  key = sfx + ".tholemodel";
68  _static_qmmm = true;
69  if (opt->exists(key + ".induce")) {
70  int induce = opt->get(key + ".induce").as< int >();
71  _static_qmmm = (induce == 1) ? false : true;
72  }
73 
74 
75  // GDMA options
76  key = sfx + ".gdma";
77  if (opt->exists(key)) {
78  _do_gdma = true;
79  string _gdma_xml = opt->get(key).as<string> ();
80  //cout << endl << "... ... Parsing " << _package_xml << endl ;
81  load_property_from_xml(_gdma_options, _gdma_xml.c_str());
82  } else {
83  _do_gdma = false;
84  }
85 
86 
87  // GWBSE options
88  key = sfx + ".gwbse";
89  if (opt->exists(key)) {
90  _do_gwbse = true;
91 
92  // PUT IN some useful type definitions for excited state
93  // - singlet -> singlet exciton
94  // - triplet -> triplet exciton
95  // - quasiparticle -> GW quasiparticle (diagqp)
96  // - kohn-sham -> DFT MO
97 
98  string _gwbse_xml = opt->get(key + ".gwbse_options").as<string> ();
99  //cout << endl << "... ... Parsing " << _package_xml << endl ;
100  load_property_from_xml(_gwbse_options, _gwbse_xml.c_str());
101  _state = opt->get(key + ".state").as< int >();
102  _type = opt->get(key + ".type").as< string >();
103  if (_type != "singlet" && _type != "triplet" && _type != "quasiparticle") {
104  throw runtime_error(" Invalid excited state type! " + _type);
105  }
106 
107  key = sfx + ".gwbse.filter";
108  if (opt->exists(key + ".oscillator_strength") && _type != "triplet") {
109  _has_osc_filter = true;
110  _osc_threshold = opt->get(key + ".oscillator_strength").as<double> ();
111  }
112  if (opt->exists(key + ".overlap")) {
113  _has_overlap_filter = true;
114  // _osc_threshold = opt->get(key + ".oscillator_strength").as<double> ();
115  }
116  if (opt->exists(key + ".localisation")) {
117  _has_loc_filter = true;
118 
119  string temp = opt->get(key + ".localisation").as<string> ();
120  Tokenizer tok_cleanup(temp, ", \n\t");
121  std::vector <std::string> strings_vec;
122  tok_cleanup.ToVector(strings_vec);
123  if (strings_vec.size()!=2){
124  throw runtime_error("qmmmachine: Fragment and localisation threshold are not separated");
125  }
126  if(strings_vec[0]=="a" || strings_vec[0]=="A"){
127  _localiseonA=true;
128  }else if(strings_vec[0]=="b" || strings_vec[0]=="B"){
129  _localiseonA=false;
130  }else{
131  throw runtime_error("qmmmachine: Fragment label not known, either A or B");
132  }
133  _loc_threshold=boost::lexical_cast<double>(strings_vec[1]);
134  }
135 
136  if (opt->exists(key + ".charge_transfer")) {
137  _has_dQ_filter = true;
138  _dQ_threshold = opt->get(key + ".charge_transfer").as<double> ();
139  }
140  if(_has_dQ_filter && _has_loc_filter){
141  throw runtime_error("Cannot use localisation and charge_transfer filter at the same time.");
142  }
143  } else {
144  _do_gwbse = false;
145  }
146 
147  return;
148  }

160  template<class QMPackage>
161  int QMMachine<QMPackage>::Evaluate(ctp::XJob *job) {
162 
163  CTP_LOG(ctp::logINFO, *_log)
164  << format("... dR %1$1.4f dQ %2$1.4f QM %3$1.4f MM %4$1.4f IT %5$d")
165  % _crit_dR % _crit_dQ % _crit_dE_QM % _crit_dE_MM % _maxIter << flush;
166 
167  // FIGURE OUT CHARGE + MULTIPLICITY
168  double dQ = 0.0;
169  for (unsigned int i = 0; i < _job->getPolarTop()->QM0().size(); ++i) {
170  dQ += _job->getPolarTop()->QM0()[i]->CalcTotQ();
171  }
172  int chrg = round(dQ);
173  int spin = ((chrg < 0) ? -chrg : chrg) % 2 + 1;
174  CTP_LOG(ctp::logINFO, *_log) << "... Q = " << chrg << ", 2S+1 = " << spin << flush;
175 
176 
177  // PREPARE JOB DIRECTORY
178  string jobFolder = "xjob_" + boost::lexical_cast<string>(_job->getId())
179  + "_" + _job->getTag();
180  bool created = boost::filesystem::create_directory(jobFolder);
181  if (created) {
182  CTP_LOG(ctp::logINFO, *_log) << "Created directory " << jobFolder << flush;
183  }
184 
185 
186  // SET ITERATION-TIME CONSTANTS
187  // TO ADJUST
188 
189  _qmpack->setCharge(chrg);
190  _qmpack->setSpin(spin);
191 
192 
193  int iterCnt = 0;
194  int iterMax = _maxIter;
195  for (; iterCnt < iterMax; ++iterCnt) {
196 
197  // check for polarized QM/MM convergence
198  int info= Iterate(jobFolder, iterCnt);
199  if(info!=0){
200  CTP_LOG(ctp::logERROR, *_log)
201  << format("Iterating job failed!")<<flush;
202  return 1;
203  }
204  if (_static_qmmm) {
205  _isConverged = true;
206  break;
207  } else if (hasConverged()) {
208  break;
209  }
210 
211  }
212 
213  if (iterCnt == iterMax - 1 && !_isConverged) {
214  CTP_LOG(ctp::logWARNING, *_log)
215  << format("Not converged within %1$d iterations.") % iterMax;
216  return 2;
217  }
218 
219  return 0;
220  }

222  template<class QMPackage>
223  bool QMMachine<QMPackage>::Iterate(string jobFolder, int iterCnt) {
224 
225  // CREATE ITERATION OBJECT & SETUP RUN DIRECTORY
226  QMMIter *thisIter = this->CreateNewIter();
227  int iter = iterCnt;
228  string runFolder = jobFolder + "/iter_" + boost::lexical_cast<string>(iter);
229 
230  bool created = boost::filesystem::create_directory(runFolder);
231  if (created)
232  CTP_LOG(ctp::logDEBUG, *_log) << "Created directory " << runFolder << flush;
233  else
234  CTP_LOG(ctp::logWARNING, *_log) << "Could not create directory " << runFolder << flush;
235 
236 
237  // RUN CLASSICAL INDUCTION & SAVE
238  _job->getPolarTop()->PrintPDB(runFolder + "/QM0_MM1_MM2.pdb");
239  _xind->Evaluate(_job);
240 
241 
242 
243  assert(_xind->hasConverged());
244  thisIter->setE_FM(_job->getEF00(), _job->getEF01(), _job->getEF02(),
245  _job->getEF11(), _job->getEF12(), _job->getEM0(),
246  _job->getEM1(), _job->getEM2(), _job->getETOT());
247 
248 
249  std::vector<ctp::Segment*> empty;
250  /* Translate atoms in QM0() to QMAtoms in orbitals object
251  * to be used in writing the QM input files for the
252  * external QMPackages or directly in internal DFT engine.
253  * DRAWBACK: QMAtom positions are fixed for all iterations
254  * unless the UPDATE function at the end of the
255  * iteration updates orb_iter_input (TO BE TESTED)
256  */
257  if (iterCnt == 0) qminterface.GenerateQMAtomsFromPolarSegs(_job->getPolarTop(), orb_iter_input);
258 
259  /* Generate list of polar segments in the MM1() and MM2()
260  * region to be used in writing the background multipoles
261  * in the external QMPackages or in the direct evaluation
262  * in the internal DFT engine.
263  */
264  std::vector<ctp::PolarSeg*> MultipolesBackground = qminterface.GenerateMultipoleList( _job->getPolarTop() );
265 
266  // if XTP DFT is used, pass this list of polar segments
267  if ( _qmpack->getPackageName() == "xtp" ) _qmpack->setMultipoleBackground( MultipolesBackground );
268 
269  // setting RUNDIR for the external QMPackages, dummy for internal
270  _qmpack->setRunDir(runFolder);
271 
272  /* Call to WriteInputFile writes the appropriate input files
273  * for the respective external QMPackages. For the internal
274  * DFT engine, this function sets logger and runs DFTENGINE's
275  * Prepare() function. ONLY the first iteration, this will
276  * initialize the atoms, basissets, ecps, etc. In all
277  * subsequent iterations it will recalculate all the "static"
278  * AOmatrices (overlap, kinetic energy, nuc/ecp) which is
279  * strictly unnecessary but at the same time also those
280  * for external point charges, dipoles, and quadrupoles.
281  * Since in polarized calculations, the dipoles can change
282  * this recalculation is required. Should be split off the
283  * Prepare() function for efficiency.
284  */
285  CTP_LOG(ctp::logDEBUG, *_log) << "Writing input file " << runFolder << flush;
286  _qmpack->WriteInputFile(empty, &orb_iter_input, MultipolesBackground);
287 
288 
289  FILE *out;
290  out = fopen((runFolder + "/system.pdb").c_str(), "w");
291  orb_iter_input.WritePDB(out);
292  fclose(out);
293 
294  /* Runs the external QMPackage or the self-consistent part of
295  * DFTENGINE
296  */
297  _qmpack->Run( &orb_iter_input );
298 
299  /* Parse information from the LOGFILE into orbitals, if
300  * external QMPackage is run. Dummy for internal DFTENGINE.
301  * The QMPackage's MOcoefficients are not automatically
302  * parsed in DFT-only calculations and ESP fits for
303  * polarized QMMM are expected in the LOGFILE.
304  * IF the internal ESPFITs should be used, the MOcoefficients
305  * need to be parsed too.
306  */
307  bool success=_qmpack->ParseLogFile(&orb_iter_input);
308  if(!success){
309  return 1;
310  }
311 
312  ub::matrix<double> DMAT_tot;
313  // GW-BSE starts here
314  double energy___ex = 0.0;
315  std::vector<int> _state_index;
316 
317  if (_do_gwbse) {
318 
319  /* Parses the MOcoefficients from the external QMPackages
320  * for GW-BSE. Internal DFTENGINE has stored coefficients
321  * into orb_iter_input already, so this is a dummy for that.
322  */
323  _qmpack->ParseOrbitalsFile(&orb_iter_input);
324  orb_iter_input.setDFTbasis(_qmpack->getBasisSetName());
325 
326  // Get a GWBSE object
327  GWBSE _gwbse = GWBSE(&orb_iter_input);
328  // define own logger for GW-BSE that is written into a runFolder logfile
329  ctp::Logger gwbse_logger(ctp::logDEBUG);
330  gwbse_logger.setMultithreading(false);
331  _gwbse.setLogger(&gwbse_logger);
332  gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
333  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
334  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
335  gwbse_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
336 
337  // Initialize with options
338  _gwbse.Initialize(&_gwbse_options);
339 
340  /* Only execute GWBSE if excited state is requested. This is a bit
341  * weird construction to have ground state calculation treated in
342  * exactly the same way for polarized QMMM.
343  */
344  if (_state > 0) {
345  CTP_LOG(ctp::logDEBUG, *_log) << "Excited state via GWBSE: " << flush;
346  CTP_LOG(ctp::logDEBUG, *_log) << " --- type: " << _type << flush;
347  CTP_LOG(ctp::logDEBUG, *_log) << " --- state: " << _state << flush;
348  if (_has_overlap_filter) {
349  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: overlap " << flush;
350  }
351  if (_has_osc_filter) {
352  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: osc.str. > " << _osc_threshold << flush;
353  }
354  if (_has_dQ_filter) {
355  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: crg.trs. > " << _dQ_threshold << flush;
356  }
357  if (_has_loc_filter){
358  if (_localiseonA){
359  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: localisation on A > " << _loc_threshold << flush;
360  }else{
361  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: localisation on B > " << _loc_threshold << flush;
362  }
363  }
364 
365  if (_has_osc_filter && _has_dQ_filter) {
366  CTP_LOG(ctp::logDEBUG, *_log) << " --- WARNING: filtering for optically active CT transition - might not make sense... " << flush;
367  }
368 
369  // actual GW-BSE run
370  _gwbse.Evaluate();
371 
372  // write logger to log file
373  ofstream ofs;
374  string gwbse_logfile = runFolder + "/gwbse.log";
375  ofs.open(gwbse_logfile.c_str(), ofstream::out);
376  if (!ofs.is_open()) {
377  throw runtime_error("Bad file handle: " + gwbse_logfile);
378  }
379  ofs << gwbse_logger << endl;
380  ofs.close();
381 
382  // PROCESSING the GW-BSE result
383  // - find the excited state of interest
384  // oscillator strength filter
385 
386  // quasiparticle filter
387  if (_has_overlap_filter) {
388  if (iter == 0) {
389 
390  // One - to - One LIST in 0th iteration
391  for (unsigned _i = 0; _i < orb_iter_input.QPdiagEnergies().size(); _i++) {
392  _state_index.push_back(_i);
393  }
394 
395  } else {
396  // get AO overlap matrix
397  AOOverlap _dftoverlap;
398  // load dft basis set (element-wise information) from xml file
399  BasisSet dftbs;
400  dftbs.LoadBasisSet(orb_iter_input.getDFTbasis());
401  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Loaded DFT Basis Set " << orb_iter_input.getDFTbasis() << flush;
402 
403  // fill auxiliary GW AO basis by going through all atoms
404  AOBasis dftbasis;
405  dftbasis.AOBasisFill(&dftbs, orb_iter_input.QMAtoms());
406  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Filled DFT Basis of size " << dftbasis.AOBasisSize() << flush;
407 
408  // Fill overlap
409  _dftoverlap.Fill(dftbasis);
410  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Filled DFT Overlap matrix of dimension: " << _dftoverlap.Matrix().size1() << flush;
411 
412 
413  // 'LAMBDA' matrix of the present iteration
414  ub::matrix<real_gwbse> lambda_N = orb_iter_input.LambdaMatrixQuasiParticle();
415 
416  // 'LAMBDA' matrix of the previous iteration
417  string runFolder_N_1 = jobFolder + "/iter_" + boost::lexical_cast<string>(iter - 1);
418  string orbfile_N_1 = runFolder_N_1 + "/system.orb";
419  Orbitals _orbitals_N_1;
420  // load the QM data from serialized orbitals object
421  std::ifstream ifs((orbfile_N_1).c_str());
422  CTP_LOG(ctp::logDEBUG, *_log) << " Loading QM data from " << orbfile_N_1 << flush;
423  boost::archive::binary_iarchive ia(ifs);
424  ia >> _orbitals_N_1;
425  ifs.close();
426 
427 
428  ub::matrix<real_gwbse> lambda_N_1 = _orbitals_N_1.LambdaMatrixQuasiParticle();
429  // calculate QP overlaps
430  ub::matrix<real_gwbse> qptemp = ub::prod(_dftoverlap.Matrix(), ub::trans(lambda_N_1));
431  ub::matrix<real_gwbse> qpoverlaps = ub::prod(lambda_N, qptemp);
432 
433  // test output
434  if (tools::globals::verbose) {
435  for (unsigned i = 0; i < qpoverlaps.size1(); i++) {
436  for (unsigned j = 0; j < qpoverlaps.size2(); j++) {
437  CTP_LOG(ctp::logDEBUG, *_log) << " [" << i << " , " << j << "]: " << qpoverlaps(i, j) << flush;
438  }
439  }
440  }
441 
442 
443  // filter for max absolute value (hopefully close to 1)
444  for (unsigned _j = 0; _j < qpoverlaps.size2(); _j++) {
445  int maxi = 0;
446  for (unsigned _i = 0; _i < qpoverlaps.size1(); _i++) {
447  if (std::abs(qpoverlaps(_i, _j)) > std::abs(qpoverlaps(maxi, _j))) {
448  maxi = _i;
449  }
450  }
451  _state_index.push_back(maxi);
452  CTP_LOG(ctp::logDEBUG, *_log) << " [" << maxi << " , " << _j << "]: " << qpoverlaps(maxi, _j) << flush;
453  }
454 
455  }
456 
457  }
458 
459  if (_has_osc_filter) {
460 
461  // go through list of singlets
462  const std::vector<double>oscs = orb_iter_input.Oscillatorstrengths();
463  for (unsigned _i = 0; _i < oscs.size(); _i++) {
464 
465  double osc = oscs[_i];
466  if (osc > _osc_threshold) _state_index.push_back(_i);
467  }
468 
469  } else {
470  const ub::vector<real_gwbse>& energies = (_type=="singlet")
471  ? orb_iter_input.BSESingletEnergies() : orb_iter_input.BSETripletEnergies();
472 
473  for (unsigned _i = 0; _i < energies.size(); _i++) {
474  _state_index.push_back(_i);
475  }
476  }
477 
478 
479  // filter according to charge transfer, go through list of excitations in _state_index
480  if (_has_dQ_filter) {
481  std::vector<int> _state_index_copy;
482  const std::vector< ub::vector<double> >& dQ_frag= (_type=="singlet")
483  ? orb_iter_input.getFragmentChargesSingEXC():orb_iter_input.getFragmentChargesTripEXC();
484  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
485  if (std::abs(dQ_frag[_state_index[_i]](0)) > _dQ_threshold) {
486  _state_index_copy.push_back(_state_index[_i]);
487  }
488  }
489  _state_index = _state_index_copy;
490  }
491  else if (_has_loc_filter) {
492  std::vector<int> _state_index_copy;
493  const std::vector< ub::vector<double> >& popE= (_type=="singlet")
494  ? orb_iter_input.getFragment_E_localisation_singlet():orb_iter_input.getFragment_E_localisation_triplet();
495  const std::vector< ub::vector<double> >& popH= (_type=="singlet")
496  ? orb_iter_input.getFragment_H_localisation_singlet():orb_iter_input.getFragment_H_localisation_triplet();
497  if(_localiseonA){
498  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
499  if (popE[_state_index[_i]](0) > _loc_threshold && popH[_state_index[_i]](0) > _loc_threshold ) {
500  _state_index_copy.push_back(_state_index[_i]);
501  }
502  }
503  }else{
504  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
505  if (popE[_state_index[_i]](1) > _loc_threshold && popH[_state_index[_i]](1) > _loc_threshold ) {
506  _state_index_copy.push_back(_state_index[_i]);
507  }
508  }
509  }
510  _state_index = _state_index_copy;
511  }
512 
513 
514  if (_state_index.size() < 1) {
515  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " WARNING: FILTER yielded no state. Taking lowest excitation" << flush;
516  _state_index.push_back(0);
517  }else{
518  if ( _type == "quasiparticle" ){
519  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Filter yielded QP index: "<<_state_index[_state - 1 - orb_iter_input.getGWAmin()]<< flush;
520  }else {
521  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Filter yielded state"<<_type<<":"<<_state_index[_state - 1]+1<< flush;
522  }
523  }
524  // - output its energy
525  if (_type == "singlet") {
526  energy___ex = orb_iter_input.BSESingletEnergies()[_state_index[_state - 1]] * tools::conv::hrt2ev; // to eV
527  } else if (_type == "triplet") {
528  energy___ex = orb_iter_input.BSETripletEnergies()[_state_index[_state - 1]] * tools::conv::hrt2ev; // to eV
529  } else if (_type == "quasiparticle") {
530  if ( _state > orb_iter_input.getNumberOfElectrons() ) {
531  // unoccupied QPs: E_a = E_0 + eps_l
532  energy___ex = orb_iter_input.QPdiagEnergies()[_state_index[_state - 1 - orb_iter_input.getGWAmin()]] * tools::conv::hrt2ev; // to eV
533  } else {
534  // occupied QPs: E_c = E_0 - eps_h
535  energy___ex = -1.0*orb_iter_input.QPdiagEnergies()[_state_index[_state - 1 - orb_iter_input.getGWAmin()]] * tools::conv::hrt2ev; // to eV
536  }
537  }
