117  template<class QMPackage>
118  void QMMachine<QMPackage>::Evaluate(ctp::XJob *job) {
119 
120  CTP_LOG(ctp::logINFO, *_log)
121  << format("... dR %1$1.4f dQ %2$1.4f QM %3$1.4f MM %4$1.4f IT %5$d")
122  % _crit_dR % _crit_dQ % _crit_dE_QM % _crit_dE_MM % _maxIter << flush;
123 
124  // FIGURE OUT CHARGE + MULTIPLICITY
125  double dQ = 0.0;
126  for (unsigned int i = 0; i < _job->getPolarTop()->QM0().size(); ++i) {
127  dQ += _job->getPolarTop()->QM0()[i]->CalcTotQ();
128  }
129  int chrg = round(dQ);
130  int spin = ((chrg < 0) ? -chrg : chrg) % 2 + 1;
131  CTP_LOG(ctp::logINFO, *_log) << "... Q = " << chrg << ", 2S+1 = " << spin << flush;
132 
133 
134  // PREPARE JOB DIRECTORY
135  string jobFolder = "xjob_" + boost::lexical_cast<string>(_job->getId())
136  + "_" + _job->getTag();
137  bool created = boost::filesystem::create_directory(jobFolder);
138  if (created) {
139  CTP_LOG(ctp::logINFO, *_log) << "Created directory " << jobFolder << flush;
140  }
141 
142 
143  // SET ITERATION-TIME CONSTANTS
144  // TO ADJUST
145 
146  _qmpack->setCharge(chrg);
147  _qmpack->setSpin(spin);
148 
149  //_qmpack->setThreads(_subthreads);
150 
151 
152  int iterCnt = 0;
153  int iterMax = _maxIter;
154  for (; iterCnt < iterMax; ++iterCnt) {
155 
156  //bool code =
157  (void) Iterate(jobFolder, iterCnt);
158  if (hasConverged()) {
159  break;
160  }
161  }
162 
163  if (iterCnt == iterMax - 1 && !_isConverged) {
164  CTP_LOG(ctp::logWARNING, *_log)
165  << format("Not converged within %1$d iterations.") % iterMax;
166  }
167 
168  return;
169  }

172  bool QMMachine<QMPackage>::Iterate(string jobFolder, int iterCnt) {
173 
174  // CREATE ITERATION OBJECT & SETUP RUN DIRECTORY
175  QMMIter *thisIter = this->CreateNewIter();
176  int iter = iterCnt;
177  string runFolder = jobFolder + "/iter_" + boost::lexical_cast<string>(iter);
178

184

188  _xind->Evaluate(_job);
189

193  thisIter->setE_FM(_job->getEF00(), _job->getEF01(), _job->getEF02(),
194  _job->getEF11(), _job->getEF12(), _job->getEM0(),
195  _job->getEM1(), _job->getEM2(), _job->getETOT());
196 
197  // WRITE AND SET QM INPUT FILE
198  Orbitals orb_iter_input;
199 
200  std::vector<ctp::Segment*> empty;
201  qminterface.GenerateQMAtomsFromPolarSegs(_job->getPolarTop(), orb_iter_input);
202 
203  _qmpack->setRunDir(runFolder);
204 
205  CTP_LOG(ctp::logDEBUG, *_log) << "Writing input file " << runFolder << flush;
206 
207  _qmpack->WriteInputFile(empty, &orb_iter_input);
208 
209  FILE *out;
210  out = fopen((runFolder + "/system.pdb").c_str(), "w");
211  orb_iter_input.WritePDB(out);
212  fclose(out);
213 
214 
215 
216  _qmpack->Run();
217 
218  // EXTRACT LOG-FILE INFOS TO ORBITALS
219  Orbitals orb_iter_output;
220  _qmpack->ParseLogFile(&orb_iter_output);
221 
222  // GW-BSE starts here
223 
224  double energy___ex = 0.0;
225 
226  if (_do_gwbse) {
227 
228 
229  // for GW-BSE, we also need to parse the orbitals file
230  _qmpack->ParseOrbitalsFile(&orb_iter_output);
231  GWBSE _gwbse=GWBSE(&orb_iter_output);
232  std::vector<int> _state_index;
233  // define own logger for GW-BSE that is written into a runFolder logfile
234  ctp::Logger gwbse_logger(ctp::logDEBUG);
235  gwbse_logger.setMultithreading(false);
236  _gwbse.setLogger(&gwbse_logger);
237  gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
238  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
239  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
240  gwbse_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
241 
242  _gwbse.Initialize(&_gwbse_options);
243 
244  if (_state > 0) {
245  CTP_LOG(ctp::logDEBUG, *_log) << "Excited state via GWBSE: " << flush;
246  CTP_LOG(ctp::logDEBUG, *_log) << " --- type: " << _type << flush;
247  CTP_LOG(ctp::logDEBUG, *_log) << " --- state: " << _state << flush;
248  if (_has_osc_filter) {
249  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: osc.str. > " << _osc_threshold << flush;
250  }
251  if (_has_dQ_filter) {
252  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: crg.trs. > " << _dQ_threshold << flush;
253  }
254 
255  if (_has_osc_filter && _has_dQ_filter) {
256  CTP_LOG(ctp::logDEBUG, *_log) << " --- WARNING: filtering for optically active CT transition - might not make sense... " << flush;
257  }
258 
259 
260 
261  // actual GW-BSE run
262  _gwbse.Evaluate();
263 
264 
265  // write logger to log file
266  ofstream ofs;
267  string gwbse_logfile = runFolder + "/gwbse.log";
268  ofs.open(gwbse_logfile.c_str(), ofstream::out);
269  if (!ofs.is_open()) {
270  throw runtime_error("Bad file handle: " + gwbse_logfile);
271  }
272  ofs << gwbse_logger << endl;
273  ofs.close();
274 
275  // PROCESSING the GW-BSE result
276  // - find the excited state of interest
277  // oscillator strength filter
278 
279 
280  if (_has_osc_filter) {
281 
282  // go through list of singlets
283  const std::vector<double>oscs = orb_iter_output.Oscillatorstrengths();
284  for (unsigned _i = 0; _i < oscs.size(); _i++) {
285 
286  double osc = oscs[_i];
287  if (osc > _osc_threshold) _state_index.push_back(_i);
288  }
289 
290 
291 
292  } else {
293 
294  if (_type == "singlet") {
295  for (unsigned _i = 0; _i < orb_iter_output.TransitionDipoles().size(); _i++) {
296  _state_index.push_back(_i);
297  }
298  } else {
299  for (unsigned _i = 0; _i < orb_iter_output.BSETripletEnergies().size(); _i++) {
300  _state_index.push_back(_i);
301  }
302  }
303  }
304 
305 
306  // filter according to charge transfer, go through list of excitations in _state_index
307  if (_has_dQ_filter) {
308  std::vector<int> _state_index_copy;
309  if (_type == "singlets") {
310  // go through list of singlets
311  const std::vector< ub::vector<double> >& dQ_frag = orb_iter_output.FragmentChargesSingEXC();
312  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
313  if (std::abs(dQ_frag[_i](0)) > _dQ_threshold) {
314  _state_index_copy.push_back(_state_index[_i]);
315  }
316  }
317  _state_index = _state_index_copy;
318  } else if (_type == "triplets") {
319  // go through list of triplets
320  const std::vector< ub::vector<double> >& dQ_frag = orb_iter_output.FragmentChargesTripEXC();
321  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
322  if (std::abs(dQ_frag[_i](0)) > _dQ_threshold) {
323  _state_index_copy.push_back(_state_index[_i]);
324  }
325  }
326  _state_index = _state_index_copy;
327 
328 
329  }
330  }
331 
332 
333  if (_state_index.size() < 1) {
334  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " WARNING: FILTER yielded no state. Taking lowest excitation"<< flush;
335  _state_index.push_back(0);
336  }
337  // - output its energy
338  if (_type == "singlet") {
339  energy___ex = orb_iter_output.BSESingletEnergies()[_state_index[_state - 1]]*tools::conv::hrt2ev; // to eV
340  } else if (_type == "triplet") {
341  energy___ex = orb_iter_output.BSETripletEnergies()[_state_index[_state - 1]]*tools::conv::hrt2ev; // to eV
342  }
343 
344  // ub::matrix<double> &_dft_orbitals_GS = orb_iter_output.MOCoefficients();
345  // int _parse_orbitals_status_GS = _qmpack->ParseOrbitalsFile( &orb_iter_output );
346 
347  } // only if state >0
348 
349  // calculate density matrix for this excited state
350  ub::matrix<double> &_dft_orbitals = orb_iter_output.MOCoefficients();
351  // load DFT basis set (element-wise information) from xml file
352  BasisSet dftbs;
353  if (orb_iter_output.getDFTbasis() != "") {
354  dftbs.LoadBasisSet(orb_iter_output.getDFTbasis());
355  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Loaded DFT Basis Set " << orb_iter_output.getDFTbasis() << flush;
356  } else {
357  dftbs.LoadBasisSet(_gwbse.get_dftbasis_name());
358  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Loaded DFT Basis Set " << _gwbse.get_dftbasis_name() << flush;
359  }
360 
361 
362  // fill DFT AO basis by going through all atoms
363  AOBasis dftbasis;
364  dftbasis.AOBasisFill(&dftbs, orb_iter_output.QMAtoms());
365  dftbasis.ReorderMOs(_dft_orbitals, orb_iter_output.getQMpackage(), "xtp");
366  // TBD: Need to switch between singlets and triplets depending on _type
367  ub::matrix<double> DMATGS = orb_iter_output.DensityMatrixGroundState(_dft_orbitals);
368 
369  ub::matrix<double> DMAT_tot = DMATGS; // Ground state + hole_contribution + electron contribution
370 
371  if (_state > 0) {
372  std::vector<ub::matrix<double> > DMAT = _gwbse.getExcitedStateDmat(_type, _state_index[_state - 1]);
373  DMAT_tot = DMAT_tot - DMAT[0] + DMAT[1]; // Ground state + hole_contribution + electron contribution
374  }
375 
376  // fill DFT AO basis by going through all atoms
377  std::vector< ctp::QMAtom* >& Atomlist = orb_iter_output.QMAtoms();
378 
379 
380 
381  Espfit esp=Espfit(_log);
382  if (_qmpack->ECPRequested()){
383  esp.setUseECPs(true);
384  }
385  esp.Fit2Density(Atomlist, DMAT_tot, dftbasis,dftbs,"medium");
386 
387 } //_do_gwbse
388 
389  // Test: go via GDMA instead of point charges, only for DFT with Gaussian!
390  GDMA _gdma;
391  if (_do_gdma) {
392  if (_qmpack->getPackageName() != "gaussian" || _qmpack->getExecutable() != "g03") {
393 
394  throw runtime_error(" Invalid QMPackage! " + _type + " Gaussian 03 only!");
395 
396  } else {
397  // get a GDMA object
398  _gdma.Initialize(&_gdma_options);
399  _gdma.setLog(_log);
400  _gdma.SetRunDir(runFolder);
401 
402  CTP_LOG(ctp::logINFO, *_log) << "Running GDMA " << flush;
403  // prepare a GDMA input file
404  _gdma.WriteInputFile();
405 
406  // run GDMA external
407  _gdma.RunExternal();
408 
409  // parse output of gdma and update multipoles_full
410  _gdma.ParseOutputFile();
411 
412  } // use gdma
413  } // _do_gdma
414 
415 
416  out = fopen((runFolder + "/InputConfig.pdb").c_str(), "w");
417  orb_iter_input.WritePDB(out);
418  fclose(out);
419 
420  assert(orb_iter_output.hasSelfEnergy());
421  assert(orb_iter_output.hasQMEnergy());
422 
423  // EXTRACT & SAVE QM ENERGIES
424  double energy___sf = orb_iter_output.getSelfEnergy();
425  double energy_qmsf = orb_iter_output.getQMEnergy();
426  double energy_qm__ = energy_qmsf - energy___sf;
427  thisIter->setQMSF(energy_qm__, energy___sf, energy___ex);
428  _job->setEnergy_QMMM(thisIter->getQMEnergy(), thisIter->getGWBSEEnergy(), thisIter->getSFEnergy(),
429  thisIter->getQMMMEnergy());
430 
431  // EXTRACT & SAVE QMATOM DATA
432  std::vector< ctp::QMAtom* > &atoms = orb_iter_output.QMAtoms();
433 
434  thisIter->UpdatePosChrgFromQMAtoms(atoms, _job->getPolarTop()->QM0());
435 
436  if (_do_gdma) {
437 
438  // update PolarTop
439  thisIter->UpdateMPSFromGDMA(_gdma.GetMultipoles(), _job->getPolarTop()->QM0());
440 
441  }
442 
443  CTP_LOG(ctp::logINFO, *_log)
444  << format("Summary - iteration %1$d:") % (iterCnt + 1) << flush;
445  CTP_LOG(ctp::logINFO, *_log)
446  << format("... QM Size = %1$d atoms") % int(atoms.size()) << flush;
447  CTP_LOG(ctp::logINFO, *_log)

483  template<class QMPackage>
484  bool QMMachine<QMPackage>::hasConverged() {
485 
486  _convg_dR = false;
487  _convg_dQ = false;
488  _convg_dE_QM = false;
489  _convg_dE_MM = false;
490 
491  if (_iters.size() > 1) {
492 
493  QMMIter *iter_0 = _iters[_iters.size() - 2];
494  QMMIter *iter_1 = _iters[_iters.size() - 1];
495 
496  double dR = iter_1->getRMSdR();
497  double dQ = iter_1->getRMSdQ();
498  double dE_QM = iter_1->getQMEnergy() - iter_0->getQMEnergy();
499  double dE_MM = iter_1->getMMEnergy() - iter_0->getMMEnergy();
500 
501  CTP_LOG(ctp::logINFO, *_log)
502  << format("... dE_QM = %1$+4.9e") % dE_QM << flush;
503  CTP_LOG(ctp::logINFO, *_log)
504  << format("... dE_MM = %1$+4.9e") % dE_MM << flush;
505 
506  if (dR <= _crit_dR) _convg_dR = true;
507  if (dQ <= _crit_dQ) _convg_dQ = true;
508  if (dE_QM * dE_QM <= _crit_dE_QM * _crit_dE_QM) _convg_dE_QM = true;
509  if (dE_MM * dE_MM <= _crit_dE_MM * _crit_dE_MM) _convg_dE_MM = true;
510  }
511 
512  _isConverged = ((_convg_dR && _convg_dQ) && (_convg_dE_QM && _convg_dE_MM));
513 
514 
515 
516  CTP_LOG(ctp::logINFO, *_log)
517  << format("... Convg dR = %s") % (_convg_dR ? "true" : "false") << flush;
518  CTP_LOG(ctp::logINFO, *_log)
519  << format("... Convg dQ = %s") % (_convg_dQ ? "true" : "false") << flush;
520  CTP_LOG(ctp::logINFO, *_log)
521  << format("... Convg QM = %s") % (_convg_dE_QM ? "true" : "false") << flush;
522  CTP_LOG(ctp::logINFO, *_log)
523  << format("... Convg MM = %s") % (_convg_dE_MM ? "true" : "false") << flush;
524 
525  return _isConverged;
526  }
