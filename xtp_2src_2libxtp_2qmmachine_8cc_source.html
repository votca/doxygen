594  template<class XQMPackage>
595  bool QMMachine<XQMPackage>::hasConverged() {
596 
597  _convg_dR = false;
598  _convg_dQ = false;
599  _convg_dE_QM = false;
600  _convg_dE_MM = false;
601 
602  if (_iters.size() > 1) {
603 
604  QMMIter *iter_0 = _iters[_iters.size() - 2];
605  QMMIter *iter_1 = _iters[_iters.size() - 1];
606 
607  double dR = iter_1->getRMSdR();
608  double dQ = iter_1->getRMSdQ();
609  double dE_QM = iter_1->getQMEnergy() - iter_0->getQMEnergy();
610  double dE_MM = iter_1->getMMEnergy() - iter_0->getMMEnergy();
611 
612  LOG(CTP::logINFO, *_log)
613  << format("... dE_QM = %1$+4.9e") % dE_QM << flush;
614  LOG(CTP::logINFO, *_log)
615  << format("... dE_MM = %1$+4.9e") % dE_MM << flush;
616 
617  if (dR <= _crit_dR) _convg_dR = true;
618  if (dQ <= _crit_dQ) _convg_dQ = true;
619  if (dE_QM * dE_QM <= _crit_dE_QM * _crit_dE_QM) _convg_dE_QM = true;
620  if (dE_MM * dE_MM <= _crit_dE_MM * _crit_dE_MM) _convg_dE_MM = true;
621  }
622 
623  _isConverged = ((_convg_dR && _convg_dQ) && (_convg_dE_QM && _convg_dE_MM));
624 
625 
626 
627  LOG(CTP::logINFO, *_log)
628  << format("... Convg dR = %s") % (_convg_dR ? "true" : "false") << flush;
629  LOG(CTP::logINFO, *_log)
630  << format("... Convg dQ = %s") % (_convg_dQ ? "true" : "false") << flush;
631  LOG(CTP::logINFO, *_log)
632  << format("... Convg QM = %s") % (_convg_dE_QM ? "true" : "false") << flush;
633  LOG(CTP::logINFO, *_log)
634  << format("... Convg MM = %s") % (_convg_dE_MM ? "true" : "false") << flush;
635 
636  return _isConverged;

703  void QMMIter::UpdatePosChrgFromQMAtoms(std::vector< CTP::QMAtom* > &qmatoms,
704  std::vector< CTP::PolarSeg* > &psegs) {
705 
706  double AA_to_NM = 0.1; // Angstrom to nanometer
707 
708  double dR_RMS = 0.0;
709  double dQ_RMS = 0.0;
710  double dQ_SUM = 0.0;
711 
712  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
713  CTP::PolarSeg *pseg = psegs[i];
714  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
715 
716  // Retrieve info from QMAtom
717  CTP::QMAtom *qmatm = qmatoms[qac];
718  vec upd_r = vec(qmatm->x, qmatm->y, qmatm->z);
719  upd_r *= AA_to_NM;
720  double upd_Q00 = qmatm->charge;
721  //cout << "updating charge to " << qmatm->charge << endl;
722 
723  // Compare to previous r, Q00
724  CTP::APolarSite *aps = (*pseg)[j];
725  vec old_r = aps->getPos();
726  double old_Q00 = aps->getQ00();
727  double dR = abs(upd_r - old_r);
728  double dQ00 = upd_Q00 - old_Q00;
729 
730  dR_RMS += dR*dR;
731  dQ_RMS += dQ00*dQ00;
732  dQ_SUM += dQ00;
733 
734  // Forward updated r, Q00 to APS
735  aps->setPos(upd_r);
736  aps->setQ00(upd_Q00, 0);
737  }
738  }
739 
740 
741  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
742  dR_RMS /= qmatoms.size();
743  dQ_RMS /= qmatoms.size();
744  dR_RMS = sqrt(dR_RMS);
745  dQ_RMS = sqrt(dQ_RMS);
746 
747  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
748 
749 
750  this->setdRdQ(dR_RMS, dQ_RMS, dQ_SUM);
751  }

753  void QMMIter::GenerateQMAtomsFromPolarSegs(CTP::PolarTop *ptop, Orbitals &orb,
754  bool split_dpl, double dpl_spacing) {
755 
756  double AA_to_NM = 0.1; // Angstrom to nanometer
757 
758  // INNER SHELL QM0
759  for (unsigned int i = 0; i < ptop->QM0().size(); ++i) {
760  CTP::PolarSeg *pseg = ptop->QM0()[i];
761  for (unsigned int j = 0; j < pseg->size(); ++j) {
762 
763  CTP::APolarSite *aps = (*pseg)[j];
764  vec pos = aps->getPos() / AA_to_NM;
765  double Q = aps->getQ00();
766  string type = "qm";
767 
768  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, false);
769 
770  }
771  }
772 
773  // MIDDLE SHELL MM1
774  for (unsigned int i = 0; i < ptop->MM1().size(); ++i) {
775  CTP::PolarSeg *pseg = ptop->MM1()[i];
776  for (unsigned int j = 0; j < pseg->size(); ++j) {
777 
778  CTP::APolarSite *aps = (*pseg)[j];
779  vec pos = aps->getPos() / AA_to_NM;
780  double Q = aps->getQ00();
781  string type = "mm";
782 
783  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
784 
785  if (split_dpl) {
786  //vec tot_dpl = vec(aps->U1x, aps->U1y, aps->U1z);
787  vec tot_dpl = aps->getU1(); // access function
788  if (aps->getRank() > 0) {
789  //tot_dpl += vec(aps->Q1x, aps->Q1y, aps->Q1z);
790  tot_dpl += aps->getQ1();
791  }
792  // Calculate virtual charge positions
793  double a = dpl_spacing; // this is in nm
794  double mag_d = abs(tot_dpl); // this is in e * nm
795  vec dir_d_0 = tot_dpl.normalize();
796  vec dir_d = dir_d_0.normalize();
797  vec A = pos + 0.5 * a * dir_d / AA_to_NM; // converted to AA
798  vec B = pos - 0.5 * a * dir_d / AA_to_NM;
799  double qA = mag_d / a;
800  double qB = -qA;
801  // Zero out if magnitude small [e*nm]
802  if (aps->getIsoP() < 1e-9 || mag_d < 1e-9) {
803  A = aps->getPos() + 0.1 * a * vec(1, 0, 0); // != pos since self-energy may diverge
804  B = aps->getPos() - 0.1 * a * vec(1, 0, 0);
805  qA = 0;
806  qB = 0;
807  }
808  orb.AddAtom("A", A.x(), A.y(), A.z(), qA, true);
809  orb.AddAtom("B", B.x(), B.y(), B.z(), qB, true);
810  }
811  }
812  }
813 
814  // OUTER SHELL MM2
815  for (unsigned int i = 0; i < ptop->MM2().size(); ++i) {
816  CTP::PolarSeg *pseg = ptop->MM2()[i];
817  for (unsigned int j = 0; j < pseg->size(); ++j) {
818 
819  CTP::APolarSite *aps = (*pseg)[j];
820  vec pos = aps->getPos() / AA_to_NM;
821  double Q = aps->getQ00();
822  string type = "mm";
823 
824  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
825  }
826  }
827  return;
828 
829 
830  }
