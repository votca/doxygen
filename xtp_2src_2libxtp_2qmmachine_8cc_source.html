34 QMMachine::QMMachine(ctp::XJob *job, ctp::XInductor *xind, QMPackage *qmpack,
35  Property *opt, string sfx)
36  : _job(job), _xind(xind), _qmpack(qmpack), _isConverged(false) {
37 
38  string key = sfx + ".qmmmconvg";
39  _crit_dR =
40  opt->ifExistsReturnElseReturnDefault<double>(key + ".dR", 0.01); // nm
41  _crit_dQ =
42  opt->ifExistsReturnElseReturnDefault<double>(key + ".dQ", 0.01); // e
43  _crit_dE_QM = opt->ifExistsReturnElseReturnDefault<double>(key + ".dQdE_QM",
44  0.001); // eV
45  _crit_dE_MM = opt->ifExistsReturnElseReturnDefault<double>(
46  key + ".dE_MM", _crit_dE_QM); // eV
47  _maxIter = opt->ifExistsReturnElseReturnDefault<int>(key + ".max_iter", 32);
48  _alpha = opt->ifExistsReturnElseReturnDefault<double>(key + ".mixing", 0.0);
49 
50  key = sfx;
51  double dpl_spacing =
52  opt->ifExistsReturnElseReturnDefault<double>(key + ".dpl_spacing", 1e-3);
53 
54  // check for archiving
55  std::string _archiving_string =
56  opt->ifExistsReturnElseReturnDefault<std::string>(key + ".archiving", "");
57  if (_archiving_string.find("iterations") != std::string::npos) {
58  _do_archive = true;
59  } else {
60  _do_archive = false;
61  }
62 
63  // check for static or polarized qmmm
64  key = sfx + ".tholemodel";
65  _static_qmmm = true;
66  qmpack->setWithPolarization(false);
67  if (opt->exists(key + ".induce")) {
68  bool induce = opt->get(key + ".induce").as<bool>();
69  if (induce) {
70  qmpack->setWithPolarization(true);
71  qmpack->setDipoleSpacing(dpl_spacing);
72  }
73  _static_qmmm = !induce;
74  }
75 
76  // GDMA options
77  key = sfx + ".gdma";
78  if (opt->exists(key)) {
79  _do_gdma = true;
80  string gdma_xml = opt->get(key).as<string>();
81  load_property_from_xml(_gdma_options, gdma_xml.c_str());
82  } else {
83  _do_gdma = false;
84  }
85 
86  // GWBSE options
87  key = sfx + ".gwbse";
88  if (opt->exists(key)) {
89  string _gwbse_xml = opt->get(key + ".gwbse_options").as<string>();
90  load_property_from_xml(_gwbse_options, _gwbse_xml.c_str());
91  key = sfx + ".gwbse.filter";
92  if (opt->exists(key)) {
93  tools::Property &filterprop = opt->get(key);
94  _filter.Initialize(filterprop);
95  }
96  }
97 
98  return;
99 }

108 int QMMachine::Evaluate(ctp::XJob *job) {
109 
110  CTP_LOG(ctp::logINFO, *_log)
111  << format("... dR %1$1.4f dQ %2$1.4f QM %3$1.4f MM %4$1.4f IT %5$d") %
112  _crit_dR % _crit_dQ % _crit_dE_QM % _crit_dE_MM % _maxIter
113  << flush;
114 
115  // FIGURE OUT CHARGE + MULTIPLICITY
116  double dQ = 0.0;
117  for (unsigned int i = 0; i < _job->getPolarTop()->QM0().size(); ++i) {
118  dQ += _job->getPolarTop()->QM0()[i]->CalcTotQ();
119  }
120  int chrg = std::round(dQ);
121  int spin = ((chrg < 0) ? -chrg : chrg) % 2 + 1;
122  CTP_LOG(ctp::logINFO, *_log)
123  << "... Q = " << chrg << ", 2S+1 = " << spin << flush;
124 
125  // PREPARE JOB DIRECTORY
126  boost::filesystem::path arg_path;
127  string frame_dir =
128  "frame_" + boost::lexical_cast<string>(_job->getTop()->getDatabaseId());
129  string jobFolder = "job_" + boost::lexical_cast<string>(_job->getId()) + "_" +
130  _job->getTag();
131  string work_dir = (arg_path / "QMMM" / frame_dir / jobFolder).c_str();
132  bool created = boost::filesystem::create_directories(work_dir);
133  if (created) {
134  CTP_LOG(ctp::logINFO, *_log) << "Created directory " << work_dir << flush;
135  }
136 
137  vector<string> split;
138  Tokenizer toker(_job->getTag(), ":");
139  toker.ToVector(split);
140  _initialstate = QMState(split.back());
141  if (_initialstate.Type().isExciton() || _initialstate.Type().isGWState()) {
142  _filter.setInitialState(_initialstate);
143  _filter.setLogger(_log);
144  _do_gwbse = true;
145  }
146 
147  _qmpack->setCharge(chrg);
148  _qmpack->setSpin(spin);
149 
150  int iterCnt = 0;
151  int iterMax = _maxIter;
152  for (; iterCnt < iterMax; ++iterCnt) {
153 
154  // check for polarized QM/MM convergence
155  int info = Iterate(work_dir, iterCnt);
156  if (info != 0) {
157  CTP_LOG(ctp::logERROR, *_log) << format("Iterating job failed!") << flush;
158  return 1;
159  }
160  if (_static_qmmm) {
161  _isConverged = true;
162  break;
163  } else if (hasConverged()) {
164  break;
165  }
166  }
167 
168  if (iterCnt == iterMax - 1 && !_isConverged) {
169  CTP_LOG(ctp::logWARNING, *_log)
170  << format("Not converged within %1$d iterations.") % iterMax;
171  return 2;
172  }
173 
174  return 0;
175 }

224 bool QMMachine::RunDFT(
225  string &runFolder,
226  std::vector<std::shared_ptr<ctp::PolarSeg> > &MultipolesBackground) {
227  ctp::Logger dft_logger(ctp::logDEBUG);
228  dft_logger.setMultithreading(false);
229  dft_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
230  dft_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
231  dft_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
232  dft_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
233  _qmpack->setLog(&dft_logger);
234 
235  _qmpack->setMultipoleBackground(MultipolesBackground);
236 
237  // setting RUNDIR for the external QMPackages, dummy for internal
238  _qmpack->setRunDir(runFolder);
239 
240  CTP_LOG(ctp::logDEBUG, *_log) << "Writing input file " << runFolder << flush;
241  CTP_LOG(ctp::logDEBUG, *_log) << "Running DFT " << flush;
242  _qmpack->WriteInputFile(orb_iter_input);
243 
244  orb_iter_input.WriteXYZ(runFolder + "/system_qm.xyz");
245 
246  _qmpack->Run();
247 
248  /* Parse information from the LOGFILE into orbitals, if
249  * external QMPackage is run. Dummy for internal DFTENGINE.
250  * The QMPackage's MOcoefficients are not automatically
251  * parsed in DFT-only calculations and ESP fits for
252  * polarized QMMM are expected in the LOGFILE.
253  * IF the internal ESPFITs should be used, the MOcoefficients
254  * need to be parsed too.
255  */
256  bool success1 = _qmpack->ParseLogFile(orb_iter_input);
257 
258  bool success2 = _qmpack->ParseOrbitalsFile(orb_iter_input);
259  _qmpack->CleanUp();
260 
261  ofstream ofs;
262  string dft_logfile = runFolder + "/dft.log";
263  ofs.open(dft_logfile.c_str(), ofstream::out);
264  if (!ofs.is_open()) {
265  throw runtime_error("Bad file handle: " + dft_logfile);
266  }
267  ofs << dft_logger << endl;
268  ofs.close();
269  _qmpack->setLog(_log);
270  return success1 && success2;
271 }

273 bool QMMachine::Iterate(string jobFolder, int iterCnt) {
274 
275  // CREATE ITERATION OBJECT & SETUP RUN DIRECTORY
276  QMMIter *thisIter = this->CreateNewIter();
277  int iter = iterCnt;
278  string runFolder = jobFolder + "/iter_" + boost::lexical_cast<string>(iter);
279 
280  bool created = boost::filesystem::create_directory(runFolder);
281  if (created)
282  CTP_LOG(ctp::logDEBUG, *_log) << "Created directory " << runFolder << flush;
283  else
284  CTP_LOG(ctp::logWARNING, *_log)
285  << "Could not create directory " << runFolder << flush;
286 
287  // RUN CLASSICAL INDUCTION & SAVE
288  _job->getPolarTop()->PrintPDB(runFolder + "/QM0_MM1_MM2.pdb");
289  _xind->Evaluate(_job);
290 
291  if (!_xind->hasConverged()) {
292  throw std::runtime_error("Classical induction did not converge");
293  }
294 
295  thisIter->setE_FM(_job->getEF00(), _job->getEF01(), _job->getEF02(),
296  _job->getEF11(), _job->getEF12(), _job->getEM0(),
297  _job->getEM1(), _job->getEM2(), _job->getETOT());
298 
299  /* Translate atoms in QM0() to QMAtoms in orbitals object
300  * to be used in writing the QM input files for the
301  * external QMPackages or directly in internal DFT engine.
302  * DRAWBACK: QMAtom positions are fixed for all iterations
303  * unless the UPDATE function at the end of the
304  * iteration updates orb_iter_input (TO BE TESTED)
305  */
306  QMInterface qminterface;
307  if (iterCnt == 0)
308  qminterface.GenerateQMAtomsFromPolarSegs(_job->getPolarTop(),
309  orb_iter_input);
310 
311  /* Generate list of polar segments in the MM1() and MM2()
312  * region to be used in writing the background multipoles
313  * in the external QMPackages or in the direct evaluation
314  * in the internal DFT engine.
315  */
316  std::vector<std::shared_ptr<ctp::PolarSeg> > MultipolesBackground =
317  qminterface.GenerateMultipoleList(_job->getPolarTop());
318 
319  bool success = RunDFT(runFolder, MultipolesBackground);
320  if (!success) {
321  return 1;
322  }
323 
324  double energy_ex = 0.0;
325  if (_do_gwbse) {
326  if (_initialstate.Type() == QMStateType::Gstate) {
327  CTP_LOG(ctp::logDEBUG, *_log) << "QMMMachine no GWBSE necessary for " +
328  _initialstate.ToLongString();
329  }
330  RunGWBSE(runFolder);
331  CTP_LOG(ctp::logDEBUG, *_log) << "Excited state via GWBSE: " << flush;
332  CTP_LOG(ctp::logDEBUG, *_log)
333  << "state:" << _initialstate.ToLongString() << flush;
334  _filter.PrintInfo();
335 
336  QMState newstate = _filter.CalcStateAndUpdate(orb_iter_input);
337  energy_ex =
338  orb_iter_input.getExcitedStateEnergy(newstate) * tools::conv::hrt2ev;
339 
340  if (!_static_qmmm) {
341  Density2Charges(newstate);
342  }
343 
344  } //_do_gwbse
345 
346  /* new ESP fit only required for
347  * - polarizable QMMM
348  * AND
349  * - GWBSE or DFT with internal DFTENGINE
350  */
351  if (!_static_qmmm && _qmpack->getPackageName() == "xtp" && !_do_gwbse) {
352  QMState gsstate = QMState(QMStateType::Gstate, 0, false);
353  Density2Charges(gsstate);
354  } // for polarized QMMM
355 
356  if (tools::globals::verbose) {
357  CTP_LOG(ctp::logDEBUG, *_log) << "Calculated partial charges" << flush;
358  for (const QMAtom *atom : orb_iter_input.QMAtoms()) {
359  CTP_LOG(ctp::logDEBUG, *_log)
360  << atom->getType() << " " << atom->getPartialcharge() << flush;
361  }
362  }
363 
364  // EXTRACT & SAVE QM ENERGIES
365  double energy_sf = orb_iter_input.getSelfEnergy() * tools::conv::hrt2ev;
366  double energy_qmsf = orb_iter_input.getQMEnergy() * tools::conv::hrt2ev;
367  double energy_qm = energy_qmsf - energy_sf;
368  thisIter->setQMSF(energy_qm, energy_sf, energy_ex);
369  _job->setEnergy_QMMM(thisIter->getQMEnergy(), thisIter->getGWBSEEnergy(),
370  thisIter->getSFEnergy(), thisIter->getQMMMEnergy());
371 
372  // EXTRACT & SAVE QMATOM DATA
373  std::vector<QMAtom *> &atoms = orb_iter_input.QMAtoms();
374 
375  thisIter->UpdatePosChrgFromQMAtoms(atoms, _job->getPolarTop()->QM0());
376 
377  if (_do_gdma) {
378  RunGDMA(thisIter, runFolder);
379  }
380 
381  // serialize this iteration
382  if (_do_archive) {
383  // save orbitals
384  std::string orb_file = runFolder + "/system.orb";
385  CTP_LOG(ctp::logDEBUG, *_log) << "Archiving data to " << orb_file << flush;
386  orb_iter_input.WriteToCpt(orb_file);
387  }
388 
389  CTP_LOG(ctp::logINFO, *_log)
390  << format("Summary - iteration %1$d:") % (iterCnt + 1) << flush;
391  CTP_LOG(ctp::logINFO, *_log)
392  << format("... QM Size = %1$d atoms") % int(atoms.size()) << flush;
393  CTP_LOG(ctp::logINFO, *_log)
394  << format("... E(QM) = %1$+4.9e") % thisIter->getQMEnergy() << flush;
395  CTP_LOG(ctp::logINFO, *_log)
396  << format("... E(GWBSE) = %1$+4.9e") % thisIter->getGWBSEEnergy()
397  << flush;
398  CTP_LOG(ctp::logINFO, *_log)
399  << format("... E(SF) = %1$+4.9e") % thisIter->getSFEnergy() << flush;
400  CTP_LOG(ctp::logINFO, *_log)
401  << format("... E(FM) = %1$+4.9e") % thisIter->getFMEnergy() << flush;
402  CTP_LOG(ctp::logINFO, *_log)
403  << format("... E(MM) = %1$+4.9e") % thisIter->getMMEnergy() << flush;
404  CTP_LOG(ctp::logINFO, *_log)
405  << format("... E(QMMM) = %1$+4.9e") % thisIter->getQMMMEnergy() << flush;
406  if (!_static_qmmm) {
407  CTP_LOG(ctp::logINFO, *_log)
408  << format("... RMS(dR) = %1$+4.9e") % thisIter->getRMSdR() << flush;
409  CTP_LOG(ctp::logINFO, *_log)
410  << format("... RMS(dQ) = %1$+4.9e") % thisIter->getRMSdQ() << flush;
411  CTP_LOG(ctp::logINFO, *_log)
412  << format("... SUM(dQ) = %1$+4.9e") % thisIter->getSUMdQ() << flush;
413  }
414  // CLEAN DIRECTORY
415 
416  return 0;
417 }

419 void QMMachine::Density2Charges(const QMState &state) {
420 
421  Eigen::MatrixXd DMAT = orb_iter_input.DensityMatrixFull(state);
422 
423  int iter = _iters.size() - 1;
424  Eigen::MatrixXd DMAT_mixed;
425  if (iter == 0) {
426  DMAT_mixed = DMAT;
427  } else {
428  DMAT_mixed = _alpha * _DMAT_old + (1 - _alpha) * DMAT;
429  }
430 
431  _DMAT_old = DMAT_mixed;
432  BasisSet dftbs;
433  if (orb_iter_input.getDFTbasisName() != "") {
434  dftbs.LoadBasisSet(orb_iter_input.getDFTbasisName());
435  CTP_LOG(ctp::logDEBUG, *_log)
436  << ctp::TimeStamp() << " Loaded DFT Basis Set "
437  << orb_iter_input.getDFTbasisName() << flush;
438  } else {
439  throw std::runtime_error("Basisset in orb_iter_input not set");
440  }
441  // fill DFT AO basis by going through all atoms
442  AOBasis dftbasis;
443  dftbasis.AOBasisFill(dftbs, orb_iter_input.QMAtoms());
444  Espfit esp = Espfit(_log);
445  esp.Fit2Density(orb_iter_input.QMAtoms(), DMAT_mixed, dftbasis, "medium");
446 
447  return;
448 }

457 bool QMMachine::hasConverged() {
458 
459  bool convg_dR = false;
460  bool convg_dQ = false;
461  bool convg_dE_QM = false;
462  bool convg_dE_MM = false;
463 
464  if (_iters.size() > 1) {
465 
466  QMMIter *iter_0 = _iters[_iters.size() - 2];
467  QMMIter *iter_1 = _iters[_iters.size() - 1];
468 
469  double dR = iter_1->getRMSdR();
470  double dQ = iter_1->getRMSdQ();
471  double dE_QM = iter_1->getQMEnergy() - iter_0->getQMEnergy();
472  double dE_MM = iter_1->getMMEnergy() - iter_0->getMMEnergy();
473 
474  CTP_LOG(ctp::logINFO, *_log)
475  << format("... dE_QM = %1$+4.9e") % dE_QM << flush;
476  CTP_LOG(ctp::logINFO, *_log)
477  << format("... dE_MM = %1$+4.9e") % dE_MM << flush;
478 
479  if (dR <= _crit_dR) convg_dR = true;
480  if (dQ <= _crit_dQ) convg_dQ = true;
481  if (dE_QM * dE_QM <= _crit_dE_QM * _crit_dE_QM) convg_dE_QM = true;
482  if (dE_MM * dE_MM <= _crit_dE_MM * _crit_dE_MM) convg_dE_MM = true;
483  }
484 
485  _isConverged = ((convg_dR && convg_dQ) && (convg_dE_QM && convg_dE_MM));
486 
487  CTP_LOG(ctp::logINFO, *_log)
488  << format("... Convg dR = %s") % (convg_dR ? "true" : "false") << flush;
489  CTP_LOG(ctp::logINFO, *_log)
490  << format("... Convg dQ = %s") % (convg_dQ ? "true" : "false") << flush;
491  CTP_LOG(ctp::logINFO, *_log)
492  << format("... Convg QM = %s") % (convg_dE_QM ? "true" : "false")
493  << flush;
494  CTP_LOG(ctp::logINFO, *_log)
495  << format("... Convg MM = %s") % (convg_dE_MM ? "true" : "false")
496  << flush;
497 
498  return _isConverged;
499 }
