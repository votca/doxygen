205 XJob XQMP::ProcessInputString(Job *job, Topology *top, QMThread *thread) {
206 
207  string input = job->getInput().as<string>();
208  vector<Segment*> qmSegs;
209  vector<string> qmSegMps;
210  vector<string> split;
211  Tokenizer toker(input, " \t\n");
212  toker.ToVector(split);
213 
214  for (unsigned int i = 0; i < split.size(); ++i) {
215 
216  string id_seg_mps = split[i];
217  vector<string> split_id_seg_mps;
218  Tokenizer toker(id_seg_mps, ":");
219  toker.ToVector(split_id_seg_mps);
220 
221  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
222  string segName = split_id_seg_mps[1];
223  string mpsFile = split_id_seg_mps[2];
224 
225  Segment *seg = top->getSegment(segId);
226  if (seg->getName() != segName) {
227  CTP_LOG(logERROR,*(thread->getLogger()))
228  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
229  << " maltagged as " << segName << ". Skip job ..." << flush;
230  throw std::runtime_error("Input does not match topology.");
231  }
232 
233  qmSegs.push_back(seg);
234  qmSegMps.push_back(mpsFile);
235  }
236 
237  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
238 }
