38BOOST_AUTO_TEST_CASE(gw_full) {
39 libint2::initialize();
40 Eigen::VectorXd mo_eigenvalues = Eigen::VectorXd::Zero(17);
41 mo_eigenvalues << -10.6784, -0.746424, -0.394948, -0.394948, -0.394948,
42 0.165212, 0.227713, 0.227713, 0.227713, 0.763971, 0.763971, 0.763971,
43 1.05054, 1.13372, 1.13372, 1.13372, 1.72964;
44 Eigen::MatrixXd mo_eigenvectors =
45 votca::tools::EigenIO_MatrixMarket::ReadMatrix(
46 std::string(XTP_TEST_DATA_FOLDER) + "/gw/mo_eigenvectors.mm");
47
48 Eigen::MatrixXd vxc = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
49 std::string(XTP_TEST_DATA_FOLDER) + "/gw/vxc.mm");
50
51 Orbitals orbitals;
52 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
53 "/gw/molecule.xyz");
54 BasisSet basis;
55 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/gw/3-21G.xml");
56 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) + "/gw/3-21G.xml");
57 AOBasis aobasis;
58 aobasis.Fill(basis, orbitals.QMAtoms());
59 orbitals.setNumberOfOccupiedLevels(4);
60 orbitals.MOs().eigenvectors() = mo_eigenvectors;
61 Logger log;
62 TCMatrix_gwbse Mmn;
63 Mmn.Initialize(aobasis.AOBasisSize(), 0, 16, 0, 16);
64 Mmn.Fill(aobasis, aobasis, mo_eigenvectors);
65
66 GW::options opt;
67 opt.ScaHFX = 0;
68 opt.homo = 4;
69 opt.qpmax = 16;
70 opt.qpmin = 0;
71 opt.rpamax = 16;
72 opt.rpamin = 0;
73 opt.gw_sc_max_iterations = 1;
74 opt.eta = 1e-3;
75 opt.sigma_integration = "ppm";
76 opt.reset_3c = 5;
77 opt.qp_solver = "grid";
78 opt.qp_grid_steps = 601;
79 opt.qp_grid_spacing = 0.005;
80 opt.gw_mixing_order = 0;
81 opt.gw_mixing_alpha = 0.7;
82 opt.g_sc_limit = 1e-5;
83 opt.g_sc_max_iterations = 50;
84 opt.gw_sc_limit = 1e-5;
85
86 GW gw(log, Mmn, vxc, mo_eigenvalues);
87 gw.configure(opt);
88
89 Eigen::MatrixXd ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
90 std::string(XTP_TEST_DATA_FOLDER) + "/gw/ref.mm");
91
92 gw.CalculateGWPerturbation();
93 Eigen::VectorXd diag = gw.getGWAResults();
94 bool check_diag = ref.diagonal().isApprox(diag, 1e-4);
95 if (!check_diag) {
96 cout << "GW energies" << endl;
97 cout << diag << endl;
98 cout << "GW energies ref" << endl;
99 cout << ref.diagonal() << endl;
100 }
101 BOOST_CHECK_EQUAL(check_diag, true);
102
103 gw.CalculateHQP();
104 Eigen::MatrixXd offdiag = gw.getHQP();
105
106 bool check_offdiag = ref.isApprox(offdiag, 1e-4);
107 if (!check_offdiag) {
108 cout << "GW energies" << endl;
109 cout << offdiag << endl;
110 cout << "GW energies ref" << endl;
111 cout << ref << endl;
112 }
113 BOOST_CHECK_EQUAL(check_offdiag, true);
114
115 libint2::finalize();
116}
