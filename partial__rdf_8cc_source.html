87 bool CsgStatApp::EvaluateOptions()
88 {
89  CsgApplication::EvaluateOptions();
90  CheckRequired("options");
91  CheckRequired("subvolume_radius");
92  CheckRequired("trj", "no trajectory file specified");
93 
94  _rdf_calculator.LoadOptions(OptionsMap()["options"].as<string>());
95 
96  _rdf_calculator.SetSubvolRadius(OptionsMap()["subvolume_radius"].as<double>());
97 
98  _rdf_calculator.WriteEvery(_write_every);
99  if(OptionsMap().count("do-blocks"))
100  _rdf_calculator.DoBlocks(true);
101 
102  if(OptionsMap().count("do-vol-corr"))
103  _rdf_calculator.DoVolumeCorrection(true);
104 
105  _rdf_calculator.Initialize();
106  return true;
107 }
