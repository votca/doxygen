76 void EulerMaclaurinGrid::getRadialCutoffs(std::vector<ctp::QMAtom* > _atoms, BasisSet* bs, string gridtype) {

78  map<string, min_exp>::iterator it;
79  std::vector< ctp::QMAtom* > ::iterator ait;
80  std::vector< ctp::QMAtom* > ::iterator bit;
81 
82  double eps = Accuracy[gridtype];
83  double _decaymin;
84  int _lvalue;
85  // cout << endl << " Setting cutoffs for grid type " << gridtype << " eps = " << eps << endl;
86  // 1) is only element based
87  // loop over atoms
88  for (ait = _atoms.begin(); ait < _atoms.end(); ++ait) {
89  // get element type of the atom
90  string name = (*ait)->type;
91  // is this element already in map?
92  it = _element_ranges.find(name);
93  // only proceed, if element data does not exist yet
94  if (it == _element_ranges.end()) {
95  // get first range estimate and add to map
96  min_exp this_atom;
97  double range_max = 0.0;
98  // get the basis set entry for this element
99  Element* _element = bs->getElement(name);
100  // and loop over all shells to figure out minimum decay constant and angular momentum of this function
101  for (Element::ShellIterator its = _element->firstShell(); its != _element->lastShell(); its++) {
102  Shell* shell = (*its);
103  _decaymin = 1e7;
104  _lvalue = 0;
105  int _lmax = shell->getLmax();
106  for (Shell::GaussianIterator itg = shell->firstGaussian(); itg != shell->lastGaussian(); itg++) {
107  GaussianPrimitive* gaussian = *itg;
108  double _decay = gaussian->decay;
109  if (_decay < _decaymin) {
110  _decaymin = _decay;
111  _lvalue = _lmax;
112  }
113  }
114  double range = DetermineCutoff(2 * _decaymin, 2 * _lvalue + 2, eps);
115  if (range > range_max) {
116  this_atom.alpha = _decaymin;
117  this_atom.l = _lvalue;
118  this_atom.range = range;
119  range_max = range;
120  }
121  } // shells
122 
123  // cout << "Element " << name << " alpha " << this_atom.alpha << " l " << this_atom.l << " Rcut " << this_atom.range << endl;
124  _element_ranges[name] = this_atom;
125  } // new element
126  } // atoms
127 
128  // calculate overlap matrix
129  AOBasis aobasis;
130  aobasis.AOBasisFill(bs, _atoms);
131  AOOverlap _overlap;
132  // initialize overlap matrix
133  _overlap.Initialize(aobasis.AOBasisSize());
134  // Fill overlap
135  _overlap.Fill(aobasis);
136 
137 
138 
139  // refining by going through all atom combinations
140  // get collapsed index list
141  int atidx = 0;
142  std::vector<int> idxstart;
143  std::vector<int> idxstop;
144  int start = 0;
145  int end = 0;
146  for (AOBasis::AOShellIterator _row = aobasis.firstShell(); _row != aobasis.lastShell(); _row++) {
147 
148  const AOShell* _shell_row = aobasis.getShell(_row);
149 
150  if (_shell_row->getIndex() == atidx) {
151 
152  end += _shell_row->getNumFunc();
153  } else {
154 
155  idxstart.push_back(start);
156  idxstop.push_back(end);
157  atidx++;
158  start = end;
159  end += _shell_row->getNumFunc();
160  }
161  }
162 
163  idxstart.push_back(start);
164  idxstop.push_back(end);
165 
166  int aidx = 0;
167 
168  for (ait = _atoms.begin(); ait < _atoms.end(); ++ait) {
169 
170  int _a_start = idxstart[aidx];
171  int _a_stop = idxstop[aidx];
172 
173  double range_max = 0.0;
174 
175  // get preset values for this atom type
176  double exp_iat = _element_ranges.at((*ait)->type).alpha;
177  int l_iat = _element_ranges.at((*ait)->type).l;
178 
179  vec pos_a = (*ait)->getPos() * tools::conv::bohr2ang;
180 
181 
182  string type_diff;
183  int bidx = 0;
184  for (bit = _atoms.begin(); bit < _atoms.end(); ++bit) {
185 
186  int _b_start = idxstart[bidx];
187  int _b_stop = idxstop[bidx];
188  vec pos_b = (*bit)->getPos() * tools::conv::bohr2ang;
189  // now do some overlap gymnastics
190  double s_max = 10.0;
191  if (aidx != bidx) {
192 
193  // find overlap block of these two atoms
194  ub::matrix<double> _overlapblock = ub::project(_overlap.Matrix(), ub::range(_a_start, _a_stop), ub::range(_b_start, _b_stop));
195  // determine abs max of this block
196  s_max = 0.0;
197  for (unsigned i = 0; i < _overlapblock.size1(); i++) {
198  for (unsigned j = 0; j < _overlapblock.size2(); j++) {
199  s_max = std::max(s_max, std::abs(_overlapblock(i, j)));
200  }
201  }
202  }
203 
204  if (s_max > 1e-5) {
205  // cout << " Atom " << aidx << " neighbor " << bidx << " s-max " << s_max << " exponents " << exp_iat << " and " << _element_ranges.at((*bit)->type).alpha << endl;
206  double range = DetermineCutoff(exp_iat + _element_ranges.at((*bit)->type).alpha, l_iat + _element_ranges.at((*bit)->type).l + 2, eps);
207  // now do some update trickery from Gaussian product formula
208  double dist = abs(pos_b - pos_a);
209  double shift_2g = dist * exp_iat / (exp_iat + _element_ranges.at((*bit)->type).alpha);
210  range += shift_2g;
211 
212 
213  if (aidx != bidx) range += dist;
214 
215  if (range > range_max) {
216  //shiftm_2g = shift_2g;
217  range_max = range;
218  //iat_diff = bidx;
219  type_diff = (*bit)->type;
220  }
221 
222  }
223 
224  bidx++;
225  }
226 
227 
228  if (round(range_max) > _element_ranges.at((*ait)->type).range) {
229  _element_ranges.at((*ait)->type).range = round(range_max);
230  }
231  aidx++;
232  }
233 
234  return;
235  } // getRadialCutoffs

273  void EulerMaclaurinGrid::setGrid(int np, double cutoff, std::vector<double>& point, std::vector<double>& weight ){
274 
275  double alpha = -cutoff/(log(1.0 - pow( (1.0 + double(np))/(2.0 + double(np)),3)) );
276 
277 
278  double factor = 3.0/(1.0+double(np));
279 
280  for ( int i = 0; i < np; i++){

298  double EulerMaclaurinGrid::DetermineCutoff(double alpha, int l, double eps){
299 
300  // determine norm of function
301 
302  /* For a function f(r) = r^k*exp(-alpha*r^2) determine
303  the radial distance r such that the fraction of the
304  function norm that is neglected if the 3D volume
305  integration is terminated at a distance r is less
306  than or equal to eps. */
307 
308 
309  double _cutoff = 1.0; // initial value
310  double _increment = 0.5; // increment
311 
312  while (_increment > 0.01) {
313  double _neglected = getNeglected(alpha,l , _cutoff);
314  // cout << "neglected is " << _neglected << endl;
315  if (_neglected > eps) {
316  _cutoff += _increment;
317  } else {
318  _cutoff -= _increment;
319  if (_cutoff < 0.0) _cutoff = 0.0;
320  _increment = 0.5 * _increment;
321  _cutoff += _increment;
322  }
323  }
324 
325  return _cutoff;
326 
327  }

340  double EulerMaclaurinGrid::RadialIntegral(double alpha, int l, double cutoff){
341 
342  const double pi = boost::math::constants::pi<double>();
343  int ilo = l % 2;
344  double value = 0.0;
345  double valexp;
346  if ( ilo == 0 ){
347  double expo = sqrt(alpha)*cutoff;
348  if ( expo > 40.0 ) {
349  value = 0.0;
350  } else {
351  value = 0.5 * sqrt( pi /alpha ) * erfc(expo);
352  }
353  }
354 
355  double exponent = alpha*cutoff*cutoff;
356  if ( exponent > 500.0 ) {
357  valexp = 0.0;
358  value = 0.0;
359  } else {
360  valexp = exp(-exponent);
361  value = valexp/2.0/alpha;
362  }
363 
364 
365  for ( int i = ilo+2; i <= l; i+=2){
366  value = ((i-1)*value + pow(cutoff,i-1)*valexp)/2.0/alpha;
367  }
368 
369  return value;
