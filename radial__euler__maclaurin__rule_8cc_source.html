327  void EulerMaclaurinGrid::setGrid(int np, double cutoff, std::vector<double>& point, std::vector<double>& weight ){
328 
329  double alpha = -cutoff/(log(1.0 - pow( (1.0 + double(np))/(2.0 + double(np)),3)) );
330 
331 
332  double factor = 3.0/(1.0+double(np));
333 
334  for ( int i = 0; i < np; i++){
335  double q = double(i+1)/(double(np)+1.0);
336  double r = -alpha*log(1.0-pow(q,3));
337  double w = factor * alpha * r*r/( 1.0 - pow(q,3) ) * pow(q,2);
338 
339  point.push_back(r);
340  weight.push_back(w);
341 
342  }

352  double EulerMaclaurinGrid::DetermineCutoff(double alpha, int l, double eps){
353 
354  // determine norm of function
355 
356  /* For a function f(r) = r^k*exp(-alpha*r^2) determine
357  the radial distance r such that the fraction of the
358  function norm that is neglected if the 3D volume
359  integration is terminated at a distance r is less
360  than or equal to eps. */
361 
362 
363  double _cutoff = 1.0; // initial value
364  double _increment = 0.5; // increment
365 
366  while (_increment > 0.01) {
367  double _neglected = getNeglected(alpha,l , _cutoff);
368  // cout << "neglected is " << _neglected << endl;
369  if (_neglected > eps) {
370  _cutoff += _increment;
371  } else {
372  _cutoff -= _increment;
373  if (_cutoff < 0.0) _cutoff = 0.0;
374  _increment = 0.5 * _increment;
375  _cutoff += _increment;
376  }
377  }
378 
379  return _cutoff;
380 
381  }

394  double EulerMaclaurinGrid::RadialIntegral(double alpha, int l, double cutoff){
395 
396  const double pi = boost::math::constants::pi<double>();
397  int ilo = l % 2;
398  double value = 0.0;
399  double valexp;
400  if ( ilo == 0 ){
401  double expo = sqrt(alpha)*cutoff;
402  if ( expo > 40.0 ) {
403  value = 0.0;
404  } else {
405  value = 0.5 * sqrt( pi /alpha ) * erfc(expo);
406  }
407  }
408 
409  double exponent = alpha*cutoff*cutoff;
410  if ( exponent > 500.0 ) {
411  valexp = 0.0;
412  value = 0.0;
413  } else {
414  valexp = exp(-exponent);
415  value = valexp/2.0/alpha;
416  }
417 
418 
419  for ( int i = ilo+2; i <= l; i+=2){
420  value = ((i-1)*value + pow(cutoff,i-1)*valexp)/2.0/alpha;
421  }
422 
423  return value;
