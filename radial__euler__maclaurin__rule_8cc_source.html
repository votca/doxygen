31  std::vector<double> EulerMaclaurinGrid::CalculatePruningIntervals(const std::string & element){
32  std::vector<double> r;
33  // get Bragg-Slater Radius for this element
34  double BSradius = _BraggSlaterRadii.at(element);
35  // row type of element
36  int RowType = _pruning_set.at(element);
37 
38  if ( RowType == 1 ){
39  r.push_back( 0.25 * BSradius );
40  r.push_back( 0.5 * BSradius );
41  r.push_back( 1.0 * BSradius );
42  r.push_back( 4.5 * BSradius );
43  } else if ( RowType == 2 ){
44  r.push_back( 0.1667 * BSradius );
45  r.push_back( 0.5 * BSradius );
46  r.push_back( 0.9 * BSradius );
47  r.push_back( 3.5 * BSradius );
48  } else if ( RowType == 3 ) {
49  r.push_back( 0.1 * BSradius );
50  r.push_back( 0.4 * BSradius );
51  r.push_back( 0.8 * BSradius );
52  r.push_back( 2.5 * BSradius );
53  } else {
54  throw std::runtime_error("EulerMaclaurinGrid::CalculatePruningIntervals:Pruning unsupported for RowType");
55  }
56  return r;
57  }

59  void EulerMaclaurinGrid::FillElementRangeMap(const AOBasis& aobasis, std::vector<QMAtom*>& atoms, double eps){
60  std::map<std::string, min_exp>::iterator it;
61  for (QMAtom* atom:atoms) {
62  std::string name = atom->getType();
63  // is this element already in map?
64  it = _element_ranges.find(name);
65  // only proceed, if element data does not exist yet
66  if (it == _element_ranges.end()) {
67  min_exp this_atom;
68  double range_max = std::numeric_limits<double>::min();
69  double decaymin = std::numeric_limits<double>::max();
70  int lvalue = std::numeric_limits<int>::min();
71  const std::vector<const AOShell*> shells=aobasis.getShellsofAtom(atom->getAtomID());
72  // and loop over all shells to figure out minimum decay constant and angular momentum of this function
73  for (const AOShell* shell:shells) {
74  int lmax = shell->getLmax();
75  if (shell->getMinDecay() < decaymin) {
76  decaymin =shell->getMinDecay();
77  lvalue = lmax;
78  }
79  double range = DetermineCutoff(2 * decaymin, 2 * lvalue + 2, eps);
80  if (range > range_max) {
81  this_atom.alpha = decaymin;
82  this_atom.l = lvalue;
83  this_atom.range = range;
84  range_max = range;
85  }
86  } // shells
87  _element_ranges[name] = this_atom;
88  } // new element
89  } // atoms
90  }

92  void EulerMaclaurinGrid::RefineElementRangeMap(const AOBasis& aobasis, std::vector<QMAtom*>& atoms, double eps){
93  AOOverlap overlap;
94  overlap.Fill(aobasis);
95 
96  // get collapsed index list
97  std::vector<int> idxstart;
98  const std::vector<int>& idxsize = aobasis.getFuncPerAtom();
99  int start = 0;
100  for (int size : idxsize) {
101  idxstart.push_back(start);
102  start += size;
103  }
104  // refining by going through all atom combinations
105  for (unsigned i = 0; i < atoms.size(); ++i) {
106  QMAtom* atom_a = atoms[i];
107  int a_start = idxstart[i];
108  int a_size = idxsize[i];
109  double range_max = std::numeric_limits<double>::min();
110  // get preset values for this atom type
111  double alpha_a = _element_ranges.at(atom_a->getType()).alpha;
112  int l_a = _element_ranges.at(atom_a->getType()).l;
113  const tools::vec& pos_a = atom_a->getPos();
114  // Cannot iterate only over j<i because it is not symmetric due to shift_2g
115  for (unsigned j = 0; j < atoms.size(); ++j) {
116  if (i == j) {
117  continue;
118  }
119  QMAtom* atom_b = atoms[j];
120  int b_start = idxstart[j];
121  int b_size = idxsize[j];
122  const tools::vec& pos_b = atom_b->getPos();
123  // find overlap block of these two atoms
124  Eigen::MatrixXd overlapblock = overlap.Matrix().block(a_start, b_start, a_size, b_size);
125  // determine abs max of this block
126  double s_max = overlapblock.cwiseAbs().maxCoeff();
127 
128  if (s_max > 1e-5) {
129  double range = DetermineCutoff(alpha_a + _element_ranges.at(atom_b->getType()).alpha, l_a + _element_ranges.at(atom_b->getType()).l + 2, eps);
130  // now do some update trickery from Gaussian product formula
131  double dist = tools::abs(pos_b - pos_a);
132  double shift_2g = dist * alpha_a / (alpha_a + _element_ranges.at(atom_b->getType()).alpha);
133  range += (shift_2g + dist);
134  if (range > range_max) {
135  range_max = range;
136  }
137  }
138  }
139  if (std::round(range_max) > _element_ranges.at(atom_a->getType()).range) {
140  _element_ranges.at(atom_a->getType()).range = std::round(range_max);
141  }
142  }
143  }

163  GridContainers::radial_grid EulerMaclaurinGrid::CalculateRadialGridforAtom(const std::string& type, const std::pair<std::string, min_exp>& element) {
164  GridContainers::radial_grid result;
165  int np = getGridParameters(element.first, type);
166  double cutoff = element.second.range;
167  result.radius=Eigen::VectorXd::Zero(np);
168  result.weight=Eigen::VectorXd::Zero(np);
169  double alpha = -cutoff / (log(1.0 - std::pow((1.0 + double(np)) / (2.0 + double(np)), 3)));
170  double factor = 3.0 / (1.0 + double(np));
171 
172  for (int i = 0; i < np; i++) {
173  double q = double(i + 1) / (double(np) + 1.0);
174  double r = -alpha * std::log(1.0 - std::pow(q, 3));
175  double w = factor * alpha * r * r / (1.0 - std::pow(q, 3)) * std::pow(q, 2);
176  result.radius[i]=r;
177  result.weight[i]=w;
178  }
179  return result;
180  }

182  double EulerMaclaurinGrid::DetermineCutoff(double alpha, int l, double eps){
183  // determine norm of function
184  /* For a function f(r) = r^k*exp(-alpha*r^2) determine
185  the radial distance r such that the fraction of the
186  function norm that is neglected if the 3D volume
187  integration is terminated at a distance r is less
188  than or equal to eps. */
189 
190  double cutoff = 1.0; // initial value
191  double increment = 0.5; // increment
192 
193  while (increment > 0.01) {
194  double residual = CalcResidual(alpha,l , cutoff);
195  if (residual > eps) {
196  cutoff += increment;
197  } else {
198  cutoff -= increment;
199  if (cutoff < 0.0) cutoff = 0.0;
200  increment = 0.5 * increment;
201  cutoff += increment;
202  }
203  }
204  return cutoff;
205  }

211  double EulerMaclaurinGrid::RadialIntegral(double alpha, int l, double cutoff){
212  const double pi = boost::math::constants::pi<double>();
213  int ilo = l % 2;
214  double value = 0.0;
215  double valexp;
216  if ( ilo == 0 ){
217  double expo = sqrt(alpha)*cutoff;
218  if ( expo > 40.0 ) {
219  value = 0.0;
220  } else {
221  value = 0.5 * sqrt( pi /alpha ) * std::erfc(expo);
222  }
223  }
224  double exponent = alpha*cutoff*cutoff;
225  if ( exponent > 500.0 ) {
226  valexp = 0.0;
227  value = 0.0;
228  } else {
229  valexp = exp(-exponent);
230  value = valexp/2.0/alpha;
231  }
232  for (int i = ilo+2; i <= l; i+=2){
233  value = ((i-1)*value + std::pow(cutoff,i-1)*valexp)/2.0/alpha;
234  }
235  return value;
236  }
