29 std::vector<double> EulerMaclaurinGrid::CalculatePruningIntervals(
30  const std::string& element) {
31  std::vector<double> r;
32  // get Bragg-Slater Radius for this element
33  double BSradius = _BraggSlaterRadii.at(element);
34  // row type of element
35  Index RowType = _pruning_set.at(element);
36 
37  if (RowType == 1) {
38  r.push_back(0.25 * BSradius);
39  r.push_back(0.5 * BSradius);
40  r.push_back(1.0 * BSradius);
41  r.push_back(4.5 * BSradius);
42  } else if (RowType == 2) {
43  r.push_back(0.1667 * BSradius);
44  r.push_back(0.5 * BSradius);
45  r.push_back(0.9 * BSradius);
46  r.push_back(3.5 * BSradius);
47  } else if (RowType == 3) {
48  r.push_back(0.1 * BSradius);
49  r.push_back(0.4 * BSradius);
50  r.push_back(0.8 * BSradius);
51  r.push_back(2.5 * BSradius);
52  } else {
53  throw std::runtime_error(
54  "EulerMaclaurinGrid::CalculatePruningIntervals:Pruning unsupported for "
55  "RowType");
56  }
57  return r;
58 }

60 void EulerMaclaurinGrid::FillElementRangeMap(const AOBasis& aobasis,
61  const QMMolecule& atoms,
62  double eps) {
63  std::map<std::string, min_exp>::iterator it;
64  for (const QMAtom& atom : atoms) {
65  std::string name = atom.getElement();
66  // is this element already in map?
67  it = _element_ranges.find(name);
68  // only proceed, if element data does not exist yet
69  if (it == _element_ranges.end()) {
70  min_exp this_atom;
71  double range_max = std::numeric_limits<double>::min();
72  double decaymin = std::numeric_limits<double>::max();
73  Index lvalue = std::numeric_limits<Index>::min();
74  const std::vector<const AOShell*> shells =
75  aobasis.getShellsofAtom(atom.getId());
76  // and loop over all shells to figure out minimum decay constant and
77  // angular momentum of this function
78  for (const AOShell* shell : shells) {
79  Index lmax = shell->getLmax();
80  if (shell->getMinDecay() < decaymin) {
81  decaymin = shell->getMinDecay();
82  lvalue = lmax;
83  }
84  double range = DetermineCutoff(2 * decaymin, 2 * lvalue + 2, eps);
85  if (range > range_max) {
86  this_atom.alpha = decaymin;
87  this_atom.l = lvalue;
88  this_atom.range = range;
89  range_max = range;
90  }
91  } // shells
92  _element_ranges[name] = this_atom;
93  } // new element
94  } // atoms
95 }

97 void EulerMaclaurinGrid::RefineElementRangeMap(const AOBasis& aobasis,
98  const QMMolecule& atoms,
99  double eps) {
100  AOOverlap overlap;
101  overlap.Fill(aobasis);
102 
103  // get collapsed index list
104  std::vector<Index> idxstart;
105  const std::vector<Index>& idxsize = aobasis.getFuncPerAtom();
106  Index start = 0;
107  for (Index size : idxsize) {
108  idxstart.push_back(start);
109  start += size;
110  }
111  // refining by going through all atom combinations
112  for (Index i = 0; i < atoms.size(); ++i) {
113  const QMAtom& atom_a = atoms[i];
114  Index a_start = idxstart[i];
115  Index a_size = idxsize[i];
116  double range_max = std::numeric_limits<double>::min();
117  // get preset values for this atom type
118  double alpha_a = _element_ranges.at(atom_a.getElement()).alpha;
119  Index l_a = _element_ranges.at(atom_a.getElement()).l;
120  const Eigen::Vector3d& pos_a = atom_a.getPos();
121  // Cannot iterate only over j<i because it is not symmetric due to shift_2g
122  for (Index j = 0; j < atoms.size(); ++j) {
123  if (i == j) {
124  continue;
125  }
126  const QMAtom& atom_b = atoms[j];
127  Index b_start = idxstart[j];
128  Index b_size = idxsize[j];
129  const Eigen::Vector3d& pos_b = atom_b.getPos();
130  // find overlap block of these two atoms
131  Eigen::MatrixXd overlapblock =
132  overlap.Matrix().block(a_start, b_start, a_size, b_size);
133  // determine abs max of this block
134  double s_max = overlapblock.cwiseAbs().maxCoeff();
135 
136  if (s_max > 1e-5) {
137  double range = DetermineCutoff(
138  alpha_a + _element_ranges.at(atom_b.getElement()).alpha,
139  l_a + _element_ranges.at(atom_b.getElement()).l + 2, eps);
140  // now do some update trickery from Gaussian product formula
141  double dist = (pos_b - pos_a).norm();
142  double shift_2g =
143  dist * alpha_a /
144  (alpha_a + _element_ranges.at(atom_b.getElement()).alpha);
145  range += (shift_2g + dist);
146  if (range > range_max) {
147  range_max = range;
148  }
149  }
150  }
151  if (std::round(range_max) > _element_ranges.at(atom_a.getElement()).range) {
152  _element_ranges.at(atom_a.getElement()).range = std::round(range_max);
153  }
154  }
155 }

180 GridContainers::radial_grid EulerMaclaurinGrid::CalculateRadialGridforAtom(
181  const std::string& type, const std::pair<std::string, min_exp>& element) {
182  GridContainers::radial_grid result;
183  Index np = getGridParameters(element.first, type);
184  double cutoff = element.second.range;
185  result.radius = Eigen::VectorXd::Zero(np);
186  result.weight = Eigen::VectorXd::Zero(np);
187  double alpha =
188  -cutoff /
189  (log(1.0 - std::pow((1.0 + double(np)) / (2.0 + double(np)), 3)));
190  double factor = 3.0 / (1.0 + double(np));
191 
192  for (Index i = 0; i < np; i++) {
193  double q = double(i + 1) / (double(np) + 1.0);
194  double r = -alpha * std::log(1.0 - std::pow(q, 3));
195  double w = factor * alpha * r * r / (1.0 - std::pow(q, 3)) * std::pow(q, 2);
196  result.radius[i] = r;
197  result.weight[i] = w;
198  }
199  return result;
200 }

202 double EulerMaclaurinGrid::DetermineCutoff(double alpha, Index l, double eps) {
203  // determine norm of function
204  /* For a function f(r) = r^k*exp(-alpha*r^2) determine
205  the radial distance r such that the fraction of the
206  function norm that is neglected if the 3D volume
207  integration is terminated at a distance r is less
208  than or equal to eps. */
209 
210  double cutoff = 1.0; // initial value
211  double increment = 0.5; // increment
212 
213  while (increment > 0.01) {
214  double residual = CalcResidual(alpha, l, cutoff);
215  if (residual > eps) {
216  cutoff += increment;
217  } else {
218  cutoff -= increment;
219  if (cutoff < 0.0) {
220  cutoff = 0.0;
221  }
222  increment = 0.5 * increment;
223  cutoff += increment;
224  }
225  }
226  return cutoff;
227 }

229 double EulerMaclaurinGrid::CalcResidual(double alpha, Index l, double cutoff) {
230  return RadialIntegral(alpha, l + 2, cutoff) /
231  RadialIntegral(alpha, l + 2, 0.0);

234 double EulerMaclaurinGrid::RadialIntegral(double alpha, Index l,
235  double cutoff) {
236  const double pi = boost::math::constants::pi<double>();
237  Index ilo = l % 2;
238  double value = 0.0;
239  double valexp;
240  if (ilo == 0) {
241  double expo = std::sqrt(alpha) * cutoff;
242  if (expo <= 40.0) {
243  value = 0.5 * std::sqrt(pi / alpha) * std::erfc(expo);
244  }
245  }
246  double exponent = alpha * cutoff * cutoff;
247  if (exponent > 500.0) {
248  valexp = 0.0;
249  value = 0.0;
250  } else {
251  valexp = std::exp(-exponent);
252  value = valexp / 2.0 / alpha;
253  }
254  for (Index i = ilo + 2; i <= l; i += 2) {
255  value = (double(i - 1) * value + std::pow(cutoff, i - 1) * valexp) / 2.0 /
256  alpha;
257  }
258  return value;
259 }
