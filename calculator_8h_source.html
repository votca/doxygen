106 inline void Calculator::UpdateWithDefaults(Property &options,
107  std::string package) {
108 
109  // copy options from the object supplied by the Application
110  std::string id = Identify();
111  Property options_id = options.get("options." + id);
112 
113  // add default values if specified in VOTCASHARE
114  char *votca_share = getenv("VOTCASHARE");
115  if (votca_share == NULL)
116  throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
117  // load the xml description of the calculator (with defaults and test values)
118  std::string xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/") +
119  package + std::string("/xml/") + id +
120  std::string(".xml");
121 
122  Property defaults, defaults_all;
123  defaults_all.LoadFromXML(xmlFile);
124  defaults = defaults_all.get("options." + id);
125 
126  // if a value not given or a tag not present, provide default values
127  AddDefaults(options_id, defaults);
128 
129  // output calculator options
130  std::string indent(" ");
131  int level = 1;
132  votca::tools::PropertyIOManipulator IndentedText(PropertyIOManipulator::TXT,
133  level, indent);
134  if (tools::globals::verbose) {
135  std::cout << "\n... ... options\n"
136  << IndentedText << options_id << "... ... options\n"
137  << std::flush;
138  }
139 }

141 inline void Calculator::AddDefaults(Property &p, const Property &defaults) {
142 
143  for (const Property &prop : defaults) {
144  std::string name = prop.path() + "." + prop.name();
145 
146  Property rootp = *p.begin();
147  if (prop.hasAttribute("default")) {
148  if (rootp.exists(name)) {
149  if (rootp.HasChildren()) rootp.value() = prop.value();
150  } else {
151  rootp.add(prop.name(), prop.value());
152  }
153  }
154  AddDefaults(p, prop);
155  }
156 }
