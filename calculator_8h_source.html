140 inline void Calculator::AddDefaults( votca::tools::Property &p, votca::tools::Property &defaults ) {
141 
142  for(std::list<votca::tools::Property>::iterator iter = defaults.begin(); iter!=defaults.end(); ++iter) {
143  std::string name = (*iter).path() + "." + (*iter).name();
144 
145  votca::tools::Property rootp = *p.begin();
146  if ( (*iter).hasAttribute("default") ) {
147  if ( rootp.exists( name ) ) {
148  //std::cout << "E " << rootp.value() << std::endl;

152  rootp.add((*iter).name(), (*iter).value());
153  }
154  }
155  AddDefaults( p, (*iter) );
156  }
