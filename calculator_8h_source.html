108  std::string package) {
109 
110  // copy options from the object supplied by the Application
111  std::string id = Identify();
112  votca::tools::Property _options = options->get("options." + id);
113 
114  // add default values if specified in VOTCASHARE
115  char *votca_share = getenv("VOTCASHARE");
116  if (votca_share == NULL)
117  throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
118  // load the xml description of the calculator (with defaults and test values)
119  std::string xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/") +
120  package + std::string("/xml/") + id +
121  std::string(".xml");
122 
123  votca::tools::Property defaults, _defaults;
124  votca::tools::load_property_from_xml(_defaults, xmlFile);
125  defaults = _defaults.get("options." + id);
126 
127  // std::cout << _options;

130  // if a value not given or a tag not present, provide default values

145 inline void Calculator::AddDefaults(votca::tools::Property &p,
146  votca::tools::Property &defaults) {
147 
148  for (std::list<votca::tools::Property>::iterator iter = defaults.begin();
149  iter != defaults.end(); ++iter) {
150  std::string name = (*iter).path() + "." + (*iter).name();
151 
152  votca::tools::Property rootp = *p.begin();
153  if ((*iter).hasAttribute("default")) {
154  if (rootp.exists(name)) {
155  // std::cout << "E " << rootp.value() << std::endl;
156  if (rootp.HasChilds()) rootp.value() = (*iter).value();
157  } else {
158  // std::cout << "N " << (*iter).name() << std::endl;
159  rootp.add((*iter).name(), (*iter).value());
160  }
161  }
162  AddDefaults(p, (*iter));
163  }
164 }
