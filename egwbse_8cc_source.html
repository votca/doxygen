119  void EGWBSE::WriteJobFile(ctp::Topology *top) {

121  cout << endl << "... ... Writing job file: " << flush;

123  ofs.open(_jobfile.c_str(), std::ofstream::out);
124  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);

126  ofs << "<jobs>" << endl;

128 
129  int jobCount = 0;
130 
131  std::vector<ctp::Segment*> segments=top->Segments();

133  for (sit = segments.begin(); sit != segments.end(); ++sit) {
134 
135  int id = ++jobCount;
136  string tag = "";
137 
138  Property Input;

141  pSegment->setAttribute<string>("type", (*sit)->getName() );
142  pSegment->setAttribute<int>("id", (*sit)->getId() );
143  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE );
144  job.ToStream(ofs,"xml");
145  }
146 
147  // CLOSE STREAM
148  ofs << "</jobs>" << endl;
149  ofs.close();

151  cout << jobCount << " jobs" << flush;

162

164  int segId = lSegments.front()->getAttribute<int>("id");

183  string orb_file = (format("%1%_%2%%3%") % "molecule" % segId % ".orb").str();
184 
185  string _mol_dir = ( format("%1%%2%%3%") % "molecule" % "_" % segId ).str();
186  string _package_append = _package + "_gwbse";
187  string _qmpackage_work_dir = (arg_path / egwbse_work_dir / _package_append / frame_dir / _mol_dir).c_str();
188 
189 
190  // get the corresponding object from the QMPackageFactory
191  QMPackage *_qmpackage = QMPackages().Create(_package);

193  _qmpackage->setLog(pLog);
194  // set the run dir
195  _qmpackage->setRunDir(_qmpackage_work_dir);
196  // get the package options
197  _qmpackage->Initialize(&_package_options);
198 
199 
200  Property _job_summary;
201  Property *_output_summary = &_job_summary.add("output", "");
202  Property *_segment_summary = &_output_summary->add("segment", "");
203  string segName = seg->getName();
204  segId = seg->getId();
205  _segment_summary->setAttribute("id", segId);
206  _segment_summary->setAttribute("type", segName);
207 
208 
209  // different tasks
210 
211  // create input for DFT
212  if ( _do_dft_input ){
213  boost::filesystem::create_directories( _qmpackage_work_dir );
214  _qmpackage->WriteInputFile( segments );
215  }
216 
217  bool _run_dft_status = false;
218  if (_do_dft_run) {
219  _run_dft_status = _qmpackage->Run();
220  if (!_run_dft_status) {
221  output += "run failed; ";
222  CTP_LOG(ctp::logERROR, *pLog) << _qmpackage->getPackageName() << " run failed" << flush;
223  cout << *pLog;
224  jres.setOutput(output);
225  jres.setStatus(ctp::Job::FAILED);
226  delete _qmpackage;
227  return jres;
228  }
229  }
230 
231  // parse the log/orbitals files
232  bool _parse_log_status = false;
233  bool _parse_orbitals_status = false;
234  if (_do_dft_parse) {
235  _parse_log_status = _qmpackage->ParseLogFile(&_orbitals);
236 
237  if (!_parse_log_status) {
238  output += "log incomplete; ";
239  CTP_LOG(ctp::logERROR, *pLog) << "LOG parsing failed" << flush;
240  cout << *pLog;
241  jres.setOutput(output);
242  jres.setStatus(ctp::Job::FAILED);
243  delete _qmpackage;
244  return jres;
245  }
246 
247  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile(&_orbitals);
248 
249  if (!_parse_orbitals_status) {

252  cout << *pLog;
253  jres.setOutput(output);
254  jres.setStatus(ctp::Job::FAILED);
255  delete _qmpackage;

258 
259 
260  } // end of the parse orbitals/log
261  else {
262 
263  // load the DFT data from serialized orbitals object
264  string DIR = egwbse_work_dir + "/molecules_gwbse/" + frame_dir;
265  std::ifstream ifs((DIR + "/" + orb_file).c_str());
266  if (_do_gwbse){

271  }

282

284  _gwbse.Initialize(&_gwbse_options);
285 
286 
287  // define own logger for GW-BSE that is written into a runFolder logfile
288  ctp::Logger gwbse_logger(ctp::logDEBUG);
289  gwbse_logger.setMultithreading(false);

291  gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...") ).str());
292  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...") ).str());
293  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...") ).str());

307 
308 
309 
310  }
311 
312 
313 
314  if (_do_esp){
315  string mps_file="";

317  esp2multipole.Initialize(&_esp_options);
318  string ESPDIR="MP_FILES/"+frame_dir+"/"+esp2multipole.GetIdentifier();

323  boost::filesystem::create_directories( ESPDIR );

326

328 
329  _segment_summary->add("partialcharges", (ESPDIR + "/" + mps_file).c_str());
330  }
331  CTP_LOG(ctp::logINFO, *pLog) << ctp::TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
332 
333  if(_do_dft_parse ||_do_gwbse ){
334  CTP_LOG(ctp::logDEBUG, *pLog) << "Saving data to " << orb_file << flush;
335  string DIR = egwbse_work_dir + "/molecules_gwbse/" + frame_dir;
336  boost::filesystem::create_directories(DIR);
337  std::ofstream ofs((DIR + "/" + orb_file).c_str());
338  boost::archive::binary_oarchive oa(ofs);

345  // output of the JOB
