213  Job::JobResult EGWBSE::EvalJob(Topology *top, Job *job, QMThread *opThread) {
214 
215 
216  Orbitals _orbitals;
217  Job::JobResult jres = Job::JobResult();
218  Property _job_input = job->getInput();
219  list<Property*> lSegments = _job_input.Select("segment");
220 
221  vector < Segment* > segments;
222  int segId = lSegments.front()->getAttribute<int>("id");
223  string segType = lSegments.front()->getAttribute<string>("type");
224 
225  Segment *seg = top->getSegment(segId);
226  assert(seg->getName() == segType);
227  segments.push_back(seg);
228 
229  Logger* pLog = opThread->getLogger();
230 
231  LOG(logINFO, *pLog) << TimeStamp() << " Evaluating site " << seg->getId() << flush;
232 
233 
234  string output;
235 
236  // directories and files
237  path arg_path;
238  string egwbse_work_dir = "OR_FILES";
239 
240  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
241  string orb_file = (format("%1%_%2%%3%") % "molecule" % segId % ".orb").str();
242 
243  string _mol_dir = ( format("%1%%2%%3%") % "molecule" % "_" % segId ).str();
244  string _package_append = _package + "_gwbse";
245  string _qmpackage_work_dir = (arg_path / egwbse_work_dir / _package_append / frame_dir / _mol_dir).c_str();
246 
247 
248  // get the corresponding object from the QMPackageFactory
249  QMPackage *_qmpackage = QMPackages().Create(_package);
250  // set a log file for the package
251  _qmpackage->setLog(pLog);
252  // set the run dir
253  _qmpackage->setRunDir(_qmpackage_work_dir);
254  // get the package options
255  _qmpackage->Initialize(&_package_options);
256 
257 
258  Property _job_summary;
259  Property *_output_summary = &_job_summary.add("output", "");
260  Property *_segment_summary = &_output_summary->add("segment", "");
261  string segName = seg->getName();
262  segId = seg->getId();
263  _segment_summary->setAttribute("id", segId);
264  _segment_summary->setAttribute("type", segName);
265 
266 
267  // different tasks
268 
269  // create input for DFT
270  if ( _do_dft_input ){
271  boost::filesystem::create_directories( _qmpackage_work_dir );
272  _qmpackage->WriteInputFile( segments );
273  }
274 
275  bool _run_dft_status = false;
276  if (_do_dft_run) {
277  _run_dft_status = _qmpackage->Run();
278  if (!_run_dft_status) {
279  output += "run failed; ";
280  LOG(logERROR, *pLog) << _qmpackage->getPackageName() << " run failed" << flush;
281  cout << *pLog;
282  jres.setOutput(output);
283  jres.setStatus(Job::FAILED);
284  delete _qmpackage;
285  return jres;
286  }
287  }
288 
289  // parse the log/orbitals files
290  bool _parse_log_status = false;
291  bool _parse_orbitals_status = false;
292  if (_do_dft_parse) {
293  _parse_log_status = _qmpackage->ParseLogFile(&_orbitals);
294 
295  if (!_parse_log_status) {
296  output += "log incomplete; ";
297  LOG(logERROR, *pLog) << "LOG parsing failed" << flush;
298  cout << *pLog;
299  jres.setOutput(output);
300  jres.setStatus(Job::FAILED);
301  delete _qmpackage;
302  return jres;
303  }
304 
305  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile(&_orbitals);
306 
307  if (!_parse_orbitals_status) {
308  output += "orbfile failed; ";
309  LOG(logERROR, *pLog) << "Orbitals parsing failed" << flush;
310  cout << *pLog;
311  jres.setOutput(output);
312  jres.setStatus(Job::FAILED);
313  delete _qmpackage;
314  return jres;
315  }
316  } // end of the parse orbitals/log
317  else {
318 
319  // load the DFT data from serialized orbitals object
320  string DIR = egwbse_work_dir + "/molecules_gwbse/" + frame_dir;
321  std::ifstream ifs((DIR + "/" + orb_file).c_str());
322  if (_do_gwbse){
323  LOG(logDEBUG, *pLog) << TimeStamp() << " Loading DFT data from " << DIR << "/" << orb_file << flush;
324  }
325  else {
326  LOG(logDEBUG, *pLog) << TimeStamp() << " Loading data from " << DIR << "/" << orb_file << flush;
327  }
328  boost::archive::binary_iarchive ia(ifs);
329  ia >> _orbitals;
330  ifs.close();
331  }
332 
333 
334  if (_do_gwbse) {
335 
336 
337 
338 
339 
340  //cout << "hallo" <<endl;
341 
342 
343  GWBSE _gwbse=GWBSE(&_orbitals);
344  // cout << "hallo1" <<endl;
345  _gwbse.setLogger(pLog);
346  _gwbse.Initialize(&_gwbse_options);
347  // _gwbse.setLogger(pLog);
348  //cout << "hallo2" <<endl;
349 
350  // define own logger for GW-BSE that is written into a runFolder logfile
351  Logger gwbse_logger(logDEBUG);
352  gwbse_logger.setMultithreading(false);
353  _gwbse.setLogger(&gwbse_logger);
354  gwbse_logger.setPreface(logINFO, (format("\nGWBSE INF ...") ).str());
355  gwbse_logger.setPreface(logERROR, (format("\nGWBSE ERR ...") ).str());
356  gwbse_logger.setPreface(logWARNING, (format("\nGWBSE WAR ...") ).str());
357  gwbse_logger.setPreface(logDEBUG, (format("\nGWBSE DBG ...") ).str());
358 
359 
360  //bool _evaluate = _gwbse.Evaluate(&_orbitals);
361  _gwbse.Evaluate();
362  _gwbse.addoutput(_segment_summary);
363  // write logger to log file
364  std::ofstream ofs;
365  string gwbse_logfile = _qmpackage_work_dir + "/gwbse.log";
366  ofs.open(gwbse_logfile.c_str(), std::ofstream::out);
367  if (!ofs.is_open()) {
368  throw runtime_error("Bad file handle: " + gwbse_logfile);
369  }
370  ofs << gwbse_logger << endl;
371  ofs.close();
372 
373 
374 
375  }
376 
377 
378 
379  if (_do_esp){
380  string mps_file="";
381  Esp2multipole esp2multipole=Esp2multipole(pLog);;
382  esp2multipole.Initialize(&_esp_options);
383  string ESPDIR="MP_FILES/"+frame_dir+"/"+esp2multipole.GetIdentifier();
384  esp2multipole.Extractingcharges(_orbitals);
385 
386 
387  mps_file = (format("%1%_%2%_%3%.mps") % segType % segId % esp2multipole.GetIdentifier() ).str();
388  boost::filesystem::create_directories( ESPDIR );
389  esp2multipole.WritetoFile((ESPDIR + "/" + mps_file).c_str(),Identify());
390 
391 
392  LOG(logDEBUG, *pLog) << "Written charges to " << (ESPDIR + "/" + mps_file).c_str() << flush;
393 
394  _segment_summary->add("partialcharges", (ESPDIR + "/" + mps_file).c_str());
395  }
396  LOG(logINFO, *pLog) << TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
397 
398  if(_do_dft_parse ||_do_gwbse ){
399  LOG(logDEBUG, *pLog) << "Saving data to " << orb_file << flush;
400  string DIR = egwbse_work_dir + "/molecules_gwbse/" + frame_dir;
401  boost::filesystem::create_directories(DIR);
402  std::ofstream ofs((DIR + "/" + orb_file).c_str());
403  boost::archive::binary_oarchive oa(ofs);
404 
405 
406  oa << _orbitals;
407  ofs.close();
408  }
409 
410  // output of the JOB
411  jres.setOutput(_job_summary);
412  jres.setStatus(Job::COMPLETE);
413 
414  // dump the LOG
415 
416  return jres;
417  }
