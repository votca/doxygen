156 void BeadStructure::AddBead(const T &bead) {
157  if (beads_.count(bead.getId())) {
158  std::string err = "Cannot add bead with Id ";
159  err += std::to_string(bead.getId());
160  err += " because each bead must have a unique Id and a bead with that Id ";
161  err += "already exists within the beadstructure";
162  throw std::invalid_argument(err);
163  }
164  UpdateOnBeadAddition_();
165  size_t numberOfBeads = beads_.size();
166  BeadInfo bead_info;
167  bead_info.mass = bead.getMass();
168  bead_info.name = bead.getName();
169 
170  beads_[bead.getId()] = bead_info;
171 
172  if (numberOfBeads != beads_.size()) {
173  single_structureUpToDate_ = false;
174  graphUpToDate = false;
175  structureIdUpToDate = false;
176  }
177 }
