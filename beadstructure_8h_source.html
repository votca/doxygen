134 template <class T>
135 void BeadStructure<T>::InitializeGraph_() {
136  if (!graphUpToDate) {
137  std::vector<TOOLS::Edge> connections_vector;
138  for (const TOOLS::Edge &edge : connections_) {
139  connections_vector.push_back(edge);
140  }
141 
142  for (std::pair<const Index, T *> &id_bead_ptr_pair : beads_) {

144  BaseBeadToGraphNode_(id_bead_ptr_pair.second);

181 template <class T>
182 void BeadStructure<T>::AddBead(T *bead) {
183  if (beads_.count(bead->getId())) {
184  std::string err = "Cannot add bead with Id ";
185  err += std::to_string(bead->getId());
186  err += " because each bead must have a unique Id and a bead with that Id ";
187  err += "already exists within the beadstructure";
188  throw std::invalid_argument(err);
189  }
190  size_t numberOfBeads = beads_.size();
191  beads_[bead->getId()] = bead;
192 
193  if (numberOfBeads != beads_.size()) {
194  single_structureUpToDate_ = false;
195  graphUpToDate = false;
196  structureIdUpToDate = false;
197  }
198 }

200 template <class T>
201 void BeadStructure<T>::ConnectBeads(Index bead1_id, Index bead2_id) {
202  if (!(beads_.count(bead1_id)) || !(beads_.count(bead2_id))) {
203  std::string err =
204  "Cannot connect beads in bead structure that do not exist";
205  throw std::invalid_argument(err);
206  }
207  if (bead1_id == bead2_id) {
208  std::string err = "Beads cannot be self-connected";
209  throw std::invalid_argument(err);
210  }
211  size_t numberOfConnections = connections_.size();
212  connections_.insert(TOOLS::Edge(bead1_id, bead2_id));
213  if (numberOfConnections != connections_.size()) {
214  single_structureUpToDate_ = false;
215  graphUpToDate = false;
216  structureIdUpToDate = false;
217  }

226 template <class T>
227 bool BeadStructure<T>::isSingleStructure() {
228 
229  InitializeGraph_();
230  if (single_structureUpToDate_ == false) {
231  std::vector<Index> vertices = graph_.getVertices();
232  if (vertices.size() == 0) {
233  single_structure_ = false;
234  return single_structure_;
235  }
236  // Choose first vertex that is actually in the graph as the starting vertex
237  TOOLS::Graph_BF_Visitor gv_breadth_first;
238  gv_breadth_first.setStartingVertex(vertices.at(0));
239  if (!singleNetwork(graph_, gv_breadth_first)) {
240  single_structure_ = false;
241  return single_structure_;
242  }
243  if (beads_.size() == 0) {
244  single_structure_ = false;
245  return single_structure_;
246  }
247  if (vertices.size() != beads_.size()) {
248  single_structure_ = false;
249  return single_structure_;
250  }
251  single_structure_ = true;
252  single_structureUpToDate_ = true;
253  }
254  return single_structure_;
255 }

258 bool BeadStructure<T>::isStructureEquivalent(BeadStructure<T> &beadstructure) {
259  if (!structureIdUpToDate) {
260  CalculateStructure_();
261  }
262  if (!beadstructure.structureIdUpToDate) {
263  beadstructure.CalculateStructure_();
