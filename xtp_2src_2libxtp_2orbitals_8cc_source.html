158 std::vector<int> Orbitals::CheckDegeneracy(int level,
159  double energy_difference) const {
160 
161  std::vector<int> result = std::vector<int>(0);
162  if (level > _mo_energies.size()) {
163  throw std::runtime_error(
164  "Level for degeneracy is higher than maximum level");
165  }
166  double MOEnergyLevel = _mo_energies(level);
167 
168  for (int i = 0; i < _mo_energies.size(); ++i) {
169  if (std::abs(_mo_energies(i) - MOEnergyLevel) * tools::conv::hrt2ev <
170  energy_difference) {
171  result.push_back(i);
172  }
173  }
174  return result;
175 }

206 Eigen::MatrixXd Orbitals::DensityMatrixFull(const QMState& state) const {
207  if (state.isTransition()) {
208  return this->TransitionDensityMatrix(state);
209  }
210  Eigen::MatrixXd result = this->DensityMatrixGroundState();
211  if (state.Type().isExciton()) {
212  std::vector<Eigen::MatrixXd> DMAT = DensityMatrixExcitedState(state);
213  result = result - DMAT[0] + DMAT[1]; // Ground state + hole_contribution +
214  // electron contribution
215  } else if (state.Type() == QMStateType::DQPstate) {
216  Eigen::MatrixXd DMATQP = DensityMatrixQuasiParticle(state);
217  if (state.Index() > getHomo()) {
218  result += DMATQP;
219  } else {
220  result -= DMATQP;
221  }
222  } else if (state.Type() != QMStateType::Gstate) {
223  throw std::runtime_error(
224  "DensityMatrixFull does not yet implement QMStateType:" +
225  state.Type().ToLongString());
226  }
227  return result;
228 }

265 Eigen::Vector3d Orbitals::CalcCoM() const {
266  tools::Elements elements;
267 
268  Eigen::Vector3d CoM = Eigen::Vector3d::Zero();
269  double totalmass = 0.0;
270  for (QMAtom* atom : _atoms) {
271  double mass = elements.getMass(atom->getType());
272  totalmass += mass;
273  CoM += mass * atom->getPos().toEigen();
274  }
275  CoM /= totalmass;
276  return CoM;
277 }

279 Eigen::Vector3d Orbitals::CalcElDipole(const QMState& state) {
280  Eigen::Vector3d CoM = CalcCoM();
281  Eigen::Vector3d nuclei_dip = Eigen::Vector3d::Zero();
282  if (!state.isTransition()) {
283  for (QMAtom* atom : _atoms) {
284  nuclei_dip += (atom->getPos().toEigen() - CoM) * atom->getNuccharge();
285  }
286  }
287 
288  BasisSet basis;
289  basis.LoadBasisSet(this->getDFTbasisName());
290  AOBasis aobasis;
291  aobasis.AOBasisFill(basis, _atoms);
292  AODipole dipole;
293  dipole.setCenter(CoM);
294  dipole.Fill(aobasis);
295  Eigen::MatrixXd dmat = this->DensityMatrixFull(state);
296  Eigen::Vector3d electronic_dip;
297  for (int i = 0; i < 3; ++i) {
298  electronic_dip(i) = dmat.cwiseProduct(dipole.Matrix()[i]).sum();
299  }
300  return nuclei_dip - electronic_dip;
301 }

303 Eigen::MatrixXd Orbitals::TransitionDensityMatrix(const QMState& state) const {
304  if (state.Type() != QMStateType::Singlet) {
305  throw runtime_error(
306  "Spin type not known for transition density matrix. Available only for "
307  "singlet");
308  }
309  const Eigen::MatrixXd& BSECoefs = _BSE_singlet_coefficients;
310  if (BSECoefs.cols() < state.Index() + 1 || BSECoefs.rows() < 2) {
311  throw runtime_error("Orbitals object has no information about state:" +
312  state.ToString());
313  }
314 
315  // The Transition dipole is sqrt2 bigger because of the spin, the excited
316  // state is a linear combination of 2 slater determinants, where either alpha
317  // or beta spin electron is excited
318 
319  /*Trying to implement D_{alpha,beta}=
320  * sqrt2*sum_{i}^{occ}sum_{j}^{virt}{BSEcoef(i,j)*MOcoef(alpha,i)*MOcoef(beta,j)}
321  */
322  // c stands for conduction band and thus virtual orbitals
323  // v stand for valence band and thus occupied orbitals
324 
325  Eigen::VectorXd coeffs = BSECoefs.col(state.Index());
326 
327  if (!_useTDA) {
328  coeffs += _BSE_singlet_coefficients_AR.col(state.Index());
329  }
330  coeffs *= std::sqrt(2.0);
331  vc2index index = vc2index(_bse_vmin, _bse_cmin, _bse_ctotal);
332  Eigen::MatrixXd dmatTS =
333  Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
334 
335  for (int i = 0; i < _bse_size; i++) {
336  dmatTS.noalias() += coeffs(i) * _mo_coefficients.col(index.v(i)) *
337  _mo_coefficients.col(index.c(i)).transpose();
338  }
339 
340  return dmatTS;
341 }

356 std::vector<Eigen::MatrixXd> Orbitals::DensityMatrixExcitedState_R(
357  const QMState& state) const {
358  if (!state.Type().isExciton()) {
359  throw runtime_error(
360  "Spin type not known for density matrix. Available are singlet and "
361  "triplet");
362  }
363 
364  const Eigen::MatrixXd& BSECoefs = (state.Type() == QMStateType::Singlet)
365  ? _BSE_singlet_coefficients
366  : _BSE_triplet_coefficients;
367  if (BSECoefs.cols() < state.Index() + 1 || BSECoefs.rows() < 2) {
368  throw runtime_error("Orbitals object has no information about state:" +
369  state.ToString());
370  }
371  /******
372  *
373  * Density matrix for GW-BSE based excitations
374  *
375  * - electron contribution
376  * D_ab = \sum{vc} \sum{c'} A_{vc}A_{vc'} mo_a(c)mo_b(c')
377  *
378  * - hole contribution
379  * D_ab = \sum{vc} \sum{v'} A_{vc}A_{v'c} mo_a(v)mo_b(v')
380  *
381  *
382  * more efficient:
383  *
384  * - electron contribution
385  * D_ab = \sum{c} \sum{c'} mo_a(c)mo_b(c') [ \sum{v} A_{vc}A_{vc'} ]
386  * = \sum{c} \sum{c'} mo_a(c)mo_b(c') A_{cc'}
387  *
388  * - hole contribution
389  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} A_{vc}A_{v'c} ]
390  * = \sum{v} \sum{v'} mo_a(v)mo_b(v') A_{vv'}
391  *
392  */
393 
394  Eigen::VectorXd coeffs = BSECoefs.col(state.Index());
395 
396  std::vector<Eigen::MatrixXd> dmatEX(2);
397  // hole part as matrix products
398  Eigen::MatrixXd occlevels = _mo_coefficients.block(
399  0, _bse_vmin, _mo_coefficients.rows(), _bse_vtotal);
400  dmatEX[0] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
401 
402  // electron part as matrix products
403  Eigen::MatrixXd virtlevels = _mo_coefficients.block(
404  0, _bse_cmin, _mo_coefficients.rows(), _bse_ctotal);
405  dmatEX[1] = virtlevels * CalcAuxMat_cc(coeffs) * virtlevels.transpose();
406 
407  return dmatEX;
408 }

410 Eigen::MatrixXd Orbitals::CalcAuxMat_vv(const Eigen::VectorXd& coeffs) const {
411  Eigen::MatrixXd Mvv = Eigen::MatrixXd::Zero(_bse_vtotal, _bse_vtotal);
412  vc2index index = vc2index(_bse_vmin, _bse_cmin, _bse_ctotal);
413  for (int idx1 = 0; idx1 < _bse_size; idx1++) {
414  int v = index.v(idx1) - _bse_vmin;
415  int c = index.c(idx1) - _bse_cmin;
416 #pragma omp parallel for
417  for (int v2 = 0; v2 < _bse_vtotal; v2++) {
418  int idx2 = index.I(v2 + _bse_vmin, c + _bse_cmin);
419  Mvv(v, v2) += coeffs(idx1) * coeffs(idx2);
420  }
421  }
422  return Mvv;
423 }

425 Eigen::MatrixXd Orbitals::CalcAuxMat_cc(const Eigen::VectorXd& coeffs) const {
426  Eigen::MatrixXd Mcc = Eigen::MatrixXd::Zero(_bse_ctotal, _bse_ctotal);
427  vc2index index = vc2index(_bse_vmin, _bse_cmin, _bse_ctotal);
428  for (int idx1 = 0; idx1 < _bse_size; idx1++) {
429  int v = index.v(idx1) - _bse_vmin;
430  int c = index.c(idx1) - _bse_cmin;
431 #pragma omp parallel for
432  for (int c2 = 0; c2 < _bse_ctotal; c2++) {
433  int idx2 = index.I(v + _bse_vmin, c2 + _bse_cmin);
434  Mcc(c, c2) += coeffs(idx1) * coeffs(idx2);
435  }
436  }
437  return Mcc;
438 }

440 std::vector<Eigen::MatrixXd> Orbitals::DensityMatrixExcitedState_AR(
441  const QMState& state) const {
442 
443  if (!state.Type().isExciton()) {
444  throw runtime_error(
445  "Spin type not known for density matrix. Available are singlet and "
446  "triplet");
447  }
448 
449  const Eigen::MatrixXd& BSECoefs_AR = (state.Type() == QMStateType::Singlet)

453  throw runtime_error("Orbitals object has no information about state:" +

457  *
458  * Density matrix for GW-BSE based excitations
459  *
460  * - electron contribution
461  * D_ab = \sum{vc} \sum{v'} B_{vc}B_{v'c} mo_a(v)mo_b(v')
462  *
463  * - hole contribution
464  * D_ab = \sum{vc} \sum{c'} B_{vc}B_{vc'} mo_a(c)mo_b(c')
465  *
466  *
467  * more efficient:
468  *
469  * - electron contribution
470  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} B_{vc}B_{v'c} ]

476  *
477  */
478 
479  Eigen::VectorXd coeffs = BSECoefs_AR.col(state.Index());

485  // electron part as matrix products
486  Eigen::MatrixXd occlevels = _mo_coefficients.block(
487  0, _bse_vmin, _mo_coefficients.rows(), _bse_vtotal);
488  dmatAR[1] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
489 
490  return dmatAR;

493 Eigen::VectorXd Orbitals::LoewdinPopulation(
494  const Eigen::MatrixXd& densitymatrix, const Eigen::MatrixXd& overlapmatrix,
495  int frag) {
496 
497  Eigen::VectorXd fragmentCharges = Eigen::VectorXd::Zero(2);
498  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es;
499  es.compute(overlapmatrix);
500  Eigen::MatrixXd sqrt = es.operatorSqrt();
501  Eigen::MatrixXd prodmat = sqrt * densitymatrix * sqrt;
502 
503  for (int i = 0; i < frag; i++) {
504  fragmentCharges(0) += prodmat(i, i);
505  }
506  for (int i = frag; i < overlapmatrix.rows(); i++) {
507  fragmentCharges(1) += prodmat(i, i);
508  }
509 
510  return fragmentCharges;
511 }

536 double Orbitals::getExcitedStateEnergy(const QMState& state) const {
537 
538  double omega = 0.0;
539  if (state.isTransition()) {
540  throw std::runtime_error(
541  "Total Energy does not exist for transition state");
542  }
543 
544  if (state.Type() == QMStateType::Singlet) {
545  if (BSESingletEnergies().size() < state.Index() + 1) {
546  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
547  state.ToString() +
548  " which has not been calculated");
549  }
550  omega = BSESingletEnergies()[state.Index()];
551  } else if (state.Type() == QMStateType::Triplet) {
552  if (BSETripletEnergies().size() < state.Index() + 1) {
553  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
554  state.ToString() +
555  " which has not been calculated");
556  }
557  omega = BSETripletEnergies()[state.Index()];
558  } else if (state.Type() == QMStateType::DQPstate) {
559  if (this->QPdiagEnergies().size() < state.Index() + 1 - getGWAmin()) {
560  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
561  state.ToString() +
562  " which has not been calculated");
563  }
564  return QPdiagEnergies()[state.Index() - getGWAmin()];
565  } else if (state.Type() == QMStateType::KSstate) {
566  if (this->MOEnergies().size() < state.Index() + 1) {
567  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
568  state.ToString() +
569  " which has not been calculated");
570  }
571  return QPdiagEnergies()[state.Index()];
572  } else if (state.Type() == QMStateType::PQPstate) {
573  if (this->_QPpert_energies.rows() < state.Index() + 1 - getGWAmin()) {
574  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
575  state.ToString() +
576  " which has not been calculated");
577  }
578  return _QPpert_energies(state.Index() - getGWAmin(), 3);
579  } else {
580  throw std::runtime_error(
581  "GetTotalEnergy only knows states:singlet,triplet,KS,DQP,PQP");
582  }

630 void Orbitals::PrepareDimerGuess(const Orbitals& orbitalsA,
631  const Orbitals& orbitalsB) {
632 
633  // constructing the direct product orbA x orbB
634  int basisA = orbitalsA.getBasisSetSize();
635  int basisB = orbitalsB.getBasisSetSize();
636 
637  int levelsA = orbitalsA.getBasisSetSize();
638  int levelsB = orbitalsB.getBasisSetSize();
639 
640  int electronsA = orbitalsA.getNumberOfAlphaElectrons();
641  int electronsB = orbitalsB.getNumberOfAlphaElectrons();
642 
643  this->MOCoefficients() =
644  Eigen::MatrixXd::Zero(basisA + basisB, levelsA + levelsB);
645 
646  // AxB = | A 0 | // A = [EA, EB] //
647  // | 0 B | // //
648  if (orbitalsA.getDFTbasisName() != orbitalsB.getDFTbasisName()) {
649  throw std::runtime_error("Basissets of Orbitals A and B differ " +
650  orbitalsA.getDFTbasisName() + ":" +
651  orbitalsB.getDFTbasisName());
652  }
653  this->setDFTbasisName(orbitalsA.getDFTbasisName());
654  if (orbitalsA.getECPName() != orbitalsB.getECPName()) {
655  throw std::runtime_error("ECPs of Orbitals A and B differ " +
656  orbitalsA.getECPName() + ":" +
657  orbitalsB.getECPName());
658  }
659  this->setECPName(orbitalsA.getECPName());
660  this->setBasisSetSize(basisA + basisB);
661  this->setNumberOfOccupiedLevels(electronsA + electronsB);
662  this->setNumberOfAlphaElectrons(electronsA + electronsB);
663 
664  this->MOCoefficients().block(0, 0, basisA, levelsA) =
665  orbitalsA.MOCoefficients();
666  this->MOCoefficients().block(basisA, levelsA, basisB, levelsB) =
667  orbitalsB.MOCoefficients();
668 
669  Eigen::VectorXd& energies = this->MOEnergies();
670  energies.resize(levelsA + levelsB);
671 
672  energies.segment(0, levelsA) = orbitalsA.MOEnergies();
673  energies.segment(levelsA, levelsB) = orbitalsB.MOEnergies();
674 
675  this->OrderMOsbyEnergy();
676 
677  return;
678 }
