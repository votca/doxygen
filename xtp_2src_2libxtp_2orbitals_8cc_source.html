86  std::vector<int> Orbitals::CheckDegeneracy(int level, double energy_difference)const{
87 
88  std::vector<int> result=std::vector<int>(0);
89  if(level>_mo_energies.size()){
90  throw std::runtime_error("Level for degeneracy is higher than maximum level");
91  }
92  double MOEnergyLevel =_mo_energies(level);
93 
94  for (int i =0; i < _mo_energies.size(); ++i) {
95  if (std::abs(_mo_energies(i) - MOEnergyLevel) * tools::conv::hrt2ev < energy_difference) {
96  result.push_back(i);
97  }
98  }
99  return result;

130  Eigen::MatrixXd Orbitals::DensityMatrixFull(const QMState& state) const{
131  if(state.isTransition()){
132  return this->TransitionDensityMatrix(state);
133  }
134  Eigen::MatrixXd result=this->DensityMatrixGroundState();
135  if(state.Type().isExciton()){
136  std::vector< Eigen::MatrixXd > DMAT = DensityMatrixExcitedState(state);
137  result=result- DMAT[0] + DMAT[1]; // Ground state + hole_contribution + electron contribution
138  }else if(state.Type()==QMStateType::DQPstate){
139  Eigen::MatrixXd DMATQP=DensityMatrixQuasiParticle(state);
140  if (state.Index() > getHomo()) {
141  result+= DMATQP;
142  } else {
143  result-= DMATQP;
144  }
145  }else if(state.Type()!=QMStateType::Gstate){
146  throw std::runtime_error("DensityMatrixFull does not yet implement QMStateType:"+state.Type().ToLongString());
147  }
148  return result;
149  }

205  Eigen::Vector3d Orbitals::CalcElDipole(const QMState& state) {
206  Eigen::Vector3d CoM=CalcCoM();
207  Eigen::Vector3d nuclei_dip = Eigen::Vector3d::Zero();
208  if (!state.isTransition()) {
209  for (QMAtom* atom : _atoms) {
210  nuclei_dip += (atom->getPos().toEigen() - CoM) * atom->getNuccharge();
211  }
212  }
213 
214  BasisSet basis;
215  basis.LoadBasisSet(this->getDFTbasis());
216  AOBasis aobasis;
217  aobasis.AOBasisFill(basis, _atoms);
218  AODipole dipole;
219  dipole.setCenter(CoM);
220  dipole.Fill(aobasis);
221  Eigen::MatrixXd dmat = this->DensityMatrixFull(state);
222  Eigen::Vector3d electronic_dip;
223  for (int i = 0; i < 3; ++i) {
224  electronic_dip(i) = dmat.cwiseProduct(dipole.Matrix()[i]).sum();
225  }
226  return nuclei_dip - electronic_dip;
227  }

229  Eigen::MatrixXd Orbitals::TransitionDensityMatrix(const QMState& state) const{
230  if (state.Type() != QMStateType::Singlet) {
231  throw runtime_error("Spin type not known for transition density matrix. Available only for singlet");
232  }
233  const MatrixXfd& BSECoefs = _BSE_singlet_coefficients;
234  if(BSECoefs.cols()<state.Index()+1 || BSECoefs.rows()<2){
235  throw runtime_error("Orbitals object has no information about state:"+state.ToString());
236  }
237  Eigen::MatrixXd dmatTS = Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
238  // The Transition dipole is sqrt2 bigger because of the spin, the excited state is a linear combination of 2 slater determinants, where either alpha or beta spin electron is excited
239  double sqrt2 = sqrt(2.0);
240  /*Trying to implement D_{alpha,beta}= sqrt2*sum_{i}^{occ}sum_{j}^{virt}{BSEcoef(i,j)*MOcoef(alpha,i)*MOcoef(beta,j)} */
241  // c stands for conduction band and thus virtual orbitals
242  // v stand for valence band and thus occupied orbitals
243 
244  // indexing info BSE vector index to occupied/virtual orbital
245  Index2MO index=BSEIndex2MOIndex();
246 
247  if (!_useTDA) {
248  const MatrixXfd& BSECoefs_AR = _BSE_singlet_coefficients_AR;
249 #pragma omp parallel for
250  for (int a = 0; a < dmatTS.rows(); a++) {
251  for (int b = 0; b < dmatTS.cols(); b++) {
252  for (int i = 0; i < _bse_size; i++) {
253  int occ = index.I2v[i];
254  int virt =index.I2c[i];
255  dmatTS(a, b) += sqrt2 * (BSECoefs(i, state.Index()) + BSECoefs_AR(i, state.Index())) * _mo_coefficients(a, occ) * _mo_coefficients(b, virt); //check factor 2??
256  }
257  }
258  }
259  } else {
260 
261 #pragma omp parallel for
262  for (int a = 0; a < dmatTS.rows(); a++) {
263  for (int b = 0; b < dmatTS.cols(); b++) {
264  for (int i = 0; i < _bse_size; i++) {
265  int occ = index.I2v[i];
266  int virt =index.I2c[i];
267  dmatTS(a, b) += sqrt2 * BSECoefs(i, state.Index()) * _mo_coefficients(a, occ) * _mo_coefficients(b, virt); //check factor 2??
268  }
269  }
270  }
271 
272  }
273  return dmatTS;
274  }

288  std::vector<Eigen::MatrixXd> Orbitals::DensityMatrixExcitedState_R(const QMState& state) const{
289  if (!state.Type().isExciton()) {
290  throw runtime_error("Spin type not known for density matrix. Available are singlet and triplet");
291  }
292 
293  const MatrixXfd & BSECoefs = (state.Type() == QMStateType::Singlet) ? _BSE_singlet_coefficients : _BSE_triplet_coefficients;
294  if(BSECoefs.cols()<state.Index()+1 || BSECoefs.rows()<2){
295  throw runtime_error("Orbitals object has no information about state:"+state.ToString());
296  }
297  /******
298  *
299  * Density matrix for GW-BSE based excitations
300  *
301  * - electron contribution
302  * D_ab = \sum{vc} \sum{c'} A_{vc}A_{vc'} mo_a(c)mo_b(c')
303  *
304  * - hole contribution
305  * D_ab = \sum{vc} \sum{v'} A_{vc}A_{v'c} mo_a(v)mo_b(v')
306  *
307  *
308  * more efficient:
309  *
310  * - electron contribution
311  * D_ab = \sum{c} \sum{c'} mo_a(c)mo_b(c') [ \sum{v} A_{vc}A_{vc'} ]
312  * = \sum{c} \sum{c'} mo_a(c)mo_b(c') A_{cc'}
313  *
314  * - hole contribution
315  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} A_{vc}A_{v'c} ]
316  * = \sum{v} \sum{v'} mo_a(v)mo_b(v') A_{vv'}
317  *
318  */
319  std::vector<Eigen::MatrixXd > dmatEX;
320  dmatEX.resize(2);
321  dmatEX[0] = Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
322  dmatEX[1] = Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
323  Index2MO index=BSEIndex2MOIndex();
324 
325  // electron assist matrix A_{cc'}
326  Eigen::MatrixXd Acc = Eigen::MatrixXd::Zero(_bse_ctotal, _bse_ctotal);
327  Eigen::MatrixXd Avv = Eigen::MatrixXd::Zero(_bse_vtotal, _bse_vtotal);
328 
329  for (int idx1 = 0; idx1 < _bse_size; idx1++) {
330  int v = index.I2v[idx1];
331  int c = index.I2c[idx1];
332  // electron assist matrix A_{cc'}
333 #pragma omp parallel for
334  for (int c2 = _bse_cmin; c2 <= _bse_cmax; c2++) {
335  int idx2 = (_bse_cmax - _bse_cmin + 1)*(v - _bse_vmin)+(c2 - _bse_cmin);
336  Acc(c - _bse_cmin, c2 - _bse_cmin) += BSECoefs(idx1, state.Index()) * BSECoefs(idx2, state.Index());
337  }
338 
339  // hole assist matrix A_{vv'}
340 #pragma omp parallel for
341  for (int v2 = _bse_vmin; v2 <= _bse_vmax; v2++) {
342  int idx2 = (_bse_cmax - _bse_cmin + 1)*(v2 - _bse_vmin)+(c - _bse_cmin);
343  Avv(v - _bse_vmin, v2 - _bse_vmin) += BSECoefs(idx1, state.Index()) * BSECoefs(idx2, state.Index());
344  }
345  }
346 
347  // hole part as matrix products
348  Eigen::MatrixXd occlevels = _mo_coefficients.block(0, _bse_vmin, _mo_coefficients.rows(), _bse_vtotal);
349  dmatEX[0] = occlevels * Avv * occlevels.transpose();
350 
351  // electron part as matrix products
352  Eigen::MatrixXd virtlevels = _mo_coefficients.block(0, _bse_cmin, _mo_coefficients.rows(), _bse_ctotal);
353  dmatEX[1] = virtlevels * Acc * virtlevels.transpose();
354  return dmatEX;
355  }

359  std::vector<Eigen::MatrixXd > Orbitals::DensityMatrixExcitedState_AR(const QMState& state) const{
360  if (state.Type() != QMStateType::Singlet) {
361  throw runtime_error("Spin type not known for density matrix. Available is singlet");
362  }
363 
364  const MatrixXfd& BSECoefs_AR = _BSE_singlet_coefficients_AR;
365  if(BSECoefs_AR.cols()<state.Index()+1 || BSECoefs_AR.rows()<2){
366  throw runtime_error("Orbitals object has no information about state:"+state.ToString());
367  }
368  /******
369  *
370  * Density matrix for GW-BSE based excitations
371  *
372  * - electron contribution
373  * D_ab = \sum{vc} \sum{v'} B_{vc}B_{v'c} mo_a(v)mo_b(v')
374  *
375  * - hole contribution
376  * D_ab = \sum{vc} \sum{c'} B_{vc}B_{vc'} mo_a(c)mo_b(c')
377  *
378  *
379  * more efficient:
380  *
381  * - electron contribution
382  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} B_{vc}B_{v'c} ]
383  * = \sum{v} \sum{v'} mo_a(v)mo_b(v') B_{vv'}
384  *
385  * - hole contribution
386  * D_ab = \sum{c} \sum{c'} mo_a(c)mo_b(c') [ \sum{v} B_{vc}B_{vc'} ]
387  * = \sum{c} \sum{c'} mo_a(c)mo_b(c') B_{cc'}
388  *
389  */
390  std::vector<Eigen::MatrixXd > dmatAR;
391  dmatAR.resize(2);
392  dmatAR[0] = Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
393  dmatAR[1] = Eigen::MatrixXd::Zero(_basis_set_size, _basis_set_size);
394  Index2MO index=BSEIndex2MOIndex();
395 
396  // hole assist matrix B_{cc'}
397  Eigen::MatrixXd Bcc = Eigen::MatrixXd::Zero(_bse_ctotal, _bse_ctotal);
398  Eigen::MatrixXd Bvv = Eigen::MatrixXd::Zero(_bse_vtotal, _bse_vtotal);
399 
400  for (int idx1 = 0; idx1 < _bse_size; idx1++) {
401  int v = index.I2v[idx1];
402  int c = index.I2c[idx1];
403  // hole assist matrix B_{cc'}
404 #pragma omp parallel for
405  for (int c2 = _bse_cmin; c2 <= _bse_cmax; c2++) {
406  int idx2 = (_bse_cmax - _bse_cmin + 1)*(v - _bse_vmin)+(c2 - _bse_cmin);
407  Bcc(c - _bse_cmin, c2 - _bse_cmin) += BSECoefs_AR(idx1, state.Index()) * BSECoefs_AR(idx2, state.Index());
408  }
409 
410  // electron assist matrix B_{vv'}
411 #pragma omp parallel for
412  for (int v2 = _bse_vmin; v2 <= _bse_vmax; v2++) {
413  int idx2 = (_bse_cmax - _bse_cmin + 1)*(v2 - _bse_vmin)+(c - _bse_cmin);
414  Bvv(v - _bse_vmin, v2 - _bse_vmin) += BSECoefs_AR(idx1, state.Index()) * BSECoefs_AR(idx2, state.Index());
415  }
416  }
417 
418  // hole part as matrix products
419  Eigen::MatrixXd occlevels = _mo_coefficients.block(0, _bse_vmin, _mo_coefficients.rows(), _bse_vtotal);
420  dmatAR[0] = occlevels * Bvv * occlevels.transpose();
421  // electron part as matrix products
422  Eigen::MatrixXd virtlevels = _mo_coefficients.block(0, _bse_cmin, _mo_coefficients.rows(), _bse_ctotal);
423  dmatAR[1] = virtlevels * Bcc * virtlevels.transpose();
424  return dmatAR;
425  }

467  double Orbitals::getExcitedStateEnergy(const QMState& state) const{
468 
469  double omega = 0.0;
470  if(state.isTransition()){
471  throw std::runtime_error("Total Energy does not exist for transition state");
472  }
473 
474  if (state.Type() == QMStateType::Singlet) {
475  if(BSESingletEnergies().size()<state.Index()+1){
476  throw std::runtime_error("Orbitals::getTotalEnergy You want "+ state.ToString()+" which has not been calculated");
477  }
478  omega = BSESingletEnergies()[state.Index()];
479  } else if (state.Type() == QMStateType::Triplet) {
480  if(BSETripletEnergies().size()<state.Index()+1){
481  throw std::runtime_error("Orbitals::getTotalEnergy You want "+ state.ToString()+" which has not been calculated");
482  }
483  omega = BSETripletEnergies()[state.Index()];
484  } else if (state.Type() == QMStateType::DQPstate) {
485  if(this->QPdiagEnergies().size()<state.Index()+1-getGWAmin()){
486  throw std::runtime_error("Orbitals::getTotalEnergy You want "+ state.ToString()+" which has not been calculated");
487  }
488  return QPdiagEnergies()[state.Index()-getGWAmin()];
489  }else if (state.Type() == QMStateType::KSstate) {
490  if(this->MOEnergies().size()<state.Index()+1){
491  throw std::runtime_error("Orbitals::getTotalEnergy You want "+ state.ToString()+" which has not been calculated");
492  }
493  return QPdiagEnergies()[state.Index()];
494  }else if (state.Type() == QMStateType::PQPstate) {
495  if(this->_QPpert_energies.rows()<state.Index()+1-getGWAmin()){
496  throw std::runtime_error("Orbitals::getTotalEnergy You want "+ state.ToString()+" which has not been calculated");
497  }
498  return _QPpert_energies(state.Index()-getGWAmin(),3);
499  } else {
500  throw std::runtime_error("GetTotalEnergy only knows states:singlet,triplet,KS,DQP,PQP");
501  }
502  return omega; // e.g. hartree
503  }

550  void Orbitals::PrepareDimerGuess(const Orbitals& orbitalsA,const Orbitals& orbitalsB) {
551 
552  // constructing the direct product orbA x orbB
553  int basisA = orbitalsA.getBasisSetSize();
554  int basisB = orbitalsB.getBasisSetSize();
555 
556  int levelsA = orbitalsA.getNumberOfLevels();
557  int levelsB = orbitalsB.getNumberOfLevels();
558 
559  int electronsA = orbitalsA.getNumberOfElectrons();
560  int electronsB = orbitalsB.getNumberOfElectrons();
561 
562 
563  this->MOCoefficients() = Eigen::MatrixXd::Zero(basisA + basisB, levelsA + levelsB);
564 
565  // AxB = | A 0 | // A = [EA, EB] //
566  // | 0 B | // //
567  if(orbitalsA.getDFTbasis()!=orbitalsB.getDFTbasis()){
568  throw std::runtime_error("Basissets of Orbitals A and B differ "+orbitalsA.getDFTbasis()+":"+orbitalsB.getDFTbasis());
569  }
570  this->setDFTbasis(orbitalsA.getDFTbasis());
571  if(orbitalsA.getECP()!=orbitalsB.getECP()){
572  throw std::runtime_error("ECPs of Orbitals A and B differ "+orbitalsA.getECP()+":"+orbitalsB.getECP());
573  }
574  this->setECP(orbitalsA.getECP());
575  this->setBasisSetSize(basisA + basisB);
576  this->setNumberOfLevels(electronsA + electronsB,
577  levelsA + levelsB - electronsA - electronsB);
578  this->setNumberOfElectrons(electronsA + electronsB);
579 
580  this->MOCoefficients().block(0, 0, basisA, levelsA) = orbitalsA.MOCoefficients();
581  this->MOCoefficients().block(basisA, levelsA, basisB, levelsB) = orbitalsB.MOCoefficients();
582 
583  Eigen::VectorXd& energies = this->MOEnergies();
584  energies.resize(levelsA + levelsB);
585 
586  energies.segment(0, levelsA) = orbitalsA.MOEnergies();
587  energies.segment(levelsA, levelsB) = orbitalsB.MOEnergies();
588 
589  this->OrderMOsbyEnergy();
590 
591  return;
