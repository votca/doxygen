00528 void CsgREupdateWorker::EvalConfiguration(Topology *conf, Topology *conf_atom){
00529 
00530   /* as per Relative Entropy Ref.  J. Chem. Phys. 134, 094112, 2011
00531    * for each CG we need to compute dU/dlamda and d2U/dlamda_i dlamda_j
00532    * here, we add running sum of the second term of eq. 51 and store it
00533    * in _DS, ensemble averaging and first term addition is done in EndEvaluate
00534    * for eq. 52, here the running sum of second and third terms is computed
00535    * and added to _HS, addition of 1st and 4th term and ensemble average
00536    * is performed in EndEvalute
00537    * 3rd term of eq. 52 can not be computed in EvalBonded/EvalNonbonded
00538    * since only one CG potential is accessible in there.
00539    * hence store current frame dU/dlamda in _dUFrame!
00540    */
00541 
00542   _dUFrame.clear();
00543 
00544   PotentialContainer::iterator potiter;
00545   for (potiter = _potentials.begin();
00546        potiter != _potentials.end(); ++potiter){
00547 
00548     PotentialInfo *potinfo = *potiter;
00549 
00550     if( potinfo->bonded )
00551       EvalBonded(conf, potinfo);
00552     else
00553       EvalNonbonded(conf,potinfo);
00554 
00555   }
00556 
00557   // update _DS and _HS
00558   for (int row = 0; row < _nlamda; row++) {
00559 
00560     _DS(row) += (-1.0 * _beta * _dUFrame(row));
00561 
00562     for (int col = row; col < _nlamda; col++)
00563       _HS(row, col) += ( _beta * _beta * _dUFrame(row) * _dUFrame(col));

00614   NBList::iterator pair_iter;
00615   int pos_start = potinfo->vec_pos;
00616   int pos_max = pos_start + potinfo->ucg->getOptParamSize();
00617   int lamda_i, lamda_j;
00618   double dU_i, d2U_ij;
00619   double U;
00620 
00621   // compute total energy
00622   U = 0.0;
00623   for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)
00624     U += potinfo->ucg->CalculateF((*pair_iter)->dist());
00625 
00626   _UavgCG += U;
00627 
00628   // computing dU/dlamda and d2U/dlamda_i dlamda_j
00629   for (int row = pos_start; row < pos_max; row++) {
00630 
00631     lamda_i = row - pos_start;
00632 
00633     dU_i = 0.0;
00634     for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)
00635       dU_i += potinfo->ucg->CalculateDF(lamda_i, (*pair_iter)->dist());
00636 
00637     _dUFrame(row) = dU_i;
00638 
00639     for (int col = row; col < pos_max; col++) {
00640 
00641       lamda_j = col - pos_start;
00642       d2U_ij = 0.0;
00643 
00644       for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)
00645         d2U_ij += potinfo->ucg->CalculateD2F(lamda_i, lamda_j, (*pair_iter)->dist());
00646 
00647       _HS(row, col) += (-1.0 * _beta * d2U_ij);
00648 
00649     } // end loop col
00650 
00651   } // end loop row
00652 
00653   delete nb;
00654 
00655 }
