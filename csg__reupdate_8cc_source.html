97void CsgREupdate::BeginEvaluate(Topology *top, Topology *) {
98
99 // initializing non-bonded interactions
100 nlamda_ = 0;
101 for (Property *prop : nonbonded_) {
102
103 string name = prop->get("name").value();
104 votca::Index id = potentials_.size();
105
106 PotentialInfo *i =
107 new PotentialInfo(id, false, nlamda_, param_in_ext_, prop, gentable_);
108
109 Table *aardf = new Table();
110 aardf->Load(name + ".dist.tgt");
111 aardfs_.push_back(aardf);
112
113 // generate the bead lists
114 BeadList beads1, beads2;
115 beads1.Generate(*top, prop->get("type1").value());
116 beads2.Generate(*top, prop->get("type2").value());
117
118 if (beads1.size() == 0) {
119 throw std::runtime_error("Topology does not have beads of type \"" +
120 prop->get("type1").value() +
121 "\"\n"
122 "This was specified in type1 of interaction \"" +
123 name + "\"");
124 }
125 if (beads2.size() == 0) {
126 throw std::runtime_error("Topology does not have beads of type \"" +
127 prop->get("type2").value() +
128 "\"\n"
129 "This was specified in type2 of interaction \"" +
130 name + "\"");
131 }
132
133 // calculate normalization factor for rdf
134 double *rdfnorm = new double();
135
136 if (prop->get("type1").value() == prop->get("type2").value()) {
137 *rdfnorm =
138 (double)(beads1.size() * beads2.size()) / 2. / top->BoxVolume();
139 } else {
140 *rdfnorm = (double)(beads1.size() * beads2.size()) / top->BoxVolume();
141 }
142
143 aardfnorms_.push_back(rdfnorm);
144
145 // let user know the properties of CG potential he/she selected
146 cout << "We have " << i->potentialName << " CG potential" << endl;
147 cout << "\t \t Between beads " << i->type1 << "-" << i->type2 << endl;
148 cout << "\t \t With Function form " << i->potentialFunction << endl;
149 cout << "\t \t And " << i->ucg->getOptParamSize()
150 << " parameters to "
151 "optimize"
152 << endl;
153 cout << "Potential range:" << endl;
154 cout << "\t \t rmin = " << i->rmin << " [nm]" << endl;
155 cout << "\t \t rcutoff = " << i->rcut << " [nm]" << endl;
156
157 // update parameter counter
158 nlamda_ += i->ucg->getOptParamSize();
159
160 potentials_.push_back(i);
161 }
162
163 cout << "Total number of parameters to optimize: " << nlamda_ << endl;
164
165 lamda_.resize(nlamda_);
166
167 // need to store initial guess of parameters in lamda_
168 for (auto &potential_ : potentials_) {
169
170 votca::Index pos_start = potential_->vec_pos;
171 votca::Index pos_max = pos_start + potential_->ucg->getOptParamSize();
172
173 for (votca::Index row = pos_start; row < pos_max; row++) {
174
175 votca::Index lamda_i = row - pos_start;
176 lamda_(row) = potential_->ucg->getOptParam(lamda_i);
177
178 } // end row loop
179
180 } // end potiter loop
181
182 DS_ = Eigen::VectorXd::Zero(nlamda_);
183 HS_ = Eigen::MatrixXd::Zero(nlamda_, nlamda_);
184 dUFrame_ = Eigen::VectorXd::Zero(nlamda_);
185 nframes_ = 0.0; // no frames processed yet!
186
187 // set Temperature
188 beta_ = (1.0 / options_.get("cg.inverse.kBT").as<double>());
189
190 // relaxation parameter for update
191 relax_ = options_.get("cg.inverse.scale").as<double>();
192
193 // whether to take steepest descent in case of non-symmetric positive H
194 // dosteep_ = options_.get("cg.inverse.re.do_steep").as<bool>();
195
196 UavgAA_ = 0.0;
197 UavgCG_ = 0.0;
198}

200void CsgREupdate::Run() {
201
202 if (!gentable_) {
203 CsgApplication::Run();
204 } else {
205
206 // only write potential tables for given parameters
207 nlamda_ = 0;
208 for (Property *prop : nonbonded_) {
209
210 string name = prop->get("name").value();
211
212 if (OptionsMap().count("interaction")) {
213
214 Tokenizer tok(OptionsMap()["interaction"].as<string>(), ";");
215 vector<string> vtok = tok.ToVector();
216 vector<string>::iterator vtok_iter =
217 find(vtok.begin(), vtok.end(), name);
218 if (vtok_iter == vtok.end()) {
219 continue;
220 }
221 }
222
223 PotentialInfo *i = new PotentialInfo(potentials_.size(), false, nlamda_,
224 param_in_ext_, prop, gentable_);
225
226 // update parameter counter
227 nlamda_ += i->ucg->getOptParamSize();
228
229 potentials_.push_back(i);
230 }
231
232 WriteOutFiles();
233 }

236void CsgREupdate::EndEvaluate() {
237
238 if (nframes_ == 0) {
239 throw std::runtime_error("No frames to process! Please check your input.");
240 }
241
242 // formulate HS_ dlamda = - DS_
243
244 REFormulateLinEq();
245
246 cout << "Updating parameters" << endl;
247 REUpdateLamda();
248
249 cout << "AA Ensemble Avg Energy :: " << UavgAA_ << endl;
250 cout << "CG Ensemble Avg Energy :: " << UavgCG_ << endl;
251
252 WriteOutFiles();
253
254 cout << "Finished RE update!\n";
255}

279// formulate HS_ x = - DS_
280void CsgREupdate::REFormulateLinEq() {
281
282 /* compute CG ensemble avges of dU,d2U by dividing its
283 * sum over all frames by total no. of frames
284 */
285
286 DS_ /= ((double)nframes_);
287 HS_ /= ((double)nframes_);
288 UavgCG_ /= ((double)nframes_);
289
290 /* adding 4th term in eq. 52 of ref J. Chem. Phys. 134, 094112, 2011
291 * to HS_
292 */
293
294 HS_ -= DS_ * DS_.transpose();
295
296 /* adding 1st term (i.e. aa ensemble avg) of eq. 51 to DS_
297 * and of eq. 52 to DH_
298 */
299 for (auto potinfo : potentials_) {
300
301 if (potinfo->bonded) {
302 AAavgBonded(potinfo);
303 } else {
304 AAavgNonbonded(potinfo);
305 }
306 }
307}

309// update lamda = lamda + relax*dlamda
310void CsgREupdate::REUpdateLamda() {
311
312 // first solve HS_ dx = - DS_
313 Eigen::LLT<Eigen::MatrixXd> cholesky(HS_); // compute the Cholesky
314 // decomposition of HS_
315 Eigen::VectorXd dlamda = cholesky.solve(-DS_);
316 if (cholesky.info() == Eigen::ComputationInfo::NumericalIssue) {
317
318 /* then can not use Newton-Raphson
319 * steepest descent with line-search may be helpful
320 * not implemented yet!
321 */
322 // if(! dosteep_){
323 if (hessian_check_) {
324 throw runtime_error(
325 "Hessian NOT a positive definite!\n"
326 "This can be a result of poor initial guess or "
327 "ill-suited CG potential settings or poor CG sampling.\n");
328 } else {
329
330 cout << "****************** WARNING *****************" << endl;
331 cout << "Hessian H NOT a positive definite!" << endl;
332 cout << "This can be a result of poor initial guess or ill-suited CG "
333 "potential settings or poor CG sampling."
334 << endl;
335 cout << "********************************************" << endl;
336 cout << endl;
337 cout << "However, you want to proceed despite non-positive definite H."
338 << endl;
339 cout << "Non-positive H does not gurantee relative entropy iterations to "
340 "converge to minimum."
341 << endl;
342 cout << "You can turn on Hessian check by setting command line option "
343 "--hessian-check=true."
344 << endl;
345
346 cout << "In this case, alternative update option is steepest descent."
347 << endl;
348 dlamda = -DS_;
349 }
350 }
351
352 lamda_ = lamda_ + relax_ * dlamda;
353
354 // now update parameters of individual cg potentials
355 for (auto &potential_ : potentials_) {
356
357 votca::Index pos_start = potential_->vec_pos;
358 votca::Index pos_max = pos_start + potential_->ucg->getOptParamSize();
359
360 for (votca::Index row = pos_start; row < pos_max; row++) {
361
362 votca::Index lamda_i = row - pos_start;
363 potential_->ucg->setOptParam(lamda_i, lamda_(row));
364
365 } // end row loop
366
367 } // end potiter loop
368}

370// do non bonded potential AA ensemble avg energy computations
371void CsgREupdate::AAavgNonbonded(PotentialInfo *potinfo) {
372
373 votca::Index pos_start = potinfo->vec_pos;
374 votca::Index pos_max = pos_start + potinfo->ucg->getOptParamSize();
375 double dU_i, d2U_ij;
376 double U;
377 votca::Index indx = potinfo->potentialIndex;
378
379 // compute avg AA energy
380 U = 0.0;
381
382 // assuming rdf bins are of same size
383 double step = aardfs_[indx]->x(2) - aardfs_[indx]->x(1);
384
385 for (votca::Index bin = 0; bin < aardfs_[indx]->size(); bin++) {
386
387 double r_hist = aardfs_[indx]->x(bin);
388 double r1 = r_hist - 0.5 * step;
389 double r2 = r1 + step;
390 double n_hist =
391 aardfs_[indx]->y(bin) * (*aardfnorms_[indx]) *
392 (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
393
394 if (n_hist > 0.0) {
395 U += n_hist * potinfo->ucg->CalculateF(r_hist);
396 }
397 }
398
399 UavgAA_ += U;
400
401 // computing dU/dlamda and d2U/dlamda_i dlamda_j
402 for (votca::Index row = pos_start; row < pos_max; row++) {
403
404 // ith parameter of this potential
405 votca::Index lamda_i = row - pos_start;
406
407 // compute dU/dlamda and add to DS_
408 dU_i = 0.0;
409
410 for (votca::Index bin = 0; bin < aardfs_[indx]->size(); bin++) {
411
412 double r_hist = aardfs_[indx]->x(bin);
413 double r1 = r_hist - 0.5 * step;
414 double r2 = r1 + step;
415 double n_hist =
416 aardfs_[indx]->y(bin) * (*aardfnorms_[indx]) *
417 (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
418
419 if (n_hist > 0.0) {
420 dU_i += n_hist * potinfo->ucg->CalculateDF(lamda_i, r_hist);
421 }
422
423 } // end loop over hist
424
425 DS_(row) += (beta_ * dU_i);
426
427 for (votca::Index col = row; col < pos_max; col++) {
428
429 votca::Index lamda_j = col - pos_start;
430
431 // compute d2U/dlamda_i dlamda_j and add to HS_
432 d2U_ij = 0.0;
433
434 for (votca::Index bin = 0; bin < aardfs_[indx]->size(); bin++) {
435
436 double r_hist = aardfs_[indx]->x(bin);
437 double r1 = r_hist - 0.5 * step;
438 double r2 = r1 + step;
439 double n_hist =
440 aardfs_[indx]->y(bin) * (*aardfnorms_[indx]) *
441 (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
442
443 if (n_hist > 0.0) {
444 d2U_ij +=
445 n_hist * potinfo->ucg->CalculateD2F(lamda_i, lamda_j, r_hist);
446 }
447
448 } // end loop pair_iter
449
450 HS_(row, col) += (beta_ * d2U_ij);
451 if (row != col) {
452 HS_(col, row) += (beta_ * d2U_ij);
453 }
454 } // end loop col
455
456 } // end loop row
457}

462std::unique_ptr<CsgApplication::Worker> CsgREupdate::ForkWorker() {
463
464 auto worker = std::make_unique<CsgREupdateWorker>();
465
466 // initialize worker
467 worker->options_ = options_;
468 worker->nonbonded_ = nonbonded_;
469 worker->nlamda_ = 0;
470
471 for (Property *prop : nonbonded_) {
472
473 PotentialInfo *i = new PotentialInfo(worker->potentials_.size(), false,
474 worker->nlamda_, param_in_ext_, prop);
475 // update parameter counter
476 worker->nlamda_ += i->ucg->getOptParamSize();
477
478 worker->potentials_.push_back(i);
479 }
480
481 worker->lamda_.resize(worker->nlamda_);
482
483 // need to store initial guess of parameters in lamda_
484 for (PotentialInfo *pot : worker->potentials_) {
485
486 votca::Index pos_start = pot->vec_pos;
487 votca::Index pos_max = pos_start + pot->ucg->getOptParamSize();
488
489 for (votca::Index row = pos_start; row < pos_max; row++) {
490
491 votca::Index lamda_i = row - pos_start;
492 worker->lamda_(row) = pot->ucg->getOptParam(lamda_i);
493
494 } // end row loop
495
496 } // end potiter loop
497
498 worker->DS_ = Eigen::VectorXd::Zero(worker->nlamda_);
499 worker->HS_ = Eigen::MatrixXd::Zero(worker->nlamda_, worker->nlamda_);
500 worker->dUFrame_ = Eigen::VectorXd::Zero(worker->nlamda_);
501 worker->nframes_ = 0.0; // no frames processed yet!
502 // set Temperature
503 worker->beta_ = (1.0 / worker->options_.get("cg.inverse.kBT").as<double>());
504 worker->UavgCG_ = 0.0;
505
506 return worker;
507}

519void CsgREupdateWorker::EvalConfiguration(Topology *conf, Topology *) {
520
521 /* as per Relative Entropy Ref. J. Chem. Phys. 134, 094112, 2011
522 * for each CG we need to compute dU/dlamda and d2U/dlamda_i dlamda_j
523 * here, we add running sum of the second term of eq. 51 and store it
524 * in DS_, ensemble averaging and first term addition is done in EndEvaluate
525 * for eq. 52, here the running sum of second and third terms is computed
526 * and added to HS_, addition of 1st and 4th term and ensemble average
527 * is performed in EndEvalute
528 * 3rd term of eq. 52 can not be computed in EvalBonded/EvalNonbonded
529 * since only one CG potential is accessible in there.
530 * hence store current frame dU/dlamda in dUFrame_!
531 */
532
533 dUFrame_.setZero();
534
535 for (PotentialInfo *potinfo : potentials_) {
536
537 if (potinfo->bonded) {
538 EvalBonded(conf, potinfo);
539 } else {
540 EvalNonbonded(conf, potinfo);
541 }
542 }
543 // update DS_ and HS_
544 DS_ -= beta_ * dUFrame_;
545 HS_ += beta_ * beta_ * dUFrame_ * dUFrame_.transpose();
546
547 nframes_++;
548}

550// do nonbonded potential related update stuff for the current frame in
551// evalconfig
552void CsgREupdateWorker::EvalNonbonded(Topology *conf, PotentialInfo *potinfo) {
553
554 BeadList beads1, beads2;
555 beads1.Generate(*conf, potinfo->type1);
556 beads2.Generate(*conf, potinfo->type2);
557
558 if (beads1.size() == 0) {
559 throw std::runtime_error("Topology does not have beads of type \"" +
560 potinfo->type1 +
561 "\"\n"
562 "This was specified in type1 of interaction \"" +
563 potinfo->potentialName + "\"");
564 }
565 if (beads2.size() == 0) {
566 throw std::runtime_error("Topology does not have beads of type \"" +
567 potinfo->type2 +
568 "\"\n"
569 "This was specified in type2 of interaction \"" +
570 potinfo->potentialName + "\"");
571 }
572
573 std::unique_ptr<NBList> nb;
574 bool gridsearch = false;
575
576 if (options_.exists("cg.nbsearch")) {
577
578 if (options_.get("cg.nbsearch").as<string>() == "grid") {
579 gridsearch = true;
580 } else if (options_.get("cg.nbsearch").as<string>() == "simple") {
581 gridsearch = false;
582 } else {
583 throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
584 }
585 }
586
587 if (gridsearch) {
588 nb = std::make_unique<NBListGrid>();
589 } else {
590 nb = std::make_unique<NBList>();
591 }
592
593 nb->setCutoff(potinfo->ucg->getCutOff());
594
595 if (potinfo->type1 == potinfo->type2) { // same beads
596 nb->Generate(beads1, true);
597 } else { // different beads
598 nb->Generate(beads1, beads2, true);
599 }
600
601 votca::Index pos_start = potinfo->vec_pos;
602 votca::Index pos_max = pos_start + potinfo->ucg->getOptParamSize();
603
604 // compute total energy
605 double U = 0.0;
606 for (auto &pair_iter : *nb) {
607 U += potinfo->ucg->CalculateF(pair_iter->dist());
608 }
609
610 UavgCG_ += U;
611
612 // computing dU/dlamda and d2U/dlamda_i dlamda_j
613 for (votca::Index row = pos_start; row < pos_max; row++) {
614
615 votca::Index lamda_i = row - pos_start;
616
617 double dU_i = 0.0;
618 for (auto &pair_iter : *nb) {
619 dU_i += potinfo->ucg->CalculateDF(lamda_i, pair_iter->dist());
620 }
621
622 dUFrame_(row) = dU_i;
623
624 for (votca::Index col = row; col < pos_max; col++) {
625
626 votca::Index lamda_j = col - pos_start;
627 double d2U_ij = 0.0;
628
629 for (auto &pair_iter : *nb) {
630 d2U_ij +=
631 potinfo->ucg->CalculateD2F(lamda_i, lamda_j, pair_iter->dist());
632 }
633
634 HS_(row, col) -= beta_ * d2U_ij;
635 if (row != col) {
636 HS_(col, row) -= beta_ * d2U_ij;
637 }
638 } // end loop col
639
640 } // end loop row
641}

646PotentialInfo::PotentialInfo(votca::Index index, bool bonded_,
647 votca::Index vec_pos_, string &param_in_ext_,
648 Property *options, bool gentable) {
649
650 potentialIndex = index;
651 bonded = bonded_;
652 vec_pos = vec_pos_;
653 options_ = options;
654
655 potentialName = options_->get("name").value();
656 type1 = options_->get("type1").value();
657 type2 = options_->get("type2").value();
658 potentialFunction = options_->get("re.function").value();
659
660 rmin = options_->get("min").as<double>();
661 rcut = options_->get("max").as<double>();
662
663 // assign the user selected function form for this potential
664 if (potentialFunction == "lj126") {
665 ucg = new PotentialFunctionLJ126(potentialName, rmin, rcut);
666 } else if (potentialFunction == "ljg") {
667 ucg = new PotentialFunctionLJG(potentialName, rmin, rcut);
668 } else if (potentialFunction == "cbspl") {
669 // get number of B-splines coefficients which are to be optimized
670 votca::Index nlam = options_->get("re.cbspl.nknots").as<votca::Index>();
671
672 if (!gentable) {
673 // determine minimum for B-spline from CG-MD rdf
674
675 Table dist;
676 string filename = potentialName + ".dist.new";
677 try {
678 dist.Load(filename);
679 // for( Index i = 0; i < dist.size(); i++)
680 // In the highly repulsive core (<0.1 nm) at some locations
681 // an unphysical non-zero RDF value may occur,
682 // so it would be better to estimate Rmin loop
683 // through RDF values from Rcut to zero instead of zero to Rcut.
684 double new_min = 0.0;
685 for (votca::Index i = dist.size() - 2; i > 0; i--) {
686 if (dist.y(i) < 1.0e-4) {
687 new_min = dist.x(i + 1);
688 break;
689 }
690 }
691 // setting extrapolation region based on CG rdf
692 rmin = new_min;
693
694 } catch (std::runtime_error &) {
695 throw std::runtime_error(
696 "Missing file for CG rdf for the interaction " + potentialName +
697 "." + " Make sure CG RDF for the pair " + potentialName +
698 " is computed after CG-MD simulation.");
699 }
700 }
701 ucg = new PotentialFunctionCBSPL(potentialName, nlam, rmin, rcut);
702 } else {
