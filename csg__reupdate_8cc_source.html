138  }

268 // formulate _HS x = -_DS
269 void CsgREupdate::REFormulateLinEq() {
270 
271  /* compute CG ensemble avges of dU,d2U by dividing its
272  * sum over all frames by total no. of frames
273  */
274 
275  _DS /= ( (double)_nframes );
276  _HS /= ( (double)_nframes );
277  _UavgCG /= ( (double)_nframes );
278 
279  /* adding 4th term in eq. 52 of ref J. Chem. Phys. 134, 094112, 2011
280  * to _HS
281  */
282  for( int row = 0; row < _nlamda; row++) {
283 
284  for( int col = row; col < _nlamda; col++){
285 
286  _HS(row,col) += (-1.0 * _DS(row) * _DS(col));
287  // since at this step _DS(i) = -beta*<dU/dlamda_i>cg
288 
289  }// end loop over col
290 
291  } // end loop over row
292 
293  /* adding 1st term (i.e. aa ensemble avg) of eq. 51 to _DS
294  * and of eq. 52 to _DH
295  */
296  PotentialContainer::iterator potiter;
297  for (potiter = _potentials.begin();
298  potiter != _potentials.end(); ++potiter){
299 
300  PotentialInfo *potinfo = *potiter;
301  if( potinfo->bonded )
302  AAavgBonded(potinfo);
303  else
304  AAavgNonbonded(potinfo);
305 
306  }
307 
308 }

310 // update lamda = lamda + relax*dlamda
311 void CsgREupdate::REUpdateLamda() {
312 
313  // first solve _HS dx = -_DS
314  ub::vector<double> dlamda(_nlamda);
315  dlamda.clear();
316  ub::vector<double> minusDS(_nlamda);
317 
318  // since linalg_cholesky_solve takes full matrix
319  // copy symmetric _HS to full matrix HS_
320  ub::matrix<double> HS_(_nlamda,_nlamda);
321  for(int row = 0; row < _nlamda; row++)
322  for(int col = 0; col<_nlamda; col++)
323  HS_(row,col) = _HS(row,col);
324 
325  minusDS = -_DS;
326 
327  try {
328 
329  votca::tools::linalg_cholesky_solve(dlamda, HS_, minusDS);
330 
331  }
332  catch (std::runtime_error){
333 
334  /* then can not use Newton-Raphson
335  * steepest descent with line-search may be helpful
336  * not implemented yet!
337  */
338  //if(!_dosteep){
339  if(_hessian_check)
340  {
341  throw runtime_error("Hessian NOT a positive definite!\n"
342  "This can be a result of poor initial guess or "
343  "ill-suited CG potential settings or poor CG sampling.\n");
344  }
345  else
346  {
347 
348  cout << "****************** WARNING *****************" << endl;
349  cout << "Hessian H NOT a positive definite!" << endl;
350  cout <<"This can be a result of poor initial guess or ill-suited CG potential settings or poor CG sampling." << endl;
351  cout << "********************************************" << endl;
352  cout << endl;
353  cout << "However, you want to proceed despite non-positive definite H." << endl;
354  cout << "Non-positive H does not gurantee relative entropy iterations to converge to minimum." << endl;
355  cout << "You can turn on Hessian check by setting command line option --hessian-check=true." <<endl;
356 
357  cout << "In this case, alternative update option is steepest descent." << endl;
358  dlamda = minusDS;
359 
360  }
361 
362  }
363 
364  _lamda = _lamda + _relax * dlamda ;
365 
366  // now update parameters of individual cg potentials
367  PotentialContainer::iterator potiter;
368  for (potiter = _potentials.begin();
369  potiter != _potentials.end(); ++potiter){
370 
371  int pos_start = (*potiter)->vec_pos;
372  int pos_max = pos_start + (*potiter)->ucg->getOptParamSize();
373 
374  for( int row = pos_start; row < pos_max; row++){
375 
376  int lamda_i = row - pos_start;
377  (*potiter)->ucg->setOptParam(lamda_i,_lamda(row));
378 
379  } // end row loop
380 
381  }// end potiter loop
382 
383 }

385 // do non bonded potential AA ensemble avg energy computations
386 void CsgREupdate::AAavgNonbonded(PotentialInfo* potinfo) {
387 
388  int pos_start = potinfo->vec_pos;
389  int pos_max = pos_start + potinfo->ucg->getOptParamSize();
390  int lamda_i,lamda_j;
391  double dU_i,d2U_ij;
392  double U;
393  int indx = potinfo->potentialIndex;
394 
395  // compute avg AA energy
396  U = 0.0;
397 
398  // assuming rdf bins are of same size
399  double step = _aardfs[indx]->x(2) - _aardfs[indx]->x(1);
400 
401  for(unsigned int bin = 0; bin < _aardfs[indx]->size(); bin++) {
402 
403  double r_hist = _aardfs[indx]->x(bin);
404  double r1 = r_hist - 0.5 * step;
405  double r2 = r1 + step;
406  double n_hist = _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
407  (4. / 3. * M_PI * (r2 * r2 * r2 - r1 * r1 * r1));
408 
409  if( n_hist > 0.0 )
410  U += n_hist * potinfo->ucg->CalculateF(r_hist);
411 
412  }
413 
414  _UavgAA += U;
415 
416  // computing dU/dlamda and d2U/dlamda_i dlamda_j
417  for( int row = pos_start; row < pos_max; row++){
418 
419  // ith parameter of this potential
420  lamda_i = row - pos_start;
421 
422  // compute dU/dlamda and add to _DS
423  dU_i = 0.0;
424 
425  for(unsigned int bin = 0; bin < _aardfs[indx]->size(); bin++) {
426 
427  double r_hist = _aardfs[indx]->x(bin);
428  double r1 = r_hist - 0.5 * step;
429  double r2 = r1 + step;
430  double n_hist = _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
431  (4. / 3. * M_PI * (r2 * r2 * r2 - r1 * r1 * r1));
432 
433  if( n_hist > 0.0 )
434  dU_i += n_hist * potinfo->ucg->CalculateDF(lamda_i,r_hist);
435 
436  } // end loop over hist
437 
438  _DS(row) += ( _beta * dU_i );
439 
440  for( int col = row; col < pos_max; col++){
441 
442  lamda_j = col - pos_start;
443 
444  // compute d2U/dlamda_i dlamda_j and add to _HS
445  d2U_ij = 0.0;
446 
447  for(unsigned int bin = 0; bin < _aardfs[indx]->size(); bin++) {
448 
449  double r_hist = _aardfs[indx]->x(bin);
450  double r1 = r_hist - 0.5 * step;
451  double r2 = r1 + step;
452  double n_hist = _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
453  (4. / 3. * M_PI * (r2 * r2 * r2 - r1 * r1 * r1));
454 
455  if ( n_hist > 0.0 )
456  d2U_ij += n_hist *
457  potinfo->ucg->CalculateD2F(lamda_i, lamda_j, r_hist);
458 
459  } // end loop pair_iter
460 
461  _HS(row,col) += (_beta * d2U_ij);
462 
463  } // end loop col
464 
465  } // end loop row
466 
467 }

538 void CsgREupdateWorker::EvalConfiguration(Topology *conf, Topology *conf_atom){
539 
540  /* as per Relative Entropy Ref. J. Chem. Phys. 134, 094112, 2011
541  * for each CG we need to compute dU/dlamda and d2U/dlamda_i dlamda_j
542  * here, we add running sum of the second term of eq. 51 and store it
543  * in _DS, ensemble averaging and first term addition is done in EndEvaluate
544  * for eq. 52, here the running sum of second and third terms is computed
545  * and added to _HS, addition of 1st and 4th term and ensemble average
546  * is performed in EndEvalute
547  * 3rd term of eq. 52 can not be computed in EvalBonded/EvalNonbonded
548  * since only one CG potential is accessible in there.
549  * hence store current frame dU/dlamda in _dUFrame!
550  */
551 
552  _dUFrame.clear();
553 
554  PotentialContainer::iterator potiter;
555  for (potiter = _potentials.begin();
556  potiter != _potentials.end(); ++potiter){
557 
558  PotentialInfo *potinfo = *potiter;
559 
560  if( potinfo->bonded )
561  EvalBonded(conf, potinfo);
562  else
563  EvalNonbonded(conf,potinfo);
564 
565  }
566 
567  // update _DS and _HS
568  for (int row = 0; row < _nlamda; row++) {
569 
570  _DS(row) += (-1.0 * _beta * _dUFrame(row));
571 
572  for (int col = row; col < _nlamda; col++)
573  _HS(row, col) += ( _beta * _beta * _dUFrame(row) * _dUFrame(col));
574 
575  }
576 
577  _nframes++;
578 
579 }

581 //do nonbonded potential related update stuff for the current frame in evalconfig
582 void CsgREupdateWorker::EvalNonbonded(Topology* conf, PotentialInfo* potinfo) {
583 
584  BeadList beads1, beads2;
585  beads1.Generate(*conf, potinfo->type1);
586  beads2.Generate(*conf, potinfo->type2);
587 
588  if(beads1.size() == 0)
589  throw std::runtime_error("Topology does not have beads of type \""
590  + potinfo->type1 + "\"\n"
591  "This was specified in type1 of interaction \""
592  + potinfo->potentialName + "\"");
593  if(beads2.size() == 0)
594  throw std::runtime_error("Topology does not have beads of type \""
595  + potinfo->type2 + "\"\n"
596  "This was specified in type2 of interaction \""
597  + potinfo->potentialName + "\"");
598 
599  NBList *nb;
600  bool gridsearch=false;
601 
602  if(_options.exists("cg.nbsearch")) {
603 
604  if(_options.get("cg.nbsearch").as<string>() == "grid")
605  gridsearch=true;
606  else if(_options.get("cg.nbsearch").as<string>() == "simple")
607  gridsearch=false;
608  else throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
609 
610  }
611 
612  if(gridsearch)
613  nb = new NBListGrid();
614  else
615  nb = new NBList();
616 
617  nb->setCutoff(potinfo->ucg->getCutOff());
618 
619  if( potinfo->type1 == potinfo->type2 )// same beads
620  nb->Generate(beads1,true);
621  else // different beads
622  nb->Generate(beads1,beads2,true);
623 
624  NBList::iterator pair_iter;
625  int pos_start = potinfo->vec_pos;
626  int pos_max = pos_start + potinfo->ucg->getOptParamSize();
627  int lamda_i, lamda_j;
628  double dU_i, d2U_ij;
629  double U;
630 
631  // compute total energy
632  U = 0.0;
633  for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)
634  U += potinfo->ucg->CalculateF((*pair_iter)->dist());
635 
636  _UavgCG += U;
637 
638  // computing dU/dlamda and d2U/dlamda_i dlamda_j
639  for (int row = pos_start; row < pos_max; row++) {
640 
641  lamda_i = row - pos_start;
642 
643  dU_i = 0.0;
644  for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)
645  dU_i += potinfo->ucg->CalculateDF(lamda_i, (*pair_iter)->dist());
646 
647  _dUFrame(row) = dU_i;
648 
649  for (int col = row; col < pos_max; col++) {
650 
651  lamda_j = col - pos_start;
652  d2U_ij = 0.0;
653 
654  for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter)

729  }

732  }
