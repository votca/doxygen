97 void CsgREupdate::BeginEvaluate(Topology *top, Topology *) {
98 
99  // initializing non-bonded interactions
100  _nlamda = 0;
101  for (Property *prop : _nonbonded) {
102 
103  string name = prop->get("name").value();
104  votca::Index id = _potentials.size();
105 
106  PotentialInfo *i =
107  new PotentialInfo(id, false, _nlamda, _param_in_ext, prop, _gentable);
108 
109  Table *aardf = new Table();
110  aardf->Load(name + ".dist.tgt");
111  _aardfs.push_back(aardf);
112 
113  // generate the bead lists
114  BeadList beads1, beads2;
115  beads1.Generate(*top, prop->get("type1").value());
116  beads2.Generate(*top, prop->get("type2").value());
117 
118  if (beads1.size() == 0) {
119  throw std::runtime_error("Topology does not have beads of type \"" +
120  prop->get("type1").value() +
121  "\"\n"
122  "This was specified in type1 of interaction \"" +
123  name + "\"");
124  }
125  if (beads2.size() == 0) {
126  throw std::runtime_error("Topology does not have beads of type \"" +
127  prop->get("type2").value() +
128  "\"\n"
129  "This was specified in type2 of interaction \"" +
130  name + "\"");
131  }
132 
133  // calculate normalization factor for rdf
134  double *rdfnorm = new double();
135 
136  if (prop->get("type1").value() == prop->get("type2").value()) {
137  *rdfnorm =
138  (double)(beads1.size() * beads2.size()) / 2. / top->BoxVolume();
139  } else {
140  *rdfnorm = (double)(beads1.size() * beads2.size()) / top->BoxVolume();
141  }
142 
143  _aardfnorms.push_back(rdfnorm);
144 
145  // let user know the properties of CG potential he/she selected
146  cout << "We have " << i->potentialName << " CG potential" << endl;
147  cout << "\t \t Between beads " << i->type1 << "-" << i->type2 << endl;
148  cout << "\t \t With Function form " << i->potentialFunction << endl;
149  cout << "\t \t And " << i->ucg->getOptParamSize()
150  << " parameters to "
151  "optimize"
152  << endl;
153  cout << "Potential range:" << endl;
154  cout << "\t \t rmin = " << i->rmin << " [nm]" << endl;
155  cout << "\t \t rcutoff = " << i->rcut << " [nm]" << endl;
156 
157  // update parameter counter
158  _nlamda += i->ucg->getOptParamSize();
159 
160  _potentials.push_back(i);
161  }
162 
163  cout << "Total number of parameters to optimize: " << _nlamda << endl;
164 
165  _lamda.resize(_nlamda);
166 
167  // need to store initial guess of parameters in _lamda
168  for (auto &_potential : _potentials) {
169 
170  votca::Index pos_start = _potential->vec_pos;
171  votca::Index pos_max = pos_start + _potential->ucg->getOptParamSize();
172 
173  for (votca::Index row = pos_start; row < pos_max; row++) {
174 
175  votca::Index lamda_i = row - pos_start;
176  _lamda(row) = _potential->ucg->getOptParam(lamda_i);
177 
178  } // end row loop
179 
180  } // end potiter loop
181 
182  _DS = Eigen::VectorXd::Zero(_nlamda);
183  _HS = Eigen::MatrixXd::Zero(_nlamda, _nlamda);
184  _dUFrame = Eigen::VectorXd::Zero(_nlamda);
185  _nframes = 0.0; // no frames processed yet!
186 
187  // set Temperature
188  _beta = (1.0 / _options.get("cg.inverse.kBT").as<double>());
189 
190  // relaxation parameter for update
191  _relax = _options.get("cg.inverse.scale").as<double>();
192 
193  // whether to take steepest descent in case of non-symmetric positive H
194  //_dosteep = _options.get("cg.inverse.re.do_steep").as<bool>();
195 
196  _UavgAA = 0.0;
197  _UavgCG = 0.0;
198 }

279 // formulate _HS x = -_DS
280 void CsgREupdate::REFormulateLinEq() {
281 
282  /* compute CG ensemble avges of dU,d2U by dividing its
283  * sum over all frames by total no. of frames
284  */
285 
286  _DS /= ((double)_nframes);
287  _HS /= ((double)_nframes);
288  _UavgCG /= ((double)_nframes);
289 
290  /* adding 4th term in eq. 52 of ref J. Chem. Phys. 134, 094112, 2011
291  * to _HS
292  */
293 
294  _HS -= _DS * _DS.transpose();
295 
296  /* adding 1st term (i.e. aa ensemble avg) of eq. 51 to _DS
297  * and of eq. 52 to _DH
298  */
299  for (auto potinfo : _potentials) {
300 
301  if (potinfo->bonded) {
302  AAavgBonded(potinfo);
303  } else {
304  AAavgNonbonded(potinfo);
305  }
306  }
307 }

309 // update lamda = lamda + relax*dlamda
310 void CsgREupdate::REUpdateLamda() {
311 
312  // first solve _HS dx = -_DS
313  Eigen::LLT<Eigen::MatrixXd> cholesky(_HS); // compute the Cholesky
314  // decomposition of _HS
315  Eigen::VectorXd dlamda = cholesky.solve(-_DS);
316  if (cholesky.info() == Eigen::ComputationInfo::NumericalIssue) {
317 
318  /* then can not use Newton-Raphson
319  * steepest descent with line-search may be helpful
320  * not implemented yet!
321  */
322  // if(!_dosteep){
323  if (_hessian_check) {
324  throw runtime_error(
325  "Hessian NOT a positive definite!\n"
326  "This can be a result of poor initial guess or "
327  "ill-suited CG potential settings or poor CG sampling.\n");
328  } else {
329 
330  cout << "****************** WARNING *****************" << endl;
331  cout << "Hessian H NOT a positive definite!" << endl;
332  cout << "This can be a result of poor initial guess or ill-suited CG "
333  "potential settings or poor CG sampling."
334  << endl;
335  cout << "********************************************" << endl;
336  cout << endl;
337  cout << "However, you want to proceed despite non-positive definite H."
338  << endl;
339  cout << "Non-positive H does not gurantee relative entropy iterations to "
340  "converge to minimum."
341  << endl;
342  cout << "You can turn on Hessian check by setting command line option "
343  "--hessian-check=true."
344  << endl;
345 
346  cout << "In this case, alternative update option is steepest descent."
347  << endl;
348  dlamda = -_DS;
349  }
350  }
351 
352  _lamda = _lamda + _relax * dlamda;
353 
354  // now update parameters of individual cg potentials
355  for (auto &_potential : _potentials) {
356 
357  votca::Index pos_start = _potential->vec_pos;
358  votca::Index pos_max = pos_start + _potential->ucg->getOptParamSize();
359 
360  for (votca::Index row = pos_start; row < pos_max; row++) {
361 
362  votca::Index lamda_i = row - pos_start;
363  _potential->ucg->setOptParam(lamda_i, _lamda(row));
364 
365  } // end row loop
366 
367  } // end potiter loop
368 }

370 // do non bonded potential AA ensemble avg energy computations
371 void CsgREupdate::AAavgNonbonded(PotentialInfo *potinfo) {
372 
373  votca::Index pos_start = potinfo->vec_pos;
374  votca::Index pos_max = pos_start + potinfo->ucg->getOptParamSize();
375  double dU_i, d2U_ij;
376  double U;
377  votca::Index indx = potinfo->potentialIndex;
378 
379  // compute avg AA energy
380  U = 0.0;
381 
382  // assuming rdf bins are of same size
383  double step = _aardfs[indx]->x(2) - _aardfs[indx]->x(1);
384 
385  for (votca::Index bin = 0; bin < _aardfs[indx]->size(); bin++) {
386 
387  double r_hist = _aardfs[indx]->x(bin);
388  double r1 = r_hist - 0.5 * step;
389  double r2 = r1 + step;
390  double n_hist =
391  _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
392  (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
393 
394  if (n_hist > 0.0) {
395  U += n_hist * potinfo->ucg->CalculateF(r_hist);
396  }
397  }
398 
399  _UavgAA += U;
400 
401  // computing dU/dlamda and d2U/dlamda_i dlamda_j
402  for (votca::Index row = pos_start; row < pos_max; row++) {
403 
404  // ith parameter of this potential
405  votca::Index lamda_i = row - pos_start;
406 
407  // compute dU/dlamda and add to _DS
408  dU_i = 0.0;
409 
410  for (votca::Index bin = 0; bin < _aardfs[indx]->size(); bin++) {
411 
412  double r_hist = _aardfs[indx]->x(bin);
413  double r1 = r_hist - 0.5 * step;
414  double r2 = r1 + step;
415  double n_hist =
416  _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
417  (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
418 
419  if (n_hist > 0.0) {
420  dU_i += n_hist * potinfo->ucg->CalculateDF(lamda_i, r_hist);
421  }
422 
423  } // end loop over hist
424 
425  _DS(row) += (_beta * dU_i);
426 
427  for (votca::Index col = row; col < pos_max; col++) {
428 
429  votca::Index lamda_j = col - pos_start;
430 
431  // compute d2U/dlamda_i dlamda_j and add to _HS
432  d2U_ij = 0.0;
433 
434  for (votca::Index bin = 0; bin < _aardfs[indx]->size(); bin++) {
435 
436  double r_hist = _aardfs[indx]->x(bin);
437  double r1 = r_hist - 0.5 * step;
438  double r2 = r1 + step;
439  double n_hist =
440  _aardfs[indx]->y(bin) * (*_aardfnorms[indx]) *
441  (4. / 3. * votca::tools::conv::Pi * (r2 * r2 * r2 - r1 * r1 * r1));
442 
443  if (n_hist > 0.0) {
444  d2U_ij +=
445  n_hist * potinfo->ucg->CalculateD2F(lamda_i, lamda_j, r_hist);
446  }
447 
448  } // end loop pair_iter
449 
450  _HS(row, col) += (_beta * d2U_ij);
451  if (row != col) {
452  _HS(col, row) += (_beta * d2U_ij);
453  }
454  } // end loop col
455 
456  } // end loop row
457 }

519 void CsgREupdateWorker::EvalConfiguration(Topology *conf, Topology *) {
520 
521  /* as per Relative Entropy Ref. J. Chem. Phys. 134, 094112, 2011
522  * for each CG we need to compute dU/dlamda and d2U/dlamda_i dlamda_j
523  * here, we add running sum of the second term of eq. 51 and store it
524  * in _DS, ensemble averaging and first term addition is done in EndEvaluate
525  * for eq. 52, here the running sum of second and third terms is computed
526  * and added to _HS, addition of 1st and 4th term and ensemble average
527  * is performed in EndEvalute
528  * 3rd term of eq. 52 can not be computed in EvalBonded/EvalNonbonded
529  * since only one CG potential is accessible in there.
530  * hence store current frame dU/dlamda in _dUFrame!
531  */
532 
533  _dUFrame.setZero();
534 
535  for (PotentialInfo *potinfo : _potentials) {
536 
537  if (potinfo->bonded) {
538  EvalBonded(conf, potinfo);
539  } else {
540  EvalNonbonded(conf, potinfo);
541  }
542  }
543  // update _DS and _HS
544  _DS -= _beta * _dUFrame;
545  _HS += _beta * _beta * _dUFrame * _dUFrame.transpose();
546 
547  _nframes++;
548 }

550 // do nonbonded potential related update stuff for the current frame in
551 // evalconfig
552 void CsgREupdateWorker::EvalNonbonded(Topology *conf, PotentialInfo *potinfo) {
553 
554  BeadList beads1, beads2;
555  beads1.Generate(*conf, potinfo->type1);
556  beads2.Generate(*conf, potinfo->type2);
557 
558  if (beads1.size() == 0) {
559  throw std::runtime_error("Topology does not have beads of type \"" +
560  potinfo->type1 +
561  "\"\n"
562  "This was specified in type1 of interaction \"" +
563  potinfo->potentialName + "\"");
564  }
565  if (beads2.size() == 0) {
566  throw std::runtime_error("Topology does not have beads of type \"" +
567  potinfo->type2 +
568  "\"\n"
569  "This was specified in type2 of interaction \"" +
570  potinfo->potentialName + "\"");
571  }
572 
573  std::unique_ptr<NBList> nb;
574  bool gridsearch = false;
575 
576  if (_options.exists("cg.nbsearch")) {
577 
578  if (_options.get("cg.nbsearch").as<string>() == "grid") {
579  gridsearch = true;
580  } else if (_options.get("cg.nbsearch").as<string>() == "simple") {
581  gridsearch = false;
582  } else {
583  throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
584  }
585  }
586 
587  if (gridsearch) {
588  nb = std::make_unique<NBListGrid>();
589  } else {
590  nb = std::make_unique<NBList>();
591  }
592 
593  nb->setCutoff(potinfo->ucg->getCutOff());
594 
595  if (potinfo->type1 == potinfo->type2) { // same beads
596  nb->Generate(beads1, true);
597  } else { // different beads
598  nb->Generate(beads1, beads2, true);
599  }
600 
601  votca::Index pos_start = potinfo->vec_pos;
602  votca::Index pos_max = pos_start + potinfo->ucg->getOptParamSize();
603 
604  // compute total energy
605  double U = 0.0;
606  for (auto &pair_iter : *nb) {
607  U += potinfo->ucg->CalculateF(pair_iter->dist());
608  }
609 
610  _UavgCG += U;
611 
612  // computing dU/dlamda and d2U/dlamda_i dlamda_j
613  for (votca::Index row = pos_start; row < pos_max; row++) {
614 
615  votca::Index lamda_i = row - pos_start;
616 
617  double dU_i = 0.0;
618  for (auto &pair_iter : *nb) {
619  dU_i += potinfo->ucg->CalculateDF(lamda_i, pair_iter->dist());
620  }
621 
622  _dUFrame(row) = dU_i;
623 
624  for (votca::Index col = row; col < pos_max; col++) {
625 
626  votca::Index lamda_j = col - pos_start;
627  double d2U_ij = 0.0;
628 
629  for (auto &pair_iter : *nb) {
630  d2U_ij +=
631  potinfo->ucg->CalculateD2F(lamda_i, lamda_j, pair_iter->dist());
632  }
633 
634  _HS(row, col) -= _beta * d2U_ij;
635  if (row != col) {
636  _HS(col, row) -= _beta * d2U_ij;
637  }
638  } // end loop col
639 
640  } // end loop row
641 }
