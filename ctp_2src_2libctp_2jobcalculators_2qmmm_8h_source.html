114 void QMMM::Initialize(Property *opt) {
115

117  UpdateWithDefaults( opt, "ctp" );
118  _options = *opt;
119 
120  cout << endl

122  << flush;
123 
124  _maverick = (_nThreads == 1) ? true : false;
125 
126 
127  string key = "options.qmmm.multipoles";
128 
129  if ( opt->exists(key) ) {
130  _xml_file = opt->get(key).as< string >();
131  }
132 
133  key = "options.qmmm.control";
134

137  }

140  }
141 
142  if ( opt->exists(key+".emp_file")) {
143  _emp_file = opt->get(key+".emp_file").as<string>();
144  }
145  else {
146  _emp_file = opt->get(key+".emp_file").as<string>();
147  }
148 
149  if ( opt->exists(key+".output") ) {
150  _outFile = opt->get(key+".output").as< string >();
151  _energies2File = true;

153  else { _energies2File = false; }
154 
155  if ( opt->exists(key+".pdb_check")) {
156  _pdb_check = opt->get(key+".pdb_check").as<string>();
157  }
158  else { _pdb_check = ""; }
159 
160  if ( opt->exists(key+".write_chk")) {
161  _write_chk_suffix = opt->get(key+".write_chk").as<string>();
162  _write_chk = true;
163  }
164  else { _write_chk = false; }
165

167  _chk_format = opt->get(key+".format_chk").as<string>();
168  }
169  else { _chk_format = "xyz"; }
170 
171  if ( opt->exists(key+".split_dpl")) {
172  _chk_split_dpl = ( opt->get(key+".split_dpl").as<int>() == 1) ?
173  true : false;
174  }
175  else { _chk_split_dpl = true; }
176 
177  if ( opt->exists(key+".dpl_spacing")) {
178  _chk_dpl_spacing = opt->get(key+".dpl_spacing").as<double>();
179  }
180  else {
181  _chk_dpl_spacing = 1.0e-6;
182  }
183 
184 
185  key = "options.qmmm.coulombmethod";
186 
187  if ( opt->exists(key+".method") ) {
188  _method = opt->get(key+".method").as< string >();
189  if (_method != "cut-off" && _method != "cutoff") {
190  throw runtime_error("Method " + _method + " not recognised.");
191  }
192  }
193  else {
194  _method = "cut-off";
195  }
196  if ( opt->exists(key+".cutoff1") ) {
197  _cutoff1 = opt->get(key+".cutoff1").as< double >();
198  if (_cutoff1) { _useCutoff = true; }
199  }
200  if ( opt->exists(key+".cutoff2") ) {
201  _cutoff2 = opt->get(key+".cutoff2").as< double >();
202  }
203  else {
204  _cutoff2 = _cutoff1;
205  }
206  if ( opt->exists(key+".subthreads") ) {
207  _subthreads = opt->get(key+".subthreads").as< int >();
208  }
209  else {
210  _subthreads = 1;
211  }
212 
213  key = "options.qmmm.qmpackage";
214 
215  if ( opt->exists(key+".package")) {
216  string package_xml = opt->get(key+".package").as< string >();
217  load_property_from_xml(_qmpack_opt, package_xml.c_str());
218  _package = _qmpack_opt.get("package.name").as< string >();
219  }

221  throw runtime_error("No QM package specified.");
222  }

224

226  // or excited state QM/MM
227  key = "options.qmmm.gwbse";
228 
229  if ( opt->exists(key)) {
230  cout << " Excited state QM/MM " << endl;
231 
232  if ( opt->exists(key+".gwbse_options")) {
233  string gwbse_xml = opt->get(key+".gwbse_options").as< string >();

235  // _gwbse = _gwbse_opt.get("package.name").as< string >();

238  throw runtime_error("GWBSE options not specified.");
239  }
240 
241  _state = opt->get(key+".state").as< int >();
242 
243 
244  } else {
245  cout << " Ground state QM/MM " << endl;
246  }
247 
248 
249 
250 
251  //cout << TXT << _options;

286 XJob QMMM::ProcessInputString(Job *job, Topology *top, QMThread *thread) {
287 
288  string input = job->getInput().as<string>();
289  vector<Segment*> qmSegs;
290  vector<string> qmSegMps;
291  vector<string> split;
292  Tokenizer toker(input, " \t\n");
293  toker.ToVector(split);
294 
295  for (unsigned int i = 0; i < split.size(); ++i) {
296 
297  string id_seg_mps = split[i];
298  vector<string> split_id_seg_mps;
299  Tokenizer toker(id_seg_mps, ":");
300  toker.ToVector(split_id_seg_mps);
301 
302  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
303  string segName = split_id_seg_mps[1];
304  string mpsFile = split_id_seg_mps[2];
305 
306  Segment *seg = top->getSegment(segId);
307  if (seg->getName() != segName) {
308  LOG(logERROR,*(thread->getLogger()))
309  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
310  << " maltagged as " << segName << ". Skip job ..." << flush;
311  throw std::runtime_error("Input does not match topology.");
312  }
313 
314  qmSegs.push_back(seg);
315  qmSegMps.push_back(mpsFile);
316  }
317 
318  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
319 }

322 Job::JobResult QMMM::EvalJob(Topology *top, Job *job, QMThread *thread) {
323 
324  // SILENT LOGGER FOR QMPACKAGE
325  Logger* log = thread->getLogger();
326  Logger* qlog = new Logger();
327  qlog->setReportLevel(logDEBUG);
328  qlog->setMultithreading(_maverick);
329  qlog->setPreface(logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
330  qlog->setPreface(logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
331  qlog->setPreface(logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
332  qlog->setPreface(logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
333 
334  // CREATE XJOB FROM JOB INPUT STRING
335  LOG(logINFO,*log)
336  << "Job input = " << job->getInput().as<string>() << flush;
337  XJob xjob = this->ProcessInputString(job, top, thread);
338 
339  // GENERATE POLAR TOPOLOGY FOR JOB
340  double co1 = _cutoff1;
341  double co2 = _cutoff2;
342  _mps_mapper.Gen_QM_MM1_MM2(top, &xjob, co1, co2, thread);
343 
344  LOG(logINFO,*log)
345  << xjob.getPolarTop()->ShellInfoStr() << flush;
346 
347  if (tools::globals::verbose)
348  xjob.getPolarTop()->PrintPDB(xjob.getTag()+"_QM0_MM1_MM2.pdb");
349 
350  // INDUCTOR, QM RUNNER, QM-MM MACHINE
351  XInductor xind = XInductor(top, &_options, "options.qmmm",
352  _subthreads, _maverick);
353  xind.setLog(thread->getLogger());
354 
355  //Gaussian qmpack = Gaussian(&_qmpack_opt);
356 
357  // get the corresponding object from the QMPackageFactory
358  QMPackage *qmpack = QMPackages().Create( _package );
359 
360  qmpack->Initialize( &_qmpack_opt );
361 
362  qmpack->setLog(qlog);
363 
364  QMMachine<QMPackage> machine = QMMachine<QMPackage>(&xjob, &xind, qmpack,
365  &_options, "options.qmmm", _subthreads, _maverick);
366  machine.setLog(thread->getLogger());
367 
368  // EVALUATE: ITERATE UNTIL CONVERGED
369  machine.Evaluate(&xjob);
370 
371  // DESTROY QMPackage
372  delete qmpack;
373 
374  // DELIVER OUTPUT & CLEAN
375  this->LockCout();
376  cout << *thread->getLogger();
377  this->UnlockCout();
378 
379  // JOT INFO STRING & CLEAN POLAR TOPOLOGY
380  xjob.setInfoLine(true,true);
381 
382  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
383  Job::JobResult jres = Job::JobResult();
384  jres.setOutput(xjob.getInfoLine());
385  jres.setStatus(Job::COMPLETE);
386 
387  if (!xind.hasConverged()) {
388  jres.setStatus(Job::FAILED);
389  jres.setError(xind.getError());
390  LOG(logERROR,*log) << xind.getError() << flush;
391  }
392 
393  return jres;
394 }
