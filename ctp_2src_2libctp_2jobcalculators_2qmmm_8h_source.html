260 void QMMM::WriteJobFile(Topology *top) {
261 
262  cout << endl << "... ... Writing job file " << flush;
263 
264  // OPEN STREAM
265  std::ofstream ofs;
266  ofs.open(_jobfile.c_str(), std::ofstream::out);
267  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
268 
269  ofs << "<jobs>" << endl;
270 
271  int jobCount = 0;
272 
273  std::vector< Segment* > segments = top->Segments();
274  std::vector< Segment* >::iterator sit;
275 
276  for (sit = segments.begin(); sit != segments.end(); ++sit) {
277 
278  int id = ++jobCount;
279  std::string tag = "";
280 
281  Property Input;
282  std::string state = "h";
283  std::string name = (*sit)->getName();
284  int idt = (*sit)->getId();
285 
286  std::string mpsFile = "MP_FILES/" + name + "_" + state + ".mps";
287 
288  std::string inp = (format("%1$d:%2$s:%3$s") % idt % name % mpsFile).str();
289 
290  (void)Input.add("input",inp);
291  Job job(id, tag, Input, Job::AVAILABLE );
292  job.ToStream(ofs,"xml");
293  }
294 
295 
296  // CLOSE STREAM
297  ofs << "</jobs>" << endl;
298  ofs.close();
299 
300 
301  //throw std::invalid_argument( "ctp_parallel -e qmmm -j write" );
302 }

304 XJob QMMM::ProcessInputString(Job *job, Topology *top, QMThread *thread) {
305 
306  string input = job->getInput().as<string>();
307  vector<Segment*> qmSegs;
308  vector<string> qmSegMps;
309  vector<string> split;
310  Tokenizer toker(input, " \t\n");
311  toker.ToVector(split);
312 
313  for (unsigned int i = 0; i < split.size(); ++i) {
314 
315  string id_seg_mps = split[i];
316  vector<string> split_id_seg_mps;
317  Tokenizer toker(id_seg_mps, ":");
318  toker.ToVector(split_id_seg_mps);
319 
320  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
321  string segName = split_id_seg_mps[1];
322  string mpsFile = split_id_seg_mps[2];
323 
324  Segment *seg = top->getSegment(segId);
325  if (seg->getName() != segName) {
326  CTP_LOG(logERROR,*(thread->getLogger()))
327  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
328  << " maltagged as " << segName << ". Skip job ..." << flush;
329  throw std::runtime_error("Input does not match topology.");
330  }
331 
332  qmSegs.push_back(seg);
333  qmSegMps.push_back(mpsFile);
334  }
335 
336  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
337 }

340 Job::JobResult QMMM::EvalJob(Topology *top, Job *job, QMThread *thread) {
341 
342  // SILENT LOGGER FOR QMPACKAGE
343  Logger* log = thread->getLogger();
344  Logger* qlog = new Logger();
345  qlog->setReportLevel(logDEBUG);
346  qlog->setMultithreading(_maverick);
347  qlog->setPreface(logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
348  qlog->setPreface(logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
349  qlog->setPreface(logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
350  qlog->setPreface(logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
351 
352  // CREATE XJOB FROM JOB INPUT STRING
353  CTP_LOG(logINFO,*log)
354  << "Job input = " << job->getInput().as<string>() << flush;
355  XJob xjob = this->ProcessInputString(job, top, thread);
356 
357  // GENERATE POLAR TOPOLOGY FOR JOB
358  double co1 = _cutoff1;
359  double co2 = _cutoff2;
360  _mps_mapper.Gen_QM_MM1_MM2(top, &xjob, co1, co2, thread);
361 
362  CTP_LOG(logINFO,*log)
363  << xjob.getPolarTop()->ShellInfoStr() << flush;
364 
365  if (tools::globals::verbose)
366  xjob.getPolarTop()->PrintPDB(xjob.getTag()+"_QM0_MM1_MM2.pdb");
367 
368  // INDUCTOR, QM RUNNER, QM-MM MACHINE
369  XInductor xind = XInductor(top, &_options, "options.qmmm",
370  _subthreads, _maverick);
371  xind.setLog(thread->getLogger());
372 
373  //Gaussian qmpack = Gaussian(&_qmpack_opt);
374 
375  // get the corresponding object from the QMPackageFactory
376  QMPackage *qmpack = QMPackages().Create( _package );
377 
378  qmpack->Initialize( &_qmpack_opt );
379 
380  qmpack->setLog(qlog);
381 
382  QMMachine<QMPackage> machine = QMMachine<QMPackage>(&xjob, &xind, qmpack,
383  &_options, "options.qmmm", _subthreads, _maverick);
384  machine.setLog(thread->getLogger());
385 
386  // EVALUATE: ITERATE UNTIL CONVERGED
387  machine.Evaluate(&xjob);
388 
389  // DESTROY QMPackage
390  delete qmpack;
391 
392  // DELIVER OUTPUT & CLEAN
393  this->LockCout();
394  cout << *thread->getLogger();
395  this->UnlockCout();
396 
397  // JOT INFO STRING & CLEAN POLAR TOPOLOGY
398  xjob.setInfoLine(true,true);
399 
400  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
401  Job::JobResult jres = Job::JobResult();
402  jres.setOutput(xjob.getInfoLine());
403  jres.setStatus(Job::COMPLETE);
404 
405  if (!xind.hasConverged()) {
406  jres.setStatus(Job::FAILED);
407  jres.setError(xind.getError());
408  CTP_LOG(logERROR,*log) << xind.getError() << flush;
409  }
410 
411  return jres;
412 }
