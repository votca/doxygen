35 void QMMIter::UpdateMPSFromGDMA(std::vector<std::vector<double> > &multipoles, std::vector< ctp::PolarSeg* > &psegs) {
36 
37 
38  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
39  ctp::PolarSeg *pseg = psegs[i];
40  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
41 
42  // Retrieve multipole info of this atom
43  std::vector<double> update = multipoles[qac];
44  while (update.size() < 9) update.push_back(0.0);
45 
46  // Convert e*(a_0)^k to e*(nm)^k where k = rank
47 
48  for (int m = 1; m < 4; m++) {
49  update[m] *= pow(tools::conv::bohr2nm, 1);
50  }
51  for (int m = 4; m < 9; m++) {
52  update[m] *= pow(tools::conv::bohr2nm, 2);
53  }
54 
55  ctp::APolarSite *aps = (*pseg)[j];
56 
57  aps->setQs(update, 0);
58 
59  }
60  }
61 
62  return;
63  }

65  void QMMIter::UpdatePosChrgFromQMAtoms(std::vector< ctp::QMAtom* > &qmatoms,
66  std::vector< ctp::PolarSeg* > &psegs) {
67 
68  double AA_to_NM = tools::conv::ang2nm;
69 
70  double dR_RMS = 0.0;
71  double dQ_RMS = 0.0;
72  double dQ_SUM = 0.0;
73 
74  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
75  ctp::PolarSeg *pseg = psegs[i];
76  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
77 
78  // Retrieve info from ctp::QMAtom
79  ctp::QMAtom *qmatm = qmatoms[qac];
80  vec upd_r = vec(qmatm->x, qmatm->y, qmatm->z);
81  upd_r *= AA_to_NM;
82  double upd_Q00 = qmatm->charge;
83  //cout << "updating charge to " << qmatm->charge << endl;
84 
85  // Compare to previous r, Q00
86  ctp::APolarSite *aps = (*pseg)[j];
87  vec old_r = aps->getPos();
88  double old_Q00 = aps->getQ00();
89  double dR = abs(upd_r - old_r);
90  double dQ00 = upd_Q00 - old_Q00;
91 
92  dR_RMS += dR*dR;
93  dQ_RMS += dQ00*dQ00;
94  dQ_SUM += dQ00;
95 
96  // Forward updated r, Q00 to APS
97  aps->setPos(upd_r);
98  aps->setQ00(upd_Q00, 0);
99  }
100  }
101 
102 
103  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
104  dR_RMS /= qmatoms.size();
105  dQ_RMS /= qmatoms.size();
106  dR_RMS = sqrt(dR_RMS);
107  dQ_RMS = sqrt(dQ_RMS);
108 
109  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
110 
111 
112  this->setdRdQ(dR_RMS, dQ_RMS, dQ_SUM);
113  return;
114  }

116  void QMMIter::GenerateQMAtomsFromPolarSegs(ctp::PolarTop *ptop, Orbitals &orb,
117  bool split_dpl, double dpl_spacing) {
118 
119  double AA_to_NM = tools::conv::ang2nm;
120 
121  // INNER SHELL QM0
122  for (unsigned int i = 0; i < ptop->QM0().size(); ++i) {
123  ctp::PolarSeg *pseg = ptop->QM0()[i];
124  for (unsigned int j = 0; j < pseg->size(); ++j) {
125 
126  ctp::APolarSite *aps = (*pseg)[j];
127  vec pos = aps->getPos() / AA_to_NM;
128  double Q = aps->getQ00();
129  string type = "qm";
130 
131  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, false);
132 
133  }
134  }
135 
136  // MIDDLE SHELL MM1
137  for (unsigned int i = 0; i < ptop->MM1().size(); ++i) {
138  ctp::PolarSeg *pseg = ptop->MM1()[i];
139  for (unsigned int j = 0; j < pseg->size(); ++j) {
140 
141  ctp::APolarSite *aps = (*pseg)[j];
142  if(aps->getRank()>1){
143  cerr<<"WARNING: Segments contain quadrupoles, votca cannot yet split them into charges. Your result will most likely be wrong."<<endl;
144  }
145  vec pos = aps->getPos() / AA_to_NM;
146  double Q = aps->getQ00();
147  string type = "mm";
148 
149  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
150 
151  if (split_dpl) {
152  vec tot_dpl = aps->getU1();
153  if (aps->getRank() > 0) {
154  tot_dpl += aps->getQ1();
155  }
156  // Calculate virtual charge positions
157  double a = dpl_spacing; // this is in nm
158  double mag_d = abs(tot_dpl); // this is in e * nm
159  vec dir_d_0 = tot_dpl.normalize();
160  vec dir_d = dir_d_0.normalize();
161  vec A = pos + 0.5 * a * dir_d / AA_to_NM; // converted to AA
162  vec B = pos - 0.5 * a * dir_d / AA_to_NM;
163  double qA = mag_d / a;
164  double qB = -qA;
165  // Zero out if magnitude small [e*nm]
166  if (aps->getIsoP() < 1e-9 || mag_d < 1e-9) {
167  A = aps->getPos() + 0.1 * a * vec(1, 0, 0); // != pos since self-energy may diverge
168  B = aps->getPos() - 0.1 * a * vec(1, 0, 0);
169  qA = 0;
170  qB = 0;
171  }
172  orb.AddAtom("A", A.x(), A.y(), A.z(), qA, true);
173  orb.AddAtom("B", B.x(), B.y(), B.z(), qB, true);
174  }
175  }
176  }
177 
178  // OUTER SHELL MM2
179  for (unsigned int i = 0; i < ptop->MM2().size(); ++i) {
180  ctp::PolarSeg *pseg = ptop->MM2()[i];
181  for (unsigned int j = 0; j < pseg->size(); ++j) {
182 
183  ctp::APolarSite *aps = (*pseg)[j];
184  vec pos = aps->getPos() / AA_to_NM;
185  double Q = aps->getQ00();
186  string type = "mm";
187 
188  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
189  }
190  }
191  return;
192  }
