35 void QMMIter::UpdateMPSFromGDMA(std::vector<std::vector<double> > &multipoles, std::vector< ctp::PolarSeg* > &psegs) {
36 
37 
38  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
39  ctp::PolarSeg *pseg = psegs[i];
40  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
41 
42  // Retrieve multipole info of this atom
43  std::vector<double> update = multipoles[qac];
44  while (update.size() < 9) update.push_back(0.0);
45 
46  // Convert e*(a_0)^k to e*(nm)^k where k = rank
47 
48  for (int m = 1; m < 4; m++) {
49  update[m] *= pow(tools::conv::bohr2nm, 1);

51  for (int m = 4; m < 9; m++) {
52  update[m] *= pow(tools::conv::bohr2nm, 2);
53  }
54 
55  ctp::APolarSite *aps = (*pseg)[j];
56 
57  aps->setQs(update, 0);

59  }

65  void QMMIter::UpdatePosChrgFromQMAtoms(std::vector< QMAtom* > &qmatoms,
66  std::vector< ctp::PolarSeg* > &psegs) {
67 
68 
69  double dR_RMS = 0.0;
70  double dQ_RMS = 0.0;
71  double dQ_SUM = 0.0;
72 
73  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
74  ctp::PolarSeg *pseg = psegs[i];
75  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
76 
77  // Retrieve info from QMAtom
78  QMAtom *qmatm = qmatoms[qac];
79  vec upd_r = qmatm->getPos();
80  upd_r *= tools::conv::bohr2nm;
81  double upd_Q00 = qmatm->getPartialcharge();
82 
83 
84  // Compare to previous r, Q00
85  ctp::APolarSite *aps = (*pseg)[j];
86  vec old_r = aps->getPos();
87  double old_Q00 = aps->getQ00();
88  double dR = abs(upd_r - old_r);
89  double dQ00 = upd_Q00 - old_Q00;
90 
91  dR_RMS += dR*dR;
92  dQ_RMS += dQ00*dQ00;
93  dQ_SUM += dQ00;
94 
95  // Forward updated r, Q00 to APS
96  aps->setPos(upd_r);
97  aps->setQ00(upd_Q00, 0);
98  }
99  }
100 
101 
102  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
103  dR_RMS /= qmatoms.size();
104  dQ_RMS /= qmatoms.size();
105  dR_RMS = sqrt(dR_RMS);
106  dQ_RMS = sqrt(dQ_RMS);
107 
108  // cout << " dR_RMS " << dR_RMS << "dQ RMS " << dQ_RMS << endl;
109 
110 
111  this->setdRdQ(dR_RMS, dQ_RMS, dQ_SUM);
112  return;
113  }
