39 void JobTopology::ModifyOptionsByJobFile(
40  std::vector<tools::Property*>& regions_def) const {
41 
42  const tools::Property& jobinput = _job.getInput();
43  std::vector<const tools::Property*> regions_def_job =
44  jobinput.Select("regions.region");
45 
46  std::string tag = "jobfile";
47  for (tools::Property* prop : regions_def) {
48  int id = prop->get("id").as<int>();
49  std::vector<std::string> paths = FindReplacePathsInOptions(*prop, tag);
50  if (!paths.empty()) {
51  XTP_LOG_SAVE(logINFO, _log) << " Region " << std::to_string(id)
52  << " is modified by jobfile" << std::flush;
53  XTP_LOG_SAVE(logINFO, _log)
54  << " Replacing the following paths with jobfile entries"
55  << std::flush;
56  for (const std::string& path : paths) {
57  XTP_LOG_SAVE(logINFO, _log) << " - " << path << std::flush;
58  }
59 
60  bool found_region_in_jobfile = false;
61  const tools::Property* job_prop = nullptr;
62  for (const tools::Property* prop_job : regions_def_job) {
63  int id2 = prop_job->get("id").as<int>();
64  if (id2 == id) {
65  job_prop = prop_job;
66  found_region_in_jobfile = true;
67  }
68  }
69  if (!found_region_in_jobfile) {
70  throw std::runtime_error("Region " + std::to_string(id) +
71  " was not found in jobfile.");
72  }
73  UpdateFromJobfile(*prop, *job_prop, paths);
74  }
75  }
76 }

78 void JobTopology::BuildRegions(const Topology& top, tools::Property options) {
79 
80  std::vector<tools::Property*> regions_def = options.Select("region");
81  CheckEnumerationOfRegions(regions_def);
82  SortRegionsDefbyId(regions_def);
83  ModifyOptionsByJobFile(regions_def);
84 
85  std::vector<std::vector<SegId> > region_seg_ids =
86  PartitionRegions(regions_def, top);
87 
88  // around this point the whole jobtopology will be centered
89  CreateRegions(options, top, region_seg_ids);
90  XTP_LOG_SAVE(logINFO, _log) << " Regions created" << std::flush;
91  for (const auto& region : _regions) {
92  XTP_LOG_SAVE(logINFO, _log) << *region << std::flush;
93  }
94 
95  return;
96 }

98 std::vector<std::string> JobTopology::FindReplacePathsInOptions(
99  const tools::Property& options, std::string tag) const {
100  std::vector<std::string> result;
101  std::string options_path = "options.qmmm.regions.region";
102  for (const tools::Property& sub : options) {
103  if (sub.HasChildren()) {
104  std::vector<std::string> subresult = FindReplacePathsInOptions(sub, tag);
105  result.insert(result.end(), subresult.begin(), subresult.end());
106  } else if (sub.value() == tag) {
107  std::string path = sub.path() + "." + sub.name();
108  std::size_t pos = path.find(options_path);
109  if (pos != std::string::npos) {
110  path.replace(pos, options_path.size(), "");
111  }
112  result.push_back(path);
113  }
114  }
115  return result;
116 }

130 template <class T>
131 void JobTopology::ShiftPBC(const Topology& top, const Eigen::Vector3d& center,
132  T& mol) const {
133  Eigen::Vector3d r_pbc = top.PbShortestConnect(center, mol.getPos());
134  Eigen::Vector3d r = mol.getPos() - center;
135  Eigen::Vector3d shift = r_pbc - r;
136  if (shift.norm() > 1e-9) {
137  mol.Translate(shift);
138  }
139 }

141 void JobTopology::CreateRegions(
142  const tools::Property& options, const Topology& top,
143  const std::vector<std::vector<SegId> >& region_seg_ids) {
144  std::string mapfile =
145  options.ifExistsReturnElseThrowRuntimeError<std::string>("mapfile");
146  std::vector<const tools::Property*> regions_def = options.Select("region");
147  // around this point the whole jobtopology will be centered for removing pbc
148  Eigen::Vector3d center = top.getSegment(region_seg_ids[0][0].Id()).getPos();
149 
150  for (const tools::Property* region_def : regions_def) {
151  int id = region_def->ifExistsReturnElseThrowRuntimeError<int>("id");
152  const std::vector<SegId>& seg_ids = region_seg_ids[id];
153  std::string type =
154  region_def->ifExistsReturnElseThrowRuntimeError<std::string>("type");
155  std::unique_ptr<Region> region;
156  QMRegion QMdummy(0, _log, "");
157  StaticRegion Staticdummy(0, _log);
158  PolarRegion Polardummy(0, _log);
159 
160  if (type == QMdummy.identify()) {
161  std::unique_ptr<QMRegion> qmregion =
162  std::unique_ptr<QMRegion>(new QMRegion(id, _log, _workdir));
163  QMMapper qmmapper(_log);
164  qmmapper.LoadMappingFile(mapfile);
165  for (const SegId& seg_index : seg_ids) {
166  const Segment& segment = top.getSegment(seg_index.Id());
167  QMMolecule mol = qmmapper.map(segment, seg_index);
168  mol.setType("qm" + std::to_string(id));
169  ShiftPBC(top, center, mol);
170  qmregion->push_back(mol);
171  }
172  region = std::move(qmregion);
173  } else if (type == Polardummy.identify()) {
174  std::unique_ptr<PolarRegion> polarregion =
175  std::unique_ptr<PolarRegion>(new PolarRegion(id, _log));
176  PolarMapper polmap(_log);
177  polmap.LoadMappingFile(mapfile);
178  for (const SegId& seg_index : seg_ids) {
179  const Segment& segment = top.getSegment(seg_index.Id());
180 
181  PolarSegment mol = polmap.map(segment, seg_index);
182 
183  ShiftPBC(top, center, mol);
184  mol.setType("mm" + std::to_string(id));
185  polarregion->push_back(mol);
186  }
187  region = std::move(polarregion);
188  } else if (type == Staticdummy.identify()) {
189  std::unique_ptr<StaticRegion> staticregion =
190  std::unique_ptr<StaticRegion>(new StaticRegion(id, _log));
191  StaticMapper staticmap(_log);
192  staticmap.LoadMappingFile(mapfile);
193  for (const SegId& seg_index : seg_ids) {
194  const Segment& segment = top.getSegment(seg_index.Id());
195  StaticSegment mol = staticmap.map(segment, seg_index);
196  mol.setType("mm" + std::to_string(id));
197  ShiftPBC(top, center, mol);
198  staticregion->push_back(mol);
199  }
200  region = std::move(staticregion);
201 
202  } else {
203  throw std::runtime_error("Region type not known!");
204  }
205  region->Initialize(*region_def);
206  _regions.push_back(std::move(region));
207  }
208 }

221 std::vector<std::vector<SegId> > JobTopology::PartitionRegions(
222  const std::vector<tools::Property*>& regions_def,
223  const Topology& top) const {
224 
225  std::vector<int> explicitly_named_segs_per_region;
226  std::vector<std::vector<SegId> > segids_per_region;
227  std::vector<bool> processed_segments =
228  std::vector<bool>(top.Segments().size(), false);
229  for (const tools::Property* region_def : regions_def) {
230 
231  if (!region_def->exists("segments") && !region_def->exists("cutoff")) {
232  throw std::runtime_error(
233  "Region definition needs either segments or a cutoff to find "
234  "segments");
235  }
236  std::vector<SegId> seg_ids;
237  if (region_def->exists("segments")) {
238  std::string seg_ids_string =
239  region_def->get("segments").as<std::string>();
240  tools::Tokenizer tok(seg_ids_string, " \n\t");
241  for (const std::string& seg_id_string : tok.ToVector()) {
242  seg_ids.push_back(SegId(seg_id_string));
243  }
244  for (const SegId& seg_id : seg_ids) {
245  if (seg_id.Id() > int(top.Segments().size() - 1)) {
246  throw std::runtime_error("Segment id is not in topology");
247  }
248  processed_segments[seg_id.Id()] = true;
249  }
250  }
251  explicitly_named_segs_per_region.push_back(seg_ids.size());
252 
253  if (region_def->exists("cutoff")) {
254  double cutoff = tools::conv::nm2bohr *
255  region_def->ifExistsReturnElseThrowRuntimeError<double>(
256  "cutoff.radius");
257 
258  std::string seg_geometry =
259  region_def->ifExistsReturnElseReturnDefault<std::string>(
260  "cutoff.geometry", "n");
261  double min = top.getBox().diagonal().minCoeff();
262  if (cutoff > 0.5 * min) {
263  throw std::runtime_error(
264  (boost::format("Cutoff is larger than half the box size. Maximum "
265  "allowed cutoff is %1$1.1f") %
266  (tools::conv::bohr2nm * 0.5 * min))
267  .str());
268  }
269  std::vector<SegId> center = seg_ids;
270  if (region_def->exists("cutoff.region")) {
271  int id = region_def->get("cutoff.region").as<int>();
272  bool only_explicit = region_def->ifExistsReturnElseReturnDefault<bool>(
273  "cutoff.relative_to_explicit_segs", false);
274  if (id < int(segids_per_region.size())) {
275  center = segids_per_region[id];
276  if (only_explicit) {
277  int no_of_segs = explicitly_named_segs_per_region[id];
278  if (no_of_segs == 0) {
279  throw std::runtime_error("Region with id '" + std::to_string(id) +
280  "' does not have");
281  }
282  center.resize(no_of_segs, SegId(0, "n"));
283  // need the second argument because resize can also increase
284  // capacity of vector and then needs a constructor,
285  // here we shrink, so should not happen
286  }
287  } else {
288  throw std::runtime_error("Region with id '" + std::to_string(id) +
289  "' used for cutoff does not exist");
290  }
291  }
292  if (center.empty()) {
293  throw std::runtime_error(
294  "Region needs either a segment or another region to which to apply "
295  "the cutoff");
296  }
297  for (const SegId& segid : center) {
298  const Segment& center_seg = top.getSegment(segid.Id());
299  for (const Segment& otherseg : top.Segments()) {
300  if (center_seg.getId() == otherseg.getId() ||
301  processed_segments[otherseg.getId()]) {
302  continue;
303  }
304  if (top.PbShortestConnect(center_seg.getPos(), otherseg.getPos())
305  .norm() < cutoff) {
306  seg_ids.push_back(SegId(otherseg.getId(), seg_geometry));
307  processed_segments[otherseg.getId()] = true;
308  }
309  }
310  }
311  }
312  segids_per_region.push_back(seg_ids);
313  }
314  return segids_per_region;
315 }

317 void JobTopology::CheckEnumerationOfRegions(
318  const std::vector<tools::Property*>& regions_def) const {
319  std::vector<int> reg_ids;
320  for (const tools::Property* region_def : regions_def) {
321  reg_ids.push_back(
322  region_def->ifExistsReturnElseThrowRuntimeError<int>("id"));
323  }
324 
325  std::vector<int> v(reg_ids.size());
326  std::iota(v.begin(), v.end(), 0);
327  if (!std::is_permutation(reg_ids.begin(), reg_ids.end(), v.begin())) {
328  throw std::runtime_error(
329  "Region id definitions are not clear. You must start at id 0 and then "
330  "ascending order. i.e. 0 1 2 3.");
331  }
332 }
