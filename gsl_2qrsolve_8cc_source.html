31 void linalg_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::vector<double> *residual)
32 {
33  // check matrix for zero column
34  int nonzero_found = 0;
35  for(size_t j=0; j<A.size2(); j++) {
36  nonzero_found = 0;
37  for(size_t i=0; i<A.size1(); i++) {
38  if(fabs(A(i,j))>0) {
39  nonzero_found = 1;
40  }
41  }
42  if(nonzero_found==0) {
43  throw "qrsolve_zero_column_in_matrix";
44  }
45  }
46 
47  gsl_matrix_view m
48  = gsl_matrix_view_array (&A(0,0), A.size1(), A.size2());
49 
50  gsl_vector_view gb
51  = gsl_vector_view_array (&b(0), b.size());
52 
53  gsl_vector *gsl_x = gsl_vector_alloc (x.size());
54  gsl_vector *tau = gsl_vector_alloc (x.size());
55  gsl_vector *gsl_residual = gsl_vector_alloc (b.size());
56 
57  gsl_linalg_QR_decomp (&m.matrix, tau);
58 
59  gsl_linalg_QR_lssolve (&m.matrix, tau, &gb.vector, gsl_x, gsl_residual);
60 
61  for (size_t i =0 ; i < x.size(); i++)
62  x(i) = gsl_vector_get(gsl_x, i);
63 
64  if(residual)
65  for (size_t i =0 ; i < residual->size(); i++)
66  (*residual)(i) = gsl_vector_get(gsl_residual, i);
67 
68  gsl_vector_free (gsl_x);
69  gsl_vector_free (tau);
70  gsl_vector_free (gsl_residual);
71 }

73 void linalg_constrained_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::matrix<double> &constr)

98  ub::identity_matrix<double> I (2*ngrid);
99  ub::vector<double> v(2*ngrid);
100 
101  Q = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
102  Q_k = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
103  v = ub::zero_vector<double>(2*ngrid);
104 
105  double *tmp = & constr(0,0);
106  gsl_matrix_view gsl_constr
107  = gsl_matrix_view_array (tmp, constr.size1(), constr.size2());
108 
109  tmp = &b(0);
110  gsl_vector_view gsl_b
111  = gsl_vector_view_array (tmp, b.size());
112 
113 
114  gsl_vector *tau_qr = gsl_vector_alloc (ngrid);
115 
116  gsl_linalg_QR_decomp (&gsl_constr.matrix, tau_qr);
117 
118  Q = I;
119 
120  for (int k = ngrid; k > 0 ; k--) {
121 
122  for (int icout = 0; icout < k - 1; icout++) {
123  v(icout) = 0;
124  }
125  v(k - 1) = 1.0;
126

129  }

132  Q = prec_prod(Q, Q_k);

134  }

144  ub::matrix<double> A2 = ub::matrix_range<ub::matrix<double> >(A,

146  );

154  gsl_vector *tau_solve = gsl_vector_alloc (ngrid); // already done!

178 }
