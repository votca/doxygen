31 void linalg_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::vector<double> *residual)
32 {
33  // check matrix for zero column
34  int nonzero_found = 0;
35  for(size_t j=0; j<A.size2(); j++) {
36  nonzero_found = 0;
37  for(size_t i=0; i<A.size1(); i++) {
38  if(fabs(A(i,j))>0) {
39  nonzero_found = 1;
40  }
41  }
42  if(nonzero_found==0) {
43  throw "qrsolve_zero_column_in_matrix";
44  }
45  }
46 
47  gsl_matrix_view m
48  = gsl_matrix_view_array (&A(0,0), A.size1(), A.size2());
49 
50  gsl_vector_view gb
51  = gsl_vector_view_array (&b(0), b.size());
52 
53  gsl_vector *gsl_x = gsl_vector_alloc (x.size());
54  gsl_vector *tau = gsl_vector_alloc (x.size());
55  gsl_vector *gsl_residual = gsl_vector_alloc (b.size());
56 
57  gsl_linalg_QR_decomp (&m.matrix, tau);
58 
59  gsl_linalg_QR_lssolve (&m.matrix, tau, &gb.vector, gsl_x, gsl_residual);
60 
61  for (size_t i =0 ; i < x.size(); i++)
62  x(i) = gsl_vector_get(gsl_x, i);
63 
64  if(residual)
65  for (size_t i =0 ; i < residual->size(); i++)
66  (*residual)(i) = gsl_vector_get(gsl_residual, i);
67 
68  gsl_vector_free (gsl_x);
69  gsl_vector_free (tau);
70  gsl_vector_free (gsl_residual);
71 }

74 void linalg_constrained_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::matrix<double> &constr)
75 {
76  // check matrix for zero column
77  int nonzero_found = 0;
78  for(size_t j=0; j<A.size2(); j++) {
79  nonzero_found = 0;
80  for(size_t i=0; i<A.size1(); i++) {
81  if(fabs(A(i,j))>0) {
82  nonzero_found = 1;
83  }
84  }
85  if(nonzero_found==0) {
86  throw std::runtime_error("constrained_qrsolve_zero_column_in_matrix");
87  }
88  }
89 
90  const int N = b.size();
91  const int ngrid = x.size()/2;
92  const int ysize = constr.size1(); //number of constraints is number of rows of constr
93  //std::cout << "ysize: " << ysize << std::endl;
94 
95  //std::cout << "constr^T.size1: " << constr.size1() << " , constr^T.size2: " << constr.size2() << std::endl;
96 
97  // Transpose constr:
98  constr = trans(constr);
99 
100  // temporary variables
101  ub::matrix<double> Q(2*ngrid, 2*ngrid);
102  // Q matrix: QR decomposition of trans(B) (trans(B) has 2*ngrid rows and ysize columns )
103  // Q matrix is orthogonal 2*ngrid x 2*ngrid matrix
104  ub::matrix<double> Q_k(2*ngrid, 2*ngrid);
105  ub::identity_matrix<double> I (2*ngrid);
106  ub::vector<double> v(2*ngrid);
107 
108  Q = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
109  Q_k = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
110  v = ub::zero_vector<double>(2*ngrid);
111 
112  double *tmp = & constr(0,0);
113  gsl_matrix_view gsl_constr
114  = gsl_matrix_view_array (tmp, constr.size1(), constr.size2());
115 
116  tmp = &b(0);
117  gsl_vector_view gsl_b
118  = gsl_vector_view_array (tmp, b.size());
119 
120 
121  //vector to store Householder coefficients of QR-decomposition of trans(B)
122  gsl_vector *tau_qr = gsl_vector_alloc (ysize);
123 
124  gsl_linalg_QR_decomp (&gsl_constr.matrix, tau_qr);
125 
126  //construct the matrix Q of trans(B)=Q * R
127  Q = I;
128 
129  //k=1,..,ysize (number of rows of B, meaning number of constraints)
130  for (int k = ysize; k > 0 ; k--) {
131 
132  for (int icout = 0; icout < k - 1; icout++) {
133  v(icout) = 0;
134  }
135  v(k - 1) = 1.0;
136 
137  for (int icout = k; icout < 2*ngrid; icout++) {
138  v(icout) = gsl_matrix_get(&gsl_constr.matrix, icout, k - 1 );
139  }
140 
141  Q_k = I - gsl_vector_get(tau_qr, k - 1 ) * outer_prod ( v, v );
142  Q = prec_prod(Q, Q_k);
143 
144  }
145 
146  //now Q is the one of trans(B)=Q * R
147  Q = trans(Q);
148  gsl_vector_free (tau_qr);
149 
150  // Calculate A * Q and store the result in A
151  A = prec_prod(A, Q);
152 
153 
154  // A = [A1 A2], so A2 is just a block of A
155  // [A1 A2] has N rows. A1 has ysize columns (range(0,ysize) ).
156  //A2 has 2*ngrid-ysize columns ( range(ysize,2*ngrid) ).
157  ub::matrix<double> A2 = ub::matrix_range<ub::matrix<double> >(A,
158  ub::range (0, N), ub::range (ysize, 2*ngrid)
159  );
160 
161  tmp = &A2(0,0);
162  gsl_matrix_view gsl_A2
163  = gsl_matrix_view_array (tmp, A2.size1(), A2.size2());
164 
165 
166 
167  //now perform QR-decomposition of A2 to solve the least-squares problem A2 * z = b
168  //A2 has N rows and (2*ngrid-ysize) columns ->
169  gsl_vector *z = gsl_vector_alloc (2*ngrid-ysize);
170  gsl_vector *tau_solve = gsl_vector_alloc (2*ngrid-ysize);
171  gsl_vector *residual = gsl_vector_alloc (N);
172 
173  gsl_linalg_QR_decomp (&gsl_A2.matrix, tau_solve);
174  gsl_linalg_QR_lssolve (&gsl_A2.matrix, tau_solve, &gsl_b.vector, z, residual);
175 
176  // Next two cycles assemble vector from y (which is zero-vector) and z
177  // (which we just got by gsl_linalg_QR_lssolve)
178 
179  //vector y has ysize components
180  for (int i = 0; i < ysize; i++ ) {
181  x[i] = 0.0;
182  }
183 
184  //vector z has (2*ngrid-ysize) components
185  for (int i = ysize; i < 2 * ngrid; i++ ) {
186  x[i] = gsl_vector_get(z, i - ysize);
187  }
188 
189  /*for (int i = 0; i < 2 * ngrid; i++ ) {
190  std::cout << "x[" << i << "]: " << x[i] << std::endl;
191  }*/
192 
193  // To get the final answer this vector should be multiplied by matrix Q
194  // TODO: here i changed the sign, check again! (victor)
195  x = -prec_prod( Q, x );
196 
197  /*for (int i = 0; i < 2 * ngrid; i++ ) {
198  std::cout << "x[" << i << "]: " << x[i] << std::endl;
199  }*/
200 
201  gsl_vector_free (z);
202  gsl_vector_free (tau_solve);
203  gsl_vector_free (residual);
204 }
