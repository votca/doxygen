00031 void linalg_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::vector<double> *residual)
00032 {
00033     // check matrix for zero column

00036         nonzero_found = 0;

00038             if(fabs(A(i,j))>0) {
00039                 nonzero_found = 1;
