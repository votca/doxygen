31 void linalg_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::vector<double> *residual)
32 {
33  // check matrix for zero column
34  int nonzero_found = 0;
35  for(size_t j=0; j<A.size2(); j++) {
36  nonzero_found = 0;
37  for(size_t i=0; i<A.size1(); i++) {
38  if(fabs(A(i,j))>0) {
39  nonzero_found = 1;
40  }
41  }
42  if(nonzero_found==0) {
43  throw "qrsolve_zero_column_in_matrix";
44  }
45  }
46 
47  gsl_matrix_view m
48  = gsl_matrix_view_array (&A(0,0), A.size1(), A.size2());
49 
50  gsl_vector_view gb
51  = gsl_vector_view_array (&b(0), b.size());
52 
53  gsl_vector *gsl_x = gsl_vector_alloc (x.size());
54  gsl_vector *tau = gsl_vector_alloc (x.size());
55  gsl_vector *gsl_residual = gsl_vector_alloc (b.size());
56 
57  gsl_linalg_QR_decomp (&m.matrix, tau);
58 
59  gsl_linalg_QR_lssolve (&m.matrix, tau, &gb.vector, gsl_x, gsl_residual);
60 
61  for (size_t i =0 ; i < x.size(); i++)
62  x(i) = gsl_vector_get(gsl_x, i);
63 
64  if(residual)
65  for (size_t i =0 ; i < residual->size(); i++)
66  (*residual)(i) = gsl_vector_get(gsl_residual, i);
67 
68  gsl_vector_free (gsl_x);
69  gsl_vector_free (tau);
70  gsl_vector_free (gsl_residual);
71 }

74 void linalg_constrained_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::matrix<double> &B)
75 {
76 
77  // check matrix for zero column
78  int nonzero_found = 0;
79  for(size_t j=0; j<A.size2(); j++) {
80  nonzero_found = 0;
81  for(size_t i=0; i<A.size1(); i++) {
82  if(fabs(A(i,j))>0) {
83  nonzero_found = 1;
84  }
85  }
86  if(nonzero_found==0) {
87  throw std::runtime_error("constrained_qrsolve_zero_column_in_matrix");
88  }
89  }
90 
91  const int NoEquations = b.size();
92  const int NoUnknowns = x.size();
93  const int Noconstrains = B.size1(); //number of constraints is number of rows of constr
94 
95  // Transpose constr:
96  B = ub::trans(B);
97 
98  // temporary variables
99  ub::matrix<double> Q(NoUnknowns, NoUnknowns);
100  // Q matrix: QR decomposition of trans(B)
101  ub::matrix<double> Q_k(NoUnknowns, NoUnknowns);
102  ub::identity_matrix<double> I (NoUnknowns);
103  ub::vector<double> v(NoUnknowns);
104 
105  Q = ub::zero_matrix<double>(NoUnknowns, NoUnknowns);
106  Q_k = ub::zero_matrix<double>(NoUnknowns, NoUnknowns);
107  v = ub::zero_vector<double>(NoUnknowns);
108 
109  double *tmp = & B(0,0);
110  gsl_matrix_view gsl_constr
111  = gsl_matrix_view_array (tmp, B.size1(), B.size2());
112 
113  tmp = &b(0);
114  gsl_vector_view gsl_b
115  = gsl_vector_view_array (tmp, b.size());
116 
117 
118  //vector to store Householder coefficients of QR-decomposition of trans(B)
119  gsl_vector *tau_qr = gsl_vector_alloc (Noconstrains);
120  gsl_linalg_QR_decomp (&gsl_constr.matrix, tau_qr);
121  //construct the matrix Q of trans(B)=Q * R
122  Q = I;
123 
124  //k=1,..,ysize (number of rows of B, meaning number of constraints)
125  for (int k = Noconstrains; k > 0 ; k--) {
126 
127  for (int icout = 0; icout < k - 1; icout++) {
128  v(icout) = 0;
129  }
130  v(k - 1) = 1.0;
131 
132  for (int icout = k; icout < NoUnknowns; icout++) {
133  v(icout) = gsl_matrix_get(&gsl_constr.matrix, icout, k - 1 );
134  }
135 
136  Q_k = I - gsl_vector_get(tau_qr, k - 1 ) * ub::outer_prod ( v, v );
137  Q = ub::prec_prod(Q, Q_k);
138 
139  }
140  //now Q is the one of trans(B)=Q * R
141  Q = ub::trans(Q);
142  gsl_vector_free (tau_qr);
143  // Calculate A * Q and store the result in A
144  A = ub::prec_prod(A, Q);
145 
146  // A = [A1 A2], so A2 is just a block of A
147  // [A1 A2] has N rows. A1 has ysize columns (range(0,ysize) ).
148  //A2 has 2*ngrid-ysize columns ( range(ysize,2*ngrid) ).
149  ub::matrix<double> A2 = ub::matrix_range<ub::matrix<double> >(A,
150  ub::range (0, NoEquations), ub::range (Noconstrains, NoUnknowns)
151  );
152 
153  tmp = &A2(0,0);
154  gsl_matrix_view gsl_A2
155  = gsl_matrix_view_array (tmp, A2.size1(), A2.size2());
156 
157 
158  //now perform QR-decomposition of A2 to solve the least-squares problem A2 * z = b
159  //A2 has N rows and (2*ngrid-ysize) columns ->
160  gsl_vector *z = gsl_vector_alloc (NoUnknowns-Noconstrains);
161  gsl_vector *tau_solve = gsl_vector_alloc (NoUnknowns-Noconstrains);
162  gsl_vector *residual = gsl_vector_alloc (NoEquations);
163  gsl_linalg_QR_decomp (&gsl_A2.matrix, tau_solve);
164  gsl_linalg_QR_lssolve (&gsl_A2.matrix, tau_solve, &gsl_b.vector, z, residual);
165  // Next two cycles assemble vector from y (which is zero-vector) and z
166  // (which we just got by gsl_linalg_QR_lssolve)
167 
168  //vector y has ysize components
169  for (int i = 0; i < Noconstrains; i++ ) {
170  x[i] = 0.0;
171  }
172 
173  for (int i = Noconstrains; i < NoUnknowns; i++ ) {
174  x[i] = gsl_vector_get(z, i - Noconstrains);
175  }
176 
177  x = ub::prec_prod( Q, x );
178 
179  gsl_vector_free (z);
180  gsl_vector_free (tau_solve);
181  gsl_vector_free (residual);
182 }
