00515 {

00518     Table target;

00520 
00521     if(!interaction->_is_bonded) {
00522         for(unsigned int i=0; i<target.y().size(); ++i) {
00523             double x1 = target.x()[i] - 0.5*interaction->_step;
00524             double x2 = x1 + interaction->_step;
00525             if(x1<0)
00526                 x1=x2=0;
00527             target.y()[i] = 1./(_avg_vol.getAvg()*interaction->_norm) *
00528                 target.y()[i] * (4./3.*M_PI*(x2*x2*x2 - x1*x1*x1));                
00529         }        
00530     }
00531     else {
00532             target.y() = (1.0 / interaction->_norm)*target.y();
00533     }
00534     if(target.y().size() !=  interaction->_average.data().y().size())
00535         throw std::runtime_error("number of grid points in target does not match the grid");
00536
