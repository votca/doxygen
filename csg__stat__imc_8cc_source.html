37 void Imc::Initialize() {
38  // do some output
39  if (_do_imc) {
40  cout << "begin to calculate inverse monte carlo parameters\n";
41  } else {
42  cout << "begin to calculate distribution functions\n";
43  }
44  cout << "# of bonded interactions: " << _bonded.size() << endl;
45  cout << "# of non-bonded interactions: " << _nonbonded.size() << endl;
46 
47  if (_bonded.size() + _nonbonded.size() == 0) {
48  throw std::runtime_error(
49  "No interactions defined in options xml-file - nothing to be done");
50  }
51 
52  // initialize non-bonded structures
53  for (tools::Property *prop : _nonbonded) {
54  interaction_t *i = AddInteraction(prop);
55  i->_is_bonded = false;
56  }
57 
58  // initialize bonded structures
59  for (tools::Property *prop : _bonded) {
60  interaction_t *i = AddInteraction(prop);
61  i->_is_bonded = true;
62  }
63 
64  // initialize the group structures
65  if (_do_imc) {
66  InitializeGroups();
67  }
68 }

70 void Imc::BeginEvaluate(Topology *top, Topology *) {
71  // we didn't process any frames so far
72  _nframes = 0;
73  _nblock = 0;
74  _processed_some_frames = false;
75 
76  // initialize non-bonded structures
77  for (tools::Property *prop : _nonbonded) {
78  string name = prop->get("name").value();
79 
80  interaction_t &i = *_interactions[name];
81 
82  // Preliminary: Quickest way to incorporate 3 body correlations
83  if (i._threebody) {
84 
85  // generate the bead lists
86  BeadList beads1, beads2, beads3;
87 
88  beads1.Generate(*top, prop->get("type1").value());
89  beads2.Generate(*top, prop->get("type2").value());
90  beads3.Generate(*top, prop->get("type3").value());
91 
92  if (beads1.size() == 0) {
93  throw std::runtime_error(
94  "Topology does not have beads of type \"" +
95  prop->get("type1").value() +
96  "\"\n"
97  "This was specified in type1 of interaction \"" +
98  name + "\"");
99  }
100  if (beads2.size() == 0) {
101  throw std::runtime_error(
102  "Topology does not have beads of type \"" +
103  prop->get("type2").value() +
104  "\"\n"
105  "This was specified in type2 of interaction \"" +
106  name + "\"");
107  }
108  if (beads3.size() == 0) {
109  throw std::runtime_error(
110  "Topology does not have beads of type \"" +
111  prop->get("type3").value() +
112  "\"\n"
113  "This was specified in type3 of interaction \"" +
114  name + "\"");
115  }
116  }
117  // 2body
118  if (!i._threebody) {
119 
120  // generate the bead lists
121  BeadList beads1, beads2;
122 
123  beads1.Generate(*top, prop->get("type1").value());
124  beads2.Generate(*top, prop->get("type2").value());
125 
126  if (beads1.size() == 0) {
127  throw std::runtime_error(
128  "Topology does not have beads of type \"" +
129  prop->get("type1").value() +
130  "\"\n"
131  "This was specified in type1 of interaction \"" +
132  name + "\"");
133  }
134  if (beads2.size() == 0) {
135  throw std::runtime_error(
136  "Topology does not have beads of type \"" +
137  prop->get("type2").value() +
138  "\"\n"
139  "This was specified in type2 of interaction \"" +
140  name + "\"");
141  }
142 
143  // calculate normalization factor for rdf
144  if (prop->get("type1").value() == prop->get("type2").value()) {
145  i._norm = 2. / (double)(beads1.size() * beads2.size());
146  } else {
147  i._norm = 1. / (double)(beads1.size() * beads2.size());
148  }
149  }
150  }
151 
152  for (tools::Property *prop : _bonded) {
153  string name = prop->get("name").value();
154 
155  std::list<Interaction *> list = top->InteractionsInGroup(name);
156  if (list.empty()) {
157  throw std::runtime_error(
158  "Bonded interaction '" + name +
159  "' defined in options xml-file, but not in topology - check name "
160  "definition in the mapping file again");
161  }
162  }

166 Imc::interaction_t *Imc::AddInteraction(tools::Property *p) {
167  string name = p->get("name").value();
168  string group;
169  if (_do_imc) {
170  group = p->get("inverse.imc.group").value();
171  } else {
172  group = "none";
173  }
174 
175  votca::Index index = Index(_interactions.size());
176  auto success = _interactions.insert(std::make_pair(
177  name, std::unique_ptr<interaction_t>(new interaction_t())));
178  interaction_t *i = success.first->second.get();
179  i->_index = index;
180  getGroup(group)->_interactions.push_back(i);
181 
182  i->_step = p->get("step").as<double>();
183  i->_min = p->get("min").as<double>();
184  i->_max = p->get("max").as<double>();
185  i->_norm = 1.0;
186  i->_p = p;
187 
188  // if option threebody does not exist, replace it by default of 0
189  i->_threebody = p->ifExistsReturnElseReturnDefault<bool>("threebody", 0);
190 
191  // if option force does not exist, replace it by default of 0
192  i->_force = p->ifExistsReturnElseReturnDefault<bool>("force", 0);
193 
194  // if option cut does not exist, replace it by default of 0.37 nm
195  i->_cut = p->ifExistsReturnElseReturnDefault<double>("cut", 0.37);
196 
197  // initialize the current and average histogram
198  votca::Index n =
199  static_cast<votca::Index>((i->_max - i->_min) / i->_step + 1.000000001);
200 
201  i->_average.Initialize(i->_min, i->_max, n);
202  if (i->_force) {
203  i->_average_force.Initialize(i->_min, i->_max, n);
204  }
205 
206  return i;
207 }

273 // process non-bonded interactions for current frame
274 void Imc::Worker::DoNonbonded(Topology *top) {
275  for (tools::Property *prop : _imc->_nonbonded) {
276  string name = prop->get("name").value();
277 
278  interaction_t &i = *_imc->_interactions[name];
279 
280  // clear the current histogram
281  _current_hists[i._index].Clear();
282  _current_hists_force[i._index].Clear();
283 
284  bool gridsearch = true;
285 
286  if (_imc->_options.exists("cg.nbsearch")) {
287  if (_imc->_options.get("cg.nbsearch").as<string>() == "grid") {
288  gridsearch = true;
289  } else if (_imc->_options.get("cg.nbsearch").as<string>() == "simple") {
290  gridsearch = false;
291  } else {
292  throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
293  }
294  }
295 
296  // Preleminary: Quickest way to incorporate 3 body correlations
297  if (i._threebody) {
298 
299  // generate the bead lists
300  BeadList beads1, beads2, beads3;
301 
302  beads1.Generate(*top, prop->get("type1").value());
303  beads2.Generate(*top, prop->get("type2").value());
304  beads3.Generate(*top, prop->get("type3").value());
305 
306  // generate the neighbour list
307  std::unique_ptr<NBList_3Body> nb;
308 
309  if (gridsearch) {
310  nb = std::unique_ptr<NBList_3Body>(new NBListGrid_3Body());
311  } else {
312  nb = std::unique_ptr<NBList_3Body>(new NBList_3Body());
313  }
314 
315  nb->setCutoff(i._cut); // implement different cutoffs for different
316  // interactions!
317  // Here, a is the distance between two beads of a triple, where the 3-body
318  // interaction is zero
319 
320  // check if type1 and type2 are the same
321  if (prop->get("type1").value() == prop->get("type2").value()) {
322  // if all three types are the same
323  if (prop->get("type2").value() == prop->get("type3").value()) {
324  nb->Generate(beads1, true);
325  }
326  // if type2 and type3 are different, use the Generate function for 2
327  // bead types
328  if (prop->get("type2").value() != prop->get("type3").value()) {
329  nb->Generate(beads1, beads3, true);
330  }
331  }
332  // if type1 != type2
333  if (prop->get("type1").value() != prop->get("type2").value()) {
334  // if the last two types are the same, use Generate function with them
335  // as the first two bead types Neighborlist_3body is constructed in a
336  // way that the two equal bead types have two be the first 2 types
337  if (prop->get("type2").value() == prop->get("type3").value()) {
338  nb->Generate(beads1, beads2, true);
339  }
340  if (prop->get("type2").value() != prop->get("type3").value()) {
341  // type1 = type3 !=type2
342  if (prop->get("type1").value() == prop->get("type3").value()) {
343  nb->Generate(beads2, beads1, true);
344  }
345  // type1 != type2 != type3
346  if (prop->get("type1").value() != prop->get("type3").value()) {
347  nb->Generate(beads1, beads2, beads3, true);
348  }
349  }
350  }
351 
352  for (auto &triple : *nb) {
353  Eigen::Vector3d rij = triple->r12();
354  Eigen::Vector3d rik = triple->r13();
355  double var = std::acos(rij.dot(rik) /
356  sqrt(rij.squaredNorm() * rik.squaredNorm()));
357  _current_hists[i._index].Process(var);
358  }
359  }
360  // 2body interaction
361  if (!i._threebody) {
362 
363  // generate the bead lists
364  BeadList beads1, beads2;
365 
366  beads1.Generate(*top, prop->get("type1").value());
367  beads2.Generate(*top, prop->get("type2").value());
368 
369  {
370  // generate the neighbour list
371  std::unique_ptr<NBList> nb;
372  if (gridsearch) {
373  nb = std::unique_ptr<NBList>(new NBListGrid());
374  } else {
375  nb = std::unique_ptr<NBList>(new NBListGrid());
376  }
377 
378  nb->setCutoff(i._max + i._step);
379 
380  IMCNBSearchHandler h(&(_current_hists[i._index]));
381 
382  nb->SetMatchFunction(&h, &IMCNBSearchHandler::FoundPair);
383 
384  // is it same types or different types?
385  if (prop->get("type1").value() == prop->get("type2").value()) {
386  nb->Generate(beads1);
387  } else {
388  nb->Generate(beads1, beads2);
389  }
390  }
391 
392  // if one wants to calculate the mean force
393  if (i._force) {
394  std::unique_ptr<NBList> nb_force;
395  if (gridsearch) {
396  nb_force = std::unique_ptr<NBList>(new NBListGrid());
397  } else {
398  nb_force = std::unique_ptr<NBList>(new NBListGrid());
399  }
400 
401  nb_force->setCutoff(i._max + i._step);
402 
403  // is it same types or different types?
404  if (prop->get("type1").value() == prop->get("type2").value()) {
405  nb_force->Generate(beads1);
406  } else {
407  nb_force->Generate(beads1, beads2);
408  }
409 
410  // process all pairs to calculate the projection of the
411  // mean force on bead 1 on the pair distance: F1 * r12
412  for (auto &pair : *nb_force) {
413  Eigen::Vector3d F2 = pair->second()->getF();
414  Eigen::Vector3d F1 = pair->first()->getF();
415  Eigen::Vector3d r12 = pair->r();
416  r12.normalize();
417  double var = pair->dist();
418  double scale = 0.5 * (F2 - F1).dot(r12);
419  _current_hists_force[i._index].Process(var, scale);
420  }
421  }
422  }
423  }
424 }

456 // initialize the groups after interactions are added
457 void Imc::InitializeGroups() {
458  if (!_do_imc) {
459  return;
460  }
461  map<string, group_t *>::iterator group_iter;
462 
463  // clear all the pairs
464 
465  // iterator over all groups
466  for (auto &group : _groups) {
467  auto &grp = group.second;
468  grp->_pairs.clear();
469 
470  auto &interactions = grp->_interactions;
471  // count number of bins needed in matrix
472  votca::Index n = std::accumulate(interactions.begin(), interactions.end(),
473  0, [](votca::Index j, interaction_t *i) {
474  return j + i->_average.getNBins();
475  });
476 
477  // handy access to matrix
478  group_matrix &M = grp->_corr;
479 
480  // initialize matrix with zeroes
481  M = Eigen::MatrixXd::Zero(n, n);
482 
483  // now create references to the sub matrices
484  // iterate over all possible cominations of pairs
485  for (votca::Index i = 0; i < votca::Index(interactions.size()); i++) {
486  votca::Index n1 = interactions[i]->_average.getNBins();
487  for (votca::Index j = i; j < votca::Index(interactions.size()); j++) {
488  votca::Index n2 = interactions[j]->_average.getNBins();
489  // create matrix proxy with sub-matrix
490  pair_matrix corr = M.block(i, j, n1, n2);
491  // add the pair
492  grp->_pairs.push_back(
493  pair_t(interactions[i], interactions[j], i, j, corr));
494  }
495  }
496  }
497 }

499 // update the correlation matrix
500 void Imc::DoCorrelations(Imc::Worker *worker) {
501  if (!_do_imc) {
502  return;
503  }
504 
505  for (auto &group : _groups) {
506  auto &grp = group.second;
507  // update correlation for all pairs
508  for (auto &pair : grp->_pairs) {
509  Eigen::VectorXd &a = worker->_current_hists[pair._i1->_index].data().y();
510  Eigen::VectorXd &b = worker->_current_hists[pair._i2->_index].data().y();
511  pair_matrix &M = pair._corr;
512 
513  M = ((((double)_nframes - 1.0) * M) + a * b.transpose()) /
514  (double)_nframes;
515  }
516  }
517 }

519 // write the distribution function
520 void Imc::WriteDist(const string &suffix) {
521  map<string, interaction_t *>::iterator iter;
522 
523  cout << std::endl; // Cosmetic, put \n before printing names of distribution
524  // files.
525  // for all interactions
526  for (auto &pair : _interactions) {
527  // calculate the rdf
528  auto &interaction = pair.second;
529  votca::tools::Table &t = interaction->_average.data();
530 
531  // if no average force calculation, dummy table
532  votca::tools::Table force;
533  // if average force calculation, table force contains force data
534  if (interaction->_force) {
535  force = interaction->_average_force.data();
536  }
537 
538  votca::tools::Table dist(t);
539  if (!interaction->_is_bonded) {
540  // Quickest way to incorporate 3 body correlations
541  if (interaction->_threebody) {
542  // \TODO normalize bond and angle differently....
543  double norm = dist.y().cwiseAbs().sum();
544  if (norm > 0) {
545  dist.y() =
546  interaction->_norm * dist.y() / (norm * interaction->_step);
547  }
548  }
549 
550  // 2body
551  if (!interaction->_threebody) {
552  // force normalization
553  // normalize by number of pairs found at a specific distance
554  for (votca::Index i = 0; i < force.y().size(); ++i) {
555  // check if any number of pairs has been found at this distance, then
556  // normalize
557  if (dist.y()[i] != 0) {
558  force.y()[i] /= dist.y()[i];
559  }
560  // else set to zero
561  else {
562  force.y()[i] = 0;
563  }
564  }
565 
566  // normalization is calculated using exact shell volume (difference of
567  // spheres)
568  for (votca::Index i = 0; i < dist.y().size(); ++i) {
569  double x1 = dist.x()[i] - 0.5 * interaction->_step;
570  double x2 = x1 + interaction->_step;
571  if (x1 < 0) {
572  dist.y()[i] = 0;
573  } else {
574  dist.y()[i] = _avg_vol.getAvg() * interaction->_norm * dist.y()[i] /
575  (4. / 3. * M_PI * (x2 * x2 * x2 - x1 * x1 * x1));
576  }
577  }
578  }
579 
580  } else {
581  // \TODO normalize bond and angle differently....
582  double norm = dist.y().cwiseAbs().sum();
583  if (norm > 0) {
584  dist.y() = interaction->_norm * dist.y() / (norm * interaction->_step);
585  }
586  }
587 
588  dist.Save((pair.first) + suffix);
589  cout << "written " << (pair.first) + suffix << "\n";
590 
591  // preliminary
592  if (interaction->_force) {
593  force.Save((pair.first) + ".force.new");
594  cout << "written " << (pair.first) + ".force.new"
595  << "\n";
596  }
597  }
598 }

606 void Imc::WriteIMCData(const string &suffix) {
607  if (!_do_imc) {
608  return;
609  }
610  // map<string, interaction_t *>::iterator ic_iter;
611  map<string, group_t *>::iterator group_iter;
612 
613  // iterate over all groups
614  for (auto &group : _groups) {
615  auto &grp = group.second;
616  string grp_name = group.first;
617 
618  // number of total bins for all interactions in group is matrix dimension
619  votca::Index n = grp->_corr.rows();
620 
621  // build full set of equations + copy some data to make
622  // code better to read
623  group_matrix gmc(grp->_corr);
624  tools::Table dS;
625  dS.resize(n);
626  // the next two variables are to later extract the individual parts
627  // from the whole data after solving equations
628  vector<std::pair<std::string, votca::tools::RangeParser> >
629  ranges; // names of the interactions & sizes of the individual
630  // interactions
631 
632  // copy all averages+r of group to one vector
633  n = 0;
634  votca::Index begin = 1;
635  for (interaction_t *ic : grp->_interactions) {
636 
637  // sub vector for dS
638  Eigen::VectorBlock<Eigen::VectorXd> sub_dS =
639  dS.y().segment(n, ic->_average.getNBins());
640 
641  // sub vector for r
642  Eigen::VectorBlock<Eigen::VectorXd> sub_r =
643  dS.x().segment(n, ic->_average.getNBins());
644 
645  // read in target and calculate dS
646  CalcDeltaS(ic, sub_dS);
647 
648  // copy r
649  sub_r = ic->_average.data().x();
650 
651  // save size
652  votca::tools::RangeParser rp;
653  votca::Index end = begin + ic->_average.getNBins() - 1;
654  rp.Add(begin, end);
655  ranges.push_back(std::pair<std::string, votca::tools::RangeParser>(
656  ic->_p->get("name").as<string>(), rp));
657  begin = end + 1;
658  // save name
659 
660  // shift subrange by size of current
661  n += ic->_average.getNBins();
662  }
663 
664  // now we need to calculate the
665  // A_ij = <S_i*S_j> - <S_i>*<S_j>
666  for (pair_t &pair : grp->_pairs) {
667  interaction_t *i1 = pair._i1;
668  interaction_t *i2 = pair._i2;
669 
670  // make reference to <S_i>
671  Eigen::VectorXd &a = i1->_average.data().y();
672  // make reference to <S_j>
673  Eigen::VectorXd &b = i2->_average.data().y();
674 
675  votca::Index i = pair._offset_i;
676  votca::Index j = pair._offset_j;
677  votca::Index n1 = i1->_average.getNBins();
678  votca::Index n2 = i2->_average.getNBins();
679 
680  pair_matrix M = gmc.block(i, j, n1, n2);
681  M = -(M - a * b.transpose());
682  // matrix is symmetric

692 // calculate deviation from target vectors
693 void Imc::CalcDeltaS(interaction_t *interaction,
694  Eigen::VectorBlock<Eigen::VectorXd> &dS) {
695  const string &name = interaction->_p->get("name").as<string>();
696 
697  tools::Table target;
698  target.Load(name + ".dist.tgt");
699 
700  if (!interaction->_is_bonded) {
701  for (votca::Index i = 0; i < target.y().size(); ++i) {
702  double x1 = target.x()[i] - 0.5 * interaction->_step;
703  double x2 = x1 + interaction->_step;
704  if (x1 < 0) {
705  x1 = x2 = 0;
706  }
707  target.y()[i] = 1. / (_avg_vol.getAvg() * interaction->_norm) *
708  target.y()[i] *
709  (4. / 3. * M_PI * (x2 * x2 * x2 - x1 * x1 * x1));
710  }
711  } else {
712  target.y() = (1.0 / interaction->_norm) * target.y();
713  }
714  if (target.y().size() != interaction->_average.data().y().size()) {
715  throw std::runtime_error(
716  "number of grid points in target does not match the grid");
717  }
718 
719  dS = interaction->_average.data().y() - target.y();
720 }

722 void Imc::WriteIMCBlock(const string &suffix) {
723 
724  if (!_do_imc) {
725  return;
726  }
727 
728  // iterate over all groups
729  for (auto &group : _groups) {
730  auto &grp = group.second;
731  string grp_name = group.first;
732  list<interaction_t *>::iterator iter;
733 
734  // number of total bins for all interactions in group is matrix dimension
735  votca::Index n = grp->_corr.rows();
736 
737  // build full set of equations + copy some data to make code better to read
738  group_matrix gmc(grp->_corr);
739  Eigen::VectorXd dS(n);
740  Eigen::VectorXd r(n);
741  // the next two variables are to later extract the individual parts
742  // from the whole data after solving equations
743  vector<votca::Index> sizes; // sizes of the individual interactions
744  vector<string> names; // names of the interactions
745 
746  // copy all averages+r of group to one vector
747  n = 0;
748  for (interaction_t *ic : grp->_interactions) {
749  // sub vector for dS
750  Eigen::VectorBlock<Eigen::VectorXd> sub_dS =
751  dS.segment(n, ic->_average.getNBins());
752 
753  // sub vector for r
754  Eigen::VectorBlock<Eigen::VectorXd> sub_r =
755  r.segment(n, ic->_average.getNBins());
756 
757  // read in target and calculate dS
758  sub_dS = ic->_average.data().y();
759  // copy r
760  sub_r = ic->_average.data().x();
761  // save size
762  sizes.push_back(ic->_average.getNBins());
763  // save name
764  names.push_back(ic->_p->get("name").as<string>());
765 
766  // shift subrange by size of current
767  n += ic->_average.getNBins();
768  }
769 
770  // write the dS
771  ofstream out_dS;
772  string name_dS = grp_name + suffix + ".S";
773  out_dS.open(name_dS);
774  out_dS << setprecision(8);
775  if (!out_dS) {
776  throw runtime_error(string("error, cannot open file ") + name_dS);
777  }
778 
779  for (votca::Index i = 0; i < dS.size(); ++i) {
780  out_dS << r[i] << " " << dS[i] << endl;
781  }
782 
783  out_dS.close();
784  cout << "written " << name_dS << endl;
785 
786  // write the correlations
787  ofstream out_cor;
788  string name_cor = grp_name + suffix + ".cor";
789  out_cor.open(name_cor);
790  out_cor << setprecision(8);
791 
792  if (!out_cor) {
793  throw runtime_error(string("error, cannot open file ") + name_cor);
794  }
795 
796  for (votca::Index i = 0; i < grp->_corr.rows(); ++i) {
797  for (votca::Index j = 0; j < grp->_corr.cols(); ++j) {
798  out_cor << grp->_corr(i, j) << " ";
799  }
800  out_cor << endl;
801  }
802  out_cor.close();
803  cout << "written " << name_cor << endl;
804  }
