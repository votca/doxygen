74 void Imc::BeginEvaluate(Topology *top, Topology *top_atom)
75 {
76  // we didn't process any frames so far
77  _nframes = 0;
78  _nblock = 0;
79  _processed_some_frames=false;
80 
81  // initialize non-bonded structures
82  for (list<Property*>::iterator iter = _nonbonded.begin();
83  iter != _nonbonded.end(); ++iter) {
84  string name = (*iter)->get("name").value();
85 
86  interaction_t &i = *_interactions[name];
87 
88  // generate the bead lists
89  BeadList beads1, beads2;
90 
91  beads1.Generate(*top, (*iter)->get("type1").value());
92  beads2.Generate(*top, (*iter)->get("type2").value());
93 
94  if(beads1.size() == 0)
95  throw std::runtime_error("Topology does not have beads of type \"" + (*iter)->get("type1").value() + "\"\n"
96  "This was specified in type1 of interaction \"" + name+ "\"");
97  if(beads2.size() == 0)
98  throw std::runtime_error("Topology does not have beads of type \"" + (*iter)->get("type2").value() + "\"\n"
99  "This was specified in type2 of interaction \"" + name + "\"");
100  // calculate normalization factor for rdf
101 
102  if ((*iter)->get("type1").value() == (*iter)->get("type2").value())
103  i._norm = 1. / (beads1.size()*(beads2.size()) / 2.);
104  else
105  i._norm = 1. / (beads1.size() * beads2.size());
106  }
107 
108  for (list<Property*>::iterator iter = _bonded.begin();
109  iter != _bonded.end(); ++iter) {
110  string name = (*iter)->get("name").value();
111 
112  std::list<Interaction *> list = top->InteractionsInGroup(name);
113  if (list.empty() )
114  throw std::runtime_error("Bonded interaction '" + name + "' defined in options xml-file, but not in topology - check name definition in the mapping file again");
115  }
116 }

119 Imc::interaction_t *Imc::AddInteraction(Property *p)
120 {
121  string name = p->get("name").value();
122  string group;
123  if(_do_imc)
124  group = p->get("inverse.imc.group").value();
125  else
126  group = "none";
127 
128  interaction_t *i = new interaction_t;
129  i->_index = _interactions.size();
130  _interactions[name] = i;
131  getGroup(group)->_interactions.push_back(i);
132 
133  i->_step = p->get("step").as<double>();
134  i->_min = p->get("min").as<double>();
135  i->_max = p->get("max").as<double>();
136  i->_norm = 1.0;
137  i->_p=p;
138 
139  // initialize the current and average histogram
140  int n = (int)((i->_max - i->_min) / i->_step + 1.000000001);
141 
142  i->_average.Initialize(i->_min, i->_max+i->_step, n);
143 
144  return i;
145 }

302 // initialize the groups after interactions are added
303 void Imc::InitializeGroups()
304 {
305  if(!_do_imc) return;
306  map<string, group_t *>::iterator group_iter;
307 
308  // clear all the pairs
309 
310  // iterator over all groups
311  for (group_iter = _groups.begin(); group_iter != _groups.end(); ++group_iter) {
312  group_t *grp = (*group_iter).second;
313  grp->_pairs.clear();
314 
315  int n = 0;
316  // count number of bins needed in matrix
317  for (list<interaction_t*>::iterator i1 = grp->_interactions.begin();
318  i1 != grp->_interactions.end(); ++i1)
319  n+=(*i1)->_average.getNBins();
320 
321  // handy access to matrix
322  group_matrix &M = grp->_corr;
323 
324  // initialize matrix with zeroes
325  M.resize(n,n);
326  M = ub::zero_matrix<double>(n, n);
327 
328  // now create references to the sub matrices
329  int i, j;
330  i=0;j=0;
331  // iterate over all possible compinations of pairs
332  for (list<interaction_t*>::iterator i1 = grp->_interactions.begin();
333  i1 != grp->_interactions.end(); ++i1) {
334  int n1 = (*i1)->_average.getNBins();
335  j = i;
336  for (list<interaction_t*>::iterator i2 = i1;
337  i2 != grp->_interactions.end(); ++i2) {
338  int n2 = (*i2)->_average.getNBins();
339 
340  // create matrix proxy with sub-matrix
341  pair_matrix corr(M, ub::range(i, i+n1), ub::range(j, j+n2));
342  // add the pair
343  grp->_pairs.push_back(pair_t(*i1, *i2, i, j, corr));
344  j+=n2;
345  }
346  i+=n1;
347  }
348  }

374 // write the distribution function
375 void Imc::WriteDist(const string &suffix)
376 {
377  map<string, interaction_t *>::iterator iter;
378 
379  // for all interactions
380  for (iter = _interactions.begin(); iter != _interactions.end(); ++iter) {
381  // calculate the rdf
382  Table &t = iter->second->_average.data();
383  Table dist(t);
384  if(!iter->second->_is_bonded) {
385  // normalization is calculated using exact shell volume (difference of spheres)
386  for(int i=0; i<dist.y().size(); ++i) {
387  double x1 = dist.x()[i] - 0.5*iter->second->_step;
388  double x2 = x1 + iter->second->_step;
389  if(x1<0) {
390  dist.y()[i]=0;
391  }
392  else {
393  dist.y()[i] = _avg_vol.getAvg()*iter->second->_norm *
394  dist.y()[i]/(4./3.*M_PI*(x2*x2*x2 - x1*x1*x1));
395  }
396  }
397  }
398  else {
399  dist.y() = iter->second->_norm * dist.y();
400  }
401 
402  dist.Save((iter->first) + suffix);
403  cout << "written " << (iter->first) + suffix << "\n";
404  }
405 }

414 void Imc::WriteIMCData(const string &suffix) {
415  if(!_do_imc) return;
416  //map<string, interaction_t *>::iterator ic_iter;
417  map<string, group_t *>::iterator group_iter;
418 
419  // iterate over all groups
420  for(group_iter = _groups.begin(); group_iter!=_groups.end(); ++group_iter) {
421  group_t *grp = (*group_iter).second;
422  string grp_name = (*group_iter).first;
423  list<interaction_t *>::iterator iter;
424 
425  // number of total bins for all interactions in group is matrix dimension
426  int n=grp->_corr.size1();
427 
428  // build full set of equations + copy some data to make
429  // code better to read
430  group_matrix gmc(grp->_corr);
431  ub::vector<double> dS(n);
432  ub::vector<double> r(n);
433  // the next two variables are to later extract the individual parts
434  // from the whole data after solving equations
435  vector<RangeParser> ranges; // sizes of the individual interactions
436  vector<string> names; // names of the interactions
437 
438  // copy all averages+r of group to one vector
439  n=0;
440  int begin=1;
441  for(iter=grp->_interactions.begin(); iter != grp->_interactions.end(); ++iter) {
442  interaction_t *ic = *iter;
443 
444  // sub vector for dS
445  ub::vector_range< ub::vector<double> > sub_dS(dS,
446  ub::range(n, n + ic->_average.getNBins()));
447 
448  // sub vector for r
449  ub::vector_range< ub::vector<double> > sub_r(r,
450  ub::range(n, n + ic->_average.getNBins()));
451 
452  // read in target and calculate dS
453  CalcDeltaS(ic, sub_dS);
454 
455  // copy r
456  sub_r = ic->_average.data().x();
457 
458  // save size
459 
460  RangeParser rp;
461  int end = begin + ic->_average.getNBins() -1;
462  rp.Add(begin, end);
463  ranges.push_back(rp);
464  begin = end+1;
465  // save name
466  names.push_back(ic->_p->get("name").as<string>());
467 
468  // shift subrange by size of current
469  n+=ic->_average.getNBins();
470  }
471 
472  // now we need to calculate the
473  // A_ij = <S_i*S_j> - <S_i>*<S_j>
474  vector<pair_t>::iterator pair;
475  for (pair = grp->_pairs.begin(); pair != grp->_pairs.end(); ++pair) {
476  interaction_t *i1 = pair->_i1;
477  interaction_t *i2 = pair->_i2;
478 
479  // make reference to <S_i>
480  ub::vector<double> &a = i1->_average.data().y();
481  // make reference to <S_j>
482  ub::vector<double> &b = i2->_average.data().y();
483 
484  int i=pair->_offset_i;
485  int j=pair->_offset_j;
486  int n1=i1->_average.getNBins();
487  int n2=i2->_average.getNBins();
488 
489  // sub matrix for these two interactions
490  // we only need to take care about one sub-matrix and not the mirrored
491  // one since ublas makes sure the matrix is symmetric
492  pair_matrix M(gmc, ub::range(i, i+n1),
493  ub::range(j, j+n2));
494  // A_ij = -(<a_i*a_j> - <a_i>*<b_j>)
495  //for(i=0; i<M.size1(); ++i)
496  // for(j=i; j<M.size2(); ++j)
497  // M(i,j) = -(M(i,j) - a(i)*b(j));
498  M = -(M - ub::outer_prod(a, b));
499  }
500 
501  imcio_write_dS(grp_name + suffix + ".imc", r, dS);
502  imcio_write_matrix(grp_name + suffix + ".gmc", gmc);
503  imcio_write_index(grp_name + suffix + ".idx", names, ranges);
504 
505  }

508 // calculate deviation from target vectors
509 void Imc::CalcDeltaS(interaction_t *interaction, ub::vector_range< ub::vector<double> > &dS)
510 {
511  const string &name = interaction->_p->get("name").as<string>();
512 
513  Table target;
514  target.Load(name + ".dist.tgt");
515 
516  if(!interaction->_is_bonded) {
517  for(int i=0; i<target.y().size(); ++i) {
518  double x1 = target.x()[i] - 0.5*interaction->_step;
519  double x2 = x1 + interaction->_step;
520  if(x1<0)
521  x1=x2=0;
522  target.y()[i] = 1./(_avg_vol.getAvg()*interaction->_norm) *
523  target.y()[i] * (4./3.*M_PI*(x2*x2*x2 - x1*x1*x1));
524  }
525  }
526  else {
527  target.y() = (1.0 / interaction->_norm)*target.y();
528  }
529  if(target.y().size() != interaction->_average.data().y().size())
530  throw std::runtime_error("number of grid points in target does not match the grid");
531 
532  dS = interaction->_average.data().y() - target.y();
533 }
