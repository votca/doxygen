41 // here the data structures are prepared to handle all the data
42 void Imc::Initialize()
43 {
44  // do some output
45  if(_do_imc)
46  cout << "begin to calculate inverse monte carlo parameters\n";
47  else
48  cout << "begin to calculate distribution functions\n";
49  cout << "# of bonded interactions: " << _bonded.size() << endl;
50  cout << "# of non-bonded interactions: " << _nonbonded.size() << endl;
51 
52  if ( _bonded.size()+_nonbonded.size() == 0 )
53  throw std::runtime_error("No interactions defined in options xml-file - nothing to be done");
54 
55

57  for (list<Property*>::iterator iter = _nonbonded.begin();
58  iter != _nonbonded.end(); ++iter) {
59  interaction_t *i = AddInteraction(*iter);

68  }
69 
70  // initialize the group structures
71  if(_do_imc)
72  InitializeGroups();

75 void Imc::BeginEvaluate(Topology *top, Topology *top_atom)
76 {
77  // we didn't process any frames so far
78  _nframes = 0;
79  _nblock = 0;
80  _processed_some_frames=false;
81 
82  // initialize non-bonded structures
83  for (list<Property*>::iterator iter = _nonbonded.begin();
84  iter != _nonbonded.end(); ++iter) {
85  string name = (*iter)->get("name").value();
86 
87  interaction_t &i = *_interactions[name];
88 
89  // generate the bead lists
90  BeadList beads1, beads2;
91 
92  beads1.Generate(*top, (*iter)->get("type1").value());
93  beads2.Generate(*top, (*iter)->get("type2").value());
94 
95  if(beads1.size() == 0)
96  throw std::runtime_error("Topology does not have beads of type \"" + (*iter)->get("type1").value() + "\"\n"
97  "This was specified in type1 of interaction \"" + name+ "\"");
98  if(beads2.size() == 0)
99  throw std::runtime_error("Topology does not have beads of type \"" + (*iter)->get("type2").value() + "\"\n"
100  "This was specified in type2 of interaction \"" + name + "\"");
101  // calculate normalization factor for rdf
102 
103  if ((*iter)->get("type1").value() == (*iter)->get("type2").value())
104  i._norm = 1. / (beads1.size()*(beads2.size()) / 2.);
105  else
106  i._norm = 1. / (beads1.size() * beads2.size());
107  }
108 
109  for (list<Property*>::iterator iter = _bonded.begin();
110  iter != _bonded.end(); ++iter) {
111  string name = (*iter)->get("name").value();
112 
113  std::list<Interaction *> list = top->InteractionsInGroup(name);
114  if (list.empty() )
115  throw std::runtime_error("Bonded interaction '" + name + "' defined in options xml-file, but not in topology - check name definition in the mapping file again");
116  }
117 }

119 // create an entry for interactions
120 Imc::interaction_t *Imc::AddInteraction(Property *p)
121 {
122  string name = p->get("name").value();
123  string group;
124  if(_do_imc)
125  group = p->get("inverse.imc.group").value();
126  else
127  group = "none";
128 
129  interaction_t *i = new interaction_t;
130  i->_index = _interactions.size();
131  _interactions[name] = i;
132  getGroup(group)->_interactions.push_back(i);
133 
134  i->_step = p->get("step").as<double>();
135  i->_min = p->get("min").as<double>();
136  i->_max = p->get("max").as<double>();
137  i->_norm = 1.0;
138  i->_p=p;
139 
140  // initialize the current and average histogram
141  int n = static_cast<int>((i->_max - i->_min) / i->_step + 1.000000001);
142 
143  i->_average.Initialize(i->_min, i->_max, n);
144 
145  return i;
146 }

304 // initialize the groups after interactions are added
305 void Imc::InitializeGroups()
306 {
307  if(!_do_imc) return;
308  map<string, group_t *>::iterator group_iter;
309 
310  // clear all the pairs
311 
312  // iterator over all groups
313  for (group_iter = _groups.begin(); group_iter != _groups.end(); ++group_iter) {
314  group_t *grp = (*group_iter).second;
315  grp->_pairs.clear();
316 
317  int n = 0;
318  // count number of bins needed in matrix
319  for (list<interaction_t*>::iterator i1 = grp->_interactions.begin();
320  i1 != grp->_interactions.end(); ++i1)
321  n+=(*i1)->_average.getNBins();
322 
323  // handy access to matrix
324  group_matrix &M = grp->_corr;
325 
326  // initialize matrix with zeroes
327  M=Eigen::MatrixXd::Zero(n,n);
328 
329  // now create references to the sub matrices
330  int i, j;
331  i=0;j=0;
332  // iterate over all possible compinations of pairs
333  for (list<interaction_t*>::iterator i1 = grp->_interactions.begin();
334  i1 != grp->_interactions.end(); ++i1) {
335  int n1 = (*i1)->_average.getNBins();
336  j = i;
337  for (list<interaction_t*>::iterator i2 = i1;
338  i2 != grp->_interactions.end(); ++i2) {
339  int n2 = (*i2)->_average.getNBins();
340 
341  // create matrix proxy with sub-matrix
342  pair_matrix corr=M.block(i,j,n1,n2);
343  // add the pair
344  grp->_pairs.push_back(pair_t(*i1, *i2, i, j, corr));
345  j+=n2;
346  }
347  i+=n1;
348  }
349  }
350 }

352 // update the correlation matrix
353 void Imc::DoCorrelations(Imc::Worker *worker) {
354  if(!_do_imc) return;
355  vector<pair_t>::iterator pair;
356  map<string, group_t *>::iterator group_iter;
357 
358  for (group_iter = _groups.begin(); group_iter != _groups.end(); ++group_iter) {
359  group_t *grp = (*group_iter).second;
360  // update correlation for all pairs
361  for (pair = grp->_pairs.begin(); pair != grp->_pairs.end(); ++pair) {
362  Eigen::VectorXd &a = worker->_current_hists[pair->_i1->_index].data().y();
363  Eigen::VectorXd &b = worker->_current_hists[pair->_i2->_index].data().y();
364  pair_matrix &M = pair->_corr;
365 
366  M = ((((double)_nframes-1.0)*M) + a*b.transpose())/(double)_nframes;
367  }
368  }

371 // write the distribution function
372 void Imc::WriteDist(const string &suffix)
373 {
374  map<string, interaction_t *>::iterator iter;
375 
376  cout << std::endl; // Cosmetic, put \n before printing names of distribution files.
377  // for all interactions
378  for (iter = _interactions.begin(); iter != _interactions.end(); ++iter) {
379  // calculate the rdf
380  Table &t = iter->second->_average.data();
381  Table dist(t);
382  if(!iter->second->_is_bonded) {
383  // normalization is calculated using exact shell volume (difference of spheres)
384  for(unsigned int i=0; i<dist.y().size(); ++i) {
385  double x1 = dist.x()[i] - 0.5*iter->second->_step;
386  double x2 = x1 + iter->second->_step;
387  if(x1<0) {
388  dist.y()[i]=0;
389  }
390  else {
391  dist.y()[i] = _avg_vol.getAvg()*iter->second->_norm *
392  dist.y()[i]/(4./3.*M_PI*(x2*x2*x2 - x1*x1*x1));
393  }
394  }
395  }
396  else {
397  // \TODO normalize bond and angle differently....
398  double norm=dist.y().cwiseAbs().sum();
399  if ( norm > 0 ) {
400  dist.y() = iter->second->_norm * dist.y() / ( norm * iter->second->_step );
401  }
402  }
403 
404  dist.Save((iter->first) + suffix);
405  cout << "written " << (iter->first) + suffix << "\n";
406  }
407 }

416 void Imc::WriteIMCData(const string &suffix) {
417  if(!_do_imc) return;
418  //map<string, interaction_t *>::iterator ic_iter;
419  map<string, group_t *>::iterator group_iter;
420 
421  // iterate over all groups
422  for(group_iter = _groups.begin(); group_iter!=_groups.end(); ++group_iter) {
423  group_t *grp = (*group_iter).second;
424  string grp_name = (*group_iter).first;
425  list<interaction_t *>::iterator iter;
426 
427  // number of total bins for all interactions in group is matrix dimension
428  int n=grp->_corr.rows();
429 
430  // build full set of equations + copy some data to make
431  // code better to read
432  group_matrix gmc(grp->_corr);
433  Eigen::VectorXd dS(n);
434  Eigen::VectorXd r(n);
435  // the next two variables are to later extract the individual parts
436  // from the whole data after solving equations
437  vector<RangeParser> ranges; // sizes of the individual interactions
438  vector<string> names; // names of the interactions
439 
440  // copy all averages+r of group to one vector
441  n=0;
442  int begin=1;
443  for(iter=grp->_interactions.begin(); iter != grp->_interactions.end(); ++iter) {
444  interaction_t *ic = *iter;
445 
446  // sub vector for dS
447  Eigen::VectorBlock< Eigen::VectorXd > sub_dS=dS.segment(n,ic->_average.getNBins());
448 
449  // sub vector for r
450  Eigen::VectorBlock< Eigen::VectorXd > sub_r=r.segment(n,ic->_average.getNBins());
451 
452  // read in target and calculate dS
453  CalcDeltaS(ic, sub_dS);
454 
455  // copy r
456  sub_r = ic->_average.data().x();
457 
458  // save size
459  RangeParser rp;
460  int end = begin + ic->_average.getNBins() -1;
461  rp.Add(begin, end);
462  ranges.push_back(rp);
463  begin = end+1;
464  // save name
465  names.push_back(ic->_p->get("name").as<string>());
466 
467  // shift subrange by size of current
468  n+=ic->_average.getNBins();
469  }
470 
471  // now we need to calculate the
472  // A_ij = <S_i*S_j> - <S_i>*<S_j>
473  vector<pair_t>::iterator pair;
474  for (pair = grp->_pairs.begin(); pair != grp->_pairs.end(); ++pair) {
475  interaction_t *i1 = pair->_i1;
476  interaction_t *i2 = pair->_i2;
477 
478  // make reference to <S_i>
479  Eigen::VectorXd &a = i1->_average.data().y();
480  // make reference to <S_j>
481  Eigen::VectorXd &b = i2->_average.data().y();
482 
483  int i=pair->_offset_i;
484  int j=pair->_offset_j;
485  int n1=i1->_average.getNBins();
486  int n2=i2->_average.getNBins();
487 
488 
489  pair_matrix M=gmc.block(i,j,n1,n2);
490  M = -(M - a*b.transpose());
491  //matrix is symmetric
492  gmc.block(j,i,n2,n1)=M.transpose().eval();
493  }
494 
495  imcio_write_dS(grp_name + suffix + ".imc", r, dS);
496  imcio_write_matrix(grp_name + suffix + ".gmc", gmc);
497  imcio_write_index(grp_name + suffix + ".idx", names, ranges);
498 
499  }
500 }

502 // calculate deviation from target vectors
503 void Imc::CalcDeltaS(interaction_t *interaction, Eigen::VectorBlock< Eigen::VectorXd > &dS)
504 {
505  const string &name = interaction->_p->get("name").as<string>();
506 
507  Table target;
508  target.Load(name + ".dist.tgt");
509 
510  if(!interaction->_is_bonded) {
511  for(unsigned int i=0; i<target.y().size(); ++i) {
512  double x1 = target.x()[i] - 0.5*interaction->_step;
513  double x2 = x1 + interaction->_step;
514  if(x1<0)
515  x1=x2=0;
516  target.y()[i] = 1./(_avg_vol.getAvg()*interaction->_norm) *
517  target.y()[i] * (4./3.*M_PI*(x2*x2*x2 - x1*x1*x1));
518  }
519  }
520  else {
521  target.y() = (1.0 / interaction->_norm)*target.y();
522  }
523  if(target.y().size() != interaction->_average.data().y().size())
524  throw std::runtime_error("number of grid points in target does not match the grid");
525 
526  dS = interaction->_average.data().y() - target.y();
527 }

635 void Imc::MergeWorker(CsgApplication::Worker* worker_)
636 {
637  _processed_some_frames = true;
638  Imc::Worker *worker = dynamic_cast<Imc::Worker *>(worker_);
639  // update the average
640  map<string, interaction_t *>::iterator ic_iter;
641  //map<string, group_t *>::iterator group_iter;
642 
643  ++_nframes;
644  _avg_vol.Process(worker->_cur_vol);
645  for (ic_iter = _interactions.begin(); ic_iter != _interactions.end(); ++ic_iter) {
646  interaction_t *i=ic_iter->second;
647  i->_average.data().y() = (((double)_nframes-1.0)*i->_average.data().y()
648  + worker->_current_hists[i->_index].data().y())/(double)_nframes;
649  }
650 
651  // update correlation matrices
652  if(_do_imc)
653  DoCorrelations(worker);
654 
655  if(_block_length != 0) {
656  if((_nframes % _block_length)==0) {
