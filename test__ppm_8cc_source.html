35BOOST_AUTO_TEST_CASE(ppm_full) {
36
37 libint2::initialize();
38 Orbitals orbitals;
39 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
40 "/ppm/molecule.xyz");
41
42 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) + "/ppm/3-21G.xml");
43 orbitals.setNumberOfOccupiedLevels(4);
44
45 AOBasis aobasis = orbitals.getDftBasis();
46
47 AOKinetic kinetic;
48 kinetic.Fill(aobasis);
49
50 AOMultipole esp;
51 esp.FillPotential(aobasis, orbitals.QMAtoms());
52
53 Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(kinetic.Matrix() +
54 esp.Matrix());
55
56 Logger log;
57 TCMatrix_gwbse Mmn;
58 Mmn.Initialize(aobasis.AOBasisSize(), 0, 16, 0, 16);
59 Mmn.Fill(aobasis, aobasis, es.eigenvectors());
60
61 RPA rpa = RPA(log, Mmn);
62 rpa.configure(4, 0, 16);
63 rpa.setRPAInputEnergies(es.eigenvalues());
64
65 PPM ppm;
66 ppm.PPM_construct_parameters(rpa);
67
68 Eigen::VectorXd ppm_freq = Eigen::VectorXd::Zero(17);
69 ppm_freq << 19.4503, 12.2429, 10.2167, 10.2167, 10.2167, 17.0832, 12.7117,
70 12.7117, 12.7117, 10.9455, 10.9455, 10.9455, 11.1861, 9.60843, 9.60843,
71 9.60843, 9.61518;
72 Eigen::VectorXd ppm_w = Eigen::VectorXd::Zero(17);
73 ppm_w << 3.59422e-05, 0.00121795, 0.00343632, 0.00343632, 0.00343632,
74 0.00394659, 0.012066, 0.012066, 0.012066, 0.0241118, 0.0241118, 0.0241118,
75 0.0286786, 0.11036, 0.11036, 0.11036, 0.191732;
76
77 bool f_check = ppm_freq.isApprox(ppm.getPpm_freq(), 0.0001);
78
79 bool w_check = ppm_w.isApprox(ppm.getPpm_weight(), 0.0001);
80
81 if (!f_check) {
82 cout << "ppm_freq" << endl;
83 cout << ppm.getPpm_freq() << endl;
84 cout << "ppm_freq_ref" << endl;
85 cout << ppm_freq << endl;
86 }
87
88 if (!w_check) {
89 cout << "ppm_w" << endl;
90 cout << ppm.getPpm_weight() << endl;
91 cout << "ppm_w_ref" << endl;
92 cout << ppm_w << endl;
93 }
94
95 BOOST_CHECK_EQUAL(f_check, 1);
96 BOOST_CHECK_EQUAL(w_check, 1);
97
98 libint2::finalize();
99}
