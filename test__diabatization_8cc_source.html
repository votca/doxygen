37BOOST_AUTO_TEST_CASE(ER_coupling_test) {
38 libint2::initialize();
39
40 // populate orbitals object
41 Orbitals dimer;
42 dimer.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
43 "/diabatization/dimer.xyz");
44 dimer.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) +
45 "/diabatization/def2-svp.xml");
46 dimer.SetupAuxBasis(std::string(XTP_TEST_DATA_FOLDER) +
47 "/diabatization/aux-def2-svp.xml");
48 dimer.setNumberOfAlphaElectrons(44);
49 dimer.setNumberOfOccupiedLevels(44);
50
51 Eigen::VectorXd ref_MOvals = votca::tools::EigenIO_MatrixMarket::ReadVector(
52 std::string(XTP_TEST_DATA_FOLDER) + "/diabatization/dimer_MOvals.mm");
53 Eigen::MatrixXd ref_MOs = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
54 std::string(XTP_TEST_DATA_FOLDER) + "/diabatization/dimer_MOs.mm");
55 dimer.MOs().eigenvalues() = ref_MOvals;
56 dimer.MOs().eigenvectors() = ref_MOs;
57
58 dimer.setGWindices(0, 130);
59 dimer.setBSEindices(0, 130);
60 dimer.setTDAApprox(false);
61
62 Eigen::VectorXd ref_singletvals =
63 votca::tools::EigenIO_MatrixMarket::ReadVector(
64 std::string(XTP_TEST_DATA_FOLDER) +
65 "/diabatization/dimer_singletE.mm");
66
67 Eigen::MatrixXd ref_spsi = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
68 std::string(XTP_TEST_DATA_FOLDER) +
69 "/diabatization/dimer_singlet_spsi.mm");
70
71 Eigen::MatrixXd ref_spsi2 = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
72 std::string(XTP_TEST_DATA_FOLDER) +
73 "/diabatization/dimer_singlet_spsi2.mm");
74
75 dimer.BSESinglets().eigenvalues() = ref_singletvals;
76 dimer.BSESinglets().eigenvectors() = ref_spsi;
77 dimer.BSESinglets().eigenvectors2() = ref_spsi2;
78
79 // set logger
80 Logger log;
81 log.setReportLevel(Log::error);
82 log.setMultithreading(true);
83 log.setCommonPreface("\n... ...");
84 ERDiabatization ERDiabatization(dimer, dimer, &log, 1, 2, "singlet", true);
85 ERDiabatization.configure();
86 ERDiabatization.setUpMatrices();
87
88 // Calculate angle
89 double angle = ERDiabatization.Calculate_angle();
90 double angle_ref = 0.7063121313715891;
91 BOOST_CHECK_CLOSE(angle_ref, angle, 1e-4);
92
93 // diabatic Hamiltonian
94 std::pair<Eigen::VectorXd, Eigen::MatrixXd> rotated_H =
95 ERDiabatization.Calculate_diabatic_H(angle);
96
97 double E1_ref = 0.21995615824390968;
98 double E2_ref = 0.22090555616542826;
99 BOOST_CHECK_CLOSE(E1_ref, rotated_H.first(0), 1e-4);
100 BOOST_CHECK_CLOSE(E2_ref, rotated_H.first(1), 1e-4);
101
102 Eigen::MatrixXd& diabatic_H = rotated_H.second;
103
104 Eigen::MatrixXd diabatic_H_ref =
105 votca::tools::EigenIO_MatrixMarket::ReadMatrix(
106 std::string(XTP_TEST_DATA_FOLDER) + "/diabatization/Hdiab_ref.mm");
107
108 BOOST_CHECK_CLOSE(diabatic_H_ref(0, 0), diabatic_H(0, 0), 1e-6);
109 BOOST_CHECK_CLOSE(diabatic_H_ref(0, 1), diabatic_H(0, 1), 1e-6);
110 BOOST_CHECK_CLOSE(diabatic_H_ref(1, 0), diabatic_H(1, 0), 1e-6);
111 BOOST_CHECK_CLOSE(diabatic_H_ref(1, 1), diabatic_H(1, 1), 1e-6);
112
113 libint2::finalize();
114}
