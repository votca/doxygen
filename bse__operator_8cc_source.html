44 template <int cqp, int cx, int cd, int cd2>
45 Eigen::RowVectorXd BSE_OPERATOR<cqp, cx, cd, cd2>::Hx_row(int index) const {
46  int thread_id = OPENMP::getThreadId();
47  if (_Hx_cache[thread_id].hasValue(index)) {
48  return _Hx_cache[thread_id].getValue(index);
49  }
50  int auxsize = _Mmn.auxsize();
51  vc2index vc = vc2index(0, 0, _bse_ctotal);
52 
53  const int vmin = _opt.vmin - _opt.rpamin;
54  const int cmin = _bse_cmin - _opt.rpamin;
55  int v1 = vc.v(index);
56  int c1 = vc.c(index);
57  int cache_size = 50;
58  if ((_bse_ctotal - c1) < cache_size) {
59  cache_size = _bse_ctotal - c1;
60  }
61  Eigen::MatrixXd H_cache = Eigen::MatrixXd::Zero(_bse_size, cache_size);
62  const Eigen::MatrixXd Mmn1T =
63  _Mmn[v1 + vmin].block(c1 + cmin, 0, cache_size, auxsize).transpose();
64  for (int v2 = 0; v2 < _bse_vtotal; v2++) {
65  const Eigen::MatrixXd& Mmn2 = _Mmn[v2 + vmin];
66  int i2 = vc.I(v2, 0);
67  H_cache.block(i2, 0, _bse_ctotal, cache_size) =
68  Mmn2.block(cmin, 0, _bse_ctotal, auxsize) * Mmn1T;
69  }
70  _Hx_cache[thread_id].FillCache(H_cache, index);
71  return H_cache.col(0).transpose();
72 }

98 Eigen::RowVectorXd BSE_OPERATOR<cqp, cx, cd, cd2>::Hqp_row(int index) const {
99  vc2index vc = vc2index(0, 0, _bse_ctotal);
100  int v1 = vc.v(index);
101  int c1 = vc.c(index);
102  Eigen::MatrixXd Result = Eigen::MatrixXd::Zero(_bse_ctotal, _bse_vtotal);
103  int cmin = _bse_cmin - _opt.qpmin;
104  // v->c
105  Result.col(v1) = _Hqp.col(c1 + cmin).segment(cmin, _bse_ctotal);
106  // c-> v
107  int vmin = _opt.vmin - _opt.qpmin;
108  Result.row(c1) -= _Hqp.col(v1 + vmin).segment(vmin, _bse_vtotal);
109  return Eigen::Map<Eigen::RowVectorXd>(Result.data(), Result.size());
110  ;
111 }
