27 void BSE_OPERATOR<cqp, cx, cd, cd2>::configure(BSEOperator_Options opt) {
28  _opt = opt;

30  _bse_cmin = _opt.homo + 1;

79 template <int cqp, int cx, int cd, int cd2>
80 Eigen::RowVectorXd BSE_OPERATOR<cqp, cx, cd, cd2>::Hx_row(int index) const {
81  int thread_id = OPENMP::getThreadId();
82  if (_Hx_cache[thread_id].hasValue(index)) {
83  return _Hx_cache[thread_id].getValue(index);
84  }
85  int auxsize = _Mmn.auxsize();
86  vc2index vc = vc2index(0, 0, _bse_ctotal);
87 
88  const int vmin = _opt.vmin - _opt.rpamin;
89  const int cmin = _bse_cmin - _opt.rpamin;
90  int v1 = vc.v(index);
91  int c1 = vc.c(index);
92  int cache_size = 50;
93  if ((_bse_ctotal - c1) < cache_size) {
94  cache_size = _bse_ctotal - c1;
95  }
96  Eigen::MatrixXd H_cache = Eigen::MatrixXd::Zero(_bse_size, cache_size);
97  const Eigen::MatrixXd Mmn1T =
98  _Mmn[v1 + vmin].block(c1 + cmin, 0, cache_size, auxsize).transpose();
99  for (int v2 = 0; v2 < _bse_vtotal; v2++) {
100  const Eigen::MatrixXd& Mmn2 = _Mmn[v2 + vmin];
101  int i2 = vc.I(v2, 0);
102  H_cache.block(i2, 0, _bse_ctotal, cache_size) =
103  Mmn2.block(cmin, 0, _bse_ctotal, auxsize) * Mmn1T;
104  }
105  _Hx_cache[thread_id].FillCache(H_cache, index);
106  return H_cache.col(0).transpose();
107 }

109 template <int cqp, int cx, int cd, int cd2>
110 Eigen::RowVectorXd BSE_OPERATOR<cqp, cx, cd, cd2>::Hd_row(int index) const {
111  int auxsize = _Mmn.auxsize();
112  vc2index vc = vc2index(0, 0, _bse_ctotal);
113 
114  const int vmin = _opt.vmin - _opt.rpamin;
115  const int cmin = _bse_cmin - _opt.rpamin;
116  int v1 = vc.v(index);
117  int c1 = vc.c(index);
118 
119  const Eigen::MatrixXd Mmn1T =
120  -(_Mmn[v1 + vmin].block(vmin, 0, _bse_vtotal, auxsize) *
121  _epsilon_0_inv.asDiagonal())
122  .transpose();
123  const Eigen::MatrixXd& Mmn2 = _Mmn[c1 + cmin];
124  Eigen::MatrixXd Mmn2xMmn1T =
125  Mmn2.block(cmin, 0, _bse_ctotal, auxsize) * Mmn1T;

129 template <int cqp, int cx, int cd, int cd2>
130 Eigen::RowVectorXd BSE_OPERATOR<cqp, cx, cd, cd2>::Hqp_row(int index) const {
131  vc2index vc = vc2index(0, 0, _bse_ctotal);
132  int v1 = vc.v(index);
133  int c1 = vc.c(index);
134  Eigen::MatrixXd Result = Eigen::MatrixXd::Zero(_bse_ctotal, _bse_vtotal);
135  int cmin = _bse_cmin - _opt.qpmin;
136  // v->c
137  Result.col(v1) = _Hqp.col(c1 + cmin).segment(cmin, _bse_ctotal);
138  // c-> v
139  int vmin = _opt.vmin - _opt.qpmin;
140  Result.row(c1) -= _Hqp.col(v1 + vmin).segment(vmin, _bse_vtotal);
141  return Eigen::Map<Eigen::RowVectorXd>(Result.data(), Result.size());
142 }
