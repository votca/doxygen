37  // partial specialization for double precision (dgemm))
38  template<class F, class A>
39  inline matrix<double,F,A> // prod( m1, m2 )
40  prod(const matrix<double,F,A> &m1, const matrix<double,F,A> &m2)
41  {
42  #ifdef GSLDEBUG
43  std::cout << "\x1b[0;34mGSL [CMD,CM]\x1b[0;39m\n" << std::flush;
44  #endif
45 
46  gsl_matrix_const_view mA = gsl_matrix_const_view_array (&m1(0,0), m1.size1(), m1.size2());
47  gsl_matrix_const_view mB = gsl_matrix_const_view_array (&m2(0,0), m2.size1(), m2.size2());
48 
49  boost::numeric::ublas::matrix<double,F,A> AxB( m1.size1(), m2.size2() );
50  gsl_matrix_view mC = gsl_matrix_view_array (&AxB(0,0), AxB.size1(), AxB.size2());
51 
52  gsl_blas_dgemm (CblasNoTrans, CblasNoTrans,
53  1.0, &mA.matrix, &mB.matrix,
54  0.0, &mC.matrix);
55  return AxB;
56  }
