36BOOST_AUTO_TEST_CASE(eigen_test) {
37 Property prop;
38 prop.add("vec", "1 2 3 4 5 6 7 8");
39 Eigen::VectorXd result = prop.get("vec").as<Eigen::VectorXd>();
40
41 Eigen::VectorXd ref;
42 ref.resize(8);
43 ref << 1, 2, 3, 4, 5, 6, 7, 8;
44 BOOST_CHECK_EQUAL(ref.isApprox(result, 0.0001), true);
45
46 BOOST_CHECK_THROW(prop.get("vec").as<Eigen::Vector3d>(), std::runtime_error);
47
48 Property prop2;
49 prop2.add("vec", "1 2 3");
50 Eigen::Vector3d result2 = prop2.get("vec").as<Eigen::Vector3d>();
51 Eigen::Vector3d ref2{1, 2, 3};
52 BOOST_CHECK_EQUAL(ref2.isApprox(result2, 0.0001), true);
53}

178BOOST_AUTO_TEST_CASE(deleteproperty) {
179
180 Property two("two", "", "");
181 two.add("goodbye", "4");
182 two.add("goodbye", "3");
183 Property& bye = two.add("bye", "1");
184 bye.add("hello", "5");
185
186 two.deleteChildren([](const Property& p) { return p.name() == "bye"; });
187 BOOST_CHECK(!two.exists("bye"));
188 BOOST_CHECK(two.exists("goodbye"));
189 two.deleteChildren([](const Property& p) {
190 return p.name() == "goodbye" && p.as<votca::Index>() == 4;
191 });
192 // there is still another goodbye tag
193 BOOST_CHECK(two.exists("goodbye"));
194 BOOST_CHECK(two.get("goodbye").as<votca::Index>() == 3);
195}
