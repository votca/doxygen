49 bool ExclusionList::IsExcluded(Bead *bead1, Bead *bead2) {
50  exclusion_t *excl;
51  if(bead1->getMolecule() != bead2->getMolecule()) return false;
52  if (bead2->getId() < bead1->getId()) swap(bead1, bead2);
53  if ((excl = GetExclusions(bead1))) {
54  if(find(excl->_exclude.begin(), excl->_exclude.end(), bead2)
55  != excl->_exclude.end()) return true;
56  }
57  return false;
58 }
