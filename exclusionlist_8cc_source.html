00049 bool ExclusionList::IsExcluded(Bead *bead1, Bead *bead2) {
00050     exclusion_t *excl;
00051     if(bead1->getMolecule() != bead2->getMolecule()) return false;
00052     if (bead2->getId() < bead1->getId()) swap(bead1, bead2);
00053     if ((excl = GetExclusions(bead1))) {
00054         if(find(excl->_exclude.begin(), excl->_exclude.end(), bead2)
