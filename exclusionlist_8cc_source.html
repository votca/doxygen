50 bool ExclusionList::IsExcluded(Bead *bead1, Bead *bead2) const {
51  if (bead1->getMoleculeId() != bead2->getMoleculeId()) {
52  return false;
53  }
54  if (bead2->getId() < bead1->getId()) {
55  swap(bead1, bead2);
56  }
57 
58  const exclusion_t *excl = GetExclusions(bead1);
59  if (excl != nullptr) {
60  if (find(excl->_exclude.begin(), excl->_exclude.end(), bead2) !=
61  excl->_exclude.end()) {
62  return true;
63  }
64  }
65  return false;
66 }
