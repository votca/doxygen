43 class QMMInterface
44 {
45 public:
46 
47  QMMInterface() { _polar_table = POLAR_TABLE(); };
48  ~QMMInterface() {};
49 
50  // CONVERSION QM -> MM
51  APolarSite *Convert(QMAtom *atm, int id = -1) {
52  double A_to_nm = 0.1;
53  vec pos = A_to_nm*vec(atm->x, atm->y, atm->z);
54  double q = atm->charge;
55  std::string elem = atm->type;
56  double pol = 0.0;
57  try {
58  pol = _polar_table.at(elem);
59  }
60  catch(out_of_range) {
61  std::cout << std::endl << "QMMInterface - no default polarizability given "
62  << "for element type '" << elem << "'. Defaulting to 1A**3" << std::flush;
63  pol = 1e-3;
64  }
65 
66  APolarSite *new_aps = new APolarSite(id, elem);
67  new_aps->setRank(0);
68  new_aps->setPos(pos);
69  new_aps->setQ00(q,0); // <- charge state 0 <> 'neutral'
70  new_aps->setIsoP(pol);
71 
72  return new_aps;
73  }
74 
75  PolarSeg *Convert(std::vector<QMAtom*> &atms) {
76  PolarSeg *new_pseg = new PolarSeg();
77  std::vector<QMAtom*>::iterator it;
78  for (it = atms.begin(); it < atms.end(); ++it) {
79  APolarSite *new_site = this->Convert(*it);
80  new_pseg->push_back(new_site);
81  }
82  return new_pseg;
83  }
84 
85  // TODO CONVERSION MM -> QM
86  QMAtom *Convert(APolarSite*);
87  std::vector<QMAtom*> Convert(PolarSeg*);
88 
89 private:
90 
91  // Allocates polarizabilities in A**3 to element types
92  std::map<std::string,double> _polar_table;
93 
94 };
