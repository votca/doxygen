88 class UnitConverter {
89  private:
91  constexpr double getDistanceValue_(
92  const DistanceUnit& enum_type) const noexcept {
93  switch (enum_type) {
94  case DistanceUnit::meters:
95  return 1E-10;
96  case DistanceUnit::centimeters:
97  return 1E-8;
98  case DistanceUnit::nanometers:
99  return 0.1;
100  case DistanceUnit::angstroms:
101  return 1.0;
102  case DistanceUnit::bohr:
103  return 1.8897161646321;
104  }
105  return 0.0;
106  }
107 
109  constexpr double getTimeValue_(const TimeUnit& enum_type) const noexcept {
110  switch (enum_type) {
111  case TimeUnit::seconds:
112  return 1E-12;
113  case TimeUnit::microseconds:
114  return 1E-6;
115  case TimeUnit::nanoseconds:
116  return 1E-3;
117  case TimeUnit::picoseconds:
118  return 1.0;
119  case TimeUnit::femtoseconds:
120  return 1000;
121  }
122  return 0.0;
123  }
124 
126  constexpr double getMassValue_(const MassUnit& enum_type) const noexcept {
127  switch (enum_type) {
128  case MassUnit::kilograms:
129  return 1.6605402E-27;
130  case MassUnit::grams:
131  return 1.6605402E-24;
132  case MassUnit::picograms:
133  return 1.6605402E-12;
134  case MassUnit::femtograms:
135  return 1.6605402E-9;
136  case MassUnit::attograms:
137  return 1.66054019E-6;
138  case MassUnit::atomic_mass_units:
139  return 1.0;
140  case MassUnit::grams_per_mole:
141  return 1.0;
142  }
143  return 0.0;
144  }
145 
147  constexpr double getEnergyValue_(const EnergyUnit& enum_type) const noexcept {
148  switch (enum_type) {
149  case EnergyUnit::kilocalories:
150  return 3.82929389E-23;
151  case EnergyUnit::kilojoules:
152  return 1.602176634E-22;
153  case EnergyUnit::joules:
154  return 1.602176634E-19;
155  case EnergyUnit::hartrees:
156  return 0.0367493;
157  case EnergyUnit::electron_volts:
158  return 1.0;
159  }
160  return 0.0;

166  switch (enum_type) {

173  case MolarEnergyUnit::hartrees_per_mole:
174  return 0.0367493;
175  case MolarEnergyUnit::electron_volts_per_mole:

177  }
178  return 0.0;
179  }
181  constexpr double getChargeValue_(const ChargeUnit& enum_type) const noexcept {
182  switch (enum_type) {

187  }
188  return 0.0;

192  constexpr double getVelocityValue_(
193  const VelocityUnit& enum_type) const noexcept {
194  switch (enum_type) {
195  case VelocityUnit::nanometers_per_picosecond:
196  return 1.0;
197  case VelocityUnit::angstroms_per_picosecond:
198  return convert(DistanceUnit::nanometers, DistanceUnit::angstroms);
199  case VelocityUnit::angstroms_per_femtosecond:
200  return convert(DistanceUnit::nanometers, DistanceUnit::angstroms) /
201  convert(TimeUnit::picoseconds, TimeUnit::femtoseconds);
202  }
203  return 0.0;
204  }
205 
207  constexpr double getForceValue_(const ForceUnit& enum_type) const noexcept {
208  switch (enum_type) {
209  case ForceUnit::kilocalories_per_angstrom:
210  return convert(EnergyUnit::kilojoules, EnergyUnit::kilocalories) /
211  convert(DistanceUnit::nanometers, DistanceUnit::angstroms);
212  case ForceUnit::newtons:
213  return convert(EnergyUnit::kilojoules, EnergyUnit::joules) /
214  convert(DistanceUnit::nanometers, DistanceUnit::meters);
215  case ForceUnit::kilojoules_per_nanometer:
216  return 1.0;
217  case ForceUnit::kilojoules_per_angstrom:
218  return (1.0) /
219  convert(DistanceUnit::nanometers, DistanceUnit::angstroms);
220  case ForceUnit::hatree_per_bohr:
221  return convert(EnergyUnit::kilojoules, EnergyUnit::hartrees) /
222  (convert(DistanceUnit::nanometers, DistanceUnit::bohr));
223  }
224  return 0.0;
225  }
226 
228  constexpr double getMolarForceValue_(
229  const MolarForceUnit& enum_type) const noexcept {
230  switch (enum_type) {
231  case MolarForceUnit::kilocalories_per_mole_angstrom:
232  return convert(MolarEnergyUnit::kilojoules_per_mole,
233  MolarEnergyUnit::kilocalories_per_mole) /
234  convert(DistanceUnit::nanometers, DistanceUnit::angstroms);
235  case MolarForceUnit::newtons_per_mole:
236  return convert(MolarEnergyUnit::kilojoules_per_mole,
237  MolarEnergyUnit::joules_per_mole) /
238  convert(DistanceUnit::nanometers, DistanceUnit::meters);
239  case MolarForceUnit::kilojoules_per_mole_nanometer:
240  return 1.0;
241  case MolarForceUnit::kilojoules_per_mole_angstrom:
242  return (1.0) /
243  convert(DistanceUnit::nanometers, DistanceUnit::angstroms);
244  case MolarForceUnit::hatree_per_mole_bohr:
245  return convert(MolarEnergyUnit::kilojoules_per_mole,
246  MolarEnergyUnit::hartrees_per_mole) /
247  (convert(DistanceUnit::nanometers, DistanceUnit::bohr));
248  }
249  return 0.0;
250  }

252  public:
253  constexpr double convert(const DistanceUnit& from,
254  const DistanceUnit& to) const noexcept {
255  return getDistanceValue_(to) / getDistanceValue_(from);
256  }
257  constexpr double convert(const TimeUnit& from,
258  const TimeUnit& to) const noexcept {
259  return getTimeValue_(to) / getTimeValue_(from);
260  }
261  constexpr double convert(const MassUnit& from,
262  const MassUnit& to) const noexcept {
263  return getMassValue_(to) / getMassValue_(from);
264  }
265  constexpr double convert(const EnergyUnit& from,
266  const EnergyUnit& to) const noexcept {
267  return getEnergyValue_(to) / getEnergyValue_(from);
268  }
269  constexpr double convert(const MolarEnergyUnit& from,
270  const MolarEnergyUnit& to) const noexcept {
271  return getMolarEnergyValue_(to) / getMolarEnergyValue_(from);
272  }
273  constexpr double convert(const ChargeUnit& from,
274  const ChargeUnit& to) const noexcept {
275  return getChargeValue_(to) / getChargeValue_(from);
276  }
277  constexpr double convert(const VelocityUnit& from,
278  const VelocityUnit& to) const noexcept {
279  return getVelocityValue_(to) / getVelocityValue_(from);
280  }
281  constexpr double convert(const ForceUnit& from,
282  const ForceUnit& to) const noexcept {
283  return (getForceValue_(to)) / (getForceValue_(from));
284  }
285  constexpr double convert(const MolarForceUnit& from,
286  const MolarForceUnit& to) const noexcept {
287  return (getMolarForceValue_(to)) / (getMolarForceValue_(from));
288  }
