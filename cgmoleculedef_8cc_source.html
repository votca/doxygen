68 void CGMoleculeDef::ParseBeads(tools::Property &options) {
69 
70  for (tools::Property *p : options.Select("cg_bead")) {
71  beaddef_t *beaddef = new beaddef_t;
72  beaddef->_options = p;
73 
74  beaddef->_name = p->get("name").as<string>();
75  beaddef->_type = p->get("type").as<string>();
76  beaddef->_mapping = p->get("mapping").as<string>();
77  if (p->exists("symmetry")) {
78  Index sym = p->get("symmetry").as<Index>();
79  if (sym == 1) {
80  beaddef->_symmetry = Bead::spherical;
81  } else if (sym == 3) {
82  beaddef->_symmetry = Bead::ellipsoidal;
83  } else {
84  throw std::runtime_error(
85  "Only beads with spherical(1) or ellipsoidal(3) symmetry "
86  "implemented.");
87  }
88  } else {
89  beaddef->_symmetry = Bead::spherical;
90  }
91 
92  if (_beads_by_name.find(beaddef->_name) != _beads_by_name.end()) {
93  throw std::runtime_error(string("bead name ") + beaddef->_name +
94  " not unique in mapping");
95  }
96  _beads.push_back(beaddef);
97  _beads_by_name[beaddef->_name] = beaddef;
98  }
99 }

111 Molecule *CGMoleculeDef::CreateMolecule(Topology &top) {
112  // add the residue names
113  Residue *res = top.CreateResidue(_name);
114  Molecule *minfo = top.CreateMolecule(_name);
115 
116  // create the atoms
117  for (auto &bead_def : _beads) {
118 
119  string type = bead_def->_type;
120  if (!top.BeadTypeExist(type)) {
121  top.RegisterBeadType(type);
122  }
123  Bead *bead = top.CreateBead(bead_def->_symmetry, bead_def->_name, type,
124  res->getId(), 0, 0);
125  minfo->AddBead(bead, bead->getName());
126  }
127 
128  // create the bonds
129  map<string, string> had_iagroup;
130 
131  for (tools::Property *prop : _bonded) {
132  std::list<Index> atoms;
133  string iagroup = prop->get("name").as<string>();
134 
135  if (had_iagroup[iagroup] == "yes") {
136  throw runtime_error(
137  string("double occurence of interactions with name ") + iagroup);
138  }
139  had_iagroup[iagroup] = "yes";
140 
141  tools::Tokenizer tok(prop->get("beads").value(), " \n\t");
142  for (auto &atom : tok) {
143  Index i = minfo->getBeadIdByName(atom);
144  if (i < 0) {
145  throw runtime_error(
146  string("error while trying to create bonded interaction, "
147  "bead " +
148  atom + " not found"));
149  }
150 
151  atoms.push_back(i);
152  }
153 
154  Index NrBeads = 1;
155  if (prop->name() == "bond") {
156  NrBeads = 2;
157  } else if (prop->name() == "angle") {
158  NrBeads = 3;
159  } else if (prop->name() == "dihedral") {
160  NrBeads = 4;
161  }
162 
163  if ((atoms.size() % NrBeads) != 0) {
164  throw runtime_error("Number of atoms in interaction '" +
165  prop->get("name").as<string>() +
166  "' is not a multiple of " +
167  lexical_cast<string>(NrBeads) + "! Missing beads?");
168  }
169 
170  Index index = 0;
171  while (!atoms.empty()) {
172  Interaction *ic;
173 
174  if (prop->name() == "bond") {
175  ic = new IBond(atoms);
176  } else if (prop->name() == "angle") {
177  ic = new IAngle(atoms);
178  } else if (prop->name() == "dihedral") {
179  ic = new IDihedral(atoms);
180  } else {
181  throw runtime_error("unknown bonded type in map: " + prop->name());
182  }
183 
184  ic->setGroup(iagroup);
185  ic->setIndex(index);
186  ic->setMolecule(minfo->getId());
187  top.AddBondedInteraction(ic);
188  minfo->AddInteraction(ic);
189  index++;
190  }
191  }
192  return minfo;
193 }

195 Map *CGMoleculeDef::CreateMap(Molecule &in, Molecule &out) {
196  if (out.BeadCount() != Index(_beads.size())) {
197  throw runtime_error(
198  "number of beads for cg molecule and mapping definition do "
199  "not match, check your molecule naming.");
200  }
201 
202  Map *map = new Map(in, out);
203  for (auto &bead : _beads) {
204 
205  Index iout = out.getBeadByName(bead->_name);
206  if (iout < 0) {
207  throw runtime_error(string("mapping error: reference molecule " +
208  bead->_name + " does not exist"));
209  }
210 
211  tools::Property *mdef = getMapByName(bead->_mapping);
212  if (!mdef) {
213  throw runtime_error(string("mapping " + bead->_mapping + " not found"));
214  }
215 
217  BeadMap *bmap;
218  switch (bead->_symmetry) {
219  case 1:
220  bmap = new Map_Sphere();
221  break;
222  case 3:
223  bmap = new Map_Ellipsoid();
224  break;
225  default:
226  throw runtime_error(string("unknown symmetry in bead definition!"));
227  }
229 
230  bmap->Initialize(&in, out.getBead(iout), (bead->_options), mdef);
231  map->AddBeadMap(bmap);
232  }
233  return map;
234 }
