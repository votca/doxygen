00058 void CGMoleculeDef::ParseBeads(Property &options) {
00059     list<Property *> beads =
00060             options.Select("cg_bead");
00061 
00062     for (list<Property *>::iterator iter = beads.begin();
00063             iter != beads.end(); ++iter) {
00064         Property *p = *iter;
00065         beaddef_t *beaddef = new beaddef_t;
00066         beaddef->_options = p;
00067 
00068         beaddef->_name = p->get("name").as<string>();
00069         beaddef->_type = p->get("type").as<string>();
00070         beaddef->_mapping = p->get("mapping").as<string>();
00071         if (p->exists("symmetry"))
00072             beaddef->_symmetry = p->get("symmetry").as<int>();
00073         else
00074             beaddef->_symmetry = 1;
00075         
00076         if (_beads_by_name.find(beaddef->_name) != _beads_by_name.end())
00077             throw std::runtime_error(string("bead name ") + beaddef->_name
00078                 + " not unique in mapping");
00079         _beads.push_back(beaddef);
00080         _beads_by_name[beaddef->_name] = beaddef;
00081     }
00082 }

00097 Molecule * CGMoleculeDef::CreateMolecule(Topology & top)
00098 {   
00099     // add the residue names
00100     Residue *res = top.CreateResidue(_name);
00101     Molecule *minfo = top.CreateMolecule(_name);
00102     
00103     // create the atoms
00104     vector<beaddef_t *>::iterator iter;
00105     for(iter = _beads.begin(); iter != _beads.end(); ++iter) {
00106         Bead *bead;
00107         BeadType *bt = top.GetOrCreateBeadType((*iter)->_type);
00108         bead = top.CreateBead((*iter)->_symmetry, (*iter)->_name, bt, res->getId(), 0, 0);
00109         minfo->AddBead(bead, bead->getName());
00110 
00111         bead->setOptions(*(*iter)->_options);
00112         
00113     }    
00114     
00115     // create the bonds
00116     list<Property *>::iterator ibnd;
00117     map<string, string> had_iagroup;
00118 
00119     for(ibnd = _bonded.begin(); ibnd!=_bonded.end(); ++ibnd) {
00120         list<int> atoms;
00121         string iagroup = (*ibnd)->get("name").as<string>();
00122 
00123         if(had_iagroup[iagroup] == "yes")
00124             throw runtime_error(
00125                 string("double occurence of interactions with name ") + iagroup);
00126         had_iagroup[iagroup] = "yes";
00127 
00128         Tokenizer tok((*ibnd)->get("beads").value(), " \n\t");
00129         for (Tokenizer::iterator atom = tok.begin(); atom != tok.end(); ++atom) {
00130             int i = minfo->getBeadIdByName(*atom);
00131             if(i < 0)
00132                 throw runtime_error(string("error while trying to create bonded interaction, "
00133                         "bead " + *atom + " not found"));
00134 
00135             atoms.push_back(i);
00136         }
00137 
00138         int index=0;
00139         while(!atoms.empty()) {
00140             Interaction *ic;
00141 
00142             if((*ibnd)->name() == "bond")
00143                 ic = new IBond(atoms);
00144             else if((*ibnd)->name() == "angle")
00145                 ic = new IAngle(atoms);
00146             else if((*ibnd)->name() == "dihedral")
00147                 ic = new IDihedral(atoms);
00148             else
00149                 throw runtime_error("unknown bonded type in map: " + (*ibnd)->name());
00150 
00151             ic->setGroup(iagroup);
00152             ic->setIndex(index);
00153             ic->setMolecule(minfo->getId());
00154             top.AddBondedInteraction(ic);
00155             minfo->AddInteraction(ic);
00156             index++;
00157         }
00158     }
00159     return minfo;
00160 }

00162 Map *CGMoleculeDef::CreateMap(Molecule &in, Molecule &out)
00163 {       
00164     if((unsigned int)out.BeadCount() != _beads.size()) {
00165         throw runtime_error("number of beads for cg molecule and mapping definition do "
00166                 "not match, check your molecule naming.");
00167     }
00168 
00169     Map *map = new Map(in, out);
00170     for(vector<beaddef_t *>::iterator def = _beads.begin();
00171             def != _beads.end(); ++def) {
00172 
00173         int iout = out.getBeadByName((*def)->_name);
00174         if(iout < 0) 
00175             throw runtime_error(string("mapping error: reference molecule "
00176                     + (*def)->_name + " does not exist"));
00177         
00178         Property *mdef = getMapByName((*def)->_mapping);
00179         if(!mdef)
00180             throw runtime_error(string("mapping " + (*def)->_mapping + " not found"));
00181         
00183         BeadMap *bmap;
00184         switch((*def)->_symmetry) {
00185         case 1:
00186             bmap = new Map_Sphere();
00187             break;
00188         case 3:
00189             bmap = new Map_Ellipsoid();
00190             break;
00191         default:
00192             throw runtime_error(string("unknown symmetry in bead definition!"));
00193         }
00195         
00196         bmap->Initialize(&in, out.getBead(iout), ((*def)->_options), mdef);
00197         map->AddBeadMap(bmap);
00198         
00199     }
00200     return map;
