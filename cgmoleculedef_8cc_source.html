58 void CGMoleculeDef::ParseBeads(Property &options) {
59  list<Property *> beads =
60  options.Select("cg_bead");
61 
62  for (list<Property *>::iterator iter = beads.begin();
63  iter != beads.end(); ++iter) {
64  Property *p = *iter;
65  beaddef_t *beaddef = new beaddef_t;
66  beaddef->_options = p;
67 
68  beaddef->_name = p->get("name").as<string>();
69  beaddef->_type = p->get("type").as<string>();
70  beaddef->_mapping = p->get("mapping").as<string>();
71  if (p->exists("symmetry"))
72  beaddef->_symmetry = p->get("symmetry").as<int>();
73  else
74  beaddef->_symmetry = 1;
75 
76  if (_beads_by_name.find(beaddef->_name) != _beads_by_name.end())
77  throw std::runtime_error(string("bead name ") + beaddef->_name
78  + " not unique in mapping");
79  _beads.push_back(beaddef);
80  _beads_by_name[beaddef->_name] = beaddef;
81  }
82 }

97 Molecule * CGMoleculeDef::CreateMolecule(Topology & top)
98 {
99  // add the residue names
100  Residue *res = top.CreateResidue(_name);
101  Molecule *minfo = top.CreateMolecule(_name);
102 
103  // create the atoms
104  vector<beaddef_t *>::iterator iter;
105  for(iter = _beads.begin(); iter != _beads.end(); ++iter) {
106  Bead *bead;
107  BeadType *bt = top.GetOrCreateBeadType((*iter)->_type);
108  bead = top.CreateBead((*iter)->_symmetry, (*iter)->_name, bt, res->getId(), 0, 0);
109  minfo->AddBead(bead, bead->getName());
110 
111  bead->setOptions(*(*iter)->_options);
112 
113  }
114 
115  // create the bonds
116  list<Property *>::iterator ibnd;
117  map<string, string> had_iagroup;
118 
119  for(ibnd = _bonded.begin(); ibnd!=_bonded.end(); ++ibnd) {
120  list<int> atoms;
121  string iagroup = (*ibnd)->get("name").as<string>();
122 
123  if(had_iagroup[iagroup] == "yes")
124  throw runtime_error(
125  string("double occurence of interactions with name ") + iagroup);
126  had_iagroup[iagroup] = "yes";
127 
128  Tokenizer tok((*ibnd)->get("beads").value(), " \n\t");
129  for (Tokenizer::iterator atom = tok.begin(); atom != tok.end(); ++atom) {
130  int i = minfo->getBeadIdByName(*atom);
131  if(i < 0)
132  throw runtime_error(string("error while trying to create bonded interaction, "
133  "bead " + *atom + " not found"));
134 
135  atoms.push_back(i);
136  }
137 
138  int NrBeads=1;
139  if((*ibnd)->name() == "bond")
140  NrBeads=2;
141  else if((*ibnd)->name() == "angle")
142  NrBeads=3;
143  else if((*ibnd)->name() == "dihedral")
144  NrBeads=4;
145 
146  if( (atoms.size() % NrBeads) != 0 )
147  throw runtime_error("Number of atoms in interaction '" + (*ibnd)->get("name").as<string>() + "' is not a multiple of " + lexical_cast<string>(NrBeads) + "! Missing beads?");
148 
149  int index=0;
150  while(!atoms.empty()) {
151  Interaction *ic;
152 
153  if((*ibnd)->name() == "bond")
154  ic = new IBond(atoms);
155  else if((*ibnd)->name() == "angle")
156  ic = new IAngle(atoms);
157  else if((*ibnd)->name() == "dihedral")
158  ic = new IDihedral(atoms);
159  else
160  throw runtime_error("unknown bonded type in map: " + (*ibnd)->name());
161 
162  ic->setGroup(iagroup);
163  ic->setIndex(index);
164  ic->setMolecule(minfo->getId());
165  top.AddBondedInteraction(ic);
166  minfo->AddInteraction(ic);
167  index++;
168  }
169  }
170  return minfo;
171 }

173 Map *CGMoleculeDef::CreateMap(Molecule &in, Molecule &out)
174 {
175  if((unsigned int)out.BeadCount() != _beads.size()) {
176  throw runtime_error("number of beads for cg molecule and mapping definition do "
177  "not match, check your molecule naming.");
178  }
179 
180  Map *map = new Map(in, out);
181  for(vector<beaddef_t *>::iterator def = _beads.begin();
182  def != _beads.end(); ++def) {
183 
184  int iout = out.getBeadByName((*def)->_name);
185  if(iout < 0)
186  throw runtime_error(string("mapping error: reference molecule "
187  + (*def)->_name + " does not exist"));
188 
189  Property *mdef = getMapByName((*def)->_mapping);
190  if(!mdef)
191  throw runtime_error(string("mapping " + (*def)->_mapping + " not found"));
192 
194  BeadMap *bmap;
195  switch((*def)->_symmetry) {
196  case 1:
197  bmap = new Map_Sphere();
198  break;
199  case 3:
200  bmap = new Map_Ellipsoid();
201  break;
202  default:
203  throw runtime_error(string("unknown symmetry in bead definition!"));
204  }
206 
207  bmap->Initialize(&in, out.getBead(iout), ((*def)->_options), mdef);
208  map->AddBeadMap(bmap);
209 
210  }
211  return map;
212 }
