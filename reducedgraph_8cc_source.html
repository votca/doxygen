44 bool compareChainWithChains_(const vector<int>& chain,
45  const vector<vector<int>>& chains) {
46  bool match = false;
47  for (const vector<int>& chain2 : chains) {
48  if (chain2.size() == chain.size()) {
49  match = true;
50  for (size_t index = 0; index < chain.size(); ++index) {
51  if (chain2.at(index) != chain.at(index)) {
52  match = false;
53  break;
54  }
55  } // Cycle vertices in each chain
56  if (match) {
57  return true;
58  }
59  } // Chains same size
60  }
61  return false;
62 }

94 set<int> getVertexJunctions_(const vector<ReducedEdge>& reduced_edges) {
95  unordered_map<int, int> vertex_count;
96  for (ReducedEdge reduced_edge : reduced_edges) {
97  // if loop, increment value is double and the first index is skipped to
98  // prevent over counting of the first number
99  int increment = 1;
100  size_t index = 0;
101  if (reduced_edge.loop()) {
102  ++index;
103  increment = 2;
104  }
105  vector<int> chain = reduced_edge.getChain();
106  for (; index < chain.size(); ++index) {
107  if (vertex_count.count(chain.at(index))) {
108  vertex_count[chain.at(index)] += increment;
109  } else {
110  vertex_count[chain.at(index)] = increment;
111  }
112  }
113  }
114  set<int> junctions;
115  for (pair<int, int> vertex_and_count : vertex_count) {
116  if (vertex_and_count.second > 2) {
117  junctions.insert(vertex_and_count.first);
118  }
119  }
120  return junctions;
121 }

192 void orderChainAfterInitialVertex_(vector<int>& chain) {
193  size_t ignore_first_and_last_vertex = 2;
194  size_t total_number_to_parse =
195  (chain.size() - ignore_first_and_last_vertex) / 2;
196  bool reverse_vector = false;
197  for (size_t count = 0; count < (total_number_to_parse); ++count) {
198  if (chain.at(chain.size() - count - 1) < chain.at(count + 1)) {
199  reverse_vector = true;
200  break;
201  }
202  }
203 
204  if (reverse_vector) {
205  reverse(chain.begin(), chain.end());
206  }
207 }

209 bool reorderAndStoreChainIfDoesNotExist_(
210  vector<Edge>& edges,
211  unordered_map<Edge, vector<vector<int>>>& expanded_edges, vector<int> chain,
212  int vertex, size_t& chain_index) {
213 
214  Edge edge(vertex, vertex);
215  edges.push_back(edge);
216  vector<int> new_chain;
217  for (size_t index = 0; index < chain.size(); ++index) {
218  if (((chain_index + index) % chain.size()) == 0) {
219  ++chain_index;
220  }
221  int new_chain_index = (chain_index + index) % chain.size();
222  new_chain.push_back(chain.at(new_chain_index));
223  }
224  // Ensure that the new_chain is sorted so after the first vertex they are
225  // ordered from smallest to largest
226  orderChainAfterInitialVertex_(new_chain);
227  bool match = compareChainWithChains_(new_chain, expanded_edges[edge]);
228  if (!match) {
229  expanded_edges[edge].push_back(new_chain);
230  return true;
231  }
232  return false;
233 }

261 void ReducedGraph::init_(vector<ReducedEdge> reduced_edges,
262  unordered_map<int, GraphNode> nodes) {
263  vector<Edge> edges;
264  nodes_ = nodes;
265 
266  junctions_ = getVertexJunctions_(reduced_edges);
267 
268  for (const ReducedEdge& reduced_edge : reduced_edges) {
269 
270  if (reduced_edge.loop() &&
271  junctions_.count(reduced_edge.getEndPoint1()) == 0) {
272  vector<int> chain = reduced_edge.getChain();
273  size_t chain_index = 0;
274  bool edge_added = false;
275  for (int vertex : chain) {
276  if (junctions_.count(vertex)) {
277  edge_added = reorderAndStoreChainIfDoesNotExist_(
278  edges, expanded_edges_, chain, vertex, chain_index);
279  break;
280  }
281  ++chain_index;
282  }
283  if (!edge_added) {
284  addEdgeIfNotLoop_(edges, reduced_edge, expanded_edges_);
285  }
286  } else {
287  addEdgeIfNotLoop_(edges, reduced_edge, expanded_edges_);
288  }
289  }
290 
291  edge_container_ = EdgeContainer(edges);
292 
293  calcId_();
294 }

315  if (nodes.size() < vertices.size()) {

348 vector<vector<Edge>> ReducedGraph::expandEdge(const Edge& edge) const {
349  vector<vector<Edge>> all_edges;
350  vector<vector<int>> chains = expanded_edges_.at(edge);
351  for (vector<int> vertices : chains) {
352  vector<Edge> edges;
353  for (size_t index = 0; index < (vertices.size() - 1); ++index) {
354  Edge edge_temp(vertices.at(index), vertices.at(index + 1));
355  edges.push_back(edge_temp);
356  }
357  all_edges.push_back(edges);
358  }
359  return all_edges;

362 vector<pair<int, GraphNode>> ReducedGraph::getNodes() const {
363  vector<int> vertices = edge_container_.getVertices();
364  vector<pair<int, GraphNode>> nodes;
365  for (const int vertex : vertices) {

367  nodes.push_back(id_and_node);
368  }
369 
370  set<int> all_connected_vertices = getAllConnectedVertices_(expanded_edges_);
371  // Grab the nodes that are not attached to any edges
372  for (pair<int, GraphNode> id_and_node : nodes_) {
373  if (!all_connected_vertices.count(id_and_node.first)) {
374  nodes.push_back(id_and_node);
