32 bool compareChainWithChains_(const vector<int>& chain,
33  const vector<vector<int>>& chains) {
34  bool match = false;
35  for (const vector<int>& chain2 : chains) {
36  if (chain2.size() == chain.size()) {
37  match = true;
38  for (size_t index = 0; index < chain.size(); ++index) {
39  if (chain2.at(index) != chain.at(index)) {
40  match = false;
41  break;
42  }
43  } // Cycle vertices in each chain
44  if (match) return true;
45  } // Chains same size
46  }
47  return false;
48 }

50 set<int> getVertexJunctions_(const vector<ReducedEdge>& reduced_edges) {
51  unordered_map<int, int> vertex_count;
52  for (ReducedEdge reduced_edge : reduced_edges) {
53  // if loop, increment value is double and the first index is skipped to
54  // prevent over counting of the first number
55  int increment = 1;
56  size_t index = 0;
57  if (reduced_edge.loop()) {
58  ++index;
59  increment = 2;
60  }
61  vector<int> chain = reduced_edge.getChain();
62  for (; index < chain.size(); ++index) {
63  if (vertex_count.count(chain.at(index))) {
64  vertex_count[chain.at(index)] += increment;
65  } else {
66  vertex_count[chain.at(index)] = increment;
67  }
68  }
69  }
70  set<int> junctions;
71  for (pair<int, int> vertex_and_count : vertex_count) {
72  if (vertex_and_count.second > 2) {
73  junctions.insert(vertex_and_count.first);
74  }
75  }
76  return junctions;
77 }

96 void orderChainAfterInitialVertex_(vector<int>& chain) {
97  size_t ignore_first_and_last_vertex = 2;
98  size_t total_number_to_parse =
99  (chain.size() - ignore_first_and_last_vertex) / 2;
100  bool reverse_vector = false;
101  for (size_t count = 0; count < (total_number_to_parse); ++count) {
102  if (chain.at(chain.size() - count - 1) < chain.at(count + 1)) {
103  reverse_vector = true;
104  break;
105  }
106  }
107 
108  if (reverse_vector) {
109  reverse(chain.begin(), chain.end());
110  }
111 }

113 bool reordereAndStoreChainIfDoesNotExist_(
114  vector<Edge>& edges,
115  unordered_map<Edge, vector<vector<int>>>& expanded_edges, vector<int> chain,
116  int vertex, size_t& chain_index) {
117 
118  Edge edge(vertex, vertex);
119  edges.push_back(edge);
120  vector<int> new_chain;
121  for (size_t index = 0; index < chain.size(); ++index) {
122  if (((chain_index + index) % chain.size()) == 0) {
123  ++chain_index;
124  }
125  int new_chain_index = (chain_index + index) % chain.size();
126  new_chain.push_back(chain.at(new_chain_index));
127  }
128  // Ensure that the new_chain is sorted so after the first vertex they are
129  // ordered from smallest to largest
130  orderChainAfterInitialVertex_(new_chain);
131  bool match = compareChainWithChains_(new_chain, expanded_edges[edge]);
132  if (!match) {
133  expanded_edges[edge].push_back(new_chain);
134  return true;
135  }
136  return false;
137 }

139 void ReducedGraph::init_(vector<ReducedEdge> reduced_edges,
140  unordered_map<int, GraphNode> nodes) {
141  vector<Edge> edges;
142  nodes_ = nodes;
143 
144  junctions_ = getVertexJunctions_(reduced_edges);
145 
146  for (const ReducedEdge& reduced_edge : reduced_edges) {
147 
148  if (reduced_edge.loop() &&
149  junctions_.count(reduced_edge.getEndPoint1()) == 0) {
150  vector<int> chain = reduced_edge.getChain();
151  size_t chain_index = 0;
152  bool edge_added = false;
153  for (int vertex : chain) {
154  if (junctions_.count(vertex)) {
155  edge_added = reordereAndStoreChainIfDoesNotExist_(
156  edges, expanded_edges_, chain, vertex, chain_index);
157  break;
158  }
159  ++chain_index;
160  }
161  if (!edge_added) {
162  addEdgeIfNotLoop_(edges, reduced_edge, expanded_edges_);
163  }
164  } else {
165  addEdgeIfNotLoop_(edges, reduced_edge, expanded_edges_);
166  }
167  }
168

171  calcId_();
172 }
