103 void PBC::RunKMC() {
104 
105 
106  std::cout << "Running KMC-PBC: " << std::endl;
107 
108  Graph graph;
109  State state;
110 
111  //Load the graph from the state file
112  std::string filename( "state.sql" );
113  graph.Load_Graph( filename );
114 
115  //Interaction still to be implemented - only Pauli exclusion available right now
116  if (_interaction == "Pauli"){
117  std::cout << "Pauli exclusion principal used as method of interaction for like charges (no double occupation)" << std::endl;
118  }
119  /*else if (_interaction == "Coulomb"){
120  std::cout << " Coulomb electrostatic interaction between like charges included " << std::endl;
121  }*/
122 
123  //option to read the rates from the state file or calculate
124  if (_rates == "read"){
125  graph.Load_Rates(filename);
126  }
127  else if ( _rates == "calculate"){
128  graph.Rates_Calculation(filename, _nelectrons, _nholes, _fieldX, _fieldY, _fieldZ, _temperature);
129  }
130  else {
131  std::cout << "The option for rates was incorrectly specified. Please choose to 'read' rates or 'calculate' rates. " << std::endl;
132  }
133 
134  //graph.Print();
135  CarrierFactory::RegisterAll();
136  EventFactory::RegisterAll();
137 
138  std::cout << "Number of Nodes: " << graph.nodes_size() << std::endl;
139  std::cout << "Number of electrons: " << _nelectrons << std::endl;
140  std::cout << "Number of holes: " << _nholes << std::endl;
141  std::cout << "Method of carrier injection: " << _injection_method << std::endl;
142 
143  std::cout << std::endl;
144 
145  if(_nelectrons>graph.nodes_size() || _nholes>graph.nodes_size()){
146  std::cout << "The number of carriers exceeds the number of available nodes!" << std::endl;
147  return;
148  }
149 
150  // use the seed to initialise the random number generator srand
151  srand(_seed);
152  votca::tools::Random2 *RandomVariable = new votca::tools::Random2();
153  RandomVariable->init(rand(), rand(), rand(), rand());
154 
155  std::cout << std::endl;
156 
157  //create electrons and place on nodes of the graph
158  if(_nelectrons != 0){
159 
160  for ( int electron = 1; electron <= _nelectrons; ++electron ) {
161 
162  // Create electrons
163  Carrier* e_carrier = state.AddCarrier( "electron" );
164  Electron* ecarrier = dynamic_cast<Electron*>(e_carrier);
165 
166  //randomly inject to a node in the graph
167  if (_injection_method == "random"){
168  int node_id = RandomVariable->rand_uniform_int(graph.nodes_size());
169  BNode* node_from = graph.GetNode(node_id + 1);
170  //carrier can only be placed if the node is free, else find a new node
171  while (ecarrier->AddNode(node_from)==false){
172  int node_id = RandomVariable->rand_uniform_int(graph.nodes_size());
173  node_from = graph.GetNode(node_id + 1);
174  }
175  if (ecarrier->AddNode(node_from)==true){ ecarrier->AddNode( node_from );}
176  }
177 
178  //inject to the first nodes of the graph to have an ordered placement of electrons
179  else if (_injection_method == "uniform") {
180  BNode* node_from = graph.GetNode(electron);
181  ecarrier->AddNode( node_from );
182  //node_from->PrintNode();
183  }
184  }
185 
186  std::cout << std::endl;
187 
188  }
189 
190  if(_nholes != 0){
191 
192  for ( int hole = 1; hole <= _nholes; ++hole ) {
193 
194  // Create holes
195  Carrier* h_carrier = state.AddCarrier( "hole" );
196  Hole* hcarrier = dynamic_cast<Hole*>(h_carrier);
197 
198  //randomly inject to a node in the graph
199  if (_injection_method == "random"){
200  int node_id = RandomVariable->rand_uniform_int(graph.nodes_size());
201  BNode* node_from = graph.GetNode(node_id + 2);
202  //only place the carrier if the node is free, otherwise find a new node
203  while (hcarrier->AddNode(node_from)==false){
204  int node_id = RandomVariable->rand_uniform_int(graph.nodes_size());
205  node_from = graph.GetNode(node_id + 2);
206  }
207  if (hcarrier->AddNode(node_from)==true){ hcarrier->AddNode( node_from );}
208  }
209 
210  //inject to the first nodes of the graph
211  else if (_injection_method == "uniform") {
212  BNode* node_from = graph.GetNode(hole);
213  hcarrier->AddNode( node_from );
214  //node_from->PrintNode();
215  }
216 
217  }
218 
219  std::cout << std::endl;
220  }
221 
222  //use vssm2_nodes to organise all the nodes and events
223  VSSM2_NODES vssm2;
224  //initialise with the state and the graph
225  vssm2.Initialize( &state, &graph );
226  //vssm2.AttachObserver(Observer, _nsteps );
227  // run the algorithm
228  vssm2.Run(_runtime, _nsteps, RandomVariable, _nelectrons, _nholes, _trajectoryfile, _outtime, _fieldX, _fieldY, _fieldZ);
229 
230 }
