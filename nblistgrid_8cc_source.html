63 void NBListGrid::InitializeGrid(const matrix &box)
64 {
65  _box_a = box.getCol(0);
66  _box_b = box.getCol(1);
67  _box_c = box.getCol(2);
68 
69  // create plane normals
70  _norm_a = _box_b ^ _box_c;
71  _norm_b = _box_c ^ _box_a;
72  _norm_c = _box_a ^ _box_b;
73 
74  _norm_a.normalize();
75  _norm_b.normalize();
76  _norm_c.normalize();
77 
78  double la = _box_a * _norm_a;
79  double lb = _box_b * _norm_b;
80  double lc = _box_c * _norm_c;
81 
82  // calculate grid size, each grid has to be at least size of cut-off
83  _box_Na = max((int)(fabs(la/_cutoff)), 1);
84  _box_Nb = max((int)(fabs(lb/_cutoff)), 1);
85  _box_Nc = max((int)(fabs(lc/_cutoff)), 1);
86 
87  _norm_a = _norm_a / (_box_a*_norm_a)*(double)_box_Na;
88  _norm_b = _norm_b / (_box_b*_norm_b)*(double)_box_Nb;
89  _norm_c = _norm_c / (_box_c*_norm_c)*(double)_box_Nc;
90 
91  //cout << "grid size: " << _box_Na << "x" << _box_Nb << "x" << _box_Nc << endl;
92  _grid.resize(_box_Na*_box_Nb*_box_Nc);
93 
94  int a1,a2,b1,b2,c1,c2;
95 
96  a1 = b1 = c1 = -1;
97  a2 = b2 = c2 = 1;
98 
99  if(_box_Na < 3) a2 = 0;
100  if(_box_Nb < 3) b2 = 0;
101  if(_box_Nc < 3) c2 = 0;
102 
103  if(_box_Na < 2) a1 = 0;
104  if(_box_Nb < 2) b1 = 0;
105  if(_box_Nc < 2) c1 = 0;
106 
107  // wow, setting up the neighbours is an ugly for construct!
108  // loop from N..2*N to avoid if and only use %
109  for(int a=_box_Na; a<2*_box_Na; ++a)
110  for(int b=_box_Nb; b<2*_box_Nb; ++b)
111  for(int c=_box_Nc; c<2*_box_Nc; ++c) {
112  cell_t &cell = getCell(a%_box_Na, b%_box_Nb, c%_box_Nc);
113  for(int aa=a+a1; aa<=a+a2; ++aa)
114  for(int bb=b+b1; bb<=b+b2; ++bb)
115  for(int cc=c+c1; cc<=c+c2; ++cc) {
116  cell_t *c = &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc);
117  if(c == &cell) continue; // ignore self
118  cell._neighbours.push_back(
119  &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc)
120  );
121  }
122  }
123 }

125 NBListGrid::cell_t &NBListGrid::getCell(const vec &r)
126 {
127  int a = (int)floor(r*_norm_a);
128  int b = (int)floor(r*_norm_b);
129  int c = (int)floor(r*_norm_c);
130 
131  if(a<0) a = _box_Na + a%_box_Na;
132  a %= _box_Na;
133 
134  if(b<0)
135  b = _box_Nb + b%_box_Nb;
136  b %= _box_Nb;
137 
138  if(c<0) c = _box_Nc + c%_box_Nc;
139  c %= _box_Nc;
140 
141  return getCell(a,b,c);
142  }
