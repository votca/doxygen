69 void NBListGrid::InitializeGrid(const Eigen::Matrix3d &box) {
70  _box_a = box.col(0);
71  _box_b = box.col(1);
72  _box_c = box.col(2);
73 
74  // create plane normals
75  _norm_a = _box_b.cross(_box_c);
76  _norm_b = _box_c.cross(_box_a);
77  _norm_c = _box_a.cross(_box_b);
78 
79  _norm_a.normalize();
80  _norm_b.normalize();
81  _norm_c.normalize();
82 
83  double la = _box_a.dot(_norm_a);
84  double lb = _box_b.dot(_norm_b);
85  double lc = _box_c.dot(_norm_c);
86 
87  // calculate grid size, each grid has to be at least size of cut-off
88  _box_Na = Index(std::max(std::abs(la / _cutoff), 1.0));
89  _box_Nb = Index(std::max(std::abs(lb / _cutoff), 1.0));
90  _box_Nc = Index(std::max(std::abs(lc / _cutoff), 1.0));
91 
92  _norm_a = _norm_a / _box_a.dot(_norm_a) * (double)_box_Na;
93  _norm_b = _norm_b / _box_b.dot(_norm_b) * (double)_box_Nb;
94  _norm_c = _norm_c / _box_c.dot(_norm_c) * (double)_box_Nc;
95 
96  _grid.resize(_box_Na * _box_Nb * _box_Nc);
97 
98  Index a1, a2, b1, b2, c1, c2;
99 
100  a1 = b1 = c1 = -1;
101  a2 = b2 = c2 = 1;
102 
103  if (_box_Na < 3) {
104  a2 = 0;
105  }
106  if (_box_Nb < 3) {
107  b2 = 0;
108  }
109  if (_box_Nc < 3) {
110  c2 = 0;
111  }
112 
113  if (_box_Na < 2) {
114  a1 = 0;
115  }
116  if (_box_Nb < 2) {
117  b1 = 0;
118  }
119  if (_box_Nc < 2) {
120  c1 = 0;
121  }
122 
123  // wow, setting up the neighbours is an ugly for construct!
124  // loop from N..2*N to avoid if and only use %
125  for (Index a = _box_Na; a < 2 * _box_Na; ++a) {
126  for (Index b = _box_Nb; b < 2 * _box_Nb; ++b) {
127  for (Index c = _box_Nc; c < 2 * _box_Nc; ++c) {
128  cell_t &cell = getCell(a % _box_Na, b % _box_Nb, c % _box_Nc);
129  for (Index aa = a + a1; aa <= a + a2; ++aa) {
130  for (Index bb = b + b1; bb <= b + b2; ++bb) {
131  for (Index cc = c + c1; cc <= c + c2; ++cc) {
132  cell_t *cell2 =
133  &getCell(aa % _box_Na, bb % _box_Nb, cc % _box_Nc);
134  if (cell2 == &cell) {
135  continue; // ignore self
136  }
137  cell._neighbours.push_back(
138  &getCell(aa % _box_Na, bb % _box_Nb, cc % _box_Nc));
139  }
140  }
141  }
142  }
143  }
144  }
145 }

147 NBListGrid::cell_t &NBListGrid::getCell(const Eigen::Vector3d &r) {
148  Index a = (Index)floor(r.dot(_norm_a));
149  Index b = (Index)floor(r.dot(_norm_b));
150  Index c = (Index)floor(r.dot(_norm_c));
151 
152  if (a < 0) {
153  a = _box_Na + a % _box_Na;
154  }
155  a %= _box_Na;
156 
157  if (b < 0) {
158  b = _box_Nb + b % _box_Nb;
159  }
160  b %= _box_Nb;
161 
162  if (c < 0) {
163  c = _box_Nc + c % _box_Nc;
164  }
165  c %= _box_Nc;
166 
167  return getCell(a, b, c);
168 }

177 void NBListGrid::TestCell(NBListGrid::cell_t &cell, Bead *bead) {
178  Eigen::Vector3d u = bead->getPos();
179 
180  for (auto &_bead : cell._beads) {
181 
182  Eigen::Vector3d v = _bead->getPos();
183  Eigen::Vector3d r = _top->BCShortestConnection(v, u);
184  double d = r.norm();
185  if (d < _cutoff) {
186  if (_do_exclusions) {
187  if (_top->getExclusions().IsExcluded(_bead, bead)) {
188  continue;
189  }
190  }
191  if ((*_match_function)(_bead, bead, r, d)) {
192  if (!FindPair(_bead, bead)) {
193  AddPair(_pair_creator(_bead, bead, r));
194  }
195  }
196  }
197  }
