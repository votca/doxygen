00061 void NBListGrid::InitializeGrid(const matrix &box)
00062 {
00063     _box_a = box.getCol(0);
00064     _box_b = box.getCol(1);
00065     _box_c = box.getCol(2);
00066 
00067     // create plane normals
00068     _norm_a = _box_b ^ _box_c;
00069     _norm_b = _box_c ^ _box_a;
00070     _norm_c = _box_a ^ _box_b;
00071 
00072     _norm_a.normalize();
00073     _norm_b.normalize();
00074     _norm_c.normalize();
00075 
00076     double la = _box_a * _norm_a;
00077     double lb = _box_b * _norm_b;
00078     double lc = _box_c * _norm_c;
00079 
00080     // calculate grid size, each grid has to be at least size of cut-off
00081     _box_Na = max((int)(fabs(la/_cutoff)), 1);
00082     _box_Nb = max((int)(fabs(lb/_cutoff)), 1);
00083     _box_Nc = max((int)(fabs(lc/_cutoff)), 1);
00084 
00085     _norm_a = _norm_a / (_box_a*_norm_a)*(double)_box_Na;
00086     _norm_b = _norm_b / (_box_b*_norm_b)*(double)_box_Nb;
00087     _norm_c = _norm_c / (_box_c*_norm_c)*(double)_box_Nc;
00088 
00089     //cout << "grid size: " << _box_Na << "x" << _box_Nb << "x" << _box_Nc << endl;
00090     _grid.resize(_box_Na*_box_Nb*_box_Nc);
00091 
00092     int a1,a2,b1,b2,c1,c2;
00093     
00094     a1 = b1 = c1 = -1;
00095     a2 = b2 = c2 = 1;
00096 
00097     if(_box_Na < 3) a2 = 0;
00098     if(_box_Nb < 3) b2 = 0;
00099     if(_box_Nc < 3) c2 = 0;
00100 
00101     if(_box_Na < 2) a1 = 0;
00102     if(_box_Nb < 2) b1 = 0;
00103     if(_box_Nc < 2) c1 = 0;
00104 
00105     // wow, setting up the neighbours is an ugly for construct!
00106     // loop from N..2*N to avoid if and only use %
00107     for(int a=_box_Na; a<2*_box_Na; ++a)
00108         for(int b=_box_Nb; b<2*_box_Nb; ++b)
00109             for(int c=_box_Nc; c<2*_box_Nc; ++c) {
00110                 cell_t &cell = getCell(a%_box_Na, b%_box_Nb, c%_box_Nc);
00111                 for(int aa=a+a1; aa<=a+a2; ++aa)
00112                     for(int bb=b+b1; bb<=b+b2; ++bb)
00113                         for(int cc=c+c1; cc<=c+c2; ++cc) {
00114                             cell_t *c = &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc);
00115                             if(c == &cell) continue; // ignore self
00116                             cell._neighbours.push_back(
00117                                 &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc)
00118                             );
00119                         }
00120             }
00121 }

00123 NBListGrid::cell_t &NBListGrid::getCell(const vec &r)
00124 {
00125     int a = (int)floor(r*_norm_a);
00126     int b = (int)floor(r*_norm_b);
00127     int c = (int)floor(r*_norm_c);
00128 
00129     if(a<0) a = _box_Na + a%_box_Na;
00130     a %= _box_Na;
00131 
00132     if(b<0)
00133       b = _box_Nb + b%_box_Nb;
00134     b %= _box_Nb;
00135 
00136     if(c<0) c = _box_Nc + c%_box_Nc;
00137     c %= _box_Nc;
00138 
00139     return getCell(a,b,c);
