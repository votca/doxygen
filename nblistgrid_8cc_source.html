36  getCell((*iter)->getPos())._beads.push_back(*iter);
37 
38 
39  for(iter = list2.begin(); iter != list2.end(); ++iter) {

61 void NBListGrid::InitializeGrid(const matrix &box)
62 {
63  _box_a = box.getCol(0);
64  _box_b = box.getCol(1);
65  _box_c = box.getCol(2);
66 
67  // create plane normals
68  _norm_a = _box_b ^ _box_c;
69  _norm_b = _box_c ^ _box_a;
70  _norm_c = _box_a ^ _box_b;
71 
72  _norm_a.normalize();
73  _norm_b.normalize();
74  _norm_c.normalize();
75 
76  double la = _box_a * _norm_a;
77  double lb = _box_b * _norm_b;
78  double lc = _box_c * _norm_c;
79 
80  // calculate grid size, each grid has to be at least size of cut-off
81  _box_Na = max((int)(fabs(la/_cutoff)), 1);
82  _box_Nb = max((int)(fabs(lb/_cutoff)), 1);
83  _box_Nc = max((int)(fabs(lc/_cutoff)), 1);
84 
85  _norm_a = _norm_a / (_box_a*_norm_a)*(double)_box_Na;
86  _norm_b = _norm_b / (_box_b*_norm_b)*(double)_box_Nb;
87  _norm_c = _norm_c / (_box_c*_norm_c)*(double)_box_Nc;
88 
89  //cout << "grid size: " << _box_Na << "x" << _box_Nb << "x" << _box_Nc << endl;
90  _grid.resize(_box_Na*_box_Nb*_box_Nc);
91 
92  int a1,a2,b1,b2,c1,c2;
93 
94  a1 = b1 = c1 = -1;
95  a2 = b2 = c2 = 1;
96 
97  if(_box_Na < 3) a2 = 0;
98  if(_box_Nb < 3) b2 = 0;
99  if(_box_Nc < 3) c2 = 0;
100 
101  if(_box_Na < 2) a1 = 0;
102  if(_box_Nb < 2) b1 = 0;
103  if(_box_Nc < 2) c1 = 0;
104 
105  // wow, setting up the neighbours is an ugly for construct!
106  // loop from N..2*N to avoid if and only use %
107  for(int a=_box_Na; a<2*_box_Na; ++a)
108  for(int b=_box_Nb; b<2*_box_Nb; ++b)
109  for(int c=_box_Nc; c<2*_box_Nc; ++c) {
110  cell_t &cell = getCell(a%_box_Na, b%_box_Nb, c%_box_Nc);
111  for(int aa=a+a1; aa<=a+a2; ++aa)
112  for(int bb=b+b1; bb<=b+b2; ++bb)
113  for(int cc=c+c1; cc<=c+c2; ++cc) {
114  cell_t *c = &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc);
115  if(c == &cell) continue; // ignore self
116  cell._neighbours.push_back(
117  &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc)
118  );
119  }
120  }
121 }

123 NBListGrid::cell_t &NBListGrid::getCell(const vec &r)
124 {
125  int a = (int)floor(r*_norm_a);
126  int b = (int)floor(r*_norm_b);
127  int c = (int)floor(r*_norm_c);
128 
129  if(a<0) a = _box_Na + a%_box_Na;
130  a %= _box_Na;
131 
132  if(b<0)
133  b = _box_Nb + b%_box_Nb;
134  b %= _box_Nb;
135 
136  if(c<0) c = _box_Nc + c%_box_Nc;
137  c %= _box_Nc;
138 
139  return getCell(a,b,c);
140  }

151 void NBListGrid::TestCell(NBListGrid::cell_t &cell, Bead *bead)
152 {
153  BeadList::iterator iter;
154  vec u = bead->getPos();
155 
156  for(iter=cell._beads.begin(); iter!=cell._beads.end(); ++iter) {
157 
158  vec v = (*iter)->getPos();
159  vec r = _top->BCShortestConnection(v, u);
160  double d = abs(r);
161  if(d < _cutoff){
162  if(_do_exclusions)
163  if(_top->getExclusions().IsExcluded((*iter), bead)) {
164  continue;
165  }
166  if((*_match_function)(*iter, bead, r, d))
167  if(!FindPair(*iter, bead))
168  AddPair(_pair_creator(*iter, bead, r));
169  }
170  }
171 }
