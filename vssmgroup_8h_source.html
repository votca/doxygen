56 double Vssmgroup::Timestep(votca::tools::Random2 *RandomVariable){
57 
58  double timestep;

105 void Vssmgroup::Perform_one_step_in_device(Events* events, Graph* graph, State* state, Globaleventinfo* globevent, votca::tools::Random2 *RandomVariable){
106 
107  double randn = RandomVariable->rand_uniform();
108 
109  long event_ID;
110  Event* chosenevent;
111 
112  if(randn<el_probsum/tot_probsum) { // electron event
113  if(randn< el_non_inject_probsum/tot_probsum) { // non injection event
114  randn *= el_non_inject_probsum;
115  event_ID = events->El_non_injection_rates->search(randn);
116  chosenevent = events->El_non_injection_events[event_ID];
117  }
118  else { // injection event
119  randn -= el_non_inject_probsum/tot_probsum;
120  randn *= el_inject_probsum;
121  event_ID = events->El_injection_rates->search(randn);
122  chosenevent = events->El_injection_events[event_ID];
123  }
124  }
125  else { //Hole event
126  randn -= el_probsum/tot_probsum;
127  if(randn< ho_non_inject_probsum/tot_probsum) { // non injection event
128  randn *= ho_non_inject_probsum;
129  event_ID = events->Ho_non_injection_rates->search(randn);
130  chosenevent = events->Ho_non_injection_events[event_ID];
131  }
132  else { // injection event
133  randn -= ho_non_inject_probsum/tot_probsum;
134  randn *= ho_inject_probsum;
135  event_ID = events->Ho_injection_rates->search(randn);
136  chosenevent = events->Ho_injection_events[event_ID];
137  }
138  }
139  events->On_execute(chosenevent, graph, state, globevent);
140 }
