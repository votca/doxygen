104 bool GW::WriteInputFile( vector<Segment* > segments, Orbitals* orbitals_guess )

108  vector< Segment* >::iterator sit;

110  //int qmatoms = 0;

116

119  string basis_name( _gwbasis );
120  gwbs.LoadBasisSet( basis_name );
121  LOG(logDEBUG,*_pLog) << "Loaded Basis Set " << basis_name << flush;

123  /* if ( _options.size() ) _com_file << _options << endl ; */

125  _gw_file << "start_dft votca " << endl;
126  _gw_file << "lattice_structure sc " << endl;
127  _gw_file << "lattice_constant_angstrom 1500.00 " << endl;
128  _gw_file << "units angstrom " << endl;
129 
130  for (sit = segments.begin() ; sit != segments.end(); ++sit) {
131 
132  _atoms = (*sit)-> Atoms();
133 
134  for (ait = _atoms.begin(); ait < _atoms.end(); ++ait) {
135 
136  if ((*ait)->HasQMPart() == false) { continue; }
137 
138  vec pos = (*ait)->getQMPos();
139  string name = (*ait)->getElement();

141  //fprintf(out, "%2s %4.7f %4.7f %4.7f \n"

149  _gw_file << "basis_gwa ";

151  for (Element::ShellIterator its = element->firstShell(); its != element->lastShell(); its++) {
152 
153  Shell* shell = (*its);
154  // shell type, number primitives, scale factor

159  //_com_file << gaussian->decay << " " << gaussian->contraction << endl;

161  }

163  _gw_file << endl;
164 
165  }
166  }
167 
168  // some more defaults
169  _gw_file << "dft_truncation_radius 500.0 " << endl;
170  _gw_file << "screening rpa_nosumrule_eps0 (0.d0, 1.d0) 1.001 " << endl;
171  _gw_file << "tolerance_threecenter 1.d-9 " << endl;
172  _gw_file << "scissors_operator_rigid_ryd .3" << endl;
173 
174  // convert _rpamax if needed and write
175  int _vbm = orbitals_guess->getNumberOfElectrons();
176  if ( _ranges == "default" ){
177  _rpamax = orbitals_guess->getNumberOfLevels();
178  } else if ( _ranges == "factor" ) {
179  _rpamax = _rpamaxfactor * _vbm;
180  }
181  _gw_file << "rpa_band_summation 1 " << _rpamax << endl;
182 
183  // VBM
184  _gw_file << "mbpt_vbm " << _vbm << endl;
185 
186  // convert _qpmin and _qpmax if needed, and write
187  if ( _ranges == "default" ){
188  _qpmin = 1;
189  _qpmax = 2 * _vbm;
190  } else if ( _ranges == "factor" ) {
191  _qpmin = _vbm - int( _qpminfactor * _vbm );
192  _qpmax = _vbm + int( _qpmaxfactor * _vbm );

196  // convert _bsemin and _bsemax if needed, and write

198  _bsemin = 1;
199  _bsemax = 2 * _vbm;
200  } else if ( _ranges == "factor" ) {
201  _bsemin = _vbm - int( _bseminfactor * _vbm );
202  _bsemax = _vbm + int( _bsemaxfactor * _vbm );
203  }
204  _gw_file << "bse_bands " << _bsemin << " " << _bsemax << endl;
205  _gw_file << "bse_momentum_operator " << endl;
206  if ( _options.size() ) _gw_file << _options << endl ;
207  _gw_file.close();
208 
209  return true;

306 bool GW::CheckLogFile() {
307 
308  // check if the log file exists
309  char ch;
310  ifstream _input_file((_run_dir + "/" + _log_file_name).c_str());
311 
312  if (_input_file.fail()) {
313  LOG(logERROR,*_pLog) << "ISOGWA LOG is not found" << flush;
314  return false;
315  };
316 
317  _input_file.seekg(0,ios_base::end); // go to the EOF
318 
319 
320  // get empty lines and end of lines out of the way
321  do {
322  _input_file.seekg(-2,ios_base::cur);
323  _input_file.get(ch);
324  //cout << "\nChar: " << ch << endl;
325  } while ( ch == '\n' || ch == ' ' || (int)_input_file.tellg() == -1 );
326 
327  // get the beginning of the line or the file
328  do {
329  _input_file.seekg(-2,ios_base::cur);
330  _input_file.get(ch);
331  //cout << "\nNext Char: " << ch << " TELL G " << (int)_input_file.tellg() << endl;
332  } while ( ch != '\n' || (int)_input_file.tellg() == -1 );
333 
334  string _line;
335  getline(_input_file,_line); // Read the current line
336  //cout << "\nResult: " << _line << '\n'; // Display it
337  _input_file.close();
338 
339  std::string::size_type self_energy_pos = _line.find("INF total program runtime");
340  if (self_energy_pos == std::string::npos) {
341  LOG(logERROR,*_pLog) << "ISOGWA is incomplete" << flush;
342  return false;
343  } else {
344  //LOG(logDEBUG,*_pLog) << "Gaussian LOG is complete" << flush;
345  return true;
346  }
347 
348  _input_file.close();
349  return true;
