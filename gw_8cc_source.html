49 double GW::CalcHomoLumoShift(Eigen::VectorXd frequencies) const {
50  double DFTgap = _dft_energies(_opt.homo + 1) - _dft_energies(_opt.homo);
51  double QPgap = frequencies(_opt.homo + 1 - _opt.qpmin) -
52  frequencies(_opt.homo - _opt.qpmin);
53  return QPgap - DFTgap;
54 }

139 void GW::CalculateGWPerturbation() {
140 
141  _Sigma_x = (1 - _opt.ScaHFX) * _sigma->CalcExchangeMatrix();
142  XTP_LOG(Log::error, _log)
143  << TimeStamp() << " Calculated Hartree exchange contribution"
144  << std::flush;
145  // dftenergies has size aobasissize
146  // rpaenergies/Mmn have size rpatotal
147  // gwaenergies/frequencies have size qptotal
148  // homo index is relative to dft_energies
149  Eigen::VectorXd dft_shifted_energies = ScissorShift_DFTlevel(_dft_energies);
150  Eigen::VectorXd rpa_energies =
151  dft_shifted_energies.segment(_opt.rpamin, _opt.rpamax - _opt.rpamin + 1);
152  _rpa.setRPAInputEnergies(rpa_energies);
153  Eigen::VectorXd frequencies =
154  dft_shifted_energies.segment(_opt.qpmin, _qptotal);
155  for (Index i_gw = 0; i_gw < _opt.gw_sc_max_iterations; ++i_gw) {
156 
157  if (i_gw % _opt.reset_3c == 0 && i_gw != 0) {
158  _Mmn.Rebuild();
159  XTP_LOG(Log::info, _log)
160  << TimeStamp() << " Rebuilding 3c integrals" << std::flush;
161  }
162  _sigma->PrepareScreening();
163  XTP_LOG(Log::info, _log)
164  << TimeStamp() << " Calculated screening via RPA" << std::flush;
165  if (_opt.qp_solver == "grid") {
166  frequencies = SolveQP_Grid(frequencies);
167  XTP_LOG(Log::error, _log)
168  << TimeStamp() << " Solved QP equation on QP grid" << std::flush;
169  } else if (_opt.qp_solver == "fixedpoint") {
170  frequencies = SolveQP_FixedPoint(frequencies);
171  XTP_LOG(Log::error, _log)
172  << TimeStamp() << " Solved QP equation self-consistently"
173  << std::flush;
174  }
175  _gwa_energies = frequencies;
176  _Sigma_c.diagonal() = _sigma->CalcCorrelationDiag(_gwa_energies);
177  XTP_LOG(Log::error, _log)
178  << TimeStamp() << " Calculated correlation diagonal" << std::flush;
179  Eigen::VectorXd rpa_energies_old = _rpa.getRPAInputEnergies();
180  _rpa.UpdateRPAInputEnergies(_dft_energies, frequencies, _opt.qpmin);
181  XTP_LOG(Log::info, _log)
182  << TimeStamp() << " GW_Iteration:" << i_gw
183  << " Shift[Hrt]:" << CalcHomoLumoShift(_gwa_energies) << std::flush;
184  if (Converged(_rpa.getRPAInputEnergies(), rpa_energies_old,
185  _opt.gw_sc_limit)) {
186  XTP_LOG(Log::error, _log) << TimeStamp() << " Converged after "
187  << i_gw + 1 << " GW iterations." << std::flush;
188  break;
189  } else if (i_gw == _opt.gw_sc_max_iterations - 1 &&
190  _opt.gw_sc_max_iterations > 1) {
191  XTP_LOG(Log::error, _log)
192  << TimeStamp()
193  << " WARNING! GW-self-consistency cycle not converged after "
194  << _opt.gw_sc_max_iterations << " iterations." << std::flush;
195  XTP_LOG(Log::error, _log)
196  << TimeStamp() << " Run continues. Inspect results carefully!"
197  << std::flush;
198  break;
199  } else {
200  double alpha = 0.0;
201  rpa_energies = (1 - alpha) * rpa_energies + alpha * rpa_energies_old;
202  }
203  }
204 
205  PrintGWA_Energies();
206 }

208 Eigen::VectorXd GW::SolveQP_Grid(const Eigen::VectorXd& frequencies) const {
209  const Index qptotal = _opt.qpmax - _opt.qpmin + 1;
210  const double range =
211  _opt.qp_grid_spacing * (double)(_opt.qp_grid_steps - 1) / 2.0;
212  const Eigen::VectorXd intercept =
213  _dft_energies.segment(_opt.qpmin, _qptotal) + _Sigma_x.diagonal() -
214  _vxc.diagonal();
215  Eigen::VectorXd frequencies_new = frequencies;
216 #pragma omp parallel for schedule(dynamic)
217  for (Index gw_level = 0; gw_level < qptotal; ++gw_level) {
218  const double frequency0 = frequencies[gw_level];
219  const double intercept0 = intercept[gw_level];
220  double freq_prev = frequency0 - range;
221  double sigc_prev = _sigma->CalcCorrelationDiagElement(gw_level, freq_prev);
222  double targ_prev = sigc_prev + intercept0 - freq_prev;
223  double qp_energy = 0.0;
224  double pole_weight_max = -1.0;
225  for (Index i_node = 1; i_node < _opt.qp_grid_steps; ++i_node) {
226  double freq = freq_prev + _opt.qp_grid_spacing;
227  double sigc = _sigma->CalcCorrelationDiagElement(gw_level, freq);
228  double targ = sigc + intercept0 - freq;
229  if (targ_prev * targ < 0.0) { // Sign change
230  double dsigc_dfreq = (sigc - sigc_prev) / _opt.qp_grid_spacing;
231  double dtarg_dfreq = (targ - targ_prev) / _opt.qp_grid_spacing;
232  // Calculate fixed-point estimate of the pole (=root)
233  double pole = freq_prev - targ_prev / dtarg_dfreq;
234  // Calculate pole weight Z \in (0, 1)
235  double pole_weight = 1.0 / (1.0 - dsigc_dfreq);
236  if (pole_weight >= 1e-5 && pole_weight > pole_weight_max) {
237  qp_energy = pole;
238  pole_weight_max = pole_weight;
239  }
240  }
241  freq_prev = freq;
242  sigc_prev = sigc;
243  targ_prev = targ;
244  }
245  if (pole_weight_max >= 0.0) {
246  frequencies_new[gw_level] = qp_energy;
247  }
248  }
249  return frequencies_new;
250 }

252 Eigen::VectorXd GW::SolveQP_FixedPoint(
253  const Eigen::VectorXd& frequencies) const {
254  const Index qptotal = _opt.qpmax - _opt.qpmin + 1;
255  const Eigen::VectorXd intercept =
256  _dft_energies.segment(_opt.qpmin, _qptotal) + _Sigma_x.diagonal() -
257  _vxc.diagonal();
258  Eigen::VectorXd frequencies_new = frequencies;
259 #pragma omp parallel for schedule(dynamic)
260  for (Index gw_level = 0; gw_level < qptotal; ++gw_level) {
261  const double intercept0 = intercept[gw_level];
262  double freq = frequencies[gw_level];
263  for (Index g_iter = 0; g_iter < _opt.g_sc_max_iterations; ++g_iter) {
264  double freq_prev = freq;
265  double sigc = _sigma->CalcCorrelationDiagElement(gw_level, freq);
266  // Update frequency via fixed-point iteration
267  freq = sigc + intercept0;
268  if (std::abs(freq - freq_prev) < _opt.g_sc_limit) {
269  break; // Converged
270  } else if (_opt.g_sc_max_iterations != 0 &&
271  g_iter == _opt.g_sc_max_iterations - 1) {
272  break; // Not converged
273  }
274  }
275  frequencies_new[gw_level] = freq;
276  }
277  return frequencies_new;
278 }

280 bool GW::Converged(const Eigen::VectorXd& e1, const Eigen::VectorXd& e2,
281  double epsilon) const {
282  Index state = 0;
283  bool energies_converged = true;
284  double diff_max = (e1 - e2).cwiseAbs().maxCoeff(&state);
285  if (diff_max > epsilon) {
286  energies_converged = false;
287  }
288  XTP_LOG(Log::info, _log) << TimeStamp() << " E_diff max=" << diff_max
289  << " StateNo:" << state << std::flush;
290  return energies_converged;
291 }
