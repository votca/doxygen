44 double GW::CalcHomoLumoShift()const{
45  double DFTgap = _dft_energies(_opt.homo + 1) - _dft_energies(_opt.homo);
46  double QPgap = _gwa_energies(_opt.homo + 1-_opt.qpmin) - _gwa_energies(_opt.homo-_opt.qpmin);
47  return QPgap - DFTgap;

137 bool GW::Converged(const Eigen::VectorXd& e1, const Eigen::VectorXd& e2, double epsilon)const {
138  int state = 0;
139  bool energies_converged = true;
140  double diff_max = (e1 - e2).cwiseAbs().maxCoeff(&state);
141  if (diff_max > epsilon) {
142  energies_converged = false;
143  }
144  if (tools::globals::verbose) {
145  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " E_diff max=" << diff_max << " StateNo:" << state << std::flush;
146  }
147  return energies_converged;

150 Eigen::VectorXd GW::CalculateExcitationFreq(Eigen::VectorXd frequencies){
151  for (int i_freq = 0; i_freq < _opt.g_sc_max_iterations; ++i_freq) {
152 
153  _Sigma_c.diagonal()= _sigma->CalcCorrelationDiag(frequencies);
154  _gwa_energies = CalcDiagonalEnergies();
155 
156  if(tools::globals::verbose){
157  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() <<" G_Iteration:" <<i_freq << " Shift[Hrt]:" << CalcHomoLumoShift() << std::flush;
158  }
159  if (Converged(_gwa_energies, frequencies, _opt.g_sc_limit)) {
160  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " Converged after " <<i_freq + 1 << " G iterations." << std::flush;
161  break;
162  } else if (i_freq == _opt.g_sc_max_iterations - 1 && _opt.g_sc_max_iterations>1) {
163  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " G-self-consistency cycle not converged after "
164  << _opt.g_sc_max_iterations << " iterations." << std::flush;
165  break;
166  } else {
167  double alpha = 0.0;
168  frequencies = (1 - alpha) * _gwa_energies + alpha*frequencies;
169  }
170 
171  }
172  return frequencies;
173 }

175 void GW::CalculateGWPerturbation() {
176 
177  _Sigma_x = (1-_opt.ScaHFX)*_sigma->CalcExchange();
178  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp()
179  << " Calculated Hartree exchange contribution " << std::flush;
180 
181  Eigen::VectorXd rpa_energies = ScissorShift_DFTlevel(_dft_energies);
182  _rpa.setRPAInputEnergies(rpa_energies);
183  Eigen::VectorXd frequencies = rpa_energies.segment(_opt.qpmin, _qptotal);
184 
185 
186  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp()
187  << " Prepared RPA " << std::flush;
188  for (int i_gw=0; i_gw < _opt.gw_sc_max_iterations; ++i_gw) {
189 
190  if(i_gw%_opt.reset_3c==0 && i_gw!=0){
191  _Mmn.Rebuild();
192  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() <<" Rebuilding 3c integrals" << std::flush;
193  }
194  _sigma->PrepareScreening();
195  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp()
196  << " Calculated screening via RPA " << std::flush;
197  frequencies=CalculateExcitationFreq(frequencies);
198  CTP_LOG(ctp::logDEBUG, _log)
199  << ctp::TimeStamp() << " Calculated diagonal part of Sigma " << std::flush;
200  Eigen::VectorXd rpa_energies_old=_rpa.getRPAInputEnergies();
201  _rpa.UpdateRPAInputEnergies(_dft_energies,frequencies,_opt.qpmin);
202 
203  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() <<" GW_Iteration:" <<i_gw << " Shift[Hrt]:" << CalcHomoLumoShift() << std::flush;
204  if(Converged(_rpa.getRPAInputEnergies(), rpa_energies_old, _opt.gw_sc_limit)){
205  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " Converged after " <<i_gw + 1 << " GW iterations." << std::flush;
206  break;
207  } else if (i_gw == _opt.gw_sc_max_iterations - 1 && _opt.gw_sc_max_iterations>1) {
208  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " WARNING! GW-self-consistency cycle not converged after "
209  << _opt.gw_sc_max_iterations << " iterations." << std::flush;
210  CTP_LOG(ctp::logDEBUG, _log)<< ctp::TimeStamp()
211  << " Run continues. Inspect results carefully!" << std::flush;
212  break;
213  } else {
214  double alpha = 0.0;
215  rpa_energies = (1 - alpha) * rpa_energies + alpha*rpa_energies_old;
216  }
217  }
218 
219  PrintGWA_Energies();
220 }
