40  _rpa.configure(_opt.homo, _opt.rpamin, _opt.rpamax);

63 double GW::CalcHomoLumoShift(Eigen::VectorXd frequencies) const {
64  double DFTgap = _dft_energies(_opt.homo + 1) - _dft_energies(_opt.homo);
65  double QPgap = frequencies(_opt.homo + 1 - _opt.qpmin) -
66  frequencies(_opt.homo - _opt.qpmin);
67  return QPgap - DFTgap;

151 void GW::CalculateGWPerturbation() {
152 
153  _Sigma_x = (1 - _opt.ScaHFX) * _sigma->CalcExchangeMatrix();
154  XTP_LOG(Log::error, _log)
155  << TimeStamp() << " Calculated Hartree exchange contribution"
156  << std::flush;
157  // dftenergies has size aobasissize
158  // rpaenergies/Mmn have size rpatotal
159  // gwaenergies/frequencies have size qptotal
160  // homo index is relative to dft_energies
161  XTP_LOG(Log::error, _log)
162  << TimeStamp() << " Scissor shifting DFT energies by: " << _opt.shift
163  << " Hrt" << std::flush;
164  Eigen::VectorXd dft_shifted_energies = ScissorShift_DFTlevel(_dft_energies);
165  _rpa.setRPAInputEnergies(
166  dft_shifted_energies.segment(_opt.rpamin, _opt.rpamax - _opt.rpamin + 1));
167  Eigen::VectorXd frequencies =
168  dft_shifted_energies.segment(_opt.qpmin, _qptotal);
169 
170  Anderson _mixing;
171  _mixing.Configure(_opt.gw_mixing_order, _opt.gw_mixing_alpha);
172 
173  for (Index i_gw = 0; i_gw < _opt.gw_sc_max_iterations; ++i_gw) {
174 
175  if (i_gw % _opt.reset_3c == 0 && i_gw != 0) {
176  _Mmn.Rebuild();
177  XTP_LOG(Log::info, _log)
178  << TimeStamp() << " Rebuilding 3c integrals" << std::flush;
179  }
180  _sigma->PrepareScreening();
181  XTP_LOG(Log::info, _log)
182  << TimeStamp() << " Calculated screening via RPA" << std::flush;
183  XTP_LOG(Log::info, _log)
184  << TimeStamp() << " Solving QP equations " << std::flush;
185  if (_opt.gw_mixing_order > 0 && i_gw > 0) {
186  _mixing.UpdateInput(frequencies);
187  }
188 
189  frequencies = SolveQP(frequencies);
190 
191  if (_opt.gw_sc_max_iterations > 1) {
192  Eigen::VectorXd rpa_energies_old = _rpa.getRPAInputEnergies();
193  if (_opt.gw_mixing_order > 0 && i_gw > 0) {
194  if (_opt.gw_mixing_order == 1) {
195  XTP_LOG(Log::debug, _log)
196  << "GWSC using linear mixing with alpha: " << _opt.gw_mixing_alpha
197  << std::flush;
198  } else {
199  XTP_LOG(Log::debug, _log)
200  << "GWSC using Anderson mixing with history "
201  << _opt.gw_mixing_order << ", alpha: " << _opt.gw_mixing_alpha
202  << std::flush;
203  }
204  _mixing.UpdateOutput(frequencies);
205  Eigen::VectorXd mixed_frequencies = _mixing.MixHistory();
206  _rpa.UpdateRPAInputEnergies(_dft_energies, mixed_frequencies,
207  _opt.qpmin);
208  frequencies = mixed_frequencies;
209 
210  } else {
211  XTP_LOG(Log::debug, _log) << "GWSC using plain update " << std::flush;
212  _rpa.UpdateRPAInputEnergies(_dft_energies, frequencies, _opt.qpmin);
213  }
214 
215  for (int i = 0; i < frequencies.size(); i++) {
216  XTP_LOG(Log::debug, _log)
217  << "... GWSC iter " << i_gw << " state " << i << " "
218  << std::setprecision(9) << frequencies(i) << std::flush;
219  }
220 
221  XTP_LOG(Log::info, _log)
222  << TimeStamp() << " GW_Iteration:" << i_gw
223  << " Shift[Hrt]:" << CalcHomoLumoShift(frequencies) << std::flush;
224  if (Converged(_rpa.getRPAInputEnergies(), rpa_energies_old,
225  _opt.gw_sc_limit)) {
226  XTP_LOG(Log::info, _log) << TimeStamp() << " Converged after "
227  << i_gw + 1 << " GW iterations." << std::flush;
228  break;
229  } else if (i_gw == _opt.gw_sc_max_iterations - 1) {
230  XTP_LOG(Log::error, _log)
231  << TimeStamp()
232  << " WARNING! GW-self-consistency cycle not converged after "
233  << _opt.gw_sc_max_iterations << " iterations." << std::flush;
234  XTP_LOG(Log::error, _log)
235  << TimeStamp() << " Run continues. Inspect results carefully!"
236  << std::flush;
237  break;
238  }
239  }
240  }
241  _Sigma_c.diagonal() = _sigma->CalcCorrelationDiag(frequencies);
242  PrintGWA_Energies();
243 }

250 Eigen::VectorXd GW::SolveQP(const Eigen::VectorXd& frequencies) const {
251  const Eigen::VectorXd intercepts =
252  _dft_energies.segment(_opt.qpmin, _qptotal) + _Sigma_x.diagonal() -
253  _vxc.diagonal();
254  Eigen::VectorXd frequencies_new = frequencies;
255  Eigen::Array<bool, Eigen::Dynamic, 1> converged =
256  Eigen::Array<bool, Eigen::Dynamic, 1>::Zero(_qptotal);
257 #pragma omp parallel for schedule(dynamic)
258  for (Index gw_level = 0; gw_level < _qptotal; ++gw_level) {
259  double initial_f = frequencies[gw_level];
260  double intercept = intercepts[gw_level];
261  boost::optional<double> newf;
262  if (_opt.qp_solver == "fixedpoint") {
263  newf = SolveQP_FixedPoint(intercept, initial_f, gw_level);
264  }
265  if (newf) {
266  frequencies_new[gw_level] = newf.value();
267  converged[gw_level] = true;
268  } else {
269  newf = SolveQP_Grid(intercept, initial_f, gw_level);
270  if (newf) {
271  frequencies_new[gw_level] = newf.value();
272  converged[gw_level] = true;
273  } else {
274  newf = SolveQP_Linearisation(intercept, initial_f, gw_level);
275  if (newf) {
276  frequencies_new[gw_level] = newf.value();
277  }
278  }
279  }
280  }
281 
282  if (!converged.all()) {
283  std::vector<Index> states;
284  for (Index s = 0; s < converged.size(); s++) {
285  if (!converged[s]) {
286  states.push_back(s);
287  }
288  }
289  IndexParser rp;
290  XTP_LOG(Log::error, _log) << TimeStamp() << " Not converged PQP states are:"
291  << rp.CreateIndexString(states) << std::flush;
292  XTP_LOG(Log::error, _log)
293  << TimeStamp() << " Increase the grid search interval" << std::flush;
294  }
295  return frequencies_new;
296 }

298 boost::optional<double> GW::SolveQP_Linearisation(double intercept0,
299  double frequency0,
300  Index gw_level) const {
301  boost::optional<double> newf = boost::none;
302 
303  double sigma = _sigma->CalcCorrelationDiagElement(gw_level, frequency0);
304  double dsigma_domega =
305  _sigma->CalcCorrelationDiagElementDerivative(gw_level, frequency0);
306  double Z = 1.0 - dsigma_domega;
307  if (std::abs(Z) > 1e-9) {
308  newf = frequency0 + (intercept0 - frequency0 + sigma) / Z;
309  }
310  return newf;
311 }

313 boost::optional<double> GW::SolveQP_Grid(double intercept0, double frequency0,
314  Index gw_level) const {
315  std::vector<std::pair<double, double>> roots;
316  const double range =
317  _opt.qp_grid_spacing * double(_opt.qp_grid_steps - 1) / 2.0;
318  boost::optional<double> newf = boost::none;
319  double freq_prev = frequency0 - range;
320  QPFunc fqp(gw_level, *_sigma.get(), intercept0);
321  double targ_prev = fqp.value(freq_prev);
322  double qp_energy = 0.0;
323  double gradient_max = std::numeric_limits<double>::max();
324  bool pole_found = false;
325  for (Index i_node = 1; i_node < _opt.qp_grid_steps; ++i_node) {
326  double freq = freq_prev + _opt.qp_grid_spacing;
327  double targ = fqp.value(freq);
328  if (targ_prev * targ < 0.0) { // Sign change
329  double f = SolveQP_Bisection(freq_prev, targ_prev, freq, targ, fqp);
330  double gradient = fqp.deriv(f);
331  double qp_weight = -1.0 / gradient;
332  roots.push_back(std::make_pair(f, qp_weight));
333  if (std::abs(gradient) < gradient_max) {
334  gradient_max = std::abs(gradient);
335  qp_energy = f;
336  pole_found = true;
337  }
338  }
339  freq_prev = freq;
340  targ_prev = targ;
341  }
342  if (Log::current_level > Log::error) {
343 #pragma omp critical
344  {
345  if (!pole_found) {
346  XTP_LOG(Log::info, _log)
347  << " No roots found for qplevel:" << gw_level << std::flush;
348  } else {
349  XTP_LOG(Log::info, _log) << " Roots found for qplevel:" << gw_level
350  << " (qpenergy:qpweight)\n\t\t";
351  for (auto& root : roots) {
352  XTP_LOG(Log::info, _log) << std::setprecision(5) << root.first << ":"
353  << root.second << " ";
354  }
355  XTP_LOG(Log::info, _log) << "Root chosen " << qp_energy << std::flush;
356  }
357  }
358  }
359 
360  if (pole_found) {
361  newf = qp_energy;
362  }
363  return newf;
364 }

366 boost::optional<double> GW::SolveQP_FixedPoint(double intercept0,
367  double frequency0,
368  Index gw_level) const {
369  boost::optional<double> newf = boost::none;
370  QPFunc f(gw_level, *_sigma.get(), intercept0);
371  NewtonRapson<QPFunc> newton = NewtonRapson<QPFunc>(
372  _opt.g_sc_max_iterations, _opt.g_sc_limit, _opt.qp_solver_alpha);
373  double freq_new = newton.FindRoot(f, frequency0);
374  if (newton.getInfo() == NewtonRapson<QPFunc>::success) {
375  newf = freq_new;
376  }
377  return newf;
378 }

380 // https://en.wikipedia.org/wiki/Bisection_method
381 double GW::SolveQP_Bisection(double lowerbound, double f_lowerbound,
382  double upperbound, double f_upperbound,
383  const QPFunc& f) const {
384 
385  if (f_lowerbound * f_upperbound > 0) {
386  throw std::runtime_error(
387  "Bisection needs a postive and negative function value");
388  }
389  double zero = 0.0;
390  while (true) {
391  double c = 0.5 * (lowerbound + upperbound);
392  if (std::abs(upperbound - lowerbound) < _opt.g_sc_limit) {
393  zero = c;
394  break;
395  }
396  double y_c = f.value(c);
397  if (std::abs(y_c) < _opt.g_sc_limit) {
398  zero = c;
399  break;
400  }
401  if (y_c * f_lowerbound > 0) {
402  lowerbound = c;
403  f_lowerbound = y_c;
404  } else {
405  upperbound = c;
406  f_upperbound = y_c;
407  }
408  }
409  return zero;
410 }

412 bool GW::Converged(const Eigen::VectorXd& e1, const Eigen::VectorXd& e2,
413  double epsilon) const {
414  Index state = 0;
415  bool energies_converged = true;

422  return energies_converged;

431 void GW::PlotSigma(std::string filename, Index steps, double spacing,
432  std::string states) const {
433 
434  Eigen::VectorXd frequencies =
435  _rpa.getRPAInputEnergies().segment(_opt.qpmin - _opt.rpamin, _qptotal);
436 
437  std::vector<Index> state_inds;
438  IndexParser rp;
439  std::vector<Index> parsed_states = rp.CreateIndexVector(states);
440  for (Index gw_level : parsed_states) {
441  if (gw_level >= _opt.qpmin && gw_level <= _opt.qpmax) {
442  state_inds.push_back(gw_level);
443  }
444  }
445  XTP_LOG(Log::error, _log)
446  << TimeStamp() << " PQP(omega) written to '" << filename
447  << "' for states " << rp.CreateIndexString(state_inds) << std::flush;
448 
449  const Index num_states = state_inds.size();
450 
451  const Eigen::VectorXd intercept =
452  _dft_energies.segment(_opt.qpmin, _qptotal) + _Sigma_x.diagonal() -
453  _vxc.diagonal();
454  Eigen::MatrixXd mat = Eigen::MatrixXd::Zero(steps, 2 * num_states);
455 #pragma omp parallel for schedule(dynamic)
456  for (Index grid_point = 0; grid_point < steps; grid_point++) {
457  const double offset =
458  ((double)grid_point - ((double)(steps - 1) / 2.0)) * spacing;
459  for (Index i = 0; i < num_states; i++) {
460  const Index gw_level = state_inds[i];
461  const double omega = frequencies(gw_level) + offset;
462  double sigma = _sigma->CalcCorrelationDiagElement(gw_level, omega);
463  mat(grid_point, 2 * i) = omega;
464  mat(grid_point, 2 * i + 1) = sigma + intercept[gw_level];
465  }
466  }
467 
468  std::ofstream out;
469  out.open(filename);
470  for (Index i = 0; i < num_states; i++) {
471  const Index gw_level = state_inds[i];
472  out << boost::format("#%1$somega_%2$d\tE_QP(omega)_%2$d") %
473  (i == 0 ? "" : "\t") % gw_level;
474  }
475  out << std::endl;
476  boost::format numFormat("%+1.6f");
477  Eigen::IOFormat matFormat(Eigen::StreamPrecision, 0, "\t", "\n");
478  out << numFormat % mat.format(matFormat) << std::endl;
479  out.close();
480 }
