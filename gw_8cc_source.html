56 double GW::CalcHomoLumoShift(Eigen::VectorXd frequencies) const {
57  double DFTgap = dft_energies_(opt_.homo + 1) - dft_energies_(opt_.homo);
58  double QPgap = frequencies(opt_.homo + 1 - opt_.qpmin) -
59  frequencies(opt_.homo - opt_.qpmin);
60  return QPgap - DFTgap;
61 }

144 void GW::CalculateGWPerturbation() {
145 
146  Sigma_x_ = (1 - opt_.ScaHFX) * sigma_->CalcExchangeMatrix();
147  XTP_LOG(Log::error, log_)
148  << TimeStamp() << " Calculated Hartree exchange contribution"
149  << std::flush;
150  // dftenergies has size aobasissize
151  // rpaenergies/Mmn have size rpatotal
152  // gwaenergies/frequencies have size qptotal
153  // homo index is relative to dft_energies
154  XTP_LOG(Log::error, log_)
155  << TimeStamp() << " Scissor shifting DFT energies by: " << opt_.shift
156  << " Hrt" << std::flush;
157  Eigen::VectorXd dft_shifted_energies = ScissorShift_DFTlevel(dft_energies_);
158  rpa_.setRPAInputEnergies(
159  dft_shifted_energies.segment(opt_.rpamin, opt_.rpamax - opt_.rpamin + 1));
160  Eigen::VectorXd frequencies =
161  dft_shifted_energies.segment(opt_.qpmin, qptotal_);
162 
163  Anderson mixing_;
164  mixing_.Configure(opt_.gw_mixing_order, opt_.gw_mixing_alpha);
165 
166  for (Index i_gw = 0; i_gw < opt_.gw_sc_max_iterations; ++i_gw) {
167 
168  if (i_gw % opt_.reset_3c == 0 && i_gw != 0) {
169  Mmn_.Rebuild();
170  XTP_LOG(Log::info, log_)
171  << TimeStamp() << " Rebuilding 3c integrals" << std::flush;
172  }
173  sigma_->PrepareScreening();
174  XTP_LOG(Log::info, log_)
175  << TimeStamp() << " Calculated screening via RPA" << std::flush;
176  XTP_LOG(Log::info, log_)
177  << TimeStamp() << " Solving QP equations " << std::flush;
178  if (opt_.gw_mixing_order > 0 && i_gw > 0) {
179  mixing_.UpdateInput(frequencies);
180  }
181 
182  frequencies = SolveQP(frequencies);
183 
184  if (opt_.gw_sc_max_iterations > 1) {
185  Eigen::VectorXd rpa_energies_old = rpa_.getRPAInputEnergies();
186  if (opt_.gw_mixing_order > 0 && i_gw > 0) {
187  if (opt_.gw_mixing_order == 1) {
188  XTP_LOG(Log::debug, log_)
189  << "GWSC using linear mixing with alpha: " << opt_.gw_mixing_alpha
190  << std::flush;
191  } else {
192  XTP_LOG(Log::debug, log_)
193  << "GWSC using Anderson mixing with history "
194  << opt_.gw_mixing_order << ", alpha: " << opt_.gw_mixing_alpha
195  << std::flush;
196  }
197  mixing_.UpdateOutput(frequencies);
198  Eigen::VectorXd mixed_frequencies = mixing_.MixHistory();
199  rpa_.UpdateRPAInputEnergies(dft_energies_, mixed_frequencies,
200  opt_.qpmin);
201  frequencies = mixed_frequencies;
202 
203  } else {
204  XTP_LOG(Log::debug, log_) << "GWSC using plain update " << std::flush;
205  rpa_.UpdateRPAInputEnergies(dft_energies_, frequencies, opt_.qpmin);
206  }
207 
208  for (int i = 0; i < frequencies.size(); i++) {
209  XTP_LOG(Log::debug, log_)
210  << "... GWSC iter " << i_gw << " state " << i << " "
211  << std::setprecision(9) << frequencies(i) << std::flush;
212  }
213 
214  XTP_LOG(Log::info, log_)
215  << TimeStamp() << " GW_Iteration:" << i_gw
216  << " Shift[Hrt]:" << CalcHomoLumoShift(frequencies) << std::flush;
217  if (Converged(rpa_.getRPAInputEnergies(), rpa_energies_old,
218  opt_.gw_sc_limit)) {
219  XTP_LOG(Log::info, log_) << TimeStamp() << " Converged after "
220  << i_gw + 1 << " GW iterations." << std::flush;
221  break;
222  } else if (i_gw == opt_.gw_sc_max_iterations - 1) {
223  XTP_LOG(Log::error, log_)
224  << TimeStamp()
225  << " WARNING! GW-self-consistency cycle not converged after "
226  << opt_.gw_sc_max_iterations << " iterations." << std::flush;
227  XTP_LOG(Log::error, log_)
228  << TimeStamp() << " Run continues. Inspect results carefully!"
229  << std::flush;
230  break;
231  }
232  }
233  }
234  Sigma_c_.diagonal() = sigma_->CalcCorrelationDiag(frequencies);
235  PrintGWA_Energies();
236 }

243 Eigen::VectorXd GW::SolveQP(const Eigen::VectorXd& frequencies) const {
244  const Eigen::VectorXd intercepts =
245  dft_energies_.segment(opt_.qpmin, qptotal_) + Sigma_x_.diagonal() -
246  vxc_.diagonal();
247  Eigen::VectorXd frequencies_new = frequencies;
248  Eigen::Array<bool, Eigen::Dynamic, 1> converged =
249  Eigen::Array<bool, Eigen::Dynamic, 1>::Zero(qptotal_);
250 #ifdef _OPENMP
251  Index use_threads =
252  OPENMP::getMaxThreads() > qptotal_ ? qptotal_ : OPENMP::getMaxThreads();
253 #endif
254 #pragma omp parallel for schedule(dynamic) num_threads(use_threads)
255  for (Index gw_level = 0; gw_level < qptotal_; ++gw_level) {
256 
257  double initial_f = frequencies[gw_level];
258  double intercept = intercepts[gw_level];
259  boost::optional<double> newf;
260  if (opt_.qp_solver == "fixedpoint") {
261  newf = SolveQP_FixedPoint(intercept, initial_f, gw_level);
262  }
263  if (newf) {
264  frequencies_new[gw_level] = newf.value();
265  converged[gw_level] = true;
266  } else {
267  newf = SolveQP_Grid(intercept, initial_f, gw_level);
268  if (newf) {
269  frequencies_new[gw_level] = newf.value();
270  converged[gw_level] = true;
271  } else {
272  newf = SolveQP_Linearisation(intercept, initial_f, gw_level);
273  if (newf) {
274  frequencies_new[gw_level] = newf.value();
275  }
276  }
277  }
278  }
279 
280  if (!converged.all()) {
281  std::vector<Index> states;
282  for (Index s = 0; s < converged.size(); s++) {
283  if (!converged[s]) {
284  states.push_back(s);
285  }
286  }
287  IndexParser rp;
288  XTP_LOG(Log::error, log_) << TimeStamp() << " Not converged PQP states are:"
289  << rp.CreateIndexString(states) << std::flush;
290  XTP_LOG(Log::error, log_)
291  << TimeStamp() << " Increase the grid search interval" << std::flush;
292  }
293  return frequencies_new;
294 }

296 boost::optional<double> GW::SolveQP_Linearisation(double intercept0,
297  double frequency0,
298  Index gw_level) const {
299  boost::optional<double> newf = boost::none;
300 
301  double sigma = sigma_->CalcCorrelationDiagElement(gw_level, frequency0);
302  double dsigma_domega =
303  sigma_->CalcCorrelationDiagElementDerivative(gw_level, frequency0);
304  double Z = 1.0 - dsigma_domega;
305  if (std::abs(Z) > 1e-9) {
306  newf = frequency0 + (intercept0 - frequency0 + sigma) / Z;
307  }
308  return newf;
309 }

311 boost::optional<double> GW::SolveQP_Grid(double intercept0, double frequency0,
312  Index gw_level) const {
313  std::vector<std::pair<double, double>> roots;
314  const double range =
315  opt_.qp_grid_spacing * double(opt_.qp_grid_steps - 1) / 2.0;
316  boost::optional<double> newf = boost::none;
317  double freq_prev = frequency0 - range;
318  QPFunc fqp(gw_level, *sigma_.get(), intercept0);
319  double targ_prev = fqp.value(freq_prev);
320  double qp_energy = 0.0;
321  double gradient_max = std::numeric_limits<double>::max();
322  bool pole_found = false;
323  for (Index i_node = 1; i_node < opt_.qp_grid_steps; ++i_node) {
324  double freq = freq_prev + opt_.qp_grid_spacing;
325  double targ = fqp.value(freq);
326  if (targ_prev * targ < 0.0) { // Sign change
327  double f = SolveQP_Bisection(freq_prev, targ_prev, freq, targ, fqp);
328  double gradient = fqp.deriv(f);
329  double qp_weight = -1.0 / gradient;
330  roots.push_back(std::make_pair(f, qp_weight));
331  if (std::abs(gradient) < gradient_max) {
332  gradient_max = std::abs(gradient);
333  qp_energy = f;
334  pole_found = true;
335  }
336  }
337  freq_prev = freq;
338  targ_prev = targ;
339  }
340  if (Log::current_level > Log::error) {
341 #pragma omp critical
342  {
343  if (!pole_found) {
344  XTP_LOG(Log::info, log_)
345  << " No roots found for qplevel:" << gw_level << std::flush;
346  } else {
347  XTP_LOG(Log::info, log_) << " Roots found for qplevel:" << gw_level
348  << " (qpenergy:qpweight)\n\t\t";
349  for (auto& root : roots) {
350  XTP_LOG(Log::info, log_) << std::setprecision(5) << root.first << ":"
351  << root.second << " ";
352  }
353  XTP_LOG(Log::info, log_) << "Root chosen " << qp_energy << std::flush;
354  }
355  }
356  }
357 
358  if (pole_found) {
359  newf = qp_energy;
360  }
361  return newf;
362 }

364 boost::optional<double> GW::SolveQP_FixedPoint(double intercept0,
365  double frequency0,
366  Index gw_level) const {
367  boost::optional<double> newf = boost::none;
368  QPFunc f(gw_level, *sigma_.get(), intercept0);
369  NewtonRapson<QPFunc> newton = NewtonRapson<QPFunc>(
370  opt_.g_sc_max_iterations, opt_.g_sc_limit, opt_.qp_solver_alpha);
371  double freq_new = newton.FindRoot(f, frequency0);
372  if (newton.getInfo() == NewtonRapson<QPFunc>::success) {
373  newf = freq_new;
374  }
375  return newf;
376 }

378 // https://en.wikipedia.org/wiki/Bisection_method
379 double GW::SolveQP_Bisection(double lowerbound, double f_lowerbound,
380  double upperbound, double f_upperbound,
381  const QPFunc& f) const {
382 
383  if (f_lowerbound * f_upperbound > 0) {
384  throw std::runtime_error(
385  "Bisection needs a postive and negative function value");
386  }
387  double zero = 0.0;
388  while (true) {
389  double c = 0.5 * (lowerbound + upperbound);
390  if (std::abs(upperbound - lowerbound) < opt_.g_sc_limit) {
391  zero = c;
392  break;
393  }
394  double y_c = f.value(c);
395  if (std::abs(y_c) < opt_.g_sc_limit) {
396  zero = c;
397  break;
398  }
399  if (y_c * f_lowerbound > 0) {
400  lowerbound = c;
401  f_lowerbound = y_c;
402  } else {
403  upperbound = c;
404  f_upperbound = y_c;
405  }
406  }
407  return zero;
408 }

410 bool GW::Converged(const Eigen::VectorXd& e1, const Eigen::VectorXd& e2,
411  double epsilon) const {
412  Index state = 0;
413  bool energies_converged = true;
414  double diff_max = (e1 - e2).cwiseAbs().maxCoeff(&state);
415  if (diff_max > epsilon) {
416  energies_converged = false;
417  }
418  XTP_LOG(Log::info, log_) << TimeStamp() << " E_diff max=" << diff_max

420  return energies_converged;

429 void GW::PlotSigma(std::string filename, Index steps, double spacing,
430  std::string states) const {
431 
432  Eigen::VectorXd frequencies =
433  rpa_.getRPAInputEnergies().segment(opt_.qpmin - opt_.rpamin, qptotal_);
434 
435  std::vector<Index> state_inds;
436  IndexParser rp;
437  std::vector<Index> parsed_states = rp.CreateIndexVector(states);
438  for (Index gw_level : parsed_states) {
439  if (gw_level >= opt_.qpmin && gw_level <= opt_.qpmax) {
440  state_inds.push_back(gw_level);
441  }
442  }
443  XTP_LOG(Log::error, log_)
444  << TimeStamp() << " PQP(omega) written to '" << filename
445  << "' for states " << rp.CreateIndexString(state_inds) << std::flush;
446 
447  const Index num_states = state_inds.size();
448 
449  const Eigen::VectorXd intercept =
450  dft_energies_.segment(opt_.qpmin, qptotal_) + Sigma_x_.diagonal() -
451  vxc_.diagonal();
452  Eigen::MatrixXd mat = Eigen::MatrixXd::Zero(steps, 2 * num_states);
453 #pragma omp parallel for schedule(dynamic)
454  for (Index grid_point = 0; grid_point < steps; grid_point++) {
455  const double offset =
456  ((double)grid_point - ((double)(steps - 1) / 2.0)) * spacing;
457  for (Index i = 0; i < num_states; i++) {
458  const Index gw_level = state_inds[i];
459  const double omega = frequencies(gw_level) + offset;
460  double sigma = sigma_->CalcCorrelationDiagElement(gw_level, omega);
461  mat(grid_point, 2 * i) = omega;
462  mat(grid_point, 2 * i + 1) = sigma + intercept[gw_level];
463  }
464  }
465 
466  std::ofstream out;
467  out.open(filename);
468  for (Index i = 0; i < num_states; i++) {
469  const Index gw_level = state_inds[i];
470  out << boost::format("#%1$somega_%2$d\tE_QP(omega)_%2$d") %
471  (i == 0 ? "" : "\t") % gw_level;
472  }
473  out << std::endl;
474  boost::format numFormat("%+1.6f");
475  Eigen::IOFormat matFormat(Eigen::StreamPrecision, 0, "\t", "\n");
476  out << numFormat % mat.format(matFormat) << std::endl;
477  out.close();
478 }
