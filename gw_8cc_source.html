305 bool GW::CheckLogFile() {
306 
307  // check if the log file exists
308  char ch;
309  ifstream _input_file((_run_dir + "/" + _log_file_name).c_str());
310 
311  if (_input_file.fail()) {
312  LOG(logERROR,*_pLog) << "ISOGWA LOG is not found" << flush;
313  return false;
314  };
315 
316  _input_file.seekg(0,ios_base::end); // go to the EOF
317 
318 
319  // get empty lines and end of lines out of the way
320  do {
321  _input_file.seekg(-2,ios_base::cur);
322  _input_file.get(ch);
323  //cout << "\nChar: " << ch << endl;
324  } while ( ch == '\n' || ch == ' ' || (int)_input_file.tellg() == -1 );
325 
326  // get the beginning of the line or the file
327  do {
328  _input_file.seekg(-2,ios_base::cur);
329  _input_file.get(ch);
330  //cout << "\nNext Char: " << ch << " TELL G " << (int)_input_file.tellg() << endl;
331  } while ( ch != '\n' || (int)_input_file.tellg() == -1 );
332 
333  string _line;
334  getline(_input_file,_line); // Read the current line
335  //cout << "\nResult: " << _line << '\n'; // Display it
336  _input_file.close();
337 
338  std::string::size_type self_energy_pos = _line.find("INF total program runtime");
339  if (self_energy_pos == std::string::npos) {
340  LOG(logERROR,*_pLog) << "ISOGWA is incomplete" << flush;
341  return false;
342  } else {
343  //LOG(logDEBUG,*_pLog) << "Gaussian LOG is complete" << flush;
344  return true;
345  }
346 
347  _input_file.close();
348
