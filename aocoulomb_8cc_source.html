27 void AOCoulomb::FillBlock(Eigen::Block<Eigen::MatrixXd>& matrix,
28  const AOShell& shell_row,
29  const AOShell& shell_col) const {
30 
31  // shell info, only lmax tells how far to go
32  Index lmax_row = Index(shell_row.getL());
33  Index lmax_col = Index(shell_col.getL());
34 
35  // set size of internal block for recursion
36  Index nrows = AOTransform::getBlockSize(lmax_row);
37  Index ncols = AOTransform::getBlockSize(lmax_col);
38  const Index mmax = lmax_row + lmax_col;
39  const Index nextra = mmax + 1;
40 
41  // get shell positions
42  const Eigen::Vector3d& pos_row = shell_row.getPos();
43  const Eigen::Vector3d& pos_col = shell_col.getPos();
44  const Eigen::Vector3d diff = pos_row - pos_col;
45  double distsq = diff.squaredNorm();
46 
47  const double pi = boost::math::constants::pi<double>();
48 
49  std::array<int, 9> n_orbitals = AOTransform::n_orbitals();
50  std::array<int, 165> nx = AOTransform::nx();
51  std::array<int, 165> ny = AOTransform::ny();
52  std::array<int, 165> nz = AOTransform::nz();
53  std::array<int, 165> i_less_x = AOTransform::i_less_x();
54  std::array<int, 165> i_less_y = AOTransform::i_less_y();
55  std::array<int, 165> i_less_z = AOTransform::i_less_z();
56 
57  // iterate over Gaussians in this shell_row
58  for (const auto& gaussian_row : shell_row) {
59  // iterate over Gaussians in this shell_col
60  const double decay_row = gaussian_row.getDecay();
61  const double rdecay_row = 0.5 / decay_row;
62  const double powfactor_row = gaussian_row.getPowfactor();
63  for (const auto& gaussian_col : shell_col) {
64 
65  // get decay constants
66  const double decay_col = gaussian_col.getDecay();
67  const double rdecay_col = 0.5 / decay_col;
68  const double powfactor_col = gaussian_col.getPowfactor();
69 
70  Eigen::Tensor<double, 3> cou(nrows, ncols, nextra);
71  cou.setZero();
72 
73  const double decay = decay_row + decay_col;
74  const double r_decay = 0.5 / decay;
75  const double r_decay_2 = 2. * r_decay;
76  const double fac_a_ac = decay_row / decay;
77  const double fac_c_ac = decay_col / decay;
78 
79  const Eigen::Vector3d wmp =
80  r_decay_2 * (decay_row * pos_row + decay_col * pos_col) - pos_row;
81  const Eigen::Vector3d wmq =
82  r_decay_2 * (decay_row * pos_row + decay_col * pos_col) - pos_col;
83 
84  const double T = fac_a_ac * decay_col * distsq;
85 
86  double fak = 2.0 * pow(pi, 2.5) /
87  (decay_row * decay_col * sqrt(decay_row + decay_col));
88  fak = fak * powfactor_col * powfactor_row;
89 
90  const Eigen::VectorXd FmT = AOTransform::XIntegrate(nextra, T);
91 
92  // get initial data from FmT -> s-s element
93  for (Index i = 0; i != nextra; ++i) {
94  cou(0, 0, i) = fak * FmT[i];
95  }
96 
97  // Integrals p - s
98  if (lmax_row > 0) {
99  for (Index m = 0; m < mmax; m++) {
100  cou(Cart::x, 0, m) = wmp(0) * cou(0, 0, m + 1);
101  cou(Cart::y, 0, m) = wmp(1) * cou(0, 0, m + 1);
102  cou(Cart::z, 0, m) = wmp(2) * cou(0, 0, m + 1);
103  }
104  }
105  //------------------------------------------------------
106 
107  // Integrals d - s
108  if (lmax_row > 1) {
109  for (Index m = 0; m < mmax - 1; m++) {
110  double term =
111  rdecay_row * (cou(0, 0, m) - fac_c_ac * cou(0, 0, m + 1));
112  cou(Cart::xx, 0, m) = wmp(0) * cou(Cart::x, 0, m + 1) + term;
113  cou(Cart::xy, 0, m) = wmp(0) * cou(Cart::y, 0, m + 1);
114  cou(Cart::xz, 0, m) = wmp(0) * cou(Cart::z, 0, m + 1);
115  cou(Cart::yy, 0, m) = wmp(1) * cou(Cart::y, 0, m + 1) + term;
116  cou(Cart::yz, 0, m) = wmp(1) * cou(Cart::z, 0, m + 1);
117  cou(Cart::zz, 0, m) = wmp(2) * cou(Cart::z, 0, m + 1) + term;
118  }
119  }
120  //------------------------------------------------------
121 
122  // Integrals f - s
123  if (lmax_row > 2) {
124  for (Index m = 0; m < mmax - 2; m++) {
125  cou(Cart::xxx, 0, m) =
126  wmp(0) * cou(Cart::xx, 0, m + 1) +
127  2 * rdecay_row *
128  (cou(Cart::x, 0, m) - fac_c_ac * cou(Cart::x, 0, m + 1));
129  cou(Cart::xxy, 0, m) = wmp(1) * cou(Cart::xx, 0, m + 1);
130  cou(Cart::xxz, 0, m) = wmp(2) * cou(Cart::xx, 0, m + 1);
131  cou(Cart::xyy, 0, m) = wmp(0) * cou(Cart::yy, 0, m + 1);
132  cou(Cart::xyz, 0, m) = wmp(0) * cou(Cart::yz, 0, m + 1);
133  cou(Cart::xzz, 0, m) = wmp(0) * cou(Cart::zz, 0, m + 1);
134  cou(Cart::yyy, 0, m) =
135  wmp(1) * cou(Cart::yy, 0, m + 1) +
136  2 * rdecay_row *
137  (cou(Cart::y, 0, m) - fac_c_ac * cou(Cart::y, 0, m + 1));
138  cou(Cart::yyz, 0, m) = wmp(2) * cou(Cart::yy, 0, m + 1);
139  cou(Cart::yzz, 0, m) = wmp(1) * cou(Cart::zz, 0, m + 1);
140  cou(Cart::zzz, 0, m) =
141  wmp(2) * cou(Cart::zz, 0, m + 1) +
142  2 * rdecay_row *
143  (cou(Cart::z, 0, m) - fac_c_ac * cou(Cart::z, 0, m + 1));
144  }
145  }
146  //------------------------------------------------------
147 
148  // Integrals g - s
149  if (lmax_row > 3) {
150  for (Index m = 0; m < mmax - 3; m++) {
151  double term_xx = rdecay_row * (cou(Cart::xx, 0, m) -
152  fac_c_ac * cou(Cart::xx, 0, m + 1));
153  double term_yy = rdecay_row * (cou(Cart::yy, 0, m) -
154  fac_c_ac * cou(Cart::yy, 0, m + 1));
155  double term_zz = rdecay_row * (cou(Cart::zz, 0, m) -
156  fac_c_ac * cou(Cart::zz, 0, m + 1));
157  cou(Cart::xxxx, 0, m) =
158  wmp(0) * cou(Cart::xxx, 0, m + 1) + 3 * term_xx;
159  cou(Cart::xxxy, 0, m) = wmp(1) * cou(Cart::xxx, 0, m + 1);
160  cou(Cart::xxxz, 0, m) = wmp(2) * cou(Cart::xxx, 0, m + 1);
161  cou(Cart::xxyy, 0, m) = wmp(0) * cou(Cart::xyy, 0, m + 1) + term_yy;
162  cou(Cart::xxyz, 0, m) = wmp(1) * cou(Cart::xxz, 0, m + 1);
163  cou(Cart::xxzz, 0, m) = wmp(0) * cou(Cart::xzz, 0, m + 1) + term_zz;
164  cou(Cart::xyyy, 0, m) = wmp(0) * cou(Cart::yyy, 0, m + 1);
165  cou(Cart::xyyz, 0, m) = wmp(0) * cou(Cart::yyz, 0, m + 1);
166  cou(Cart::xyzz, 0, m) = wmp(0) * cou(Cart::yzz, 0, m + 1);
167  cou(Cart::xzzz, 0, m) = wmp(0) * cou(Cart::zzz, 0, m + 1);
168  cou(Cart::yyyy, 0, m) =
169  wmp(1) * cou(Cart::yyy, 0, m + 1) + 3 * term_yy;
170  cou(Cart::yyyz, 0, m) = wmp(2) * cou(Cart::yyy, 0, m + 1);
171  cou(Cart::yyzz, 0, m) = wmp(1) * cou(Cart::yzz, 0, m + 1) + term_zz;
172  cou(Cart::yzzz, 0, m) = wmp(1) * cou(Cart::zzz, 0, m + 1);
173  cou(Cart::zzzz, 0, m) =
174  wmp(2) * cou(Cart::zzz, 0, m + 1) + 3 * term_zz;
175  }
176  }
177  //------------------------------------------------------
178 
179  // Integrals h - s
180  if (lmax_row > 4) {
181  for (Index m = 0; m < mmax - 4; m++) {
182  double term_xxx = rdecay_row * (cou(Cart::xxx, 0, m) -
183  fac_c_ac * cou(Cart::xxx, 0, m + 1));
184  double term_yyy = rdecay_row * (cou(Cart::yyy, 0, m) -
185  fac_c_ac * cou(Cart::yyy, 0, m + 1));
186  double term_zzz = rdecay_row * (cou(Cart::zzz, 0, m) -
187  fac_c_ac * cou(Cart::zzz, 0, m + 1));
188  cou(Cart::xxxxx, 0, m) =
189  wmp(0) * cou(Cart::xxxx, 0, m + 1) + 4 * term_xxx;
190  cou(Cart::xxxxy, 0, m) = wmp(1) * cou(Cart::xxxx, 0, m + 1);
191  cou(Cart::xxxxz, 0, m) = wmp(2) * cou(Cart::xxxx, 0, m + 1);
192  cou(Cart::xxxyy, 0, m) =
193  wmp(1) * cou(Cart::xxxy, 0, m + 1) + term_xxx;
194  cou(Cart::xxxyz, 0, m) = wmp(1) * cou(Cart::xxxz, 0, m + 1);
195  cou(Cart::xxxzz, 0, m) =
196  wmp(2) * cou(Cart::xxxz, 0, m + 1) + term_xxx;
197  cou(Cart::xxyyy, 0, m) =
198  wmp(0) * cou(Cart::xyyy, 0, m + 1) + term_yyy;
199  cou(Cart::xxyyz, 0, m) = wmp(2) * cou(Cart::xxyy, 0, m + 1);
200  cou(Cart::xxyzz, 0, m) = wmp(1) * cou(Cart::xxzz, 0, m + 1);
201  cou(Cart::xxzzz, 0, m) =
202  wmp(0) * cou(Cart::xzzz, 0, m + 1) + term_zzz;
203  cou(Cart::xyyyy, 0, m) = wmp(0) * cou(Cart::yyyy, 0, m + 1);
204  cou(Cart::xyyyz, 0, m) = wmp(0) * cou(Cart::yyyz, 0, m + 1);
205  cou(Cart::xyyzz, 0, m) = wmp(0) * cou(Cart::yyzz, 0, m + 1);
206  cou(Cart::xyzzz, 0, m) = wmp(0) * cou(Cart::yzzz, 0, m + 1);
207  cou(Cart::xzzzz, 0, m) = wmp(0) * cou(Cart::zzzz, 0, m + 1);
208  cou(Cart::yyyyy, 0, m) =
209  wmp(1) * cou(Cart::yyyy, 0, m + 1) + 4 * term_yyy;
210  cou(Cart::yyyyz, 0, m) = wmp(2) * cou(Cart::yyyy, 0, m + 1);
211  cou(Cart::yyyzz, 0, m) =
212  wmp(2) * cou(Cart::yyyz, 0, m + 1) + term_yyy;
213  cou(Cart::yyzzz, 0, m) =
214  wmp(1) * cou(Cart::yzzz, 0, m + 1) + term_zzz;
215  cou(Cart::yzzzz, 0, m) = wmp(1) * cou(Cart::zzzz, 0, m + 1);
216  cou(Cart::zzzzz, 0, m) =
217  wmp(2) * cou(Cart::zzzz, 0, m + 1) + 4 * term_zzz;
218  }
219  }
220  //------------------------------------------------------
221 
222  // Integrals i - s
223  if (lmax_row > 5) {
224  for (Index m = 0; m < mmax - 5; m++) {
225  double term_xxxx =
226  rdecay_row *
227  (cou(Cart::xxxx, 0, m) - fac_c_ac * cou(Cart::xxxx, 0, m + 1));
228  double term_xyyy =
229  rdecay_row *
230  (cou(Cart::xyyy, 0, m) - fac_c_ac * cou(Cart::xyyy, 0, m + 1));
231  double term_xzzz =
232  rdecay_row *
233  (cou(Cart::xzzz, 0, m) - fac_c_ac * cou(Cart::xzzz, 0, m + 1));
234  double term_yyyy =
235  rdecay_row *
236  (cou(Cart::yyyy, 0, m) - fac_c_ac * cou(Cart::yyyy, 0, m + 1));
237  double term_yyzz =
238  rdecay_row *
239  (cou(Cart::yyzz, 0, m) - fac_c_ac * cou(Cart::yyzz, 0, m + 1));
240  double term_yzzz =
241  rdecay_row *
242  (cou(Cart::yzzz, 0, m) - fac_c_ac * cou(Cart::yzzz, 0, m + 1));
243  double term_zzzz =
244  rdecay_row *
245  (cou(Cart::zzzz, 0, m) - fac_c_ac * cou(Cart::zzzz, 0, m + 1));
246  cou(Cart::xxxxxx, 0, m) =
247  wmp(0) * cou(Cart::xxxxx, 0, m + 1) + 5 * term_xxxx;
248  cou(Cart::xxxxxy, 0, m) = wmp(1) * cou(Cart::xxxxx, 0, m + 1);
249  cou(Cart::xxxxxz, 0, m) = wmp(2) * cou(Cart::xxxxx, 0, m + 1);
250  cou(Cart::xxxxyy, 0, m) =
251  wmp(1) * cou(Cart::xxxxy, 0, m + 1) + term_xxxx;
252  cou(Cart::xxxxyz, 0, m) = wmp(1) * cou(Cart::xxxxz, 0, m + 1);
253  cou(Cart::xxxxzz, 0, m) =
254  wmp(2) * cou(Cart::xxxxz, 0, m + 1) + term_xxxx;
255  cou(Cart::xxxyyy, 0, m) =
256  wmp(0) * cou(Cart::xxyyy, 0, m + 1) + 2 * term_xyyy;
257  cou(Cart::xxxyyz, 0, m) = wmp(2) * cou(Cart::xxxyy, 0, m + 1);
258  cou(Cart::xxxyzz, 0, m) = wmp(1) * cou(Cart::xxxzz, 0, m + 1);
259  cou(Cart::xxxzzz, 0, m) =
260  wmp(0) * cou(Cart::xxzzz, 0, m + 1) + 2 * term_xzzz;
261  cou(Cart::xxyyyy, 0, m) =
262  wmp(0) * cou(Cart::xyyyy, 0, m + 1) + term_yyyy;
263  cou(Cart::xxyyyz, 0, m) = wmp(2) * cou(Cart::xxyyy, 0, m + 1);
264  cou(Cart::xxyyzz, 0, m) =
265  wmp(0) * cou(Cart::xyyzz, 0, m + 1) + term_yyzz;
266  cou(Cart::xxyzzz, 0, m) = wmp(1) * cou(Cart::xxzzz, 0, m + 1);
267  cou(Cart::xxzzzz, 0, m) =
268  wmp(0) * cou(Cart::xzzzz, 0, m + 1) + term_zzzz;
269  cou(Cart::xyyyyy, 0, m) = wmp(0) * cou(Cart::yyyyy, 0, m + 1);
270  cou(Cart::xyyyyz, 0, m) = wmp(0) * cou(Cart::yyyyz, 0, m + 1);
271  cou(Cart::xyyyzz, 0, m) = wmp(0) * cou(Cart::yyyzz, 0, m + 1);
272  cou(Cart::xyyzzz, 0, m) = wmp(0) * cou(Cart::yyzzz, 0, m + 1);
273  cou(Cart::xyzzzz, 0, m) = wmp(0) * cou(Cart::yzzzz, 0, m + 1);
274  cou(Cart::xzzzzz, 0, m) = wmp(0) * cou(Cart::zzzzz, 0, m + 1);
275  cou(Cart::yyyyyy, 0, m) =
276  wmp(1) * cou(Cart::yyyyy, 0, m + 1) + 5 * term_yyyy;
277  cou(Cart::yyyyyz, 0, m) = wmp(2) * cou(Cart::yyyyy, 0, m + 1);
278  cou(Cart::yyyyzz, 0, m) =
279  wmp(2) * cou(Cart::yyyyz, 0, m + 1) + term_yyyy;
280  cou(Cart::yyyzzz, 0, m) =
281  wmp(1) * cou(Cart::yyzzz, 0, m + 1) + 2 * term_yzzz;
282  cou(Cart::yyzzzz, 0, m) =
283  wmp(1) * cou(Cart::yzzzz, 0, m + 1) + term_zzzz;
284  cou(Cart::yzzzzz, 0, m) = wmp(1) * cou(Cart::zzzzz, 0, m + 1);
285  cou(Cart::zzzzzz, 0, m) =
286  wmp(2) * cou(Cart::zzzzz, 0, m + 1) + 5 * term_zzzz;
287  }
288  }

295  cou(0, Cart::x, m) = wmq(0) * cou(0, 0, m + 1);

317  for (Index i = 4; i < n_orbitals[lmax_row]; i++) {

335  double term =
336  rdecay_col * (cou(0, 0, m) - fac_a_ac * cou(0, 0, m + 1));

367  wmq(2) * cou(i, Cart::z, m + 1) +

373  } // end if (lmax_col > 1)

375  if (lmax_col > 2) {

377  // Integrals s - f
378  for (Index m = 0; m < lmax_col - 2; m++) {
379  cou(0, Cart::xxx, m) =
380  wmq(0) * cou(0, Cart::xx, m + 1) +
381  2 * rdecay_col *

402  for (Index m = 0; m < lmax_col - 2; m++) {
