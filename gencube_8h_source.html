168  void GenCube::calculateCube(){
169 
170  CTP_LOG(ctp::logDEBUG, _log) << "Reading serialized QM data from " << _orbfile << flush;
171 
172  Orbitals _orbitals;
173 
174  CTP_LOG(ctp::logDEBUG, _log) << " Loading QM data from " << _orbfile << flush;
175  _orbitals.Load(_orbfile);
176 
177  if (_do_qp && !_orbitals.hasQPdiag()){
178  throw std::runtime_error("Orbitals file does not contain QP coefficients");
179  }
180 
181  if (_do_bse && _do_singlet && !_orbitals.hasBSESinglets()){
182  throw std::runtime_error("Orbitals file does not contain Singlet BSE coefficients");
183  }
184  if (_do_bse && _do_triplet && !_orbitals.hasBSETriplets()){
185  throw std::runtime_error("Orbitals file does not contain Triplet BSE coefficients");
186  }
187 
188 
189 
190  // load the QM data from serialized orbitals object
191 
192 
193 
194 
195  // get atoms
196  std::vector<ctp::QMAtom*> _atoms = _orbitals.QMAtoms();
197 
198  // determine min and max in each cartesian direction
199  double xmin = std::numeric_limits<double>::max();
200  double xmax = std::numeric_limits<double>::min();
201  double ymin = xmin;
202  double ymax = xmax;
203  double zmin = xmin;
204  double zmax = xmax;
205 
206  vector< ctp::QMAtom* > ::iterator ait;
207  for (ait = _atoms.begin(); ait != _atoms.end(); ++ait) {
208  // get center coordinates in Bohr
209  double x = (*ait)->x * tools::conv::ang2bohr;
210  double y = (*ait)->y * tools::conv::ang2bohr;
211  double z = (*ait)->z * tools::conv::ang2bohr;
212 
213  if (x > xmax) xmax = x;
214  if (x < xmin) xmin = x;
215  if (y > ymax) ymax = y;
216  if (y < ymin) ymin = y;
217  if (z > zmax) zmax = z;
218  if (z < zmin) zmin = z;
219 
220 
221  }
222  // generate cube grid
223  double xstart = xmin - _padding;
224  double xstop = xmax + _padding;
225  double ystart = ymin - _padding;
226  double ystop = ymax + _padding;
227  double zstart = zmin - _padding;
228  double zstop = zmax + _padding;
229 
230  double xincr = (xstop - xstart) / double(_xsteps);
231  double yincr = (ystop - ystart) / double(_ysteps);
232  double zincr = (zstop - zstart) / double(_zsteps);
233 
234 
235  // open output stream
236  FILE *out;
237  out = fopen(_output_file.c_str(), "w");
238 
239  // write cube header
240  if ( _do_groundstate && !_do_bse ){
241  fprintf(out, "Electron density of neutral state \n" );
242  } else if ( _do_bse ){
243  if ( _do_groundstate ){
244  fprintf(out, "Total electron density of excited state %i spin %s \n", _state, _spin.c_str());
245  }
246  else{
247  fprintf(out, "Difference electron density of excited state %i spin %s \n", _state, _spin.c_str());
248  }
249  }
250  else if ( _do_qp ){
251  fprintf(out, "Quasiparticle state %i \n", _state);
252  }
253  else if ( _do_transition ){
254  fprintf(out, "Transition state between Groundstate and state %i \n", _state);
255  }
256  fprintf(out, "Created by VOTCA-XTP \n");
257  if ( _do_qp ){
258  fprintf(out, "-%lu %f %f %f \n", _atoms.size(), xstart, ystart, zstart);
259  } else {
260  fprintf(out, "%lu %f %f %f \n", _atoms.size(), xstart, ystart, zstart);
261  }
262 
263  fprintf(out, "%d %f 0.0 0.0 \n", _xsteps + 1, xincr);
264  fprintf(out, "%d 0.0 %f 0.0 \n", _ysteps + 1, yincr);
265  fprintf(out, "%d 0.0 0.0 %f \n", _zsteps + 1, zincr);
266  Elements _elements;
267  for (ait = _atoms.begin(); ait != _atoms.end(); ++ait) {
268  // get center coordinates in Bohr
269  double x = (*ait)->x * tools::conv::ang2bohr;
270  double y = (*ait)->y * tools::conv::ang2bohr;
271  double z = (*ait)->z * tools::conv::ang2bohr;
272 
273  string element = (*ait)->type;
274  int atnum =_elements.getEleNum(element);
275  double crg = _elements.getNucCrgECP(element);
276 
277  fprintf(out, "%d %f %f %f %f\n", atnum, crg, x, y, z);
278 
279 
280  }
281 
282  if ( _do_qp ){
283  fprintf(out, " 1 %d \n", _state);
284  }
285 
286  // load DFT basis set (element-wise information) from xml file
287  BasisSet dftbs;
288  dftbs.LoadBasisSet(_orbitals.getDFTbasis());
289  CTP_LOG(ctp::logDEBUG, _log) << " Loaded DFT Basis Set " << _orbitals.getDFTbasis() << flush;
290 
291  // fill DFT AO basis by going through all atoms
292  AOBasis dftbasis;
293  dftbasis.AOBasisFill(&dftbs, _orbitals.QMAtoms());
294 
295 
296 
297  // now depending on the type of cube
298  if (_do_groundstate || _do_bse || _do_transition ) {
299 
300 
301  ub::matrix<double> DMAT_tot = ub::zero_matrix<double>(dftbasis.AOBasisSize(), dftbasis.AOBasisSize());
302 
303  // ground state only if requested
304  if ( _do_groundstate ) {
305  ub::matrix<double> DMATGS = _orbitals.DensityMatrixGroundState();
306  DMAT_tot = DMATGS; // Ground state + hole_contribution + electron contribution
307  CTP_LOG(ctp::logDEBUG, _log) << " Calculated ground state density matrix " << flush;
308  }
309 
310  if(_state>0){
311 
312 
313  if ( _do_transition ){
314  DMAT_tot=_orbitals.TransitionDensityMatrix(_spin, _state - 1);
315  CTP_LOG(ctp::logDEBUG, _log) << " Calculated transition state density matrix " << flush;
316  }
317 
318  // excited state if requested
319  else if ( _do_bse ) {
320  std::vector< ub::matrix<double> > DMAT=_orbitals.DensityMatrixExcitedState(_spin, _state - 1);
321 
322  DMAT_tot =DMAT_tot+DMAT[1]-DMAT[0];// Ground state + hole_contribution + electron contribution
323  CTP_LOG(ctp::logDEBUG, _log) << " Calculated excited state density matrix " << flush;
324  }
325  }
326 
327 
328  CTP_LOG(ctp::logDEBUG, _log) << " Calculating cube data ... \n" << flush;
329  _log.setPreface(ctp::logDEBUG, (format(" ... ...") ).str());
330 
331  boost::progress_display progress(_xsteps) ;
332  // eval density at cube grid points
333  for (int _ix = 0; _ix <= _xsteps; _ix++) {
334  double _x = xstart + double(_ix) * xincr;
335  for (int _iy = 0; _iy <= _ysteps; _iy++) {
336  double _y = ystart + double(_iy) * yincr;
337 
338  int Nrecord = 0;
339  for (int _iz = 0; _iz <= _zsteps; _iz++) {
340  double _z = zstart + double(_iz) * zincr;
341  Nrecord++;
342  vec pos=vec(_x, _y, _z);
343  // get value of orbitals at each gridpoint
344  ub::matrix<double> tmat = ub::zero_matrix<double>( 1,dftbasis.AOBasisSize());
345 
346  for (AOBasis::AOShellIterator _row = dftbasis.firstShell(); _row != dftbasis.lastShell(); _row++) {
347 
348  const double decay=(*_row)->getMinDecay();
349  const tools::vec& shellpos=(*_row)->getPos();
350 
351 
352  tools::vec dist=shellpos-pos;
353  double distsq=dist*dist;
354  // if contribution is smaller than -ln(1e-10), calc density
355  if ( (decay * distsq) < 20.7 ){
356  ub::matrix_range< ub::matrix<double> > _submatrix = ub::subrange(tmat,0,1, (*_row)->getStartIndex(), (*_row)->getStartIndex()+(*_row)->getNumFunc());
357  (*_row)->EvalAOspace(_submatrix, pos);
358  }
359  }
360 
361 
362  ub::matrix<double> _tempmat = ub::prod( tmat,DMAT_tot); // tempmat can be reused for density gradient
363  double density_at_grid = ub::prod(_tempmat,ub::trans(tmat))(0,0);
364 
365  if (Nrecord == 6 || _iz == _zsteps) {
366  fprintf(out, "%E \n", density_at_grid);
367  Nrecord = 0;
368  } else {
369  fprintf(out, "%E ", density_at_grid);
370  }
371  }// z-component
372 
373 
374 
375  }// y-component
376 
377  ++progress;
378 
379 
380  } // x-component
381 
382 
383  } // ground or excited state
384  _log.setPreface(ctp::logDEBUG, (format("\n ... ...") ).str());
385 
386  // diagonalized QP, if requested
387  if ( _do_qp && _state > 0 ){
388 
389  int GWAmin = _orbitals.getGWAmin();
390  int GWAmax = _orbitals.getGWAmax();
391 
392  //cout << _orbitals.hasQPdiag()<< endl;
393  ub::matrix<double> QPcoefs = ub::project(_orbitals.QPdiagCoefficients(),ub::range(GWAmin, GWAmax + 1), ub::range(_state-1, _state ) ); // get QPdiag coefficients for the requested state
394 
395 
396  ub::matrix<double> MOs = ub::project(_orbitals.MOCoefficients(),ub::range(GWAmin, GWAmax + 1), ub::range(0, dftbasis.AOBasisSize())) ; // get DFT MO coefficients
397 
398  ub::matrix<double> Ftemp = ub::prod( ub::trans(MOs),QPcoefs );
399 
400  for (int _ix = 0; _ix <= _xsteps; _ix++) {
401  double _x = xstart + double(_ix) * xincr;
402  for (int _iy = 0; _iy <= _ysteps; _iy++) {
403  double _y = ystart + double(_iy) * yincr;
404 
405  int Nrecord = 0;
406  for (int _iz = 0; _iz <= _zsteps; _iz++) {
407  double _z = zstart + double(_iz) * zincr;
408  Nrecord++;
409  // get value of orbitals at each gridpoint
410  ub::matrix<double> tmat = ub::zero_matrix<double>(1,dftbasis.AOBasisSize());
411  vec pos=vec(_x, _y, _z);
412  for (AOBasis::AOShellIterator _row = dftbasis.firstShell(); _row != dftbasis.lastShell(); _row++) {
413 
414  ub::matrix_range< ub::matrix<double> > _submatrix = ub::subrange(tmat,0,1, (*_row)->getStartIndex(), (*_row)->getStartIndex()+(*_row)->getNumFunc());
415  (*_row)->EvalAOspace(_submatrix,pos);
416  }
417 
418  double QP_at_grid = 0.0;
419  for (unsigned _i = 0; _i < Ftemp.size1(); _i++) {
420  QP_at_grid += Ftemp(_i,0) * tmat(0,_i);
421  }
422 
423  if (Nrecord == 6 || _iz == _zsteps) {
424  fprintf(out, "%E \n", QP_at_grid);
425  Nrecord = 0;
426  } else {
427  fprintf(out, "%E ", QP_at_grid);
428  }
429  }// z-component
430  }// y-component
431  } // x-component
432 
433 
434 
435  }
436 
437 
438  fclose(out);
439 
440 
441  CTP_LOG(ctp::logDEBUG, _log) << "Wrote cube data to " << _output_file << flush;
442 
443  return;
444  }
