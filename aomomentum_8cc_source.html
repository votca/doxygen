26 void AOMomentum::FillBlock(std::vector<Eigen::Block<Eigen::MatrixXd> >& matrix,
27  const AOShell& shell_row,
28  const AOShell& shell_col) const {
29 
30  /* Calculating the AO matrix of the gradient operator requires
31  * the raw overlap matrix (i.e. in unnormalized cartesians)
32  * with lmax of the column shell increased by one:
33  *
34  * phi_(ijk) = x^i y^j z^k exp(-beta r^2)
35  * => d/dx phi_(ijk) = (i*x^(i-1) - 2beta x^(i+1)) y^j z^k exp(-beta r^2)
36  * d/dy phi_(ijk) = x^i (j*y^(j-1) - 2beta y^(j+1)) z^k exp(-beta r^2)
37  * d/dz phi_(ijk) = x^i y^j (k*z^(k-1) - 2beta z^(k+1)) exp(-beta r^2)
38  *
39  * i.e.: d/dx phi_s = d/dx phi_(000) = -2beta phi_(100) = -2beta phi_px
40  * d/dy phi_px = d/dy phi_(100) = -2beta phi_(110) = -2beta phi_dxy
41  * d/dz phi_pz = d/dz phi_(001) = phi_(000) - 2beta phi_(002)
42  * = phi_s - 2beta phi_dxx
43  *
44  * and with that
45  * <s|d/dx|s> = -2beta <s|px>
46  * <s|d/dy|px> = -2beta <s|dxy>
47  * <s|d/dz|pz> = <s|s> - 2beta <s|dxx>
48  * ...
49  *
50  */
51 
52  // shell info, only lmax tells how far to go
53  Index lmax_row = shell_row.getLmax();
54  Index lmax_col = shell_col.getLmax();
55 
56  if (lmax_col > 4) {
57  throw std::runtime_error(
58  "Momentum transition dipoles only implemented for S,P,D,F,G functions "
59  "in DFT basis!");
60  }
61 
62  // set size of internal block for recursion
63  Index nrows = AOTransform::getBlockSize(lmax_row);
64  Index ncols = AOTransform::getBlockSize(lmax_col);
65 
66  // initialize local matrix block for unnormalized cartesians
67  std::array<Eigen::MatrixXd, 3> mom;
68  for (Index i_comp = 0; i_comp < 3; i_comp++) {
69  mom[i_comp] = Eigen::MatrixXd ::Zero(nrows, ncols);
70  }
71 
72  std::array<Eigen::MatrixXd, 6> scd_mom;
73  for (Index i_comp = 0; i_comp < 6; i_comp++) {
74  scd_mom[i_comp] = Eigen::MatrixXd ::Zero(nrows, ncols);
75  }
76 
77  const Eigen::Vector3d& pos_row = shell_row.getPos();
78  const Eigen::Vector3d& pos_col = shell_col.getPos();
79  const Eigen::Vector3d diff = pos_row - pos_col;
80 
81  double distsq = diff.squaredNorm();
82 
83  std::array<int, 165> nx = AOTransform::nx();
84  std::array<int, 165> ny = AOTransform::ny();
85  std::array<int, 165> nz = AOTransform::nz();
86  std::array<int, 165> i_less_x = AOTransform::i_less_x();
87  std::array<int, 165> i_less_y = AOTransform::i_less_y();
88  std::array<int, 165> i_less_z = AOTransform::i_less_z();
89  std::array<int, 120> i_more_x = AOTransform::i_more_x();
90  std::array<int, 120> i_more_y = AOTransform::i_more_y();
91  std::array<int, 120> i_more_z = AOTransform::i_more_z();
92 
93  for (const auto& gaussian_row : shell_row) {
94  const double decay_row = gaussian_row.getDecay();
95 
96  for (const auto& gaussian_col : shell_col) {
97 
98  const double decay_col = gaussian_col.getDecay();
99 
100  const double fak = 0.5 / (decay_row + decay_col);
101  const double fak2 = 2.0 * fak;
102  double _exparg = fak2 * decay_row * decay_col * distsq;
103 
105 
106  if (_exparg > 30.0) {
107  continue;
108  }
109 
110  AOOverlap overlap;
111  Index L_offset = 1;
112  Eigen::MatrixXd ol =
113  overlap.Primitive_Overlap(gaussian_row, gaussian_col, L_offset);
114 
115  double alpha2 = 2.0 * decay_row;
116  double beta2 = 2.0 * decay_col;
117  for (Index i = 0; i < ncols; i++) {
118 
119  int nx_i = nx[i];
120  int ny_i = ny[i];
121  int nz_i = nz[i];
122  int ilx_i = i_less_x[i];
123  int ily_i = i_less_y[i];
124  int ilz_i = i_less_z[i];
125  int imx_i = i_more_x[i];
126  int imy_i = i_more_y[i];
127  int imz_i = i_more_z[i];
128 
129  for (Index j = 0; j < nrows; j++) {
130 
131  mom[0](j, i) = nx_i * ol(j, ilx_i) - beta2 * ol(j, imx_i);
132  mom[1](j, i) = ny_i * ol(j, ily_i) - beta2 * ol(j, imy_i);
133  mom[2](j, i) = nz_i * ol(j, ilz_i) - beta2 * ol(j, imz_i);
134 
135  int nx_j = nx[j];
136  int ny_j = ny[j];
137  int nz_j = nz[j];
138  int ilx_j = i_less_x[j];
139  int ily_j = i_less_y[j];
140  int ilz_j = i_less_z[j];
141  int imx_j = i_more_x[j];
142  int imy_j = i_more_y[j];
143  int imz_j = i_more_z[j];
144 
145  scd_mom[0](j, i) =
146  nx_j * (beta2 * ol(ilx_j, imx_i) - nx_i * ol(ilx_j, ilx_i)) -
147  alpha2 * (beta2 * ol(imx_j, imx_i) -
148  nx_i * ol(imx_j, ilx_i)); // d2/(dxdx)
149  scd_mom[1](j, i) =
150  nx_j * (beta2 * ol(ilx_j, imy_i) - ny_i * ol(ilx_j, ily_i)) -
151  alpha2 * (beta2 * ol(imx_j, imy_i) -
152  ny_i * ol(imx_j, ily_i)); // d2/(dxdy)
153  scd_mom[2](j, i) =
154  nx_j * (beta2 * ol(ilx_j, imz_i) - nz_i * ol(ilx_j, ilz_i)) -
155  alpha2 * (beta2 * ol(imx_j, imz_i) -
156  nz_i * ol(imx_j, ilz_i)); // d2/(dxdz)
157 
158  scd_mom[3](j, i) =
159  ny_j * (beta2 * ol(ily_j, imy_i) - ny_i * ol(ily_j, ily_i)) -
160  alpha2 * (beta2 * ol(imy_j, imy_i) -
161  ny_i * ol(imy_j, ily_i)); // d2/(dydy)
162  scd_mom[4](j, i) =
163  ny_j * (beta2 * ol(ily_j, imz_i) - nz_i * ol(ily_j, ilz_i)) -
164  alpha2 * (beta2 * ol(imy_j, imz_i) -
165  nz_i * ol(imy_j, ilz_i)); // d2/(dydz)
166 
167  scd_mom[5](j, i) =
168  nz_j * (beta2 * ol(ilz_j, imz_i) - nz_i * ol(ilz_j, ilz_i)) -
169  alpha2 * (beta2 * ol(imz_j, imz_i) -
170  nz_i * ol(imz_j, ilz_i)); // d2/(dzdz)
171  }
172  }
173 
174  Eigen::MatrixXd trafo_row = AOTransform::getTrafo(gaussian_row);
175  Eigen::MatrixXd trafo_col = AOTransform::getTrafo(gaussian_col);
176  // cartesian -> spherical
177  for (Index i = 0; i < 3; i++) {
178  Eigen::MatrixXd mom_sph =
179  trafo_row.transpose() *
180  mom[i].bottomRightCorner(shell_row.getCartesianNumFunc(),
181  shell_col.getCartesianNumFunc()) *
182  trafo_col;
183  // save to matrix
184  matrix[i] += mom_sph;
185  }
186 
187  } // shell_col Gaussians
188  } // shell_row Gaussians
189 }
