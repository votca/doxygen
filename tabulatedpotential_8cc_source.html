00242 void TabulatedPotential::CalcForce(vector<double> &U, vector<double> &F, double dx, bool bPeriodic)
00243 {
00244     size_t n=U.size();
00245     double f = 0.5/dx;
00246     F.resize(n);
00247     if(bPeriodic)
00248         F[n-1] = F[0] = -(U[1] - U[n-2])*f;
00249     else {
00250         F[0] = -(U[1] - U[0])*2*f;
00251         F[n-1] = -(U[n-1] - U[n-2])*2*f;
00252     }
00253     for(size_t i=1; i<n-1; i++)
00254         F[i] = -(U[i+1] - U[i-1])*f;

00296 void TabulatedPotential::Smooth(vector<double> &data, bool bPeriodic)
00297 {
00298     double old[3];
00299     int n=data.size();
00300     if(bPeriodic) {
00301         old[0] = data[n-3];
00302         old[1] = data[n-2];
00303     }
00304     else {
00305         old[0] = data[0];
00306         old[1] = data[0];
00307     }
00308     size_t i;
00309     for(i=0; i<data.size()-2; i++) {
00310         old[2] = data[i];
00311         data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 2.*data[i+1] + data[i+2])/9.;
00312         old[0]=old[1];
00313         old[1]=old[2];;
00314     }
00315     if(bPeriodic) {
00316         data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 2.*data[i+1] + data[0])/9.;
00317         old[0]=old[1];old[1]=data[i];
00318         data[n-1] = data[0];
00319     }
00320     else {
00321         data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 3.*data[i+1])/9.;
00322         old[0]=old[1];old[1]=data[i];
00323         i++;
00324         data[i] = (old[0] + 2.*old[1] + 6.*data[i])/9.;    
00325     }
00326 }

00328 void TabulatedPotential::BoltzmannInvert(vector<double> &data, double T)
00329 {
00330     double _min, _max;
00331     
00332     _min = numeric_limits<double>::max();
00333     _max = numeric_limits<double>::min();
00334     
00335     for(size_t i=0; i<data.size(); i++) {
00336         _max = max(data[i], _max);
00337         if(data[i] > 0) _min = min(data[i], _min);
00338     }
00339     _max = -8.3109*T*log(_max)*0.001;
00340     _min = -8.3109*T*log(_min)*0.001-_max;
00341     
00342     for(size_t i=0; i<data.size(); i++) {
00343         if(data[i] == 0) data[i] = _min;
00344         else
00345             data[i] = -8.3109*T*log(data[i])*0.001 - _max;
00346     }
