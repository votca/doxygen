49 void TabulatedPotential::Command(BondedStatistics &bs, const string &cmd,
50  vector<string> &args) {
51  if (args[0] == "set") {
52  if (cmd == "hist") {
53  SetOption_(_hist_options, args);
54  } else if (cmd == "tab") {
55  if (!SetOption_(_tab_options, args)) {
56  if (args.size() > 2) {
57  if (args[1] == "smooth_pdf") {
58  _tab_smooth1 = boost::lexical_cast<Index>(args[2]);
59  } else if (args[1] == "smooth_pot") {
60  _tab_smooth2 = boost::lexical_cast<Index>(args[2]);
61  } else if (args[1] == "T") {
62  _Temperature = boost::lexical_cast<double>(args[2]);
63  } else {
64  cout << "unknown option " << args[2] << endl;
65  return;
66  }
67  }
68  }
69  if (args.size() <= 2) {
70  cout << "smooth_pdf: " << _tab_smooth1 << endl;
71  cout << "smooth_pot: " << _tab_smooth2 << endl;
72  cout << "T: " << _Temperature << endl;
73  }
74  }
75  } else if (args.size() >= 2) {
76  if (cmd == "hist") {
77  WriteHistogram(bs, args);
78  } else if (cmd == "tab") {
79  WritePotential(bs, args);
80  }
81  } else {
82  cout << "wrong number of arguments" << endl;
83  }
84 }

215 void TabulatedPotential::WritePotential(BondedStatistics &bs,
216  vector<string> &args) {
217  ofstream out;
218  DataCollection<double>::selection *sel = nullptr;
219 
220  // Appends all the interactions that are specified in args to selection
221  // pointer given by sel
222 
223  for (size_t i = 1; i < args.size(); i++) {
224  sel = bs.BondedValues().select(args[i], sel);
225  }
226 
227  Histogram h(_tab_options);
228 
229  h.ProcessData(sel);
230  for (Index i = 0; i < _tab_smooth1; ++i) {
231  Smooth_(h.getPdf(), _tab_options._periodic);
232  }
233  BoltzmannInvert_(h.getPdf());
234  for (Index i = 0; i < _tab_smooth2; ++i) {
235  Smooth_(h.getPdf(), _tab_options._periodic);
236  }
237  out.open(args[0].c_str());
238 
239  vector<double> F;
240  assert(h.getInterval() > 0 && "Interval for pdf histogram is 0");
241  CalcForce_(h.getPdf(), F, h.getInterval(), _tab_options._periodic);
242  for (Index i = 0; i < h.getN(); i++) {
243  out << h.getMin() + h.getInterval() * ((double)i) << " " << h.getPdf()[i]
244  << " " << F[i] << endl;
245  }
246  out.close();
247  cout << "histogram created using " << sel->size() << " data-rows, written to "
248  << args[0] << endl;
249  delete sel;
250 }

294 void TabulatedPotential::CalcForce_(vector<double> &U, vector<double> &F,
295  double dx, bool bPeriodic) {
296  size_t n = U.size();
297  double f = 0.5 / dx;
298  F.resize(n);
299  if (bPeriodic) {
300  F[n - 1] = F[0] = -(U[1] - U[n - 2]) * f;
301  } else {
302  F[0] = -(U[1] - U[0]) * 2 * f;
303  F[n - 1] = -(U[n - 1] - U[n - 2]) * 2 * f;
304  }
305  for (size_t i = 1; i < n - 1; i++) {
306  F[i] = -(U[i + 1] - U[i - 1]) * f;
307  }
308 }

310 void TabulatedPotential::Smooth_(vector<double> &data, bool bPeriodic) {
311  double old[3];
312  Index n = data.size();
313  if (bPeriodic) {
314  old[0] = data[n - 3];
315  old[1] = data[n - 2];
316  } else {
317  old[0] = data[0];
318  old[1] = data[0];
319  }
320  size_t i;
321  for (i = 0; i < data.size() - 2; i++) {
322  old[2] = data[i];
323  data[i] =
324  (old[0] + 2. * old[1] + 3. * data[i] + 2. * data[i + 1] + data[i + 2]) /
325  9.;
326  old[0] = old[1];
327  old[1] = old[2];
328  ;
329  }
330  if (bPeriodic) {
331  data[i] =
332  (old[0] + 2. * old[1] + 3. * data[i] + 2. * data[i + 1] + data[0]) / 9.;
333  old[0] = old[1];
334  old[1] = data[i];
335  data[n - 1] = data[0];
336  } else {
337  data[i] = (old[0] + 2. * old[1] + 3. * data[i] + 3. * data[i + 1]) / 9.;
338  old[0] = old[1];
339  old[1] = data[i];
340  i++;
341  data[i] = (old[0] + 2. * old[1] + 6. * data[i]) / 9.;
342  }
343 }

345 void TabulatedPotential::BoltzmannInvert_(vector<double> &data) {
346  double _min, _max;
347 
348  _min = numeric_limits<double>::max();
349  _max = numeric_limits<double>::min();
350 
351  for (double i : data) {
352  _max = max(i, _max);
353  if (i > 0) {
354  _min = min(i, _min);
355  }
356  }
357  _max = -conv::kB * conv::ev2kj_per_mol * _Temperature * log(_max);
358  _min = -conv::kB * conv::ev2kj_per_mol * _Temperature * log(_min) - _max;
359 
360  for (double &i : data) {
361  if (i == 0) {
362  i = _min;
363  } else {
364  i = -conv::kB * conv::ev2kj_per_mol * _Temperature * log(i) - _max;
365  }
366  }
367 }
