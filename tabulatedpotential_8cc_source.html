45 void TabulatedPotential::Command(BondedStatistics &bs, string cmd, vector<string> &args)
46 {
47  if(args[0] == "set") {
48  if(cmd == "hist") SetOption(_hist_options, args);
49  else if(cmd == "tab") {
50  if(!SetOption(_tab_options, args)) {
51  if(args.size() >2) {
52  if(args[1] == "smooth_pdf")
53  _tab_smooth1 = lexical_cast<int>(args[2]);
54  else if(args[1] == "smooth_pot")
55  _tab_smooth2 = lexical_cast<int>(args[2]);
56  else if(args[1] == "T")
57  _T = lexical_cast<double>(args[2]);
58  else {
59  cout << "unknown option " << args[2] << endl;
60  return;
61  }
62  }
63  }
64  if(args.size() <=2) {
65  cout << "smooth_pdf: " << _tab_smooth1 << endl;
66  cout << "smooth_pot: " << _tab_smooth2 << endl;
67  cout << "T: " << _T << endl;
68  }
69  }
70  }
71  else if(args.size() >= 2) {
72  if(cmd == "hist") WriteHistogram(bs, args);
73  else if(cmd == "tab") WritePotential(bs, args);
74  }
75  else cout << "wrong number of arguments" << endl;
76 }

242 void TabulatedPotential::CalcForce(vector<double> &U, vector<double> &F, double dx, bool bPeriodic)
243 {
244  size_t n=U.size();
245  double f = 0.5/dx;
246  F.resize(n);
247  if(bPeriodic)
248  F[n-1] = F[0] = -(U[1] - U[n-2])*f;
249  else {
250  F[0] = -(U[1] - U[0])*2*f;
251  F[n-1] = -(U[n-1] - U[n-2])*2*f;
252  }
253  for(size_t i=1; i<n-1; i++)
254  F[i] = -(U[i+1] - U[i-1])*f;
255 }

296 void TabulatedPotential::Smooth(vector<double> &data, bool bPeriodic)
297 {
298  double old[3];
299  int n=data.size();
300  if(bPeriodic) {
301  old[0] = data[n-3];
302  old[1] = data[n-2];
303  }
304  else {
305  old[0] = data[0];
306  old[1] = data[0];
307  }
308  size_t i;
309  for(i=0; i<data.size()-2; i++) {
310  old[2] = data[i];
311  data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 2.*data[i+1] + data[i+2])/9.;
312  old[0]=old[1];
313  old[1]=old[2];;
314  }
315  if(bPeriodic) {
316  data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 2.*data[i+1] + data[0])/9.;
317  old[0]=old[1];old[1]=data[i];
318  data[n-1] = data[0];
319  }
320  else {
321  data[i] = (old[0] + 2.*old[1] + 3.*data[i] + 3.*data[i+1])/9.;
322  old[0]=old[1];old[1]=data[i];
323  i++;
324  data[i] = (old[0] + 2.*old[1] + 6.*data[i])/9.;
325  }
326 }

328 void TabulatedPotential::BoltzmannInvert(vector<double> &data, double T)
329 {
330  double _min, _max;
331 
332  _min = numeric_limits<double>::max();
333  _max = numeric_limits<double>::min();
334 
335  for(size_t i=0; i<data.size(); i++) {
336  _max = max(data[i], _max);
337  if(data[i] > 0) _min = min(data[i], _min);
338  }
339  _max = -8.3109*T*log(_max)*0.001;
340  _min = -8.3109*T*log(_min)*0.001-_max;
341 
342  for(size_t i=0; i<data.size(); i++) {
343  if(data[i] == 0) data[i] = _min;
344  else
345  data[i] = -8.3109*T*log(data[i])*0.001 - _max;
346  }
347 }
