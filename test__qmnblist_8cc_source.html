32BOOST_AUTO_TEST_CASE(add_pair) {
33
34 Segment seg1("one", 1);
35 Atom atm1(1, "C", Eigen::Vector3d::Ones());
36 seg1.push_back(atm1);
37
38 Segment seg2("two", 2);
39 Atom atm2(2, "C", -Eigen::Vector3d::Ones());
40 seg2.push_back(atm2);
41
42 Segment seg3("three", 3);
43 Atom atm3(3, "Ca", -Eigen::Vector3d::UnitX());
44 seg3.push_back(atm3);
45
46 QMNBList qmnb;
47 BOOST_CHECK_EQUAL(qmnb.empty(), true);
48 qmnb.Add(seg1, seg3, 0.5 * Eigen::Vector3d::Ones());
49 qmnb.Add(seg1, seg2, -0.5 * Eigen::Vector3d::Ones());
50 BOOST_CHECK_EQUAL(qmnb.size(), 2);
51 BOOST_CHECK_EQUAL(qmnb.empty(), false);
52
53 QMPair* p0 = qmnb.FindPair(&seg1, &seg2);
54
55 QMPair* p1 = qmnb.FindPair(&seg1, &seg3);
56 BOOST_CHECK_EQUAL(p0->R().isApprox(-0.5 * Eigen::Vector3d::Ones(), 1e-5),
57 true);
58 BOOST_CHECK_EQUAL(p0->getId(), 1);
59 BOOST_CHECK_EQUAL(p1->getId(), 0);
60
61 // sort qmpairs by seg1id and then by seg2id then reindex the pair id
62 // according to that.
63 qmnb.sortAndReindex([](QMPair* a, QMPair* b) {
64 if (a->Seg1()->getId() != b->Seg1()->getId()) {
65 return a->Seg1()->getId() < b->Seg1()->getId();
66 }
67 return a->Seg2()->getId() < b->Seg2()->getId();
68 });
69
70 BOOST_CHECK_EQUAL(p0->getId(), 0);
71 BOOST_CHECK_EQUAL(p1->getId(), 1);
72
73 votca::csg::PairList<const Segment*, QMPair>::partners* Partners =
74 qmnb.FindPartners(&seg1);
75
76 QMPair* p_ref0 = Partners->at(&seg2);
77 QMPair* p_ref1 = Partners->at(&seg3);
78 BOOST_CHECK_EQUAL(p_ref0, p0);
79 BOOST_CHECK_EQUAL(p_ref1, p1);
80}
