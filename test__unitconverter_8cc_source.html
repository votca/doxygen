36 BOOST_AUTO_TEST_CASE(unitconverter_test_distance) {
37  UnitConverter converter;
38 
39  double distance = 1; // Assuming in meters
40  double distance_new =
41  converter.convert(DistanceUnit::meters, DistanceUnit::nanometers) *
42  distance;
43  BOOST_CHECK_EQUAL(1E9, distance_new);
44 
45  distance_new =
46  converter.convert(DistanceUnit::meters, DistanceUnit::centimeters) *
47  distance;
48  BOOST_CHECK_EQUAL(100, distance_new);
49 
50  distance_new =
51  converter.convert(DistanceUnit::meters, DistanceUnit::angstroms) *
52  distance;
53  BOOST_CHECK_EQUAL(1E10, distance_new);
54 
55  distance_new =
56  converter.convert(DistanceUnit::meters, DistanceUnit::bohr) * distance;
57  BOOST_CHECK_CLOSE(1.8897e+10, distance_new, 0.01);
58 
59  // Now assuming distance in angstroms
60  distance_new =
61  converter.convert(DistanceUnit::angstroms, DistanceUnit::nanometers) *
62  distance;
63  BOOST_CHECK_CLOSE(0.1, distance_new, 0.01);
64 
65  distance_new =
66  converter.convert(DistanceUnit::angstroms, DistanceUnit::meters) *
67  distance;
68  BOOST_CHECK_CLOSE(1.E-10, distance_new, 0.01);
69 
70  // Now assuming distance in Bohr
71  distance_new =
72  converter.convert(DistanceUnit::nanometers, DistanceUnit::bohr) *
73  distance;
74  BOOST_CHECK_CLOSE(18.8973, distance_new, 0.01);
75 }
