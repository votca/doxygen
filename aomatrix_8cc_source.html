33  template< class T>
34  void AOMatrix<T>::Fill(const AOBasis& aobasis) {
35  _aomatrix = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(aobasis.AOBasisSize(),aobasis.AOBasisSize());
36  // loop row
37 #pragma omp parallel for schedule(guided)
38  for (unsigned row = 0; row < aobasis.getNumofShells(); row++) {
39  const AOShell* shell_row = aobasis.getShell(row);
40  int row_start = shell_row->getStartIndex();
41 
42  // AOMatrix is symmetric, restrict explicit calculation to triangular matrix
43  for (unsigned col = row; col < aobasis.getNumofShells(); col++) {
44  const AOShell* shell_col = aobasis.getShell(col);
45  // figure out the submatrix
46  int col_start = shell_col->getStartIndex();
47  Eigen::Block< Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> > block=
48  _aomatrix.block(row_start,col_start, shell_row->getNumFunc(),shell_col->getNumFunc());
49  // Fill block
50  FillBlock(block, shell_row, shell_col);
51  }
52  }
53  // Fill whole matrix by copying
54  for ( unsigned i=0; i < _aomatrix.rows(); i++){
55  for (unsigned j = 0; j < i; j++) {
56  _aomatrix(i, j) = _aomatrix(j, i);
57  }
58  }
59 
60  return;
61  }

64  void AOMatrix3D::Fill(const AOBasis& aobasis) {
65  _aomatrix.resize(3);
66  for (int i = 0; i < 3; i++) {
67  _aomatrix[ i ] = Eigen::MatrixXd::Zero(aobasis.AOBasisSize(),aobasis.AOBasisSize());
68  }
69  // loop row
70 #pragma omp parallel for
71  for (unsigned row = 0; row < aobasis.getNumofShells(); row++) {
72  const AOShell* shell_row = aobasis.getShell(row);
73  int row_start = shell_row->getStartIndex();
74  // loop column
75  for (const AOShell* shell_col:aobasis) {
76  // figure out the submatrix
77  int col_start = shell_col->getStartIndex();
78  std::vector< Eigen::Block<Eigen::MatrixXd> > submatrix;
79  for (int i = 0; i < 3; i++) {
80  Eigen::Block<Eigen::MatrixXd> block=_aomatrix[i].block( row_start,col_start,shell_row->getNumFunc(),shell_col->getNumFunc());
81  submatrix.push_back(block );
82  }
83  // Fill block
84  FillBlock(submatrix, shell_row, shell_col);
85  }
86  }
87  return;
88  }

370  std::vector<double> AOMatrix<T>::XIntegrate(int size, double U) {
371  std::vector<double> FmU = std::vector<double>(size, 0.0);
372  const int mm = FmU.size() - 1;
373  const double pi = boost::math::constants::pi<double>();
374  if (mm < 0) {
375  std::cerr << "mm is: " << mm << " This should not have happened!" << std::flush;
376  exit(1);
377  }
378 
379  if (U < 0.0) {
380  std::cerr << "U is: " << U << " This should not have happened!" << std::flush;
381  exit(1);
382  }
383 
384  if (U >= 10.0) {
385  // forward iteration
386  FmU[0] = 0.50 * sqrt(pi / U) * erf(sqrt(U));
387 
388  for (unsigned m = 1; m < FmU.size(); m++) {
389  FmU[m] = (2.0 * m - 1) * FmU[m - 1] / (2.0 * U) - exp(-U) / (2.0 * U);
390  }
391  }
392 
393  if (U < 1e-10) {
394  for (unsigned m = 0; m < FmU.size(); m++) {
395  FmU[m] = 1.0 / (2.0 * m + 1.0) - U / (2.0 * m + 3.0);
396  }
397  }
398 
399 
400  if (U >= 1e-10 && U < 10.0) {
401  // backward iteration
402  double fm = 0.0;
403  for (int m = 60; m >= mm; m--) {
404  fm = (2.0 * U) / (2.0 * m + 1.0) * (fm + exp(-U) / (2.0 * U));
405  }
406  FmU[mm] = fm;
407  for (int m = mm - 1; m >= 0; m--) {
408  FmU[m] = (2.0 * U) / (2.0 * m + 1.0) * (FmU[m + 1] + exp(-U) / (2.0 * U));
409  }
410  }
411 
412  return FmU;
413  }

415 int AOSuperMatrix::getBlockSize(int lmax) {
416  //Each cartesian shells has (l+1)(l+2)/2 elements
417  //Sum of all shells up to _lmax leads to blocksize=1+11/6 l+l^2+1/6 l^3
418  int blocksize = 6 + 11 * lmax + 6 * lmax * lmax + lmax * lmax*lmax;
419  blocksize /= 6;
420  return blocksize;
421  }
