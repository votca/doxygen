34 template <libint2::Operator obtype,
35  typename OperatorParams =
36  typename libint2::operator_traits<obtype>::oper_params_type>
37 std::array<MatrixLibInt, libint2::operator_traits<obtype>::nopers>
38  computeOneBodyIntegrals(const AOBasis& aobasis,
39  OperatorParams oparams = OperatorParams()) {
40 
41  Index nthreads = OPENMP::getMaxThreads();
42  std::vector<libint2::Shell> shells = aobasis.GenerateLibintBasis();
43 
44  std::vector<std::vector<Index>> shellpair_list = aobasis.ComputeShellPairs();
45 
46  Index nopers = static_cast<Index>(libint2::operator_traits<obtype>::nopers);
47  std::array<MatrixLibInt, libint2::operator_traits<obtype>::nopers> result;
48  for (MatrixLibInt& r : result) {
49  r = MatrixLibInt::Zero(aobasis.AOBasisSize(), aobasis.AOBasisSize());
50  }
51 
52  std::vector<libint2::Engine> engines(nthreads);
53  engines[0] = libint2::Engine(obtype, aobasis.getMaxNprim(),
54  static_cast<int>(aobasis.getMaxL()), 0);
55  engines[0].set_params(oparams);
56  for (Index i = 1; i < nthreads; ++i) {
57  engines[i] = engines[0];
58  }
59 
60  std::vector<Index> shell2bf = aobasis.getMapToBasisFunctions();
61 
62 #pragma omp parallel for schedule(dynamic)
63  for (Index s1 = 0; s1 < aobasis.getNumofShells(); ++s1) {
64  Index thread_id = OPENMP::getThreadId();
65  libint2::Engine& engine = engines[thread_id];
66  const libint2::Engine::target_ptr_vec& buf = engine.results();
67 
68  Index bf1 = shell2bf[s1];
69  Index n1 = shells[s1].size();
70 
71  for (Index s2 : shellpair_list[s1]) {
72 
73  engine.compute(shells[s1], shells[s2]);
74  if (buf[0] == nullptr) {
75  continue; // if all integrals screened out, skip to next shell set
76  }
77  Index bf2 = shell2bf[s2];
78  Index n2 = shells[s2].size();
79  for (unsigned int op = 0; op != nopers; ++op) {
80  Eigen::Map<const MatrixLibInt> buf_mat(buf[op], n1, n2);
81  result[op].block(bf1, bf2, n1, n2) = buf_mat;
82  if (s1 != s2) { // if s1 >= s2, copy {s1,s2} to the corresponding
83  // {s2,s1} block, note the transpose!
84  result[op].block(bf2, bf1, n2, n1) = buf_mat.transpose();
85  }
86  }
87  }
88  }
89  return result;
90 }

106 Eigen::MatrixXd AOOverlap::singleShellOverlap(const AOShell& shell) const {
107  libint2::Shell::do_enforce_unit_normalization(false);
108  libint2::Operator obtype = libint2::Operator::overlap;
109  try {
110  libint2::Engine engine(obtype, shell.getSize(),
111  static_cast<int>(shell.getL()), 0);
112 
113  const libint2::Engine::target_ptr_vec& buf = engine.results();
114 
115  libint2::Shell s = shell.LibintShell();
116  engine.compute(s, s);
117 
118  Eigen::Map<const MatrixLibInt> buf_mat(buf[0], shell.getNumFunc(),
119  shell.getNumFunc());
120  return buf_mat;
121  } catch (const libint2::Engine::lmax_exceeded& error) {
122  std::ostringstream oss;
123  oss << "\nA libint error occured:\n"
124  << error.what() << "\n"
125  << "\nYou requested a computation for a shell with angular momentum "
126  << error.lmax_requested()
127  << ",\nbut your libint package only supports angular momenta upto "
128  << error.lmax_limit() - 1 << ".\n"
129  << "\nTo fix this error you will need to reinstall libint with "
130  "support\n"
131  "for higher angular momenta. If you installed your own libint it\n"
132  "should be reconfigured and installed with the option "
133  "--with-max-am=<maxAngularMomentum>.\n"
134  "If you installed libint with your OS package manager, you will\n"
135  "need to setup you own libint installation with the \n"
136  "--with-max-am=<maxAngularMomentum> option set."
137  << std::endl;
138  throw std::runtime_error(oss.str());

142 Eigen::MatrixXd AOOverlap::Pseudo_InvSqrt(double etol) {
143  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(aomatrix_);
144  smallestEigenvalue = es.eigenvalues()(0);
145  Eigen::VectorXd diagonal = Eigen::VectorXd::Zero(es.eigenvalues().size());
146  removedfunctions = 0;
147  for (Index i = 0; i < diagonal.size(); ++i) {
148  if (es.eigenvalues()(i) < etol) {
149  removedfunctions++;
150  } else {
151  diagonal(i) = 1.0 / std::sqrt(es.eigenvalues()(i));
152  }
153  }
154 
155  return es.eigenvectors() * diagonal.asDiagonal() *
156  es.eigenvectors().transpose();
157 }

172 void AOCoulomb::computeCoulombIntegrals(const AOBasis& aobasis) {
173  Index nthreads = OPENMP::getMaxThreads();
174  std::vector<libint2::Shell> shells = aobasis.GenerateLibintBasis();
175  std::vector<Index> shell2bf = aobasis.getMapToBasisFunctions();
176 
177  aomatrix_ =
178  Eigen::MatrixXd::Zero(aobasis.AOBasisSize(), aobasis.AOBasisSize());
179 
180  // build engines for each thread
181  std::vector<libint2::Engine> engines(nthreads);
182  engines[0] =
183  libint2::Engine(libint2::Operator::coulomb, aobasis.getMaxNprim(),
184  static_cast<int>(aobasis.getMaxL()), 0);
185  engines[0].set(libint2::BraKet::xs_xs);
186  for (Index i = 1; i != nthreads; ++i) {
187  engines[i] = engines[0];
188  }
189 
190 #pragma omp parallel for schedule(dynamic)
191  for (Index s1 = 0; s1 < aobasis.getNumofShells(); ++s1) {
192  libint2::Engine& engine = engines[OPENMP::getThreadId()];
193  const libint2::Engine::target_ptr_vec& buf = engine.results();
194 
195  Index bf1 = shell2bf[s1];
196  Index n1 = shells[s1].size();
197  // cannot use shellpairs because this is still a two-center integral and
198  // overlap screening would give wrong result
199  for (Index s2 = 0; s2 <= s1; ++s2) {
200 
201  engine.compute2<libint2::Operator::coulomb, libint2::BraKet::xs_xs, 0>(
202  shells[s1], libint2::Shell::unit(), shells[s2],
203  libint2::Shell::unit());
204 
205  if (buf[0] == nullptr) {
206  continue; // if all integrals screened out, skip to next shell set
207  }
208  Index bf2 = shell2bf[s2];
209  Index n2 = shells[s2].size();
210 
211  Eigen::Map<const MatrixLibInt> buf_mat(buf[0], n1, n2);
212  aomatrix_.block(bf1, bf2, n1, n2) = buf_mat;
213  if (s1 != s2) { // if s1 >= s2, copy {s1,s2} to the corresponding
214  // {s2,s1} block, note the transpose!
215  aomatrix_.block(bf2, bf1, n2, n1) = buf_mat.transpose();
216  }
217  }
218  }
219 }

221 // This converts V into ((S-1/2 V S-1/2)-1/2 S-1/2)T, which is needed to
222 // construct 4c integrals,
223 Eigen::MatrixXd AOCoulomb::Pseudo_InvSqrt_GWBSE(const AOOverlap& auxoverlap,
224  double etol) {
225 
226  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> eo(auxoverlap.Matrix());
227  removedfunctions = 0;
228  Eigen::VectorXd diagonal_overlap =
229  Eigen::VectorXd::Zero(eo.eigenvalues().size());
230  for (Index i = 0; i < diagonal_overlap.size(); ++i) {
231  if (eo.eigenvalues()(i) < etol) {
232  removedfunctions++;
233  } else {
234  diagonal_overlap(i) = 1.0 / std::sqrt(eo.eigenvalues()(i));
235  }
236  }
237  Eigen::MatrixXd Ssqrt = eo.eigenvectors() * diagonal_overlap.asDiagonal() *
238  eo.eigenvectors().transpose();
239 
240  Eigen::MatrixXd ortho = Ssqrt * aomatrix_ * Ssqrt;
241  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(ortho);
242  Eigen::VectorXd diagonal = Eigen::VectorXd::Zero(es.eigenvalues().size());
243 
244  for (Index i = 0; i < diagonal.size(); ++i) {
245  if (es.eigenvalues()(i) < etol) {
246  removedfunctions++;
247  } else {
248  diagonal(i) = 1.0 / std::sqrt(es.eigenvalues()(i));
249  }
250  }
251 
252  Eigen::MatrixXd Vm1 =
253  es.eigenvectors() * diagonal.asDiagonal() * es.eigenvectors().transpose();
254  Eigen::MatrixXd result = (Vm1 * Ssqrt).transpose();
255  return result;
256 }
