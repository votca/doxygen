45  void AOMatrix::Fill(const AOBasis& aobasis,vec r, AOBasis* ecp ) {
46  _aomatrix = ub::zero_matrix<double>(aobasis.AOBasisSize());
47  _gridpoint = r;
48  // loop row
49  #pragma omp parallel for
50  for (unsigned _row = 0; _row < aobasis.getNumofShells() ; _row++ ){
51 
52  const AOShell* _shell_row = aobasis.getShell( _row );
53  int _row_start = _shell_row->getStartIndex();
54  int _row_end = _row_start + _shell_row->getNumFunc();
55 
56  // AOMatrix is symmetric, restrict explicit calculation to triangular matrix
57  for ( unsigned _col = 0; _col <= _row ; _col++ ){
58 
59  const AOShell* _shell_col = aobasis.getShell( _col );
60 
61  // figure out the submatrix
62  int _col_start = _shell_col->getStartIndex();
63  int _col_end = _col_start + _shell_col->getNumFunc();
64 
65  ub::matrix_range< ub::matrix<double> > _submatrix = ub::subrange(_aomatrix, _row_start, _row_end, _col_start, _col_end);
66 
67  // Fill block
68  FillBlock( _submatrix, _shell_row, _shell_col, ecp );
69 
70  }
71  }
72 
73  // Fill whole matrix by copying
74  for ( unsigned _i=0; _i < _aomatrix.size1(); _i++){
75  for ( unsigned _j=0; _j < _i; _j++){
76  _aomatrix(_j,_i) = _aomatrix(_i,_j);
77 
78  }
79  }
80 
81  // check symmetry
82  bool _is_symmetric = true;
83 
84  // Copy stuff to fill lower triangular part
85  for ( unsigned _i=0; _i < this->_aomatrix.size1(); _i++){
86  for (unsigned _j=0; _j <= _i; _j++){
87 
88  if ( std::abs(this->_aomatrix(_i,_j) - this->_aomatrix(_j,_i) ) > 1e-4 ) {
89 
90  cerr << _i << ":" << _j << " == " << this->_aomatrix(_i,_j) << " vs " << this->_aomatrix(_j,_i) << endl;
91  _is_symmetric = false;
92  }
93 
94  }
95  }
96  if ( !_is_symmetric) {cerr << " Error: AOMatrix is not symmetric! "; exit(1);}
97 
98  return;
99  }

459  std::vector<double> AOMatrix::XIntegrate(int _n, double _T ){
460  std::vector<double> _FmT=std::vector<double>(_n,0.0);
461  const int _mm = _FmT.size() - 1;
462  const double pi = boost::math::constants::pi<double>();
463  if ( _mm < 0){
464  cerr << "mm is: " << _mm << " This should not have happened!" << flush;
465  exit(1);
466  }
467 
468  if ( _T < 0.0 ) {
469  cerr << "T is: " << _T << " This should not have happened!" << flush;
470  exit(1);
471  }
472 
473  if ( _T >= 10.0 ) {
474  // forward iteration
475  _FmT[0]=0.50*sqrt(pi/_T)* erf(sqrt(_T));
476 
477  for (unsigned m = 1; m < _FmT.size(); m++ ){
478  _FmT[m] = (2*m-1) * _FmT[m-1]/(2.0*_T) - exp(-_T)/(2.0*_T) ;
479  }
480  }
481 
482  if ( _T < 1e-10 ){
483  for ( unsigned m=0; m < _FmT.size(); m++){
484  _FmT[m] = 1.0/(2.0*m+1.0) - _T/(2.0*m+3.0);
485  }
486  }
487 
488 
489  if ( _T >= 1e-10 && _T < 10.0 ){
490  // backward iteration
491  double fm = 0.0;
492  for ( int m = 60; m >= _mm; m--){
493  fm = (2.0*_T)/(2.0*m+1.0) * ( fm + exp(-_T)/(2.0*_T));
494  }
495  _FmT[_mm] = fm;
496  for (int m = _mm-1 ; m >= 0; m--){
497  _FmT[m] = (2.0*_T)/(2.0*m+1.0) * (_FmT[m+1] + exp(-_T)/(2.0*_T));
498  }
499  }
500 
501  return _FmT;
502  }

505  int AOSuperMatrix::getBlockSize( int _lmax){
506  int _block_size;
507  if ( _lmax == 0 ) { _block_size = 1 ;} // s
508  else if ( _lmax == 1 ) { _block_size = 4 ;} // p
