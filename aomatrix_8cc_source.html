291  void AOMatrix::XIntegrate(vector<double>& _FmT, const double& _T ){
292 
293  const int _mm = _FmT.size() - 1;
294  const double pi = boost::math::constants::pi<double>();
295  if ( _mm < 0 || _mm > 10){
296  cerr << "mm is: " << _mm << " This should not have happened!" << flush;
297  exit(1);
298  }
299 
300  if ( _T < 0.0 ) {
301  cerr << "T is: " << _T << " This should not have happened!" << flush;
302  exit(1);
303  }
304 
305  if ( _T >= 10.0 ) {
306  // forward iteration
307  _FmT[0]=0.50*sqrt(pi/_T)* erf(sqrt(_T));
308 
309  for (unsigned m = 1; m < _FmT.size(); m++ ){
310  _FmT[m] = (2*m-1) * _FmT[m-1]/(2.0*_T) - exp(-_T)/(2.0*_T) ;
311  }
312  }
313 
314  if ( _T < 1e-10 ){
315  for ( unsigned m=0; m < _FmT.size(); m++){
316  _FmT[m] = 1.0/(2.0*m+1.0) - _T/(2.0*m+3.0);
317  }
318  }
319 
320 
321  if ( _T >= 1e-10 && _T < 10.0 ){
322  // backward iteration
323  double fm = 0.0;
324  for ( int m = 60; m >= _mm; m--){
325  fm = (2.0*_T)/(2.0*m+1.0) * ( fm + exp(-_T)/(2.0*_T));
326  }
327  _FmT[_mm] = fm;
328  for (int m = _mm-1 ; m >= 0; m--){
329  _FmT[m] = (2.0*_T)/(2.0*m+1.0) * (_FmT[m+1] + exp(-_T)/(2.0*_T));
330  }
331  }
