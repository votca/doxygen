304 Eigen::MatrixXd AOTransform::getTrafo(const AOGaussianPrimitive& gaussian) {
305 
306  const AOShell& shell = gaussian.getShell();
307  if (!shell.isCombined()) {
308  return AOTransform::getPrimitiveShellTrafo(
309  shell.getLmax(), gaussian.getDecay(),
310  gaussian.getContraction()(shell.getLmax()));
311  }
312 
313  Eigen::MatrixXd trafo =
314  Eigen::MatrixXd::Zero(shell.getCartesianNumFunc(), shell.getNumFunc());
315 
316  int rowoffset = 0;
317  int coloffset = 0;
318  for (int l = shell.getLmin(); l <= shell.getLmax(); l++) {
319  int cart_size = AOTransform::getCartesianSize(l);
320  int spherical_size = AOTransform::getSphericalSize(l);
321  trafo.block(rowoffset, coloffset, cart_size, spherical_size) =
322  AOTransform::getPrimitiveShellTrafo(l, gaussian.getDecay(),
323  gaussian.getContraction()(l));
324  rowoffset += cart_size;
325  coloffset += spherical_size;
326  }
327  return trafo;
328 }

330 Eigen::VectorXd AOTransform::XIntegrate(int size, double U) {
331  Eigen::VectorXd FmU = Eigen::VectorXd::Zero(size);
332  const int mm = size - 1;
333  const double pi = boost::math::constants::pi<double>();
334  if (mm < 0) {
335  throw std::runtime_error("mm is: " + std::to_string(mm) +
336  " This should not have happened!");
337  }
338 
339  if (U < 0.0) {
340  throw std::runtime_error("U is: " + std::to_string(U) +
341  " This should not have happened!");
342  }
343 
344  if (U >= 10.0) {
345  // forward iteration
346  FmU[0] = 0.50 * std::sqrt(pi / U) * std::erf(std::sqrt(U));
347 
348  const double expU = std::exp(-U);
349  for (unsigned m = 1; m < FmU.size(); m++) {
350  FmU[m] = (2.0 * m - 1) * FmU[m - 1] / (2.0 * U) - expU / (2.0 * U);
351  }
352  }
353 
354  else if (U < 1e-10) {
355  for (unsigned m = 0; m < FmU.size(); m++) {
356  FmU[m] = 1.0 / (2.0 * m + 1.0) - U / (2.0 * m + 3.0);
357  }
358  }
359 
360  else if (U >= 1e-10 && U < 10.0) {
361  // backward iteration
362  double fm = 0.0;
363  const double expU = std::exp(-U);
364  for (int m = 60; m >= mm; m--) {
365  fm = (2.0 * U) / (2.0 * m + 1.0) * (fm + expU / (2.0 * U));
366  }
367  FmU[mm] = fm;
368  for (int m = mm - 1; m >= 0; m--) {
369  FmU[m] = (2.0 * U) / (2.0 * m + 1.0) * (FmU[m + 1] + expU / (2.0 * U));
370  }
371  }
372 
373  return FmU;
374 }

376 int AOTransform::getBlockSize(int lmax) {
377  // Each cartesian shells has (l+1)(l+2)/2 elements
378  // Sum of all shells up to _lmax leads to blocksize=1+11/6 l+l^2+1/6 l^3
379  int blocksize = 6 + 11 * lmax + 6 * lmax * lmax + lmax * lmax * lmax;
380  blocksize /= 6;
381  return blocksize;
