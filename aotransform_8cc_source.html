26double AOTransform::getNorm(L l, const AOGaussianPrimitive& gaussian) {
27 const double contraction = gaussian.getContraction();
28 const double decay = gaussian.getDecay();
29 switch (l) {
30 case L::S: {
31 return contraction;
32 }
33 case L::P: {
34 return 2. * std::sqrt(decay) * contraction;
35 }
36 case L::D: {
37 return 4. * decay * contraction / std::sqrt(3);
38 }
39 case L::F: {
40 return 8 * std::pow(decay, 1.5) * contraction / std::sqrt(5 * 3);
41 }
42 case L::G: {
43 return decay * decay * contraction * 16.0 / std::sqrt(5. * 3. * 7.);
44 }
45 default:
46 throw std::runtime_error("No norms for shells higher than g (l=4)");
47 }
48
49 return 0;
50}

53template <typename Matrix>
54Matrix AOTransform::tform(L l_row, L l_col, const Matrix& cartesian) {
55 const auto& coefs_row =
56 libint2::solidharmonics::SolidHarmonicsCoefficients<double>::instance(
57 int(l_row));
58 const auto& coefs_col =
59 libint2::solidharmonics::SolidHarmonicsCoefficients<double>::instance(
60 int(l_col));
61 int npure_row = 2 * int(l_row) + 1;
62 int npure_col = 2 * int(l_col) + 1;
63 Matrix spherical = Matrix::Zero(npure_row, npure_col);
64 // loop over row shg
65 for (auto s1 = 0; s1 != npure_row; ++s1) {
66 const auto nc1 =
67 coefs_row.nnz(s1); // # of cartesians contributing to shg s1
68 const auto* c1_idxs =
69 coefs_row.row_idx(s1); // indices of cartesians contributing to shg s1
70 const auto* c1_vals = coefs_row.row_values(
71 s1); // coefficients of cartesians contributing to shg s1
72 // loop over col shg
73 for (auto s2 = 0; s2 != npure_col; ++s2) {
74 const auto nc2 =
75 coefs_col.nnz(s2); // # of cartesians contributing to shg s2
76 const auto* c2_idxs = coefs_col.row_idx(s2); // indices of cartesians
77 // contributing to shg s2
78 const auto* c2_vals = coefs_col.row_values(
79 s2); // coefficients of cartesians contributing to shg s2
80 for (size_t ic1 = 0; ic1 != nc1;
81 ++ic1) { // loop over contributing cartesians
82 auto c1 = c1_idxs[ic1];
83 auto s1_c1_coeff = c1_vals[ic1];
84 for (size_t ic2 = 0; ic2 != nc2;
85 ++ic2) { // loop over contributing cartesians
86 auto c2 = c2_idxs[ic2];
87 auto s2_c2_coeff = c2_vals[ic2];
88 spherical(s1, s2) += cartesian(c1, c2) * s1_c1_coeff * s2_c2_coeff;
89 } // cart2
90
91 } // cart1
92
93 } // shg2
94 }
95 return spherical;
96}

102Eigen::VectorXd AOTransform::XIntegrate(Index size, double U) {
103
104 Eigen::VectorXd FmU = Eigen::VectorXd::Zero(size);
105 const Index mm = size - 1;
106 const double pi = boost::math::constants::pi<double>();
107 if (mm < 0) {
108 throw std::runtime_error("mm is: " + std::to_string(mm) +
109 " This should not have happened!");
110 }
111
112 if (U < 0.0) {
113 throw std::runtime_error("U is: " + std::to_string(U) +
114 " This should not have happened!");
115 }
116
117 if (U >= 10.0) {
118 // forward iteration
119 FmU[0] = 0.50 * std::sqrt(pi / U) * std::erf(std::sqrt(U));
120
121 const double expU = std::exp(-U);
122 for (Index m = 1; m < FmU.size(); m++) {
123 FmU[m] =
124 (2.0 * double(m) - 1) * FmU[m - 1] / (2.0 * U) - expU / (2.0 * U);
125 }
126 }
127
128 else if (U < 1e-10) {
129 for (Index m = 0; m < FmU.size(); m++) {
130 FmU[m] = 1.0 / (2.0 * double(m) + 1.0) - U / (2.0 * double(m) + 3.0);
131 }
132 }
133
134 else if (U >= 1e-10 && U < 10.0) {
135 // backward iteration
136 double fm = 0.0;
137 const double expU = std::exp(-U);
138 for (Index m = 60; m >= mm; m--) {
139 fm = (2.0 * U) / (2.0 * double(m) + 1.0) * (fm + expU / (2.0 * U));
140 }
141 FmU[mm] = fm;
142 for (Index m = mm - 1; m >= 0; m--) {
143 FmU[m] =
144 (2.0 * U) / (2.0 * double(m) + 1.0) * (FmU[m + 1] + expU / (2.0 * U));
145 }
146 }
147
148 return FmU;
149}

151Index AOTransform::getBlockSize(Index lmax) {
152 // Each cartesian shells has (l+1)(l+2)/2 elements
153 // Sum of all shells up to lmax_ leads to blocksize=1+11/6 l+l^2+1/6 l^3
154 Index blocksize = 6 + 11 * lmax + 6 * lmax * lmax + lmax * lmax * lmax;
155 blocksize /= 6;
156 return blocksize;
157}
