90 bool QMAnalyze::Evaluate() {
91 
92  _log.setReportLevel( logDEBUG );
93  _log.setMultithreading( true );
94 
95  _log.setPreface(logINFO, "\n... ...");
96  _log.setPreface(logERROR, "\n... ...");
97  _log.setPreface(logWARNING, "\n... ...");
98  _log.setPreface(logDEBUG, "\n... ...");
99 
100  LOG(logDEBUG, _log) << "Analyzing serialized QM data in " << _orbfile << flush;
101 
102  Orbitals _orbitals;
103  // load the QM data from serialized orbitals object
104 
105  std::ifstream ifs( (_orbfile).c_str());
106  LOG(logDEBUG, _log) << " Loading QM data from " << _orbfile << flush;
107  boost::archive::binary_iarchive ia(ifs);
108  ia >> _orbitals;
109  ifs.close();
110 
111  // output info about contents of serialized data
112  CheckContent( _orbitals );
113 
114 
115 
116 
117  // LOG(logDEBUG, _log) << "Written text data to " << _output_file << flush;
118 
119 
120  return true;
121 }

126 void QMAnalyze::CheckContent( Orbitals& _orbitals ){
127 
128 
129 
130  LOG(logDEBUG, _log) << "===== Summary of serialized content ===== " << flush;
131  LOG(logDEBUG, _log) << " Information about DFT:" << flush;
132 
133 
134  // DFT atoms
135  if ( _orbitals.hasQMAtoms() ) {
136  LOG(logDEBUG, _log) << " atoms: " << _orbitals.QMAtoms().size() << flush;
137  } else {
138  LOG(logDEBUG, _log) << " atoms: not stored "<< flush;
139  }
140 
141  // QM package
142  if ( _orbitals.hasQMpackage() ){
143  LOG(logDEBUG, _log) << " QM package: " << _orbitals.getQMpackage() << flush;
144 
145  } else {
146  LOG(logDEBUG, _log) << " QM package: not stored " << flush;
147  }
148 
149 
150  // DFT basis set
151  if ( _orbitals.hasDFTbasis() ) {
152  LOG(logDEBUG, _log) << " basis set: " << _orbitals.getDFTbasis() << flush;
153  } else {
154  LOG(logDEBUG, _log) << " basis set: not stored "<< flush;
155  }
156 
157  // DFT basis set size
158  if ( _orbitals.hasBasisSetSize() ) {
159  LOG(logDEBUG, _log) << " basis set size: " << _orbitals.getBasisSetSize() << flush;
160 
161  } else {
162  LOG(logDEBUG, _log) << " basis set size: not stored "<< flush;
163  }
164 
165  // DFT number of electrons
166  if ( _orbitals.hasNumberOfElectrons() ) {
167  LOG(logDEBUG, _log) << " number of electrons: " << _orbitals.getNumberOfElectrons() << flush;
168  } else {
169  LOG(logDEBUG, _log) << " number of electrons: not stored "<< flush;
170  }
171 
172  // DFT number of levels
173  if ( _orbitals.hasNumberOfLevels() ) {
174  LOG(logDEBUG, _log) << " number of levels: " << _orbitals.getNumberOfLevels() << flush;
175  } else {
176  LOG(logDEBUG, _log) << " number of levels: not stored "<< flush;
177  }
178 
179  // DFT orbital energies
180  if ( _orbitals.hasMOEnergies() ) {
181  LOG(logDEBUG, _log) << " MO energies: " << _orbitals.getEnergies()->size() << flush;
182  } else {
183  LOG(logDEBUG, _log) << " MO energies: not stored "<< flush;
184  }
185 
186  // DFT orbital coefficients
187  if ( _orbitals.hasMOCoefficients() ) {
188  LOG(logDEBUG, _log) << " MO coefficients: " << _orbitals.MOCoefficients().size1() << " x " << _orbitals.MOCoefficients().size2() << flush;
189  } else {
190  LOG(logDEBUG, _log) << " MO coefficients: not stored "<< flush;
191  }
192 
193  // DFT AO overlap matrix
194  if ( _orbitals.hasAOOverlap() ) {
195  LOG(logDEBUG, _log) << " AO overlap matrix: " << _orbitals.getOverlap()->size1() << " x " << _orbitals.getOverlap()->size2() << flush;
196  } else {
197  LOG(logDEBUG, _log) << " AO overlap matrix: not stored "<< flush;
198  }
199 
200  // DFT AO XC matrix
201  if ( _orbitals.hasAOVxc() ) {
202  LOG(logDEBUG, _log) << " AO XC matrix: " << _orbitals.AOVxc().size1() << " x " << _orbitals.AOVxc().size2() << flush;
203  } else {
204  LOG(logDEBUG, _log) << " AO XC matrix: not stored "<< flush;
205  }
206 
207  // QM total energy
208  if ( _orbitals.hasQMEnergy() ){
209  LOG(logDEBUG, _log) << " QM energy: " << _orbitals.getQMEnergy() << flush;
210  } else{
211  LOG(logDEBUG, _log) << " QM energy: not stored " << flush;
212  }
213 
214  // MM self-energy
215  if ( _orbitals.hasSelfEnergy() ){
216  LOG(logDEBUG, _log) << " MM self energy: " << _orbitals.getSelfEnergy() << flush;
217  } else{
218  LOG(logDEBUG, _log) << " MM self energy: not stored " << flush;
219  }
220 
221  // DFT transfer integrals
222  if ( _orbitals.hasMOCouplings() ) {
223  LOG(logDEBUG, _log) << " DFT transfer integrals: " << _orbitals.MOCouplings().size1() << " x " << _orbitals.MOCouplings().size2() << flush;
224 
225  } else {
226  LOG(logDEBUG, _log) << " DFT transfer integrals: not stored "<< flush;
227  }
