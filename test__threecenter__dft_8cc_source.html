38BOOST_AUTO_TEST_CASE(small_basis) {
39 libint2::initialize();
40 QMMolecule mol(" ", 0);
41 mol.LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
42 "/threecenter_dft/molecule.xyz");
43 BasisSet basis;
44 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/threecenter_dft/3-21G.xml");
45 AOBasis aobasis;
46 aobasis.Fill(basis, mol);
47 TCMatrix_dft threec;
48 threec.Fill(aobasis, aobasis);
49
50 Eigen::MatrixXd Ref0 = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
51 std::string(XTP_TEST_DATA_FOLDER) + "/threecenter_dft/Ref0.mm");
52
53 Eigen::MatrixXd Ref4 = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
54 std::string(XTP_TEST_DATA_FOLDER) + "/threecenter_dft/Ref4.mm");
55
56 bool check_three1 = Ref0.isApprox(threec[0].FullMatrix(), 0.00001);
57 if (!check_three1) {
58 std::cout << "Res0" << std::endl;
59 std::cout << threec[0].FullMatrix() << std::endl;
60 std::cout << "0_ref" << std::endl;
61 std::cout << Ref0 << std::endl;
62 }
63 BOOST_CHECK_EQUAL(check_three1, true);
64 bool check_three2 = Ref4.isApprox(threec[4].FullMatrix(), 0.00001);
65 if (!check_three2) {
66 std::cout << "Res4" << std::endl;
67 std::cout << threec[4].FullMatrix() << std::endl;
68 std::cout << "4_ref" << std::endl;
69 std::cout << Ref4 << std::endl;
70 }
71 BOOST_CHECK_EQUAL(check_three2, true);
72
73 libint2::finalize();
74}
