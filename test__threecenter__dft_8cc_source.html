30 Eigen::MatrixXd ReadMatrixFromString(const std::string& matrix) {
31  votca::tools::Tokenizer lines(matrix, "\n");
32 
33  std::vector<double> entries;
34  int cols = 0;
35  int rows = 0;
36  for (auto line : lines) {
37  if (line[0] == '#') {
38  continue;
39  }
40  votca::tools::Tokenizer entries_tok(line, " ");
41  std::vector<std::string> temp = entries_tok.ToVector();
42  cols = temp.size();
43  rows++;
44  for (const auto& s : temp) {
45  entries.push_back(std::stod(s));
46  }
47  }
48 
49  return Eigen::Map<Eigen::MatrixXd>(entries.data(), rows, cols);
50 }
