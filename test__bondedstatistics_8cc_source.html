38 BOOST_AUTO_TEST_CASE(test_bondedstatistics_begin) {
39  Topology top;
40  // Create two bonded interactions
41  string interaction_group = "covalent_bond1";
42  string interaction_group_compare = ":covalent_bond1";
43  auto bond1 = new IBond(0, 1);
44  bond1->setGroup(interaction_group);
45 
46  string interaction_group2 = "covalent_bond2";
47  string interaction_group_compare2 = ":covalent_bond2";
48  auto bond2 = new IBond(1, 2);
49  bond2->setGroup(interaction_group2);
50 
51  top.AddBondedInteraction(bond1);
52  top.AddBondedInteraction(bond2);
53 
54  BondedStatistics bonded_statistics;
55  bonded_statistics.BeginCG(&top, nullptr);
56 
57  DataCollection<double> &data_collection = bonded_statistics.BondedValues();
58  vector<DataCollection<double>::array *> &vector_of_arrays =
59  data_collection.Data();
60  BOOST_CHECK_EQUAL(vector_of_arrays.size(), 2);
61  BOOST_CHECK_EQUAL(vector_of_arrays.at(0)->getName(),
62  interaction_group_compare);
63  BOOST_CHECK_EQUAL(vector_of_arrays.at(1)->getName(),
64  interaction_group_compare2);
65  // The arrays do not store any numbers at this point
66  BOOST_CHECK_EQUAL(vector_of_arrays.at(0)->size(), 0);
67  BOOST_CHECK_EQUAL(vector_of_arrays.at(1)->size(), 0);
68  top.Cleanup();
