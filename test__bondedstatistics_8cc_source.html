43BOOST_AUTO_TEST_CASE(test_bondedstatistics_begin) {
44 Topology top;
45 // Create two bonded interactions
46 string interaction_group = "covalent_bond1";
47 string interaction_group_compare = ":covalent_bond1";
48 auto bond1 = new IBond(0, 1);
49 bond1->setGroup(interaction_group);
50
51 string interaction_group2 = "covalent_bond2";
52 string interaction_group_compare2 = ":covalent_bond2";
53 auto bond2 = new IBond(1, 2);
54 bond2->setGroup(interaction_group2);
55
56 top.AddBondedInteraction(bond1);
57 top.AddBondedInteraction(bond2);
58
59 BondedStatistics bonded_statistics;
60 bonded_statistics.BeginCG(&top, nullptr);
61
62 DataCollection<double> &data_collection = bonded_statistics.BondedValues();
63 vector<DataCollection<double>::array *> &vector_of_arrays =
64 data_collection.Data();
65 BOOST_CHECK_EQUAL(vector_of_arrays.size(), 2);
66 BOOST_CHECK_EQUAL(vector_of_arrays.at(0)->getName(),
67 interaction_group_compare);
68 BOOST_CHECK_EQUAL(vector_of_arrays.at(1)->getName(),
69 interaction_group_compare2);
70 // The arrays do not store any numbers at this point
71 BOOST_CHECK_EQUAL(vector_of_arrays.at(0)->size(), 0);
72 BOOST_CHECK_EQUAL(vector_of_arrays.at(1)->size(), 0);
73 top.Cleanup();
74}

98 bead_ptr->setId(0);
99 bead_ptr->setPos(pos_bead1);
100
101 string bead_name2 = "bead_test2";
102 Eigen::Vector3d pos_bead2(5.0, 4.0, 5.0);
103 auto bead_ptr2 = top.CreateBead(Bead::spherical, bead_name2, bead_type_name,
104 residue_number, mass, charge);
105 bead_ptr2->setId(1);
106 bead_ptr2->setPos(pos_bead2);
107
108 string bead_name3 = "bead_test3";
109 Eigen::Vector3d pos_bead3(5.0, 6.0, 5.0);
110 auto bead_ptr3 = top.CreateBead(Bead::spherical, bead_name3, bead_type_name,
111 residue_number, mass, charge);
112 bead_ptr3->setId(2);
113 bead_ptr3->setPos(pos_bead3);
114
115 // Create two bonded interactions
116 string interaction_group = "covalent_bond1";
117 auto bond1 = new IBond(0, 1);
118 bond1->setGroup(interaction_group);
119 string interaction_group2 = "covalent_bond2";
120 auto bond2 = new IBond(1, 2);
121 bond2->setGroup(interaction_group2);
122
123 top.AddBondedInteraction(bond1);
124 top.AddBondedInteraction(bond2);
125 }
126
127 BondedStatistics bonded_statistics;
128 bonded_statistics.BeginCG(&top, nullptr);
129
130 // Calling EvalConfiguration on IBond structures will store the distances
131 // between the beads in the BondedStatitics class
132 bonded_statistics.EvalConfiguration(&top, nullptr);
133
134 DataCollection<double> &data_collection = bonded_statistics.BondedValues();
135 vector<DataCollection<double>::array *> &vector_of_arrays =
136 data_collection.Data();
137
138 // Distance between bead 0 and bead 1
139 BOOST_CHECK_EQUAL(vector_of_arrays.at(0)->at(0), 1.0);
140 // Distance between bead 1 and bead 2
141 BOOST_CHECK_EQUAL(vector_of_arrays.at(1)->at(0), 2.0);
142
143 top.Cleanup();
144}
