58 void CGForceMatching::BeginEvaluate(Topology *top, Topology *top_atom)
59 {
60  // set counters to zero value:
61  _nblocks = 0;
62  _line_cntr = _col_cntr = 0;
63 
64  // Number of CG beads in topology
65  _nbeads = top->BeadCount();
66  // Set frame counter to zero
67  _frame_counter = 0;
68 
69  // accuracy for evaluating the difference in bead positions (default 1e-5)
70  _dist = 1e-5;
71  if (_options.exists("cg.fmatch.dist")) {
72  _dist = _options.get("cg.fmatch.dist").as<double>();
73  }
74  //std::cout << "_dist: " << _dist << std::endl;
75 
76  // read _nframes from input file
77  _nframes = _options.get("cg.fmatch.frames_per_block").as<int>();
78  // read _constr_least_sq from input file
79  _constr_least_sq = _options.get("cg.fmatch.constrainedLS").as<bool>();
80 
81  // initializing bonded interactions
82  for (list<Property*>::iterator iter = _bonded.begin();
83  iter != _bonded.end(); ++iter) {
84  SplineInfo *i = new SplineInfo(_splines.size(), true, _col_cntr, *iter);
85  //adjust initial matrix dimensions:
86  _line_cntr += i->num_gridpoints;
87  _col_cntr += 2 * i->num_gridpoints;
88  //if periodic potential, one additional constraint has to be taken into account -> 1 additional line in matrix
89  if(i->periodic != 0){
90  _line_cntr += 1;
91  }
92 
93  // add spline to container
94  _splines.push_back(i);
95  }
96 
97  // initializing non-bonded interactions
98  for (list<Property*>::iterator iter = _nonbonded.begin();
99  iter != _nonbonded.end(); ++iter) {
100  SplineInfo *i = new SplineInfo(_splines.size(), false, _col_cntr, *iter);
101  //adjust initial matrix dimensions:
102  _line_cntr += i->num_gridpoints;
103  _col_cntr += 2 * i->num_gridpoints;
104 
105  // add spline to container
106  _splines.push_back(i);
107  }
108 
109  cout << "\nYou are using VOTCA!\n";
110  cout << "\nhey, somebody wants to forcematch!\n";
111 
112 
113  // now initialize _A, _b, _x and probably _B_constr
114  // depending on least-squares algorithm used
115  if (_constr_least_sq) { // Constrained Least Squares
116 
117  cout << "\nUsing constrained Least Squares!\n " << endl;
118 
119  // assign _least_sq_offset
120  _least_sq_offset = 0;
121 
122  // resize and clear _B_constr
123  _B_constr=Eigen::MatrixXd::Zero(_line_cntr, _col_cntr);
124 
125  // resize matrix _A
126  _A=Eigen::MatrixXd::Zero(3 * _nbeads *_nframes, _col_cntr);
127  // resize vector _b
128  _b=Eigen::VectorXd::Zero(3 * _nbeads *_nframes);
129 
130  // in case of constrained least squares smoothing conditions
131  // are assigned to matrix _B_constr
132  FmatchAssignSmoothCondsToMatrix(_B_constr);
133  } else { // Simple Least Squares
134 
135  cout << "\nUsing simple Least Squares! " << endl;
136  // assign _least_sq_offset
137  _least_sq_offset = _line_cntr;
138 
139  // resize matrix _A
140  _A=Eigen::MatrixXd::Zero(_line_cntr + 3 * _nbeads *_nframes, _col_cntr);
141  // resize vector _b
142  _b=Eigen::VectorXd::Zero(_line_cntr + 3 * _nbeads *_nframes);
143 
144  // in case of simple least squares smoothing conditions
145  // are assigned to matrix _A
146  FmatchAssignSmoothCondsToMatrix(_A);
147  // clear _b (only necessary in simple least squares)
148  _b.setZero();
149  }
150  // resize and clear _x
151  _x=Eigen::VectorXd::Zero(_col_cntr);
152 
153  if(_has_existing_forces) {
154  _top_force.CopyTopologyData(top);
155  _trjreader_force = TrjReaderFactory().Create(_op_vm["trj-force"].as<string>());

161  _trjreader_force->FirstFrame(_top_force);

232 void CGForceMatching::WriteOutFiles()
233 {
234  string file_extension = ".force";
235  string file_name;
236  Table force_tab;
237 
238  // table with error column
239  force_tab.SetHasYErr(true);
240 
241  SplineContainer::iterator is;
242 
243  for (is = _splines.begin(); is != _splines.end(); ++is) {
244  // construct meaningful outfile name
245  file_name = (*is)->splineName;
246  file_name = file_name + file_extension;
247 
248  // resize table
249  force_tab.resize((*is)->num_outgrid);
250 
251  // print output file names on stdout
252  cout << "Updating file: " << file_name << endl;
253 
254  // loop over output grid points
255  for (int i = 0; i < (*is)->num_outgrid; i++) {
256  // average value
257  (*is)->result[i] = (*is)->resSum[i] / _nblocks;
258  // standard deviation of the average
259  (*is)->error[i] = sqrt( ((*is)->resSum2[i] / _nblocks - (*is)->result[i] * (*is)->result[i])/_nblocks );
260  }
261 
262  // first output point = first grid point
263  double out_x = (*is)->Spline.getGridPoint(0);
264  // loop over output grid
265  for (int i = 0; i < (*is)->num_outgrid; i++) {
266  // put point, result, flag and accuracy at point out_x into the table
267  force_tab.set(i, out_x, (-1.0) * (*is)->result[i], 'i', (*is)->error[i]);
268  // update out_x for the next iteration
269  out_x += (*is)->dx_out;
270  }
271  // save table in the file
272  force_tab.Save(file_name);
273  // clear the table for the next spline
274  force_tab.clear();
275  }

278 void CGForceMatching::EvalConfiguration(Topology *conf, Topology *conf_atom)
279 {
280  SplineContainer::iterator spiter;
281  if(conf->BeadCount() == 0)
282  throw std::runtime_error("CG Topology has 0 beads, check your mapping file!");
283  if(_has_existing_forces) {
284  if(conf->BeadCount() != _top_force.BeadCount())
285  throw std::runtime_error("number of beads in topology and force topology does not match");
286  for(int i=0; i<conf->BeadCount(); ++i) {
287 // cout << "conf->getBead(" << i << ")->getPos(): " << conf->getBead(i)->getPos() << endl;
288 // cout << "_top_force->getBead(" << i << ")->getPos(): " << _top_force.getBead(i)->getPos() << endl;
289  conf->getBead(i)->F() -= _top_force.getBead(i)->getF();
290  vec d = conf->getBead(i)->getPos() - _top_force.getBead(i)->getPos();
291 // cout << "vec d of bead " << i << ": " << d << "abs(d): " << abs(d) << endl;
292  if(abs(d) > _dist)//default is 1e-5, otherwise it can be a too strict criterion
293 // cout << "conf->getBead(" << i << ")->getPos(): " << conf->getBead(i)->getPos() << endl;
294 // cout << "_top_force->getBead(" << i << ")->getPos(): " << _top_force.getBead(i)->getPos() << endl;
295  throw std::runtime_error("One or more bead positions in mapped and reference force trajectory differ by more than 1e-5");
296  }
297  }
298 
299  for (spiter = _splines.begin(); spiter != _splines.end(); ++spiter) {
300  SplineInfo *sinfo = *spiter;
301  if (sinfo->bonded) // bonded interaction
302  EvalBonded(conf, sinfo);
303  else // non-bonded interaction
304  EvalNonbonded(conf, sinfo);
305  }
306 
307  // loop for the forces vector:
308  // hack, chage the Has functions..
309  if (conf->getBead(0)->HasF()) {
310  vec Force(0., 0., 0.);
311  for (int iatom = 0; iatom < _nbeads; ++iatom) {
312  Force = conf->getBead(iatom)->getF();
313  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + iatom) = Force.x();
314  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + iatom) = Force.y();
315  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + iatom) = Force.z();
316  }
317  } else {
318  cerr << "\nERROR in csg_fmatch::EvalConfiguration - No forces in configuration!\n" << endl;
319  exit(-1);
320  }
321  // update the frame counter
322  _frame_counter += 1;
323 
324  if (_frame_counter % _nframes == 0) { // at this point we processed _nframes frames, which is enough for one block
325  // update block counter
326  _nblocks++;
327  // solve FM equations and accumulate the result
328  FmatchAccumulateData();
329  // print status information
330  cout << "\nBlock No" << _nblocks << " done!" << endl;
331  // write results to output files
332  WriteOutFiles();
333 
334  // we must count frames from zero again for the next block
335  _frame_counter = 0;
336  if (_constr_least_sq) { //Constrained Least Squares
337  // Matrices should be cleaned after each block is evaluated
338  _A.setZero();
339  _b.setZero();
340  // clear and assign smoothing conditions to _B_constr
341  FmatchAssignSmoothCondsToMatrix(_B_constr);
342  } else { // Simple Least Squares
343  // Matrices should be cleaned after each block is evaluated
344  // clear and assign smoothing conditions to _A
345  FmatchAssignSmoothCondsToMatrix(_A);
346  _b.setZero();
347  }
348  }
349  if(_has_existing_forces)
350  _trjreader_force->NextFrame(_top_force);
351 }

353 void CGForceMatching::FmatchAccumulateData()
354 {
355  if (_constr_least_sq) { // Constrained Least Squares
356  // Solving linear equations system
357  votca::tools::linalg_constrained_qrsolve(_x, _A, _b, _B_constr);
358  } else { // Simple Least Squares
359 
360  Eigen::HouseholderQR<Eigen::MatrixXd> dec(_A);
361  _x=dec.solve(_b);
362  Eigen::VectorXd residual=_b-_A*_x;
363  // calculate FM residual - quality of FM
364  // FM residual is initially calculated in (kJ/(mol*nm))^2
365  double fm_resid = residual.cwiseAbs2().sum();
366 
367  // strange number is units conversion -> now (kcal/(mol*angstrom))^2
368  fm_resid /= 3 * _nbeads * _frame_counter * 1750.5856;
369 
370  cout << endl;
371  cout << "#### Force matching residual ####" << endl;
372  cout << " Chi_2 = " << fm_resid << endl;
373  cout << "#################################" << endl;
374  cout << endl;
375  }
376 
377  SplineContainer::iterator is;
378  for (is = _splines.begin(); is != _splines.end(); ++is) {
379  int &mp = (*is)->matr_pos;
380  int &ngp = (*is)->num_gridpoints;
381 
382  // _x contains results for all splines. Here we cut the results for one spline
383  for (int i = 0; i < ngp; i++) {
384  (*is)->block_res_f[i] = _x[ i + mp ];
385  (*is)->block_res_f2[i] = _x[ i + mp + ngp];
386  }
387  // result cutted before is assigned to the corresponding spline
388  (*is)->Spline.setSplineData((*is)->block_res_f, (*is)->block_res_f2);
389 
390  // first output point = first grid point
391  double out_x = (*is)->Spline.getGridPoint(0);
392 
393  // point in the middle of the output grid for printing debug information
394  int grid_point_debug = (*is)->num_outgrid / 2;
395 
396  // loop over output grid
397  for (int i = 0; i < (*is)->num_outgrid; i++) {
398  // update resSum (add result of a particular block)
399  (*is)->resSum[i] += (*is)->Spline.Calculate(out_x);
400  // print useful debug information
401  if (i == grid_point_debug) cout << "This should be a number: " << (*is)->Spline.Calculate(out_x) << " " << endl;
402  // update resSum2 (add result of a particular block)
403  (*is)->resSum2[i] += (*is)->Spline.Calculate(out_x) * (*is)->Spline.Calculate(out_x);
404  // output point for the next iteration
405  out_x += (*is)->dx_out;
406  }
407  }
408 }

410 void CGForceMatching::FmatchAssignSmoothCondsToMatrix(Eigen::MatrixXd &Matrix)
411 {
412 // This function assigns Spline smoothing conditions to the Matrix.
413 // For the simple least squares the function is used for matrix _A
414 // For constrained least squares - for matrix _B_constr
415  int line_tmp, col_tmp;
416  line_tmp = 0;
417  col_tmp = 0;
418 
419  Matrix.setZero();
420 
421 
422  SplineContainer::iterator is;
423  for (is = _splines.begin(); is != _splines.end(); ++is) {
424  int sfnum = (*is)->num_splinefun;
425  (*is)->Spline.AddBCToFitMatrix(Matrix, line_tmp, col_tmp);
426  //if periodic potential, one additional constraint has to be taken into account!
427  if ((*is)->periodic != 0){
428  (*is)->Spline.AddBCSumZeroToFitMatrix(Matrix, line_tmp, col_tmp);
429  //update counter
430  line_tmp += 1;
431  }
432  // update counters
433  line_tmp += sfnum + 1;
434  col_tmp += 2 * (sfnum + 1);
435  }
436 }

445 void CGForceMatching::EvalBonded(Topology *conf, SplineInfo *sinfo)
446 {
447  std::list<Interaction *> interList;
448  std::list<Interaction *>::iterator interListIter;
449 
450  interList = conf->InteractionsInGroup(sinfo->splineName);
451 
452  for (interListIter = interList.begin(); interListIter != interList.end(); ++interListIter) {
453 
454  int beads_in_int = (*interListIter)->BeadCount(); // 2 for bonds, 3 for angles, 4 for dihedrals
455 
456  CubicSpline &SP = sinfo->Spline;
457 
458  int &mpos = sinfo->matr_pos;
459 
460  double var = (*interListIter)->EvaluateVar(*conf); // value of bond, angle, or dihedral
461 
462  for (int loop = 0; loop < beads_in_int; loop++) {
463  int ii = (*interListIter)->getBeadId(loop);
464  vec gradient = (*interListIter)->Grad(*conf, loop);
465 
466  SP.AddToFitMatrix(_A, var,
467  _least_sq_offset + 3 * _nbeads * _frame_counter + ii, mpos, -gradient.x());
468  SP.AddToFitMatrix(_A, var,
469  _least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + ii, mpos, -gradient.y());
470  SP.AddToFitMatrix(_A, var,
471  _least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + ii, mpos, -gradient.z());
472  }
