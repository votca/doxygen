61 void CGForceMatching::BeginEvaluate(Topology *top, Topology *top_atom)
62 {
63  // set counters to zero value:
64  _nblocks = 0;
65  _line_cntr = _col_cntr = 0;
66 
67  // Number of CG beads in topology
68  _nbeads = top->BeadCount();
69  // Set frame counter to zero
70  _frame_counter = 0;
71 
72  // accuracy for evaluating the difference in bead positions (default 1e-5)
73  _dist = 1e-5;
74  if (_options.exists("cg.fmatch.dist")) {
75  _dist = _options.get("cg.fmatch.dist").as<double>();
76  }
77  //std::cout << "_dist: " << _dist << std::endl;
78 
79  // read _nframes from input file
80  _nframes = _options.get("cg.fmatch.frames_per_block").as<int>();
81  // read _constr_least_sq from input file
82  _constr_least_sq = _options.get("cg.fmatch.constrainedLS").as<bool>();
83 
84  // initializing bonded interactions
85  for (list<Property*>::iterator iter = _bonded.begin();
86  iter != _bonded.end(); ++iter) {
87  SplineInfo *i = new SplineInfo(_splines.size(), true, _col_cntr, *iter);
88  //adjust initial matrix dimensions:
89  _line_cntr += i->num_gridpoints;
90  _col_cntr += 2 * i->num_gridpoints;
91  //if periodic potential, one additional constraint has to be taken into account -> 1 additional line in matrix
92  if(i->periodic != 0){
93  _line_cntr += 1;
94  }
95 
96  // add spline to container
97  _splines.push_back(i);
98  }
99 
100  // initializing non-bonded interactions
101  for (list<Property*>::iterator iter = _nonbonded.begin();
102  iter != _nonbonded.end(); ++iter) {
103  SplineInfo *i = new SplineInfo(_splines.size(), false, _col_cntr, *iter);
104  //adjust initial matrix dimensions:
105  _line_cntr += i->num_gridpoints;
106  _col_cntr += 2 * i->num_gridpoints;
107 
108  // add spline to container
109  _splines.push_back(i);
110  }
111 
112  cout << "\nYou are using VOTCA!\n";
113  cout << "\nhey, somebody wants to forcematch!\n";
114 
115 
116  // now initialize _A, _b, _x and probably _B_constr
117  // depending on least-squares algorithm used
118  if (_constr_least_sq) { // Constrained Least Squares
119 
120  cout << "\nUsing constrained Least Squares!\n " << endl;
121 
122  // assign _least_sq_offset
123  _least_sq_offset = 0;
124 
125  // resize and clear _B_constr
126  _B_constr.resize(_line_cntr, _col_cntr, false);
127  _B_constr.clear();
128 
129  // resize matrix _A
130  _A.resize(3 * _nbeads *_nframes, _col_cntr, false);
131  // resize vector _b
132  _b.resize(3 * _nbeads *_nframes, false);
133 
134  // in case of constrained least squares smoothing conditions
135  // are assigned to matrix _B_constr
136  FmatchAssignSmoothCondsToMatrix(_B_constr);
137  } else { // Simple Least Squares
138 
139  cout << "\nUsing simple Least Squares! " << endl;
140  // assign _least_sq_offset
141  _least_sq_offset = _line_cntr;
142 
143  // resize matrix _A
144  _A.resize(_line_cntr + 3 * _nbeads *_nframes, _col_cntr, false);
145  // resize vector _b
146  _b.resize(_line_cntr + 3 * _nbeads *_nframes, false);
147 
148  // in case of simple least squares smoothing conditions
149  // are assigned to matrix _A
150  FmatchAssignSmoothCondsToMatrix(_A);
151  // clear _b (only necessary in simple least squares)
152  _b.clear();
153  }
154  // resize and clear _x
155  _x.resize(_col_cntr);
156  _x.clear();
157 
158  if(_has_existing_forces) {
159  _top_force.CopyTopologyData(top);
160  _trjreader_force = TrjReaderFactory().Create(_op_vm["trj-force"].as<string>());
161  if(_trjreader_force == NULL)
162  throw runtime_error(string("input format not supported: ") + _op_vm["trj-force"].as<string>());
163  // open the trajectory
164  _trjreader_force->Open(_op_vm["trj-force"].as<string>());
165  // read in first frame
166  _trjreader_force->FirstFrame(_top_force);
167  }

233 void CGForceMatching::WriteOutFiles()
234 {
235  // sanity check
236  if (_nblocks == 0) {
237  cerr << "\nERROR in csg_fmatch::EndCG - No blocks have been processed so far" << endl;
238  cerr << "It might be that you are using trajectory, which is smaller than needed for one block" << endl;
239  cerr << "Check your input!" << endl;
240  exit(-1);
241  }
242 
243  string file_extension = ".force";
244  string file_name;
245  Table force_tab;
246 
247  // table with error column
248  force_tab.SetHasYErr(true);
249 
250  SplineContainer::iterator is;
251 
252  for (is = _splines.begin(); is != _splines.end(); ++is) {
253  // construct meaningful outfile name
254  file_name = (*is)->splineName;
255  file_name = file_name + file_extension;
256 
257  // resize table
258  force_tab.resize((*is)->num_outgrid, false);
259 
260  // print output file names on stdout
261  cout << "Updating file: " << file_name << endl;
262 
263  // loop over output grid points
264  for (int i = 0; i < (*is)->num_outgrid; i++) {
265  // average value
266  (*is)->result[i] = (*is)->resSum[i] / _nblocks;
267  // standard deviation of the average
268  (*is)->error[i] = sqrt( ((*is)->resSum2[i] / _nblocks - (*is)->result[i] * (*is)->result[i])/_nblocks );
269  }
270 
271  // first output point = first grid point
272  double out_x = (*is)->Spline.getGridPoint(0);
273  // loop over output grid
274  for (int i = 0; i < (*is)->num_outgrid; i++) {
275  // put point, result, flag and accuracy at point out_x into the table
276  force_tab.set(i, out_x, (-1.0) * (*is)->result[i], 'i', (*is)->error[i]);
277  // update out_x for the next iteration
278  out_x += (*is)->dx_out;
279  }
280  // save table in the file
281  force_tab.Save(file_name);
282  // clear the table for the next spline
283  force_tab.clear();
284  }

287 void CGForceMatching::EvalConfiguration(Topology *conf, Topology *conf_atom)
288 {
289  SplineContainer::iterator spiter;
290  if(_has_existing_forces) {
291  if(conf->BeadCount() != _top_force.BeadCount())
292  throw std::runtime_error("number of beads in topology and force topology does not match");
293  for(int i=0; i<conf->BeadCount(); ++i) {
294 // cout << "conf->getBead(" << i << ")->getPos(): " << conf->getBead(i)->getPos() << endl;
295 // cout << "_top_force->getBead(" << i << ")->getPos(): " << _top_force.getBead(i)->getPos() << endl;
296  conf->getBead(i)->F() -= _top_force.getBead(i)->getF();
297  vec d = conf->getBead(i)->getPos() - _top_force.getBead(i)->getPos();
298 // cout << "vec d of bead " << i << ": " << d << "abs(d): " << abs(d) << endl;
299  if(abs(d) > _dist)//default is 1e-5, otherwise it can be a too strict criterion
300 // cout << "conf->getBead(" << i << ")->getPos(): " << conf->getBead(i)->getPos() << endl;
301 // cout << "_top_force->getBead(" << i << ")->getPos(): " << _top_force.getBead(i)->getPos() << endl;
302  throw std::runtime_error("One or more bead positions in mapped and reference force trajectory differ by more than 1e-5");
303  }
304  }
305 
306  for (spiter = _splines.begin(); spiter != _splines.end(); ++spiter) {
307  SplineInfo *sinfo = *spiter;
308  if (sinfo->bonded) // bonded interaction
309  EvalBonded(conf, sinfo);
310  else // non-bonded interaction
311  EvalNonbonded(conf, sinfo);
312  }
313 
314  // loop for the forces vector:
315  // hack, chage the Has functions..
316  if (conf->getBead(0)->HasF()) {
317  vec Force(0., 0., 0.);
318  for (int iatom = 0; iatom < _nbeads; ++iatom) {
319  Force = conf->getBead(iatom)->getF();
320  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + iatom) = Force.x();
321  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + iatom) = Force.y();
322  _b(_least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + iatom) = Force.z();
323  }
324  } else {
325  cerr << "\nERROR in csg_fmatch::EvalConfiguration - No forces in configuration!\n" << endl;
326  exit(-1);
327  }
328  // update the frame counter
329  _frame_counter += 1;
330 
331  if (_frame_counter % _nframes == 0) { // at this point we processed _nframes frames, which is enough for one block
332  // update block counter
333  _nblocks++;
334  // solve FM equations and accumulate the result
335  FmatchAccumulateData();
336  // print status information
337  cout << "\nBlock No" << _nblocks << " done!" << endl;
338  // write results to output files
339  WriteOutFiles();
340 
341  // we must count frames from zero again for the next block
342  _frame_counter = 0;
343  if (_constr_least_sq) { //Constrained Least Squares
344  // Matrices should be cleaned after each block is evaluated
345  _A.clear();
346  _b.clear();
347  // clear and assign smoothing conditions to _B_constr
348  FmatchAssignSmoothCondsToMatrix(_B_constr);
349  } else { // Simple Least Squares
350  // Matrices should be cleaned after each block is evaluated
351  // clear and assign smoothing conditions to _A
352  FmatchAssignSmoothCondsToMatrix(_A);
353  _b.clear();
354  }
355  }
356  if(_has_existing_forces)
357  _trjreader_force->NextFrame(_top_force);
358 }

360 void CGForceMatching::FmatchAccumulateData()
361 {
362  _x.clear();
363  if (_constr_least_sq) { // Constrained Least Squares
364  // Solving linear equations system
365  ub::matrix<double> B_constr = _B_constr;
366  votca::tools::linalg_constrained_qrsolve(_x, _A, _b, B_constr);
367  _x = -_x;
368 
369 
370  } else { // Simple Least Squares
371 
372  ub::vector<double> residual(_b.size());
373  votca::tools::linalg_qrsolve(_x, _A, _b, &residual);
374 
375  // calculate FM residual - quality of FM
376  // FM residual is initially calculated in (kJ/(mol*nm))^2
377  double fm_resid = 0;
378 
379  for (size_t i = 0; i < _b.size(); i++)
380  fm_resid += residual(i) * residual(i);
381 
382  // strange number is units conversion -> now (kcal/(mol*angstrom))^2
383  fm_resid /= 3 * _nbeads * _frame_counter * 1750.5856;
384 
385  cout << endl;
386  cout << "#### Force matching residual ####" << endl;
387  cout << " Chi_2 = " << fm_resid << endl;
388  cout << "#################################" << endl;
389  cout << endl;
390  }
391 
392  SplineContainer::iterator is;
393  for (is = _splines.begin(); is != _splines.end(); ++is) {
394  int &mp = (*is)->matr_pos;
395  int &ngp = (*is)->num_gridpoints;
396 
397  // _x contains results for all splines. Here we cut the results for one spline
398  for (int i = 0; i < ngp; i++) {
399  (*is)->block_res_f[i] = _x[ i + mp ];
400  (*is)->block_res_f2[i] = _x[ i + mp + ngp];
401  }
402  // result cutted before is assigned to the corresponding spline
403  (*is)->Spline.setSplineData((*is)->block_res_f, (*is)->block_res_f2);
404 
405  // first output point = first grid point
406  double out_x = (*is)->Spline.getGridPoint(0);
407 
408  // point in the middle of the output grid for printing debug information
409  int grid_point_debug = (*is)->num_outgrid / 2;
410 
411  // loop over output grid
412  for (int i = 0; i < (*is)->num_outgrid; i++) {
413  // update resSum (add result of a particular block)
414  (*is)->resSum[i] += (*is)->Spline.Calculate(out_x);
415  // print useful debug information
416  if (i == grid_point_debug) cout << "This should be a number: " << (*is)->Spline.Calculate(out_x) << " " << endl;
417  // update resSum2 (add result of a particular block)
418  (*is)->resSum2[i] += (*is)->Spline.Calculate(out_x) * (*is)->Spline.Calculate(out_x);
419  // output point for the next iteration
420  out_x += (*is)->dx_out;
421  }
422  }
423 }

425 void CGForceMatching::FmatchAssignSmoothCondsToMatrix(ub::matrix<double> &Matrix)
426 {
427 // This function assigns Spline smoothing conditions to the Matrix.
428 // For the simple least squares the function is used for matrix _A
429 // For constrained least squares - for matrix _B_constr
430  int line_tmp, col_tmp;
431  line_tmp = 0;
432  col_tmp = 0;
433 
434  Matrix.clear();
435 
436 
437  SplineContainer::iterator is;
438  for (is = _splines.begin(); is != _splines.end(); ++is) {
439  int sfnum = (*is)->num_splinefun;
440  (*is)->Spline.AddBCToFitMatrix(Matrix, line_tmp, col_tmp);
441  //if periodic potential, one additional constraint has to be taken into account!
442  if ((*is)->periodic != 0){
443  (*is)->Spline.AddBCSumZeroToFitMatrix(Matrix, line_tmp, col_tmp);
444  //update counter
445  line_tmp += 1;
446  }
447  // update counters
448  line_tmp += sfnum + 1;
449  col_tmp += 2 * (sfnum + 1);
450  }
451 }

460 void CGForceMatching::EvalBonded(Topology *conf, SplineInfo *sinfo)
461 {
462  std::list<Interaction *> interList;
463  std::list<Interaction *>::iterator interListIter;
464 
465  interList = conf->InteractionsInGroup(sinfo->splineName);
466 
467  for (interListIter = interList.begin(); interListIter != interList.end(); ++interListIter) {
468 
469  int beads_in_int = (*interListIter)->BeadCount(); // 2 for bonds, 3 for angles, 4 for dihedrals
470 
471  CubicSpline &SP = sinfo->Spline;
472 
473  int &mpos = sinfo->matr_pos;
474 
475  double var = (*interListIter)->EvaluateVar(*conf); // value of bond, angle, or dihedral
476 
477  for (int loop = 0; loop < beads_in_int; loop++) {
478  int ii = (*interListIter)->getBeadId(loop);
479  vec gradient = (*interListIter)->Grad(*conf, loop);
480 
481  SP.AddToFitMatrix(_A, var,
482  _least_sq_offset + 3 * _nbeads * _frame_counter + ii, mpos, -gradient.x());
483  SP.AddToFitMatrix(_A, var,
484  _least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + ii, mpos, -gradient.y());
485  SP.AddToFitMatrix(_A, var,
486  _least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + ii, mpos, -gradient.z());
487  }
488  }

491 void CGForceMatching::EvalNonbonded(Topology *conf, SplineInfo *sinfo)
492 {
493  // generate the neighbour list
494  // generate the neighbour list
495  NBList *nb;
496 
497  bool gridsearch=false;
498 
499  if(_options.exists("cg.nbsearch")) {
500  if(_options.get("cg.nbsearch").as<string>() == "grid")
501  gridsearch=true;
502  else if(_options.get("cg.nbsearch").as<string>() == "simple")
503  gridsearch=false;
504  else throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
505  }
506  if(gridsearch)
507  nb = new NBListGrid();
508  else
509  nb = new NBList();
510 
511  nb->setCutoff(sinfo->_options->get("fmatch.max").as<double>()); // implement different cutoffs for different interactions!
512 
513  // generate the bead lists
514  BeadList beads1, beads2;
515  beads1.Generate(*conf, sinfo->type1);
516  beads2.Generate(*conf, sinfo->type2);
517 
518  // is it same types or different types?
519  if (sinfo->type1 == sinfo->type2)
520  nb->Generate(beads1, true);
521  else
522  nb->Generate(beads1, beads2, true);
523 
524  NBList::iterator pair_iter;
525  // iterate over all pairs
526  for (pair_iter = nb->begin(); pair_iter != nb->end(); ++pair_iter) {
527  int iatom = (*pair_iter)->first->getId();
528  int jatom = (*pair_iter)->second->getId();
529  double var = (*pair_iter)->dist();
530  vec gradient = (*pair_iter)->r();
531  gradient.normalize();
532 
533  CubicSpline &SP = sinfo->Spline;
534 
535  int &mpos = sinfo->matr_pos;
536 
537  // add iatom
538  SP.AddToFitMatrix(_A, var,
539  _least_sq_offset + 3 * _nbeads * _frame_counter + iatom, mpos, gradient.x());
540  SP.AddToFitMatrix(_A, var,
541  _least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + iatom, mpos, gradient.y());
542  SP.AddToFitMatrix(_A, var,
543  _least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + iatom, mpos, gradient.z());
544 
545  // add jatom
546  SP.AddToFitMatrix(_A, var,
547  _least_sq_offset + 3 * _nbeads * _frame_counter + jatom, mpos, -gradient.x());
548  SP.AddToFitMatrix(_A, var,
549  _least_sq_offset + 3 * _nbeads * _frame_counter + _nbeads + jatom, mpos, -gradient.y());
550  SP.AddToFitMatrix(_A, var,
551  _least_sq_offset + 3 * _nbeads * _frame_counter + 2 * _nbeads + jatom, mpos, -gradient.z());
552  }
553  delete nb;
