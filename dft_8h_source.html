158 
159  if (_reporting == "silent") _log.setReportLevel(ctp::logERROR); // only output ERRORS, GEOOPT info, and excited state info for trial geometry
160  if (_reporting == "noisy") _log.setReportLevel(ctp::logDEBUG); // OUTPUT ALL THE THINGS
161  if (_reporting == "default") _log.setReportLevel(ctp::logINFO); //
162 
163  _log.setMultithreading(true);
164  _log.setPreface(ctp::logINFO, "\n... ...");
165  _log.setPreface(ctp::logERROR, "\n... ...");
166  _log.setPreface(ctp::logWARNING, "\n... ...");
167  _log.setPreface(ctp::logDEBUG, "\n... ...");
168 
169  //TLogLevel _ReportLevel = _log.getReportLevel( ); // backup report level
170

173 
174  if (_do_guess) {

176  _orbitals.Load(_guess_file);
177 
178  } else {
179  CTP_LOG(ctp::logDEBUG, _log) << "Reading structure from " << _xyzfile << flush;
180  _orbitals.LoadFromXYZ(_xyzfile);
181  }
182 
183 
184 
185  // initialize the DFTENGINE
186  DFTENGINE _dft;
187  _dft.Initialize(&_dftengine_options);
188  _dft.setLogger(&_log);
189  ;
190 
191  if (_do_external) {
192  CTP_LOG (ctp::logDEBUG, _log) << " Let's create the background " << flush;
193  vector<ctp::APolarSite*> sites = ctp::APS_FROM_MPS(_mpsfile, 0);
194  std::vector<ctp::PolarSeg*> polar_segments;
195  //ctp::PolarSeg *thisPolarSegment = NULL;
196  ctp::PolarSeg *newPolarSegment = new ctp::PolarSeg(0, sites);
197  polar_segments.push_back(newPolarSegment);
198  polar_segments[0]->WriteMPS("test.mps", "test");
199  _dft.setExternalcharges(polar_segments);
200  }
201 
202  // RUN
203  _dft.Prepare(&_orbitals);
204  _dft.Evaluate(&_orbitals);
205 
206 
207  CTP_LOG(ctp::logDEBUG, _log) << "Saving data to " << _output_file << flush;
208  std::ofstream ofs((_output_file).c_str());
209  boost::archive::binary_oarchive oa(ofs);
210 
211 
212 
213  oa << _orbitals;
214  ofs.close();
215 
216  return true;
