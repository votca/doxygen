155 bool DFT::Evaluate() {
156 
157 
158  if ( _reporting == "silent") _log.setReportLevel( ctp::logERROR ); // only output ERRORS, GEOOPT info, and excited state info for trial geometry
159  if ( _reporting == "noisy") _log.setReportLevel( ctp::logDEBUG ); // OUTPUT ALL THE THINGS
160  if ( _reporting == "default") _log.setReportLevel( ctp::logINFO ); //
161 
162  _log.setMultithreading( true );
163  _log.setPreface(ctp::logINFO, "\n... ...");
164  _log.setPreface(ctp::logERROR, "\n... ...");
165  _log.setPreface(ctp::logWARNING, "\n... ...");
166  _log.setPreface(ctp::logDEBUG, "\n... ...");
167 
168  //TLogLevel _ReportLevel = _log.getReportLevel( ); // backup report level
169 
170  // Create new orbitals object and fill with atom coordinates
171  Orbitals _orbitals;
172 
173  if(_do_guess){
174  CTP_LOG(ctp::logDEBUG,_log) << "Reading guess from " << _guess_file << flush;
175  _orbitals.Load(_guess_file);
176 
177  }
178  else{
179  CTP_LOG(ctp::logDEBUG,_log) << "Reading structure from " << _xyzfile << flush;
180  XYZ2Orbitals( &_orbitals, _xyzfile );
181 
182  }
183 
184 
185 
186  // initialize the DFTENGINE
187  DFTENGINE _dft;
188  _dft.Initialize( &_dftengine_options );
189  _dft.setLogger(&_log);
190 
191  if(_do_external){
192  vector<ctp::APolarSite*> sites=ctp::APS_FROM_MPS(_mpsfile,0);
193  ctp::PolarSeg polarseg=ctp::PolarSeg(0,sites);
194  _dft.setExternalcharges(polarseg);
195  }
196 
197  // RUN
198  _dft.Evaluate( &_orbitals );
199 
200 
201  CTP_LOG(ctp::logDEBUG,_log) << "Saving data to " << _output_file << flush;
202  std::ofstream ofs( ( _output_file).c_str() );
203  boost::archive::binary_oarchive oa( ofs );
204 
205 
206 
207  oa << _orbitals;
208  ofs.close();
209 
210 
211 
212 
213 
214  Property _summary;
215  _summary.add("output","");
216  //Property *_job_output = &_summary.add("output","");
217  tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 1, "");
218 
219  //ofs (_output_file.c_str(), std::ofstream::out);
220  //ofs << *_job_output;
