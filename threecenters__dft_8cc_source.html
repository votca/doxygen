87  void TCMatrix_dft::FillBlock(const AOShell* _shell, const AOBasis& dftbasis) {
88 
89

92  // alpha-loop over the "left" DFT basis function
93  for (AOBasis::AOShellIterator _row = dftbasis.firstShell(); _row != dftbasis.lastShell(); ++_row) {

100  int _col_start = _shell_col->getStartIndex();
101 
102  // get 3-center overlap directly as _subvector

104 
105  bool nonzero = FillThreeCenterRepBlock(_subvector, _shell, _shell_row, _shell_col);
106 
107 
108

110  // and put it into the block it belongs to
111  // functions in ONE AUXshell
112  for (int _aux = 0; _aux < _shell->getNumFunc(); _aux++) {
113  // column in ONE DFTshell
114  for (int _col = 0; _col < _shell_col->getNumFunc(); _col++) {
