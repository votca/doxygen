38 void OptionsHandler::ResolveLinks(Property &prop) const {
39 
40  if (prop.hasAttribute("link")) {
41  std::string relative_path =
42  "subpackages/" + prop.getAttribute<std::string>("link");
43  std::string file_path = defaults_path_ + relative_path;
44  tools::Property package;
45  package.LoadFromXML(file_path);
46  const tools::Property &options = *(package.begin());
47  for (Property::const_AttributeIterator attr = options.firstAttribute();
48  attr != options.lastAttribute(); ++attr) {
49  if (!prop.hasAttribute(attr->first)) {
50  prop.setAttribute(attr->first, attr->second);
51  }
52  }
53 
54  for (const auto &child : options) {
55  prop.add(child);
56  }
57  }
58 
59  for (tools::Property &child : prop) {
60  ResolveLinks(child);
61  }
62 }

162 void OptionsHandler::OverwriteDefaultsWithUserInput(const Property &user_input,
163  Property &defaults) const {
164  // There are 3 distinct cases
165  // a) normal option that can be copied over
166  // b) a list attribute is discovered
167  // c) an unchecked attriute is found,then the whole set is simply copied
168  // over from the user_options, should be done afterwards, as an unchecked
169  // session is not checked and simply copied over
170 
171  if (!defaults.hasAttribute("list")) {
172  defaults.value() = user_input.value();
173  defaults.setAttribute("injected", "true");
174  for (auto &child : defaults) {
175  if (user_input.exists(child.name())) {
176  OverwriteDefaultsWithUserInput(user_input.get(child.name()), child);
177  }
178  }
179  } else {
180  defaults.setAttribute("injected", "true");
181  std::map<std::string, Index> tags;
182  for (const auto &child : defaults) {
183  tags[child.name()]++;
184  }
185 
186  for (const auto &tag : tags) {
187  if (tag.second > 1) {
188  throw std::runtime_error(
189  "Developers: Each distinct tag in list should only appear once.");
190  }
191  std::vector<const tools::Property *> inputs =
192  user_input.Select(tag.first);
193 
194  // if the input has no elements with that tag, remove all children from
195  // default with that tag as well
196  if (inputs.empty()) {
197  defaults.deleteChildren(
198  [&](const Property &prop) { return prop.name() == tag.first; });
199  } else {
200  // copy the element from defaults if the user_input has the element more
201  // than once
202  Property copy = defaults.get(tag.first);
203  for (Index i = 1; i < Index(inputs.size()); i++) {
204  defaults.add(copy);
205  }
206  std::vector<Property *> newdefault_elements =
207  defaults.Select(tag.first);
208  // for each element in both lists do the overwrite again
209  for (Index i = 0; i < Index(inputs.size()); i++) {
210  OverwriteDefaultsWithUserInput(*inputs[i], *newdefault_elements[i]);
211  }
212  }
213  }
214  }
215 
216  if (defaults.hasAttribute("unchecked")) {
217  for (const auto &child : user_input) {
218  defaults.add(child);
219  }
220  }
221 }

237 void OptionsHandler::RecursivelyCheckOptions(const Property &p) const {
238 
239  for (const Property &prop : p) {
240  if (prop.HasChildren()) {
241  RecursivelyCheckOptions(prop);
242  } else {
243  std::vector<std::string> choices = GetPropertyChoices(prop);
244  if (choices.empty()) {
245  continue;
246  }
247  if (!IsValidOption(prop, choices)) {
248  std::ostringstream oss;
249  oss << "\nThe input value for \"" << prop.name() << "\"";
250  if (choices.size() == 1) {
251  oss << " should be a \"" << choices.front() << "\"";
252  } else {
253  oss << " should be one of the following values: ";
254  for (const std::string &c : choices) {
255  oss << "\"" << c << "\""
256  << " ";
257  }
258  }
259  oss << " But \"" << prop.value()
260  << "\" cannot be converted into one.\n";
261  throw std::runtime_error(oss.str());
262  }
263  }
264  }

267 bool OptionsHandler::IsValidOption(
268  const Property &prop, const std::vector<std::string> &choices) const {
269 
270  const std::string &head = choices.front();
271  std::ostringstream oss;
272  bool is_valid = true;
273  if (head == "bool") {
274  is_valid = IsValidCast<bool>(prop);
275  } else if (head == "float") {
276  is_valid = IsValidCast<double>(prop);
277  } else if (head == "float+") {
278  is_valid = IsValidCast<double>(prop) && (prop.as<double>() >= 0.0);
279  } else if (head == "int") {
280  is_valid = IsValidCast<Index>(prop);
281  } else if (head == "int+") {
282  is_valid = IsValidCast<Index>(prop) && (prop.as<Index>() >= 0);
283  } else {
284  std::string value = prop.as<std::string>();
285  std::string att = prop.getAttribute<std::string>("choices");
286  std::size_t start_bracket = att.find('[');
287  if (start_bracket == std::string::npos) {
288  // There is a single choice out of multiple default valid choices
289  auto it = std::find(std::cbegin(choices), std::cend(choices), value);
290  is_valid = (it != std::cend(choices));
291  } else {
292  // there are multiple valid choices
293  Tokenizer tok{value, " ,"};
294  for (const std::string &word : tok) {
295  auto it = std::find(std::cbegin(choices), std::cend(choices), word);
296  if (it == std::cend(choices)) {
297  is_valid = false;
298  break;
299  }
300  }
301  }
302  }
303  if (std::find(additional_choices_.begin(), additional_choices_.end(),
304  prop.as<std::string>()) != additional_choices_.end()) {
305  is_valid = true;
306  }
307  return is_valid;
308 }
