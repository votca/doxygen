88 void Diode::RunKMC() {
89 
90  //Setup random number generator
91  seed = 1;
92  srand(seed); // srand expects any integer in order to initialise the random number generator
93  votca::tools::Random2 *RandomVariable = new votca::tools::Random2();
94  RandomVariable->init(rand(), rand(), rand(), rand());
95 
96  //Initialize all structures
97  graph->hopdist = hopdist;
98  graph->Generate_cubic_graph(nx, ny, nz, lattice_constant, disorder_strength,RandomVariable, disorder_ratio,
99  correlation_type, left_electrode_distance, right_electro_distance,globevent);
100  state->Init();
101  events->Initialize_eventvector(graph, state, globevent);
102  events->Initialize_longrange (graph, globevent);
103  events->Recompute_all_injection_events(graph, globevent);
104  events->Recompute_all_non_injection_events(graph, state, globevent);
105 
106 
107  sim_time = 0.0;
108  for (long it = 0; it < 2*nr_equilsteps + nr_timesteps; it++) {
109 
110  // Update longrange cache (expensive, so not done at every timestep)
111  if(ldiv(it, steps_update_longrange).rem == 0 && it>0){
112  events->longrange->Update_cache(graph->sim_box_size, globevent);
113  events->Recompute_all_injection_events(graph, globevent);
114  events->Recompute_all_non_injection_events(graph,state,globevent);
115  }
116 
117  vssmgroup->Recompute_in_device(events);
118  sim_time += vssmgroup->Timestep(RandomVariable);
119  vssmgroup->Perform_one_step_in_device(events,graph,state,globevent,RandomVariable);
120  }
121 }
