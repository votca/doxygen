86 void Diode::RunKMC() {
87 
88  //Setup random number generator
89  seed = 1;
90  srand(seed); // srand expects any integer in order to initialise the random number generator
91  votca::tools::Random2 *RandomVariable = new votca::tools::Random2();
92  RandomVariable->init(rand(), rand(), rand(), rand());
93 
94  //Initialize all structures
95  graph->hopdist = hopdist;
96  graph->Generate_cubic_graph(nx, ny, nz, lattice_constant, disorder_strength,RandomVariable, disorder_ratio,
97  correlation_type, left_electrode_distance, right_electro_distance,globevent);
98  state->Init();
99  events->Initialize_eventvector(graph, state, globevent);
100  events->Initialize_longrange (graph, globevent);
101  events->Recompute_all_injection_events(graph, globevent);
102  events->Recompute_all_non_injection_events(graph, state, globevent);
103 
104 
105  sim_time = 0.0;
106  for (long it = 0; it < 2*nr_equilsteps + nr_timesteps; it++) {
107 
108  // Update longrange cache (expensive, so not done at every timestep)
109  if(ldiv(it, steps_update_longrange).rem == 0 && it>0){
110  events->longrange->Update_cache(graph->sim_box_size, globevent);
111  events->Recompute_all_injection_events(graph, globevent);
112  events->Recompute_all_non_injection_events(graph,state,globevent);
113  }
114 
115  vssmgroup->Recompute_in_device(events);
116  sim_time += vssmgroup->Timestep(RandomVariable);
117  vssmgroup->Perform_one_step_in_device(events,graph,state,globevent,RandomVariable);
118  }
119 }

121 }}
