33 void Topology::Cleanup()
34 {
35  // cleanup beads
36  {
37  BeadContainer::iterator i;
38  for(i=_beads.begin();i<_beads.end();++i)
39  delete *i;
40  _beads.clear();
41  }
42  // cleanup molecules
43  {
44  MoleculeContainer::iterator i;
45  for(i=_molecules.begin();i<_molecules.end();++i)
46  delete *i;
47  _molecules.clear();
48  }
49  // cleanup residues
50  {
51  ResidueContainer::iterator i;
52  for(i=_residues.begin();i<_residues.end();++i)
53  delete (*i);
54  _residues.clear();
55  }
56  // cleanup interactions
57  {
58  InteractionContainer::iterator i;
59  for(i=_interactions.begin();i<_interactions.end();++i)
60  delete (*i);
61  _interactions.clear();
62  }
63  // cleanup _bc object
64  if(_bc)
65  delete (_bc);
66  _bc = new OpenBox();
67 }

71 {

74  int res_offset=0;
75 
76  BeadContainer::iterator bead;
77  for(bead=_beads.begin(); bead!=_beads.end(); ++bead) {

82  res_offset = (*bead)->getResnr();

129 void Topology::Add(Topology *top)
130 {
131  BeadContainer::iterator bead;
132  ResidueContainer::iterator res;
133  MoleculeContainer::iterator mol;
134 
135  int res0=ResidueCount();
136 
137  for(bead=top->_beads.begin(); bead!=top->_beads.end(); ++bead) {
138  Bead *bi = *bead;
139  BeadType *type = GetOrCreateBeadType(bi->getType()->getName());
140  CreateBead(bi->getSymmetry(), bi->getName(), type, bi->getResnr()+res0, bi->getM(), bi->getQ());
141  }
142 
143  for(res=top->_residues.begin();res!=top->_residues.end(); ++res) {
144  CreateResidue((*res)->getName());
145  }
146 
147  // \todo beadnames in molecules!!

149  Molecule *mi = CreateMolecule((*mol)->getName());

314 BoundaryCondition::eBoxtype Topology::autoDetectBoxType(const matrix &box) {
315  // set the box type to OpenBox in case "box" is the zero matrix,
316  // to OrthorhombicBox in case "box" is a diagonal matrix,
317  // or to TriclinicBox otherwise
318  if(box.get(0,0)==0 && box.get(0,1)==0 && box.get(0,2)==0 &&
319  box.get(1,0)==0 && box.get(1,1)==0 && box.get(1,2)==0 &&
320  box.get(2,0)==0 && box.get(2,1)==0 && box.get(2,2)==0) {
321  //cout << "box open\n";
322  return BoundaryCondition::typeOpen;
323  }
324  else
325  if(box.get(0,1)==0 && box.get(0,2)==0 &&
326  box.get(1,0)==0 && box.get(1,2)==0 &&
327  box.get(2,0)==0 && box.get(2,1)==0) {
328  //cout << "box orth\n";
329  return BoundaryCondition::typeOrthorhombic;
330  }
331  else {
332  //cout << "box tric\n";
333  return BoundaryCondition::typeTriclinic;
334  }
335  return BoundaryCondition::typeOpen;
336 }

338 double Topology::ShortestBoxSize()
339 {
340  vec _box_a = getBox().getCol(0);
341  vec _box_b = getBox().getCol(1);
342  vec _box_c = getBox().getCol(2);
343 
344  // create plane normals
345  vec _norm_a = _box_b ^ _box_c;
346  vec _norm_b = _box_c ^ _box_a;
347  vec _norm_c = _box_a ^ _box_b;
348 
349  _norm_a.normalize();
350  _norm_b.normalize();
351  _norm_c.normalize();
352 
353  double la = _box_a * _norm_a;
354  double lb = _box_b * _norm_b;
355  double lc = _box_c * _norm_c;
356 
357  return min(la, min(lb, lc));
358 }
