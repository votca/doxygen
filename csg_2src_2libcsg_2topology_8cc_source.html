50 
51  for (BeadContainer::iterator i = _beads.begin(); i < _beads.end(); ++i) {

82  delete (_bc);
83  }
84  _bc = new OpenBox();

88 void Topology::CreateMoleculesByRange(string name, Index first, Index nbeads,
89  Index nmolecules) {
90  Molecule *mol = CreateMolecule(name);
91  Index beadcount = 0;
92  Index res_offset = 0;
93 
94  for (auto &_bead : _beads) {
95  // xml numbering starts with 1
96  if (--first > 0) {
97  continue;
98  }
99  // This is not 100% correct, but let's assume for now that the resnr do
100  // increase
101  if (beadcount == 0) {
102  res_offset = _bead->getResnr();
103  }
104  stringstream bname;
105  bname << _bead->getResnr() - res_offset + 1 << ":"
106  << getResidue(_bead->getResnr())->getName() << ":"
107  << _bead->getName();
108  mol->AddBead(_bead, bname.str());
109  if (++beadcount == nbeads) {
110  if (--nmolecules <= 0) {
111  break;
112  }
113  mol = CreateMolecule(name);
114  beadcount = 0;
115  }
116  }
117 }

284 void Topology::RegisterBeadType(string type) {
285  unordered_set<Index> ids;
286  for (pair<const string, Index> type_and_id : beadtypes_) {
287  ids.insert(type_and_id.second);
288  }
289 
290  Index id = 0;
291  // If the type is also a number use it as the id as well provided it is not
292  // already taken
293  if (is_digits(type)) {
294  id = boost::lexical_cast<Index>(type);
295  assert(!ids.count(id) &&
296  "The type passed in is a number and has already"
297  " been registered. It is likely that you are passing in numbers as "
298  "bead types as well as strings, choose one or the other do not mix "
299  "between using numbers and strings ");
300  }
301 
302  while (ids.count(id)) {
303  ++id;
304  }
305  beadtypes_[type] = id;
306 }

322 BoundaryCondition::eBoxtype Topology::autoDetectBoxType(
323  const Eigen::Matrix3d &box) const {
324  // set the box type to OpenBox in case "box" is the zero matrix,
325  // to OrthorhombicBox in case "box" is a diagonal matrix,
326  // or to TriclinicBox otherwise
327  if (box.isApproxToConstant(0)) {
328  return BoundaryCondition::typeOpen;
329  } else if ((box - Eigen::Matrix3d(box.diagonal().asDiagonal()))
330  .isApproxToConstant(0)) {
331  return BoundaryCondition::typeOrthorhombic;
332  } else {
333  return BoundaryCondition::typeTriclinic;
334  }
335  return BoundaryCondition::typeOpen;
336 }

338 double Topology::ShortestBoxSize() const {
339  Eigen::Vector3d box_a = getBox().col(0);
340  Eigen::Vector3d box_b = getBox().col(1);
341  Eigen::Vector3d box_c = getBox().col(2);
342 
343  // create plane normals
344  Eigen::Vector3d norm_a = box_b.cross(box_c);
345  Eigen::Vector3d norm_b = box_c.cross(box_a);
346  Eigen::Vector3d norm_c = box_a.cross(box_b);
347 
348  norm_a.normalize();
349  norm_b.normalize();
350  norm_c.normalize();
351 
352  double la = box_a.dot(norm_a);
353  double lb = box_b.dot(norm_b);
354  double lc = box_c.dot(norm_c);
355 
356  return std::min(la, std::min(lb, lc));
357 }
