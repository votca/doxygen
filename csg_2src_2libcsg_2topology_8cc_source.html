314 BoundaryCondition::eBoxtype Topology::autoDetectBoxType(const matrix &box) {
315  // set the box type to OpenBox in case "box" is the zero matrix,
316  // to OrthorhombicBox in case "box" is a diagonal matrix,
317  // or to TriclinicBox otherwise
318  if(box.get(0,0)==0 && box.get(0,1)==0 && box.get(0,2)==0 &&
319  box.get(1,0)==0 && box.get(1,1)==0 && box.get(1,2)==0 &&
320  box.get(2,0)==0 && box.get(2,1)==0 && box.get(2,2)==0) {
321  //cout << "box open\n";
322  return BoundaryCondition::typeOpen;
323  }
324  else
325  if(box.get(0,1)==0 && box.get(0,2)==0 &&
326  box.get(1,0)==0 && box.get(1,2)==0 &&
327  box.get(2,0)==0 && box.get(2,1)==0) {
328  //cout << "box orth\n";
329  return BoundaryCondition::typeOrthorhombic;
330  }
331  else {
332  //cout << "box tric\n";
333  return BoundaryCondition::typeTriclinic;
334  }
335  return BoundaryCondition::typeOpen;
336 }

338 double Topology::ShortestBoxSize()
339 {
340  vec _box_a = getBox().getCol(0);
341  vec _box_b = getBox().getCol(1);
342  vec _box_c = getBox().getCol(2);
343 
344  // create plane normals
345  vec _norm_a = _box_b ^ _box_c;
346  vec _norm_b = _box_c ^ _box_a;
347  vec _norm_c = _box_a ^ _box_b;
348 
349  _norm_a.normalize();
350  _norm_b.normalize();
351  _norm_c.normalize();
352 
353  double la = _box_a * _norm_a;
354  double lb = _box_b * _norm_b;
355  double lc = _box_c * _norm_c;
356 
357  return min(la, min(lb, lc));
358 }
