31 bool DLPOLYTrajectoryReader::Open(const string &file)
32 // open the original dlpoly configuration or trajectory file
33 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention: ".dlpc"="CONFIG", ".dlph"="HISTORY")
34 {
35  boost::filesystem::path filepath(file.c_str());
36  string inp_name="HISTORY";
37 
38  if ( boost::filesystem::extension(filepath).size() == 0 ) {
39 
40  throw std::ios_base::failure("Error on opening dlpoly file '" + file + "' - extension is expected, use .dlph or .dlpc");
41 
42  } else if( boost::filesystem::extension(filepath)==".dlpc" ) {
43 
44  _isConfig=true;
45  inp_name="CONFIG";
46 
47  } else if( boost::filesystem::extension(filepath)==".dlph" ) {
48 
49  _isConfig=false;
50 
51  } else {
52  throw std::ios_base::failure("Error on opening dlpoly file '" + file + "' - wrong extension, use .dlph or .dlpc");
53  }
54 
55  if ( boost::filesystem::basename(filepath).size() == 0 ) {
56  if (filepath.parent_path().string().size() == 0) {
57  _fname=inp_name;
58  } else {
59  _fname=filepath.parent_path().string() + "/" + inp_name;
60  }
61  } else {
62  _fname=file;
63  }
64 
65  _fl.open(_fname.c_str());
66  if(!_fl.is_open())
67  throw std::ios_base::failure("Error on opening dlpoly file '"+ _fname + "'");
68  return true;
69 }

84 bool DLPOLYTrajectoryReader::NextFrame(Topology &conf)
85 {
86  static bool hasVs = false;
87  static bool hasFs = false;
88  static int mavecs = 0; // number of 3d vectors per atom = keytrj in DL_POLY manuals
89  static int mpbct = 0; // cell PBC type = imcon in DL_POLY manuals
90  static int matoms = 0; // number of atoms/beads in a frame
91  const double scale = 0.1; // A -> nm factor
92 
93  static int nerrt = 0;
94 
95  string line;
96 
97  BoundaryCondition::eBoxtype pbc_type=BoundaryCondition::typeAuto;
98 
99  if (_first_frame) {
100 
101  getline(_fl, line); // title
102 
103 #ifdef DEBUG
104  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - header" << endl;
105 #endif
106 
107  getline(_fl, line); // 2nd header line
108 
109 #ifdef DEBUG
110  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - directives line" << endl;
111 #endif
112 
113  Tokenizer tok(line, " \t");
114  vector<string> fields;
115  tok.ToVector(fields);
116 
117  if( fields.size() < 3 )
118  throw std::runtime_error("Error: too few directive switches (<3) in '"+_fname+"' header (check its 2-nd line)");
119 
120  mavecs = boost::lexical_cast<int>(fields[0]);
121  mpbct = boost::lexical_cast<int>(fields[1]);
122  matoms = boost::lexical_cast<int>(fields[2]);
123 
124  hasVs = (mavecs > 0); // 1 or 2 => in DL_POLY frame velocity vector follows coords for each atom/bead
125  hasFs = (mavecs > 1); // 2 => in DL_POLY frame force vector follows velocities for each atom/bead
126 
127 #ifdef DEBUG
128  if(hasVs != conf.HasVel() || hasFs != conf.HasForce()) {
129  cout << "WARNING: N of atom vectors (keytrj) in '" << _fname << "' header differs from that read with topology" << endl;
130  }
131 #endif
132 
133  conf.SetHasVel(hasVs);
134  conf.SetHasForce(hasFs);
135 
136 #ifdef DEBUG
137  cout << "Read from dlpoly file '" << _fname << "' : keytrj - " << mavecs
138  << ", hasV - " << conf.HasVel() << ", hasF - " << conf.HasForce() << endl;
139 #endif
140 
141  if(matoms != conf.BeadCount())
142  throw std::runtime_error("Number of atoms/beads in '"+_fname+"' header differs from that read with topology");
143 
144  if(mpbct == 0) {
145  pbc_type=BoundaryCondition::typeOpen;
146  }
147  else if(mpbct == 1 || mpbct == 2 ) {
148  pbc_type=BoundaryCondition::typeOrthorhombic;
149  }
150  else if(mpbct == 3) {
151  pbc_type=BoundaryCondition::typeTriclinic;
152  }
153 
154 #ifdef DEBUG
155  cout << "Read from dlpoly file '" << _fname << "' : pbc_type (imcon) - '" << pbc_type << "'" << endl;
156 
157  if(pbc_type != conf.getBoxType())
158  cout << "WARNING: PBC type in dlpoly file '" << _fname << "' header differs from that read with topology" << endl;
159  //throw std::runtime_error("Error: Boundary conditions in '"+_fname+"' header differs from that read with topology");
160 #endif
161  }
162  else if( _isConfig ) {
163 
164  return false;
165  }
166  //read normal frame
167 
168  if( !_isConfig ) {
169  getline(_fl, line); // timestep line - only present in HISTORY, and not in CONFIG
170 #ifdef DEBUG
171  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "'" << endl;
172 #endif
173  }
174 
175  if(!_fl.eof()) {
176  double dtime,stime;
177  int nstep;
178  int natoms;
179  int navecs;
180  int npbct;
181 
182  if( _isConfig ) {
183  // use the above read specs from the header, and skip the data missing in CONFIG
184 
185  natoms = matoms;
186  navecs = mavecs;
187  npbct = mpbct;
188 
189  conf.SetHasVel(hasVs);
190  conf.SetHasForce(hasFs);
191 
192 #ifdef DEBUG
193  cout << "Read from CONFIG: traj_key - " << navecs << ", hasV - " << conf.HasVel() << ", hasF - " << conf.HasForce() << endl;
194 #endif
195 
196  } else {
197 
198  Tokenizer tok(line, " \t");
199  vector<string> fields;
200  tok.ToVector(fields);
201 
202  if( fields.size() < 6 )
203  throw std::runtime_error("Error: too few directive switches (<6) in 'timestep' record");
204 
205  nstep = boost::lexical_cast<int>(fields[1]);
206  natoms = boost::lexical_cast<int>(fields[2]);
207  navecs = boost::lexical_cast<int>(fields[3]);
208  npbct = boost::lexical_cast<int>(fields[4]);
209  dtime = boost::lexical_cast<double>(fields[5]); // normally it is the 5-th column in 'timestep' line
210  stime = boost::lexical_cast<double>(fields[fields.size()-1]); // normally it is the last column in 'timestep' line
211 
212 #ifdef DEBUG
213  cout << "Read from dlpoly file '" << _fname << "' : natoms = " << natoms << ", levcfg = " << fields[3];
214  cout << ", dt = " << fields[5] << ", time = " << stime << endl;
215 #endif
216 
217  if(natoms != conf.BeadCount())
218  throw std::runtime_error("Error: N of atoms/beads in '"+_fname+"' header differs from that found in topology");
219  if(natoms != matoms)
220  throw std::runtime_error("Error: N of atoms/beads in '"+_fname+"' header differs from that found in the frame");
221  if(navecs != mavecs)
222  throw std::runtime_error("Error: N of atom vectors (keytrj) in '"+_fname+"' header differs from that found in the frame");
223  if(npbct != mpbct)
224  throw std::runtime_error("Error: boundary conditions (imcon) in '"+_fname+"' header differs from that found in the frame");
225 
226  // total time - calculated as product due to differences between DL_POLY versions in HISTORY formats
227  conf.setTime(nstep*dtime);
228  conf.setStep(nstep);
229 
230  if( std::abs(stime - conf.getTime()) > 1.e-8 ) {
231  nerrt++;
232  if( nerrt < 11 ) {
233  cout << "Check: nstep = " << nstep << ", dt = " << dtime << ", time = " << stime << " (correct?)" << endl;
234  //cout << "Check: nstep = " << nstep << ", dt = " << dtime << ", time = " << conf.getTime() << " (correct?)" << endl;
235  }
236  else if( nerrt == 11 ) {
237  cout << "Check: timestep - more than 10 mismatches in total time found..." << endl;
238  }
239  }
240 
241  if(npbct == 0) {
242  pbc_type=BoundaryCondition::typeOpen;
243  }
244  else if(npbct == 1 || npbct == 2 ) {
245  pbc_type=BoundaryCondition::typeOrthorhombic;
246  }
247  else if(npbct == 3) {
248  pbc_type=BoundaryCondition::typeTriclinic;
249  }
250  }
251 
252  vec box_vectors[3]{0.0,0.0,0.0};
253  for (int i=0;i<3;i++) { // read 3 box/cell lines
254 
255  getline(_fl, line);
256 
257 #ifdef DEBUG
258  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - box vector # " << i+1 << endl;
259 #endif
260 
261  if(_fl.eof())
262  throw std::runtime_error("Error: unexpected EOF in dlpoly file '"+_fname+"', when reading box vector"+
263  boost::lexical_cast<string>(i));
264 
265  Tokenizer tok(line, " \t");
266  vector<double> fields;
267  tok.ConvertToVector<double>(fields);
268  //Angs -> nm
269  box_vectors[i]=vec(fields[0]*scale,fields[1]*scale,fields[2]*scale);
270  }
271  matrix box(box_vectors[0],box_vectors[1],box_vectors[2]);
272 
273  conf.setBox(box,pbc_type);
274 
275  for (int i=0;i<natoms;i++){
276 
277  {
278  getline(_fl, line); //atom header line
279 
280 #ifdef DEBUG
281  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "'" << endl;
282 #endif
283 
284  if(_fl.eof())
285  throw std::runtime_error("Error: unexpected EOF in dlpoly file '"+_fname+"', when reading atom/bead # " +
286  boost::lexical_cast<string>(i+1));
287 
288  vector<string> fields;
289  Tokenizer tok(line, " \t");
290  tok.ToVector(fields);
291  int id=boost::lexical_cast<int>(fields[1]);
292  if (i+1 != id )
293  throw std::runtime_error("Error: unexpected atom/bead index in dlpoly file '"+_fname+"' : expected " +
294  boost::lexical_cast<string>(i+1) + " but got " + boost::lexical_cast<string>(id));
295  }
296 
297  Bead *b = conf.getBead(i);
298  vec atom_vec[3]{0.0,0.0,0.0};
299  for (int j=0;j<min(navecs,2)+1;j++){
300 
301  getline(_fl, line); //read atom positions
302 
303 #ifdef DEBUG
304  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "'" << endl;
305 #endif
306 
307  if(_fl.eof())
308  throw std::runtime_error("Error: unexpected EOF in dlpoly file '"+_fname+"', when reading atom/bead vector # " +
309  boost::lexical_cast<string>(j) + " of atom " + boost::lexical_cast<string>(i+1) );
310 
311  vector<double> fields;
312  Tokenizer tok(line, " \t");
313  tok.ConvertToVector<double>(fields);
314  //Angs -> nm
315  atom_vec[j]=vec(fields[0]*scale,fields[1]*scale,fields[2]*scale);
316  }
317 
318  b->setPos(atom_vec[0]);
319 #ifdef DEBUG
320  cout << "Crds from dlpoly file '" << _fname << "' : " << atom_vec[0] << endl;
321 #endif
322  if (navecs > 0) {
323  b->setVel(atom_vec[1]);
324 #ifdef DEBUG
325  cout << "Vels from dlpoly file '" << _fname << "' : " << atom_vec[1] << endl;
326 #endif
327  if (navecs > 1) {
328  b->setF(atom_vec[2]);
329 #ifdef DEBUG
330  cout << "Frcs from dlpoly file '" << _fname << "' : " << atom_vec[2] << endl;
331 #endif
332  }
333  }
334  }
335  }
336 
337  return !_fl.eof();
338 }
