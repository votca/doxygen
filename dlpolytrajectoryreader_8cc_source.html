00032 // open the original dlpoly configuration or trajectory file
00033 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention: ".dlpc"="CONFIG", ".dlph"="HISTORY")
00034 {
00035     boost::filesystem::path filepath(file.c_str());

00037

00039

00041 
00042     } else if( boost::filesystem::extension(filepath)==".dlpc" ) {
00043 
00044       _isConfig=true;
00045       inp_name="CONFIG";
00046 
00047     } else if( boost::filesystem::extension(filepath)==".dlph" ) {
00048 
00049       _isConfig=false;
00050 
00051     } else {
00052       throw std::ios_base::failure("Error on opening dlpoly file '" + file + "' - wrong extension, use .dlph or .dlpc");      
00053     }
00054 
00055     if ( boost::filesystem::basename(filepath).size() == 0 ) {
00056       if (filepath.parent_path().string().size() == 0) {
00057         _fname=inp_name;
00058       } else {
00059         _fname=filepath.parent_path().string() + "/" + inp_name;
00060       }
00061     } else {
00062       _fname=file;
00063     }
00064 
00065     _fl.open(_fname.c_str());
00066     if(!_fl.is_open())
00067         throw std::ios_base::failure("Error on opening dlpoly file '"+ _fname + "'");
00068     return true;
