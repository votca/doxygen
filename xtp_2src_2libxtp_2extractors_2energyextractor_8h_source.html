44 bool EnergyExtractor::EvaluateFrame(ctp::Topology *top) {
45 
46  std::string xmlfile = Identify() + ".xml";
47 
48  tools::Property state("state", "", "");
49  tools::Property &segs = state.add("segments", "");
50  tools::Property &pairs = state.add("pairs", "");
51  tools::Property *next = NULL;
52 
53  using boost::format;
54 
55  // SEGMENTS
56  std::vector<ctp::Segment *>::iterator sit;
57  next = &segs;
58  for (sit = top->Segments().begin(); sit < top->Segments().end(); ++sit) {
59  ctp::Segment *seg = *sit;
60  tools::Property &segprop = next->add("segment", "");
61  segprop.add("id", (format("%1$d") % seg->getId()).str());
62  segprop.add("name", seg->getName());
63  if (seg->hasChrgState(+1)) {
64  tools::Property &channel = segprop.add("channel", "");
65  channel.setAttribute("type", "hole");
66  channel.add("energy_hn",
67  (format("%1$+1.7f") % seg->getSiteEnergy(+1)).str());
68  channel.add("lambda_hn",
69  (format("%1$+1.7f") % seg->getU_nC_nN(+1)).str());
70  channel.add("lambda_nh",
71  (format("%1$+1.7f") % seg->getU_cN_cC(+1)).str());
72  }
73  if (seg->hasChrgState(-1)) {
74  tools::Property &channel = segprop.add("channel", "");
75  channel.setAttribute("type", "electron");
76  channel.add("energy_en",
77  (format("%1$+1.7f") % seg->getSiteEnergy(-1)).str());
78  channel.add("lambda_en",
79  (format("%1$+1.7f") % seg->getU_nC_nN(-1)).str());
80  channel.add("lambda_ne",
81  (format("%1$+1.7f") % seg->getU_cN_cC(-1)).str());
82  }
83  }
84 
85  if (tools::globals::verbose) {
86  // PAIRS
87  ctp::QMNBList::iterator pit;
88  ctp::QMNBList &nb = top->NBList();
89  next = &pairs;
90  for (pit = nb.begin(); pit != nb.end(); ++pit) {
91  ctp::QMPair *qmp = *pit;
92  tools::Property &pairprop = next->add("pair", "");
93  pairprop.add("id1", (format("%1$d") % qmp->Seg1()->getId()).str());
94  pairprop.add("name1", (format("%1$s") % qmp->Seg1()->getName()).str());
95  pairprop.add("id2", (format("%1$d") % qmp->Seg2()->getId()).str());
96  pairprop.add("name2", (format("%1$s") % qmp->Seg2()->getName()).str());
97 
98  if (qmp->isPathCarrier(+1)) {
99  tools::Property &channel = pairprop.add("channel", "");
100  channel.setAttribute("type", "hole");
101  channel.add("deltaE_h12", (format("%1$1.7f") % qmp->getdE12(+1)).str());
102  channel.add("lambdaI_h12",
103  (format("%1$1.7f") % qmp->getReorg12(+1)).str());
104  channel.add("lambdaI_h21",
105  (format("%1$1.7f") % qmp->getReorg21(+1)).str());
106  channel.add("lambdaO_h",
107  (format("%1$1.7f") % qmp->getLambdaO(+1)).str());
108  }
109  if (qmp->isPathCarrier(-1)) {
110  tools::Property &channel = pairprop.add("channel", "");
111  channel.setAttribute("type", "electron");
112  channel.add("deltaE_e12", (format("%1$1.7f") % qmp->getdE12(-1)).str());
113  channel.add("lambdaI_e12",
114  (format("%1$1.7f") % qmp->getReorg12(-1)).str());
115  channel.add("lambdaI_e21",
116  (format("%1$1.7f") % qmp->getReorg21(-1)).str());
117  channel.add("lambdaO_e",
118  (format("%1$1.7f") % qmp->getLambdaO(-1)).str());
119  }
120  }
121  }
122 
123  std::ofstream ofs;
124  ofs.open(xmlfile.c_str(), std::ofstream::out);
125  if (!ofs.is_open()) {
126  throw std::runtime_error("Bad file handle: " + xmlfile);
127  }
128  ofs << tools::XML << state;
129  ofs.close();
130 
131  return true;
132 }
