156 void EDFT::WriteJobFile(Topology *top) {
157 
158  cout << endl << "... ... Writing job file: " << flush;
159  ofstream ofs;
160  ofs.open(_jobfile.c_str(), ofstream::out);
161  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
162 
163  ofs << "<jobs>" << endl;
164 
165  QMNBList::iterator pit;
166  QMNBList &nblist = top->NBList();
167 
168 
169  int jobCount = 0;
170  if (nblist.size() == 0) {
171  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
172  return;
173  }
174 
175  // regenerate the list of bridging segments for every pair
176  // (Donor - Bridge1 - Bridge2 - ... - Acceptor) type
177  nblist.GenerateSuperExchange();
178 
179  map< int,Segment* > segments;
180  map< int,Segment* >::iterator sit;
181 
182  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
183 
184  int id1 = (*pit)->Seg1()->getId();
185  int id2 = (*pit)->Seg2()->getId();
186  segments[id1] = (*pit)->Seg1();
187  segments[id2] = (*pit)->Seg2();
188 
189  /* loop over bridging segments if any and add them to the map
190  this in principle is not needed since all pairs between
191  donors, acceptors, and bridges are already in the list
192  */
193  vector<Segment*> bridges = (*pit)->getBridgingSegments();
194  for ( vector<Segment*>::const_iterator bsit = bridges.begin(); bsit != bridges.end(); bsit++ ) {
195  //cout << "Bridging segment " << (*bsit)->getId() << " : " << (*bsit)->getName() << endl;
196  segments[ (*bsit)->getId() ] = (*bsit);
197  }
198 
199  }
200 
201 
202 
203  for (sit = segments.begin(); sit != segments.end(); ++sit) {
204 
205  int id = ++jobCount;
206  string tag = "";
207 
208  Property Input;
209  Property *pInput = &Input.add("input","");
210  Property *pSegment = &pInput->add("segment" , (format("%1$s") % sit->first).str() );
211  pSegment->setAttribute<string>("type", sit->second->getName() );
212  pSegment->setAttribute<int>("id", sit->second->getId() );
213  Job job(id, tag, Input, Job::AVAILABLE );
214  job.ToStream(ofs,"xml");
215  }
216 
217 
218  // CLOSE STREAM
219  ofs << "</jobs>" << endl;
220  ofs.close();
221 
222  cout << jobCount << " jobs" << flush;
223 
224 }

227 Job::JobResult EDFT::EvalJob(Topology *top, Job *job, QMThread *opThread) {
228 
229  string output;
230 
231  bool _run_status;
232  bool _parse_log_status;
233  bool _parse_orbitals_status;
234  bool _convert_status;
235 
236 
237  Orbitals _orbitals;
238  Job::JobResult jres = Job::JobResult();
239  Property _job_input = job->getInput();
240  list<Property*> lSegments = _job_input.Select( "segment" );
241 
242  vector < Segment* > segments;
243  int segId = lSegments.front()->getAttribute<int>( "id" );
244  string segType = lSegments.front()->getAttribute<string>( "type" );
245 
246  Segment *seg = top->getSegment( segId );
247  assert( seg->getName() == segType );
248  segments.push_back( seg );
249  Logger* pLog = opThread->getLogger();
250  LOG(logINFO,*pLog) << TimeStamp() << " Evaluating site " << seg->getId() << flush;
251 
252  // log, com, and orbital files will be stored in ORB_FILES/package_name/frame_x/mol_ID/
253  // extracted information will be stored in ORB_FILES/molecules/frame_x/molecule_ID.orb
254 
255  string edft_work_dir = "OR_FILES";
256  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
257  string ID = boost::lexical_cast<string>( seg->getId() );
258 
259  // get the corresponding object from the QMPackageFactory
260  QMPackage *_qmpackage = QMPackages().Create( _package );
261 
262  _qmpackage->setLog( pLog );
263  _qmpackage->Initialize( &_package_options );
264 
265  string orbitals_storage_dir;
266  string qmpackage_work_dir;
267 
268  /* To avoid possible mixup between standard DFT runs, e.g., for coupling
269  * elements and GWBSE DFT basis runs, modify storage files/directories */
270  if ( _qmpackage->ECPRequested() ){
271  orbitals_storage_dir = "molecules_gwbse";
272  qmpackage_work_dir = edft_work_dir + "/" + _package + "_gwbse/" + frame_dir + "/mol_" + ID;
273  } else {
274  orbitals_storage_dir = "molecules";
275  qmpackage_work_dir = edft_work_dir + "/" + _package + "/" + frame_dir + "/mol_" + ID;
276  }
277 
278  string ORB_DIR = edft_work_dir + "/" + orbitals_storage_dir + "/" + frame_dir;
279  boost::filesystem::create_directories( qmpackage_work_dir );
280  boost::filesystem::create_directories(ORB_DIR);
281 
282  _qmpackage->setRunDir( qmpackage_work_dir );
283 
284 
285  // if asked, prepare the input files
286  if ( _do_input ) {
287  _qmpackage->WriteInputFile( segments );
288  }
289 
290  // Run the executable
291  if ( _do_run ) {
292  _run_status = _qmpackage->Run( );
293  if ( !_run_status ) {
294  output += "run failed; " ;
295  LOG(logERROR,*pLog) << _package << " run failed" << flush;
296  jres.setOutput( output );
297  jres.setStatus(Job::FAILED);
298  delete _qmpackage;
299  return jres;
300  } else {
301  output += "run completed; " ;
302  }
303  }
304 
305  // Parse log files
306  if ( _do_parse ) {
307  _parse_log_status = _qmpackage->ParseLogFile( &_orbitals );
308  if ( !_parse_log_status ) {
309  output += "log incomplete; ";
310  LOG(logERROR,*pLog) << "QM log incomplete" << flush;
311  jres.setOutput( output );
312  jres.setStatus(Job::FAILED);
313  delete _qmpackage;
314  return jres;
315  } else {
316  output += "log parsed; " ;
317  }
318 
319  // Parse orbitals file
320  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile( &_orbitals );
321  if ( !_parse_orbitals_status ) {
322  output += "orbitals failed; " ;
323  LOG(logERROR,*pLog) << "QM orbitals not parsed" << flush;
324  jres.setOutput( output );
325  jres.setStatus(Job::FAILED);
326  delete _qmpackage;
327  return jres;
328  } else {
329  output += "orbitals parsed; " ;
330  }
331  }
332 
333  // Trim virtual orbitals
334  if ( _do_trim ) {
335 
336  int factor = 2;
337  if ( !_do_parse ) { // orbitals must be loaded from a file
338  boost::filesystem::path arg_path;
339  string ORB_FILE = ( arg_path / ORB_DIR / (format("molecule_%1%.orb") % ID ).str() ).c_str() ;
340  LOG(logDEBUG,*pLog) << "Loading orbitals from " << ORB_FILE << flush;
341  if ( ! _orbitals.Load(ORB_FILE) ) { // did not manage to load
342  LOG(logERROR,*pLog) << "Failed loading orbitals from " << ORB_FILE << flush;
343  output += "failed loading " + ORB_FILE;
344  jres.setOutput( output );
345  jres.setStatus(Job::FAILED);
346  delete _qmpackage;
347  return jres;
348  }
349  }
350 
351  _orbitals.Trim(factor);
352  LOG(logDEBUG,*pLog) << "Trimming virtual orbitals from "
353  << _orbitals.getNumberOfLevels() - _orbitals.getNumberOfElectrons() << " to "
354  << _orbitals.getNumberOfElectrons()*factor << flush;
355  output += "orbitals trimmed; " ;
356  }
357 
358 
359  if ( _do_parse ){
360  // save orbitals
361  string ORB_FILE = "molecule_" + ID + ".orb";
362  LOG(logDEBUG,*pLog) << "Serializing to " << ORB_FILE << flush;
363  std::ofstream ofs( (ORB_DIR + "/" + ORB_FILE).c_str() );
364  boost::archive::binary_oarchive oa( ofs );
365  oa << _orbitals;
366  ofs.close();
367  }
368 
369  // Convert to GW
370  if ( _do_convert ) {
371 
372  if ( !_do_parse ) { // orbitals must be loaded from a file
373  string ORB_FILE = "molecule_" + ID + ".orb";
374  LOG(logDEBUG,*pLog) << "Loading orbitals from " << ORB_FILE << flush;
375  std::ifstream ifs( (ORB_DIR + "/" + ORB_FILE).c_str() );
376  boost::archive::binary_iarchive ia( ifs );
377  ia >> _orbitals;
378  ifs.close();
379  }
380 
381  _convert_status = _qmpackage->ConvertToGW( &_orbitals );
382  if ( !_convert_status ) {
383  output += "conversion failed; " ;
384  LOG(logERROR,*pLog) << _package << " conversion failed" << flush;
385  jres.setOutput( output );
386  jres.setStatus(Job::FAILED);
387  delete _qmpackage;
388  return jres;
389  } else {
390  output += "conversion completed; " ;
391  }
392 
393  }
394 
395 
396  // to run any of the GW-BSE steps, initialize QMPackage
397  if (_do_gwbse_input || _do_gwbse_run || _do_gwbse_parse) {
398 
399  // initialize GW as another QMPackage
400  QMPackage *_gwpackage = QMPackages().Create("gw");
401  _gwpackage->Initialize(&_gwpackage_options);
402  _gwpackage->setLog(pLog);
403  _gwpackage->setRunDir(qmpackage_work_dir);
404 
405 
406  if (_do_gwbse_input || _do_gwbse_parse ) {
407 
408  // DFT data must be read from storage or have been parsed
409  if (!_do_parse) { // orbitals must be loaded from a file
410  string ORB_FILE = "molecule_" + ID + ".orb";
411  LOG(logDEBUG, *pLog) << "Loading orbitals from " << ORB_FILE << flush;
412  std::ifstream ifs((ORB_DIR + "/" + ORB_FILE).c_str());
413  boost::archive::binary_iarchive ia(ifs);
414  ia >> _orbitals;
415  ifs.close();
416  }
417 
418  _gwpackage->WriteInputFile(segments, &_orbitals);
419 
420  }
421 
422  if (_do_gwbse_run) {
423  // run the GWBSE executable
424  _run_status = _gwpackage->Run();
425  if (!_run_status) {
426  output += "run failed; ";
427  LOG(logERROR, *pLog) << " GWBSE run failed" << flush;
428  jres.setOutput(output);
429  jres.setStatus(Job::FAILED);
430  delete _gwpackage;
431  return jres;
432  } else {
433  output += "run completed; ";
434  }
435  }
436 
437  if (_do_gwbse_parse) {
438  // parse and store output from GWBSE
439  _parse_log_status = _gwpackage->ParseLogFile(&_orbitals);
440  if (!_parse_log_status) {
441  output += "log incomplete; ";
442  LOG(logERROR, *pLog) << "GWBSE log incomplete" << flush;
443  jres.setOutput(output);
444  jres.setStatus(Job::FAILED);
445  delete _gwpackage;
446  return jres;
447  } else {
448  output += "log parsed; ";
449  // save orbitals
450  string ORB_FILE = "molecule_" + ID + ".orb";
451  LOG(logDEBUG,*pLog) << "Serializing to " << ORB_FILE << flush;
452  std::ofstream ofs( (ORB_DIR + "/" + ORB_FILE).c_str() );
453  boost::archive::binary_oarchive oa( ofs );
454  oa << _orbitals;
455  ofs.close();
456  }
457 
458 
459  }
460  }
461 
462  // Clean run
463  _qmpackage->CleanUp();
464  delete _qmpackage;
465 
466  LOG(logINFO,*pLog) << TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
467 
468  Property _job_summary;
469  Property *_output_summary = &_job_summary.add("output","");
470  Property *_segment_summary = &_output_summary->add("segment","");
471  string segName = seg->getName();
472  segId = seg->getId();
473  _segment_summary->setAttribute("id", segId);
474  _segment_summary->setAttribute("type", segName);
475  _segment_summary->setAttribute("homo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() ));
476  _segment_summary->setAttribute("lumo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() + 1 ));
477 
478  // output of the JOB
479  jres.setOutput( _job_summary );
480  jres.setStatus(Job::COMPLETE);
481 
482  // dump the LOG
483  //cout << *pLog;
484 
485  return jres;
486 
487 }
