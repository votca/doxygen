120 void EDFT::WriteJobFile(Topology *top) {
121 
122  cout << endl << "... ... Writing job file: " << flush;
123  ofstream ofs;
124  ofs.open(_jobfile.c_str(), ofstream::out);
125  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
126 
127  ofs << "<jobs>" << endl;
128 
129  QMNBList::iterator pit;
130  QMNBList &nblist = top->NBList();
131 
132 
133  int jobCount = 0;
134  if (nblist.size() == 0) {
135  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
136  return;
137  }
138 
139  // regenerate the list of bridging segments for every pair
140  // (Donor - Bridge1 - Bridge2 - ... - Acceptor) type
141  nblist.GenerateSuperExchange();
142 
143  map< int,Segment* > segments;
144  map< int,Segment* >::iterator sit;
145 
146  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
147 
148  int id1 = (*pit)->Seg1()->getId();
149  int id2 = (*pit)->Seg2()->getId();
150  segments[id1] = (*pit)->Seg1();
151  segments[id2] = (*pit)->Seg2();
152 
153  /* loop over bridging segments if any and add them to the map
154  this in principle is not needed since all pairs between
155  donors, acceptors, and bridges are already in the list
156  */
157  vector<Segment*> bridges = (*pit)->getBridgingSegments();
158  for ( vector<Segment*>::const_iterator bsit = bridges.begin(); bsit != bridges.end(); bsit++ ) {
159  //cout << "Bridging segment " << (*bsit)->getId() << " : " << (*bsit)->getName() << endl;
160  segments[ (*bsit)->getId() ] = (*bsit);
161  }
162 
163  }
164 
165 
166 
167  for (sit = segments.begin(); sit != segments.end(); ++sit) {
168 
169  int id = ++jobCount;
170  string tag = "";
171 
172  Property Input;
173  Property *pInput = &Input.add("input","");
174  Property *pSegment = &pInput->add("segment" , (format("%1$s") % sit->first).str() );
175  pSegment->setAttribute<string>("type", sit->second->getName() );
176  pSegment->setAttribute<int>("id", sit->second->getId() );
177  Job job(id, tag, Input, Job::AVAILABLE );
178  job.ToStream(ofs,"xml");
179  }
180 
181 
182  // CLOSE STREAM
183  ofs << "</jobs>" << endl;
184  ofs.close();
185 
186  cout << jobCount << " jobs" << flush;
187 
188 }

191 Job::JobResult EDFT::EvalJob(Topology *top, Job *job, QMThread *opThread) {
192 
193  string output;
194 
195  bool _run_status;
196  bool _parse_log_status;
197  bool _parse_orbitals_status;
198 
199  Orbitals _orbitals;
200  Job::JobResult jres = Job::JobResult();
201  Property _job_input = job->getInput();
202  list<Property*> lSegments = _job_input.Select( "segment" );
203  vector < Segment* > segments;
204  int segId = lSegments.front()->getAttribute<int>( "id" );
205  string segType = lSegments.front()->getAttribute<string>( "type" );
206 
207  Segment *seg = top->getSegment( segId );
208  assert( seg->getName() == segType );
209  segments.push_back( seg );
210  Logger* pLog = opThread->getLogger();
211  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Evaluating site " << seg->getId() << flush;
212 
213  // log, com, and orbital files will be stored in ORB_FILES/package_name/frame_x/mol_ID/
214  // extracted information will be stored in ORB_FILES/molecules/frame_x/molecule_ID.orb
215 
216  string edft_work_dir = "OR_FILES";
217  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
218  string ID = boost::lexical_cast<string>( seg->getId() );
219 
220  // get the corresponding object from the QMPackageFactory
221  QMPackage *_qmpackage = QMPackages().Create( _package );
222 
223  _qmpackage->setLog( pLog );
224  _qmpackage->Initialize( &_package_options );
225 
226  string orbitals_storage_dir;
227  string qmpackage_work_dir;
228 
229  orbitals_storage_dir = "molecules";
230  qmpackage_work_dir = edft_work_dir + "/" + _package + "/" + frame_dir + "/mol_" + ID;
231 
232 
233  string ORB_DIR = edft_work_dir + "/" + orbitals_storage_dir + "/" + frame_dir;
234  boost::filesystem::create_directories( qmpackage_work_dir );
235  boost::filesystem::create_directories(ORB_DIR);
236 
237  _qmpackage->setRunDir( qmpackage_work_dir );
238 
239 
240  // if asked, prepare the input files
241  if ( _do_input ) {
242  _qmpackage->WriteInputFile( segments );
243  }
244 
245  // Run the executable
246  if ( _do_run ) {
247  _run_status = _qmpackage->Run( );
248  if ( !_run_status ) {
249  output += "run failed; " ;
250  CTP_LOG(logERROR,*pLog) << _package << " run failed" << flush;
251  jres.setOutput( output );
252  jres.setStatus(Job::FAILED);
253  delete _qmpackage;
254  return jres;
255  } else {
256  output += "run completed; " ;
257  }
258  }
259 
260  // Parse log files
261  if ( _do_parse ) {
262  _parse_log_status = _qmpackage->ParseLogFile( &_orbitals );
263  if ( !_parse_log_status ) {
264  output += "log incomplete; ";
265  CTP_LOG(logERROR,*pLog) << "QM log incomplete" << flush;
266  jres.setOutput( output );
267  jres.setStatus(Job::FAILED);
268  delete _qmpackage;
269  return jres;
270  } else {
271  output += "log parsed; " ;
272  }
273 
274  // Parse orbitals file
275  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile( &_orbitals );
276  if ( !_parse_orbitals_status ) {
277  output += "orbitals failed; " ;
278  CTP_LOG(logERROR,*pLog) << "QM orbitals not parsed" << flush;
279  jres.setOutput( output );
280  jres.setStatus(Job::FAILED);
281  delete _qmpackage;
282  return jres;
283  } else {
284  output += "orbitals parsed; " ;
285  }
286  }
287 
288  // Trim virtual orbitals
289  if ( _do_trim ) {
290 
291  int factor = 2;
292  if ( !_do_parse ) { // orbitals must be loaded from a file
293  boost::filesystem::path arg_path;
294  string ORB_FILE = ( arg_path / ORB_DIR / (format("molecule_%1%.orb") % ID ).str() ).c_str() ;
295  CTP_LOG(logDEBUG,*pLog) << "Loading orbitals from " << ORB_FILE << flush;
296  if ( ! _orbitals.Load(ORB_FILE) ) { // did not manage to load
297  CTP_LOG(logERROR,*pLog) << "Failed loading orbitals from " << ORB_FILE << flush;
298  output += "failed loading " + ORB_FILE;
299  jres.setOutput( output );
300  jres.setStatus(Job::FAILED);
301  delete _qmpackage;
302  return jres;
303  }
304  }
305 
306  _orbitals.Trim(factor);
307  CTP_LOG(logDEBUG,*pLog) << "Trimming virtual orbitals from "
308  << _orbitals.getNumberOfLevels() - _orbitals.getNumberOfElectrons() << " to "
309  << _orbitals.getNumberOfElectrons()*factor << flush;
310  output += "orbitals trimmed; " ;
311  }
312 
313 
314  if ( _do_parse ){
315  // save orbitals
316  string ORB_FILE = "molecule_" + ID + ".orb";
317  CTP_LOG(logDEBUG,*pLog) << "Serializing to " << ORB_FILE << flush;
318  std::ofstream ofs( (ORB_DIR + "/" + ORB_FILE).c_str() );
319  boost::archive::binary_oarchive oa( ofs );
320  oa << _orbitals;
321  // ofs.close();
322 
323  CTP_LOG(logDEBUG,*pLog) << "Done serializing " << ORB_FILE << flush;
324  }
325 
326  // Clean run
327  _qmpackage->CleanUp();
328  delete _qmpackage;
329 
330  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
331 
332  Property _job_summary;
333  Property *_output_summary = &_job_summary.add("output","");
334  Property *_segment_summary = &_output_summary->add("segment","");
335  string segName = seg->getName();
336  segId = seg->getId();
337  _segment_summary->setAttribute("id", segId);
338  _segment_summary->setAttribute("type", segName);
339  _segment_summary->setAttribute("homo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() ));
340  _segment_summary->setAttribute("lumo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() + 1 ));
341 
342  // output of the JOB
343  jres.setOutput( _job_summary );
344  jres.setStatus(Job::COMPLETE);
345 
346  // dump the LOG
347  //cout << *pLog;
348 
349  return jres;
350 
351 }
