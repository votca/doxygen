48BOOST_AUTO_TEST_CASE(density_test) {
49 libint2::initialize();
50 QMMolecule mol("none", 0);
51
52 mol.LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
53 "/densityintegration/molecule.xyz");
54 AOBasis aobasis = CreateBasis(mol);
55
56 Eigen::MatrixXd dmat = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
57 std::string(XTP_TEST_DATA_FOLDER) + "/densityintegration/dmat.mm");
58
59 Vxc_Grid grid;
60 grid.GridSetup("medium", mol, aobasis);
61 DensityIntegration<Vxc_Grid> num(grid);
62
63 double ntot = num.IntegrateDensity(dmat);
64 BOOST_CHECK_CLOSE(ntot, 8.000000, 1e-5);
65
66 Eigen::Vector3d pos = {3, 3, 3};
67
68 BOOST_CHECK_CLOSE(num.IntegratePotential(pos), -1.543242, 1e-4);
69
70 Eigen::Vector3d field = num.IntegrateField(pos);
71 Eigen::Vector3d field_ref = {0.172802, 0.172802, 0.172802};
72 bool field_check = field.isApprox(field_ref, 1e-5);
73 if (!field_check) {
74 std::cout << "field" << std::endl;
75 std::cout << field.transpose() << std::endl;
76 std::cout << "ref" << std::endl;
77 std::cout << field_ref.transpose() << std::endl;
78 }
79 libint2::finalize();
80}

98
99 Eigen::MatrixXd dmat = votca::tools::EigenIO_MatrixMarket::ReadMatrix(

101 Vxc_Grid grid;
102 grid.GridSetup("medium", mol, aobasis);
103 DensityIntegration<Vxc_Grid> num(grid);
104
105 Gyrationtensor tensor = num.IntegrateGyrationTensor(dmat);
106 BOOST_CHECK_CLOSE(tensor.mass, 8.0000005, 1e-5);
107
108 Eigen::Vector3d dip_ref = Eigen::Vector3d::Zero();
109 dip_ref << 1.88973, 1.88973, 1.88973;
110 bool centroid_check = dip_ref.isApprox(tensor.centroid, 1e-5);
111 BOOST_CHECK_EQUAL(centroid_check, true);
112 if (!centroid_check) {
113 std::cout << "centroid" << std::endl;
114 std::cout << tensor.centroid.transpose() << std::endl;
115 std::cout << "ref" << std::endl;
116 std::cout << dip_ref.transpose() << std::endl;
117 }
118 Eigen::Matrix3d gyro_ref = Eigen::Matrix3d::Zero();
119 gyro_ref << 0.596158, 2.85288e-12, 2.86873e-12, 2.85289e-12, 0.596158,
120 2.87163e-12, 2.86874e-12, 2.87161e-12, 0.596158;
121 bool gyro_check = gyro_ref.isApprox(tensor.gyration, 1e-5);
122 BOOST_CHECK_EQUAL(gyro_check, true);
123 if (!gyro_check) {
124 std::cout << "gyro" << std::endl;
125 std::cout << tensor.gyration << std::endl;
126 std::cout << "ref" << std::endl;
127 std::cout << gyro_ref << std::endl;
