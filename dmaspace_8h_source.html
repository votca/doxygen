226  cmplx X1x1 =

229  Xlm_shifted.push_back(X1_1);
230  Xlm_shifted.push_back(X10);
231  Xlm_shifted.push_back(X1x1);

234  cmplx X2_2 =
235  SqrtBinBin(2,-2,0,0) * Xlm.X00()*Clm.C2_2() // L=2 M=-2 l=0 m=0
