40 // REGULAR SPHERICAL HARMONICS
41 // o NOTE Currently only up to rank l=2 (<> quadrupole)
42 // o TODO For higher ranks, use recursive construction
43 // o Notation key: C1_1 => l=1 m=-1 and C1x1 => l=1 m=+1
44 class RegularSphericalHarmonics
45 {
46 public:
47  RegularSphericalHarmonics(votca::tools::vec r)
48  : _x(r.getX()), _y(r.getY()), _z(r.getZ()), _r2(r*r) {}
49  ~RegularSphericalHarmonics() {}
50 
51  // Real spherical harmonics
52  inline double R00() { return 1; }
53  inline double R10() { return _z; }
54  inline double R11c() { return _x; }
55  inline double R11s() { return _y; }
56  inline double R20() { return 0.5*(3*_z*_z - _r2); }
57  inline double R21c() { return std::sqrt(3)*_x*_z; }
58  inline double R21s() { return std::sqrt(3)*_y*_z; }
59  inline double R22c() { return 0.5*std::sqrt(3)*(_x*_x - _y*_y); }
60  inline double R22s() { return std::sqrt(3)*_x*_y; }
61 
62  // Complex spherical harmonics
63  inline cmplx C00() { return cmplx(R00(), 0.0); }
64  inline cmplx C1_1() { return std::sqrt(0.5)*cmplx(+1*R11c(), -1*R11s()); }
65  inline cmplx C10() { return cmplx(R10(), 0.0); }
66  inline cmplx C1x1() { return std::sqrt(0.5)*cmplx(-1*R11c(), -1*R11s()); }
67  inline cmplx C2_2() { return std::sqrt(0.5)*cmplx(+1*R22c(), -1*R22s()); }
68  inline cmplx C2_1() { return std::sqrt(0.5)*cmplx(+1*R21c(), -1*R21s()); }
69  inline cmplx C20() { return cmplx(R20(), 0.0); }
70  inline cmplx C2x1() { return std::sqrt(0.5)*cmplx(-1*R21c(), -1*R21s()); }
71  inline cmplx C2x2() { return std::sqrt(0.5)*cmplx(+1*R22c(), +1*R22s()); }
72 
73 private:
74  double _x, _y, _z;
75  double _r2;
76 };

228  vector<cmplx> Shift(ComplexSphericalMoments &Xlm,
229  RegularSphericalHarmonics &Clm) {
230 
231  vector<cmplx> Xlm_shifted;
232 
233  // CHARGE
234  cmplx X00 =
235  SqrtBinBin(0,0,0,0) * Xlm.X00()*Clm.C00(); // L=0 M=0 l=0 m=0
236  Xlm_shifted.push_back(X00);
237 
238  // DIPOLE
239  cmplx X1_1 =
240  SqrtBinBin(1,-1,0,0) * Xlm.X00()*Clm.C1_1() // L=1 M=-1 l=0 m=0
241  + SqrtBinBin(1,-1,1,-1) * Xlm.X1_1()*Clm.C00(); // L=1 M=-1 l=1 m=-1
242  cmplx X10 =
243  SqrtBinBin(1,0,0,0) * Xlm.X00()*Clm.C10() // L=1 M=0 l=0 m=0
244  + SqrtBinBin(1,0,1,0) * Xlm.X10()*Clm.C00(); // L=1 M=0 l=1 m=0
245  cmplx X1x1 =
246  SqrtBinBin(1,1,0,0) * Xlm.X00()*Clm.C1x1() // L=1 M=+1 l=0 m=0
247  + SqrtBinBin(1,1,1,1) * Xlm.X1x1()*Clm.C00(); // L=1 M=+1 l=+1 m=+1
248  Xlm_shifted.push_back(X1_1);
249  Xlm_shifted.push_back(X10);
250  Xlm_shifted.push_back(X1x1);
251 
252  // QUADRUPOLE
253  cmplx X2_2 =
254  SqrtBinBin(2,-2,0,0) * Xlm.X00()*Clm.C2_2() // L=2 M=-2 l=0 m=0
255  + SqrtBinBin(2,-2,1,-1) * Xlm.X1_1()*Clm.C1_1() // L=2 M=-2 l=1 m=-1
256  + SqrtBinBin(2,-2,2,-2) * Xlm.X2_2()*Clm.C00(); // L=2 M=-2 l=2 m=-2
257  cmplx X2_1 =
258  SqrtBinBin(2,-1,0,0) * Xlm.X00()*Clm.C2_1() // ...
259  + SqrtBinBin(2,-1,1,0) * Xlm.X10()*Clm.C1_1()
260  + SqrtBinBin(2,-1,1,-1) * Xlm.X1_1()*Clm.C10()
261  + SqrtBinBin(2,-1,2,-1) * Xlm.X2_1()*Clm.C00();
262  cmplx X20 =
263  SqrtBinBin(2,0,0,0) * Xlm.X00()*Clm.C20()
264  + SqrtBinBin(2,0,1,0) * Xlm.X10()*Clm.C10()
265  + SqrtBinBin(2,0,1,-1) * Xlm.X1_1()*Clm.C1x1()
266  + SqrtBinBin(2,0,1,+1) * Xlm.X1x1()*Clm.C1_1()
267  + SqrtBinBin(2,0,2,0) * Xlm.X20()*Clm.C00();
268  cmplx X2x1 =
269  SqrtBinBin(2,1,0,0) * Xlm.X00()*Clm.C2x1()
270  + SqrtBinBin(2,1,1,0) * Xlm.X10()*Clm.C1x1()
271  + SqrtBinBin(2,1,1,1) * Xlm.X1x1()*Clm.C10()
272  + SqrtBinBin(2,1,2,1) * Xlm.X2x1()*Clm.C00();
273  cmplx X2x2 =
274  SqrtBinBin(2,2,0,0) * Xlm.X00()*Clm.C2x2()
275  + SqrtBinBin(2,2,1,1) * Xlm.X1x1()*Clm.C1x1()
276  + SqrtBinBin(2,2,2,2) * Xlm.X2x2()*Clm.C00();
277  Xlm_shifted.push_back(X2_2);
278  Xlm_shifted.push_back(X2_1);
279  Xlm_shifted.push_back(X20);
280  Xlm_shifted.push_back(X2x1);
281  Xlm_shifted.push_back(X2x2);
282 
283  return Xlm_shifted;
284  }

287  return std::sqrt( binom(L+M, l+m) * binom(L-M, l-m) );
