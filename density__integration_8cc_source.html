26 template <class Grid>
27 double DensityIntegration<Grid>::IntegratePotential(
28  const Eigen::Vector3d& rvector) const {
29 
30  double result = 0.0;
31  assert(!_densities.empty() && "Density not calculated");
32  for (Index i = 0; i < _grid.getBoxesSize(); i++) {
33  const std::vector<Eigen::Vector3d>& points = _grid[i].getGridPoints();
34  const std::vector<double>& densities = _densities[i];
35  for (Index j = 0; j < _grid[i].size(); j++) {
36  double dist = (points[j] - rvector).norm();
37  result -= densities[j] / dist;
38  }
39  }
40  return result;

45  const Eigen::Vector3d& rvector) const {
46 
47  Eigen::Vector3d result = Eigen::Vector3d::Zero();

50  const std::vector<Eigen::Vector3d>& points = _grid[i].getGridPoints();

68 template <class Grid>
69 double DensityIntegration<Grid>::IntegrateDensity(
70  const Eigen::MatrixXd& density_matrix) {
71 
72  Index nthreads = OPENMP::getMaxThreads();
73  std::vector<double> N_thread = std::vector<double>(nthreads, 0.0);
74  SetupDensityContainer();
75 
76 #pragma omp parallel for schedule(guided)
77  for (Index i = 0; i < _grid.getBoxesSize(); ++i) {
78  const GridBox& box = _grid[i];
79  if (!box.Matrixsize()) {
80  continue;
81  }
82  double N_box = 0.0;
83  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
84  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
85  const std::vector<double>& weights = box.getGridWeights();
86  // iterate over gridpoints
87  for (Index p = 0; p < box.size(); p++) {
88  Eigen::VectorXd ao = box.CalcAOValues(points[p]);
89  double rho = (ao.transpose() * DMAT_here * ao)(0, 0) * weights[p];
90  _densities[i][p] = rho;
91  N_box += rho;
92  }
93  N_thread[OPENMP::getThreadId()] += N_box;
94  }
95  double N = std::accumulate(N_thread.begin(), N_thread.end(), 0.0);
96  return N;
97 }

99 template <class Grid>
100 Gyrationtensor DensityIntegration<Grid>::IntegrateGyrationTensor(
101  const Eigen::MatrixXd& density_matrix) {
102 
103  Index nthreads = OPENMP::getMaxThreads();
104  std::vector<double> N_thread = std::vector<double>(nthreads, 0.0);
105  std::vector<Eigen::Vector3d> centroid_thread =
106  std::vector<Eigen::Vector3d>(nthreads, Eigen::Vector3d::Zero());
107  std::vector<Eigen::Matrix3d> gyration_thread =
108  std::vector<Eigen::Matrix3d>(nthreads, Eigen::Matrix3d::Zero());
109 
110  SetupDensityContainer();
111 #pragma omp parallel for schedule(guided)
112  for (Index i = 0; i < _grid.getBoxesSize(); ++i) {
113  const GridBox& box = _grid[i];
114  if (!box.Matrixsize()) {
115  continue;
116  }
117  double N_box = 0.0;
118  Eigen::Vector3d centroid_box = Eigen::Vector3d::Zero();
119  Eigen::Matrix3d gyration_box = Eigen::Matrix3d::Zero();
120  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
121  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
122  const std::vector<double>& weights = box.getGridWeights();
123  // iterate over gridpoints
124  for (Index p = 0; p < box.size(); p++) {
125  Eigen::VectorXd ao = box.CalcAOValues(points[p]);
126  double rho = (ao.transpose() * DMAT_here * ao).value() * weights[p];
127  _densities[i][p] = rho;
128  N_box += rho;
129  centroid_box += rho * points[p];
130  gyration_box += rho * points[p] * points[p].transpose();
131  }
132  N_thread[OPENMP::getThreadId()] += N_box;
133  centroid_thread[OPENMP::getThreadId()] += centroid_box;
134  gyration_thread[OPENMP::getThreadId()] += gyration_box;
135  }
136  double N = std::accumulate(N_thread.begin(), N_thread.end(), 0.0);
137  Eigen::Vector3d centroid =
138  std::accumulate(centroid_thread.begin(), centroid_thread.end(),
139  Eigen::Vector3d::Zero().eval());
140  Eigen::Matrix3d gyration =
141  std::accumulate(gyration_thread.begin(), gyration_thread.end(),
142  Eigen::Matrix3d::Zero().eval());
143 
144  // Normalize
145  centroid = centroid / N;
146  gyration = gyration / N;
147  gyration = gyration - centroid * centroid.transpose();
148  Gyrationtensor gyro;
149  gyro.mass = N;
150  gyro.centroid = centroid;
151  gyro.gyration = gyration;
152 
153  return gyro;
154 }
155 template <class Grid>
156 Eigen::MatrixXd DensityIntegration<Grid>::IntegratePotential(
157  const AOBasis& externalbasis) const {
158  Eigen::MatrixXd Potential = Eigen::MatrixXd::Zero(
159  externalbasis.AOBasisSize(), externalbasis.AOBasisSize());
160 
161  assert(!_densities.empty() && "Density not calculated");
162  for (Index i = 0; i < _grid.getBoxesSize(); i++) {
163  const std::vector<Eigen::Vector3d>& points = _grid[i].getGridPoints();
164  const std::vector<double>& densities = _densities[i];
165  for (Index j = 0; j < _grid[i].size(); j++) {
166  if (densities[j] < 1e-12) {
167  continue;
168  }
169  AOMultipole esp;
170  esp.FillPotential(externalbasis, points[j]);
171  Potential += densities[j] * esp.Matrix();
172  }
173  }
174  return Potential;
175 }
