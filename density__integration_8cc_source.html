27template <class Grid>
28double DensityIntegration<Grid>::IntegratePotential(
29 const Eigen::Vector3d& rvector) const {
30
31 double result = 0.0;
32 assert(!densities_.empty() && "Density not calculated");
33 for (Index i = 0; i < grid_.getBoxesSize(); i++) {
34 const std::vector<Eigen::Vector3d>& points = grid_[i].getGridPoints();
35 const std::vector<double>& densities = densities_[i];
36 for (Index j = 0; j < grid_[i].size(); j++) {
37 double dist = (points[j] - rvector).norm();
38 result -= densities[j] / dist;
39 }
40 }
41 return result;
42}

44template <class Grid>
45Eigen::Vector3d DensityIntegration<Grid>::IntegrateField(
46 const Eigen::Vector3d& rvector) const {
47
48 Eigen::Vector3d result = Eigen::Vector3d::Zero();
49 assert(!densities_.empty() && "Density not calculated");
50 for (Index i = 0; i < grid_.getBoxesSize(); i++) {
51 const std::vector<Eigen::Vector3d>& points = grid_[i].getGridPoints();
52 const std::vector<double>& densities = densities_[i];
53 for (Index j = 0; j < grid_[i].size(); j++) {
54 Eigen::Vector3d r = points[j] - rvector;
55 result -= densities[j] * r / std::pow(r.norm(), 3); // x,y,z
56 }
57 }
58 return result;
59}

69template <class Grid>
70double DensityIntegration<Grid>::IntegrateDensity(
71 const Eigen::MatrixXd& density_matrix) {
72
73 double N = 0.0;
74 SetupDensityContainer();
75
76#pragma omp parallel for schedule(guided) reduction(+ : N)
77 for (Index i = 0; i < grid_.getBoxesSize(); ++i) {
78 const GridBox& box = grid_[i];
79 if (!box.Matrixsize()) {
80 continue;
81 }
82 const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
83 const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
84 const std::vector<double>& weights = box.getGridWeights();
85 // iterate over gridpoints
86 for (Index p = 0; p < box.size(); p++) {
87 AOShell::AOValues ao = box.CalcAOValues(points[p]);
88 double rho = ao.values.dot(DMAT_here * ao.values) * weights[p];
89 densities_[i][p] = rho;
90 N += rho;
91 }
92 }
93 return N;
94}

96template <class Grid>
97Gyrationtensor DensityIntegration<Grid>::IntegrateGyrationTensor(
98 const Eigen::MatrixXd& density_matrix) {
99 double N = 0.0;
100 Eigen::Vector3d centroid = Eigen::Vector3d::Zero();
101 Eigen::Matrix3d gyration = Eigen::Matrix3d::Zero();
102
103 SetupDensityContainer();
104#pragma omp parallel for schedule(guided)reduction(+:N)reduction(+:centroid)reduction(+:gyration)
105 for (Index i = 0; i < grid_.getBoxesSize(); ++i) {
106 const GridBox& box = grid_[i];
107 if (!box.Matrixsize()) {
108 continue;
109 }
110
111 const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
112 const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
113 const std::vector<double>& weights = box.getGridWeights();
114 // iterate over gridpoints
115 for (Index p = 0; p < box.size(); p++) {
116 AOShell::AOValues ao = box.CalcAOValues(points[p]);
117 double rho = ao.values.dot(DMAT_here * ao.values) * weights[p];
118 densities_[i][p] = rho;
119 N += rho;
120 centroid += rho * points[p];
121 gyration += rho * points[p] * points[p].transpose();
122 }
123 }
124
125 // Normalize
126 centroid = centroid / N;
127 gyration = gyration / N;
128 gyration = gyration - centroid * centroid.transpose();
129 Gyrationtensor gyro;
130 gyro.mass = N;
131 gyro.centroid = centroid;
132 gyro.gyration = gyration;
133
134 return gyro;
135}
136template <class Grid>
137Eigen::MatrixXd DensityIntegration<Grid>::IntegratePotential(
138 const AOBasis& externalbasis) const {
139 Eigen::MatrixXd Potential = Eigen::MatrixXd::Zero(
140 externalbasis.AOBasisSize(), externalbasis.AOBasisSize());
141
142 assert(!densities_.empty() && "Density not calculated");
143 for (Index i = 0; i < grid_.getBoxesSize(); i++) {
144 const std::vector<Eigen::Vector3d>& points = grid_[i].getGridPoints();
145 const std::vector<double>& densities = densities_[i];
146 for (Index j = 0; j < grid_[i].size(); j++) {
147 if (densities[j] < 1e-12) {
148 continue;
149 }
150 AOMultipole esp;
151 esp.FillPotential(externalbasis, points[j]);
152 Potential += densities[j] * esp.Matrix();
153 }
154 }
155 return Potential;
156}
