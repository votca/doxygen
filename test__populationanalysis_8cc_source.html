113 orb.BSESinglets().eigenvalues() = se_ref;
114
115 // reference coefficients
116 Eigen::MatrixXd spsi_ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
117 std::string(XTP_TEST_DATA_FOLDER) + "/populationanalysis/spsi_ref.mm");
118
119 orb.BSESinglets().eigenvectors() = spsi_ref;
120 orb.setBSEindices(0, 16);
121 orb.setTDAApprox(true);
122
123 std::vector<QMFragment<BSE_Population> > frags;
124 QMFragment<BSE_Population> f1(0, "0 1");
125 QMFragment<BSE_Population> f2(1, "2 3 4");
126 frags.push_back(f1);
127 frags.push_back(f2);
128
129 orb.DensityMatrixGroundState();
130 QMState n("n");
131 Lowdin low;
132 BOOST_REQUIRE_THROW(low.CalcChargeperFragment(frags, orb, n.Type()),
133 std::runtime_error);
134 QMState s1("s1");
135 low.CalcChargeperFragment(frags, orb, s1.Type());
136
137 BOOST_CHECK_CLOSE(frags[0].value().Gs, 0.5164649, 1e-5);
138 BOOST_CHECK_CLOSE(frags[1].value().Gs, -0.5164628, 1e-5);

148
149 Eigen::VectorXd f2H_ref = Eigen::VectorXd::Zero(3);
150 f2H_ref << 0.342785, 0.377565, 0.34525;
