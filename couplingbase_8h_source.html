66 inline Eigen::MatrixXd CouplingBase::CalculateOverlapMatrix(Orbitals& orbitalsAB){
67  BasisSet dftbasisset;
68  AOBasis dftbasis;
69  dftbasisset.LoadBasisSet(orbitalsAB.getDFTbasisName());

77 inline void CouplingBase::CheckAtomCoordinates(const Orbitals& orbitalsA,
78  const Orbitals& orbitalsB, const Orbitals& orbitalsAB){
79  const std::vector<QMAtom*>& atomsA=orbitalsA.QMAtoms();
80  const std::vector<QMAtom*>& atomsB=orbitalsB.QMAtoms();
81  const std::vector<QMAtom*>& atomsAll = orbitalsAB.QMAtoms();
82  bool coordinates_agree=true;
83  for (unsigned i = 0; i < atomsAll.size(); i++) {
84  QMAtom* dimer = atomsAll[i];
85  QMAtom* monomer = NULL;
86 
87  if (i < atomsA.size()) {
88  monomer = atomsA[i];
89  } else if (i < atomsB.size() + atomsA.size()) {
90  monomer = atomsB[i - atomsA.size()];
91  } else {
92  // Linker
93  CTP_LOG(ctp::logERROR, *_pLog) << (boost::format("Neither Monomer A nor Monomer B contains "
94  "atom %s on line %u. Hence, this atom is part of a linker.") %dimer->getType() %(i+1) ).str()<<std::flush;
95  continue;
96  }
97 
98  if(!monomer->getPos().isClose(dimer->getPos(), 0.001)){
99  coordinates_agree=false;
100  }
101 
102  if (monomer->getType() != dimer->getType()) {
103  throw std::runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
104  }
105  }
106 
107  if(!coordinates_agree){
108  CTP_LOG(ctp::logINFO, *_pLog) << "======WARNING=======\n Coordinates of monomer "
109  "and dimer atoms do not agree" << std::flush;
110  }
111 }
