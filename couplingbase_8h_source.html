68 inline void CouplingBase::CheckAtomCoordinates(
69  const Orbitals& orbitalsA, const Orbitals& orbitalsB,
70  const Orbitals& orbitalsAB) const {
71  const QMMolecule& atomsA = orbitalsA.QMAtoms();
72  const QMMolecule& atomsB = orbitalsB.QMAtoms();
73  const QMMolecule& atomsAll = orbitalsAB.QMAtoms();
74  bool coordinates_agree = true;
75  for (int i = 0; i < atomsAll.size(); i++) {
76  const QMAtom& dimer = atomsAll[i];
77  const QMAtom* monomer = nullptr;
78 
79  if (i < atomsA.size()) {
80  monomer = &atomsA[i];
81  } else if (i < atomsB.size() + atomsA.size()) {
82  monomer = &atomsB[i - atomsA.size()];
83  } else {
84  // Linker
85  XTP_LOG(logERROR, *_pLog)
86  << (boost::format(
87  "Neither Monomer A nor Monomer B contains "
88  "atom %s on line %u. Hence, this atom is part of a linker.") %
89  dimer.getElement() % (i + 1))
90  .str()
91  << std::flush;
92  continue;
93  }
94 
95  if (!monomer->getPos().isApprox(dimer.getPos(), 0.001)) {
96  coordinates_agree = false;
97  }
98 
99  if (monomer->getElement() != dimer.getElement()) {
100  throw std::runtime_error(
101  "\nERROR: Atom types do not agree in dimer and monomers\n");
102  }
103  }
104 
105  if (!coordinates_agree) {
106  XTP_LOG(logINFO, *_pLog) << "======WARNING=======\n Coordinates of monomer "
107  "and dimer atoms do not agree"
108  << std::flush;
109  }
110 }
