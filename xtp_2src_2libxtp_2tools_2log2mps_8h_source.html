74 bool Log2Mps::Evaluate() {
75 
76  // Logger (required for QM package, so we can just as well use it)
77  ctp::Logger log;
78  log.setPreface(ctp::logINFO, "\n... ...");
79  log.setPreface(ctp::logDEBUG, "\n... ...");
80  log.setReportLevel(ctp::logDEBUG);
81  log.setMultithreading(true);
82 
83  // Set-up QM package
84  LOG_SAVE(ctp::logINFO,log) << "Using package <" << _package << ">" << flush;
85  QMPackage *qmpack = QMPackages().Create(_package);
86  qmpack->doGetCharges(true);
87  qmpack->setLog(&log);
88  qmpack->setRunDir(".");
89  qmpack->setLogFileName(_logfile);
90 
91  // Create orbitals, fill with life & extract QM atoms
92  Orbitals orbs;
93  int cdx = qmpack->ParseLogFile(&orbs);
94  if (!cdx) {
95  cout << "\nERROR Parsing " << _logfile << "failed. Abort." << endl;
96  throw std::runtime_error("(see above, parsing error)");
97  }
98  vector<ctp::QMAtom*> &qmatoms = *orbs.getAtoms();
99  vector<ctp::QMAtom*>::iterator it;
100 
101  // Sanity checks, total charge
102  double Q = 0.0;
103  for (it = qmatoms.begin(); it < qmatoms.end(); ++it) {
104  Q += (*it)->charge;
105  }
106 
107  if (qmatoms.size() < 1) {
108  cout << "\nERROR No charges extracted from " << _logfile
109  << ". Abort.\n" << flush;
110  throw std::runtime_error("(see above, input or parsing error)");
111  }
112  LOG_SAVE(ctp::logINFO,log)
113  << qmatoms.size() << " QM atoms, total charge Q = " << Q << flush;
114 
115 
116  // Convert to polar segment & write mps-file
117  QMMInterface qmmface;
118  ctp::PolarSeg *pseg = qmmface.Convert(qmatoms);
119 
120  string tag = "::LOG2MPS "
121  + (boost::format("(log-file='%1$s' : %2$d QM atoms)")
122  % _logfile % qmatoms.size()).str();
123  pseg->WriteMPS(_mpsfile, tag);
124  return true;
125 }
