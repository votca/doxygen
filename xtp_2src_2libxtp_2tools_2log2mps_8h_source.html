78 bool Log2Mps::Evaluate() {
79 
80  // Logger (required for QM package, so we can just as well use it)
81  ctp::Logger log;
82  log.setPreface(ctp::logINFO, "\n... ...");
83  log.setPreface(ctp::logDEBUG, "\n... ...");
84  log.setReportLevel(ctp::logDEBUG);
85  log.setMultithreading(true);
86 
87  // Set-up QM package
88 
89  CTP_LOG_SAVE(ctp::logINFO,log) << "Using package <" << _package << ">" << flush;
90  QMPackage *qmpack = QMPackages().Create(_package);
91  qmpack->doGetCharges(true);
92  qmpack->setLog(&log);
93  qmpack->setRunDir(".");
94  qmpack->setLogFileName(_logfile);
95 
96  // Create orbitals, fill with life & extract QM atoms
97  Orbitals orbs;
98  int cdx = qmpack->ParseLogFile(orbs);
99  if (!cdx) {
100  cout << "\nERROR Parsing " << _logfile << "failed. Abort." << endl;
101  throw std::runtime_error("(see above, parsing error)");
102  }
103  vector<QMAtom*> &qmatoms = orbs.QMAtoms();
104  vector<QMAtom*>::iterator it;
105 
106  // Sanity checks, total charge
107  double Q = 0.0;
108  for (it = qmatoms.begin(); it < qmatoms.end(); ++it) {
109  Q += (*it)->getPartialcharge();
110  }
111 
112  if (qmatoms.size() < 1) {
113  cout << "\nERROR No charges extracted from " << _logfile
114  << ". Abort.\n" << flush;
115  throw std::runtime_error("(see above, input or parsing error)");
116  }
117  CTP_LOG_SAVE(ctp::logINFO,log)
118  << qmatoms.size() << " QM atoms, total charge Q = " << Q << flush;
119 
120 
121  // Convert to polar segment & write mps-file
122  QMInterface qmmface;
123  ctp::PolarSeg pseg = qmmface.Convert(qmatoms);
124 
125  string tag = "::LOG2MPS "
126  + (boost::format("(log-file='%1$s' : %2$d QM atoms)")
127  % _logfile % qmatoms.size()).str();
128  pseg.WriteMPS(_mpsfile, tag);
129  return true;
130 }
