212 inline void BasisSet::LoadBasisSet ( string name )
213 {
214  Property basis_property;
215 
216  // get the path to the shared folders with xml files
217  char *votca_share = getenv("VOTCASHARE");
218  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
219  string xmlFile = string(getenv("VOTCASHARE")) + string("/ctp/basis_sets/") + name + string(".xml");
220 
221  bool success = load_property_from_xml(basis_property, xmlFile);
222 
223  if ( !success ) {; }
224 
225  list<Property*> elementProps = basis_property.Select("basis.element");
226 
227  for (list<Property*> ::iterator ite = elementProps.begin(); ite != elementProps.end(); ++ite)
228  {
229  string elementName = (*ite)->getAttribute<string>("name");
230  Element *element = addElement( elementName );
231  //cout << "\nElement " << elementName;
232 
233  list<Property*> shellProps = (*ite)->Select("shell");
234  for (list<Property*> ::iterator its = shellProps.begin(); its != shellProps.end(); ++its)
235  {
236  string shellType = (*its)->getAttribute<string>("type");
237  double shellScale = (*its)->getAttribute<double>("scale");
238 
239  Shell* shell = element->addShell( shellType, shellScale );
240  //cout << "\n\tShell " << shellType;
241 
242  list<Property*> constProps = (*its)->Select("constant");
243  for (list<Property*> ::iterator itc = constProps.begin(); itc != constProps.end(); ++itc)
244  {
245  double decay = (*itc)->getAttribute<double>("decay");
246  double contraction = (*itc)->getAttribute<double>("contraction");
247  shell->addGaussian(decay, contraction);
248  //cout << "\n\t\t" << decay << " " << contraction << endl;
249  }
250 
251  }
252 
253  }
