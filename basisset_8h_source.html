316 inline void BasisSet::LoadBasisSet ( std::string name )
317 {
318  Property basis_property;
319 
320  // get the path to the shared folders with xml files
321  char *votca_share = getenv("VOTCASHARE");
322  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
323  std::string xmlFile = std::string(getenv("VOTCASHARE")) + std::string("/xtp/basis_sets/") + name + std::string(".xml");
324 
325  bool success = load_property_from_xml(basis_property, xmlFile);
326 
327  if ( !success ) {; }
328 
329  list<Property*> elementProps = basis_property.Select("basis.element");
330 
331  for (list<Property*> ::iterator ite = elementProps.begin(); ite != elementProps.end(); ++ite)
332  {
333  std::string elementName = (*ite)->getAttribute<std::string>("name");
334  Element *element = addElement( elementName );
335  //cout << "\nElement " << elementName;
336 
337  list<Property*> shellProps = (*ite)->Select("shell");
338  for (list<Property*> ::iterator its = shellProps.begin(); its != shellProps.end(); ++its)
339  {
340  std::string shellType = (*its)->getAttribute<std::string>("type");
341  double shellScale = (*its)->getAttribute<double>("scale");
342 
343  Shell* shell = element->addShell( shellType, shellScale );
344  //cout << "\n\tShell " << shellType;
345 
346  list<Property*> constProps = (*its)->Select("constant");
347  for (list<Property*> ::iterator itc = constProps.begin(); itc != constProps.end(); ++itc)
348  {
349  double decay = (*itc)->getAttribute<double>("decay");
350  // << " decay "<<decay<<endl;
351  std::vector<double> contraction;
352  contraction.resize(shell->getLmax()+1);
353  list<Property*> contrProps = (*itc)->Select("contractions");
354  for (list<Property*> ::iterator itcont = contrProps.begin(); itcont != contrProps.end(); ++itcont)
355  {
356  std::string contrType = (*itcont)->getAttribute<std::string>("type");
357  double contrFactor = (*itcont)->getAttribute<double>("factor");
358  //cout << " factor " << contrFactor << endl;
359  if ( contrType == "S" ) contraction[0] = contrFactor;
360  if ( contrType == "P" ) contraction[1] = contrFactor;
361  if ( contrType == "D" ) contraction[2] = contrFactor;
362  if ( contrType == "F" ) contraction[3] = contrFactor;
363  if ( contrType == "G" ) contraction[4] = contrFactor;
364  }
365  shell->addGaussian(decay, contraction);
366  }
367 
368  }
369 
370  }
