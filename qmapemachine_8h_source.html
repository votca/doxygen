43 class QMAPEInterface
44 {
45 public:
46 
47  QMAPEInterface() { _polar_table = CTP::POLAR_TABLE(); };
48  ~QMAPEInterface() {};
49 
50  // CONVERSION QM -> MM
51  CTP::APolarSite *Convert(CTP::QMAtom *atm, int id = -1) {
52  double A_to_nm = 0.1;
53  vec pos = A_to_nm*vec(atm->x, atm->y, atm->z);
54  double q = atm->charge;
55  std::string elem = atm->type;
56  double pol = 0.0;
57  try {
58  pol = _polar_table.at(elem);
59  }
60  catch(out_of_range) {
61  cout << endl << "QMAPEInterface - no default polarizability given "
62  << "for element type '" << elem << "'. Defaulting to 1A**3" << flush;
63  pol = 1e-3;
64  }
65 
66  CTP::APolarSite *new_aps = new CTP::APolarSite(id, elem);
67  new_aps->setRank(0);
68  new_aps->setPos(pos);
69  new_aps->setQ00(q,0); // <- charge state 0 <> 'neutral'
70  new_aps->setIsoP(pol);
71 
72  return new_aps;
73  }
74 
75  CTP::PolarSeg *Convert(std::vector<CTP::QMAtom*> &atms) {
76  CTP::PolarSeg *new_pseg = new CTP::PolarSeg();
77  std::vector<CTP::QMAtom*>::iterator it;
78  for (it = atms.begin(); it < atms.end(); ++it) {
79  CTP::APolarSite *new_site = this->Convert(*it);
80  new_pseg->push_back(new_site);
81  }
82  return new_pseg;
83  }
84 
85  // TODO CONVERSION MM -> QM
86  CTP::QMAtom *Convert(CTP::APolarSite*);
87  std::vector<CTP::QMAtom*> Convert(CTP::PolarSeg*);
88 
89 private:
90 
91  // Allocates polarizabilities in A**3 to element types
92  std::map<std::string,double> _polar_table;
93 
94 };
