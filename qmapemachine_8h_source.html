42 class QMAPEInterface
43 {
44 public:
45 
46  QMAPEInterface() { _polar_table = POLAR_TABLE(); };
47  ~QMAPEInterface() {};
48 
49  // CONVERSION QM -> MM
50  APolarSite *Convert(QMAtom *atm, int id = -1) {
51  double A_to_nm = 0.1;
52  vec pos = A_to_nm*vec(atm->x, atm->y, atm->z);
53  double q = atm->charge;
54  std::string elem = atm->type;
55  double pol = 0.0;
56  try {
57  pol = _polar_table.at(elem);
58  }
59  catch(out_of_range) {
60  cout << endl << "QMAPEInterface - no default polarizability given "
61  << "for element type '" << elem << "'. Defaulting to 1A**3" << flush;
62  pol = 1e-3;
63  }
64 
65  APolarSite *new_aps = new APolarSite(id, elem);
66  new_aps->setRank(0);
67  new_aps->setPos(pos);
68  new_aps->setQ00(q,0); // <- charge state 0 <> 'neutral'
69  new_aps->setIsoP(pol);
70 
71  return new_aps;
72  }
73 
74  PolarSeg *Convert(std::vector<QMAtom*> &atms) {
75  PolarSeg *new_pseg = new PolarSeg();
76  std::vector<QMAtom*>::iterator it;
77  for (it = atms.begin(); it < atms.end(); ++it) {
78  APolarSite *new_site = this->Convert(*it);
79  new_pseg->push_back(new_site);
80  }
81  return new_pseg;
82  }
83 
84  // TODO CONVERSION MM -> QM
85  QMAtom *Convert(APolarSite*);
86  std::vector<QMAtom*> Convert(PolarSeg*);
87 
88 private:
89 
90  // Allocates polarizabilities in A**3 to element types
91  std::map<std::string,double> _polar_table;
92 
93 };
