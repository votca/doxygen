110 /* Custom basis sets are written on a per-element basis to
111  * the system.bas/aux file(s), which are then included in the
112  * Orca input file using GTOName = "system.bas/aux"
113  */
114 void Orca::WriteBasisset(std::vector<QMAtom*>& qmatoms, std::string& bs_name,
115  std::string& el_file_name) {
116 
117  std::vector<std::string> UniqueElements = FindUniqueElements(qmatoms);
118 
119  tools::Elements elementInfo;
120  BasisSet bs;
121  bs.LoadBasisSet(bs_name);
122  CTP_LOG(ctp::logDEBUG, *_pLog) << "Loaded Basis Set " << bs_name << flush;
123  ofstream el_file;
124 
125  el_file.open(el_file_name.c_str());
126  el_file << "$DATA" << endl;
127 
128  for (const std::string& element_name : UniqueElements) {
129  const Element& element = bs.getElement(element_name);
130  el_file << elementInfo.getEleFull(element_name) << endl;
131  for (const Shell& shell : element) {
132  string type = shell.getType();
133  // check combined shells
134  for (unsigned i = 0; i < type.size(); ++i) {
135  string subtype = string(type, i, 1);
136  el_file << subtype << " " << shell.getSize() << endl;
137  int sh_idx = 0;
138  for (const GaussianPrimitive& gaussian : shell) {
139  sh_idx++;
140  el_file << " " << sh_idx << " " << indent(gaussian._decay);
141  el_file << " " << indent(gaussian._contraction[FindLmax(subtype)]);
142  el_file << endl;
143  }
144  }
145  }
146  }
147  el_file << "STOP\n";
148  el_file.close();
149 
150  return;
151 }

156 void Orca::WriteCoordinates(std::ofstream& inp_file,
157  std::vector<QMAtom*>& qmatoms) {
158 
159  for (QMAtom* atom : qmatoms) {

161  inp_file << setw(3) << atom->getType().c_str() << setw(12)
162  << setiosflags(ios::fixed) << setprecision(5) << pos.getX()
163  << setw(12) << setiosflags(ios::fixed) << setprecision(5)
164  << pos.getY() << setw(12) << setiosflags(ios::fixed)
165  << setprecision(5) << pos.getZ() << endl;
166  }
167  inp_file << "* \n" << endl;
168  return;

174 void Orca::WriteECP(std::ofstream& inp_file, std::vector<QMAtom*>& qmatoms) {
175 
176  inp_file << endl;
177  std::vector<std::string> UniqueElements = FindUniqueElements(qmatoms);
178 
179  BasisSet ecp;
180  ecp.LoadPseudopotentialSet(_ecp_name);
181 
182  CTP_LOG(ctp::logDEBUG, *_pLog)
183  << "Loaded Pseudopotentials " << _ecp_name << flush;
184 
185  for (const std::string& element_name : UniqueElements) {
186  try {
187  ecp.getElement(element_name);
188  } catch (std::runtime_error& error) {
189  CTP_LOG(ctp::logDEBUG, *_pLog)
190  << "No pseudopotential for " << element_name << " available" << flush;
191  continue;
192  }
193  const Element& element = ecp.getElement(element_name);
194 
195  inp_file << "\n"
196  << "NewECP"
197  << " " << element_name << endl;
198  inp_file << "N_core"
199  << " " << element.getNcore() << endl;
200  inp_file << "lmax"
201  << " " << getLName(element.getLmax()) << endl;
202  // For Orca the order doesn't matter but let's write it in ascending order
203  // write remaining shells in ascending order s,p,d...
204  for (int i = 0; i <= element.getLmax(); i++) {
205  for (const Shell& shell : element) {
206  if (shell.getLmax() == i) {
207  // shell type, number primitives, scale factor
208  inp_file << shell.getType() << " " << shell.getSize() << endl;
209  int sh_idx = 0;
210  for (const GaussianPrimitive& gaussian : shell) {
211  sh_idx++;
212  inp_file << sh_idx << " " << gaussian._decay << " "
213  << gaussian._contraction[0] << " " << gaussian._power
214  << endl;
215  }
216  }
217  }
218  }
219  inp_file << "end\n "
220  << "\n"
221  << endl;
222  }
223  return;
224 }

233 /* For QM/MM the molecules in the MM environment are represented by
234  * their atomic partial charge distributions. ORCA expects them in
235  * q,x,y,z format in a separate file "background.crg"
236  */
237 void Orca::WriteBackgroundCharges() {
238 
239  std::ofstream crg_file;
240  std::string _crg_file_name_full = _run_dir + "/background.crg";
241  crg_file.open(_crg_file_name_full.c_str());
242  int total_background = 0;
243 
244  for (std::shared_ptr<ctp::PolarSeg> seg : _PolarSegments) {
245  for (ctp::APolarSite* site : *seg) {
246  if (site->getQ00() != 0.0) total_background++;
247  if (site->getRank() > 0 || _with_polarization) {
248  std::vector<std::vector<double>> _split_multipoles =
249  SplitMultipoles(site);
250  total_background += _split_multipoles.size();
251  }
252  }
253  } // counting only
254 
255  crg_file << total_background << endl;
256  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
257  // now write
258  for (std::shared_ptr<ctp::PolarSeg> seg : _PolarSegments) {
259  for (ctp::APolarSite* site : *seg) {
260  string sitestring =
261  boost::str(fmt % site->getQ00() %
262  ((site->getPos().getX()) * votca::tools::conv::nm2ang) %
263  (site->getPos().getY() * votca::tools::conv::nm2ang) %
264  (site->getPos().getZ() * votca::tools::conv::nm2ang));
265  if (site->getQ00() != 0.0) crg_file << sitestring << endl;
266  if (site->getRank() > 0 || _with_polarization) {
267  std::vector<std::vector<double>> _split_multipoles =
268  SplitMultipoles(site);
269  for (const auto& mpoles : _split_multipoles) {
270  string multipole =
271  boost::str(fmt % mpoles[3] % mpoles[0] % mpoles[1] % mpoles[2]);
272  crg_file << multipole << endl;
273  }
274  }
275  }
276  }
277 
278  return;
279 }

286 bool Orca::WriteInputFile(Orbitals& orbitals) {
287 
288  std::vector<std::string> results;
289  std::string temp_suffix = "/id";
290  std::string scratch_dir_backup = _scratch_dir;
291  std::ofstream inp_file;
292  std::string inp_file_name_full = _run_dir + "/" + _input_file_name;
293  inp_file.open(inp_file_name_full.c_str());
294  // header
295  inp_file << "* xyz " << _charge << " " << _spin << endl;
296 
297  std::vector<QMAtom*> qmatoms = orbitals.QMAtoms();
298  // put coordinates
299  WriteCoordinates(inp_file, qmatoms);
300  // add parallelization info
301  inp_file << "%pal\n "
302  << "nprocs " << _threads << "\nend"
303  << "\n"
304  << endl;
305  // basis set info
306  if (_write_basis_set) {
307  std::string el_file_name = _run_dir + "/" + "system.bas";
308  WriteBasisset(qmatoms, _basisset_name, el_file_name);
309  inp_file << "%basis\n " << endl;
310  inp_file << "GTOName"
311  << " "
312  << "="
313  << "\"system.bas\";" << endl;
314  if (_auxbasisset_name != "") {
315  std::string aux_file_name = _run_dir + "/" + "system.aux";
316  WriteBasisset(qmatoms, _auxbasisset_name, aux_file_name);
317  inp_file << "GTOAuxName"
318  << " "
319  << "="
320  << "\"system.aux\";" << endl;
321  }
322  } // write_basis set
323 
324  // ECPs
325  /* WRITING ECP INTO system.inp FILE for ORCA**/
326  if (_write_pseudopotentials) {
327  WriteECP(inp_file, qmatoms);
328  } // write pseudopotentials
329  /* END OF WRITING BASISSET/ECP INTO system.inp FILE for ORCA*************/
330  inp_file << "end\n "
331  << "\n"
332  << endl; // This end is for the basis set block
333  if (_write_charges) {
334  WriteBackgroundCharges();
335  }
336 
337  inp_file << _options << "\n";
338  inp_file << endl;
339  inp_file.close();
340  // and now generate a shell script to run both jobs, if neccessary
341  CTP_LOG(ctp::logDEBUG, *_pLog)
342  << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
343  _scratch_dir = scratch_dir_backup + temp_suffix;
344  WriteShellScript();
345  _scratch_dir = scratch_dir_backup;
346  return true;
347 }

395  << _input_file_name << " failed to start" << flush;

405 void Orca::CleanUp() {
406

409  remove((_run_dir + "/" + "molB.gbw").c_str());
410  remove((_run_dir + "/" + "dimer.gbw").c_str());

413  if (_cleanup.size() != 0) {
414  tools::Tokenizer tok_cleanup(_cleanup, ",");
415  std::vector<std::string> cleanup_info;
416  tok_cleanup.ToVector(cleanup_info);
417  for (const std::string& substring : cleanup_info) {
418  if (substring == "inp") {
419  std::string file_name = _run_dir + "/" + _input_file_name;
420  remove(file_name.c_str());
421  }
422 
423  if (substring == "bas") {
424  std::string file_name = _run_dir + "/system.bas";
425  remove(file_name.c_str());
426  }
427 
428  if (substring == "log") {
429  std::string file_name = _run_dir + "/" + _log_file_name;
430  remove(file_name.c_str());
431  }
432 
433  if (substring == "gbw") {
434  std::string file_name = _run_dir + "/" + _orb_file_name;
435  remove(file_name.c_str());
436  }
437 
438  if (substring == "ges") {

451 bool Orca::ParseLogFile(Orbitals& orbitals) {
452  bool found_success = false;
453  orbitals.setQMpackage("orca");
454  orbitals.setDFTbasisName(_basisset_name);
455  if (_write_pseudopotentials) {
456  orbitals.setECPName(_ecp_name);
457  }
458  CTP_LOG(ctp::logDEBUG, *_pLog) << "Parsing " << _log_file_name << flush;
459  std::string log_file_name_full = _run_dir + "/" + _log_file_name;
460  // check if LOG file is complete
461  if (!CheckLogFile()) return false;
462  std::map<int, double> energies;
463  std::map<int, double> occupancy;
464 
465  std::string line;
466  unsigned levels = 0;
467  int number_of_electrons = 0;
468  std::vector<std::string> results;
469 
470  std::ifstream input_file(log_file_name_full.c_str());
471 
472  if (input_file.fail()) {
473  CTP_LOG(ctp::logERROR, *_pLog)
474  << "File " << log_file_name_full << " not found " << flush;
475  return false;
476  } else {
477  CTP_LOG(ctp::logDEBUG, *_pLog)
478  << "Reading Coordinates and occupationnumbers and energies from "
479  << log_file_name_full << flush;
480  }
481  // Coordinates of the final configuration depending on whether it is an
482  // optimization or not
483  while (input_file) {
484  getline(input_file, line);
485  boost::trim(line);
486 
487  if (_is_optimization) {
488  throw runtime_error("Not implemented yet!");
489  }
490  bool found_optimization = true;
491  std::string::size_type coordinates_pos =
492  line.find("CARTESIAN COORDINATES (ANGSTROEM)");
493 
494  if (found_optimization && coordinates_pos != std::string::npos) {
495  CTP_LOG(ctp::logDEBUG, *_pLog) << "Getting the coordinates" << flush;
496  bool has_QMAtoms = orbitals.hasQMAtoms();
497  // three garbage lines
498  getline(input_file, line);
499  // now starts the data in format
500  // _id type Qnuc x y z
501  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
502  int nfields = row.size();
503  int atom_id = 0;
504  while (nfields == 4) {
505  std::string atom_type = row.at(0);
506  double x = boost::lexical_cast<double>(row.at(1));
507  double y = boost::lexical_cast<double>(row.at(2));
508  double z = boost::lexical_cast<double>(row.at(3));
509  row = GetLineAndSplit(input_file, "\t ");
510  nfields = row.size();
511  tools::vec pos = tools::vec(x, y, z);
512  pos *= tools::conv::ang2bohr;
513  if (has_QMAtoms == false) {
514  orbitals.AddAtom(atom_id, atom_type, pos);
515  } else {
516  QMAtom* pAtom = orbitals.QMAtoms().at(atom_id);
517  pAtom->setPos(pos);
518  }
519  atom_id++;
520  }
521  }
522 
523  std::string::size_type energy_pos = line.find("FINAL SINGLE");
524  if (energy_pos != std::string::npos) {
525  boost::algorithm::split(results, line, boost::is_any_of(" "),
526  boost::algorithm::token_compress_on);
527  std::string energy = results[4];
528  boost::trim(energy);
529  orbitals.setQMEnergy(boost::lexical_cast<double>(energy));
530  CTP_LOG(ctp::logDEBUG, *_pLog)
531  << (boost::format("QM energy[Hrt]: %4.8f ") % orbitals.getQMEnergy())
532  .str()
533  << flush;
534  }
535 
536  std::string::size_type HFX_pos = line.find("Fraction HF Exchange ScalHFX");
537  if (HFX_pos != std::string::npos) {
538  boost::algorithm::split(results, line, boost::is_any_of(" "),
539  boost::algorithm::token_compress_on);
540  double ScaHFX = boost::lexical_cast<double>(results.back());
541  orbitals.setScaHFX(ScaHFX);
542  CTP_LOG(ctp::logDEBUG, *_pLog)
543  << "DFT with " << ScaHFX << " of HF exchange!" << flush;
544  }
545 
546  std::string::size_type dim_pos = line.find("Basis Dimension");
547  if (dim_pos != std::string::npos) {
548  boost::algorithm::split(results, line, boost::is_any_of(" "),
549  boost::algorithm::token_compress_on);
550  std::string dim =
551  results[4]; // The 4th element of results vector is the Basis Dim
552  boost::trim(dim);
553  levels = boost::lexical_cast<int>(dim);
554  CTP_LOG(ctp::logDEBUG, *_pLog) << "Basis Dimension: " << levels << flush;
555  CTP_LOG(ctp::logDEBUG, *_pLog) << "Energy levels: " << levels << flush;
556  }
557 
558  std::string::size_type OE_pos = line.find("ORBITAL ENERGIES");
559  if (OE_pos != std::string::npos) {
560  number_of_electrons = 0;
561  getline(input_file, line);
562  getline(input_file, line);
563  getline(input_file, line);
564  if (line.find("E(Eh)") == std::string::npos) {
565  CTP_LOG(ctp::logDEBUG, *_pLog)
566  << "Warning: Orbital Energies not found in log file" << flush;
567  }
568  for (unsigned i = 0; i < levels; i++) {
569  results = GetLineAndSplit(input_file, " ");
570  std::string no = results[0];
571  boost::trim(no);
572  unsigned levelnumber = boost::lexical_cast<unsigned>(no);
573  if (levelnumber != i) {
574  CTP_LOG(ctp::logDEBUG, *_pLog)
575  << "Have a look at the orbital energies something weird is going "
576  "on"
577  << flush;
578  }
579  std::string oc = results[1];
580  boost::trim(oc);
581  double occ = boost::lexical_cast<double>(oc);
582  // We only count alpha electrons, each orbital must be empty or doubly
583  // occupied
584  if (occ == 2 || occ == 1) {
585  number_of_electrons++;
586  occupancy[i] = occ;
587  } else if (occ == 0) {
588  occupancy[i] = occ;
589  } else {
590  if (occ == 1) {
591  CTP_LOG(ctp::logDEBUG, *_pLog)
592  << "Watch out! No distinction between alpha and beta "
593  "electrons. Check if occ = 1 is suitable for your "
594  "calculation "
595  << flush;
596  number_of_electrons++;
597  occupancy[i] = occ;
598  } else {
599  throw runtime_error(
600  "Only empty or doubly occupied orbitals are allowed not "

631  }
632  pAtom->setPartialcharge(atom_charge);
633  }

639  found_success = true;
640  }
641  }
642

644  << "Alpha electrons: " << number_of_electrons << flush;
645  int occupied_levels = number_of_electrons;
646  int unoccupied_levels = levels - occupied_levels;
647  CTP_LOG(ctp::logDEBUG, *_pLog)
648  << "Occupied levels: " << occupied_levels << flush;
649  CTP_LOG(ctp::logDEBUG, *_pLog)
650  << "Unoccupied levels: " << unoccupied_levels << flush;

653

655 
656  orbitals.setBasisSetSize(levels);
657  orbitals.setNumberOfAlphaElectrons(number_of_electrons);
658  orbitals.setNumberOfOccupiedLevels(occupied_levels);
659  orbitals.setSelfEnergy(0.0);
660 
661  // copying energies to a vector
662  orbitals.MOEnergies().resize(levels);
663  //_level = 1;
664  for (int i = 0; i < orbitals.MOEnergies().size(); i++) {
665  orbitals.MOEnergies()[i] = energies[i];
666  }
667 
668  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done reading Log file" << flush;
669  return found_success;

707 bool Orca::ParseOrbitalsFile(Orbitals& orbitals) {
708  if (!CheckLogFile()) return false;
709  std::vector<double> coefficients;
710  int basis_size = orbitals.getBasisSetSize();
711  int levels = orbitals.getBasisSetSize();
712  if (basis_size == 0 || levels == 0) {
713  throw runtime_error(
714  "Basis size not set, calculator does not parse log file first");
715  }
716  CTP_LOG(ctp::logDEBUG, *_pLog)
717  << "Reading the gbw file, this may or may not work so be careful: "
718  << flush;
719  ifstream infile;
720  infile.open((_run_dir + "/" + _orb_file_name).c_str(), ios::binary | ios::in);
721  if (!infile) {
722  throw runtime_error("Could not open " + _orb_file_name + " file");
723  }
724  infile.seekg(24, ios::beg);
725  char* buffer = new char[8];
726  infile.read(buffer, 8);
727  long int offset = *((long int*)buffer);
728 
729  infile.seekg(offset, ios::beg);
730  infile.read(buffer, 4);
731  int op_read = *((int*)buffer);
732  infile.seekg(offset + 4, ios::beg);
733  infile.read(buffer, 4);
734  int dim_read = *((int*)buffer);
735  infile.seekg(offset + 8, ios::beg);
736  CTP_LOG(ctp::logDEBUG, *_pLog) << "Number of operators: " << op_read
737  << " Basis dimension: " << dim_read << flush;
738  int n = op_read * dim_read * dim_read;
739  delete[] buffer;
740  buffer = new char[8];
741  for (int i = 0; i < n; i++) {
742  infile.read(buffer, 8);
743  double mocoeff = *((double*)buffer);
744  // CTP_LOG(logDEBUG,*_pLog) << mocoeff<< flush ;
745  coefficients.push_back(mocoeff);
746  }
747  delete[] buffer;
748 
749  infile.close();
750  // i -> MO, j -> AO
751  (orbitals.MOCoefficients()).resize(levels, basis_size);
752  for (int i = 0; i < orbitals.MOCoefficients().rows(); i++) {
753  for (int j = 0; j < orbitals.MOCoefficients().cols(); j++) {
754  orbitals.MOCoefficients()(j, i) = coefficients[j * basis_size + i];
755  }
756  }
757  ReorderOutput(orbitals);
758  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done parsing" << flush;
759  return true;
760 }
