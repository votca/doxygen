287  void Orca::WriteBackgroundCharges(std::vector<ctp::PolarSeg*> segments) {
288  std::ofstream _crg_file;
289  std::string _crg_file_name_full = _run_dir + "/background.crg";
290  _crg_file.open(_crg_file_name_full.c_str());
291  int _total_background = 0;
292 
293  std::vector< ctp::PolarSeg* >::iterator it;
294  for (it = segments.begin(); it < segments.end(); it++) {
295  vector<ctp::APolarSite*> ::iterator pit;
296  for (pit = (*it)->begin(); pit < (*it)->end(); ++pit) {
297  if ((*pit)->getQ00() != 0.0) _total_background++;
298 
299  if ((*pit)->getRank() > 0 || _with_polarization ) {
300 
301  std::vector<std::vector<double>> _split_multipoles = SplitMultipoles(*pit);
302  _total_background+= _split_multipoles.size();
303  }
304  }
305  } //counting only
306 
307 
308  _crg_file << _total_background << endl;
309  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
310  //now write
311  for (it = segments.begin(); it < segments.end(); it++) {
312  vector<ctp::APolarSite*> ::iterator pit;
313  for (pit = (*it)->begin(); pit < (*it)->end(); ++pit) {
314 
315  string site=boost::str(fmt % (*pit)->getQ00() % (((*pit)->getPos().getX())*votca::tools::conv::nm2ang)
316  % ((*pit)->getPos().getY()*votca::tools::conv::nm2ang)
317  % ((*pit)->getPos().getZ()*votca::tools::conv::nm2ang)
318  );
319  if ((*pit)->getQ00() != 0.0) _crg_file << site << endl;
320 
321  if ((*pit)->getRank() > 0 || _with_polarization ) {
322 
323  std::vector< std::vector<double> > _split_multipoles = SplitMultipoles(*pit);
324  for (const auto& mpoles:_split_multipoles){
325  string multipole=boost::str( fmt % mpoles[3] % mpoles[0] % mpoles[1] % mpoles[2] );
326  _crg_file << multipole << endl;
327 
328  }
329  }
330  }
331  }

333  return;

340  bool Orca::WriteInputFile(std::vector< ctp::Segment* > segments, Orbitals* orbitals_guess , std::vector<ctp::PolarSeg*> PolarSegments ) {
341 
342  std::vector<std::string> results;
343  std::string temp_suffix = "/id";
344  std::string scratch_dir_backup = _scratch_dir;
345  std::ofstream _com_file;
346 
347  std::string _com_file_name_full = _run_dir + "/" + _input_file_name;
348 
349  _com_file.open(_com_file_name_full.c_str());
350  // header
351  _com_file << "* xyz " << _charge << " " << _spin << endl;
352 
353  std::vector< ctp::QMAtom* > qmatoms;
354  // This is needed for the QM/MM scheme, since only orbitals have
355  // updated positions of the QM region, hence vector<Segments*> is
356  // NULL in the QMMachine and the QM region is also printed here
357  if (_write_charges) {
358  qmatoms = orbitals_guess->QMAtoms();
359  } else {
360  QMMInterface qmmface;
361  qmatoms = qmmface.Convert(segments);
362  }
363 
364  // put coordinates
365  WriteCoordinates(_com_file, qmatoms);
366 
367  // add parallelization info
368  _com_file << "%pal\n " << "nprocs " << _threads << "\nend" << "\n" << endl;
369 
370  // basis set info
371  if (_write_basis_set) {
372 
373  std::string _el_file_name = _run_dir + "/" + "system.bas";
374  WriteBasisset(qmatoms, _basisset_name, _el_file_name);
375  _com_file << "%basis\n " << endl;
376  _com_file << "GTOName" << " " << "=" << "\"system.bas\";" << endl;
377 
378  if (_auxbasisset_name != "") {
379  std::string _aux_file_name = _run_dir + "/" + "system.aux";
380  WriteBasisset(qmatoms, _auxbasisset_name, _aux_file_name);
381  _com_file << "GTOAuxName" << " " << "=" << "\"system.aux\";" << endl;
382  }
383  } // write_basis set
384 
385  // ECPs
386  /* WRITING ECP INTO system.inp FILE for ORCA**/
387  if (_write_pseudopotentials) {
388  WriteECP(_com_file, qmatoms);
389  } // write pseudopotentials
390 
391 
392  /* END OF WRITING BASISSET/ECP INTO system.inp FILE for ORCA*************/
393  _com_file << "end\n " << "\n" << endl; //This end is for the basis set block
394 
395 
396  if (_write_charges) {
397  // actual writing of charges
398  WriteBackgroundCharges( PolarSegments);
399  }
400 
401  _com_file << _options << "\n";
402 
403  if (_write_guess) {
404  throw std::runtime_error("Not implemented in orca");
405  }
406 
407  _com_file << endl;
408  _com_file.close();
409 
410  // and now generate a shell script to run both jobs, if neccessary
411  CTP_LOG(ctp::logDEBUG, *_pLog) << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
412 
413  _scratch_dir = scratch_dir_backup + temp_suffix;
414  WriteShellScript();
415  _scratch_dir = scratch_dir_backup;
416 
417  return true;
418  }

545  // check if LOG file is complete

579

831  bool Orca::ParseOrbitalsFile(Orbitals* _orbitals) {
832  if (!CheckLogFile()) return false;
833  std::vector<double> _coefficients;
834  int _basis_size = _orbitals->getBasisSetSize();
835  int _levels = _orbitals->getNumberOfLevels();

841  CTP_LOG(ctp::logDEBUG, *_pLog) << "Reading the gbw file, this may or may not work so be careful: " << flush;

844  if (!infile) {
845  throw runtime_error("Could not open " + _orb_file_name + " file");
846  }
847  infile.seekg(24, ios::beg);
848  char* buffer = new char [8];
849  infile.read(buffer, 8);
850  long int offset = *((long int*) buffer);
851 
852  infile.seekg(offset, ios::beg);
853  infile.read(buffer, 4);

855  infile.seekg(offset + 4, ios::beg);

860  int n = op_read * dim_read*dim_read;
861  delete[] buffer;
862  buffer = new char [8];
863  for (int i = 0; i < n; i++) {
864  infile.read(buffer, 8);
865  double mocoeff = *((double*) buffer);
866  //CTP_LOG(logDEBUG,*_pLog) << mocoeff<< flush ;
867  _coefficients.push_back(mocoeff);
868  }
869  delete[] buffer;
870 
871  infile.close();
872  //cout<< "basissize " <<_basis_size << endl;
873  //cout << "coeffvektor size "<< _coefficients.size() << endl;
874 
875 
876  // i -> MO, j -> AO
877  (_orbitals->MOCoefficients()).resize(_levels, _basis_size);
878  for (size_t i = 0; i < _orbitals->MOCoefficients().size1(); i++) {
879  for (size_t j = 0; j < _orbitals->MOCoefficients().size2(); j++) {
880  _orbitals->MOCoefficients()(i, j) = _coefficients[j * _basis_size + i];
881 
882  }
883  }
884 
885  ReorderOutput(_orbitals);
886 
887 
888  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done parsing" << flush;
889  return true;
