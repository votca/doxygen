49 /* Custom basis sets are written on a per-element basis to
50  * the system.bas/aux file(s), which are then included in the
51  * Orca input file using GTOName = "system.bas/aux"
52  */
53 void Orca::WriteBasisset(const QMMolecule& qmatoms, std::string& bs_name,
54  std::string& el_file_name) {
55 
56  std::vector<std::string> UniqueElements = qmatoms.FindUniqueElements();
57 
58  tools::Elements elementInfo;
59  BasisSet bs;
60  bs.Load(bs_name);
61  XTP_LOG(Log::error, *_pLog) << "Loaded Basis Set " << bs_name << flush;
62  ofstream el_file;
63 
64  el_file.open(el_file_name);
65  el_file << "$DATA" << endl;
66 
67  for (const std::string& element_name : UniqueElements) {
68  const Element& element = bs.getElement(element_name);
69  el_file << elementInfo.getEleFull(element_name) << endl;
70  for (const Shell& shell : element) {
71  for (const char& subtype : shell.getType()) {
72  el_file << subtype << " " << shell.getSize() << endl;
73  Index sh_idx = 0;
74  for (const GaussianPrimitive& gaussian : shell) {
75  sh_idx++;
76  el_file << " " << sh_idx << " " << indent(gaussian.decay());
77  el_file << " "
78  << indent(gaussian.Contractions()[FindLmax(
79  std::string(1, subtype))]);
80  el_file << endl;
81  }
82  }
83  }
84  }
85  el_file << "STOP\n";
86  el_file.close();
87 
88  return;
89 }

91 /* Coordinates are written in standard Element,x,y,z format to the
92  * input file.
93  */
94 void Orca::WriteCoordinates(std::ofstream& inp_file,
95  const QMMolecule& qmatoms) {
96 
97  for (const QMAtom& atom : qmatoms) {
98  Eigen::Vector3d pos = atom.getPos() * tools::conv::bohr2ang;
99  inp_file << setw(3) << atom.getElement() << setw(12)
100  << setiosflags(ios::fixed) << setprecision(5) << pos.x()
101  << setw(12) << setiosflags(ios::fixed) << setprecision(5)
102  << pos.y() << setw(12) << setiosflags(ios::fixed)
103  << setprecision(5) << pos.z() << endl;
104  }
105  inp_file << "* \n" << endl;
106  return;
107 }

112 void Orca::WriteECP(std::ofstream& inp_file, const QMMolecule& qmatoms) {
113 
114  inp_file << endl;
115  std::vector<std::string> UniqueElements = qmatoms.FindUniqueElements();
116 
117  ECPBasisSet ecp;
118  ecp.Load(_settings.get("ecp"));
119 
120  XTP_LOG(Log::error, *_pLog)
121  << "Loaded Pseudopotentials " << _settings.get("ecp") << flush;
122 
123  for (const std::string& element_name : UniqueElements) {
124  try {
125  ecp.getElement(element_name);
126  } catch (std::runtime_error& error) {
127  XTP_LOG(Log::error, *_pLog)
128  << "No pseudopotential for " << element_name << " available" << flush;
129  continue;
130  }
131  const ECPElement& element = ecp.getElement(element_name);
132 
133  inp_file << "\n"
134  << "NewECP"
135  << " " << element_name << endl;
136  inp_file << "N_core"
137  << " " << element.getNcore() << endl;
138  inp_file << "lmax"
139  << " " << getLName(element.getLmax()) << endl;
140  // For Orca the order doesn't matter but let's write it in ascending order
141  // write remaining shells in ascending order s,p,d...
142  for (Index i = 0; i <= element.getLmax(); i++) {
143  for (const ECPShell& shell : element) {
144  if (shell.getL() == i) {
145  // shell type, number primitives, scale factor
146  inp_file << shell.getType() << " " << shell.getSize() << endl;
147  Index sh_idx = 0;
148  for (const ECPGaussianPrimitive& gaussian : shell) {
149  sh_idx++;
150  inp_file << sh_idx << " " << gaussian._decay << " "
151  << gaussian._contraction << " " << gaussian._power << endl;
152  }
153  }
154  }
155  }
156  inp_file << "end\n "
157  << "\n"
158  << endl;
159  }
160  return;
161 }

168 /* For QM/MM the molecules in the MM environment are represented by
169  * their atomic partial charge distributions. ORCA expects them in
170  * q,x,y,z format in a separate file "background.crg"
171  */
172 void Orca::WriteBackgroundCharges() {
173 
174  std::ofstream crg_file;
175  std::string _crg_file_name_full = _run_dir + "/background.crg";
176  crg_file.open(_crg_file_name_full);
177  Index total_background = 0;
178 
179  for (const std::unique_ptr<StaticSite>& site : _externalsites) {
180  if (site->getCharge() != 0.0) {
181  total_background++;
182  }
183  std::vector<MinimalMMCharge> split_multipoles = SplitMultipoles(*site);
184  total_background += split_multipoles.size();
185  } // counting only
186 
187  crg_file << total_background << endl;
188  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
189  // now write
190  for (const std::unique_ptr<StaticSite>& site : _externalsites) {
191  Eigen::Vector3d pos = site->getPos() * tools::conv::bohr2ang;
192  string sitestring =
193  boost::str(fmt % site->getCharge() % pos.x() % pos.y() % pos.z());
194  if (site->getCharge() != 0.0) {
195  crg_file << sitestring << endl;
196  }
197  std::vector<MinimalMMCharge> split_multipoles = SplitMultipoles(*site);
198  for (const auto& mpoles : split_multipoles) {
199  Eigen::Vector3d pos2 = mpoles._pos * tools::conv::bohr2ang;
200  string multipole =
201  boost::str(fmt % mpoles._q % pos2.x() % pos2.y() % pos2.z());
202  crg_file << multipole << endl;
203  }
204  }
205 
206  return;
207 }

214 bool Orca::WriteInputFile(const Orbitals& orbitals) {
215 
216  std::vector<std::string> results;
217  std::string temp_suffix = "/id";
218  std::string scratch_dir_backup = _scratch_dir;
219  std::ofstream inp_file;
220  std::string inp_file_name_full = _run_dir + "/" + _input_file_name;
221  inp_file.open(inp_file_name_full);
222  // header
223  inp_file << "* xyz " << _charge << " " << _spin << endl;
224  Index threads = OPENMP::getMaxThreads();
225  const QMMolecule& qmatoms = orbitals.QMAtoms();
226  // put coordinates
227  WriteCoordinates(inp_file, qmatoms);
228  // add parallelization info
229  inp_file << "%pal\n"
230  << "nprocs " << threads << "\nend"
231  << "\n"
232  << endl;
233  // basis set info
234  std::string el_file_name = _run_dir + "/" + "system.bas";
235  WriteBasisset(qmatoms, _basisset_name, el_file_name);
236  inp_file << "%basis\n";
237  inp_file << "GTOName"
238  << " "
239  << "="
240  << "\"system.bas\";" << endl;
241  if (_settings.has_key("auxbasisset")) {
242  std::string aux_file_name = _run_dir + "/" + "system.aux";
243  std::string auxbasisset_name = _settings.get("auxbasisset");
244  WriteBasisset(qmatoms, auxbasisset_name, aux_file_name);
245  inp_file << "GTOAuxName"
246  << " "
247  << "="
248  << "\"system.aux\";" << endl;
249  } // write_auxbasis set
250 
251  // ECPs
252  if (_settings.has_key("ecp")) {
253  WriteECP(inp_file, qmatoms);
254  }
255  inp_file << "end\n "
256  << "\n"
257  << endl; // This end is for the basis set block
258  if (_settings.get<bool>("write_charges")) {
259  WriteBackgroundCharges();
260  }
261 
262  // Write Orca section specified by the user
263  for (const auto& prop : this->_settings.property("orca")) {
264  const std::string& prop_name = prop.name();
265  if (prop_name != "method") {
266  _options += this->CreateInputSection("orca." + prop_name);
267  }
268  }
269  // Write main DFT method
270  _options += this->WriteMethod();
271  inp_file << _options;
272  inp_file.close();
273  // and now generate a shell script to run both jobs, if neccessary
274 
275  XTP_LOG(Log::info, *_pLog)
276  << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
277  _scratch_dir = scratch_dir_backup + temp_suffix;
278  WriteShellScript();
279  _scratch_dir = scratch_dir_backup;
280  return true;
281 }

314

326  XTP_LOG(Log::error, *_pLog) << "Orca job failed" << flush;

330  << _input_file_name << " failed to start" << flush;
331  return false;

386 StaticSegment Orca::GetCharges() const {
387 
388  StaticSegment result("charges", 0);
389 
390  XTP_LOG(Log::error, *_pLog) << "Parsing " << _log_file_name << flush;
391  std::string log_file_name_full = _run_dir + "/" + _log_file_name;
392  std::string line;
393 
394  std::ifstream input_file(log_file_name_full);
395  while (input_file) {
396  getline(input_file, line);
397  boost::trim(line);
398  GetCoordinates(result, line, input_file);
399 
400  std::string::size_type charge_pos = line.find("CHELPG Charges");
401 
402  if (charge_pos != std::string::npos) {
403  XTP_LOG(Log::error, *_pLog) << "Getting charges" << flush;
404  getline(input_file, line);
405  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
406  Index nfields = Index(row.size());
407  bool hasAtoms = result.size() > 0;
408  while (nfields == 4) {
409  Index atom_id = boost::lexical_cast<Index>(row.at(0));
410  std::string atom_type = row.at(1);
411  double atom_charge = boost::lexical_cast<double>(row.at(3));
412  row = GetLineAndSplit(input_file, "\t ");
413  nfields = Index(row.size());
414  if (hasAtoms) {
415  StaticSite& temp = result.at(atom_id);
416  if (temp.getElement() != atom_type) {
417  throw std::runtime_error(
418  "Getting charges failed. Mismatch in elemts:" +
419  temp.getElement() + " vs " + atom_type);
420  }
421  temp.setCharge(atom_charge);
422  } else {
423  StaticSite temp =
424  StaticSite(atom_id, atom_type, Eigen::Vector3d::Zero());
425  temp.setCharge(atom_charge);
426  result.push_back(temp);
427  }
428  }
429  }
430  }
431  return result;
432 }

434 Eigen::Matrix3d Orca::GetPolarizability() const {
435  std::string line;
436  ifstream input_file((_run_dir + "/" + _log_file_name));
437  bool has_pol = false;
438 
439  Eigen::Matrix3d pol = Eigen::Matrix3d::Zero();
440  while (input_file) {
441  getline(input_file, line);
442  boost::trim(line);
443 
444  std::string::size_type pol_pos = line.find("THE POLARIZABILITY TENSOR");
445  if (pol_pos != std::string::npos) {
446  XTP_LOG(Log::error, *_pLog) << "Getting polarizability" << flush;
447  getline(input_file, line);
448  getline(input_file, line);
449  getline(input_file, line);
450 
451  if (line.find("The raw cartesian tensor (atomic units)") ==
452  std::string::npos) {
453  throw std::runtime_error(
454  "Could not find cartesian polarisation tensor");
455  }
456 
457  for (Index i = 0; i < 3; i++) {
458  getline(input_file, line);
459  tools::Tokenizer tok2(line, " ");
460  std::vector<std::string> values = tok2.ToVector();
461  if (values.size() != 3) {
462  throw std::runtime_error("Polarisation line " + line +
463  " cannot be parsed");
464  }
465  Eigen::Vector3d row;
466  row << std::stod(values[0]), std::stod(values[1]), std::stod(values[2]);
467  pol.row(i) = row;
468  }
469 
470  has_pol = true;
471  }
472  }
473  if (!has_pol) {
474  throw std::runtime_error("Could not find polarisation in logfile");
475  }
476  return pol;
477 }

479 bool Orca::ParseLogFile(Orbitals& orbitals) {
480  bool found_success = false;
481  orbitals.setQMpackage(getPackageName());
482  orbitals.setDFTbasisName(_basisset_name);
483  if (_settings.has_key("ecp")) {
484  orbitals.setECPName(_settings.get("ecp"));
485  }
486 
487  XTP_LOG(Log::error, *_pLog) << "Parsing " << _log_file_name << flush;
488  std::string log_file_name_full = _run_dir + "/" + _log_file_name;
489  // check if LOG file is complete
490  if (!CheckLogFile()) {
491  return false;
492  }
493  std::map<Index, double> energies;
494  std::map<Index, double> occupancy;
495 
496  std::string line;
497  Index levels = 0;
498  Index number_of_electrons = 0;
499  std::vector<std::string> results;
500 
501  std::ifstream input_file(log_file_name_full);
502 
503  if (input_file.fail()) {
504  XTP_LOG(Log::error, *_pLog)
505  << "File " << log_file_name_full << " not found " << flush;
506  return false;
507  } else {
508  XTP_LOG(Log::error, *_pLog)
509  << "Reading Coordinates and occupationnumbers and energies from "
510  << log_file_name_full << flush;
511  }
512  // Coordinates of the final configuration depending on whether it is an
513  // optimization or not
514 
515  QMMolecule& mol = orbitals.QMAtoms();
516  while (input_file) {
517  getline(input_file, line);
518  boost::trim(line);
519 
520  GetCoordinates(mol, line, input_file);
521 
522  std::string::size_type energy_pos = line.find("FINAL SINGLE");
523  if (energy_pos != std::string::npos) {
524 
525  boost::algorithm::split(results, line, boost::is_any_of(" "),
526  boost::algorithm::token_compress_on);
527  std::string energy = results[4];
528  boost::trim(energy);
529  orbitals.setQMEnergy(boost::lexical_cast<double>(energy));
530  XTP_LOG(Log::error, *_pLog) << (boost::format("QM energy[Hrt]: %4.6f ") %
531  orbitals.getDFTTotalEnergy())
532  .str()
533  << flush;
534  }
535 
536  std::string::size_type HFX_pos = line.find("Fraction HF Exchange ScalHFX");
537  if (HFX_pos != std::string::npos) {
538  boost::algorithm::split(results, line, boost::is_any_of(" "),
539  boost::algorithm::token_compress_on);
540  double ScaHFX = boost::lexical_cast<double>(results.back());
541  orbitals.setScaHFX(ScaHFX);
542  XTP_LOG(Log::error, *_pLog)
543  << "DFT with " << ScaHFX << " of HF exchange!" << flush;
544  }
545 
546  std::string::size_type dim_pos = line.find("Basis Dimension");
547  if (dim_pos != std::string::npos) {
548  boost::algorithm::split(results, line, boost::is_any_of(" "),
549  boost::algorithm::token_compress_on);
550  std::string dim =
551  results[4]; // The 4th element of results vector is the Basis Dim
552  boost::trim(dim);
553  levels = boost::lexical_cast<Index>(dim);
554  XTP_LOG(Log::info, *_pLog) << "Basis Dimension: " << levels << flush;
555  XTP_LOG(Log::info, *_pLog) << "Energy levels: " << levels << flush;
556  }
557 
558  std::string::size_type OE_pos = line.find("ORBITAL ENERGIES");
559  if (OE_pos != std::string::npos) {
560 
561  number_of_electrons = 0;
562  getline(input_file, line);
563  getline(input_file, line);
564  getline(input_file, line);
565  if (line.find("E(Eh)") == std::string::npos) {
566  XTP_LOG(Log::error, *_pLog)
567  << "Warning: Orbital Energies not found in log file" << flush;
568  }
569  for (Index i = 0; i < levels; i++) {
570  results = GetLineAndSplit(input_file, " ");
571  std::string no = results[0];
572  boost::trim(no);
573  Index levelnumber = boost::lexical_cast<Index>(no);
574  if (levelnumber != i) {
575  XTP_LOG(Log::error, *_pLog) << "Have a look at the orbital energies "
576  "something weird is going on"
577  << flush;
578  }
579  std::string oc = results[1];
580  boost::trim(oc);
581  double occ = boost::lexical_cast<double>(oc);
582  // We only count alpha electrons, each orbital must be empty or doubly
583  // occupied
584  if (occ == 2 || occ == 1) {
585  number_of_electrons++;
586  occupancy[i] = occ;
587  if (occ == 1) {
588  XTP_LOG(Log::error, *_pLog)
589  << "Watch out! No distinction between alpha and beta "
590  "electrons. Check if occ = 1 is suitable for your "
591  "calculation "
592  << flush;
593  }
594  } else if (occ == 0) {
595  occupancy[i] = occ;
596  } else {
597  throw runtime_error(
598  "Only empty or doubly occupied orbitals are allowed not "
599  "running the right kind of DFT calculation");
600  }
601  std::string e = results[2];
602  boost::trim(e);
603  energies[i] = boost::lexical_cast<double>(e);
604  }
605  }
606 
607  std::string::size_type success =
608  line.find("* SUCCESS *");
609  if (success != std::string::npos) {
610  found_success = true;
611  }
612  }
613 
614  XTP_LOG(Log::info, *_pLog)
615  << "Alpha electrons: " << number_of_electrons << flush;
616  Index occupied_levels = number_of_electrons;
617  Index unoccupied_levels = levels - occupied_levels;
618  XTP_LOG(Log::info, *_pLog) << "Occupied levels: " << occupied_levels << flush;
619  XTP_LOG(Log::info, *_pLog)
620  << "Unoccupied levels: " << unoccupied_levels << flush;
621 
622  /************************************************************/
623 
624  // copying information to the orbitals object
625 
626  orbitals.setBasisSetSize(levels);
627  orbitals.setNumberOfAlphaElectrons(number_of_electrons);
628  orbitals.setNumberOfOccupiedLevels(occupied_levels);
629 
630  // copying energies to a vector
631  orbitals.MOs().eigenvalues().resize(levels);

660  double x = boost::lexical_cast<double>(row.at(1));
661  double y = boost::lexical_cast<double>(row.at(2));

664  nfields = Index(row.size());

666  pos *= tools::conv::ang2bohr;

669  } else {
670  Atom& pAtom = mol.at(atom_id);
671  pAtom.setPos(pos);
672  }
673  atom_id++;
674  }

713 bool Orca::ParseMOsFile(Orbitals& orbitals) {
714  if (!CheckLogFile()) {
715  return false;
716  }
717  std::vector<double> coefficients;
718  Index basis_size = orbitals.getBasisSetSize();
719  if (basis_size == 0) {
720  throw runtime_error(
721  "Basis size not set, calculator does not parse log file first");
722  }
723 
724  XTP_LOG(Log::error, *_pLog)
725  << "Reading the gbw file, this may or may not work so be careful: "
726  << flush;
727  ifstream infile;
728  infile.open(_run_dir + "/" + _mo_file_name, ios::binary | ios::in);
729  if (!infile) {
730  throw runtime_error("Could not open " + _mo_file_name + " file");
731  }
732  infile.seekg(24, ios::beg);
733  std::array<char, 8> buffer;
734  infile.read(buffer.data(), 8);
735  if (!infile) {
736  infile.close();
737  return false;
738  }
739  Index offset = *((Index*)buffer.data());
740 
741  infile.seekg(offset, ios::beg);
742  infile.read(buffer.data(), 4);
743  if (!infile) {
744  infile.close();
745  return false;
746  }
747  int op_read = *((int*)buffer.data());
748  infile.seekg(offset + 4, ios::beg);
749  infile.read(buffer.data(), 4);
750  if (!infile) {
751  infile.close();
752  return false;
753  }
754  int dim_read = *((int*)buffer.data());
755  infile.seekg(offset + 8, ios::beg);
756  XTP_LOG(Log::info, *_pLog) << "Number of operators: " << op_read
757  << " Basis dimension: " << dim_read << flush;
758  Index n = op_read * dim_read * dim_read;
759  for (Index i = 0; i < n; i++) {
760  infile.read(buffer.data(), 8);
761  if (!infile) {
762  infile.close();
763  return false;
764  }
765  double mocoeff = *((double*)buffer.data());
766  coefficients.push_back(mocoeff);
767  }
768 
769  infile.close();
770  // i -> MO, j -> AO
771  orbitals.MOs().eigenvectors().resize(basis_size, basis_size);
772  for (Index i = 0; i < basis_size; i++) {
773  for (Index j = 0; j < basis_size; j++) {
774  orbitals.MOs().eigenvectors()(j, i) = coefficients[j * basis_size + i];
775  }
776  }
777  ReorderOutput(orbitals);
778  XTP_LOG(Log::error, *_pLog) << "Done parsing" << flush;
779  return true;
780 }
