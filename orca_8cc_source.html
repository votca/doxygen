996  LOG(logDEBUG,*_pLog) << "Reading the gbw file, this may or may not work so be careful: " << flush ;

1001  }

1003  char* buffer= new char [8];
1004  infile.read(buffer,8);
1005  long int offset= *((long int*)buffer);
1006 
1007  infile.seekg(offset,ios::beg);
1008  infile.read(buffer,4);
1009  int op_read= *((int*)buffer);
1010  infile.seekg(offset+4,ios::beg);
1011  infile.read(buffer,4);
1012  int dim_read= *((int*)buffer);

1014  LOG(logDEBUG,*_pLog) << "Number of operators: "<< op_read<< " Basis dimension: "<<dim_read<< flush ;

1016  delete[] buffer;
1017  buffer =new char [8];
1018  for (int i=0;i<n;i++){
1019  infile.read(buffer,8);
1020  double mocoeff=*((double*)buffer);
