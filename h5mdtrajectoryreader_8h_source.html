67  template <typename T1>
68  T1 *ReadVectorData(hid_t ds, hid_t ds_data_type, Index row) {
69  hsize_t offset[3];
70  offset[0] = hsize_t(row);
71  offset[1] = 0;
72  offset[2] = 0;
73  hsize_t chunk_rows[3];
74  chunk_rows[0] = 1;
75  chunk_rows[1] = N_particles_;
76  chunk_rows[2] = vec_components_;
77  hid_t dsp = H5Dget_space(ds);
78  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, nullptr, chunk_rows,
79  nullptr);
80  hid_t mspace1 = H5Screate_simple(vec_components_, chunk_rows, nullptr);
81  T1 *data_out = new T1[N_particles_ * vec_components_];
82  herr_t status =
83  H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
84  if (status < 0) {
85  throw std::runtime_error("Error ReadVectorData: " +
86  boost::lexical_cast<std::string>(status));
87  } else {
88  return data_out;
89  }
90  }

93  template <typename T1>
94  T1 *ReadScalarData(hid_t ds, hid_t ds_data_type, Index row) {
95  hsize_t offset[2];
96  offset[0] = row;
97  offset[1] = 0;
98  hsize_t ch_rows[2];
99  ch_rows[0] = 1;
100  ch_rows[1] = N_particles_;
101  hid_t dsp = H5Dget_space(ds);
102  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, nullptr, ch_rows, nullptr);
103  hid_t mspace1 = H5Screate_simple(2, ch_rows, nullptr);
104  T1 *data_out = new T1[N_particles_];
105  herr_t status =
106  H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
107  if (status < 0) {
108  throw std::runtime_error("Error ReadScalarData: " +
109  boost::lexical_cast<std::string>(status));
110  } else {
111  return data_out;
