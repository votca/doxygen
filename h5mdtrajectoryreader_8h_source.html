68  template <typename T1>
69  T1* ReadVectorData(hid_t ds, hid_t ds_data_type, int row) {
70  hsize_t offset[3]; offset[0] = row; offset[1] = 0; offset[2] = 0;
71  hsize_t chunk_rows[3]; chunk_rows[0] = 1; chunk_rows[1] = N_particles_; chunk_rows[2] = vec_components_;
72  hid_t dsp = H5Dget_space(ds);
73  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, NULL, chunk_rows, NULL);
74  hid_t mspace1 = H5Screate_simple(vec_components_, chunk_rows, NULL);
75  T1 *data_out = new T1[N_particles_ * vec_components_];
76  herr_t status = H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
77  if (status < 0) {
78  throw std::runtime_error("Error ReadVectorData: " + boost::lexical_cast<string>(status));
79  } else
80  return data_out;
81  }

84  template <typename T1>
85  T1* ReadScalarData(hid_t ds, hid_t ds_data_type, int row) {
86  hsize_t offset[2]; offset[0] = row; offset[1] = 0;
87  hsize_t ch_rows[2]; ch_rows[0] = 1; ch_rows[1] = N_particles_;

89  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, NULL, ch_rows, NULL);
