00072     hid_t dsp = H5Dget_space(ds);
00073     H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, NULL, chunk_rows, NULL);
00074     hid_t mspace1 = H5Screate_simple(vec_components_, chunk_rows, NULL);
00075     T1 *data_out = new T1[N_particles_ * vec_components_];
00076     herr_t status = H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
00077     if (status < 0) {
00078       throw std::runtime_error("Error ReadVectorData: " + boost::lexical_cast<string>(status));
00079     } else
