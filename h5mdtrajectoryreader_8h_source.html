67  template <typename T1>
68  T1* ReadVectorData(hid_t ds, hid_t ds_data_type, int row) {
69  hsize_t offset[3]; offset[0] = row; offset[1] = 0; offset[2] = 0;
70  hsize_t chunk_rows[3]; chunk_rows[0] = 1; chunk_rows[1] = N_particles_; chunk_rows[2] = vec_components_;
71  hid_t dsp = H5Dget_space(ds);
72  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, NULL, chunk_rows, NULL);
73  hid_t mspace1 = H5Screate_simple(vec_components_, chunk_rows, NULL);
74  T1 *data_out = new T1[N_particles_ * vec_components_];
75  herr_t status = H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
76  if (status < 0) {
77  throw std::runtime_error("Error ReadVectorData: " + boost::lexical_cast<std::string>(status));
78  } else
79  return data_out;
80  }

83  template <typename T1>
84  T1* ReadScalarData(hid_t ds, hid_t ds_data_type, int row) {
85  hsize_t offset[2]; offset[0] = row; offset[1] = 0;
86  hsize_t ch_rows[2]; ch_rows[0] = 1; ch_rows[1] = N_particles_;
87  hid_t dsp = H5Dget_space(ds);
88  H5Sselect_hyperslab(dsp, H5S_SELECT_SET, offset, NULL, ch_rows, NULL);
89  hid_t mspace1 = H5Screate_simple(2, ch_rows, NULL);
90  T1 *data_out = new T1[N_particles_];
91  herr_t status = H5Dread(ds, ds_data_type, mspace1, dsp, H5P_DEFAULT, data_out);
92  if (status < 0) {
93  throw std::runtime_error("Error ReadScalarData: " + boost::lexical_cast<std::string>(status));
94  } else {
95  return data_out;
