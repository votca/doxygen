67 void IDFT::ParseOptionsXML( votca::tools::Property *options ) {

69  // Orbitals are in fort.7 file; number of electrons in .log file

71  string key = "options." + Identify();

75  string _tasks_string = options->get(key+".tasks").as<string> ();

77  if (_tasks_string.find("run") != std::string::npos) _do_run = true;
78  if (_tasks_string.find("parse") != std::string::npos) _do_parse = true;
79  if (_tasks_string.find("project") != std::string::npos) _do_project = true;

81  if (_tasks_string.find("extract") != std::string::npos) _do_extract = true;

83  string _store_string = options->get(key+".store").as<string> ();
84  if (_store_string.find("orbitals") != std::string::npos) _store_orbitals = true;
85  if (_store_string.find("overlap") != std::string::npos) _store_overlap = true;
86  if (_store_string.find("integrals") != std::string::npos) _store_integrals = true;
87 
88  _max_occupied_levels = options->get(key+".levels").as<int> ();
89  _max_unoccupied_levels = _max_occupied_levels;

95  _trim_factor=-1;

101

110  key = "options."+Identify();

564 void IDFT::PrepareGuess( Orbitals* _orbitalsA, Orbitals* _orbitalsB, Orbitals* _orbitalsAB, Logger *log )
565 {
566 
567  LOG(logDEBUG,*log) << "Constructing the guess for dimer orbitals" << flush;
568 
569  // constructing the direct product orbA x orbB
570  int _basisA = _orbitalsA->getBasisSetSize();
571  int _basisB = _orbitalsB->getBasisSetSize();
572 
573  int _levelsA = _orbitalsA->getNumberOfLevels();
574  int _levelsB = _orbitalsB->getNumberOfLevels();
575 
576  int _electronsA = _orbitalsA->getNumberOfElectrons();
577  int _electronsB = _orbitalsB->getNumberOfElectrons();
578 
579  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
580  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
581 
582  ub::matrix<double>* _mo_coefficients = _orbitalsAB->getOrbitals();
583  //cout << "MO coefficients " << *_mo_coefficients << endl;
584 
585  // AxB = | A 0 | // A = [EA, EB] //
586  // | 0 B | // //
587  _mo_coefficients->resize( _levelsA + _levelsB, _basisA + _basisB );
588  _orbitalsAB->setBasisSetSize( _basisA + _basisB );
589  _orbitalsAB->setNumberOfLevels( _electronsA - _electronsB ,
590  _levelsA + _levelsB - _electronsA - _electronsB );
591  _orbitalsAB->setNumberOfElectrons( _electronsA + _electronsB );
592 
593  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
594  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
595  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
596  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
597 
598  //cout << "MO coefficients " << *_mo_coefficients << endl;
599 
600  ub::vector<double>* _energies = _orbitalsAB->getEnergies();
601  _energies->resize( _levelsA + _levelsB );
602 
603  ub::project( *_energies, ub::range (0, _levelsA ) ) = *_orbitalsA->getEnergies();
604  ub::project( *_energies, ub::range (_levelsA, _levelsA + _levelsB ) ) = *_orbitalsB->getEnergies();
605 
606  //cout << "MO energies " << *_energies << endl;
607 
609
