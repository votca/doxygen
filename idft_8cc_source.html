43 void IDFT::Initialize(votca::tools::Property* options ) {
44 
45  _energy_difference = 0.0;
46 
47  _do_input = false;
48  _do_run = false;
49  _do_parse = false;
50  _do_project = false;
51  _do_trim = false;
52  _do_extract = false;
53 
54  _store_orbitals = false;
55  _store_overlap = false;
56  _store_integrals = false;
57 
58  // update options with the VOTCASHARE defaults
59  UpdateWithDefaults( options, "ctp" );

61 
62  // register all QM packages (Gaussian, turbomole, etc))
63  QMPackageFactory::RegisterAll();

67 void IDFT::ParseOptionsXML( votca::tools::Property *options ) {
68 
69  // Orbitals are in fort.7 file; number of electrons in .log file
70 
71  string key = "options." + Identify();
72  _energy_difference = options->get( key + ".degeneracy" ).as< double > ();
73 
74 
75  string _tasks_string = options->get(key+".tasks").as<string> ();
76  if (_tasks_string.find("input") != std::string::npos) _do_input = true;
77  if (_tasks_string.find("run") != std::string::npos) _do_run = true;
78  if (_tasks_string.find("parse") != std::string::npos) _do_parse = true;
79  if (_tasks_string.find("project") != std::string::npos) _do_project = true;
80  if (_tasks_string.find("trim") != std::string::npos) _do_trim = true;
81  if (_tasks_string.find("extract") != std::string::npos) _do_extract = true;
82 
83  string _store_string = options->get(key+".store").as<string> ();
84  if (_store_string.find("orbitals") != std::string::npos) _store_orbitals = true;
85  if (_store_string.find("overlap") != std::string::npos) _store_overlap = true;
86  if (_store_string.find("integrals") != std::string::npos) _store_integrals = true;
87 
88  _max_occupied_levels = options->get(key+".levels").as<int> ();
89  _max_unoccupied_levels = _max_occupied_levels;
90 
91  if ( options->exists(key+".trim")){
92  _trim_factor = options->get(key + ".trim").as< int > ();
93  }
94  else{
95  _trim_factor=-1;
96  if(!_do_trim){
97  cout << "WARNING: Are you sure you do not want to trim your orbitals, enable trimming by adding ""trim"" to tasks, default for trimming is -1 e.g. trimming to HOMO/LUMO respectively " << endl;;
98  }
99  }
100 
101  string _package_xml = options->get(key+".package").as<string> ();
102 
103  load_property_from_xml( _package_options, _package_xml.c_str() );
104 
105  key = "package";
106  _package = _package_options.get(key+".name").as<string> ();
107 
108  key = "options."+Identify();
109 
110  if ( options->exists(key+".job_file")) {
111  _jobfile = options->get(key+".job_file").as<string>();
112  }
113  else {
114  throw std::runtime_error("Job-file not set. Abort.");
115  }
116

120 void IDFT::LoadOrbitals(string file_name, Orbitals* orbitals, Logger *log ) {
121 
122  CTP_LOG(logDEBUG, *log) << "Loading " << file_name << flush;
123  std::ifstream ifs( file_name.c_str() );
124  boost::archive::binary_iarchive ia( ifs );
125  try {
126  ia >> *orbitals;
127  } catch(std::exception &err) {
128  CTP_LOG(logDEBUG, *log) << "Could not load orbitals from " << file_name << flush;
129  std::cerr << "An error occurred:\n" << err.what() << endl;
130  }
131  ifs.close();

135 Job::JobResult IDFT::EvalJob(Topology *top, Job *job, QMThread *opThread) {
136 
137  string idft_work_dir = "OR_FILES";
138  string edft_work_dir = "OR_FILES";
139  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
140 
141  bool _run_status = false;
142  bool _parse_log_status = false ;
143  bool _parse_orbitals_status = false;
144  bool _calculate_integrals = false;
145  stringstream sout;
146  string output;
147 
148  int HOMO_A;
149  int HOMO_B;
150  int LUMO_A;
151  int LUMO_B;
152  Orbitals _orbitalsA, _orbitalsB;
153  Overlap _overlap;
154 
155  Property _job_summary;
156  Property *_job_output = &_job_summary.add("output","");
157 
158  // report back to the progress observer
159  Job::JobResult jres = Job::JobResult();
160 
161  // get the logger from the thread
162  Logger* pLog = opThread->getLogger();
163 
164  // get the information about the job executed by the thread
165  int _job_ID = job->getId();
166  Property _job_input = job->getInput();
167  list<Property*> segment_list = _job_input.Select( "segment" );
168  int ID_A = segment_list.front()->getAttribute<int>( "id" );
169  string type_A = segment_list.front()->getAttribute<string>( "type" );
170  int ID_B = segment_list.back()->getAttribute<int>( "id" );
171  string type_B = segment_list.back()->getAttribute<string>( "type" );
172 
173  // set the folders
174  string _pair_dir = ( format("%1%%2%%3%%4%%5%") % "pair" % "_" % ID_A % "_" % ID_B ).str();
175 
176  path arg_path, arg_pathA, arg_pathB, arg_pathAB;
177 
178  string orbFileA = (arg_pathA / edft_work_dir / "molecules" / frame_dir / (format("%1%_%2%%3%") % "molecule" % ID_A % ".orb").str()).c_str();
179  string orbFileB = (arg_pathB / edft_work_dir / "molecules" / frame_dir / (format("%1%_%2%%3%") % "molecule" % ID_B % ".orb").str()).c_str();
180  string orbFileAB = (arg_pathAB / idft_work_dir / "pairs" / frame_dir / (format("%1%%2%%3%%4%%5%") % "pair_" % ID_A % "_" % ID_B % ".orb" ).str()).c_str();
181  string _orb_dir = (arg_path / idft_work_dir / "pairs" / frame_dir).c_str();
182 
183  Segment *seg_A = top->getSegment( ID_A );
184  assert( seg_A->getName() == type_A );
185 
186  Segment *seg_B = top->getSegment( ID_B );
187  assert( seg_B->getName() == type_B );
188 
189  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Evaluating pair "
190  << _job_ID << " [" << ID_A << ":" << ID_B << "] out of " <<
191  (top->NBList()).size() << flush;
192 
193 
194  string _qmpackage_work_dir = (arg_path / idft_work_dir / _package / frame_dir / _pair_dir).c_str();
195  // get the corresponding object from the QMPackageFactory
196  QMPackage *_qmpackage = QMPackages().Create( _package );
197  // set a log file for the package
198  _qmpackage->setLog( pLog );
199  // set the run dir
200  _qmpackage->setRunDir( _qmpackage_work_dir );
201  // get the package options
202  _qmpackage->Initialize( &_package_options );
203 
204 
205  // if asked, prepare the input files
206  if (_do_input) {
207  boost::filesystem::create_directories( _qmpackage_work_dir );
208  Orbitals *_orbitalsAB = NULL;
209  if ( _qmpackage->GuessRequested() ) { // do not want to do an SCF loop for a dimer
210  CTP_LOG(logINFO,*pLog) << "Guess requested, reading molecular orbitals" << flush;
211  Orbitals _orbitalsA, _orbitalsB;
212  _orbitalsAB = new Orbitals();
213  // load the corresponding monomer orbitals and prepare the dimer guess
214 
215  // failed to load; wrap-up and finish current job
216  if ( !_orbitalsA.Load( orbFileA ) ) {
217  CTP_LOG(logERROR,*pLog) << "Do input: failed loading orbitals from " << orbFileA << flush;
218  cout << *pLog;
219  output += "failed on " + orbFileA;
220  jres.setOutput( output );
221  jres.setStatus(Job::FAILED);
222  delete _qmpackage;
223  return jres;
224  }
225 
226  if ( !_orbitalsB.Load( orbFileB ) ) {
227  CTP_LOG(logERROR,*pLog) << "Do input: failed loading orbitals from " << orbFileB << flush;
228  cout << *pLog;
229  output += "failed on " + orbFileB;
230  jres.setOutput( output );
231  jres.setStatus(Job::FAILED);
232  delete _qmpackage;
233  return jres;
234  }
235 
236  PrepareGuess(&_orbitalsA, &_orbitalsB, _orbitalsAB, pLog);
237  }
238 
239  // if a pair object is available, take into account PBC, otherwise write as is
240  QMNBList* nblist = &top->NBList();
241  QMPair* pair = nblist->FindPair(seg_A, seg_B);
242 
243  if ( pair == NULL ) {
244  vector < Segment* > segments;
245  segments.push_back( seg_A );
246  segments.push_back( seg_B );
247  CTP_LOG(logWARNING,*pLog) << "PBCs are not taken into account when writing the coordinate file!" << flush;
248  _qmpackage->WriteInputFile(segments, _orbitalsAB);
249  } else {
250  _qmpackage->WriteInputFilePBC(pair, _orbitalsAB);
251  }
252 
253  delete _orbitalsAB;
254  } // end of the input
255 
256 
257  // run the executable
258  if ( _do_run ) {
259  _run_status = _qmpackage->Run( );
260  if ( !_run_status ) {
261  output += "run failed; " ;
262  CTP_LOG(logERROR,*pLog) << _qmpackage->getPackageName() << " run failed" << flush;
263  cout << *pLog;
264  jres.setOutput( output );
265  jres.setStatus(Job::FAILED);
266  delete _qmpackage;
267  return jres;
268  }
269  } // end of the run
270 
271 
272  // This will be later used to write orbitals of the dimer to a file
273  // SOMETHING TO CLEANUP
274  Orbitals _orbitalsAB;
275  // parse the log/orbitals files
276  if ( _do_parse ) {
277  _parse_log_status = _qmpackage->ParseLogFile( &_orbitalsAB );
278 
279  if ( !_parse_log_status ) {
280  output += "log incomplete; ";
281  CTP_LOG(logERROR,*pLog) << "CTP_LOG parsing failed" << flush;
282  cout << *pLog;
283  jres.setOutput( output );
284  jres.setStatus(Job::FAILED);
285  delete _qmpackage;
286  return jres;
287  }
288 
289  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
290 
291  if ( !_parse_orbitals_status ) {
292  output += "fort7 failed; " ;
293  CTP_LOG(logERROR,*pLog) << "Orbitals parsing failed" << flush;
294  cout << *pLog;
295  jres.setOutput( output );
296  jres.setStatus(Job::FAILED);
297  delete _qmpackage;
298  return jres;
299  }
300  } // end of the parse orbitals/log
301 
302  // orbital file used to archive parsed data
303  string _pair_file = ( format("%1%%2%%3%%4%%5%") % "pair_" % ID_A % "_" % ID_B % ".orb" ).str();
304  ub::matrix<double> _JAB;
305 
306 
307  // Orbitals _orbitalsA, _orbitalsB;
308 
309  int _degAH = 1;
310  int _degAL = 1;
311  int _degBH = 1;
312  int _degBL = 1;
313  if ( _do_project ) {
314 
315  // orbitals must be loaded from a file
316  if ( !_do_parse ) LoadOrbitals( orbFileAB, &_orbitalsAB, pLog );
317 
318 
319  // failed to load; wrap-up and finish current job
320  if ( !_orbitalsA.Load( orbFileA ) ) {
321  CTP_LOG(logERROR,*pLog) << "Failed loading orbitals from " << orbFileA << flush;
322  cout << *pLog;
323  output += "failed on " + orbFileA;
324  jres.setOutput( output );
325  jres.setStatus(Job::FAILED);
326  delete _qmpackage;
327  return jres;
328  }
329 
330  if ( !_orbitalsB.Load( orbFileB ) ) {
331  CTP_LOG(logERROR,*pLog) << "Failed loading orbitals from " << orbFileB << flush;
332  cout << *pLog;
333  output += "failed on " + orbFileB;
334  jres.setOutput( output );
335  jres.setStatus(Job::FAILED);
336  delete _qmpackage;
337  return jres;
338  }
339 
340 
341 
342  if (_trim_factor == -1 ) {
343 
344  // find degeneracy of HOMOs and LUMOs
345  std::vector<int> list_levelsAH = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons()-1, _energy_difference ));
346  _degAH = list_levelsAH.size();
347  std::vector<int> list_levelsAL = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons(), _energy_difference ));
348  _degAL = list_levelsAL.size();
349 
350  std::vector<int> list_levelsBH = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons()-1, _energy_difference ));
351  _degBH = list_levelsBH.size();
352  std::vector<int> list_levelsBL = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons(), _energy_difference ));
353  _degBL = list_levelsBL.size();
354  }
355 
356  if ( _do_trim ) {
357 
358 
359  if (_trim_factor == -1 ) {
360  CTP_LOG(logDEBUG,*pLog) << "Trimming orbitals to minimal set " << flush;
361  CTP_LOG(logDEBUG,*pLog) << "HOMO(A)-" << _degAH << " to " << "LUMO(A)+" << _degAL << flush;
362  _orbitalsA.Trim(_degAH,_degAL);
363  CTP_LOG(logDEBUG,*pLog) << "HOMO(B)-" << _degBH << " to " << "LUMO(B)+" << _degBL << flush;
364  _orbitalsB.Trim(_degBH,_degBL);
365 
366  } else {
367 
368 
369  CTP_LOG(logDEBUG,*pLog) << "Trimming virtual orbitals A:"
370  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
371  << _orbitalsA.getNumberOfElectrons()*(_trim_factor-1);
372  _orbitalsA.Trim(_trim_factor);
373 
374  CTP_LOG(logDEBUG,*pLog) << " B:"
375  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
376  << _orbitalsB.getNumberOfElectrons()*(_trim_factor-1) << flush;
377  _orbitalsB.Trim(_trim_factor);
378 
379  }
380  } // _do_trim
381 
382  _overlap.setLogger(pLog);
383 
384  // 10 seconds for a small system
385  //_calculate_integrals = _overlap.CalculateIntegralsOptimized( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
386  // 7 seconds with GSL overloading
387  _calculate_integrals = _overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
388 
389  if ( !_calculate_integrals ) {
390  output += "integrals failed; " ;
391  CTP_LOG(logERROR,*pLog) << "Calculating integrals failed" << flush;
392  cout << *pLog;
393  jres.setOutput( output );
394  jres.setStatus(Job::FAILED);
395  return jres;
396  }
397 
398  HOMO_A = _orbitalsA.getNumberOfElectrons() ;
399  HOMO_B = _orbitalsB.getNumberOfElectrons() ;
400  LUMO_A = HOMO_A + 1;
401  LUMO_B = HOMO_B + 1;
402 
403  double J_h;
404  double J_e;
405 
406  if ( _trim_factor == -1 ){
407  J_h = _overlap.getCouplingElement( _degAH , _degBH, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
408  J_e = _overlap.getCouplingElement( _degAH +1 ,_degBH +1, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
409  } else{
410  J_h = _overlap.getCouplingElement( HOMO_A , HOMO_B, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
411  J_e = _overlap.getCouplingElement( LUMO_A , LUMO_B, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
412  }
413  CTP_LOG(logINFO,*pLog) << "Couplings h/e " << ID_A << ":" << ID_B << " " << J_h << ":" << J_e << flush;
414 
415  // Output the thread run summary and clean the Logger
416  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Finished evaluating pair " << ID_A << ":" << ID_B << flush;
417  //cout << *pLog;
418 
419  // save orbitals
420  boost::filesystem::create_directories(_orb_dir);
421 
422  CTP_LOG(logDEBUG,*pLog) << "Saving orbitals to " << _pair_file << flush;
423  std::ofstream ofs( orbFileAB.c_str() );
424  boost::archive::binary_oarchive oa( ofs );
425 
426  if ( !( _store_orbitals && _do_parse && _parse_orbitals_status) ) {
427  _store_orbitals = false;
428  CTP_LOG(logINFO,*pLog) << "Not storing orbitals" << flush;
429  }
430  if ( !( _store_overlap && _do_parse && _parse_log_status) ) {
431  _store_overlap = false;
432  CTP_LOG(logINFO,*pLog) << "Not storing overlap" << flush;
433  }
434  if ( !( _store_integrals && _do_project && _calculate_integrals) ) {
435  _store_integrals = false;
436  CTP_LOG(logINFO,*pLog) << "Not storing integrals" << flush;
437  } else {
438  // _orbitalsAB.setIntegrals( &_JAB );
439  ub::matrix<double>& _JAB_store = _orbitalsAB.MOCouplings();
440  _JAB_store = _JAB;
441  }
442 
443  _orbitalsAB.setStorage( _store_orbitals, _store_overlap, _store_integrals );
444 
445 
446  oa << _orbitalsAB;
447  ofs.close();
448 
449  /* save project summary
450  * <pair idA="" idB="" typeA="" typeB="">
451  * <overlap orbA="" orbB="" enA="" enB="" ></overlap>
452  * </pair>
453  */
454 
455  } // end of the projection loop
456 
457 
458  if ( _do_extract ) {
459  LoadOrbitals( orbFileAB, &_orbitalsAB, pLog );
460  LoadOrbitals( orbFileA, &_orbitalsA, pLog );
461  LoadOrbitals( orbFileB, &_orbitalsB, pLog );
462  if (_trim_factor == -1 ) {
463 
464  // find degeneracy of HOMOs and LUMOs
465  std::vector<int> list_levelsAH = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons()-1, _energy_difference ));
466  _degAH = list_levelsAH.size();
467  std::vector<int> list_levelsAL = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons(), _energy_difference ));
468  _degAL = list_levelsAL.size();
469 
470  std::vector<int> list_levelsBH = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons()-1, _energy_difference ));
471  _degBH = list_levelsBH.size();
472  std::vector<int> list_levelsBL = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons(), _energy_difference ));
473  _degBL = list_levelsBL.size();
474 
475  _orbitalsA.Trim(_degAH,_degAL);
476  _orbitalsB.Trim(_degBH,_degBL);
477 
478  } else {
479  _orbitalsA.Trim(_trim_factor);
480  _orbitalsB.Trim(_trim_factor);
481  }
482  _JAB = _orbitalsAB.MOCouplings();
483  HOMO_A = _orbitalsA.getNumberOfElectrons() ;
484  HOMO_B = _orbitalsB.getNumberOfElectrons() ;
485  LUMO_A = HOMO_A + 1;
486  LUMO_B = HOMO_B + 1;
487 
488 
489  }
490 
491  if ( _do_project || _do_extract ) {
492  Property *_pair_summary = &_job_output->add("pair","");
493  string nameA = seg_A->getName();
494  string nameB = seg_B->getName();
495  _pair_summary->setAttribute("idA", ID_A);
496  _pair_summary->setAttribute("idB", ID_B);
497  _pair_summary->setAttribute("homoA", HOMO_A);
498  _pair_summary->setAttribute("homoB", HOMO_B);
499  _pair_summary->setAttribute("typeA", nameA);
500  _pair_summary->setAttribute("typeB", nameB);
501 
502  if ( _trim_factor == -1 ) {
503 
504  // HOMO-HOMO coupling
505  double JAB = _overlap.getCouplingElement(_degAH, _degBH , &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference);
506  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
507  double energyA = _orbitalsA.getEnergy(_degAH);
508  double energyB = _orbitalsB.getEnergy(_degBH);
509  _overlap_summary->setAttribute("orbA", HOMO_A);
510  _overlap_summary->setAttribute("orbB", HOMO_B);
511  //_overlap_summary->setAttribute("jAB", JAB);
512  _overlap_summary->setAttribute("eA", energyA);
513  _overlap_summary->setAttribute("eB", energyB);
514 
515  // LUMO-LUMO coupling
516  JAB = _overlap.getCouplingElement(_degAH+1, _degBH+1 , &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference);
517  _overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
518  energyA = _orbitalsA.getEnergy(_degAH +1);
519  energyB = _orbitalsB.getEnergy(_degBH +1);
520  _overlap_summary->setAttribute("orbA", LUMO_A);
521  _overlap_summary->setAttribute("orbB", LUMO_B);
522  //_overlap_summary->setAttribute("jAB", JAB);
523  _overlap_summary->setAttribute("eA", energyA);
524  _overlap_summary->setAttribute("eB", energyB);
525 
526  } else {
527 
528 
529  for (int levelA = HOMO_A - _max_occupied_levels +1; levelA <= LUMO_A + _max_unoccupied_levels - 1; ++levelA ) {
530  for (int levelB = HOMO_B - _max_occupied_levels + 1; levelB <= LUMO_B + _max_unoccupied_levels -1 ; ++levelB ) {
531  Property *_overlap_summary = &_pair_summary->add("overlap","");
532  double JAB = _overlap.getCouplingElement( levelA , levelB, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
533  double energyA = _orbitalsA.getEnergy( levelA );
534  double energyB = _orbitalsB.getEnergy( levelB );
535  _overlap_summary->setAttribute("orbA", levelA);

538  _overlap_summary->setAttribute("eA", energyA);
539  _overlap_summary->setAttribute("eB", energyB);
540  }
541  }
542  }
