162 
163  LOG(logINFO,*pLog) << TimeStamp() << " Evaluating pair "

193  jres.setOutput( output );
194  jres.setStatus(Job::FAILED);

197  }
198

200  LOG(logERROR,*pLog) << "Do input: failed loading orbitals from " << orbFileB << flush;

202  output += "failed on " + orbFileB;

206  return jres;

211

223  _qmpackage->WriteInputFilePBC(pair, _orbitalsAB);

225 
226  delete _orbitalsAB;
227  } // end of the input
228 
229 
230  // run the executable
231  if ( _do_run ) {
232  _run_status = _qmpackage->Run( );
233  if ( !_run_status ) {
234  output += "run failed; " ;
235  LOG(logERROR,*pLog) << _qmpackage->getPackageName() << " run failed" << flush;
236  cout << *pLog;
237  jres.setOutput( output );
238  jres.setStatus(Job::FAILED);
239  delete _qmpackage;
240  return jres;
241  }
242  } // end of the run
243 
244 
245  // This will be later used to write orbitals of the dimer to a file
246  // SOMETHING TO CLEANUP
247  Orbitals _orbitalsAB;
248  // parse the log/orbitals files
249  if ( _do_parse ) {
250  _parse_log_status = _qmpackage->ParseLogFile( &_orbitalsAB );
251 
252  if ( !_parse_log_status ) {
253  output += "log incomplete; ";
254  LOG(logERROR,*pLog) << "LOG parsing failed" << flush;
255  cout << *pLog;
256  jres.setOutput( output );
257  jres.setStatus(Job::FAILED);
258  delete _qmpackage;
259  return jres;
260  }
261 
262  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
263 
264  if ( !_parse_orbitals_status ) {
265  output += "fort7 failed; " ;
266  LOG(logERROR,*pLog) << "Orbitals parsing failed" << flush;
267  cout << *pLog;
268  jres.setOutput( output );
269  jres.setStatus(Job::FAILED);
270  delete _qmpackage;
271  return jres;
272  }
273  } // end of the parse orbitals/log
274 
275  // orbital file used to archive parsed data
276  string _pair_file = ( format("%1%%2%%3%%4%%5%") % "pair_" % ID_A % "_" % ID_B % ".orb" ).str();
277  ub::matrix<double> _JAB;
278 
279  Property _job_summary;
280  if ( _do_project ) {
281 
282  // orbitals must be loaded from a file
283  if ( !_do_parse ) LoadOrbitals( orbFileAB, &_orbitalsAB, pLog );
284 
285  Orbitals _orbitalsA, _orbitalsB;
286 
287  // failed to load; wrap-up and finish current job
288  if ( !_orbitalsA.Load( orbFileA ) ) {
289  LOG(logERROR,*pLog) << "Failed loading orbitals from " << orbFileA << flush;
290  cout << *pLog;
291  output += "failed on " + orbFileA;
292  jres.setOutput( output );
293  jres.setStatus(Job::FAILED);
294  delete _qmpackage;
295  return jres;
296  }
297 
298  if ( !_orbitalsB.Load( orbFileB ) ) {
299  LOG(logERROR,*pLog) << "Failed loading orbitals from " << orbFileB << flush;
300  cout << *pLog;
301  output += "failed on " + orbFileB;
302  jres.setOutput( output );
303  jres.setStatus(Job::FAILED);
304  delete _qmpackage;
305  return jres;
306  }
307 
308  if ( _do_trim ) {
309  LOG(logDEBUG,*pLog) << "Trimming virtual orbitals A:"
310  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
311  << _orbitalsA.getNumberOfElectrons()*(_trim_factor-1);
312  _orbitalsA.Trim(_trim_factor);
313 
314  LOG(logDEBUG,*pLog) << " B:"
315  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
316  << _orbitalsB.getNumberOfElectrons()*(_trim_factor-1) << flush;
317  _orbitalsB.Trim(_trim_factor);
318  }
319 
320  Overlap _overlap;
321  _overlap.setLogger(pLog);
322 
323  // 10 seconds for a small system
324  //_calculate_integrals = _overlap.CalculateIntegralsOptimized( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
325  // 7 seconds with GSL overloading
326  _calculate_integrals = _overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
327 
328  if ( !_calculate_integrals ) {
329  output += "integrals failed; " ;
330  LOG(logERROR,*pLog) << "Calculating integrals failed" << flush;
331  cout << *pLog;
332  jres.setOutput( output );
333  jres.setStatus(Job::FAILED);
334  return jres;
335  }
336 
337  int HOMO_A = _orbitalsA.getNumberOfElectrons() ;
338  int HOMO_B = _orbitalsB.getNumberOfElectrons() ;
339 
340  int LUMO_A = HOMO_A + 1;
341  int LUMO_B = HOMO_B + 1;
342 
343  double J_h = _overlap.getCouplingElement( HOMO_A , HOMO_B, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
344  double J_e = _overlap.getCouplingElement( LUMO_A , LUMO_B, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
345 
346  LOG(logINFO,*pLog) << "Couplings h/e " << ID_A << ":" << ID_B << " " << J_h << ":" << J_e << flush;
347 
348  // Output the thread run summary and clean the Logger
349  LOG(logINFO,*pLog) << TimeStamp() << " Finished evaluating pair " << ID_A << ":" << ID_B << flush;
350  //cout << *pLog;
351 
352  // save orbitals
353  boost::filesystem::create_directories(_orb_dir);
354 
355  LOG(logDEBUG,*pLog) << "Saving orbitals to " << _pair_file << flush;
356  std::ofstream ofs( orbFileAB.c_str() );
357  boost::archive::binary_oarchive oa( ofs );
358 
359  if ( !( _store_orbitals && _do_parse && _parse_orbitals_status) ) {
360  _store_orbitals = false;
361  LOG(logINFO,*pLog) << "Not storing orbitals" << flush;
362  }
363  if ( !( _store_overlap && _do_parse && _parse_log_status) ) {
364  _store_overlap = false;
365  LOG(logINFO,*pLog) << "Not storing overlap" << flush;
366  }
367  if ( !( _store_integrals && _do_project && _calculate_integrals) ) {
368  _store_integrals = false;
369  LOG(logINFO,*pLog) << "Not storing integrals" << flush;
370  } else {
371  _orbitalsAB.setIntegrals( &_JAB );
372  }
373 
374  _orbitalsAB.setStorage( _store_orbitals, _store_overlap, _store_integrals );
375 
376 
377  oa << _orbitalsAB;
378  ofs.close();
379 
380  /* save project summary
381  * <pair idA="" idB="" typeA="" typeB="">
382  * <overlap orbA="" orbB="" enA="" enB="" ></overlap>
383  * </pair>
384  */
385 
386  Property *_job_output = &_job_summary.add("output","");
387  Property *_pair_summary = &_job_output->add("pair","");
388  string nameA = seg_A->getName();
389  string nameB = seg_B->getName();
390  _pair_summary->setAttribute("idA", ID_A);
391  _pair_summary->setAttribute("idB", ID_B);
392  _pair_summary->setAttribute("homoA", HOMO_A);
393  _pair_summary->setAttribute("homoB", HOMO_B);
394  _pair_summary->setAttribute("typeA", nameA);
395  _pair_summary->setAttribute("typeB", nameB);
396  for (int levelA = HOMO_A - _max_occupied_levels +1; levelA <= LUMO_A + _max_unoccupied_levels - 1; ++levelA ) {
397  for (int levelB = HOMO_B - _max_occupied_levels + 1; levelB <= LUMO_B + _max_unoccupied_levels -1 ; ++levelB ) {
398  Property *_overlap_summary = &_pair_summary->add("overlap","");
399  double JAB = _overlap.getCouplingElement( levelA , levelB, &_orbitalsA, &_orbitalsB, &_JAB, _energy_difference );
400  double energyA = _orbitalsA.getEnergy( levelA );
401  double energyB = _orbitalsB.getEnergy( levelB );
402  _overlap_summary->setAttribute("orbA", levelA);
403  _overlap_summary->setAttribute("orbB", levelB);
404  _overlap_summary->setAttribute("jAB", JAB);
405  _overlap_summary->setAttribute("eA", energyA);
406  _overlap_summary->setAttribute("eB", energyB);
407  }
408  }
409 
410  votca::tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 1, "");
411  sout << iomXML << _job_summary;
412  } // end of the projection loop
413 
414  // cleanup whatever is not needed
415  _qmpackage->CleanUp();
416  delete _qmpackage;
417 
418  jres.setOutput( _job_summary );
419  jres.setStatus(Job::COMPLETE);
420 
421  return jres;

425 void IDFT::PrepareGuess( Orbitals* _orbitalsA, Orbitals* _orbitalsB, Orbitals* _orbitalsAB, Logger *log )
426 {
427 
428  LOG(logDEBUG,*log) << "Constructing the guess for dimer orbitals" << flush;
429 
430  // constructing the direct product orbA x orbB
431  int _basisA = _orbitalsA->getBasisSetSize();
432  int _basisB = _orbitalsB->getBasisSetSize();
433 
434  int _levelsA = _orbitalsA->getNumberOfLevels();
435  int _levelsB = _orbitalsB->getNumberOfLevels();
436 
437  int _electronsA = _orbitalsA->getNumberOfElectrons();
438  int _electronsB = _orbitalsB->getNumberOfElectrons();
439 
440  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
441  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
442 
443  ub::matrix<double>* _mo_coefficients = _orbitalsAB->getOrbitals();
444  //cout << "MO coefficients " << *_mo_coefficients << endl;
445 
446  // AxB = | A 0 | // A = [EA, EB] //
447  // | 0 B | // //
448  _mo_coefficients->resize( _levelsA + _levelsB, _basisA + _basisB );
449  _orbitalsAB->setBasisSetSize( _basisA + _basisB );
450  _orbitalsAB->setNumberOfLevels( _electronsA - _electronsB ,
451  _levelsA + _levelsB - _electronsA - _electronsB );
452  _orbitalsAB->setNumberOfElectrons( _electronsA + _electronsB );
453 
454  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
455  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
456  ub::project( *_mo_coefficients, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = *_orbitalsA->getOrbitals();
457  ub::project( *_mo_coefficients, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = *_orbitalsB->getOrbitals();
458 
459  //cout << "MO coefficients " << *_mo_coefficients << endl;
460 
461  ub::vector<double>* _energies = _orbitalsAB->getEnergies();
462  _energies->resize( _levelsA + _levelsB );
463 
464  ub::project( *_energies, ub::range (0, _levelsA ) ) = *_orbitalsA->getEnergies();
465  ub::project( *_energies, ub::range (_levelsA, _levelsA + _levelsB ) ) = *_orbitalsB->getEnergies();
466 
467  //cout << "MO energies " << *_energies << endl;
468 
470 
471 }

615 void IDFT::ReadJobFile(Topology *top) {
616 
617  Property xml;
618 
619  vector<Property*> records;
620 
621  // gets the neighborlist from the topology
622  QMNBList &nblist = top->NBList();
623  int _number_of_pairs = nblist.size();
624  int _current_pairs = 0;
625  int _incomplete_jobs = 0;
626 
627  // output using logger
628  Logger _log;
629  _log.setReportLevel(logINFO);
630 
631  // generate lists of bridges for superexchange pairs
632  nblist.GenerateSuperExchange();
633 
634  // load the QC results in a vector indexed by the pair ID
635  load_property_from_xml(xml, _jobfile);
636  list<Property*> jobProps = xml.Select("jobs.job");
637 
638  records.resize( jobProps.size() + 1 );
639 
640  // loop over all jobs = pair records in the job file
641  for (list<Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
642 
643  // if job produced an output, then continue with analysis
644  if ( (*it)->exists("output") && (*it)->exists("output.pair") ) {
645 
646  // get the output records
647  Property poutput = (*it)->get("output.pair");
648  // id's of two segments of a pair
649  int idA = poutput.getAttribute<int>("idA");
650  int idB = poutput.getAttribute<int>("idB");
651  // segments which correspond to these ids
652  Segment *segA = top->getSegment(idA);
653  Segment *segB = top->getSegment(idB);
654  // pair that corresponds to the two segments
