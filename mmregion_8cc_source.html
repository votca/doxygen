44template <class T>
45void MMRegion<T>::WriteToCpt(CheckpointWriter& w) const {
46 w(id_, "id");
47 w(identify(), "type");
48 Index size = Index(segments_.size());
49 w(size, "size");
50 CheckpointWriter ww = w.openChild("segments");
51 for (const auto& seg : segments_) {
52 CheckpointWriter www =
53 ww.openChild(seg.identify() + "_" + std::to_string(seg.getId()));
54 seg.WriteToCpt(www);
55 }
56}
57template <class T>
58void MMRegion<T>::ReadFromCpt(CheckpointReader& r) {
59 r(id_, "id");
60 Index size;
61 r(size, "size");
62 segments_.clear();
63 segments_.reserve(size);
64 T dummy("dummy", 0);
65 CheckpointReader rr = r.openChild("segments");
66 for (Index i = 0; i < size; i++) {
67 CheckpointReader rrr =
68 rr.openChild(dummy.identify() + "_" + std::to_string(i));
69 segments_.push_back(T(rrr));
70 }
71}
