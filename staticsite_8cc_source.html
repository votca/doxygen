31 Eigen::Matrix3d StaticSite::CalculateCartesianMultipole() const {
32  // We are transforming here just quadrupoles
33  // const Eigen::VectorXd& MP = _multipole;
34  const Vector9d& MP = _Q;
35  Eigen::Matrix3d theta = Eigen::Matrix3d::Zero();
36  if (_rank > 1) {
37  double sqr3 = std::sqrt(3);
38  theta(0, 0) = 0.5 * (-MP(4) + sqr3 * MP(7)); // theta_xx
39  theta(1, 1) = 0.5 * (-MP(4) - sqr3 * MP(7)); // theta_yy
40  theta(2, 2) = MP(4); // theta_zz
41  theta(0, 1) = theta(1, 0) = 0.5 * sqr3 * MP(8); // theta_xy = theta_yx
42  theta(0, 2) = theta(2, 0) = 0.5 * sqr3 * MP(5); // theta_xz = theta_zx
43  theta(1, 2) = theta(2, 1) = 0.5 * sqr3 * MP(6); // theta_yz = theta_zy
44  }
45  return theta;
46 }

48 Eigen::VectorXd StaticSite::CalculateSphericalMultipole(
49  const Eigen::Matrix3d& quad_cart) {
50  Eigen::VectorXd quadrupole_polar = Eigen::VectorXd::Zero(5);
51  const double sqr3 = std::sqrt(3);
52  quadrupole_polar(0) = quad_cart(2, 2);
53  quadrupole_polar(1) = (2. / sqr3) * quad_cart(0, 2);
54  quadrupole_polar(2) = (2. / sqr3) * quad_cart(1, 2);
55  quadrupole_polar(3) = (1. / sqr3) * (quad_cart(0, 0) - quad_cart(1, 1));
56  quadrupole_polar(4) = (2. / sqr3) * quad_cart(0, 1);
57  return quadrupole_polar;
58 }

60 void StaticSite::Rotate(const Eigen::Matrix3d& R,
61  const Eigen::Vector3d& refPos) {
62  Eigen::Vector3d dir = _pos - refPos;
63  dir = R * dir;
64  _pos = refPos + dir; // Rotated Position
65  if (_rank > 0) {
66  const Eigen::Vector3d temp = R * _Q.segment<3>(1);
67  _Q.segment<3>(1) = temp;
68  }
69  if (_rank > 1) {
70  Eigen::Matrix3d cartesianquad = CalculateCartesianMultipole();
71  Eigen::Matrix3d rotated = R * cartesianquad * R.transpose();
72  _Q.segment<5>(4) = CalculateSphericalMultipole(rotated);
73  }
74  return;
75 }

82 std::string StaticSite::writePolarisation() const {
83  tools::Elements e;

85  try {
86  default_pol =

94 std::string StaticSite::WriteMpsLine(string unit) const {
95  double conv_pos = 1.;
96  if (unit == "angstrom") {
97  conv_pos = tools::conv::bohr2ang;
98  } else if (unit == "bohr") {
99  conv_pos = 1.;
100  } else {
101  throw std::runtime_error(
102  " StaticSite::WriteMpsLine: Unit conversion not known");
103  }
104  std::string output = "";
105  output += (boost::format(" %1$2s %2$+1.7f %3$+1.7f %4$+1.7f Rank %5$d\n") %
106  _element % (_pos(0) * conv_pos) % (_pos(1) * conv_pos) %
107  (_pos(2) * conv_pos) % _rank)
108  .str();
109  output += (boost::format(" %1$+1.7f\n") % getCharge()).str();
110  if (_rank > 0) {
111  // Dipole z x y
112  output += (boost::format(" %1$+1.7f %2$+1.7f %3$+1.7f\n") % _Q(3) %
113  _Q(1) % _Q(2))
114  .str();
115  if (_rank > 1) {
116  // Quadrupole 20 21c 21s 22c 22s
117  output +=
118  (boost::format(" %1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f %5$+1.7f\n") %
119  _Q(4) % _Q(5) % _Q(6) % _Q(7) % _Q(8))
120  .str();
121  }
122  }
123  // Polarizability
124  output += writePolarisation();
125  return output;
126 }
