72 void IGWBSE::ParseOptionsXML( votca::tools::Property *opt ) {
73 
74 
75  // parsing general ibse options
76  string key = "options." + Identify();
77  // _energy_difference = opt->get( key + ".degeneracy" ).as< double > ();
78 
79 
80  // job tasks
81  string _tasks_string = opt->get(key+".tasks").as<string> ();
82  if (_tasks_string.find("input") != std::string::npos) _do_dft_input = true;
83  if (_tasks_string.find("dft") != std::string::npos) _do_dft_run = true;
84  if (_tasks_string.find("parse") != std::string::npos) _do_dft_parse = true;
85  if (_tasks_string.find("gwbse") != std::string::npos) _do_gwbse = true;
86  if (_tasks_string.find("coupling") != std::string::npos) _do_coupling = true;
87 
88  // storage options
89  string _store_string = opt->get(key+".store").as<string> ();
90  if (_store_string.find("dft") != std::string::npos) _store_dft = true;
91  if (_store_string.find("singlets") != std::string::npos) _store_singlets = true;
92  if (_store_string.find("triplets") != std::string::npos) _store_triplets = true;
93  if (_store_string.find("ehint") != std::string::npos) _store_ehint = true;
94  if (_store_dft || _store_singlets || _store_triplets || _store_ehint){
95  _write_orbfile=true;
96  }
97  // options for gwbse
98  string _gwbse_xml = opt->get(key+".gwbse_options").as<string> ();
99  //cout << endl << "... ... Parsing " << _package_xml << endl ;
100  load_property_from_xml( _gwbse_options, _gwbse_xml.c_str() );
101  string _coupling_xml=opt->get(key + ".bsecoupling_options").as<string>();
102  load_property_from_xml(_coupling_options, _coupling_xml.c_str());
103 
104  // options for dft package
105  string _package_xml = opt->get(key+".dftpackage").as<string> ();
106  //cout << endl << "... ... Parsing " << _package_xml << endl ;
107  load_property_from_xml( _package_options, _package_xml.c_str() );
108  key = "package";
109  _package = _package_options.get(key+".name").as<string> ();
110 
111  // job file specification
112  key = "options."+Identify();
113 
114  if ( opt->exists(key+".job_file")) {
115  _jobfile = opt->get(key+".job_file").as<string>();
116  }
117  else {
118  throw std::runtime_error("Job-file not set. Abort.");
119  }
120 
121  //options for parsing data into sql file
122  key ="options." + Identify()+".readjobfile";
123  if ( opt->exists(key+".singlets")) {
124  string _parse_string_s = opt->get(key+".singlets").as<string> ();
125  _singlet_levels=FillParseMaps(_parse_string_s);
126  }
127  if ( opt->exists(key+".triplets")) {
128  string _parse_string_t = opt->get(key+".triplets").as<string> ();
129  _triplet_levels=FillParseMaps(_parse_string_t);
130  }
131 
132  return;
133 }

135 std::map<std::string, int> IGWBSE::FillParseMaps(string Mapstring){
136  std::vector<string> strings_vec;
137  boost::algorithm::split( strings_vec, Mapstring, boost::is_any_of("\t,\n"),boost::token_compress_on );
138  std::vector<string>::iterator sit;
139  std::map<std::string, int> type2level;
140  for(sit=strings_vec.begin();sit<strings_vec.end();++sit){
141  //std::vector<string>temp;
142  std::vector<string> segmentpnumber;
143  string temp=*sit;
144  boost::trim(temp);
145  boost::algorithm::split(segmentpnumber , temp, boost::is_any_of(": "),boost::token_compress_on );
146  if (segmentpnumber.size()!=2){
147 
148  throw runtime_error("Parser igwbse: Segment and exciton labels are not separated properly");
149  }
150  if (segmentpnumber[1].size()!=2){
151  throw runtime_error("State identifier "+segmentpnumber[1]+" unknown, right now only states up to number 9 are parsed. s1,s2,t1, etc..");
152  }
153 
154  //boost::algorithm::split( temp, (*sit), boost::is_any_of(""),boost::token_compress_on );
155  int number=boost::lexical_cast<int>(segmentpnumber[1].at(1))-1;
156  string type=boost::lexical_cast<string>(segmentpnumber[0]);
157  type2level[type]=number;
158  }
159  return type2level;
160 }

536 void IGWBSE::ReadJobFile(ctp::Topology *top) {
537 
538  Property xml;
539 
540  vector<Property*> records;
541 
542  // gets the neighborlist from the topology
543  ctp::QMNBList &nblist = top->NBList();
544  int _number_of_pairs = nblist.size();
545  int _current_pairs = 0;
546  int _incomplete_jobs = 0;
547 
548  // output using logger
549  ctp::Logger _log;
550  _log.setReportLevel(ctp::logINFO);
551 
552 
553  // load the QC results in a vector indexed by the pair ID
554  load_property_from_xml(xml, _jobfile);
555  list<Property*> jobProps = xml.Select("jobs.job");
556  records.resize( nblist.size() + 1 );
557  //to skip pairs which are not in the jobfile
558  for (unsigned i=0;i<records.size();i++){
559  records[i]=NULL;
560  }
561  // loop over all jobs = pair records in the job file
562  for (list<Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
563 
564  //int level_segA=1;
565  //int level_segB=1;
566 
567  // if job produced an output, then continue with analysis
568  if ( (*it)->exists("output") && (*it)->exists("output.pair") ) {
569 
570  // get the output records
571  Property poutput = (*it)->get("output.pair");
572  // job file is stupid, because segment ids are only in input have to get them out l
573  list<Property*> pinput = (*it)->Select("input.segment");
574  vector<int> id;
575  for (list<Property*> ::iterator iit = pinput.begin(); iit != pinput.end(); ++iit) {
576  id.push_back((*iit)->getAttribute<int>("id"));
577  }
578  if (id.size()!=2) throw std::runtime_error("Getting pair ids from jobfile failed, check jobfile.");
579 
580  double idA=id[0];
581  double idB=id[1];
582 
583  // segments which correspond to these ids
584  ctp::Segment *segA = top->getSegment(idA);
585  ctp::Segment *segB = top->getSegment(idB);
586  // pair that corresponds to the two segments
587  ctp::QMPair *qmp = nblist.FindPair(segA,segB);
