213 inline bool XJob::isWithinDist(const vec &pt, double dist, Topology *top) {
214 // // This will not work for pairs far apart
215 // bool yesno = false;
216 //
217 // double dR = abs(top->PbShortestConnect(_center, pt));
218 // if (dR <= dist) { yesno = true; }
219 //
220 // return yesno;
221 
222 
223  bool inCenter = false;
224 
225  for (unsigned int i = 0; i < _qmSegs.size(); ++i) {
226  Segment *seg = _qmSegs[i];
227  double dR = abs(top->PbShortestConnect(seg->getPos(), pt));
228  if (dR <= dist) { inCenter = true; break; }
229  }
230 
231  return inCenter;
232 }
