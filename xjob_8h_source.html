210 inline bool XJob::isWithinDist(const vec &pt, double dist, Topology *top) {
211 // // This will not work for pairs far apart
212 // bool yesno = false;
213 //
214 // double dR = abs(top->PbShortestConnect(_center, pt));
215 // if (dR <= dist) { yesno = true; }
216 //
217 // return yesno;
218 
219 
220  bool inCenter = false;
221 
222  for (int i = 0; i < _qmSegs.size(); ++i) {
223  Segment *seg = _qmSegs[i];
224  double dR = abs(top->PbShortestConnect(seg->getPos(), pt));
225  if (dR <= dist) { inCenter = true; break; }
226  }
227 
228  return inCenter;
229 }
