105  vec b = box.getCol(1);
106  vec c = box.getCol(2);
107 
108  if (_refmol == "") {
109 
110  _ref = (a + b + c) / 2;
111 
112  cout << "Refernce is center of box " << _ref << endl;
113  }
114 
115  boxl = abs(a) / 2;
116  if (_rbinw > 0) {
117  _rbins = boxl / _rbinw + 1;
118  cout << "radial bins " << _rbins << endl;

127  _hist_v = new double *[_rbins];
128  _hist_w = new double *[_rbins];

131  _hist_v[i] = new double[_nbin];
132  _hist_w[i] = new double[_nbin];

144  void EndEvaluate() {
145 
146  cout << "Average number of molecules within cutoff " << endl;
147  for (int i = 0; i < _rbins; i++) {
148  cout << i * _rbinw << " " << (double)_nmol[i] / _n << endl;
149  }
150 
151  double exp_value = (double)1 / _nbin;
152  double orderparam = 0;
153 
154  for (int n = 0; n < _nbin; n++) {
155  _hist_u[0][n] /= (double)_nmol[0]; // normalize to numberframes and avg.
156  // number of molecules
157  _hist_v[0][n] /= (double)_nmol[0];
158  _hist_w[0][n] /= (double)_nmol[0];
159 
160  _file_u << (double)n * 2 / (_nbin)-1 << " " << _hist_u[0][n] << endl;
161  _file_v << (double)n * 2 / (_nbin)-1 << " " << _hist_v[0][n] << endl;
162  _file_w << (double)n * 2 / (_nbin)-1 << " " << _hist_w[0][n] << endl;
163 
164  orderparam += (_hist_u[0][n] - exp_value) * (_hist_u[0][n] - exp_value);
165  }
166 
167  orderparam = sqrt(orderparam / _nbin);
168 
169  cout << "Orderparam " << _radialcutoff << " " << orderparam << endl;
170 
171  _file_u.close();
172  _file_v.close();
173  _file_w.close();
174  };

176  void EvalConfiguration(Topology *conf, Topology *conf_atom = 0) {
177 
178  vec eR;
179  int nu, nv, nw;
180  vec u, v, w;
181 
182  if (_refmol != "") {
183  for (BeadContainer::iterator iter = conf->Beads().begin();
184  iter != conf->Beads().end(); ++iter) {
185  Bead *bead = *iter;
186  if (wildcmp(_refmol.c_str(), bead->getName().c_str())) {
187  _ref = bead->getPos();
188  // cout << " Solute pos " << _ref << endl;
189  }
190  }
191  }
192 
193  for (BeadContainer::iterator iter = conf->Beads().begin();
194  iter != conf->Beads().end(); ++iter) {
195  Bead *bead = *iter;
196  if (!wildcmp(_filter.c_str(), bead->getName().c_str())) continue;
197  if (wildcmp(_refmol.c_str(), bead->getName().c_str())) continue;
198 
199  eR = bead->getPos() - _ref;
200  if ((abs(eR) < _radialcutoff && abs(eR) > _minrad) || _rbins != 1) {
201  // cout << eR << endl;
202  int rb = 0;
203  if (_rbinw > 0) {
204  rb = (int)((abs(eR)) / boxl * (double)_rbins);
205  }
206  if (rb >= _rbins) continue;
207  // cout << "rb " << rb << endl;
208 
209  eR.normalize();
210  u = bead->getU();
211  v = bead->getV();
212  w = bead->getW();
213  u.normalize();
214  v.normalize();
215  w.normalize();
216 
217  nu = (int)((((eR * u) + 1) / 2) * _nbin);
218  nv = (int)((((eR * v) + 1) / 2) * _nbin);
219  nw = (int)((((eR * w) + 1) / 2) * _nbin);
220 
221  // cout << "nu" << nu << "nv" << nv << "nw" << nw << endl;
222  _hist_u[rb][nu] += 1;
223  _hist_v[rb][nv] += 1;
224  _hist_w[rb][nw] += 1;
225  _nmol[rb]++;
226  }
227  }
228 
229  _n++;
230  }
