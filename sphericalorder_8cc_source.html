135  void EndEvaluate() override {
136 
137  cout << "Average number of molecules within cutoff " << endl;
138  for (votca::Index i = 0; i < _rbins; i++) {
139  cout << (double)i * _rbinw << " " << (double)_nmol[i] / (double)_n
140  << endl;

142 
143  double exp_value = 1.0 / (double)_nbin;
144  double orderparam = 0;
145 
146  for (votca::Index n = 0; n < _nbin; n++) {
147  _hist_u(0, n) /= (double)_nmol[0]; // normalize to numberframes and avg.
148  // number of molecules
149  _hist_v(0, n) /= (double)_nmol[0];
150  _hist_w(0, n) /= (double)_nmol[0];
151 
152  _file_u << (double)n * 2 / double(_nbin - 1) << " " << _hist_u(0, n)
153  << endl;
154  _file_v << (double)n * 2 / double(_nbin - 1) << " " << _hist_v(0, n)
155  << endl;
156  _file_w << (double)n * 2 / double(_nbin - 1) << " " << _hist_w(0, n)
157  << endl;
158 
159  orderparam += (_hist_u(0, n) - exp_value) * (_hist_u(0, n) - exp_value);
160  }
161 
162  orderparam = sqrt(orderparam / (double)_nbin);
163 
164  cout << "Orderparam " << _radialcutoff << " " << orderparam << endl;

166  _file_u.close();

168  _file_w.close();

171  void EvalConfiguration(Topology *conf, Topology * = nullptr) override {
172 
173  Eigen::Vector3d eR;
174  votca::Index nu, nv, nw;
175  Eigen::Vector3d u, v, w;
176 
177  if (_refmol != "") {
178  for (Bead *bead : conf->Beads()) {
179  if (votca::tools::wildcmp(_refmol, bead->getName())) {
180  _ref = bead->getPos();
181  }
182  }
183  }
184 
185  for (Bead *bead : conf->Beads()) {
186  if (!votca::tools::wildcmp(_filter, bead->getName())) {
187  continue;
188  }
189  if (votca::tools::wildcmp(_refmol, bead->getName())) {
190  continue;
191  }
192 
193  eR = bead->getPos() - _ref;
194  if ((eR.norm() < _radialcutoff && eR.norm() > _minrad) || _rbins != 1) {
195  // cout << eR << endl;
196  votca::Index rb = 0;
197  if (_rbinw > 0) {
198  rb = (votca::Index)((eR.norm()) / boxl * (double)_rbins);
199  }
200  if (rb >= _rbins) {
201  continue;
202  }
203 
204  eR.normalize();
205  u = bead->getU();
206  v = bead->getV();
207  w = bead->getW();
208  u.normalize();
209  v.normalize();
210  w.normalize();
211 
212  nu = (votca::Index)((eR.dot(u) + 1) / 2) * _nbin;
213  nv = (votca::Index)((eR.dot(v) + 1) / 2) * _nbin;
214  nw = (votca::Index)((eR.dot(w) + 1) / 2) * _nbin;
215 
216  _hist_u(rb, nu) += 1;
217  _hist_v(rb, nv) += 1;
218  _hist_w(rb, nw) += 1;
219  _nmol[rb]++;
220  }
221  }
222 
223  _n++;
224  }
