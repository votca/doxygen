174  void EvalConfiguration(Topology *conf, Topology*conf_atom = 0) {
175 
176  vec eR;
177  int nu, nv, nw;
178  vec u, v, w;
179 
180  if (_refmol != ""){
181  for(BeadContainer::iterator iter = conf->Beads().begin();
182  iter!=conf->Beads().end();++iter) {
183  Bead *bead = *iter;
184  if(wildcmp(_refmol.c_str(), bead->getName().c_str())){
185  _ref = bead->getPos();
186  //cout << " Solute pos " << _ref << endl;
187  }
188 
189  }
190  }
191 
192  for(BeadContainer::iterator iter = conf->Beads().begin();
193  iter!=conf->Beads().end();++iter) {
194  Bead *bead = *iter;
195  if(!wildcmp(_filter.c_str(), bead->getName().c_str())) continue;
196  if(wildcmp(_refmol.c_str(), bead->getName().c_str())) continue;
197 
198  eR = bead->getPos()-_ref;
199  if ((abs(eR) < _radialcutoff && abs(eR)>_minrad) ||_rbins!=1) {
200  // cout << eR << endl;
201  int rb=0;
202  if (_rbinw>0){
203  rb = (int) ((abs(eR))/boxl*(double)_rbins);
204 
205  }
206  if (rb>=_rbins) continue;
207  //cout << "rb " << rb << endl;
208 
209  eR.normalize();
210  u = bead->getU();
211  v = bead->getV();
212  w = bead->getW();
213  u.normalize();
214  v.normalize();
215  w.normalize();
216 
217  nu = (int) ((((eR * u) + 1) / 2) * _nbin);
218  nv = (int) ((((eR * v) + 1) / 2) * _nbin);
219  nw = (int) ((((eR * w) + 1) / 2) * _nbin);
220 
221  //cout << "nu" << nu << "nv" << nv << "nw" << nw << endl;
222  _hist_u[rb][nu] += 1;
223  _hist_v[rb][nv] += 1;
224  _hist_w[rb][nw] += 1;
225  _nmol[rb]++;
226 
227  }
228  }
229 
230  _n++;
231  }
