127 inline void GraphSQL<TNode,TLink>::Initialize_cubic(Eventinfo* eventinfo, votca::tools::Random2 *RandomVariable)
128 {
129  int NX = eventinfo->NX;
130  int NY = eventinfo->NY;
131  int NZ = eventinfo->NZ;
132  double lat_const = eventinfo->lat_const;
133 
134  int node_index = 0;
135 
136  for(int ix=0; ix<NX; ix++) {
137  for(int iy=0; iy<NY; iy++) {
138  for(int iz=0; iz<NZ; iz++) {
139 
140  int id = node_index;
141  votca::tools::vec position = votca::tools::vec(ix*lat_const,iy*lat_const,iz*lat_const);
142  TNode* newTNode = this->AddNode(id,position);
143 
144  node_index++;
145 
146  double randn_el = RandomVariable->rand_gaussian(eventinfo->el_disorder);
147  double UcCnNe = randn_el; //disorder
148  double randn_ho=0;
149  double UcCnNh;
150  if(eventinfo->el_ho_correlation == 0) { //no correlation
151  randn_ho = RandomVariable->rand_gaussian(eventinfo->ho_disorder);
152  }
153  else if(eventinfo->el_ho_correlation == 1) { //perfect correlation
154  randn_ho = 1.0*randn_el;
155  }
156  else if(eventinfo->el_ho_correlation == -1) { // perfect anti-correlation
157  randn_ho = -1.0*randn_el;
158  }
159  UcCnNh = randn_ho;
160 
161  double eAnion = eventinfo->lumo; //lumo
162  double eNeutral = 0.0;
163  double eCation = eventinfo->homo; //homo
164 
165  newTNode->setE(eAnion, eNeutral, eCation);
166  newTNode->setu(UcCnNe, UcCnNh);
167 
168  }
169  }
170  }
171 
172  std::cout << "nodes created" << endl;
173 
174  node_index = 0;
175  int hopd = ceil(eventinfo->hop_distance);
176  long link_index = 0;
177 
178  for(int ix=0; ix<NX; ix++) {
179  for(int iy=0; iy<NY; iy++) {
180  for(int iz=0; iz<NZ; iz++) {
181 
182  int node1_id = node_index;
183  TNode* node1 = this->GetNode(node1_id);
184  for(int dx = - hopd; dx <= hopd; dx++) {
185  for(int dy = - hopd; dy <= hopd; dy++) {
186  for(int dz = - hopd; dz <= hopd; dz++) {
187  if(!(dx == 0 && (dy == 0 && dz == 0))) {
188 
189  int new_ix = ix + dx;
190  int new_iy = iy + dy;
191  int new_iz = iz + dz;
192 
193  if(new_ix<0) new_ix += NX;
194  if(new_ix>NX-1) new_ix -= NX;
195  if(new_iy<0) new_iy += NY;
196  if(new_iy>NY-1) new_iy -= NY;
197  if(new_iz<0) new_iz += NZ;
198  if(new_iz>NZ-1) new_iz -= NZ;
199 
200  int node2_id = new_iz + new_iy*NZ + new_ix*NZ*NY;
201  TNode* node2 = this->GetNode(node2_id);
202 
203  double drX = 1.0*eventinfo->lat_const*dx;
204  double drY = 1.0*eventinfo->lat_const*dy;
205  double drZ = 1.0*eventinfo->lat_const*dz;
206  votca::tools::vec r12(drX,drY,drZ);
207  this->AddLink(link_index,node1, node2, r12);
208  //TLink* newTLink = this->AddLink(link_index,node1, node2, r12);
209  link_index++;
210  }
211  }
212  }
213  }
214 
215  node_index++;
216  }
217  }
218  }
219 
220  std::cout << "links created" << endl;
