158  row3.setX(0);
159  row3.setY(1);
160  row3.setZ(0);

163  vec center;
164  center.setX(0.0);
165  center.setY(0.0);
166  center.setZ(0.0);
167 
168  matrix rot_mat(row1,row2,row3);
169 
170  // The qmpos is not used to calculate the positions of the fragment
171  // when calcPos is called with the MD tag which is also the default
172  vec qmpos;
173  qmpos.setX(2.0);
174  qmpos.setY(2.0);
175  qmpos.setZ(2.0);
176 
177  vec pos;
178  pos.setX(3.0);
179  pos.setY(3.0);
180  pos.setZ(3.0);
181 
182  // If set to false cannot calculate the QM position of fragment when
183  // the atom is added
184  bool hasQM = true;
185  Atom * atm = new Atom(nullptr, "res1",1,"CSP",2,hasQM,3,qmpos,"C",1.0);
186  atm->setPos(pos);

189  frag.AddAtom(atm);
190  // Default is MD
191  frag.calcPos();
