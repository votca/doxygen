54bool XTPDFT::RunDFT() {
55 DFTEngine xtpdft;
56 xtpdft.Initialize(options_);
57 xtpdft.setLogger(pLog_);
58
59 if (!externalsites_.empty()) {
60 xtpdft.setExternalcharges(&externalsites_);
61 }
62 bool success = xtpdft.Evaluate(orbitals_);
63 std::string file_name = run_dir_ + "/" + log_file_name_;
64 XTP_LOG(Log::error, *pLog_)
65 << "Writing result to " << log_file_name_ << flush;
66 orbitals_.WriteToCpt(file_name);
67 return success;
68}

83void XTPDFT::CleanUp() {
84 if (cleanup_.size() != 0) {
85 XTP_LOG(Log::info, *pLog_) << "Removing " << cleanup_ << " files" << flush;
86 std::vector<std::string> cleanup_info =
87 tools::Tokenizer(cleanup_, ", ").ToVector();
88 for (const std::string& substring : cleanup_info) {
89 if (substring == "log") {
90 std::string file_name = run_dir_ + "/" + log_file_name_;
91 remove(file_name.c_str());
92 }
93 }
94 }
95
96 return;
97}

104bool XTPDFT::ParseLogFile(Orbitals& orbitals) {
105 try {
106 std::string file_name = run_dir_ + "/" + log_file_name_;

112 } catch (std::runtime_error& error) {
