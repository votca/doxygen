63 void XTPDFT::CleanUp() {
64  if (_cleanup.size() != 0) {
65  XTP_LOG(logDEBUG, *_pLog) << "Removing " << _cleanup << " files" << flush;
66  tools::Tokenizer tok_cleanup(_cleanup, ", ");
67  std::vector<std::string> cleanup_info;
68  tok_cleanup.ToVector(cleanup_info);
69  for (const std::string& substring : cleanup_info) {
70  if (substring == "log") {
71  std::string file_name = _run_dir + "/" + _log_file_name;
72  remove(file_name.c_str());
73  }
74  }
75  }
76 
77  return;
78 }
