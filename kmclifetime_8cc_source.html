138  vector<GNode*> KMCLifetime::LoadGraph() {

142  votca::tools::Database db;
143  db.Open(_filename);
144 
145  votca::tools::Statement *stmt;
146  if (_carriertype == "h" || _carriertype == "e") {
147  stmt = db.Prepare("SELECT _id-1, name, posX, posY, posZ, UnCnN" + _carriertype + ", UcNcC" + _carriertype + ",eAnion,eNeutral,eCation,UcCnN" + _carriertype + " FROM segments;");
148  } else if (_carriertype == "s" || _carriertype == "t") {
149  stmt = db.Prepare("SELECT _id-1, name, posX, posY, posZ, UnXnN" + _carriertype + ", UxNxX" + _carriertype + ",eSinglet,eNeutral,eTriplet,UxXnN" + _carriertype + " FROM segments;");
150  } else{
151  throw runtime_error("Carriertype "+_carriertype+" not known.");
152  }
153 
154  int i = 0;
155  while (stmt->Step() != SQLITE_DONE) {
156  GNode *newNode = new GNode();
157  node.push_back(newNode);
158 
159  int newid = stmt->Column<int>(0);
160  string name = stmt->Column<string>(1);
161  node[i]->id = newid;
162  vec nodeposition = vec(stmt->Column<double>(2)*1E-9, stmt->Column<double>(3)*1E-9, stmt->Column<double>(4)*1E-9); // converted from nm to m
163  node[i]->position = nodeposition;
164  node[i]->reorg_intorig = stmt->Column<double>(5); // UnCnN or UnXnN
165  node[i]->reorg_intdest = stmt->Column<double>(6); // UcNcC or UxNxX
166  double eAnion = stmt->Column<double>(7);
167  //double eNeutral = stmt->Column<double>(8);
168  double eCation = stmt->Column<double>(9);
169  double internalenergy = stmt->Column<double>(10); // UcCnN or UxXnN
170  double siteenergy = 0;
171  if (_carriertype == "e" || _carriertype == "s") {
172  siteenergy = eAnion + internalenergy;
173  } else if (_carriertype == "h" || _carriertype == "t") {
174  siteenergy = eCation + internalenergy;
175  }
176 
177  node[i]->siteenergy = siteenergy;
178  if (votca::tools::wildcmp(_injection_name.c_str(), name.c_str())) {
179  node[i]->injectable = 1;
180  } else {
181  node[i]->injectable = 0;
182  }
183  i++;
184  }
185  delete stmt;
186 
187  ReadLifetimeFile(_lifetimefile,node);
188 
189  // Load pairs and rates
190  int numberofpairs = 0;
191  stmt = db.Prepare("SELECT seg1-1 AS 'segment1', seg2-1 AS 'segment2', rate12" + _carriertype + " AS 'rate', drX, drY, drZ, Jeff2" + _carriertype + ", lO" + _carriertype + " FROM pairs UNION SELECT seg2-1 AS 'segment1', seg1-1 AS 'segment2', rate21" + _carriertype + " AS 'rate', -drX AS 'drX', -drY AS 'drY', -drZ AS 'drZ', Jeff2" + _carriertype + ", lO" + _carriertype + " FROM pairs ORDER BY segment1;");
192  while (stmt->Step() != SQLITE_DONE) {
193  int seg1 = stmt->Column<int>(0);
194  int seg2 = stmt->Column<int>(1);
195 
196  double rate12 = stmt->Column<double>(2);
197 
198  vec dr = vec(stmt->Column<double>(3)*1E-9, stmt->Column<double>(4)*1E-9, stmt->Column<double>(5)*1E-9); // converted from nm to m
199  double Jeff2 = stmt->Column<double>(6);
200  double reorg_out = stmt->Column<double>(7);
201  node[seg1]->AddEvent(seg2, rate12, dr, Jeff2, reorg_out);
202  numberofpairs++;
203  }
204  delete stmt;
205 
206  if (votca::tools::globals::verbose) {
207  cout << "pairs: " << numberofpairs / 2 << endl;
208  }
209 
210  //get boxsize
211  double boxsize = 1.0;
212  stmt = db.Prepare("SELECT box11,box12,box13,box21,box22,box23,box31,box32,box33 FROM frames");
213  while (stmt->Step() != SQLITE_DONE) {
214  vec vecx = vec(stmt->Column<double>(0), stmt->Column<double>(1), stmt->Column<double>(2));
215  vec vecy = vec(stmt->Column<double>(3), stmt->Column<double>(4), stmt->Column<double>(5));
216  vec vecz = vec(stmt->Column<double>(6), stmt->Column<double>(7), stmt->Column<double>(8));
217  boxsize = std::abs(((vecx^vecy) * vecz));
218  }
219 
220  cout << "spatial density: " << _numberofcharges / boxsize << " nm^-3" << endl;
221 
222  for (unsigned int i = 0; i < node.size(); i++) {
223  node[i]->InitEscapeRate();
224  }
225  return node;
226  }

284  void KMCLifetime::ReadLifetimeFile(string filename,vector<GNode*> node){
285  Property xml;
286  load_property_from_xml(xml, filename);
287  list<Property*> jobProps = xml.Select("lifetimes.site");
288  if (jobProps.size()!=node.size()){
289  throw runtime_error((boost::format("The number of sites in the sqlfile: %i does not match the number in the lifetimefile: %i") % node.size() % jobProps.size()).str());
290 
291  }
292 
293  for (list<Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
294  int site_id =(*it)->getAttribute<int>("id");
295  double lifetime=boost::lexical_cast<double>((*it)->value());
296  bool check=false;
297  for (unsigned i=0;i<node.size();i++){
298  if (node[i]->id==site_id-1 && !(node[i]->hasdecay)){
299  node[i]->AddDecayEvent(1.0/lifetime);
300  check=true;
301  break;
302  }
303  else if(node[i]->id==site_id && node[i]->hasdecay){
304  throw runtime_error((boost::format("Node %i appears twice in your list") %site_id).str());
305  }
306 
307  }
308  if (!check){
309  throw runtime_error((boost::format("Site from file with id: %i not found in sql") %site_id).str());
310  }

315  void KMCLifetime::InitialRates(vector<GNode*> node) {
316  cout << endl << "Calculating initial Marcus rates." << endl;
317  cout << " Temperature T = " << _temperature << " K." << endl;
318 
319  cout << " carriertype: " << _carriertype << endl;
320  unsigned numberofsites = node.size();
321  cout << " Rates for " << numberofsites << " sites are computed." << endl;
322  double maxreldiff = 0;
323  int totalnumberofrates = 0;
324  for (unsigned int i = 0; i < numberofsites; i++) {
325  unsigned numberofneighbours = node[i]->event.size();
326  for (unsigned int j = 0; j < numberofneighbours; j++) {
327  if(node[i]->event[j].decayevent){
328  //if event is a decay event there is no point in calculating its rate, because it already has that from the reading in.
329  continue;
330  }
331 
332 
333 
334  double destindex = node[i]->event[j].destination;
335  double reorg = node[i]->reorg_intorig + node[destindex]->reorg_intdest + node[i]->event[j].reorg_out;
336 
337  double dG = node[destindex]->siteenergy - node[i]->siteenergy;
338 
339  double J2 = node[i]->event[j].Jeff2;
340 
341  double rate = 2 * Pi / conv::hbar * J2 / sqrt(4 * Pi * reorg * conv::kB * _temperature) * exp(-(dG + reorg)*(dG + reorg) / (4 * reorg * conv::kB * _temperature));
342 
343  // calculate relative difference compared to values in the table
344  double reldiff = (node[i]->event[j].rate - rate) / node[i]->event[j].rate;
345  if (reldiff > maxreldiff) {
346  maxreldiff = reldiff;
347  }
348  reldiff = (node[i]->event[j].rate - rate) / rate;
349  if (reldiff > maxreldiff) {
350  maxreldiff = reldiff;
351  }
352 
353  // set rates to calculated values
354  node[i]->event[j].rate = rate;
355  node[i]->event[j].initialrate = rate;
356 
357  totalnumberofrates++;
358 
359  }
360 
361  // Initialise escape rates
362  for (unsigned int i = 0; i < node.size(); i++) {
363  node[i]->InitEscapeRate();
364  }
365 
366  }
367  cout << " " << totalnumberofrates << " rates have been calculated." << endl;
368  if (maxreldiff < 0.01) {
369  cout << " Good agreement with rates in the state file. Maximal relative difference: " << maxreldiff * 100 << " %" << endl;
370  } else {
371  cout << " WARNING: Rates differ from those in the state file up to " << maxreldiff * 100 << " %." << " If the rates in the state file are calculated for a different temperature/field or if they are not Marcus rates, this is fine. Otherwise something might be wrong here." << endl;
372  }
373  return;
374  }

376  double KMCLifetime::Promotetime(double cumulated_rate,votca::tools::Random2 * RandomVariable){
377  double dt = 0;
378  double rand_u = 1 - RandomVariable->rand_uniform();
379  while (rand_u == 0) {
380  cout << "WARNING: encountered 0 as a random variable! New try." << endl;
381  rand_u = 1 - RandomVariable->rand_uniform();
382  }
383  dt = -1 / cumulated_rate * log(rand_u);
384  return dt;

560  affectedcarrier->node->occupied=1;
561  affectedcarrier->resetCarrier();
562  insertioncount++;
563  affectedcarrier->id=numberofcharges-1+insertioncount;
564  secondlevel=false;
565  break;
566  }
567  else{
568  newnode = node[event->destination];
569  dr = &(event->dr);
570  }
571 
572  // check after the event if this was allowed
573  if (CheckForbidden(newnode->id, forbiddendests)) {
574  continue;
575  }
576 
577  // if the new segment is unoccupied: jump; if not: add to forbidden list and choose new hopping destination
578  if (newnode->occupied == 1) {
579  if (CheckSurrounded(oldnode, forbiddendests)) {
580  AddtoForbiddenlist(oldnode->id, forbiddennodes);
581  break; // select new escape node (ends level 2 but without setting level1step to 1)
582  }
583  AddtoForbiddenlist(newnode->id, forbiddendests);
584  continue; // select new destination
585  } else {
586  newnode->occupied = 1;
587  oldnode->occupied = 0;
588  affectedcarrier->node = newnode;
589  affectedcarrier->dr_travelled += *(dr);
590  secondlevel=false;
591 
592  break; // this ends LEVEL 2 , so that the time is updated and the next MC step started
593  }
594 
595 
596  // END LEVEL 2
597  }
598  // END LEVEL 1
599  }
600  }
601 
602 
603  cout<<endl;
604  cout << "Total runtime:\t\t\t\t\t"<< simtime << " s"<< endl;
605  cout << "Total KMC steps:\t\t\t\t"<< steps << endl;
606  cout << "Average lifetime:\t\t\t\t"<<avlifetime/insertions<< " s"<<endl;
607  cout << "Mean freepath\t l=<|r_x-r_o|> :\t\t"<<(meanfreepath/insertions)<< " nm"<<endl;
608  cout << "Average diffusionlength\t d=sqrt(<(r_x-r_o)^2>)\t"<<sqrt(abs(difflength)/insertions)<< " nm"<<endl;
609  cout<<endl;
610 
611  for(unsigned int j=0; j<node.size(); j++)
612  {
613  occP[j] = node[j]->occupationtime / simtime;
614  }
615 
616  return occP;
617  }
