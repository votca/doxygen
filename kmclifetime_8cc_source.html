73 void KMCLifetime::WriteDecayProbability(string filename) {
74 
75  Eigen::VectorXd outrates = Eigen::VectorXd::Zero(_nodes.size());
76  Eigen::VectorXd inrates = Eigen::VectorXd::Zero(_nodes.size());
77  Eigen::VectorXd decayrates = Eigen::VectorXd::Ones(_nodes.size());
78 
79  for (unsigned i = 0; i < _nodes.size(); i++) {
80  GNode& node = _nodes[i];
81  if (node.canDecay()) {
82  for (const GLink& event : node.Events()) {
83  if (event.isDecayEvent()) {
84  decayrates[i] = event.getRate();
85  } else {
86  inrates[event.getDestination()->getId()] += event.getRate();
87  outrates[i] += event.getRate();
88  }
89  }
90  }
91  }
92  outrates = decayrates.cwiseQuotient(outrates + decayrates);
93  inrates = decayrates.cwiseQuotient(inrates + decayrates);
94 
95  fstream probs;
96  probs.open(filename, fstream::out);
97  probs << "#SiteID, Relative Prob outgoing, Relative Prob ingoing" << endl;
98  for (unsigned i = 0; i < _nodes.size(); i++) {
99  probs << _nodes[i].getId() << " " << outrates[i] << " " << inrates[i]
100  << endl;
101  }
102  probs.close();
103  return;
104 }

106 void KMCLifetime::ReadLifetimeFile(std::string filename) {
107  tools::Property xml;
108  xml.LoadFromXML(filename);
109  std::vector<tools::Property*> jobProps = xml.Select("lifetimes.site");
110  if (jobProps.size() != _nodes.size()) {
111  throw runtime_error(
112  (boost::format("The number of sites in the topology: %i does not match "
113  "the number in the lifetimefile: %i") %
114  _nodes.size() % jobProps.size())
115  .str());
116  }
117 
118  for (tools::Property* prop : jobProps) {
119  Index site_id = prop->getAttribute<Index>("id");
120  double lifetime = boost::lexical_cast<double>(prop->value());
121  bool check = false;
122  for (auto& node : _nodes) {
123  if (node.getId() == site_id && !(node.canDecay())) {
124  node.AddDecayEvent(1.0 / lifetime);
125  check = true;
126  break;
127  } else if (node.getId() == site_id && node.canDecay()) {
128  throw runtime_error(
129  (boost::format("Node %i appears twice in your list") % site_id)
130  .str());
131  }
132  }
133  if (!check) {
134  throw runtime_error(
135  (boost::format("Site from file with id: %i not found in state file") %
136  site_id)
137  .str());
138  }
139  }
140  for (auto& node : _nodes) {
141  node.InitEscapeRate();
142  node.MakeHuffTree();
143  }
144  return;
145 }

160 void KMCLifetime::RunVSSM() {
161 
162  Index realtime_start = time(nullptr);
163  XTP_LOG(Log::error, _log)
164  << "\nAlgorithm: VSSM for Multiple Charges with finite Lifetime\n"
165  "number of charges: "
166  << _numberofcarriers << "\nnumber of nodes: " << _nodes.size() << flush;
167 
168  if (_numberofcarriers > Index(_nodes.size())) {
169  throw runtime_error(
170  "ERROR in kmclifetime: specified number of charges is greater than the "
171  "number of nodes. This conflicts with single occupation.");
172  }
173 
174  fstream traj;
175  fstream energyfile;
176 
177  XTP_LOG(Log::error, _log)
178  << "Writing trajectory to " << _trajectoryfile << "." << flush;
179  traj.open(_trajectoryfile, fstream::out);
180  traj << "#Simtime [s]\t Insertion\t Carrier ID\t Lifetime[s]\tSteps\t Last "
181  "Segment\t x_travelled[nm]\t y_travelled[nm]\t z_travelled[nm]"
182  << endl;
183 
184  if (_do_carrierenergy) {
185 
186  XTP_LOG(Log::error, _log)
187  << "Tracking the energy of one charge carrier and exponential average "
188  "with alpha="
189  << _alpha << " to " << _energy_outputfile << flush;
190  energyfile.open(_energy_outputfile, fstream::out);
191  energyfile << "Simtime [s]\tSteps\tCarrier ID\tEnergy_a=" << _alpha
192  << "[eV]" << endl;
193  }
194 
195  // Injection
196  XTP_LOG(Log::error, _log)
197  << "\ninjection method: " << _injectionmethod << flush;
198 
199  RandomlyCreateCharges();
200 
201  unsigned long insertioncount = 0;
202  unsigned long step = 0;
203  double simtime = 0.0;
204 
205  std::vector<GNode*> forbiddennodes;
206  std::vector<GNode*> forbiddendests;
207 
208  time_t now = time(nullptr);
209  tm* localtm = localtime(&now);
210  XTP_LOG(Log::error, _log) << "Run started at " << asctime(localtm) << flush;
211 
212  double avlifetime = 0.0;
213  double meanfreepath = 0.0;
214  Eigen::Vector3d difflength_squared = Eigen::Vector3d::Zero();
215 
216  double avgenergy = _carriers[0].getCurrentEnergy();
217  Index carrieridold = _carriers[0].getId();
218 
219  while (insertioncount < _insertions) {
220  if ((time(nullptr) - realtime_start) > Index(_maxrealtime * 60. * 60.)) {
221  XTP_LOG(Log::error, _log)
222  << "\nReal time limit of " << _maxrealtime << " hours ("
223  << Index(_maxrealtime * 60 * 60 + 0.5)
224  << " seconds) has been reached. Stopping here.\n"
225  << flush;
226  break;
227  }
228 
229  double cumulated_rate = 0;
230 
231  for (const auto& carrier : _carriers) {
232  cumulated_rate += carrier.getCurrentEscapeRate();
233  }
234  if (cumulated_rate == 0) { // this should not happen: no possible jumps
235  // defined for a node
236  throw runtime_error(
237  "ERROR in kmclifetime: Incorrect rates in the database file. All the "
238  "escape rates for the current setting are 0.");
239  }
240  // go forward in time
241  double dt = Promotetime(cumulated_rate);
242 
243  if (_do_carrierenergy) {
244  bool print = false;
245  if (_carriers[0].getId() > carrieridold) {
246  avgenergy = _carriers[0].getCurrentEnergy();
247  print = true;
248  carrieridold = _carriers[0].getId();
249  } else if (step % _outputsteps == 0) {
250  avgenergy =
251  _alpha * _carriers[0].getCurrentEnergy() + (1 - _alpha) * avgenergy;
252  print = true;
253  }
254  if (print) {
255  energyfile << simtime << "\t" << step << "\t" << _carriers[0].getId()
256  << "\t" << avgenergy * tools::conv::hrt2ev << endl;
257  }
258  }
259 
260  simtime += dt;
261  step++;
262  for (auto& carrier : _carriers) {
263  carrier.updateLifetime(dt);
264  carrier.updateSteps(1);
265  carrier.updateOccupationtime(dt);
266  }
267 
268  ResetForbiddenlist(forbiddennodes);
269  bool secondlevel = true;
270  while (secondlevel) {
271 
272  // determine which carrier will escape
273  GNode* newnode = nullptr;
274  Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
275 
276  if (CheckForbidden(affectedcarrier->getCurrentNode(), forbiddennodes)) {
277  continue;
278  }
279 
280  // determine where it will jump to
281  ResetForbiddenlist(forbiddendests);
282  boost::progress_display progress(_insertions);
283 
284  while (true) {
285  // LEVEL 2
286 
287  newnode = nullptr;
288  const GLink& event =
289  ChooseHoppingDest(affectedcarrier->getCurrentNode());
290 
291  if (event.isDecayEvent()) {
292  const Eigen::Vector3d& dr_travelled =
293  affectedcarrier->get_dRtravelled();
294  avlifetime += affectedcarrier->getLifetime();
295  meanfreepath += dr_travelled.norm();
296  difflength_squared += dr_travelled.cwiseAbs2();
297  WriteToTraj(traj, insertioncount, simtime, *affectedcarrier);
298  ++progress;
299  RandomlyAssignCarriertoSite(*affectedcarrier);
300  affectedcarrier->resetCarrier();
301  insertioncount++;
302  affectedcarrier->setId(_numberofcarriers - 1 + insertioncount);
303  secondlevel = false;
304  break;
305  } else {
306  newnode = event.getDestination();
307  }
308 
309  // check after the event if this was allowed
310  if (CheckForbidden(*newnode, forbiddendests)) {
311  continue;
312  }
313 
314  // if the new segment is unoccupied: jump; if not: add to forbidden list
315  // and choose new hopping destination
316  if (newnode->isOccupied()) {
317  if (CheckSurrounded(affectedcarrier->getCurrentNode(),
318  forbiddendests)) {
319  AddtoForbiddenlist(affectedcarrier->getCurrentNode(),
320  forbiddennodes);
321  break; // select new escape node (ends level 2 but without setting
322  // level1step to 1)
323  }
324  AddtoForbiddenlist(*newnode, forbiddendests);
325  continue; // select new destination
326  } else {
327  affectedcarrier->jumpAccordingEvent(event);
328  secondlevel = false;
329 
330  break; // this ends LEVEL 2 , so that the time is updated and the
331  // next MC step started
332  }
333 
334  // END LEVEL 2
335  }
336  // END LEVEL 1
337  }
338  }
339 
340  XTP_LOG(Log::error, _log)
341  << "\nTotal runtime:\t\t\t\t\t" << simtime
342  << " s\n"
343  "Total KMC steps:\t\t\t\t"
344  << step << "\nAverage lifetime:\t\t\t\t"
345  << avlifetime / double(insertioncount) << " s\n"
346  << "Mean freepath\t l=<|r_x-r_o|> :\t\t"
347  << (meanfreepath * tools::conv::bohr2nm / double(insertioncount))
348  << " nm\n"
349  << "Average diffusionlength\t d=sqrt(<(r_x-r_o)^2>)\t"
350  << std::sqrt(difflength_squared.norm() / double(insertioncount)) *
351  tools::conv::bohr2nm
352  << " nm\n"
353  << flush;
354 
355  WriteOccupationtoFile(simtime, _occfile);
356  traj.close();
357  if (_do_carrierenergy) {
358  energyfile.close();
359  }
360  return;
361 }

363 bool KMCLifetime::EvaluateFrame(Topology& top) {
364  XTP_LOG(Log::error, _log) << "\n-----------------------------------"
365  "\n KMCLIFETIME started"
366  "\n-----------------------------------\n"
367  << std::flush;
368 
369  XTP_LOG(Log::info, _log) << "\nInitialising random number generator" << flush;
370 
371  _RandomVariable.init(_seed);
372  LoadGraph(top);
373  ReadLifetimeFile(_lifetimefile);
374 
375  if (!_probfile.empty()) {
376  WriteDecayProbability(_probfile);
377  }
378  RunVSSM();
379 
380  time_t now = time(nullptr);
381  tm* localtm = localtime(&now);
382  XTP_LOG(Log::error, _log)
383  << " KMCLIFETIME finished at:" << asctime(localtm) << std::flush;
384 
385  return true;
