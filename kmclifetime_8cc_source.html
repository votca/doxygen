83  void KMCLifetime::WriteDecayProbability(string filename){
84 
85  Eigen::VectorXd outrates=Eigen::VectorXd::Zero(_nodes.size());
86  Eigen::VectorXd inrates=Eigen::VectorXd::Zero(_nodes.size());
87  Eigen::VectorXd decayrates=Eigen::VectorXd::Ones(_nodes.size());
88 
89  for (unsigned i=0;i<_nodes.size();i++){
90  GNode* node=_nodes[i];
91  if(node->hasdecay){
92  for(const GLink& event:node->events){
93  if(event.decayevent){
94  decayrates[i]=event.rate;
95  }else{
96  inrates[event.destination]+=event.rate;
97  outrates[i]+=event.rate;
98  }
99  }
100  }
101  }
102  outrates=decayrates.cwiseQuotient(outrates+decayrates);
103  inrates=decayrates.cwiseQuotient(inrates+decayrates);
104 
105  fstream probs;
106  probs.open ( filename.c_str(), fstream::out);
107  probs<<"#SiteID, Relative Prob outgoing, Relative Prob ingoing"<<endl;
108  for (unsigned i=0;i<_nodes.size();i++){
109  probs<<_nodes[i]->id<<" "<<outrates[i]<<" "<<inrates[i]<<endl;
110  }
111  probs.close();
112  return;
113  }

116  void KMCLifetime::ReadLifetimeFile(std::string filename){
117  tools::Property xml;
118  load_property_from_xml(xml, filename);
119  list<tools::Property*> jobProps = xml.Select("lifetimes.site");
120  if (jobProps.size()!=_nodes.size()){
121  throw runtime_error((boost::format("The number of sites in the sqlfile: %i does not match the number in the lifetimefile: %i")
122  % _nodes.size() % jobProps.size()).str());
123  }
124 
125  for (list<tools::Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
126  int site_id =(*it)->getAttribute<int>("id")-1;
127  double lifetime=boost::lexical_cast<double>((*it)->value());
128  bool check=false;
129  for (unsigned i=0;i<_nodes.size();i++){
130  if (_nodes[i]->id==site_id && !(_nodes[i]->hasdecay)){
131  _nodes[i]->AddDecayEvent(1.0/lifetime);
132  check=true;
133  break;
134  }
135  else if(_nodes[i]->id==site_id && _nodes[i]->hasdecay){
136  throw runtime_error((boost::format("Node %i appears twice in your list") %site_id).str());
137  }
138 
139  }
140  if (!check){
141  throw runtime_error((boost::format("Site from file with id: %i not found in sql") %site_id).str());
142  }
143  }
144 
145  return;
146  }

149  void KMCLifetime::RunVSSM(ctp::Topology *top) {
150 
151  int realtime_start = time(NULL);
152  cout << endl << "Algorithm: VSSM for Multiple Charges with finite Lifetime" << endl;
153  cout << "number of charges: " << _numberofcharges << endl;
154  cout << "number of nodes: " << _nodes.size() << endl;
155 
156  if (_numberofcharges > _nodes.size()) {
157  throw runtime_error("ERROR in kmclifetime: specified number of charges is greater than the number of nodes. This conflicts with single occupation.");
158  }
159 
160  fstream traj;
161  fstream energyfile;
162 
163  cout << "Writing trajectory to " << _trajectoryfile << "." << endl;
164  traj.open ( _trajectoryfile.c_str(), fstream::out);
165  traj << "#Simtime [s]\t Insertion\t Carrier ID\t Lifetime[s]\tSteps\t Last Segment\t x_travelled[nm]\t y_travelled[nm]\t z_travelled[nm]"<<endl;
166 
167  if(_do_carrierenergy){
168 
169  cout << "Tracking the energy of one charge carrier and exponential average with alpha=" << _alpha << " to "<<_energy_outputfile << endl;
170  energyfile.open(_energy_outputfile.c_str(),fstream::out);
171  energyfile << "Simtime [s]\tSteps\tCarrier ID\tEnergy_a="<<_alpha<<"[eV]"<<endl;
172  }
173 
174  // Injection
175  cout << endl << "injection method: " << _injectionmethod << endl;
176 
177  RandomlyCreateCharges();
178 
179  unsigned insertioncount = 0;
180  unsigned long step=0;
181  double simtime=0.0;
182 
183  std::vector<int> forbiddennodes;
184  std::vector<int> forbiddendests;
185 
186  time_t now = time(0);
187  tm* localtm = localtime(&now);
188  cout << "Run started at " << asctime(localtm) << endl;
189 
190  double avlifetime=0.0;
191  double meanfreepath=0.0;
192  tools::vec difflength=tools::vec(0,0,0);
193 
194  double avgenergy=_carriers[0]->getCurrentEnergy();
195  int carrieridold=_carriers[0]->id;
196 
197  while (insertioncount < _insertions) {
198  if ((time(NULL) - realtime_start) > _maxrealtime * 60. * 60.) {
199  cout << endl << "Real time limit of " << _maxrealtime << " hours (" << int(_maxrealtime * 60 * 60 + 0.5) << " seconds) has been reached. Stopping here." << endl << endl;
200  break;
201  }
202 
203 
204  double cumulated_rate = 0;
205 
206  for (unsigned int i = 0; i < _carriers.size(); i++) {
207  cumulated_rate += _carriers[i]->getCurrentEscapeRate();
208  }
209  if (cumulated_rate == 0) { // this should not happen: no possible jumps defined for a node
210  throw runtime_error("ERROR in kmclifetime: Incorrect rates in the database file. All the escape rates for the current setting are 0.");
211  }
212  // go forward in time
213  double dt=Promotetime(cumulated_rate);
214 
215  if(_do_carrierenergy){
216  bool print=false;
217  if (_carriers[0]->id>carrieridold){
218  avgenergy=_carriers[0]->getCurrentEnergy();
219  print=true;
220  carrieridold=_carriers[0]->id;
221  }
222  else if(step%_outputsteps==0){
223  avgenergy=_alpha*_carriers[0]->getCurrentEnergy()+(1-_alpha)*avgenergy;
224  print=true;
225  }
226  if(print){
227  energyfile << simtime<<"\t"<<step <<"\t"<<_carriers[0]->id<<"\t"<<avgenergy<<endl;
228  }
229  }
230 
231  simtime += dt;
232  step++;
233  for (unsigned int i = 0; i < _carriers.size(); i++) {
234  _carriers[i]->updateLifetime(dt);
235  _carriers[i]->updateSteps(1);
236  _carriers[i]->updateOccupationtime(dt);
237  }
238 
239  ResetForbiddenlist(forbiddennodes);
240  bool secondlevel=true;
241  while (secondlevel){
242 
243  // determine which carrier will escape
244  GNode* newnode=NULL;
245  Chargecarrier* affectedcarrier=ChooseAffectedCarrier(cumulated_rate);
246 
247 
248 
249  if (CheckForbidden(affectedcarrier->getCurrentNodeId(), forbiddennodes)) {
250  continue;
251  }
252 
253  // determine where it will jump to
254  ResetForbiddenlist(forbiddendests);
255 
256  while (true) {
257  // LEVEL 2
258 
259  newnode = NULL;
260  GLink* event=ChooseHoppingDest(affectedcarrier->getCurrentNode());
261 
262  if (event->decayevent){
263 
264  avlifetime+=affectedcarrier->getLifetime();
265  meanfreepath+=tools::abs(affectedcarrier->dr_travelled);
266  difflength+=tools::elementwiseproduct(affectedcarrier->dr_travelled,affectedcarrier->dr_travelled);
267  traj << simtime<<"\t"<<insertioncount<< "\t"<< affectedcarrier->id<<"\t"<< affectedcarrier->getLifetime()<<"\t"<<affectedcarrier->getSteps()<<"\t"<< affectedcarrier->getCurrentNodeId()+1<<"\t"<<affectedcarrier->dr_travelled.getX()<<"\t"<<affectedcarrier->dr_travelled.getY()<<"\t"<<affectedcarrier->dr_travelled.getZ()<<endl;
268  if( tools::globals::verbose &&(_insertions<1500 ||insertioncount% (_insertions/1000)==0 || insertioncount<0.001*_insertions)){
269  std::cout << "\rInsertion " << insertioncount+1<<" of "<<_insertions;
270  std::cout << std::flush;
271  }
272  RandomlyAssignCarriertoSite(affectedcarrier);
273  affectedcarrier->resetCarrier();
274  insertioncount++;
275  affectedcarrier->id=_numberofcharges-1+insertioncount;
276  secondlevel=false;
277  break;
278  }
279  else{
280  newnode = _nodes[event->destination];
281  }
282 
283  // check after the event if this was allowed
284  if (CheckForbidden(newnode->id, forbiddendests)) {
285  continue;
286  }
287 
288  // if the new segment is unoccupied: jump; if not: add to forbidden list and choose new hopping destination
289  if (newnode->occupied) {
290  if (CheckSurrounded(affectedcarrier->getCurrentNode(), forbiddendests)) {
291  AddtoForbiddenlist(affectedcarrier->getCurrentNodeId(), forbiddennodes);
292  break; // select new escape node (ends level 2 but without setting level1step to 1)
293  }
294  AddtoForbiddenlist(newnode->id, forbiddendests);
295  continue; // select new destination
296  } else {
297  affectedcarrier->jumpfromCurrentNodetoNode(newnode);
298  affectedcarrier->dr_travelled += event->dr;
299  AddtoJumplengthdistro(event,dt);
300  secondlevel=false;
301 
302  break; // this ends LEVEL 2 , so that the time is updated and the next MC step started
303  }
304 
305 
306  // END LEVEL 2
307  }
308  // END LEVEL 1
309  }
310  }
311 
312 
313  cout<<endl;
314  cout << "Total runtime:\t\t\t\t\t"<< simtime << " s"<< endl;
315  cout << "Total KMC steps:\t\t\t\t"<< step << endl;
316  cout << "Average lifetime:\t\t\t\t"<<avlifetime/insertioncount<< " s"<<endl;
317  cout << "Mean freepath\t l=<|r_x-r_o|> :\t\t"<<(meanfreepath/insertioncount)<< " nm"<<endl;
318  cout << "Average diffusionlength\t d=sqrt(<(r_x-r_o)^2>)\t"<<sqrt(abs(difflength)/insertioncount)<< " nm"<<endl;
319  cout<<endl;
320 
321  PrintJumplengthdistro();
322 
323  vector< ctp::Segment* >& seg = top->Segments();
324 
325  for (unsigned i = 0; i < seg.size(); i++) {
326  double occupationprobability=_nodes[i]->occupationtime / simtime;
327  seg[i]->setOcc(occupationprobability,_carriertype);
328  }
329  traj.close();
330  if(_do_carrierenergy){
331  energyfile.close();
332  }
333  return;
334  }

338  bool KMCLifetime::EvaluateFrame(ctp::Topology *top) {
339  std::cout << "-----------------------------------" << std::endl;
340  std::cout << " KMCLIFETIME started" << std::endl;
341  std::cout << "-----------------------------------" << std::endl << std::endl;
342 
343  // Initialise random number generator
344  if (votca::tools::globals::verbose) {
345  cout << endl << "Initialising random number generator" << endl;
346  }
347  std::srand(_seed); // srand expects any integer in order to initialise the random number generator
348  _RandomVariable = tools::Random2();
349  _RandomVariable.init(rand(), rand(), rand(), rand());
350  LoadGraph(top);
351  ReadLifetimeFile(_lifetimefile);
352 
353 
354 
355  if(_rates == "calculate")
356  {
357  cout << "Calculating rates (i.e. rates from state file are not used)." << endl;
358  InitialRates();
359  }
360  else
361  {
362  cout << "Using rates from state file." << endl;
363  }
364  if(_probfile!=""){
365  WriteDecayProbability(_probfile);
366  }
367  RunVSSM(top);
368 
369  time_t now = time(0);

371  std::cout << " KMCLIFETIME finished at:" <<asctime(localtm) << std::endl;
372 
373  return true;
