66 void KMCLifetime::WriteDecayProbability(std::string filename) {
67 
68  Eigen::VectorXd outrates = Eigen::VectorXd::Zero(_nodes.size());
69  Eigen::VectorXd inrates = Eigen::VectorXd::Zero(_nodes.size());
70  Eigen::VectorXd decayrates = Eigen::VectorXd::Ones(_nodes.size());
71 
72  for (unsigned i = 0; i < _nodes.size(); i++) {
73  GNode& node = _nodes[i];
74  if (node.canDecay()) {
75  for (const GLink& event : node.Events()) {
76  if (event.isDecayEvent()) {
77  decayrates[i] = event.getRate();
78  } else {
79  inrates[event.getDestination()->getId()] += event.getRate();
80  outrates[i] += event.getRate();
81  }
82  }
83  }
84  }
85  outrates = decayrates.cwiseQuotient(outrates + decayrates);
86  inrates = decayrates.cwiseQuotient(inrates + decayrates);
87 
88  std::fstream probs;
89  probs.open(filename, std::fstream::out);
90  probs << "#SiteID, Relative Prob outgoing, Relative Prob ingoing\n";
91  for (unsigned i = 0; i < _nodes.size(); i++) {
92  probs << _nodes[i].getId() << " " << outrates[i] << " " << inrates[i]
93  << std::endl;
94  }
95  probs.close();
96  return;
97 }

99 void KMCLifetime::ReadLifetimeFile(std::string filename) {
100  tools::Property xml;
101  xml.LoadFromXML(filename);
102  std::vector<tools::Property*> jobProps = xml.Select("lifetimes.site");
103  if (jobProps.size() != _nodes.size()) {
104  throw std::runtime_error(
105  (boost::format("The number of sites in the topology: %i does not match "
106  "the number in the lifetimefile: %i") %
107  _nodes.size() % jobProps.size())
108  .str());
109  }
110 
111  for (tools::Property* prop : jobProps) {
112  Index site_id = prop->getAttribute<Index>("id");
113  double lifetime = boost::lexical_cast<double>(prop->value());
114  bool check = false;
115  for (auto& node : _nodes) {
116  if (node.getId() == site_id && !(node.canDecay())) {
117  node.AddDecayEvent(1.0 / lifetime);
118  check = true;
119  break;
120  } else if (node.getId() == site_id && node.canDecay()) {
121  throw std::runtime_error(
122  (boost::format("Node %i appears twice in your list") % site_id)
123  .str());
124  }
125  }
126  if (!check) {
127  throw std::runtime_error(
128  (boost::format("Site from file with id: %i not found in state file") %
129  site_id)
130  .str());
131  }
132  }
133  for (auto& node : _nodes) {
134  node.InitEscapeRate();
135  node.MakeHuffTree();
136  }
137  return;
138 }

153 void KMCLifetime::RunVSSM() {
154 
155  std::chrono::time_point<std::chrono::system_clock> realtime_start =
156  std::chrono::system_clock::now();
157  XTP_LOG(Log::error, _log)
158  << "\nAlgorithm: VSSM for Multiple Charges with finite Lifetime\n"
159  "number of charges: "
160  << _numberofcarriers << "\nnumber of nodes: " << _nodes.size()
161  << std::flush;
162 
163  if (_numberofcarriers > Index(_nodes.size())) {
164  throw std::runtime_error(
165  "ERROR in kmclifetime: specified number of charges is greater than the "
166  "number of nodes. This conflicts with single occupation.");
167  }
168 
169  std::fstream traj;
170  std::fstream energyfile;
171 
172  XTP_LOG(Log::error, _log)
173  << "Writing trajectory to " << _trajectoryfile << "." << std::flush;
174  traj.open(_trajectoryfile, std::fstream::out);
175  traj << "#Simtime [s]\t Insertion\t Carrier ID\t Lifetime[s]\tSteps\t Last "
176  "Segment\t x_travelled[nm]\t y_travelled[nm]\t z_travelled[nm]\n";
177 
178  if (_do_carrierenergy) {
179 
180  XTP_LOG(Log::error, _log)
181  << "Tracking the energy of one charge carrier and exponential average "
182  "with alpha="
183  << _alpha << " to " << _energy_outputfile << std::flush;
184  energyfile.open(_energy_outputfile, std::fstream::out);
185  energyfile << "Simtime [s]\tSteps\tCarrier ID\tEnergy_a=" << _alpha
186  << "[eV]\n";
187  }
188 
189  // Injection
190  XTP_LOG(Log::error, _log)
191  << "\ninjection method: " << _injectionmethod << std::flush;
192 
193  RandomlyCreateCharges();
194 
195  unsigned long insertioncount = 0;
196  unsigned long step = 0;
197  double simtime = 0.0;
198 
199  std::vector<GNode*> forbiddennodes;
200  std::vector<GNode*> forbiddendests;
201 
202  time_t now = time(nullptr);
203  tm* localtm = localtime(&now);
204  XTP_LOG(Log::error, _log)
205  << "Run started at " << asctime(localtm) << std::flush;
206 
207  double avlifetime = 0.0;
208  double meanfreepath = 0.0;
209  Eigen::Vector3d difflength_squared = Eigen::Vector3d::Zero();
210 
211  double avgenergy = _carriers[0].getCurrentEnergy();
212  Index carrieridold = _carriers[0].getId();
213 
214  while (insertioncount < _insertions) {
215  std::chrono::duration<double> elapsed_time =
216  std::chrono::system_clock::now() - realtime_start;
217  if (elapsed_time.count() > (_maxrealtime * 60. * 60.)) {
218  XTP_LOG(Log::error, _log)
219  << "\nReal time limit of " << _maxrealtime << " hours ("
220  << Index(_maxrealtime * 60 * 60 + 0.5)
221  << " seconds) has been reached. Stopping here.\n"
222  << std::flush;
223  break;
224  }
225 
226  double cumulated_rate = 0;
227 
228  for (const auto& carrier : _carriers) {
229  cumulated_rate += carrier.getCurrentEscapeRate();
230  }
231  if (cumulated_rate == 0) { // this should not happen: no possible jumps
232  // defined for a node
233  throw std::runtime_error(
234  "ERROR in kmclifetime: Incorrect rates in the database file. All the "
235  "escape rates for the current setting are 0.");
236  }
237  // go forward in time
238  double dt = Promotetime(cumulated_rate);
239 
240  if (_do_carrierenergy) {
241  bool print = false;
242  if (_carriers[0].getId() > carrieridold) {
243  avgenergy = _carriers[0].getCurrentEnergy();
244  print = true;
245  carrieridold = _carriers[0].getId();
246  } else if (step % _outputsteps == 0) {
247  avgenergy =
248  _alpha * _carriers[0].getCurrentEnergy() + (1 - _alpha) * avgenergy;
249  print = true;
250  }
251  if (print) {
252  energyfile << simtime << "\t" << step << "\t" << _carriers[0].getId()
253  << "\t" << avgenergy * tools::conv::hrt2ev << std::endl;
254  }
255  }
256 
257  simtime += dt;
258  step++;
259  for (auto& carrier : _carriers) {
260  carrier.updateLifetime(dt);
261  carrier.updateSteps(1);
262  carrier.updateOccupationtime(dt);
263  }
264 
265  ResetForbiddenlist(forbiddennodes);
266  bool secondlevel = true;
267  while (secondlevel) {
268 
269  // determine which carrier will escape
270  GNode* newnode = nullptr;
271  Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
272 
273  if (CheckForbidden(affectedcarrier->getCurrentNode(), forbiddennodes)) {
274  continue;
275  }
276 
277  // determine where it will jump to
278  ResetForbiddenlist(forbiddendests);
279  boost::progress_display progress(_insertions);
280 
281  while (true) {
282  // LEVEL 2
283 
284  newnode = nullptr;
285  const GLink& event =
286  ChooseHoppingDest(affectedcarrier->getCurrentNode());
287 
288  if (event.isDecayEvent()) {
289  const Eigen::Vector3d& dr_travelled =
290  affectedcarrier->get_dRtravelled();
291  avlifetime += affectedcarrier->getLifetime();
292  meanfreepath += dr_travelled.norm();
293  difflength_squared += dr_travelled.cwiseAbs2();
294  WriteToTraj(traj, insertioncount, simtime, *affectedcarrier);
295  ++progress;
296  RandomlyAssignCarriertoSite(*affectedcarrier);
297  affectedcarrier->resetCarrier();
298  insertioncount++;
299  affectedcarrier->setId(_numberofcarriers - 1 + insertioncount);
300  secondlevel = false;
301  break;
302  } else {
303  newnode = event.getDestination();
304  }
305 
306  // check after the event if this was allowed
307  if (CheckForbidden(*newnode, forbiddendests)) {
308  continue;
309  }
310 
311  // if the new segment is unoccupied: jump; if not: add to forbidden list
312  // and choose new hopping destination
313  if (newnode->isOccupied()) {
314  if (CheckSurrounded(affectedcarrier->getCurrentNode(),
315  forbiddendests)) {
316  AddtoForbiddenlist(affectedcarrier->getCurrentNode(),
317  forbiddennodes);
318  break; // select new escape node (ends level 2 but without setting
319  // level1step to 1)
320  }
321  AddtoForbiddenlist(*newnode, forbiddendests);
322  continue; // select new destination
323  } else {
324  affectedcarrier->jumpAccordingEvent(event);
325  secondlevel = false;
326 
327  break; // this ends LEVEL 2 , so that the time is updated and the
328  // next MC step started
329  }
330 
331  // END LEVEL 2
332  }
333  // END LEVEL 1
334  }
335  }
336 
337  XTP_LOG(Log::error, _log)
338  << "\nTotal runtime:\t\t\t\t\t" << simtime
339  << " s\n"
340  "Total KMC steps:\t\t\t\t"
341  << step << "\nAverage lifetime:\t\t\t\t"
342  << avlifetime / double(insertioncount) << " s\n"
343  << "Mean freepath\t l=<|r_x-r_o|> :\t\t"
344  << (meanfreepath * tools::conv::bohr2nm / double(insertioncount))
345  << " nm\n"
346  << "Average diffusionlength\t d=sqrt(<(r_x-r_o)^2>)\t"
347  << std::sqrt(difflength_squared.norm() / double(insertioncount)) *
348  tools::conv::bohr2nm
349  << " nm\n"
350  << std::flush;
351 
352  WriteOccupationtoFile(simtime, _occfile);
353  traj.close();
354  if (_do_carrierenergy) {
355  energyfile.close();
356  }
357  return;
358 }

360 bool KMCLifetime::EvaluateFrame(Topology& top) {
361  XTP_LOG(Log::error, _log) << "\n-----------------------------------"
362  "\n KMCLIFETIME started"
363  "\n-----------------------------------\n"
364  << std::flush;
365 
366  XTP_LOG(Log::info, _log) << "\nInitialising random number generator"
367  << std::flush;
368 
369  _RandomVariable.init(_seed);
370  LoadGraph(top);
371  ReadLifetimeFile(_lifetimefile);
372 
373  if (!_probfile.empty()) {
374  WriteDecayProbability(_probfile);
375  }
376  RunVSSM();
377 
378  time_t now = time(nullptr);
379  tm* localtm = localtime(&now);
380  XTP_LOG(Log::error, _log)
381  << " KMCLIFETIME finished at:" << asctime(localtm) << std::flush;
382 
383  return true;
384 }
