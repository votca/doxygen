75  void EAnalyze::Initialize(tools::Property *opt) {
76  _skip_corr = false;
77  _skip_sites = false;
78  _skip_pairs = false;
79  // update options with the VOTCASHARE defaults
80  UpdateWithDefaults(opt, "xtp");
81  std::string key = "options." + Identify();
82  if (opt->exists(key + ".resolution_pairs")) {
83  _resolution_pairs = opt->get(key + ".resolution_pairs").as< double >();
84  } else {
85  _skip_pairs = true;
86  }
87  if (opt->exists(key + ".resolution_sites")) {
88  _resolution_sites = opt->get(key + ".resolution_sites").as< double >();
89  } else {
90  _skip_sites = true;
91  }
92  if (opt->exists(key + ".resolution_space")) {
93  _resolution_space = opt->get(key + ".resolution_space").as< double >();
94  } else {
95  _skip_corr = true;
96  }
97 
98  if (opt->exists(key + ".pattern")) {
99  _seg_pattern = opt->get(key + ".pattern").as<std::string>();
100  } else _seg_pattern = "*";
101 
102  if (opt->exists(key + ".states")) {
103  std::string statestrings = opt->get(key + ".states").as< std::string>();
104  tools::Tokenizer tok(statestrings, ",\n\t ");
105  std::vector<std::string> string_vec;
106  tok.ToVector(string_vec);
107  for (std::string& state : string_vec) {
108  _states.push_back(QMStateType(state));
109  }
110  } else {
111  _states.push_back(QMStateType(QMStateType::Electron));
112  _states.push_back(QMStateType(QMStateType::Hole));
113  }
114 
115  _doenergy_landscape = opt->ifExistsReturnElseReturnDefault<bool>(key + ".energy_landscape", false);
116 
117 
118  if (opt->exists(key + ".distancemode")) {
119  std::vector<std::string> choices = {"segment", "centerofmass"};
120  _distancemode = opt->ifExistsAndinListReturnElseThrowRuntimeError<std::string>(key + ".distancemode", choices);
121  } else {
122  _distancemode = "segment";
123  }

127  bool EAnalyze::EvaluateFrame(ctp::Topology *top) {
128 
129  // Short-list segments according to pattern
130  for (ctp::Segment* seg : top->Segments()) {
131  std::string seg_name = seg->getName();
132  if (votca::tools::wildcmp(_seg_pattern.c_str(), seg_name.c_str())) {
133  _seg_shortlist.push_back(seg);
134  }
135  }
136  std::cout << std::endl << "... ... Short-listed " << _seg_shortlist.size()
137  << " segments (pattern='" << _seg_pattern << "')" << std::flush;
138  std::cout << std::endl << "... ... ... NOTE Statistics of site energies and spatial"
139  << " correlations thereof are based on the short-listed segments only. "
140  << std::flush;
141  std::cout << std::endl << "... ... ... "
142  << "Statistics of site-energy differences operate on the full list."
143  << std::flush;
144 
145  // Calculate
146  // ... Site-energy histogram, mean, width
147  // ... Pair-energy histogram, mean, width
148  // ... Site-energy correlation
149 
150  ctp::QMNBList &nblist = top->NBList();
151 
152  for (QMStateType state : _states) {
153  std::cout << std::endl << "... ... excited state " << state.ToString() << std::flush;
154 
155  if (!_seg_shortlist.size()) {
156  std::cout << std::endl << "... ... ... No segments short-listed. Skip ... "
157  << std::flush;
158  } else {
159  if (_skip_sites) {
160  std::cout << std::endl << "... ... ... Skip site-energy hist." << std::flush;
161  } else {
162  SiteHist(top, state);
163  }
164  if (_skip_corr) {
165  std::cout << std::endl << "... ... ... Skip correlation ..." << std::flush;
166  } else {
167  SiteCorr(top, state);
168  }
169  }
170 
171  if (!nblist.size()) {
172  std::cout << std::endl << "... ... ... No pairs in topology. Skip ... "
173  << std::flush;
174  } else {
175  if (_skip_pairs) {
176  std::cout << std::endl << "... ... ... Skip pair-energy hist." << std::flush;
177  } else {
178  PairHist(top, state);
179  }
180  }
181  }
182 
183  return true;
184  }

186  void EAnalyze::SiteHist(ctp::Topology *top, QMStateType state) {
187 
188  std::vector< double > Es;
189  Es.reserve(_seg_shortlist.size());
190  for (ctp::Segment* seg : _seg_shortlist) {
191  double E = seg->getSiteEnergy(state.ToCTPIndex());
192  Es.push_back(E);
193  }
194 
195  double MAX = *std::max_element(Es.begin(), Es.end());
196  double MIN = *std::min_element(Es.begin(), Es.end());
197  double sum = std::accumulate(Es.begin(), Es.end(), 0.0);
198  double AVG = sum / Es.size();
199  double sq_sum = std::inner_product(Es.begin(), Es.end(), Es.begin(), 0.0);
200  double STD = std::sqrt(sq_sum / Es.size() - AVG * AVG);
201 
202  // Prepare bins
203  int BIN = int( (MAX - MIN) / _resolution_sites + 0.5) + 1;
204 
205  tools::HistogramNew hist;
206  hist.Initialize(MIN, MAX, BIN);
207  hist.ProcessRange<std::vector<double>::iterator>(Es.begin(), Es.end());
208  tools::Table& tab = hist.data();
209  tab.flags()= std::vector<char>(tab.size(), ' ');
210  std::string comment = (boost::format("EANALYZE: SITE-ENERGY HISTOGRAM \n # AVG %1$4.7f STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") % AVG % STD % MIN % MAX).str();
211  std::string filename = "eanalyze.sitehist_" + state.ToString() + ".out";
212  tab.set_comment(comment);
213  tab.Save(filename);
214 
215  // Write "seg x y z energy" with atomic {x,y,z}
216  if (_doenergy_landscape) {
217  std::string filename = "eanalyze.landscape_" + state.ToString() + ".out";
218  std::ofstream out;
219  out.open(filename.c_str());
220  if (!out) throw std::runtime_error("error, cannot open file " + filename);
221  for (ctp::Segment* seg : _seg_shortlist) {
222  if (seg->getId() < _first_seg) {
223  continue;
224  }
225  if (seg->getId() == _last_seg) {
226  break;
227  }
228  double E = seg->getSiteEnergy(state.ToCTPIndex());
229  for (ctp::Atom *atm : seg->Atoms()) {
230  out << boost::format("%1$3s %2$4.7f %3$4.7f %4$4.7f %5$4.7f\n")
231  % seg->getName()
232  % atm->getPos().getX() % atm->getPos().getY() % atm->getPos().getZ()
233  % E;
234  }
235  }
236  out.close();
237  }

240  void EAnalyze::PairHist(ctp::Topology *top, QMStateType state) {
241 
242  ctp::QMNBList &nblist = top->NBList();
243 
244  std::string filenamelist = "eanalyze.pairlist_" + state.ToString() + ".out";
245 
246  // Collect site-energy differences from neighbourlist
247  std::vector< double > dE;
248  dE.reserve(2 * nblist.size());
249  std::ofstream out;
250  out.open(filenamelist.c_str());
251  if (!out) throw std::runtime_error("error, cannot open file " + filenamelist);
252  for (ctp::QMPair *pair : nblist) {
253  ctp::Segment *seg1 = pair->Seg1();
254  ctp::Segment *seg2 = pair->Seg2();
255  double deltaE = seg2->getSiteEnergy(state.ToCTPIndex()) - seg1->getSiteEnergy(state.ToCTPIndex());
256  dE.push_back(deltaE);
257  dE.push_back(-deltaE);
258  out << boost::format("%1$5d %2$5d %3$4.7f \n") % seg1->getId() % seg2->getId() % deltaE;
259  }
260  out.close();
261 
262  double MAX = *std::max_element(dE.begin(), dE.end());
263  double MIN = *std::min_element(dE.begin(), dE.end());
264  double sum = std::accumulate(dE.begin(), dE.end(), 0.0);
265  double AVG = sum / dE.size();
266  double sq_sum = std::inner_product(dE.begin(), dE.end(), dE.begin(), 0.0);
267  double STD = std::sqrt(sq_sum / dE.size() - AVG * AVG);
268  int BIN = int( (MAX - MIN) / _resolution_pairs + 0.5) + 1;
269 
270  std::string filename2 = "eanalyze.pairhist_" + state.ToString() + ".out";
271  tools::HistogramNew hist;
272  hist.Initialize(MIN, MAX, BIN);
273  hist.ProcessRange<std::vector<double>::iterator>(dE.begin(), dE.end());
274  tools::Table& tab = hist.data();
275  std::string comment = (boost::format("EANALYZE: PAIR-ENERGY HISTOGRAM \n # AVG %1$4.7f STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") % AVG % STD % MIN % MAX).str();
276  tab.set_comment(comment);
277  tab.flags()= std::vector<char>(tab.size(), ' ');
278  tab.Save(filename2);
279  }

281  void EAnalyze::SiteCorr(ctp::Topology *top, QMStateType state) {
282 
283  std::vector< double > Es;
284  Es.reserve(_seg_shortlist.size());
285  for (ctp::Segment* seg : _seg_shortlist) {
286  double E = seg->getSiteEnergy(state.ToCTPIndex());
287  Es.push_back(E);
288  }
289 
290  double sum = std::accumulate(Es.begin(), Es.end(), 0.0);
291  double AVG = sum / Es.size();
292  double sq_sum = std::inner_product(Es.begin(), Es.end(), Es.begin(), 0.0);
293  double VAR = sq_sum / Es.size() - AVG * AVG;
294  double STD = std::sqrt(VAR);
295 
296  // Collect inter-site distances, correlation product
297  std::vector< ctp::Segment* > ::iterator sit1;
298  std::vector< ctp::Segment* > ::iterator sit2;
299 
300  tools::Table tabcorr;
301  int length = _seg_shortlist.size()*(_seg_shortlist.size() - 1) / 2;
302  tabcorr.resize(length);
303  int index = 0;
304  for (sit1 = _seg_shortlist.begin(); sit1 < _seg_shortlist.end(); ++sit1) {
305  for (sit2 = sit1 + 1; sit2 < _seg_shortlist.end(); ++sit2) {
306  double R = abs(top->PbShortestConnect((*sit1)->getPos(),
307  (*sit2)->getPos()));
308  if (_distancemode == "segment") {
309  for (ctp::Fragment* frag1 : (*sit1)->Fragments()) {
310  for (ctp::Fragment* frag2 : (*sit2)->Fragments()) {
311  double R_FF = tools::abs(top->PbShortestConnect(frag1->getPos(),
312  frag2->getPos()));
313  if (R_FF < R) {
314  R = R_FF;
315  }
316  }
317  }
318  }
319  double C = ((*sit1)->getSiteEnergy(state.ToCTPIndex()) - AVG)
320  * ((*sit2)->getSiteEnergy(state.ToCTPIndex()) - AVG);
321  tabcorr.set(index, R, C);
322  index++;
323  }
324  }
325 
326  double MIN = tabcorr.x().minCoeff();
327  double MAX = tabcorr.x().maxCoeff();
328  std::string corrfile = "eanalyze.sitecorr.atomic_" + state.ToString() + ".out";
329  tabcorr.Save(corrfile);
330 
331  // Prepare bins
332  int BIN = int( (MAX - MIN) / _resolution_space + 0.5) + 1;
333  std::vector< std::vector<double> > histCs;
334  histCs.resize(BIN);
335 
336  for (int i = 0; i < tabcorr.size(); ++i) {
337  int bin = int((tabcorr.x()[i] - MIN) / _resolution_space + 0.5);
338  histCs[bin].push_back(tabcorr.y()[i]);
339  }
340 
341  tools::Table histC;
342  histC.SetHasYErr(true);
343  // Calculate spatial correlation
344  histC.resize(BIN);
345  for (int bin = 0; bin < BIN; ++bin) {
346  double corr = 0.0;
347  double dcorr2 = 0.0;
348  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
349  corr += histCs[bin][i] / VAR;
350  }
351  corr = corr / histCs[bin].size();
352  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
353  dcorr2 += (histCs[bin][i] / VAR / histCs[bin].size() - corr)*(histCs[bin][i] / VAR / histCs[bin].size() - corr);
354  }
355  // error on mean value
356  dcorr2 = dcorr2 / histCs[bin].size() / (histCs[bin].size() - 1);
357  double R = MIN + bin*_resolution_space;
358  histC.set(bin, R, corr, ' ', std::sqrt(dcorr2));
359  }
360 
361  std::string filename = "eanalyze.sitecorr_" + state.ToString() + ".out";
362  std::string comment = (boost::format("EANALYZE: SPATIAL SITE-ENERGY CORRELATION \n # AVG %1$4.7f STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") % AVG % STD % MIN % MAX).str();
363  histC.set_comment(comment);
364  histC.Save(filename);
365 
366  }
367 
368  }
