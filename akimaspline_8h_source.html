00094 inline double AkimaSpline::Calculate(const double &r)
00095 {
00096     int interval =  getInterval(r);
00097     double z = r-_r[interval];
00098     return  p0(interval)
00099             + p1(interval)*z
00100             + p2(interval)*z*z
00101             + p3(interval)*z*z*z;
