94 inline double AkimaSpline::Calculate(const double &r)
95 {
96  int interval = getInterval(r);
97  double z = r-_r[interval];
98  return p0(interval)
99  + p1(interval)*z
100  + p2(interval)*z*z
101  + p3(interval)*z*z*z;
102 }

104 inline double AkimaSpline::CalculateDerivative(const double &r)
105 {

107  double z = r-_r[interval];
108  return + p1(interval)
109  + 2.0*p2(interval)*z
110  + 3.0*p3(interval)*z*z;
111 }

113 inline double AkimaSpline::getSlope(double m1, double m2, double m3, double m4)
114 {
115  if ((m1==m2) && (m3==m4)) {
116  return (m2+m3)/2.0;
117  } else {
118  return (fabs(m4-m3)*m2 + fabs(m2-m1)*m3) / (fabs(m4-m3) + fabs(m2-m1));
119  }
120 }
