83 inline double AkimaSpline::Calculate(double r) {
84  Index interval = getInterval(r);
85  double z = r - _r[interval];
86  return p0(interval) + p1(interval) * z + p2(interval) * z * z +
87  p3(interval) * z * z * z;
88 }
89 
90 inline double AkimaSpline::CalculateDerivative(double r) {
91  Index interval = getInterval(r);
92  double z = r - _r[interval];
93  return +p1(interval) + 2.0 * p2(interval) * z + 3.0 * p3(interval) * z * z;
94 }

96 inline double AkimaSpline::getSlope(double m1, double m2, double m3,
97  double m4) {
98  if ((m1 == m2) && (m3 == m4)) {
99  return (m2 + m3) / 2.0;
100  } else {
101  return (std::abs(m4 - m3) * m2 + std::abs(m2 - m1) * m3) /
102  (std::abs(m4 - m3) + std::abs(m2 - m1));
103  }
104 }
