93 inline double AkimaSpline::Calculate(const double &r) {
94  int interval = getInterval(r);
95  double z = r - _r[interval];
96  return p0(interval) + p1(interval) * z + p2(interval) * z * z +
97  p3(interval) * z * z * z;
98 }

100 inline double AkimaSpline::CalculateDerivative(const double &r) {
101  int interval = getInterval(r);
102  double z = r - _r[interval];
103  return +p1(interval) + 2.0 * p2(interval) * z + 3.0 * p3(interval) * z * z;
104 }

106 inline double AkimaSpline::getSlope(double m1, double m2, double m3,
107  double m4) {
108  if ((m1 == m2) && (m3 == m4)) {
109  return (m2 + m3) / 2.0;
110  } else {
111  return (fabs(m4 - m3) * m2 + fabs(m2 - m1) * m3) /
112  (fabs(m4 - m3) + fabs(m2 - m1));
113  }
114 }
