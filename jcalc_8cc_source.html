41 void JCalc::ParseCrgUnitTypes(Property &options){
42  string namecoord;
43  string nameorb;
44  string namecrg;
45  string nameneutr;
46  vector <int> transorbs;
47 
48  string name;
49  string molname;
50  string namebasis;
51  vector < vector <int> > list_atoms_monomer;
52  vector < vector <double> > list_weights_monomer;
53 
54  list<Property *> types = options.Select("segment");
55  list<Property *>::iterator iter;
56 
57  for (iter = types.begin();iter != types.end(); ++iter){
58  namecoord = (*iter)->get("coordinates").as<string>();
59 
60  bool estatics = (*iter)->exists("qneutral");
61  if(estatics==true){
62  nameneutr = (*iter)->get("qneutral").as<string>();
63  namecrg = (*iter)->get("qcharged").as<string>();
64  }
65 
66  if((*iter)->exists("torbital")) transorbs = (*iter)->get("torbital").as<vector <int> >();
67  for(int i=0; i<transorbs.size(); ++i)
68  transorbs[i]--;
69 
70  if(transorbs.size() > 0) nameorb = (*iter)->get("orbitals").as<string>();
71 
72  name = (*iter)->get("name").as<string>();
73 
74  namebasis = "INDO";
75  if((*iter)->exists("basisset") > 0) namebasis = (*iter)->get("basisset").as<string>();
76 
77  string all_monomer = (*iter)->get("map").as<string>();
78  for (string::iterator c = all_monomer.begin(); c != all_monomer.end(); ++c) {
79  if (*c == '\n') *c = ' ';
80  if (*c == '\t') *c = ' ';
81  }
82  Tokenizer tok1(all_monomer, ":");
83  Tokenizer::iterator it_mon;
84  for (it_mon = tok1.begin(); it_mon != tok1.end(); ++it_mon) {
85  vector <int> list_atoms;
86  Tokenizer tok2(*it_mon, " ");
87  tok2.ConvertToVector<int>(list_atoms);
88  for(int i=0; i<list_atoms.size(); ++i) {
89  if(list_atoms[i] == 0)
90  throw std::runtime_error("An atom with index=0 was specified in listcharges. Counting starts at 1!");
91 
92  list_atoms[i]--;
93  }
94  list_atoms_monomer.push_back(list_atoms);
95  }
96 
97  string all_weights = (*iter)->get("weights").as<string>();
98  for (string::iterator c = all_weights.begin(); c != all_weights.end(); ++c) {
99  if (*c == '\n') *c = ' ';
100  if (*c == '\t') *c = ' ';
101  }
102  Tokenizer tok3(all_weights, ":");
103  for (it_mon = tok3.begin(); it_mon != tok3.end(); ++it_mon) {
104  vector <double> list_weights;
105  Tokenizer tok2(*it_mon, " ");
106  tok2.ConvertToVector<double>(list_weights);
107  list_weights_monomer.push_back(list_weights);
108  }
109 
110  CrgUnitType* crgunittype = new CrgUnitType(namecoord.c_str(), nameorb.c_str(),
111  nameneutr.c_str(), namecrg.c_str(), namebasis,
112  transorbs, _listCrgUnitType.size(),
113  name, list_atoms_monomer, list_weights_monomer);
114  crgunittype->setOptions(*iter);
115  _mapCrgUnitByName.insert(make_pair(name, crgunittype));
116  _listCrgUnitType.push_back(crgunittype);
117 
118  clearListList(list_atoms_monomer);
119  clearListList(list_weights_monomer);
120  }
121 }

124 JCalc::JCalcData * JCalc::InitJCalcData(CrgUnitType * mol1, CrgUnitType *mol2)
125 {
126  JCalcData * data = new JCalcData;
127 
128  data->_type1 = mol1;
129  data->_type2 = mol2;
130 
131  data->_orblabels.first = data->_type1 -> GetTransOrbs();
132  data->_orblabels.second = data->_type2 -> GetTransOrbs();
133 
134  int nrorbs1 = data->_orblabels.first.size();
135  int nrorbs2 = data->_orblabels.second.size();
136  /*for (int i = 0; i < nrorbs; i++) {
137  if (data->_orblabels.first[i] != i && data->_orblabels.second[i] != i) {
138  cout << "orblabels: " << data->_orblabels.first[i] << " " << data->_orblabels.second[i] << endl;
139  throw "Error in RateCalculator, the charge unit types do not have stripped orbitals";
140  }
141  }*/
142  //initialise the first copy of the molecules + orbitals
143  data->_mol1.define_bs(mol1->GetBS());
144  data->_mol1.cp_atompos(mol1->GetCrgUnit());
145  data->_mol1.cp_atoms(mol1->GetCrgUnit());
146  data->_orb1.init_orbitals_stripped(mol1->GetOrb(), nrorbs1);
147  data->_mol1.assign_orb(&data->_orb1);
148  data->_mol1.cp_crg(mol1->GetCrgUnit());
149 
150  //inititalise the second copy of molecules + orbitals
151  data->_mol2.define_bs(mol2->GetBS());
152  data->_mol2.cp_atompos(mol2->GetCrgUnit());
153  data->_mol2.cp_atoms(mol2->GetCrgUnit());
154  data->_orb2.init_orbitals_stripped(mol2->GetOrb(), nrorbs2);
155  data->_mol2.assign_orb(&data->_orb2);
156  data->_mol2.cp_crg(mol2->GetCrgUnit());
157 
158  // we have stripped the orbs to the bone
159  for (int i = 0; i < data->_orblabels.first.size(); i++) {
160  data->_orblabels.first[i] = i;
161  }
162  for (int i =0;i < data->_orblabels.second.size(); i++) {
163  data->_orblabels.second[i] = i;
164  }
165 
166  //initialise the fock matrix
167  data->_fock.init(data->_mol1, data->_mol2);
168 
169  // add a quick reference to find calculator for a pair
170  _maplistfock.insert(make_pair(make_pair(mol1, mol2), data));
171 
172  return data;
173 }

189 vector <double> JCalc::CalcJ(CrgUnit & one, CrgUnit & two)
190 {
191  if(one.getType()->_transorbs.size() == 0)
192  throw std::runtime_error("no orbital information found for conjugated segment " + one.getName());
193  if(two.getType()->_transorbs.size() == 0)
194  throw std::runtime_error("no orbital information found for conjugated segment " + two.getName());
195 
196  if (one.getType()->getId() > two.getType()->getId())
197  return CalcJ(two, one);
198 
199  JCalcData * jdata = getJCalcData(one, two);
200 
201  //calculate J
202  one.rot_two_mol(two, jdata->_mol1, jdata->_mol2);
203 
204  // calculating orbitals for all pairs
205 
206  if (tools::globals::verbose) {
207  cout << " id1:id2 [J] " << endl;
208  cout << one.getType()->getId() << ":"
209  << two.getType()->getId() << " [";
210  }
211 
212  vector < double > Js; // vector with all Js
213  vector < pair <int, int> > _pois; // vector with all pairs of interest
214  pair <int, int> poi; // pair of interest
215  for (int i = 0; i < jdata->_orblabels.first.size(); i++) {
216  for (int j = 0; j < jdata->_orblabels.first.size(); j++) {
217  poi.first = jdata->_orblabels.first[i];
218  poi.second = jdata->_orblabels.second[j];
219  Js.push_back(jdata->_fock.calcJ(poi));
220  _pois.push_back(poi);
221  if (tools::globals::verbose) { cout << jdata->_fock.calcJ(poi) << " " ; }
222  }
223  if (tools::globals::verbose) { cout << "]" << endl; }
224  }
225 
226 
227 
228 
229  int my = one.getType()->getId();
230  cout << my << endl;
231 
232  return Js;
233 }

235 int JCalc::WriteProJ(CrgUnit & one, CrgUnit & two,string namedir)
236 {
237  // only write them when one.id < two.id
238  if (one.getType()->getId() < two.getType()->getId())
239  return 1;
240 
241  //rotate the molecule and orbitals
242  JCalcData * jdata = getJCalcData(one, two);
243  one.rot_two_mol(two, jdata->_mol1, jdata->_mol2);
244 
245 
246  ofstream out;
247  string name1, nameorb1, name2, nameorb2, namedim;
248  name1 =namedir + boost::lexical_cast<string>(one.getId()) + ".xyz";
249  name2 = namedir +boost::lexical_cast<string>(two.getId()) + ".xyz";
250  nameorb1 =namedir + boost::lexical_cast<string>(one.getId()) + ".fort.7";
251  nameorb2 = namedir +boost::lexical_cast<string>(two.getId()) + ".fort.7";
252  namedim = namedir +boost::lexical_cast<string>(one.getId())+"and"+
253  boost::lexical_cast<string>(two.getId()) +".com";
254 
255 
256  //write the info for molecule1
257  out.open(name1.c_str());
258  (jdata->_mol1).print(out);
259  out.close();
260 
261  (jdata->_orb1).print_g03(nameorb1);
262  //write the info for molecule2
263  out.open(name2.c_str());
264  (jdata->_mol2).print(out);
265  out.close();
266 
267  (jdata->_orb2).print_g03(nameorb2);
268 
269  //write the input file for both
270  orb dimerorb;
271  dimerorb.dimerise_orbs((jdata->_orb1), (jdata->_orb2), (jdata->_mol1).n_el,(jdata->_mol2).n_el);
272 
273  out.open(namedim.c_str());
274  // write header
275  out << "%nproc=1" <<'\n' <<
276  "%mem=1Gb" << '\n'<<
277  "#p b3lyp/TZVP-6D guess(cards) scf(MaxCycle=1) "<<
278  "Nosymm IOp(3/33=1) IOp(5/13=1) punch=mo" << '\n' <<
279  '\n' <<
280  "autogen" << '\n' << '\n'<<
281  "0 1" <<'\n';
282  (jdata->_mol1).print(out);
283  (jdata->_mol2).print(out);
284  //write blank line
285  out << '\n';
286  out.close();
287  dimerorb.print_g03(namedim, string ("a"));
288  return 0;
