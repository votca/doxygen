65 bool Partialcharges::Evaluate() {
66  OPENMP::setMaxThreads(_nThreads);
67  _log.setReportLevel(Log::current_level);
68  _log.setMultithreading(true);
69 
70  _log.setCommonPreface("\n... ...");
71 
72  Orbitals orbitals;
73  XTP_LOG(Log::error, _log)
74  << " Loading QM data from " << _orbfile << std::flush;
75  orbitals.ReadFromCpt(_orbfile);
76  Esp2multipole esp2multipole = Esp2multipole(_log);
77  esp2multipole.Initialize(_esp_options);
78  StaticSegment seg = esp2multipole.Extractingcharges(orbitals);
79  seg.WriteMPS(_output_file, esp2multipole.GetStateString());
80 
81  XTP_LOG(Log::error, _log)
82  << "Written charges to " << _output_file << std::flush;
83 
84  return true;
85 }
