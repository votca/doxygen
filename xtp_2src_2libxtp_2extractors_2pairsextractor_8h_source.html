51 bool PairsExtractor::EvaluateFrame(ctp::Topology *top) {
52 
53  // Rigidify std::system (if possible)
54  if (!top->Rigidify()) return 0;
55 
56  std::string xmlfile = Identify() + ".xml";
57 
58  tools::Property state("state", "", "");
59  tools::Property &pairs = state.add("pairs","");
60 
61  using boost::format;
62 
63  // PAIRS
64  ctp::QMNBList::iterator pit;
65  ctp::QMNBList &nb = top->NBList();
66  for (pit = nb.begin(); pit != nb.end(); ++pit) {
67  ctp::QMPair *qmp = *pit;
68  tools::Property &pairprop = pairs.add("pair", "");
69  pairprop.add("id1", (format("%1$d") % qmp->Seg1()->getId()).str());
70  pairprop.add("name1", (format("%1$s") % qmp->Seg1()->getName()).str());
71  pairprop.add("id2", (format("%1$d") % qmp->Seg2()->getId()).str());
72  pairprop.add("name2", (format("%1$s") % qmp->Seg2()->getName()).str());
73 
74  pairprop.add("pairvector", (format("%1$+1.4f %2$+1.4f %3$+1.4f")
75  % qmp->R().getX() % qmp->R().getY() % qmp->R().getZ()).str());
76 
77  if (qmp->isPathCarrier(+1)) {
78  tools::Property &channel = pairprop.add("channel", "");
79  channel.setAttribute("type","hole");
80  channel.add("jeff2_h", (format("%1$1.7e") % qmp->getJeff2(+1)).str());
81  channel.add("deltaE_h12", (format("%1$1.7f") % qmp->getdE12(+1)).str());
82  channel.add("lambdaI_h12", (format("%1$1.7f") % qmp->getReorg12(+1)).str());
83  channel.add("lambdaI_h21", (format("%1$1.7f") % qmp->getReorg21(+1)).str());
84  channel.add("lambdaO_h", (format("%1$1.7f") % qmp->getLambdaO(+1)).str());
85  channel.add("rate_h12", (format("%1$1.7e") % qmp->getRate12(+1)).str());
86  channel.add("rate_h21", (format("%1$1.7e") % qmp->getRate12(+1)).str());
87  }
88  if (qmp->isPathCarrier(-1)) {
89  tools::Property &channel = pairprop.add("channel", "");
90  channel.setAttribute("type","electron");
91  channel.add("jeff2_e", (format("%1$1.7e") % qmp->getJeff2(-1)).str());
92  channel.add("deltaE_e12", (format("%1$1.7f") % qmp->getdE12(-1)).str());
93  channel.add("lambdaI_e12", (format("%1$1.7f") % qmp->getReorg12(-1)).str());
94  channel.add("lambdaI_e21", (format("%1$1.7f") % qmp->getReorg21(-1)).str());
95  channel.add("lambdaO_e", (format("%1$1.7f") % qmp->getLambdaO(-1)).str());
96  channel.add("rate_e12", (format("%1$1.7e") % qmp->getRate12(-1)).str());
97  channel.add("rate_e21", (format("%1$1.7e") % qmp->getRate12(-1)).str());
98  }
99  }
100 
101  std::ofstream ofs;
102  ofs.open(xmlfile.c_str(), std::ofstream::out);
103  if (!ofs.is_open()) {
104  throw std::runtime_error("Bad file handle: " + xmlfile);
105  }
106  ofs << tools::XML << state;
107  ofs.close();
108 
109  return true;
110 }
