61void XtpParallel::AddCommandLineOpt() {
62 namespace propt = boost::program_options;
63 AddProgramOptions()("ompthreads,x", propt::value<Index>()->default_value(1),
64 " number of openmp threads to create in each thread");
65 AddProgramOptions()("restart,r",

68 AddProgramOptions()("cache,q", propt::value<Index>()->default_value(8),

70 AddProgramOptions()("jobs,j",

72 " task(s) to perform: write, run, read");

88void XtpParallel::ConfigCalculator() {
89 std::cout << "... " << calc_->Identify() << std::endl;
90
91 progObs_.InitCmdLineOpts(OptionsMap());
92 calc_->setnThreads(OptionsMap()["nthreads"].as<Index>());
93 calc_->setOpenMPThreads(OptionsMap()["ompthreads"].as<Index>());
94 calc_->setProgObserver(&progObs_);
95 calc_->Initialize(options_);
96 std::cout << std::endl;
97}

99bool XtpParallel::EvaluateFrame(xtp::Topology& top) {
100 std::cout << "... " << calc_->Identify() << " " << std::flush;
101 if (generate_input_) {
102 calc_->WriteJobFile(top);
103 } else if (run_) {
104 calc_->EvaluateFrame(top);
105 } else if (import_) {
106 calc_->ReadJobFile(top);
107 }
108 std::cout << std::endl;
109 return true;
110}
