67 bool XYZReader::NextFrame(Topology &top)
68 {
69  string line;
70  getline(_fl, line); ++_line;
71  //cout << "natoms : " << line << endl;
72  if(!_fl.eof()) {
73  // read the number of atoms
74  _natoms = boost::lexical_cast<int>(line);
75  if(!_topology && _natoms !=top.BeadCount())
76  throw std::runtime_error("number of beads in topology and trajectory differ");
77 
78  // the title line
79  getline(_fl, line); ++_line;
80  //cout << "title : " << line << endl;
81 
82  // read atoms
83  for(int i=0; i<_natoms; ++i) {
84  getline(_fl, line); ++_line;
85  //cout << "coords : " << line << endl;
86  if(_fl.eof())
87  throw std::runtime_error("unexpected end of file in xyz file");
88 
89  vector<string> fields;
90  Tokenizer tok(line, " ");
91  tok.ToVector(fields);
92 
93  if(fields.size() != 4)
94  throw std::runtime_error("invalide line " +
95  boost::lexical_cast<string>(_line) +
96  " in xyz file\n" + line);
97 
98  Bead *b;
99  if(_topology)
100  b = top.CreateBead(1, fields[0]+boost::lexical_cast<string>(i),
101  top.GetOrCreateBeadType(fields[0]), 0, 0, 0);
102  else
103  b = top.getBead(i);
104 
105  // convert to nm from A
106  b->setPos(vec(
107  boost::lexical_cast<double>(fields[1])/10.0,
108  boost::lexical_cast<double>(fields[2])/10.0,
109  boost::lexical_cast<double>(fields[3])/10.0
110  ));
111 
112  }
113  }
114  return !_fl.eof();;
115 }
