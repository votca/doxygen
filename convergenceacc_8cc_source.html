35  Eigen::MatrixXd ConvergenceAcc::Iterate(const Eigen::MatrixXd& dmat,Eigen::MatrixXd& H,Eigen::VectorXd &MOenergies,Eigen::MatrixXd &MOs,double totE){
36  Eigen::MatrixXd H_guess=Eigen::MatrixXd::Zero(H.rows(),H.cols());
37 
38  if(int(_mathist.size())==_opt.histlength){
39  _totE.erase(_totE.begin()+_maxerrorindex);
40  _mathist.erase(_mathist.begin()+_maxerrorindex);
41  _dmatHist.erase(_dmatHist.begin()+_maxerrorindex);
42  }
43 
44  _totE.push_back(totE);
45  if(_opt.mode!=KSmode::fractional){
46  double gap=MOenergies(_nocclevels)-MOenergies(_nocclevels-1);
47  if((_diiserror>_opt.levelshiftend && _opt.levelshift>0.0) || gap<1e-6){
48  Levelshift(H);
49  }
50  }
51  const Eigen::MatrixXd& S=_S->Matrix();
52  Eigen::MatrixXd errormatrix=Sminusahalf.transpose()*(H*dmat*S-S*dmat*H)*Sminusahalf;
53  _diiserror=errormatrix.cwiseAbs().maxCoeff();
54 
55  _mathist.push_back(H);
56  _dmatHist.push_back(dmat);
57 
58  if(_opt.maxout){
59  if (_diiserror>_maxerror){
60  _maxerror=_diiserror;
61  _maxerrorindex=_mathist.size()-1;
62  }
63  }
64 
65  _diis.Update(_maxerrorindex,errormatrix);
66  bool diis_error=false;
67  CTP_LOG(ctp::logDEBUG,*_log) << ctp::TimeStamp()
68  << " DIIs error " << getDIIsError() << std::flush;
69 
70  CTP_LOG(ctp::logDEBUG,*_log) << ctp::TimeStamp() <<" Delta Etot "<<getDeltaE()<< std::flush;
71 
72  if ((_diiserror<_opt.adiis_start ||_diiserror<_opt.diis_start) && _opt.usediis && _mathist.size()>2){
73  Eigen::VectorXd coeffs;
74  //use ADIIs if energy has risen a lot in current iteration
75 
76  if(_diiserror>_opt.diis_start || _totE.back()>0.9*_totE[_totE.size()-2]){
77  coeffs=_adiis.CalcCoeff(_dmatHist,_mathist);
78  diis_error=!_adiis.Info();
79  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using ADIIS for next guess" << std::flush;
80 
81  }
82  else{
83  coeffs=_diis.CalcCoeff();
84  diis_error=!_diis.Info();
85  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using DIIS for next guess" << std::flush;
86 
87  }
88  if(diis_error){
89  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " (A)DIIS failed using mixing instead" << std::flush;
90  H_guess=H;
91  }else{
92  for (int i=0;i<coeffs.size();i++){
93  if(std::abs(coeffs(i))<1e-8){ continue;}
94  H_guess+=coeffs(i)*_mathist[i];
95  }
96  }
97 
98  }
99  else{
100  H_guess=H;
101  }
102 
103 
104  SolveFockmatrix( MOenergies,MOs,H_guess);
105  Eigen::MatrixXd dmatout=DensityMatrix(MOs,MOenergies);
106 
107  if(_diiserror>_opt.adiis_start ||!_opt.usediis || diis_error ||_mathist.size()<=2 ){
108  _usedmixing=true;
109  dmatout=_opt.mixingparameter*dmat+(1.0-_opt.mixingparameter)*dmatout;
110  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using Mixing with alpha="<<_opt.mixingparameter << std::flush;
111  }else{
112  _usedmixing=false;
113  }
114  return dmatout;
115  }

136  void ConvergenceAcc::SolveFockmatrix(Eigen::VectorXd& MOenergies,Eigen::MatrixXd& MOs,const Eigen::MatrixXd&H){
137  //transform to orthogonal for
138  Eigen::MatrixXd H_ortho=Sminusahalf.transpose()*H*Sminusahalf;
139  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(H_ortho);
140 
141  if(es.info()!=Eigen::ComputationInfo::Success){
142  throw std::runtime_error("Matrix Diagonalisation failed. DiagInfo"+std::to_string(es.info()));
143  }
144 
145  MOsinv=es.eigenvectors().transpose()*Sonehalf;
146  MOenergies=es.eigenvalues();
147 
148  MOs=Sminusahalf*es.eigenvectors();
149  return;
150  }

152  void ConvergenceAcc::Levelshift(Eigen::MatrixXd& H) const{
153  if(_opt.levelshift<1e-9){
154  return;
155  }
156  if(MOsinv.rows()<1){
157  throw std::runtime_error("ConvergenceAcc::Levelshift: Call SolveFockmatrix before Levelshift, MOsinv not initialized");
158  }
159  Eigen::MatrixXd virt = Eigen::MatrixXd::Zero(H.rows(),H.cols());
160  for (int i = _nocclevels; i < H.rows(); i++) {
161  virt(i, i) = _opt.levelshift;
162  }
163 
164  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using levelshift:" << _opt.levelshift << " Hartree" << std::flush;
165  Eigen::MatrixXd vir= MOsinv.transpose()*virt*MOsinv ;
166  H+=vir;
167  return;
168  }

199  Eigen::MatrixXd ConvergenceAcc::DensityMatrixGroundState_frac(const Eigen::MatrixXd& MOs, const Eigen::VectorXd& MOEnergies)const {
200  if (_opt.numberofelectrons == 0) {
201  return Eigen::MatrixXd::Zero(MOs.rows(),MOs.cols());
202  }
203  int numofelec=_opt.numberofelectrons;
204  Eigen::VectorXd occupation = Eigen::VectorXd::Zero(MOEnergies.size());
205 
206  std::vector< std::vector<int> > degeneracies;
207  double buffer=1e-4;
208  degeneracies.push_back(std::vector<int>{0});
209  for (int i=1;i<occupation.size();i++){
210  if(MOEnergies(i)<MOEnergies(degeneracies[degeneracies.size()-1][0])+buffer){
211  degeneracies[degeneracies.size()-1].push_back(i);
212  }
213  else{
214  degeneracies.push_back(std::vector<int>{i});
215  }
216  }
217  for (const std::vector<int>& deglevel:degeneracies){
218  int numofpossibleelectrons=2*deglevel.size();
219  if(numofpossibleelectrons<=numofelec){
220  for (const int& i:deglevel){
221  occupation(i)=2;
222  }
223  numofelec-=numofpossibleelectrons;
224  }else if(numofpossibleelectrons>numofelec){
225  double occ=double(numofelec)/double(deglevel.size());
226  for (const int& i:deglevel){
227  occupation(i)=occ;
228  }
229  break;
230  }
231  }
232  Eigen::MatrixXd dmatGS =MOs*occupation.asDiagonal()*MOs.transpose();
233  return dmatGS;
234  }
