37  Eigen::MatrixXd ConvergenceAcc::Iterate(const Eigen::MatrixXd& dmat,Eigen::MatrixXd& H,Eigen::VectorXd &MOenergies,Eigen::MatrixXd &MOs,double totE){
38  Eigen::MatrixXd H_guess=Eigen::MatrixXd::Zero(H.rows(),H.cols());
39 
40  if(int(_mathist.size())>_histlength){
41  delete _mathist[_maxerrorindex];
42  delete _dmatHist[_maxerrorindex];
43  _totE.erase(_totE.begin()+_maxerrorindex);
44  _mathist.erase(_mathist.begin()+_maxerrorindex);
45  _dmatHist.erase(_dmatHist.begin()+_maxerrorindex);
46  }
47 
48  _totE.push_back(totE);
49  if(_mode!=KSmode::fractional){
50  double gap=MOenergies(_nocclevels)-MOenergies(_nocclevels-1);
51  if((_diiserror>_levelshiftend && _levelshift>0.0) || gap<1e-6){
52  Levelshift(H);
53  }
54  }
55  const Eigen::MatrixXd& S=_S->Matrix();
56  Eigen::MatrixXd errormatrix=Sminusahalf.transpose()*(H*dmat*S-S*dmat*H)*Sminusahalf;
57  _diiserror=errormatrix.cwiseAbs().maxCoeff();
58 
59  Eigen::MatrixXd* old=new Eigen::MatrixXd;
60  *old=H;
61  _mathist.push_back(old);
62  Eigen::MatrixXd* dold=new Eigen::MatrixXd;
63  *dold=dmat;
64  _dmatHist.push_back(dold);
65 
66  if(_maxout){
67  if (_diiserror>_maxerror){
68  _maxerror=_diiserror;
69  _maxerrorindex=_mathist.size();
70  }
71  }
72 
73  _diis.Update(_maxerrorindex,errormatrix);
74  bool diis_error=false;
75 
76 
77  if ((_diiserror<_adiis_start ||_diiserror<_diis_start) && _usediis && _mathist.size()>2){
78  Eigen::VectorXd coeffs;
79  //use ADIIs if energy has risen a lot in current iteration
80 
81  if(_diiserror>_diis_start || _totE[_totE.size()-1]>0.9*_totE[_totE.size()-2]){
82  coeffs=_adiis.CalcCoeff(_dmatHist,_mathist);
83  diis_error=!_adiis.Info();
84  if(_noisy){
85  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using ADIIS" << std::flush;
86  }
87  }
88  else{
89  coeffs=_diis.CalcCoeff();
90  diis_error=!_diis.Info();
91  if(_noisy){
92  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using DIIS" << std::flush;
93  }
94  }
95  if(diis_error){
96  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " DIIS failed using mixing instead" << std::flush;
97  H_guess=H;
98  }else{
99  for (int i=0;i<coeffs.size();i++){
100  if(std::abs(coeffs(i))<1e-8){ continue;}
101  H_guess+=coeffs(i)*(*_mathist[i]);
102  }
103  }
104 
105  }
106  else{
107  H_guess=H;
108  }
109 
110 
111  SolveFockmatrix( MOenergies,MOs,H_guess);
112  Eigen::MatrixXd dmatout=DensityMatrix(MOs,MOenergies);
113 
114 
115  if(_diiserror>_adiis_start ||!_usediis || diis_error ||_mathist.size()<=2 ){
116  _usemixing=true;
117  dmatout=_mixingparameter*dmat+(1.0-_mixingparameter)*dmatout;
118  if(_noisy){
119  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using Mixing with alpha="<<_mixingparameter << std::flush;
120  }
121  }else{
122  _usemixing=false;
123  }
124  return dmatout;
125  }

127  void ConvergenceAcc::SolveFockmatrix(Eigen::VectorXd& MOenergies,Eigen::MatrixXd& MOs,const Eigen::MatrixXd&H){
128  //transform to orthogonal for
129  Eigen::MatrixXd H_ortho=Sminusahalf.transpose()*H*Sminusahalf;
130  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(H_ortho);
131  if(es.info()!=Eigen::ComputationInfo::Success){
132  std::cerr<<"DiagInfo "<<es.info()<<std::endl;
133  throw std::runtime_error("Matrix Diagonalisation failed");
134  }
135 
136  MOsinv=es.eigenvectors().transpose()*Sonehalf;
137  MOenergies=es.eigenvalues();
138 
139  MOs=Sminusahalf*es.eigenvectors();
140  return;
141  }

143  void ConvergenceAcc::Levelshift(Eigen::MatrixXd& H) {
144  if(_levelshift<1e-9){
145  return;
146  }
147  if(MOsinv.rows()<1){
148  throw std::runtime_error("ConvergenceAcc::Levelshift: Call SolveFockmatrix before Levelshift, MOsinv not initialized");
149  }
150  Eigen::MatrixXd virt = Eigen::MatrixXd::Zero(H.rows(),H.cols());
151  for (int i = _nocclevels; i < H.rows(); i++) {
152  virt(i, i) = _levelshift;
153  }
154 
155  if(_noisy){
156  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using levelshift:" << _levelshift << " Hartree" << std::flush;
157  }
158  Eigen::MatrixXd vir= MOsinv.transpose()*virt*MOsinv ;
159  H+=vir;
160 
161  return;
162  }

193  Eigen::MatrixXd ConvergenceAcc::DensityMatrixGroundState_frac(const Eigen::MatrixXd& MOs, const Eigen::VectorXd& MOEnergies) {
194  if (_numberofelectrons == 0) {
195  return Eigen::MatrixXd::Zero(MOs.rows(),MOs.cols());
196  }
197  int numofelec=_numberofelectrons;
198  Eigen::VectorXd occupation = Eigen::VectorXd::Zero(MOEnergies.size());
199 
200  std::vector< std::vector<int> > degeneracies;
201  double buffer=1e-4;
202  degeneracies.push_back(std::vector<int>{0});
203  for (int i=1;i<occupation.size();i++){
204  if(MOEnergies(i)<MOEnergies(degeneracies[degeneracies.size()-1][0])+buffer){
205  degeneracies[degeneracies.size()-1].push_back(i);
206  }
207  else{
208  degeneracies.push_back(std::vector<int>{i});
209  }
210  }
211  for (const std::vector<int>& deglevel:degeneracies){
212  int numofpossibleelectrons=2*deglevel.size();
213  if(numofpossibleelectrons<=numofelec){
214  for (const int& i:deglevel){
215  occupation(i)=2;
216  }
217  numofelec-=numofpossibleelectrons;
218  }else if(numofpossibleelectrons>numofelec){
219  double occ=double(numofelec)/double(deglevel.size());
220  for (const int& i:deglevel){
221  occupation(i)=occ;
222  }
223  break;
224  }
225  }
226  Eigen::MatrixXd _dmatGS =MOs*occupation.asDiagonal()*MOs.transpose();
227  return _dmatGS;
228  }
