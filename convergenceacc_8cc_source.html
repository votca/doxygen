27  void ConvergenceAcc::setOverlap(const Eigen::MatrixXd* _S,double etol){
28  S=_S;
29  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es((*S));
30  if(_noisy){
31  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Smallest value of AOOverlap matrix is "<<es.eigenvalues()(0) << std::flush;
32  }
33  Eigen::VectorXd diagonal=Eigen::VectorXd::Zero(es.eigenvalues().size());
34  int removedfunctions=0;
35  for (unsigned i=0;i<diagonal.size();++i){
36  if(es.eigenvalues()(i)<etol){
37  removedfunctions++;
38  }else{
39  diagonal(i)=1.0/std::sqrt(es.eigenvalues()(i));
40  }
41  }
42  if(_noisy){
43  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Removed "<<removedfunctions<<" basisfunction from inverse overlap matrix" << std::flush;
44  }
45  Sminusahalf =es.eigenvectors() * diagonal.asDiagonal() * es.eigenvectors().transpose();
46  Sonehalf=es.operatorSqrt();
47  mix.Configure(_mixingparameter,S);
48  return;
49  }

52  Eigen::MatrixXd ConvergenceAcc::Iterate(const Eigen::MatrixXd& dmat,Eigen::MatrixXd& H,Eigen::VectorXd &MOenergies,Eigen::MatrixXd &MOs,double totE){
53  Eigen::MatrixXd H_guess=Eigen::MatrixXd::Zero(H.rows(),H.cols());
54 
55  if(_mathist.size()>_histlength){
56  delete _mathist[_maxerrorindex];
57  delete _dmatHist[_maxerrorindex];
58  _totE.erase(_totE.begin()+_maxerrorindex);
59  _mathist.erase(_mathist.begin()+_maxerrorindex);
60  _dmatHist.erase(_dmatHist.begin()+_maxerrorindex);
61 
62  }
63 
64  _totE.push_back(totE);
65 
66  double gap=MOenergies(_nocclevels)-MOenergies(_nocclevels-1);
67 
68  if((_diiserror>_levelshiftend && _levelshift>0.0) || gap<1e-6){
69  if(_mode!=KSmode::fractional){
70  Levelshift(H);
71  }
72  }
73 
74  Eigen::MatrixXd errormatrix=Sminusahalf.transpose()*(H*dmat*(*S)-(*S)*dmat*H)*Sminusahalf;
75  _diiserror=errormatrix.cwiseAbs().maxCoeff();
76 
77  Eigen::MatrixXd* old=new Eigen::MatrixXd;
78  *old=H;
79  _mathist.push_back(old);
80  Eigen::MatrixXd* dold=new Eigen::MatrixXd;
81  *dold=dmat;
82  _dmatHist.push_back(dold);
83 
84  if(_maxout){
85  if (_diiserror>_maxerror){
86  _maxerror=_diiserror;
87  _maxerrorindex=_mathist.size();
88  }
89  }
90 
91  diis.Update(_maxerrorindex,errormatrix);
92  bool diis_error=false;
93 
94 
95  if ((_diiserror<_adiis_start ||_diiserror<_diis_start) && _usediis && _mathist.size()>2){
96  Eigen::VectorXd coeffs;
97  //use ADIIs if energy has risen a lot in current iteration
98 
99  if(_diiserror>_diis_start || _totE[_totE.size()-1]>0.9*_totE[_totE.size()-2]){
100  coeffs=adiis.CalcCoeff(_dmatHist,_mathist);
101  diis_error=!adiis.Info();
102  if(_noisy){
103  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using ADIIS" << std::flush;
104  }
105  }
106  else{
107  coeffs=diis.CalcCoeff();
108  diis_error=!diis.Info();
109  if(_noisy){
110  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using DIIS" << std::flush;
111  }
112  }
113  if(diis_error){
114  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " DIIS failed using mixing instead" << std::flush;
115  H_guess=H;
116  }else{
117  for (unsigned i=0;i<coeffs.size();i++){
118  if(std::abs(coeffs(i))<1e-8){ continue;}
119  H_guess+=coeffs(i)*(*_mathist[i]);
120  }
121  }
122 
123  }
124  else{
125  H_guess=H;
126  }
127 
128 
129  SolveFockmatrix( MOenergies,MOs,H_guess);
130  Eigen::MatrixXd dmatout=DensityMatrix(MOs,MOenergies);
131 
132  mix.Updatemix(dmat,dmatout);
133  if(_diiserror>_adiis_start ||!_usediis || diis_error ||_mathist.size()<=2 ){
134  _usemixing=true;
135  dmatout=mix.MixDmat(dmat,dmatout);
136  if(_noisy){
137  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using Mixing with alpha="<<mix.getAlpha() << std::flush;
138  }
139  }else{
140  _usemixing=false;
141  }
142  return dmatout;
143  }

145  void ConvergenceAcc::SolveFockmatrix(Eigen::VectorXd& MOenergies,Eigen::MatrixXd& MOs,const Eigen::MatrixXd&H){
146  //transform to orthogonal for
147  Eigen::MatrixXd H_ortho=Sminusahalf.transpose()*H*Sminusahalf;
148  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(H_ortho);
149  if(es.info()!=Eigen::ComputationInfo::Success){
150  throw std::runtime_error("SolveFockmatrix: Matrix Diagonalisation failed!");
151  }
152  MOsinv=es.eigenvectors().transpose()*Sonehalf;
153  MOenergies=es.eigenvalues();
154 
155  MOs=Sminusahalf*es.eigenvectors();
156  return;
157  }

159  void ConvergenceAcc::Levelshift(Eigen::MatrixXd& H) {
160  if(_levelshift<1e-9){
161  return;
162  }
163  if(MOsinv.rows()<1){
164  throw std::runtime_error("ConvergenceAcc::Levelshift: Call SolveFockmatrix before Levelshift, MOsinv not initialized");
165  }
166  Eigen::MatrixXd virt = Eigen::MatrixXd::Zero(H.rows(),H.cols());
167  for (unsigned _i = _nocclevels; _i < H.rows(); _i++) {
168  virt(_i, _i) = _levelshift;
169  }
170 
171  if(_noisy){
172  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using levelshift:" << _levelshift << " Hartree" << std::flush;
173  }
174  Eigen::MatrixXd vir= MOsinv.transpose()*virt*MOsinv ;
175  H+=vir;
176 
177  return;
178  }

209  Eigen::MatrixXd ConvergenceAcc::DensityMatrixGroundState_frac(const Eigen::MatrixXd& MOs, const Eigen::VectorXd& MOEnergies) {
210  if (_nocclevels == 0) {
211  return Eigen::MatrixXd::Zero(MOs.rows(),MOs.cols());
212  }
213  unsigned numofelec=2*_nocclevels;
214  Eigen::VectorXd occupation = Eigen::VectorXd::Zero(MOEnergies.size());
215 
216  double buffer = 0.0001;
217  double homo_energy = MOEnergies(numofelec - 1);
218  std::vector<unsigned> degeneracies;
219 
220  for (unsigned _level = 0; _level < occupation.size(); _level++) {
221  if (MOEnergies(_level)<(homo_energy - buffer)) {
222  occupation(_level) = 1.0;
223  numofelec--;
224  } else if (std::abs(MOEnergies(_level) - homo_energy) < buffer) {
225  degeneracies.push_back(_level);
226  } else if (MOEnergies(_level)>(homo_energy + buffer)) {
227  occupation(_level) = 0.0;
228  }
229  }
230  double deg_occupation = double(numofelec) / double(degeneracies.size());
231  for (unsigned _level = 0; _level < degeneracies.size(); _level++) {
232  occupation(degeneracies[_level]) = deg_occupation;
233  }
234  Eigen::MatrixXd _dmatGS =MOs*occupation.asDiagonal()*MOs.transpose();
235  return _dmatGS;
236  }
