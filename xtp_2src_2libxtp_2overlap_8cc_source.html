44 double Overlap::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
45  Orbitals* _orbitalsB, ub::matrix<double>* _JAB, double _energy_difference ) {
46 
47 
48  int _levelsA = _orbitalsA->getNumberOfLevels();
49 
50  if ( _energy_difference != 0 ) {
51  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
52  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
53 
54  double _JAB_sq = 0; double _JAB_one_level;
55 
56  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
57  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
58  //cout << *iA << ':' << *iB << endl;
59  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
60  _JAB_sq += _JAB_one_level*_JAB_one_level ;
61  }
62  }
63 
64  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
65 
66  } else {
67 
68  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
69 
70  }
71  // the matrix should be symmetric, could also return this element
72  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );
73 }

86 bool Overlap::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
87  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
88 
89  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating electronic couplings" << flush;
90 
91  const std::vector<ctp::QMAtom*> atomsA=_orbitalsA->QMAtoms();
92  const std::vector<ctp::QMAtom*> atomsB=_orbitalsB->QMAtoms();
93  const std::vector<ctp::QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
94 
95  for (unsigned i=0;i<atomsAB.size();i++){
96  ctp::QMAtom* dimer=atomsAB[i];
97  ctp::QMAtom* monomer=NULL;
98  if (i<atomsA.size()){
99  monomer=atomsA[i];
100  }
101  else if (i<atomsB.size()+atomsA.size() ){
102  monomer=atomsB[i-atomsA.size()];
103  }
104  else{
105  throw runtime_error((format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
106  }
107 
108  if(monomer->type != dimer->type){
109  throw runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
110  }
111  if(std::abs(monomer->x-dimer->x)>0.001 || std::abs(monomer->y-dimer->y)>0.001 || std::abs(monomer->z-dimer->z)>0.001){
112  CTP_LOG(ctp::logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << flush;
113  break;
114  }
115 
116  }
117 
118  // constructing the direct product orbA x orbB
119  int _basisA = _orbitalsA->getBasisSetSize();
120  int _basisB = _orbitalsB->getBasisSetSize();
121 
122  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
123  CTP_LOG(ctp::logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
124  return false;
125  }
126 
127  int _levelsA = _orbitalsA->getNumberOfLevels();
128  int _levelsB = _orbitalsB->getNumberOfLevels();
129 
130  //boost::timer t; // start timing
131  //double _st = t.elapsed();
132 
133  CTP_LOG(ctp::logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
134  << " B[" << _levelsB << ":" << _basisB << "]" << flush;
135 
136  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
137  CTP_LOG(ctp::logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
138  return false;
139  }
140 
141  // | Orbitals_A 0 | | Overlap_A |
142  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
143  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
144  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
145  ub::matrix<double> _psi_AxB ( _levelsA + _levelsB, _basisA + _basisB );
146 
147 
148  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing direct product AxB ["
149  << _psi_AxB.size1() << "x"
150  << _psi_AxB.size2() << "]" << flush;
151 
152  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
153  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
154  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = _orbitalsA->MOCoefficients();
155  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = _orbitalsB->MOCoefficients();
156 
157  // psi_AxB * S_AB * psi_AB
158  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << flush;
159  if (_orbitalsAB->hasAOOverlap() ) {
160  CTP_LOG(ctp::logERROR,*_pLog) << "Overlap matrix is not stored";
161  return false;
162  }
163  ub::matrix<double> overlap= _orbitalsAB->AOOverlap();
164  ub::matrix<double> _psi_AB = ub::prod(overlap, ub::trans( _orbitalsAB->MOCoefficients()) );
165  ub::matrix<double> _psi_AxB_dimer_basis = ub::prod( _psi_AxB, _psi_AB );
166  _psi_AB.clear();
167 
168  //check to see if projection quality is sufficient
169  for (unsigned i=0;i<_psi_AxB_dimer_basis.size1();i++){
170  double mag=0.0;
171  for (unsigned j=0;j<_psi_AxB_dimer_basis.size2();j++){
172  mag+=_psi_AxB_dimer_basis(i,j)*_psi_AxB_dimer_basis(i,j);
173 
174  }
175  if (mag<0.95){
176  throw runtime_error("\nERROR: Projection of monomer orbitals on dimer is insufficient, maybe the orbital order is screwed up, otherwise increase dimer basis.\n");
177  }
178  }
179 
180 
181  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
182  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << flush;
183  ub::diagonal_matrix<double> _fock_AB( _orbitalsAB->getNumberOfLevels(), _orbitalsAB->MOEnergies().data() );
184  ub::matrix<double> _temp = ub::prod( _fock_AB, ub::trans( _psi_AxB_dimer_basis ) ) ;
185  ub::matrix<double> JAB_dimer = ub::prod( _psi_AxB_dimer_basis, _temp);
186 
187  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
188  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing Overlap matrix" << flush;
189  ub::matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
190 
191  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
192  << JAB_dimer.size1() << "x"
193  << JAB_dimer.size2() << "]" << flush;
194  double smalleig=linalg_loewdin(JAB_dimer, _S_AxB);
195  CTP_LOG(ctp::logDEBUG,*_pLog) << "Smallest eigenvalue of overlap matrix is "<<smalleig<< flush;
196  (*_JAB) = JAB_dimer;
197 
198 
199  CTP_LOG(ctp::logDEBUG,*_pLog) << "Done with electronic couplings" << flush;
200 
201  return true;
202 
203 }
