81 bool CsgApplication::EvaluateOptions() {
82  _do_mapping = false;
83  if (NeedsTopology()) {
84  CheckRequired("top", "no topology file specified");
85  }
86 
87  // check for mapping options
88  if (DoMapping()) {
89  // default mapping is on
90  if (DoMappingDefault()) {
91  // if the user does not explicitly ask to turn it off, cg is needed
92  if (OptionsMap().count("no-map") == 0) {
93  CheckRequired("cg", "no coarse graining definition specified");
94  _do_mapping = true;
95  }
96  if (OptionsMap().count("no-map") && OptionsMap().count("cg")) {
97  ShowHelpText(std::cout);
98  throw std::runtime_error(
99  "no-map and cg options are mutually exclusive!");
100  }
101  } // default mapping is off, if user gives cg, then do mapping
102  else if (OptionsMap().count("cg")) {
103  _do_mapping = true;
104  }
105  }
106 
107  /* check threading options */
108  if (DoThreaded()) {
109  _nthreads = _op_vm["nt"].as<Index>();
110  /* TODO
111  * does the number of threads make sense?
112  * which criteria should be used? smaller than system's cores?
113  */
114  }
115 
116  return true;
117 }

131 void CsgApplication::Worker::Run() {
132  while (_app->ProcessData(this)) {
133  if (_app->SynchronizeThreads()) {
134  Index id = getId();
135  _app->_threadsMutexesOut[id]->Lock();
136  _app->MergeWorker(this);
137  _app->_threadsMutexesOut[(id + 1) % _app->_nthreads]->Unlock();
138  }
139  }

142 bool CsgApplication::ProcessData(Worker *worker) {
143 
144  Index id;
145  id = worker->getId();
146 
147  if (SynchronizeThreads()) {
148  // wait til its your turn
149  _threadsMutexesIn[id]->Lock();
150  }
151  _traj_readerMutex.Lock();
152  if (_nframes == 0) {
153  _traj_readerMutex.Unlock();
154 
155  if (SynchronizeThreads()) {
156  // done processing? don't forget to unlock next worker anyway
157  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
158  }
159 
160  return false;
161  }
162  _nframes--;
163  if (!_is_first_frame || worker->getId() != 0) {
164  // get frame
165  bool tmpRes = _traj_reader->NextFrame(worker->_top);
166  if (!tmpRes) {
167  _traj_readerMutex.Unlock();
168  if (SynchronizeThreads()) {
169  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
170  }
171  return false;
172  }
173  }
174  if (worker->getId() == 0) {
175  _is_first_frame = false;
176  }
177 
178  _traj_readerMutex.Unlock();
179  if (SynchronizeThreads()) {
180  // unlock next frame for input
181  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
182  }
183  // evaluate
184  if (_do_mapping) {
185  worker->_map->Apply();
186  worker->EvalConfiguration(&worker->_top_cg, &worker->_top);
187  } else {
188  worker->EvalConfiguration(&worker->_top);
189  }
190 
191  return true;
192 }

194 void CsgApplication::Run(void) {
195  TopologyReader *reader;
196  // create reader for atomistic topology
197  reader = TopReaderFactory().Create(_op_vm["top"].as<std::string>());
198  if (reader == nullptr) {
199  throw std::runtime_error(std::string("input format not supported: ") +
200  _op_vm["top"].as<std::string>());
201  }
202 
203  class DummyWorker : public Worker {
204  public:
205  void EvalConfiguration(Topology *top, Topology *top_ref) override {
206  _app->EvalConfiguration(top, top_ref);
207  }
208  };
209 
210  // create the master worker
211  Worker *master = nullptr;
212  if (DoThreaded()) {
213  master = ForkWorker();
214  } else {
215  master = new DummyWorker();
216  }
217 
218  master->setApplication(this);
219  master->setId(0);
220  _myWorkers.push_back(master);
221 
222  CGEngine cg;
223 
225  // read in the topology for master
227  reader->ReadTopology(_op_vm["top"].as<std::string>(), master->_top);
228  // Ensure that the coarse grained topology will have the same boundaries
229  master->_top_cg.setBox(master->_top.getBox());
230 
231  std::cout << "I have " << master->_top.BeadCount() << " beads in "
232  << master->_top.MoleculeCount() << " molecules" << std::endl;
233  master->_top.CheckMoleculeNaming();
234 
235  if (_do_mapping) {
236  // read in the coarse graining definitions (xml files)
237  cg.LoadMoleculeType(_op_vm["cg"].as<std::string>());
238  // create the mapping + cg topology
239 
240  if (_op_vm.count("map-ignore") != 0) {
241  tools::Tokenizer tok(_op_vm["map-ignore"].as<std::string>(), ";");
242  for (std::string str : tok) {
243  boost::trim(str);
244  if (str.length() > 0) {
245  cg.AddIgnore(str);
246  }
247  }
248  }
249 
250  master->_map = cg.CreateCGTopology(master->_top, master->_top_cg);
251 
252  std::cout << "I have " << master->_top_cg.BeadCount() << " beads in "
253  << master->_top_cg.MoleculeCount()
254  << " molecules for the coarsegraining" << std::endl;
255 
256  // If the trajectory reader is off but mapping flag is specified do apply
257  // the mapping, this switch is necessary in cases where xml files are
258  // specified, which do not contain positional information. In such cases
259  // it is not possible to apply the positional mapping, a trajectory file
260  // must be read in.
261  if (DoTrajectory() == false) {
262  master->_map->Apply();
263  if (!EvaluateTopology(&master->_top_cg, &master->_top)) {
264  return;
265  }
266  }
267  } else if (!EvaluateTopology(&master->_top)) {
268  return;
269  }
270 
272  // Here trajectory parsing starts
274  if (DoTrajectory() && _op_vm.count("trj")) {
275  double begin = 0;
276  Index first_frame;
277  bool has_begin = false;
278 
279  if (_op_vm.count("begin")) {
280  has_begin = true;
281  begin = _op_vm["begin"].as<double>();
282  }
283 
284  _nframes = -1;
285  if (_op_vm.count("nframes")) {
286  _nframes = _op_vm["nframes"].as<Index>();
287  }
288 
289  first_frame = _op_vm["first-frame"].as<Index>();
290 
291  // create reader for trajectory
292  _traj_reader = TrjReaderFactory().Create(_op_vm["trj"].as<std::string>());
293  if (_traj_reader == nullptr) {
294  throw std::runtime_error(std::string("input format not supported: ") +
295  _op_vm["trj"].as<std::string>());
296  }
297  // open the trajectory
298  _traj_reader->Open(_op_vm["trj"].as<std::string>());
299 
301  // Create all the workers
303  for (Index thread = 1; thread < _nthreads && DoThreaded(); thread++) {
304  Worker *myWorker = ForkWorker();
305  myWorker->setApplication(this);
306  myWorker->setId(thread);
307  _myWorkers.push_back(myWorker);
308 
309  // this will be changed to CopyTopologyData
310  // read in the topology
311 
312  reader->ReadTopology(_op_vm["top"].as<std::string>(), myWorker->_top);
313  myWorker->_top.CheckMoleculeNaming();
314 
315  if (_do_mapping) {
316  // create the mapping + cg topology
317  myWorker->_map = cg.CreateCGTopology(myWorker->_top, myWorker->_top_cg);
318  }
319  }
320 
322  // Proceed to first frame of interest
324 
325  _traj_reader->FirstFrame(master->_top);
326  if (master->_top.getBoxType() == BoundaryCondition::typeOpen) {
327  std::cout
328  << "NOTE: You are using OpenBox boundary conditions. Check if this "
329  "is intended.\n"
330  << std::endl;
331  }
332  // seek first frame, let thread0 do that
333  bool bok;
334  for (bok = true; bok == true; bok = _traj_reader->NextFrame(master->_top)) {
335  if ((has_begin && (master->_top.getTime() < begin)) || first_frame > 1) {
336  first_frame--;
337  continue;
338  }
339  break;
340  }
341  if (!bok) { // trajectory was too short and we did not proceed to first
342  // frame
343  _traj_reader->Close();
344  delete _traj_reader;
345 
346  throw std::runtime_error(
347  "trajectory was too short, did not process a single frame");
348  }
349 
350  // notify all observers that coarse graining has begun
351  if (_do_mapping) {
352  master->_map->Apply();
353  BeginEvaluate(&master->_top_cg, &master->_top);
354  } else {
355  BeginEvaluate(&master->_top);
356  }
357 
358  _is_first_frame = true;
360  // start threads
361  if (DoThreaded()) {
362  for (size_t thread = 0; thread < _myWorkers.size(); thread++) {
363 
364  if (SynchronizeThreads()) {
365  tools::Mutex *myMutexIn = new tools::Mutex;
366  _threadsMutexesIn.push_back(myMutexIn);
367  // lock each worker for input
368  myMutexIn->Lock();
369 
370  tools::Mutex *myMutexOut = new tools::Mutex;
371  _threadsMutexesOut.push_back(myMutexOut);
372  // lock each worker for output
373  myMutexOut->Lock();
374  }
375  }
376  for (auto &_myWorker : _myWorkers) {
377  _myWorker->Start();
378  }
379 
380  if (SynchronizeThreads()) {
381  // unlock first thread and start ordered input/output
382  _threadsMutexesIn[0]->Unlock();
383  _threadsMutexesOut[0]->Unlock();
384  }
385  // mutex needed for merging if SynchronizeThreads()==False
386  tools::Mutex mergeMutex;
387  for (auto &myWorker : _myWorkers) {
388  myWorker->WaitDone();
389  if (!SynchronizeThreads()) {
390  mergeMutex.Lock();
391  MergeWorker(myWorker);
392  mergeMutex.Unlock();
393  }
394  delete myWorker;
395  }
396  for (size_t thread = 0; thread < _threadsMutexesIn.size(); ++thread) {
397  delete _threadsMutexesIn[thread];
398  delete _threadsMutexesOut[thread];
399  }
400 
401  } else {
402  master->Start();
403  master->WaitDone();
404  delete master;
405  }
406 
407  EndEvaluate();
408 
409  _myWorkers.clear();
410  _threadsMutexesIn.clear();
411  _threadsMutexesOut.clear();
412  _traj_reader->Close();
413 
414  delete _traj_reader;
415  }
416 
417  delete reader;
418 }
