79  bool CsgApplication::EvaluateOptions(void) {
80  _do_mapping = false;
81  if (NeedsTopology()) {
82  CheckRequired("top", "no topology file specified");
83  }
84 
85  // check for mapping options
86  if (DoMapping()) {
87  // default mapping is on
88  if (DoMappingDefault()) {
89  // if the user does not explicitly ask to turn it off, cg is needed
90  if (OptionsMap().count("no-map") == 0) {
91  CheckRequired("cg", "no coarse graining definition specified");
92  _do_mapping = true;
93  }
94  if (OptionsMap().count("no-map") && OptionsMap().count("cg")){
95  ShowHelpText(cout);
96  throw runtime_error("no-map and cg options are mutually exclusive!");
97  }
98  }// default mapping is off, if user gives cg, then do mapping
99  else if (OptionsMap().count("cg")) {
100  _do_mapping = true;
101  }
102  }
103 
104  /* check threading options */
105  if (DoThreaded()) {
106  _nthreads = _op_vm["nt"].as<int > ();
107  /* TODO
108  * does the number of threads make sense?
109  * which criteria should be used? smaller than system's cores?
110  */
111  }
112 
113  return true;

139  bool CsgApplication::ProcessData(Worker * worker) {
140 
141  int id;
142  id = worker->getId();
143 
144  if (SynchronizeThreads()) {
145  //wait til its your turn
146  _threadsMutexesIn[id]->Lock();
147  }
148  _traj_readerMutex.Lock();
149  if (_nframes == 0) {
150  _traj_readerMutex.Unlock();
151 
152  if (SynchronizeThreads()) {
153  //done processing? don't forget to unlock next worker anyway
154  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
155  }
156 
157  return false;
158  }
159  _nframes--;
160  if (!_is_first_frame || worker->getId() != 0) {
161  //get frame
162  bool tmpRes = _traj_reader->NextFrame(worker->_top);
163  if (!tmpRes) {
164  _traj_readerMutex.Unlock();
165  if (SynchronizeThreads())
166  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
167  return false;
168  }
169  }
170  if (worker->getId() == 0)
171  _is_first_frame = false;
172 
173  _traj_readerMutex.Unlock();
174  if (SynchronizeThreads()) {
175  //unlock next frame for input
176  _threadsMutexesIn[(id + 1) % _nthreads]->Unlock();
177  }
178  //evaluate
179  if (_do_mapping) {
180  worker->_map->Apply();
181  worker->EvalConfiguration(&worker->_top_cg, &worker->_top);
182  } else
183  worker->EvalConfiguration(&worker->_top);
184 
185  return true;
186  }

188  void CsgApplication::Run(void) {
189  TopologyReader *reader;
190  // create reader for atomistic topology
191  reader = TopReaderFactory().Create(_op_vm["top"].as<string > ());
192  if (reader == NULL)
193  throw runtime_error(string("input format not supported: ") + _op_vm["top"].as<string > ());
194 
195  class DummyWorker : public Worker {
196  public:
197 
198  void EvalConfiguration(Topology *top, Topology *top_ref) {
199  _app->EvalConfiguration(top, top_ref);
200  }
201  };
202 
203  // create the master worker
204  Worker *master = NULL;
205  if (DoThreaded())
206  master = ForkWorker();
207  else
208  master = new DummyWorker();
209 
210  master->setApplication(this);
211  master->setId(0);
212  _myWorkers.push_back(master);
213 
214  CGEngine cg;
215 
217  // read in the topology for master
219  reader->ReadTopology(_op_vm["top"].as<string > (), master->_top);
220  cout << "I have " << master->_top.BeadCount() << " beads in " << master->_top.MoleculeCount() << " molecules" << endl;
221  master->_top.CheckMoleculeNaming();
222 
223  if (_do_mapping) {
224  // read in the coarse graining definitions (xml files)
225  cg.LoadMoleculeType(_op_vm["cg"].as<string > ());
226  // create the mapping + cg topology
227 
228  if(_op_vm.count("map-ignore") != 0) {
229  Tokenizer tok(_op_vm["map-ignore"].as<string>(), ";");
230  Tokenizer::iterator iter;
231  for(iter=tok.begin(); iter!=tok.end(); ++iter) {
232  string str=*iter;
233  boost::trim(str);
234  if(str.length() > 0)
235  cg.AddIgnore(str);
236  }
237  }
238 
239  master->_map = cg.CreateCGTopology(master->_top, master->_top_cg);
240 
241  cout << "I have " << master->_top_cg.BeadCount() << " beads in " << master->_top_cg.MoleculeCount() << " molecules for the coarsegraining" << endl;
242  master->_map->Apply();
243  if (!EvaluateTopology(&master->_top_cg, &master->_top))
244  return;
245  } else
246  if (!EvaluateTopology(&master->_top))
247  return;
248 
250  // Here trajectory parsing starts
252  if (DoTrajectory() && _op_vm.count("trj")) {
253  double begin;
254  int first_frame;
255  bool has_begin = false;
256 
257  if (_op_vm.count("begin")) {
258  has_begin = true;
259  begin = _op_vm["begin"].as<double>();
260  }
261 
262  _nframes = -1;
263  if (_op_vm.count("nframes")) {
264  _nframes = _op_vm["nframes"].as<int>();
265  }
266 
267  first_frame = _op_vm["first-frame"].as<int>();
268 
269  // create reader for trajectory
270  _traj_reader = TrjReaderFactory().Create(_op_vm["trj"].as<string > ());
271  if (_traj_reader == NULL)
272  throw runtime_error(string("input format not supported: ") + _op_vm["trj"].as<string > ());
273  // open the trajectory
274  _traj_reader->Open(_op_vm["trj"].as<string > ());
275 
277  // Create all the workers
279  for (int thread = 1; thread < _nthreads && DoThreaded(); thread++) {
280  Worker *myWorker = ForkWorker();
281  myWorker->setApplication(this);
282  myWorker->setId(thread);
283  _myWorkers.push_back(myWorker);
284 
285  // this will be changed to CopyTopologyData
286  // read in the topology
287  reader->ReadTopology(_op_vm["top"].as<string > (), myWorker->_top);
288  myWorker->_top.CheckMoleculeNaming();
289 
290  if (_do_mapping) {
291  // create the mapping + cg topology
292  myWorker->_map = cg.CreateCGTopology(myWorker->_top, myWorker->_top_cg);
293  }
294  }
295 
297  // Proceed to first frame of interest
299 
300  _traj_reader->FirstFrame(master->_top);
301  if(master->_top.getBoxType()==BoundaryCondition::typeOpen) {
302  cout << "NOTE: You are using OpenBox boundary conditions. Check if this is intended.\n" << endl;
303  }
304  //seek first frame, let thread0 do that
305  bool bok;
306  for (bok = true; bok == true; bok = _traj_reader->NextFrame(master->_top)) {
307  if (((master->_top.getTime() < begin) && has_begin) || first_frame > 1) {
308  first_frame--;
309  continue;
310  }
311  break;
312  }
313  if (!bok) { // trajectory was too short and we did not proceed to first frame
314  _traj_reader->Close();
315  delete _traj_reader;
316 
317  throw std::runtime_error("trajectory was too short, did not process a single frame");
318  }
319 
320  // notify all observers that coarse graining has begun
321  if (_do_mapping) {
322  master->_map->Apply();
323  BeginEvaluate(&master->_top_cg, &master->_top);
324  } else
325  BeginEvaluate(&master->_top);
326 
327  _is_first_frame = true;
329  //start threads
330  if (DoThreaded()) {
331  for (size_t thread = 0; thread < _myWorkers.size(); thread++) {
332 
333  if (SynchronizeThreads()) {
334  Mutex *myMutexIn = new Mutex;
335  _threadsMutexesIn.push_back(myMutexIn);
336  // lock each worker for input
337  myMutexIn->Lock();
338 
339  Mutex *myMutexOut = new Mutex;
340  _threadsMutexesOut.push_back(myMutexOut);
341  // lock each worker for output
342  myMutexOut->Lock();
343  }
344  }
345  for (size_t thread = 0; thread < _myWorkers.size(); thread++)
346  _myWorkers[thread]->Start();
347 
348  if (SynchronizeThreads()) {
349  //unlock first thread and start ordered input/output
350  _threadsMutexesIn[0]->Unlock();
351  _threadsMutexesOut[0]->Unlock();
352  }
353  // mutex needed for merging if SynchronizeThreads()==False
354  Mutex mergeMutex;
355  for (size_t thread = 0; thread < _myWorkers.size(); thread++) {
356  _myWorkers[thread]->WaitDone();
357  if (!SynchronizeThreads()) {
358  mergeMutex.Lock();
359  MergeWorker(_myWorkers[thread]);
360  mergeMutex.Unlock();
361  }
362  delete _myWorkers[thread];
363  }
364  for (size_t thread = 0; thread < _threadsMutexesIn.size(); ++thread) {
365  delete _threadsMutexesIn[thread];
366  delete _threadsMutexesOut[thread];
367  }
368 
369 
370  } else {
371  master->Run();
372  delete master;
373  }
374 
375  EndEvaluate();
376 
377  _myWorkers.clear();
378  _threadsMutexesIn.clear();
379  _threadsMutexesOut.clear();
380  _traj_reader->Close();
381 
382  delete _traj_reader;
383  }
384 
385  delete reader;
386 
387  }
