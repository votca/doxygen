44 class QMMInterface
45 {
46 public:
47 
48  QMMInterface() { _polar_table = ctp::POLAR_TABLE(); };
49  ~QMMInterface() {};
50 
51  // CONVERSION QM -> MM
52  ctp::APolarSite *Convert(ctp::QMAtom *atm, int id = -1) {
53  double A_to_nm = 0.1;
54  vec pos = A_to_nm*vec(atm->x, atm->y, atm->z);
55  double q = atm->charge;
56  std::string elem = atm->type;
57  double pol = 0.0;
58  try {
59  pol = _polar_table.at(elem);
60  }
61  catch(out_of_range) {
62  std::cout << std::endl << "QMMInterface - no default polarizability given "
63  << "for element type '" << elem << "'. Defaulting to 1A**3" << std::flush;
64  pol = 1e-3;
65  }
66 
67  ctp::APolarSite *new_aps = new ctp::APolarSite(id, elem);
68  new_aps->setRank(0);
69  new_aps->setPos(pos);
70  new_aps->setQ00(q,0); // <- charge state 0 <> 'neutral'
71  new_aps->setIsoP(pol);
72 
73  return new_aps;
74  }
75 
76  ctp::PolarSeg *Convert(std::vector<ctp::QMAtom*> &atms) {
77  ctp::PolarSeg *new_pseg = new ctp::PolarSeg();
78  std::vector<ctp::QMAtom*>::iterator it;
79  for (it = atms.begin(); it < atms.end(); ++it) {
80  ctp::APolarSite *new_site = this->Convert(*it);
81  new_pseg->push_back(new_site);
82  }
83  return new_pseg;
84  }

87  ctp::QMAtom *Convert(ctp::APolarSite*);
88  std::vector<ctp::QMAtom*> Convert(ctp::PolarSeg*);
89 
90 private:
91 
92  // Allocates polarizabilities in A**3 to element types
93  std::map<std::string,double> _polar_table;
94 
95 };
