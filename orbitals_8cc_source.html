46 std::vector<Index> Orbitals::CheckDegeneracy(Index level,
47  double energy_difference) const {
48 
49  std::vector<Index> result;
50  if (level > _mos.eigenvalues().size()) {
51  throw std::runtime_error(
52  "Level for degeneracy is higher than maximum level");
53  }
54  double MOEnergyLevel = _mos.eigenvalues()(level);
55 
56  for (Index i = 0; i < _mos.eigenvalues().size(); ++i) {
57  if (std::abs(_mos.eigenvalues()(i) - MOEnergyLevel) < energy_difference) {
58  result.push_back(i);
59  }
60  }
61 
62  if (result.empty()) {
63  result.push_back(level);
64  }
65  return result;
66 }

77 Eigen::MatrixXd Orbitals::DensityMatrixFull(const QMState& state) const {
78  if (state.isTransition()) {
79  return this->TransitionDensityMatrix(state);
80  }
81  Eigen::MatrixXd result = this->DensityMatrixGroundState();
82  if (state.Type().isExciton()) {
83  std::array<Eigen::MatrixXd, 2> DMAT = DensityMatrixExcitedState(state);
84  result = result - DMAT[0] + DMAT[1]; // Ground state + hole_contribution +
85  // electron contribution
86  } else if (state.Type() == QMStateType::DQPstate) {
87  Eigen::MatrixXd DMATQP = DensityMatrixQuasiParticle(state);
88  if (state.StateIdx() > getHomo()) {
89  result += DMATQP;
90  } else {
91  result -= DMATQP;
92  }
93  } else if (state.Type() != QMStateType::Gstate) {
94  throw std::runtime_error(
95  "DensityMatrixFull does not yet implement QMStateType:" +
96  state.Type().ToLongString());
97  }
98  return result;
99 }

135 Eigen::Vector3d Orbitals::CalcElDipole(const QMState& state) const {
136  Eigen::Vector3d nuclei_dip = Eigen::Vector3d::Zero();
137  if (!state.isTransition()) {
138  for (const QMAtom& atom : _atoms) {
139  nuclei_dip += (atom.getPos() - _atoms.getPos()) * atom.getNuccharge();
140  }
141  }
142  AOBasis basis = SetupDftBasis();
143  AODipole dipole;
144  dipole.setCenter(_atoms.getPos());
145  dipole.Fill(basis);
146 
147  Eigen::MatrixXd dmat = this->DensityMatrixFull(state);
148  Eigen::Vector3d electronic_dip;
149  for (Index i = 0; i < 3; ++i) {
150  electronic_dip(i) = dmat.cwiseProduct(dipole.Matrix()[i]).sum();
151  }
152  return nuclei_dip - electronic_dip;
153 }

155 Eigen::MatrixXd Orbitals::TransitionDensityMatrix(const QMState& state) const {
156  if (state.Type() != QMStateType::Singlet) {
157  throw std::runtime_error(
158  "Spin type not known for transition density matrix. Available only for "
159  "singlet");
160  }
161  const Eigen::MatrixXd& BSECoefs = _BSE_singlet.eigenvectors();
162  if (BSECoefs.cols() < state.StateIdx() + 1 || BSECoefs.rows() < 2) {
163  throw std::runtime_error("Orbitals object has no information about state:" +
164  state.ToString());
165  }
166 
167  // The Transition dipole is sqrt2 bigger because of the spin, the excited
168  // state is a linear combination of 2 slater determinants, where either alpha
169  // or beta spin electron is excited
170 
171  /*Trying to implement D_{alpha,beta}=
172  * sqrt2*sum_{i}^{occ}sum_{j}^{virt}{BSEcoef(i,j)*MOcoef(alpha,i)*MOcoef(beta,j)}
173  */
174  // c stands for conduction band and thus virtual orbitals
175  // v stand for valence band and thus occupied orbitals
176 
177  Eigen::VectorXd coeffs = BSECoefs.col(state.StateIdx());
178 
179  if (!_useTDA) {
180  coeffs += _BSE_singlet.eigenvectors2().col(state.StateIdx());
181  }
182  coeffs *= std::sqrt(2.0);
183  auto occlevels = _mos.eigenvectors().block(
184  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
185  auto virtlevels = _mos.eigenvectors().block(
186  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
187  Eigen::Map<const Eigen::MatrixXd> mat(coeffs.data(), _bse_ctotal,
188  _bse_vtotal);
189 
190  return occlevels * mat.transpose() * virtlevels.transpose();
191 }

207 std::array<Eigen::MatrixXd, 2> Orbitals::DensityMatrixExcitedState_R(
208  const QMState& state) const {
209  if (!state.Type().isExciton()) {
210  throw std::runtime_error(
211  "Spin type not known for density matrix. Available are singlet and "
212  "triplet");
213  }
214 
215  const Eigen::MatrixXd& BSECoefs = (state.Type() == QMStateType::Singlet)
216  ? _BSE_singlet.eigenvectors()
217  : _BSE_triplet.eigenvectors();
218  if (BSECoefs.cols() < state.StateIdx() + 1 || BSECoefs.rows() < 2) {
219  throw std::runtime_error("Orbitals object has no information about state:" +
220  state.ToString());
221  }
222  /******
223  *
224  * Density matrix for GW-BSE based excitations
225  *
226  * - electron contribution
227  * D_ab = \sum{vc} \sum{c'} A_{vc}A_{vc'} mo_a(c)mo_b(c')
228  *
229  * - hole contribution
230  * D_ab = \sum{vc} \sum{v'} A_{vc}A_{v'c} mo_a(v)mo_b(v')
231  *
232  */
233 
234  Eigen::VectorXd coeffs = BSECoefs.col(state.StateIdx());
235 
236  std::array<Eigen::MatrixXd, 2> dmatEX;
237  // hole part as matrix products
238  Eigen::MatrixXd occlevels = _mos.eigenvectors().block(
239  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
240  dmatEX[0] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
241 
242  // electron part as matrix products
243  Eigen::MatrixXd virtlevels = _mos.eigenvectors().block(
244  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
245  dmatEX[1] = virtlevels * CalcAuxMat_cc(coeffs) * virtlevels.transpose();
246 
247  return dmatEX;
248 }

262 std::array<Eigen::MatrixXd, 2> Orbitals::DensityMatrixExcitedState_AR(
263  const QMState& state) const {
264 
265  if (!state.Type().isExciton()) {
266  throw std::runtime_error(
267  "Spin type not known for density matrix. Available are singlet and "
268  "triplet");
269  }
270 
271  const Eigen::MatrixXd& BSECoefs_AR = (state.Type() == QMStateType::Singlet)
272  ? _BSE_singlet.eigenvectors2()
273  : _BSE_triplet.eigenvectors2();
274  if (BSECoefs_AR.cols() < state.StateIdx() + 1 || BSECoefs_AR.rows() < 2) {
275  throw std::runtime_error("Orbitals object has no information about state:" +
276  state.ToString());
277  }
278  /******
279  *
280  * Density matrix for GW-BSE based excitations
281  *
282  * - electron contribution
283  * D_ab = \sum{vc} \sum{v'} B_{vc}B_{v'c} mo_a(v)mo_b(v')
284  *
285  * - hole contribution
286  * D_ab = \sum{vc} \sum{c'} B_{vc}B_{vc'} mo_a(c)mo_b(c')
287  *
288  *
289  * more efficient:
290  *
291  * - electron contribution
292  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} B_{vc}B_{v'c} ]
293  * = \sum{v} \sum{v'} mo_a(v)mo_b(v') B_{vv'}
294  *
295  * - hole contribution
296  * D_ab = \sum{c} \sum{c'} mo_a(c)mo_b(c') [ \sum{v} B_{vc}B_{vc'} ]
297  * = \sum{c} \sum{c'} mo_a(c)mo_b(c') B_{cc'}
298  *
299  */
300 
301  Eigen::VectorXd coeffs = BSECoefs_AR.col(state.StateIdx());
302 
303  std::array<Eigen::MatrixXd, 2> dmatAR;
304  Eigen::MatrixXd virtlevels = _mos.eigenvectors().block(
305  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
306  dmatAR[0] = virtlevels * CalcAuxMat_cc(coeffs) * virtlevels.transpose();
307  // electron part as matrix products
308  Eigen::MatrixXd occlevels = _mos.eigenvectors().block(
309  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
310  dmatAR[1] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
311 
312  return dmatAR;
313 }

315 Eigen::VectorXd Orbitals::Oscillatorstrengths() const {
316 
317  Index size = Index(_transition_dipoles.size());

338 double Orbitals::getExcitedStateEnergy(const QMState& state) const {
339 
340  double omega = 0.0;
341  if (state.isTransition()) {
342  throw std::runtime_error(
343  "Total Energy does not exist for transition state");
344  }
345 
346  if (state.Type() == QMStateType::Singlet) {
347  if (_BSE_singlet.eigenvalues().size() < state.StateIdx() + 1) {
348  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
349  state.ToString() +
350  " which has not been calculated");
351  }
352  omega = _BSE_singlet.eigenvalues()[state.StateIdx()];
353  } else if (state.Type() == QMStateType::Triplet) {
354  if (_BSE_triplet.eigenvalues().size() < state.StateIdx() + 1) {
355  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
356  state.ToString() +
357  " which has not been calculated");
358  }
359  omega = _BSE_triplet.eigenvalues()[state.StateIdx()];
360  } else if (state.Type() == QMStateType::DQPstate) {
361  if (_QPdiag.eigenvalues().size() < state.StateIdx() + 1 - getGWAmin()) {
362  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
363  state.ToString() +
364  " which has not been calculated");
365  }
366  return _QPdiag.eigenvalues()[state.StateIdx() - getGWAmin()];
367  } else if (state.Type() == QMStateType::KSstate) {
368  if (_mos.eigenvalues().size() < state.StateIdx() + 1) {
369  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
370  state.ToString() +
371  " which has not been calculated");
372  }
373  return _mos.eigenvalues()[state.StateIdx()];
374  } else if (state.Type() == QMStateType::PQPstate) {
375  if (this->_QPpert_energies.rows() < state.StateIdx() + 1 - getGWAmin()) {
376  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
377  state.ToString() +
378  " which has not been calculated");
379  }
380  return _QPpert_energies(state.StateIdx() - getGWAmin(), 3);
381  } else {
382  throw std::runtime_error(
383  "GetTotalEnergy only knows states:singlet,triplet,KS,DQP,PQP");
384  }
385  return omega; // e.g. hartree
386 }

388 std::array<Eigen::MatrixXd, 3> Orbitals::CalcFreeTransition_Dipoles() const {
389  const Eigen::MatrixXd& dft_orbitals = _mos.eigenvectors();
390  AOBasis basis = SetupDftBasis();
391  // Testing electric dipole AOMatrix
392  AODipole dft_dipole;
393  dft_dipole.Fill(basis);
394 
395  // now transition dipole elements for free interlevel transitions
396  std::array<Eigen::MatrixXd, 3> interlevel_dipoles;
397 
398  Eigen::MatrixXd empty =
399  dft_orbitals.block(0, _bse_cmin, basis.AOBasisSize(), _bse_ctotal);
400  Eigen::MatrixXd occ =
401  dft_orbitals.block(0, _bse_vmin, basis.AOBasisSize(), _bse_vtotal);
402  for (Index i = 0; i < 3; i++) {
403  interlevel_dipoles[i] = empty.transpose() * dft_dipole.Matrix()[i] * occ;
404  }
405  return interlevel_dipoles;
406 }

408 void Orbitals::CalcCoupledTransition_Dipoles() {
409  std::array<Eigen::MatrixXd, 3> interlevel_dipoles =
410  CalcFreeTransition_Dipoles();
411  Index numofstates = _BSE_singlet.eigenvalues().size();
412  _transition_dipoles.resize(0);
413  _transition_dipoles.reserve(numofstates);
414  const double sqrt2 = sqrt(2.0);
415  for (Index i_exc = 0; i_exc < numofstates; i_exc++) {
416 
417  Eigen::VectorXd coeffs = _BSE_singlet.eigenvectors().col(i_exc);
418  if (!_useTDA) {
419  coeffs += _BSE_singlet.eigenvectors2().col(i_exc);
420  }
421  Eigen::Map<Eigen::MatrixXd> mat(coeffs.data(), _bse_ctotal, _bse_vtotal);
422  Eigen::Vector3d tdipole = Eigen::Vector3d::Zero();
423  for (Index i = 0; i < 3; i++) {
424  tdipole[i] = mat.cwiseProduct(interlevel_dipoles[i]).sum();
425  }
426  // The Transition dipole is sqrt2 bigger because of the spin, the
427  // excited state is a linear combination of 2 slater determinants,
428  // where either alpha or beta spin electron is excited
429  _transition_dipoles.push_back(-sqrt2 * tdipole); //- because electrons are
430  // negative
431  }
432 }

453 void Orbitals::PrepareDimerGuess(const Orbitals& orbitalsA,
454  const Orbitals& orbitalsB) {
455 
456  // constructing the direct product orbA x orbB
457  Index basisA = orbitalsA.getBasisSetSize();
458  Index basisB = orbitalsB.getBasisSetSize();
459 
460  Index electronsA = orbitalsA.getNumberOfAlphaElectrons();
461  Index electronsB = orbitalsB.getNumberOfAlphaElectrons();
462 
463  _mos.eigenvectors() = Eigen::MatrixXd::Zero(basisA + basisB, basisA + basisB);
464 
465  // AxB = | A 0 | // A = [EA, EB] //
466  // | 0 B | // //
467  if (orbitalsA.getDFTbasisName() != orbitalsB.getDFTbasisName()) {
468  throw std::runtime_error("Basissets of Orbitals A and B differ " +
469  orbitalsA.getDFTbasisName() + ":" +
470  orbitalsB.getDFTbasisName());
471  }
472  this->setDFTbasisName(orbitalsA.getDFTbasisName());
473  if (orbitalsA.getECPName() != orbitalsB.getECPName()) {
474  throw std::runtime_error("ECPs of Orbitals A and B differ " +
475  orbitalsA.getECPName() + ":" +
476  orbitalsB.getECPName());
477  }
478  this->setECPName(orbitalsA.getECPName());
479  this->setBasisSetSize(basisA + basisB);
480  this->setNumberOfOccupiedLevels(electronsA + electronsB);
481  this->setNumberOfAlphaElectrons(electronsA + electronsB);
482 
483  _mos.eigenvectors().topLeftCorner(basisA, basisA) =
484  orbitalsA.MOs().eigenvectors();
485  _mos.eigenvectors().bottomRightCorner(basisB, basisB) =
486  orbitalsB.MOs().eigenvectors();
487 
488  _mos.eigenvalues().resize(basisA + basisB);
489 
490  _mos.eigenvalues().head(basisA) = orbitalsA.MOs().eigenvalues();
491  _mos.eigenvalues().tail(basisB) = orbitalsB.MOs().eigenvalues();
492 
493  OrderMOsbyEnergy();
494 
495  return;
496 }
