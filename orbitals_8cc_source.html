44 std::vector<Index> Orbitals::CheckDegeneracy(Index level,
45  double energy_difference) const {
46 
47  std::vector<Index> result;
48  if (level > _mos.eigenvalues().size()) {
49  throw std::runtime_error(
50  "Level for degeneracy is higher than maximum level");
51  }
52  double MOEnergyLevel = _mos.eigenvalues()(level);
53 
54  for (Index i = 0; i < _mos.eigenvalues().size(); ++i) {
55  if (std::abs(_mos.eigenvalues()(i) - MOEnergyLevel) < energy_difference) {
56  result.push_back(i);
57  }
58  }
59 
60  if (result.empty()) {
61  result.push_back(level);
62  }
63  return result;
64 }

66 std::vector<Index> Orbitals::SortEnergies() {
67  std::vector<Index> index = std::vector<Index>(_mos.eigenvalues().size());

71  });
72  return index;

75 Eigen::MatrixXd Orbitals::DensityMatrixFull(const QMState& state) const {
76  if (state.isTransition()) {
77  return this->TransitionDensityMatrix(state);
78  }
79  Eigen::MatrixXd result = this->DensityMatrixGroundState();
80  if (state.Type().isExciton()) {
81  std::array<Eigen::MatrixXd, 2> DMAT = DensityMatrixExcitedState(state);
82  result = result - DMAT[0] + DMAT[1]; // Ground state + hole_contribution +
83  // electron contribution
84  } else if (state.Type() == QMStateType::DQPstate) {
85  Eigen::MatrixXd DMATQP = DensityMatrixQuasiParticle(state);
86  if (state.StateIdx() > getHomo()) {
87  result += DMATQP;
88  } else {
89  result -= DMATQP;
90  }
91  } else if (state.Type() != QMStateType::Gstate) {
92  throw std::runtime_error(
93  "DensityMatrixFull does not yet implement QMStateType:" +
94  state.Type().ToLongString());
95  }
96  return result;
97 }

121 Eigen::MatrixXd Orbitals::DensityMatrixQuasiParticle(
122  const QMState& state) const {
123  if (state.Type() != QMStateType::DQPstate) {

125  " is not a quasiparticle state");

133 Eigen::Vector3d Orbitals::CalcElDipole(const QMState& state) const {
134  Eigen::Vector3d nuclei_dip = Eigen::Vector3d::Zero();
135  if (!state.isTransition()) {
136  for (const QMAtom& atom : _atoms) {
137  nuclei_dip += (atom.getPos() - _atoms.getPos()) * atom.getNuccharge();
138  }
139  }
140  AOBasis basis = SetupDftBasis();
141  AODipole dipole;
142  dipole.setCenter(_atoms.getPos());
143  dipole.Fill(basis);
144 
145  Eigen::MatrixXd dmat = this->DensityMatrixFull(state);
146  Eigen::Vector3d electronic_dip;
147  for (Index i = 0; i < 3; ++i) {
148  electronic_dip(i) = dmat.cwiseProduct(dipole.Matrix()[i]).sum();
149  }
150  return nuclei_dip - electronic_dip;
151 }

153 Eigen::MatrixXd Orbitals::TransitionDensityMatrix(const QMState& state) const {
154  if (state.Type() != QMStateType::Singlet) {
155  throw std::runtime_error(
156  "Spin type not known for transition density matrix. Available only for "
157  "singlet");
158  }
159  const Eigen::MatrixXd& BSECoefs = _BSE_singlet.eigenvectors();
160  if (BSECoefs.cols() < state.StateIdx() + 1 || BSECoefs.rows() < 2) {
161  throw std::runtime_error("Orbitals object has no information about state:" +
162  state.ToString());
163  }
164 
165  // The Transition dipole is sqrt2 bigger because of the spin, the excited
166  // state is a linear combination of 2 slater determinants, where either alpha
167  // or beta spin electron is excited
168 
169  /*Trying to implement D_{alpha,beta}=
170  * sqrt2*sum_{i}^{occ}sum_{j}^{virt}{BSEcoef(i,j)*MOcoef(alpha,i)*MOcoef(beta,j)}
171  */
172  // c stands for conduction band and thus virtual orbitals
173  // v stand for valence band and thus occupied orbitals
174 
175  Eigen::VectorXd coeffs = BSECoefs.col(state.StateIdx());
176 
177  if (!_useTDA) {
178  coeffs += _BSE_singlet.eigenvectors2().col(state.StateIdx());
179  }
180  coeffs *= std::sqrt(2.0);
181  auto occlevels = _mos.eigenvectors().block(
182  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
183  auto virtlevels = _mos.eigenvectors().block(
184  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
185  Eigen::Map<const Eigen::MatrixXd> mat(coeffs.data(), _bse_ctotal,
186  _bse_vtotal);
187 
188  return occlevels * mat.transpose() * virtlevels.transpose();
189 }

205 std::array<Eigen::MatrixXd, 2> Orbitals::DensityMatrixExcitedState_R(
206  const QMState& state) const {
207  if (!state.Type().isExciton()) {
208  throw std::runtime_error(
209  "Spin type not known for density matrix. Available are singlet and "
210  "triplet");
211  }
212 
213  const Eigen::MatrixXd& BSECoefs = (state.Type() == QMStateType::Singlet)
214  ? _BSE_singlet.eigenvectors()
215  : _BSE_triplet.eigenvectors();
216  if (BSECoefs.cols() < state.StateIdx() + 1 || BSECoefs.rows() < 2) {
217  throw std::runtime_error("Orbitals object has no information about state:" +
218  state.ToString());
219  }
220  /******
221  *
222  * Density matrix for GW-BSE based excitations
223  *
224  * - electron contribution
225  * D_ab = \sum{vc} \sum{c'} A_{vc}A_{vc'} mo_a(c)mo_b(c')
226  *
227  * - hole contribution
228  * D_ab = \sum{vc} \sum{v'} A_{vc}A_{v'c} mo_a(v)mo_b(v')
229  *
230  */
231 
232  Eigen::VectorXd coeffs = BSECoefs.col(state.StateIdx());
233 
234  std::array<Eigen::MatrixXd, 2> dmatEX;
235  // hole part as matrix products
236  Eigen::MatrixXd occlevels = _mos.eigenvectors().block(
237  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
238  dmatEX[0] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
239 
240  // electron part as matrix products
241  Eigen::MatrixXd virtlevels = _mos.eigenvectors().block(
242  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
243  dmatEX[1] = virtlevels * CalcAuxMat_cc(coeffs) * virtlevels.transpose();
244 
245  return dmatEX;
246 }

260 std::array<Eigen::MatrixXd, 2> Orbitals::DensityMatrixExcitedState_AR(
261  const QMState& state) const {
262 
263  if (!state.Type().isExciton()) {
264  throw std::runtime_error(
265  "Spin type not known for density matrix. Available are singlet and "
266  "triplet");
267  }
268 
269  const Eigen::MatrixXd& BSECoefs_AR = (state.Type() == QMStateType::Singlet)
270  ? _BSE_singlet.eigenvectors2()
271  : _BSE_triplet.eigenvectors2();
272  if (BSECoefs_AR.cols() < state.StateIdx() + 1 || BSECoefs_AR.rows() < 2) {
273  throw std::runtime_error("Orbitals object has no information about state:" +
274  state.ToString());
275  }
276  /******
277  *
278  * Density matrix for GW-BSE based excitations
279  *
280  * - electron contribution
281  * D_ab = \sum{vc} \sum{v'} B_{vc}B_{v'c} mo_a(v)mo_b(v')
282  *
283  * - hole contribution
284  * D_ab = \sum{vc} \sum{c'} B_{vc}B_{vc'} mo_a(c)mo_b(c')
285  *
286  *
287  * more efficient:
288  *
289  * - electron contribution
290  * D_ab = \sum{v} \sum{v'} mo_a(v)mo_b(v') [ \sum{c} B_{vc}B_{v'c} ]
291  * = \sum{v} \sum{v'} mo_a(v)mo_b(v') B_{vv'}
292  *
293  * - hole contribution
294  * D_ab = \sum{c} \sum{c'} mo_a(c)mo_b(c') [ \sum{v} B_{vc}B_{vc'} ]
295  * = \sum{c} \sum{c'} mo_a(c)mo_b(c') B_{cc'}
296  *
297  */
298 
299  Eigen::VectorXd coeffs = BSECoefs_AR.col(state.StateIdx());
300 
301  std::array<Eigen::MatrixXd, 2> dmatAR;
302  Eigen::MatrixXd virtlevels = _mos.eigenvectors().block(
303  0, _bse_cmin, _mos.eigenvectors().rows(), _bse_ctotal);
304  dmatAR[0] = virtlevels * CalcAuxMat_cc(coeffs) * virtlevels.transpose();
305  // electron part as matrix products
306  Eigen::MatrixXd occlevels = _mos.eigenvectors().block(
307  0, _bse_vmin, _mos.eigenvectors().rows(), _bse_vtotal);
308  dmatAR[1] = occlevels * CalcAuxMat_vv(coeffs) * occlevels.transpose();
309 
310  return dmatAR;
311 }

313 Eigen::VectorXd Orbitals::Oscillatorstrengths() const {
314 
315  Index size = Index(_transition_dipoles.size());
316  if (size > _BSE_singlet.eigenvalues().size()) {
317  size = _BSE_singlet.eigenvalues().size();
318  }
319  Eigen::VectorXd oscs = Eigen::VectorXd::Zero(size);
320  for (Index i = 0; i < size; ++i) {
321  oscs(i) = _transition_dipoles[i].squaredNorm() * 2.0 / 3.0 *

336 double Orbitals::getExcitedStateEnergy(const QMState& state) const {
337 
338  double omega = 0.0;
339  if (state.isTransition()) {
340  throw std::runtime_error(
341  "Total Energy does not exist for transition state");
342  }
343 
344  if (state.Type() == QMStateType::Singlet) {
345  if (_BSE_singlet.eigenvalues().size() < state.StateIdx() + 1) {
346  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
347  state.ToString() +
348  " which has not been calculated");
349  }
350  omega = _BSE_singlet.eigenvalues()[state.StateIdx()];
351  } else if (state.Type() == QMStateType::Triplet) {
352  if (_BSE_triplet.eigenvalues().size() < state.StateIdx() + 1) {
353  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
354  state.ToString() +
355  " which has not been calculated");
356  }
357  omega = _BSE_triplet.eigenvalues()[state.StateIdx()];
358  } else if (state.Type() == QMStateType::DQPstate) {
359  if (_QPdiag.eigenvalues().size() < state.StateIdx() + 1 - getGWAmin()) {
360  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
361  state.ToString() +
362  " which has not been calculated");
363  }
364  return _QPdiag.eigenvalues()[state.StateIdx() - getGWAmin()];
365  } else if (state.Type() == QMStateType::KSstate) {
366  if (_mos.eigenvalues().size() < state.StateIdx() + 1) {
367  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
368  state.ToString() +
369  " which has not been calculated");
370  }
371  return _mos.eigenvalues()[state.StateIdx()];
372  } else if (state.Type() == QMStateType::PQPstate) {
373  if (this->_QPpert_energies.rows() < state.StateIdx() + 1 - getGWAmin()) {
374  throw std::runtime_error("Orbitals::getTotalEnergy You want " +
375  state.ToString() +
376  " which has not been calculated");
377  }
378  return _QPpert_energies(state.StateIdx() - getGWAmin(), 3);
379  } else {
380  throw std::runtime_error(
381  "GetTotalEnergy only knows states:singlet,triplet,KS,DQP,PQP");
382  }
383  return omega; // e.g. hartree
384 }

386 std::array<Eigen::MatrixXd, 3> Orbitals::CalcFreeTransition_Dipoles() const {
387  const Eigen::MatrixXd& dft_orbitals = _mos.eigenvectors();
388  AOBasis basis = SetupDftBasis();
389  // Testing electric dipole AOMatrix
390  AODipole dft_dipole;
391  dft_dipole.Fill(basis);
392 
393  // now transition dipole elements for free interlevel transitions
394  std::array<Eigen::MatrixXd, 3> interlevel_dipoles;
395 
396  Eigen::MatrixXd empty =
397  dft_orbitals.block(0, _bse_cmin, basis.AOBasisSize(), _bse_ctotal);
398  Eigen::MatrixXd occ =
399  dft_orbitals.block(0, _bse_vmin, basis.AOBasisSize(), _bse_vtotal);
400  for (Index i = 0; i < 3; i++) {
401  interlevel_dipoles[i] = empty.transpose() * dft_dipole.Matrix()[i] * occ;
402  }
403  return interlevel_dipoles;
404 }

406 void Orbitals::CalcCoupledTransition_Dipoles() {
407  std::array<Eigen::MatrixXd, 3> interlevel_dipoles =
408  CalcFreeTransition_Dipoles();
409  Index numofstates = _BSE_singlet.eigenvalues().size();
410  _transition_dipoles.resize(0);
411  _transition_dipoles.reserve(numofstates);
412  const double sqrt2 = sqrt(2.0);
413  for (Index i_exc = 0; i_exc < numofstates; i_exc++) {
414 
415  Eigen::VectorXd coeffs = _BSE_singlet.eigenvectors().col(i_exc);
416  if (!_useTDA) {
417  coeffs += _BSE_singlet.eigenvectors2().col(i_exc);
418  }
419  Eigen::Map<Eigen::MatrixXd> mat(coeffs.data(), _bse_ctotal, _bse_vtotal);
420  Eigen::Vector3d tdipole = Eigen::Vector3d::Zero();
421  for (Index i = 0; i < 3; i++) {
422  tdipole[i] = mat.cwiseProduct(interlevel_dipoles[i]).sum();
423  }
424  // The Transition dipole is sqrt2 bigger because of the spin, the
425  // excited state is a linear combination of 2 slater determinants,
426  // where either alpha or beta spin electron is excited
427  _transition_dipoles.push_back(-sqrt2 * tdipole); //- because electrons are
428  // negative
429  }
430 }

451 void Orbitals::PrepareDimerGuess(const Orbitals& orbitalsA,
452  const Orbitals& orbitalsB) {
453 
454  // constructing the direct product orbA x orbB
455  Index basisA = orbitalsA.getBasisSetSize();
456  Index basisB = orbitalsB.getBasisSetSize();
457 
458  Index electronsA = orbitalsA.getNumberOfAlphaElectrons();
459  Index electronsB = orbitalsB.getNumberOfAlphaElectrons();
460 
461  _mos.eigenvectors() = Eigen::MatrixXd::Zero(basisA + basisB, basisA + basisB);
462 
463  // AxB = | A 0 | // A = [EA, EB] //
464  // | 0 B | // //
465  if (orbitalsA.getDFTbasisName() != orbitalsB.getDFTbasisName()) {
466  throw std::runtime_error("Basissets of Orbitals A and B differ " +
467  orbitalsA.getDFTbasisName() + ":" +
468  orbitalsB.getDFTbasisName());
469  }
470  this->setDFTbasisName(orbitalsA.getDFTbasisName());
471  if (orbitalsA.getECPName() != orbitalsB.getECPName()) {
472  throw std::runtime_error("ECPs of Orbitals A and B differ " +
473  orbitalsA.getECPName() + ":" +
474  orbitalsB.getECPName());
475  }
476  this->setECPName(orbitalsA.getECPName());
477  this->setBasisSetSize(basisA + basisB);
478  this->setNumberOfOccupiedLevels(electronsA + electronsB);
479  this->setNumberOfAlphaElectrons(electronsA + electronsB);
480 
481  _mos.eigenvectors().topLeftCorner(basisA, basisA) =
482  orbitalsA.MOs().eigenvectors();
483  _mos.eigenvectors().bottomRightCorner(basisB, basisB) =
484  orbitalsB.MOs().eigenvectors();
485 
486  _mos.eigenvalues().resize(basisA + basisB);
487 
488  _mos.eigenvalues().head(basisA) = orbitalsA.MOs().eigenvalues();
489  _mos.eigenvalues().tail(basisB) = orbitalsB.MOs().eigenvalues();
490 
491  OrderMOsbyEnergy();
492 
493  return;
494 }
