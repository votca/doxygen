143 bool Orbitals::CheckDegeneracy( double _energy_difference ) {
144 
145  ub::vector<double>::iterator it1 = _mo_energies.begin();
146  bool _degenerate = false;
147 
148  if ( tools::globals::verbose ) cout << endl <<"... ... Checking level degeneracy " << endl;
149 
150  _level_degeneracy.clear();
151 
152  while ( it1 !=_mo_energies.end() ) {
153 
154  // in all containers counters start with 0; real life - with 1
155  int _level1 = std::distance(_mo_energies.begin(), it1) + 1;
156 
157  // add the level itself - it is easier to loo over all levels later
158  _level_degeneracy[_level1].push_back( _level1 );
159 
160  ub::vector<double>::iterator it2 = it1;
161  it2++;
162 
163  while ( it2 !=_mo_energies.end() ) {
164  //cout << _level1 << ":" << *it1 << ":" << *it2 << endl;
165  double energy1 = *it1;
166  double energy2 = *it2;
167 
168  // in all containers counters start with 0; real life - with 1
169  int _level2 = std::distance(_mo_energies.begin(), it2) + 1;
170 
171  if ( abs(energy1 - energy2)*_conv_Hrt_eV < _energy_difference ) {
172  _level_degeneracy[_level1].push_back( _level2 );
173  _level_degeneracy[_level2].push_back( _level1 );
174  _degenerate = true;
175  }
176  it2++;
177  }
178  it1++;
179  }
180 
181  if ( tools::globals::verbose ){
182 
183  if ( _degenerate ) {
184  cout << "... ... Some levels are degenerate" << endl;
185  for (std::map<int, std::vector<int> >::iterator it = _level_degeneracy.begin();
186  it != _level_degeneracy.end();
187  ++it) {
188  // output only degenerate levels
189  if ( (it->second).size() > 1 ) {
190  std::cout << "... ... level "<< it->first << " : ";
191  for (vector<int>::iterator lev = (it->second).begin(); lev != (it->second).end(); lev++)
192  cout << *lev << " ";
193  cout << endl;
194  }
195  }
196  } else {
197  cout << "... ... No degeneracy found" << endl;
198  }
199 
200  cout << "... ... Done checking level degeneracy" << endl;
201 
202  }
203 
204  _has_level_degeneracy = true;
205  return _degenerate;
206 
207 }
