110 bool ExcitonCoupling::Evaluate() {
111  Property *_job_output=NULL;
112  Property _summary;
113  _log.setReportLevel( ctp::logDEBUG );
114  _log.setMultithreading( true );
115 
116  _log.setPreface( ctp::logINFO, "\n... ...");
117  _log.setPreface( ctp::logERROR, "\n... ...");
118  _log.setPreface( ctp::logWARNING, "\n... ...");
119  _log.setPreface( ctp::logDEBUG, "\n... ...");
120 
121  // get the corresponding object from the QMPackageFactory
122  if(!_classical){
123  Orbitals _orbitalsA, _orbitalsB, _orbitalsAB;
124  // load the QM data from serialized orbitals objects
125 
126  std::ifstream ifa( (_orbA ).c_str());
127  LOG( ctp::logDEBUG, _log) << " Loading QM data for molecule A from " << _orbA << flush;
128  boost::archive::binary_iarchive ia(ifa);
129  ia >> _orbitalsA;
130  ifa.close();
131 
132  std::ifstream ifb( (_orbB ).c_str());
133  LOG( ctp::logDEBUG, _log) << " Loading QM data for molecule B from " << _orbB << flush;
134  boost::archive::binary_iarchive ib(ifb);
135  ib >> _orbitalsB;
136  ifb.close();
137 
138 
139  std::ifstream ifab( (_orbAB ).c_str());
140  LOG( ctp::logDEBUG, _log) << " Loading QM data for dimer AB from " << _orbAB << flush;
141  boost::archive::binary_iarchive iab(ifab);
142  iab >> _orbitalsAB;
143  ifab.close();
144 
145  BSECoupling _bsecoupling;
146  _bsecoupling.setLogger(&_log);
147  _bsecoupling.Initialize(&_coupling_options);
148  //bool _doSinglets=_bsecoupling.get_doSinglets();
149  //bool _dotriplets=_bsecoupling.get_doTriplets();
150 
151 
152  //bool _calculate_integrals = _bsecoupling.CalculateCouplings_OLD( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB_singlet );
153  bool _calculate_integrals = _bsecoupling.CalculateCouplings( &_orbitalsA, &_orbitalsB, &_orbitalsAB );
154  std::cout << _log;
155 
156  if ( _calculate_integrals ){
157  // output the results
158 
159  _job_output = &_summary.add("output","");
160  Property *_pair_summary = &_job_output->add("pair","");
161  Property *_type_summary = &_pair_summary->add("type","");
162  _bsecoupling.addoutput(_type_summary,&_orbitalsA, & _orbitalsB);
163 
164 
165 
166  }
167  }
168 
169  else if (_classical){
170  LOG( ctp::logDEBUG, _log) << "Calculating electronic coupling using classical transition charges." << _orbB << flush;
171  std::vector< ctp::APolarSite*> seg1= ctp::APS_FROM_MPS(_mpsA, 0);
172  std::vector< ctp::APolarSite*> seg2= ctp::APS_FROM_MPS(_mpsB, 0);
173 
174  ctp::PolarSeg* Seg1 = new ctp::PolarSeg(1,seg1);
175  ctp::PolarSeg* Seg2 = new ctp::PolarSeg(2,seg2);
176  ctp::XInteractor actor;
177  actor.ResetEnergy();
178  vec s = vec(0,0,0);
179 
180  //LOG(logINFO, *pLog) << "Evaluate pair for debugging " << Seg1->getId() << ":" <<Seg2->getId() << " Distance "<< abs(s) << flush;
181  ctp::PolarSeg::iterator pit1;
182  ctp::PolarSeg::iterator pit2;
183  double E = 0.0;
184  for (pit1 = Seg1->begin(); pit1 < Seg1->end(); ++pit1) {
185  for (pit2 = Seg2->begin(); pit2 < Seg2->end(); ++pit2) {
186 
187  actor.BiasIndu(*(*pit1), *(*pit2), s);
188  (*pit1)->Depolarize();
189  (*pit2)->Depolarize();
190 
191  E += actor.E_f(*(*pit1), *(*pit2));
192 
193 
194  }
195  }
196 
197  double J=E*conv::int2eV;
198 
199 
200  _job_output = &_summary.add("output","");
201  Property *_pair_summary = &_job_output->add("pair","");
202  _pair_summary->setAttribute("idA", 1);
203  _pair_summary->setAttribute("idB", 2);
204  _pair_summary->setAttribute("typeA", _mpsA);
205  _pair_summary->setAttribute("typeB", _mpsB);
206  Property *_coupling_summary = &_pair_summary->add("Coupling","");
207  _coupling_summary->setAttribute("jABstatic", J);
208  }
209 
210  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
211 
212  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
213  ofs << *_job_output;
214  ofs.close();
215  return true;
216 }
