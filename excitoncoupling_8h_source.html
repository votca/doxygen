81 bool ExcitonCoupling::Evaluate() {
82  OPENMP::setMaxThreads(_nThreads);
83  _log.setReportLevel(Log::current_level);
84  _log.setMultithreading(true);
85 
86  _log.setCommonPreface("\n... ...");
87  tools::Property summary;
88  tools::Property& job_output = summary.add("output", "");
89  // get the corresponding object from the QMPackageFactory
90  if (!_classical) {
91  Orbitals orbitalsA, orbitalsB, orbitalsAB;
92  // load the QM data from serialized orbitals objects
93 
94  XTP_LOG(Log::error, _log)
95  << " Loading QM data for molecule A from " << _orbA << std::flush;
96  orbitalsA.ReadFromCpt(_orbA);
97 
98  XTP_LOG(Log::error, _log)
99  << " Loading QM data for molecule B from " << _orbB << std::flush;
100  orbitalsB.ReadFromCpt(_orbB);
101 
102  XTP_LOG(Log::error, _log)
103  << " Loading QM data for dimer AB from " << _orbAB << std::flush;
104  orbitalsAB.ReadFromCpt(_orbAB);
105 
106  BSECoupling bsecoupling;
107  bsecoupling.setLogger(&_log);
108  bsecoupling.Initialize(_coupling_options);
109 
110  bsecoupling.CalculateCouplings(orbitalsA, orbitalsB, orbitalsAB);
111  std::cout << _log;
112 
113  tools::Property& pair_summary = job_output.add("pair", "");
114  tools::Property& type_summary = pair_summary.add("type", "");
115  bsecoupling.Addoutput(type_summary, orbitalsA, orbitalsB);
116 
117  }
118 
119  else if (_classical) {
120  XTP_LOG(Log::error, _log)
121  << "Calculating electronic coupling using classical transition charges."
122  << _orbB << std::flush;
123  PolarSegment seg1 = PolarSegment("A", 0);
124  PolarSegment seg2 = PolarSegment("B", 1);
125  seg1.LoadFromFile(_mpsA);
126  seg2.LoadFromFile(_mpsB);
127  eeInteractor ee;
128  double J = ee.CalcStaticEnergy(seg1, seg2);
129 
130  tools::Property& pair_summary = job_output.add("pair", "");
131  pair_summary.setAttribute("idA", 1);
132  pair_summary.setAttribute("idB", 2);
133  pair_summary.setAttribute("typeA", _mpsA);
134  pair_summary.setAttribute("typeB", _mpsB);
135  tools::Property& coupling_summary = pair_summary.add("Coupling", "");
136  coupling_summary.setAttribute("jABstatic", J);
137  }
138 
139  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
140 
141  std::ofstream ofs(_output_file, std::ofstream::out);
142  ofs << job_output;
143  ofs.close();
144  return true;
145 }
