108 bool ExcitonCoupling::Evaluate() {
109 
110  _log.setReportLevel( ctp::logDEBUG );
111  _log.setMultithreading( true );
112 
113  _log.setPreface( ctp::logINFO, "\n... ...");
114  _log.setPreface( ctp::logERROR, "\n... ...");
115  _log.setPreface( ctp::logWARNING, "\n... ...");
116  _log.setPreface( ctp::logDEBUG, "\n... ...");
117  Property summary;
118  Property& job_output = summary.add("output","");
119  // get the corresponding object from the QMPackageFactory
120  if(!_classical){
121  Orbitals orbitalsA, orbitalsB, orbitalsAB;
122  // load the QM data from serialized orbitals objects
123 
124  CTP_LOG( ctp::logDEBUG, _log) << " Loading QM data for molecule A from " << _orbA << flush;
125  orbitalsA.ReadFromCpt(_orbA);
126 
127  CTP_LOG( ctp::logDEBUG, _log) << " Loading QM data for molecule B from " << _orbB << flush;
128  orbitalsB.ReadFromCpt(_orbB);
129 
130  CTP_LOG( ctp::logDEBUG, _log) << " Loading QM data for dimer AB from " << _orbAB << flush;
131  orbitalsAB.ReadFromCpt(_orbAB);
132 
133  BSECoupling bsecoupling;
134  bsecoupling.setLogger(&_log);
135  bsecoupling.Initialize(_coupling_options);
136 
137  bsecoupling.CalculateCouplings( orbitalsA,orbitalsB, orbitalsAB );
138  std::cout << _log;
139 
140  Property& pair_summary = job_output.add("pair","");
141  Property& type_summary = pair_summary.add("type","");
142  bsecoupling.Addoutput(type_summary,orbitalsA, orbitalsB);
143 
144  }
145 
146  else if (_classical){
147  CTP_LOG( ctp::logDEBUG, _log) << "Calculating electronic coupling using classical transition charges." << _orbB << flush;
148  std::vector< ctp::APolarSite*> seg1= ctp::APS_FROM_MPS(_mpsA, 0);
149  std::vector< ctp::APolarSite*> seg2= ctp::APS_FROM_MPS(_mpsB, 0);
150 
151  ctp::PolarSeg Seg1 = ctp::PolarSeg(1,seg1);
152  ctp::PolarSeg Seg2 = ctp::PolarSeg(2,seg2);
153  ctp::XInteractor actor;
154  actor.ResetEnergy();
155  vec s = vec(0,0,0);
156  double E = 0.0;
157  for (ctp::APolarSite* site1:Seg1) {
158  for (ctp::APolarSite* site2:Seg2) {
159  actor.BiasIndu(*site1, *site2, s);
160  site1->Depolarize();
161  site2->Depolarize();
162  E += actor.E_f(*site1, *site2);
163  }
164  }
165 
166  double J=E*conv::int2eV;
167 
168  Property &pair_summary = job_output.add("pair","");
169  pair_summary.setAttribute("idA", 1);
170  pair_summary.setAttribute("idB", 2);
171  pair_summary.setAttribute("typeA", _mpsA);
172  pair_summary.setAttribute("typeB", _mpsB);
173  Property & coupling_summary =pair_summary.add("Coupling","");
174  coupling_summary.setAttribute("jABstatic", J);
175  }
176 
177  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
178 
179  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
180  ofs << job_output;
181  ofs.close();
182  return true;
183 }
