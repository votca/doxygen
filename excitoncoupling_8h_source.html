56 void ExcitonCoupling::Initialize(tools::Property& options) {
57 
58  std::string key = "options." + Identify();

64  _classical = false;

66 
67  if (!_classical) {
68

74  _orbB = options.get(key + ".orbitalsB").as<std::string>();
75  _orbAB = options.get(key + ".orbitalsAB").as<std::string>();

77  } else {
78  _mpsA = options.get(key + ".mpsA").as<std::string>();
79  _mpsB = options.get(key + ".mpsB").as<std::string>();
80  }
81  _output_file = options.get(key + ".output").as<std::string>();

83  // get the path to the shared folders with xml files

89 bool ExcitonCoupling::Evaluate() {
90  OPENMP::setMaxThreads(_nThreads);
91  _log.setReportLevel(logDEBUG);
92  _log.setMultithreading(true);
93 
94  _log.setPreface(logINFO, "\n... ...");
95  _log.setPreface(logERROR, "\n... ...");
96  _log.setPreface(logWARNING, "\n... ...");
97  _log.setPreface(logDEBUG, "\n... ...");
98  tools::Property summary;
99  tools::Property& job_output = summary.add("output", "");
100  // get the corresponding object from the QMPackageFactory
101  if (!_classical) {
102  Orbitals orbitalsA, orbitalsB, orbitalsAB;
103  // load the QM data from serialized orbitals objects
104 
105  XTP_LOG_SAVE(logDEBUG, _log)
106  << " Loading QM data for molecule A from " << _orbA << std::flush;
107  orbitalsA.ReadFromCpt(_orbA);
108 
109  XTP_LOG_SAVE(logDEBUG, _log)
110  << " Loading QM data for molecule B from " << _orbB << std::flush;
111  orbitalsB.ReadFromCpt(_orbB);
112 
113  XTP_LOG_SAVE(logDEBUG, _log)
114  << " Loading QM data for dimer AB from " << _orbAB << std::flush;
115  orbitalsAB.ReadFromCpt(_orbAB);
116 
117  BSECoupling bsecoupling;
118  bsecoupling.setLogger(&_log);
119  bsecoupling.Initialize(_coupling_options);
120 
121  bsecoupling.CalculateCouplings(orbitalsA, orbitalsB, orbitalsAB);
122  std::cout << _log;
123 
124  tools::Property& pair_summary = job_output.add("pair", "");
125  tools::Property& type_summary = pair_summary.add("type", "");
126  bsecoupling.Addoutput(type_summary, orbitalsA, orbitalsB);
127 
128  }
129 
130  else if (_classical) {
131  XTP_LOG_SAVE(logDEBUG, _log)
132  << "Calculating electronic coupling using classical transition charges."
133  << _orbB << std::flush;
134  PolarSegment seg1 = PolarSegment("A", 0);
135  PolarSegment seg2 = PolarSegment("B", 1);
136  seg1.LoadFromFile(_mpsA);
137  seg2.LoadFromFile(_mpsB);
138  eeInteractor ee;
139  double J = ee.CalcStaticEnergy(seg1, seg2);
140 
141  tools::Property& pair_summary = job_output.add("pair", "");
142  pair_summary.setAttribute("idA", 1);
143  pair_summary.setAttribute("idB", 2);
144  pair_summary.setAttribute("typeA", _mpsA);
145  pair_summary.setAttribute("typeB", _mpsB);
146  tools::Property& coupling_summary = pair_summary.add("Coupling", "");
147  coupling_summary.setAttribute("jABstatic", J);
148  }
149 
150  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
151 
152  std::ofstream ofs(_output_file, std::ofstream::out);
153  ofs << job_output;
154  ofs.close();
155  return true;
156 }
