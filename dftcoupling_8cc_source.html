36 double DFTcoupling::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
37  Orbitals* _orbitalsB, Eigen::MatrixXd* _JAB, double _energy_difference ) {
38 
39 
40  int _levelsA = _orbitalsA->getNumberOfLevels();
41 
42  if ( _energy_difference != 0 ) {
43  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
44  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
45 
46  double _JAB_sq = 0; double _JAB_one_level;
47 
48  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
49  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
50  _JAB_one_level = (*_JAB)( *iA - 1 , *iB -1 + _levelsA );
51  _JAB_sq += _JAB_one_level*_JAB_one_level ;
52  }
53  }
54 
55  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
56 
57  } else {
58 
59  return (*_JAB)( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
60 
61  }
62  // the matrix should be symmetric, could also return this element
63  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );
64 }

77 bool DFTcoupling::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
78  Orbitals* _orbitalsAB, Eigen::MatrixXd* _JAB) {
79 
80  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating electronic couplings" << flush;
81 
82  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
83  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
84  const std::vector<QMAtom*> atomsAll = _orbitalsAB->QMAtoms();
85 
86  for (unsigned i = 0; i < atomsAll.size(); i++) {
87  QMAtom* dimer = atomsAll[i];
88  QMAtom* monomer = NULL;
89 
90  if (i < atomsA.size()) {
91  monomer = atomsA[i];
92  } else if (i < atomsB.size() + atomsA.size()) {
93  monomer = atomsB[i - atomsA.size()];
94  } else {
95  // Linker
96  CTP_LOG(ctp::logERROR, *_pLog) << (format("Neither Monomer A nor Monomer B contains atom %s on line %u. Hence, this atom is part of a linker.") %dimer->getType() %(i+1) ).str()<<flush;
97  continue;
98  }
99 
100  if(!monomer->getPos().isClose(dimer->getPos(), 0.001)){
101  CTP_LOG(ctp::logINFO, *_pLog) << "======WARNING=======\n Coordinates of monomer and dimer atoms do not agree, do you know what you are doing?" << flush;
102  }
103 
104  if (monomer->getType() != dimer->getType()) {
105  throw std::runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
106  }
107 }
108 
109  // constructing the direct product orbA x orbB
110  int _basisA = _orbitalsA->getBasisSetSize();
111  int _basisB = _orbitalsB->getBasisSetSize();
112 
113  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
114  CTP_LOG(ctp::logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
115  return false;
116  }
117 
118  int _levelsA = _orbitalsA->getNumberOfLevels();
119  int _levelsB = _orbitalsB->getNumberOfLevels();
120 
121  //boost::timer t; // start timing
122  //double _st = t.elapsed();
123 
124  CTP_LOG(ctp::logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
125  << " B[" << _levelsB << ":" << _basisB << "]" << flush;
126 
127  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
128  CTP_LOG(ctp::logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
129  return false;
130  }
131 
132  // | Orbitals_A 0 | | Overlap_A |
133  // | 0 Orbitals_B |.T X | Overlap_B | X ( Orbitals_AB )
134 
135 
136  Eigen::MatrixXd _psi_AxB=Eigen::MatrixXd::Zero( _basisA + _basisB, _levelsA + _levelsB );
137 
138  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing direct product AxB ["
139  << _psi_AxB.rows() << "x"
140  << _psi_AxB.cols() << "]" << flush;
141 
142  // constructing merged orbitals
143  _psi_AxB.block(0,0, _basisA,_levelsA) = _orbitalsA->MOCoefficients();
144  _psi_AxB.block(_basisA,_levelsA, _basisB,_levelsB) =_orbitalsB->MOCoefficients();
145 
146  Eigen::MatrixXd overlap;
147  if ( _orbitalsAB->hasAOOverlap() ) {
148  CTP_LOG(ctp::logDEBUG,*_pLog) << "Reading overlap matrix from orbitals" << flush;
149  overlap= _orbitalsAB->AOOverlap();
150  }else{
151  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating overlap matrix for basisset: "<< _orbitalsAB->getDFTbasis()<< flush;
152  BasisSet _dftbasisset;
153  AOBasis _dftbasis;
154  _dftbasisset.LoadBasisSet(_orbitalsAB->getDFTbasis());
155 
156  _dftbasis.AOBasisFill(&_dftbasisset, _orbitalsAB->QMAtoms());
157  AOOverlap _dftAOoverlap;
158  _dftAOoverlap.Fill(_dftbasis);
159  overlap=_dftAOoverlap.Matrix();
160  }
161  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << flush;
162  Eigen::MatrixXd _psi_AxB_dimer_basis =_psi_AxB.transpose()*overlap*_orbitalsAB->MOCoefficients();
163 
164  unsigned int LevelsA = _levelsA;
165  for (unsigned i=0;i<_psi_AxB_dimer_basis.rows();i++){
166  double mag=_psi_AxB_dimer_basis.row(i).squaredNorm();
167  if (mag<0.95){
168  int monomer = 0;
169  int level = 0;
170  if ( i < LevelsA ) {
171  monomer = 1;
172  level = i;
173  } else {
174  monomer = 2;
175  level = i -_levelsA;
176 
177  }
178  CTP_LOG(ctp::logERROR,*_pLog) << "\nERROR: " << i << " Projection of orbital " << level << " of monomer " << monomer << " on dimer is insufficient,mag="<<mag<<" maybe the orbital order is screwed up, otherwise increase dimer basis.\n"<<flush;
179  }
180  }
181  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
182  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << flush;
183 
184  Eigen::MatrixXd JAB_dimer = _psi_AxB_dimer_basis*_orbitalsAB->MOEnergies().asDiagonal()*_psi_AxB_dimer_basis.transpose();
185  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
186  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing Overlap matrix" << flush;
187  Eigen::MatrixXd _S_AxB = _psi_AxB_dimer_basis*_psi_AxB_dimer_basis.transpose();
188 
189  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
190  << JAB_dimer.rows() << "x"
191  << JAB_dimer.cols() << "]" << flush;
192 
193  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(_S_AxB);
194  Eigen::MatrixXd Sm1=es.operatorInverseSqrt();
195  CTP_LOG(ctp::logDEBUG,*_pLog) << "Smallest eigenvalue of overlap matrix is "<<es.eigenvalues()(0)<< flush;
196  (*_JAB) = Sm1*JAB_dimer*Sm1;
197 
198  CTP_LOG(ctp::logDEBUG,*_pLog) << "Done with electronic couplings" << flush;
199 
200  return true;
201 
202 }
