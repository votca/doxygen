43 double DFTcoupling::getCouplingElement( int levelA, int levelB, Orbitals* _orbitalsA,
44  Orbitals* _orbitalsB, ub::matrix<double>* _JAB, double _energy_difference ) {
45 
46 
47  int _levelsA = _orbitalsA->getNumberOfLevels();
48 
49  if ( _energy_difference != 0 ) {
50  std::vector<int> list_levelsA = *_orbitalsA->getDegeneracy( levelA, _energy_difference );
51  std::vector<int> list_levelsB = *_orbitalsA->getDegeneracy( levelB, _energy_difference );
52 
53  double _JAB_sq = 0; double _JAB_one_level;
54 
55  for (std::vector<int>::iterator iA = list_levelsA.begin()++; iA != list_levelsA.end(); iA++) {
56  for (std::vector<int>::iterator iB = list_levelsB.begin()++; iB != list_levelsB.end(); iB++) {
57  //cout << *iA << ':' << *iB << endl;
58  _JAB_one_level = _JAB->at_element( *iA - 1 , *iB -1 + _levelsA );
59  _JAB_sq += _JAB_one_level*_JAB_one_level ;
60  }
61  }
62 
63  return sqrt(_JAB_sq / ( list_levelsA.size() * list_levelsB.size() ) ) * tools::conv::hrt2ev ;
64 
65  } else {
66 
67  return _JAB->at_element( levelA - 1 , levelB -1 + _levelsA ) * tools::conv::hrt2ev;
68 
69  }
70  // the matrix should be symmetric, could also return this element
71  // _JAB.at_element( _levelsA + levelB - 1 , levelA - 1 );
72 }

85 bool DFTcoupling::CalculateIntegrals(Orbitals* _orbitalsA, Orbitals* _orbitalsB,
86  Orbitals* _orbitalsAB, ub::matrix<double>* _JAB) {
87 
88  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating electronic couplings" << flush;
89 
90  const std::vector<QMAtom*> atomsA=_orbitalsA->QMAtoms();
91  const std::vector<QMAtom*> atomsB=_orbitalsB->QMAtoms();
92  const std::vector<QMAtom*> atomsAB=_orbitalsAB->QMAtoms();
93 
94  for (unsigned i=0;i<atomsAB.size();i++){
95  QMAtom* dimer=atomsAB[i];
96  QMAtom* monomer=NULL;
97  if (i<atomsA.size()){
98  monomer=atomsA[i];
99  }
100  else if (i<atomsB.size()+atomsA.size() ){
101  monomer=atomsB[i-atomsA.size()];
102  }
103  else{
104  throw runtime_error((format("Number of Atoms in dimer %3i and the two monomers A:%3i B:%3i does not agree") %atomsAB.size() %atomsA.size() %atomsB.size()).str());
105  }
106 
107  if(monomer->getType() != dimer->getType()){
108  throw runtime_error("\nERROR: Atom types do not agree in dimer and monomers\n");
109  }
110  if(tools::abs(monomer->getPos()-dimer->getPos())>0.001){
111  CTP_LOG(ctp::logERROR,*_pLog) << "======WARNING=======\n Coordinates of monomers and dimer atoms do not agree, do you know what you are doing?\n " << flush;
112  break;
113  }
114 
115  }
116 
117  // constructing the direct product orbA x orbB
118  int _basisA = _orbitalsA->getBasisSetSize();
119  int _basisB = _orbitalsB->getBasisSetSize();
120 
121  if ( ( _basisA == 0 ) || ( _basisB == 0 ) ) {
122  CTP_LOG(ctp::logERROR,*_pLog) << "Basis set size is not stored in monomers" << flush;
123  return false;
124  }
125 
126  int _levelsA = _orbitalsA->getNumberOfLevels();
127  int _levelsB = _orbitalsB->getNumberOfLevels();
128 
129  //boost::timer t; // start timing
130  //double _st = t.elapsed();
131 
132  CTP_LOG(ctp::logDEBUG,*_pLog) << "Levels:Basis A[" << _levelsA << ":" << _basisA << "]"
133  << " B[" << _levelsB << ":" << _basisB << "]" << flush;
134 
135  if ( ( _levelsA == 0 ) || (_levelsB == 0) ) {
136  CTP_LOG(ctp::logERROR,*_pLog) << "No information about number of occupied/unoccupied levels is stored" << flush;
137  return false;
138  }
139 
140  // | Orbitals_A 0 | | Overlap_A |
141  // | 0 Orbitals_B | X | Overlap_B | X Transpose( Orbitals_AB )
142  ub::zero_matrix<double> zeroB( _levelsA, _basisB ) ;
143  ub::zero_matrix<double> zeroA( _levelsB, _basisA ) ;
144  ub::matrix<double> _psi_AxB ( _levelsA + _levelsB, _basisA + _basisB );
145 
146 
147  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing direct product AxB ["
148  << _psi_AxB.size1() << "x"
149  << _psi_AxB.size2() << "]" << flush;
150 
151 
152 
153  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( _basisA, _basisA +_basisB ) ) = zeroB;
154  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( 0, _basisA ) ) = zeroA;
155  ub::project( _psi_AxB, ub::range (0, _levelsA ), ub::range ( 0, _basisA ) ) = _orbitalsA->MOCoefficients();
156  ub::project( _psi_AxB, ub::range (_levelsA, _levelsA + _levelsB ), ub::range ( _basisA, _basisA + _basisB ) ) = _orbitalsB->MOCoefficients();
157 
158  // psi_AxB * S_AB * psi_AB
159 
160 
161  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting dimer onto monomer orbitals" << flush;
162  ub::matrix<double> overlap;
163  if ( _orbitalsAB->hasAOOverlap() ) {
164  CTP_LOG(ctp::logDEBUG,*_pLog) << "Reading overlap matrix from orbitals" << flush;
165  overlap= _orbitalsAB->AOOverlap();
166  }else{
167  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating overlap matrix for basisset: "<< _orbitalsAB->getDFTbasis()<< flush;
168  BasisSet _dftbasisset;
169  AOBasis _dftbasis;
170  _dftbasisset.LoadBasisSet(_orbitalsAB->getDFTbasis());
171 
172  _dftbasis.AOBasisFill(&_dftbasisset, _orbitalsAB->QMAtoms());
173  AOOverlap _dftAOoverlap;
174  _dftAOoverlap.Fill(_dftbasis);
175  overlap=_dftAOoverlap.Matrix();
176  }
177 
178  ub::matrix<double> _psi_AB = ub::prod(overlap, ub::trans( _orbitalsAB->MOCoefficients()) );
179  ub::matrix<double> _psi_AxB_dimer_basis = ub::prod( _psi_AxB, _psi_AB );
180  _psi_AB.clear();
181 
182  //check to see if projection quality is sufficient
183  for (unsigned i=0;i<_psi_AxB_dimer_basis.size1();i++){
184  double mag=0.0;
185  for (unsigned j=0;j<_psi_AxB_dimer_basis.size2();j++){
186  mag+=_psi_AxB_dimer_basis(i,j)*_psi_AxB_dimer_basis(i,j);
187 
188  }
189  if (mag<0.95){
190  throw runtime_error("\nERROR: Projection of monomer orbitals on dimer is insufficient, maybe the orbital order is screwed up, otherwise increase dimer basis.\n");
191  }
192  }
193 
194  // J = psi_AxB_dimer_basis * FAB * psi_AxB_dimer_basis^T
195  CTP_LOG(ctp::logDEBUG,*_pLog) << "Projecting the Fock matrix onto the dimer basis" << flush;
196  ub::diagonal_matrix<double> _fock_AB( _orbitalsAB->getNumberOfLevels(), _orbitalsAB->MOEnergies().data() );
197  ub::matrix<double> _temp = ub::prod( _fock_AB, ub::trans( _psi_AxB_dimer_basis ) ) ;
198  ub::matrix<double> JAB_dimer = ub::prod( _psi_AxB_dimer_basis, _temp);
199 
200  // S = psi_AxB_dimer_basis * psi_AxB_dimer_basis^T
201  CTP_LOG(ctp::logDEBUG,*_pLog) << "Constructing Overlap matrix" << flush;
202  ub::matrix<double> _S_AxB = ub::prod( _psi_AxB_dimer_basis, ub::trans( _psi_AxB_dimer_basis ));
203 
204  CTP_LOG(ctp::logDEBUG,*_pLog) << "Calculating the effective overlap JAB ["
205  << JAB_dimer.size1() << "x"
206  << JAB_dimer.size2() << "]" << flush;
207  double smalleig=linalg_loewdin(JAB_dimer, _S_AxB);
208  CTP_LOG(ctp::logDEBUG,*_pLog) << "Smallest eigenvalue of overlap matrix is "<<smalleig<< flush;
209  (*_JAB) = JAB_dimer;
210 
211 
212  CTP_LOG(ctp::logDEBUG,*_pLog) << "Done with electronic couplings" << flush;
213 
214  return true;
215 
216 }
