59 void DFTcoupling::Addoutput(tools::Property& type_summary,
60  const Orbitals& orbitalsA,
61  const Orbitals& orbitalsB) const {
62  tools::Property& dftcoupling = type_summary.add(Identify(), "");
63  dftcoupling.setAttribute("homoA", orbitalsA.getHomo());
64  dftcoupling.setAttribute("homoB", orbitalsB.getHomo());
65  tools::Property& hole_summary = dftcoupling.add("hole", "");
66  // hole hole
67  for (Index a = Range_orbA.first; a <= orbitalsA.getHomo(); ++a) {
68  for (Index b = Range_orbB.first; b <= orbitalsB.getHomo(); ++b) {
69  WriteToProperty(hole_summary, orbitalsA, orbitalsB, a, b);
70  }
71  }
72  tools::Property& electron_summary = dftcoupling.add("electron", "");
73  // electron-electron
74  for (Index a = orbitalsA.getLumo();
75  a <= Range_orbA.first + Range_orbA.second - 1; ++a) {
76  for (Index b = orbitalsB.getLumo();
77  b <= Range_orbB.first + Range_orbB.second - 1; ++b) {
78  WriteToProperty(electron_summary, orbitalsA, orbitalsB, a, b);
79  }
80  }
81  return;
82 }

84 std::pair<int, Index> DFTcoupling::DetermineRangeOfStates(
85  const Orbitals& orbital, Index numberofstates) const {
86  const Eigen::VectorXd& MOEnergies = orbital.MOs().eigenvalues();
87  if (std::abs(MOEnergies(orbital.getHomo()) - MOEnergies(orbital.getLumo())) <
88  degeneracy_) {
89  throw std::runtime_error(
90  "Homo Lumo Gap is smaller than degeneracy. "
91  "Either your degeneracy is too large or your Homo and Lumo are "
92  "degenerate");
93  }
94 
95  Index minimal = orbital.getHomo() - numberofstates + 1;
96  Index maximal = orbital.getLumo() + numberofstates - 1;
97 
98  std::vector<Index> deg_min = orbital.CheckDegeneracy(minimal, degeneracy_);
99  minimal = *std::min_element(deg_min.begin(), deg_min.end());
100 
101  std::vector<Index> deg_max = orbital.CheckDegeneracy(maximal, degeneracy_);
102  maximal = *std::max_element(deg_max.begin(), deg_max.end());
103 
104  std::pair<Index, Index> result;
105  result.first = minimal; // start
106  result.second = maximal - minimal + 1; // size
107 
108  return result;
109 }

111 double DFTcoupling::getCouplingElement(Index levelA, Index levelB,
112  const Orbitals& orbitalsA,
113  const Orbitals& orbitalsB) const {
114 
115  Index levelsA = Range_orbA.second;
116  if (degeneracy_ != 0) {
117  std::vector<Index> list_levelsA =
118  orbitalsA.CheckDegeneracy(levelA, degeneracy_);
119  std::vector<Index> list_levelsB =
120  orbitalsB.CheckDegeneracy(levelB, degeneracy_);
121 
122  double JAB_sq = 0;
123 
124  for (Index iA : list_levelsA) {
125  Index indexA = iA - Range_orbA.first;
126  for (Index iB : list_levelsB) {
127  Index indexB = iB - Range_orbB.first + levelsA;
128  double JAB_one_level = JAB(indexA, indexB);
129  JAB_sq += JAB_one_level * JAB_one_level;
130  }
131  }
132  return std::sqrt(JAB_sq /
133  double(list_levelsA.size() * list_levelsB.size())) *
134  tools::conv::hrt2ev;
135  } else {
136  Index indexA = levelA - Range_orbA.first;
137  Index indexB = levelB - Range_orbB.first + levelsA;
138  return JAB(indexA, indexB) * tools::conv::hrt2ev;
139  }
140 }

148 void DFTcoupling::CalculateCouplings(const Orbitals& orbitalsA,
149  const Orbitals& orbitalsB,
150  const Orbitals& orbitalsAB) {
151 
152  XTP_LOG(Log::error, *pLog_) << "Calculating electronic couplings" << flush;
153 
154  CheckAtomCoordinates(orbitalsA, orbitalsB, orbitalsAB);
155 
156  // constructing the direct product orbA x orbB
157  Index basisA = orbitalsA.getBasisSetSize();
158  Index basisB = orbitalsB.getBasisSetSize();
159 
160  if ((basisA == 0) || (basisB == 0)) {
161  throw std::runtime_error("Basis set size is not stored in monomers");
162  }
163 
164  Range_orbA = DetermineRangeOfStates(orbitalsA, numberofstatesA_);
165  Range_orbB = DetermineRangeOfStates(orbitalsB, numberofstatesB_);
166 
167  Index levelsA = Range_orbA.second;
168  Index levelsB = Range_orbB.second;
169 
170  XTP_LOG(Log::error, *pLog_)
171  << "Levels:Basis A[" << levelsA << ":" << basisA << "]"
172  << " B[" << levelsB << ":" << basisB << "]" << flush;
173 
174  if ((levelsA == 0) || (levelsB == 0)) {
175  throw std::runtime_error(
176  "No information about number of occupied/unoccupied levels is stored");
177  }
178 
179  // constructing merged orbitals
180  auto MOsA = orbitalsA.MOs().eigenvectors().block(0, Range_orbA.first, basisA,
181  Range_orbA.second);
182  auto MOsB = orbitalsB.MOs().eigenvectors().block(0, Range_orbB.first, basisB,
183  Range_orbB.second);
184 
185  XTP_LOG(Log::info, *pLog_) << "Calculating overlap matrix for basisset: "
186  << orbitalsAB.getDFTbasisName() << flush;
187 
188  Eigen::MatrixXd overlap =
189  CalculateOverlapMatrix(orbitalsAB) * orbitalsAB.MOs().eigenvectors();
190 
191  XTP_LOG(Log::info, *pLog_)
192  << "Projecting monomers onto dimer orbitals" << flush;
193  Eigen::MatrixXd A_AB = MOsA.transpose() * overlap.topRows(basisA);
194  Eigen::MatrixXd B_AB = MOsB.transpose() * overlap.bottomRows(basisB);
195  Eigen::VectorXd mag_A = A_AB.rowwise().squaredNorm();
196  if (mag_A.any() < 0.95) {
197  XTP_LOG(Log::error, *pLog_)
198  << "\nWarning: "
199  << "Projection of orbitals of monomer A on dimer is insufficient,mag="
200  << mag_A.minCoeff() << flush;
201  }
202  Eigen::VectorXd mag_B = B_AB.rowwise().squaredNorm();
203  if (mag_B.any() < 0.95) {
204  XTP_LOG(Log::error, *pLog_)
205  << "\nWarning: "
206  << "Projection of orbitals of monomer B on dimer is insufficient,mag="
207  << mag_B.minCoeff() << flush;
208  }
209 
210  Eigen::MatrixXd psi_AxB_dimer_basis(A_AB.rows() + B_AB.rows(), A_AB.cols());
211  psi_AxB_dimer_basis.topRows(A_AB.rows()) = A_AB;
212  psi_AxB_dimer_basis.bottomRows(B_AB.rows()) = B_AB;
213 
214  XTP_LOG(Log::info, *pLog_)
215  << "Projecting the Fock matrix onto the dimer basis" << flush;
216  Eigen::MatrixXd JAB_dimer = psi_AxB_dimer_basis *
217  orbitalsAB.MOs().eigenvalues().asDiagonal() *
218  psi_AxB_dimer_basis.transpose();
219  XTP_LOG(Log::info, *pLog_) << "Constructing Overlap matrix" << flush;
220  Eigen::MatrixXd S_AxB = psi_AxB_dimer_basis * psi_AxB_dimer_basis.transpose();
221  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(S_AxB);
222  Eigen::MatrixXd Sm1 = es.operatorInverseSqrt();
223  XTP_LOG(Log::info, *pLog_) << "Smallest eigenvalue of overlap matrix is "
224  << es.eigenvalues()(0) << flush;
225  JAB = Sm1 * JAB_dimer * Sm1;
226  XTP_LOG(Log::error, *pLog_) << "Done with electronic couplings" << flush;
227 }
