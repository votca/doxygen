58 void DFTcoupling::Addoutput(tools::Property& type_summary,
59  const Orbitals& orbitalsA,
60  const Orbitals& orbitalsB) {
61  tools::Property& dftcoupling = type_summary.add(Identify(), "");
62  dftcoupling.setAttribute("homoA", orbitalsA.getHomo());
63  dftcoupling.setAttribute("homoB", orbitalsB.getHomo());
64  tools::Property& hole_summary = dftcoupling.add("hole", "");
65  // hole hole
66  for (int a = Range_orbA.first; a <= orbitalsA.getHomo(); ++a) {
67  for (int b = Range_orbB.first; b <= orbitalsB.getHomo(); ++b) {
68  WriteToProperty(hole_summary, orbitalsA, orbitalsB, a, b);
69  }
70  }
71  tools::Property& electron_summary = dftcoupling.add("electron", "");
72  // electron-//electron
73  for (int a = orbitalsA.getLumo(); a <= Range_orbA.first + Range_orbA.second;
74  ++a) {
75  for (int b = orbitalsB.getLumo(); b <= Range_orbB.first + Range_orbB.second;
76  ++b) {
77  WriteToProperty(electron_summary, orbitalsA, orbitalsB, a, b);
78  }
79  }
80  return;
81 }

83 std::pair<int, int> DFTcoupling::DetermineRangeOfStates(
84  const Orbitals& orbital, int numberofstates) const {
85  const Eigen::VectorXd& MOEnergies = orbital.MOEnergies();
86  if (std::abs(MOEnergies(orbital.getHomo()) - MOEnergies(orbital.getLumo())) <
87  _degeneracy) {
88  throw std::runtime_error(
89  "Homo Lumo Gap is smaller than degeneracy. "
90  "Either your degeneracy is too large or your Homo and Lumo are "
91  "degenerate");
92  }
93 
94  int minimal = orbital.getHomo() - numberofstates + 1;
95  int maximal = orbital.getLumo() + numberofstates - 1;
96 
97  std::vector<int> deg_min = orbital.CheckDegeneracy(minimal, _degeneracy);
98  for (int i : deg_min) {
99  if (i < minimal) {
100  minimal = i;
101  }
102  }
103 
104  std::vector<int> deg_max = orbital.CheckDegeneracy(maximal, _degeneracy);
105  for (int i : deg_max) {
106  if (i > maximal) {
107  maximal = i;
108  }
109  }
110 
111  std::pair<int, int> result;
112  result.first = minimal; // start
113  result.second = maximal - minimal + 1; // size
114 
115  return result;
116 }

118 double DFTcoupling::getCouplingElement(int levelA, int levelB,
119  const Orbitals& orbitalsA,
120  const Orbitals& orbitalsB) const {
121 
122  int levelsA = Range_orbA.second;
123 
124  if (_degeneracy != 0) {
125  std::vector<int> list_levelsA =
126  orbitalsA.CheckDegeneracy(levelA, _degeneracy);
127  std::vector<int> list_levelsB =
128  orbitalsA.CheckDegeneracy(levelB, _degeneracy);
129 
130  double JAB_sq = 0;
131 
132  for (int iA : list_levelsA) {
133  for (int iB : list_levelsB) {
134  double JAB_one_level = JAB(iA - 1, iB - 1 + levelsA);
135  JAB_sq += JAB_one_level * JAB_one_level;
136  }
137  }
138  return std::sqrt(JAB_sq / (list_levelsA.size() * list_levelsB.size())) *
139  tools::conv::hrt2ev;
140  } else {
141  return JAB(levelA - 1, levelB - 1 + levelsA) * tools::conv::hrt2ev;
142  }
143 }

151 void DFTcoupling::CalculateCouplings(const Orbitals& orbitalsA,
152  const Orbitals& orbitalsB,
153  Orbitals& orbitalsAB) {
154 
155  CTP_LOG(ctp::logDEBUG, *_pLog) << "Calculating electronic couplings" << flush;
156 
157  CheckAtomCoordinates(orbitalsA, orbitalsB, orbitalsAB);
158 
159  // constructing the direct product orbA x orbB
160  int basisA = orbitalsA.getBasisSetSize();
161  int basisB = orbitalsB.getBasisSetSize();
162 
163  if ((basisA == 0) || (basisB == 0)) {
164  throw std::runtime_error("Basis set size is not stored in monomers");
165  }
166 
167  Range_orbA = DetermineRangeOfStates(orbitalsA, _numberofstatesA);
168  Range_orbB = DetermineRangeOfStates(orbitalsB, _numberofstatesB);
169 
170  int levelsA = Range_orbA.second;
171  int levelsB = Range_orbB.second;
172 
173  CTP_LOG(ctp::logDEBUG, *_pLog)
174  << "Levels:Basis A[" << levelsA << ":" << basisA << "]"
175  << " B[" << levelsB << ":" << basisB << "]" << flush;
176 
177  if ((levelsA == 0) || (levelsB == 0)) {
178  throw std::runtime_error(
179  "No information about number of occupied/unoccupied levels is stored");
180  }
181 
182  // | Orbitals_A 0 | | Overlap_A |
183  // | 0 Orbitals_B |.T X | Overlap_B | X ( Orbitals_AB )
184 
185  Eigen::MatrixXd psi_AxB =
186  Eigen::MatrixXd::Zero(basisA + basisB, levelsA + levelsB);
187 
188  CTP_LOG(ctp::logDEBUG, *_pLog)
189  << "Constructing direct product AxB [" << psi_AxB.rows() << "x"
190  << psi_AxB.cols() << "]" << flush;
191 
192  // constructing merged orbitals
193  psi_AxB.block(0, 0, basisA, levelsA) = orbitalsA.MOCoefficients().block(
194  0, Range_orbA.first, basisA, Range_orbA.second);
195  psi_AxB.block(basisA, levelsA, basisB, levelsB) =
196  orbitalsB.MOCoefficients().block(0, Range_orbB.first, basisB,
197  Range_orbB.second);
198 
199  Eigen::MatrixXd overlap;
200  if (orbitalsAB.hasAOOverlap()) {
201  CTP_LOG(ctp::logDEBUG, *_pLog)
202  << "Reading overlap matrix from orbitals" << flush;
203  overlap = orbitalsAB.AOOverlap();
204  } else {
205  CTP_LOG(ctp::logDEBUG, *_pLog)
206  << "Calculating overlap matrix for basisset: "
207  << orbitalsAB.getDFTbasisName() << flush;
208  overlap = CalculateOverlapMatrix(orbitalsAB);
209  }
210  CTP_LOG(ctp::logDEBUG, *_pLog)
211  << "Projecting dimer onto monomer orbitals" << flush;
212  Eigen::MatrixXd psi_AxB_dimer_basis =
213  psi_AxB.transpose() * overlap * orbitalsAB.MOCoefficients();
214 
215  unsigned int LevelsA = levelsA;
216  for (unsigned i = 0; i < psi_AxB_dimer_basis.rows(); i++) {
217  double mag = psi_AxB_dimer_basis.row(i).squaredNorm();
218  if (mag < 0.95) {
219  int monomer = 0;
220  int level = 0;
221  if (i < LevelsA) {
222  monomer = 1;
223  level = i;
224  } else {
225  monomer = 2;
226  level = i - levelsA;
227  }
228  CTP_LOG(ctp::logERROR, *_pLog)
229  << "\nWarning: " << i << " Projection of orbital " << level
230  << " of monomer " << monomer
231  << " on dimer is insufficient,mag=" << mag
232  << " maybe the orbital order is screwed up, otherwise increase dimer "
233  "basis.\n"
234  << flush;
235  }
236  }
237  CTP_LOG(ctp::logDEBUG, *_pLog)
238  << "Projecting the Fock matrix onto the dimer basis" << flush;
239  Eigen::MatrixXd JAB_dimer = psi_AxB_dimer_basis *
240  orbitalsAB.MOEnergies().asDiagonal() *
241  psi_AxB_dimer_basis.transpose();
242  CTP_LOG(ctp::logDEBUG, *_pLog) << "Constructing Overlap matrix" << flush;
243  Eigen::MatrixXd S_AxB = psi_AxB_dimer_basis * psi_AxB_dimer_basis.transpose();
244  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(S_AxB);
245  Eigen::MatrixXd Sm1 = es.operatorInverseSqrt();
246  CTP_LOG(ctp::logDEBUG, *_pLog) << "Smallest eigenvalue of overlap matrix is "
247  << es.eigenvalues()(0) << flush;
248  JAB = Sm1 * JAB_dimer * Sm1;
249  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done with electronic couplings" << flush;
250 }
