56 void DFTcoupling::Addoutput(tools::Property& type_summary,
57  const Orbitals& orbitalsA,
58  const Orbitals& orbitalsB) const {
59  tools::Property& dftcoupling = type_summary.add(Identify(), "");
60  dftcoupling.setAttribute("homoA", orbitalsA.getHomo());
61  dftcoupling.setAttribute("homoB", orbitalsB.getHomo());
62  tools::Property& hole_summary = dftcoupling.add("hole", "");
63  // hole hole
64  for (int a = Range_orbA.first; a <= orbitalsA.getHomo(); ++a) {
65  for (int b = Range_orbB.first; b <= orbitalsB.getHomo(); ++b) {
66  WriteToProperty(hole_summary, orbitalsA, orbitalsB, a, b);
67  }
68  }
69  tools::Property& electron_summary = dftcoupling.add("electron", "");
70  // electron-electron
71  for (int a = orbitalsA.getLumo();
72  a <= Range_orbA.first + Range_orbA.second - 1; ++a) {
73  for (int b = orbitalsB.getLumo();
74  b <= Range_orbB.first + Range_orbB.second - 1; ++b) {
75  WriteToProperty(electron_summary, orbitalsA, orbitalsB, a, b);
76  }
77  }
78  return;
79 }

81 std::pair<int, int> DFTcoupling::DetermineRangeOfStates(
82  const Orbitals& orbital, int numberofstates) const {
83  const Eigen::VectorXd& MOEnergies = orbital.MOs().eigenvalues();
84  if (std::abs(MOEnergies(orbital.getHomo()) - MOEnergies(orbital.getLumo())) <
85  _degeneracy) {
86  throw std::runtime_error(
87  "Homo Lumo Gap is smaller than degeneracy. "
88  "Either your degeneracy is too large or your Homo and Lumo are "
89  "degenerate");
90  }
91 
92  int minimal = orbital.getHomo() - numberofstates + 1;
93  int maximal = orbital.getLumo() + numberofstates - 1;
94 
95  std::vector<int> deg_min = orbital.CheckDegeneracy(minimal, _degeneracy);
96  minimal = *std::min_element(deg_min.begin(), deg_min.end());
97 
98  std::vector<int> deg_max = orbital.CheckDegeneracy(maximal, _degeneracy);
99  maximal = *std::max_element(deg_max.begin(), deg_max.end());
100 
101  std::pair<int, int> result;
102  result.first = minimal; // start
103  result.second = maximal - minimal + 1; // size
104 
105  return result;
106 }

108 double DFTcoupling::getCouplingElement(int levelA, int levelB,
109  const Orbitals& orbitalsA,
110  const Orbitals& orbitalsB) const {
111 
112  int levelsA = Range_orbA.second;
113  if (_degeneracy != 0) {
114  std::vector<int> list_levelsA =
115  orbitalsA.CheckDegeneracy(levelA, _degeneracy);
116  std::vector<int> list_levelsB =
117  orbitalsA.CheckDegeneracy(levelB, _degeneracy);
118 
119  double JAB_sq = 0;
120 
121  for (int iA : list_levelsA) {
122  int indexA = iA - Range_orbA.first;
123  for (int iB : list_levelsB) {
124  int indexB = iB - Range_orbB.first + levelsA;
125  double JAB_one_level = JAB(indexA, indexB);
126  JAB_sq += JAB_one_level * JAB_one_level;
127  }
128  }
129  return std::sqrt(JAB_sq / (list_levelsA.size() * list_levelsB.size())) *
130  tools::conv::hrt2ev;
131  } else {
132  int indexA = levelA - Range_orbA.first;
133  int indexB = levelB - Range_orbB.first + levelsA;
134  return JAB(indexA, indexB) * tools::conv::hrt2ev;
135  }
136 }

144 void DFTcoupling::CalculateCouplings(const Orbitals& orbitalsA,
145  const Orbitals& orbitalsB,
146  const Orbitals& orbitalsAB) {
147 
148  XTP_LOG(logDEBUG, *_pLog) << "Calculating electronic couplings" << flush;
149 
150  CheckAtomCoordinates(orbitalsA, orbitalsB, orbitalsAB);
151 
152  // constructing the direct product orbA x orbB
153  int basisA = orbitalsA.getBasisSetSize();
154  int basisB = orbitalsB.getBasisSetSize();
155 
156  if ((basisA == 0) || (basisB == 0)) {
157  throw std::runtime_error("Basis set size is not stored in monomers");
158  }
159 
160  Range_orbA = DetermineRangeOfStates(orbitalsA, _numberofstatesA);
161  Range_orbB = DetermineRangeOfStates(orbitalsB, _numberofstatesB);
162 
163  int levelsA = Range_orbA.second;
164  int levelsB = Range_orbB.second;
165 
166  XTP_LOG(logDEBUG, *_pLog)
167  << "Levels:Basis A[" << levelsA << ":" << basisA << "]"
168  << " B[" << levelsB << ":" << basisB << "]" << flush;
169 
170  if ((levelsA == 0) || (levelsB == 0)) {
171  throw std::runtime_error(
172  "No information about number of occupied/unoccupied levels is stored");
173  }
174 
175  // constructing merged orbitals
176  auto MOsA = orbitalsA.MOs().eigenvectors().block(0, Range_orbA.first, basisA,
177  Range_orbA.second);
178  auto MOsB = orbitalsB.MOs().eigenvectors().block(0, Range_orbB.first, basisB,
179  Range_orbB.second);
180 
181  XTP_LOG(logDEBUG, *_pLog) << "Calculating overlap matrix for basisset: "
182  << orbitalsAB.getDFTbasisName() << flush;
183 
184  Eigen::MatrixXd overlap =
185  CalculateOverlapMatrix(orbitalsAB) * orbitalsAB.MOs().eigenvectors();
186 
187  XTP_LOG(logDEBUG, *_pLog)
188  << "Projecting monomers onto dimer orbitals" << flush;
189  Eigen::MatrixXd A_AB = MOsA.transpose() * overlap.topRows(basisA);
190  Eigen::MatrixXd B_AB = MOsB.transpose() * overlap.bottomRows(basisB);
191  Eigen::VectorXd mag_A = A_AB.rowwise().squaredNorm();
192  if (mag_A.any() < 0.95) {
193  XTP_LOG(logERROR, *_pLog)
194  << "\nWarning: "
195  << "Projection of orbitals of monomer A on dimer is insufficient,mag="
196  << mag_A.minCoeff() << flush;
197  }
198  Eigen::VectorXd mag_B = B_AB.rowwise().squaredNorm();
199  if (mag_B.any() < 0.95) {
200  XTP_LOG(logERROR, *_pLog)
201  << "\nWarning: "
202  << "Projection of orbitals of monomer B on dimer is insufficient,mag="
203  << mag_B.minCoeff() << flush;
204  }
205 
206  Eigen::MatrixXd psi_AxB_dimer_basis(A_AB.rows() + B_AB.rows(), A_AB.cols());
207  psi_AxB_dimer_basis.topRows(A_AB.rows()) = A_AB;
208  psi_AxB_dimer_basis.bottomRows(B_AB.rows()) = B_AB;
209 
210  XTP_LOG(logDEBUG, *_pLog)
211  << "Projecting the Fock matrix onto the dimer basis" << flush;
212  Eigen::MatrixXd JAB_dimer = psi_AxB_dimer_basis *
213  orbitalsAB.MOs().eigenvalues().asDiagonal() *
214  psi_AxB_dimer_basis.transpose();
215  XTP_LOG(logDEBUG, *_pLog) << "Constructing Overlap matrix" << flush;
216  Eigen::MatrixXd S_AxB = psi_AxB_dimer_basis * psi_AxB_dimer_basis.transpose();
217  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(S_AxB);
218  Eigen::MatrixXd Sm1 = es.operatorInverseSqrt();
219  XTP_LOG(logDEBUG, *_pLog) << "Smallest eigenvalue of overlap matrix is "
220  << es.eigenvalues()(0) << flush;
221  JAB = Sm1 * JAB_dimer * Sm1;
222  XTP_LOG(logDEBUG, *_pLog) << "Done with electronic couplings" << flush;
223 }
