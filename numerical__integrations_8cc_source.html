308  std::vector< std::vector< std::vector< std::vector< GridContainers::integration_grid* > > > > boxes;

312

314  for ( unsigned j = 0 ; j < grid[i].size(); j++){
315  const tools::vec& pos= grid[i][j].grid_pos;

317  max.x()=pos.getX();

326  min.y()=pos.getY();
327  }

329  max.z()=pos.getZ();

332  min.z()=pos.getZ();

334  }
335  }
336 
337  vec molextension=(max-min);

388  void NumericalIntegration::FindSignificantShells(){
389 
390  for (unsigned i=0;i<_grid_boxes.size();++i){
391  GridBox & box=_grid_boxes[i];
392  for (AOBasis::AOShellIterator _row = _basis->firstShell(); _row != _basis->lastShell(); _row++) {
393  AOShell* _store=(*_row);
394  const double decay=(*_row)->getMinDecay();
395  const tools::vec& shellpos=(*_row)->getPos();
396 
397  for(const auto& point : box.getGridPoints()){
398  tools::vec dist=shellpos-point;
399  double distsq=dist*dist;
400  // if contribution is smaller than -ln(1e-10), add atom to list
401  if ( (decay * distsq) < 20.7 ){
402  box.addShell(_store);
403  break;
404  }
405  }
406  }
407  //cout<<box.significant_shells.size()<<" "<<box.grid_pos.size()<<endl;
408  }

410  std::vector< GridBox > _grid_boxes_copy;
411 
412  int combined=0;
413  std::vector<bool> Compared=std::vector<bool>(_grid_boxes.size(),false);
414  for (unsigned i=0;i<_grid_boxes.size();i++){
415  if(Compared[i]){continue;}
416  GridBox box=_grid_boxes[i];
417  if(box.Shellsize()<1){continue;}
418  Compared[i]=true;
419  for (unsigned j=i+1;j<_grid_boxes.size();j++){
420  if(GridBox::compareGridboxes(_grid_boxes[i],_grid_boxes[j])){
421  Compared[j]=true;
422  box.addGridBox(_grid_boxes[j]);
423  combined++;
424  }
425 
426  }
427  _grid_boxes_copy.push_back(box);
428  }
429 
430 
431 
432 
433  std::vector<unsigned> sizes;
434  sizes.reserve(_grid_boxes_copy.size());

436  sizes.push_back(box.size()*box.Matrixsize());
437  }

439

441  std::iota(indexes.begin(), indexes.end(), 0);

447  #endif

454  }
455 
456

458  unsigned thread=0;

460  for(unsigned i=0;i<scores.size();++i){
461  if(scores[i]<minimum){
462  minimum=scores[i];
463  thread=i;
464  }
465  }

477  thread_start.push_back(start);
478  stop=start+thread_index.size();
479  thread_stop.push_back(stop);
480  start=stop;
481  for(const unsigned index:thread_index){
482  GridBox newbox=_grid_boxes_copy[index];
483  newbox.setIndexoffirstgridpoint(indexoffirstgridpoint);
484  indexoffirstgridpoint+=newbox.size();
485  newbox.PrepareForIntegration();

488  }

495  ub::matrix<double> NumericalIntegration::IntegrateVXC(const ub::matrix<double>& _density_matrix){
496  ub::matrix<double> Vxc=ub::zero_matrix<double>(_density_matrix.size1());
497  EXC = 0;
498 
499  unsigned nthreads = 1;
500  #ifdef _OPENMP
501  nthreads = omp_get_max_threads();
502  #endif
503  std::vector<ub::matrix<double> >vxc_thread;
504  std::vector<double> Exc_thread=std::vector<double>(nthreads,0.0);
505  for(unsigned i=0;i<nthreads;++i){
506  ub::matrix<double> Vxc_thread=ub::zero_matrix<double>(_density_matrix.size1());
507  vxc_thread.push_back(Vxc_thread);
508  }
509 
510 
511  #pragma omp parallel for
512  for (unsigned thread=0;thread<nthreads;++thread){
513  for (unsigned i = thread_start[thread]; i < thread_stop[thread]; ++i) {
514 
515  double EXC_box=0.0;
516  GridBox& box = _grid_boxes[i];
517 
518  const ub::matrix<double> DMAT_here=box.ReadFromBigMatrix(_density_matrix);
519 
520  ub::matrix<double> Vxc_here=ub::zero_matrix<double>(DMAT_here.size1());
521  const std::vector<tools::vec>& points=box.getGridPoints();
522  const std::vector<double>& weights=box.getGridWeights();
523 
524  ub::range one=ub::range(0,1);
525  ub::range three=ub::range(0,3);
526  ub::matrix<double> _temp = ub::matrix<double>(1,box.Matrixsize());
527  ub::matrix<double> _tempgrad = ub::matrix<double>(3,box.Matrixsize());
528  ub::matrix<double> ao=ub::matrix<double>(1,box.Matrixsize());
529  ub::matrix<double> ao_grad=ub::matrix<double>(3,box.Matrixsize());
530 
531  //iterate over gridpoints
532  for(unsigned p=0;p<box.size();p++){
533  ao=ub::zero_matrix<double>(1,box.Matrixsize());
534  ao_grad=ub::zero_matrix<double>(3,box.Matrixsize());
535  const std::vector<ub::range>& aoranges=box.getAOranges();
536  const std::vector<const AOShell* >& shells=box.getShells();
537 
538  for(unsigned j=0;j<box.Shellsize();++j){
539 
540  ub::matrix_range< ub::matrix<double> > aoshell=ub::project(ao,one,aoranges[j]);
541  ub::matrix_range< ub::matrix<double> > ao_grad_shell=ub::project(ao_grad,three,aoranges[j]);
542  shells[j]->EvalAOspace(aoshell,ao_grad_shell,points[p]);
543 
544  }
545 
546  _temp=ub::prod( ao, DMAT_here);
547 
548  _tempgrad=ub::prod(ao_grad,DMAT_here);
549 
550  double rho=ub::prod(_temp, ub::trans( ao) )(0,0);
551 
552  ub::matrix<double> rho_grad=ub::prod(_temp, ub::trans(ao_grad))+ub::prod(ao,ub::trans(_tempgrad));
553 
554  if ( rho < 1.e-15 ) continue; // skip the rest, if density is very small
555 
556  double f_xc; // E_xc[n] = int{n(r)*eps_xc[n(r)] d3r} = int{ f_xc(r) d3r }
557  double df_drho; // v_xc_rho(r) = df/drho
558  double df_dsigma; // df/dsigma ( df/dgrad(rho) = df/dsigma * dsigma/dgrad(rho) = df/dsigma * 2*grad(rho))
559  EvaluateXC( rho,rho_grad,f_xc, df_drho, df_dsigma);
560 
561  double weight=weights[p];
562  ub::matrix<double> _addXC = weight * df_drho * ao *0.5;
563 
564  _addXC+= 2.0*df_dsigma * weight * ub::prod(rho_grad,ao_grad);
565 
566  // Exchange correlation energy
567  EXC_box += weight * rho * f_xc;
568 
569  Vxc_here+=ub::prod( ub::trans(_addXC), ao);
570 
571  }
572 
573  box.AddtoBigMatrix(vxc_thread[thread],Vxc_here);
574 
575  Exc_thread[thread]+=EXC_box;
576 
577  }
578 
579  }
580  for(unsigned i=0;i<nthreads;++i){
581  Vxc+=vxc_thread[i];
582  EXC+=Exc_thread[i];
583  }
584  Vxc+=ub::trans(Vxc);
585 
586  return Vxc;
587  }
