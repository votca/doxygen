51  double NumericalIntegration::getExactExchange(const std::string& functional) {
52 
53  double exactexchange = 0.0;
54  Vxc_Functionals map;
55  std::vector<std::string> strs;
56 
57  boost::split(strs, functional, boost::is_any_of(" "));
58  if (strs.size() > 2) {
59  throw std::runtime_error("Too many functional names");
60  } else if (strs.size() < 1) {
61  throw std::runtime_error("Specify at least one functional");
62  }
63 
64  for (unsigned i = 0; i < strs.size(); i++) {
65 
66  int func_id = map.getID(strs[i]);
67  if (func_id < 0) {
68  exactexchange = 0.0;
69  break;
70  }
71  xc_func_type func;
72  if (xc_func_init(&func, func_id, XC_UNPOLARIZED) != 0) {
73  throw std::runtime_error((boost::format("Functional %s not found\n") %strs[i]).str());
74  }
75  if (exactexchange > 0 && func.cam_alpha > 0) {
76  throw std::runtime_error("You have specified two functionals with exact exchange");
77  }
78  exactexchange += func.cam_alpha;
79  xc_func_end(&func);
80  }
81 
82  return exactexchange;
83 
84  }

87  void NumericalIntegration::setXCfunctional(const std::string& functional) {
88 
89  Vxc_Functionals map;
90  std::vector<std::string> strs;
91  tools::Tokenizer tok(functional," ,\n\t");
92  tok.ToVector(strs);
93  xfunc_id = 0;
94  _use_separate = false;
95  cfunc_id = 0;
96  if (strs.size() == 1) {
97  xfunc_id = map.getID(strs[0]);
98  } else if (strs.size() == 2) {
99  xfunc_id = map.getID(strs[0]);
100  cfunc_id = map.getID(strs[1]);
101  _use_separate = true;
102  } else {
103  std::cout << "LIBXC " << strs.size() << std::endl;
104  throw std::runtime_error("LIBXC. Please specify one combined or an exchange and a correlation functionals");
105  }
106 
107  if (xc_func_init(&xfunc, xfunc_id, XC_UNPOLARIZED) != 0) {
108  throw std::runtime_error((boost::format("Functional %s not found\n") %strs[0]).str());
109  }
110  if (xfunc.info->kind != 2 && !_use_separate) {
111  throw std::runtime_error("Your functional misses either correlation or exchange, please specify another functional, separated by whitespace");
112  }
113  if (_use_separate) {
114  if (xc_func_init(&cfunc, cfunc_id, XC_UNPOLARIZED) != 0) {
115  throw std::runtime_error((boost::format("Functional %s not found\n") %strs[1]).str());
116  }
117  if ((xfunc.info->kind + cfunc.info->kind) != 1) {
118  throw std::runtime_error("Your functionals are not one exchange and one correlation");
119  }
120  }
121  _setXC = true;
122  return;
123  }

126  void NumericalIntegration::EvaluateXC(const double rho, const double sigma, double& f_xc, double& df_drho, double& df_dsigma) {
127 
128  double exc[1];
129  double vsigma[1]; // libxc
130  double vrho[1]; // libxc df/drho
131  switch (xfunc.info->family) {
132  case XC_FAMILY_LDA:
133  xc_lda_exc_vxc(&xfunc, 1, &rho, exc, vrho);
134  break;
135  case XC_FAMILY_GGA:
136  case XC_FAMILY_HYB_GGA:
137  xc_gga_exc_vxc(&xfunc, 1, &rho, &sigma, exc, vrho, vsigma);
138  break;
139  }
140  f_xc = exc[0];
141  df_drho = vrho[0];
142  df_dsigma = vsigma[0];
143  if (_use_separate) {
144  // via libxc correlation part only
145  switch (cfunc.info->family) {
146  case XC_FAMILY_LDA:
147  xc_lda_exc_vxc(&cfunc, 1, &rho, exc, vrho);
148  break;
149  case XC_FAMILY_GGA:
150  case XC_FAMILY_HYB_GGA:
151  xc_gga_exc_vxc(&cfunc, 1, &rho, &sigma, exc, vrho, vsigma);
152  break;
153  }
154 
155  f_xc += exc[0];
156  df_drho += vrho[0];
157  df_dsigma += vsigma[0];
158  }
159 
160  return;
161  }

164  double NumericalIntegration::IntegratePotential(const tools::vec& rvector) {
165 
166  double result = 0.0;
167  assert(_density_set && "Density not calculated");
168  for (unsigned i = 0; i < _grid_boxes.size(); i++) {
169  const std::vector<tools::vec>& points = _grid_boxes[i].getGridPoints();
170  const std::vector<double>& weights = _grid_boxes[i].getGridWeights();
171  const std::vector<double>& densities = _grid_boxes[i].getGridDensities();
172  for (unsigned j = 0; j < points.size(); j++) {
173  double dist = abs(points[j] - rvector);
174  result -= weights[j] * densities[j] / dist;
175  }
176  }
177  return result;
178  }

183  void NumericalIntegration::SortGridpointsintoBlocks(std::vector< std::vector< GridContainers::Cartesian_gridpoint > >& grid) {
184  const double boxsize = 1;//1 bohr
185 
186  std::vector< std::vector< std::vector< std::vector< GridContainers::Cartesian_gridpoint* > > > > boxes;
187  tools::vec min = tools::vec(std::numeric_limits<double>::max());
188  tools::vec max = tools::vec(std::numeric_limits<double>::min());
189 
190  for (unsigned i = 0; i < grid.size(); i++) {
191  for (unsigned j = 0; j < grid[i].size(); j++) {
192  const tools::vec& pos = grid[i][j].grid_pos;
193  if (pos.getX() > max.getX()) {
194  max.x() = pos.getX();
195  } else if (pos.getX() < min.getX()) {
196  min.x() = pos.getX();
197  }
198  if (pos.getY() > max.getY()) {
199  max.y() = pos.getY();
200  } else if (pos.getY() < min.getY()) {
201  min.y() = pos.getY();
202  }
203  if (pos.getZ() > max.getZ()) {
204  max.z() = pos.getZ();
205  } else if (pos.getZ() < min.getZ()) {
206  min.z() = pos.getZ();
207  }
208  }
209  }
210 
211  tools::vec molextension = (max - min);
212  tools::vec numberofboxes = molextension / boxsize;
213  tools::vec roundednumofbox = tools::vec(std::ceil(numberofboxes.getX()), std::ceil(numberofboxes.getY()), std::ceil(numberofboxes.getZ()));
214 
215  //creating temparray
216  for (unsigned i = 0; i<unsigned(roundednumofbox.getX()); i++) {
217  std::vector< std::vector< std::vector< GridContainers::Cartesian_gridpoint* > > > boxes_yz;
218  for (unsigned j = 0; j<unsigned(roundednumofbox.getY()); j++) {
219  std::vector< std::vector< GridContainers::Cartesian_gridpoint* > > boxes_z;
220  for (unsigned k = 0; k<unsigned(roundednumofbox.getZ()); k++) {
221  std::vector< GridContainers::Cartesian_gridpoint* > box;
222  box.reserve(100);
223  boxes_z.push_back(box);
224  }
225  boxes_yz.push_back(boxes_z);
226  }
227  boxes.push_back(boxes_yz);
228  }
229 
230  for (auto & atomgrid : grid) {
231  for (auto & gridpoint : atomgrid) {
232  tools::vec pos = gridpoint.grid_pos - min;
233  tools::vec index = pos / boxsize;
234  int i_x = int(index.getX());
235  int i_y = int(index.getY());
236  int i_z = int(index.getZ());
237  boxes[i_x][i_y][i_z].push_back(&gridpoint);
238  }
239  }
240 
241  for (auto& boxes_xy : boxes) {
242  for (auto& boxes_z : boxes_xy) {
243  for (auto& box : boxes_z) {
244  if (box.size() < 1) {
245  continue;
246  }
247  GridBox gridbox;
248 
249  for (const auto&point : box) {
250  gridbox.addGridPoint(*point);
251  }
252  _grid_boxes.push_back(gridbox);
253  }
254  }
255  }
256  return;
257  }

259  void NumericalIntegration::FindSignificantShells(const AOBasis& basis) {
260  for (unsigned i = 0; i < _grid_boxes.size(); ++i) {
261  GridBox & box = _grid_boxes[i];
262  for (const AOShell* store:basis) {
263  const double decay = store->getMinDecay();
264  const tools::vec& shellpos =store->getPos();
265  for (const auto& point : box.getGridPoints()) {
266  tools::vec dist = shellpos - point;
267  double distsq = dist*dist;
268  // if contribution is smaller than -ln(1e-10), add shell to list
269  if ((decay * distsq) < 20.7) {
270  box.addShell(store);
271  break;
272  }
273  }
274  }
275  }
276 
277  std::vector< GridBox > grid_boxes_copy;
278  int combined = 0;
279  //use vecot of bool to indicate if a gridbox has already been merged into another
280  std::vector<bool> Merged = std::vector<bool>(_grid_boxes.size(), false);
281  for (unsigned i = 0; i < _grid_boxes.size(); i++) {
282  if (Merged[i]) {
283  continue;
284  }
285  GridBox box = _grid_boxes[i];
286  if (box.Shellsize() < 1) {
287  continue;
288  }
289  Merged[i] = true;
290  for (unsigned j = i + 1; j < _grid_boxes.size(); j++) {
291  if (GridBox::compareGridboxes(_grid_boxes[i], _grid_boxes[j])) {
292  Merged[j] = true;
293  box.addGridBox(_grid_boxes[j]);
294  combined++;
295  }
296  }
297  grid_boxes_copy.push_back(box);
298  }
299  std::vector<unsigned> sizes;
300  sizes.reserve(grid_boxes_copy.size());
301  for (auto& box : grid_boxes_copy) {
302  sizes.push_back(box.size() * box.Matrixsize());
303  }
304  std::vector<unsigned> indexes = std::vector<unsigned>(sizes.size());
305  std::iota(indexes.begin(), indexes.end(), 0);
306  std::sort(indexes.begin(), indexes.end(), [&sizes](unsigned i1, unsigned i2) {
307  return sizes[i1] > sizes[i2];
308  });
309 
310  unsigned nthreads = 1;
311 #ifdef _OPENMP
312  nthreads = omp_get_max_threads();
313 #endif
314  std::vector<unsigned> scores = std::vector<unsigned>(nthreads, 0);
315  std::vector< std::vector<unsigned> > indices;
316  for (unsigned i = 0; i < nthreads; ++i) {
317  std::vector<unsigned> thread_box_indices;
318  indices.push_back(thread_box_indices);
319  }
320  for (const auto index : indexes) {
321  unsigned thread = 0;
322  unsigned minimum = std::numeric_limits<unsigned>::max();
323  for (unsigned i = 0; i < scores.size(); ++i) {
324  if (scores[i] < minimum) {
325  minimum = scores[i];
326  thread = i;
327  }
328  }
329  indices[thread].push_back(index);
330  scores[thread] += sizes[index];
331  }
332  thread_start = std::vector<unsigned>(0);
333  thread_stop = std::vector<unsigned>(0);
334  unsigned start = 0;
335  unsigned stop = 0;
336  unsigned indexoffirstgridpoint = 0;
337  _grid_boxes.resize(0);
338  for (const std::vector<unsigned>& thread_index : indices) {
339  thread_start.push_back(start);
340  stop = start + thread_index.size();
341  thread_stop.push_back(stop);
342  start = stop;
343  for (const unsigned index : thread_index) {
344  GridBox newbox = grid_boxes_copy[index];
345  newbox.setIndexoffirstgridpoint(indexoffirstgridpoint);
346  indexoffirstgridpoint += newbox.size();
347  newbox.PrepareForIntegration();
348  _grid_boxes.push_back(newbox);
349  }
350  }
351  return;
352  }

356  Eigen::MatrixXd NumericalIntegration::IntegrateVXC(const Eigen::MatrixXd& density_matrix) {
357  Eigen::MatrixXd Vxc = Eigen::MatrixXd::Zero(density_matrix.rows(), density_matrix.cols());
358  _EXC = 0;
359  unsigned nthreads = 1;
360 #ifdef _OPENMP
361  nthreads = omp_get_max_threads();
362 #endif
363  std::vector<Eigen::MatrixXd >vxc_thread;
364  std::vector<double> Exc_thread = std::vector<double>(nthreads, 0.0);
365  for (unsigned i = 0; i < nthreads; ++i) {
366  vxc_thread.push_back(Eigen::MatrixXd::Zero(density_matrix.rows(), density_matrix.cols()));
367  }
368 
369 #pragma omp parallel for
370  for (unsigned thread = 0; thread < nthreads; ++thread) {
371  for (unsigned i = thread_start[thread]; i < thread_stop[thread]; ++i) {
372 
373  double EXC_box = 0.0;
374  const GridBox& box = _grid_boxes[i];
375  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
376  const Eigen::MatrixXd DMAT_symm= DMAT_here+DMAT_here.transpose();
377  double cutoff=1.e-40/density_matrix.rows()/density_matrix.rows();
378  if (DMAT_here.cwiseAbs2().maxCoeff()<cutoff ){
379  continue;
380  }
381  Eigen::MatrixXd Vxc_here = Eigen::MatrixXd::Zero(DMAT_here.rows(), DMAT_here.cols());
382  const std::vector<tools::vec>& points = box.getGridPoints();
383  const std::vector<double>& weights = box.getGridWeights();
384  const std::vector<GridboxRange>& aoranges = box.getAOranges();
385  const std::vector<const AOShell* >& shells = box.getShells();
386 
387  //iterate over gridpoints
388  for (unsigned p = 0; p < box.size(); p++) {
389  Eigen::VectorXd ao = Eigen::VectorXd::Zero(box.Matrixsize());
390  Eigen::MatrixX3d ao_grad= Eigen::MatrixX3d::Zero(box.Matrixsize(),3);
391  for (unsigned j = 0; j < box.Shellsize(); ++j) {
392  Eigen::Block<Eigen::MatrixX3d> grad_block=ao_grad.block(aoranges[j].start,0,aoranges[j].size,3);
393  Eigen::VectorBlock<Eigen::VectorXd> ao_block=ao.segment(aoranges[j].start,aoranges[j].size);
394  shells[j]->EvalAOspace(ao_block,grad_block,points[p]);
395  }
396  const double rho =0.5* (ao.transpose()*DMAT_symm*ao).value();
397  const double weight = weights[p];
398  if (rho*weight < 1.e-20) continue; // skip the rest, if density is very small
399  const Eigen::Vector3d rho_grad = ao.transpose()*DMAT_symm*ao_grad;
400  const double sigma = (rho_grad.transpose()*rho_grad).value();
401  const Eigen::VectorXd grad =ao_grad*rho_grad;
402  double f_xc; // E_xc[n] = int{n(r)*eps_xc[n(r)] d3r} = int{ f_xc(r) d3r }
403  double df_drho; // v_xc_rho(r) = df/drho
404  double df_dsigma; // df/dsigma ( df/dgrad(rho) = df/dsigma * dsigma/dgrad(rho) = df/dsigma * 2*grad(rho))
405  EvaluateXC(rho, sigma, f_xc, df_drho, df_dsigma);
406  EXC_box += weight * rho * f_xc;
407  auto addXC = weight * (0.5*df_drho * ao+ 2.0 * df_dsigma *grad);
408  // Exchange correlation energy
409  Vxc_here.noalias() += addXC* ao.transpose();
410  }
411  box.AddtoBigMatrix(vxc_thread[thread], Vxc_here);
412  Exc_thread[thread] += EXC_box;
413  }
414  }
415  for (unsigned i = 0; i < nthreads; ++i) {
416  Vxc += vxc_thread[i];
417  _EXC += Exc_thread[i];
418  }
419  return Vxc+Vxc.transpose();
420  }

423  Eigen::MatrixXd NumericalIntegration::IntegrateExternalPotential(const std::vector<double>& Potentialvalues) {
424 
425  Eigen::MatrixXd ExternalMat = Eigen::MatrixXd::Zero(_AOBasisSize, _AOBasisSize);
426  unsigned nthreads = 1;
427 #ifdef _OPENMP
428  nthreads = omp_get_max_threads();
429 #endif
430  std::vector<Eigen::MatrixXd >vex_thread;
431  std::vector<double> Exc_thread = std::vector<double>(nthreads, 0.0);
432  for (unsigned i = 0; i < nthreads; ++i) {
433  Eigen::MatrixXd Vex_thread = Eigen::MatrixXd::Zero(ExternalMat.rows(), ExternalMat.cols());
434  vex_thread.push_back(Vex_thread);
435  }
436 
437 #pragma omp parallel for
438  for (unsigned thread = 0; thread < nthreads; ++thread) {
439  for (unsigned i = thread_start[thread]; i < thread_stop[thread]; ++i) {
440 
441  const GridBox& box = _grid_boxes[i];
442  Eigen::MatrixXd Vex_here = Eigen::MatrixXd::Zero(box.Matrixsize(), box.Matrixsize());
443  const std::vector<tools::vec>& points = box.getGridPoints();
444  const std::vector<double>& weights = box.getGridWeights();
445 
446  //iterate over gridpoints
447  for (unsigned p = 0; p < box.size(); p++) {
448  Eigen::VectorXd ao = Eigen::VectorXd::Zero(box.Matrixsize());
449  const std::vector<GridboxRange>& aoranges = box.getAOranges();
450  const std::vector<const AOShell* > shells = box.getShells();
451  for (unsigned j = 0; j < box.Shellsize(); ++j) {
452  Eigen::VectorBlock<Eigen::VectorXd> ao_block=ao.segment(aoranges[j].start,aoranges[j].size);
453  shells[j]->EvalAOspace(ao_block, points[p]);
454  }

456  Vex_here += addEX.transpose() * ao;
457  }
458  box.AddtoBigMatrix(vex_thread[thread], Vex_here);
459  }
460  }
461  for (unsigned i = 0; i < nthreads; ++i) {
462  ExternalMat += vex_thread[i];
463  }
464  ExternalMat += ExternalMat.transpose();
465  return ExternalMat;
466  }

473 #endif

478  for (unsigned i = thread_start[thread]; i < thread_stop[thread]; ++i) {
479

509  Gyrationtensor NumericalIntegration::IntegrateGyrationTensor(const Eigen::MatrixXd& density_matrix) {
510  double N = 0;
511  tools::vec centroid = tools::vec(0.0);
512  tools::matrix gyration = tools::matrix(0.0);
513  unsigned nthreads = 1;
514 #ifdef _OPENMP
515  nthreads = omp_get_max_threads();
516 #endif
517  std::vector<double> N_thread = std::vector<double>(nthreads, 0.0);
518  // centroid
519  std::vector<tools::vec> centroid_thread;
520  std::vector<tools::matrix> gyration_thread;
521  for (unsigned thread = 0; thread < nthreads; ++thread) {
522  tools::vec tempvec = tools::vec(0.0);
523  centroid_thread.push_back(tempvec);
524  tools::matrix tempmatrix = tools::matrix(0.0);
525  gyration_thread.push_back(tempmatrix);
526  }
527 
528 #pragma omp parallel for
529  for (unsigned thread = 0; thread < nthreads; ++thread) {
530  for (unsigned i = thread_start[thread]; i < thread_stop[thread]; ++i) {
531  double N_box = 0.0;
532  tools::vec centroid_box = tools::vec(0.0);
533  tools::matrix gyration_box = tools::matrix(0.0);
534  GridBox& box = _grid_boxes[i];
535  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
536  const std::vector<tools::vec>& points = box.getGridPoints();
537  const std::vector<double>& weights = box.getGridWeights();
538  box.prepareDensity();
539  //iterate over gridpoints
540  for (unsigned p = 0; p < box.size(); p++) {
541  Eigen::VectorXd ao = Eigen::VectorXd::Zero(box.Matrixsize());
542  const std::vector<GridboxRange>& aoranges = box.getAOranges();
543  const std::vector<const AOShell* > shells = box.getShells();
544  for (unsigned j = 0; j < box.Shellsize(); ++j) {
545  Eigen::VectorBlock<Eigen::VectorXd> ao_block=ao.segment(aoranges[j].start,aoranges[j].size);
546  shells[j]->EvalAOspace(ao_block, points[p]);
547  }
548  double rho =(ao.transpose()*DMAT_here*ao)(0, 0);
549  box.addDensity(rho);
550  N_box += rho * weights[p];
551  centroid_box+=rho * weights[p] * points[p];
552  gyration_box += rho * weights[p] * (points[p]|points[p]);
553  }
554  N_thread[thread] += N_box;
555  centroid_thread[thread] += centroid_box;
556  gyration_thread[thread] += gyration_box;
557  }
558  }
559  for (unsigned i = 0; i < nthreads; ++i) {
560  N += N_thread[i];
561  centroid += centroid_thread[i];

628 int NumericalIntegration::UpdateOrder(LebedevGrid& sphericalgridofElement, int maxorder, std::vector<double>& PruningIntervals, double r){
629  int order;
630  int maxindex=sphericalgridofElement.getIndexFromOrder(maxorder);
631  if (maxindex == 1) {
632  // smallest possible grid anyway, nothing to do
633  order = maxorder;
634  } else if (maxindex == 2) {
635  // only three intervals
636  if (r < PruningIntervals[0]) {
637  order = sphericalgridofElement.getOrderFromIndex(1); //1;
638  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[3])) {
639  order = sphericalgridofElement.getOrderFromIndex(2);
640  } else {
641  order = sphericalgridofElement.getOrderFromIndex(1);
642  } // maxorder == 2
643  } else {
644  // five intervals
645  if (r < PruningIntervals[0]) {
646  order = sphericalgridofElement.getOrderFromIndex(2);
647  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[1])) {
648  order = sphericalgridofElement.getOrderFromIndex(4);
649  } else if ((r >= PruningIntervals[1]) && (r < PruningIntervals[2])) {
650  order = sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 4));
651  } else if ((r >= PruningIntervals[2]) && (r < PruningIntervals[3])) {
652  order = maxorder;
653  } else {
654  order = sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 1));
655  }
656  }
657  return order;
658  }

706 void NumericalIntegration::GridSetup(const std::string& type, std::vector<QMAtom*> atoms,const AOBasis& basis) {
707  _AOBasisSize=basis.AOBasisSize();
708  GridContainers initialgrids;
709  // get radial grid per element
710  EulerMaclaurinGrid radialgridofElement;
711  initialgrids.radial_grids=radialgridofElement.CalculateAtomicRadialGrids(basis, atoms, type); // this checks out 1:1 with NWChem results! AWESOME
712  LebedevGrid sphericalgridofElement;
713  initialgrids.spherical_grids=sphericalgridofElement.CalculateSphericalGrids(atoms,type);
714 
715  // for the partitioning, we need all inter-center distances later, stored in matrix
716  Eigen::MatrixXd Rij=CalcInverseAtomDist(atoms);
717  _totalgridsize = 0;
718  std::vector< std::vector< GridContainers::Cartesian_gridpoint > > grid;
719 
720  for (unsigned i_atom=0;i_atom<atoms.size();++i_atom) {
721  QMAtom* atom=atoms[i_atom];
722 
723  const tools::vec & atomA_pos =atom->getPos();
724  const std::string & name = atom->getType();
725  GridContainers::radial_grid radial_grid = initialgrids.radial_grids.at(name);
726  GridContainers::spherical_grid spherical_grid = initialgrids.spherical_grids.at(name);
727 
728  // maximum order (= number of points) in spherical integration grid
729  int maxorder = sphericalgridofElement.Type2MaxOrder(name, type);
730  // for pruning of integration grid, get interval boundaries for this element
731  std::vector<double> PruningIntervals = radialgridofElement.CalculatePruningIntervals(name);
732  int current_order = 0;
733  // for each radial value
734  std::vector< GridContainers::Cartesian_gridpoint > atomgrid;
735  for (unsigned i_rad = 0; i_rad < radial_grid.radius.size(); i_rad++) {
736  double r = radial_grid.radius[i_rad];
737 
738  // which Lebedev order for this point?
739  int order=UpdateOrder(sphericalgridofElement, maxorder, PruningIntervals, r);
740  // get new spherical grid, if order changed
741  if (order != current_order) {
742  spherical_grid=sphericalgridofElement.CalculateUnitSphereGrid(order);
743  current_order = order;
744  }
745 
746  for (unsigned i_sph = 0; i_sph < spherical_grid.phi.size(); i_sph++) {
747  GridContainers::Cartesian_gridpoint gridpoint=CreateCartesianGridpoint(atomA_pos, radial_grid, spherical_grid, i_rad,i_sph);
748  atomgrid.push_back(gridpoint);
749  } // spherical gridpoints
750  } // radial gridpoint
751 
752  SSWpartitionAtom(atoms, atomgrid, i_atom,Rij);
753 
754  // now remove points from the grid with negligible weights
755  for (std::vector<GridContainers::Cartesian_gridpoint >::iterator git = atomgrid.begin(); git != atomgrid.end();) {
756  if (git->grid_weight < 1e-13) {
757  git = atomgrid.erase(git);
758  } else {
759  ++git;
760  }
761  }
762  _totalgridsize += atomgrid.size();
763  grid.push_back(atomgrid);
764  } // atoms
765  SortGridpointsintoBlocks(grid);
766  FindSignificantShells(basis);
767  return;
768  }

770  Eigen::VectorXd NumericalIntegration::SSWpartition(int igrid, const Eigen::MatrixXd & rq,const Eigen::MatrixXd& Rij) {
771  const double ass = 0.725;
772  // initialize partition vector to 1.0
773  Eigen::VectorXd p=Eigen::VectorXd::Ones(rq.rows());
774  const double tol_scr = 1e-10;
775  const double leps = 1e-6;
776  // go through centers
777  for (int i = 1; i < rq.rows(); i++) {
778  double rag = rq(i,igrid);
779  // through all other centers (one-directional)
780  for (int j = 0; j < i; j++) {
781  if ((std::abs(p[i]) > tol_scr) || (std::abs(p[j]) > tol_scr)) {
782  double mu = (rag - rq(j,igrid)) * Rij(j,i);
783  if (mu > ass) {
784  p[i] = 0.0;
785  } else if (mu < -ass) {
786  p[j] = 0.0;
787  } else {
788  double sk;
789  if (std::abs(mu) < leps) {
790  sk = -1.88603178008 * mu + 0.5;
791  } else {
792  sk =erf1c(mu);
793  }
794  if (mu > 0.0) sk = 1.0 - sk;
795  p[j] = p[j] * sk;
796  p[i] = p[i] * (1.0 - sk);
797  }
798  }
799  }
800  }
801  return p;
802  }
