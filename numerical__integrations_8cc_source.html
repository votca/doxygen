45 double NumericalIntegration::getExactExchange(
46  const std::string& functional) const {
47 
48  double exactexchange = 0.0;
49  Vxc_Functionals map;
50  std::vector<std::string> strs;
51 
52  boost::split(strs, functional, boost::is_any_of(" "));
53  if (strs.size() > 2) {
54  throw std::runtime_error("Too many functional names");
55  } else if (strs.size() < 1) {
56  throw std::runtime_error("Specify at least one functional");
57  }
58 
59  for (unsigned i = 0; i < strs.size(); i++) {
60 
61  int func_id = map.getID(strs[i]);
62  if (func_id < 0) {
63  exactexchange = 0.0;
64  break;
65  }
66  xc_func_type func;
67  if (xc_func_init(&func, func_id, XC_UNPOLARIZED) != 0) {
68  throw std::runtime_error(
69  (boost::format("Functional %s not found\n") % strs[i]).str());
70  }
71  if (exactexchange > 0 && func.cam_alpha > 0) {
72  throw std::runtime_error(
73  "You have specified two functionals with exact exchange");
74  }
75  exactexchange += func.cam_alpha;
76  xc_func_end(&func);
77  }
78 
79  return exactexchange;
80 }

82 void NumericalIntegration::setXCfunctional(const std::string& functional) {
83 
84  Vxc_Functionals map;
85  std::vector<std::string> strs;
86  tools::Tokenizer tok(functional, " ,\n\t");
87  tok.ToVector(strs);
88  xfunc_id = 0;
89  _use_separate = false;
90  cfunc_id = 0;
91  if (strs.size() == 1) {
92  xfunc_id = map.getID(strs[0]);
93  } else if (strs.size() == 2) {
94  xfunc_id = map.getID(strs[0]);
95  cfunc_id = map.getID(strs[1]);
96  _use_separate = true;
97  } else {
98  throw std::runtime_error(
99  "LIBXC. Please specify one combined or an exchange and a correlation "
100  "functionals");
101  }
102 
103  if (xc_func_init(&xfunc, xfunc_id, XC_UNPOLARIZED) != 0) {
104  throw std::runtime_error(
105  (boost::format("Functional %s not found\n") % strs[0]).str());
106  }
107  if (xfunc.info->kind != 2 && !_use_separate) {
108  throw std::runtime_error(
109  "Your functional misses either correlation or exchange, please specify "
110  "another functional, separated by whitespace");
111  }
112  if (_use_separate) {
113  if (xc_func_init(&cfunc, cfunc_id, XC_UNPOLARIZED) != 0) {
114  throw std::runtime_error(
115  (boost::format("Functional %s not found\n") % strs[1]).str());
116  }
117  if ((xfunc.info->kind + cfunc.info->kind) != 1) {
118  throw std::runtime_error(
119  "Your functionals are not one exchange and one correlation");
120  }
121  }
122  _setXC = true;
123  return;
124 }

126 NumericalIntegration::XC_entry NumericalIntegration::EvaluateXC(
127  double rho, double sigma) const {
128 
129  NumericalIntegration::XC_entry result;
130  switch (xfunc.info->family) {
131  case XC_FAMILY_LDA:
132  xc_lda_exc_vxc(&xfunc, 1, &rho, &result.f_xc, &result.df_drho);
133  break;
134  case XC_FAMILY_GGA:
135  case XC_FAMILY_HYB_GGA:
136  xc_gga_exc_vxc(&xfunc, 1, &rho, &sigma, &result.f_xc, &result.df_drho,
137  &result.df_dsigma);
138  break;
139  }
140  if (_use_separate) {
141  NumericalIntegration::XC_entry temp;
142  // via libxc correlation part only
143  switch (cfunc.info->family) {
144  case XC_FAMILY_LDA:
145  xc_lda_exc_vxc(&cfunc, 1, &rho, &temp.f_xc, &temp.df_drho);
146  break;
147  case XC_FAMILY_GGA:
148  case XC_FAMILY_HYB_GGA:
149  xc_gga_exc_vxc(&cfunc, 1, &rho, &sigma, &temp.f_xc, &temp.df_drho,
150  &temp.df_dsigma);
151  break;
152  }
153 
154  result.f_xc += temp.f_xc;
155  result.df_drho += temp.df_drho;
156  result.df_dsigma += temp.df_dsigma;
157  }
158 
159  return result;
160 }

162 double NumericalIntegration::IntegratePotential(
163  const Eigen::Vector3d& rvector) const {
164 
165  double result = 0.0;
166  assert(_density_set && "Density not calculated");
167  for (const auto& box : _grid_boxes) {
168  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
169  const std::vector<double>& weights = box.getGridWeights();
170  const std::vector<double>& densities = box.getGridDensities();
171  for (unsigned j = 0; j < points.size(); j++) {
172  double charge = -weights[j] * densities[j];
173  double dist = (points[j] - rvector).norm();
174  result += charge / dist;
175  }
176  }
177  return result;
178 }

182 
183  Eigen::Vector3d result = Eigen::Vector3d::Zero();

189  for (unsigned j = 0; j < points.size(); j++) {
190  double charge = -weights[j] * densities[j];

193  }

198 void NumericalIntegration::SortGridpointsintoBlocks(
199  std::vector<std::vector<GridContainers::Cartesian_gridpoint> >& grid) {
200  const double boxsize = 1; // 1 bohr
201 
202  Eigen::Vector3d min =
203  Eigen::Vector3d::Ones() * std::numeric_limits<double>::max();
204  Eigen::Vector3d max =
205  Eigen::Vector3d::Ones() * std::numeric_limits<double>::min();
206 
207  for (unsigned i = 0; i < grid.size(); i++) {
208  for (unsigned j = 0; j < grid[i].size(); j++) {
209  const Eigen::Vector3d& pos = grid[i][j].grid_pos;
210  if (pos[0] > max[0]) {
211  max[0] = pos[0];
212  } else if (pos[0] < min[0]) {
213  min[0] = pos[0];
214  }
215  if (pos[1] > max[1]) {
216  max[1] = pos[1];
217  } else if (pos[1] < min[1]) {
218  min[1] = pos[1];
219  }
220  if (pos[2] > max[2]) {
221  max[2] = pos[2];
222  } else if (pos[2] < min[2]) {
223  min[2] = pos[2];
224  }
225  }
226  }
227 
228  Eigen::Vector3d molextension = max - min;
229  Eigen::Vector3d numberofboxes = molextension / boxsize;
230  Eigen::Vector3d roundednumofbox(std::ceil(numberofboxes[0]),
231  std::ceil(numberofboxes[1]),
232  std::ceil(numberofboxes[2]));
233 
234  std::vector<std::vector<
235  std::vector<std::vector<GridContainers::Cartesian_gridpoint*> > > >
236  boxes;
237  // creating temparray
238  for (unsigned i = 0; i < unsigned(roundednumofbox[0]); i++) {
239  std::vector<
240  std::vector<std::vector<GridContainers::Cartesian_gridpoint*> > >
241  boxes_yz;
242  for (unsigned j = 0; j < unsigned(roundednumofbox[1]); j++) {
243  std::vector<std::vector<GridContainers::Cartesian_gridpoint*> > boxes_z;
244  for (unsigned k = 0; k < unsigned(roundednumofbox[2]); k++) {
245  std::vector<GridContainers::Cartesian_gridpoint*> box;
246  box.reserve(100);
247  boxes_z.push_back(box);
248  }
249  boxes_yz.push_back(boxes_z);
250  }
251  boxes.push_back(boxes_yz);
252  }
253 
254  for (auto& atomgrid : grid) {
255  for (auto& gridpoint : atomgrid) {
256  Eigen::Vector3d pos = gridpoint.grid_pos - min;
257  Eigen::Vector3d index = pos / boxsize;
258  int i_x = int(index[0]);
259  int i_y = int(index[1]);
260  int i_z = int(index[2]);
261  boxes[i_x][i_y][i_z].push_back(&gridpoint);
262  }
263  }
264  for (auto& boxes_xy : boxes) {
265  for (auto& boxes_z : boxes_xy) {
266  for (auto& box : boxes_z) {
267  if (box.size() < 1) {
268  continue;
269  }
270  GridBox gridbox;
271  for (const auto& point : box) {
272  gridbox.addGridPoint(*point);
273  }
274  _grid_boxes.push_back(gridbox);
275  }
276  }
277  }
278  return;
279 }

281 void NumericalIntegration::FindSignificantShells(const AOBasis& basis) {
282  for (GridBox& box : _grid_boxes) {
283  for (const AOShell& store : basis) {
284  const double decay = store.getMinDecay();
285  const Eigen::Vector3d& shellpos = store.getPos();
286  for (const auto& point : box.getGridPoints()) {
287  Eigen::Vector3d dist = shellpos - point;
288  double distsq = dist.squaredNorm();
289  // if contribution is smaller than -ln(1e-10), add shell to list
290  if ((decay * distsq) < 20.7) {
291  box.addShell(&store);
292  break;
293  }
294  }
295  }
296  }
297 
298  std::vector<GridBox> grid_boxes_copy;
299  int combined = 0;
300  // use vector of bool to indicate if a gridbox has already been merged into
301  // another
302  std::vector<bool> Merged = std::vector<bool>(_grid_boxes.size(), false);
303  for (unsigned i = 0; i < _grid_boxes.size(); i++) {
304  if (Merged[i]) {
305  continue;
306  }
307  GridBox box = _grid_boxes[i];
308  if (box.Shellsize() < 1) {
309  continue;
310  }
311  Merged[i] = true;
312  for (unsigned j = i + 1; j < _grid_boxes.size(); j++) {
313  if (GridBox::compareGridboxes(_grid_boxes[i], _grid_boxes[j])) {
314  Merged[j] = true;
315  box.addGridBox(_grid_boxes[j]);
316  combined++;
317  }
318  }
319  grid_boxes_copy.push_back(box);
320  }
321 
322  _totalgridsize = 0;
323  _grid_boxes = grid_boxes_copy;
324  for (auto& box : _grid_boxes) {
325  _totalgridsize += box.size();
326  box.PrepareForIntegration();
327  }
328 }

330 Eigen::VectorXd NumericalIntegration::CalcAOValue_and_Grad(
331  Eigen::MatrixX3d& ao_grad, const GridBox& box,
332  const Eigen::Vector3d& point) const {
333  Eigen::VectorXd ao = Eigen::VectorXd::Zero(box.Matrixsize());
334  const std::vector<GridboxRange>& aoranges = box.getAOranges();
335  const std::vector<const AOShell*>& shells = box.getShells();
336  for (unsigned j = 0; j < box.Shellsize(); ++j) {
337  Eigen::Block<Eigen::MatrixX3d> grad_block =
338  ao_grad.block(aoranges[j].start, 0, aoranges[j].size, 3);
339  Eigen::VectorBlock<Eigen::VectorXd> ao_block =
340  ao.segment(aoranges[j].start, aoranges[j].size);
341  shells[j]->EvalAOspace(ao_block, grad_block, point);
342  }
343  return ao;

346 Mat_p_Energy NumericalIntegration::IntegrateVXC(
347  const Eigen::MatrixXd& density_matrix) const {
348 
349  int nthreads = OPENMP::getMaxThreads();
350 
351  std::vector<Eigen::MatrixXd> vxc_thread = std::vector<Eigen::MatrixXd>(
352  nthreads,
353  Eigen::MatrixXd::Zero(density_matrix.rows(), density_matrix.cols()));
354  std::vector<double> Exc_thread = std::vector<double>(nthreads, 0.0);
355 
356 #pragma omp parallel for schedule(guided)
357  for (unsigned i = 0; i < _grid_boxes.size(); ++i) {
358 
359  double EXC_box = 0.0;
360  const GridBox& box = _grid_boxes[i];
361  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
362  const Eigen::MatrixXd DMAT_symm = DMAT_here + DMAT_here.transpose();
363  double cutoff = 1.e-40 / density_matrix.rows() / density_matrix.rows();
364  if (DMAT_here.cwiseAbs2().maxCoeff() < cutoff) {
365  continue;
366  }
367  Eigen::MatrixXd Vxc_here =
368  Eigen::MatrixXd::Zero(DMAT_here.rows(), DMAT_here.cols());
369  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
370  const std::vector<double>& weights = box.getGridWeights();
371 
372  // iterate over gridpoints
373  for (unsigned p = 0; p < box.size(); p++) {
374 
375  Eigen::MatrixX3d ao_grad = Eigen::MatrixX3d::Zero(box.Matrixsize(), 3);
376  Eigen::VectorXd ao = CalcAOValue_and_Grad(ao_grad, box, points[p]);
377  const double rho = 0.5 * (ao.transpose() * DMAT_symm * ao).value();
378  const double weight = weights[p];
379  if (rho * weight < 1.e-20) {
380  continue; // skip the rest, if density is very small
381  }
382  const Eigen::Vector3d rho_grad = ao.transpose() * DMAT_symm * ao_grad;
383  const double sigma = (rho_grad.transpose() * rho_grad).value();
384  const Eigen::VectorXd grad = ao_grad * rho_grad;
385  NumericalIntegration::XC_entry xc = EvaluateXC(rho, sigma);
386  EXC_box += weight * rho * xc.f_xc;
387  auto addXC = weight * (0.5 * xc.df_drho * ao + 2.0 * xc.df_dsigma * grad);
388  Vxc_here.noalias() += addXC * ao.transpose();
389  }
390  box.AddtoBigMatrix(vxc_thread[OPENMP::getThreadId()], Vxc_here);
391  Exc_thread[OPENMP::getThreadId()] += EXC_box;
392  }
393 
394  double EXC = std::accumulate(Exc_thread.begin(), Exc_thread.end(), 0.0);
395  Eigen::MatrixXd Vxc = std::accumulate(
396  vxc_thread.begin(), vxc_thread.end(),
397  Eigen::MatrixXd::Zero(density_matrix.rows(), density_matrix.cols())
398  .eval());
399 
400  Mat_p_Energy Oxc(EXC, Vxc + Vxc.transpose());
401  return Oxc;
402 }

404 double NumericalIntegration::IntegrateDensity(
405  const Eigen::MatrixXd& density_matrix) {
406 
407  int nthreads = OPENMP::getMaxThreads();
408  std::vector<double> N_thread = std::vector<double>(nthreads, 0.0);
409 
410 #pragma omp parallel for schedule(guided)
411  for (unsigned i = 0; i < _grid_boxes.size(); ++i) {
412  double N_box = 0.0;
413  GridBox& box = _grid_boxes[i];
414  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
415  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
416  const std::vector<double>& weights = box.getGridWeights();
417  box.prepareDensity();
418  // iterate over gridpoints
419  for (unsigned p = 0; p < box.size(); p++) {
420  Eigen::VectorXd ao = CalcAOValues(box, points[p]);
421  double rho = (ao.transpose() * DMAT_here * ao)(0, 0);
422  box.addDensity(rho);
423  N_box += rho * weights[p];
424  }
425  N_thread[OPENMP::getThreadId()] += N_box;
426  }
427  double N = std::accumulate(N_thread.begin(), N_thread.end(), 0.0);
428  _density_set = true;
429  return N;
430 }

445 Gyrationtensor NumericalIntegration::IntegrateGyrationTensor(
446  const Eigen::MatrixXd& density_matrix) {
447 
448  int nthreads = OPENMP::getMaxThreads();
449  std::vector<double> N_thread = std::vector<double>(nthreads, 0.0);
450  std::vector<Eigen::Vector3d> centroid_thread =
451  std::vector<Eigen::Vector3d>(nthreads, Eigen::Vector3d::Zero());
452  std::vector<Eigen::Matrix3d> gyration_thread =
453  std::vector<Eigen::Matrix3d>(nthreads, Eigen::Matrix3d::Zero());
454 
455 #pragma omp parallel for schedule(guided)
456  for (unsigned i = 0; i < _grid_boxes.size(); ++i) {
457  double N_box = 0.0;
458  Eigen::Vector3d centroid_box = Eigen::Vector3d::Zero();
459  Eigen::Matrix3d gyration_box = Eigen::Matrix3d::Zero();
460  GridBox& box = _grid_boxes[i];
461  const Eigen::MatrixXd DMAT_here = box.ReadFromBigMatrix(density_matrix);
462  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
463  const std::vector<double>& weights = box.getGridWeights();
464  box.prepareDensity();
465  // iterate over gridpoints
466  for (unsigned p = 0; p < box.size(); p++) {
467  Eigen::VectorXd ao = CalcAOValues(box, points[p]);
468  double rho = (ao.transpose() * DMAT_here * ao).value();
469  box.addDensity(rho);
470  N_box += rho * weights[p];
471  centroid_box += rho * weights[p] * points[p];
472  gyration_box += rho * weights[p] * points[p] * points[p].transpose();
473  }
474  N_thread[OPENMP::getThreadId()] += N_box;
475  centroid_thread[OPENMP::getThreadId()] += centroid_box;
476  gyration_thread[OPENMP::getThreadId()] += gyration_box;
477  }
478  double N = std::accumulate(N_thread.begin(), N_thread.end(), 0.0);
479  Eigen::Vector3d centroid =
480  std::accumulate(centroid_thread.begin(), centroid_thread.end(),
481  Eigen::Vector3d::Zero().eval());
482  Eigen::Matrix3d gyration =
483  std::accumulate(gyration_thread.begin(), gyration_thread.end(),
484  Eigen::Matrix3d::Zero().eval());
485 
486  _density_set = true;
487  // Normalize
488  centroid = centroid / N;
489  gyration = gyration / N;
490  gyration = gyration - centroid * centroid.transpose();
491  Gyrationtensor gyro;
492  gyro.mass = N;
493  gyro.centroid = centroid;
494  gyro.gyration = gyration;
495 
496  return gyro;
497 }

512 Eigen::MatrixXd NumericalIntegration::IntegratePotential(
513  const AOBasis& externalbasis) const {
514  Eigen::MatrixXd Potential = Eigen::MatrixXd::Zero(
515  externalbasis.AOBasisSize(), externalbasis.AOBasisSize());
516 
517  assert(_density_set && "Density not calculated");
518  for (const auto& box : _grid_boxes) {
519  const std::vector<Eigen::Vector3d>& points = box.getGridPoints();
520  const std::vector<double>& weights = box.getGridWeights();
521  const std::vector<double>& densities = box.getGridDensities();
522  for (unsigned j = 0; j < points.size(); j++) {
523  double weighteddensity = weights[j] * densities[j];
524  if (weighteddensity < 1e-12) {
525  continue;
526  }
527  AOMultipole esp;
528  esp.FillPotential(externalbasis, points[j]);
529  Potential += weighteddensity * esp.Matrix();
530  }
531  }
532  return Potential;
533 }

535 Eigen::MatrixXd NumericalIntegration::CalcInverseAtomDist(
536  const QMMolecule& atoms) const {
537  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(atoms.size(), atoms.size());

541  for (int j = 0; j < i; ++j) {

549 int NumericalIntegration::UpdateOrder(LebedevGrid& sphericalgridofElement,
550  int maxorder,
551  std::vector<double>& PruningIntervals,
552  double r) const {
553  int order;
554  int maxindex = sphericalgridofElement.getIndexFromOrder(maxorder);
555  if (maxindex == 1) {
556  // smallest possible grid anyway, nothing to do
557  order = maxorder;
558  } else if (maxindex == 2) {
559  // only three intervals
560  if (r < PruningIntervals[0]) {
561  order = sphericalgridofElement.getOrderFromIndex(1); // 1;
562  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[3])) {
563  order = sphericalgridofElement.getOrderFromIndex(2);
564  } else {
565  order = sphericalgridofElement.getOrderFromIndex(1);
566  } // maxorder == 2
567  } else {
568  // five intervals
569  if (r < PruningIntervals[0]) {
570  order = sphericalgridofElement.getOrderFromIndex(2);
571  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[1])) {
572  order = sphericalgridofElement.getOrderFromIndex(4);
573  } else if ((r >= PruningIntervals[1]) && (r < PruningIntervals[2])) {
574  order =
575  sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 4));
576  } else if ((r >= PruningIntervals[2]) && (r < PruningIntervals[3])) {
577  order = maxorder;
578  } else {
579  order =
580  sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 1));
581  }
582  }
583  return order;

586 GridContainers::Cartesian_gridpoint
587  NumericalIntegration::CreateCartesianGridpoint(
588  const Eigen::Vector3d& atomA_pos,
589  GridContainers::radial_grid& radial_grid,
590  GridContainers::spherical_grid& spherical_grid, int i_rad,
591  int i_sph) const {
592  GridContainers::Cartesian_gridpoint gridpoint;
593  double p = spherical_grid.phi[i_sph];
594  double t = spherical_grid.theta[i_sph];
595  const Eigen::Vector3d s =
596  Eigen::Vector3d{sin(p) * cos(t), sin(p) * sin(t), cos(p)};
597  double r = radial_grid.radius[i_rad];
598  gridpoint.grid_pos = atomA_pos + r * s;
599  gridpoint.grid_weight =
600  radial_grid.weight[i_rad] * spherical_grid.weight[i_sph];
601  return gridpoint;
602 }

604 Eigen::MatrixXd NumericalIntegration::CalcDistanceAtomsGridpoints(
605  const QMMolecule& atoms,
606  std::vector<GridContainers::Cartesian_gridpoint>& atomgrid) const {
607  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(atoms.size(), atomgrid.size());
608 #pragma omp parallel for
609  for (int i = 0; i < atoms.size(); ++i) {
610  const Eigen::Vector3d& atom_pos = atoms[i].getPos();
611  for (unsigned j = 0; j < atomgrid.size(); ++j) {
612  const auto& gridpoint = atomgrid[j];
613  result(i, j) = (atom_pos - gridpoint.grid_pos).norm();
614  }
615  }
616  return result;
617 }

619 void NumericalIntegration::SSWpartitionAtom(
620  const QMMolecule& atoms,
621  std::vector<GridContainers::Cartesian_gridpoint>& atomgrid, int i_atom,
622  const Eigen::MatrixXd& Rij) const {
623  Eigen::MatrixXd AtomGridDist = CalcDistanceAtomsGridpoints(atoms, atomgrid);
624 
625 #pragma omp parallel for schedule(guided)
626  for (int i_grid = 0; i_grid < int(atomgrid.size()); i_grid++) {
627  Eigen::VectorXd p = SSWpartition(AtomGridDist.col(i_grid), Rij);
628  // check weight sum
629  double wsum = p.sum();
630  if (wsum != 0.0) {
631  // update the weight of this grid point
632  atomgrid[i_grid].grid_weight *= p[i_atom] / wsum;
633  } else {
634  std::cerr << "\nSum of partition weights of grid point " << i_grid
635  << " of atom " << i_atom << " is zero! ";
636  throw std::runtime_error("\nThis should never happen!");
637  }
638  } // partition weight for each gridpoint
639 }

641 void NumericalIntegration::GridSetup(const std::string& type,
642  const QMMolecule& atoms,
643  const AOBasis& basis) {
644  GridContainers initialgrids;
645  // get radial grid per element
646  EulerMaclaurinGrid radialgridofElement;
647  initialgrids.radial_grids = radialgridofElement.CalculateAtomicRadialGrids(
648  basis, atoms, type); // this checks out 1:1 with NWChem results! AWESOME
649  LebedevGrid sphericalgridofElement;
650  initialgrids.spherical_grids =
651  sphericalgridofElement.CalculateSphericalGrids(atoms, type);
652 
653  // for the partitioning, we need all inter-center distances later, stored in
654  // matrix
655  Eigen::MatrixXd Rij = CalcInverseAtomDist(atoms);
656  std::vector<std::vector<GridContainers::Cartesian_gridpoint> > grid;
657 
658  for (int i_atom = 0; i_atom < atoms.size(); ++i_atom) {
659  const QMAtom& atom = atoms[i_atom];
660 
661  const Eigen::Vector3d& atomA_pos = atom.getPos();
662  const std::string& name = atom.getElement();
663  GridContainers::radial_grid radial_grid =
664  initialgrids.radial_grids.at(name);
665  GridContainers::spherical_grid spherical_grid =
666  initialgrids.spherical_grids.at(name);
667 
668  // maximum order (= number of points) in spherical integration grid
669  int maxorder = sphericalgridofElement.Type2MaxOrder(name, type);
670  // for pruning of integration grid, get interval boundaries for this element
671  std::vector<double> PruningIntervals =
672  radialgridofElement.CalculatePruningIntervals(name);
673  int current_order = 0;
674  // for each radial value
675  std::vector<GridContainers::Cartesian_gridpoint> atomgrid;
676  for (int i_rad = 0; i_rad < radial_grid.radius.size(); i_rad++) {
677  double r = radial_grid.radius[i_rad];
678 
679  // which Lebedev order for this point?
680  int order =
681  UpdateOrder(sphericalgridofElement, maxorder, PruningIntervals, r);
682  // get new spherical grid, if order changed
683  if (order != current_order) {
684  spherical_grid = sphericalgridofElement.CalculateUnitSphereGrid(order);
685  current_order = order;
686  }
687 
688  for (int i_sph = 0; i_sph < spherical_grid.phi.size(); i_sph++) {
689  GridContainers::Cartesian_gridpoint gridpoint =
690  CreateCartesianGridpoint(atomA_pos, radial_grid, spherical_grid,
691  i_rad, i_sph);
692  atomgrid.push_back(gridpoint);
693  } // spherical gridpoints
694  } // radial gridpoint
695 
696  SSWpartitionAtom(atoms, atomgrid, i_atom, Rij);
697  // now remove points from the grid with negligible weights
698  std::vector<GridContainers::Cartesian_gridpoint> atomgrid_cleanedup;
699  for (const auto& point : atomgrid) {
700  if (point.grid_weight > 1e-13) {
701  atomgrid_cleanedup.push_back(point);
702  }
703  }
704  grid.push_back(atomgrid_cleanedup);
705  } // atoms
706  SortGridpointsintoBlocks(grid);
707  FindSignificantShells(basis);
708  return;
709 }

711 Eigen::VectorXd NumericalIntegration::SSWpartition(
712  const Eigen::VectorXd& rq_i, const Eigen::MatrixXd& Rij) const {
713  const double ass = 0.725;
714  // initialize partition vector to 1.0
715  Eigen::VectorXd p = Eigen::VectorXd::Ones(rq_i.size());
716  const double tol_scr = 1e-10;
717  const double leps = 1e-6;
718  // go through centers
719  for (int i = 1; i < rq_i.size(); i++) {
720  double rag = rq_i(i);
721  // through all other centers (one-directional)
722  for (int j = 0; j < i; j++) {
723  if ((std::abs(p[i]) > tol_scr) || (std::abs(p[j]) > tol_scr)) {
724  double mu = (rag - rq_i(j)) * Rij(j, i);
725  if (mu > ass) {
726  p[i] = 0.0;
727  } else if (mu < -ass) {
728  p[j] = 0.0;
729  } else {
730  double sk;
731  if (std::abs(mu) < leps) {
732  sk = -1.88603178008 * mu + 0.5;
733  } else {
734  sk = erf1c(mu);
735  }
736  if (mu > 0.0) {
737  sk = 1.0 - sk;
738  }
739  p[j] = p[j] * sk;
740  p[i] = p[i] * (1.0 - sk);
741  }
742  }
743  }
744  }
745  return p;
746 }

748 double NumericalIntegration::erf1c(double x) const {
749  const static double alpha_erf1 = 1.0 / 0.30;
750  return 0.5 * std::erfc(std::abs(x / (1.0 - x * x)) * alpha_erf1);
751 }
