41 void GMXTrajectoryWriter::Write(Topology *conf) {
42  static Index step = 0;
43  Index N = conf->BeadCount();
44  t_trxframe frame;
45  rvec *x = new rvec[N];
46  rvec *v = nullptr;
47  rvec *f = nullptr;
48  Eigen::Matrix3d box = conf->getBox();
49 
50  frame.natoms = (int)N;
51  frame.bTime = true;
52  frame.time = real(conf->getTime());
53  frame.bStep = true;
54  frame.step = conf->getStep();
55  frame.x = x;
56  frame.bLambda = false;
57  frame.bAtoms = false;
58  frame.bPrec = false;
59  frame.bX = true;
60  frame.bF = conf->HasForce();
61  frame.bBox = true;
62  frame.bV = conf->HasVel();
63 
64  for (Index i = 0; i < 3; i++) {
65  for (Index j = 0; j < 3; j++) {
66  frame.box[j][i] = real(box(i, j));
67  }
68  }
69 
70  for (Index i = 0; i < N; ++i) {
71  Eigen::Vector3d pos = conf->getBead(i)->getPos();
72  x[i][0] = real(pos.x());
73  x[i][1] = real(pos.y());
74  x[i][2] = real(pos.z());
75  }
76 
77  if (frame.bV) {
78  v = new rvec[N];
79  for (Index i = 0; i < N; ++i) {
80  frame.v = v;
81  Eigen::Vector3d vel = conf->getBead(i)->getVel();
82  v[i][0] = real(vel.x());
83  v[i][1] = real(vel.y());
84  v[i][2] = real(vel.z());
85  }
86  }
87  if (frame.bF) {
88  f = new rvec[N];
89  for (Index i = 0; i < N; ++i) {
90  frame.f = f;
91  Eigen::Vector3d force = conf->getBead(i)->getF();
92  f[i][0] = real(force.x());
93  f[i][1] = real(force.y());
94  f[i][2] = real(force.z());
95  }
96  }
97 
98  write_trxframe(_file, &frame, nullptr);
99 
100  step++;
101  delete[] x;
102  if (frame.bV) {
103  delete[] v;
104  }
105  if (frame.bF) {
106  delete[] f;
107  }
108 }
