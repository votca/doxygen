41  rvec *v;
42  rvec *f;

44

46  frame.bTime = true;
47  frame.time = conf->getTime();
48  frame.bStep = true;
49  frame.step = conf->getStep();;
50  frame.x = x;
51  frame.bTitle=false;
52  frame.bLambda=false;
53  frame.bAtoms=false;
54  frame.bPrec=false;
55  frame.bX = true;

59 
60  for(int i=0; i<3; i++)
61  for(int j=0; j<3; j++)
62  frame.box[j][i] = box[i][j];
63 
64 for(int i=0; i<N; ++i) {
65  vec pos = conf->getBead(i)->getPos();
66  x[i][0] = pos.getX();
67  x[i][1] = pos.getY();
68  x[i][2] = pos.getZ();
69  }
70 
71 if (frame.bV){

73  for(int i=0; i<N; ++i) {
74  frame.v = v;

79  }
80 }

89  }
90 }
91 
92  write_trxframe(_file, &frame, NULL);
93 
94  step++;

98 }
