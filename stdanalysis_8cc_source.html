52 void StdAnalysis::Help(string cmd, vector<string> &args)

61  if(cmd == "cor") {
62  cout << "cor <file> <selection>\n"
63  << "Calculate linear correlation coefficient of the first item in selection with all the other items\n"
64  << "WARNING: for evaluating correlations in the system, it is not sufficient to calculate the "
65  << "linear correlation coefficient, 2D histograms with data from the vals command should be used instead!\n";
66  }
67  if(cmd == "autocor") {

71  }
72  if(cmd == "list") {
