64  cout << "cor <file> <selection>\n"

67  << "WARNING: for evaluating correlations in the system, it is not "

69  << "linear correlation coefficient, 2D histograms with data from the "
70  "vals command should be used instead!\n";

72  if (cmd == "autocor") {

75  "The output is periodic since FFTW3 is used to "
