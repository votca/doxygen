34 void IAnalyze::Initialize(const tools::Property &user_options) {
35 
36  tools::Property options =
37  LoadDefaultsAndUpdateWithUserOptions("xtp", user_options);
38 
39  std::string statestrings = options.get(".states").as<std::string>();
40  tools::Tokenizer tok(statestrings, ",\n\t ");

42  tok.ToVector(string_vec);
43  for (std::string &state : string_vec) {
44  _states.push_back(QMStateType(state));

46 
47  _resolution_logJ2 = options.get(".resolution_logJ2").as<double>();
48  if (options.get(".do_pairtype").as<bool>()) {
49  _do_pairtype = true;

51  if (_store_stdstring.find("Hopping") != std::string::npos) {
52  _pairtype.push_back(QMPair::Hopping);

54  if (_store_stdstring.find("Excitoncl") != std::string::npos) {
55  _pairtype.push_back(QMPair::Excitoncl);
56  }
57  if (!_pairtype.size()) {
58  std::cout << "\n... ... No pairtypes recognized will output all pairs. ";
59  _do_pairtype = false;
60  }
61  }
62  if (options.get(".do_resolution_spatial").as<bool>()) {
63  _resolution_spatial = options.get(".resolution_spatial").as<double>();
64  if (_resolution_spatial != 0.0) {
65  _do_IRdependence = true;
66  }
67  }

104 void IAnalyze::IHist(Topology &top, QMStateType state) {
105  QMNBList &nblist = top.NBList();
106 
107  // Collect J2s from pairs
108  std::vector<double> J2s;
109  for (QMPair *pair : nblist) {
110  if (_do_pairtype) {
111  QMPair::PairType pairtype = pair->getType();
112  if (!(std::find(_pairtype.begin(), _pairtype.end(), pairtype) !=
113  _pairtype.end())) {
114  continue;
115  }
116  }
117  double test =
118  pair->getJeff2(state) * tools::conv::hrt2ev * tools::conv::hrt2ev;
119  if (test <= 0) {
120  continue;
121  } // avoid -inf in output
122  double J2 = std::log10(test);
123  J2s.push_back(J2);
124  }
125 
126  if (J2s.size() < 1) {
127  std::cout << "WARNING:" + state.ToLongString() +
128  " Couplings are all zero. You have not yet imported them! "
129  << std::endl;
130  return;
131  }
132 
133  double MAX = *std::max_element(J2s.begin(), J2s.end());
134  double MIN = *std::min_element(J2s.begin(), J2s.end());
135  double sum = std::accumulate(J2s.begin(), J2s.end(), 0.0);
136  double AVG = sum / double(J2s.size());
137  double sq_sum = std::inner_product(J2s.begin(), J2s.end(), J2s.begin(), 0.0);
138  double STD = std::sqrt(sq_sum / double(J2s.size()) - AVG * AVG);
139  // Prepare bins
140  Index BIN = Index((MAX - MIN) / _resolution_logJ2 + 0.5) + 1;
141 
142  tools::HistogramNew hist;
143  hist.Initialize(MIN, MAX, BIN);
144  hist.ProcessRange<std::vector<double>::iterator>(J2s.begin(), J2s.end());
145  tools::Table &tab = hist.data();
146  std::string comment =
147  (boost::format("IANALYZE: PAIR-INTEGRAL J2 HISTOGRAM \n # AVG %1$4.7f "
148  "STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") %
149  AVG % STD % MIN % MAX)
150  .str();
151  std::string filename = "ianalyze.ihist_" + state.ToString() + ".out";
152  tab.set_comment(comment);
153  tab.flags() = std::vector<char>(tab.size(), ' ');
154  tab.Save(filename);
155 }

157 void IAnalyze::IRdependence(Topology &top, QMStateType state) {
158 
159  QMNBList &nblist = top.NBList();
160 
161  // Collect J2s from pairs
162  std::vector<double> J2s;
163  J2s.reserve(nblist.size());
164  std::vector<double> distances;
165  distances.reserve(nblist.size());
166 
167  for (QMPair *pair : nblist) {
168  double test =
169  pair->getJeff2(state) * tools::conv::hrt2ev * tools::conv::hrt2ev;
170  if (test <= 0) {
171  continue;
172  } // avoid -inf in output
173  double J2 = std::log10(test);
174  double distance = pair->R().norm() * tools::conv::bohr2nm;
175  distances.push_back(distance);
176  J2s.push_back(J2);
177  }
178 
179  double MAXR = *std::max_element(distances.begin(), distances.end());
180  double MINR = *std::min_element(distances.begin(), distances.end());
181 
182  // Prepare R bins
183  Index pointsR = Index((MAXR - MINR) / _resolution_spatial);
184  std::vector<std::vector<double> > rJ2;
185  rJ2.resize(pointsR);
186 
187  // Loop over distance
188  for (Index i = 0; i < pointsR; ++i) {
189  double thisMINR = MINR + double(i) * _resolution_spatial;
190  double thisMAXR = MINR + double(i + 1) * _resolution_spatial;
191  // now count Js that lie within this R range
192  for (Index j = 0; j < Index(J2s.size()); ++j) {
193  if (thisMINR < distances[j] && distances[j] < thisMAXR) {
194  rJ2[i].push_back(J2s[j]);
195  }
196  }
197  }
198 
199  tools::Table tab;
200  tab.SetHasYErr(true);
201  tab.resize(pointsR);
202 
203  // make plot values
204  for (Index i = 0; i < Index(rJ2.size()); i++) {
205  const std::vector<double> &vec = rJ2[i];
206  double sum = std::accumulate(vec.begin(), vec.end(), 0.0);
207  double AVG = sum / double(vec.size());
208  double thisR = MINR + (double(i) + 0.5) * _resolution_spatial;
209  double sq_sum =
210  std::inner_product(vec.begin(), vec.end(), vec.begin(), 0.0);
211  double STD = std::sqrt(sq_sum / double(vec.size()) - AVG * AVG);
212  tab.set(i, thisR, AVG, ' ', STD);
213  }
214  std::string filename = "ianalyze.ispatial_" + state.ToString() + ".out";
215  std::string comment =
216  "# IANALYZE: SPATIAL DEPENDENCE OF log10(J2) "
217  "[r[nm],log10(J)[eV^2],error]";
218  tab.setErrorDetails(comment);
219  tab.Save(filename);
220 }
