63 bool GROReader::NextFrame(Topology &top)
64 {
65  string tmp;
66  getline(_fl, tmp);//title
67  getline(_fl, tmp);//number atoms
68  int natoms = atoi(tmp.c_str());
69  if(!_topology && natoms != top.BeadCount())
70  throw std::runtime_error("number of beads in topology and trajectory differ");
71 
72  for(int i=0;i<natoms; i++) {
73  string line;
74  getline(_fl, line);
75  string resNum,resName, atName, x,y,z;
76  try {
77  resNum= string(line,0,5); // %5i
78  resName = string(line,5,5); //%5s
79  atName = string(line,10,5); // %5s
80  //atNum= string(line,15,5); // %5i not needed
81  x = string(line,20,8); // %8.3f
82  y = string(line,28,8); // %8.3f
83  z = string(line,36,8); // %8.3f
84  } catch (std::out_of_range& err) {
85  throw std::runtime_error("Misformated gro file");
86  }
87  boost::algorithm::trim(atName);
88  boost::algorithm::trim(resName);
89  boost::algorithm::trim(resNum);
90  boost::algorithm::trim(x);
91  boost::algorithm::trim(y);
92  boost::algorithm::trim(z);
93  string vx,vy,vz;
94  bool hasVel=true;
95  try {
96  vx = string(line,44,8); // %8.4f
97  vy = string(line,52,8); // %8.4f
98  vz = string(line,60,8); // %8.4f
99  } catch (std::out_of_range& err) {
100  hasVel=false;
101  }
102 
103  Bead *b;
104  if(_topology){
105  int resnr = boost::lexical_cast<int>(resNum);
106  if(resnr >= top.ResidueCount()) {
107  if (top.ResidueCount()==0) //gro resnr start with 1 but VOTCA starts with 0
108  top.CreateResidue("ZERO"); // create 0 res, to allow to keep gro numbering
109  top.CreateResidue(resName);
110  }
111  //this is not correct, but still better than no type at all!
112  BeadType *type = top.GetOrCreateBeadType(atName);
113 
114  b = top.CreateBead(1, atName, type, resnr, 1., 0.);
115  } else {
116  b = top.getBead(i);
117  }
118 
119  b->setPos(vec(
120  boost::lexical_cast<double>(x),
121  boost::lexical_cast<double>(y),
122  boost::lexical_cast<double>(z)
123  ));
124  if (hasVel) {
125  boost::algorithm::trim(vx);
126  boost::algorithm::trim(vy);
127  boost::algorithm::trim(vz);
128  b->setVel(vec(
129  boost::lexical_cast<double>(vx),
130  boost::lexical_cast<double>(vy),
131  boost::lexical_cast<double>(vz)
132  ));
133  }
134  }
135 
136  getline(_fl, tmp); //read box line
137  if(_fl.eof())
138  throw std::runtime_error("unexpected end of file in poly file, when boxline");
139  Tokenizer tok(tmp, " ");
140  vector<double> fields;
141  tok.ConvertToVector<double>(fields);
142  matrix box;
143  if ( fields.size() == 3 ) {
144  box.ZeroMatrix();
145  for (int i=0;i<3;i++) box[i][i] = fields[i];
146  } else if ( fields.size() == 9 ) {
147  box[0][0]= fields[0];
148  box[1][1]= fields[1];
149  box[2][2]= fields[2];
150  box[1][0]= fields[3];
151  box[2][0]= fields[4];
152  box[0][1]= fields[5];
153  box[2][1]= fields[6];
154  box[0][2]= fields[7];
155  box[1][2]= fields[8];
156  } else {
157  throw std::runtime_error("Error while reading box (last) line");
158  }
159  top.setBox(box);
160 
161  _fl.close();
162 
163  cout << "WARNING: topology created from .gro file, masses and charges are wrong!\n";
164 
165  return true;
166 }
