63 bool GROReader::NextFrame(Topology &top)
64 {
65  string tmp;
66  getline(_fl, tmp);//title
67  if (_fl.eof()) {
68  return !_fl.eof();
69  }
70  getline(_fl, tmp);//number atoms
71  int natoms = atoi(tmp.c_str());
72  if(!_topology && natoms != top.BeadCount())
73  throw std::runtime_error("number of beads in topology and trajectory differ");
74 
75  for(int i=0;i<natoms; i++) {
76  string line;
77  getline(_fl, line);
78  string resNum,resName, atName, x,y,z;
79  try {
80  resNum= string(line,0,5); // %5i
81  resName = string(line,5,5); //%5s
82  atName = string(line,10,5); // %5s
83  //atNum= string(line,15,5); // %5i not needed
84  x = string(line,20,8); // %8.3f
85  y = string(line,28,8); // %8.3f
86  z = string(line,36,8); // %8.3f
87  } catch (std::out_of_range& err) {
88  throw std::runtime_error("Misformated gro file");
89  }
90  boost::algorithm::trim(atName);
91  boost::algorithm::trim(resName);
92  boost::algorithm::trim(resNum);
93  boost::algorithm::trim(x);
94  boost::algorithm::trim(y);
95  boost::algorithm::trim(z);
96  string vx,vy,vz;
97  bool hasVel=true;
98  try {
99  vx = string(line,44,8); // %8.4f
100  vy = string(line,52,8); // %8.4f
101  vz = string(line,60,8); // %8.4f
102  } catch (std::out_of_range& err) {
103  hasVel=false;
104  }
105 
106  Bead *b;
107  if(_topology){
108  int resnr = boost::lexical_cast<int>(resNum);
109  if (resnr < 1)
110  throw std::runtime_error("Misformated gro file, resnr has to be > 0");
111  //TODO: fix the case that resnr is not in ascending order
112  if(resnr > top.ResidueCount()) {
113  while ((resnr-1)>top.ResidueCount()){ //gro resnr should start with 1 but accept sloppy files
114  top.CreateResidue("DUMMY"); // create dummy residue, hopefully it will never show
115  cout << "Warning: residue numbers not continous, create DUMMY residue with nr " << top.ResidueCount() << endl;
116  }
117  top.CreateResidue(resName);
118  }
119  //this is not correct, but still better than no type at all!
120  BeadType *type = top.GetOrCreateBeadType(atName);
121 
122  // res -1 as internal number starts with 0
123  b = top.CreateBead(1, atName, type, resnr-1, 1., 0.);
124  } else {
125  b = top.getBead(i);
126  }
127 
128  b->setPos(vec(
129  boost::lexical_cast<double>(x),
130  boost::lexical_cast<double>(y),
131  boost::lexical_cast<double>(z)
132  ));
133  if (hasVel) {
134  boost::algorithm::trim(vx);
135  boost::algorithm::trim(vy);
136  boost::algorithm::trim(vz);
137  b->setVel(vec(
138  boost::lexical_cast<double>(vx),
139  boost::lexical_cast<double>(vy),
140  boost::lexical_cast<double>(vz)
141  ));
142  }
143  }
144 
145  getline(_fl, tmp); //read box line
146  if(_fl.eof())
147  throw std::runtime_error("unexpected end of file in poly file, when boxline");
148  Tokenizer tok(tmp, " ");
149  vector<double> fields;
150  tok.ConvertToVector<double>(fields);
151  matrix box;
152  if ( fields.size() == 3 ) {
153  box.ZeroMatrix();
154  for (int i=0;i<3;i++) box[i][i] = fields[i];
155  } else if ( fields.size() == 9 ) {
156  box[0][0]= fields[0];
157  box[1][1]= fields[1];
158  box[2][2]= fields[2];
159  box[1][0]= fields[3];
160  box[2][0]= fields[4];
161  box[0][1]= fields[5];
162  box[2][1]= fields[6];
163  box[0][2]= fields[7];
164  box[1][2]= fields[8];
165  } else {
166  throw std::runtime_error("Error while reading box (last) line");
167  }
168  top.setBox(box);
169 
170  if (_topology) {
171  cout << "WARNING: topology created from .gro file, masses and charges are wrong!\n";
172  }
173 
174  return !_fl.eof();
175 }
