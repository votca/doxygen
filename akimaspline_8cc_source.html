27 void AkimaSpline::Interpolate(Eigen::VectorXd &x, Eigen::VectorXd &y) {
28  if (x.size() != y.size()) {
29  throw std::invalid_argument(
30  "error in AkimaSpline::Interpolate : sizes of vectors x and y do not "
31  "match");
32  }
33 
34  // Akima splines require at least 4 points
35  if (x.size() < 4) {
36  throw std::invalid_argument(
37  "error in AkimaSpline::Interpolate : vectors x and y have to contain "
38  "at least 4 points");
39  }
40 
41  const long int N = x.size();
42 
43  // copy the grid points into f
44  _r = x;
45 
46  // initialize vectors p1,p2,p3,p4 and t
47  p0 = Eigen::VectorXd::Zero(N);
48  p1 = Eigen::VectorXd::Zero(N);
49  p2 = Eigen::VectorXd::Zero(N);
50  p3 = Eigen::VectorXd::Zero(N);
51  t = Eigen::VectorXd::Zero(N);
52 
53  double m1, m2, m3, m4;
54 
55  double temp, g0, g1, g2, x1, x2, y1, y2, x4, x5, y4, y5, m5;
56 
57  // boundary conditions
58  // >> determine t(0), t(1) and t(N-2), t(N-1)
59  switch (_boundaries) {
60  case splineNormal:
61  // Akima method: estimation of two more points on each side using a
62  // degree two polyomial
63  // resulting slopes t(0), t(1), t(N-2), t(N-1) are directly calculated
64 
65  // left side: t(0), t(1)
66  temp = (x(1) - x(0)) / (x(2) - x(0));
67  temp = temp * temp;
68  g0 = y(0);
69  g1 = ((y(1) - y(0)) - temp * (y(2) - y(0))) /
70  ((x(1) - x(0)) - temp * (x(2) - x(0)));
71  g2 = ((y(2) - y(0)) - g1 * (x(2) - x(0))) /
72  ((x(2) - x(0)) * (x(2) - x(0)));
73  x1 = x(0) - (x(2) - x(0));
74  x2 = x(1) - (x(2) - x(0));
75  y1 = g0 + g1 * (x1 - x(0)) + g2 * (x1 - x(0)) * (x1 - x(0));
76  y2 = g0 + g1 * (x2 - x(0)) + g2 * (x2 - x(0)) * (x2 - x(0));
77  m1 = (y2 - y1) / (x2 - x1);
78  m2 = (y(0) - y2) / (x(0) - x2);
79  m3 = (y(1) - y(0)) / (x(1) - x(0));
80  m4 = (y(2) - y(1)) / (x(2) - x(1));
81  t(0) = getSlope(m1, m2, m3, m4);
82  m5 = (y(3) - y(2)) / (x(3) - x(2));
83  t(1) = getSlope(m2, m3, m4, m5);
84 
85  // right side: t(N-2), t(N-1)
86  temp = (x(N - 2) - x(N - 1)) / (x(N - 3) - x(N - 1));
87  temp = temp * temp;
88  g0 = y(N - 1);
89  g1 = ((y(N - 2) - y(N - 1)) - temp * (y(N - 3) - y(N - 1))) /
90  ((x(N - 2) - x(N - 1)) - temp * (x(N - 3) - x(N - 1)));
91  g2 = ((y(N - 3) - y(N - 1)) - g1 * (x(N - 3) - x(N - 1))) /
92  ((x(N - 3) - x(N - 1)) * (x(N - 3) - x(N - 1)));
93  x4 = x(N - 2) + (x(N - 1) - x(N - 3));
94  x5 = x(N - 1) + (x(N - 1) - x(N - 3));
95  y4 = g0 + g1 * (x4 - x(N - 1)) + g2 * (x4 - x(N - 1)) * (x4 - x(N - 1));
96  y5 = g0 + g1 * (x5 - x(N - 1)) + g2 * (x5 - x(N - 1)) * (x5 - x(N - 1));
97  m1 = (y(N - 3) - y(N - 4)) / (x(N - 3) - x(N - 4));
98  m2 = (y(N - 2) - y(N - 3)) / (x(N - 2) - x(N - 3));
99  m3 = (y(N - 1) - y(N - 2)) / (x(N - 1) - x(N - 2));
100  m4 = (y4 - y(N - 1)) / (x4 - x(N - 1));
101  m5 = (y5 - y4) / (x5 - x4);
102  t(N - 2) = getSlope(m1, m2, m3, m4);
103  t(N - 1) = getSlope(m2, m3, m4, m5);
104  break;
105  case splinePeriodic:
106  // left: last two points determine the slopes t(0), t(1)
107  m1 = (y(N - 1) - y(N - 2)) / (x(N - 1) - x(N - 2));
108  m2 = (y(0) - y(N - 1)) / (x(0) - x(N - 1));
109  m3 = (y(1) - y(0)) / (x(1) - x(0));
110  m4 = (y(2) - y(1)) / (x(2) - x(1));
111  m5 = (y(3) - y(2)) / (x(3) - x(2));
112  t(0) = getSlope(m1, m2, m3, m4);
113  t(1) = getSlope(m2, m3, m4, m5);
114  // right: first two points determine the slopes t(N-2), t(N-1)
115  m1 = (y(N - 3) - y(N - 4)) / (x(N - 3) - x(N - 4));
116  m2 = (y(N - 2) - y(N - 3)) / (x(N - 2) - x(N - 3));
117  m3 = (y(N - 1) - y(N - 2)) / (x(N - 1) - x(N - 2));
118  m4 = (y(0) - y(N - 1)) / (x(0) - x(N - 1));
119  m5 = (y(1) - y(0)) / (x(1) - x(0));
120  t(N - 2) = getSlope(m1, m2, m3, m4);
121  t(N - 1) = getSlope(m2, m3, m4, m5);
122  break;
123  case splineDerivativeZero:
124  throw std::runtime_error(
125  "erro in AkimaSpline::Interpolate: case splineDerivativeZero not "
126  "implemented yet");
127  break;
128  }
129 
130  // calculate t's for all inner points [2,N-3]
131  for (int i = 2; i < N - 2; i++) {
132  m1 = (y(i - 1) - y(i - 2)) / (x(i - 1) - x(i - 2));
133  m2 = (y(i) - y(i - 1)) / (x(i) - x(i - 1));
134  m3 = (y(i + 1) - y(i)) / (x(i + 1) - x(i));
135  m4 = (y(i + 2) - y(i + 1)) / (x(i + 2) - x(i + 1));
136  t(i) = getSlope(m1, m2, m3, m4);
137  }
138 
139  // calculate p0,p1,p2,p3 for all intervals 0..(N-2), where interval
140  // [x(i),x(i+1)] shall have number i (this means that the last interval
141  // has number N-2)
142  for (int i = 0; i < N - 1; i++) {
143  p0(i) = y(i);
144  p1(i) = t(i);
145  p2(i) =
146  (3.0 * (y(i + 1) - y(i)) / (x(i + 1) - x(i)) - 2.0 * t(i) - t(i + 1)) /
147  (x(i + 1) - x(i));
148  p3(i) = (t(i) + t(i + 1) - 2.0 * (y(i + 1) - y(i)) / (x(i + 1) - x(i))) /
149  ((x(i + 1) - x(i)) * (x(i + 1) - x(i)));
150  }
151 }
