75  // returns number of basisfunctions for minimal basis, less for ecps because no core shells
76  std::vector<int> getMinimalBasis(std::string name,bool ecp){
77  const ub::matrix<int> temp=_Atomconfig.at(name);
78  std::vector<int> result=std::vector<int>(temp.size1(),0);
79 
80  //cout<<temp<<endl;
81 
82 
83  for ( int i=temp.size2()-1;i>=0;i-- ){
84  //cout <<i<<endl;
85  if(temp(0,i)<1) continue;
86  for (unsigned j=0;j<temp.size1();j++){
87  if(temp(j,i)>0){
88  result[j]+=2*j+1;
89  //cout <<"["<<i<<":"<<j<<"]"<<endl;
90  }
91  }
92  if(ecp) break;
93  }
94 
95 
96  return result;
97  }
