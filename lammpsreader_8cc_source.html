65 bool LAMMPSReader::NextFrame(Topology &top)
66 {
67  string line;
68  getline(_fl, line);
69  while(!_fl.eof()) {
70  if(line.substr(0, 5) != "ITEM:")
71  throw std::ios_base::failure("unexpected line in lammps file:\n"+line);
72  if(line.substr(6,8) == "TIMESTEP") {
73  ReadTimestep(top, line);
74  }
75  else if(line.substr(6,15) == "NUMBER OF ATOMS") {
76  ReadNumAtoms(top, line);
77  }
78  else if(line.substr(6,10) == "BOX BOUNDS") {
79  ReadBox(top, line);
80  }
81  else if(line.substr(6, 5) == "ATOMS") {
82  ReadAtoms(top, line);
83  break;
84  }
85 
86  else {
87  throw std::ios_base::failure("unknown item lammps file : " + line.substr(6));
88  }
89  getline(_fl, line);
90  }
91  if (_topology) {
92  cout << "WARNING: topology created from .dump file, masses, charges, types, residue names are wrong!\n";
93  }
94  return !_fl.eof();;
95 }

105 void LAMMPSReader::ReadBox(Topology &top, string itemline)
106 {
107  string s;
108 
109  matrix m;
110  m.ZeroMatrix();
111 
112  for(int i=0; i<3; ++i) {
113  getline(_fl, s);
114  Tokenizer tok(s, " ");
115  vector<double> v;
116  tok.ConvertToVector(v);
117  if(v.size()!=2)
118  throw std::ios_base::failure("invalid box format");
119  m[i][i] = v[1] - v[0];
120  }
121  top.setBox(m );
122 }

133 void LAMMPSReader::ReadAtoms(Topology &top, string itemline) {
134  if(_topology){
135  top.CreateResidue("dum");
136  for(int i=0; i<_natoms; ++i) {
137  (void)top.CreateBead(1, "no", top.GetOrCreateBeadType("no"), 0, 0, 0);
138  }
139  }
140 
141  bool pos=false;
142  bool force=false;
143  bool vel=false;
144  int id=-1;
145 
146  vector<string> fields;
147 
148  {
149  Tokenizer tok(itemline.substr(12), " ");
150  tok.ToVector(fields);
151  int j=0;
152  for(Tokenizer::iterator i=tok.begin(); i!=tok.end(); ++i,++j) {
153  if(*i == "x" || *i == "y" || *i == "z")
154  pos =true;
155  else if(*i == "xu" || *i == "yu" || *i == "zu")
156  pos =true;
157  else if(*i == "xs" || *i == "ys" || *i == "zs")
158  pos =true;
159  else if(*i == "vx" || *i == "vy" || *i == "vz")
160  vel =true;
161  else if(*i == "fx" || *i == "fy" || *i == "fz")
162  force=true;
163  else if(*i == "id")
164  id=j;
165  }
166  }
167  if (id<0){
168  throw std::runtime_error("error, id not found in any column of the atoms section");
169  }
170 
171  for(int i=0; i<_natoms; ++i) {
172  string s;
173  getline(_fl, s);
174  if (_fl.eof())
175  throw std::runtime_error("Error: unexpected end of lammps file '" + _fname + "' only " + boost::lexical_cast<string>(i) + " atoms of " + boost::lexical_cast<string>(_natoms) + " read.");
176 
177  Tokenizer tok(s, " ");
178  Tokenizer::iterator itok= tok.begin();
179  vector<string> fields2;
180  tok.ToVector(fields2);
181  // internal numbering begins with 0
182  int atom_id = boost::lexical_cast<int>(fields2[id]);
183  if (atom_id > _natoms)
184  throw std::runtime_error("Error: found atom with id "+ boost::lexical_cast<string>(atom_id) + " but only "+ boost::lexical_cast<string>(_natoms) + " atoms defined in header of file '" + _fname + "'");
185  Bead *b = top.getBead(atom_id-1);
186  b->HasPos(pos);
187  b->HasF(force);
188  b->HasVel(vel);
189  matrix m=top.getBox();
190 
191  for(size_t j=0; itok!=tok.end(); ++itok, ++j) {
192  if(j == fields.size())
193  throw std::runtime_error("error, wrong number of columns in atoms section");
194  else if(fields[j] == "x")
195  b->Pos().x() = boost::lexical_cast<double>(*itok);
196  else if(fields[j] == "y")
197  b->Pos().y() = boost::lexical_cast<double>(*itok);
198  else if(fields[j] == "z")
199  b->Pos().z() = boost::lexical_cast<double>(*itok);
200  else if(fields[j] == "xu")
201  b->Pos().x() = boost::lexical_cast<double>(*itok);
202  else if(fields[j] == "yu")
203  b->Pos().y() = boost::lexical_cast<double>(*itok);
204  else if(fields[j] == "zu")
205  b->Pos().z() = boost::lexical_cast<double>(*itok);
206  else if(fields[j] == "xs")
207  b->Pos().x() = boost::lexical_cast<double>(*itok)*m[0][0];
208  else if(fields[j] == "ys")
209  b->Pos().y() = boost::lexical_cast<double>(*itok)*m[1][1];
210  else if(fields[j] == "zs")
211  b->Pos().z() = boost::lexical_cast<double>(*itok)*m[2][2];
212  else if(fields[j] == "vx")
213  b->Vel().x() = boost::lexical_cast<double>(*itok);
214  else if(fields[j] == "vy")
215  b->Vel().y() = boost::lexical_cast<double>(*itok);
216  else if(fields[j] == "vz")
217  b->Vel().z() = boost::lexical_cast<double>(*itok);
218  else if(fields[j] == "fx")
219  b->F().x() = boost::lexical_cast<double>(*itok);
