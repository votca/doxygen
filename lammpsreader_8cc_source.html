00065 bool LAMMPSReader::NextFrame(Topology &top)
00066 {
00067     string line;
00068     getline(_fl, line);
00069     while(!_fl.eof()) {
00070         if(line.substr(0, 5) != "ITEM:")
00071             throw std::ios_base::failure("unexpected line in lammps file:\n"+line);
00072         if(line.substr(6,8) == "TIMESTEP") {
00073                 ReadTimestep(top, line);
00074         }
00075         else if(line.substr(6,15) == "NUMBER OF ATOMS") {
00076                 ReadNumAtoms(top, line);
00077         }
00078         else if(line.substr(6,10) == "BOX BOUNDS") {
00079                 ReadBox(top, line);
00080         }
00081         else if(line.substr(6, 5) == "ATOMS") {
00082                 ReadAtoms(top, line);
00083                 break;
00084         }
00085 
00086         else {
00087             throw std::ios_base::failure("unknown item lammps file : " + line.substr(6));
00088         }
00089         getline(_fl, line);
00090     }
00091     if (_topology) {
00092       cout << "WARNING: topology created from .dump file, masses, charges, types, residue names are wrong!\n";
00093     }
00094     return !_fl.eof();;
00095 }
