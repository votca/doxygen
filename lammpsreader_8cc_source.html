63 bool LAMMPSReader::NextFrame(Topology &top)
64 {
65  string line;
66  getline(_fl, line);
67  while(!_fl.eof()) {
68  if(line.substr(0, 5) != "ITEM:")
69  throw std::ios_base::failure("unexpected line in lammps file:\n"+line);
70  if(line.substr(6,8) == "TIMESTEP") {
71  ReadTimestep(top, line);
72  }
73  else if(line.substr(6,15) == "NUMBER OF ATOMS") {
74  ReadNumAtoms(top, line);
75  }
76  else if(line.substr(6,10) == "BOX BOUNDS") {
77  ReadBox(top, line);
78  }
79  else if(line.substr(6, 5) == "ATOMS") {
80  ReadAtoms(top, line);
81  break;
82  }
83 
84  else {
85  throw std::ios_base::failure("unknown item lammps file : " + line.substr(6));
86  }
87  getline(_fl, line);
88  }
89  if (_topology) {
90  cout << "WARNING: topology created from .dump file, masses, charges, types, residue names are wrong!\n";
91  }
92  return !_fl.eof();;
93 }
