159  std::vector<std::unique_ptr<StaticSite> > externalsites;

166 
167  Eigen::MatrixXd dip_ref = Eigen::MatrixXd::Zero(17, 17);
168 
169  double a = 0.1; // this is in a0
170  double mag_d = seg[0].getDipole().norm(); // this is in e * a0
171  const Eigen::Vector3d dir_d = seg[0].getDipole().normalized();
172  const Eigen::Vector3d A = seg[0].getPos() + 0.5 * a * dir_d;
173  const Eigen::Vector3d B = seg[0].getPos() - 0.5 * a * dir_d;
174  double qA = mag_d / a;
175  double qB = -qA;
176  StaticSite site1 = StaticSite(0, "", A);
177  site1.setCharge(qA);
178  StaticSite site2 = StaticSite(1, "", B);
179  site2.setCharge(qB);
180  std::vector<std::unique_ptr<StaticSite> > externalsites_mono;
181  externalsites_mono.push_back(
182  std::unique_ptr<StaticSite>(new StaticSite(site1)));
183  externalsites_mono.push_back(
184  std::unique_ptr<StaticSite>(new StaticSite(site2)));
185  AOMultipole mono2;
186  mono2.FillPotential(aobasis, externalsites_mono);
187 
188  bool dip_check = mono2.Matrix().isApprox(dip.Matrix(), 1e-4);
189  BOOST_CHECK_EQUAL(dip_check, 1);
190  if (!dip_check) {
191  std::cout << "mono2 Ref" << endl;
192  std::cout << mono2.Matrix() << endl;
193  std::cout << "Dip" << endl;
194  std::cout << dip.Matrix() << endl;
