44 Mat_p_Energy ERIs::CalculateERIs(const Eigen::MatrixXd& DMAT) const {
45  Index nthreads = OPENMP::getMaxThreads();
46  std::vector<Eigen::MatrixXd> ERIS_thread = std::vector<Eigen::MatrixXd>(
47  nthreads, Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols()));
48  Symmetric_Matrix dmat_sym = Symmetric_Matrix(DMAT);
49 #pragma omp parallel for
50  for (Index i = 0; i < _threecenter.size(); i++) {
51  const Symmetric_Matrix& threecenter = _threecenter[i];
52  // Trace over prod::DMAT,I(l)=componentwise product over
53  const double factor = threecenter.TraceofProd(dmat_sym);
54  Eigen::SelfAdjointView<Eigen::MatrixXd, Eigen::Upper> m =
55  ERIS_thread[OPENMP::getThreadId()].selfadjointView<Eigen::Upper>();
56  threecenter.AddtoEigenUpperMatrix(m, factor);
57  }
58 
59  Eigen::MatrixXd ERIs2 =
60  std::accumulate(ERIS_thread.begin(), ERIS_thread.end(),
61  Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols()).eval());
62  ERIs2 = ERIs2.selfadjointView<Eigen::Upper>();
63  double energy = CalculateEnergy(DMAT, ERIs2);
64  return Mat_p_Energy(energy, ERIs2);
65 }

110 Mat_p_Energy ERIs::CalculateERIs_4c_small_molecule(
111  const Eigen::MatrixXd& DMAT) const {
112 
113  Eigen::MatrixXd ERIs2 = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
114 
115  const Eigen::VectorXd& fourc_vector = _fourcenter.get_4c_vector();
116 
117  Index dftBasisSize = DMAT.rows();
118  Index vectorSize = (dftBasisSize * (dftBasisSize + 1)) / 2;
119 #pragma omp parallel for
120  for (Index i = 0; i < dftBasisSize; i++) {
121  Index sum_i = (i * (i + 1)) / 2;
122  for (Index j = i; j < dftBasisSize; j++) {
123  Index index_ij = dftBasisSize * i - sum_i + j;
124  Index index_ij_kl_a =
125  vectorSize * index_ij - (index_ij * (index_ij + 1)) / 2;
126  for (Index k = 0; k < dftBasisSize; k++) {
127  Index sum_k = (k * (k + 1)) / 2;
128  for (Index l = k; l < dftBasisSize; l++) {
129  Index index_kl = dftBasisSize * k - sum_k + l;
130 
131  Index index_ij_kl = index_ij_kl_a + index_kl;
132  if (index_ij > index_kl) {
133  index_ij_kl = vectorSize * index_kl -
134  (index_kl * (index_kl + 1)) / 2 + index_ij;
135  }
136 
137  if (l == k) {
138  ERIs2(i, j) += DMAT(k, l) * fourc_vector(index_ij_kl);
139  } else {
140  ERIs2(i, j) += 2. * DMAT(k, l) * fourc_vector(index_ij_kl);
141  }
142  }
143  }
144  ERIs2(j, i) = ERIs2(i, j);
145  }
146  }
147 
148  double energy = CalculateEnergy(DMAT, ERIs2);
149  return Mat_p_Energy(energy, ERIs2);
150 }

152 Mat_p_Energy ERIs::CalculateEXX_4c_small_molecule(
153  const Eigen::MatrixXd& DMAT) const {
154  std::vector<Eigen::MatrixXd> EXX_thread = std::vector<Eigen::MatrixXd>(
155  OPENMP::getMaxThreads(), Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols()));
156 
157  const Eigen::VectorXd& fourc_vector = _fourcenter.get_4c_vector();
158 
159  Index dftBasisSize = DMAT.rows();
160  Index vectorSize = (dftBasisSize * (dftBasisSize + 1)) / 2;
161 #pragma omp parallel for
162  for (Index i = 0; i < dftBasisSize; i++) {
163  Index thread = OPENMP::getThreadId();
164  Index sum_i = (i * (i + 1)) / 2;
165  for (Index j = i; j < dftBasisSize; j++) {
166  Index index_ij = DMAT.cols() * i - sum_i + j;
167  Index index_ij_kl_a =
168  vectorSize * index_ij - (index_ij * (index_ij + 1)) / 2;
169  for (Index k = 0; k < dftBasisSize; k++) {
170  Index sum_k = (k * (k + 1)) / 2;
171  for (Index l = k; l < dftBasisSize; l++) {
172  Index index_kl = DMAT.cols() * k - sum_k + l;
173 
174  Index _index_ij_kl = index_ij_kl_a + index_kl;
175  if (index_ij > index_kl) {
176  _index_ij_kl = vectorSize * index_kl -
177  (index_kl * (index_kl + 1)) / 2 + index_ij;
178  }
179  double factorij = 1;
180  if (i == j) {
181  factorij = 0.5;
182  }
183  double factorkl = 1;
184  if (l == k) {
185  factorkl = 0.5;
186  }
187  double factor = factorij * factorkl;
188  EXX_thread[thread](i, l) +=
189  factor * DMAT(j, k) * fourc_vector(_index_ij_kl);
190  EXX_thread[thread](j, l) +=
191  factor * DMAT(i, k) * fourc_vector(_index_ij_kl);
192  EXX_thread[thread](i, k) +=
193  factor * DMAT(j, l) * fourc_vector(_index_ij_kl);
194  EXX_thread[thread](j, k) +=
195  factor * DMAT(i, l) * fourc_vector(_index_ij_kl);
196  }
197  }
198  }
199  }
200 
201  Eigen::MatrixXd EXXs =
202  std::accumulate(EXX_thread.begin(), EXX_thread.end(),
203  Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols()).eval());
204 
205  double energy = CalculateEnergy(DMAT, EXXs);
206  return Mat_p_Energy(energy, EXXs);
207 }

209 Mat_p_Energy ERIs::CalculateERIs_4c_direct(const AOBasis& dftbasis,
210  const Eigen::MatrixXd& DMAT) const {
211 
212  // Number of shells
213  Index numShells = dftbasis.getNumofShells();
214 
215  // Initialize ERIs matrix
216  Eigen::MatrixXd ERIs2 = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
217 
218 #pragma omp parallel
219  { // Begin omp parallel
220 
221  Eigen::MatrixXd ERIs_thread =
222  Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
223 
224 #pragma omp for
225  for (Index iShell_3 = 0; iShell_3 < numShells; iShell_3++) {
226  const AOShell& shell_3 = dftbasis.getShell(iShell_3);
227  Index numFunc_3 = shell_3.getNumFunc();
228  for (Index iShell_4 = iShell_3; iShell_4 < numShells; iShell_4++) {
229  const AOShell& shell_4 = dftbasis.getShell(iShell_4);
230  Index numFunc_4 = shell_4.getNumFunc();
231  for (Index iShell_1 = iShell_3; iShell_1 < numShells; iShell_1++) {
232  const AOShell& shell_1 = dftbasis.getShell(iShell_1);
233  Index numFunc_1 = shell_1.getNumFunc();
234  for (Index iShell_2 = iShell_1; iShell_2 < numShells; iShell_2++) {
235  const AOShell& shell_2 = dftbasis.getShell(iShell_2);
236  Index numFunc_2 = shell_2.getNumFunc();
237 
238  // Pre-screening
239  if (_with_screening && CheckScreen(_screening_eps, shell_1, shell_2,
240  shell_3, shell_4)) {
241  continue;
242  }
243 
244  // Get the current 4c block
245  Eigen::Tensor<double, 4> block(numFunc_1, numFunc_2, numFunc_3,
246  numFunc_4);
247  block.setZero();
248  bool nonzero = _fourcenter.FillFourCenterRepBlock(
249  block, shell_1, shell_2, shell_3, shell_4);
250 
251  // If there are only zeros, we don't need to put anything in the
252  // ERIs matrix
253  if (!nonzero) {
254  continue;
255  }
256 
257  // Begin fill ERIs matrix
258 
259  FillERIsBlock<false>(ERIs_thread, DMAT, block, shell_1, shell_2,
260  shell_3, shell_4);
261 
262  // Symmetry 1 <--> 2
263  if (iShell_1 != iShell_2) {
264  FillERIsBlock<false>(ERIs_thread, DMAT, block, shell_2, shell_1,
265  shell_3, shell_4);
266  }
267 
268  // Symmetry 3 <--> 4
269  if (iShell_3 != iShell_4) {
270  FillERIsBlock<false>(ERIs_thread, DMAT, block, shell_1, shell_2,
271  shell_4, shell_3);
272  }
273 
274  // Symmetry 1 <--> 2 and 3 <--> 4
275  if (iShell_1 != iShell_2 && iShell_3 != iShell_4) {
276  FillERIsBlock<false>(ERIs_thread, DMAT, block, shell_2, shell_1,
277  shell_4, shell_3);
278  }
279 
280  // Symmetry (1, 2) <--> (3, 4)
281  if (iShell_1 != iShell_3) {
282 
283  FillERIsBlock<true>(ERIs_thread, DMAT, block, shell_3, shell_4,
284  shell_1, shell_2);
285 
286  // Symmetry 1 <--> 2
287  if (iShell_1 != iShell_2) {
288  FillERIsBlock<true>(ERIs_thread, DMAT, block, shell_3, shell_4,
289  shell_2, shell_1);
290  }
291 
292  // Symmetry 3 <--> 4
293  if (iShell_3 != iShell_4) {
294  FillERIsBlock<true>(ERIs_thread, DMAT, block, shell_4, shell_3,
295  shell_1, shell_2);
296  }
297 
298  // Symmetry 1 <--> 2 and 3 <--> 4
299  if (iShell_1 != iShell_2 && iShell_3 != iShell_4) {
300  FillERIsBlock<true>(ERIs_thread, DMAT, block, shell_4, shell_3,
301  shell_2, shell_1);
302  }
303  }
304 
305  // End fill ERIs matrix
306  } // End loop over shell 2
307  } // End loop over shell 1
308  } // End loop over shell 4
309  } // End loop over shell 3
310 
311 #pragma omp critical
312  { ERIs2 += ERIs_thread; }
313  }
314 
315  ERIs2 = ERIs2.selfadjointView<Eigen::Upper>();
316  double energy = CalculateEnergy(DMAT, ERIs2);
317  return Mat_p_Energy(energy, ERIs2);
318 }

320 template <bool transposed_block>
321 void ERIs::FillERIsBlock(Eigen::MatrixXd& ERIsCur, const Eigen::MatrixXd& DMAT,
322  const Eigen::Tensor<double, 4>& block,
323  const AOShell& shell_1, const AOShell& shell_2,
324  const AOShell& shell_3, const AOShell& shell_4) const {
325 
326  for (Index iFunc_3 = 0; iFunc_3 < shell_3.getNumFunc(); iFunc_3++) {
327  Index ind_3 = shell_3.getStartIndex() + iFunc_3;
328  for (Index iFunc_4 = 0; iFunc_4 < shell_4.getNumFunc(); iFunc_4++) {
329  Index ind_4 = shell_4.getStartIndex() + iFunc_4;
330 
331  // Symmetry
332  if (ind_3 > ind_4) {
333  continue;
334  }
335 
336  for (Index iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
337  Index ind_1 = shell_1.getStartIndex() + iFunc_1;
338  for (Index iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
339  Index ind_2 = shell_2.getStartIndex() + iFunc_2;
340 
341  // Symmetry
342  if (ind_1 > ind_2) {
343  continue;
344  }
345 
346  // Symmetry for diagonal elements
347  double multiplier = (ind_1 == ind_2 ? 1.0 : 2.0);
348  // Fill ERIs matrix
349  if (!transposed_block) {
350  ERIsCur(ind_3, ind_4) += multiplier * DMAT(ind_1, ind_2) *
351  block(iFunc_1, iFunc_2, iFunc_3, iFunc_4);
352  } else {
353  ERIsCur(ind_3, ind_4) += multiplier * DMAT(ind_1, ind_2) *
354  block(iFunc_3, iFunc_4, iFunc_1, iFunc_2);
355  }
356 
357  } // End loop over functions in shell 2
358  } // End loop over functions in shell 1
359  } // End loop over functions in shell 4
360  } // End loop over functions in shell 3
361 
362  return;
363 }

365 void ERIs::CalculateERIsDiagonals(const AOBasis& dftbasis) {
366  // Number of shells
367  Index numShells = dftbasis.getNumofShells();
368  // Total number of functions
369  Index dftBasisSize = dftbasis.AOBasisSize();
370 
371  _diagonals = Eigen::MatrixXd::Zero(dftBasisSize, dftBasisSize);
372 
373  for (Index iShell_1 = 0; iShell_1 < numShells; iShell_1++) {
374  const AOShell& shell_1 = dftbasis.getShell(iShell_1);
375  Index numFunc_1 = shell_1.getNumFunc();
376  for (Index iShell_2 = iShell_1; iShell_2 < numShells; iShell_2++) {
377  const AOShell& shell_2 = dftbasis.getShell(iShell_2);
378  Index numFunc_2 = shell_2.getNumFunc();
379 
380  // Get the current 4c block
381  Eigen::Tensor<double, 4> block(numFunc_1, numFunc_2, numFunc_1,
382  numFunc_2);
383  block.setZero();
384  bool nonzero = _fourcenter.FillFourCenterRepBlock(block, shell_1, shell_2,
385  shell_1, shell_2);
386 
387  if (!nonzero) {
388  continue;
389  }
390 
391  for (Index iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
392  Index ind_1 = shell_1.getStartIndex() + iFunc_1;
393  for (Index iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
394  Index ind_2 = shell_2.getStartIndex() + iFunc_2;
395 
396  // Symmetry
397  if (ind_1 > ind_2) {
398  continue;
399  }
400 
401  _diagonals(ind_1, ind_2) = block(iFunc_1, iFunc_2, iFunc_1, iFunc_2);
402 
403  // Symmetry
404  if (ind_1 != ind_2) {
405  _diagonals(ind_2, ind_1) = _diagonals(ind_1, ind_2);
406  }
407  }
408  }
409  }
410  }
411 
412  return;

415 bool ERIs::CheckScreen(double eps, const AOShell& shell_1,
416  const AOShell& shell_2, const AOShell& shell_3,
417  const AOShell& shell_4) const {
418 
419  const double eps2 = eps * eps;
420 
421  for (Index iFunc_3 = 0; iFunc_3 < shell_3.getNumFunc(); iFunc_3++) {
422  Index ind_3 = shell_3.getStartIndex() + iFunc_3;
423  for (Index iFunc_4 = 0; iFunc_4 < shell_4.getNumFunc(); iFunc_4++) {
424  Index ind_4 = shell_4.getStartIndex() + iFunc_4;
425 
426  // Symmetry
427  if (ind_3 > ind_4) {
428  continue;
429  }
430 
431  for (Index iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
432  Index ind_1 = shell_1.getStartIndex() + iFunc_1;
433  for (Index iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
434  Index ind_2 = shell_2.getStartIndex() + iFunc_2;
435 
436  // Symmetry
437  if (ind_1 > ind_2) {
438  continue;
439  }
440 
441  // Cauchy–Schwarz
442  // <ab|cd> <= sqrt(<ab|ab>) * sqrt(<cd|cd>)
443  double ub = _diagonals(ind_1, ind_2) * _diagonals(ind_3, ind_4);
444 
445  // Compare with tolerance
446  if (ub > eps2) {
447  return false; // We must compute ERIS for the whole block
448  }
449  }
450  }
451  }
452  }
453 
454  return true; // We can skip the whole block
455 }
