47  void ERIs::CalculateERIs(const Eigen::MatrixXd &DMAT) {
48 
49  int nthreads = 1;
50  #ifdef _OPENMP
51  nthreads = omp_get_max_threads();
52  #endif
53  std::vector<Eigen::MatrixXd >ERIS_thread;
54 
55  for (int i = 0; i < nthreads; ++i) {
56  Eigen::MatrixXd thread = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
57  ERIS_thread.push_back(thread);
58  }
59 
60  #pragma omp parallel for
61  for (int thread = 0; thread < nthreads; ++thread) {
62  Symmetric_Matrix dmat_sym = Symmetric_Matrix(DMAT);
63  for (int i = thread; i < _threecenter.size(); i += nthreads) {
64  const Symmetric_Matrix &threecenter = _threecenter[i];
65  // Trace over prod::DMAT,I(l)=componentwise product over
66  const double factor = threecenter.TraceofProd(dmat_sym);
67  Eigen::SelfAdjointView<Eigen::MatrixXd,Eigen::Upper> m=ERIS_thread[thread].selfadjointView<Eigen::Upper>();
68  threecenter.AddtoEigenUpperMatrix(m, factor);
69  }
70  }
71 
72  _ERIs = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
73  for (const auto& thread : ERIS_thread) {
74  _ERIs += thread;
75  }
76  _ERIs+=_ERIs.triangularView<Eigen::StrictlyUpper>().transpose();
77 
78  CalculateEnergy(DMAT);
79  return;
80  }

143  void ERIs::CalculateERIs_4c_small_molecule(const Eigen::MatrixXd &DMAT) {
144 
145  _ERIs = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
146 
147  const Eigen::VectorXd& _4c_vector = _fourcenter.get_4c_vector();
148 
149  int dftBasisSize = DMAT.rows();
150  int vectorSize = (dftBasisSize*(dftBasisSize+1))/2;
151  #pragma omp parallel for
152  for (int _i = 0; _i < dftBasisSize; _i++) {
153  int sum_i = (_i*(_i+1))/2;
154  for (int _j = _i; _j < dftBasisSize; _j++) {
155  int _index_ij = dftBasisSize * _i - sum_i + _j;
156  int _index_ij_kl_a = vectorSize * _index_ij - (_index_ij*(_index_ij+1))/2;
157  for (int _k = 0; _k < dftBasisSize; _k++) {
158  int sum_k = (_k*(_k+1))/2;
159  for (int _l = _k; _l < dftBasisSize; _l++) {
160  int _index_kl = dftBasisSize * _k - sum_k + _l;
161 
162  unsigned _index_ij_kl = _index_ij_kl_a + _index_kl;
163  if (_index_ij > _index_kl) _index_ij_kl = vectorSize * _index_kl - (_index_kl*(_index_kl+1))/2 + _index_ij;
164 
165  if (_l == _k) {
166  _ERIs(_i, _j) += DMAT(_k, _l) * _4c_vector(_index_ij_kl);
167  } else {
168  _ERIs(_i, _j) += 2. * DMAT(_k, _l) * _4c_vector(_index_ij_kl);
169  }
170 
171  }
172  }
173  _ERIs(_j, _i) = _ERIs(_i, _j);
174  }
175  }
176 
177  CalculateEnergy(DMAT);
178  return;
179  }

182  void ERIs::CalculateEXX_4c_small_molecule(const Eigen::MatrixXd &DMAT) {
183  int nthreads = 1;
184  #ifdef _OPENMP
185  nthreads = omp_get_max_threads();
186  #endif
187 
188  std::vector<Eigen::MatrixXd >EXX_thread;
189 
190  for (int i = 0; i < nthreads; ++i) {
191  Eigen::MatrixXd thread = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
192  EXX_thread.push_back(thread);
193  }
194 
195  const Eigen::VectorXd& _4c_vector = _fourcenter.get_4c_vector();
196 
197  int dftBasisSize = DMAT.rows();
198  int vectorSize = (dftBasisSize*(dftBasisSize+1))/2;
199  #pragma omp parallel for
200  for (int thread = 0; thread < nthreads; ++thread) {
201  for (int _i = thread; _i < dftBasisSize; _i+= nthreads) {
202  int sum_i = (_i*(_i+1))/2;
203  for (int _j = _i; _j < dftBasisSize; _j++) {
204  int _index_ij = DMAT.cols() * _i - sum_i + _j;
205  int _index_ij_kl_a = vectorSize * _index_ij - (_index_ij*(_index_ij+1))/2;
206  for (int _k = 0; _k < dftBasisSize; _k++) {
207  int sum_k = (_k*(_k+1))/2;
208  for (int _l = _k; _l < dftBasisSize; _l++) {
209  int _index_kl = DMAT.cols() * _k - sum_k + _l;
210 
211  int _index_ij_kl = _index_ij_kl_a + _index_kl;
212  if (_index_ij > _index_kl) _index_ij_kl = vectorSize * _index_kl - (_index_kl*(_index_kl+1))/2 + _index_ij;
213  double factorij=1;
214  if(_i==_j){factorij=0.5;}
215  double factorkl=1;
216  if(_l==_k){factorkl=0.5;}
217  double factor=factorij*factorkl;
218  EXX_thread[thread](_i, _l) +=factor*DMAT(_j, _k) * _4c_vector(_index_ij_kl);
219  EXX_thread[thread](_j, _l) +=factor*DMAT(_i, _k) * _4c_vector(_index_ij_kl);
220  EXX_thread[thread](_i, _k) +=factor*DMAT(_j, _l) * _4c_vector(_index_ij_kl);
221  EXX_thread[thread](_j, _k) +=factor*DMAT(_i, _l) * _4c_vector(_index_ij_kl);
222  }
223  }
224  }
225  }
226  }
227  _EXXs = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
228  for (const auto& thread : EXX_thread) {
229  _EXXs += thread;
230  }
231 
232  CalculateEXXEnergy(DMAT);
233  return;

237  void ERIs::CalculateERIs_4c_direct(const AOBasis& dftbasis, const Eigen::MatrixXd &DMAT) {
238 
239  tensor4d::extent_gen extents;
240 
241  // Number of shells
242  int numShells = dftbasis.getNumofShells();
243 
244  // Initialize ERIs matrix
245  _ERIs = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
246 
247  #pragma omp parallel
248  { // Begin omp parallel
249 
250  Eigen::MatrixXd ERIs_thread = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
251 
252  #pragma omp for
253  for (int iShell_3 = 0; iShell_3 < numShells; iShell_3++) {
254  const AOShell& shell_3 = *dftbasis.getShell(iShell_3);
255  int numFunc_3 = shell_3.getNumFunc();
256  for (int iShell_4 = iShell_3; iShell_4 < numShells; iShell_4++) {
257  const AOShell& shell_4 = *dftbasis.getShell(iShell_4);
258  int numFunc_4 = shell_4.getNumFunc();
259  for (int iShell_1 = iShell_3; iShell_1 < numShells; iShell_1++) {
260  const AOShell& shell_1 = *dftbasis.getShell(iShell_1);
261  int numFunc_1 = shell_1.getNumFunc();
262  for (int iShell_2 = iShell_1; iShell_2 < numShells; iShell_2++) {
263  const AOShell& shell_2 = *dftbasis.getShell(iShell_2);
264  int numFunc_2 = shell_2.getNumFunc();
265 
266  // Pre-screening
267  if (_with_screening && CheckScreen(_screening_eps, shell_1, shell_2, shell_3, shell_4))
268  continue;
269 
270  // Get the current 4c block
271  tensor4d block(extents[ range(0, numFunc_1) ][ range(0, numFunc_2) ][ range(0, numFunc_3)][ range(0, numFunc_4)]);
272  for (int i = 0; i < numFunc_1; ++i) {
273  for (int j = 0; j < numFunc_2; ++j) {
274  for (int k = 0; k < numFunc_3; ++k) {
275  for (int l = 0; l < numFunc_4; ++l) {
276  block[i][j][k][l] = 0.0;
277  }
278  }
279  }
280  }
281  bool nonzero=_fourcenter.FillFourCenterRepBlock(block, &shell_1, &shell_2, &shell_3, &shell_4);
282 
283  // If there are only zeros, we don't need to put anything in the ERIs matrix
284  if (!nonzero)
285  continue;
286 
287  // Begin fill ERIs matrix
288 
289  FillERIsBlock(ERIs_thread, DMAT, block, shell_1, shell_2, shell_3, shell_4);
290 
291  // Symmetry 1 <--> 2
292  if (iShell_1 != iShell_2)
293  FillERIsBlock(ERIs_thread, DMAT, block, shell_2, shell_1, shell_3, shell_4);
294 
295  // Symmetry 3 <--> 4
296  if (iShell_3 != iShell_4)
297  FillERIsBlock(ERIs_thread, DMAT, block, shell_1, shell_2, shell_4, shell_3);
298 
299  // Symmetry 1 <--> 2 and 3 <--> 4
300  if (iShell_1 != iShell_2 && iShell_3 != iShell_4)
301  FillERIsBlock(ERIs_thread, DMAT, block, shell_2, shell_1, shell_4, shell_3);
302 
303  // Symmetry (1, 2) <--> (3, 4)
304  if (iShell_1 != iShell_3) {
305 
306  // We need the 'transpose' of block
307  tensor4d block2(extents[ range(0, numFunc_3) ][ range(0, numFunc_4) ][ range(0, numFunc_1)][ range(0, numFunc_2)]);
308  for (int i = 0; i < numFunc_1; ++i) {
309  for (int j = 0; j < numFunc_2; ++j) {
310  for (int k = 0; k < numFunc_3; ++k) {
311  for (int l = 0; l < numFunc_4; ++l) {
312  block2[k][l][i][j] = block[i][j][k][l];
313  }
314  }
315  }
316  }
317 
318  FillERIsBlock(ERIs_thread, DMAT, block2, shell_3, shell_4, shell_1, shell_2);
319 
320  // Symmetry 1 <--> 2
321  if (iShell_1 != iShell_2)
322  FillERIsBlock(ERIs_thread, DMAT, block2, shell_3, shell_4, shell_2, shell_1);
323 
324  // Symmetry 3 <--> 4
325  if (iShell_3 != iShell_4)
326  FillERIsBlock(ERIs_thread, DMAT, block2, shell_4, shell_3, shell_1, shell_2);
327 
328  // Symmetry 1 <--> 2 and 3 <--> 4
329  if (iShell_1 != iShell_2 && iShell_3 != iShell_4)
330  FillERIsBlock(ERIs_thread, DMAT, block2, shell_4, shell_3, shell_2, shell_1);
331  }
332 
333  // End fill ERIs matrix
334  } // End loop over shell 2
335  } // End loop over shell 1
336  } // End loop over shell 4
337  } // End loop over shell 3
338 
339  #pragma omp critical
340  {
341  _ERIs += ERIs_thread;
342  }
343  }
344 
345  // Fill lower triangular part using symmetry
346  for (int i = 0; i < DMAT.cols(); i++){
347  for (int j = i + 1; j < DMAT.rows(); j++){
348  _ERIs(j, i) = _ERIs(i, j);
349  }
350  }
351 
352  CalculateEnergy(DMAT);
353  return;
354  }

357  void ERIs::FillERIsBlock(Eigen::MatrixXd& ERIsCur, const Eigen::MatrixXd& DMAT,
358  const tensor4d& block,
359  const AOShell& shell_1, const AOShell& shell_2,
360  const AOShell& shell_3, const AOShell& shell_4) {
361 
362  for (int iFunc_3 = 0; iFunc_3 < shell_3.getNumFunc(); iFunc_3++) {
363  int ind_3 = shell_3.getStartIndex() + iFunc_3;
364  for (int iFunc_4 = 0; iFunc_4 < shell_4.getNumFunc(); iFunc_4++) {
365  int ind_4 = shell_4.getStartIndex() + iFunc_4;
366 
367  // Symmetry
368  if (ind_3 > ind_4) {
369  continue;
370  }
371 
372  for (int iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
373  int ind_1 = shell_1.getStartIndex() + iFunc_1;
374  for (int iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
375  int ind_2 = shell_2.getStartIndex() + iFunc_2;
376 
377  // Symmetry
378  if (ind_1 > ind_2) {
379  continue;
380  }
381 
382  // Symmetry for diagonal elements
383  double multiplier = (ind_1 == ind_2 ? 1.0 : 2.0);
384  // Fill ERIs matrix
385  ERIsCur(ind_3, ind_4) += multiplier * DMAT(ind_1, ind_2) * block[iFunc_1][iFunc_2][iFunc_3][iFunc_4];
386 
387  } // End loop over functions in shell 2
388  } // End loop over functions in shell 1
389  } // End loop over functions in shell 4
390  } // End loop over functions in shell 3
391 
392  return;
393  }

396  void ERIs::CalculateERIsDiagonals(const AOBasis& dftbasis) {
397 
398  tensor4d::extent_gen extents;
399 
400  // Number of shells
401  int numShells = dftbasis.getNumofShells();
402  // Total number of functions
403  int dftBasisSize = dftbasis.AOBasisSize();
404 
405  _diagonals = Eigen::MatrixXd::Zero(dftBasisSize,dftBasisSize);
406 
407  for (int iShell_1 = 0; iShell_1 < numShells; iShell_1++) {
408  const AOShell& shell_1 = *dftbasis.getShell(iShell_1);
409  int numFunc_1 = shell_1.getNumFunc();
410  for (int iShell_2 = iShell_1; iShell_2 < numShells; iShell_2++) {
411  const AOShell& shell_2 = *dftbasis.getShell(iShell_2);
412  int numFunc_2 = shell_2.getNumFunc();
413 
414  // Get the current 4c block
415  tensor4d block(extents[ range(0, numFunc_1) ][ range(0, numFunc_2) ][ range(0, numFunc_1)][ range(0, numFunc_2)]);
416  for (int i = 0; i < numFunc_1; ++i) {
417  for (int j = 0; j < numFunc_2; ++j) {
418  for (int k = 0; k < numFunc_1; ++k) {
419  for (int l = 0; l < numFunc_2; ++l) {
420  block[i][j][k][l] = 0.0;
421  }
422  }
423  }
424  }
425  bool nonzero=_fourcenter.FillFourCenterRepBlock(block, &shell_1, &shell_2, &shell_1, &shell_2);
426 
427  if (!nonzero) {
428  continue;
429  }
430 
431  for (int iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
432  int ind_1 = shell_1.getStartIndex() + iFunc_1;
433  for (int iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
434  int ind_2 = shell_2.getStartIndex() + iFunc_2;
435 
436  // Symmetry
437  if (ind_1 > ind_2) {
438  continue;
439  }
440 
441  _diagonals(ind_1, ind_2) = block[iFunc_1][iFunc_2][iFunc_1][iFunc_2];
442 
443  // Symmetry
444  if (ind_1 != ind_2) {
445  _diagonals(ind_2, ind_1) = _diagonals(ind_1, ind_2);
446  }
447 
448  }
449  }
450  }
451  }
452 
453  return;

457  bool ERIs::CheckScreen(double eps,
458  const AOShell& shell_1, const AOShell& shell_2,
459  const AOShell& shell_3, const AOShell& shell_4) {
460 
461  const double eps2=eps * eps;
462 
463  for (int iFunc_3 = 0; iFunc_3 < shell_3.getNumFunc(); iFunc_3++) {
464  int ind_3 = shell_3.getStartIndex() + iFunc_3;
465  for (int iFunc_4 = 0; iFunc_4 < shell_4.getNumFunc(); iFunc_4++) {
466  int ind_4 = shell_4.getStartIndex() + iFunc_4;
467 
468  // Symmetry
469  if (ind_3 > ind_4) {
470  continue;
471  }
472 
473  for (int iFunc_1 = 0; iFunc_1 < shell_1.getNumFunc(); iFunc_1++) {
474  int ind_1 = shell_1.getStartIndex() + iFunc_1;
475  for (int iFunc_2 = 0; iFunc_2 < shell_2.getNumFunc(); iFunc_2++) {
476  int ind_2 = shell_2.getStartIndex() + iFunc_2;
477 
478  // Symmetry
479  if (ind_1 > ind_2) {
480  continue;
481  }
482 
483  // CauchySchwarz
484  // <ab|cd> <= sqrt(<ab|ab>) * sqrt(<cd|cd>)
485  double ub = _diagonals(ind_1, ind_2) * _diagonals(ind_3, ind_4);
486 
487  // Compare with tolerance
488  if (ub > eps2) {
489  return false; // We must compute ERIS for the whole block
490  }
491 
492  }
493  }
494  }
495  }
496 
497  return true; // We can skip the whole block
498  }
