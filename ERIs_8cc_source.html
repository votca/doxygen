95  void ERIs::CalculateERIs_4c_small_molecule(const ub::matrix<double> &DMAT) {
96 
97  _ERIs = ub::zero_matrix<double>(DMAT.size1(), DMAT.size2());
98  const ub::vector<double> dmatasarray = DMAT.data();
99  const ub::vector<double>& _4c_vector = _fourcenter.get_4c_vector();
100 
101  int dftBasisSize = DMAT.size1();
102  int vectorSize = (dftBasisSize*(dftBasisSize+1))/2;
103  #pragma omp parallel for
104  for (unsigned _i = 0; _i < DMAT.size1(); _i++) {
105  unsigned sum_i = (_i*(_i+1))/2;
106  for (unsigned _j = _i; _j < DMAT.size2(); _j++) {
107  unsigned _index_ij = DMAT.size2() * _i - sum_i + _j;
108  unsigned _index_ij_kl_a = vectorSize * _index_ij - (_index_ij*(_index_ij+1))/2;
109  for (unsigned _k = 0; _k < DMAT.size1(); _k++) {
110  unsigned sum_k = (_k*(_k+1))/2;
111  for (unsigned _l = _k; _l < DMAT.size2(); _l++) {
112  unsigned _index_kl = DMAT.size2() * _k - sum_k + _l;
113 
114  unsigned _index_ij_kl = _index_ij_kl_a + _index_kl;
115  if (_index_ij > _index_kl) _index_ij_kl = vectorSize * _index_kl - (_index_kl*(_index_kl+1))/2 + _index_ij;
116 
117  if (_l == _k) {
118  _ERIs(_i, _j) += DMAT(_k, _l) * _4c_vector(_index_ij_kl);
119  } else {
120  _ERIs(_i, _j) += 2. * DMAT(_k, _l) * _4c_vector(_index_ij_kl);
121  }
122 
123  }
124  }
125  _ERIs(_j, _i) = _ERIs(_i, _j);
126  }
127  }
128 
129  CalculateEnergy(dmatasarray);
130  return;
131  }
