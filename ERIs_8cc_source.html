50 Eigen::MatrixXd ERIs::ComputeSchwarzShells(const AOBasis& basis) const {
51 
52  Index noshells = basis.getNumofShells();
53 
54  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(noshells, noshells);
55  Index nthreads = OPENMP::getMaxThreads();
56  std::vector<libint2::Engine> engines(nthreads);
57  double epsilon = 0.0;
58  engines[0] = libint2::Engine(libint2::Operator::coulomb, basis.getMaxNprim(),
59  static_cast<int>(basis.getMaxL()), 0, epsilon);
60 
61  for (Index i = 1; i < nthreads; ++i) {
62  engines[i] = engines[0];
63  }
64 
65  std::vector<libint2::Shell> shells = basis.GenerateLibintBasis();
66  using MatrixLibInt =
67  Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic, Eigen::RowMajor>;
68 
69 #pragma omp parallel for schedule(dynamic)
70  for (Index s1 = 0l; s1 < basis.getNumofShells(); ++s1) {
71  Index thread_id = OPENMP::getThreadId();
72  libint2::Engine& engine = engines[thread_id];
73  const libint2::Engine::target_ptr_vec& buf = engine.results();
74  Index n1 = shells[s1].size();
75 
76  for (Index s2 = 0l; s2 <= s1; ++s2) {
77  Index n2 = shells[s2].size();
78  Index n12 = n1 * n2;
79 
80  engines[thread_id]
81  .compute2<libint2::Operator::coulomb, libint2::BraKet::xx_xx, 0>(
82  shells[s1], shells[s2], shells[s1], shells[s2]);
83 
84  Eigen::Map<const MatrixLibInt> buf_mat(buf[0], n12, n12);
85 
86  result(s2, s1) = std::sqrt(buf_mat.cwiseAbs().maxCoeff());
87  }
88  }
89  return result.selfadjointView<Eigen::Upper>();
90 }

92 std::vector<std::vector<libint2::ShellPair>> ERIs::ComputeShellPairData(
93  const std::vector<libint2::Shell>& basis,
94  const std::vector<std::vector<Index>>& shellpairs) const {
95  std::vector<std::vector<libint2::ShellPair>> shellpairdata(basis.size());
96  const double ln_max_engine_precision =
97  std::log(std::numeric_limits<double>::epsilon() * 1e-10);
98 
99 #pragma omp parallel for schedule(dynamic)
100  for (Index s1 = 0; s1 < Index(shellpairs.size()); s1++) {
101  for (Index s2 : shellpairs[s1]) {
102  shellpairdata[s1].emplace_back(
103  libint2::ShellPair(basis[s1], basis[s2], ln_max_engine_precision));
104  }
105  }
106  return shellpairdata;
107 }

109 Eigen::MatrixXd ERIs::ComputeShellBlockNorm(const Eigen::MatrixXd& dmat) const {
110  Eigen::MatrixXd result =
111  Eigen::MatrixXd::Zero(starts_.size(), starts_.size());
112 #pragma omp parallel for schedule(dynamic)
113  for (Index s1 = 0l; s1 < Index(basis_.size()); ++s1) {
114  Index bf1 = starts_[s1];
115  Index n1 = basis_[s1].size();
116  for (Index s2 = 0l; s2 <= s1; ++s2) {
117  Index bf2 = starts_[s2];
118  Index n2 = basis_[s2].size();
119 
120  result(s2, s1) = dmat.block(bf2, bf1, n2, n1).cwiseAbs().maxCoeff();
121  }
122  }
123  return result.selfadjointView<Eigen::Upper>();
124 }

126 template <bool with_exchange>
127 std::array<Eigen::MatrixXd, 2> ERIs::Compute4c(const Eigen::MatrixXd& dmat,
128  double error) const {
129  assert(schwarzscreen_.rows() > 0 && schwarzscreen_.cols() > 0 &&
130  "Please call Initialize_4c before running this");
131  Index nthreads = OPENMP::getMaxThreads();
132 
133  Eigen::MatrixXd hartree = Eigen::MatrixXd::Zero(dmat.rows(), dmat.cols());
134  Eigen::MatrixXd exchange;
135  if (with_exchange) {
136  exchange = Eigen::MatrixXd::Zero(dmat.rows(), dmat.cols());
137  }
138  Eigen::MatrixXd dnorm_block = ComputeShellBlockNorm(dmat);
139  double fock_precision = error;
140  // engine precision controls primitive truncation, assume worst-case scenario
141  // (all primitive combinations add up constructively)
142  Index max_nprim4 = maxnprim_ * maxnprim_ * maxnprim_ * maxnprim_;
143  double engine_precision = std::min(fock_precision / dnorm_block.maxCoeff(),
144  std::numeric_limits<double>::epsilon()) /
145  double(max_nprim4);
146  std::vector<libint2::Engine> engines(nthreads);
147  engines[0] = libint2::Engine(libint2::Operator::coulomb, int(maxnprim_),
148  int(maxL_), 0);
149  engines[0].set_precision(engine_precision); // shellset-dependent precision
150  // control will likely break
151  // positive definiteness
152  // stick with this simple recipe
153  for (Index i = 1; i < nthreads; ++i) {
154  engines[i] = engines[0];
155  }
156  Index nshells = basis_.size();
157 
158 #pragma omp parallel for schedule(dynamic)reduction(+ : hartree)reduction(+: exchange)
159  for (Index s1 = 0; s1 < nshells; ++s1) {
160  Index thread_id = OPENMP::getThreadId();
161  libint2::Engine& engine = engines[thread_id];
162  const auto& buf = engine.results();
163  Index start_1 = starts_[s1];
164  const libint2::Shell& shell1 = basis_[s1];
165  Index n1 = shell1.size();
166 
167  auto sp12_iter = shellpairdata_[s1].begin();
168  for (Index s2 : shellpairs_[s1]) {
169  Index start_2 = starts_[s2];
170  const libint2::Shell& shell2 = basis_[s2];
171  Index n2 = shell2.size();
172  double dnorm_12 = dnorm_block(s1, s2);
173  const libint2::ShellPair* sp12 = &(*sp12_iter);
174  ++sp12_iter;
175 
176  for (Index s3 = 0; s3 <= s1; ++s3) {
177 
178  Index start_3 = starts_[s3];
179  const libint2::Shell& shell3 = basis_[s3];
180  Index n3 = shell3.size();
181  auto sp34_iter = shellpairdata_[s3].begin();
182  double dnorm_123 = std::max(dnorm_block(s1, s3),
183  std::max(dnorm_block(s2, s3), dnorm_12));
184  Index s4max = (s1 == s3) ? s2 : s3;
185  for (Index s4 : shellpairs_[s3]) {
186  if (s4 > s4max) {
187  break;
188  } // for each s3, s4 are stored in monotonically increasing
189  // order
190 
191  const libint2::ShellPair* sp34 = &(*sp34_iter);
192  // must update the iter even if going to skip s4
193  ++sp34_iter;
194  double dnorm_1234 =
195  std::max(dnorm_block(s1, s4),
196  std::max(dnorm_block(s2, s4),
197  std::max(dnorm_block(s3, s4), dnorm_123)));
198 
199  if (dnorm_1234 * schwarzscreen_(s1, s2) * schwarzscreen_(s3, s4) <
200  fock_precision) {
201  continue;
202  }
203 
204  const libint2::Shell& shell4 = basis_[s4];
205  engine
206  .compute2<libint2::Operator::coulomb, libint2::BraKet::xx_xx, 0>(
207  shell1, shell2, shell3, shell4, sp12, sp34);
208  const auto* buf_1234 = buf[0];
209  if (buf_1234 == nullptr) {
210  continue; // if all integrals screened out, skip to next quartet
211  }
212  Index start_4 = starts_[s4];
213  Index n4 = shell4.size();
214  Index s12_deg = (s1 == s2) ? 1 : 2;
215  Index s34_deg = (s3 == s4) ? 1 : 2;
216  Index s12_34_deg = (s1 == s3) ? (s2 == s4 ? 1 : 2) : 2;
217  Index s1234_deg = s12_deg * s34_deg * s12_34_deg;
218 
219  for (Index f1 = 0, f1234 = 0; f1 != n1; ++f1) {
220  const Index bf1 = f1 + start_1;
221  for (Index f2 = 0; f2 != n2; ++f2) {
222  const Index bf2 = f2 + start_2;
223  for (Index f3 = 0; f3 != n3; ++f3) {
224  const Index bf3 = f3 + start_3;
225  for (Index f4 = 0; f4 != n4; ++f4, ++f1234) {
226  const Index bf4 = f4 + start_4;
227 
228  const double value = buf_1234[f1234];
229 
230  const double value_scal_by_deg = value * double(s1234_deg);
231 
232  hartree(bf1, bf2) += dmat(bf3, bf4) * value_scal_by_deg;
233  hartree(bf3, bf4) += dmat(bf1, bf2) * value_scal_by_deg;
234  if (with_exchange) {
235  exchange(bf1, bf3) -= dmat(bf2, bf4) * value_scal_by_deg;
236  exchange(bf2, bf3) -= dmat(bf1, bf4) * value_scal_by_deg;
237  exchange(bf2, bf4) -= dmat(bf1, bf3) * value_scal_by_deg;
238  exchange(bf1, bf4) -= dmat(bf2, bf3) * value_scal_by_deg;
239  }
240  }
241  }
242  }
243  }
244  }
245  }
246  }
247  }
248  std::array<Eigen::MatrixXd, 2> result2;
249  // 0.25=0.5(symmetrisation)*0.5(our dmat has a factor 2)
250  result2[0] = 0.25 * (hartree + hartree.transpose());
251  if (with_exchange) {
252  // prefactor
253  result2[1] = 0.125 * (exchange + exchange.transpose());
254  }
255  return result2;
256 }

263 Eigen::MatrixXd ERIs::CalculateERIs_3c(const Eigen::MatrixXd& DMAT) const {
264  assert(_threecenter.size() > 0 &&
265  "Please call Initialize before running this");
266  Eigen::MatrixXd ERIs2 = Eigen::MatrixXd::Zero(DMAT.rows(), DMAT.cols());
267  Symmetric_Matrix dmat_sym = Symmetric_Matrix(DMAT);
268 #pragma omp parallel for schedule(guided) reduction(+ : ERIs2)
269  for (Index i = 0; i < _threecenter.size(); i++) {
270  const Symmetric_Matrix& threecenter = _threecenter[i];
271  // Trace over prod::DMAT,I(l)=componentwise product over
272  const double factor = threecenter.TraceofProd(dmat_sym);
273  Eigen::SelfAdjointView<Eigen::MatrixXd, Eigen::Upper> m =
274  ERIs2.selfadjointView<Eigen::Upper>();
275  threecenter.AddtoEigenUpperMatrix(m, factor);
276  }
277 
278  return ERIs2.selfadjointView<Eigen::Upper>();
279 }
