44 bool IntegralsExtractor::EvaluateFrame(ctp::Topology *top) {
45 
46  std::string xmlfile = Identify() + ".xml";
47 
48  tools::Property state("state", "", "");
49  tools::Property &pairs = state.add("pairs", "");
50 
51  using boost::format;
52 
53  // PAIRS
54  ctp::QMNBList::iterator pit;
55  ctp::QMNBList &nb = top->NBList();
56  for (pit = nb.begin(); pit != nb.end(); ++pit) {
57  ctp::QMPair *qmp = *pit;
58  tools::Property &pairprop = pairs.add("pair", "");
59  pairprop.add("id1", (format("%1$d") % qmp->Seg1()->getId()).str());
60  pairprop.add("name1", (format("%1$s") % qmp->Seg1()->getName()).str());
61  pairprop.add("id2", (format("%1$d") % qmp->Seg2()->getId()).str());
62  pairprop.add("name2", (format("%1$s") % qmp->Seg2()->getName()).str());
63 
64  if (qmp->isPathCarrier(+1)) {
65  tools::Property &channel = pairprop.add("channel", "");
66  channel.setAttribute("type", "hole");
67  channel.add("jeff2_h", (format("%1$1.7e") % qmp->getJeff2(+1)).str());
68  }
69  if (qmp->isPathCarrier(-1)) {
70  tools::Property &channel = pairprop.add("channel", "");
71  channel.setAttribute("type", "electron");
72  channel.add("jeff2_e", (format("%1$1.7e") % qmp->getJeff2(-1)).str());
73  }
74  }
75 
76  std::ofstream ofs;
77  ofs.open(xmlfile.c_str(), std::ofstream::out);
78  if (!ofs.is_open()) {
79  throw std::runtime_error("Bad file handle: " + xmlfile);
80  }
81  ofs << tools::XML << state;
82  ofs.close();
83 
84  return true;
85 }
