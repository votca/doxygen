51 string getIntStringId_(const unordered_map<string, Index> int_vals) {
52  vector<string> keys;
53  // Grab integer keys sort alphabetically and store in string_id
54  string int_string_id;
55  for (auto it : int_vals) {
56  keys.push_back(it.first);
57  }
58  sort(keys.begin(), keys.end());
59  for (auto key : keys) {
60  int_string_id.append(key);
61  auto it = int_vals.find(key);
62  int_string_id.append(lexical_cast<string>(it->second));
63  }
64  return int_string_id;
65 }
