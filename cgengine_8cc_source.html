00047     for(iter=mols.begin(); iter!=mols.end(); ++iter) {
00048         Molecule *mol = *iter;

00058                  << "--------------------------------------\n";
00059             continue;
00060         }
00061         Molecule *mcg = def->CreateMolecule(out);
00062         Map *map = def->CreateMap(*mol, *mcg);
