42 TopologyMap *CGEngine::CreateCGTopology(Topology &in, Topology &out)
43 {
44  MoleculeContainer &mols = in.Molecules();
45  MoleculeContainer::iterator iter;
46  TopologyMap *m = new TopologyMap(&in, &out);
47  for(iter=mols.begin(); iter!=mols.end(); ++iter) {
48  Molecule *mol = *iter;
49  if(IsIgnored(mol->getName())) continue;
50  CGMoleculeDef *def = getMoleculeDef(mol->getName());
51  if(!def) {
52  cout << "--------------------------------------\n"
53  << "WARNING: unknown molecule \"" << mol->getName() << "\" with id "
54  << mol->getId() << " in topology" << endl
55  << "molecule will not be mapped to CG representation\n"
56  << "Check weather a mapping file for all molecule exists, was specified in --cg "
57  << "separated by ; and the ident tag in xml-file matches the molecule name\n"
58  << "--------------------------------------\n";
59  continue;
60  }
61  Molecule *mcg = def->CreateMolecule(out);
62  Map *map = def->CreateMap(*mol, *mcg);
63  m->AddMoleculeMap(map);
64  }
65  out.RebuildExclusions();
66  return m;
67 }
