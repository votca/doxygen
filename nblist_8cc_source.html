35 void NBList::Generate(BeadList &list1, BeadList &list2, bool do_exclusions)
36 {
37  BeadList::iterator iter1;
38  BeadList::iterator iter2;
39  _do_exclusions = do_exclusions;
40 
41  if(list1.empty()) return;
42  if(list2.empty()) return;
43 
44  assert(list1.getTopology() == list2.getTopology());
45  Topology *top = list1.getTopology();
46 
47  for(iter1 = list1.begin(); iter1 != list1.end(); ++iter1) {
48  if(&list1 == &list2) {
49  iter2=iter1;
50  ++iter2;
51  }
52  else
53  iter2 = list2.begin();
54 
55  if(*iter1 == *iter2) continue;
56 
57  for(; iter2 != list2.end(); ++iter2) {
58  vec u = (*iter1)->getPos();
59  vec v = (*iter2)->getPos();
60 
61  vec r = top->BCShortestConnection(u, v);
62  double d = abs(r);
63  if(d < _cutoff){
64  if(_do_exclusions)
65  if(top->getExclusions().IsExcluded(*iter1, *iter2)) {
66  continue;
67  }
68  if((*_match_function)(*iter1, *iter2, r, d))
69  if(!FindPair(*iter1, *iter2))
70  AddPair( _pair_creator(*iter1, *iter2, r));
71  }
72  }
73  }
74 }
