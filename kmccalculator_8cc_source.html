32  void KMCCalculator::LoadGraph(ctp::Topology *top) {
33 
34  std::vector< ctp::Segment* >& seg = top->Segments();
35 
36  if(seg.size()<1){
37  throw std::runtime_error("Your sql file contains no segments!");
38  }
39 
40  for (unsigned i = 0; i < seg.size(); i++) {
41  GNode *newNode = new GNode();
42  newNode->ReadfromSegment(seg[i], _carriertype);
43  if (tools::wildcmp(_injection_name.c_str(), seg[i]->getName().c_str())) {
44  newNode->injectable = true;
45  } else {
46  newNode->injectable = false;
47  }
48  _nodes.push_back(newNode);
49  }
50 
51  ctp::QMNBList &nblist = top->NBList();
52  if(nblist.size()<1){
53  throw std::runtime_error("Your sql file contains no pairs!");
54  }
55 
56 
57  for (ctp::QMNBList::iterator it = nblist.begin(); it < nblist.end(); ++it) {
58  _nodes[(*it)->Seg1()->getId()-1]->AddEventfromQmPair(*it, _carriertype);
59  _nodes[(*it)->Seg2()->getId()-1]->AddEventfromQmPair(*it, _carriertype);
60  }
61 
62  unsigned events=0;
63  unsigned max=std::numeric_limits<unsigned>::min();
64  unsigned min=std::numeric_limits<unsigned>::max();
65  minlength=std::numeric_limits<double>::max();
66  double maxlength=0;
67  for(const auto& node:_nodes){
68 
69  unsigned size=node->events.size();
70  for( const auto& event:node->events){
71  if(event.decayevent){continue;}
72  double dist=abs(event.dr);
73  if(dist>maxlength){
74  maxlength=dist;
75  } else if(dist<minlength){
76  minlength=dist;
77  }
78  }
79 
80  events+=size;
81  if(size==0){
82  cout<<"Node "<<node->id<<" has 0 jumps"<<endl;
83  }
84  else if(size<min){
85  min=size;
86  }
87  else if(size>max){
88  max=size;
89  }
90  }
91  double avg=double(events)/double(_nodes.size());
92  double deviation=0.0;
93  for(const auto& node:_nodes){
94  double size=node->events.size();
95  deviation+=(size-avg)*(size-avg);
96  }
97  deviation=std::sqrt(deviation/double(_nodes.size()));
98 
99  cout<<"Nblist has "<<nblist.size()<<" pairs. Nodes contain "<<events<<" jump events"<<endl;
100  cout<<"with avg="<<avg<<" std="<<deviation<<" max="<<max<<" min="<<min<<endl;
101  cout<<"Minimum jumpdistance ="<<minlength<<" nm Maximum distance ="<<maxlength<<" nm"<<endl;
102  cout<<"Grouping into "<<lengthdistribution<<" boxes"<<endl;
103  lengthresolution=(1.00001*maxlength-minlength)/double(lengthdistribution);
104  cout<<"Resolution is "<<lengthresolution<<" nm"<<endl;
105 
106  _jumplengthdistro=std::vector<long unsigned>(lengthdistribution,0);
107  _jumplengthdistro_weighted=std::vector<double>(lengthdistribution,0);
108 
109 
110  cout << "spatial density: " << _numberofcharges / top->BoxVolume() << " nm^-3" << endl;
111 
112  for (unsigned int i = 0; i < _nodes.size(); i++) {
113  _nodes[i]->InitEscapeRate();
114  }
115 
116  return;
117  }

256  void KMCCalculator::InitialRates() {
257 
258  cout << endl << "Calculating initial Marcus rates." << endl;
259  cout << " Temperature T = " << _temperature << " K." << endl;
260 
261  cout << " carriertype: " << CarrierInttoLongString(_carriertype) << endl;
262  unsigned numberofsites = _nodes.size();
263  cout << " Rates for " << numberofsites << " sites are computed." << endl;
264  double charge=0.0;
265  if (_carriertype == -1)
266  {
267  charge = -1.0;
268  }
269  else if (_carriertype == 1)
270  {
271  charge = 1.0;
272  }
273  cout<<"electric field ="<<_field<<" V/nm"<<endl;
274 
275  double maxreldiff = 0;
276  double maxrate=0;
277  double minrate=std::numeric_limits<double>::max();
278  int totalnumberofrates = 0;
279  for (unsigned int i = 0; i < numberofsites; i++) {
280  unsigned numberofneighbours = _nodes[i]->events.size();
281  for (unsigned int j = 0; j < numberofneighbours; j++) {
282  if(_nodes[i]->events[j].decayevent){
283  //if event is a decay event there is no point in calculating its rate, because it already has that from the reading in.
284  continue;
285  }
286 
287  double destindex = _nodes[i]->events[j].destination;
288  double reorg = _nodes[i]->reorg_intorig + _nodes[destindex]->reorg_intdest + _nodes[i]->events[j].reorg_out;
289  if(std::abs(reorg)<1e-12){
290  throw std::runtime_error("Reorganisation energy for a pair is extremly close to zero,\n"
291  " you probably forgot to import reorganisation energies into your sql file.");
292  }
293  double dG_Field =0.0;
294  if(charge!=0.0){
295  dG_Field=charge * (_nodes[i]->events[j].dr*_field);
296  }
297  double dG_Site = _nodes[destindex]->siteenergy - _nodes[i]->siteenergy;
298  double dG=dG_Site-dG_Field;
299  double J2 = _nodes[i]->events[j].Jeff2;
300 
301  double rate = 2 * tools::conv::Pi / tools::conv::hbar * J2 / sqrt(4 * tools::conv::Pi * reorg * tools::conv::kB * _temperature)
302  * exp(-(dG + reorg)*(dG + reorg) / (4 * reorg * tools::conv::kB * _temperature));
303 
304  // calculate relative difference compared to values in the table
305  double reldiff = (_nodes[i]->events[j].rate - rate) / _nodes[i]->events[j].rate;
306  if (reldiff > maxreldiff) {
307  maxreldiff = reldiff;
308  }
309  reldiff = (_nodes[i]->events[j].rate - rate) / rate;
310  if (reldiff > maxreldiff) {
311  maxreldiff = reldiff;
312  }
313 
314  // set rates to calculated values
315  _nodes[i]->events[j].rate = rate;
316  _nodes[i]->events[j].initialrate = rate;
317 
318  if(rate>maxrate){
319  maxrate=rate;
320  }
321  else if(rate<minrate){
322  minrate=rate;
323  }
324 
325  totalnumberofrates++;
326  }
327 
328  // Initialise escape rates
329  for (unsigned int i = 0; i < _nodes.size(); i++) {
330  _nodes[i]->InitEscapeRate();
331  }
332 
333  }
334 
335  cout << " " << totalnumberofrates << " rates have been calculated." << endl;
336  cout<< " Largest rate="<<maxrate<<" 1/s Smallest rate="<<minrate<<" 1/s"<<endl;
337  if (maxreldiff < 0.01) {
338  cout << " Good agreement with rates in the state file. Maximal relative difference: " << maxreldiff * 100 << " %" << endl;
339  } else {
340  cout << " WARNING: Rates differ from those in the state file up to " << maxreldiff * 100 << " %." << " If the rates in the state file are calculated for a different temperature/field or if they are not Marcus rates, this is fine. Otherwise something might be wrong here." << endl;
341  }
342 
343  return;
344  }
