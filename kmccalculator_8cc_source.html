55 void KMCCalculator::LoadGraph(Topology& top) {
56 
57  std::vector<Segment>& segs = top.Segments();
58 
59  if (segs.size() < 1) {
60  throw std::runtime_error("Your state file contains no segments!");
61  }
62  _nodes.reserve(segs.size());
63  for (Segment& seg : segs) {
64  bool injectable = false;
65  if (tools::wildcmp(_injection_name, seg.getType())) {
66  injectable = true;
67  }
68  _nodes.push_back(GNode(seg, _carriertype, injectable));
69  }
70 
71  QMNBList& nblist = top.NBList();
72  if (nblist.size() < 1) {
73  throw std::runtime_error("neighborlist contains no pairs!");
74  }
75  if (_temperature <= 0) {
76  throw std::runtime_error(
77  "Your Temperature is negative or zero, please specify the temperature "
78  "in Kelvin.");
79  }
80 
81  Rate_Engine rate_engine(_temperature, _field);
82  cout << endl << "Calculating initial rates." << endl;
83  cout << rate_engine << endl;
84  cout << " carriertype: " << _carriertype.ToLongString() << endl;
85 
86  for (const QMPair* pair : nblist) {
87  Rate_Engine::PairRates rates = rate_engine.Rate(*pair, _carriertype);
88  _nodes[pair->Seg1()->getId()].AddEventfromQmPair(*pair, _nodes,
89  rates.rate12);
90  _nodes[pair->Seg2()->getId()].AddEventfromQmPair(*pair, _nodes,
91  rates.rate21);
92  }
93  _RandomVariable.setMaxInt(Index(_nodes.size()));
94  cout << " Rates for " << _nodes.size() << " sites are computed." << endl;
95  WriteRatestoFile(_ratefile, nblist);
96 
97  Index events = 0;
98  Index max = std::numeric_limits<Index>::min();
99  Index min = std::numeric_limits<Index>::max();
100  double minlength = std::numeric_limits<double>::max();
101  double maxlength = 0;
102  for (const auto& node : _nodes) {
103 
104  Index size = Index(node.Events().size());
105  for (const auto& event : node.Events()) {
106  if (event.isDecayEvent()) {
107  continue;
108  }
109  double dist = event.getDeltaR().norm();
110  if (dist > maxlength) {
111  maxlength = dist;
112  } else if (dist < minlength) {
113  minlength = dist;
114  }
115  }
116 
117  events += size;
118  if (size == 0) {
119  cout << "Node " << node.getId() << " has 0 jumps" << endl;
120  } else if (size < min) {
121  min = size;
122  } else if (size > max) {
123  max = size;
124  }
125  }
126  double avg = double(events) / double(_nodes.size());
127  double deviation = 0.0;
128  for (const auto& node : _nodes) {
129  double size = double(node.Events().size());
130  deviation += (size - avg) * (size - avg);
131  }
132  deviation = std::sqrt(deviation / double(_nodes.size()));
133 
134  cout << "Nblist has " << nblist.size() << " pairs. Nodes contain " << events
135  << " jump events" << endl;
136  cout << "with avg=" << avg << " std=" << deviation << " max=" << max
137  << " min=" << min << " jumps per site" << endl;
138  cout << "Minimum jumpdistance =" << minlength * tools::conv::bohr2nm
139  << " nm Maximum distance =" << maxlength * tools::conv::bohr2nm << " nm"
140  << endl;
141  double conv = std::pow(tools::conv::bohr2nm, 3);
142  cout << "spatial carrier density: "
143  << double(_numberofcarriers) / (top.BoxVolume() * conv) << " nm^-3"
144  << endl;
145 
146  for (auto& node : _nodes) {
147  node.InitEscapeRate();
148  node.MakeHuffTree();
149  }
150  return;
151 }

210 void KMCCalculator::RandomlyAssignCarriertoSite(Chargecarrier& Charge) {
211  Index nodeId_guess = -1;
212  do {
213  nodeId_guess = _RandomVariable.rand_uniform_int();
214  } while (_nodes[nodeId_guess].isOccupied() ||
215  _nodes[nodeId_guess].isInjectable() ==
216  false); // maybe already occupied? or maybe not injectable?
217  if (Charge.hasNode()) {
218  Charge.ReleaseNode();
219  }
220  Charge.settoNote(&_nodes[nodeId_guess]);
221 
222  return;

237 Chargecarrier* KMCCalculator::ChooseAffectedCarrier(double cumulated_rate) {
238  if (_carriers.size() == 1) {
239  return &_carriers[0];
240  }
241  Chargecarrier* carrier = nullptr;
242  double u = 1 - _RandomVariable.rand_uniform();
243  for (Index i = 0; i < _numberofcarriers; i++) {
244  u -= _carriers[i].getCurrentEscapeRate() / cumulated_rate;
245  if (u <= 0 || i == _numberofcarriers - 1) {
246  carrier = &_carriers[i];
247  break;
248  }
249  }
250  return carrier;
251 }
