64 void KMCCalculator::LoadGraph(Topology& top) {
65 
66  std::vector<Segment>& segs = top.Segments();
67 
68  if (segs.size() < 1) {
69  throw std::runtime_error("Your state file contains no segments!");
70  }
71  _nodes.reserve(segs.size());
72  for (Segment& seg : segs) {
73  bool injectable = false;
74  if (tools::wildcmp(_injection_name.c_str(), seg.getType().c_str())) {
75  injectable = true;
76  }
77  _nodes.push_back(GNode(seg, _carriertype, injectable));
78  }
79 
80  QMNBList& nblist = top.NBList();
81  if (nblist.size() < 1) {
82  throw std::runtime_error("neighborlist contains no pairs!");
83  }
84 
85  Rate_Engine rate_engine(_temperature, _field);
86  cout << endl << "Calculating initial rates." << endl;
87  cout << rate_engine << endl;
88  cout << " carriertype: " << _carriertype.ToLongString() << endl;
89 
90  for (const QMPair* pair : nblist) {
91  Rate_Engine::PairRates rates = rate_engine.Rate(*pair, _carriertype);
92  _nodes[pair->Seg1()->getId()].AddEventfromQmPair(*pair, _nodes,
93  rates.rate12);
94  _nodes[pair->Seg2()->getId()].AddEventfromQmPair(*pair, _nodes,
95  rates.rate21);
96  }
97  _RandomVariable.setMaxInt(_nodes.size());
98  cout << " Rates for " << _nodes.size() << " sites are computed." << endl;
99  WriteRatestoFile(_ratefile, nblist);
100 
101  unsigned events = 0;
102  unsigned max = std::numeric_limits<unsigned>::min();
103  unsigned min = std::numeric_limits<unsigned>::max();
104  double minlength = std::numeric_limits<double>::max();
105  double maxlength = 0;
106  for (const auto& node : _nodes) {
107 
108  unsigned size = node.Events().size();
109  for (const auto& event : node.Events()) {
110  if (event.isDecayEvent()) {
111  continue;
112  }
113  double dist = event.getDeltaR().norm();
114  if (dist > maxlength) {
115  maxlength = dist;
116  } else if (dist < minlength) {
117  minlength = dist;
118  }
119  }
120 
121  events += size;
122  if (size == 0) {
123  cout << "Node " << node.getId() << " has 0 jumps" << endl;
124  } else if (size < min) {
125  min = size;
126  } else if (size > max) {
127  max = size;
128  }
129  }
130  double avg = double(events) / double(_nodes.size());
131  double deviation = 0.0;
132  for (const auto& node : _nodes) {
133  double size = node.Events().size();
134  deviation += (size - avg) * (size - avg);
135  }
136  deviation = std::sqrt(deviation / double(_nodes.size()));
137 
138  cout << "Nblist has " << nblist.size() << " pairs. Nodes contain " << events
139  << " jump events" << endl;
140  cout << "with avg=" << avg << " std=" << deviation << " max=" << max
141  << " min=" << min << " jumps per site" << endl;
142  cout << "Minimum jumpdistance =" << minlength * tools::conv::bohr2nm
143  << " nm Maximum distance =" << maxlength * tools::conv::bohr2nm << " nm"
144  << endl;
145  double conv = std::pow(tools::conv::bohr2nm, 3);
146  cout << "spatial carrier density: "
147  << _numberofcarriers / (top.BoxVolume() * conv) << " nm^-3" << endl;
148 
149  for (auto& node : _nodes) {
150  node.InitEscapeRate();
151  node.MakeHuffTree();
152  }
153  return;
154 }

182  bool surrounded = true;
183  for (const auto& event : node.Events()) {
184  bool thisevent_possible = true;

186  if (event.getDestination() == fnode) {
187  thisevent_possible = false;

190  }

192  surrounded = false;
193  break;
194  }
195  }

213 void KMCCalculator::RandomlyAssignCarriertoSite(Chargecarrier& Charge) {
214  int nodeId_guess = -1;
215  do {
216  nodeId_guess = _RandomVariable.rand_uniform_int();
217  } while (_nodes[nodeId_guess].isOccupied() ||
218  _nodes[nodeId_guess].isInjectable() ==
219  false); // maybe already occupied? or maybe not injectable?
220  if (Charge.hasNode()) {
221  Charge.ReleaseNode();
222  }
223  Charge.settoNote(&_nodes[nodeId_guess]);
224 
225  return;

240 Chargecarrier* KMCCalculator::ChooseAffectedCarrier(double cumulated_rate) {
241  if (_carriers.size() == 1) {
242  return &_carriers[0];
243  }
244  Chargecarrier* carrier = NULL;
245  double u = 1 - _RandomVariable.rand_uniform();
246  for (int i = 0; i < _numberofcarriers; i++) {
247  u -= _carriers[i].getCurrentEscapeRate() / cumulated_rate;
248  if (u <= 0 || i == _numberofcarriers - 1) {
249  carrier = &_carriers[i];
250  break;
251  }
252  }
253  return carrier;
254 }
