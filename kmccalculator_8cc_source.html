35  void KMCCalculator::LoadGraph(ctp::Topology *top) {
36 
37  vector< ctp::Segment* >& seg = top->Segments();
38 
39  for (unsigned i = 0; i < seg.size(); i++) {
40  GNode *newNode = new GNode();
41  _nodes.push_back(newNode);
42  _nodes[i]->ReadfromSegment(seg[i], _carriertype);
43  if (tools::wildcmp(_injection_name.c_str(), seg[i]->getName().c_str())) {
44  _nodes[i]->injectable = true;
45  } else {
46  _nodes[i]->injectable = false;
47  }
48  }
49 
50  ctp::QMNBList &nblist = top->NBList();
51 
52  for (ctp::QMNBList::iterator it = nblist.begin(); it < nblist.end(); ++it) {
53  _nodes[(*it)->Seg1()->getId()]->AddEventfromQmPair(*it, _carriertype);
54  }
55 
56 
57  if (votca::tools::globals::verbose) {
58  cout << "pairs: " << nblist.size() / 2 << endl;
59  }
60  cout << "spatial density: " << _numberofcharges / top->BoxVolume() << " nm^-3" << endl;
61 
62  for (unsigned int i = 0; i < _nodes.size(); i++) {
63  _nodes[i]->InitEscapeRate();
64  }
65 
66  return;
67  }

206  void KMCCalculator::InitialRates() {
207 
208  cout << endl << "Calculating initial Marcus rates." << endl;
209  cout << " Temperature T = " << _temperature << " K." << endl;
210 
211  cout << " carriertype: " << CarrierInttoLongString(_carriertype) << endl;
212  unsigned numberofsites = _nodes.size();
213  cout << " Rates for " << numberofsites << " sites are computed." << endl;
214  double charge=0.0;
215  if (_carriertype == -1)
216  {
217  charge = -1.0;
218  }
219  else if (_carriertype == 1)
220  {
221  charge = 1.0;
222  }
223 
224 
225  double maxreldiff = 0;
226  int totalnumberofrates = 0;
227  for (unsigned int i = 0; i < numberofsites; i++) {
228  unsigned numberofneighbours = _nodes[i]->events.size();
229  for (unsigned int j = 0; j < numberofneighbours; j++) {
230  if(_nodes[i]->events[j].decayevent){
231  //if event is a decay event there is no point in calculating its rate, because it already has that from the reading in.
232  continue;
233  }
234 
235  double destindex = _nodes[i]->events[j].destination;
236  double reorg = _nodes[i]->reorg_intorig + _nodes[destindex]->reorg_intdest + _nodes[i]->events[j].reorg_out;
237 
238  double dG_Field =0.0;
239  if(charge!=0.0){
240  dG_Field=charge * (_nodes[i]->events[j].dr*_field);
241  }
242  double dG_Site = _nodes[destindex]->siteenergy - _nodes[i]->siteenergy;
243  double dG=dG_Site-dG_Field;
244  double J2 = _nodes[i]->events[j].Jeff2;
245 
246  double rate = 2 * tools::conv::Pi / tools::conv::hbar * J2 / sqrt(4 * tools::conv::Pi * reorg * tools::conv::kB * _temperature)
247  * exp(-(dG + reorg)*(dG + reorg) / (4 * reorg * tools::conv::kB * _temperature));
248 
249  // calculate relative difference compared to values in the table
250  double reldiff = (_nodes[i]->events[j].rate - rate) / _nodes[i]->events[j].rate;
251  if (reldiff > maxreldiff) {
252  maxreldiff = reldiff;
253  }
254  reldiff = (_nodes[i]->events[j].rate - rate) / rate;
255  if (reldiff > maxreldiff) {
256  maxreldiff = reldiff;
257  }
258 
259  // set rates to calculated values
260  _nodes[i]->events[j].rate = rate;
261  _nodes[i]->events[j].initialrate = rate;
262 
263  totalnumberofrates++;
264  }
265 
266  // Initialise escape rates
267  for (unsigned int i = 0; i < _nodes.size(); i++) {
268  _nodes[i]->InitEscapeRate();
269  }
270 
271  }
272 
273  cout << " " << totalnumberofrates << " rates have been calculated." << endl;
274  if (maxreldiff < 0.01) {
275  cout << " Good agreement with rates in the state file. Maximal relative difference: " << maxreldiff * 100 << " %" << endl;
276  } else {
277  cout << " WARNING: Rates differ from those in the state file up to " << maxreldiff * 100 << " %." << " If the rates in the state file are calculated for a different temperature/field or if they are not Marcus rates, this is fine. Otherwise something might be wrong here." << endl;
278  }
279 
280  return;
281  }

286  double KMCCalculator::Promotetime(double cumulated_rate){
287  double dt = 0;
288  double rand_u = 1 - _RandomVariable->rand_uniform();
289  while (rand_u == 0) {
290  cout << "WARNING: encountered 0 as a random variable! New try." << endl;
291  rand_u = 1 - _RandomVariable->rand_uniform();
292  }
293  dt = -1 / cumulated_rate * log(rand_u);
