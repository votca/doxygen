56void KMCCalculator::LoadGraph(Topology& top) {
57
58 std::vector<Segment>& segs = top.Segments();
59
60 if (segs.size() < 1) {
61 throw std::runtime_error("Your state file contains no segments!");
62 }
63 nodes_.reserve(segs.size());
64 for (Segment& seg : segs) {
65 bool injectable = false;
66 if (tools::wildcmp(injection_name_, seg.getType())) {
67 injectable = true;
68 }
69 nodes_.push_back(GNode(seg, carriertype_, injectable));
70 }
71
72 QMNBList& nblist = top.NBList();
73 if (nblist.size() < 1) {
74 throw std::runtime_error("neighborlist contains no pairs!");
75 }
76 if (temperature_ <= 0) {
77 throw std::runtime_error(
78 "Your Temperature is negative or zero, please specify the temperature "
79 "in Kelvin.");
80 }
81
82 Rate_Engine rate_engine(temperature_, field_);
83 XTP_LOG(Log::error, log_) << "\nCalculating initial rates." << std::flush;
84 XTP_LOG(Log::error, log_) << rate_engine << std::flush;
85 XTP_LOG(Log::error, log_)
86 << " carriertype: " << carriertype_.ToLongString() << std::flush;
87
88 for (const QMPair* pair : nblist) {
89 Rate_Engine::PairRates rates = rate_engine.Rate(*pair, carriertype_);
90 nodes_[pair->Seg1()->getId()].AddEventfromQmPair(*pair, nodes_,
91 rates.rate12);
92 nodes_[pair->Seg2()->getId()].AddEventfromQmPair(*pair, nodes_,
93 rates.rate21);
94 }
95 RandomVariable_.setMaxInt(Index(nodes_.size()));
96 XTP_LOG(Log::error, log_) << " Rates for " << nodes_.size()
97 << " sites are computed." << std::flush;
98 WriteRatestoFile(ratefile_, nblist);
99
100 Index events = 0;
101 Index max = std::numeric_limits<Index>::min();
102 Index min = std::numeric_limits<Index>::max();
103 double minlength = std::numeric_limits<double>::max();
104 double maxlength = 0;
105 for (const auto& node : nodes_) {
106
107 Index size = Index(node.Events().size());
108 for (const auto& event : node.Events()) {
109 if (event.isDecayEvent()) {
110 continue;
111 }
112 double dist = event.getDeltaR().norm();
113 if (dist > maxlength) {
114 maxlength = dist;
115 } else if (dist < minlength) {
116 minlength = dist;
117 }
118 }
119
120 events += size;
121 if (size == 0) {
122 XTP_LOG(Log::info, log_)
123 << "Node " << node.getId() << " has 0 jumps" << std::flush;
124 } else if (size < min) {
125 min = size;
126 } else if (size > max) {
127 max = size;
128 }
129 }
130 double avg = double(events) / double(nodes_.size());
131 double deviation = 0.0;
132 for (const auto& node : nodes_) {
133 double size = double(node.Events().size());
134 deviation += (size - avg) * (size - avg);
135 }
136 deviation = std::sqrt(deviation / double(nodes_.size()));
137
138 XTP_LOG(Log::error, log_)
139 << "Nblist has " << nblist.size() << " pairs. Nodes contain " << events
140 << " jump events\n"
141 << "with avg=" << avg << " std=" << deviation << " max=" << max
142 << " min=" << min << " jumps per site\n"
143 << "Minimum jumpdistance =" << minlength * tools::conv::bohr2nm
144 << " nm Maximum distance =" << maxlength * tools::conv::bohr2nm << " nm\n"
145 << std::flush;
146 double conv = std::pow(tools::conv::bohr2nm, 3);
147 XTP_LOG(Log::error, log_)
148 << "spatial carrier density: "
149 << double(numberofcarriers_) / (top.BoxVolume() * conv) << " nm^-3"
150 << std::flush;
151
152 for (auto& node : nodes_) {
153 node.InitEscapeRate();
154 node.MakeHuffTree();
155 }
156 return;
157}

171bool KMCCalculator::CheckForbidden(
172 const GNode& node, const std::vector<GNode*>& forbiddenlist) const {
173 bool forbidden = false;
174 for (const GNode* fnode : forbiddenlist) {

178 }

183bool KMCCalculator::CheckSurrounded(
184 const GNode& node, const std::vector<GNode*>& forbiddendests) const {
185 bool surrounded = true;
186 for (const auto& event : node.Events()) {
187 bool thisevent_possible = true;
188 for (const GNode* fnode : forbiddendests) {
189 if (event.getDestination() == fnode) {
190 thisevent_possible = false;
191 break;
192 }
193 }
194 if (thisevent_possible == true) {
195 surrounded = false;
196 break;
197 }
198 }
199 return surrounded;
200}

203
204 XTP_LOG(Log::error, log_) << "looking for injectable nodes..." << std::flush;
205 for (Index i = 0; i < numberofcarriers_; i++) {
206 Chargecarrier newCharge(i);
207 RandomlyAssignCarriertoSite(newCharge);

217void KMCCalculator::RandomlyAssignCarriertoSite(Chargecarrier& Charge) {
218 Index nodeId_guess = -1;
219 do {
220 nodeId_guess = RandomVariable_.rand_uniform_int();
221 } while (nodes_[nodeId_guess].isOccupied() ||
222 nodes_[nodeId_guess].isInjectable() ==
223 false); // maybe already occupied? or maybe not injectable?
224 if (Charge.hasNode()) {
225 Charge.ReleaseNode();
226 }
227 Charge.settoNote(&nodes_[nodeId_guess]);
228
229 return;
230}

232double KMCCalculator::Promotetime(double cumulated_rate) {
233 double dt = 0;
234 double rand_u = 1 - RandomVariable_.rand_uniform();
235 dt = -1 / cumulated_rate * std::log(rand_u);
236 return dt;
237}

244Chargecarrier* KMCCalculator::ChooseAffectedCarrier(double cumulated_rate) {
245 if (carriers_.size() == 1) {
246 return &carriers_[0];
247 }
248 Chargecarrier* carrier = nullptr;
249 double u = 1 - RandomVariable_.rand_uniform();
250 for (Index i = 0; i < numberofcarriers_; i++) {
251 u -= carriers_[i].getCurrentEscapeRate() / cumulated_rate;
252 if (u <= 0 || i == numberofcarriers_ - 1) {
253 carrier = &carriers_[i];
254 break;
255 }
256 }
257 return carrier;
258}
259void KMCCalculator::WriteRatestoFile(std::string filename,
260 const QMNBList& nblist) {
261 XTP_LOG(Log::error, log_)
262 << "\nRates are written to " << filename << std::flush;
263 fstream ratefs;
264 ratefs.open(filename, fstream::out);
265 ratefs << "#PairID,SiteID1,SiteID2, ,rate12[1/s],rate21[1/s] at "
266 << temperature_ * tools::conv::hrt2ev / tools::conv::kB
267 << "K for carrier:" << carriertype_.ToString() << endl;
268
269 Rate_Engine rate_engine(temperature_, field_);
270 for (const QMPair* pair : nblist) {
271 Rate_Engine::PairRates rates = rate_engine.Rate(*pair, carriertype_);
272 ratefs << pair->getId() << " " << pair->Seg1()->getId() << " "
273 << pair->Seg2()->getId() << " " << rates.rate12 << " "
274 << rates.rate21 << "\n";
275 }
276 ratefs << std::flush;
277 ratefs.close();
278}
