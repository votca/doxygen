29 void PMLocalization::computePML(Orbitals &orbitals) {
30  occupied_orbitals = orbitals.MOs().eigenvectors().leftCols(
31  orbitals.getNumberOfAlphaElectrons());
32  aobasis_ = orbitals.getDftBasis();
33  AOOverlap overlap;
34  overlap.Fill(aobasis_);
35  overlap_ = overlap.Matrix();
36 
37  XTP_LOG(Log::error, log_) << std::flush;
38  XTP_LOG(Log::error, log_)
39  << TimeStamp() << " Starting localization of orbitals" << std::flush;
40 
41  // determine initial penalty_function
42  initial_penalty();
43 
44  Index iteration = 1;
45 
46  while (iteration < nrOfIterations_) {
47 
48  XTP_LOG(Log::info, log_) << "Iteration: " << iteration << std::flush;
49 
50  Index maxrow, maxcol;
51  double max_penalty = PM_penalty_.maxCoeff(&maxrow, &maxcol);
52 
53  XTP_LOG(Log::info, log_)
54  << "maximum of penalty function: " << max_penalty << std::flush;
55 
56  if (max_penalty < convergence_limit_) break;
57 
58  XTP_LOG(Log::info, log_)
59  << "Orbitals to be changed: " << maxrow << " " << maxcol << std::flush;
60 
61  Eigen::MatrixX2d max_orbs(occupied_orbitals.rows(), 2);
62  max_orbs << occupied_orbitals.col(maxrow), occupied_orbitals.col(maxcol);
63  Eigen::MatrixX2d rotated_orbs = rotateorbitals(max_orbs, maxrow, maxcol);
64  occupied_orbitals.col(maxrow) = rotated_orbs.col(0);
65  occupied_orbitals.col(maxcol) = rotated_orbs.col(1);
66 
67  update_penalty(maxrow, maxcol);
68 
69  iteration++;
70  }
71  if (iteration == nrOfIterations_) {
72  throw std::runtime_error(
73  "Localization with Jacobi-Sweeps did not converge");
74  }
75  XTP_LOG(Log::error, log_) << TimeStamp() << " Orbitals localized after "
76  << iteration + 1 << " iterations" << std::flush;
77  orbitals.setPMLocalizedOrbital(occupied_orbitals);
78 }

80 // Function to rotate the 2 maximum orbitals (s and t)
81 Eigen::MatrixX2d PMLocalization::rotateorbitals(const Eigen::MatrixX2d &maxorbs,
82  const Index s, const Index t) {
83  const double gamma =
84  0.25 *
85  asin(B_(s, t) / sqrt((A_(s, t) * A_(s, t)) + (B_(s, t) * B_(s, t))));
86  Eigen::MatrixX2d rotatedorbitals(maxorbs.rows(), 2);
87  rotatedorbitals.col(0) =
88  (std::cos(gamma) * maxorbs.col(0)) + (std::sin(gamma) * maxorbs.col(1));
89  rotatedorbitals.col(1) = -1 * (std::sin(gamma) * maxorbs.col(0)) +
90  (std::cos(gamma) * maxorbs.col(1));
91  XTP_LOG(Log::info, log_) << "Sine of the rotation angle = " << std::sin(gamma)
92  << std::flush;
93  return rotatedorbitals;
94 }

96 Eigen::VectorXd PMLocalization::pop_per_atom(const Eigen::VectorXd &orbital) {
97 
98  Eigen::RowVectorXd MullikenPop_orb_per_basis =
99  (orbital.asDiagonal() * overlap_ * orbital.asDiagonal()).colwise().sum();
100  Index start = 0;
101 
102  Eigen::VectorXd per_atom = Eigen::VectorXd::Zero(Index(numfuncpatom_.size()));
103  for (Index atom_id = 0; atom_id < Index(numfuncpatom_.size()); atom_id++) {
104  per_atom(atom_id) =
105  MullikenPop_orb_per_basis.segment(start, numfuncpatom_[atom_id]).sum();
106  start += numfuncpatom_[atom_id];
107  }
108 
109  return per_atom;
110 }

112 // Determine PM cost function based on Mulliken populations
113 void PMLocalization::initial_penalty() {
114 
115  PM_penalty_ =
116  Eigen::MatrixXd::Zero(occupied_orbitals.cols(), occupied_orbitals.cols());
117  // Variable names A and B are used directly as described in the paper above
118  A_ =
119  Eigen::MatrixXd::Zero(occupied_orbitals.cols(), occupied_orbitals.cols());
120  B_ =
121  Eigen::MatrixXd::Zero(occupied_orbitals.cols(), occupied_orbitals.cols());
122 
123  numfuncpatom_ = aobasis_.getFuncPerAtom();
124 
125  // get the s-s elements first ("diagonal in orbital")
126  MullikenPop_orb_per_atom_ = Eigen::MatrixXd::Zero(
127  occupied_orbitals.cols(), Index(numfuncpatom_.size()));
128 #pragma omp parallel for
129  for (Index s = 0; s < occupied_orbitals.cols(); s++) {
130  MullikenPop_orb_per_atom_.row(s) = pop_per_atom(occupied_orbitals.col(s));
131  }
132 
133 // now we only need to calculate the off-diagonals explicitly
134 #pragma omp parallel for
135  for (Index s = 0; s < occupied_orbitals.cols(); s++) {
136  Eigen::MatrixXd s_overlap =
137  occupied_orbitals.col(s).asDiagonal() * overlap_;
138 
139  for (Index t = s + 1; t < occupied_orbitals.cols(); t++) {
140 
141  Eigen::Vector2d temp = offdiag_penalty_elements(s_overlap, s, t);
142 
143  A_(s, t) = temp(0);
144  B_(s, t) = temp(1);
145  PM_penalty_(s, t) =
146  A_(s, t) + sqrt((A_(s, t) * A_(s, t)) + (B_(s, t) * B_(s, t)));
147  }
148  }
149  return;
150 }

152 Eigen::Vector2d PMLocalization::offdiag_penalty_elements(
153  const Eigen::MatrixXd &s_overlap, Index s, Index t) {
154 
155  Eigen::MatrixXd splitwiseMullikenPop_orb_SandT =
156  s_overlap * occupied_orbitals.col(t).asDiagonal();
157  Eigen::RowVectorXd MullikenPop_orb_SandT_per_basis =
158  0.5 * (splitwiseMullikenPop_orb_SandT.colwise().sum() +
159  splitwiseMullikenPop_orb_SandT.rowwise().sum().transpose());
160 
161  Index start =
162  0; // This helps to sum only over the basis functions on an atom
163  double Ast = 0;
164  double Bst = 0;
165 
166  for (Index atom_id = 0; atom_id < Index(numfuncpatom_.size()); atom_id++) {
167  double MullikenPop_orb_SandT_per_atom =
168  MullikenPop_orb_SandT_per_basis.segment(start, numfuncpatom_[atom_id])
169  .sum();
170 
171  Ast += MullikenPop_orb_SandT_per_atom * MullikenPop_orb_SandT_per_atom -
172  0.25 * ((MullikenPop_orb_per_atom_(s, atom_id) -
173  MullikenPop_orb_per_atom_(t, atom_id)) *
174  (MullikenPop_orb_per_atom_(s, atom_id) -
175  MullikenPop_orb_per_atom_(t, atom_id)));
176 
177  Bst += MullikenPop_orb_SandT_per_atom *
178  (MullikenPop_orb_per_atom_(s, atom_id) -
179  MullikenPop_orb_per_atom_(t, atom_id));
180  start += numfuncpatom_[atom_id];
181  }
182 
183  Eigen::Vector2d out(Ast, Bst);
184 
185  return out;
186 }

188 // Update PM cost function based on Mulliken populations after rotations
189 void PMLocalization::update_penalty(Index orb1, Index orb2) {
190 
191  // update the get the s-s elements for orb1 and orb2
192 #pragma omp parallel for
193  for (Index s = 0; s < occupied_orbitals.cols(); s++) {
194  if (s == orb1 || s == orb2) {
195 
196  MullikenPop_orb_per_atom_.row(s) = pop_per_atom(occupied_orbitals.col(s));
197  }
198  }
199 
200 // now we only need to calculate the off-diagonals explicitly for all
201 // pairs involving orb1 or orb2
202 #pragma omp parallel for
203  for (Index s = 0; s < occupied_orbitals.cols(); s++) {
204  Eigen::MatrixXd s_overlap =
205  occupied_orbitals.col(s).asDiagonal() * overlap_;
206 
207  for (Index t = s + 1; t < occupied_orbitals.cols(); t++) {
208 
209  // we do this only if any of s or t matches orb1 or orb2
210  if (s == orb1 || s == orb2 || t == orb1 || t == orb2) {
211 
212  Eigen::Vector2d temp = offdiag_penalty_elements(s_overlap, s, t);
213  A_(s, t) = temp(0);
214  B_(s, t) = temp(1);
215  PM_penalty_(s, t) =
216  A_(s, t) + sqrt((A_(s, t) * A_(s, t)) + (B_(s, t) * B_(s, t)));
217  }
218  }
219  }
220  return;
221 }
