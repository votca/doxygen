00135     BeadList b;

00137     NBListGrid nb;

00146 // the user is required to define how to merge the single data
00147 // belonging to each thread into the main data belonging to CsgTestApp
00148 void CsgTestApp::MergeWorker(Worker *worker) {
00149     RDFWorker * myRDFWorker;
00150 // cast generel Worker into your derived worker class(here RDFWorker)
00151     myRDFWorker = dynamic_cast<RDFWorker*> (worker);
00152 
00153 // the next comment block explains how mutexes are used internally for this function:
00154 // mutexes are used to exclusively work on data
00155 // e.g., if you read or write global data, make sure that nobody else (i.e. no other worker)
00156 // works on that very same piece of data at the same time; otherwise,
00157 // you will end up with wrong results that you struggle to understand
00158 // the parent class handles a "merging mutex" for you internally; this is what happens:
00159 // first, a mutex is created, e.g.
00160 //        Mutex rdfMutex;
00161 // then, for each worker, the mutex is first locked
00162 //        rdfMutex.Lock())
00163 // and MergeWorker(worker) is called (i.e. the code you define here is executed)
00164 // after MergeWorker exits, the mutex is unlocked
00165 //        rdfMutex.Unlock();
00166 // and allows other threads to get a lock and start merging
00167 
00168 // now follows your code
00169 
00170 // merging of data in this simple example is easy and does not have to follow
00171 // the original order of frames (since plain summing is commutative)
00172     _rdf.data().y() = _rdf.data().y() + myRDFWorker->_rdf.data().y();
00173 
00174 }
