52  atoms.push_back(QMAtom(32, "O", Eigen::Vector3d::Random()));
53  atoms.push_back(QMAtom(100, "O", Eigen::Vector3d::Random()));

55  atoms.push_back(QMAtom(3145, "N", Eigen::Vector3d::Random()));

58  double qmEnergy = -2.1025e-3;

60  std::string qmPackage = "NOPE";

62  Index rpaMin = '?';

84  Eigen::VectorXd BSETripletEnergiesTest = Eigen::VectorXd::Random(33);
85  Eigen::MatrixXd BSETripletCoefficientsTest = Eigen::MatrixXd::Random(33, 31);
86
