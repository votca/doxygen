39BOOST_AUTO_TEST_CASE(checkpoint_file_test) {
40 libint2::initialize();
41
42 Index occupiedLevels = 4;
43 Index unoccupiedLevels = 13;
44 Index numElectrons = 12;
45
46 Eigen::VectorXd moeTest = Eigen::VectorXd::Random(17);
47 Eigen::MatrixXd mocTest = Eigen::MatrixXd::Random(17, 17);
48
49 QMMolecule atoms = QMMolecule(" ", 0);
50 atoms.LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) + "/hdf5/molecule.xyz");
51
52 double qmEnergy = -2.1025e-3;
53
54 std::string qmPackage = "NOPE";
55
56 Index rpaMin = '?';
57 Index rpaMax = 1e3;
58
59 Index bseVmin = -6019386;
60 Index bseCmax = 42;
61
62 double scaHfx = 3.14159;
63
64 bool useTDA = true;
65
66 Eigen::MatrixXd vxcTest = Eigen::MatrixXd::Random(200, 200);
67 std::string someECP = "aye aye Cap'n";
68
69 Eigen::VectorXd QPpertEnergiesTest = Eigen::VectorXd::Random(31);
70 Eigen::MatrixXd QPdiagEnergiesTest = Eigen::VectorXd::Random(21);
71 Eigen::MatrixXd QPdiagCoefficientsTest = Eigen::MatrixXd::Identity(42, 42);
72
73 Eigen::VectorXd BSESingletEnergiesTest = Eigen::VectorXd::Random(25);
74 Eigen::MatrixXd BSESingletCoefficientsTest = Eigen::MatrixXd::Random(25, 38);
75 Eigen::MatrixXd BSESingletCoefficientsARTest =
76 Eigen::MatrixXd::Random(42, 42);
77
78 Eigen::VectorXd BSETripletEnergiesTest = Eigen::VectorXd::Random(33);
79 Eigen::MatrixXd BSETripletCoefficientsTest = Eigen::MatrixXd::Random(33, 31);
80
81 Index basisSetSize = 0;
82
83 // Write orbitals
84 Orbitals orbWrite;
85 orbWrite.setNumberOfOccupiedLevels(occupiedLevels);
86 orbWrite.setNumberOfAlphaElectrons(numElectrons);
87 orbWrite.MOs().eigenvalues() = moeTest;
88 orbWrite.MOs().eigenvectors() = mocTest;
89
90 orbWrite.QMAtoms() = atoms;
91 orbWrite.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) + "/hdf5/3-21G.xml");
92 basisSetSize = orbWrite.getBasisSetSize();
93
94 orbWrite.setQMEnergy(qmEnergy);
95 orbWrite.setQMpackage(qmPackage);
96 orbWrite.setRPAindices(rpaMin, rpaMax);
97 // no need to write qpmin, qpmax
98 orbWrite.setBSEindices(bseVmin, bseCmax);
99 orbWrite.setScaHFX(scaHfx);
100 orbWrite.setTDAApprox(useTDA);
101 orbWrite.setECPName(someECP);
102 orbWrite.QPpertEnergies() = QPpertEnergiesTest;
103 orbWrite.QPdiag().eigenvalues() = QPdiagEnergiesTest;
104 orbWrite.QPdiag().eigenvectors() = QPdiagCoefficientsTest;
105 orbWrite.BSESinglets().eigenvalues() = BSESingletEnergiesTest;
106 orbWrite.BSESinglets().eigenvectors() = BSESingletCoefficientsTest;
107 orbWrite.BSESinglets().eigenvectors2() = BSESingletCoefficientsARTest;
108 orbWrite.BSETriplets().eigenvalues() = BSETripletEnergiesTest;
109 orbWrite.BSETriplets().eigenvectors() = BSETripletCoefficientsTest;
110
111 orbWrite.WriteToCpt("xtp_testing.hdf5");
112
113 // Read Orbitals
114 Orbitals orbRead;
115 orbRead.ReadFromCpt("xtp_testing.hdf5");
116
117 double tol = 1e-6;
118
119 // Test the read values
120 BOOST_CHECK_EQUAL(orbRead.getBasisSetSize(), basisSetSize);
121 BOOST_CHECK_EQUAL(orbRead.getBasisSetSize(),
122 occupiedLevels + unoccupiedLevels);
123 BOOST_CHECK_EQUAL(orbRead.getNumberOfAlphaElectrons(), numElectrons);
124 BOOST_CHECK(orbRead.MOs().eigenvalues().isApprox(moeTest, tol));
125
126 BOOST_CHECK(orbRead.MOs().eigenvectors().isApprox(mocTest, tol));
127 BOOST_CHECK_CLOSE(orbRead.getDFTTotalEnergy(), qmEnergy, tol);
128 BOOST_CHECK_EQUAL(orbRead.getQMpackage(), qmPackage);
129 BOOST_CHECK_EQUAL(orbRead.getRPAmin(), rpaMin);
130 BOOST_CHECK_EQUAL(orbRead.getRPAmax(), rpaMax);
131
132 BOOST_CHECK_EQUAL(orbRead.getBSEvmin(), bseVmin);
133 BOOST_CHECK_EQUAL(orbRead.getBSEcmax(), bseCmax);
134
135 BOOST_CHECK_CLOSE(orbRead.getScaHFX(), scaHfx, tol);
136 BOOST_CHECK_EQUAL(orbRead.getTDAApprox(), useTDA);
137 BOOST_CHECK_EQUAL(orbRead.getECPName(), someECP);
138 BOOST_CHECK(orbRead.QPpertEnergies().isApprox(QPpertEnergiesTest, tol));
139 BOOST_CHECK(orbRead.QPdiag().eigenvalues().isApprox(QPdiagEnergiesTest, tol));
140 BOOST_CHECK(orbRead.QPdiag().eigenvectors().isApprox(QPdiagCoefficientsTest));
141 BOOST_CHECK(orbRead.BSESinglets().eigenvalues().isApprox(
142 BSESingletEnergiesTest, tol));
143 BOOST_CHECK(orbRead.BSESinglets().eigenvectors().isApprox(
144 BSESingletCoefficientsTest, tol));
145 BOOST_CHECK(orbRead.BSESinglets().eigenvectors2().isApprox(
146 BSESingletCoefficientsARTest, tol));
147 BOOST_CHECK(orbRead.BSETriplets().eigenvalues().isApprox(
148 BSETripletEnergiesTest, tol));
149 BOOST_CHECK(orbRead.BSETriplets().eigenvectors().isApprox(
150 BSETripletCoefficientsTest, tol));
151
152 BOOST_REQUIRE_EQUAL(orbRead.QMAtoms().size(), atoms.size());
153
154 for (Index i = 0; i < atoms.size(); ++i) {
155 const auto& atomRead = orbRead.QMAtoms()[i];
156 const auto& atomTest = atoms[i];
157 BOOST_CHECK_EQUAL(atomRead.getId(), atomTest.getId());
158 BOOST_CHECK(atomRead.getPos().isApprox(atomTest.getPos(), tol));
159 BOOST_CHECK_EQUAL(atomRead.getNuccharge(), atomTest.getNuccharge());
160 BOOST_CHECK_EQUAL(atomRead.getElement(), atomTest.getElement());
161 }
162
163 // Check if basissets are equal after a write read cycle
164 std::stringstream ssWrite;
165 ssWrite << orbWrite.getDftBasis();
166 std::stringstream ssRead;
167 ssRead << orbRead.getDftBasis();
168
169 bool basissets_equal = (ssWrite.str() == ssRead.str());
170 BOOST_CHECK(basissets_equal);
171
172 if (!basissets_equal) {
173 std::cout << "Thing we wrote:" << std::endl;
174 std::cout << ssWrite.str();
175 std::cout << "Thing we read:" << std::endl;
176 std::cout << ssRead.str();
177 }
178
179 libint2::finalize();
180}

232BOOST_AUTO_TEST_CASE(staticsegment) {
233
234 CheckpointFile cpf("xtp_staticsegment.hdf5");
235 CheckpointWriter w = cpf.getWriter();
236 StaticSegment seg = StaticSegment("test", 0);
237 for (Index i = 0; i < 10; ++i) {
238 seg.push_back(StaticSite(0, "O", Eigen::Vector3d::Random()));
239 seg.push_back(StaticSite(25, "O", Eigen::Vector3d::Random()));
240 seg.push_back(StaticSite(32, "O", Eigen::Vector3d::Random()));
241 seg.push_back(StaticSite(100, "O", Eigen::Vector3d::Random()));
242 seg.push_back(StaticSite(2, "Si", Eigen::Vector3d::Random()));
243 seg.push_back(StaticSite(3145, "N", Eigen::Vector3d::Random()));
244 }
245
246 seg.WriteToCpt(w);
247 CheckpointReader r = cpf.getReader();
248 StaticSegment seg2 = StaticSegment(r);
249
250 BOOST_REQUIRE_EQUAL(seg2.size(), seg.size());
251 for (Index i = 0; i < seg.size(); ++i) {
252 const auto& atomRead = seg2[i];
253 const auto& atomTest = seg[i];
254 BOOST_CHECK_EQUAL(atomRead.getId(), atomTest.getId());
255 BOOST_CHECK(atomRead.getPos().isApprox(atomTest.getPos(), 1e-7));
256 BOOST_CHECK_EQUAL(atomRead.getCharge(), atomTest.getCharge());
257 BOOST_CHECK_EQUAL(atomRead.getElement(), atomTest.getElement());
258 }
259}
