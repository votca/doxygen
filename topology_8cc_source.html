313 BoundaryCondition::eBoxtype Topology::autoDetectBoxType(const matrix &box) {
314  // set the box type to OpenBox in case "box" is the zero matrix,
315  // to OrthorhombicBox in case "box" is a diagonal matrix,
316  // or to TriclinicBox otherwise
317  if(box.get(0,0)==0 && box.get(0,1)==0 && box.get(0,2)==0 &&
318  box.get(1,0)==0 && box.get(1,1)==0 && box.get(1,2)==0 &&
319  box.get(2,0)==0 && box.get(2,1)==0 && box.get(2,2)==0) {
320  //cout << "box open\n";
321  return BoundaryCondition::typeOpen;
322  }
323  else
324  if(box.get(0,1)==0 && box.get(0,2)==0 &&
325  box.get(1,0)==0 && box.get(1,2)==0 &&
326  box.get(2,0)==0 && box.get(2,1)==0) {
327  //cout << "box orth\n";
328  return BoundaryCondition::typeOrthorhombic;
329  }
330  else {
331  //cout << "box tric\n";
332  return BoundaryCondition::typeTriclinic;
333  }
334  return BoundaryCondition::typeOpen;
335 }

337 double Topology::ShortestBoxSize()
338 {
339  vec _box_a = getBox().getCol(0);
340  vec _box_b = getBox().getCol(1);
341  vec _box_c = getBox().getCol(2);
342 
343  // create plane normals
344  vec _norm_a = _box_b ^ _box_c;
345  vec _norm_b = _box_c ^ _box_a;
346  vec _norm_c = _box_a ^ _box_b;
347 
348  _norm_a.normalize();
349  _norm_b.normalize();
350  _norm_c.normalize();
351 
352  double la = _box_a * _norm_a;
353  double lb = _box_b * _norm_b;
354  double lc = _box_c * _norm_c;
355 
356  return min(la, min(lb, lc));
357 }
