00029 void linalg_invert( ub::matrix<double> &A, ub::matrix<double> &V){
00030     // matrix inversion using MKL
00031     // input matrix is destroyed, make local copy
00032     ub::matrix<double> work = A;
00033     
00034     // define LAPACK variables
00035     MKL_INT n = A.size1();
00036     MKL_INT info;
00037     MKL_INT ipiv[n];
00038     
00039     // initialize V
00040     V = ub::identity_matrix<double>(n,n);
00041 
00042     // pointers for LAPACK
00043     double * pV = const_cast<double*>(&V.data().begin()[0]);
00044     double * pwork = const_cast<double*>(&work.data().begin()[0]);
00045     
00046     // solve
00047     info = LAPACKE_dgesv( LAPACK_ROW_MAJOR, n, n, pwork , n, ipiv, pV, n );
00048 }
