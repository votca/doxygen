247 bool KMCParallel::EvaluateFrame() {
248 
249  cout << "Evaluate Frame " << endl;
250  cout << "... KMCParallel" << endl;
251 
252  vector<KMCSingleOp*> kmcOps;
253 
254  for (int id = 0; id < _nThreads; ++id) {
255  KMCSingleOp *newOp = new KMCSingleOp(id, this);
256  kmcOps.push_back(newOp);
257  }
258 
259  for (int id = 0; id < _nThreads; ++id) {
260  kmcOps[id]->InitSlotData();
261  }
262 
263  for (int id = 0; id < _nThreads; ++id) {
264  kmcOps[id]->Start();
265  }
266 
267  for (int id = 0; id < _nThreads; ++id) {
268  kmcOps[id]->WaitDone();
269  }
270 
271  for (int id = 0; id < _nThreads; ++id) {
272  delete kmcOps[id];
273  }
274 
275  kmcOps.clear();
276 
277 
278  FILE *out;
279  out = fopen(_outFile.c_str(),"w");
280 
281  map< int, vector<vec> > ::iterator mit;
282  vector<vec> ::iterator vit;
283  vec AVGVEL = vec(0,0,0);
284  int AVG_N = 0;
285 
286  for (mit = _log_inj_vel.begin();
287  mit != _log_inj_vel.end();
288  ++mit) {
289  for (vit = (*mit).second.begin();
290  vit < (*mit).second.end();
291  ++vit) {
292  fprintf(out, "%5d %4.7e %4.7e %4.7e \n",
293  (*mit).first, (*vit).getX(), (*vit).getY(), (*vit).getZ());
294  AVGVEL += (*vit);
295  AVG_N += 1;
296  }
297  }
298 
299  AVGVEL /= AVG_N;
300  fprintf(out, "AVG VELOCITY %4.7e %4.7e %4.7e \n",
301  AVGVEL.getX(), AVGVEL.getY(), AVGVEL.getZ());
302  fclose(out);
303 
304  cout << "... ... AVG VELOCITY = " << AVGVEL << " m/s." << endl;
305  return true;
306 }

341 void KMCParallel::KMCSingleOp::InitSlotData() {
342 
343  // Initialise random-number generator
344  cout << "... ... OP " << this->_id << ": " << flush;
345  this->_random.init(rand(), rand(), rand(), rand());
346 
347  Database db;
348  db.Open(_master->_stateFile);
349 
350  if (_master->_maverick) {
351  cout << flush
352  << "... ... Loading graph from " << _master->_stateFile
353  << endl;
354  }
355 
356  // Load segments <=> nodes
357  Statement *stmt = db.Prepare("SELECT id, name FROM segments;");
358 
359  while (stmt->Step() != SQLITE_DONE) {
360 
361  int id = stmt->Column<int>(0);
362  string name = stmt->Column<string>(1);
363 
364  NodeBoxed *newnode = new NodeBoxed(id);
365  newnode->SetRNG(&this->_random);
366 
367  _nodes.push_back(newnode);
368  _nodes_lookup[id] = _nodes.back();
369 
370  if (wildcmp(_master->_injName.c_str(), name.c_str())) {
371  _injection.push_back(newnode);
372  _injection_lookup[id] = _injection.back();
373  }
374  }
375 
376  delete stmt;
377 
378  // Load pairs <=> links
379  int linkCount = 0;
380  if (_master->_channel == "hole") {
381  stmt = db.Prepare("SELECT seg1, seg2, "
382  "rate12h, rate21h, "
383  "drX, drY, drZ "
384  "FROM pairs;");
385  }
386  else if (_master->_channel == "electron") {
387  stmt = db.Prepare("SELECT seg1, seg2, "
388  "rate12e, rate21e, "
389  "drX, drY, drZ "
390  "FROM pairs;");
391  }
392  else {
393  throw std::runtime_error(" Invalid channel option '" +
394  _master->_channel + "'. ");
395  }
396 
397  while (stmt->Step() != SQLITE_DONE) {
398 
399  NodeBoxed *node1 = _nodes_lookup[stmt->Column<int>(0)];
400  NodeBoxed *node2 = _nodes_lookup[stmt->Column<int>(1)];
401 
402  double rate12 = stmt->Column<double>(2);
403  double rate21 = stmt->Column<double>(3);
404 
405  vec dr = vec(stmt->Column<double>(4),
406  stmt->Column<double>(5),
407  stmt->Column<double>(6));
408 
409  node1->AddEvent(new LinkBoxed(node2, rate12, dr, this));
410  node2->AddEvent(new LinkBoxed(node1, rate21, -dr, this));
411  ++(++linkCount);
412  }
413 
414  delete stmt;
415  stmt = NULL;
416 
417  if (_master->_maverick) {
418  cout << flush
419  << "... ... Created graph with "
420  << _nodes.size() << " nodes, "
421  << linkCount << " links. "
422  << endl;
423  }
424 }

444 void KMCParallel::KMCSingleOp::EvalKMC() {
445 
446  // Pick injection site
447  int inj = _random.rand_uniform_int(_injection.size());
448  _current = _injection[inj];
449  _current->UpdateWaitingTime();
450  _pos = vec(0,0,0);
451 
452  if (_master->_maverick) {
453  cout << "... ... ... Starting run at node "
454  << _current->_id << " (" << inj << "). "
455  << endl;
456  }
457 
458  // Init KMC
459  int steps = 0;
460  double t_run = 0.0;
461  double t_out = _master->_outtime;
462  double t_max = _master->_runtime;
463 
464  // Prepare trajectory log file
465  FILE *out;
466  string logTrajFile = "log_traj_"
467  + boost::lexical_cast<string>(_id)
468  + ".dat";
469  out = fopen(logTrajFile.c_str(), "a");
470  fprintf(out, "Injection at node %5d \n", inj);
471  fprintf(out, "%5d %4.7e %4.7f %4.7f %4.7f \n",
472  _current->_id, t_run, _pos.getX(), _pos.getY(), _pos.getZ());
473 
474  // Run KMC
475  while (t_run < t_max) {
476 
477  t_run += _current->WaitingTime();
478  _current->OnExecute();
479 
480  if (t_run > t_out) {
481  t_out = t_run + _master->_outtime;
482  fprintf(out, "%5d %4.7e %4.7f %4.7f %4.7f \n",
483  _current->_id, t_run, _pos.getX(), _pos.getY(), _pos.getZ());
484  }
485 
486  ++steps;
487  }
488  fprintf(out, "%5d %4.7e %4.7f %4.7f %4.7f \n",
489  _current->_id, t_run, _pos.getX(), _pos.getY(), _pos.getZ());
490  fprintf(out, "Finished KMC in %9d steps. \n", steps);
491  fprintf(out, "AVG velocity %4.7e %4.7e %4.7e \n\n",
492  (_pos/t_run*1e-9).getX(),
493  (_pos/t_run*1e-9).getY(),
494  (_pos/t_run*1e-9).getZ());
495  fclose(out);
496 
497  _master->_logMutex.Lock();
498  _master->_log_inj_vel[inj].push_back(_pos/t_run*1e-9);
499  _master->_logMutex.Unlock();
500 
501  // Post-process
502  this->WriteOcc();
503  this->Reset();
504 }

553 template<typename event_t>
554 event_t *KMCParallel::VSSMGroupBoxed<event_t>::SelectEvent_BinarySearch() {
555 
556  //double max = Rate();
557  double u = 1.-_random->rand_uniform();
558  u=u*Rate();
559 
560  int imin = 0;
561  int imax = _acc_rate.size();
562 
563  while(imax - imin > 1) {
564  int imid = (int) ((imin+imax)*0.5);
565  if(u <= _acc_rate[imid]) imax=imid;
566  else imin=imid;
567  }
568 
569  return _events[imin];
570 }
