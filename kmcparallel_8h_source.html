428

456 
457  // Run KMC

459 
460  t_run += _current->WaitingTime();
461  _current->OnExecute();
462 
463  if (t_run > t_out) {
464  t_out = t_run + _master->_outtime;

467  }

472  _current->_id, t_run, _pos.getX(), _pos.getY(), _pos.getZ());
473  fprintf(out, "Finished KMC in %9d steps. \n", steps);
474  fprintf(out, "AVG velocity %4.7e %4.7e %4.7e \n\n",
