287  //cout << "\nLoaded 1\n" << flush;
288  if ( _has_basis_set_size ) { ar & _basis_set_size; }
289  if ( _has_occupied_levels ) { ar & _occupied_levels; }
290  if ( _has_unoccupied_levels ) { ar & _unoccupied_levels; }
291  if ( _has_number_of_electrons ) { ar & _number_of_electrons; }
292  if ( _has_level_degeneracy ) { ar & _level_degeneracy; }
293  if ( _has_mo_energies ) { ar & _mo_energies; }
294  if ( _has_mo_coefficients ) { ar & _mo_coefficients; }
295  if ( _has_overlap ) {
296  // symmetric matrix does not serialize by default
297  if (Archive::is_saving::value) {
298  unsigned int size = _overlap.size1();
299  ar & size;
300  }
301 
302  // copy the values back if loading
303  if (Archive::is_loading::value) {
304  unsigned int size;
305  ar & size;
306  _overlap.resize(size);
307  }
308 
309  for (unsigned int i = 0; i < _overlap.size1(); ++i)
310  for (unsigned int j = 0; j <= i; ++j)
311  ar & _overlap(i, j);
312  }
313 
314  if ( _has_atoms ) { ar & _atoms; }
315  if ( _has_qm_energy ) { ar & _qm_energy; }
316  if ( _has_qm_package ) { ar & _qm_package; }
317  if ( _has_self_energy ) { ar & _self_energy; }
318  if ( _has_integrals ) { ar & _integrals; }
319 
320  //cout << "\nLoaded 2\n" << flush;
321  // GW-BSE storage
322  if(version > 0) {
323  ar & _has_vxc;
324  ar & _has_QPpert;
325  ar & _has_QPdiag;
326  ar & _has_BSE_singlets;
327  ar & _has_BSE_triplets;
328  if ( _has_QPpert ) { ar & _QP_levels_index; ar & _QPpert_energies; }
329  if ( _has_QPdiag ) { ar & _QPdiag_energies; ar & _QPdiag_coefficients; }
330  if ( _has_BSE_singlets || _has_BSE_triplets ) { ar & _BSE_levels_indices; }
331  if ( _has_BSE_singlets ) { ar & _BSE_singlet_energies; ar & _BSE_singlet_coefficients; }
332  if ( _has_BSE_triplets ) { ar & _BSE_triplet_energies; ar & _BSE_triplet_coefficients; }
333 
334  //cout << "\nLoaded 3\n" << flush;
335  if ( _has_vxc ) {
336  // symmetric matrix does not serialize by default
337  if (Archive::is_saving::value) {
338  unsigned int size = _vxc.size1();
