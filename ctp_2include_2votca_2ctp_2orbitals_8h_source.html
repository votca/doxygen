244  // serialization itself (template implementation stays in the header)
245  template<typename Archive>
246  void serialize(Archive& ar, const unsigned int version) {
247 
248  ar & _basis_set_size;
249  ar & _occupied_levels;
250  ar & _unoccupied_levels;
251  ar & _number_of_electrons;
252  ar & _level_degeneracy;

254  ar & _mo_coefficients;
255 
256 
257  // symmetric matrix does not serialize by default
258  if (Archive::is_saving::value) {
259  unsigned int size = _overlap.size1();
260  ar & size;
261  }
262 
263  // copy the values back if loading
264  if (Archive::is_loading::value) {
265  unsigned int size;
266  ar & size;
267  _overlap.resize(size);
268  }
269 
270  for (unsigned int i = 0; i < _overlap.size1(); ++i)
271  for (unsigned int j = 0; j <= i; ++j)
272  ar & _overlap(i, j);
273 
274  ar & _atoms;
275  ar & _qm_energy;
276  ar & _qm_package;
277  ar & _self_energy;
278  ar & _mo_couplings;
279 
280 
281  }// end of serialization
