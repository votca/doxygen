216  void GWBSE::BSE_FragmentPopulations(AOBasis& dftbasis, ub::matrix<double>& _dft_orbitals, ub::matrix<real_gwbse>& _bse_coefficients, std::vector<double>& popHA, std::vector<double>& popEA, std::vector<double>& popHB, std::vector<double>& popEB, std::vector<double> &_CrgsA, std::vector<double> &_CrgsB) {
217  double &_popA = _orbitals->FragmentAChargesGS();
218  double &_popB = _orbitals->FragmentBChargesGS();
219 
220 
221 
222  // Mulliken fragment population analysis
223  if (_fragA > 0) {
224 
225  // get overlap matrix for DFT basisset
226  AOOverlap _dftoverlap;
227  // initialize overlap matrix
228  _dftoverlap.Initialize(dftbasis._AOBasisSize);
229  // Fill overlap
230  _dftoverlap.Fill(&dftbasis);
231  LOG(CTP::logDEBUG, *_pLog) << CTP::TimeStamp() << " Filled DFT Overlap matrix of dimension: " << _dftoverlap._aomatrix.size1() << flush;
232  LOG(CTP::logDEBUG, *_pLog) << CTP::TimeStamp() << " Running Excitation fragment population analysis " << flush;
233  // ground state populations
234  ub::matrix<double> DMAT = _orbitals->DensityMatrixGroundState(_dft_orbitals);
235  double nucA;
236  double nucB;
237  _orbitals->FragmentNuclearCharges(_fragA, nucA, nucB);
238  _orbitals->MullikenPopulation(DMAT, _dftoverlap._aomatrix, dftbasis._AOBasisFragA, _popA, _popB);
239  // population to electron charges and add nuclear charges
240  _popA = nucA - _popA;
241  _popB = nucB - _popB;
242 
243  for (int _i_state = 0; _i_state < _bse_nprint; _i_state++) {
244 
245  // checking Density Matrices
246  std::vector<ub::matrix<double> > DMAT = _orbitals->DensityMatrixExcitedState(_dft_orbitals, _bse_coefficients, _i_state);
247 
248  double _totalA;
249  double _totalB;
250 
251  // hole part
252  _orbitals->MullikenPopulation(DMAT[0], _dftoverlap._aomatrix, dftbasis._AOBasisFragA, _totalA, _totalB);
253 
254  popHA.push_back(std::abs(_totalA));
255  popHB.push_back(std::abs(_totalB));
256 
257  // electron part
258  _orbitals->MullikenPopulation(DMAT[1], _dftoverlap._aomatrix, dftbasis._AOBasisFragA, _totalA, _totalB);
259  popEA.push_back(_totalA);
260  popEB.push_back(_totalB);
261 
262  // update effective charges
263  _CrgsA.push_back(popHA[_i_state] - popEA[_i_state]);
264  _CrgsB.push_back(popHB[_i_state] - popEB[_i_state]);
265 
266  }

301  void GWBSE::BSE_CoupledTransition_Dipoles_BTDA(ub::vector<real_gwbse>& _bse_energies,
302  ub::matrix<real_gwbse>& _bse_coefficients,
303  ub::matrix<real_gwbse>& _bse_coefficients_AR,
304  std::vector<double>& _oscillator_strength,
305  std::vector<double>& _transition_dipole_strength) {
306 
307  double sqrt2 = sqrt(2.0);
308  std::vector<ub::vector<double> >& _transition_dipoles = _orbitals->TransitionDipoles();
309  for (int _i_exc = 0; _i_exc < _bse_nprint; _i_exc++) {
310  ub::vector<double> _tdipole = ub::zero_vector<double>(3);
311 
312  for (unsigned _v = 0; _v < _bse_vtotal; _v++) {
313  for (unsigned _c = 0; _c < _bse_ctotal; _c++) {
314 
315  int index_vc = _bse_ctotal * _v + _c;
316 
317  for (int _i_comp = 0; _i_comp < 3; _i_comp++) {
318  // The Transition dipole is sqrt2 bigger because of the spin, the excited state is a linear combination of 2 slater determinants, where either alpha or beta spin electron is excited
319  _tdipole(_i_comp) += sqrt2 * _bse_singlet_coefficients(index_vc, _i_exc) * _interlevel_dipoles[_i_comp](_v, _c);
320  // anti-resonant part
321  _tdipole(_i_comp) += sqrt2 * _bse_singlet_coefficients_AR(index_vc, _i_exc) * _interlevel_dipoles[_i_comp](_v, _c);
322  }
323 
324  }
325 
326  }
327  // cout << _i_exc << " : " << _tdipole[0] << " : " << _tdipole[1] << " : " << _tdipole[2] << endl;
328  _transition_dipoles.push_back(_tdipole);
329  _transition_dipole_strength.push_back((ub::inner_prod(_tdipole, _tdipole)));
330  //cout<< ub::inner_prod(_tdipole,_tdipole)<< " "<< _tdipole<< endl;
