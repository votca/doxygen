38BOOST_AUTO_TEST_CASE(jacobisweeps_test) {
39
40 libint2::initialize();
41 Orbitals orbitals;
42 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
43 "/pmlocalization/ch3oh.xyz");
44 orbitals.setNumberOfOccupiedLevels(9);
45 orbitals.setNumberOfAlphaElectrons(9);
46
47 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) +
48 "/pmlocalization/def2-tzvp.xml");
49
50 orbitals.MOs().eigenvectors() =
51 votca::tools::EigenIO_MatrixMarket::ReadMatrix(
52 std::string(XTP_TEST_DATA_FOLDER) +
53 "/pmlocalization/orbitalsMOs_ref.mm");
54 orbitals.MOs().eigenvalues() = votca::tools::EigenIO_MatrixMarket::ReadVector(
55 std::string(XTP_TEST_DATA_FOLDER) +
56 "/pmlocalization/ch3oh_energies_ref.mm");
57
58 Logger log;
59 tools::Property options;
60 options.add("max_iterations", "1000");
61 options.add("convergence_limit", "1e-12");
62 options.add("method", "jacobi-sweeps");
63
64 PMLocalization pml(log, options);
65 pml.computePML(orbitals);
66
67 Eigen::MatrixXd LMOs = orbitals.getLMOs();
68 Eigen::VectorXd LMOs_energies = orbitals.getLMOs_energies();
69 Eigen::VectorXd test_LMOs_energies =
70 votca::tools::EigenIO_MatrixMarket::ReadVector(
71 std::string(XTP_TEST_DATA_FOLDER) +
72 "/pmlocalization/ch3oh_energies.mm");
73
74 Eigen::MatrixXd test_LMOs = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
75 std::string(XTP_TEST_DATA_FOLDER) + "/pmlocalization/ch3oh.mm");
76
77 bool checkLMOs = LMOs.isApprox(test_LMOs, 2e-6);
78 bool checkLMOs_energies = LMOs_energies.isApprox(test_LMOs_energies, 2e-6);
79
80 BOOST_CHECK_EQUAL(checkLMOs, 1);
81 BOOST_CHECK_EQUAL(checkLMOs_energies, 1);
82
83 libint2::finalize();
84}
