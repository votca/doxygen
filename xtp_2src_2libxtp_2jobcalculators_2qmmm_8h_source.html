259 ctp::XJob QMMM::ProcessInputString(ctp::Job *job, ctp::Topology *top, ctp::QMThread *thread) {
260 
261  string input = job->getInput().as<string>();
262  vector<ctp::Segment*> qmSegs;
263  vector<string> qmSegMps;
264  vector<string> split;
265  Tokenizer toker(input, " \t\n");
266  toker.ToVector(split);
267 
268  for (unsigned int i = 0; i < split.size(); ++i) {
269 
270  string id_seg_mps = split[i];
271  vector<string> split_id_seg_mps;
272  Tokenizer toker(id_seg_mps, ":");
273  toker.ToVector(split_id_seg_mps);
274 
275  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
276  string segName = split_id_seg_mps[1];
277  string mpsFile = split_id_seg_mps[2];
278 
279  ctp::Segment *seg = top->getSegment(segId);
280  if (seg->getName() != segName) {
281  CTP_LOG(ctp::logERROR,*(thread->getLogger()))
282  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
283  << " maltagged as " << segName << ". Skip job ..." << flush;
284  throw std::runtime_error("Input does not match topology.");
285  }
286 
287  qmSegs.push_back(seg);
288  qmSegMps.push_back(mpsFile);
289  }
290 
291  return ctp::XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
292 }

295 ctp::Job::JobResult QMMM::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *thread) {
296 
297  // SILENT LOGGER FOR QMPACKAGE
298  ctp::Logger* log = thread->getLogger();
299  ctp::Logger* qlog = new ctp::Logger();
300  qlog->setReportLevel(ctp::logDEBUG);
301  qlog->setMultithreading(_maverick);
302  qlog->setPreface(ctp::logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
303  qlog->setPreface(ctp::logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
304  qlog->setPreface(ctp::logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
305  qlog->setPreface(ctp::logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
306 
307  // CREATE XJOB FROM JOB INPUT STRING
308  CTP_LOG(ctp::logINFO,*log)
309  << "Job input = " << job->getInput().as<string>() << flush;
310  ctp::XJob xjob = this->ProcessInputString(job, top, thread);
311 
312  // GENERATE POLAR TOPOLOGY FOR JOB
313  double co1 = _cutoff1;
314  double co2 = _cutoff2;
315  _mps_mapper.Gen_QM_MM1_MM2(top, &xjob, co1, co2, thread);
316 
317 
318 
319 
320  const matrix box=xjob.getTop()->getBox();
321  //check if box is non orthogonal
322 
323  double min=box.get(0,0);
324  if(min>box.get(1,1)){min=box.get(1,1);}
325  if(min>box.get(2,2)){min=box.get(2,2);}
326 
327  if(_cutoff2>0.5*min){
328  throw runtime_error((format("Cutoff is larger than half the box size. Maximum allowed cutoff is %1$1.1f - molecule extension.") % (0.5*min)).str());
329  }
330 
331 
332  CTP_LOG(ctp::logINFO,*log)
333  << xjob.getPolarTop()->ShellInfoStr() << flush;
334 
335  if (tools::globals::verbose){
336  xjob.getPolarTop()->PrintPDB(xjob.getTag()+"_QM0_MM1_MM2.pdb");
337  }
338  // INDUCTOR, QM RUNNER, QM-MM MACHINE
339  ctp::XInductor xind = ctp::XInductor(top, &_options, "options."+Identify(),
340  _subthreads, _maverick);
341  xind.setLog(thread->getLogger());
342 
343  //Gaussian qmpack = Gaussian(&_qmpack_opt);
344 
345  // get the corresponding object from the QMPackageFactory
346  QMPackage *qmpack = QMPackages().Create( _package );
347 
348  qmpack->Initialize( &_qmpack_opt );
349 
350  qmpack->setLog(qlog);
351 
352 
353  QMMachine<QMPackage> machine = QMMachine<QMPackage>(&xjob, &xind, qmpack,
354  &_options, "options."+Identify(), _subthreads, _maverick);
355  machine.setLog(thread->getLogger());
356 
357  // EVALUATE: ITERATE UNTIL CONVERGED
358  int error=machine.Evaluate(&xjob);
359 
360  // DESTROY QMPackage
361  delete qmpack;
362 
363  // DELIVER OUTPUT & CLEAN
364  this->LockCout();
365  cout << *thread->getLogger();
366  this->UnlockCout();
367 
368  // JOT INFO STRING & CLEAN POLAR TOPOLOGY
369  xjob.setInfoLine(true,true);
370 
371  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
372  ctp::Job::JobResult jres = ctp::Job::JobResult();
373  jres.setOutput(xjob.getInfoLine());
374  jres.setStatus(ctp::Job::COMPLETE);
375 
376  if (!xind.hasConverged()) {
377  jres.setStatus(ctp::Job::FAILED);
378  jres.setError(xind.getError());
379  CTP_LOG(ctp::logERROR,*log) << xind.getError() << flush;
380  }
381  if(error!=0){
382  jres.setStatus(ctp::Job::FAILED);
383  }
384 
385  return jres;
386 }
