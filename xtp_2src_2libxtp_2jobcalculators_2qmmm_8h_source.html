90 void QMMM::Initialize(Property *options) {
91 
92  // update options with the VOTCASHARE defaults
93  UpdateWithDefaults( options, "xtp" );
94  _options = *options;
95 
96  cout << endl
97  << "... ... Initialized with " << _nThreads << " threads. "
98  << flush;
99 
100  _maverick = (_nThreads == 1) ? true : false;
101 
102 
103  string key = "options."+Identify();
104  _xml_file= options->ifExistsReturnElseThrowRuntimeError<string>(key+".mapping");
105  _jobfile = options->ifExistsReturnElseThrowRuntimeError<string>(key+".job_file");
106  _emp_file = options->ifExistsReturnElseThrowRuntimeError<string>(key+".emp_file");
107 
108  key = "options."+Identify()+".coulombmethod";
109 
110  std::vector<string> choices={"cutoff","cut-off"};
111  _method = options->ifExistsAndinListReturnElseThrowRuntimeError<string>(key+".method",choices);
112  _cutoff1 = options->ifExistsReturnElseThrowRuntimeError<double>(key+".cutoff1");
113  _cutoff2 = options->ifExistsReturnElseReturnDefault<double>(key+".cutoff2",_cutoff1);
114 
115  if(_cutoff1>_cutoff2){
116  throw runtime_error("Cutoff1 must be smaller or equal Cutoff2");
117  }
118 
119  _subthreads = options->ifExistsReturnElseReturnDefault<int>(key+".subthreads",1);
120 
121 
122  key = "options."+Identify();
123 
124  string package_xml = options->ifExistsReturnElseThrowRuntimeError<string>(key+".dftpackage");
125  load_property_from_xml(_qmpack_opt, package_xml.c_str());
126  _package = _qmpack_opt.get("package.name").as< string >();
127 
128 
129 
130  // GWBSE options, depending on whether it is there, decide for ground
131  // or excited state QM/MM
132  key = "options."+Identify()+".gwbse";
133 
134  if ( options->exists(key)) {
135  string gwbse_xml = options->ifExistsReturnElseThrowRuntimeError<string>(key+".gwbse_options");
136  load_property_from_xml(_gwbse_opt, gwbse_xml.c_str());
137  }
138 
139  // register all QM packages (Gaussian, turbomole, etc))
140  QMPackageFactory::RegisterAll();
141 
142 }

172 ctp::XJob QMMM::ProcessInputString(ctp::Job *job, ctp::Topology *top, ctp::QMThread *thread) {
173 
174  string input = job->getInput().as<string>();
175  vector<ctp::Segment*> qmSegs;
176  vector<string> qmSegMps;
177  vector<string> split;
178  Tokenizer toker(input, " \t\n");
179  toker.ToVector(split);
180 
181  for (unsigned int i = 0; i < split.size(); ++i) {
182 
183  string id_seg_mps = split[i];
184  vector<string> split_id_seg_mps;
185  Tokenizer toker(id_seg_mps, ":");
186  toker.ToVector(split_id_seg_mps);
187 
188  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
189  string segName = split_id_seg_mps[1];
190  string mpsFile = split_id_seg_mps[2];
191 
192  ctp::Segment *seg = top->getSegment(segId);
193  if (seg->getName() != segName) {
194  CTP_LOG(ctp::logERROR,*(thread->getLogger()))
195  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
196  << " maltagged as " << segName << ". Skip job ..." << flush;
197  throw std::runtime_error("Input does not match topology.");
198  }
199 
200  qmSegs.push_back(seg);
201  qmSegMps.push_back(mpsFile);
202  }
203 
204  return ctp::XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
205 }

208 ctp::Job::JobResult QMMM::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *thread) {
209 
210  // SILENT LOGGER FOR QMPACKAGE
211  ctp::Logger* log = thread->getLogger();
212  ctp::Logger qlog;
213  qlog.setReportLevel(ctp::logDEBUG);
214  qlog.setMultithreading(_maverick);
215  qlog.setPreface(ctp::logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
216  qlog.setPreface(ctp::logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
217  qlog.setPreface(ctp::logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
218  qlog.setPreface(ctp::logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
219 
220  // CREATE XJOB FROM JOB INPUT STRING
221  CTP_LOG(ctp::logINFO,*log)
222  << "Job input = " << job->getInput().as<string>() << flush;
223  ctp::XJob xjob = this->ProcessInputString(job, top, thread);
224 
225  // GENERATE POLAR TOPOLOGY FOR JOB
226  double co1 = _cutoff1;
227  double co2 = _cutoff2;
228  _mps_mapper.Gen_QM_MM1_MM2(top, &xjob, co1, co2, thread);
229 
230  const matrix box=xjob.getTop()->getBox();
231  //check if box is non orthogonal
232 
233  double min=box.get(0,0);
234  if(min>box.get(1,1)){min=box.get(1,1);}
235  if(min>box.get(2,2)){min=box.get(2,2);}
236 
237  if(_cutoff2>0.5*min){
238  throw runtime_error((format("Cutoff is larger than half the box size. Maximum allowed cutoff is %1$1.1f - molecule extension.") % (0.5*min)).str());
239  }
240 
241 
242  CTP_LOG(ctp::logINFO,*log)
243  << xjob.getPolarTop()->ShellInfoStr() << flush;
244 
245  if (tools::globals::verbose){
246  xjob.getPolarTop()->PrintPDB(xjob.getTag()+"_QM0_MM1_MM2.pdb");
247  }
248  // INDUCTOR, QM RUNNER, QM-MM MACHINE
249  ctp::XInductor xind = ctp::XInductor(top, &_options, "options."+Identify(),
250  _subthreads, _maverick);
251  xind.setLog(thread->getLogger());
252 
253  // get the corresponding object from the QMPackageFactory
254  QMPackage *qmpack = QMPackages().Create( _package );
255  qmpack->Initialize( _qmpack_opt );
256  qmpack->setLog(&qlog);
257 
258 
259  QMMachine machine = QMMachine(&xjob, &xind, qmpack,
260  &_options, "options."+Identify());
261  machine.setLog(thread->getLogger());
262 
263  // EVALUATE: ITERATE UNTIL CONVERGED
264  int error=machine.Evaluate(&xjob);
265 
266  // DESTROY QMPackage
267  delete qmpack;
268 
269  // DELIVER OUTPUT & CLEAN
270  this->LockCout();
271  cout << *thread->getLogger();
272  this->UnlockCout();
273 
274  // JOT INFO STRING & CLEAN POLAR TOPOLOGY
275  xjob.setInfoLine(true,true);
276 
277  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
278  ctp::Job::JobResult jres = ctp::Job::JobResult();
279  jres.setOutput(xjob.getInfoLine());
280  jres.setStatus(ctp::Job::COMPLETE);
281 
282  if (!xind.hasConverged()) {
283  jres.setStatus(ctp::Job::FAILED);
284  jres.setError(xind.getError());
285  CTP_LOG(ctp::logERROR,*log) << xind.getError() << flush;
286  }
287  if(error!=0){
288  jres.setStatus(ctp::Job::FAILED);
289  }
290 
291  return jres;
292 }
