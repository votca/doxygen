39 std::vector<double> VAverage::ReadOccfile(std::string filename) const {
40  std::vector<double> result;
41  std::ifstream intt;
42  intt.open(filename);
43  if (!intt.is_open()) {
44  throw std::runtime_error("File:" + filename + " could not be opened");
45  }
46  std::string line;
47  Index id = 0;
48  while (intt.good()) {
49 
50  std::getline(intt, line);
51  std::vector<std::string> split;
52  tools::Tokenizer toker(line, " \t");
53  toker.ToVector(split);
54  if (!split.size() || split[0] == "#" || split[0].substr(0, 1) == "#") {
55  continue;
56  }
57  if (split.size() != 2) {
58  throw std::runtime_error("Row should only contain id and occupation");
59  }
60 
61  Index id_readin = std::stoi(split[0]);
62  if (id_readin != id) {
63  throw std::runtime_error("Ids should be sorted and start at zero.");
64  }
65  id++;
66  result.push_back(std::stod(split[1]));
67  }
68  return result;
69 }

71 std::vector<Rate_Engine::PairRates> VAverage::ReadRatefile(

95  Index id_readin = std::stoi(split[0]);

102  pair.rate21 = std::stod(split[4]);
103  result.push_back(pair);
104  }
105  return result;

108 bool VAverage::EvaluateFrame(Topology& top) {
109  std::cout << std::endl
110  << "... ... Computing velocity average for all sites\n";
111  std::cout << "Reading in site occupations from " << _occfile << std::endl;
112  std::vector<double> occ = ReadOccfile(_occfile);
113  if (top.Segments().size() != occ.size()) {
114  throw std::runtime_error(
115  "Number of occupations is" + std::to_string(occ.size()) +
116  " Topology has size:" + std::to_string(top.Segments().size()));
117  }
118  std::cout << "Reading in rates from " << _ratefile << std::endl;
119  std::vector<Rate_Engine::PairRates> rates = ReadRatefile(_ratefile);
120  if (top.NBList().size() != Index(rates.size())) {
121  throw std::runtime_error(
122  "Number of pairs in file is" + std::to_string(rates.size()) +
123  " Topology has size:" + std::to_string(top.NBList().size()));
124  }
125  std::vector<Eigen::Vector3d> velocities =
126  std::vector<Eigen::Vector3d>(occ.size(), Eigen::Vector3d::Zero());
127  for (const QMPair* pair : top.NBList()) {
128  Index id1 = pair->Seg1()->getId();
129  Index id2 = pair->Seg1()->getId();
130  Index pairid = pair->getId();
131  double p1 = occ[id1];
132  double p2 = occ[id2];
133  double w12 = rates[pairid].rate12;
134  double w21 = rates[pairid].rate21;
135  const Eigen::Vector3d& dR12 =
136  pair->R(); // points from seg. (1) to seg. (2)
137  // Split current symmetrically among sites (1) and (2)
138  Eigen::Vector3d v_12_21 = 0.5 * (p1 * w12 - p2 * w21) * dR12;
139  velocities[id1] += v_12_21;
140  velocities[id2] += v_12_21;
141  }
142 
143  std::ofstream ofs;
144  ofs.open(_outputfile, std::ofstream::out);
145  if (!ofs.is_open()) {
146  throw std::runtime_error("Bad file handle: " + _outputfile);
147  }
148  ofs << "# 1 | 2 | 3 4 5 | 6 7 8 |" << std::endl;
149  ofs << "# ----|---------|--------------|-------------------|" << std::endl;
150  ofs << "# ID | NAME | X Y Z [nm] | VX VY VZ [nm/s] |" << std::endl;
151  for (const Segment& seg : top.Segments()) {
152  const Eigen::Vector3d r = seg.getPos() * tools::conv::bohr2nm;
153  const Eigen::Vector3d v = velocities[seg.getId()] * tools::conv::bohr2nm;
154  ofs << (boost::format("%1$4d %2$-10s %3$+1.7e %4$+1.7e "
155  "%5$+1.7e %6$+1.7e %7$+1.7e %8$+1.7e") %
156  seg.getId() % seg.getType() % r.x() % r.y() % r.z() % v.x() %
157  v.y() % v.z())
158  << std::endl;
159  }
160  std::cout << "Writing velocities to " << _outputfile << std::endl;
161  return true;
162 }
