89 bool ParallelSiteCalculator::EvaluateFrame(Topology *top) {
90 
91  // Rigidify if (a) not rigid yet (b) rigidification at all possible
92  if (!top->isRigid()) {
93  bool isRigid = top->Rigidify();
94  if (!isRigid) { return 0; }
95  }
96  else { cout << endl << "... ... System is already rigidified."; }
97  cout << endl;
98 
99  vector<SiteOperator*> siteOps;
100  this->InitSlotData(top);
101 
102  _nextSite = top->Segments().begin();
103 
104  for (int id = 0; id < _nThreads; id++) {
105  SiteOperator *newOp = new SiteOperator(id, top, this);
106  siteOps.push_back(newOp);
107  }
108 
109  for (int id = 0; id < _nThreads; id++) {
110  siteOps[id]->Start();
111  }
112 
113  for (int id = 0; id < _nThreads; id++) {
114  siteOps[id]->WaitDone();
115  }
116 
117  for (int id = 0; id < _nThreads; id++) {
118  delete siteOps[id];
119  }
120 
121  siteOps.clear();
122 
123  this->PostProcess(top);
124  return 1;
125 }

132 Segment *ParallelSiteCalculator::RequestNextSite(int opId, Topology *top) {

134  _nextSiteMutex.Lock();
