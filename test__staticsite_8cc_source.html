72BOOST_AUTO_TEST_CASE(rotation_test) {
73 StaticSite st(1, "st2", Eigen::Vector3d::UnitY());
74
75 Eigen::Matrix3d R = Eigen::Matrix3d::Zero(); // Rotation around z axes
76 R << 0, -1, 0, 1, 0, 0, 0, 0, 1;
77
78 Eigen::VectorXd multipoles = Eigen::VectorXd::Zero(9);
79 multipoles << 1, 1, 0, 0, 0, 1, 0, 0,
80 0; // q=1, mu_x=1 and Q_21c=1 the rest is 0
81 st.setMultipole(multipoles, 2);
82 st.Rotate(R, Eigen::Vector3d::Zero());
83 bool equalpos = st.getPos().isApprox(Eigen::Vector3d(-1, 0, 0), 1e-5);
84 if (!equalpos) {
85 std::cout << "Result " << std::endl;
86 std::cout << st.getPos() << std::endl;
87 std::cout << "Reference" << std::endl;
88 std::cout << Eigen::Vector3d(-1, 0, 0) << std::endl;
89 }
90 BOOST_CHECK_EQUAL(equalpos, true);
91
92 Eigen::VectorXd rotmultipoles = Eigen::VectorXd::Zero(9);
93 rotmultipoles << 1, 0, 1, 0, 0, 0, 1, 0, 0; // q=1, mu_y=1 and Q_21s=1 is 0
94 bool equalmultipoles = rotmultipoles.isApprox(st.Q(), 1e-5);
95 if (!equalmultipoles) {
96 std::cout << "Result " << std::endl;
97 std::cout << st.Q() << std::endl;
98 std::cout << "Reference" << std::endl;
99 std::cout << rotmultipoles << std::endl;
100 }
101 BOOST_CHECK_EQUAL(equalmultipoles, true);
102}
