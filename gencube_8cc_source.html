80 void GenCube::subtractCubes() {
81 
82  // open infiles for reading
83  ifstream in1;
84  XTP_LOG(Log::error, _log) << " Reading first cube from " << _infile1 << flush;
85  in1.open(_infile1, ios::in);
86  ifstream in2;
87  XTP_LOG(Log::error, _log)
88  << " Reading second cube from " << _infile2 << flush;
89  in2.open(_infile2, ios::in);
90  string s;
91 
92  std::ofstream out(_output_file);
93 
94  // first two lines of header are garbage
95  getline(in1, s);
96  out << s << "\n";
97  getline(in1, s);
98  out << s << " substraction\n";
99  getline(in2, s);
100  getline(in2, s);
101 
102  // read rest from header
103  Index natoms;
104  double xstart;
105  double ystart;
106  double zstart;
107  // first line
108  in1 >> natoms;
109  bool do_amplitude = false;
110  if (natoms < 0) {
111  do_amplitude = true;
112  }
113  in1 >> xstart;
114  in1 >> ystart;
115  in1 >> zstart;
116  // check from second file
117  Index tempint;
118  double tempdouble;
119  in2 >> tempint;
120  if (tempint != natoms) {
121  throw std::runtime_error("Atom numbers do not match");
122  }
123  in2 >> tempdouble;
124  if (tempdouble != xstart) {
125  throw std::runtime_error("Xstart does not match");
126  }
127  in2 >> tempdouble;
128  if (tempdouble != ystart) {
129  throw std::runtime_error("Ystart does not match");
130  }
131  in2 >> tempdouble;
132  if (tempdouble != zstart) {
133  throw std::runtime_error("Zstart does not match");
134  }
135 
136  out << boost::format("%1$lu %2$f %3$f %4$f \n") % natoms % xstart % ystart %
137  zstart;
138 
139  // grid information from first cube
140  double xincr;
141  double yincr;
142  double zincr;
143  Index xsteps;
144  Index ysteps;
145  Index zsteps;
146  in1 >> xsteps;
147  in1 >> xincr;
148  in1 >> tempdouble;
149  in1 >> tempdouble;
150  in1 >> ysteps;
151  in1 >> tempdouble;
152  in1 >> yincr;
153  in1 >> tempdouble;
154  in1 >> zsteps;
155  in1 >> tempdouble;
156  in1 >> tempdouble;
157  in1 >> zincr;
158 
159  // check second cube
160  in2 >> tempint;
161  if (tempint != xsteps) {
162  throw std::runtime_error("xsteps does not match");
163  }
164  in2 >> tempdouble;
165  if (tempdouble != xincr) {
166  throw std::runtime_error("xincr does not match");
167  }
168  in2 >> tempdouble;
169  in2 >> tempdouble;
170  in2 >> tempint;
171  if (tempint != ysteps) {
172  throw std::runtime_error("ysteps does not match");
173  }
174  in2 >> tempdouble;
175  in2 >> tempdouble;
176  if (tempdouble != yincr) {
177  throw std::runtime_error("yincr does not match");
178  }
179  in2 >> tempdouble;
180  in2 >> tempint;
181  if (tempint != zsteps) {
182  throw std::runtime_error("zsteps does not match");
183  }
184  in2 >> tempdouble;
185  in2 >> tempdouble;
186  in2 >> tempdouble;
187  if (tempdouble != zincr) {
188  throw std::runtime_error("zincr does not match");
189  }
190 
191  out << boost::format("%1$d %2$f 0.0 0.0 \n") % xsteps % xincr;
192  out << boost::format("%1$d 0.0 %2$f 0.0 \n") % ysteps % yincr;
193  out << boost::format("%1$d 0.0 0.0 %2$f \n") % zsteps % zincr;
194 
195  // atom information
196 
197  for (Index iatom = 0; iatom < std::abs(natoms); iatom++) {
198  // get center coordinates in Bohr
199  double x;
200  double y;
201  double z;
202  Index atnum;
203  double crg;
204 
205  // get from first cube
206  in1 >> atnum;
207  in1 >> crg;
208  in1 >> x;
209  in1 >> y;
210  in1 >> z;
211 
212  // check second cube
213  in2 >> tempint;
214  if (tempint != atnum) {
215  throw std::runtime_error("atnum does not match");
216  }
217  in2 >> tempdouble;
218  if (tempdouble != crg) {
219  throw std::runtime_error("crg does not match");
220  }
221  in2 >> tempdouble;
222  if (tempdouble != x) {
223  throw std::runtime_error("x does not match");
224  }
225  in2 >> tempdouble;
226  if (tempdouble != y) {
227  throw std::runtime_error("y does not match");
228  }
229  in2 >> tempdouble;
230  if (tempdouble != z) {
231  throw std::runtime_error("z does not match");
232  }
233  out << boost::format("%1$d %2$f %3$f %4$f %5$f\n") % atnum % crg % x % y %
234  z;
235  }
236 
237  if (do_amplitude) {
238  Index ntotal;
239  Index nis;
240  in1 >> ntotal;
241  in1 >> nis;
242 
243  in2 >> tempint;
244  if (tempint != ntotal) {
245  throw std::runtime_error("ntotal does not match");
246  }
247  in2 >> tempint;
248  if (tempint != nis) {
249  throw std::runtime_error("nis does not match");
250  }
251  out << boost::format(" 1 %1$d \n") % nis;
252  }
253  // now read data
254  double val1;
255  double val2;
256  for (Index ix = 0; ix < xsteps; ix++) {
257  for (Index iy = 0; iy < ysteps; iy++) {
258  Index Nrecord = 0;
259  for (Index iz = 0; iz < zsteps; iz++) {
260  Nrecord++;
261  in1 >> val1;
262  in2 >> val2;
263  if (Nrecord == 6 || iz == zsteps - 1) {
264  out << boost::format("%1$E \n") % (val1 - val2);
265  Nrecord = 0;
266  } else {
267  out << boost::format("%1$E ") % (val1 - val2);
268  }
269  }
270  }
271  }
272 
273  out.close();
274  XTP_LOG(Log::error, _log)
275  << "Wrote subtracted cube data to " << _output_file << flush;
276 }
