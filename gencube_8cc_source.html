34 void GenCube::Initialize(tools::Property& options) {
35 
36  // update options with the VOTCASHARE defaults
37  UpdateWithDefaults(options, "xtp");

39  string key = "options." + Identify();

69 void GenCube::calculateCube() {
70 
71  XTP_LOG_SAVE(logDEBUG, _log)
72  << "Reading serialized QM data from " << _orbfile << flush;
73 
74  Orbitals orbitals;
75  XTP_LOG_SAVE(logDEBUG, _log) << " Loading QM data from " << _orbfile << flush;
76  orbitals.ReadFromCpt(_orbfile);
77 
78  const QMMolecule& atoms = orbitals.QMAtoms();
79 
80  std::pair<Eigen::Vector3d, Eigen::Vector3d> minmax =
81  atoms.CalcSpatialMinMax();
82 
83  // generate cube grid
84  double xstart = minmax.first.x() - _padding;
85  double xstop = minmax.second.x() + _padding;
86  double ystart = minmax.first.y() - _padding;
87  double ystop = minmax.second.y() + _padding;
88  double zstart = minmax.first.z() - _padding;
89  double zstop = minmax.second.z() + _padding;
90 
91  double xincr = (xstop - xstart) / double(_xsteps);
92  double yincr = (ystop - ystart) / double(_ysteps);
93  double zincr = (zstop - zstart) / double(_zsteps);
94 
95  std::ofstream out(_output_file);
96  if (!out.is_open()) {
97  throw std::runtime_error("Bad file handle: " + _output_file);
98  }
99  bool do_amplitude = (_state.Type().isSingleParticleState());
100 
101  // write cube header
102  if (_state.isTransition()) {
103  out << boost::format("Transition state: %1$s \n") % _state.ToString();
104  } else if (do_amplitude) {
105  out << boost::format("%1$s with energy %2$f eV \n") % _state.ToString() %
106  (orbitals.getExcitedStateEnergy(_state) * tools::conv::hrt2ev);
107  } else {
108  if (_dostateonly) {
109  out << boost::format(
110  "Difference electron density of excited state %1$s \n") %
111  _state.ToString();
112  } else {
113  out << boost::format("Total electron density of %1$s state\n") %
114  _state.ToString();
115  }
116  }
117 
118  out << "Created by VOTCA-XTP \n";
119  if (do_amplitude) {
120  out << boost::format("-%1$lu %2$f %3$f %4$f \n") % atoms.size() % xstart %
121  ystart % zstart;
122  } else {
123  out << boost::format("%1$lu %2$f %3$f %4$f \n") % atoms.size() % xstart %
124  ystart % zstart;
125  }
126 
127  out << boost::format("%1$d %2$f 0.0 0.0 \n") % (_xsteps + 1) % xincr;
128  out << boost::format("%1$d 0.0 %2$f 0.0 \n") % (_ysteps + 1) % yincr;
129  out << boost::format("%1$d 0.0 0.0 %2$f \n") % (_zsteps + 1) % zincr;
130  tools::Elements elements;
131  for (const QMAtom& atom : atoms) {
132  double x = atom.getPos().x();
133  double y = atom.getPos().y();
134  double z = atom.getPos().z();
135  string element = atom.getElement();
136  Index atnum = elements.getEleNum(element);
137  Index crg = atom.getNuccharge();
138  out << boost::format("%1$d %2$d %3$f %4$f %5$f\n") % atnum % crg % x % y %
139  z;
140  }
141 
142  if (do_amplitude) {
143  out << boost::format(" 1 %1$d \n") % (_state.StateIdx() + 1);
144  }
145 
146  // load DFT basis set (element-wise information) from xml file
147  BasisSet dftbs;
148  dftbs.Load(orbitals.getDFTbasisName());
149  XTP_LOG_SAVE(logDEBUG, _log)
150  << " Loaded DFT Basis Set " << orbitals.getDFTbasisName() << flush;
151 
152  // fill DFT AO basis by going through all atoms
153  AOBasis dftbasis;
154  dftbasis.Fill(dftbs, orbitals.QMAtoms());
155 
156  Eigen::MatrixXd mat =
157  Eigen::MatrixXd::Zero(dftbasis.AOBasisSize(), dftbasis.AOBasisSize());
158  if (_dostateonly) {
159  if (_state.Type().isExciton()) {
160  std::array<Eigen::MatrixXd, 2> DMAT =
161  orbitals.DensityMatrixExcitedState(_state);
162  mat = DMAT[1] - DMAT[0];
163  }
164  } else {
165  mat = orbitals.DensityMatrixFull(_state);
166  }
167  Index amplitudeindex = 0;
168  if (do_amplitude) {
169  if (_state.Type() == QMStateType::DQPstate) {
170  mat = orbitals.CalculateQParticleAORepresentation();
171  amplitudeindex = _state.StateIdx() - orbitals.getGWAmin();
172  } else {
173  mat = orbitals.MOs().eigenvectors();
174  amplitudeindex = _state.StateIdx();
175  }
176  }
177 
178  XTP_LOG_SAVE(logDEBUG, _log) << " Calculating cube data ... \n" << flush;
179  _log.setPreface(logDEBUG, "... ...");
180 
181  boost::progress_display progress(_xsteps);
182  // eval density at cube grid points
183  for (Index ix = 0; ix <= _xsteps; ix++) {
184  double x = xstart + double(ix) * xincr;
185  for (Index iy = 0; iy <= _ysteps; iy++) {
186  double y = ystart + double(iy) * yincr;
187  Index Nrecord = 0;
188  for (Index iz = 0; iz <= _zsteps; iz++) {
189  double z = zstart + double(iz) * zincr;
190  Nrecord++;
191  Eigen::Vector3d pos(x, y, z);
192  Eigen::VectorXd tmat = EvaluateBasisAtPosition(dftbasis, pos);
193  double value = 0.0;
194  if (do_amplitude) {
195  value = (mat.col(amplitudeindex).transpose() * tmat).value();
196  } else {
197  value = (tmat.transpose() * mat * tmat).value();
198  }
199  if (Nrecord == 6 || iz == _zsteps) {
200  out << boost::format("%1$E \n") % value;
201  Nrecord = 0;
202  } else {
203  out << boost::format("%1$E ") % value;
204  }
205  } // z-component
206  } // y-component
207  ++progress;
208  } // x-component
209 
210  out.close();
211  XTP_LOG_SAVE(logDEBUG, _log)
212  << "Wrote cube data to " << _output_file << flush;
213  return;
214 }

216 Eigen::VectorXd GenCube::EvaluateBasisAtPosition(const AOBasis& dftbasis,
217  const Eigen::Vector3d& pos) {
218 
219  // get value of orbitals at each gridpoint
220  Eigen::VectorXd tmat = Eigen::VectorXd::Zero(dftbasis.AOBasisSize());
221  for (const AOShell& shell : dftbasis) {
222  const double decay = shell.getMinDecay();
223  const Eigen::Vector3d& shellpos = shell.getPos();
224  Eigen::Vector3d dist = shellpos - pos;
225  double distsq = dist.squaredNorm();
226  // if contribution is smaller than -ln(1e-10), calc density
227  if ((decay * distsq) < 20.7) {
228  Eigen::VectorBlock<Eigen::VectorXd> tmat_block =
229  tmat.segment(shell.getStartIndex(), shell.getNumFunc());
230  shell.EvalAOspace(tmat_block, pos);
231  }
232  }
233  return tmat;
234 }

236 void GenCube::subtractCubes() {
237 
238  // open infiles for reading
239  ifstream in1;
240  XTP_LOG_SAVE(logDEBUG, _log)
241  << " Reading first cube from " << _infile1 << flush;
242  in1.open(_infile1, ios::in);
243  ifstream in2;
244  XTP_LOG_SAVE(logDEBUG, _log)
245  << " Reading second cube from " << _infile2 << flush;
246  in2.open(_infile2, ios::in);
247  string s;
248 
249  std::ofstream out(_output_file);
250 
251  // first two lines of header are garbage
252  getline(in1, s);
253  out << s << "\n";
254  getline(in1, s);
255  out << s << " substraction\n";
256  getline(in2, s);
257  getline(in2, s);
258 
259  // read rest from header
260  Index natoms;
261  double xstart;
262  double ystart;
263  double zstart;
264  // first line
265  in1 >> natoms;
266  bool do_amplitude = false;
267  if (natoms < 0) {
268  do_amplitude = true;
269  }
270  in1 >> xstart;
271  in1 >> ystart;
272  in1 >> zstart;
273  // check from second file
274  Index tempint;
275  double tempdouble;
276  in2 >> tempint;
277  if (tempint != natoms) {
278  throw std::runtime_error("Atom numbers do not match");
279  }
280  in2 >> tempdouble;
281  if (tempdouble != xstart) {
282  throw std::runtime_error("Xstart does not match");
283  }
284  in2 >> tempdouble;
285  if (tempdouble != ystart) {
286  throw std::runtime_error("Ystart does not match");
287  }
288  in2 >> tempdouble;
289  if (tempdouble != zstart) {
290  throw std::runtime_error("Zstart does not match");
291  }
292 
293  out << boost::format("%1$lu %2$f %3$f %4$f \n") % natoms % xstart % ystart %
294  zstart;
295 
296  // grid information from first cube
297  double xincr;
298  double yincr;
299  double zincr;
300  in1 >> _xsteps;
301  in1 >> xincr;
302  in1 >> tempdouble;
303  in1 >> tempdouble;
304  in1 >> _ysteps;
305  in1 >> tempdouble;
306  in1 >> yincr;
307  in1 >> tempdouble;
308  in1 >> _zsteps;
309  in1 >> tempdouble;
310  in1 >> tempdouble;
311  in1 >> zincr;
312 
313  // check second cube
314  in2 >> tempint;
315  if (tempint != _xsteps) {
316  throw std::runtime_error("xsteps does not match");
317  }
318  in2 >> tempdouble;
319  if (tempdouble != xincr) {
320  throw std::runtime_error("xincr does not match");
321  }
322  in2 >> tempdouble;
323  in2 >> tempdouble;
324  in2 >> tempint;
325  if (tempint != _ysteps) {
326  throw std::runtime_error("ysteps does not match");
327  }
328  in2 >> tempdouble;
329  in2 >> tempdouble;
330  if (tempdouble != yincr) {
331  throw std::runtime_error("yincr does not match");
332  }
333  in2 >> tempdouble;
334  in2 >> tempint;
335  if (tempint != _zsteps) {
336  throw std::runtime_error("zsteps does not match");
337  }
338  in2 >> tempdouble;
339  in2 >> tempdouble;
340  in2 >> tempdouble;
341  if (tempdouble != zincr) {
342  throw std::runtime_error("zincr does not match");
343  }
344 
345  out << boost::format("%1$d %2$f 0.0 0.0 \n") % _xsteps % xincr;
346  out << boost::format("%1$d 0.0 %2$f 0.0 \n") % _ysteps % yincr;
347  out << boost::format("%1$d 0.0 0.0 %2$f \n") % _zsteps % zincr;
348 
349  // atom information
350 
351  for (Index iatom = 0; iatom < std::abs(natoms); iatom++) {
352  // get center coordinates in Bohr
353  double x;
354  double y;
355  double z;
356  Index atnum;
357  double crg;
358 
359  // get from first cube
360  in1 >> atnum;
361  in1 >> crg;
362  in1 >> x;
363  in1 >> y;
364  in1 >> z;
365 
366  // check second cube
367  in2 >> tempint;
368  if (tempint != atnum) {
369  throw std::runtime_error("atnum does not match");
370  }
371  in2 >> tempdouble;
372  if (tempdouble != crg) {
373  throw std::runtime_error("crg does not match");
374  }
375  in2 >> tempdouble;
376  if (tempdouble != x) {
377  throw std::runtime_error("x does not match");
378  }
379  in2 >> tempdouble;
380  if (tempdouble != y) {
381  throw std::runtime_error("y does not match");
382  }
383  in2 >> tempdouble;
384  if (tempdouble != z) {
385  throw std::runtime_error("z does not match");
386  }
387  out << boost::format("%1$d %2$f %3$f %4$f %5$f\n") % atnum % crg % x % y %
388  z;
389  }
390 
391  if (do_amplitude) {
392  Index ntotal;
393  Index nis;
394  in1 >> ntotal;
395  in1 >> nis;
396 
397  in2 >> tempint;
398  if (tempint != ntotal) {
399  throw std::runtime_error("ntotal does not match");
400  }
401  in2 >> tempint;
402  if (tempint != nis) {
403  throw std::runtime_error("nis does not match");
404  }
405  out << boost::format(" 1 %1$d \n") % nis;
406  }
407  // now read data
408  double val1;
409  double val2;
410  for (Index ix = 0; ix < _xsteps; ix++) {
411  for (Index iy = 0; iy < _ysteps; iy++) {
412  Index Nrecord = 0;
413  for (Index iz = 0; iz < _zsteps; iz++) {
414  Nrecord++;
415  in1 >> val1;
416  in2 >> val2;
417  if (Nrecord == 6 || iz == _zsteps - 1) {
418  out << boost::format("%1$E \n") % (val1 - val2);
419  Nrecord = 0;
420  } else {
421  out << boost::format("%1$E ") % (val1 - val2);
422  }
423  }
424  }
425  }
426 
427  out.close();
428  XTP_LOG_SAVE(logDEBUG, _log)
429  << "Wrote subtracted cube data to " << _output_file << flush;
430 }
