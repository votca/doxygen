33void GeometryOptimization::Initialize(tools::Property& options) {
34
35 opt_state_ = options.get(".state").as<QMState>();
36 if (!opt_state_.Type().isExciton()) {
37 throw std::runtime_error(
38 "At the moment only excitonic states can be optimized");
39 }
40 // default convergence parameters from ORCA
41 conv_.deltaE = options.get(".convergence.energy").as<double>(); // Hartree
42 conv_.RMSForce =
43 options.get(".convergence.RMSForce").as<double>(); // Hartree/Bohr
44 conv_.MaxForce =
45 options.get(".convergence.MaxForce").as<double>(); // Hartree/Bohr
46 conv_.RMSStep = options.get(".convergence.RMSStep").as<double>(); // Bohr
47 conv_.MaxStep = options.get(".convergence.MaxStep").as<double>(); // Bohr
48 trust_radius_ = options.get("optimizer.trust").as<double>(); // Angstrom
49 trust_radius_ *= tools::conv::ang2bohr; // initial trust radius in a.u.
50
51 max_iteration_ = options.get(".maxiter").as<Index>();
52 trajfile_ = options.get(".trajectory_file").as<std::string>();
53 optimizer_ = options.get(".optimizer.method").as<std::string>();
54 force_options_ = options.get(".forces");
55 statetracker_options_ = options.get(".statetracker");
56}

58void GeometryOptimization::Evaluate() {
59 XTP_LOG(Log::error, *pLog_)
60 << "Requested geometry optimization of excited state "
61 << opt_state_.ToString() << std::flush;
62
63 StateTracker tracker;
64 tracker.Initialize(statetracker_options_);
65 tracker.setInitialState(opt_state_);
66 tracker.setLogger(pLog_);
67 tracker.PrintInfo();
68
69 // get a force object
70 Forces force_engine(gwbse_engine_, tracker);
71 force_engine.Initialize(force_options_);
72 force_engine.setLog(pLog_);
73 XTP_LOG(Log::error, *pLog_)
74 << (boost::format("Convergence of total energy: %1$8.6f Hartree ") %
75 conv_.deltaE)
76 .str()
77 << std::flush;
78 XTP_LOG(Log::error, *pLog_)
79 << (boost::format("Convergence of RMS Force: %1$8.6f Hartree/Bohr ") %
80 conv_.RMSForce)
81 .str()
82 << std::flush;
83 XTP_LOG(Log::error, *pLog_)
84 << (boost::format("Convergence of Max Force: %1$8.6f Hartree/Bohr ") %
85 conv_.MaxForce)
86 .str()
87 << std::flush;
88 XTP_LOG(Log::error, *pLog_)
89 << (boost::format("Convergence of RMS Step: %1$8.6f Bohr ") %
90 conv_.RMSStep)
91 .str()
92 << std::flush;
93 XTP_LOG(Log::error, *pLog_)
94 << (boost::format("Convergence of Max Step: %1$8.6f Bohr ") %
95 conv_.MaxStep)
96 .str()
97 << std::flush;
98 XTP_LOG(Log::error, *pLog_)
99 << (boost::format("Initial trust radius: %1$8.6f Bohr") %
100 trust_radius_)
101 .str()
102 << std::flush;
103
104 Energy_costfunction e_cost =
105 Energy_costfunction(gwbse_engine_, tracker, orbitals_, force_engine);
106 e_cost.setConvergenceParameters(conv_);
107 e_cost.setLog(pLog_);
108 // get the optimizer
109 if (optimizer_ == "BFGS-TRM") {
110 BFGSTRM bfgstrm(e_cost);
111 std::vector<std::function<void()> > callbacks;
112 std::function<void()> reporting = std::bind(
113 Report, std::cref(bfgstrm), std::cref(force_engine), std::ref(*pLog_));
114 callbacks.push_back(reporting);
115 std::function<void()> filewrite = std::bind(
116 WriteTrajectory, trajfile_, orbitals_.QMAtoms(), std::cref(bfgstrm));
117 callbacks.push_back(filewrite);
118 bfgstrm.setCallbacks(callbacks);
119 bfgstrm.setNumofIterations(max_iteration_);
120 bfgstrm.setTrustRadius(trust_radius_);
121 bfgstrm.setLog(pLog_);
122 bfgstrm.Optimize(Energy_costfunction::QMAtoms2Vector(orbitals_.QMAtoms()));
123 }
124 return;
125}

127void GeometryOptimization::Report(const BFGSTRM& bfgstrm, const Forces& forces,
128 Logger& pLog) {
129
130 XTP_LOG(Log::error, pLog) << std::flush;
131 XTP_LOG(Log::error, pLog)
132 << (boost::format("=========== OPTIMIZATION SUMMARY "
133 "================================= "))
134 .str()
135 << std::flush;
136 XTP_LOG(Log::error, pLog)
137 << "At iteration " << bfgstrm.getIteration() << std::flush;
138 XTP_LOG(Log::error, pLog)
139 << (boost::format(" ---- POSITIONS (Angstrom) ")).str() << std::flush;
140 XTP_LOG(Log::error, pLog)
141 << (boost::format(" Atom\t x\t y\t z ")).str() << std::flush;
142 const Eigen::VectorXd& atomvec = bfgstrm.getParameters();
143 for (Index i = 0; i < atomvec.size(); i += 3) {
144 XTP_LOG(Log::error, pLog)
145 << (boost::format("%1$4d %2$+1.4f %3$+1.4f %4$+1.4f") % (i / 3) %
146 (atomvec(i) * votca::tools::conv::bohr2ang) %
147 (atomvec(i + 1) * votca::tools::conv::bohr2ang) %
148 (atomvec(i + 2) * votca::tools::conv::bohr2ang))
149 .str()
150 << std::flush;
151 }
152 XTP_LOG(Log::error, pLog)
153 << (boost::format(" Total energy: %1$12.8f Hartree ") %
154 bfgstrm.getCost())
155 .str()
156 << std::flush;
157 XTP_LOG(Log::error, pLog)
158 << (boost::format(" Trust radius: %1$12.8f Bohr ") %
159 bfgstrm.getTrustRadius())
160 .str()
161 << std::flush;
162 forces.Report();
163 return;
164}

166void GeometryOptimization::WriteTrajectory(const std::string& filename,
167 QMMolecule& atoms,
168 const BFGSTRM& bfgstrm) {
169 std::ofstream ofs;
170 if (bfgstrm.getIteration() == 0) {
171 ofs.open(filename, std::ofstream::out);
172 } else {
173 ofs.open(filename, std::ofstream::app);
174 }
175 if (!ofs.is_open()) {
176 throw std::runtime_error("Bad file handle: " + filename);
177 }
178 // write coordinates as xyz file
179 ofs << atoms.size() << std::endl;
180 ofs << "iteration " << bfgstrm.getIteration() << " energy "
181 << bfgstrm.getCost() << " Hartree" << std::endl;
182 Energy_costfunction::Vector2QMAtoms(bfgstrm.getParameters(), atoms);
183 for (const QMAtom& atom : atoms) {
184 const Eigen::Vector3d pos = atom.getPos() * tools::conv::bohr2ang;
185 ofs << atom.getElement() << " " << pos.x() << " " << pos.y() << " "
186 << pos.z() << std::endl;
187 }
188 ofs.close();
189 return;
190}
