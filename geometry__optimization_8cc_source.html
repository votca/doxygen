30 void GeometryOptimization::Initialize(tools::Property& options) {
31 
32  std::string statestring =
33  options.ifExistsReturnElseThrowRuntimeError<std::string>(".state");
34  _opt_state.FromString(statestring);
35  if (!_opt_state.Type().isExciton()) {
36  throw std::runtime_error(
37  "At the moment only excitonic states can be optimized");
38  }
39  // default convergence parameters from ORCA
40  _conv.deltaE = options.ifExistsReturnElseReturnDefault<double>(
41  ".convergence.energy", 1.e-6); // Hartree
42  _conv.RMSForce = options.ifExistsReturnElseReturnDefault<double>(
43  ".convergence.RMSForce", 3.e-5); // Hartree/Bohr
44  _conv.MaxForce = options.ifExistsReturnElseReturnDefault<double>(
45  ".convergence.MaxForce", 1.e-4); // Hartree/Bohr
46  _conv.RMSStep = options.ifExistsReturnElseReturnDefault<double>(
47  ".convergence.RMSStep", 6.e-4); // Bohr
48  _conv.MaxStep = options.ifExistsReturnElseReturnDefault<double>(
49  ".convergence.MaxStep", 1.e-3); // Bohr
50  _trust_radius = options.ifExistsReturnElseReturnDefault<double>(
51  ".trust", 0.01); // Angstrom
52  _trust_radius *= tools::conv::ang2bohr; // initial trust radius in a.u.
53 
54  _max_iteration =
55  options.ifExistsReturnElseReturnDefault<Index>(".maxiter", 50);
56 
57  _trajfile = options.ifExistsReturnElseReturnDefault<std::string>(
58  ".trajectory_file", "optimisation.trj");
59 
60  std::vector<std::string> choices = {"BFGS-TRM"};
61  _optimizer =
62  options.ifExistsAndinListReturnElseThrowRuntimeError<std::string>(
63  ".optimizer.method", choices);
64 
65  if (options.exists(".forces")) {
66  _force_options = options.get(".forces");
67  } else {
68  throw std::runtime_error("No forces options provided");
69  }
70  if (options.exists(".statetracker")) {
71  _statetracker_options = options.get(".statetracker");
72  } else {
73  throw std::runtime_error("No filter options set");
74  }
75 
76  return;
77 }

79 void GeometryOptimization::Evaluate() {
80  XTP_LOG(Log::error, *_pLog)
81  << "Requested geometry optimization of excited state "
82  << _opt_state.ToString() << std::flush;
83 
84  StateTracker tracker;
85  tracker.Initialize(_statetracker_options);
86  tracker.setInitialState(_opt_state);
87  tracker.setLogger(_pLog);
88  tracker.PrintInfo();
89 
90  // get a force object
91  Forces force_engine(_gwbse_engine, tracker);
92  force_engine.Initialize(_force_options);
93  force_engine.setLog(_pLog);
94  XTP_LOG(Log::error, *_pLog)
95  << (boost::format("Convergence of total energy: %1$8.6f Hartree ") %
96  _conv.deltaE)
97  .str()
98  << std::flush;
99  XTP_LOG(Log::error, *_pLog)
100  << (boost::format("Convergence of RMS Force: %1$8.6f Hartree/Bohr ") %
101  _conv.RMSForce)
102  .str()
103  << std::flush;
104  XTP_LOG(Log::error, *_pLog)
105  << (boost::format("Convergence of Max Force: %1$8.6f Hartree/Bohr ") %
106  _conv.MaxForce)
107  .str()
108  << std::flush;
109  XTP_LOG(Log::error, *_pLog)
110  << (boost::format("Convergence of RMS Step: %1$8.6f Bohr ") %
111  _conv.RMSStep)
112  .str()
113  << std::flush;
114  XTP_LOG(Log::error, *_pLog)
115  << (boost::format("Convergence of Max Step: %1$8.6f Bohr ") %
116  _conv.MaxStep)
117  .str()
118  << std::flush;
119  XTP_LOG(Log::error, *_pLog)
120  << (boost::format("Initial trust radius: %1$8.6f Bohr") %
121  _trust_radius)
122  .str()
123  << std::flush;
124 
125  Energy_costfunction e_cost =
126  Energy_costfunction(_gwbse_engine, tracker, _orbitals, force_engine);
127  e_cost.setConvergenceParameters(_conv);
128  e_cost.setLog(_pLog);
129  // get the optimizer
130  if (_optimizer == "BFGS-TRM") {
131  BFGSTRM bfgstrm(e_cost);
132  std::vector<std::function<void()> > callbacks;
133  std::function<void()> reporting = std::bind(
134  Report, std::cref(bfgstrm), std::cref(force_engine), std::ref(*_pLog));
135  callbacks.push_back(reporting);
136  std::function<void()> filewrite = std::bind(
137  WriteTrajectory, _trajfile, _orbitals.QMAtoms(), std::cref(bfgstrm));
138  callbacks.push_back(filewrite);
139  bfgstrm.setCallbacks(callbacks);
140  bfgstrm.setNumofIterations(_max_iteration);
141  bfgstrm.setTrustRadius(_trust_radius);
142  bfgstrm.setLog(_pLog);
143  bfgstrm.Optimize(Energy_costfunction::QMAtoms2Vector(_orbitals.QMAtoms()));
144  }
145  return;
146 }

148 void GeometryOptimization::Report(const BFGSTRM& bfgstrm, const Forces& forces,
149  Logger& pLog) {
150 
151  XTP_LOG(Log::error, pLog) << std::flush;
152  XTP_LOG(Log::error, pLog)
153  << (boost::format("=========== OPTIMIZATION SUMMARY "
154  "================================= "))
155  .str()
156  << std::flush;
157  XTP_LOG(Log::error, pLog)
158  << "At iteration " << bfgstrm.getIteration() << std::flush;
159  XTP_LOG(Log::error, pLog)
160  << (boost::format(" ---- POSITIONS (Angstrom) ")).str() << std::flush;
161  XTP_LOG(Log::error, pLog)
162  << (boost::format(" Atom\t x\t y\t z ")).str() << std::flush;
163  const Eigen::VectorXd& atomvec = bfgstrm.getParameters();
164  for (Index i = 0; i < atomvec.size(); i += 3) {
165  XTP_LOG(Log::error, pLog)
166  << (boost::format("%1$4d %2$+1.4f %3$+1.4f %4$+1.4f") % (i / 3) %
167  (atomvec(i) * votca::tools::conv::bohr2ang) %
168  (atomvec(i + 1) * votca::tools::conv::bohr2ang) %
169  (atomvec(i + 2) * votca::tools::conv::bohr2ang))
170  .str()
171  << std::flush;
172  }
173  XTP_LOG(Log::error, pLog)
174  << (boost::format(" Total energy: %1$12.8f Hartree ") %
175  bfgstrm.getCost())
176  .str()
177  << std::flush;
178  XTP_LOG(Log::error, pLog)
179  << (boost::format(" Trust radius: %1$12.8f Bohr ") %
180  bfgstrm.getTrustRadius())
181  .str()
182  << std::flush;
183  forces.Report();
184  return;
185 }

187 void GeometryOptimization::WriteTrajectory(const std::string& filename,
188  QMMolecule& atoms,
189  const BFGSTRM& bfgstrm) {
190  std::ofstream ofs;
191  if (bfgstrm.getIteration() == 0) {
192  ofs.open(filename, std::ofstream::out);
193  } else {
194  ofs.open(filename, std::ofstream::app);
195  }
196  if (!ofs.is_open()) {
197  throw std::runtime_error("Bad file handle: " + filename);
198  }
199  // write coordinates as xyz file
200  ofs << atoms.size() << std::endl;
201  ofs << "iteration " << bfgstrm.getIteration() << " energy "
202  << bfgstrm.getCost() << " Hartree" << std::endl;
203  Energy_costfunction::Vector2QMAtoms(bfgstrm.getParameters(), atoms);
204  for (const QMAtom& atom : atoms) {
205  const Eigen::Vector3d pos = atom.getPos() * tools::conv::bohr2ang;
206  ofs << atom.getElement() << " " << pos.x() << " " << pos.y() << " "
207  << pos.z() << std::endl;
208  }
209  ofs.close();
210  return;
211 }
