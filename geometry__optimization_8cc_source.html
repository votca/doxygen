30  void GeometryOptimization::Initialize(tools::Property &options) {
31 
32  std::string statestring = options.ifExistsReturnElseThrowRuntimeError<std::string>(".state");
33  _opt_state.FromString(statestring);
34  if (!_opt_state.Type().isExciton()) {
35  throw std::runtime_error("At the moment only excitonic states can be optimized");
36  }
37  // default convergence parameters from ORCA
38  _conv.deltaE = options.ifExistsReturnElseReturnDefault<double>(".convergence.energy", 1.e-6); // Hartree
39  _conv.RMSForce = options.ifExistsReturnElseReturnDefault<double>(".convergence.RMSForce", 3.e-5); // Hartree/Bohr
40  _conv.MaxForce = options.ifExistsReturnElseReturnDefault<double>(".convergence.MaxForce", 1.e-4); // Hartree/Bohr
41  _conv.RMSStep = options.ifExistsReturnElseReturnDefault<double>(".convergence.RMSStep", 6.e-4); // Bohr
42  _conv.MaxStep = options.ifExistsReturnElseReturnDefault<double>(".convergence.MaxStep", 1.e-3); // Bohr
43  _trust_radius = options.ifExistsReturnElseReturnDefault<double>(".trust", 0.01); // Angstrom
44  _trust_radius *= tools::conv::ang2bohr; // initial trust radius in a.u.
45 
46  _max_iteration = options.ifExistsReturnElseReturnDefault<unsigned>(".maxiter", 50);
47 
48  _trajfile = options.ifExistsReturnElseReturnDefault<std::string>(".trajectory_file", "optimisation.trj");
49 
50 
51  std::vector<std::string> choices = {"BFGS-TRM"};
52  _optimizer = options.ifExistsAndinListReturnElseThrowRuntimeError<std::string>(".optimizer.method", choices);
53 
54  if (options.exists(".forces")) {
55  _force_options = options.get(".forces");
56  } else {
57  throw std::runtime_error("No forces options provided");
58  }
59  if (options.exists(".filter")) {
60  _filter_options = options.get(".filter");
61  } else {
62  throw std::runtime_error("No filter options set");
63  }
64 
65  return;
66  }

68  void GeometryOptimization::Evaluate() {
69  CTP_LOG(ctp::logINFO, *_pLog) << "Requested geometry optimization of excited state " << _opt_state.ToString() << std::flush;
70 
71  Statefilter filter;
72  filter.Initialize(_filter_options);
73  filter.setInitialState(_opt_state);
74  filter.setLogger(_pLog);
75  filter.PrintInfo();
76 
77  // get a force object
78  Forces force_engine(_gwbse_engine, filter);
79  force_engine.Initialize(_force_options);
80  force_engine.setLog(_pLog);
81  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Convergence of total energy: %1$8.6f Hartree ") % _conv.deltaE).str() << std::flush;
82  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Convergence of RMS Force: %1$8.6f Hartree/Bohr ") % _conv.RMSForce).str() << std::flush;
83  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Convergence of Max Force: %1$8.6f Hartree/Bohr ") % _conv.MaxForce).str() << std::flush;
84  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Convergence of RMS Step: %1$8.6f Bohr ") % _conv.RMSStep).str() << std::flush;
85  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Convergence of Max Step: %1$8.6f Bohr ") % _conv.MaxStep).str() << std::flush;
86  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("Initial trust radius: %1$8.6f Bohr") % _trust_radius).str() << std::flush;
87 
88  Energy_costfunction e_cost = Energy_costfunction(_gwbse_engine, filter, _orbitals, force_engine);
89  e_cost.setConvergenceParameters(_conv);
90  e_cost.setLog(_pLog);
91  // get the optimizer
92  if (_optimizer == "BFGS-TRM") {
93  BFGSTRM bfgstrm(e_cost);
94  std::vector<std::function<void()> > callbacks;
95  std::function<void()> reporting=std::bind(Report,std::cref(bfgstrm),std::cref(force_engine),std::cref(_pLog));
96  callbacks.push_back(reporting);
97  std::function<void()> filewrite=std::bind(WriteTrajectory,_trajfile, _orbitals.QMAtoms(), std::cref(bfgstrm));
98  callbacks.push_back(filewrite);
99  bfgstrm.setCallbacks(callbacks);
100  bfgstrm.setNumofIterations(_max_iteration);
101  bfgstrm.setTrustRadius(_trust_radius);
102  bfgstrm.setLog(_pLog);
103  bfgstrm.Optimize(Energy_costfunction::QMAtoms2Vector(_orbitals.QMAtoms()));
104  }
105  return;
106  }

108  void GeometryOptimization::Report(const BFGSTRM& bfgstrm,const Forces& forces,ctp::Logger* pLog){
109 
110  CTP_LOG(ctp::logINFO, *pLog) << std::flush;
111  CTP_LOG(ctp::logINFO, *pLog) << (boost::format("=========== OPTIMIZATION SUMMARY ================================= ")).str() << std::flush;
112  CTP_LOG(ctp::logINFO, *pLog) << "At iteration " << bfgstrm.getIteration() << std::flush;
113  CTP_LOG(ctp::logINFO, *pLog) << (boost::format(" ---- POSITIONS (Angstrom) ")).str() << std::flush;
114  CTP_LOG(ctp::logINFO, *pLog) << (boost::format(" Atom\t x\t y\t z ")).str() << std::flush;
115  const Eigen::VectorXd& atomvec = bfgstrm.getParameters();
116  for (unsigned i = 0; i < atomvec.size(); i += 3) {
117  CTP_LOG(ctp::logINFO, *pLog) << (boost::format("%1$4d %2$+1.4f %3$+1.4f %4$+1.4f")
118  % (i/3) % (atomvec(i) * votca::tools::conv::bohr2ang) % (atomvec(i + 1) * votca::tools::conv::bohr2ang) % (atomvec(i + 2) * votca::tools::conv::bohr2ang)).str() << std::flush;
119  }
120  CTP_LOG(ctp::logINFO, *pLog) << (boost::format(" Total energy: %1$12.8f Hartree ") % bfgstrm.getCost()).str() << std::flush;
121  CTP_LOG(ctp::logINFO, *pLog) << (boost::format(" Trust radius: %1$12.8f Bohr ") % bfgstrm.getTrustRadius()).str() << std::flush;
122  forces.Report();
123  return;
124  }

128  void GeometryOptimization::WriteTrajectory(const std::string& filename, std::vector< QMAtom* >& atoms, const BFGSTRM& bfgstrm){
129  std::ofstream ofs;
130  if (bfgstrm.getIteration() == 0) {
131  ofs.open(filename.c_str(), std::ofstream::out);
132  } else {
133  ofs.open(filename.c_str(), std::ofstream::app);
134  }
135  if (!ofs.is_open()) {
136  throw std::runtime_error("Bad file handle: " + filename);
137  }
138  // write coordinates as xyz file
139  ofs << atoms.size() << std::endl;
140  ofs << "iteration " << bfgstrm.getIteration() << " energy " << bfgstrm.getCost() << " Hartree" << std::endl;
141  Energy_costfunction::Vector2QMAtoms(bfgstrm.getParameters(), atoms);
142  for (const QMAtom* atom : atoms) {
143  tools::vec pos = atom->getPos() * tools::conv::bohr2ang;
144  ofs << atom->getType() << " " << pos.getX() << " " << pos.getY() << " " << pos.getZ() << std::endl;
145  }
146  ofs.close();
147  return;
148  }
