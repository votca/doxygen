118 void KMCApplication::LoadGraph() {
119 
120  Database db;
121  db.Open(OptionsMap()["graph"].as<string > ());
122 
123  Statement *stmt = db.Prepare("SELECT _id,name FROM conjsegs;");
124 
125  while (stmt->Step() != SQLITE_DONE) {
126  int id = stmt->Column<int>(0);
127  string name = stmt->Column<string>(1);
128  node_t *n =new node_t(id);
129  _nodes.push_back(n);
130  _nodes_lookup[id] = _nodes.back();
131  if (wildcmp(_injection_name.c_str(), name.c_str())) {
132  _injection.push_back(n);
133  _injection_lookup[id] = _injection.back();
134  }
135  }
136  //delete stmt;
137  cout << "Nodes: " << _nodes.size() << endl;
138  cout << "Injection Points: " << _injection.size() << endl;
139 
140  delete stmt;
141 
142 
143  int links = 0;
144  stmt = db.Prepare("SELECT conjseg1, conjseg2, rate12, rate21, r_x, r_y, r_z FROM pairs;");
145  while (stmt->Step() != SQLITE_DONE) {
146  node_t *n1 = _nodes_lookup[stmt->Column<int>(0)];
147  node_t *n2 = _nodes_lookup[stmt->Column<int>(1)];
148  double rate12 = stmt->Column<double>(2);
149  double rate21 = stmt->Column<double>(3);
150  vec r = vec(stmt->Column<double>(4), stmt->Column<double>(5), stmt->Column<double>(6));
151  n1->AddEvent(new link_t(n2, rate12, r));
152  n2->AddEvent(new link_t(n1, rate21, -r));
153  links += 2;
154  }
155  delete stmt;
156  cout << "Links: " << links << endl;
157 }

159 void KMCApplication::RunKMC(void)
160 {
161  double t = 0;
162  current=_injection[Random::rand_uniform_int(_injection.size())];
163  cout <<"starting simulation at node: "<<current->_id-1<<endl;
164  double next_output = _dt;
165  int i=0;
166  while(t<_runtime) {
167  t+=current->WaitingTime();
168  current->onExecute();
169  if(t > next_output) {
170  next_output = t + _dt;
171  cout << t << ": " << r << endl;
172  }
173  }
174  _runtime = t;
175  WriteOcc();
176  cout << std::scientific << "\nKMC run finished\nAverage velocity (m/s): " << r/t*1e-9 << endl;

179 void KMCApplication::WriteOcc()
180 {
181  Database db;
182  db.Open(OptionsMap()["graph"].as<string>());
183  db.Exec("BEGIN;");
184  Statement *stmt = db.Prepare("UPDATE conjsegs SET occ = ? WHERE _id = ?;");
185  for(int i=0; i<_nodes.size(); ++i) {
186  stmt->Reset();
187  stmt->Bind(1, _nodes[i]->_occ/_runtime);
188  stmt->Bind(2, _nodes[i]->_id);
189  stmt->Step();
190  }
191  db.Exec("END;");
192  delete stmt;
193 }
