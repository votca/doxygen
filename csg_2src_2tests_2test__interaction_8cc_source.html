111

114 double charge = 1.0;
115 votca::Index resid = 1;
116 Bead* bead1 = top.CreateBead(Bead::spherical, "a1", "C", resid, mass, charge);
117 Eigen::Vector3d pos1(1, 0, 0);
118 bead1->setPos(pos1);
119 Bead* bead2 = top.CreateBead(Bead::spherical, "a2", "C", resid, mass, charge);
120 Eigen::Vector3d pos2(0, 0, 0);
121 bead2->setPos(pos2);
122 IBond bond1(0, 1);
123 double length = bond1.EvaluateVar(top);
124 Eigen::Vector3d grad0 = bond1.Grad(top, 0);
125 Eigen::Vector3d grad1 = bond1.Grad(top, 1);
126 Eigen::Vector3d grad0_ref(1, 0, 0);
127 Eigen::Vector3d grad1_ref(-1, 0, 0);
128 BOOST_CHECK_CLOSE(length, 1.0, 1e-5);
129 bool grad0_check = grad0.isApprox(grad0_ref, 1e-5);
130 BOOST_CHECK(grad0_check);
131 if (!grad0_check) {
132 std::cout << "ref" << std::endl;
133 std::cout << grad0_ref << std::endl;
134 std::cout << "result" << std::endl;
135 std::cout << grad0 << std::endl;
136 }
137 bool grad1_check = grad1.isApprox(grad1_ref, 1e-5);
