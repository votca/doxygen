63  CudaMatrix(const Eigen::MatrixXd &matrix, const cudaStream_t &stream)
64  : _rows{static_cast<int>(matrix.rows())},
65  _cols{static_cast<int>(matrix.cols())} {
66  _data = alloc_matrix_in_gpu(size_matrix());
67  _stream = stream;
68  cudaError_t err = cudaMemcpyAsync(_data.get(), matrix.data(), size_matrix(),
69  cudaMemcpyHostToDevice, stream);
70  if (err != 0) {
71  throw std::runtime_error("Error copy arrays to device");
72  }

101  double *dmatrix;

103  checkCuda(cudaMalloc(&dmatrix, size_arr));

109  void throw_if_not_enough_memory_in_gpu(size_t requested_memory) const {
110  size_t free, total;
111  checkCuda(cudaMemGetInfo(&free, &total));
112 
113  std::ostringstream oss;
114  oss << "There were requested : " << requested_memory
115  << "bytes int the device\n";
116  oss << "Device Free memory (bytes): " << free
117  << "\nDevice total Memory (bytes): " << total << "\n";
118 
119  // Raise an error if there is not enough total or free memory in the device
120  if (requested_memory > free) {
121  oss << "There is not enough memory in the Device!\n";
122  throw std::runtime_error(oss.str());
123  }
124  }
