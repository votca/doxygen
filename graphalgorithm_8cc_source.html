57 vector<shared_ptr<Graph>> decoupleIsolatedSubGraphs(Graph g) {
58 
59  auto vertices_list = vectorToList_(g.getVertices());
60  vector<shared_ptr<Graph>> subGraphs;
61  {
62  Graph_BF_Visitor gv_breadth_first;
63  gv_breadth_first.setStartingVertex(*vertices_list.begin());
64  if (singleNetwork(g, gv_breadth_first)) {
65  subGraphs.push_back(make_shared<Graph>(g));
66  return subGraphs;
67  }
68  }
69 
70  auto v_it = vertices_list.begin();
71  unordered_map<int, GraphNode> sub_graph_nodes;
72  while (v_it != vertices_list.end()) {
73  Graph_BF_Visitor gv_breadth_first;
74  gv_breadth_first.setStartingVertex(*v_it);
75  exploreGraph(g, gv_breadth_first);
76 
77  ++v_it;
78  auto sub_graph_explored_vertices = gv_breadth_first.getExploredVertices();
79  set<Edge> sub_graph_edges;
80 
81  auto sub_graph_vertex_it = sub_graph_explored_vertices.begin();
82  while(sub_graph_vertex_it!=sub_graph_explored_vertices.end()) {
83  if (*v_it == *sub_graph_vertex_it) {
84  ++v_it;
85  }
86  vertices_list.remove(*sub_graph_vertex_it);
87 
88  auto sub_graph_neigh_edges = g.getNeighEdges(*sub_graph_vertex_it);
89  for (auto sub_graph_edge : sub_graph_neigh_edges) {
90  sub_graph_edges.insert(sub_graph_edge);
91  }
92 
93  sub_graph_nodes[*sub_graph_vertex_it] = g.getNode(*sub_graph_vertex_it);
94  ++sub_graph_vertex_it;
95  }
96 
97  auto sub_graph_vector_edges = edgeSetToVector_(sub_graph_edges);
98 
99  subGraphs.push_back(
100  make_shared<Graph>(Graph(sub_graph_vector_edges, sub_graph_nodes)));
101  }
102  return subGraphs;
103 }

105 void exploreGraph(Graph& g, GraphVisitor& gv) {
106 
107  if (!g.vertexExist(gv.getStartingVertex())) {
108  string err = "Cannot explore graph starting at vertex " +
109  to_string(gv.getStartingVertex()) +
110  " because it does not exist in the "
111  "graph. You can change the starting vertex by using the "
112  "setStartingVertex method of the visitor instance.";
113  throw invalid_argument(err);
114  }
115  // Create a list of all vertices and determine if they have all been
116  // explored when the visitor que is empty
117  gv.initialize(g);
118  while (!gv.queEmpty()) {
119  auto ed = gv.nextEdge(g);
120  gv.exec(g, ed);
121  }
122 }
123 }
