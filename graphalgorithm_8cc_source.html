46 std::set<Edge> exploreBranch(Graph g, long int starting_vertex,
47  const Edge& edge) {
48  // Check if the starting vertex is in the graph
49  if (!g.vertexExist(starting_vertex)) {
50  throw invalid_argument(
51  "Cannot explore a branch of the graph when the "
52  "exploration is started from a vertex that does not exist within the "
53  "graph.");
54  }
55  if (!g.edgeExist(edge)) {
56  throw invalid_argument(
57  "Edge does not exist in the graph so exploration of"
58  " the graph branch cannot continue");
59  }
60  if (!edge.contains(starting_vertex)) {
61  throw invalid_argument(
62  "The edge determining which branch to explore does "
63  "not contain the starting vertex.");
64  }
65 
66  set<Edge> branch_edges;
67  if (edge.getEndPoint1() == edge.getEndPoint2()) {
68  branch_edges.insert(edge);
69  return branch_edges;
70  }
71  Graph_BF_Visitor gv_breadth_first;
72  GraphNode gn;
73  pair<long int, GraphNode> pr_gn(starting_vertex, gn);
74  gv_breadth_first.exploreNode(pr_gn, g);
75  gv_breadth_first.setStartingVertex(edge.getOtherEndPoint(starting_vertex));
76  gv_breadth_first.initialize(g);
77 
78  branch_edges.insert(edge);
79  while (!gv_breadth_first.queEmpty()) {
80  Edge ed = gv_breadth_first.nextEdge(g);
81  branch_edges.insert(ed);
82  gv_breadth_first.exec(g, ed);
83  }
84 
85  vector<Edge> neigh_edges = g.getNeighEdges(starting_vertex);
86  for (Edge& ed : neigh_edges) {
87  long int neigh_vertex = ed.getOtherEndPoint(starting_vertex);
88  if (neigh_vertex != starting_vertex) {
89  if (gv_breadth_first.vertexExplored(neigh_vertex)) {
90  branch_edges.insert(ed);
91  }
92  }
93  }
94 
95  return branch_edges;
96 }

111  class ExplorationRecord {
112  private:
113  unordered_map<long int, std::pair<bool, long int>> vertex_explored_;
114  size_t unexplored_vertex_count_;
115 
116  public:
117  explicit ExplorationRecord(

121  unexplored_vertex_count_(vertex_explored.size()){};

130  long int getUnexploredVertex() {
131 
132  vector<long int> remaining_unexplored;
133  for (const pair<long int, pair<bool, long int>>& vertex_record :
134  vertex_explored_) {
135  bool vertex_explored = vertex_record.second.first;
136  if (!vertex_explored) {
137  long int degree = vertex_record.second.second;
138  if (degree > 2) {
139  return vertex_record.first;
140  }
141  remaining_unexplored.push_back(vertex_record.first);
142  }
143  }
144 
145  // Search tips next
146  for (const long int& vertex : remaining_unexplored) {
147  if (vertex_explored_[vertex].second == 1) {
148  return vertex;
149  }
150  }
151 
152  // Finally if there are no tips or junctions left we will return a vertex
153  // of degree 2 if one exists
154  for (const long int& vertex : remaining_unexplored) {
155  if (!vertex_explored_[vertex].first) {
156  return vertex;
157  }
158  }
159 
160  throw runtime_error(
161  "You cannot get an unexplored vertex as they have all"
162  " been explored.");
163  }
164  }; // Class ExplorationRecord
165 
166  unordered_map<long int, pair<bool, long int>> unexplored_vertices;
167  vector<long int> vertices = graph.getVertices();
168  for (const long int& vertex : vertices) {
169  unexplored_vertices[vertex] =
170  pair<bool, long int>(false, graph.getDegree(vertex));
171  }
172 
173  ExplorationRecord exploration_record(unexplored_vertices);
174 
175  vector<vector<long int>> chains;
176 
177  while (exploration_record.unexploredVerticesExist()) {
178  Graph_DF_Visitor graph_visitor;
179  long int starting_vertex = exploration_record.getUnexploredVertex();
180  exploration_record.explore(starting_vertex);
181  graph_visitor.setStartingVertex(starting_vertex);
182  graph_visitor.initialize(graph);
183 
184  vector<long int> chain{starting_vertex};
185  long int old_vertex = starting_vertex;
186  bool new_chain = false;
187  while (!graph_visitor.queEmpty()) {
188  Edge edge = graph_visitor.nextEdge(graph);
189  vector<long int> unexplored_vertex =
190  graph_visitor.getUnexploredVertex(edge);
191 
192  if (new_chain) {
193  if (unexplored_vertex.size() == 0) {
194  old_vertex = edge.getEndPoint1();
195  chain.push_back(old_vertex);
196  new_chain = false;
197  } else {
198  old_vertex = edge.getOtherEndPoint(unexplored_vertex.at(0));
199  chain.push_back(old_vertex);
200  new_chain = false;
201  }
202  }
203  long int new_vertex = edge.getOtherEndPoint(old_vertex);
204  if (unexplored_vertex.size() == 0) {
205  chain.push_back(new_vertex);
206  chains.push_back(chain);
207  chain.clear();
208  new_chain = true;
209  } else if (graph.getDegree(new_vertex) == 1) {
210  chain.push_back(new_vertex);
211  chains.push_back(chain);
212  chain.clear();
213  exploration_record.explore(new_vertex);
214  new_chain = true;
215  } else if (graph.getDegree(new_vertex) > 2) {
216  chain.push_back(new_vertex);
217  chains.push_back(chain);
218  chain.clear();
219  exploration_record.explore(new_vertex);
220  new_chain = true;
221  } else if (unexplored_vertex.size() == 1) {
222  chain.push_back(new_vertex);
223  old_vertex = new_vertex;
224  exploration_record.explore(new_vertex);
225  }
226 
227  graph_visitor.exec(graph, edge);
228  }
229  }
230  vector<ReducedEdge> reduced_edges;
231  for (vector<long int> chain : chains) {
232  ReducedEdge reduced_edge(chain);
233  reduced_edges.push_back(reduced_edge);
234  }
235 
236  ReducedGraph reduced_g(reduced_edges);
237  auto nodes_graph = graph.getNodes();
238  reduced_g.clearNodes();
239  reduced_g.copyNodes(graph);
240  auto nodes_reduced_g = reduced_g.getNodes();
241  return reduced_g;
242 }

244 vector<Graph> decoupleIsolatedSubGraphs(Graph graph) {
245 
246  const std::vector<long int>& vertices = graph.getVertices();
247  // bool vector to see if vertex is already part of graph
248  std::vector<bool> vertex_analysed = std::vector<bool>(vertices.size(), false);
249 
250  std::vector<Graph> subGraphs;
251  unsigned i = 0;
252  while (i < vertices.size()) {
253  if (vertex_analysed[i]) {
254  i++;
255  continue;
256  }
257  Graph_BF_Visitor graph_visitor_breadth_first;
258  graph_visitor_breadth_first.setStartingVertex(vertices[i]);
259  exploreGraph(graph, graph_visitor_breadth_first);
260  set<long int> sub_graph_explored_vertices =
261  graph_visitor_breadth_first.getExploredVertices();
262 
263  for (long int vertex : sub_graph_explored_vertices) {
264  for (unsigned j = 0; j < vertices.size(); j++) {
265  if (vertex_analysed[j]) {
266  continue;
267  }
268  if (vertex == vertices[j]) {
269  vertex_analysed[j] = true;
270  break;
271  }
272  }
273  }
274 
275  set<Edge> sub_graph_edges;
276  unordered_map<int long, GraphNode> sub_graph_nodes;
277  for (long int vertex : sub_graph_explored_vertices) {
278 
279  for (const Edge& sub_graph_edge : graph.getNeighEdges(vertex)) {
280  sub_graph_edges.insert(sub_graph_edge);
281  }
282  sub_graph_nodes[vertex] = graph.getNode(vertex);
283  }
284  subGraphs.push_back(
285  Graph(std::vector<Edge>(sub_graph_edges.begin(), sub_graph_edges.end()),
286  sub_graph_nodes));
287  }
288  return subGraphs;
289 }

291 void exploreGraph(Graph& graph, GraphVisitor& graph_visitor) {
292 
293  if (!graph.vertexExist(graph_visitor.getStartingVertex())) {
294  string err = "Cannot explore graph starting at vertex " +
295  to_string(graph_visitor.getStartingVertex()) +
296  " because it does not exist in the "
297  "graph. You can change the starting vertex by using the "
298  "setStartingVertex method of the visitor instance.";
299  throw invalid_argument(err);
300  }
301  // Create a list of all vertices and determine if they have all been
302  // explored when the visitor que is empty
303  graph_visitor.initialize(graph);
304  while (!graph_visitor.queEmpty()) {
305  Edge edge = graph_visitor.nextEdge(graph);
306  graph_visitor.exec(graph, edge);
307  }
308 }
