00073 void GmxTopolApp::WriteAtoms(ostream &out, Molecule &cg)
00074 {
00075     out << "[atoms]\n";
00076     out << "; nr type resnr residue atom cgnr charge mass\n";
00077     for(int i=0; i<cg.BeadCount(); ++i) {
00078         Bead *b=cg.getBead(i);
00079        
00080         out << format("%d %s 1 RES %s %d %f %f\n")
00081             % (i+1) % b->getType()->getName() % b->getName() % (i+1) % b->getQ() % b->getM();
00082     }
00083     out << endl;
00084 }

00086 void GmxTopolApp::WriteInteractions(ostream &out, Molecule &cg)
00087 {
00088     int nb=-1;
00089     
00090     Interaction *ic;
00091     vector<Interaction *>::iterator iter;
00092   
00093     InteractionContainer &ics=cg.getParent()->BondedInteractions();
00094 
00095     for(iter=ics.begin(); iter!=ics.end(); ++iter) {
00096         ic = *iter;
00097         if(ic->getMolecule() != cg.getId()) continue;
00098         if(nb != ic->BeadCount()) {
00099             nb=ic->BeadCount();
00100             switch(nb) {
00101                 case 2:
00102                     out << "\n[ bonds ]\n";
00103                     break;
00104                 case 3:
00105                     out << "\n[ angles ]\n";
00106                     break;
00107                 case 4:
00108                     out << "\n[ dihedrals ]\n";
00109                     break;
00110                 default:
00111                     throw runtime_error(string("cannot handle number of beads in interaction:") +
00112                             ic->getName());
00113             }
00114         }
00115         for(int i=0; i<nb; ++i)
00116             out << ic->getBeadId(i)+1 << " ";
00117         out << "  1  ; " << ic->getName() << endl;
00118     }
00119 }
