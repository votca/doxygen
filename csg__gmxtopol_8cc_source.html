60 bool GmxTopolApp::EvaluateTopology(Topology *top, Topology *top_ref)
61 {

73 void GmxTopolApp::WriteAtoms(ostream &out, Molecule &cg)
74 {
75  out << "[atoms]\n";
76  out << "; nr type resnr residue atom cgnr charge mass\n";
77  for(int i=0; i<cg.BeadCount(); ++i) {
78  Bead *b=cg.getBead(i);
79 
80  out << format("%d %s 1 RES %s %d %f %f\n")
81  % (i+1) % b->getType()->getName() % b->getName() % (i+1) % b->getQ() % b->getM();
82  }
83  out << endl;
84 }

86 void GmxTopolApp::WriteInteractions(ostream &out, Molecule &cg)
87 {
88  int nb=-1;
89 
90  Interaction *ic;
91  vector<Interaction *>::iterator iter;
92 
93  InteractionContainer &ics=cg.getParent()->BondedInteractions();
94 
95  for(iter=ics.begin(); iter!=ics.end(); ++iter) {
96  ic = *iter;
97  if(ic->getMolecule() != cg.getId()) continue;
98  if(nb != ic->BeadCount()) {
99  nb=ic->BeadCount();
100  switch(nb) {
101  case 2:
102  out << "\n[ bonds ]\n";
103  break;
104  case 3:
105  out << "\n[ angles ]\n";
106  break;
107  case 4:
108  out << "\n[ dihedrals ]\n";
109  break;
110  default:
111  throw runtime_error(string("cannot handle number of beads in interaction:") +
112  ic->getName());
113  }
114  }
115  for(int i=0; i<nb; ++i)
116  out << ic->getBeadId(i)+1 << " ";
117  out << " 1 ; " << ic->getName() << endl;
118  }
119 }
