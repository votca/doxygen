58 bool GmxTopolApp::EvaluateTopology(Topology *top, Topology *) {
59  if (top->MoleculeCount() > 1) {
60  cout << "WARNING: cannot create topology for topology with"
61  "multiple molecules, using only first molecule\n";
62  }
63  ofstream fl;
64  fl.open((OptionsMap()["out"].as<string>() + ".top"));
65  WriteMolecule(fl, *top, *(top->MoleculeByIndex(0)));
66  fl.close();
67  return true;
68 }

70 void GmxTopolApp::WriteAtoms(ostream &out, Molecule &cg) {
71  out << "[atoms]\n";
72  out << "; nr type resnr residue atom cgnr charge mass\n";
73  for (votca::Index i = 0; i < cg.BeadCount(); ++i) {
74  Bead *b = cg.getBead(i);
75  out << format("%d %s 1 RES %s %d %f %f\n") % (i + 1) % b->getType() %
76  b->getName() % (i + 1) % b->getQ() % b->getMass();
77  }
78  out << endl;
79 }

81 void GmxTopolApp::WriteInteractions(ostream &out, const Topology &top,
82  Molecule &cg) {
83  votca::Index nb = -1;
84 
85  for (const Interaction *ic : top.BondedInteractions()) {
86  if (ic->getMolecule() != cg.getId()) {
87  continue;
88  }
89  if (nb != ic->BeadCount()) {
90  nb = ic->BeadCount();
91  switch (nb) {
92  case 2:
93  out << "\n[ bonds ]\n";
94  break;
95  case 3:
96  out << "\n[ angles ]\n";
97  break;
98  case 4:
99  out << "\n[ dihedrals ]\n";
100  break;
101  default:
102  string err = "cannot handle number of beads in interaction:";
103  err += to_string(ic->getMolecule() + 1) + ":" + ic->getGroup();
104  err += ":" + to_string(ic->getIndex() + 1);
105  throw runtime_error(err);
106  }
107  }
108  for (votca::Index i = 0; i < nb; ++i) {
109  out << ic->getBeadId(i) + 1 << " ";
110  }
111  out << " 1 ; ";
112  out << to_string(ic->getMolecule() + 1);
113  out << ":" + ic->getGroup();
114  out << ":" + to_string(ic->getIndex() + 1) << endl;
115  }
116 }
