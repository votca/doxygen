27 void Statefilter::Initialize(tools::Property& options) {
28  if (options.exists("oscillator_strength")) {
29  _use_oscfilter = true;
30  _oscthreshold = options.ifExistsReturnElseThrowRuntimeError<double>(
31  "oscillator_strength");
32  }
33  if (options.exists("overlap")) {
34  _use_overlapfilter = true;
35  _overlapthreshold =
36  options.ifExistsReturnElseReturnDefault<double>("overlap", 0.0);
37  }
38  if (options.exists("localisation")) {
39  _use_localisationfilter = true;
40  std::string temp = options.get("localisation").as<std::string>();
41  tools::Tokenizer tok_cleanup(temp, ", \n\t");
42  std::vector<std::string> strings_vec;
43  tok_cleanup.ToVector(strings_vec);
44  if (strings_vec.size() != 2) {
45  throw std::runtime_error(
46  "statefiler: Fragment and localisation threshold are not separated");
47  }
48  if (strings_vec[0] == "a" || strings_vec[0] == "A") {
49  _localiseonA = true;
50  } else if (strings_vec[0] == "b" || strings_vec[0] == "B") {
51  _localiseonA = false;
52  } else {
53  throw std::runtime_error(
54  "statefiler: Fragment label not known, either A or B");
55  }
56  _loc_threshold = boost::lexical_cast<double>(strings_vec[1]);
57  }
58 
59  if (options.exists("charge_transfer")) {
60  _use_dQfilter = true;
61  _dQ_threshold =
62  options.ifExistsReturnElseThrowRuntimeError<double>("charge_transfer");
63  }
64  if (_use_dQfilter && _use_localisationfilter) {
65  throw std::runtime_error(
66  "Cannot use localisation and charge_transfer filter at the same time.");
67  }
68 }

70 void Statefilter::PrintInfo() const {
71  CTP_LOG(ctp::logDEBUG, *_log)
72  << "Initial state: " << _statehist[0].ToString() << flush;
73  if (_statehist.size() > 1) {
74  CTP_LOG(ctp::logDEBUG, *_log)
75  << "Last state: " << _statehist.back().ToString() << flush;
76  }
77  if (_use_oscfilter) {
78  CTP_LOG(ctp::logDEBUG, *_log)
79  << "Using oscillator strength filter with cutoff " << _oscthreshold
80  << flush;
81  }
82  if (_use_overlapfilter) {
83  if (_overlapthreshold == 0.0) {
84  CTP_LOG(ctp::logDEBUG, *_log)
85  << "Using overlap filer with no cutoff " << flush;
86  } else {
87  CTP_LOG(ctp::logDEBUG, *_log)
88  << "Using overlap filer with cutoff " << _overlapthreshold << flush;
89  }
90  }
91  if (_use_localisationfilter) {
92  std::string fragment = "A";
93  if (!_localiseonA) {
94  fragment = "B";
95  }
96  CTP_LOG(ctp::logDEBUG, *_log)
97  << "Using localisation filter for fragment" << fragment
98  << " with cutoff " << _loc_threshold << flush;
99  }
100  if (_use_dQfilter) {
101  CTP_LOG(ctp::logDEBUG, *_log)
102  << "Using Delta Q filter with cutoff " << _dQ_threshold << flush;
103  }
104  if (_use_oscfilter && _use_dQfilter) {
105  CTP_LOG(ctp::logDEBUG, *_log) << "WARNING: filtering for optically active "
106  "CT transition - might not make sense... "
107  << flush;
108  }
109  if (_use_dQfilter + _use_oscfilter + _use_localisationfilter +
110  _use_oscfilter <
111  1) {
112  CTP_LOG(ctp::logDEBUG, *_log) << "WARNING: No filter is used " << flush;
113  }

116 std::vector<int> Statefilter::ComparePairofVectors(
117  std::vector<int>& vec1, std::vector<int>& vec2) const {
118  std::vector<int> result(std::min(vec1, vec2));
119  std::vector<int>::iterator it;
120  std::sort(vec1.begin(), vec1.end());
121  std::sort(vec2.begin(), vec2.end());
122  it = std::set_intersection(vec1.begin(), vec1.end(), vec2.begin(), vec2.end(),
123  result.begin());
124  result.resize(it - result.begin());
125  return result;
126 }

141 QMState Statefilter::CalcState(Orbitals& orbitals) const {
142 
143  if (_use_dQfilter + _use_oscfilter + _use_localisationfilter +
144  _use_oscfilter <
145  1) {
146  return _statehist[0];
147  }
148 
149  std::vector<std::vector<int> > results;
150  if (_use_oscfilter) {
151  results.push_back(OscFilter(orbitals));
152  }
153  if (_use_localisationfilter) {
154  results.push_back(LocFilter(orbitals));
155  }
156  if (_use_overlapfilter) {
157  results.push_back(OverlapFilter(orbitals));
158  }
159  if (_use_dQfilter) {
160  results.push_back(DeltaQFilter(orbitals));
161  }
162 
163  std::vector<int> result = CollapseResults(results);
164  QMState state;
165  if (result.size() < 1) {
166  state = _statehist.back();
167  CTP_LOG(ctp::logDEBUG, *_log)
168  << "No State found by filter using last state: " << state.ToString()
169  << flush;
170  } else {
171  state = QMState(_statehist.back().Type(), result[0], false);
172  CTP_LOG(ctp::logDEBUG, *_log)
173  << "Next State is: " << state.ToString() << flush;
174  }
175  return state;
176 }

196 std::vector<int> Statefilter::LocFilter(const Orbitals& orbitals) const {
197  std::vector<int> indexes;
198  const std::vector<Eigen::VectorXd>& popE =
199  (_statehist[0].Type() == QMStateType::Singlet)
200  ? orbitals.getFragment_E_localisation_singlet()
201  : orbitals.getFragment_E_localisation_triplet();
202  const std::vector<Eigen::VectorXd>& popH =
203  (_statehist[0].Type() == QMStateType::Singlet)
204  ? orbitals.getFragment_H_localisation_singlet()
205  : orbitals.getFragment_H_localisation_triplet();
206  int fragmentindex = 1;
207  if (_localiseonA) {
208  fragmentindex = 0;
209  }
210  for (unsigned i = 0; i < popE.size(); i++) {
211  if (popE[i](fragmentindex) > _loc_threshold &&
212  popH[i](fragmentindex) > _loc_threshold) {
213  indexes.push_back(i);
214  }
215  }
216  return indexes;
217 }

219 std::vector<int> Statefilter::DeltaQFilter(const Orbitals& orbitals) const {
220  std::vector<int> indexes;
221  const std::vector<Eigen::VectorXd>& dQ_frag =
222  (_statehist[0].Type() == QMStateType::Singlet)
223  ? orbitals.getFragmentChargesSingEXC()
224  : orbitals.getFragmentChargesTripEXC();
225  for (unsigned i = 0; i < dQ_frag.size(); i++) {
226  if (std::abs(dQ_frag[i](0)) > _dQ_threshold) {
227  indexes.push_back(i);
228  }
229  }
230  return indexes;

233 Eigen::VectorXd Statefilter::CalculateOverlap(Orbitals& orbitals) const {
234  BasisSet dftbs;
235  dftbs.LoadBasisSet(orbitals.getDFTbasisName());
236  AOBasis dftbasis;
237  dftbasis.AOBasisFill(dftbs, orbitals.QMAtoms());
238  AOOverlap dftoverlap;
239  dftoverlap.Fill(dftbasis);
240  _S_onehalf = dftoverlap.Pseudo_InvSqrt(1e-8);
241  Eigen::MatrixXd ortho_coeffs = CalcOrthoCoeffs(orbitals);
242  Eigen::VectorXd overlap = (ortho_coeffs * _laststatecoeff).cwiseAbs2();
243  return overlap;
244 }

246 Eigen::MatrixXd Statefilter::CalcOrthoCoeffs(Orbitals& orbitals) const {
247  QMStateType type = _statehist[0].Type();
248  Eigen::MatrixXd ortho_coeffs;
249  if (type.isSingleParticleState()) {
250  Eigen::MatrixXd coeffs;
251  if (type == QMStateType::DQPstate) {
252  coeffs = orbitals.CalculateQParticleAORepresentation();
253  } else {
254  coeffs = orbitals.MOCoefficients();
255  }
256  ortho_coeffs = _S_onehalf * coeffs;
257  } else {
258  throw std::runtime_error("Overlap for excitons not implemented yet");
259  }
260  return ortho_coeffs;
261 }

263 void Statefilter::UpdateLastCoeff(Orbitals& orbitals) {
264  Eigen::MatrixXd ortho_coeffs = CalcOrthoCoeffs(orbitals);
265  int offset = 0;
266  if (_statehist[0].Type() == QMStateType::DQPstate) {
267  offset = orbitals.getGWAmin();
268  }
269  _laststatecoeff = ortho_coeffs.col(_statehist.back().Index() - offset);
270 }

272 std::vector<int> Statefilter::OverlapFilter(Orbitals& orbitals) const {
273  std::vector<int> indexes;
274  if (_statehist.size() <= 1) {
275  indexes = std::vector<int>{_statehist[0].Index()};
276  return indexes;
277  }
278 
279  Eigen::VectorXd Overlap = CalculateOverlap(orbitals);
280  int validelements = Overlap.size();
281  for (int i = 0; i < Overlap.size(); i++) {
282  if (Overlap(i) < _overlapthreshold) {
283  validelements--;
284  }
285  }
286 
287  std::vector<int> index = std::vector<int>(Overlap.size());
288  std::iota(index.begin(), index.end(), 0);
289  std::stable_sort(index.begin(), index.end(), [&Overlap](int i1, int i2) {
290  return Overlap[i1] > Overlap[i2];
291  });
292 
293  int offset = 0;
294  if (_statehist[0].Type() == QMStateType::DQPstate) {
295  offset = orbitals.getGWAmin();
296  }
297 
298  for (int i : index) {
299  if (int(indexes.size()) == validelements) {
300  break;
301  }
302  indexes.push_back(i + offset);
303  }
304  return indexes;
305 }
