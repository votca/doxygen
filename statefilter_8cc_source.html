27  void Statefilter::Initialize(tools::Property& options){
28  if (options.exists("oscillator_strength")) {
29  _use_oscfilter = true;
30  _oscthreshold = options.ifExistsReturnElseThrowRuntimeError<double>("oscillator_strength");
31  }
32  if (options.exists("overlap")) {
33  _use_overlapfilter = true;
34  _overlapthreshold = options.ifExistsReturnElseReturnDefault<double>("overlap",0.0);
35  }
36  if (options.exists("localisation")) {
37  _use_localisationfilter = true;
38  std::string temp = options.get("localisation").as<std::string> ();
39  tools::Tokenizer tok_cleanup(temp, ", \n\t");
40  std::vector <std::string> strings_vec;
41  tok_cleanup.ToVector(strings_vec);
42  if (strings_vec.size() != 2) {
43  throw std::runtime_error("statefiler: Fragment and localisation threshold are not separated");
44  }
45  if (strings_vec[0] == "a" || strings_vec[0] == "A") {
46  _localiseonA = true;
47  } else if (strings_vec[0] == "b" || strings_vec[0] == "B") {
48  _localiseonA = false;
49  } else {
50  throw std::runtime_error("statefiler: Fragment label not known, either A or B");
51  }
52  _loc_threshold = boost::lexical_cast<double>(strings_vec[1]);
53  }
54 
55  if (options.exists("charge_transfer")) {
56  _use_dQfilter = true;
57  _dQ_threshold = options.ifExistsReturnElseThrowRuntimeError<double>("charge_transfer");
58  }
59  if (_use_dQfilter && _use_localisationfilter) {
60  throw std::runtime_error("Cannot use localisation and charge_transfer filter at the same time.");
61  }
62 
63  }

79  }

81  std::string fragment="A";
82  if(!_localiseonA){
83  fragment="B";

86  }
87  if(_use_dQfilter){
88  CTP_LOG(ctp::logDEBUG, *_log) << "Using Delta Q filter with cutoff "<<_dQ_threshold << flush;
89  }
90  if (_use_oscfilter && _use_dQfilter) {
91  CTP_LOG(ctp::logDEBUG, *_log) << "WARNING: filtering for optically active CT transition - might not make sense... " << flush;
92  }

100  std::vector<int> Statefilter::ComparePairofVectors(std::vector<int> & vec1,std::vector<int> & vec2)const{
101  std::vector<int> result(std::min(vec1,vec2));
102  std::vector<int>::iterator it;
103  std::sort (vec1.begin(), vec1.end());
104  std::sort (vec2.begin(), vec2.end());
105  it=std::set_intersection (vec1.begin(), vec1.end(), vec2.begin(), vec2.end(), result.begin());
106  result.resize(it-result.begin());
107  return result;

122  QMState Statefilter::CalcState(Orbitals& orbitals)const{
123 
124  if(_use_dQfilter+_use_oscfilter+_use_localisationfilter+_use_oscfilter<1){
125  return _statehist[0];
126  }
127 
128  std::vector< std::vector<int> > results;
129  if(_use_oscfilter){
130  results.push_back(OscFilter(orbitals));
131  }
132  if(_use_localisationfilter){
133  results.push_back(LocFilter(orbitals));
134  }
135  if(_use_overlapfilter){
136  results.push_back(OverlapFilter(orbitals));
137  }
138  if(_use_dQfilter){
139  results.push_back(DeltaQFilter(orbitals));
140  }
141 
142  std::vector<int> result=CollapseResults(results);
143  QMState state;
144  if(result.size()<1){
145  state=_statehist.back();
146  CTP_LOG(ctp::logDEBUG, *_log) << "No State found by filter using last state: "<<state.ToString()<< flush;
147  }else{
148  state=QMState(_statehist.back().Type(),result[0],false);
149  CTP_LOG(ctp::logDEBUG, *_log) << "Next State is: "<<state.ToString()<< flush;
150  }
151  return state;
152  }

174  std::vector<int> Statefilter::LocFilter(const Orbitals& orbitals)const{
175  std::vector<int> indexes;
176  const std::vector< Eigen::VectorXd >& popE = (_statehist[0].Type()== QMStateType::Singlet)
177  ? orbitals.getFragment_E_localisation_singlet() : orbitals.getFragment_E_localisation_triplet();
178  const std::vector< Eigen::VectorXd >& popH = (_statehist[0].Type()== QMStateType::Singlet)
179  ? orbitals.getFragment_H_localisation_singlet() : orbitals.getFragment_H_localisation_triplet();
180  int fragmentindex=1;
181  if (_localiseonA) {
182  fragmentindex=0;
183  }
184  for (unsigned i = 0; i < popE.size(); i++) {
185  if (popE[i](fragmentindex) > _loc_threshold && popH[i](fragmentindex) > _loc_threshold) {
186  indexes.push_back(i);
187  }
188  }
189  return indexes;
190  }

205  Eigen::VectorXd Statefilter::CalculateOverlap(Orbitals & orbitals)const{
206  BasisSet dftbs;
207  dftbs.LoadBasisSet(orbitals.getDFTbasisName());
208  AOBasis dftbasis;
209  dftbasis.AOBasisFill(dftbs, orbitals.QMAtoms());
210  AOOverlap dftoverlap;
211  dftoverlap.Fill(dftbasis);
212  _S_onehalf=dftoverlap.Pseudo_InvSqrt(1e-8);
213  Eigen::MatrixXd ortho_coeffs=CalcOrthoCoeffs(orbitals);
214  Eigen::VectorXd overlap=(ortho_coeffs*_laststatecoeff).cwiseAbs2();
215  return overlap;
216  }

218  Eigen::MatrixXd Statefilter::CalcOrthoCoeffs(Orbitals& orbitals)const{
219  QMStateType type=_statehist[0].Type();
220  Eigen::MatrixXd ortho_coeffs;
221  if(type.isSingleParticleState()){
222  Eigen::MatrixXd coeffs;
223  if(type==QMStateType::DQPstate){
224  coeffs=orbitals.CalculateQParticleAORepresentation();
225  }else{
226  coeffs=orbitals.MOCoefficients();
227  }
228  ortho_coeffs=_S_onehalf*coeffs;
229  }else{
230  throw std::runtime_error("Overlap for excitons not implemented yet");
231  }
232  return ortho_coeffs;
233  }

235  void Statefilter::UpdateLastCoeff(Orbitals& orbitals){
236  Eigen::MatrixXd ortho_coeffs=CalcOrthoCoeffs(orbitals);
237  int offset=0;
238  if(_statehist[0].Type()==QMStateType::DQPstate){
239  offset=orbitals.getGWAmin();
240  }
241  _laststatecoeff=ortho_coeffs.col(_statehist.back().Index()-offset);

244  std::vector<int> Statefilter::OverlapFilter(Orbitals & orbitals)const{
245  std::vector<int> indexes;
246  if (_statehist.size() <= 1) {
247  indexes=std::vector<int>{_statehist[0].Index()};
248  return indexes;
249  }
250 
251  Eigen::VectorXd Overlap=CalculateOverlap(orbitals);
252  int validelements=Overlap.size();
253  for(int i=0;i<Overlap.size();i++){
254  if(Overlap(i)<_overlapthreshold){
255  validelements--;
256  }
257  }
258 
259  std::vector<int>index = std::vector<int>(Overlap.size());
260  std::iota(index.begin(), index.end(), 0);
261  std::stable_sort(index.begin(), index.end(), [&Overlap](int i1, int i2) {
262  return Overlap[i1] > Overlap[i2];
263  });
264 
265  int offset=0;
266  if(_statehist[0].Type()==QMStateType::DQPstate){
267  offset=orbitals.getGWAmin();
268  }
269 
270  for (int i:index){
271  if(int(indexes.size())==validelements){
272  break;
273  }
274  indexes.push_back(i+offset);
275  }
276  return indexes;
277  }
