00062     if (beads.size() != weights.size())
00063         throw runtime_error(string("number of subbeads in " +
00064                 opts_bead->get("name").as<string>()
00065                 + " and number of weights in map "
00066                 + opts_map->get("name").as<string>() + " do not match"));

00069     double norm = 1./ std::accumulate(weights.begin(), weights.end(), 0.);
00070     
00071     transform(weights.begin(), weights.end(), weights.begin(), bind2nd(multiplies<double>(), norm));
00072     // get the d vector if exists or initialize same as weights
00073     vector<double> d;

00075         Tokenizer tok_weights(_opts_map->get("d").value(), " \n\t");

00078         norm = 1./std::accumulate(d.begin(), d.end(), 0.);
00079         transform(d.begin(), d.end(), d.begin(), bind2nd(multiplies<double>(), norm));
00080     } else {
00081         // initialize force-weights with weights
00082         d.resize(weights.size());
00083         copy(weights.begin(), weights.end(), d.begin());
00084     }
00085 
00086     // check weather number of d coeffs is correct
00087     if (beads.size() != d.size()) {
00088         throw runtime_error(string("number of subbeads in " +
00089             opts_bead->get("name").as<string>()
00090             + " and number of d-coefficients in map "
00091             + opts_map->get("name").as<string>() + " do not match"));
00092     }
00093 
00094     fweights.resize(weights.size());
00095     // calculate force weights by d_i/w_i
00096     for(size_t i=0; i<weights.size(); ++i) {
00097         if(weights[i] == 0 && d[i]!=0) {
00098             throw runtime_error(
00099                 "A d coefficient is nonzero while weights is zero in mapping "
00100                 + opts_map->get("name").as<string>());
00101         }
00102         if(weights[i] != 0)
00103             fweights[i] = d[i] / weights[i];
00104         else
00105             fweights[i] = 0;        
00106     }
00107 
00108     for (size_t i = 0; i < beads.size(); ++i) {
00109         int iin = in->getBeadByName(beads[i]);
00110         if (iin < 0)
00111             throw std::runtime_error(string("mapping error: molecule " + beads[i] + " does not exist"));
00112         AddElem(in->getBead(iin), weights[i], fweights[i]);
00113     }
