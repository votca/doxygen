59 void Map_Sphere::Initialize(Molecule *in, Bead *out, Property *opts_bead,
60  Property *opts_map) {
61  BeadMap::Initialize(in, out, opts_bead, opts_map);
62 
63  vector<string> beads;
64  vector<double> weights;
65  vector<double> fweights;
66 
67  // get the beads
68  string s(_opts_bead->get("beads").value());
69  Tokenizer tok_beads(s, " \n\t");
70  tok_beads.ToVector(beads);
71 
72  // get vector of weights
73  Tokenizer tok_weights(_opts_map->get("weights").value(), " \n\t");
74  tok_weights.ConvertToVector<double>(weights);
75 
76  // check weather weights and # beads matches
77  if (beads.size() != weights.size()) {
78  throw runtime_error(
79  string("number of subbeads in " + opts_bead->get("name").as<string>() +
80  " and number of weights in map " +
81  opts_map->get("name").as<string>() + " do not match"));
82  }
83 
84  // normalize the weights
85  double norm = 1. / std::accumulate(weights.begin(), weights.end(), 0.);
86 
87  transform(weights.begin(), weights.end(), weights.begin(),
88  bind2nd(multiplies<double>(), norm));
89  // get the d vector if exists or initialize same as weights
90  vector<double> d;
91  if (_opts_map->exists("d")) {
92  Tokenizer tok_weights2(_opts_map->get("d").value(), " \n\t");
93  tok_weights2.ConvertToVector(d);
94  // normalize d coefficients
95  norm = 1. / std::accumulate(d.begin(), d.end(), 0.);
96  transform(d.begin(), d.end(), d.begin(),
97  bind2nd(multiplies<double>(), norm));
98  } else {
99  // initialize force-weights with weights
100  d.resize(weights.size());
101  copy(weights.begin(), weights.end(), d.begin());
102  }
103 
104  // check weather number of d coeffs is correct
105  if (beads.size() != d.size()) {
106  throw runtime_error(
107  string("number of subbeads in " + opts_bead->get("name").as<string>() +
108  " and number of d-coefficients in map " +
109  opts_map->get("name").as<string>() + " do not match"));
110  }
111 
112  fweights.resize(weights.size());
113  // calculate force weights by d_i/w_i
114  for (size_t i = 0; i < weights.size(); ++i) {
115  if (weights[i] == 0 && d[i] != 0) {
116  throw runtime_error(
117  "A d coefficient is nonzero while weights is zero in mapping " +
118  opts_map->get("name").as<string>());
119  }
120  if (weights[i] != 0) {
121  fweights[i] = d[i] / weights[i];
122  } else {
123  fweights[i] = 0;
124  }
125  }
126 
127  for (size_t i = 0; i < beads.size(); ++i) {
128  Index iin = in->getBeadByName(beads[i]);
129  if (iin < 0) {
130  throw std::runtime_error(
131  string("mapping error: molecule " + beads[i] + " does not exist"));
132  }
133  AddElem(in->getBead(iin), weights[i], fweights[i]);
134  }
135 }

137 void Map_Sphere::Apply(const BoundaryCondition &bc) {
138 
139  assert(_matrix.size() > 0 && "Cannot map to sphere there are no beads");
140 
141  bool bPos, bVel, bF;
142  bPos = bVel = bF = false;
143  _out->ClearParentBeads();
144 
145  // the following is needed for pbc treatment
146  Eigen::Vector3d r0 = Eigen::Vector3d::Zero();
147  string name0;
148  Index id0 = 0;
149  if (_matrix.size() > 0) {
150  if (_matrix.front()._in->HasPos()) {
151  r0 = _matrix.front()._in->getPos();
152  name0 = _matrix.front()._in->getName();
153  id0 = _matrix.front()._in->getId();
154  }
155  }
156 
157  double M = 0;
158  Eigen::Vector3d cg = Eigen::Vector3d::Zero();
159  Eigen::Vector3d f = Eigen::Vector3d::Zero();
160  Eigen::Vector3d vel = Eigen::Vector3d::Zero();
161 
162  Bead *bead_max_dist = _matrix.at(0)._in;
163  double max_bead_dist =
164  bc.BCShortestConnection(r0, bead_max_dist->getPos()).norm();
165 
166  for (auto &iter : _matrix) {
167  Bead *bead = iter._in;
168  _out->AddParentBead(bead->getId());
169  M += bead->getMass();
170  if (bead->HasPos()) {
171  Eigen::Vector3d r = bc.BCShortestConnection(r0, bead->getPos());
172  if (r.norm() > max_bead_dist) {
173  max_bead_dist = r.norm();
174  bead_max_dist = bead;
175  }
176  cg += iter._weight * (r + r0);
177  bPos = true;
178  }
179  }
180 
183  if (bc.getBoxType() != BoundaryCondition::eBoxtype::typeOpen) {
184  double max_dist = 0.5 * bc.getShortestBoxDimension();
185  if (max_bead_dist > max_dist) {
186  cout << r0 << " " << bead_max_dist->getPos() << endl;
187  throw std::runtime_error(
188  "coarse-grained bead is bigger than half the box \n "
189  "(atoms " +
190  name0 + " (id " + boost::lexical_cast<string>(id0 + 1) + ")" + ", " +
191  bead_max_dist->getName() + " (id " +
192  boost::lexical_cast<string>(bead_max_dist->getId() + 1) + ")" +
193  +" , molecule " +
194  boost::lexical_cast<string>(bead_max_dist->getMoleculeId() + 1) +
195  ")");
196  }
197  }
198 
199  for (auto &iter : _matrix) {
200  Bead *bead = iter._in;
201  if (bead->HasVel()) {
202  vel += iter._weight * bead->getVel();
203  bVel = true;
204  }
205  if (bead->HasF()) {
206  f += iter._force_weight * bead->getF();
207  bF = true;
208  }
209  }
210  _out->setMass(M);
211  if (bPos) {
212  _out->setPos(cg);
213  }
214  if (bVel) {
215  _out->setVel(vel);
216  }
217  if (bF) {
218  _out->setF(f);
219  }
220 }

223 void Map_Ellipsoid::Apply(const BoundaryCondition &bc) {
224 
225  assert(_matrix.size() > 0 && "Cannot map to ellipsoid there are no beads");
226 
227  bool bPos, bVel, bF;
228  bPos = bVel = bF = false;
229 
230  // the following is needed for pbc treatment
231  Eigen::Vector3d r0 = Eigen::Vector3d::Zero();
232  string name0;
233  Index id0 = 0;
234  if (_matrix.size() > 0) {
235  if (_matrix.front()._in->HasPos()) {
236  r0 = _matrix.front()._in->getPos();
237  name0 = _matrix.front()._in->getName();
238  id0 = _matrix.front()._in->getId();
239  }
240  }
241  Eigen::Vector3d cg = Eigen::Vector3d::Zero();
242  Eigen::Vector3d c = Eigen::Vector3d::Zero();
243  Eigen::Vector3d f = Eigen::Vector3d::Zero();
244  Eigen::Vector3d vel = Eigen::Vector3d::Zero();
245 
246  Index n;
247  n = 0;
248  _out->ClearParentBeads();
249 
250  Bead *bead_max_dist = _matrix.at(0)._in;
251  double max_bead_dist =
252  bc.BCShortestConnection(r0, bead_max_dist->getPos()).norm();
253 
254  for (auto &iter : _matrix) {
255  Bead *bead = iter._in;
256  _out->AddParentBead(bead->getId());
257  if (bead->HasPos()) {
258  Eigen::Vector3d r = bc.BCShortestConnection(r0, bead->getPos());
259  if (r.norm() > max_bead_dist) {
260  max_bead_dist = r.norm();
261  bead_max_dist = bead;
262  }
263  cg += iter._weight * (r + r0);
264  bPos = true;
265  }
266  }
267 
270  if (bc.getBoxType() != BoundaryCondition::eBoxtype::typeOpen) {
271  double max_dist = 0.5 * bc.getShortestBoxDimension();
272  if (max_bead_dist > max_dist) {
273  cout << r0 << " " << bead_max_dist->getPos() << endl;
274  throw std::runtime_error(
275  "coarse-grained bead is bigger than half the box \n "
276  "(atoms " +
277  name0 + " (id " + boost::lexical_cast<string>(id0 + 1) + ")" + ", " +
278  bead_max_dist->getName() + " (id " +
279  boost::lexical_cast<string>(bead_max_dist->getId() + 1) + ")" +
280  +" , molecule " +
281  boost::lexical_cast<string>(bead_max_dist->getMoleculeId() + 1) +
282  ")");
283  }
284  }
285 
286  for (auto &iter : _matrix) {
287  Bead *bead = iter._in;
288  if (bead->HasVel() == true) {
289  vel += iter._weight * bead->getVel();
290  bVel = true;
291  }
292  if (bead->HasF()) {
295  // f += (*iter)._weight * _in->getBeadF((*iter)._in);
296  f += iter._force_weight * bead->getF();
297  bF = true;
298  }
299 
300  if (iter._weight > 0 && bead->HasPos()) {
301  c += bead->getPos();
302  n++;
303  }
304  }
305 
306  if (bPos) {
307  _out->setPos(cg);
308  }
309  if (bVel) {
310  _out->setVel(vel);
311  }
312  if (bF) {
313  _out->setF(f);
314  }
315 
316  if (!_matrix[0]._in->HasPos()) {
317  _out->setU(Eigen::Vector3d::UnitX());
318  _out->setV(Eigen::Vector3d::UnitY());
319  _out->setW(Eigen::Vector3d::UnitZ());
320  return;
321  }
322 
323  Eigen::Matrix3d m = Eigen::Matrix3d::Zero();
324  // calculate the tensor of gyration
325  c = c / (double)n;
326  for (auto &iter : _matrix) {
327  if (iter._weight == 0) {
328  continue;
329  }
330  Bead *bead = iter._in;
331  Eigen::Vector3d v = bead->getPos() - c;
332  // v = vec(1, 0.5, 0) * 0.*(drand48()-0.5)
333  // + vec(0.5, -1, 0) * (drand48()-0.5)
334  // + vec(0, 0, 1) * (drand48()-0.5);
335 
336  // Normalize the tensor with 1/number_of_atoms_per_bead
337  m += v * v.transpose() / (double)_matrix.size();
338  }
339 
340  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> eig;
341  eig.computeDirect(m);
342 
343  Eigen::Vector3d u = eig.eigenvectors().col(0);
344  Eigen::Vector3d v = _matrix[1]._in->getPos() - _matrix[0]._in->getPos();
345  v.normalize();
346 
347  _out->setV(v);
348 
349  Eigen::Vector3d w = _matrix[2]._in->getPos() - _matrix[0]._in->getPos();
350  w.normalize();
351 
352  if (v.cross(w).dot(u) < 0) {
353  u = -u;
354  }
355  _out->setU(u);
356 
357  // write out w
358  w = u.cross(v);
359  w.normalize();
360  _out->setW(w);
361 }
