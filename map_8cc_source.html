45 void Map_Sphere::Initialize(Molecule *in, Bead *out, Property *opts_bead, Property *opts_map) {
46  BeadMap::Initialize(in, out, opts_bead, opts_map);
47 
48  vector<string> beads;
49  vector<double> weights;
50  vector<double> fweights;
51 
52  // get the beads
53  string s(_opts_bead->get("beads").value());
54  Tokenizer tok_beads(s, " \n\t");
55  tok_beads.ToVector(beads);
56 
57  // get vector of weights
58  Tokenizer tok_weights(_opts_map->get("weights").value(), " \n\t");
59  tok_weights.ConvertToVector<double>(weights);
60 
61  // check weather weights and # beads matches
62  if (beads.size() != weights.size())
63  throw runtime_error(string("number of subbeads in " +
64  opts_bead->get("name").as<string>()
65  + " and number of weights in map "
66  + opts_map->get("name").as<string>() + " do not match"));
67 
68  // normalize the weights
69  double norm = 1./ std::accumulate(weights.begin(), weights.end(), 0.);
70 
71  transform(weights.begin(), weights.end(), weights.begin(), bind2nd(multiplies<double>(), norm));
72  // get the d vector if exists or initialize same as weights
73  vector<double> d;
74  if(_opts_map->exists("d")) {
75  Tokenizer tok_weights(_opts_map->get("d").value(), " \n\t");
76  tok_weights.ConvertToVector(d);
77  // normalize d coefficients
78  norm = 1./std::accumulate(d.begin(), d.end(), 0.);
79  transform(d.begin(), d.end(), d.begin(), bind2nd(multiplies<double>(), norm));
80  } else {
81  // initialize force-weights with weights
82  d.resize(weights.size());
83  copy(weights.begin(), weights.end(), d.begin());
84  }
85 
86  // check weather number of d coeffs is correct
87  if (beads.size() != d.size()) {
88  throw runtime_error(string("number of subbeads in " +
89  opts_bead->get("name").as<string>()
90  + " and number of d-coefficients in map "
91  + opts_map->get("name").as<string>() + " do not match"));
92  }
93 
94  fweights.resize(weights.size());
95  // calculate force weights by d_i/w_i
96  for(size_t i=0; i<weights.size(); ++i) {
97  if(weights[i] == 0 && d[i]!=0) {
98  throw runtime_error(
99  "A d coefficient is nonzero while weights is zero in mapping "
100  + opts_map->get("name").as<string>());
101  }
102  if(weights[i] != 0)
103  fweights[i] = d[i] / weights[i];
104  else
105  fweights[i] = 0;
106  }
107 
108  for (size_t i = 0; i < beads.size(); ++i) {
109  int iin = in->getBeadByName(beads[i]);
110  if (iin < 0)
111  throw std::runtime_error(string("mapping error: molecule " + beads[i] + " does not exist"));
112  AddElem(in->getBead(iin), weights[i], fweights[i]);
113  }
114 }

116 void Map_Sphere::Apply()
117 {
118  vector<element_t>::iterator iter;
119  vec cg(0., 0., 0.), f(0.,0.,0.), vel(0.,0.,0.);
120  bool bPos, bVel, bF;
121  bPos=bVel=bF=false;
122  _out->ParentBeads().clear();
123 
124  // the following is needed for pbc treatment
125  Topology *top = _out->getParent();
126  double max_dist = 0.5*top->ShortestBoxSize();
127  vec r0 = vec(0,0,0);
128  string name0;
129  int id0=0;
130  if(_matrix.size() > 0) {
131  if(_matrix.front()._in->HasPos()) {
132  r0=_matrix.front()._in->getPos();
133  name0 = _matrix.front()._in->getName();
134  id0 = _matrix.front()._in->getId();
135  }
136  }
137 
138  double M = 0;
139 
140  for(iter = _matrix.begin(); iter != _matrix.end(); ++iter) {
141  Bead *bead = iter->_in;
142  _out->ParentBeads().push_back(bead->getId());
143  M+=bead->getM();
144  if(bead->HasPos()) {
145  vec r = top->BCShortestConnection(r0, bead->getPos());
146  if(abs(r) > max_dist) {
147  cout << r0 << " " << bead->getPos() << endl;
148  throw std::runtime_error("coarse-grained bead is bigger than half the box \n (atoms "
149  + name0 + " (id " + boost::lexical_cast<string>(id0+1) + ")" + ", " + bead->getName() + " (id " + boost::lexical_cast<string>(bead->getId()+1) + ")" + +" , molecule "
150  + boost::lexical_cast<string>(bead->getMolecule()->getId()+1) + ")" );
151  }
152  cg += (*iter)._weight * (r+r0);
153  bPos=true;
154  }
155  if(bead->HasVel()) {
156  vel += (*iter)._weight * bead->getVel();
157  bVel = true;
158  }
159  if(bead->HasF()) {
160  f += (*iter)._force_weight * bead->getF();
161  bF = true;
162  }
163  }
164  _out->setM(M);
165  if(bPos)
166  _out->setPos(cg);
167  if(bVel)
168  _out->setVel(vel);
169  if(bF)
170  _out->setF(f);
171 }

174 void Map_Ellipsoid::Apply()
175 {
176  vector<element_t>::iterator iter;
177  vec cg(0., 0., 0.), c(0., 0., 0.), f(0.,0.,0.), vel(0.,0.,0.);
178  matrix m(0.);
179  bool bPos, bVel, bF;
180  bPos=bVel=bF=false;
181 
182  // the following is needed for pbc treatment
183  Topology *top = _out->getParent();
184  double max_dist = 0.5*top->ShortestBoxSize();
185  vec r0 = vec(0,0,0);
186  if(_matrix.size() > 0) {
187  if(_matrix.front()._in->HasPos()) {
188  r0=_matrix.front()._in->getPos();
189  }
190  }
191 
192  int n;
193  n = 0;
194  _out->ParentBeads().clear();
195  for(iter = _matrix.begin(); iter != _matrix.end(); ++iter) {
196  Bead *bead = iter->_in;
197  _out->ParentBeads().push_back(bead->getId());
198  if(bead->HasPos()) {
199  vec r = top->BCShortestConnection(r0, bead->getPos());
200  if(abs(r) > max_dist)
201  throw std::runtime_error("coarse-grained bead is bigger than half the box");
202  cg += (*iter)._weight * (r+r0);
203  bPos=true;
204  }
205  if(bead->HasVel() == true) {
206  vel += (*iter)._weight * bead->getVel();
207  bVel = true;
208  }
209  if(bead->HasF()) {
211  //f += (*iter)._weight * _in->getBeadF((*iter)._in);
212  f += (*iter)._force_weight * bead->getF();
213  bF = true;
214  }
215 
216  if((*iter)._weight>0 && bead->HasPos()) {
217  c += bead->getPos();
218  n++;
219  }
220  }
221 
222  if(bPos)
223  _out->setPos(cg);
224  if(bVel)
225  _out->setVel(vel);
226  if(bF)
227  _out->setF(f);
228 
229  if(!_matrix[0]._in->HasPos()) {
230  _out->setU(vec(1.0,0,0));
231  _out->setV(vec(.0,1,0));
232  _out->setW(vec(.0,0,1));
233  return;
234  }
235 
236  // calculate the tensor of gyration
237  c=c/(double)n;
238  for(iter = _matrix.begin(); iter != _matrix.end(); ++iter) {
239  if((*iter)._weight == 0) continue;
240  Bead *bead = iter->_in;
241  vec v = bead->getPos() - c;
242  //v = vec(1, 0.5, 0) * 0.*(drand48()-0.5)
243  // + vec(0.5, -1, 0) * (drand48()-0.5)
244  // + vec(0, 0, 1) * (drand48()-0.5);
245 
246  //Normalize the tensor with 1/number_of_atoms_per_bead
247  m[0][0] += v.getX()*v.getX()/(double)_matrix.size();
248  m[0][1] += v.getX()*v.getY()/(double)_matrix.size();
249  m[0][2] += v.getX()*v.getZ()/(double)_matrix.size();
250  m[1][1] += v.getY()*v.getY()/(double)_matrix.size();
251  m[1][2] += v.getY()*v.getZ()/(double)_matrix.size();
252  m[2][2] += v.getZ()*v.getZ()/(double)_matrix.size();
253 
254  }
255  m[1][0] = m[0][1];
256  m[2][0] = m[0][2];
257  m[2][1] = m[1][2];
258 
259  // calculate the eigenvectors
260  matrix::eigensystem_t es;
261  m.SolveEigensystem(es);
262 
263  //vec eigenv1=es.eigenvecs[0];
264  //vec eigenv2=es.eigenvecs[1];
265  //vec eigenv3=es.eigenvecs[2];
266 
267 /* _out->seteigenvec1(eigenv1);
268  _out->seteigenvec2(eigenv2);
269  _out->seteigenvec3(eigenv3);
270  */
271 
272  vec u = es.eigenvecs[0];
273  vec v = _matrix[1]._in->getPos() - _matrix[0]._in->getPos();
274  v.normalize();
275 
276  _out->setV(v);
277 
278  vec w = _matrix[2]._in->getPos() - _matrix[0]._in->getPos();
279  w.normalize();
280 
281  if((v^w)*u < 0) u=vec(0.,0.,0.)-u;
282  _out->setU(u);
283 
284  //write out w
285  w=u^v;
286  w.normalize();
287  _out->setW(w);
288

290

292 }
