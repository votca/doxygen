51 void Md2QmEngine::Initialize(const string &xmlfile) {
52 
53  Property typology;
54  load_property_from_xml(typology, xmlfile.c_str());
55 
56  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
57  // XML to Types: Molecules => Segments => Fragments => Atoms //
58  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
59 
60  string key = "topology.molecules.molecule";
61  list<Property *> molecules = typology.Select(key);
62  list<Property *>::iterator it_molecule;
63  int molecule_id = 1;
64  int qmunit_id = 1; // Counts segment types
65 
66  for ( it_molecule = molecules.begin();
67  it_molecule != molecules.end();
68  ++it_molecule ) {
69 
70  XTP::Molecule *molecule = AddMoleculeType(molecule_id++, *it_molecule);
71  string molMdName = (*it_molecule)->get("mdname").as<string>();
72 
73  // +++++++++++++ //
74  // Load segments //
75  // +++++++++++++ //
76 
77  key = "segments.segment";
78  list<Property *> segments = (*it_molecule)->Select(key);
79  list<Property *>::iterator it_segment;
80  int segment_id = 1;
81  int md_atom_id = 1; // <- atom id count with respect to molecule
82 
83  for ( it_segment = segments.begin();
84  it_segment != segments.end();
85  ++it_segment ) {
86 
87  // Create new segment + associated type (QM Unit)
88  XTP::Segment *segment = AddSegmentType(segment_id++, *it_segment );
89  XTP::SegmentType *qmUnit = AddQMUnit(qmunit_id, *it_segment );
90  ++qmunit_id;
91 
92  segment->setType(qmUnit);
93  molecule->AddSegment(segment);
94 
95  // Load internal (i.e. QM-) coord.s and MOO-related properties
96  string qmcoordsFile = "";
97  map<int, pair<string, vec> > intCoords;
98  if ( (*it_segment)->exists("qmcoords") ) {
99  qmcoordsFile = (*it_segment)->get("qmcoords").as<string>();
100  // QM ID Element Position
101  this->getIntCoords(qmcoordsFile, intCoords);
102  }
103 
104  // ++++++++++++++ //
105  // Load fragments //
106  // ++++++++++++++ //
107 
108  map<string,bool> fragname_isTaken;
109 
110  key = "fragments.fragment";
111  list<Property *> fragments = (*it_segment)->Select(key);
112  list<Property *>::iterator it_fragment;
113  int fragment_id = 1;
114 
115  for ( it_fragment = fragments.begin();
116  it_fragment != fragments.end();
117  ++it_fragment ) {
118 
119  // Create new fragment
120  XTP::Fragment* fragment=AddFragmentType(fragment_id++,*it_fragment);
121  segment->AddFragment( fragment );
122 
123  // Verify that this fragment name is not taken already
124  try {
125  // This should throw
126  bool taken = fragname_isTaken.at(fragment->getName());
127  cout << "ERROR Fragment name '" << fragment->getName()
128  << "' in segment '" << segment->getName()
129  << "' occurs more than once." << endl;
130  if (taken)
131  throw runtime_error("(see above, naming collision)");
132  }
133  catch (out_of_range) {
134  fragname_isTaken[fragment->getName()] = true;
135  }
136 
137  // Load local-frame definition
138  vector<int> trihedron = (*it_fragment)->get("localframe")
139  .as< vector<int> >();
140  while (trihedron.size() < 3) {
141  trihedron.push_back(-1);
142  }
143  fragment->setTrihedron(trihedron);
144 
145 
146  // ++++++++++ //
147  // Load atoms //
148  // ++++++++++ //
149 
150  string mdatoms = (*it_fragment)->get("mdatoms").as<string>();
151  string qmatoms = (*it_fragment)->get("qmatoms").as<string>();
152  string weights = (*it_fragment)->get("weights").as<string>();
153 
154  Tokenizer tok_md_atoms(mdatoms, " \t\n");
155  Tokenizer tok_qm_atoms(qmatoms, " \t\n");
156  Tokenizer tok_weights(weights, " \t\n");
157 
158  vector <string> md_atoms_info;
159  vector <string> qm_atoms_info;
160  vector <string> atom_weights;;
161 
162  tok_md_atoms.ToVector(md_atoms_info);
163  tok_qm_atoms.ToVector(qm_atoms_info);
164  tok_weights.ToVector(atom_weights);
165 
166  //assert(md_atoms_info.size() == qm_atoms_info.size());
167  //assert(md_atoms_info.size() == atom_weights.size());
168 
169  if ( (md_atoms_info.size() != qm_atoms_info.size()) ||
170  (md_atoms_info.size() != atom_weights.size() ) ) {
171  cout << "ERROR: "
172  << "Could not allocate MD atoms to QM atoms or weights"
173  << " in fragment " << fragment->getName()
174  << " in segment " << segment->getName()
175  << " in molecule " << molMdName
176  << " due to inconsistent number of columns"
177  << " (MD: " << md_atoms_info.size() << ","
178  << " QM: " << qm_atoms_info.size() << ")"
179  << " Weights: " << atom_weights.size() << ")."
180  << endl;
181  cout << "NOTE: "
182  << "To define an MD atom without QM counterpart, insert "
183  << "a single ':' in the associated QM-atoms column and "
184  << "specify a mapping weight of 0."
185  << endl;
186  throw runtime_error( "Inconsistency in mapping file." );
187  }
188 
189  vector<string> ::iterator it_md_atom_name;
190  vector<string> ::iterator it_qm_atom_name;
191  vector<string> ::iterator it_atom_weight;
192 
193  for ( it_md_atom_name = md_atoms_info.begin(),
194  it_qm_atom_name = qm_atoms_info.begin(),
195  it_atom_weight = atom_weights.begin();
196  it_md_atom_name != md_atoms_info.end();
197  ++it_md_atom_name,
198  ++it_qm_atom_name,
199  ++ it_atom_weight) {
200 
201  // ++++++++++++++++++ //
202  // Create single atom //
203  // ++++++++++++++++++ //
204 
205  Tokenizer tok_md((*it_md_atom_name), ":");
206  Tokenizer tok_qm((*it_qm_atom_name), ":");
207 
208  vector<string> md_atom_specs;
209  vector<string> qm_atom_specs;
210 
211  tok_md.ToVector( md_atom_specs );
212  tok_qm.ToVector( qm_atom_specs );
213 
214  // MD atom information
215  int residue_number = boost::lexical_cast<int>(md_atom_specs[0]);
216  string residue_name = md_atom_specs[1];
217  string md_atom_name = md_atom_specs[2];
218 
219  // QM atom information
220  bool hasQMPart = false;
221  int qm_atom_id = -1;
222  vec qmPos = vec(0,0,0);
223  // The atomic element is first taken as first character of
224  // the MD name; for atoms with QM counterpart, the element
225  // is read from the coordinates file
226  string element = md_atom_name.substr(0,1);
227  // Check whether MD atom has QM counterpart
228  if (qm_atom_specs.size() == 2) {
229  hasQMPart = true;
230  qm_atom_id = boost::lexical_cast<int>(qm_atom_specs[0]);
231 
232  // Look up qm coordinates in table created from xyz file
233  try {
234  qmPos = intCoords.at(qm_atom_id).second;
235  element = intCoords.at(qm_atom_id).first;
236  // Check whether elements of md and qm match
237  if (intCoords.at(qm_atom_id).first.substr(0,1)
238  != md_atom_name.substr(0,1) ) {
239  cout << "WARNING: Atom " <<md_atom_name << " in mol. "
240  << molMdName << " appears to have element type "
241  << md_atom_name.substr(0,1)
242  << ", but QM partner (ID " << qm_atom_id
243  << ") has element type "
244  << intCoords.at(qm_atom_id).first << endl;
245  }
246  }
247  catch (out_of_range) {
248  ; // No QM coordinates specified
249  }
250  }
251 
252  // Mapping weight
253  double weight = boost::lexical_cast<double>(*it_atom_weight);
254  if (!hasQMPart && weight != 0) {
255  cout << "ERROR: "
256  << "Atom " << md_atom_name << " in residue "
257  << residue_name << " in molecule " << molMdName
258  << " has no QM counterpart despite non-zero weight. "
259  << endl;
260  throw runtime_error( "Error in mapping file" );
261  }
262 
263  // Create atom
264  XTP::Atom *atom = AddAtomType(molecule,
265  residue_name, residue_number,
266  md_atom_name, md_atom_id++,
267  hasQMPart, qm_atom_id,
268  qmPos, element,
269  weight);
270 
271  try {
272  this->_map_mol_resNr_atm_atmType.at(molMdName)
273  .at(residue_number)
274  .at(md_atom_name);
275  // If this succeeded, atom has already been defined:
276  cout << "ERROR: "
277  << "Ambiguous atom definition in molecule "
278  << molMdName << ": "
279  << "Atom " << md_atom_name << " in residue "
280  << residue_number << " exists more than once. "
281  << endl;
282  throw runtime_error( "Ambiguity in atom definition." );
283  }
284  catch ( out_of_range ) {
285  this->_map_mol_resNr_atm_atmType[molMdName]
286  [residue_number]
287  [md_atom_name] = atom;
288  }
289 
290  fragment->AddAtom(atom);
291  segment->AddAtom(atom);
292  molecule->AddAtom(atom);
293 
294  } /* exit loop over atoms */
295 
296  } /* exit loop over fragments */
297 
298  } /* exit loop over segments */
299 
300  } /* exit loop over molecules */
301 }

313

322  qmtop->setCanRigidify(true);
323 
324  // Add types (=> Segment types / QM units)
325  vector< XTP::SegmentType* > ::iterator typeit;
326  for (typeit = this->_qmUnits.begin();
327  typeit != this->_qmUnits.end();

338  segType->setQMCoordsFile(qmcoords);
339  segType->setCanRigidify(canRigidify);
340  if (!canRigidify) { qmtop->setCanRigidify(false); }
341  }
342 
343  // Populate topology in a trickle-down manner
344  // (i.e. molecules => ... ... => atoms)
345  CSG::MoleculeContainer::iterator mit;
346  for (mit = mdtop->Molecules().begin();
347  mit != mdtop->Molecules().end();
348  mit++ ) {
349 
350  // MD molecule + name

352  string nameMolMD = molMD->getName();
353 
354  // Find QM counterpart
355  XTP::Molecule *molQM = this->MoleculeFactory(molMD);
356  string nameMolQM = molQM->getName();
357 
358  // Generate and export

361  }

371 XTP::Molecule *Md2QmEngine::MoleculeFactory(CSG::Molecule *molMDTemplate) {
372 
373  string nameMolQM = this->getMoleculeName(molMDTemplate->getName());
374  XTP::Molecule *molQMTemplate = this->getMoleculeType(nameMolQM);
375 
376  int resnrOffset = molMDTemplate->getBead(0)->getResnr();
377 
378  for (int i = 0; i < molMDTemplate->BeadCount(); i++) {
379  CSG::Bead *atomMD = molMDTemplate->getBead(i);

408 XTP::Molecule *Md2QmEngine::ExportMolecule(XTP::Molecule *refMol,
409  XTP::Topology *qmtop) {
410 
411  XTP::Molecule *newMol = qmtop->AddMolecule(refMol->getName());
412 
413  // Note: The topology is responsible for allocating IDs to
414  // molecules, segments, ...
415 
416  vector<XTP::Segment *> ::iterator segIt;
417  for (segIt = refMol->Segments().begin();
418  segIt < refMol->Segments().end();
419  segIt++) {
420 
421  XTP::Segment *refSeg = *segIt;
422  XTP::Segment *newSeg = qmtop->AddSegment(refSeg->getName());
423  newSeg->setType( qmtop->getSegmentType(refSeg->getType()->getId()) );
424 
425  vector<XTP::Fragment *> ::iterator fragIt;
426  for (fragIt = refSeg->Fragments().begin();
427  fragIt < refSeg->Fragments().end();
428  fragIt++) {
429 
430  XTP::Fragment *refFrag = *fragIt;
431  XTP::Fragment *newFrag = qmtop->AddFragment(refFrag->getName());
432  newFrag->setTrihedron(refFrag->getTrihedron());
433 
434  vector<XTP::Atom *> ::iterator atomIt;
435  for (atomIt = refFrag->Atoms().begin();
436  atomIt < refFrag->Atoms().end();
437  atomIt++) {
438 
439  XTP::Atom *refAtom = *atomIt;
440  XTP::Atom *newAtom = qmtop->AddAtom(refAtom->getName());
441 
442  newAtom->setWeight(refAtom->getWeight());
443  newAtom->setResnr(refAtom->getResnr());
444  newAtom->setResname(refAtom->getResname());
445  newAtom->setElement(refAtom->getElement());
446  if (refAtom->HasQMPart()) {
447  newAtom->setQMPart(refAtom->getQMId(),
448  refAtom->getQMPos());
449  }
450  newAtom->setPos(refAtom->getPos());
451 
452  newFrag->AddAtom(newAtom);
453  newSeg->AddAtom(newAtom);
454  newMol->AddAtom(newAtom);
455  } /* exit loop over template atoms */
456  newFrag->calcPos();
457  newSeg->AddFragment(newFrag);
458  newMol->AddFragment(newFrag);
459  } /* exit loop over template fragments */
460  newSeg->calcPos();
461  newMol->AddSegment(newSeg);
462  } /* exit loop over template molecules */
463 
464  return newMol;
465 }

571 void Md2QmEngine::getIntCoords(string &file,
572  map<int, pair<string,vec> > &intCoords) {
573 
574 
575  int atomCount = 0;
576 
577  std::string line;
578  std::ifstream intt;
579  intt.open(file.c_str());
580 
581  if (intt.is_open() ) {
582  while ( intt.good() ) {
583  std::getline(intt, line);
584 
585  vector< string > split;
586  Tokenizer toker(line, " \t");
587  toker.ToVector(split);
588  if ( !split.size() ||
589  split.size() != 4 ||
590  split[0] == "#" ||
591  split[0].substr(0,1) == "#" ) { continue; }
592 
593  // Interesting information written here: e.g. 'C 0.000 0.000 0.000'
594  atomCount++;
595  string element = split[0];
596  double x = boost::lexical_cast<double>( split[1] ) / 10.; //°A to NM
597  double y = boost::lexical_cast<double>( split[2] ) / 10.;
598  double z = boost::lexical_cast<double>( split[3] ) / 10.;
599  vec qmPos = vec(x,y,z);
600 
601  pair<string, vec> qmTypePos(element, qmPos);
602  intCoords[atomCount] = qmTypePos;
603  }
604  }
605  else {
606  throw std::runtime_error("No such file: '"+file+"'.");
607  }
608 }
