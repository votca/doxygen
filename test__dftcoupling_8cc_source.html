27 Eigen::MatrixXd ReadMatrixFromString(const std::string& matrix) {
28  votca::tools::Tokenizer lines(matrix, "\n");
29 
30  std::vector<double> entries;
31  Index cols = 0;
32  Index rows = 0;
33  for (auto line : lines) {
34  if (line[0] == '#') {
35  continue;
36  }
37  votca::tools::Tokenizer entries_tok(line, " ");
38  std::vector<std::string> temp = entries_tok.ToVector();
39  cols = Index(temp.size());
40  rows++;
41  for (const auto& s : temp) {
42  entries.push_back(std::stod(s));
43  }
44  }
45 
46  return Eigen::Map<Eigen::MatrixXd>(entries.data(), rows, cols);
47 }
