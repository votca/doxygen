363 bool XNeighborlist::StochasticConnectOrNot(double thisdistance, vector<double> distances, vector<double> probabilities, votca::tools::Random2 *RandomVariable){
364  double MINR = distances[0];
365  double MAXR = distances[distances.size()-1];
366  if(thisdistance == 0){
367  // no connection with itself
368  return false;
369  }
370  else if(thisdistance < MINR) {
371  return true;
372  }
373  else if(thisdistance > MAXR){
374  return false;
375  }
376  else{
377  //int numberofpoints = distances.size();
378  double thisprobability=0.0;
379  for(unsigned i = 0; i<distances.size()-1; i++){
380  if(distances[i] < thisdistance && thisdistance < distances[i+1]){
381  // linear interpolations
382  thisprobability = (probabilities[i+1]-probabilities[i])/(distances[i+1]-distances[i])*(thisdistance-distances[i])+probabilities[i];
383  break;
384  }
385  }
386  double uniformnumber = RandomVariable->rand_uniform();
387  if(uniformnumber < thisprobability) {
388  // accept connection
389  return true;
390  }
391  else{
392  // don't accept connection
393  return false;
394  }
395  }

398 void XNeighborlist::StochasticConnectivity(CTP::Topology *top, string filename) {
399  cout << endl << "... ... Creating connectivity based on the provided probability function that" << endl;
400  cout << "... ... describes the centre of mass-dependent probability of two sites being" << endl;
401  cout << "... ... conncected (coarse grained model). This file can be generated using the" << endl;
402  cout << "... ... panalyze calculator." << endl;
403 
404  // read in probability function
405  cout << endl << "... ... probability function:" << endl;
406  vector<double> distances;
407  vector<double> probabilities;
408  std::string line;
409  std::ifstream intt;
410  intt.open(filename.c_str());
411  int linenumber = 0;
412  if (intt.is_open() ) {
413  while ( intt.good() ) {
414 
415  std::getline(intt, line);
416  if (linenumber > 0){
417  vector<string> split;
418  CTP::Tokenizer toker(line, " \t");
419  toker.ToVector(split);
420 
421  if ( !split.size() ||
422  split[0] == "!" ||
423  split[0].substr(0,1) == "!" ) { continue; }
424 
425  double distance = boost::lexical_cast<double>(split[0]);
426  double probability = boost::lexical_cast<double>(split[1]);
427  cout << " " << distance << " " << probability << endl;
428  if(linenumber > 1){
429  if (probability == 1 && probabilities[probabilities.size()-1] == 1){
430  // first entry should be largest distance with probability 1
431  probabilities.pop_back();
432  distances.pop_back();
433  }
434  }
435  distances.push_back(distance);
436  probabilities.push_back(probability);
437  }
438  linenumber++;
439  }
440  }
441  else { cout << endl << "ERROR: No such file " << filename << endl;
442  throw std::runtime_error("Supply input file."); }
443

448  RandomVariable->init(rand(), rand(), rand(), rand());

450 
451  // get segments
452  vector <CTP::Segment*> segments = top->Segments();
453  vector <CTP::Segment*>::iterator seg1;
454  vector <CTP::Segment*>::iterator seg2;
455 
456  for (seg1 = segments.begin(); seg1!= segments.end(); seg1++){

462  TOOLS::vec r2 = (*seg2)->getPos();
463  double distance = abs( top->PbShortestConnect(r1, r2));

466  // add pair to neighbor list
