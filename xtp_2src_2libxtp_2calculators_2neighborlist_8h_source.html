65  void Neighborlist::Initialize(tools::Property *options) {
66 
67  // update options with the VOTCASHARE defaults
68  UpdateWithDefaults(options, "xtp");
69  std::string key = "options." + Identify();
70 
71  std::list< tools::Property* > segs = options->Select(key + ".segments");
72 
73  for (tools::Property* segprop : segs) {
74  std::string types = segprop->get("type").as< std::string>();
75  double cutoff = segprop->get("cutoff").as<double>();
76 
77  tools::Tokenizer tok(types, " ");
78  std::vector< std::string > names;
79  tok.ToVector(names);
80 
81  if (names.size() != 2) {
82  throw std::runtime_error("ERROR: Faulty pair definition for cut-off's: Need two segment names separated by a space");
83  }
84  _cutoffs[names[0]][names[1]] = cutoff;
85  _cutoffs[names[1]][names[0]] = cutoff;
86  if (std::find(_included_segments.begin(), _included_segments.end(), names[0]) == _included_segments.end()) {
87  _included_segments.push_back(names[0]);
88  }
89  if (std::find(_included_segments.begin(), _included_segments.end(), names[1]) == _included_segments.end()) {
90  _included_segments.push_back(names[1]);
91  }
92  }
93 
94  if (options->exists(key + ".constant")) {
95  _useConstantCutoff = true;
96  _constantCutoff = options->get(key + ".constant").as< double >();
97  } else {
98  _useConstantCutoff = false;
99  }
100  if (options->exists(key + ".exciton_cutoff")) {
101  _useExcitonCutoff = true;
102  _excitonqmCutoff = options->get(key + ".exciton_cutoff").as< double >();
103  } else {
104  _useExcitonCutoff = false;
105  }
106  if (options->exists(key + ".generate_from")) {
107  _generate_from_file = true;
108  _pairfilename = options->get(key + ".generate_from").as< std::string >();
109  } else {
110  _generate_from_file = false;
111  _pairfilename = "nofile";
112  }
113 
114  }

116  void Neighborlist::DetClassicalPairs(ctp::Topology* top){
117  std::cout << std::endl << " ... ... Determining classical pairs " << std::endl;
118  for (ctp::QMPair* pair:top->NBList()) {
119  tools::vec r1;
120  tools::vec r2;
121  ctp::Segment* seg1=pair->Seg1();
122  ctp::Segment* seg2=pair->Seg2();
123  bool stopLoop = false;
124  for (ctp::Fragment* frag1:seg1->Fragments()) {
125  if (stopLoop) {
126  break;
127  }
128  for (ctp::Fragment* frag2:seg2->Fragments()) {
129  r1 = frag1->getPos();
130  r2 = frag2->getPos();
131  if (tools::abs(top->PbShortestConnect(r1, r2)) > _excitonqmCutoff) {
132  pair->setType(3);
133  continue;
134  } else {
135  pair->setType(0);
136  stopLoop = true;
137  break;
138  }
139  } /* exit loop frag2 */
140  } /* exit loop frag1 */
141  } //Type 3 Exciton_classical approx
142  }

144  bool Neighborlist::EvaluateFrame(ctp::Topology *top) {
145  top->NBList().Cleanup();
146 
147  if (_generate_from_file) {
148  GenerateFromFile(top, _pairfilename);
149  }
150  else {
151  if (tools::globals::verbose) {
152  std::cout << std::endl << "... ..." << std::flush;
153  }
154 
155  const tools::matrix box = top->getBox();
156  double min = box.get(0, 0);
157  if (min > box.get(1, 1)) {
158  min = box.get(1, 1);
159  }
160  if (min > box.get(2, 2)) {
161  min = box.get(2, 2);
162  }
163 
164  std::vector< ctp::Segment* > segs;
165  for (ctp::Segment* seg:top->Segments()) {
166  if (_useConstantCutoff || std::find(_included_segments.begin(), _included_segments.end(), seg->getName()) != _included_segments.end()) {
167  segs.push_back(seg);
168  seg->calcPos();
169  seg->calcApproxSize();
170  }
171  }
172  std::cout << std::endl;
173  std::cout << "Evaluating " << segs.size() << " segments for neighborlist. "
174  << top->Segments().size() - segs.size() << " segments are not taken into account as specified" << std::endl;
175  if (!_useConstantCutoff) {
176  std::cout << "The following segments are used in the neigborlist creation" << std::endl;
177  std::cout << "\t" << std::flush;
178  for (const std::string& st:_included_segments) {
179  std::cout << " " << st << std::flush;
180  }
181  std::cout << std::endl;
182  }
183 
184  std::cout << "\r ... ... Evaluating " << std::flush;
185  std::vector<std::string> skippedpairs;
186 
187  for (std::vector< ctp::Segment* >::iterator segit1 = segs.begin(); segit1 < segs.end(); ++segit1) {
188  ctp::Segment *seg1 = *segit1;
189 
190  std::vector< ctp::Segment* > ::iterator segit2;
191  double cutoff;
192  tools::vec r1;
193  tools::vec r2;
194 
195  if (tools::globals::verbose) {
196  std::cout << "\r ... ... NB List Seg " << seg1->getId() << std::flush;
197  }
198 
199  for (segit2 = segit1 + 1; segit2 < segs.end(); ++segit2) {
200  ctp::Segment *seg2 = *segit2;
201  if (!_useConstantCutoff) {
202  try {
203  cutoff = _cutoffs.at(seg1->getName())
204  .at(seg2->getName());
205  } catch (const std::exception& out_of_range) {
206  std::string pairstring = seg1->getName() + "/" + seg2->getName();
207  if (std::find(skippedpairs.begin(), skippedpairs.end(), pairstring) == skippedpairs.end()) {
208  skippedpairs.push_back(pairstring);
209  }
210  continue;
211  }
212  }
213  else {
214  cutoff = _constantCutoff;
215  }
216 
217  if (cutoff > 0.5 * min) {
218  throw std::runtime_error((boost::format("Cutoff is larger than half the box size. Maximum allowed cutoff is %1$1.1f") % (0.5 * min)).str());
219  }
220  double cutoff2 = cutoff*cutoff;
221  tools::vec segdistance = top->PbShortestConnect(seg1->getPos(), seg2->getPos());
222  double segdistance2 = segdistance*segdistance;
223  double outside = cutoff + seg1->getApproxSize() + seg2->getApproxSize();
224 
225  if (segdistance2 < cutoff2) {
226  top->NBList().Add(seg1, seg2);
227  } else if (segdistance2 > (outside * outside)) {
228  continue;
229  } else {
230  bool stopLoop = false;
231  for (ctp::Fragment* frag1:seg1->Fragments()) {
232  if (stopLoop) {
233  break;
234  }
235  r1 = frag1->getPos();
236  for (ctp::Fragment* frag2:seg2->Fragments()) {
237  r2 = frag2->getPos();
238  tools::vec distance = top->PbShortestConnect(r1, r2);
239  double dist2 = distance*distance;
240  if (dist2 > cutoff2) {
241  continue;
242  } else {
243  top->NBList().Add(seg1, seg2);
244  stopLoop = true;
245  break;
246  }
247  } /* exit loop frag2 */
248  } /* exit loop frag1 */
249  }
250  } /* exit loop seg2 */
251  } /* exit loop seg1 */
252 
253  if (skippedpairs.size() > 0) {
254  std::cout << "WARNING: No cut-off specified for segment pairs of type " << std::endl;
255  for (const std::string& st:skippedpairs) {
256  std::cout << st << std::endl;
257  }
258  std::cout << "pairs were skipped" << std::endl;
259  }
260 
261  std::cout << std::endl << " ... ... Created " << top->NBList().size() << " direct pairs.";
262  if (_useExcitonCutoff) {
263  DetClassicalPairs(top);
264  }
265  }
266  return true;
267  }
