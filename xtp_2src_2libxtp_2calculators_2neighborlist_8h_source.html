77 void Neighborlist::Initialize(tools::Property *options) {
78 
79  // update options with the VOTCASHARE defaults
80  UpdateWithDefaults( options, "xtp" );
81  std::string key = "options." + Identify();
82 
83 
84  std::list< tools::Property* > segs = options->Select(key+".segments");
85  std::list< tools::Property* > ::iterator segsIt;
86 
87  for (segsIt = segs.begin();
88  segsIt != segs.end();
89  segsIt++) {
90 
91  std::string types = (*segsIt)->get("type").as< std::string>();
92  double cutoff = (*segsIt)->get("cutoff").as<double>();
93 
94  tools::Tokenizer tok(types, " ");
95  std::vector< std::string > names;
96  tok.ToVector(names);
97 
98 
99  if (names.size() != 2) {
100  std::cout << "ERROR: Faulty pair definition for cut-off's: "
101  << "Need two segment names separated by a space" << std::endl;
102  throw std::runtime_error("Error in options file.");
103  }
104 
105  _cutoffs[names[0]][names[1]] = cutoff;
106  _cutoffs[names[1]][names[0]] = cutoff;
107  if(std::find(_included_segments.begin(), _included_segments.end(), names[0]) == _included_segments.end()){
108  _included_segments.push_back(names[0]);
109  }
110  if(std::find(_included_segments.begin(), _included_segments.end(), names[1]) == _included_segments.end()){
111  _included_segments.push_back(names[1]);
112  }
113 
114 
115  }
116 
117  if (options->exists(key+".constant")) {
118  _useConstantCutoff = true;
119  _constantCutoff = options->get(key+".constant").as< double >();
120  }
121  else {
122  _useConstantCutoff = false;
123  }
124  if (options->exists(key+".exciton_cutoff")) {
125  _useExcitonCutoff = true;
126  _excitonqmCutoff = options->get(key+".exciton_cutoff").as< double >();
127  }
128  else {
129  _useExcitonCutoff = false;
130  }
131  if (options->exists(key+".generate_from")) {
132  _generate_from_file = true;
133  _generate_from = options->get(key+".generate_from").as< std::string >();
134  }
135  else {
136  _generate_from_file = false;
137  _generate_from = "nofile";
138  }
139 
140 
141 
142  // if superexchange is given
143  if (options->exists(key + ".superexchange")) {
144  _do_bridging = true;
145 
146 
147 
148  std::list< tools::Property* > _se = options->Select(key + ".superexchange");
149  std::list< tools::Property* > ::iterator seIt;
150 
151  for (seIt = _se.begin(); seIt != _se.end(); seIt++) {
152  std::string types = (*seIt)->get("type").as< std::string>();
153  ctp::QMNBList::SuperExchangeType* _su = new ctp::QMNBList::SuperExchangeType(types);
154  _superexchange.push_back(_su);
155  }
156  }
157  else{
158  _do_bridging=false;
159  }

163 bool Neighborlist::EvaluateFrame(ctp::Topology *top) {
164 
165 
166  top->NBList().Cleanup();
167 
168  if (_generate_from_file) {
169  GenerateFromFile(top, _generate_from);
170  }
171  else {
172 
173 
174  if (tools::globals::verbose) {
175  std::cout << std::endl << "... ..." << std::flush;
176  }
177 
178  const tools::matrix box=top->getBox();
179  double min=box.get(0,0);
180  if(min>box.get(1,1)){min=box.get(1,1);}
181  if(min>box.get(2,2)){min=box.get(2,2);}
182 
183  std::vector< ctp::Segment* > segs;
184 
185  for (std::vector< ctp::Segment* > ::iterator segit1 = top->Segments().begin();
186  segit1 < top->Segments().end();++segit1) {
187  ctp::Segment *seg1 = *segit1;
188  if(_useConstantCutoff || std::find(_included_segments.begin(), _included_segments.end(), seg1->getName()) != _included_segments.end()){
189  segs.push_back(seg1);
190  seg1->calcPos();
191  seg1->calcApproxSize();
192  }
193  }
194  std::cout<<std::endl;
195  std::cout <<"Evaluating "<<segs.size()<<" segments for neighborlist. "<< top->Segments().size()-segs.size()<<" segments are not taken into account as specified"<< std::endl;
196  if(!_useConstantCutoff){
197  std::cout << "The following segments were ignored in the neigborlist creation"<<std::endl;
198  std::cout<<"\t"<<std::flush;
199  for(std::vector< std::string >::iterator st=_included_segments.begin();st!=_included_segments.end();++st){
200  std::cout<<" "<<(*st)<<std::flush;
201  }
202  std::cout <<std::endl;
203  }
204 
205  std::cout << "\r ... ... Evaluating " <<std::flush;
206  std::vector<std::string> skippedpairs;
207 
208  for (std::vector< ctp::Segment* > ::iterator segit1 = segs.begin();segit1 < segs.end();++segit1) {
209  ctp::Segment *seg1 = *segit1;
210 
211  std::vector< ctp::Segment* > ::iterator segit2;
212  std::vector< ctp::Fragment* > ::iterator fragit1;
213  std::vector< ctp::Fragment* > ::iterator fragit2;
214  double cutoff;
215  tools::vec r1;
216  tools::vec r2;
217 
218 
219  if (tools::globals::verbose) {
220  std::cout << "\r ... ... NB List Seg " << seg1->getId() << std::flush;
221 
222  }
223 
224  for (segit2 = segit1 + 1;segit2 < segs.end();++segit2) {
225 
226  ctp::Segment *seg2 = *segit2;
227 
228  if (!_useConstantCutoff) {
229  // Find cut-off
230  try {
231  cutoff = _cutoffs.at(seg1->getName())
232  .at(seg2->getName());
233  }
234  catch (std::out_of_range) {
235  std::string pairstring=seg1->getName()+"/"+seg2->getName();
236  if(std::find(skippedpairs.begin(), skippedpairs.end(), pairstring) == skippedpairs.end()){
237  skippedpairs.push_back(pairstring);
238  }
239  continue;
240  }
241  }
242 
243  else { cutoff = _constantCutoff; }
244 
245  if (cutoff>0.5*min){
246  throw std::runtime_error((boost::format("Cutoff is larger than half the box size. Maximum allowed cutoff is %1$1.1f") % (0.5*min)).str());
247  }
248  double cutoff2=cutoff*cutoff;
249  tools::vec segdistance=top->PbShortestConnect(seg1->getPos(),seg2->getPos());
250  double segdistance2=segdistance*segdistance;
251  double outside=cutoff+seg1->getApproxSize()+seg2->getApproxSize();
252 
253  if(segdistance2<cutoff2){
254  top->NBList().Add(seg1, seg2);
255  }
256  else if(segdistance2>(outside*outside)){
257  continue;
258  }
259  else {
260  bool stopLoop = false;
261  for (fragit1 = seg1->Fragments().begin();
262  fragit1 < seg1->Fragments().end();
263  fragit1++) {
264 
265  if (stopLoop) {
266  break; }
267  r1 = (*fragit1)->getPos();
268  for (fragit2 = seg2->Fragments().begin();
269  fragit2 < seg2->Fragments().end();
270  fragit2++) {
271 
272 
273 
274  r2 = (*fragit2)->getPos();
275  tools::vec distance = top->PbShortestConnect(r1, r2);
276  double dist2 = distance*distance;
277  if (dist2 > cutoff2) {
278  continue;
279  } else {
280  top->NBList().Add(seg1, seg2);
281  stopLoop = true;
282  break;
283  }
284 
285  } /* exit loop frag2 */
286  } /* exit loop frag1 */
287  }
288 
289 
290  } /* exit loop seg2 */
291 
292 
293 
294  } /* exit loop seg1 */
295 
296 
297 
298 
299  if(skippedpairs.size()>0){
300  std::cout << "WARNING: No cut-off specified for segment pairs of type "<<std::endl;
301  for(std::vector< std::string >::iterator st=skippedpairs.begin();st!=skippedpairs.end();++st){
302  std::cout<<(*st)<<std::endl;
303  }
304  std::cout << "pairs were skipped"<<std::endl;
305  }
306 
307 
308  std::cout << std::endl << " ... ... Created " << top->NBList().size() << " direct pairs.";
309  if(_useExcitonCutoff){
310  std::cout << std::endl << " ... ... Determining classical pairs "<<std::endl;
311  ctp::QMNBList &nblist = top->NBList();
312  for (ctp::QMNBList::iterator pit = nblist.begin(); pit != nblist.end(); ++pit) {
313  tools::vec r1;
314  tools::vec r2;
315  std::vector< ctp::Fragment* > ::iterator fragit1;
316  std::vector< ctp::Fragment* > ::iterator fragit2;
317 
318  bool stopLoop = false;
319  for (fragit1 = (*pit)->Seg1()->Fragments().begin();
320  fragit1 < (*pit)->Seg1()->Fragments().end();
321  fragit1 ++) {
322  if (stopLoop) { break; }
323 
324  for (fragit2 = (*pit)->Seg2()->Fragments().begin();
325  fragit2 < (*pit)->Seg2()->Fragments().end();
326  fragit2++) {
327 
328 
329  r1 = (*fragit1)->getPos();
330  r2 = (*fragit2)->getPos();
331  if( abs( top->PbShortestConnect(r1, r2) ) > _excitonqmCutoff ) {
332  (*pit)->setType(3);
333  continue;
334  }
335  else {
336  (*pit)->setType(0);
337  stopLoop = true;
338  break;
339  }
340 
341  } /* exit loop frag2 */
342  } /* exit loop frag1 */
343  } //Type 3 Exciton_classical approx
344  }
345  if (_do_bridging){
346  // add superexchange pairs
347  top->NBList().setSuperExchangeTypes(_superexchange);
348  top->NBList().GenerateSuperExchange();
349 
350  // DEBUG output
351  if (votca::tools::globals::verbose) {
352 
353  tools::Property bridges_summary;
354  tools::Property *_bridges = &bridges_summary.add("bridges","");
355 
356  std::cout << "Bridged Pairs \n [idA:idB] com distance" << std::endl;
357  for (ctp::QMNBList::iterator ipair = top->NBList().begin(); ipair != top->NBList().end(); ++ipair) {
358  ctp::QMPair *pair = *ipair;
359  ctp::Segment* segment1 = pair->Seg1PbCopy();
360  ctp::Segment* segment2 = pair->Seg2PbCopy();
361 
362  std::cout << " [" << segment1->getId() << ":" << segment2->getId()<< "] "
363  << pair->Dist()<< " bridges: "
364  << (pair->getBridgingSegments()).size()
365  << " type: "
366  << pair->getType()
367  << " | " << std::flush;
368 
369  std::vector<ctp::Segment*> bsegments = pair->getBridgingSegments();
370 
371  tools::Property *_pair_property = &_bridges->add("pair","");
372 
373  _pair_property->setAttribute("id1", segment1->getId());
374  _pair_property->setAttribute("id2", segment2->getId());
375  _pair_property->setAttribute("name1", segment1->getName());
376  _pair_property->setAttribute("name2", segment2->getName());
377  _pair_property->setAttribute("r12", pair->Dist());
378 
379  tools::Property *_bridge_property = &_pair_property->add("bridge","");
380 
381  for ( std::vector<ctp::Segment*>::iterator itb = bsegments.begin(); itb != bsegments.end(); itb++ ) {
382  std::cout << (*itb)->getId() << " " ;
383  _bridge_property->setAttribute("id", (*itb)->getId());
384  }
385 
386  std::cout << std::endl;
387  }
388  //std::cout << bridges_summary;
389  }
390  }
391  }
392  return true;
393 }
