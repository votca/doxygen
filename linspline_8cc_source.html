00029 void LinSpline::Interpolate(ub::vector<double> &x, ub::vector<double> &y)
00030 {
00031     if(x.size() != y.size())
00032         throw std::invalid_argument("error in LinSpline::Interpolate : sizes of vectors x and y do not match");
00033 
00034     if(x.size()<2)
00035         throw std::invalid_argument("error in LinSpline::Interpolate : vectors x and y have to contain at least 2 points");
00036 
00037     const int N = x.size();
00038 
00039     // adjust the grid
00040     _r.resize(N);
00041     
00042     // copy the grid points into f
00043     _r = x;
00044     
00045     // LINEAR SPLINE: a(i) * x + b(i)
00046     // where i=number of interval
00047 
00048     // initialize vectors a,b
00049     a = ub::zero_vector<double>(N);
00050     b = ub::zero_vector<double>(N);
00051 
00052     // boundary conditions not applicable
00053     
00054     // calculate a,b for all intervals 0..(N-2), where interval
00055     // [x(i),x(i+1)] shall have number i (this means that the last interval
00056     // has number N-2)
00057     for (int i=0; i<N-1; i++) {
00058         a(i) = (y(i+1)-y(i))/(x(i+1)-x(i));
00059         b(i) = y(i)-a(i)*x(i);
00060     }
00061 }

00063 void LinSpline::Fit(ub::vector<double> &x, ub::vector<double> &y)
00064 {
00065     if(x.size() != y.size())
00066         throw std::invalid_argument("error in LinSpline::Fit : sizes of vectors x and y do not match");
00067 
00068     const int N = x.size();
00069     const int ngrid = _r.size();
00070 
00071     // construct the equation
00072     // A*u = b
00073     // The matrix A contains all conditions
00074     // s_i(x) = (y(i+1)-y(i)) * (x-r(i))/(r(i+1)-r(i)) + y(i)
00075     // where y(i) are the unknown values at grid points r(i), and
00076     // the condition y=s_i(x) is to be satisfied at all input points:
00077     // therefore b=y and u=vector of all unknown y(i)
00078     
00079     ub::matrix<double> A(N, ngrid);
00080     A = ub::zero_matrix<double>(N, ngrid);
00081     int interval;
00082 
00083     // construct matrix A
00084     for (int i=0; i<N; i++) {
00085         interval = getInterval(x(i));
00086         A(i,interval)   = 1 - (x(i)-_r(interval))/(_r(interval+1)-_r(interval));
00087         A(i,interval+1) = (x(i)-_r(interval))/(_r(interval+1)-_r(interval));
00088     }
00089 
00090     // now do a qr solve
00091     ub::vector<double> sol(ngrid);
00092     votca::tools::linalg_qrsolve(sol, A, y);
00093 
00094     // vector "sol" contains all y-values of fitted linear splines at each
00095     // interval border
00096     // get a(i) and b(i) for piecewise splines out of solution vector "sol"
00097     a = ub::zero_vector<double>(ngrid-1);
00098     b = ub::zero_vector<double>(ngrid-1);
00099     for (int i=0; i<ngrid-1; i++) {
00100         a(i) = (sol(i+1)-sol(i))/(_r(i+1)-_r(i));
00101         b(i) = -a(i)*_r(i) + sol(i);
00102     }
00103 }
