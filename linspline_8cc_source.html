29 void LinSpline::Interpolate(ub::vector<double> &x, ub::vector<double> &y)
30 {
31  if(x.size() != y.size())
32  throw std::invalid_argument("error in LinSpline::Interpolate : sizes of vectors x and y do not match");
33 
34  if(x.size()<2)
35  throw std::invalid_argument("error in LinSpline::Interpolate : vectors x and y have to contain at least 2 points");
36 
37  const int N = x.size();
38 
39  // adjust the grid
40  _r.resize(N);
41 
42  // copy the grid points into f
43  _r = x;
44 
45  // LINEAR SPLINE: a(i) * x + b(i)
46  // where i=number of interval
47 
48  // initialize vectors a,b
49  a = ub::zero_vector<double>(N);
50  b = ub::zero_vector<double>(N);
51 
52  // boundary conditions not applicable
53 
54  // calculate a,b for all intervals 0..(N-2), where interval
55  // [x(i),x(i+1)] shall have number i (this means that the last interval
56  // has number N-2)
57  for (int i=0; i<N-1; i++) {
58  a(i) = (y(i+1)-y(i))/(x(i+1)-x(i));
59  b(i) = y(i)-a(i)*x(i);
60  }
61 }

63 void LinSpline::Fit(ub::vector<double> &x, ub::vector<double> &y)
64 {
65  if(x.size() != y.size())
66  throw std::invalid_argument("error in LinSpline::Fit : sizes of vectors x and y do not match");
67 
68  const int N = x.size();
69  const int ngrid = _r.size();
70 
71  // construct the equation
72  // A*u = b
73  // The matrix A contains all conditions
74  // s_i(x) = (y(i+1)-y(i)) * (x-r(i))/(r(i+1)-r(i)) + y(i)
75  // where y(i) are the unknown values at grid points r(i), and
76  // the condition y=s_i(x) is to be satisfied at all input points:
77  // therefore b=y and u=vector of all unknown y(i)
78 
79  ub::matrix<double> A(N, ngrid);
80  A = ub::zero_matrix<double>(N, ngrid);
81  int interval;
82 
83  // construct matrix A
84  for (int i=0; i<N; i++) {
85  interval = getInterval(x(i));
86  A(i,interval) = 1 - (x(i)-_r(interval))/(_r(interval+1)-_r(interval));
87  A(i,interval+1) = (x(i)-_r(interval))/(_r(interval+1)-_r(interval));
88  }
89 
90  // now do a qr solve
91  ub::vector<double> sol(ngrid);
92  votca::tools::linalg_qrsolve(sol, A, y);
93 
94  // vector "sol" contains all y-values of fitted linear splines at each
95  // interval border
96  // get a(i) and b(i) for piecewise splines out of solution vector "sol"
97  a = ub::zero_vector<double>(ngrid-1);
98  b = ub::zero_vector<double>(ngrid-1);
99  for (int i=0; i<ngrid-1; i++) {
100  a(i) = (sol(i+1)-sol(i))/(_r(i+1)-_r(i));
101  b(i) = -a(i)*_r(i) + sol(i);
102  }
103 }
