28 // MPS files have a weird format positions can be in bohr or angstroem,
29 // multipoles are in q*bohr^k, with k rank of multipole and polarisabilities are
30 // in angstroem^3
31 template <class T>
32 void ClassicalSegment<T>::LoadFromFile(std::string filename) {
33 
34  std::string line;
35  std::ifstream intt;
36  intt.open(filename);
37  double unit_conversion = tools::conv::ang2bohr;
38 
39  Index readinmultipoles = 0;
40  Index numberofmultipoles = 0;
41  Vector9d multipoles = Vector9d::Zero();
42  Index rank = 0;
43 
44  if (!intt.is_open()) {
45  throw std::runtime_error("File:" + filename + " could not be opened");
46  }
47  while (intt.good()) {
48 
49  std::getline(intt, line);
50  tools::Tokenizer toker(line, " \t");
51  std::vector<std::string> split = toker.ToVector();
52 
53  if (!split.size() || split[0] == "!" || split[0].substr(0, 1) == "!") {
54  continue;
55  }
56 
57  // ! Interesting information here, e.g.
58  // ! DCV2T opt
59  // ! SP RB3LYP 6-311+G(d,p)
60  // Units bohr
61  //
62  // C -4.2414603400 -3.8124751600 0.0017575736 Rank 2
63  // -0.3853409355
64  // -0.0002321905 0.2401559510 0.6602334308
65  // -0.7220625314 0.0004894995 -0.0003833545 0.4526409813 -0.50937399
66  // P 1.75
67 
68  // Units used
69  if (split[0] == "Units") {
70  std::string units = split[1];
71  if (units != "bohr" && units != "angstrom") {
72  throw std::runtime_error("Unit " + units + " in file " + filename +
73  " not supported.");
74  }
75  if (units == "bohr") {
76  unit_conversion = 1.0;
77  }
78  } else if (split.size() == 6) {
79  // element, position, rank limit convert to bohr
80  std::string name = split[0];
81  Eigen::Vector3d pos;
82  Index id = Index(this->_atomlist.size());
83  pos[0] = boost::lexical_cast<double>(split[1]);
84  pos[1] = boost::lexical_cast<double>(split[2]);
85  pos[2] = boost::lexical_cast<double>(split[3]);
86  rank = boost::lexical_cast<Index>(split[5]);
87  numberofmultipoles = (rank + 1) * (rank + 1);
88  multipoles = Vector9d::Zero();
89  pos *= unit_conversion;
90  this->_atomlist.push_back(T(id, name, pos));
91  }
92  // 'P', dipole polarizability
93  else if (split[0] == "P") {
94  Eigen::Matrix3d p1;
95  if (split.size() == 7) {
96  double pxx = boost::lexical_cast<double>(split[1]);
97  double pxy = boost::lexical_cast<double>(split[2]);
98  double pxz = boost::lexical_cast<double>(split[3]);
99  double pyy = boost::lexical_cast<double>(split[4]);
100  double pyz = boost::lexical_cast<double>(split[5]);
101  double pzz = boost::lexical_cast<double>(split[6]);
102  p1 << pxx, pxy, pxz, pxy, pyy, pyz, pxz, pyz, pzz;
103  } else if (split.size() == 2) {
104  double pxx = boost::lexical_cast<double>(split[1]);
105  p1 = pxx * Eigen::Matrix3d::Identity();
106  } else {
107  throw std::runtime_error("Invalid line in " + filename + ": " + line);
108  }
109  double unit_conversion_3 = std::pow(tools::conv::ang2bohr, 3);
110  p1 = p1 * unit_conversion_3;
111  this->_atomlist.back().setPolarisation(p1);
112  }
113  // Multipole lines
114  else {
115  // stay in bohr
116  for (const std::string& entry : split) {
117  double qXYZ = boost::lexical_cast<double>(entry);
118  if (multipoles.size() < readinmultipoles) {
119  throw std::runtime_error("ReadMpsFile: File" + filename +
120  "is not properly formatted");
121  }
122  multipoles(readinmultipoles) = qXYZ;
123  readinmultipoles++;
124  }
125  if (readinmultipoles == numberofmultipoles) {
126  Eigen::Vector3d temp_dipoles = multipoles.segment<3>(1);
127  // mps format for dipoles is z x y
128  // we need x y z
129  multipoles(1) = temp_dipoles(1);
130  multipoles(2) = temp_dipoles(2);
131  multipoles(3) = temp_dipoles(0);
132  this->_atomlist.back().setMultipole(multipoles, rank);
133  readinmultipoles = 0;
134  }
135  }
136  }
137  this->calcPos();
138 }

140 template <class T>
141 double ClassicalSegment<T>::CalcTotalQ() const {
142  double Q = 0;
143  for (const T& site : this->_atomlist) {
144  Q += site.getCharge();
145  }
146  return Q;

149 template <class T>
150 Eigen::Vector3d ClassicalSegment<T>::CalcDipole() const {
151  Eigen::Vector3d dipole = Eigen::Vector3d::Zero();
152 
153  Eigen::Vector3d CoM = this->getPos();
154  for (const T& site : this->_atomlist) {
155  dipole += (site.getPos() - CoM) * site.getCharge();
156  dipole += site.getDipole();
157  }
158  return dipole;
159 }

161 template <class T>
162 void ClassicalSegment<T>::WriteMPS(std::string filename,
163  std::string header) const {
164 
165  std::ofstream ofs;
166  ofs.open(filename, std::ofstream::out);
167  if (!ofs.is_open()) {
168  throw std::runtime_error("Bad file handle: " + filename);
169  }
170 
171  ofs << (boost::format("! GENERATED BY VOTCA::XTP::%1$s\n") % header);
172  ofs << (boost::format("! N=%2$d Q[e]=%1$+1.7f\n") % CalcTotalQ() %
173  this->size());
174  ofs << boost::format("Units angstrom\n");
175 
176  for (const T& site : this->_atomlist) {
177  ofs << site.WriteMpsLine("angstrom");
178  }
179  ofs.close();
180 }
