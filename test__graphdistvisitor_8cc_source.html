44BOOST_AUTO_TEST_CASE(basic_test) {
45 // Create edge
46 Edge ed(0, 1);
47 vector<Edge> edges;
48 edges.push_back(ed);
49
50 // Create Graph nodes
51 GraphNode gn1;
52 GraphNode gn2;
53
54 unordered_map<votca::Index, GraphNode> nodes;
55 nodes[0] = gn1;
56 nodes[1] = gn2;
57
58 Graph g(edges, nodes);
59
60 GraphDistVisitor gdv;
61 BOOST_CHECK(gdv.queEmpty());
62
63 BOOST_CHECK_THROW(gdv.exec(g, ed), runtime_error);
64 // Default starts with node index 0
65 gdv.initialize(g);
66 BOOST_CHECK_EQUAL(gdv.queEmpty(), false);
67 // No exception should be thrown at this point
68 Edge ed1 = gdv.nextEdge(g);
69 BOOST_CHECK_EQUAL(ed, ed1);
70 gdv.exec(g, ed1);
71 BOOST_CHECK(gdv.queEmpty());
72 GraphNode gn3 = g.getNode(0);
73 votca::Index dist = gn3.getInt("Dist");
74 BOOST_CHECK_EQUAL(dist, 0);
75 GraphNode gn4 = g.getNode(1);
76 dist = gn4.getInt("Dist");
77 BOOST_CHECK_EQUAL(dist, 1);
78}

101 edges.push_back(ed3);
102 edges.push_back(ed4);
103 edges.push_back(ed5);
104 edges.push_back(ed6);
105
106 // Create Graph nodes
107 GraphNode gn1;
108 GraphNode gn2;
109 GraphNode gn3;
110 GraphNode gn4;
111 GraphNode gn5;
112 GraphNode gn6;
113 GraphNode gn7;
114
115 unordered_map<votca::Index, GraphNode> nodes;
116 nodes[0] = gn1;
117 nodes[1] = gn2;
118 nodes[2] = gn3;
119 nodes[3] = gn4;
120 nodes[4] = gn5;
121 nodes[5] = gn6;
122 nodes[6] = gn7;
123
124 Graph g(edges, nodes);
125
126 GraphDistVisitor gdv;
127 BOOST_CHECK(gdv.queEmpty());
128
129 BOOST_CHECK_THROW(gdv.exec(g, ed), runtime_error);
130 // Default starts with node index 0
131 gdv.initialize(g);
132 BOOST_CHECK_EQUAL(gdv.queEmpty(), false);
133 // No exception should be thrown at this point
134
135 // First two edges that should be explored are edges ed and ed6
136 vector<Edge> temp;
137 Edge ed7 = gdv.nextEdge(g);
138 temp.push_back(ed7);
139 gdv.exec(g, ed7);
140 ed7 = gdv.nextEdge(g);
141 gdv.exec(g, ed7);
142 temp.push_back(ed7);
143
144 bool found_ed = false;
145 bool found_ed6 = false;
146 for (auto ed_temp : temp) {
147 if (ed_temp == ed) {
148 found_ed = true;
149 }
150 if (ed_temp == ed6) {
151 found_ed6 = true;
152 }
153 }
154 BOOST_CHECK(found_ed);
155 BOOST_CHECK(found_ed6);
156
157 // Explore the whole graph
158 while (!gdv.queEmpty()) {
159 ed7 = gdv.nextEdge(g);
160 gdv.exec(g, ed7);
161 }
162
163 // Now let's check that the distances for each node
164 // are correct. The distances are counted from the starting
165 // node which by default is node 0.
166
167 GraphNode gn8 = g.getNode(0);
168 votca::Index dist = gn8.getInt("Dist");
169 BOOST_CHECK_EQUAL(dist, 0);
170
171 gn8 = g.getNode(1);
172 dist = gn8.getInt("Dist");
173 BOOST_CHECK_EQUAL(dist, 1);
174
175 gn8 = g.getNode(2);
176 dist = gn8.getInt("Dist");
177 BOOST_CHECK_EQUAL(dist, 2);
178
179 gn8 = g.getNode(3);
180 dist = gn8.getInt("Dist");
181 BOOST_CHECK_EQUAL(dist, 3);
182
183 gn8 = g.getNode(4);
184 dist = gn8.getInt("Dist");
185 BOOST_CHECK_EQUAL(dist, 2);
186
187 gn8 = g.getNode(5);
188 dist = gn8.getInt("Dist");
189 BOOST_CHECK_EQUAL(dist, 3);
190
191 gn8 = g.getNode(6);
192 dist = gn8.getInt("Dist");
193 BOOST_CHECK_EQUAL(dist, 1);
194}
