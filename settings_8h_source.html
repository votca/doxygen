78  template <typename T = std::string>
79  T get(const std::string& key) const {
80  auto delimiter = ".";
81  std::string primary_key = key;
82  std::string secondary_key;
83 
84  auto iter = key.find(delimiter);
85  if (iter != std::string::npos) {
86  primary_key = key.substr(0, key.find(delimiter));
87  secondary_key = key.substr(key.find(delimiter) + 1);
88  }
89 
90  auto it = this->_nodes.find(primary_key);
91  if (it == this->_nodes.end()) {
92  std::ostringstream oss;
93  oss << "Unknown keyword: " << key << "\n";
94  throw std::runtime_error(oss.str());
95  } else if (secondary_key.empty()) {
96  return it->second.as<T>();
97  } else {
98  return it->second.get(secondary_key).as<T>();
99  }
100  }
