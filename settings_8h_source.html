74  template <typename T = std::string>
75  T get(const std::string& key) const {
76  auto delimiter = ".";
77  std::string primary_key = key;
78  std::string secondary_key;
79 
80  auto iter = key.find(delimiter);
81  if (iter != std::string::npos) {
82  primary_key = key.substr(0, key.find(delimiter));
83  secondary_key = key.substr(key.find(delimiter) + 1);
84  }
85 
86  auto it = this->_nodes.find(primary_key);
87  if (it == this->_nodes.end()) {
88  std::ostringstream oss;
89  oss << "Unknown keyword: " << key << "\n";
90  throw std::runtime_error(oss.str());
91  } else if (secondary_key.empty()) {
92  return it->second.as<T>();
93  } else {
94  return it->second.get(secondary_key).as<T>();
95  }
96  }
