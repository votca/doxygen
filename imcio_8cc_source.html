120 void imcio_read_matrix(const string &filename, ub::matrix<double> &gmc)
121 {
122  ifstream in;
123  in.open(filename.c_str());
124 
125  bool is_initialized = false;
126  if(!in)
127  throw runtime_error(string("error, cannot open file ") + filename);
128 
129  size_t line_count =0;
130  string line;
131  // read till the first data line
132  while(getline(in, line)) {
133  // remove comments and xmgrace stuff
134  line = line.substr(0, line.find("#"));
135  line = line.substr(0, line.find("@"));
136 
137  // tokenize string and put it to vector
138  Tokenizer tok(line, " \t");
139  vector<string> tokens;
140  tok.ToVector(tokens);
141 
142  // skip empty lines
143  if(tokens.size()==0) continue;
144  if(!is_initialized)
145  gmc.resize(tokens.size(),tokens.size());
146  is_initialized=true;
147 
148  if(gmc.size1()!=tokens.size())
149  throw runtime_error(string("error loading ")
150  + filename + ": size mismatchm, number of columns differ");
151  for(size_t i=0; i<tokens.size(); ++i)
152  gmc(line_count,i) = boost::lexical_cast<double>(tokens[i]);
153  ++line_count;
154  }
155  if(line_count != gmc.size1())
156  throw runtime_error(string("error loading ")
157  + filename + ": size mismatch, not enough lines");
158  in.close();
159 }

161 void imcio_read_index(const string &filename, vector<string> &names, vector<RangeParser> &ranges)
162 {
163  ifstream in;
164  in.open(filename.c_str());
165  if(!in)
166  throw runtime_error(string("error, cannot open file ") + filename);
167 
168  names.clear();
169  ranges.clear();
170  string line;
171 
172  // read till the first data line
173  while(getline(in, line)) {
174  // remove comments and xmgrace stuff
175  line = line.substr(0, line.find("#"));
176  line = line.substr(0, line.find("@"));
177 
178  boost::trim(line);
179 
180  size_t found;
181  found = line.find(" ");
182  if(found==string::npos)
183  throw runtime_error(string("wrong format in ") + filename);
184 
185  string name = line.substr(0, found);
186 
187  string range = line.substr(found);
188 
189  RangeParser rp;
190  rp.Parse(range);
191  names.push_back(name);
192  ranges.push_back(rp);
193 
194  }
195  in.close();
196 }
