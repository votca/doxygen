106 Eigen::MatrixXd imcio_read_matrix(const std::string &filename) {
107  std::ifstream intt;
108  intt.open(filename);
109  if (!intt) {
110  throw std::runtime_error(std::string("error, cannot open file ") +
111  filename);
112  }
113 
114  std::string line;
115  std::vector<double> result;
116  Index numrows = 0;
117  size_t numcols = 0;
118  while (getline(intt, line)) {
119  if (line[0] == '#') {
120  continue;
121  }
122  votca::tools::Tokenizer tok(line, " \t");
123  std::vector<std::string> tokens = tok.ToVector();
124  if (numrows == 0) {
125  numcols = tokens.size();
126  } else if (numcols != tokens.size()) {
127  throw std::runtime_error(
128  "Matrix has not the same number of entries in each row.");
129  }
130  for (const std::string &s : tokens) {
131  result.push_back(std::stod(s));
132  }
133  numrows++;
134  }
135  intt.close();
136 
137  return Eigen::Map<Eigen::MatrixXd>(result.data(), numrows, numcols);
138 }

140 std::vector<std::pair<std::string, tools::RangeParser> > imcio_read_index(
141  const string &filename) {
142  ifstream in;
143  in.open(filename);
144  if (!in) {
145  throw runtime_error(string("error, cannot open file ") + filename);
146  }
147 
148  std::vector<std::pair<std::string, tools::RangeParser> > indeces;
149  string line;
150  // read till the first data line
151  while (getline(in, line)) {
152  // remove comments and xmgrace stuff
153  line = line.substr(0, line.find("#"));
154  line = line.substr(0, line.find("@"));
155 
156  boost::trim(line);
157  size_t found = line.find(" ");
158  if (found == string::npos) {
159  throw runtime_error(string("wrong format in ") + filename);
160  }
161 
162  string name = line.substr(0, found);
163  string range = line.substr(found);
164  tools::RangeParser rp;
165  rp.Parse(range);
166  indeces.push_back(std::pair<std::string, tools::RangeParser>(name, rp));
167  }
168  in.close();
169  return indeces;
170 }
