114 void imcio_read_matrix(const string &filename, Eigen::MatrixXd &gmc) {
115  ifstream in;
116  in.open(filename);
117 
118  bool is_initialized = false;
119  if (!in) throw runtime_error(string("error, cannot open file ") + filename);
120 
121  int line_count = 0;
122  string line;
123  // read till the first data line
124  while (getline(in, line)) {
125  // remove comments and xmgrace stuff
126  line = line.substr(0, line.find("#"));
127  line = line.substr(0, line.find("@"));
128 
129  // tokenize string and put it to vector
130  tools::Tokenizer tok(line, " \t");
131  vector<string> tokens = tok.ToVector();
132 
133  // skip empty lines
134  if (tokens.size() == 0) continue;
135  if (!is_initialized) gmc.resize(tokens.size(), tokens.size());
136  is_initialized = true;
137 
138  if (gmc.rows() != int(tokens.size()))
139  throw runtime_error(string("error loading ") + filename +
140  ": size mismatchm, number of columns differ");
141  for (size_t i = 0; i < tokens.size(); ++i)
142  gmc(line_count, i) = stod(tokens[i]);
143  ++line_count;
144  }
145  if (line_count != gmc.rows())
146  throw runtime_error(string("error loading ") + filename +
147  ": size mismatch, not enough lines");
148  in.close();
149 }

151 void imcio_read_index(const string &filename, vector<string> &names,
152  vector<tools::RangeParser> &ranges) {
153  ifstream in;
154  in.open(filename);
155  if (!in) throw runtime_error(string("error, cannot open file ") + filename);
156 
157  names.clear();
158  ranges.clear();
159  string line;
160 
161  // read till the first data line
162  while (getline(in, line)) {
163  // remove comments and xmgrace stuff
164  line = line.substr(0, line.find("#"));
165  line = line.substr(0, line.find("@"));
166 
167  boost::trim(line);
168 
169  size_t found;
170  found = line.find(" ");
171  if (found == string::npos)
172  throw runtime_error(string("wrong format in ") + filename);
173 
174  string name = line.substr(0, found);
175 
176  string range = line.substr(found);
177 
178  tools::RangeParser rp;
179  rp.Parse(range);
180  names.push_back(name);
181  ranges.push_back(rp);
182  }
183  in.close();
184 }
