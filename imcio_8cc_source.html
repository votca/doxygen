00141 
00142         // skip empty lines
00143         if(tokens.size()==0) continue;
00144         if(!is_initialized)
00145             gmc.resize(tokens.size(),tokens.size());

00147 
00148         if(gmc.size1()!=tokens.size())
00149             throw runtime_error(string("error loading ")
00150                     + filename + ": size mismatchm, number of columns differ");
00151         for(size_t i=0; i<tokens.size(); ++i)
00152             gmc(line_count,i) = boost::lexical_cast<double>(tokens[i]);
00153         ++line_count;
00154     }
00155     if(line_count != gmc.size1())
00156             throw runtime_error(string("error loading ")
00157                     + filename + ": size mismatch, not enough lines");
