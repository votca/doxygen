76 void PDB2Top::Initialize(tools::Property* options) {
77  // read options

79 
80  // set boolean constants to false

82  _has_gro = false;
83 
84  // find PDB, then GRO, then error
85  if ( options->exists(key+"pdb") ){
86  _input_pdb = options->get(key+"pdb").as<string> ();
87  _has_pdb = true;
88  cout << endl << "... ... PDB file: \t" << _input_pdb;
89  }
90  else if ( options->exists(key+"gro") ){
91  _input_gro = options->get(key+"gro").as<string> ();
92  _has_gro = true;
93  cout << endl << "... ... GRO file: \t" << _input_gro;
94  }
95  else{
96  error1( "... ... Error. Unsupported input file format. \n"
97  "... ... Supported formats: pdb, gro \n");
98  }
99 
100  if ( options->exists(key+"top") ){
101  _out_top = options->get(key+"top").as<string> ();
102  cout << "\n" "... ... User defined top file: \t" << _numMol;
103  }
104  else{
105  _out_top = "topol.top";
106  cout <<"\n" "... ... Default top file: topol.top \n";
107  }
108 
109 
110  if ( options->exists(key+"num") ){
111  _numMol = options->get(key+"num").as<int> ();
112  cout << "\n" "... ... User defined num of mols: \t" << _numMol;
113  }
114  else{_numMol = 1; cout <<"\n" "... ... Default num of mols: 1 \n";}

132 void PDB2Top::top2txt(){
133 
134  // generating part
135  ctp::Topology * _topPtr = &_top;
136 
137  // preps
138  stringstream ss, sbody, stype;
139 
140  string weight("1.000"), charge("0.000"), resref("-1"),
141  sigma ("0.00000e+00"), eps ("0.00000e+00");
142 
143  vector <char> atTypesLst;
144 
145  // iterations
146  vector <ctp::Fragment*> _fragsPtr = _topPtr->Fragments();
147  vector <ctp::Fragment*>::iterator _fragIt;
148  for (_fragIt = _fragsPtr.begin(); _fragIt!=_fragsPtr.end(); _fragIt++){
149 
150  // read atoms from fragments
151  // iterate
152  ctp::Fragment* _frag = *(_fragIt);
153  vector <ctp::Atom*> _atsPtr = _frag->Atoms();
154  vector <ctp::Atom*>::iterator _atIt;
155 
156  // print name as comment
157  sbody << "; res " << _frag->getName() << endl;
158 
159  for (_atIt = _atsPtr.begin(); _atIt!=_atsPtr.end(); _atIt++){
160 
161  ctp::Atom* _at = *(_atIt);
162 
163  // print name, mass, charge, ptype, sigma, eps
164  // if not found add, else dont
165  if( find(atTypesLst.begin(), atTypesLst.end(),
166  _at->getName()[0]) == atTypesLst.end()){
167 
168  atTypesLst.push_back(_at->getName()[0]);
169 
170  stype << format("%1$6s%2$11s%3$12s%4$9s%5$13s%6$13s\n")
171  % _at->getName()[0] % weight
172  % charge % "A"
173  % sigma % eps ;
174  }
175 
176  // print atid, elem, resid,resname, atname, id, charge, mass
177  sbody << format( "%1$10s%2$10s%3$10s%4$10s"
178  "%5$10s%6$10s%7$10s%8$10s\n" )
179  % _at->getId() % _at->getName()[0]
180  % _at->getResnr() % _at->getResname()
181  % _at->getName() % _at->getId()
182  % charge % weight ;
183  }
184  }
185  // output part
186  ss << "; " << '\n'
187  << "; generated by votca top2map " << '\n'
188  << "; for xtp_map ONLY" << '\n'
189  << "; you CAN NOT use this top " << '\n'
190  << "; for simulations! " << '\n'
191  << "; " << '\n'
192  << "" << '\n'
193  << "[ defaults ]" << '\n'
194  << format(";%1$7s%2$17s%3$16s%4$14s%5$8s\n")
195  % "nbfunc" % "comb-rule"
196  % "gen-pairs" % "fudgeLJ"
197  % "fudgeLJ"
198  << format("%1$8s%2$17s%3$16s%4$14s%5$8s\n")
199  % '1' % '3'
200  % "yes" % "0.5"
201  % "0.5"
202  << "" << '\n'
203  << "[ atomtypes ]" << '\n'
204  << format(";%1$5s%2$11s%3$12s%4$9s%5$13s%6$13s\n")
205  % "name" % "mass"
206  % "charge" % "ptype"
207  % "sigma" % "eps" ;
208  // types
209  ss << stype.str();
210 
211  //
212  ss << "" << '\n'
213  << "[ moleculetype ]" << '\n'
214  << "; Name nrexcl" << '\n'
215  << "Other 3" << '\n'
216  << "" << '\n'
217  << "[ atoms ]" << '\n'
218  << format( ";%1$9s%2$10s%3$10s%4$10s"
219  "%5$10s%6$10s%7$10s%8$10s\n" )
220  % "nr" % "type"
221  % "resnr" % "residue"
222  % "atom" % "cgnr"
223  % "charge" % "mass" ;
224  // body
225  ss << sbody.str();
226 
227  ss << "" << '\n'
228  << "[ system ]" << '\n'
229  << "; Name" << '\n'
230  << "Protein" << '\n'
231  << "" << '\n'
232  << "[ molecules ]" << '\n'
233  << "; Compound #mols" << '\n'
234  << format("Other %s")
235  % _numMol << '\n' ;
236 
237 
238  // print entire stream to file;
239  std::ofstream _outfile( _out_top.c_str());
240  _outfile << ss.str();
241  _outfile.close();

244 void PDB2Top::readPDB(){
245  cout << endl << "... ... Assuming: PDB";
246 
247  // set molecule >> segment >> fragment
248  // reconnect them all
249  ctp::Topology * _topPtr = 0;
250  _topPtr = &_top;
251 
252  ctp::Molecule * _molPtr = 0;
253  // direct
254  _molPtr = _topPtr->AddMolecule("M1");
255  // inverse
256  _molPtr->setTopology(_topPtr);
257 
258  ctp::Segment * _segPtr = 0;
259  // direct
260  _segPtr = _topPtr->AddSegment("S1");
261  _molPtr->AddSegment(_segPtr);
262  // inverse
263  _segPtr->setTopology(_topPtr);
264  _segPtr->setMolecule(_molPtr);
265 
266  // try: read PDB file
267  std::ifstream _file( _input_pdb.c_str());
268  if (_file.fail()) {
269  error1( "... ... Can not open: " + _input_pdb + "\n"
270  "... ... Does it exist? Is it correct file name?\n");
271  }
272  else{
273  cout << endl <<
274  ("... ... File " + _input_pdb + ""
275  " was opened successfully.\n");
276  }
277 
278  // read PDB line by line
279  string _line;
280 
281  // counters for loops
282 // int _atom_id = 0;
283  int _newResNum = 0;
284 
285  while ( std::getline(_file, _line,'\n') ){
286  if( boost::find_first(_line, "ATOM" ) ||
287  boost::find_first(_line, "HETATM")
288  ){
289 
290  // according to PDB format
291  string _recType (_line,( 1-1),6); // str, "ATOM", "HETATM"
292  string _atNum (_line,( 7-1),6); // int, Atom serial number
293  string _atName (_line,(13-1),4); // str, Atom name
294  string _atAltLoc (_line,(17-1),1); // char, Alternate location indicator
295  string _resName (_line,(18-1),4); // str, Residue name
296  string _chainID (_line,(22-1),1); // char, Chain identifier
297  string _resNum (_line,(23-1),4); // int, Residue sequence number
298  string _atICode (_line,(27-1),1); // char, Code for insertion of res
299  string _x (_line,(31-1),8); // float 8.3 ,x
300  string _y (_line,(39-1),8); // float 8.3 ,y
301  string _z (_line,(47-1),8); // float 8.3 ,z
302  string _atOccup (_line,(55-1),6); // float 6.2, Occupancy
303  string _atTFactor (_line,(61-1),6); // float 6.2, Temperature factor
304  string _segID (_line,(73-1),4); // str, Segment identifier
305  string _atElement (_line,(77-1),2); // str, Element symbol
306  string _atCharge (_line,(79-1),2); // str, Charge on the atom
307 
308  ba::trim(_recType);
309  ba::trim(_atNum);
310  ba::trim(_atName);
311  ba::trim(_atAltLoc);
312  ba::trim(_resName);
313  ba::trim(_chainID);
314  ba::trim(_resNum);
315  ba::trim(_atICode);
316  ba::trim(_x);
317  ba::trim(_y);
318  ba::trim(_z);
319  ba::trim(_atOccup);
320  ba::trim(_atTFactor);
321  ba::trim(_segID);
322  ba::trim(_atElement);
323  ba::trim(_atCharge);
324 
325  double _xd(0),_yd(0),_zd(0);
326  int _resNumInt(0);
327 
328  try
329  {
330  _xd = boost::lexical_cast<double>(_x);
331  _yd = boost::lexical_cast<double>(_y);
332  _zd = boost::lexical_cast<double>(_z);
333  _resNumInt = boost::lexical_cast<int>(_resNum);
334  }
335  catch(boost::bad_lexical_cast &)
336  {
337  error1( "... ... Can not convert PDB coord line!\n"
338  "... ... Atom number: " + _atNum + "\n"
339  "... ... Make sure this line is PDB style\n");
340  }
341 
342  tools::vec r(_xd , _yd , _zd);
343 
344  // set fragment
345  // reconnect to topology, molecule, segment
346  ctp::Fragment * _fragPtr = 0;
347  // make new frag for new res number
348  // otherwise use last created
349  if ( _newResNum != _resNumInt ){
350 
351  _newResNum = _resNumInt;
352  string _newResName = _resName+'_'+_resNum;
353 
354  // direct
355  _fragPtr = _topPtr->AddFragment(_newResName);
356  _molPtr->AddFragment(_fragPtr);
357  _segPtr->AddFragment(_fragPtr);
358  // inverse
359  _fragPtr->setTopology(_topPtr);
360  _fragPtr->setMolecule(_molPtr);
361  _fragPtr->setSegment(_segPtr);
362  }
363  else{
364  _fragPtr = _topPtr->Fragments().back();
365  }
366  if (_fragPtr==0) {error1("Zero pointer in GRO reader. Why?");}
367 
368  // set atom
369  // reconnect to topology, molecule, segment, fragment
370  ctp::Atom * _atmPtr = 0;
371  // direct
372  _atmPtr = _topPtr->AddAtom(_atName);
373  _molPtr->AddAtom(_atmPtr);
374  _segPtr->AddAtom(_atmPtr);
375  _fragPtr->AddAtom(_atmPtr);
376  // inverse
377  _atmPtr->setTopology(_topPtr);
378  _atmPtr->setMolecule(_molPtr);
379  _atmPtr->setSegment(_segPtr);
380  _atmPtr->setFragment(_fragPtr);
381 
382  _atmPtr->setResnr (_resNumInt);
383  _atmPtr->setResname (_resName);
384  _atmPtr->setPos (r);
385  }
386  }
387 
388  return;
