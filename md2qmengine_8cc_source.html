24 void Md2QmEngine::CheckMappingFile(tools::Property& topology_map) const {
25  std::string molkey = "topology.molecules.molecule";
26  std::vector<tools::Property*> molecules = topology_map.Select(molkey);
27  if (SameValueForMultipleEntries<std::string>(molecules, "mdname")) {
28  throw std::runtime_error("Multiple molecules have same mdname");
29  }
30  std::string segkey = "segments.segment";
31  std::vector<tools::Property*> segments_all;
32  for (tools::Property* mol : molecules) {
33  std::vector<tools::Property*> segments = mol->Select(segkey);
34  if (SameValueForMultipleEntries<std::string>(segments, "name")) {
35  throw std::runtime_error("Multiple segments in molecule:" +
36  mol->get("mdname").as<std::string>() +
37  " have same name");
38  }
39  segments_all.insert(segments_all.end(), segments.begin(), segments.end());
40  for (tools::Property* seg : segments) {
41  std::string fragkey = "fragments.fragment";
42  std::vector<tools::Property*> fragments = seg->Select(fragkey);
43  if (SameValueForMultipleEntries<std::string>(fragments, "name")) {
44  throw std::runtime_error(
45  "Multiple fragments have same name in molecule " +
46  mol->get("mdname").as<std::string>() + " segment " +
47  seg->get("name").as<std::string>());
48  }
49 
50  std::vector<std::string> atomnames_seg;
51  for (tools::Property* frag : fragments) {
52  std::string mdatoms = frag->get("mdatoms").as<std::string>();
53  tools::Tokenizer tok_md_atoms(mdatoms, " \t\n");
54  std::vector<std::string> atomnames = tok_md_atoms.ToVector();
55  atomnames_seg.insert(atomnames_seg.end(), atomnames.begin(),
56  atomnames.end());
57  }
58  std::sort(atomnames_seg.begin(), atomnames_seg.end()); // O(N log N)
59  if (adjacent_find(atomnames_seg.begin(), atomnames_seg.end()) !=
60  atomnames_seg.end()) {
61  throw std::runtime_error(
62  "Multiple mdatoms have same identifier in molecule " +
63  mol->get("mdname").as<std::string>() + " segment " +
64  seg->get("name").as<std::string>());
65  }
66  }
67  }
68  if (SameValueForMultipleEntries<std::string>(segments_all, "name")) {
69  throw std::runtime_error("Multiple segments have same name");
70  }
71 }

73 Index Md2QmEngine::DetermineAtomNumOffset(
74  const csg::Molecule* mol, const std::vector<Index>& atom_ids_map) const {
75  std::vector<Index> IDs;
76  IDs.reserve(mol->BeadCount());
77  for (const csg::Bead* bead : mol->Beads()) {
78  IDs.push_back(bead->getId());
79  }
80  std::sort(IDs.begin(), IDs.end());
81  Index offset = IDs[0] - atom_ids_map[0];
82  for (Index i = 1; i < Index(IDs.size()); i++) {
83  if (IDs[i] - atom_ids_map[i] != offset) {
84  throw std::runtime_error(
85  "AtomIds offset could not be determined, either our MD trajectory or "
86  "your mapping file have wrong Atom ids");
87  }
88  }
89  return offset;
90 }

92 bool Md2QmEngine::CheckMolWhole(const Topology& top, const Segment& seg) const {
93  Eigen::Vector3d CoM = seg.getPos();
94  bool whole = true;
95  for (const Atom& a : seg) {
96  Eigen::Vector3d r = a.getPos() - CoM;
97  Eigen::Vector3d r_pbc = top.PbShortestConnect(CoM, a.getPos());
98  Eigen::Vector3d shift = r_pbc - r;
99  if (shift.norm() > 1e-9) {
100  whole = false;
101  break;
102  }
103  }
104  return whole;
105 }

107 void Md2QmEngine::MakeSegmentsWholePBC(Topology& top) const {
108  for (Segment& seg : top.Segments()) {
109  seg.calcPos();
110  while (!CheckMolWhole(top, seg)) {
111  Eigen::Vector3d CoM = seg.getPos();
112  for (Atom& a : seg) {
113  Eigen::Vector3d r = a.getPos() - CoM;
114  Eigen::Vector3d r_pbc = top.PbShortestConnect(CoM, a.getPos());
115  Eigen::Vector3d shift = r_pbc - r;
116  if (shift.norm() > 1e-9) {
117  a.Translate(shift);
118  }
119  }
120  seg.calcPos();
121  }
122  }
123 }

125 Topology Md2QmEngine::map(const csg::Topology& top) const {
126 
127  tools::Property topology_map;
128  topology_map.LoadFromXML(_mapfile);
129  CheckMappingFile(topology_map);
130  Topology xtptop;
131  xtptop.setStep(top.getStep());
132  xtptop.setTime(top.getTime());
133  xtptop.setBox(top.getBox() * tools::conv::nm2bohr, top.getBoxType());
134 
135  // which segmentname does an atom belong to molname atomid
136  std::map<std::string, std::map<Index, std::string> > MolToSegMap;
137 
138  // which atomids belong to molname
139  std::map<std::string, std::vector<Index> > MolToAtomIds;
140 
141  // names of segments in one molecule;
142  std::map<std::string, std::vector<std::string> > SegsinMol;
143 
144  std::string molkey = "topology.molecules.molecule";
145  std::vector<tools::Property*> molecules = topology_map.Select(molkey);
146  std::string segkey = "segments.segment";
147  for (tools::Property* mol : molecules) {
148  std::string molname = mol->get("mdname").as<std::string>();
149  std::vector<tools::Property*> segments = mol->Select(segkey);
150  std::vector<std::string> segnames;
151  std::vector<Index> atomids;
152  for (tools::Property* seg : segments) {
153  std::string segname = seg->get("name").as<std::string>();
154  segnames.push_back(segname);
155  std::string fragkey = "fragments.fragment";
156  std::vector<tools::Property*> fragments = seg->Select(fragkey);
157  for (tools::Property* frag : fragments) {
158  std::string mdatoms = frag->get("mdatoms").as<std::string>();
159  tools::Tokenizer tok_md_atoms(mdatoms, " \t\n");
160  std::vector<std::string> atomnames;
161  tok_md_atoms.ToVector(atomnames);
162  for (const std::string& atomname : atomnames) {
163  tools::Tokenizer tok_atom_name(atomname, ":");
164  std::vector<std::string> entries = tok_atom_name.ToVector();
165  if (entries.size() != 3) {
166  throw std::runtime_error("Atom entry " + atomname +
167  " is not well formatted");
168  }
169  // format should be RESNUM:ATOMNAME:ATOMID we do not care about the
170  // first two
171  Index atomid = 0;
172  try {
173  atomid = std::stoi(entries[2]);
174  } catch (std::invalid_argument& e) {
175  throw std::runtime_error("Atom entry " + atomname +
176  " is not well formatted");
177  }
178  atomids.push_back(atomid);
179  MolToSegMap[molname][atomid] = segname;
180  }
181  }
182  }
183  std::sort(atomids.begin(), atomids.end());
184  MolToAtomIds[molname] = atomids;
185  SegsinMol[molname] = segnames;
186  }
187 
188  for (const csg::Molecule* mol : top.Molecules()) {
189  const std::vector<std::string> segnames = SegsinMol[mol->getName()];
190 
191  std::map<std::string, Segment*> segments; // we first add them to topology
192  // and then modify them via
193  // pointers;
194  for (const std::string& segname : segnames) {
195  segments[segname] = &xtptop.AddSegment(segname);
196  segments[segname]->AddMoleculeId(mol->getId());
197  }
198 
199  Index IdOffset = DetermineAtomNumOffset(mol, MolToAtomIds[mol->getName()]);
200  for (const csg::Bead* bead : mol->Beads()) {
201  Segment* seg =
202  segments[MolToSegMap[mol->getName()][bead->getId() - IdOffset]];
203 
204  Atom atom(bead->getResnr(), bead->getName(), bead->getId(),
205  bead->getPos() * tools::conv::nm2bohr, bead->getType());
206  seg->push_back(atom);
207  }
208  }
209 
210  MakeSegmentsWholePBC(xtptop);
211 
212  return xtptop;
213 }

215 template <class T>
216 bool Md2QmEngine::SameValueForMultipleEntries(
217  const std::vector<tools::Property*>& props, std::string valuetag) const {
218  std::vector<T> entries;
219  for (tools::Property* prop : props) {
220  entries.push_back(prop->get(valuetag).as<T>());
221  }
222  std::sort(entries.begin(), entries.end()); // O(N log N)
223  return adjacent_find(entries.begin(), entries.end()) != entries.end();
224 }
