27 Segment::Segment(int id, string name)
28  : _id(id), _name(name),
29  _has_e(false), _has_h(false) { _eMpoles.resize(3); }
30 
31 // This constructor creates a copy of the stencil segment, without
32 // adding it to any containers further up in the hierarchy; i.e. the topology
33 // and molecules will neither know about the existence of this segment, nor
34 // be able to access it. Used for creating the ghost in PB corrected pairs.
35 Segment::Segment(Segment *stencil)
36  : _id(stencil->getId()), _name(stencil->getName()+"_ghost"),
37  _typ(stencil->getType()), _top(NULL), _mol(NULL),
38  _CoM(stencil->getPos()),
39  _has_e(false), _has_h(false) { _eMpoles.resize(3);
40 
41  vector<Fragment*> ::iterator fit;
42  for (fit = stencil->Fragments().begin();
43  fit < stencil->Fragments().end();
44  fit++) {
45 
46  Fragment *newFrag = new Fragment(*fit);
47  this->AddFragment(newFrag);
48 
49  vector<Atom*> ::iterator ait;
50  for (ait = newFrag->Atoms().begin();
51  ait < newFrag->Atoms().end();
52  ait++) {
53  this->AddAtom(*ait);
54  }
55  }
56 }
