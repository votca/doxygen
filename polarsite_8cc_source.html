31 PolarSite::PolarSite(Index id, std::string element, Eigen::Vector3d pos)
32  : StaticSite(id, element, pos) {
33  tools::Elements e;
34  double default_pol = std::pow(tools::conv::ang2bohr, 3);
35  try {
36  default_pol =
37  e.getPolarizability(element) * std::pow(tools::conv::nm2bohr, 3);
38  } catch (const std::runtime_error&) {
39  ;
40  }
41  setPolarisation(default_pol * Eigen::Matrix3d::Identity());
42 }

52 void PolarSite::setPolarisation(const Eigen::Matrix3d& pol) {
53  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
54  es.computeDirect(pol);
55  _pinv = es.eigenvectors() * es.eigenvalues().cwiseInverse().asDiagonal() *
56  es.eigenvectors().transpose();
57  _eigendamp_invsqrt = 1.0 / std::sqrt(es.eigenvalues().maxCoeff());
58 }

60 std::string PolarSite::writePolarisation() const {
61  double conv_pol = std::pow(tools::conv::bohr2ang, 3);
62  Eigen::MatrixX3d pol = _pinv.inverse() * conv_pol;
63  return (boost::format(" P %1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f %5$+1.7f "
64  "%6$+1.7f\n") %
65  pol(0, 0) % pol(1, 0) % pol(2, 0) % pol(1, 1) % pol(1, 2) % pol(2, 2))
66  .str();
67 }

124  d.Q21s = _Q[6];
125  d.Q22c = _Q[7];
126  d.Q22s = _Q[8];
127 
128  d.Vx = _V[0];
129  d.Vy = _V[1];
130  d.Vz = _V[2];
131 
132  d.Vx_noE = _V_noE[0];
133  d.Vy_noE = _V_noE[1];
134  d.Vz_noE = _V_noE[2];
135 
136  Eigen::Matrix3d P = _pinv.inverse();

139  d.pxz = P(0, 2);

141  d.pyz = P(1, 2);
142  d.pzz = P(2, 2);
143 
144  d.d_x_ind = _induced_dipole.x();

146  d.d_z_ind = _induced_dipole.z();
