26 void AOMatrix3D::Fill(const AOBasis& aobasis) {
27  for (Index i = 0; i < 3; i++) {
28  _aomatrix[i] =
29  Eigen::MatrixXd::Zero(aobasis.AOBasisSize(), aobasis.AOBasisSize());
30  }
31  // loop row
32 #pragma omp parallel for
33  for (Index row = 0; row < aobasis.getNumofShells(); row++) {
34  const AOShell& shell_row = aobasis.getShell(row);
35  Index row_start = shell_row.getStartIndex();
36  // loop column
37  for (const AOShell& shell_col : aobasis) {
38  // figure out the submatrix
39  Index col_start = shell_col.getStartIndex();
40  std::vector<Eigen::Block<Eigen::MatrixXd> > submatrix;
41  for (Index i = 0; i < 3; i++) {
42  Eigen::Block<Eigen::MatrixXd> block =
43  _aomatrix[i].block(row_start, col_start, shell_row.getNumFunc(),
44  shell_col.getNumFunc());
45  submatrix.push_back(block);
46  }
47  // Fill block
48  FillBlock(submatrix, shell_row, shell_col);
49  }
50  }
51  return;
52 }
