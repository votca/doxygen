75 void IAnalyze::IHist(Topology *top, int state) {
76 
77  QMNBList &nblist = top->NBList();
78  QMNBList::iterator nit;
79 
80 
81  double MIN = log10(nblist.front()->getJeff2(state));
82  double MAX = log10(nblist.front()->getJeff2(state));
83 
84  // Collect J2s from pairs
85  vector< double > J2s;
86  J2s.reserve(nblist.size());
87 
88  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
89  double test = (*nit)->getJeff2(state);
90  double J2;
91  // Check if coupling constant is zero, if yes it is skipped from the histogramm.
92  if (test==0.0){
93  J2=-300;
94  int id=(*nit)->getId();
95  cout <<"\r\n....Jeff2 of pair " <<id << " is zero, skipping it."<<endl;
96 
97  }
98  else{
99  J2=log10(test);
100 
101  MIN = (J2 < MIN) ? J2 : MIN;
102  MAX = (J2 > MAX) ? J2 : MAX;
103 
104  J2s.push_back(J2);
105  }
106  }
107 
108  // Prepare bins
109  int BIN = ( (MAX-MIN)/_resolution_logJ2 + 0.5 ) + 1;
110 
111  vector< vector<double> > histJ2;
112 
113  histJ2.resize(BIN);
114 
115  // Execute binning
116  vector< double > ::iterator jit;
117  for (jit = J2s.begin(); jit < J2s.end(); ++jit) {
118 
119  int bin = int( (*jit-MIN)/_resolution_logJ2 + 0.5 );
120  histJ2[bin].push_back(*jit);
121  }
122 
123  vector< int > histN;
124  histN.resize(BIN);
125  for (int bin = 0; bin < BIN; ++bin) {
126  histN[bin] = histJ2[bin].size();
127  }
128  FILE *out;
129  string tag = boost::lexical_cast<string>("ianalyze.ihist_") + ( (state == -1) ? "e" : "h" ) + ".out";
130  out = fopen(tag.c_str(), "w");
131 
132  fprintf(out, "# IANALYZE: PAIR-INTEGRAL J2 HISTOGRAM\n");
133  fprintf(out, "# STATE %1d\n", state);
134 
135  for (int bin = 0; bin < BIN; ++bin) {
136  double J2 = MIN + bin*_resolution_logJ2;
137  fprintf(out, "%4.7f %4d \n", J2, histN[bin]);
138  }
139  fclose(out);
140 }

144 }}
