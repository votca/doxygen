77 std::vector<std::string> ECPAOBasis::Fill(const ECPBasisSet& bs,
78  QMMolecule& atoms) {
79  _AOBasisSize = 0;
80  _aoshells.clear();
81  _ncore_perAtom.clear();
82  _name = bs.Name();
83  std::vector<std::string> non_ecp_elements;
84  Index index = 0;
85  for (QMAtom& atom : atoms) {
86  std::string name = atom.getElement();
87 
88  bool element_exists = true;
89 
90  try {
91  bs.getElement(name);
92  } catch (std::runtime_error& error) {
93  element_exists = false;
94 
95  if (std::find(non_ecp_elements.begin(), non_ecp_elements.end(), name) !=
96  non_ecp_elements.end()) {
97  non_ecp_elements.push_back(name);
98  }
99  }
100 
101  if (element_exists) {
102  const ECPElement& element = bs.getElement(name);
103  _ncore_perAtom.push_back(element.getNcore());
104  L lmax = element.getLmax();
105  for (const ECPShell& shell : element) {
106  ECPAOShell& aoshell = addShell(shell, atom, _AOBasisSize, lmax);
107  index++;
108  _AOBasisSize += NumFuncShell(shell.getL());
109  for (const ECPGaussianPrimitive& gaussian : shell) {
110  aoshell.addGaussian(gaussian);
111  }
112  }
113  } else {
114  _ncore_perAtom.push_back(0);
115  }
116  }
117 
118  AddECPChargeToMolecule(atoms);
119  return non_ecp_elements;
120 }

122 void ECPAOBasis::WriteToCpt(CheckpointWriter& w) const {
123  w(_name, "name");
124  w(_AOBasisSize, "basissize");
125 
126  w(_ncore_perAtom, "atomic ecp charges");
127  Index numofprimitives = 0;
128  for (const auto& shell : _aoshells) {
129  numofprimitives += shell.getSize();

131 
132  CptTable table =
133  w.openTable<ECPAOGaussianPrimitive>("Contractions", numofprimitives);
134 
135  std::vector<ECPAOGaussianPrimitive::data> dataVec(numofprimitives);
136  Index i = 0;
137  for (const auto& shell : _aoshells) {
138  for (const auto& gaussian : shell) {
139  gaussian.WriteData(dataVec[i], shell);
140  i++;
141  }
142  }
143 
144  table.write(dataVec);
145 }

147 void ECPAOBasis::ReadFromCpt(CheckpointReader& r) {
148  clear();
149  r(_name, "name");
150  r(_AOBasisSize, "basissize");
151 
152  if (_AOBasisSize > 0) {
153  r(_ncore_perAtom, "atomic ecp charges");
154 
155  CptTable table = r.openTable<ECPAOGaussianPrimitive>("Contractions");
156  std::vector<ECPAOGaussianPrimitive::data> dataVec(table.numRows());
157  table.read(dataVec);
158  Index laststartindex = -1;
159  for (std::size_t i = 0; i < table.numRows(); ++i) {
160  if (dataVec[i].startindex != laststartindex) {
161  _aoshells.push_back(ECPAOShell(dataVec[i]));
162  laststartindex = dataVec[i].startindex;
163  } else {
164  _aoshells.back()._gaussians.push_back(
165  ECPAOGaussianPrimitive(dataVec[i]));
166  }
167  }
168  }
169 }
