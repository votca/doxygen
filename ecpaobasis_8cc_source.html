33 std::vector<std::string> ECPAOBasis::Fill(const ECPBasisSet& bs,
34  QMMolecule& atoms) {
35  _AOBasisSize = 0;
36  _aoshells.clear();
37  _shells_perAtom.clear();
38  std::vector<std::vector<Index>> shellindex_per_atom;
39 
40  std::vector<std::string> non_ecp_elements;
41  Index index = 0;
42  for (QMAtom& atom : atoms) {
43  std::string name = atom.getElement();
44 
45  bool element_exists = true;
46  std::vector<Index> shellindex;
47 
48  try {
49  bs.getElement(name);
50  } catch (std::runtime_error& error) {
51  element_exists = false;
52  if (std::find(non_ecp_elements.begin(), non_ecp_elements.end(), name) !=
53  non_ecp_elements.end()) {
54  non_ecp_elements.push_back(name);
55  }
56  }
57 
58  if (element_exists) {
59  const ECPElement& element = bs.getElement(name);
60  atom._ecpcharge = element.getNcore();
61  Index lmax = element.getLmax();
62  for (const ECPShell& shell : element) {
63  ECPAOShell& aoshell = addShell(shell, atom, _AOBasisSize, lmax);
64  shellindex.push_back(index);
65  index++;
66  _AOBasisSize += NumFuncShell(shell.getType());
67  for (const ECPGaussianPrimitive& gaussian : shell) {
68  aoshell.addGaussian(gaussian);
69  }
70  }
71  }
72  shellindex_per_atom.push_back(shellindex);
73  }
74 
75  // have to do it via indeces because if _aoshells resizes pointers are
76  // invalidated
77  for (const auto& atom_index : shellindex_per_atom) {
78  std::vector<const ECPAOShell*> temp;
79  for (Index lindex : atom_index) {
80  temp.push_back(&_aoshells[lindex]);
81  }
82  _shells_perAtom.push_back(temp);
83  }
84 
85  return non_ecp_elements;
86 }
