49 void Profile::Initialize(Property *options) {
50 
51  // update options with the VOTCASHARE defaults
52  UpdateWithDefaults( options );
53 
54  string key = "options." + Identify();
55 
56  // GEOMETRICAL OPTIONS
57  // axis along which to calculate profiles
58  _axis = options->get(key+".axis.direction").as< vec >();
59  // spatial resolution of the profile
60  _resolution = options->get(key+".axis.bin").as< double >();
61  // do binning automatically
62  int autobin = options->get(key+".axis.auto").as< int >();
63  _auto_bin = (autobin == 1) ? true : false;
64  // min and max for manual binning
65  if (!_auto_bin) {
66  _min = options->get(key+"axis.min").as< double >();
67  _max = options->get(key+"axis.max").as< double >();
68  }
69  // Normalize axis
70  _axis = _axis / sqrt(_axis*_axis);
71 
72  // OUTPUT OPTIONS
73  // output file for the density profile
74  _outfile = options->get(key+".output.density").as< string >();
75  // output file for the site energy profile
76  _outfile_EA_IP = options->get(key+".output.energy").as< string >();
77 
78 
79  // PROPERTY OPTIONS
80  // to use segments or atoms centers of mass
81  _density_type = options->get(key+".particles.type").as< string >();
82  // limit statistics to the first:last range
83  _firstSegId = options->get(key+".particles.first").as<int>();
84  _lastSegId = options->get(key+".particles.last").as<int>();
85 
86 }

90 bool Profile::EvaluateFrame(Topology *top) {
91 
92  map< string, vector< double > > map_seg_zs; // for atomistic number density
93  map< string, vector< double > > map_com_zs; // for segment number density
94  map< string, vector< Segment* > > map_com_seg;
95  map< string, bool > set_seg;
96 
97  vector< Segment* > ::iterator sit;
98  vector< Atom* > ::iterator ait;
99 
100  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
101  // Collect profile positions from atoms in system, order by segment name //
102  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
103 
104  double MAX = -1e100;
105  double MIN = 1e100;
106  //double RES = _resolution;
107 
108  for (sit = top->Segments().begin();
109  sit < top->Segments().end();
110  ++sit) {
111 
112  Segment *seg = *sit;
113 
114  // Within specified segment range?
115  if (seg->getId() < _firstSegId) { continue; }
116  if (seg->getId() == _lastSegId+1) { break; }
117 
118  if (!set_seg.count(seg->getName())) { set_seg[seg->getName()] = true; }
119 
120  double z_com = seg->getPos() * _axis;
121 
122  map_com_zs[seg->getName()].push_back(z_com);
123  map_com_seg[seg->getName()].push_back(seg);
124 
125  // Include this segment in density profile?
126  // if (!wildcmp(seg->getName().c_str(),_seg_pattern.c_str())) {
127  // continue;
128  // }
129 
130  for (ait = seg->Atoms().begin();
131  ait < seg->Atoms().end();
132  ++ait) {
133 
134  double z = (*ait)->getPos() * _axis;
135 
136  MAX = (z > MAX) ? z : MAX;
137  MIN = (z < MIN) ? z : MIN;
138 
139  map_seg_zs[seg->getName()].push_back(z);
140  }
141  }
142 
143  if (!_auto_bin) {
144  MAX = _max;
145  MIN = _min;
146  }
147 
148  int BIN = int( (MAX-MIN)/_resolution + 0.5 ) + 1;
149  int SEG = set_seg.size();
150 
151  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
152  // Calculate density profile from z-list, segment-name based //
153  // +++++++++++++++++++++++++++++++++++++++++++++++++++++++++ //
154 
155  vector< vector< int > > seg_hist_Ns;
156  vector< vector< int > > com_hist_Ns;
157  vector< vector< double > > com_profile_ea;
158  vector< vector< double > > com_profile_ec;
159  vector< vector< double > > com_profile_ea_std;
160  vector< vector< double > > com_profile_ec_std;
161  map< string, bool > ::iterator setit;
162 
163  for (setit = set_seg.begin(); setit != set_seg.end(); ++setit) {
164 
165  // Retrieve appropriate z list
166  string segName = setit->first;
167  vector< double > seg_zs = map_seg_zs[segName];
168  vector< double > com_zs = map_com_zs[segName];
169  vector< Segment* > com_seg = map_com_seg[segName];
170  vector< vector< double > > hist_zs;
171  vector< vector< double > > hist_zs_com;
172  vector< vector< double > > binned_ea; // anion site energies (<> EA)
173  vector< vector< double > > binned_ec; // cation site energies (<> IP)
174  hist_zs.resize(BIN);
175  hist_zs_com.resize(BIN);
176  binned_ea.resize(BIN);
177  binned_ec.resize(BIN);
178 
179  // Perform binning
180  vector< double > ::iterator zit;
181  vector< Segment* > ::iterator sit;
182  for (zit = seg_zs.begin(); zit < seg_zs.end(); ++zit) {
183 
184  int bin = int( ((*zit)-MIN)/_resolution + 0.5 );
185  hist_zs[bin].push_back((*zit));
186  }
187 
188  for (zit = com_zs.begin(), sit = com_seg.begin();
189  zit < com_zs.end();
190  ++zit, ++sit) {
191 
192  int bin = int( ((*zit)-MIN)/_resolution + 0.5 );
193  hist_zs_com[bin].push_back((*zit));
194  binned_ea[bin].push_back((*sit)->getSiteEnergy(-1));
195  binned_ec[bin].push_back((*sit)->getSiteEnergy(+1));
196  }
197 
198  // Reduce bins
199  vector< int > hist_Ns;
200  vector< int > hist_Ns_com;
201  vector< double > binned_avg_ea;
202  vector< double > binned_avg_ec;
203  vector< double > binned_std_ea;
204  vector< double > binned_std_ec;
205  hist_Ns.resize(BIN);
206  hist_Ns_com.resize(BIN);
207  binned_avg_ea.resize(BIN);
208  binned_avg_ec.resize(BIN);
209  binned_std_ea.resize(BIN);
210  binned_std_ec.resize(BIN);
211  // Atomistic number density
212  for (int bin = 0; bin < BIN; ++bin) {
213  hist_Ns[bin] = hist_zs[bin].size();
214  }
215  // Segment-based number density
216  for (int bin = 0; bin < BIN; ++bin) {
217  hist_Ns_com[bin] = hist_zs_com[bin].size();
218  }
219  // IP and EA
220  for (int bin = 0; bin < BIN; ++bin) {
221  double avg_ea = 0.0;
222  double avg_ec = 0.0;
223 
224  for (unsigned int entry = 0; entry < binned_ea[bin].size(); ++entry) {
225  avg_ea += binned_ea[bin][entry];
226  avg_ec += binned_ec[bin][entry];
227  }
228  avg_ea /= binned_ea[bin].size();
229  avg_ec /= binned_ec[bin].size();
230 
231  binned_avg_ea[bin] = avg_ea;
232  binned_avg_ec[bin] = avg_ec;
233  }
234 
235  // STD DEVIATION WITHIN LAYERS ( = LOCAL WIDTH DOS)
236  for (int bin = 0; bin < BIN; ++bin) {
237  double std_ea = 0.0;
238  double std_ec = 0.0;
239  for (unsigned int entry = 0; entry < binned_ea[bin].size(); ++entry) {
240  double avg_ea = binned_avg_ea[bin];
241  double avg_ec = binned_avg_ec[bin];
242  std_ea += (binned_ea[bin][entry] - avg_ea)
243  *(binned_ea[bin][entry] - avg_ea);
244  std_ec += (binned_ec[bin][entry] - avg_ec)
245  *(binned_ec[bin][entry] - avg_ec);
246  }
247  std_ea /= binned_ea[bin].size();
248  std_ec /= binned_ec[bin].size();
249  std_ea = sqrt(std_ea);
250  std_ec = sqrt(std_ec);
251 
252  binned_std_ea[bin] = std_ea;
253  binned_std_ec[bin] = std_ec;
254  }
255 
256  seg_hist_Ns.push_back(hist_Ns);
257  com_hist_Ns.push_back(hist_Ns_com);
258  com_profile_ea.push_back(binned_avg_ea);
259  com_profile_ec.push_back(binned_avg_ec);
260  com_profile_ea_std.push_back(binned_std_ea);
261  com_profile_ec_std.push_back(binned_std_ec);
262  }
263 
264 
265  // +++++++++++++++++++++++ //
266  // Output number densities //
267  // +++++++++++++++++++++++ //
268 
269  FILE *out;
270  out = fopen(_outfile.c_str(), "w");
271 
272  fprintf(out, "# DENSITY PROFILE ALONG AXIS z = %4.7f %4.7f %4.7f \n",
273  _axis.getX(), _axis.getY(), _axis.getZ());
274  fprintf(out, "# MIN z %4.7f MAX z %4.7f \n", MIN, MAX);
275 
276 
277  fprintf(out, "# z");
278  for (setit = set_seg.begin(); setit != set_seg.end(); ++setit) {
279  fprintf(out, " N(%1s,z) ", setit->first.c_str());
280  }
281  fprintf(out, " N(z) \n");
282 
283  for (int bin = 0; bin < BIN; ++bin) {
284 
285  double z = MIN + bin*_resolution;
286  fprintf(out, "%4.7f", z);
287  int N_z = 0;
288  if (_density_type == "atoms") {
289  for (int s = 0; s < SEG; ++s) {
290  fprintf(out, " %7d ", seg_hist_Ns[s][bin]);
291  N_z += seg_hist_Ns[s][bin];
292  }
293  }
294  else if (_density_type == "segments") {
295  for (int s = 0; s < SEG; ++s) {
296  fprintf(out, " %7d ", com_hist_Ns[s][bin]);
297  N_z += com_hist_Ns[s][bin];
298  }
299  }
300  else {
301  cout << endl << "ERROR: Invalid density type " << _density_type
302  << ". Should be either atoms or segments" << flush;
303  break;
304  }
305 
306  fprintf(out, " %7d \n", N_z);
307  }
308 
309  fclose(out);
310 
311 
312  // ++++++++++++++++++++++++++++++++ //
313  // Output spatially-resolved EA, IP //
314  // ++++++++++++++++++++++++++++++++ //
315 
316  out = fopen(_outfile_EA_IP.c_str(), "w");
317 
318  fprintf(out, "# ENERGY LANDSCAPE ALONG AXIS z = %4.7f %4.7f %4.7f \n",
319  _axis.getX(), _axis.getY(), _axis.getZ());
320  fprintf(out, "# MIN z %4.7f MAX z %4.7f \n", MIN, MAX);
321 
322  fprintf(out, "# z");
323  for (setit = set_seg.begin(); setit != set_seg.end(); ++setit) {
324  fprintf(out, " EA(%1s,z) +/- ", setit->first.c_str());
325  fprintf(out, " IP(%1s,z) +/- ", setit->first.c_str());
326  }
327  fprintf(out, " \n");
328 
329  for (int bin = 0; bin < BIN; ++bin) {
330 
331  double z = MIN + bin*_resolution;
332  fprintf(out, "%4.7f", z);
333 
334  for (int s = 0; s < SEG; ++s) {
335  fprintf(out, " %4.7f ", com_profile_ea[s][bin]);
336  fprintf(out, " %4.7f ", com_profile_ec[s][bin]);
337  fprintf(out, " %4.7f ", com_profile_ea_std[s][bin]);
338  fprintf(out, " %4.7f ", com_profile_ec_std[s][bin]);
339  }
340 
341  fprintf(out, " \n");
342  }
343  fclose(out);
344 
345  return true;
346 }
