54 void SqlApplication::Run() {
55 
56  std::string name = ProgramName();
57  if (VersionString() != "") name = name + ", version " + VersionString();
58  votca::xtp::HelpTextHeader(name);
59  // EVALUATE OPTIONS
60  int nThreads = OptionsMap()["nthreads"].as<int>();
61  int nframes = OptionsMap()["nframes"].as<int>();
62  int fframe = OptionsMap()["first-frame"].as<int>();
63  if (fframe-- == 0)
64  throw runtime_error(
65  "ERROR: First frame is 0, counting "
66  "in VOTCA::XTP starts from 1.");
67  int save = OptionsMap()["save"].as<int>();
68 
69  // STATESAVER & PROGRESS OBSERVER
70  string statefile = OptionsMap()["file"].as<string>();
71  StateSaverSQLite statsav;
72  statsav.Open(_top, statefile);
73 
74  // INITIALIZE & RUN CALCULATORS
75  cout << "Initializing calculators " << endl;
76  BeginEvaluate(nThreads);
77 
78  int frameId = -1;
79  int framesDone = 0;
80  while (statsav.NextFrame() && framesDone < nframes) {
81  frameId += 1;
82  if (frameId < fframe) continue;
83  cout << "Evaluating frame " << _top.getDatabaseId() << endl;
84  EvaluateFrame();
85  if (save == 1) {
86  statsav.WriteFrame();
87  } else {
88  cout << "Changes have not been written to state file." << endl;
89  }
90  framesDone += 1;
91  }
92 
93  if (framesDone == 0)
94  cout << "Input requires first frame = " << fframe + 1
95  << ", # frames = " << nframes << " => No frames processed.";
96 
97  statsav.Close();
98  EndEvaluate();
99 }
