71 void Bulk::Initialize(const char *filename, Property *options, const char *outputfile)
72 {
73 
74  std::cout << "===================================================" << "\n";
75  std::cout << "= Initialization phase =" << "\n";
76  std::cout << "===================================================" << "\n";
77 
78  eventinfo = new Eventinfo();
79  eventinfo->device = false;
80  eventinfo->Read_bulk(options);
81 
82  //Setup random number generator
83  srand(eventinfo->random_seed); // srand expects any integer in order to initialise the random number generator
84  randomvariable = new votca::tools::Random2();
85  randomvariable->init(rand(), rand(), rand(), rand());
86 
87  graph = new GraphKMC();
88  graph->Initialize_sql(filename);
89  std::cout << "number of nodes before graph manipulations: " << graph->Numberofnodes() << "\n";
90  std::cout << "simulation box size before graph manipulations: " << graph->Determine_Sim_Box_Size() << "\n";
91 
92  graph->Setup_bulk_graph(eventinfo);
93 
94  eventinfo->Graph_Parameters(graph->hopdist(), graph->mindist(), graph->simboxsize(), graph->maxpairdegree(),graph->Average_hole_node_energy(), graph->stddev_hole_node_energy(), graph->Average_electron_node_energy(), graph-> Hole_inject_reorg(), graph->Electron_inject_reorg());
95 
96  std::cout << "graph object initialized" << "\n";
97  std::cout << "max pair degree: " << graph->maxpairdegree() << "\n";
98  std::cout << "hopping distance: " << graph->hopdist() << "\n";
99  std::cout << "simulation box size: " << graph->simboxsize() << "\n";
100  std::cout << "number of nodes " << graph->Numberofnodes() << "\n";
101 
102  state = new StateReservoir();
103  state->InitStateReservoir();
104  std::cout << "state reservoir object initialized" << "\n";
105 
106  site_inject_probs = new Bsumtree();
107  site_inject_probs->initialize(graph->Numberofnodes());
108  std::cout << "Random node injector tree initialized" << "\n";
109 
110  //Random charge distribution
111  int nrholes = 0;
112  int nrelectrons = 0;
113 
114  nrholes = ceil(eventinfo->hole_density*graph->Numberofnodes());
115  nrelectrons = ceil(eventinfo->electron_density*graph->Numberofnodes());
116 
117  if(nrholes!=0 && nrelectrons != 0) { std::cout << "Double carrier simulation" << "\n";}
118  else { std::cout << "Single carrier simulation" << "\n";}
119  if(nrholes+nrelectrons>graph->Numberofnodes()){ std::cout<< "WARNING: number of electrons and holes is larger than the number of available nodes" << "\n";}
120  if(nrholes+nrelectrons == 0) { std::cout << "WARNING: without electrons and/or holes nothing happens";}
121 
122  state->Random_init_injection(nrelectrons, nrholes, site_inject_probs, graph, eventinfo, randomvariable);
123  std::cout << "randomly placed : " << nrelectrons + nrholes << " charges of which " << nrelectrons << " electrons and " << nrholes << " holes" << "\n";
124 
125 
126  if(state->ReservoirEmpty()) state->Grow(eventinfo->growsize, eventinfo->maxpairdegree);
127  std::cout << "state grown after random charge injection" << "\n";
128 
129 
130  non_injection_rates = new Bsumtree();
131  left_injection_rates = new Bsumtree();
132  right_injection_rates = new Bsumtree();
133  std::cout << "rate binary tree initialized" << "\n";
134 
135  longrange = new Longrange();
136 
137  events = new Events();
138  if(eventinfo->coulomb_strength > 0.0) {
139  events->Init_non_injection_meshes(eventinfo);
140  std::cout << "mesh structure for Coulomb interaction calculations initialized" << "\n";
141  }
142  events->Initialize_bulk_eventvector(graph,state,eventinfo);
143  std::cout << "Initialize (bulk) eventvector (setting number of injection events to zero)" << "\n";
144  events->Initialize_rates(non_injection_rates, left_injection_rates, right_injection_rates, eventinfo);
145  std::cout << "Fill rate binary trees" << "\n";
146  events->Initialize_after_charge_placement(graph,state, longrange, non_injection_rates, left_injection_rates, right_injection_rates, eventinfo);
147  std::cout << "Initialize event and rates object after initial placement of charges" << "\n";
148 
149  if(eventinfo->carrier_trajectory) {state->Init_trajectory(eventinfo->trajectory_filename); std::cout << "Trajectory file initialized" << "\n";}
150 
151  vssmgroup = new Vssmgroup();
152  std::cout << "vssm group initialized" << "\n";
153 
154  numoutput = new Numoutput();
155  numoutput->Initialize(eventinfo);
156  std::cout << "output object initialized" << "\n";
157 
158

184 void Bulk::RunKMC() {
185 
186 
187  // to check whether anti repeating methods are useful
188  if(eventinfo->repeat_counting) numoutput->Repeat_count_init();
189 
190  // convergence criteria
191  //bool direct_iv_convergence = false;
192  //bool direct_reco_convergence = false;
193  //int direct_iv_counter = 0; //if the convergence criterium is counted ten times in a row, result is converged
194  //int direct_reco_counter = 0;
195 
196  if(eventinfo->filament_visualisation) numoutput->Init_visualisation(graph, eventinfo);
197  if(eventinfo->carrier_trajectory) eventinfo->number_of_equilibration_steps = 0;
198 
199  std::cout << "total link z distance : " << graph->total_link_distance_z() << "\n";
200  std::cout << "average hole site energy : " << eventinfo->avholeenergy << "\n";
201  std::cout << "average electron site energy : " << eventinfo->avelectronenergy << "\n";
202  std::cout << "standard deviation (disorder strength) of hole site energies: " << graph->stddev_hole_node_energy() << "\n";
203  std::cout << "standard deviation (disorder strength) of electron site energies: " << graph->stddev_electron_node_energy() << "\n";
204  std::cout << "\n";
205  std::cout << "===================================================" << "\n";
206  std::cout << "= Start of bulk simulation =" << "\n";
207  std::cout << "===================================================" << "\n";
208  std::cout << "\n";
209 
210  sim_time = 0.0;
211  for (long it = 0; it < eventinfo->number_of_equilibration_steps + eventinfo->number_of_steps; it++) {
212 
213  vssmgroup->Recompute_bulk(non_injection_rates);
214 
215  double timestep = vssmgroup->Timestep(randomvariable);
216  sim_time += timestep;
217 
218  Event* chosenevent;
219  chosenevent = vssmgroup->Choose_event_bulk(events, non_injection_rates, randomvariable);
220 
221  if(eventinfo->filament_visualisation && it <= eventinfo->visualisation_at_nr_steps) numoutput->Update_visualisation(chosenevent);
222  if(eventinfo->filament_visualisation && it == eventinfo->visualisation_at_nr_steps) numoutput->Print_visualisation();
223 
224  if(eventinfo->carrier_trajectory && ldiv(it,eventinfo->number_of_trajectory_report_steps).rem == 0) state->Print_trajectory(sim_time);
225 
226  // check for direct repeats
227  if(eventinfo->repeat_counting) numoutput->Repeat_count_update(chosenevent);
228 
229  if(!eventinfo->carrier_trajectory) numoutput->Update(chosenevent, sim_time, timestep);
230  events->On_execute(chosenevent, graph, state, longrange, non_injection_rates, left_injection_rates, right_injection_rates, eventinfo);
231 
232  // equilibration
233 
234  if(!eventinfo->carrier_trajectory && it == eventinfo->number_of_equilibration_steps) {
235  numoutput->Init_convergence_check(sim_time);
236  numoutput->Initialize_equilibrate(eventinfo);
237  sim_time = 0.0;
238  }
239  // convergence checking
240 
241  if(!eventinfo->carrier_trajectory && (ldiv(it,eventinfo->number_of_report_steps ).rem==0 && it> eventinfo->number_of_equilibration_steps)) numoutput->Convergence_check(sim_time, eventinfo);
242 
243  // direct output
244  if(!eventinfo->carrier_trajectory && ldiv(it,eventinfo->number_of_report_steps ).rem==0){
245  numoutput->Write(it, sim_time, timestep, eventinfo);
246  std::cout << "\n";
247  }
248 
249  if(eventinfo->carrier_trajectory && ldiv(it,eventinfo->number_of_report_steps ).rem==0){
250  std::cout << "step " << it << " of " << eventinfo->number_of_equilibration_steps + eventinfo->number_of_steps << " timesteps" << "\n";
251  }
252 
253  // break out of loop
254  if(!eventinfo->carrier_trajectory && numoutput->iv_conv()) {break;}
255 
256  }
257 
258 }
