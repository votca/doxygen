44 // This function calculates and stores inverses of the microscopic dielectric
45 // matrix in a matrix vector
46 void ImaginaryAxisIntegration::CalcDielInvVector(
47  const RPA& rpa, const Eigen::MatrixXd& kDielMxInv_zero) {
48  _dielinv_matrices_r.resize(_gq->Order());
49 
50  for (Index j = 0; j < _gq->Order(); j++) {
51  double newpoint = _gq->ScaledPoint(j);
52  Eigen::MatrixXd eps_inv_j = rpa.calculate_epsilon_i(newpoint).inverse();
53  eps_inv_j.diagonal().array() -= 1.0;
54  _dielinv_matrices_r[j] =
55  -eps_inv_j +
56  kDielMxInv_zero * std::exp(-std::pow(_opt.alpha * newpoint, 2));
57  }
58 }

60 class FunctionEvaluation {
61  public:
62  FunctionEvaluation(const Eigen::MatrixXd& Imx, const Eigen::ArrayXcd& DeltaE,
63  const std::vector<Eigen::MatrixXd>& dielinv_matrices_r)
64  : _Imx(Imx), _DeltaE(DeltaE), _dielinv_matrices_r(dielinv_matrices_r){};
65 
66  double operator()(Index j, double point, bool symmetry) const {
67  Eigen::VectorXcd denominator;
68  const std::complex<double> cpoint(0.0, point);
69  if (symmetry) {
70  denominator =
71  (_DeltaE + cpoint).cwiseInverse() + (_DeltaE - cpoint).cwiseInverse();
72  } else {
73  denominator = (_DeltaE + cpoint).cwiseInverse();
74  }
75  return 0.5 / tools::conv::Pi *
76  ((_Imx * (_dielinv_matrices_r[j].conjugate()))
77  .cwiseProduct(denominator.asDiagonal() * _Imx))
78  .sum()
79  .real();
80  }
81 
82  private:
83  const Eigen::MatrixXd& _Imx;
84  const Eigen::ArrayXcd& _DeltaE;
85  const std::vector<Eigen::MatrixXd>& _dielinv_matrices_r;
86 };

88 double ImaginaryAxisIntegration::SigmaGQDiag(double frequency, Index gw_level,
89  double eta) const {
90  Index lumo = _opt.homo + 1;
91  const Index occ = lumo - _opt.rpamin;
92  const Index unocc = _opt.rpamax - _opt.homo;
93  Index gw_level_offset = gw_level + _opt.qpmin - _opt.rpamin;
94  const Eigen::MatrixXd& Imx = _Mmn[gw_level_offset];
95  Eigen::ArrayXcd DeltaE = frequency - _energies.array();
96  DeltaE.imag().head(occ) = eta;
97  DeltaE.imag().tail(unocc) = -eta;
98  FunctionEvaluation f = FunctionEvaluation(Imx, DeltaE, _dielinv_matrices_r);
99  return _gq->Integrate(f);
100 }
