45 // This function calculates and stores inverses of the microscopic dielectric
46 // matrix in a matrix vector
47 void ImaginaryAxisIntegration::CalcDielInvVector(
48  const RPA& rpa, const Eigen::MatrixXd& kDielMxInv_zero) {
49  dielinv_matrices_r_.resize(gq_->Order());
50 
51  for (Index j = 0; j < gq_->Order(); j++) {
52  double newpoint = gq_->ScaledPoint(j);
53  Eigen::MatrixXd eps_inv_j = rpa.calculate_epsilon_i(newpoint).inverse();
54  eps_inv_j.diagonal().array() -= 1.0;
55  dielinv_matrices_r_[j] =
56  -eps_inv_j +
57  kDielMxInv_zero * std::exp(-std::pow(opt_.alpha * newpoint, 2));
58  }
59 }

61 class FunctionEvaluation {
62  public:
63  FunctionEvaluation(const Eigen::MatrixXd& Imx, const Eigen::ArrayXcd& DeltaE,
64  const std::vector<Eigen::MatrixXd>& dielinv_matrices_r)
65  : Imx_(Imx), DeltaE_(DeltaE), dielinv_matrices_r_(dielinv_matrices_r){};
66 
67  double operator()(Index j, double point, bool symmetry) const {
68  Eigen::VectorXcd denominator;
69  const std::complex<double> cpoint(0.0, point);
70  if (symmetry) {
71  denominator =
72  (DeltaE_ + cpoint).cwiseInverse() + (DeltaE_ - cpoint).cwiseInverse();
73  } else {
74  denominator = (DeltaE_ + cpoint).cwiseInverse();
75  }
76  return 0.5 / tools::conv::Pi *
77  ((Imx_ * (dielinv_matrices_r_[j].conjugate()))
78  .cwiseProduct(denominator.asDiagonal() * Imx_))
79  .sum()
80  .real();
81  }
82 
83  private:
84  const Eigen::MatrixXd& Imx_;
85  const Eigen::ArrayXcd& DeltaE_;
86  const std::vector<Eigen::MatrixXd>& dielinv_matrices_r_;
87 };

89 double ImaginaryAxisIntegration::SigmaGQDiag(double frequency, Index gw_level,
90  double eta) const {
91  Index lumo = opt_.homo + 1;
92  const Index occ = lumo - opt_.rpamin;
93  const Index unocc = opt_.rpamax - opt_.homo;
94  Index gw_level_offset = gw_level + opt_.qpmin - opt_.rpamin;
95  const Eigen::MatrixXd& Imx = Mmn_[gw_level_offset];
96  Eigen::ArrayXcd DeltaE = frequency - energies_.array();
97  DeltaE.imag().head(occ) = eta;
98  DeltaE.imag().tail(unocc) = -eta;
99  FunctionEvaluation f(Imx, DeltaE, dielinv_matrices_r_);
100  return gq_->Integrate(f);
101 }
