00159 boost::program_options::options_description_easy_init
00160     Application::AddProgramOptions(const string &group)
00161 {
00162     // if no group is given, add it to standard options
00163     if(group == "")
00164         return _op_desc.add_options();
00165     
00166     // does group already exist, if yes, add it there
00167     std::map<string, boost::program_options::options_description>::iterator iter;
00168     iter = _op_groups.find(group);
00169     if(iter!=_op_groups.end())
00170         return iter->second.add_options();
00171 
00172     // no group with given name was found -> create group
00173     _op_groups.insert(make_pair(group, boost::program_options::options_description(group)));
00174 
00175     return _op_groups[group].add_options();
00176 }

00193     
00194     // parse the command line
00195     try {
00196         po::store(po::parse_command_line(argc, argv, _op_desc), _op_vm);
00197         po::notify(_op_vm);

00199     catch(boost::program_options::error err) {
00200         throw runtime_error(string("error parsing command line: ") + err.what());
