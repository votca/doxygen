109 int Application::Exec(int argc, char **argv)
110 {
111  try {
112  //_continue_execution = true;
113  AddProgramOptions()("help,h", " display this help and exit");
114  AddProgramOptions()("verbose,v", " be loud and noisy");
115  AddProgramOptions("Hidden")("man", " output man-formatted manual pages");
116  AddProgramOptions("Hidden")("tex", " output tex-formatted manual pages");
117 
118  Initialize(); // initialize program-specific parameters
119 
120  ParseCommandLine(argc, argv); // initialize general parameters & read input file
121 
122  if (_op_vm.count("verbose")) {
123  globals::verbose = true;
124  }
125 
126  if (_op_vm.count("man")) {
127  ShowManPage(cout);
128  return 0;
129  }
130 
131  if (_op_vm.count("tex")) {
132  ShowTEXPage(cout);
133  return 0;
134  }
135 
136  if (_op_vm.count("help")) {
137  ShowHelpText(cout);
138  return 0;
139  }
140 
141  if(!EvaluateOptions()) {
142  ShowHelpText(cout);
143  return -1;
144  }
145 
146  if(_continue_execution)
147  Run();
148  else cout << "nothing to be done - stopping here\n";
149  }
150  catch(std::exception &error) {
151  cerr << "an error occurred:\n" << error.what() << endl;
152  return -1;
153  }
154  return 0;
155 }

157 boost::program_options::options_description_easy_init
158  Application::AddProgramOptions(const string &group)
159 {
160  // if no group is given, add it to standard options
161  if(group == "")
162  return _op_desc.add_options();
163 
164  // does group already exist, if yes, add it there
165  std::map<string, boost::program_options::options_description>::iterator iter;
166  iter = _op_groups.find(group);
167  if(iter!=_op_groups.end())
168  return iter->second.add_options();
169 
170  // no group with given name was found -> create group
171  _op_groups.insert(make_pair(group, boost::program_options::options_description(group)));
172 
173  return _op_groups[group].add_options();
174 }

177 void Application::ParseCommandLine(int argc, char **argv)
178 {
179  namespace po = boost::program_options;
180 
181  std::map<string, boost::program_options::options_description>::iterator iter;
182 
183  // default options should be added to visible (the rest is handled via a map))
184  _visible_options.add(_op_desc);
185 
186  // add all categories to list of available options
187  for(iter=_op_groups.begin(); iter!=_op_groups.end(); ++iter) {
188  _op_desc.add(iter->second);
189  if ( iter->first != "Hidden" ) _visible_options.add(iter->second);
190  }
191 
192  // parse the command line
193  try {
194  po::store(po::parse_command_line(argc, argv, _op_desc), _op_vm);
195  po::notify(_op_vm);
196  }
197  catch(boost::program_options::error err) {
198  throw runtime_error(string("error parsing command line: ") + err.what());
199  }
200 }
