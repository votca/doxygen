111 int Application::Exec(int argc, char **argv)
112 {
113  try {
114  //_continue_execution = true;
115  AddProgramOptions()("help,h", " display this help and exit");
116  AddProgramOptions()("verbose,v", " be loud and noisy");
117  AddProgramOptions("Hidden")("man", " output man-formatted manual pages");
118  AddProgramOptions("Hidden")("tex", " output tex-formatted manual pages");
119 
120  Initialize(); // initialize program-specific parameters
121 
122  ParseCommandLine(argc, argv); // initialize general parameters & read input file
123 
124  if (_op_vm.count("verbose")) {
125  globals::verbose = true;
126  }
127 
128  if (_op_vm.count("man")) {
129  ShowManPage(cout);
130  return 0;
131  }
132 
133  if (_op_vm.count("tex")) {
134  ShowTEXPage(cout);
135  return 0;
136  }
137 
138  if (_op_vm.count("help")) {
139  ShowHelpText(cout);
140  return 0;
141  }
142 
143  if(!EvaluateOptions()) {
144  ShowHelpText(cout);
145  return -1;
146  }
147 
148  if(_continue_execution)
149  Run();
150  else cout << "nothing to be done - stopping here\n";
151  }
152  catch(std::exception &error) {
153  cerr << "an error occurred:\n" << error.what() << endl;
154  return -1;
155  }
156  return 0;
157 }

159 boost::program_options::options_description_easy_init
160  Application::AddProgramOptions(const string &group)
161 {
162  // if no group is given, add it to standard options
163  if(group == "")
164  return _op_desc.add_options();
165 
166  // does group already exist, if yes, add it there
167  std::map<string, boost::program_options::options_description>::iterator iter;
168  iter = _op_groups.find(group);
169  if(iter!=_op_groups.end())
170  return iter->second.add_options();
171 
172  // no group with given name was found -> create group
173  _op_groups.insert(make_pair(group, boost::program_options::options_description(group)));
174 
175  return _op_groups[group].add_options();
176 }

179 void Application::ParseCommandLine(int argc, char **argv)
180 {
181  namespace po = boost::program_options;
182 
183  std::map<string, boost::program_options::options_description>::iterator iter;
184 
185  // default options should be added to visible (the rest is handled via a map))
186  _visible_options.add(_op_desc);
187 
188  // add all categories to list of available options
189  for(iter=_op_groups.begin(); iter!=_op_groups.end(); ++iter) {
190  _op_desc.add(iter->second);
191  if ( iter->first != "Hidden" ) _visible_options.add(iter->second);
192  }
193 
194  // parse the command line
195  try {
196  po::store(po::parse_command_line(argc, argv, _op_desc), _op_vm);
197  po::notify(_op_vm);
198  }
199  catch(boost::program_options::error err) {
200  throw runtime_error(string("error parsing command line: ") + err.what());
201  }
202 }

212 void Application::PrintDescription(std::ostream &out, const string &calculator_name, const string help_path, HelpType help_type )
213 {
214  boost::format _format("%|3t|%1% %|20t|%2% \n");
215  string help_string;
216  boost::filesystem::path arg_path;
217  Property options;
218  string xmlFile = (arg_path / string(getenv("VOTCASHARE")) / help_path / (boost::format("%1%.%2%") % calculator_name % "xml").str()).string().c_str();
219 
220  // loading the documentation xml file from VOTCASHARE
221  char *votca_share = getenv("VOTCASHARE");
222  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
223 
224  try {
225 
226  load_property_from_xml(options, xmlFile);
227  Property &calculator_options = options.get("options." + calculator_name);
228  Property::AttributeIterator atr_it = calculator_options.findAttribute("help");
229 
230  if (atr_it != calculator_options.lastAttribute()) {
231  help_string = (*atr_it).second;
232  } else {
233  if (tools::globals::verbose) out << _format % calculator_name % "Undocumented";
234  return;
235  }
236 
237  switch (help_type) {
238  default:
239  break;
240  case HelpShort: // short description of the calculator
241  out << _format % calculator_name % help_string;
242  break;
243  case HelpLong:
244  votca::tools::PropertyIOManipulator iom(votca::tools::PropertyIOManipulator::HLP, 2, "");
245  out << iom << options;
246  break;
247  }
248 
249  } catch (std::exception &error) {
250  if (tools::globals::verbose) out << _format % calculator_name % "Undocumented";
251  }
252 
253 }
