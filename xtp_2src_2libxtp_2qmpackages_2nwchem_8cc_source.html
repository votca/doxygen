61  _basisset_name = options.get(key + ".basisset").as<std::string>();

65 
66  if (_write_pseudopotentials)

68

135 int NWChem::WriteBackgroundCharges(ofstream& nw_file) {
136 
137  int numberofcharges = 0;
138  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
139  for (std::shared_ptr<ctp::PolarSeg> seg : _PolarSegments) {
140  for (ctp::APolarSite* site : *seg) {
141  string sitestring = boost::str(
142  fmt % ((site->getPos().getX()) * votca::tools::conv::nm2ang) %
143  (site->getPos().getY() * votca::tools::conv::nm2ang) %
144  (site->getPos().getZ() * votca::tools::conv::nm2ang) %
145  site->getQ00());
146  if (site->getQ00() != 0.0) {
147  nw_file << sitestring << endl;
148  numberofcharges++;
149  }
150  if (site->getRank() > 0 || _with_polarization) {
151  std::vector<std::vector<double> > _split_multipoles =
152  SplitMultipoles(site);
153  for (const auto& mpoles : _split_multipoles) {
154  string multipole =
155  boost::str(fmt % mpoles[0] % mpoles[1] % mpoles[2] % mpoles[3]);
156  nw_file << multipole << endl;
157  numberofcharges++;
158  }
159  }
160  }
161  }
162  nw_file << endl;
163  return numberofcharges;
164 }

166 bool NWChem::WriteGuess(Orbitals& orbitals) {
167  ofstream orb_file;
168  std::string orb_file_name_full = _run_dir + "/" + _orb_file_name;
169  // get name of temporary ascii file and open it
170  std::vector<std::string> results;
171  boost::algorithm::split(results, _orb_file_name, boost::is_any_of("."),
172  boost::algorithm::token_compress_on);
173  std::string orb_file_name_ascii = _run_dir + "/" + results.front() + ".mos";
174  orb_file.open(orb_file_name_ascii.c_str());
175 
176  // header
177  orb_file << "#generated by VOTCA\nbasisum\ngeomsum\n\nscf\nFri Sep 13 "
178  "00:00:00 2013\nscf\n1\n\n8\nao basis\n1\n"
179  << flush;
180  int size_of_basis = (orbitals.MOEnergies()).size();
181  orb_file << size_of_basis << endl;
182  orb_file << size_of_basis << endl;
183  ReorderMOsBack(orbitals);
184  int level = 1;
185  int ncolumns = 3;
186  // write occupations as double in three columns
187  // occupied levels
188  int column = 1;
189  for (int i = 0; i < orbitals.getNumberOfAlphaElectrons(); i++) {
190  orb_file << FortranFormat(2.0);
191  if (column == ncolumns) {
192  orb_file << endl;
193  column = 0;
194  }
195  column++;
196  }
197  // unoccupied levels
198  for (int i = orbitals.getNumberOfAlphaElectrons(); i < size_of_basis; i++) {
199  orb_file << FortranFormat(0.0);
200  if (column == ncolumns) {
201  orb_file << endl;
202  column = 0;
203  }
204  column++;
205  }
206  // extra endl
207  if (column != 1) {
208  orb_file << endl;
209  }
210 
211  // write all energies in same format
212  column = 1;
213  for (int i = 0; i < orbitals.MOEnergies().size(); ++i) {
214  double energy = (orbitals.MOEnergies())[i];
215  orb_file << FortranFormat(energy);
216  if (column == ncolumns) {
217  orb_file << endl;
218  column = 0;
219  }
220  column++;
221  }
222  if (column != 1) orb_file << endl;
223 
224  // write coefficients in same format
225  for (int i = 0; i < orbitals.MOCoefficients().cols(); ++i) {
226  Eigen::VectorXd mr = orbitals.MOCoefficients().col(i);
227  column = 1;
228  for (unsigned j = 0; j < mr.size(); ++j) {
229  orb_file << FortranFormat(mr[j]);
230  if (column == ncolumns) {
231  orb_file << endl;
232  column = 0;
233  }
234  column++;
235  }
236  level++;
237  if (column != 1) orb_file << endl;
238  }
239  orb_file << " 0.0000 0.0000" << endl;
240  orb_file.close();
241  // now convert this ascii file to binary
242  std::string command;
243  command = "cd " + _run_dir +
244  "; asc2mov 5000 system.mos system.movecs > convert.log";
245  int i = std::system(command.c_str());
246  if (i == 0) {
247  CTP_LOG(ctp::logDEBUG, *_pLog)
248  << "Converted MO file from ascii to binary" << flush;
249  } else {
250  CTP_LOG(ctp::logERROR, *_pLog)
251  << "Conversion of binary MO file to binary failed. " << flush;
252  return false;
253  }
254  return true;
255 }

261 bool NWChem::WriteInputFile(Orbitals& orbitals) {
262 
263  std::string temp_suffix = "/id";
264  std::string scratch_dir_backup = _scratch_dir;
265  std::ofstream nw_file;
266  std::ofstream crg_file;
267 
268  std::string nw_file_name_full = _run_dir + "/" + _input_file_name;
269  std::string crg_file_name_full = _run_dir + "/background.crg";
270 
271  nw_file.open(nw_file_name_full.c_str());
272  // header
273  nw_file << "geometry noautoz noautosym" << endl;
274 
275  std::vector<QMAtom*> qmatoms = orbitals.QMAtoms();
276 
277  for (const QMAtom* atom : qmatoms) {
278  tools::vec pos = atom->getPos() * tools::conv::bohr2ang;
279  nw_file << setw(3) << atom->getType().c_str() << setw(12)
280  << setiosflags(ios::fixed) << setprecision(5) << pos.getX()
281  << setw(12) << setiosflags(ios::fixed) << setprecision(5)
282  << pos.getY() << setw(12) << setiosflags(ios::fixed)
283  << setprecision(5) << pos.getZ() << endl;
284  }
285  nw_file << "end\n";
286  if (_write_charges) {
287  // part for the MM charge coordinates
288  crg_file.open(crg_file_name_full.c_str());
289  int numberofcharges = WriteBackgroundCharges(crg_file);
290  crg_file << endl;
291  crg_file.close();
292  nw_file << endl;
293  nw_file << "set bq:max_nbq " << numberofcharges << endl;
294  nw_file << "bq background" << endl;
295  nw_file << "load background.crg format 1 2 3 4" << endl;
296  nw_file << "end\n" << endl;
297  }
298 
299  if (_write_basis_set) {
300  WriteBasisset(nw_file, qmatoms);
301  }
302 
303  if (_write_pseudopotentials) {
304  WriteECP(nw_file, qmatoms);
305  }
306 
307  // write charge of the molecule
308  nw_file << "\ncharge " << _charge << "\n";
309 
310  // writing scratch_dir info
311  if (_scratch_dir != "") {
312  std::string _temp("scratch_dir " + _scratch_dir + temp_suffix + "\n");
313  nw_file << _temp;
314  }
315  if (_charge != 0.0) {
316  std::string dft = "dft";
317  if (_options.find(dft) != std::string::npos) {
318  int dftpos = _options.find(dft);
319  dftpos += dft.size();
320  std::string openshell =
321  "\nodft\n" + (boost::format("mult %1%\n") % _spin).str();
322  _options.insert(dftpos, openshell, 0, openshell.size());
323  } else {
324  throw runtime_error("NWCHEM: dft input data missing");
325  }
326  }
327  nw_file << _options << "\n";
328  if (_write_guess) {
329  bool worked = WriteGuess(orbitals);
330  if (!worked) {
331  return false;
332  }
333  }
334 
335  nw_file << endl;
336  nw_file.close();
337  // and now generate a shell script to run both jobs, if neccessary
338  CTP_LOG(ctp::logDEBUG, *_pLog)
339  << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
340 
341  _scratch_dir = scratch_dir_backup + temp_suffix;
342 
343  WriteShellScript();
344  _scratch_dir = scratch_dir_backup;
345 
346  return true;
347 }

450 bool NWChem::ParseOrbitalsFile(Orbitals& orbitals) {
451  std::map<int, std::vector<double> > coefficients;
452  std::map<int, double> energies;
453  std::map<int, double> occupancy;
454 
455  std::string line;
456  unsigned levels = 0;
457  // unsigned _level;
458  unsigned basis_size = 0;
459  int number_of_electrons = 0;
460 
461  /* maybe we DO need to convert from fortran binary to ASCII first to avoid
462  compiler-dependent issues */
463  std::string orb_file_name_bin = _run_dir + "/" + _orb_file_name;
464  std::string command;
465  command = "cd " + _run_dir +
466  "; mov2asc 10000 system.movecs system.mos > convert.log";
467  int i = std::system(command.c_str());
468  if (i == 0) {
469  CTP_LOG(ctp::logDEBUG, *_pLog)
470  << "Converted MO file from binary to ascii" << flush;
471  } else {
472  CTP_LOG(ctp::logERROR, *_pLog)
473  << "Conversion of binary MO file to ascii failed. " << flush;
474  return false;
475  }
476 
477  // opening the ascii MO file
478  std::string orb_file_name_full = _run_dir + "/" + "system.mos";
479  std::ifstream input_file(orb_file_name_full.c_str());
480 
481  if (input_file.fail()) {
482  CTP_LOG(ctp::logERROR, *_pLog)
483  << "File " << orb_file_name_full
484  << " with molecular orbitals is not found " << flush;
485  return false;
486  } else {
487  CTP_LOG(ctp::logDEBUG, *_pLog)
488  << "Reading MOs from " << orb_file_name_full << flush;
489  }
490 
491  // the first 12 lines are garbage info
492  for (i = 1; i < 13; i++) {
493  getline(input_file, line);
494  }
495  // next line has basis set size
496  input_file >> basis_size;
497  CTP_LOG(ctp::logDEBUG, *_pLog) << "Basis set size: " << basis_size << flush;
498 
499  // next line has number of stored MOs
500  input_file >> levels;
501  CTP_LOG(ctp::logDEBUG, *_pLog) << "Energy levels: " << levels << flush;
502 
503  /* next lines contain information about occupation of the MOs
504  * - each line has 3 numbers
505  * - from _occ we can determine the number of electrons/2 */
506  int n_lines = ((levels - 1) / 3);
507  int n_rest = levels - 3 * n_lines;
508  // read in the data
509  int imo = 0;
510  for (i = 1; i <= n_lines; i++) {
511  for (int j = 0; j < 3; j++) {
512  input_file >> occupancy[imo];
513  if (occupancy[imo] == 2.0) {
514  number_of_electrons++;
515  }
516  imo++;
517  }
518  }
519  if (n_rest != 0) {
520  for (i = 0; i < n_rest; i++) {
521  input_file >> occupancy[imo];
522  imo++;
523  }
524  }
525  CTP_LOG(ctp::logDEBUG, *_pLog)
526  << "Alpha electrons: " << number_of_electrons << flush;
527 
528  int occupied_levels = number_of_electrons;
529  int unoccupied_levels = levels - occupied_levels;
530  CTP_LOG(ctp::logDEBUG, *_pLog)
531  << "Occupied levels: " << occupied_levels << flush;
532  CTP_LOG(ctp::logDEBUG, *_pLog)
533  << "Unoccupied levels: " << unoccupied_levels << flush;
534 
535  // reset index and read MO energies the same way
536  imo = 0;
537  for (i = 1; i <= n_lines; i++) {
538  for (int j = 0; j < 3; j++) {
539  input_file >> energies[imo];
540  imo++;
541  }
542  }
543  if (n_rest != 0) {
544  for (i = 0; i < n_rest; i++) {
545  input_file >> energies[imo];
546  imo++;
547  }
548  }
549 
550  // Now, the same for the coefficients
551  double coef;
552  for (unsigned imo = 0; imo < levels; imo++) {
553  for (i = 1; i <= n_lines; i++) {
554  for (int j = 0; j < 3; j++) {
555  input_file >> coef;
556  coefficients[imo].push_back(coef);
557  }
558  }
559  if (n_rest != 0) {
560  for (i = 0; i < n_rest; i++) {
561  input_file >> coef;
562  coefficients[imo].push_back(coef);
563  }
564  }
565  }
566 
567  // copying information to the orbitals object
568  orbitals.setBasisSetSize(basis_size);
569  orbitals.setNumberOfAlphaElectrons(number_of_electrons);
570  orbitals.setNumberOfOccupiedLevels(occupied_levels);
571  // copying energies to a matrix
572  orbitals.MOEnergies().resize(levels);
573  //_level = 1;
574  for (int i = 0; i < orbitals.MOEnergies().size(); i++) {
575  orbitals.MOEnergies()[i] = energies[i];
576  }
577 
578  // copying orbitals to the matrix
579  (orbitals.MOCoefficients()).resize(levels, basis_size);
580  for (int i = 0; i < orbitals.MOCoefficients().rows(); i++) {
581  for (int j = 0; j < orbitals.MOCoefficients().cols(); j++) {
582  orbitals.MOCoefficients()(j, i) = coefficients[i][j];
583  }
584  }
585 
586  // when all is done, we can trash the ascii file
587  std::string file_name = _run_dir + "/system.mos";
588  remove(file_name.c_str());
589 
590  ReorderOutput(orbitals);
591  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done reading MOs" << flush;
592 
593  return true;
594 }

596 bool NWChem::CheckLogFile() {
597 
598  // check if the log file exists
599 
600  ifstream input_file((_run_dir + "/" + _log_file_name).c_str());
601 
602  if (input_file.fail()) {
603  CTP_LOG(ctp::logERROR, *_pLog) << "NWChem LOG is not found" << flush;
604  return false;
605  };
606 
607  if (input_file.peek() == std::ifstream::traits_type::eof()) {
608  CTP_LOG(ctp::logERROR, *_pLog)
609  << "NWChem run failed. Check OpenMPI version!" << flush;
610  return false;
611  };
612 
613  /* Checking the log file is a pain in the *** since NWChem throws an error
614  * for our 'iterations 1' runs (even though it outputs the required data
615  * correctly. The only way that works for both scf and noscf runs is to
616  * check for "Total DFT energy" near the end of the log file.
617  */
618 
619  input_file.seekg(0, ios_base::end); // go to the EOF
620  char ch;
621  std::string::size_type total_energy_pos = std::string::npos;
622  std::string::size_type diis_pos = std::string::npos;
623  do {
624  // get the beginning of the line
625  do {
626  input_file.seekg(-2, ios_base::cur);
627  input_file.get(ch);
628  // cout << "\nNext Char: " << ch << " TELL G " << (int)_input_file.tellg()
629  // << endl;
630  } while (ch != '\n' || (int)input_file.tellg() == -1);
631 
632  std::string line;
633  getline(input_file, line); // Read the current line

661 bool NWChem::ParseLogFile(Orbitals& orbitals) {
662 
663  std::string line;
664  std::vector<std::string> results;
665 
666  bool has_overlap_matrix = false;
667  bool has_charges = false;
668  bool has_qm_energy = false;
669  bool has_self_energy = false;
670  bool has_basis_set_size = false;
671 
672  bool found_optimization = false;

674 
675  CTP_LOG(ctp::logDEBUG, *_pLog) << "Parsing " << _log_file_name << flush;
676  std::string log_file_name_full = _run_dir + "/" + _log_file_name;
677  // check if LOG file is complete
678  if (!CheckLogFile()) return false;
679 
680  // save qmpackage name
681  orbitals.setQMpackage("nwchem");
682  orbitals.setDFTbasisName(_basisset_name);
683  if (_write_pseudopotentials) {
684  orbitals.setECPName(_ecp_name);
685  }
686  // set _found_optimization to true if this is a run without optimization
687  if (!_is_optimization) {
688  found_optimization = true;
689  }
690 
691  // Start parsing the file line by line
692  ifstream input_file(log_file_name_full.c_str());
693  while (input_file) {
694  getline(input_file, line);
695  boost::trim(line);
696  /*
697  * basis set size (is only required for overlap matrix reading, rest is
698  * in orbitals file and could be skipped
699  */
700  std::string::size_type basis_pos = line.find("number of functions");
701  if (basis_pos != std::string::npos) {
702  // cout << _line << endl;
703  boost::algorithm::split(results, line, boost::is_any_of(":"),
704  boost::algorithm::token_compress_on);
705  has_basis_set_size = true;
706  std::string bf = results.back();
707  boost::trim(bf);
708  basis_set_size = boost::lexical_cast<int>(bf);
709  orbitals.setBasisSetSize(basis_set_size);
710  CTP_LOG(ctp::logDEBUG, *_pLog)
711  << "Basis functions: " << basis_set_size << flush;
712  }
713 
714  std::string::size_type energy_pos = line.find("Total DFT energy");
715  if (energy_pos != std::string::npos) {
716  boost::algorithm::split(results, line, boost::is_any_of("="),
717  boost::algorithm::token_compress_on);
718  std::string energy = results.back();
719  boost::trim(energy);
720  orbitals.setQMEnergy(boost::lexical_cast<double>(energy));
721  CTP_LOG(ctp::logDEBUG, *_pLog)
722  << (boost::format("QM energy[Hrt]: %4.8f ") % orbitals.getQMEnergy())
723  .str()
724  << flush;
725  has_qm_energy = true;
726  }
727 
728  std::string::size_type charge_pos = line.find("ESP");
729  if (charge_pos != std::string::npos && _get_charges) {
730  CTP_LOG(ctp::logDEBUG, *_pLog) << "Getting charges" << flush;
731  has_charges = true;
732  // two empty lines
733  getline(input_file, line);
734  getline(input_file, line);
735 
736  // now starts the data in format
737  // _id type x y z q
738 
739  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
740  int nfields = row.size();
741 
742  while (nfields == 6) {
743  int atom_id = boost::lexical_cast<int>(row.at(0));
744  std::string atom_type = row.at(1);
745  double atom_charge = boost::lexical_cast<double>(row.at(5));
746  row = GetLineAndSplit(input_file, "\t ");
747  nfields = row.size();
748  QMAtom* pAtom;
749  if (orbitals.hasQMAtoms() == false) {
750  pAtom = orbitals.AddAtom(atom_id - 1, atom_type, tools::vec(0.0));
751  } else {
752  pAtom = orbitals.QMAtoms().at(atom_id - 1);
753  }
754  pAtom->setPartialcharge(atom_charge);
755  }
756  }
757 
758  // Coordinates of the final configuration
759  // depending on whether it is an optimization or not
760  if (_is_optimization) {
761  std::string::size_type optimize_pos = line.find("Optimization converged");
762  if (optimize_pos != std::string::npos) {
763  found_optimization = true;
764  }
765  }
766 
767  std::string::size_type coordinates_pos = line.find("Output coordinates");
768  if (found_optimization && coordinates_pos != std::string::npos) {
769  CTP_LOG(ctp::logDEBUG, *_pLog) << "Getting the coordinates" << flush;
770  //_has_coordinates = true;
771  bool has_QMAtoms = orbitals.hasQMAtoms();
772  // three garbage lines
773  getline(input_file, line);
774  getline(input_file, line);
775  getline(input_file, line);
776  // now starts the data in format
777  // _id type Qnuc x y z
778  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
779  int nfields = row.size();
780 
781  while (nfields == 6) {
782  int atom_id = boost::lexical_cast<int>(row.at(0)) - 1;
783  std::string _atom_type = row.at(1);
784  double x = boost::lexical_cast<double>(row.at(3));
785  double y = boost::lexical_cast<double>(row.at(4));
786  double z = boost::lexical_cast<double>(row.at(5));
787  tools::vec pos = tools::vec(x, y, z);
788  pos *= tools::conv::ang2bohr;
789  if (has_QMAtoms == false) {
790  orbitals.AddAtom(atom_id, _atom_type, pos);
791  } else {
792  QMAtom* pAtom = orbitals.QMAtoms().at(atom_id);
793  pAtom->setPos(pos);
794  }
795  atom_id++;
796  row = GetLineAndSplit(input_file, "\t ");
797  nfields = row.size();
798  }
799  }
800 
801  /*
802  * Vxc matrix
803  * stored after the global array: g vxc
804  */
805  if (_output_Vxc) {
806  std::string::size_type vxc_pos = line.find("global array: g vxc");
807  if (vxc_pos != std::string::npos) {
808  Eigen::MatrixXd vxc = orbitals.AOVxc();
809  vxc.resize(basis_set_size, basis_set_size);
810  std::vector<int> j_indeces;
811  int n_blocks = 1 + ((basis_set_size - 1) / 6);
812 
813  for (int block = 0; block < n_blocks; block++) {
814  // first line is garbage
815  getline(input_file, line);
816  // second line gives the j index in the matrix
817  getline(input_file, line);
818  boost::tokenizer<> tok(line);
819 
821  std::transform(tok.begin(), tok.end(), std::back_inserter(j_indeces),
822  &boost::lexical_cast<int, std::string>);
823 
824  // third line is garbage again
825  getline(input_file, line);
826 
827  // read the block of max _basis_size lines + the following header
828  for (int i = 0; i < basis_set_size; i++) {
829  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
830  int i_index = boost::lexical_cast<int>(row.front());
831  row.erase(row.begin());
832  std::vector<int>::iterator j_iter = j_indeces.begin();
833  for (std::string& coefficient : row) {
834  int j_index = *j_iter;
835  vxc(i_index - 1, j_index - 1) =
836  boost::lexical_cast<double>(coefficient);
837  vxc(j_index - 1, i_index - 1) =
838  boost::lexical_cast<double>(coefficient);
839  j_iter++;
840  }
841  }
842  // clear the index for the next block
843  j_indeces.clear();
844  } // end of the blocks
845  CTP_LOG(ctp::logDEBUG, *_pLog) << "Read the Vxc matrix" << flush;
846  }
847  }
848 
849  // Check for ScaHFX = factor of HF exchange included in functional
850  std::string::size_type HFX_pos = line.find("Hartree-Fock (Exact) Exchange");
851  if (HFX_pos != std::string::npos) {
852  boost::algorithm::split(results, line, boost::is_any_of("\t "),
853  boost::algorithm::token_compress_on);
854  double ScaHFX = boost::lexical_cast<double>(results.back());
855  orbitals.setScaHFX(ScaHFX);
856  CTP_LOG(ctp::logDEBUG, *_pLog)
857  << "DFT with " << ScaHFX << " of HF exchange!" << flush;
858  }
859 
860  // overlap matrix
861  // stored after the global array: Temp Over line
862  std::string::size_type overlap_pos = line.find("global array: Temp Over");
863  if (overlap_pos != std::string::npos) {
864  // prepare the container
865  (orbitals.AOOverlap()).resize(basis_set_size, basis_set_size);
866  has_overlap_matrix = true;
867  std::vector<int> j_indeces;
868 
869  int n_blocks = 1 + ((basis_set_size - 1) / 6);
870 
871  for (int block = 0; block < n_blocks; block++) {
872  // first line is garbage
873  getline(input_file, line);
874  // second line gives the j index in the matrix
875  getline(input_file, line);
876  boost::tokenizer<> tok(line);
877 
879  std::transform(tok.begin(), tok.end(), std::back_inserter(j_indeces),
880  &boost::lexical_cast<int, std::string>);
881 
882  // third line is garbage again
883  getline(input_file, line);
884 
885  // read the block of max _basis_size lines + the following header
886  for (int i = 0; i < basis_set_size; i++) {
887  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
888  int i_index = boost::lexical_cast<int>(row.front());
889  row.erase(row.begin());
890  std::vector<int>::iterator j_iter = j_indeces.begin();
891  for (std::string& coefficient : row) {
892  int j_index = *j_iter;
893  orbitals.AOOverlap()(i_index - 1, j_index - 1) =
894  boost::lexical_cast<double>(coefficient);
895  orbitals.AOOverlap()(j_index - 1, i_index - 1) =
896  boost::lexical_cast<double>(coefficient);
897  j_iter++;
898  }
899  }
900  // clear the index for the next block
901  j_indeces.clear();
902  } // end of the blocks
903  CTP_LOG(ctp::logDEBUG, *_pLog) << "Read the overlap matrix" << flush;
904  } // end of the if "Overlap" found
905 
906  /*
907  * TODO Self-energy of external charges
908  */
909  std::string::size_type self_energy_pos =
910  line.find("Self energy of the charges");
911  if (self_energy_pos != std::string::npos) {
912  CTP_LOG(ctp::logDEBUG, *_pLog) << "Getting the self energy\n";
913  std::vector<std::string> block;
914  std::vector<std::string> energy;
915  boost::algorithm::split(block, line, boost::is_any_of("="),
916  boost::algorithm::token_compress_on);
917  boost::algorithm::split(energy, block[1], boost::is_any_of("\t "),
918  boost::algorithm::token_compress_on);
919  orbitals.setSelfEnergy(boost::lexical_cast<double>(energy[1]));
920  CTP_LOG(ctp::logDEBUG, *_pLog)
921  << "Self energy " << orbitals.getSelfEnergy() << flush;
922  }
923 
924  // check if all information has been accumulated and quit
925  if (has_basis_set_size && has_overlap_matrix && has_charges &&
926  has_qm_energy && has_self_energy)
927  break;
928 
929  } // end of reading the file line-by-line
930 
931  CTP_LOG(ctp::logDEBUG, *_pLog) << "Done parsing" << flush;
932  return true;
