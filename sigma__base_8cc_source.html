29 Eigen::MatrixXd Sigma_base::CalcExchangeMatrix() const {
30  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(_qptotal, _qptotal);
31  Index occlevel = _opt.homo - _opt.rpamin + 1;
32  Index qpmin = _opt.qpmin - _opt.rpamin;
33 #pragma omp parallel for schedule(dynamic)
34  for (Index gw_level1 = 0; gw_level1 < _qptotal; gw_level1++) {
35  const Eigen::MatrixXd& Mmn1 = _Mmn[gw_level1 + qpmin];
36  for (Index gw_level2 = gw_level1; gw_level2 < _qptotal; gw_level2++) {
37  const Eigen::MatrixXd& Mmn2 = _Mmn[gw_level2 + qpmin];
38  double sigma_x =
39  -(Mmn1.topRows(occlevel).cwiseProduct(Mmn2.topRows(occlevel))).sum();
40  result(gw_level2, gw_level1) = sigma_x;
41  }
42  }
43  result = result.selfadjointView<Eigen::Lower>();
44  return result;
45 }
