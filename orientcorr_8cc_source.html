155 // evaluates a frame
156 void MyWorker::EvalConfiguration(Topology *top, Topology *top_ref) {
157 
158  // first genearate a mapped topology
159  // the beads are sitting on the bonds and have an orientation which
160  // is pointing along bond direction
161  Topology mapped;
162  cout << "generating mapped topology...";
163 
164  // copy box size
165  mapped.setBox(top->getBox());
166 
167  // loop over all molecules
168  for(MoleculeContainer::iterator iter = top->Molecules().begin();
169  iter != top->Molecules().end(); ++iter) {
170  Molecule *mol_src = *iter;
171  // create a molecule in mapped topology
172  Molecule *mol = mapped.CreateMolecule(mol_src->getName());
173  // loop over beads in molecule
174  for(int i=0; i<mol_src->BeadCount()-1; ++i) {
175  // create a bead in mapped topology
176  Bead *b = mapped.CreateBead(3, "A", mapped.GetOrCreateBeadType("A"), 1, 0.0, 0.0);
177  vec p1 = mol_src->getBead(i)->getPos();
178  vec p2 = mol_src->getBead(i+1)->getPos();
179  // position is in middle of bond
180  vec pos = 0.5*(p1 + p2);
181  // orientation pointing along bond
182  vec v = p2 - p1;
183  v.normalize();
184  b->setPos(pos);
185  b->setV(v);
186  mol->AddBead(b, "A");
187  }
188  }
189  cout << "done\n";
190 
191  // the neighbor search only finds pairs, add self-self correlation parts here
192  _cor.Process(0.0f, mapped.BeadCount());
193  _count.Process(0.0f, mapped.BeadCount());
194 
195  // search for all beads
196  BeadList b;
197  b.Generate(mapped, "*");
198 
199  // create/initialize neighborsearch
200  NBList *nb = OrientCorrApp::CreateNBSearch();
201  nb->setCutoff(_cut_off);
202 
203  // set callback for each pair found
204  nb->SetMatchFunction(this, &MyWorker::FoundPair);
205 
206  // execute the search
207  nb->Generate(b);
208 
209  delete nb;
210 }

212 // process a pair, since return value is falsed, pairs are not cached which
213 // saves a lot of memory for the big systems
214 bool MyWorker::FoundPair(Bead *b1, Bead *b2, const vec &r, const double dist)
215 {
216  double tmp = b1->getV()*b2->getV();
217  double P2 = 3./2.*tmp*tmp - 0.5;
218 
219  // calculate average without exclusions
220  _cor.Process(dist, P2);
221  _count.Process(dist);
222 
223  if(b1->getMolecule() == b2->getMolecule())
224  return false;
225 
226  // calculate average with excluding intramolecular contributions
227  _cor_excl.Process(dist, P2);
228  _count_excl.Process(dist);
229 
230  return false;
231 }
