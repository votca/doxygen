157 // evaluates a frame
158 void MyWorker::EvalConfiguration(Topology *top, Topology *) {
159 
160  // first genearate a mapped topology
161  // the beads are sitting on the bonds and have an orientation which
162  // is pointing along bond direction
163  Topology mapped;
164  cout << "generating mapped topology...";
165 
166  // copy box size
167  mapped.setBox(top->getBox());
168 
169  // loop over all molecules
170  for (auto mol_src : top->Molecules()) {
171  // create a molecule in mapped topology
172  Molecule *mol = mapped.CreateMolecule(mol_src->getName());
173  // loop over beads in molecule
174  for (votca::Index i = 0; i < mol_src->BeadCount() - 1; ++i) {
175  // create a bead in mapped topology
176  string bead_type = "A";
177  if (mapped.BeadTypeExist(bead_type) == false) {
178  mapped.RegisterBeadType(bead_type);
179  }
180  Bead *b =
181  mapped.CreateBead(Bead::ellipsoidal, "A", bead_type, 1, 0.0, 0.0);
182  Eigen::Vector3d p1 = mol_src->getBead(i)->getPos();
183  Eigen::Vector3d p2 = mol_src->getBead(i + 1)->getPos();
184  // position is in middle of bond
185  Eigen::Vector3d pos = 0.5 * (p1 + p2);
186  // orientation pointing along bond
187  Eigen::Vector3d v = p2 - p1;
188  v.normalize();
189  b->setPos(pos);
190  b->setV(v);
191  mol->AddBead(b, "A");
192  }
193  }
194  cout << "done\n";
195 
196  // the neighbor search only finds pairs, add self-self correlation parts here
197  _cor.Process(0.0f, (double)mapped.BeadCount());
198  _count.Process(0.0f, (double)mapped.BeadCount());
199 
200  // search for all beads
201  BeadList b;
202  b.Generate(mapped, "*");
203 
204  // create/initialize neighborsearch
205  std::unique_ptr<NBList> nb =
206  std::unique_ptr<NBList>(OrientCorrApp::CreateNBSearch());
207  nb->setCutoff(_cut_off);
208 
209  // set callback for each pair found
210  nb->SetMatchFunction(this, &MyWorker::FoundPair);
211 
212  // execute the search
213  nb->Generate(b);
214 }

216 // process a pair, since return value is falsed, pairs are not cached which
217 // saves a lot of memory for the big systems
218 bool MyWorker::FoundPair(Bead *b1, Bead *b2, const Eigen::Vector3d &,
219  const double dist) {
220  double tmp = b1->getV().dot(b2->getV());
221  double P2 = 3. / 2. * tmp * tmp - 0.5;
222 
223  // calculate average without exclusions
224  _cor.Process(dist, P2);
225  _count.Process(dist);
226 
227  if (b1->getMoleculeId() == b2->getMoleculeId()) {
228  return false;
229  }
230 
231  // calculate average with excluding intramolecular contributions
232  _cor_excl.Process(dist, P2);
233  _count_excl.Process(dist);
234 
235  return false;
236 }
