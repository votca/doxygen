139 void MyWorker::EvalConfiguration(Topology *top, Topology *top_ref) {
140 
141  // first genearate a mapped topology
142  // the beads are sitting on the bonds and have an orientation which
143  // is pointing along bond direction
144  Topology mapped;
145  cout << "generating mapped topology...";
146 
147  // copy box size
148  mapped.setBox(top->getBox());
149 
150  // loop over all molecules
151  for(MoleculeContainer::iterator iter = top->Molecules().begin();
152  iter != top->Molecules().end(); ++iter) {
153  Molecule *mol_src = *iter;
154  // create a molecule in mapped topology
155  Molecule *mol = mapped.CreateMolecule(mol_src->getName());
156  // loop over beads in molecule
157  for(int i=0; i<mol_src->BeadCount()-1; ++i) {
158  // create a bead in mapped topology
159  Bead *b = mapped.CreateBead(3, "A", mapped.GetOrCreateBeadType("A"), 1, 0.0, 0.0);
160  vec p1 = mol_src->getBead(i)->getPos();
161  vec p2 = mol_src->getBead(i+1)->getPos();
162  // position is in middle of bond
163  vec pos = 0.5*(p1 + p2);
164  // orientation pointing along bond
165  vec v = p2 - p1;
166  v.normalize();
167  b->setPos(pos);
168  b->setV(v);
169  mol->AddBead(b, "A");
170  }
171  }
172  cout << "done\n";
173 
174  // the neighbor search only finds pairs, add self-self correlation parts here
175  _cor.Process(0.0f, mapped.BeadCount());
176  _count.Process(0.0f, mapped.BeadCount());
177 
178  // search for all beads
179  BeadList b;
180  b.Generate(mapped, "*");
181 
182  // create/initialize neighborsearch
183  NBList *nb = OrientCorrApp::CreateNBSearch();
184  nb->setCutoff(_cut_off);
185 
186  // set callback for each pair found
187  nb->SetMatchFunction(this, &MyWorker::FoundPair);
188 
189  // execute the search
190  nb->Generate(b);
191 
192  delete nb;
193 }

197 bool MyWorker::FoundPair(Bead *b1, Bead *b2, const vec &r, const double dist)
198 {
199  double tmp = b1->getV()*b2->getV();
200  double P2 = 3./2.*tmp*tmp - 0.5;
201 
202  // calculate average without exclusions
