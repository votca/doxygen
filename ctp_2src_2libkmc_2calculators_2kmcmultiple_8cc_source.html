592 void KMCMultiple::InitialRates(vector<Node*> node)
593 {
594  cout << endl <<"Calculating initial Marcus rates." << endl;
595  cout << " Temperature T = " << _temperature << " K." << endl;
596  cout << " Field: (" << _fieldX << ", " << _fieldY << ", " << _fieldZ << ") V/m" << endl;
597  if (_carriertype == "e")
598  {
599  _q = -1.0;
600  }
601  else if (_carriertype == "h")
602  {
603  _q = 1.0;
604  }
605  cout << " carriertype: " << _carriertype << " (charge: " << _q << " eV)" << endl;
606  unsigned int numberofsites = node.size();
607  cout << " Rates for "<< numberofsites << " sites are computed." << endl;
608  double maxreldiff = 0;
609  int totalnumberofrates = 0;
610  for(unsigned int i=0; i<numberofsites; i++)
611  {
612  unsigned int numberofneighbours = node[i]->event.size();
613  for(unsigned int j=0; j<numberofneighbours; j++)
614  {
615  double dX = node[i]->event[j].dr.x();
616  double dY = node[i]->event[j].dr.y();
617  double dZ = node[i]->event[j].dr.z();
618  double dG_Field = _q * (dX*_fieldX + dY*_fieldY + dZ*_fieldZ);
619 
620  double destindex = node[i]->event[j].destination;
621  double reorg = node[i]->reorg_intorig + node[destindex]->reorg_intdest + node[i]->event[j].reorg_out;
622 
623  double dG_Site = node[destindex]->siteenergy - node[i]->siteenergy;
624 
625  double J2 = node[i]->event[j].Jeff2;
626 
627  double dG = dG_Site - dG_Field;
628 
629  double rate = 2*Pi/hbar * J2/sqrt(4*Pi*reorg*kB*_temperature) * exp(-(dG+reorg)*(dG+reorg) / (4*reorg*kB*_temperature));
630 
631  // calculate relative difference compared to values in the table
632  double reldiff = (node[i]->event[j].rate - rate) / node[i]->event[j].rate;
633  if (reldiff > maxreldiff) {maxreldiff = reldiff;}
634  reldiff = (node[i]->event[j].rate - rate) / rate;
635  if (reldiff > maxreldiff) {maxreldiff = reldiff;}
636 
637  // set rates to calculated values
638  node[i]->event[j].rate = rate;
639  node[i]->event[j].initialrate = rate;
640 
641  totalnumberofrates ++;
642 
643  }
644 
645  // Initialise escape rates
646  for(unsigned int i=0; i<node.size(); i++)
647  {
648  node[i]->InitEscapeRate();
649  }
650 
651  }
652  cout << " " << totalnumberofrates << " rates have been calculated." << endl;
653  if(maxreldiff < 0.01)
654  {
655  cout << " Good agreement with rates in the state file. Maximal relative difference: " << maxreldiff*100 << " %"<< endl;
656  }
657  else
658  {
659  cout << " WARNING: Rates differ from those in the state file up to " << maxreldiff*100 << " %." << " If the rates in the state file are calculated for a different temperature/field or if they are not Marcus rates, this is fine. Otherwise something might be wrong here." << endl;
