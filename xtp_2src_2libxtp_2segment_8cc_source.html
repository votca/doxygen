48 // This constructor creates a copy of the stencil segment, without
49 // adding it to any containers further up in the hierarchy; i.e. the topology
50 // and molecules will neither know about the existence of this segment, nor
51 // be able to access it. Used for creating the ghost in PB corrected pairs.
52 Segment::Segment(Segment *stencil)
53  : _id(stencil->getId()),
54  _name(stencil->getName() + "_ghost"),
55  _typ(stencil->getType()),
56  _top(NULL),
57  _mol(NULL),
58  _CoM(stencil->getPos()),
59  _has_e(false),
60  _has_h(false),
61  _has_s(false),
62  _has_t(false) {
63  _eMpoles.resize(5);
64 
65  vector<Fragment *>::iterator fit;
66  for (fit = stencil->Fragments().begin(); fit < stencil->Fragments().end();
67  fit++) {
68 
69  Fragment *newFrag = new Fragment(*fit);
70  this->AddFragment(newFrag);
71 
72  vector<Atom *>::iterator ait;
73  for (ait = newFrag->Atoms().begin(); ait < newFrag->Atoms().end(); ait++) {
74  this->AddAtom(*ait);
75  }
76  }
77 }

323 void Segment::calcApproxSize() {
324  _approxsize = 0.0;
325 
326  tools::vec min = vec(numeric_limits<double>::max());
327  tools::vec max = vec(numeric_limits<double>::min());
328  vector<Fragment *>::iterator fragit1;
329  for (fragit1 = Fragments().begin(); fragit1 < Fragments().end(); fragit1++) {
330  const tools::vec &pos = (*fragit1)->getPos();
331  if (pos.getX() > max.getX()) {
332  max.x() = pos.getX();
333  } else if (pos.getX() < min.getX()) {
334  min.x() = pos.getX();
335  }
336  if (pos.getY() > max.getY()) {
337  max.y() = pos.getY();
338  } else if (pos.getY() < min.getY()) {
339  min.y() = pos.getY();
340  }
341  if (pos.getZ() > max.getZ()) {
342  max.z() = pos.getZ();
343  } else if (pos.getZ() < min.getZ()) {
344  min.z() = pos.getZ();
345  }
346  }
347 
348  _approxsize = abs(max - min);
349 
350  return;
351 }
