274  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
275  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
276  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
277  e12 = pol2.getPos() - pol1.getPos();
278  R = 1/abs(e12);
279  R2 = R*R;
280  R3 = R2*R;
281  R4 = R3*R;
