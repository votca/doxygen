326  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
327  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
328  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
329  e12 = pol2.getPos() - pol1.getPos();
330  R = 1/abs(e12);
331  R2 = R*R;
332  R3 = R2*R;
333  R4 = R3*R;
334  R5 = R4*R;
335  e12 *= R;
336 
337  // Thole damping init.
338 // double P_proj_1 = fabs(e12 * pol1.pax) * pol1.eigenpxx
339 // + fabs(e12 * pol1.pay) * pol1.eigenpyy
340 // + fabs(e12 * pol1.paz) * pol1.eigenpzz;
341 // double P_proj_2 = fabs(e12 * pol2.pax) * pol2.eigenpxx

343 // + fabs(e12 * pol2.paz) * pol2.eigenpzz;
344 
345  //u3 = 1 / (R3 * sqrt(pol1.getIsoP() * pol2.getIsoP()));
346  u3 = 1 / (R3 * sqrt(
347  1./3.*(pol1.Pxx*pol2.Pxx + pol1.Pyy*pol2.Pyy + pol1.Pzz*pol2.Pzz) ));
348 
349  // rax = pol1._locX * e12;
350  // ray = pol1._locY * e12;
351  // raz = pol1._locZ * e12;
352  // rbx = - pol2._locX * e12;
353  // rby = - pol2._locY * e12;
354  // rbz = - pol2._locZ * e12;
355 
356  rax = e12.getX();
357  ray = e12.getY();
358  raz = e12.getZ();
359  rbx = - rax;
360  rby = - ray;
361  rbz = - raz;

394  pol1.FUz += TU1z_1z() * pol2.U1z;
