261  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
262  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
263  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
264  e12 = pol2.getPos() - pol1.getPos();
265  R = 1/abs(e12);
266  R2 = R*R;
267  R3 = R2*R;
268  R4 = R3*R;
269  R5 = R4*R;
270  e12 *= R;
271 
272  // Thole damping init.
273 // double P_proj_1 = fabs(e12 * pol1.pax) * pol1.eigenpxx
274 // + fabs(e12 * pol1.pay) * pol1.eigenpyy
275 // + fabs(e12 * pol1.paz) * pol1.eigenpzz;
276 // double P_proj_2 = fabs(e12 * pol2.pax) * pol2.eigenpxx
277 // + fabs(e12 * pol2.pay) * pol2.eigenpyy
278 // + fabs(e12 * pol2.paz) * pol2.eigenpzz;
279 
280  //u3 = 1 / (R3 * sqrt(pol1.getIsoP() * pol2.getIsoP()));
281  u3 = 1 / (R3 * sqrt(
282  1./3.*(pol1.Pxx*pol2.Pxx + pol1.Pyy*pol2.Pyy + pol1.Pzz*pol2.Pzz) ));
283 
284  // rax = pol1._locX * e12;
285  // ray = pol1._locY * e12;
286  // raz = pol1._locZ * e12;
287  // rbx = - pol2._locX * e12;
288  // rby = - pol2._locY * e12;
289  // rbz = - pol2._locZ * e12;
290 
291  rax = e12.getX();
292  ray = e12.getY();
293  raz = e12.getZ();
294  rbx = - rax;
295  rby = - ray;
296  rbz = - raz;
297 
298  // cxx = pol1._locX * pol2._locX;
299  // cxy = pol1._locX * pol2._locY;
300  // cxz = pol1._locX * pol2._locZ;
301  // cyx = pol1._locY * pol2._locX;
302  // cyy = pol1._locY * pol2._locY;
303  // cyz = pol1._locY * pol2._locZ;
304  // czx = pol1._locZ * pol2._locX;
305  // czy = pol1._locZ * pol2._locY;

308  cxx = 1;
309  cxy = 0;
310  cxz = 0;

312  cyy = 1;
313  cyz = 0;
314  czx = 0;
315  czy = 0;
316  czz = 1;

319 
320  if (a*u3 < 40.0) {
321  pol1.FUx += TU1x_1x() * pol2.U1x;
322  pol1.FUx += TU1x_1y() * pol2.U1y;
323  pol1.FUx += TU1x_1z() * pol2.U1z;
324  pol1.FUy += TU1y_1x() * pol2.U1x;
325  pol1.FUy += TU1y_1y() * pol2.U1y;
326  pol1.FUy += TU1y_1z() * pol2.U1z;
327  pol1.FUz += TU1z_1x() * pol2.U1x;
328  pol1.FUz += TU1z_1y() * pol2.U1y;
329  pol1.FUz += TU1z_1z() * pol2.U1z;
330 
331  pol2.FUx += TU1x_1x() * pol1.U1x;
332  pol2.FUx += TU1y_1x() * pol1.U1y;
333  pol2.FUx += TU1z_1x() * pol1.U1z;

336  pol2.FUy += TU1z_1y() * pol1.U1z;
