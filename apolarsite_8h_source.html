281  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
282  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
283  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
284  e12 = pol2.getPos() - pol1.getPos();
285  R = 1/abs(e12);
286  R2 = R*R;
287  R3 = R2*R;
288  R4 = R3*R;

302  1./3.*(pol1.Pxx*pol2.Pxx + pol1.Pyy*pol2.Pyy + pol1.Pzz*pol2.Pzz) ));

306  // raz = pol1._locZ * e12;
307  // rbx = - pol2._locX * e12;
