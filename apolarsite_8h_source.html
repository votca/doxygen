294  inline void FieldInduAlpha(APolarSite &pol1, APolarSite &pol2) {
295  // NOTE >>> Exp. damping factor a must have been set <<< NOTE //
296  // NOTE >>> e12 points from polar site 1 to polar site 2 <<< NOTE //
297  e12 = pol2.getPos() - pol1.getPos();
298  R = 1/abs(e12);
299  R2 = R*R;
300  R3 = R2*R;
301  R4 = R3*R;
302  R5 = R4*R;
303  e12 *= R;
304 
305  // Thole damping init.
306 // double P_proj_1 = fabs(e12 * pol1.pax) * pol1.eigenpxx
307 // + fabs(e12 * pol1.pay) * pol1.eigenpyy
308 // + fabs(e12 * pol1.paz) * pol1.eigenpzz;
309 // double P_proj_2 = fabs(e12 * pol2.pax) * pol2.eigenpxx
310 // + fabs(e12 * pol2.pay) * pol2.eigenpyy
311 // + fabs(e12 * pol2.paz) * pol2.eigenpzz;
312 
313  //u3 = 1 / (R3 * sqrt(pol1.getIsoP() * pol2.getIsoP()));
314  u3 = 1 / (R3 * sqrt(
315  1./3.*(pol1.Pxx*pol2.Pxx + pol1.Pyy*pol2.Pyy + pol1.Pzz*pol2.Pzz) ));
316 
317  // rax = pol1._locX * e12;
318  // ray = pol1._locY * e12;
319  // raz = pol1._locZ * e12;
320  // rbx = - pol2._locX * e12;
321  // rby = - pol2._locY * e12;
322  // rbz = - pol2._locZ * e12;
323 
324  rax = e12.getX();
325  ray = e12.getY();
326  raz = e12.getZ();
327  rbx = - rax;
328  rby = - ray;
329  rbz = - raz;
330 
331  // cxx = pol1._locX * pol2._locX;
332  // cxy = pol1._locX * pol2._locY;
333  // cxz = pol1._locX * pol2._locZ;
334  // cyx = pol1._locY * pol2._locX;
335  // cyy = pol1._locY * pol2._locY;
336  // cyz = pol1._locY * pol2._locZ;
337  // czx = pol1._locZ * pol2._locX;
338  // czy = pol1._locZ * pol2._locY;
339  // czz = pol1._locZ * pol2._locZ;
340 
341  cxx = 1;
342  cxy = 0;
343  cxz = 0;
344  cyx = 0;
345  cyy = 1;
346  cyz = 0;
347  czx = 0;
348  czy = 0;
349  czz = 1;
350 
351  // Fields generated by rank-1 induced m'poles
352 
353  if (a*u3 < 40.0) {
354  pol1.FUx += TU1x_1x() * pol2.U1x;
355  pol1.FUx += TU1x_1y() * pol2.U1y;
356  pol1.FUx += TU1x_1z() * pol2.U1z;
357  pol1.FUy += TU1y_1x() * pol2.U1x;
358  pol1.FUy += TU1y_1y() * pol2.U1y;
359  pol1.FUy += TU1y_1z() * pol2.U1z;
360  pol1.FUz += TU1z_1x() * pol2.U1x;
361  pol1.FUz += TU1z_1y() * pol2.U1y;
362  pol1.FUz += TU1z_1z() * pol2.U1z;
363 
364  pol2.FUx += TU1x_1x() * pol1.U1x;
365  pol2.FUx += TU1y_1x() * pol1.U1y;
366  pol2.FUx += TU1z_1x() * pol1.U1z;
367  pol2.FUy += TU1x_1y() * pol1.U1x;
368  pol2.FUy += TU1y_1y() * pol1.U1y;
369  pol2.FUy += TU1z_1y() * pol1.U1z;
370  pol2.FUz += TU1x_1z() * pol1.U1x;
371  pol2.FUz += TU1y_1z() * pol1.U1y;
372  pol2.FUz += TU1z_1z() * pol1.U1z;
373  }
374  else {
375  pol1.FUx += T1x_1x() * pol2.U1x;
376  pol1.FUx += T1x_1y() * pol2.U1y;
377  pol1.FUx += T1x_1z() * pol2.U1z;
378  pol1.FUy += T1y_1x() * pol2.U1x;
379  pol1.FUy += T1y_1y() * pol2.U1y;
380  pol1.FUy += T1y_1z() * pol2.U1z;
381  pol1.FUz += T1z_1x() * pol2.U1x;
382  pol1.FUz += T1z_1y() * pol2.U1y;
383  pol1.FUz += T1z_1z() * pol2.U1z;
384 
385  pol2.FUx += T1x_1x() * pol1.U1x;
386  pol2.FUx += T1y_1x() * pol1.U1y;
387  pol2.FUx += T1z_1x() * pol1.U1z;
388  pol2.FUy += T1x_1y() * pol1.U1x;
389  pol2.FUy += T1y_1y() * pol1.U1y;
390  pol2.FUy += T1z_1y() * pol1.U1z;
391  pol2.FUz += T1x_1z() * pol1.U1x;
392  pol2.FUz += T1y_1z() * pol1.U1y;
