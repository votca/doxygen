153  template <typename T>
154  void ReadData(const CptLoc& loc, Eigen::MatrixBase<T>& matrix,
155  const std::string& name) const {
156 
157  const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
158 
159  H5::DataSet dataset = loc.openDataSet(name);
160 
161  H5::DataSpace dp = dataset.getSpace();
162 
163  hsize_t dims[2];
164  dp.getSimpleExtentDims(dims, NULL); // ndims is always 2 for us
165 
166  hsize_t matRows = dims[0];
167  hsize_t matCols = dims[1];
168 
169  matrix.derived().resize(matRows, matCols);
170 
171  hsize_t matColSize = matrix.derived().outerStride();
172 
173  hsize_t fileRows = matCols;
174 
175  hsize_t fStride[2] = {1, fileRows};
176  hsize_t fCount[2] = {1, 1};
177  hsize_t fBlock[2] = {1, fileRows};
178 
179  hsize_t mStride[2] = {matColSize, 1};
180  hsize_t mCount[2] = {1, 1};
181  hsize_t mBlock[2] = {matCols, 1};
182 
183  hsize_t mDim[2] = {matCols, matColSize};
184  H5::DataSpace mspace(2, mDim);
185 
186  for (hsize_t i = 0; i < matRows; i++) {
187  hsize_t fStart[2] = {i, 0};
188  hsize_t mStart[2] = {0, i};
189  dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
190  mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
191  dataset.read(matrix.derived().data(), *dataType, mspace, dp);
192  }
193  }
