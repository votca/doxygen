160  template <typename T>
161  void ReadData(const CptLoc& loc, Eigen::MatrixBase<T>& matrix,
162  const std::string& name) const {
163 
164  const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
165 
166  H5::DataSet dataset = loc.openDataSet(name);
167 
168  H5::DataSpace dp = dataset.getSpace();
169 
170  hsize_t dims[2];
171  dp.getSimpleExtentDims(dims, nullptr); // ndims is always 2 for us
172 
173  hsize_t matRows = dims[0];
174  hsize_t matCols = dims[1];
175 
176  matrix.derived().resize(matRows, matCols);
177  if (matrix.size() == 0) {
178  return;
179  }
180 
181  hsize_t matColSize = matrix.derived().outerStride();
182 
183  hsize_t fileRows = matCols;
184 
185  hsize_t fStride[2] = {1, fileRows};
186  hsize_t fCount[2] = {1, 1};
187  hsize_t fBlock[2] = {1, fileRows};
188 
189  hsize_t mStride[2] = {matColSize, 1};
190  hsize_t mCount[2] = {1, 1};
191  hsize_t mBlock[2] = {matCols, 1};
192 
193  hsize_t mDim[2] = {matCols, matColSize};
194  H5::DataSpace mspace(2, mDim);
195 
196  for (hsize_t i = 0; i < matRows; i++) {
197  hsize_t fStart[2] = {i, 0};
198  hsize_t mStart[2] = {0, i};
199  dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
200  mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
201  dataset.read(matrix.derived().data(), *dataType, mspace, dp);
202  }
203  }
