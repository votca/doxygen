80 bool XtpTools::EvaluateOptions() {
81 
82  std::string helpdir = "xtp/xml";
83 
84  if (OptionsMap().count("list")) {
85  std::cout << "Available XTP tools: \n";
86 
87  for (const auto& tool : xtp::QMTools().getObjects()) {
88  PrintDescription(std::cout, tool.first, helpdir, Application::HelpShort);
89  }
90  StopExecution();
91  return true;
92  }
93 
94  if (OptionsMap().count("description")) {
95  CheckRequired("description", "no tool is given");
96  tools::Tokenizer tok(OptionsMap()["description"].as<std::string>(),
97  " ,\n\t");
98  // loop over the names in the description string
99  for (const std::string& n : tok) {
100  // loop over tools
101  bool printerror = true;
102  for (const auto& tool : xtp::QMTools().getObjects()) {
103  if (n.compare(tool.first) == 0) {
104  PrintDescription(std::cout, tool.first, helpdir,
105  Application::HelpLong);
106  printerror = false;
107  break;
108  }
109  }
110  if (printerror) {
111  std::cout << "Tool " << n << " does not exist\n";
112  }
113  }
114  StopExecution();
115  return true;
116  }
117 
118  CheckRequired("execute", "Please provide the name of the tool to execute");
119 
120  tools::Tokenizer xtools(OptionsMap()["execute"].as<std::string>(), " ,\n\t");
121  std::vector<std::string> calc_string = xtools.ToVector();
122  if (calc_string.size() != 1) {
123  throw std::runtime_error(
124  "You can only run one calculator at the same time.");
125  }
126 
127  CheckRequired(
128  "name", "Please provide the job name to run (same as the xyz file name)");
129 
130  bool found_calc = false;
131  for (const auto& tool : xtp::QMTools().getObjects()) {
132  if (calc_string[0].compare(tool.first) == 0) {
133  this->SetTool(xtp::QMTools().Create(calc_string[0]));
134  found_calc = true;
135  break;
136  }
137  }
138  if (!found_calc) {
139  std::cout << "Tool " << calc_string[0] << " does not exist\n";
140  StopExecution();
141  } else {
142  std::cout << "Registered " << calc_string[0];
143  }
144  return true;
145 }

147 void XtpTools::Run() {
148 
149  auto it = _op_vm.find("options");
150  if (it != _op_vm.cend()) {
151  std::string optionsFile = _op_vm["options"].as<std::string>();
152  _options.LoadFromXML(optionsFile);
153  } else {
154  // Empty user options
155  tools::Property& opts = _options.add("options", "");
156  opts.add(_tool->Identify(), "");
157  }
158 
159  std::string job_name = _op_vm["name"].as<std::string>();
160  tools::Property& opts = _options.get("options." + _tool->Identify());
161  opts.add("job_name", job_name);
162 
163  Index nThreads = OptionsMap()["nthreads"].as<Index>();
164  std::string name = ProgramName();
165  if (VersionString() != "") {
166  name = name + ", version " + VersionString();
167  }
168  xtp::HelpTextHeader(name);
169  std::cout << "Initializing tool\n";
170  BeginEvaluate(nThreads);
171 
172  std::cout << "Evaluating tool\n";
173 
174  Evaluate();
175 }
