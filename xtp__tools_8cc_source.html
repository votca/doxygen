94 bool XtpTools::EvaluateOptions() {
95 
96  string helpdir = "xtp/xml";
97  string ctphelpdir = "ctp/xml";
98 
99  if (OptionsMap().count("list")) {
100  cout << "Available XTP tools: \n";
101  for (const auto& tool:xtp::QMTools().getObjects()) {
102  PrintDescription(std::cout, tool.first, helpdir, Application::HelpShort);
103  }
104  cout << "Available CTP tools: \n";
105  for (const auto& tool:ctp::QMTools().getObjects()) {
106  bool printctp = true;
107  std::string ctpcalc = (tool.first).c_str();
108  for (const auto& xtptool:xtp::QMTools().getObjects()) {
109  if (ctpcalc.compare((xtptool.first).c_str()) == 0) {
110  printctp = false;
111  break;
112  }
113  }
114  if (printctp) {
115  PrintDescription(std::cout, tool.first, ctphelpdir, Application::HelpShort);
116  }
117  }
118  StopExecution();
119  return true;
120  }
121 
122 
123  if (OptionsMap().count("description")) {
124  CheckRequired("description", "no tool is given");
125  tools::Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
126  // loop over the names in the description string
127  for (const std::string& n :tok) {
128  // loop over tools
129  bool printerror = true;
130  for ( const auto& tool:xtp::QMTools().getObjects()) {
131  if (n.compare(tool.first.c_str()) == 0) {
132  PrintDescription(std::cout, tool.first, helpdir, Application::HelpLong);
133  printerror = false;
134  break;
135  }
136  }
137  for (const auto& tool:ctp::QMTools().getObjects()) {
138  if (n.compare(tool.first.c_str()) == 0) {
139  bool printctp = true;
140  std::string ctpcalc = tool.first.c_str();
141  for (const auto& xtptool:xtp::QMTools().getObjects()) {
142  if (ctpcalc.compare(xtptool.first.c_str()) == 0) {
143  printctp = false;
144  break;
145  }
146  }
147  if (printctp) {
148  PrintDescription(std::cout, tool.first, "ctp/xml", Application::HelpLong);
149  printerror = false;
150  break;
151  }
152  }
153  }
154  if (printerror) cout << "Tool " << n << " does not exist\n";
155  }
156  StopExecution();
157  return true;
158  }
159 
160  Application::EvaluateOptions();
161  CheckRequired("execute", "Nothing to do here: Abort.");
162  CheckRequired("options", "Please provide an xml file with tool options");
163 
164  tools::Tokenizer xtools(OptionsMap()["execute"].as<string>(), " ,\n\t");
165  for (const std::string& n :xtools) {
166  bool _found_calc = false;
167  for (const auto& tool:xtp::QMTools().getObjects()) {
168  if (n.compare(tool.first.c_str()) == 0) {
169  cout << " This is a XTP app" << endl;
170  this->AddTool(xtp::QMTools().Create(n.c_str()));
171  _found_calc = true;
172  }
173  }
174  if (!_found_calc) {
175  for (const auto& tool:ctp::QMTools().getObjects()) {
176 
177  if (n.compare(tool.first.c_str()) == 0) {
178  cout << " This is a CTP app" << endl;
179  this->AddTool(ctp::QMTools().Create(n.c_str()));
180  _found_calc = true;
181  }
182  }
183  }
184  if (!_found_calc) {
185  cout << "Tool " << n << " does not exist\n";
186  StopExecution();
187  }else{
188  cout << "Registered " << n << endl;
189  }
190  }
191  return 1;
192 }
