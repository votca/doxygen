100 template <bool topology, class T>
101 inline bool XYZReader::ReadFrame(T &container) {
102  string line;
103  getline(_fl, line);
104  ++_line;
105  if (!_fl.eof()) {
106  // read the number of atoms
107  Tokenizer tok1(line, " \t");
108  std::vector<std::string> line1;
109  tok1.ToVector(line1);
110  if (line1.size() != 1) {
111  throw std::runtime_error("First line of xyz file should contain number "
112  "of atoms/beads, nothing else.");
113  }
114  int natoms = boost::lexical_cast<int>(line1[0]);
115  if (!topology && natoms != getContainerSize(container)) {
116  throw std::runtime_error(
117  "number of beads in topology and trajectory differ");
118  }
119  // the title line
120  getline(_fl, line);
121  ++_line;
122  // read atoms
123  for (int i = 0; i < natoms; ++i) {
124  getline(_fl, line);
125  ++_line;
126  if (_fl.eof()) {
127  throw std::runtime_error("unexpected end of file in xyz file");
128  }
129  vector<string> fields;
130  Tokenizer tok(line, " ");
131  tok.ToVector(fields);
132  if (fields.size() != 4) {
133  throw std::runtime_error("invalide line " +
134  boost::lexical_cast<string>(_line) +
135  " in xyz file\n" + line);
136  }
137  tools::vec pos = vec(boost::lexical_cast<double>(fields[1]),
138  boost::lexical_cast<double>(fields[2]),
139  boost::lexical_cast<double>(fields[3]));
140 
141  AddAtom<topology, T>(container, fields[0], i, pos);
142  }
143  }
144  return !_fl.eof();
145 }
146 }
