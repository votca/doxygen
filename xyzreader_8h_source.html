103 template <bool topology, class T>
104 inline bool XYZReader::ReadFrame(T &container) {
105  std::string line;
106  std::getline(_fl, line);
107  ++_line;
108  if (!_fl.eof()) {
109  // read the number of atoms
110  tools::Tokenizer tok1(line, " \t");
111  std::vector<std::string> line1 = tok1.ToVector();
112  if (line1.size() != 1) {
113  throw std::runtime_error(
114  "First line of xyz file should contain number "
115  "of atoms/beads, nothing else.");
116  }
117  int natoms = boost::lexical_cast<int>(line1[0]);
118  if (!topology && natoms != getContainerSize(container)) {
119  throw std::runtime_error(
120  "number of beads in topology and trajectory differ");
121  }
122  // the title line
123  getline(_fl, line);
124  ++_line;
125  // read atoms
126  for (int i = 0; i < natoms; ++i) {
127  getline(_fl, line);
128  ++_line;
129  if (_fl.eof()) {
130  throw std::runtime_error("unexpected end of file in xyz file");
131  }
132  tools::Tokenizer tok(line, " ");
133  std::vector<std::string> fields = tok.ToVector();
134  if (fields.size() != 4) {
135  throw std::runtime_error("invalide line " +
136  boost::lexical_cast<std::string>(_line) +
137  " in xyz file\n" + line);
138  }
139  Eigen::Vector3d pos =
140  Eigen::Vector3d(boost::lexical_cast<double>(fields[1]),
141  boost::lexical_cast<double>(fields[2]),
142  boost::lexical_cast<double>(fields[3]));
143 
144  AddAtom<topology, T>(container, fields[0], i, pos);
145  }
146  }
147  return !_fl.eof();
148 }
