113 template <bool topology, class T>
114 inline bool XYZReader::ReadFrame(T &container) {
115  std::string line;
116  tools::getline(_fl, line);
117  ++_line;
118  if (!_fl.eof()) {
119  // read the number of atoms
120  tools::Tokenizer tok1(line, " \t");
121  std::vector<std::string> line1 = tok1.ToVector();
122  if (line1.size() != 1) {
123  throw std::runtime_error(
124  "First line of xyz file should contain number "
125  "of atoms/beads, nothing else.");
126  }
127  Index natoms = boost::lexical_cast<Index>(line1[0]);
128  if (!topology && natoms != getContainerSize(container)) {
129  throw std::runtime_error(
130  "number of beads in topology and trajectory differ");
131  }
132  // the title line
133  tools::getline(_fl, line);
134  ++_line;
135  // read atoms
136  for (Index i = 0; i < natoms; ++i) {
137  tools::getline(_fl, line);
138  ++_line;
139  if (_fl.eof()) {
140  throw std::runtime_error("unexpected end of file in xyz file");
141  }
142  tools::Tokenizer tok(line, " ");
143  std::vector<std::string> fields = tok.ToVector();
144  if (fields.size() != 4) {
145  throw std::runtime_error("invalide line " +
146  boost::lexical_cast<std::string>(_line) +
147  " in xyz file\n" + line);
148  }
149  Eigen::Vector3d pos =
150  Eigen::Vector3d(boost::lexical_cast<double>(fields[1]),
151  boost::lexical_cast<double>(fields[2]),
152  boost::lexical_cast<double>(fields[3]));
153 
154  AddAtom<topology, T>(container, fields[0], i, pos);
155  }
156  }
157  return !_fl.eof();
158 }
