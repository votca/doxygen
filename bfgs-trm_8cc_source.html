83  void BFGSTRM::Optimize() {
84 
85  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM convergence of total energy: %1$8.6f Hartree ") % _convergence).str() << flush;
86  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM convergence of RMS Force: %1$8.6f Hartree/Bohr ") % _RMSForce_convergence).str() << flush;
87  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM convergence of Max Force: %1$8.6f Hartree/Bohr ") % _MaxForce_convergence).str() << flush;
88  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM convergence of RMS Step: %1$8.6f Bohr ") % _RMSStep_convergence).str() << flush;
89  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM convergence of Max Step: %1$8.6f Bohr ") % _MaxStep_convergence).str() << flush;
90  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM initial trust radius: %1$8.6f Bohr") % _trust_radius).str() << flush;
91  bool _converged = false;
92 
93  // in 1st iteration, get the energy of the initial configuration
94  if (_iteration == 0) {
95  _last_energy = GetEnergy(); // in Hartree
96  WriteTrajectory();
97  }
98 
99  // now change geometry until convergence
100  while (!_converged) {
101  _iteration++;
102  _update_hessian = true; // always update Hessian after each accepted step
103 
104  // calculate the forces for the present configuration
105  _force_engine.Calculate(_last_energy);
106  _force = _force_engine.GetForces();
107 
108  _step_accepted = false;
109  while (!_step_accepted) {
110 
111  // determine new trial configuration according to BFGS
112  BFGSStep();
113 
114  // update coordinates in segment
115  UpdateSegment();
116 
117  // for the updated geometry, get new reference energy
118  _new_energy = GetEnergy();
119  _energy_delta = _new_energy - _last_energy;
120 
121  // check the energy at the trial coordinates, accept/reject step, and adjust trust radius
122  AcceptReject();
123 
124 
125  } // checking step to be trusted
126 
127  // after the step is accepted, we can shift the stored data
128  _last_energy = _new_energy;
129  _old_xyz = _current_xyz;
130  _current_xyz = _trial_xyz;
131  _force_old = _force;
132 
133  // Write to trajectory
134  WriteTrajectory();
135 
136 
137  // Check convergence criteria
138  _converged = GeometryConverged();
139 
140  // Report summary of the iteration
141  Report();
142 
143  if (_iteration == _max_iteration) {
144  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: not converged after %2$d iterations ") % _iteration % _max_iteration).str() << flush;
145  break;
146  }
147 
148  } // convergence
149 
150  return;

154  /* Accept/reject the new geometry and adjust trust radius, if required */
155  void BFGSTRM::AcceptReject() {
156 
157  if (_energy_delta > 0.0) {
158  // total energy has unexpectedly increased, half the trust radius
159  _trust_radius = 0.5 * _trust_radius;
160  // Hessian should not be updated for reduced trust radius
161  _update_hessian = false;
162  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: step rejected ") % _iteration).str() << flush;
163  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: new trust radius %2$8.6f Bohr") % _iteration % _trust_radius).str() << flush;
164  if (_trust_radius < 1e-5) throw runtime_error("Trust radius vanishing! Stopping optimization! ");
165  // repeat BGFSStep with new trust radius
166  } else {
167  // total energy has decreased, we accept the step but might update the trust radius
168  _step_accepted = true;
169  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: step accepted ") % _iteration).str() << flush;
170  // adjust trust radius, if required
171  double _tr_check = _energy_delta / _delta_energy_estimate;
172  if (_tr_check > 0.75 && 1.25 * sqrt(_norm_delta_pos) > _trust_radius) {
173  _trust_radius = 2.0 * _trust_radius;
174  // if ( _trust_radius > _trust_radius_max) _trust_radius = _trust_radius_max;
175  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: new trust radius %2$8.6f Bohr") % _iteration % _trust_radius).str() << flush;
176  } else if (_tr_check < 0.25) {
177  _trust_radius = 0.25 * sqrt(_norm_delta_pos);
178  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: new trust radius %2$8.6f Bohr") % _iteration % _trust_radius).str() << flush;
179  }
180 
181  }
182 
183  return;

337  /* Update the Hessian */
338  void BFGSTRM::UpdateHessian() {
339 
340  // delta is new - old
341  _delta_pos = _current_pos - _previous_pos;
342  _norm_delta_pos = ub::inner_prod(_delta_pos, _delta_pos);
343 
344  // we have no Hessian in the first iteration => start with something
345  if (_iteration == 1) {
346  for (unsigned _i = 0; _i < _dim; _i++) {
347  _hessian(_i, _i) = 1.0; // unit matrix
348  }
349 
350  } else {
351  /* for later iteration, we can make use of an iterative refinement of
352  * the initial Hessian based on the gradient (force) history
353  */
354 
355  ub::vector<double> _delta_gradient = _current_gradient - _previous_gradient;
356 
357  // second term in BFGS update (needs current Hessian)
358  ub::vector<double> _temp1 = ub::prod(_hessian, _delta_pos);
359  ub::vector<double> _temp2 = ub::prod(ub::trans(_delta_pos), _hessian);
360  _hessian -= ub::outer_prod(_temp1, _temp2) / ub::inner_prod(_delta_pos, _temp1);
361 
362  // first term in BFGS update
363  _hessian += ub::outer_prod(_delta_gradient, _delta_gradient) / ub::inner_prod(_delta_gradient, _delta_pos);
364 
365  // symmetrize Hessian (since d2E/dxidxj should be symmetric)
366  for (unsigned _i = 0; _i < _dim; _i++) {
367  for (unsigned _j = _i + 1; _j < _dim; _j++) {
368  double _sum = 0.5 * (_hessian(_i, _j) + _hessian(_j, _i));
369  _hessian(_i, _j) = _sum;
370  _hessian(_j, _i) = _sum;
371  }
372  }
373  } // update Hessian
374 
375  return;
376  }

404  /* Regularize step in case of prediction outside of Trust Region */
405  void BFGSTRM::RegularizeStep() {
406 
407  double _max_step_squared = 0.0;
408 
409  // get eigenvalues and eigenvectors of Hessian
410  ub::matrix<double> _eigenvectors;
411  ub::vector<double> _eigenvalues;
412  votca::tools::linalg_eigenvalues(_hessian, _eigenvalues, _eigenvectors);
413 
414  // start value for lambda a bit lower than lowest eigenvalue of Hessian
415  double _lambda;
416  if (_eigenvalues(0) > 0.0) {
417  _lambda = -0.05 * std::abs(_eigenvalues(0));
418  } else {
419  _lambda = 1.05 * _eigenvalues(0);
420  }
421 
422  // for constrained step, we expect
423  _max_step_squared = _norm_delta_pos;
424  while (OutsideTrustRegion(_max_step_squared)) {
425  _max_step_squared = 0.0;
426  _lambda -= 0.05 * std::abs(_eigenvalues(0));
427  for (unsigned _i = 0; _i < _dim; _i++) {
428  ub::vector<double> _slice(_dim, 0.0);
429  for (unsigned _j = 0; _j < _dim; _j++) {
430  _slice(_j) = _eigenvectors(_j, _i);
431  }
432 
433  //cout << " forece is of dim " << _current_force.size1() << " " << _current_force.size2() << endl;
434  double _temp = ub::inner_prod(_slice, _current_gradient);
435  //cout << " slice is of dim " << _slice.size1() << " " << _slice.size2() << endl; cout << " tmep is of dim " << _temp.size1() << " " << _temp.size2() << endl;
436  // cout << " into max_step_sq " << _temp << " and " << ( _eigenvalues(_i) - _lambda ) << endl;
437  _max_step_squared += _temp * _temp / (_eigenvalues(_i) - _lambda) / (_eigenvalues(_i) - _lambda);
438  }
439  }
440 
441  //CTP_LOG(ctp::logDEBUG,_log) << " BFGS-TRM: with lambda " << _lambda << " max step sq is " << _max_step_squared << flush;
442 
443  _delta_pos = ub::zero_vector<double>(_dim);
444  for (unsigned _i = 0; _i < _dim; _i++) {
445  ub::vector<double> _slice(_dim, 0.0);
446  for (unsigned _j = 0; _j < _dim; _j++) {
447  _slice(_j) = _eigenvectors(_j, _i);
448  }
449 
450  _delta_pos -= _slice * ub::inner_prod(_slice, _current_gradient) / (_eigenvalues(_i) - _lambda);
451  }
452 
453  _norm_delta_pos = ub::inner_prod(_delta_pos, _delta_pos); //_max_step_squared;
454 
455  return;
456  }

509 
510 
511  std::vector< ctp::Atom* > _atoms;
512  std::vector< ctp::Atom* > ::iterator ait;
513  _atoms = _segments[0]->Atoms();
514 
515  // write logger to log file
