27  void BFGSTRM::Optimize(const Eigen::VectorXd& initialparameters) {
28  _parameters = initialparameters;
29 
30  Eigen::VectorXd gradient = Eigen::VectorXd::Zero(_parameters.size());
31  Eigen::VectorXd delta_p_trial = Eigen::VectorXd::Zero(_parameters.size());
32 
33  double lastcost = _costfunction.EvaluateCost(_parameters);
34  Eigen::VectorXd last_gradient = Eigen::VectorXd::Zero(_parameters.size());
35  double delta_cost = 0;
36  for (_iteration = 0; _iteration < _max_iteration; _iteration++) {
37  gradient = _costfunction.EvaluateGradient(_parameters);
38  bool step_accepted = false;
39  for (int i = 0; i < 100; i++) {
40  TrustRegion subproblem;
41  delta_p_trial = subproblem.CalculateStep(gradient, _hessian, _trust_radius);
42  double trialcost = _costfunction.EvaluateCost(_parameters + delta_p_trial);
43  delta_cost = trialcost - lastcost;
44  step_accepted = AcceptRejectStep(delta_p_trial, gradient, delta_cost);
45  if (step_accepted) {
46  _cost = trialcost;
47  _parameters += delta_p_trial;
48  break;
49  }
50  }
51  if (_iteration > 0) {
52  UpdateHessian(delta_p_trial, gradient - last_gradient);
53  }
54  lastcost = _cost;
55  last_gradient = gradient;
56  for (auto& func : _callbacks) {
57  func();
58  }
59  if (_costfunction.Converged(delta_p_trial, delta_cost, gradient)) {
60  break;
61  } else if (_iteration == _max_iteration - 1) {
62  _success = false;
63  if (_logging) {
64  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: not converged after %2$d iterations ")
65  % _iteration % _max_iteration).str() << std::flush;
66  }
67  }
68  }
69  return;
70  }

72  /* Accept/reject the new geometry and adjust trust radius, if required */
73  bool BFGSTRM::AcceptRejectStep(const Eigen::VectorXd& delta_p, const Eigen::VectorXd& gradient, double cost_delta) {
74  bool step_accepted = false;
75  if (cost_delta > 0.0) {
76  // total energy has unexpectedly increased, half the trust radius
77  _trust_radius = 0.25 * _trust_radius;
78  if (_logging) {
79  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: step rejected ")
80  % _iteration).str() << std::flush;
81  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: new trust radius %2$8.10e")
82  % _iteration % _trust_radius).str() << std::flush;
83  }
84  } else {
85  // total energy has decreased, we accept the step but might update the trust radius
86  step_accepted = true;
87  // adjust trust radius, if required
88  double tr_check = cost_delta / QuadraticEnergy(gradient, delta_p);
89  double norm_delta_p = delta_p.squaredNorm();
90  if (tr_check > 0.75 && 1.25 * norm_delta_p > _trust_radius * _trust_radius) {
91  _trust_radius = 2.0 * _trust_radius;
92  } else if (tr_check < 0.25) {
93  _trust_radius = 0.25 * _trust_radius;
94  }
95  if (_logging) {
96  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: step accepted ")
97  % _iteration).str() << std::flush;
98  CTP_LOG(ctp::logINFO, *_pLog) << (boost::format("BFGS-TRM @iteration %1$d: new trust radius %2$8.10e")
99  % _iteration % _trust_radius).str() << std::flush;
100  }
101  }
102  return step_accepted;
103  }

105  void BFGSTRM::UpdateHessian(const Eigen::VectorXd& delta_pos, const Eigen::VectorXd& delta_gradient) {
106  // second term in BFGS update (needs current Hessian)
107  _hessian -= _hessian * delta_pos * delta_pos.transpose() * _hessian.transpose() / (delta_pos.transpose() * _hessian * delta_pos).value();
108  // first term in BFGS update
109  _hessian += (delta_gradient * delta_gradient.transpose()) / (delta_gradient.transpose() * delta_pos);
110  // symmetrize Hessian (since d2E/dxidxj should be symmetric)
111  _hessian = 0.5 * (_hessian + _hessian.transpose());
112  return;
113  }

115  /* Estimate energy change based on quadratic approximation */
116  double BFGSTRM::QuadraticEnergy(const Eigen::VectorXd& gradient, const Eigen::VectorXd& delta_pos) const {
117  return (gradient.transpose() * delta_pos).value() + 0.5 * (delta_pos.transpose() * _hessian * delta_pos).value();
118  }
