25 double Vssmgroup::Timestep(votca::tools::Random2 *RandomVariable){
26  double timestep;
27 
28  double rand_u = 1-RandomVariable->rand_uniform();
29  while(rand_u == 0) {
30  rand_u = 1-RandomVariable->rand_uniform();
31  }
32 
33  timestep = (-1.0/tot_probsum)*log(rand_u);
34  return timestep;
35 
36 }

67 Event* Vssmgroup::Choose_event_device(Events* events, Bsumtree* non_injection_rates, Bsumtree* left_injection_rates, Bsumtree* right_injection_rates, votca::tools::Random2 *RandomVariable){
68 
69  double randn = 0.0;
70  while(randn == 0.0) {
71  randn = tot_probsum*RandomVariable->rand_uniform();
72  }
73 
74  long event_ID;
75  Event* chosenevent;
76  if(randn<left_inject_probsum) { // injection event
77  event_ID = left_injection_rates->search(randn);
78  chosenevent = events->get_injection_event(0,event_ID);
79  }
80  else {
81  randn -= left_inject_probsum;
82  if(randn<right_inject_probsum){
83  event_ID = right_injection_rates->search(randn);
84  chosenevent = events->get_injection_event(1,event_ID);
85  }
86  else {
87  randn -= right_inject_probsum;
88  event_ID = non_injection_rates->search(randn);
89  chosenevent = events->get_non_injection_event(event_ID);
90  }
91  }
92  return chosenevent;
93 }
