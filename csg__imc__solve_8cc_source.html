52 void CG_IMC_solve::Run() {
53  std::string imcfile = _op_vm["imcfile"].as<std::string>();
54  std::string gmcfile = _op_vm["gmcfile"].as<std::string>();
55 
56  double reg = _op_vm["regularization"].as<double>();
57 
58  Eigen::MatrixXd A = votca::csg::imcio_read_matrix(gmcfile);
59  votca::tools::Table B;
60  B.Load(imcfile);
61 
62  votca::tools::Table x;
63  x.resize(B.size());
64  x.x() = B.x();
65 
66  // Calculating https://en.wikipedia.org/wiki/Tikhonov_regularization
67 
68  Eigen::MatrixXd ATA = A.transpose() * A;
69 
70  // We could add and invert but for stability reasons we will diagonalize the
71  // matrix which is symmetric by construction
72 
73  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(ATA);
74 
75  Eigen::VectorXd inv_diag = Eigen::VectorXd::Zero(es.eigenvalues().size());
76  double etol = 1e-12;
77 
78  votca::Index numerics_unstable = 0;
79  // Constructing pseudoinverse if not stable
80  // https://en.wikipedia.org/wiki/Moore%E2%80%93Penrose_inverse
81  for (votca::Index i = 0; i < es.eigenvalues().size(); i++) {
82  if (std::abs(es.eigenvalues()[i] + reg) < etol) {
83  numerics_unstable++;
84  } else {
85  inv_diag[i] = 1 / (es.eigenvalues()[i] + reg);
86  }
87  }
88 
89  if (numerics_unstable > 0) {
90  std::cout
91  << "Regularisation parameter was to small, a pseudo inverse was "
92  "constructed instead.\n Use a larger regularisation parameter R."
93  " Smallest (eigenvalue+R)="
94  << (es.eigenvalues().array() + reg).abs().minCoeff() << " Found "
95  << numerics_unstable << " eigenvalues of " << es.eigenvalues().size()
96  << " below " << etol << std::endl;
97  }
98 
99  Eigen::MatrixXd inverse =
100  es.eigenvectors() * inv_diag.asDiagonal() * es.eigenvectors().transpose();
101 
102  x.y() = -inverse * A.transpose() * B.y();
103 
104  std::string idxfile = _op_vm["idxfile"].as<std::string>();
105  std::vector<std::pair<std::string, votca::tools::RangeParser> > ranges =
106  votca::csg::imcio_read_index(idxfile);
107  for (std::pair<std::string, votca::tools::RangeParser>& range : ranges) {
108  votca::tools::Table tbl;
109  for (votca::Index r : range.second) {
110  tbl.push_back(x.x(r - 1), x.y(r - 1), 'i');
111  }
112  tbl.Save(range.first + ".dpot.imc");
113  }
114 }
