74 void QMNBList::GenerateSuperExchange() {
75 
76  //QMNBList bridged_nblist;
77 
78  // loop over all donor/acceptor pair types
79  for (std::list<SuperExchangeType*>::iterator itDA = _superexchange.begin(); itDA != _superexchange.end(); itDA++) {
80 
81  cout << endl << " ... ... Processing superexchange pairs of type " << (*itDA)->asString() << "\n" << flush;
82  int _bridged_pairs = 0;
83  int _bridged_and_direct_pairs = 0;
84  //int bridged_molecules = 0;
85 
86  // vector of neighboring segments of the donor/acceptor type
87  map< int, Segment*> _ns;
88 
89  // loop over all segments in the topology
90  for (std::vector< Segment* >::iterator segit = _top->Segments().begin(); segit != _top->Segments().end(); segit++) {
91 
92  // check if this is a bridge
93  Segment* segment = *segit;
94  string name = segment->getName();
95 
96  if ((*itDA)->isOfBridge(name)) {
97  QMNBList::partners *_partners = FindPartners(segment);
98 
99  // loop over all partners of a segment
100  QMNBList::partners::iterator itp;
101  if (_partners != NULL) {
102  map< int, Segment*> _neighbors;
103  for (itp = _partners->begin(); itp != _partners->end(); itp++) {
104  Segment *nb = itp->first;
105  //QMPair *pair = itp->second;
106  // check if the neighbor is of a donor or acceptor type
107  if ((*itDA)->isOfDonorAcceptor(nb->getName())) _neighbors[ nb->getId() ] = nb;
108  } // end of the loop of all partners of a segment
109 
110  // create new pairs if there are more than one neighbors of DA type
111  if (_neighbors.size() > 1) {
112  for (map<int, Segment*>::iterator it1 = _neighbors.begin(); it1 != _neighbors.end(); it1++) {
113  map<int, Segment*>::iterator it_diag = it1;
114  for (map<int, Segment*>::iterator it2 = ++it_diag; it2 != _neighbors.end(); it2++) {
115 
116  QMPair *pair = FindPair(it1->second, it2->second);
117 
118  if (pair == NULL ) { // no connection between donor and acceptor
119  QMPair* _pair = Add(it1->second, it2->second);
120  _pair->setType( QMPair::SuperExchange );
121  _pair->AddBridgingSegment( segment );
122  _bridged_pairs++;
123  } else { // pair type is already there
124  if ( pair->getType() == QMPair::Hopping ) {
125  _bridged_and_direct_pairs++;
126  pair->setType( QMPair::SuperExchangeAndHopping );
127  }
128  pair->AddBridgingSegment( segment );
129  }
130  }
131  }
132  } //
133  } // end of the check of zero partners
134  } // end of if this is a bridged pair
135  } // end of the loop of all segments
136 
137  cout << "Added " << _bridged_pairs + _bridged_and_direct_pairs << " superexchange with " << _bridged_and_direct_pairs << " mixed pairs" << endl;
138 
139  } // end of the loop over donor/acceptor types
140 
141  map<int, int> npairs;
142  for ( QMNBList::iterator pit = this->begin(); pit != this->end(); pit++ ) {
143  QMPair *pair = (*pit);
144  npairs[pair->getType()] += 1;
145  }
146 
147  cout << "Hopping only pairs: " << npairs[QMPair::Hopping] << endl;
148  cout << "Superexchange pairs: " << npairs[QMPair::SuperExchange] << endl;
149  cout << "Superexchange and hopping pairs: " << npairs[QMPair::SuperExchangeAndHopping] << endl;
150 
151  return;
152 }
