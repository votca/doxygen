90 std::complex<double> ccgamma(std::complex<double> z,int OPT)
91 {
92  std::complex<double> g,z0,z1;
93  double x0,q1,q2,x,y,th,th1,th2,g0,gr,gi,gr1,gi1;
94  double na=0,t,x1=0,y1,sr,si;
95  //int i,j,k;
96  int j,k;
97 
98  static double a[] = {
99  8.333333333333333e-02,
100  -2.777777777777778e-03,
101  7.936507936507937e-04,
102  -5.952380952380952e-04,
103  8.417508417508418e-04,
104  -1.917526917526918e-03,
105  6.410256410256410e-03,
106  -2.955065359477124e-02,
107  1.796443723688307e-01,
108  -1.39243221690590};
109 
110  x = real(z);
111  y = imag(z);
112  if (x > 171) return std::complex<double>(1e308,0);
113  if ((y == 0.0) && (x == (int)x) && (x <= 0.0))
114  return std::complex<double>(1e308,0);
115  else if (x < 0.0) {
116  x1 = x;
117  y1 = y;
118  x = -x;
119  y = -y;
120  }
121  x0 = x;
122  if (x <= 7.0) {
123  na = (int)(7.0-x);
124  x0 = x+na;
125  }
126  q1 = sqrt(x0*x0+y*y);
127  th = atan(y/x0);
128  gr = (x0-0.5)*log(q1)-th*y-x0+0.5*log(2.0*M_PI);
129  gi = th*(x0-0.5)+y*log(q1)-y;
130  for (k=0;k<10;k++){
131  t = pow(q1,-1.0-2.0*k);
132  gr += (a[k]*t*cos((2.0*k+1.0)*th));
133  gi -= (a[k]*t*sin((2.0*k+1.0)*th));
134  }
135  if (x <= 7.0) {
136  gr1 = 0.0;
137  gi1 = 0.0;
138  for (j=0;j<na;j++) {
139  gr1 += (0.5*log((x+j)*(x+j)+y*y));
140  gi1 += atan(y/(x+j));
141  }
142  gr -= gr1;
143  gi -= gi1;
144  }
145  if (x1 <= 0.0) {
146  q1 = sqrt(x*x+y*y);
147  th1 = atan(y/x);
148  sr = -sin(M_PI*x)*cosh(M_PI*y);
149  si = -cos(M_PI*x)*sinh(M_PI*y);
150  q2 = sqrt(sr*sr+si*si);
151  th2 = atan(si/sr);
152  if (sr < 0.0) th2 += M_PI;
153  gr = log(M_PI/(q1*q2))-gr;
154  gi = -th1-th2-gi;
155  x = x1;
156  y = y1;
157  }
158  if (OPT == 0) {
159  g0 = exp(gr);
160  gr = g0*cos(gi);
161  gi = g0*sin(gi);
162  }
163  g = std::complex<double>(gr,gi);
164  return g;
165 }
