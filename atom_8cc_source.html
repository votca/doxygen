29 Atom::Atom(Index resnr, std::string md_atom_name, Index atom_id,
30  Eigen::Vector3d pos, std::string type)
31  : _id(atom_id), _name(md_atom_name), _resnr(resnr), _pos(pos) {
32 
33  std::string elename = GetElementFromString(md_atom_name);
34  std::string eletype = GetElementFromString(type);
35  tools::Elements ele;
36  bool found_element_name = true;
37  bool found_element_type = true;
38  try {
39  ele.getMass(elename);
40  } catch (std::runtime_error&) {
41  found_element_name = false;
42  }
43 
44  try {
45  ele.getMass(eletype);
46  } catch (std::runtime_error&) {
47  found_element_type = false;
48  }
49 
50  if (found_element_name && found_element_type) {
51  if (elename != eletype) {
52  throw std::runtime_error("Elements " + elename + " and" + eletype +
53  " from atom name: " + md_atom_name +
54  " and atom type:" + type + " do not match.");
55  }
56  _element = elename;
57  } else if (found_element_name) {
58  _element = elename;
59  } else if (found_element_type) {
60  _element = elename;
61  } else {
62  throw std::runtime_error("Could not get Element from atom name:" +
63  md_atom_name + " or atom type:" + type);
64  }
65 }
