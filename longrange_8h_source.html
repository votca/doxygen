74 void Longrange::Initialize (Graph* graph, Globaleventinfo* globevent) {
75 
76  number_of_layers = ceil(graph->sim_box_size.x()/graph->hopdist);
77 
78  vector<double> positional_sum;
79  vector<int> number_of_charges;
80  vector<bool> flagged_for_deletion;
81 
82  for (int ilayer=0; ilayer<number_of_layers; ilayer++) {
83  positional_sum.push_back(0.0);
84  number_of_charges.push_back(0);
85  flagged_for_deletion.push_back(false);
86  }
87 
88  for (unsigned int inode=0; inode<graph->nodes.size(); inode++) {
89  double posx = graph->nodes[inode]->node_position.x();
90  int iposx = floor(posx/graph->hopdist);
91  positional_sum[iposx] += posx;
92  number_of_charges[iposx]++;
93  graph->nodes[inode]->layer_index = iposx;
94  }
95 
96  int rem_layers = 0;
97 
98  for (int ilayer=0; ilayer<number_of_layers; ilayer++) {
99  if(number_of_charges[ilayer] != 0) {
100  positional_average.push_back(positional_sum[ilayer]/number_of_charges[ilayer]);
101  }
102  else {
103  flagged_for_deletion[ilayer] = true;
104  rem_layers++;
105  }
106  }
107 
108  number_of_layers -= rem_layers;
109 
110  layercharge.resize(number_of_layers);
111  longrange_cache.resize(number_of_layers);
112 
113  precalculate_disc_contrib.resize(number_of_layers);
114  first_contributing_layer.resize(number_of_layers);
115  final_contributing_layer.resize(number_of_layers);
116 
117  for (int ilayer=0;ilayer<number_of_layers;ilayer++) {
118  // define for every layer, how many other layers are within the coulomb cut off radius from this layer
119  double define_layer = positional_average[ilayer];
120 
121  int start_index = 0;
122  bool startfound = false;
123  while (!startfound) {
124  double start_layer = positional_average[start_index];
125  if((define_layer-start_layer)<=globevent->coulcut) {
126  startfound = true;
127  first_contributing_layer[ilayer]=start_index;
128  }
129  start_index++;
130  }
131 
132  int final_index = number_of_layers-1;
133  bool finalfound = false;
134  while (!finalfound) {
135  double final_layer = positional_average[final_index];
136  if((final_layer-define_layer)<=globevent->coulcut) {
137  finalfound = true;
138  final_contributing_layer[ilayer]=final_index;
139  }
140  final_index--;
141  }
142 
143  int number_of_contributing_layers = final_contributing_layer[ilayer]-first_contributing_layer[ilayer]+1;
144  precalculate_disc_contrib[ilayer].resize(number_of_contributing_layers);
145  }
146 
147  for(int i=0; i<number_of_layers; i++) {
148  layercharge[i] = 0.0;
149  longrange_cache[i] = 0.0;
150  int first_layer = first_contributing_layer[i];
151  int final_layer = final_contributing_layer[i];
152  for (int j=first_layer; j<=final_layer; j++) {
153  precalculate_disc_contrib[i][j-first_layer] = Calculate_disc_contrib(i,j,graph->sim_box_size,globevent);
154  }
155  }
156 }

158 double Longrange::Calculate_disc_contrib(int calculate_layer, int contrib_layer, myvec sim_box_size,Globaleventinfo* globevent) {
159 
160  double calcpos = positional_average[calculate_layer];
161  double contribpos = positional_average[contrib_layer];
162  double rdist = contribpos-calcpos;
163 
164  double contrib = globevent->coulcut-fabs(rdist); // Direct contribution (no image), factor 2 pi is missing, included in compute_longrange
165  double radiussqr = globevent->coulcut*globevent->coulcut-rdist*rdist; //radius of contributing disc
166 
167  for (long i=0; i<globevent->nr_of_lr_images; i++) {
168 
169  // Calculate contribution from images
170  long dist1;
171  long dist2;
172  long sign;
173  if (ldiv(i,2).rem==0) { // even generation (x-position of image charges is -p.x + 2*j*L, j=...,-1,0,1,...)
174  sign = -1;
175  dist1 = i*sim_box_size.x() + 2*contribpos - rdist;
176  dist2 = i*sim_box_size.x() + 2*sim_box_size.x() - 2*contribpos + rdist;
177  }
178  else { // odd generation (x-position of image charges is -p.x + 2*j*L, j=...,-1,0,1,...)
179  sign = 1;
180  dist1 = (i+1)*sim_box_size.x() + rdist;
181  dist2 = (i+1)*sim_box_size.x() - rdist;
182  }
183  double diag1 = sqrt(radiussqr+dist1*dist1);
184  double diag2 = sqrt(radiussqr+dist2*dist2);
185  contrib += sign*(diag1-fabs(dist1)+diag2-fabs(dist2));
186  }
187  return contrib;
188 }

190 double Longrange::Calculate_longrange(int layer, bool cut_out_discs, myvec sim_box_size, Globaleventinfo* globevent) {
191  // Potential is expressed in multiples of e/(4*pi*epsilon) (with e the elementary charge>0)
192  double plate_contrib1 = 0.0;
193  double disc_contrib = 0.0;
194  double PI = 3.14159265358979323846264338327950288419716939937510;
195  for(int i=0; i<layer; i++) {
196  double charge_i = 1.0*layercharge[i];
197  double position_i = 1.0*positional_average[i];
198  plate_contrib1 += position_i*charge_i; // potential of a charged plate between two electrodes
199  double distance = layer-position_i;
200  int first_layer = first_contributing_layer[layer];
201  if (distance<=globevent->coulcut) {
202  // Cut out short-range sphere
203  disc_contrib -= charge_i*precalculate_disc_contrib[layer][i-first_layer];
204  }
205  }
206  double plate_contrib2 = 0.0;
207  for(int i=layer; i<number_of_layers; i++) {
208  double charge_i = 1.0*layercharge[i];
209  double rel_position_i = 1.0*(sim_box_size.x()-positional_average[i]);
210  plate_contrib2 += rel_position_i*charge_i; // potential of a charged plate between two electrodes
211  double distance = positional_average[i]-layer;
212  int first_layer = first_contributing_layer[layer];
213  if (distance<=globevent->coulcut) {
214  // Cut out short-range sphere
215  disc_contrib -= charge_i*precalculate_disc_contrib[layer][i-first_layer];
216  }
217  }
218  if (!cut_out_discs) { disc_contrib = 0.0; }
219  double layerpos = positional_average[layer];
220  return 4*PI*(plate_contrib1*(1-layerpos/sim_box_size.x()) + plate_contrib2*(layerpos/sim_box_size.x()) + 0.5*disc_contrib)/(sim_box_size.y()*sim_box_size.z());
221 }
