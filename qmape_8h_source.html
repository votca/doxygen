109 
110  key = "options."+Identify();
111 
112 
113  string package_xml = options->ifExistsReturnElseThrowRuntimeError<string>(key+".dft.dftengine");
114  load_property_from_xml(_dft_opt, package_xml.c_str());
115 
116  key = "options."+Identify()+".gwbse";
117  if ( options->exists(key)) {
118  cout << endl << "... ... Configure for excited states (DFT+GWBSE)" << flush;
119  string gwbse_xml = options->ifExistsReturnElseThrowRuntimeError<string>(key+".gwbse_options");

157 ctp::XJob QMAPE::ProcessInputString(ctp::Job *job,ctp::Topology *top, ctp::QMThread *thread) {
158 
159  // Input string looks like this:
160  // <id1>:<name1>:<mpsfile1> <id2>:<name2>: ... ... ...
161 
162  string input = job->getInput().as<string>();
163  vector<ctp::Segment*> qmSegs;
164  vector<string> qmSegMps;
165  vector<string> split;
166  Tokenizer toker(input, " \t\n");
167  toker.ToVector(split);
168 
169  for (unsigned i = 0; i < split.size(); ++i) {
170 
171  string id_seg_mps = split[i];
172  vector<string> split_id_seg_mps;
173  Tokenizer toker(id_seg_mps, ":");
174  toker.ToVector(split_id_seg_mps);
175 
176  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
177  string segName = split_id_seg_mps[1];
178  string mpsFile = split_id_seg_mps[2];
179 
180  ctp::Segment *seg = top->getSegment(segId);
181  if (seg->getName() != segName) {
182  CTP_LOG(ctp::logERROR,*(thread->getLogger()))
183  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
184  << " maltagged as " << segName << ". Skip job ..." << flush;
185  throw std::runtime_error("Input does not match topology.");
186  }
187 
188  qmSegs.push_back(seg);
189  qmSegMps.push_back(mpsFile);
190  }
191 
192  return ctp::XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
193 }

196 ctp::Job::JobResult QMAPE::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *thread) {
197 
198  // SILENT LOGGER FOR QMPACKAGE
199  ctp::Logger* log = thread->getLogger();
200  ctp::Logger qlog;
201  qlog.setReportLevel(ctp::logDEBUG);
202  qlog.setMultithreading(_maverick);
203  qlog.setPreface(ctp::logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
204  qlog.setPreface(ctp::logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
205  qlog.setPreface(ctp::logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
206  qlog.setPreface(ctp::logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
207 
208  // CREATE XJOB FROM JOB INPUT STRING
209  CTP_LOG(ctp::logINFO,*log)
210  << "Job input = " << job->getInput().as<string>() << flush;
211  ctp::XJob xjob = this->ProcessInputString(job, top, thread);
212 
213  // SETUP POLAR TOPOLOGY (GENERATE VS LOAD IF PREPOLARIZED)
214  if (_polar_bg_arch == "") {
215  CTP_LOG(ctp::logINFO,*log) << "Mps-Mapper: Generate FGC FGN BGN" << flush;
216  _mps_mapper.Gen_FGC_FGN_BGN(top, &xjob, thread);
217  }
218  else {
219  CTP_LOG(ctp::logINFO,*log) << "Mps-Mapper: Generate FGC, load FGN BGN from '"
220  << _polar_bg_arch << "'" << flush;
221  _mps_mapper.Gen_FGC_Load_FGN_BGN(top, &xjob, _polar_bg_arch, thread);
222  }
223  CTP_LOG(ctp::logINFO,*log) << xjob.getPolarTop()->ShellInfoStr() << flush;
224 
225  // SETUP MM METHOD
226  ctp::PEwald3D3D cape = ctp::PEwald3D3D(top, xjob.getPolarTop(), _options,
227  thread->getLogger());
228  if (_pdb_check)
229  cape.WriteDensitiesPDB(xjob.getTag()+".densities.pdb");
230 
231  // SETUP QMAPE
232  QMAPEMachine machine = QMAPEMachine(&xjob, &cape, _options, "options.qmape");
233  machine.setLog(thread->getLogger());
234 
235  // EVALUATE: ITERATE UNTIL CONVERGED
236  machine.Evaluate(&xjob);
237 
238  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
239  ctp::Job::JobResult jres = ctp::Job::JobResult();
240  jres.setOutput(xjob.getInfoLine());
241  jres.setStatus(ctp::Job::COMPLETE);
242 
243  return jres;
244 }
