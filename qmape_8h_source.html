102 void QMAPE::Initialize(Property *options) {
103 
104  _options = options;
105 
106  cout << endl
107  << "... ... Initialized with " << _nThreads << " threads. "
108  << flush;
109 
110  _maverick = (_nThreads == 1) ? true : false;
111 
112  string key = "options."+Identify();
113 
114  if ( options->exists(key+".job_file")) {
115  _jobfile = options->get(key+".job_file").as<string>();
116  }
117  else {
118  throw std::runtime_error("Job-file not set. Abort.");
119  }
120 
121  key = "options"+Identify()+".ewald";
122  if (options->exists(key+".mapping")) {
123  _xml_file = options->get(key+".mapping").as< string >();
124  }
125  else {
126  cout << endl;
127  throw std::runtime_error("Multipole mapping file not set. Abort.");
128  }
129  if ( options->exists(key+".mps_table")) {
130  _mps_table = options->get(key+".mps_table").as<string>();
131  }
132  else {
133  cout << endl;
134  throw std::runtime_error("Background mps table not set. Abort.");
135  }
136  if (options->exists(key+".polar_bg")) {
137  _polar_bg_arch = options->get(key+".polar_bg").as<string>();
138  }
139  else { _polar_bg_arch = ""; }
140  if (options->exists(key+".pdb_check")) {
141  _pdb_check = options->get(key+".pdb_check").as<bool>();
142  }
143  else { _pdb_check = false; }
144  if (options->exists(key+".ptop_check")) {
145  _ptop_check = options->get(key+".ptop_check").as<bool>();
146  }
147  else { _ptop_check = false; }
148 
149 
150  key = "options."+Identify();
151  if ( options->exists(key+".dftpackage")) {
152  string package_xml = options->get(key+".dftpackage").as< string >();
153  load_property_from_xml(_qmpack_opt, package_xml.c_str());
154  _package = _qmpack_opt.get("package.name").as< string >();
155  }
156  else {
157  throw runtime_error("No DFT package specified.");
158  }
159 
160 
161  key = "options."+Identify()+".gwbse";
162  if ( options->exists(key)) {
163  cout << endl << "... ... Configure for excited states (DFT+GWBSE)" << flush;
164  if ( options->exists(key+".gwbse_options")) {
165  string gwbse_xml = options->get(key+".gwbse_options").as< string >();
166  load_property_from_xml(_gwbse_opt, gwbse_xml.c_str());
167  // _gwbse = _gwbse_opt.get("package.name").as< string >();
168  }
169  else {
170  throw runtime_error("GWBSE options not specified.");
171  }
172  _state = options->get(key+".state").as< int >();
173  }
174  else {
175  cout << endl << "... ... Configure for ground states (DFT)" << flush;
176  }
177 
178  QMPackageFactory::RegisterAll();
179 }

209 ctp::XJob QMAPE::ProcessInputString(ctp::Job *job,ctp::Topology *top, ctp::QMThread *thread) {
210 
211  // Input string looks like this:
212  // <id1>:<name1>:<mpsfile1> <id2>:<name2>: ... ... ...
213 
214  string input = job->getInput().as<string>();
215  vector<ctp::Segment*> qmSegs;
216  vector<string> qmSegMps;
217  vector<string> split;
218  Tokenizer toker(input, " \t\n");
219  toker.ToVector(split);
220 
221  for (unsigned i = 0; i < split.size(); ++i) {
222 
223  string id_seg_mps = split[i];
224  vector<string> split_id_seg_mps;
225  Tokenizer toker(id_seg_mps, ":");
226  toker.ToVector(split_id_seg_mps);
227 
228  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
229  string segName = split_id_seg_mps[1];
230  string mpsFile = split_id_seg_mps[2];
231 
232  ctp::Segment *seg = top->getSegment(segId);
233  if (seg->getName() != segName) {
234  LOG(ctp::logERROR,*(thread->getLogger()))
235  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
236  << " maltagged as " << segName << ". Skip job ..." << flush;
237  throw std::runtime_error("Input does not match topology.");
238  }
239 
240  qmSegs.push_back(seg);
241  qmSegMps.push_back(mpsFile);
242  }
243 
244  return ctp::XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
245 }

248 ctp::Job::JobResult QMAPE::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *thread) {
249 
250  // SILENT LOGGER FOR QMPACKAGE
251  ctp::Logger* log = thread->getLogger();
252  ctp::Logger* qlog = new ctp::Logger();
253  qlog->setReportLevel(ctp::logDEBUG);
254  qlog->setMultithreading(_maverick);
255  qlog->setPreface(ctp::logINFO, (format("\nQ%1$02d ... ...") % thread->getId()).str());
256  qlog->setPreface(ctp::logERROR, (format("\nQ%1$02d ERR ...") % thread->getId()).str());
257  qlog->setPreface(ctp::logWARNING, (format("\nQ%1$02d WAR ...") % thread->getId()).str());
258  qlog->setPreface(ctp::logDEBUG, (format("\nQ%1$02d DBG ...") % thread->getId()).str());
259 
260  // CREATE XJOB FROM JOB INPUT STRING
261  LOG(ctp::logINFO,*log)
262  << "Job input = " << job->getInput().as<string>() << flush;
263  ctp::XJob xjob = this->ProcessInputString(job, top, thread);
264 
265  // SETUP POLAR TOPOLOGY (GENERATE VS LOAD IF PREPOLARIZED)
266  if (_polar_bg_arch == "") {
267  LOG(ctp::logINFO,*log) << "Mps-Mapper: Generate FGC FGN BGN" << flush;
268  _mps_mapper.Gen_FGC_FGN_BGN(top, &xjob, thread);
269  }
270  else {
271  LOG(ctp::logINFO,*log) << "Mps-Mapper: Generate FGC, load FGN BGN from '"
272  << _polar_bg_arch << "'" << flush;
273  _mps_mapper.Gen_FGC_Load_FGN_BGN(top, &xjob, _polar_bg_arch, thread);
274  }
275  LOG(ctp::logINFO,*log) << xjob.getPolarTop()->ShellInfoStr() << flush;
276 
277  // SETUP MM METHOD
278  ctp::PEwald3D3D cape = ctp::PEwald3D3D(top, xjob.getPolarTop(), _options,
279  thread->getLogger());
280  if (_pdb_check)
281  cape.WriteDensitiesPDB(xjob.getTag()+".densities.pdb");
282 
283  // SETUP QM HANDLERS
284  QMPackage *qmpack = QMPackages().Create(_package);
285  qmpack->Initialize(&_qmpack_opt);
286  qmpack->setLog(qlog);
287 
288  // SETUP QMAPE
289  QMAPEMachine<QMPackage> machine = QMAPEMachine<QMPackage>(&xjob, &cape, qmpack,
290  _options, "options.qmape", _subthreads);
291  machine.setLog(thread->getLogger());
292 
293  // EVALUATE: ITERATE UNTIL CONVERGED
294  machine.Evaluate(&xjob);
295 
296  // GENERATE OUTPUT AND FORWARD TO PROGRESS OBSERVER (RETURN)
297  ctp::Job::JobResult jres = ctp::Job::JobResult();
298  jres.setOutput(xjob.getInfoLine());
299  jres.setStatus(ctp::Job::COMPLETE);
300 
301  // CLEAN-UP
302  delete qmpack;
303  delete qlog;
304 
305  return jres;
306 }
