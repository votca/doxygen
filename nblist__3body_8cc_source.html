31 void NBList_3Body::Generate(BeadList &list1, BeadList &list2, BeadList &list3,
32  bool do_exclusions) {
33  BeadList::iterator iter1;
34  BeadList::iterator iter2;
35  BeadList::iterator iter3;
36  _do_exclusions = do_exclusions;
37 
38  if (list1.empty()) return;
39  if (list2.empty()) return;
40  if (list3.empty()) return;
41 
42  // check if all bead lists "have" the same topology
43  assert(list1.getTopology() == list2.getTopology());
44  assert(list1.getTopology() == list3.getTopology());
45  assert(list2.getTopology() == list3.getTopology());
46  Topology *top = list1.getTopology();
47 
48  // builds neighbor lists for all cases, where all list are of different bead
49  // typess (list1 neq list2 neq list3), list2 and list3 are of the same type
50  // (list1 neq (list2 = list3)) or all three lists are the same
51  // (list1=list2=list3)!
52  for (iter1 = list1.begin(); iter1 != list1.end(); ++iter1) {
53 
54  // in all cases iterate over the full bead lists list1 and list2
55  // if both lists are the same (the 3 bead types are the same),
56  // still necessary as both permutations (1,2,*) (2,1,*) included in neighbor
57  // list! (i,jneqi,k>j)
58  iter2 = list2.begin();
59 
60  for (; iter2 != list2.end(); ++iter2) {
61 
62  // do not include the same beads twice in one triple!
63  if (*iter1 == *iter2) {
64  continue;
65  }
66 
67  // if list2 and list3 are the same, set iter3 to "iter2+1" (i,jneqi,k>j)
68  if (&list2 == &list3) {
69  iter3 = iter2;
70  ++iter3;
71  } else {
72  iter3 = list3.begin();
73  }
74 
75  for (; iter3 != list3.end(); ++iter3) {
76 
77  // do not include the same beads twice in one triple!
78  if (*iter1 == *iter3) {
79  continue;
80  }
81  if (*iter2 == *iter3) {
82  continue;
83  }
84 
85  vec u = (*iter1)->getPos();
86  vec v = (*iter2)->getPos();
87  vec z = (*iter3)->getPos();
88 
89  vec r12 = top->BCShortestConnection(u, v);
90  vec r13 = top->BCShortestConnection(u, z);
91  vec r23 = top->BCShortestConnection(v, z);
92  double d12 = abs(r12);
93  double d13 = abs(r13);
94  double d23 = abs(r23);
95  // to do: at the moment use only one cutoff value
96  // to do: so far only check the distance between bead 1 (central bead)
97  // and bead2 and bead 3
98  if ((d12 < _cutoff) && (d13 < _cutoff)) {
101  if (_do_exclusions)
102  if ((top->getExclusions().IsExcluded(*iter1, *iter2)) ||
103  (top->getExclusions().IsExcluded(*iter1, *iter3)) ||
104  (top->getExclusions().IsExcluded(*iter2, *iter3))) {
105  continue;
106  }
107  if ((*_match_function)(*iter1, *iter2, *iter3, r12, r13, r23, d12,
108  d13, d23))
109  if (!FindTriple(*iter1, *iter2, *iter3))
110  AddTriple(_triple_creator(*iter1, *iter2, *iter3, r12, r13, r23));
111  }
112  }
113  }
114  }
115 }
