69BOOST_AUTO_TEST_CASE(cubicspline_fit_test) {
70
71 votca::Index size = 80;
72 Eigen::VectorXd x = Eigen::VectorXd::Zero(size);
73 Eigen::VectorXd y = Eigen::VectorXd::Zero(size);
74 for (votca::Index i = 0; i < size; ++i) {
75 x(i) = 0.25 * double(i);
76 y(i) = std::sin(x(i));
77 }
78 CubicSpline cspline;
79 cspline.setBCInt(0);
80 cspline.GenerateGrid(0.4, 0.6, 0.1);
81 Eigen::VectorXd gridpoints_ref = Eigen::VectorXd::Zero(3);
82 gridpoints_ref << 0.4, 0.5, 0.6;
83 bool grid_check = cspline.getX().isApprox(gridpoints_ref, 1e-5);
84 BOOST_CHECK_EQUAL(grid_check, true);
85 if (!grid_check) {
86 std::cout << "result value" << std::endl;
87 std::cout << cspline.getX().transpose() << std::endl;
88 std::cout << "ref value" << std::endl;
89 std::cout << gridpoints_ref.transpose() << std::endl;
90 }
91 cspline.Fit(x, y);
92
93 Eigen::VectorXd rs = Eigen::VectorXd::Zero(10);
94 rs << 0.45, 0.47, 0.8, 0.75, 0.6, 0.4, 0.9, 0.55, 0, 0;
95 Eigen::VectorXd values_ref = Eigen::VectorXd::Zero(10);
96 values_ref << 0.311213, 0.310352, 0.296153, 0.298304, 0.304759, 0.313364,
97 0.291851, 0.30691, 0.33058, 0.33058;
98 Eigen::VectorXd derivatives_ref = Eigen::VectorXd::Zero(10);
99 derivatives_ref << -0.0430277, -0.0430282, -0.0430231, -0.0430267, -0.0430313,
100 -0.0430272, -0.0430128, -0.0430308, -0.04306, -0.04306;
101 Eigen::VectorXd values = cspline.Calculate(rs);
102 Eigen::VectorXd derivatives = cspline.CalculateDerivative(rs);
103
104 bool equal_val = values_ref.isApprox(values, 1e-5);
105
106 if (!equal_val) {
107 std::cout << "result value" << std::endl;
108 std::cout << values.transpose() << std::endl;
109 std::cout << "ref value" << std::endl;
110 std::cout << values_ref.transpose() << std::endl;
111 }
112 BOOST_CHECK_EQUAL(equal_val, true);
113
114 bool equal_derivative = derivatives_ref.isApprox(derivatives, 1e-5);
115
116 if (!equal_derivative) {
117 std::cout << "result value" << std::endl;
118 std::cout << derivatives.transpose() << std::endl;
119 std::cout << "ref value" << std::endl;
120 std::cout << derivatives_ref.transpose() << std::endl;
121 }
122 BOOST_CHECK_EQUAL(equal_derivative, true);
123}
