135 ImagePotential::ImagePotential(long N) {
136  // Precalculate the image potential for all relevant configurations
137  for (int il=1; il<L; il++) {
138  for (int ix=-RC; ix<=RC; ix++) {
139  for (int iy=-RC; iy<=RC; iy++) {
140  for (int iz=-RC; iz<=RC; iz++) {
141  s_imagepot s(il,ix,iy,iz);
142  precalc[il-1][ix+RC][iy+RC][iz+RC] = image_potential(s,N);
143  }
144  }
145  }
146  }

149 double ImagePotential::image_potential(s_imagepot &s,long N) {
150  long distsyz = s.dy*s.dy + s.dz*s.dz;
151  long dists = s.dx*s.dx + distsyz;
152  double result = 0;
153  if (dists!=0) { // dists==0 is special case: charge interacts with its own image charges
154  result = 1.0/sqrt(double(dists));
155  }
156  for (int i=0; i<100; i++) {
157  long dist1;
158  long dist2;
159  long sign;
160  if (div(i,2).rem==0) { // even generation (x-position of image charges is -p.x + 2*j*L, j=...,-1,0,1,...)
161  sign = -1;
162  dist1 = i*L + 2*s.mx + s.dx;
163  dist2 = i*L + 2*L - 2*s.mx - s.dx;
164  }
165  else { // odd generation (x-position of image charges is -p.x + 2*j*L, j=...,-1,0,1,...)
166  sign = 1;
167  dist1 = (i+1)*L + s.dx;
168  dist2 = (i+1)*L - s.dx;
169  }
170  long dists1 = dist1*dist1 + distsyz;
171  long dists2 = dist2*dist2 + distsyz;
172  result += sign*(1.0/sqrt(double(dists1)) + 1.0/sqrt(double(dists2)));
173  }
174  return result;
175 }
