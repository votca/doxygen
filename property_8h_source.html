314 inline T Property::ifExistsAndinListReturnElseThrowRuntimeError(
315  const std::string &key, std::vector<T> possibleReturns) const {
316  T result;
317  result = ifExistsReturnElseThrowRuntimeError<T>(key);
318  if (std::find(possibleReturns.begin(), possibleReturns.end(), result) ==
319  possibleReturns.end()) {
320  std::stringstream s;
321  s << "Allowed options are: ";
322  for (unsigned i = 0; i < possibleReturns.size(); ++i) {
323  s << possibleReturns[i] << " ";
324  }
325  s << std::endl;
