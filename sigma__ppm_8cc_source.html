34 Eigen::VectorXd Sigma_PPM::CalcCorrelationDiag(
35  const Eigen::VectorXd& frequencies) const {
36 
37  Eigen::VectorXd RPAEnergies = _rpa.getRPAInputEnergies();
38  Eigen::VectorXd result = Eigen::VectorXd::Zero(_qptotal);
39  const int levelsum = _Mmn.nsize(); // total number of bands
40  const int auxsize = _Mmn.auxsize(); // size of the GW basis
41  const int lumo = _opt.homo + 1;
42  const int qpmin_offset = _opt.qpmin - _opt.rpamin;
43 #pragma omp parallel for
44  for (int gw_level = 0; gw_level < _qptotal; gw_level++) {
45  const double qpmin = frequencies(gw_level);
46  double sigma_c = 0.0;
47 
48  for (int i_aux = 0; i_aux < auxsize; i_aux++) {
49  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc
50  // PPM_construct_parameters
51  if (_ppm.getPpm_weight()(i_aux) < 1.e-9) {
52  continue;
53  }
54 
55  const Eigen::VectorXd Mmn2 =
56  _Mmn[gw_level + qpmin_offset].col(i_aux).cwiseAbs2();
57  const double ppm_freq = _ppm.getPpm_freq()(i_aux);
58  const double fac = 0.5 * _ppm.getPpm_weight()(i_aux) * ppm_freq;
59  Eigen::ArrayXd denom = qpmin - RPAEnergies.array();
60  denom.segment(0, lumo) += ppm_freq;
61  denom.segment(lumo, levelsum - lumo) -= ppm_freq;
62  Stabilize(denom);
63  sigma_c += fac * (denom.inverse() * Mmn2.array()).sum();
64  }
65  result(gw_level) = sigma_c;
66  }
67  return result;
68 }

70 void Sigma_PPM::Stabilize(Eigen::ArrayXd& denom) const {
71  const double fourpi = 4 * boost::math::constants::pi<double>();
72  for (int i = 0; i < denom.size(); ++i) {
73  if (std::abs(denom[i]) < 0.25) {
74  denom[i] = denom[i] / (0.5 * (1.0 - std::cos(fourpi * denom[i])));
75  }
76  }
77 }

79 Eigen::MatrixXd Sigma_PPM::CalcCorrelationOffDiag(
80  const Eigen::VectorXd& frequencies) const {
81 
82  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(_qptotal, _qptotal);
83 
84 #pragma omp parallel
85  {
86  const int lumo = _opt.homo + 1;
87  const int levelsum = _Mmn.nsize(); // total number of bands
88  const int auxsize = _Mmn.auxsize(); // size of the GW basis
89  const Eigen::VectorXd ppm_weight = _ppm.getPpm_weight();
90  const Eigen::VectorXd ppm_freqs = _ppm.getPpm_freq();
91  const int qpmin_offset = _opt.qpmin - _opt.rpamin;
92  const Eigen::VectorXd rpaenergies_thread = _rpa.getRPAInputEnergies();
93 #pragma omp for schedule(dynamic)
94  for (int gw_level1 = 0; gw_level1 < _qptotal; gw_level1++) {
95  const Eigen::MatrixXd& Mmn1 = _Mmn[gw_level1 + qpmin_offset];
96  const double qpmin1 = frequencies(gw_level1);
97  for (int gw_level2 = gw_level1 + 1; gw_level2 < _qptotal; gw_level2++) {
98  const Eigen::MatrixXd& Mmn2 = _Mmn[gw_level2 + qpmin_offset];
99  const double qpmin2 = frequencies(gw_level2);
100  double sigma_c = 0;
101  for (int i_aux = 0; i_aux < auxsize; i_aux++) {
102  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc
103  // PPM_construct_parameters
104  if (ppm_weight(i_aux) < 1.e-9) {
105  continue;
106  }
107  const double ppm_freq = ppm_freqs(i_aux);
108  const double fac = 0.25 * ppm_weight(i_aux) * ppm_freq;
109  const Eigen::VectorXd Mmn1xMmn2 =
110  Mmn1.col(i_aux).cwiseProduct(Mmn2.col(i_aux));
111  Eigen::ArrayXd denom1 = rpaenergies_thread;
112  denom1.segment(0, lumo) -= ppm_freq;
113  denom1.segment(lumo, levelsum - lumo) += ppm_freq;
114  Eigen::ArrayXd denom2 = (qpmin2 - denom1);
115  Stabilize(denom2);
116  denom1 = (qpmin1 - denom1);
117  Stabilize(denom1);
118  sigma_c +=
119  fac *
120  ((denom1.inverse() + denom2.inverse()) * Mmn1xMmn2.array()).sum();
121  }
122  result(gw_level1, gw_level2) = sigma_c;
123  result(gw_level2, gw_level1) = sigma_c;
124  } // GW row
125  } // GW col
126  }
127  return result;
128 }
