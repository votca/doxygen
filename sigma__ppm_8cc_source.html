34 double Sigma_PPM::CalcCorrelationDiagElement(Index gw_level,
35  double frequency) const {
36  const Index lumo = _opt.homo + 1;
37  const Index levelsum = _Mmn.nsize(); // total number of bands
38  const Index auxsize = _Mmn.auxsize(); // size of the GW basis
39  const Eigen::VectorXd ppm_weight = _ppm.getPpm_weight();
40  const Eigen::VectorXd ppm_freqs = _ppm.getPpm_freq();
41  const Index qpmin_offset = _opt.qpmin - _opt.rpamin;
42  const Eigen::VectorXd RPAEnergies = _rpa.getRPAInputEnergies();
43  double sigma_c = 0;
44  for (Index i_aux = 0; i_aux < auxsize; i_aux++) {
45  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc
46  // PPM_construct_parameters
47  if (ppm_weight(i_aux) < 1.e-9) {
48  continue;
49  }
50  const double ppm_freq = ppm_freqs(i_aux);
51  const double fac = 0.5 * ppm_weight(i_aux) * ppm_freq;
52  const Eigen::VectorXd Mmn2 =
53  _Mmn[gw_level + qpmin_offset].col(i_aux).cwiseAbs2();
54  Eigen::ArrayXd denom = frequency - RPAEnergies.array();
55  denom.segment(0, lumo) += ppm_freq;
56  denom.segment(lumo, levelsum - lumo) -= ppm_freq;
57  Stabilize(denom);
58  sigma_c += fac * (denom.inverse() * Mmn2.array()).sum();
59  }
60  return sigma_c;
61 }

63 double Sigma_PPM::CalcCorrelationOffDiagElement(Index gw_level1,
64  Index gw_level2,
65  double frequency1,
66  double frequency2) const {
67  const Index lumo = _opt.homo + 1;
68  const Index levelsum = _Mmn.nsize(); // total number of bands
69  const Index auxsize = _Mmn.auxsize(); // size of the GW basis
70  const Eigen::VectorXd ppm_weight = _ppm.getPpm_weight();
71  const Eigen::VectorXd ppm_freqs = _ppm.getPpm_freq();
72  const Index qpmin_offset = _opt.qpmin - _opt.rpamin;
73  const Eigen::VectorXd RPAEnergies = _rpa.getRPAInputEnergies();
74  double sigma_c = 0;
75  for (Index i_aux = 0; i_aux < auxsize; i_aux++) {
76  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc
77  // PPM_construct_parameters
78  if (ppm_weight(i_aux) < 1.e-9) {
79  continue;
80  }
81  const double ppm_freq = ppm_freqs(i_aux);
82  const double fac = 0.25 * ppm_weight(i_aux) * ppm_freq;
83  const Eigen::MatrixXd& Mmn1 = _Mmn[gw_level1 + qpmin_offset];
84  const Eigen::MatrixXd& Mmn2 = _Mmn[gw_level2 + qpmin_offset];
85  const Eigen::VectorXd Mmn1xMmn2 =
86  Mmn1.col(i_aux).cwiseProduct(Mmn2.col(i_aux));
87  Eigen::ArrayXd denom1 = RPAEnergies;
88  denom1.segment(0, lumo) -= ppm_freq;
89  denom1.segment(lumo, levelsum - lumo) += ppm_freq;
90  Eigen::ArrayXd denom2 = (frequency2 - denom1);
91  Stabilize(denom2);
92  denom1 = (frequency1 - denom1);
93  Stabilize(denom1);
94  sigma_c +=
95  fac * ((denom1.inverse() + denom2.inverse()) * Mmn1xMmn2.array()).sum();
96  }
97  return sigma_c;
98 }

100 void Sigma_PPM::Stabilize(Eigen::ArrayXd& denom) const {
101  const double fourpi = 4 * boost::math::constants::pi<double>();
102  for (Index i = 0; i < denom.size(); ++i) {
103  if (std::abs(denom[i]) < 0.25) {
104  denom[i] = denom[i] / (0.5 * (1.0 - std::cos(fourpi * denom[i])));
105  }
106  }
107 }
