25 void DIIS::Update(Index maxerrorindex, const Eigen::MatrixXd& errormatrix) {
26 
27  if (int(_errormatrixhist.size()) == _histlength) {
28  _errormatrixhist.erase(_errormatrixhist.begin() + maxerrorindex);
29  _Diis_Bs.erase(_Diis_Bs.begin() + maxerrorindex);
30  for (std::vector<double>& subvec : _Diis_Bs) {
31  subvec.erase(subvec.begin() + maxerrorindex);
32  }
33  }
34 
35  _errormatrixhist.push_back(errormatrix);
36 
37  std::vector<double> Bijs;
38  for (Index i = 0; i < Index(_errormatrixhist.size()) - 1; i++) {
39  double value =
40  errormatrix.cwiseProduct((_errormatrixhist[i]).transpose()).sum();
41  Bijs.push_back(value);
42  _Diis_Bs[i].push_back(value);
43  }
44  Bijs.push_back(errormatrix.cwiseProduct(errormatrix.transpose()).sum());
45  _Diis_Bs.push_back(Bijs);
46  return;
47 }

49 Eigen::VectorXd DIIS::CalcCoeff() {
50  success = true;
51  const Index size = Index(_errormatrixhist.size());
52 
53  // C2-DIIS
54  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(size, size);
55 
56  for (Index i = 0; i < B.rows(); i++) {
57  for (Index j = 0; j <= i; j++) {
58  B(i, j) = _Diis_Bs[i][j];
59  if (i != j) {
60  B(j, i) = B(i, j);
61  }
62  }
63  }
64  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(B);
65  Eigen::MatrixXd eigenvectors = Eigen::MatrixXd::Zero(size, size);
66 
67  for (Index i = 0; i < es.eigenvectors().cols(); i++) {
68  double norm = es.eigenvectors().col(i).sum();
69  eigenvectors.col(i) = es.eigenvectors().col(i) / norm;
70  }
71 
72  // Choose solution by picking out solution with smallest absolute error
73  Eigen::VectorXd errors =
74  (eigenvectors.transpose() * B * eigenvectors).diagonal().cwiseAbs();
75 
76  double MaxWeight = 10.0;
77  Index mincoeff = 0;
78  success = false;
79  for (Index i = 0; i < errors.size(); i++) {
80  errors.minCoeff(&mincoeff);
81  if (std::abs(eigenvectors.col(mincoeff).maxCoeff()) > MaxWeight) {
82  errors[mincoeff] = std::numeric_limits<double>::max();
83  } else {
84  success = true;
85  break;
86  }
87  }
88 
89  Eigen::VectorXd coeffs = eigenvectors.col(mincoeff);
90 
91  if (std::abs(coeffs[coeffs.size() - 1]) < 0.001) {
92  success = false;
93  }
94 
95  return coeffs;
96 }
