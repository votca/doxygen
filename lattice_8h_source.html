346 void Lattice:: Add_Remove_Carrier(Node* carnode, CarrierType cartype, enum action) {
347 
348  double RC = _coulomb_cutoff;
349  double RCSQR = RC*RC;
350  double hopdist = _hopping_distance;
351  int arsign; // arsign = 1 for add, arsign = -1 for remove
352 
353  int carrier1;
354  Carrier* pcarrier;
355  int charge1;
356  Myvec carpos = carnode->_position;
357  int carnumberofpairs;
358 
359  if(action == Add) {
360  // Add new carrier to the lattice
361  arsign = 1;
362  if(cartype==Hole) {
363  carrier1 = _hole_state.Buy();
364  pcarrier = _hole_state.Get_item(carrier1);
365  carnode->_hole_number = carrier1;
366  }
367  else if(cartype==Electron) {
368  carrier1 = _electron_state.Buy();
369  pcarrier = _electron_state.Get_item(carrier1);
370  carnode->_electron_number = carrier1;
371  }
372  pcarrier->_carriertype = cartype;
373  pcarrier->_node_index = carnode->_node_id;
374  carnode->_carrier = pcarrier1;
375 
376  if (cartype == Hole) {
377  if(_nholes == _maxholes){
378  int growsize = 10;
379  lattice_grow(growsize);
380  }
381  _nholes++;
382  charge1 = 1;
383  pcarrier->_shortrange_coulomb.resize(1+carnode->_number_of_ho_pairs);
384  carnumberofpairs = _number_of_ho_pairs;
385  }
386  else if (cartype == Electron) {
387  if(_nelectrons == _maxelectrons){
388  int growsize = 10;
389  lattice_grow(growsize);
390  }
391  _nelectrons++;
392  charge1 = -1;
393  pcarrier->_shortrange_coulomb.resize(1+carnode->_number_of_el_pairs);
394  carnumberofpairs = _number_of_el_pairs;
395  }
396  _ncarriers++;
397 
398  // add charge to charge list
399  int iposx = floor(carpos.x()/_coul_box_dimX);
400  int iposy = floor(carpos.y()/_coul_box_dimY);
401  int iposz = floor(carpos.z()/_coul_box_dimZ);
402  chargelist[iposx][iposy][iposz].push_back(carrier1);
403  }
404  else if(action == Remove) {
405  // Remove existing carrier from lattice
406  arsign = -1;
407  Carriertype rem_cartype = carnode->_carrier->_carriertype;
408  if (rem_cartype == Hole) {
409  _nholes--;
410  charge1 = 1;
411  carnode->_hole_number = -1; //empty (necessary to set to -1)
412  }
413  else if (rem_cartype == Electron) {
414  _nelectrons--;
415  charge1 = -1;
416  carnode->_electron_number = -1; //empty (necessary to set to -1)
417  }
418  carnode->_carrier = NULL;
419  _ncarriers--;
420  }
421 
422  int layerindex = carnode->_layer_index;
423  _longrange->_layercharge[layerindex] += arsign*charge1;
424 
425 
426  // Define cubic boundaries in non-periodic coordinates
427  double ix1 = carpos.x-RC-hopdist; double ix2 = carpos.x+RC+hopdist;
428  double iy1 = carpos.y-RC-hopdist; double iy2 = carpos.y+RC+hopdist;
429  double iz1 = carpos.z-RC-hopdist; double iz2 = carpos.z+RC+hopdist;
430 
431  // Break periodicity in x-direction
432  if(_device) {
433  if (ix1<0.0) ix1 = 0.0;
434  if (ix2>=_NX) ix2 = _NX;
435  }
436 
437  // Translate cubic boundaries to sublattice boundaries in non-periodic coordinates
438  int sx1 = floor(ix1,_coul_box_dimX);
439  int sx2 = floor(ix2,_coul_box_dimX);
440  int sy1 = floor(iy1,_coul_box_dimY);
441  int sy2 = floor(iy2,_coul_box_dimY);
442  int sz1 = floor(iz1,_coul_box_dimZ);
443  int sz2 = floor(iz2,_coul_box_dimZ);
444 
445  // Now visit all relevant sublattices
446  for (int isz=sz1; isz<=sz2; isz++) {
447  int r_isz = isz;
448  while (r_isz < 0) r_isz += _NCHARGE_Z;
449  while (r_isz >= _NCHARGE_Z) r_isz -= _NCHARGE_Z;
450  for (int isy=sy1; isy<=sy2; isy++) {
451  int r_isy = isy;
452  while (r_isy < 0) r_isy += _NCHARGE_Y;
453  while (r_isy >= _NCHARGE_Y) r_isy -= _NCHARGE_Y;
454  for (int isx=sx1; isx<=sx2; isx++) {
455  int r_isx = isx;
456  while (r_isx < 0) r_isx += _NCHARGE_X;
457  while (r_isx >= _NCHARGE_X) r_isx -= _NCHARGE_X;
458 
459  // Ask a list of all charges in this sublattice
460  std::list<int>::iterator li1,li2,li3;
461  std::list<int> *pList = &chargelist[r_isx][r_isy][r_isz];
462  li1 = pList->begin();
463  li2 = pList->end();
464  for (li3=li1; li3!=li2; li3++) {
465  int carrier2 = *li3;
466  Carrier* probecarrier = carriers.get_item(carrier2);
467  Node* probenode = _graph->_nodes[probercarrier->_node_index];
468  myvec probepos = probenode->_position;
469  Carriertype probecartype = probecarrier->_carriertype;
470  int probecharge;
471  if(probecartype==Electron) {
472  probecharge = -1;
473  }
474  else {
475  probecharge = 1;
476  }
477 
478  int interaction_sign = arsign*charge1*probecharge;
479 
480 
481  // Compute coordinates in non-periodic lattice
482  myvec periodic_convert = myvec((isx-r_isx)*_coul_box_dimX,(isy-r_isy)*_coul_box_dimY,(isz-r_isz)*_coul_box_dimZ);
483  myvec np_probepos = probepos + periodic_convert;
484  myvec difference = np_probepos-carpos;
485 
486  double distancesqr = difference.x()*difference.x() + difference.y()*difference.y()+difference.z()*difference.z();
487 
488  if (distancesqr < pow(0.5*_graph->_minimum_distance,2.0)) { // Charge interacting with its own images, special case
489  if (action==ADD) {
490  pcarrier ->_shortrange_coulomb[0] += _self_image_prefactor*compute_Coulomb_potential(carpos.x(),myvec(0.0,0.0,0.0));
491 
492  // Adjust Coulomb potential for neighbours of carrier1
493  for (int jump=0; jump < carnumberofpairs; jump++) {
494  myvec jumpdistance;
495  if(cartype==Hole) {
496  jumpdistance = carnode->_ho_pair_distances[jump];
497  }
498  else {
499  jumpdistance = carnode->_el_pair_distances[jump];
500  }
501  pcarrier->_shortrange_coulomb[jump+1] += self_image_prefactor*compute_Coulomb_potential(carpos.x(),jumpdistance);
502  }
503  }
504  }
505  else { // dists>0, Charge interacting with another charge and its images
506  if (distancesqr <= RCSQR) {
507  if (action==ADD) pcarrier->_shortrange_coulomb[0] +=interaction_sign*compute_Coulomb_potential(np_probepos.x(),-1.0*difference);
508  probecarrier->_shortrange_coulomb[0] += interaction_sign*compute_Coulomb_potential(carpos.x(),difference);
509  }
510 
511  if (action==ADD) {
512  // Adjust Coulomb potential for neighbours of carrier1
513  for (int jump=0; jump < carnumberofpairs; jump++) {
514  myvec jumpdistance;
515  if(cartype == Hole){
516  jumpdistance = carnode->_ho_pair_distances[jump];
517  }
518  else {
519  jumpdistance = carnode->_el_pair_distances[jump];
520  }
521  myvec jumppos = carnode->_position + jumpdistance;
522  myvec jumpdifference = np_probepos - jumppos;
523  double distsqr = jumpdifference.x()*jumpdifference.x() + jumpdifference.y()*jumpdifference.y()+jumpdifference.z()*jumpdifference.z*();
524 
525  if(distsqr <= RCSQR) {
526  pcarrier->_shortrange_coulomb[jump+1] += interaction_sign*compute_Coulomb_potential(np_probepos.x(),jumpdifference);
527  }
528  }
529  }
530 
531  // Adjust Coulomb potential for neighbours of carrier2
532  if (probecartype == Hole) {
533  probenumberofpairs = probenode->_number_of_ho_pairs;
534  }
535  else if(probecartype == Electron) {
536  probenumberofpairs = probenode->_number_of_el_pairs;
537  }
538  for (int jump=0; jump < probenumberofpairs; jump++) {
539  myvec jumpdistance;
540  Node* jumpnode;
541  if(probecartype == Hole) {
542  jumpdistance = probenode->_ho_pair_distances[jump];
543  jumpnode = probenode->_ho_pair_nodes[jump];
544  }
545  else {
546  jumpdistance = probenode->_el_pair_distances[jump];
547  jumpnode = probenode->_ho_pair_nodes[jump];
548  }
549  myvec jumppos = np_probepos + jumpdistance;
550  myvec jumpdifference = carpos - jumppos;
551  double distsqr = jumpdifference.x()*jumpdifference.x() + jumpdifference.y()*jumpdifference.y()+jumpdifference.z()*jumpdifference.z*();
552  if(distsqr <= RCSQR) {
553  probecarrier->_shortrange_coulomb[jump+1] += interaction_sign*compute_Coulomb_potential(carpos.x(),jumpdifference);
554  }
555 
556  if(probecartype == Electron) {
557 
558  double rate = compute_jump_rate(jumpnode,probenode,Electron);
559  _electron_rates.set_jump_rate(carrier2,jump,rate);
560  }
561  else {
562  double rate = compute_jump_rate(jumpnode,probenode,Hole);
563  _hole_rates.set_jump_rate(carrier2,jump,rate);
564  }
565  }
566  }
567  if (action==REMOVE) {
568  // Reset Coulomb potential for carrier1 and its neighbours
569  for (int jump=0; jump < carnumberofpairs+1; jump++) {
570  pcarrier->_shortrange_coulomb[jump] = 0.0;
571  }
572  }
573  }
574  }
575  }
576  }
577 
578 
579  // update the injection rates
580 
581  if (carpos.x() <= RC) { // Distance to left electrode
582 
583  //maximum entry inside device is hopdist (make this variable!!)
584 
585  double bound = sqrt(double(RCSQR - carpos.x()*carpos.x()));
586 
587  // Define cubic boundaries in non-periodic coordinates
588  double iy1 = carpos.y-bound-hopdist; double iy2 = carpos.y+bound+hopdist;

942 double Lattice::Compute_Coulomb_potential(double startx, myvec dif) {
943 
944  double coulpot;
945  double RC = _coulomb_cutoff
946  double RCSQR = RC*RC;
947  int nimages = _number_of_shortrange_images;
948 
949  if(!_device) {
950  coulpot = 1.0/abs(dif)-1.0/RC;
951  }
952  else {
953  coulpot = 0.0;
954  double distsqr_planar = dif.y()*dif.y() + dif.z()*dif.z();
955  double distsqr = dif.x()*dif.x() + distsqr_planar;
956  if (distsqr >= 0.5*_graph->_minimum_distance) {
957  coulpot += 1.0/abs(dif)-1.0/RC;
958  }
959 
960  int sign;
961  double distx_1;
962  double distx_2;
963  double distancesqr_1;
964  double distancesqr_2;
965  bool outside_cut_off1 = false;
966  bool outside_cut_off2 = false;
967 
968  while(!(outside_cut_off1&&outside_cut_off2)) {
969  for (int i=0;i<nimages; i++) {
970  if (div(i,2).rem==0) { // even generation
971  sign = -1;
972  distx_1 = i*_NX + 2*startx + dif.x();
973  distx_2 = (i+2)*_NX - 2*startx - dif.x();
974  }
975  else {
976  sign = 1;
977  distx_1 = (i+1)*_NX + dif.x();
978  distx_2 = (i+1)*_NX - dif.x();
979  }
980  distancesqr_1 = distx_1*distx_1 + distsqr_planar;
981  if (distancesqr_1<=RCSQR) {
982  coulpot += sign*1.0/sqrt(distancesqr_1)-1.0/(RC);
983  }
984  else {
985  outside_cut_off1 = true;
986  }
987  distancesqr_2 = distx_2*distx_2 + distsqr_planar;
988  if (distancesqr_2<=RCSQR) {
989  coulpot += sign*1.0/sqrt(distancesqr_2)-1.0/(RC);
990  }
991  else {
992  outside_cut_off2 = true;
993  }
994  }
995  }
996  }
