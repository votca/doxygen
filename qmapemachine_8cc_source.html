35 QMAPEMachine::QMAPEMachine(ctp::XJob *job, ctp::Ewald3DnD *cape,
36  Property *opt, string sfx, int nst)
37  : _job(job), _cape(cape),_isConverged(false) {
38 
39  // CONVERGENCE THRESHOLDS
40  string key = sfx + ".qmmmconvg";
41  _crit_dR =opt->ifExistsReturnElseReturnDefault<double>(key + ".dR",0.01);//nm
42  _crit_dQ =opt->ifExistsReturnElseReturnDefault<double>(key + ".dQ",0.01);//e
43  _crit_dE_QM =opt->ifExistsReturnElseReturnDefault<double>(key + ".dQdE_QM",0.001);//eV
44  _crit_dE_MM =opt->ifExistsReturnElseReturnDefault<double>(key + ".dE_MM",_crit_dE_QM);//eV
45  _maxIter=opt->ifExistsReturnElseReturnDefault<int>(key + ".max_iter",32);
46 
47  // TASKS
48  key = sfx + ".tasks";
49  _run_ape = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_ape",true);
50  _run_dft = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_dft",true);
51  _run_gwbse = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_gwbse",true);
52 
53 
54  if(_run_dft){
55  key=sfx+".dft";
56  string dft_xml = opt->ifExistsReturnElseThrowRuntimeError<string>(key + ".dftengine");
57  load_property_from_xml(_dft_options, dft_xml.c_str());
58  _externalgridaccuracy=opt->ifExistsReturnElseReturnDefault<string>(key+".externalgrid","medium");
59  }
60  if(_run_gwbse){
61  // GWBSE CONFIG
62  key = sfx + ".gwbse";
63  string gwbse_xml = opt->ifExistsReturnElseThrowRuntimeError<string>(key + ".gwbse_options");
64  load_property_from_xml(_gwbse_options, gwbse_xml.c_str());
65 
66  _state = opt->ifExistsReturnElseReturnDefault<int>(key+".state",1);
67  std::vector<string> choices={ "singlet","triplet" };
68  _type = opt->ifExistsAndinListReturnElseThrowRuntimeError<string>(key+".type",choices);
69 
70 
71  key = sfx + ".gwbse.filter";
72  if (opt->exists(key + ".oscillator_strength") && _type != "triplet") {
73  _has_osc_filter = true;
74  _osc_threshold = opt->get(key + ".oscillator_strength").as<double> ();
75  }
76  if (opt->exists(key + ".localisation")) {
77  _has_loc_filter = true;
78 
79  string temp = opt->get(key + ".localisation").as<string> ();
80  Tokenizer tok_cleanup(temp, ", \n\t");
81  std::vector <std::string> strings_vec;
82  tok_cleanup.ToVector(strings_vec);
83  if (strings_vec.size()!=2){
84  throw runtime_error("qmmmachine: Fragment and localisation threshold are not separated");
85  }
86  if(strings_vec[0]=="a" || strings_vec[0]=="A"){
87  _localiseonA=true;
88  }else if(strings_vec[0]=="b" || strings_vec[0]=="B"){
89  _localiseonA=false;
90  }else{
91  throw runtime_error("qmmmachine: Fragment label not known, either A or B");
92  }

94  }
95 
96  if (opt->exists(key + ".charge_transfer")) {
97  _has_dQ_filter = true;
98  _dQ_threshold = opt->get(key + ".charge_transfer").as<double> ();
99  }
100  if(_has_dQ_filter && _has_loc_filter){
101  throw runtime_error("Cannot use localisation and charge_transfer filter at the same time.");
102  }
103  }
104  return;
105  }

130 void QMAPEMachine::Evaluate(ctp::XJob *job) {
131 
132  // PREPARE JOB DIRECTORY
133  string jobFolder = "qmapejob_" + boost::lexical_cast<string>(_job->getId())
134  + "_" + _job->getTag();
135  bool created = boost::filesystem::create_directory(jobFolder);
136 
137  CTP_LOG(ctp::logINFO,*_log) << flush;
138  if (created) {
139  CTP_LOG(ctp::logINFO,*_log) << "Created directory " << jobFolder << flush;
140  }
141 
142  CTP_LOG(ctp::logINFO,*_log)
143  << format("... dR %1$1.4f dQ %2$1.4f QM %3$1.4f MM %4$1.4f IT %5$d")
144  % _crit_dR % _crit_dQ % _crit_dE_QM % _crit_dE_MM % _maxIter << flush;
145 
146  // FIGURE OUT CHARGE + MULTIPLICITY
147  double dQ = 0.0;
148  for (unsigned i = 0; i < _job->getPolarTop()->QM0().size(); ++i) {
149  dQ += _job->getPolarTop()->QM0()[i]->CalcTotQ();
150  }
151  int chrg = round(dQ);
152  int spin = ( (chrg < 0) ? -chrg:chrg ) % 2 + 1;
153  CTP_LOG(ctp::logINFO,*_log) << "... Q = " << chrg << ", 2S+1 = " << spin << flush;
154 
155  if(chrg!=0){
156  throw runtime_error("Charged DFT calculations are not possible at the moment");
157  }
158  qminterface.setMultipoleSplitting(false,1);
159 
160 
161  int iterCnt = 0;
162  int iterMax = _maxIter;
163  for ( ; iterCnt < iterMax; ++iterCnt) {
164  CTP_LOG(ctp::logINFO,*_log) << "QMMM ITERATION:" << iterCnt+1<<" of "<<iterMax << flush;
165  //bool code = Iterate(jobFolder, iterCnt);
166  Iterate(jobFolder, iterCnt);
167  if (hasConverged()) {
168  CTP_LOG(ctp::logINFO,*_log) << "QMMM CONVERGED after:" << iterCnt+1<<" iterations."<< flush;
169 
170  break; }
171  }
172 
173  if (iterCnt == iterMax-1 && !_isConverged) {
174  CTP_LOG(ctp::logWARNING,*_log)
175  << format("Not converged within %1$d iterations.") % iterMax;
176  }
177 
178  return;
179 }

182 bool QMAPEMachine::Iterate(string jobFolder, int iterCnt) {
183 
184  // CREATE ITERATION OBJECT & SETUP RUN DIRECTORY
185  QMMIter *thisIter = this->CreateNewIter();
186  int iter = iterCnt;
187  string runFolder = jobFolder + "/iter_" + boost::lexical_cast<string>(iter);
188 
189  CTP_LOG(ctp::logINFO,*_log) << flush;
190  bool created = boost::filesystem::create_directory(runFolder);
191  if (created)
192  CTP_LOG(ctp::logDEBUG,*_log) << "Created directory " << runFolder << flush;
193  else
194  CTP_LOG(ctp::logWARNING,*_log) << "Could not create directory " << runFolder << flush;
195 
196  Orbitals orb_iter_input;
197 
198  qminterface.GenerateQMAtomsFromPolarSegs(_job->getPolarTop(), orb_iter_input);
199 
200 
201 
202 
203  if (iterCnt == 0) {
204  dftengine.Initialize(&_dft_options);
205  dftengine.setLogger(_log);
206  dftengine.ConfigureExternalGrid(_externalgridaccuracy);
207  dftengine.Prepare(&orb_iter_input);
208  SetupPolarSiteGrids(dftengine.getExternalGridpoints(),orb_iter_input.QMAtoms());
209  }
210 
211  // COMPUTE POLARIZATION STATE WITH QM0(0)
212  if (_run_ape) {
213  if (iterCnt == 0) {
214  _cape->ShowAgenda(_log);
215  // Reset FGC, start from BGP state, apply FP fields (BG & FG)
216  _cape->EvaluateInductionQMMM(true, true, true, true, true);
217  // Add BG, do not add MM1 & QM0
218  _cape->EvaluatePotential(target_bg, true, false, false);
219  }
220  // Do not add BG & QM0, add MM1
221  _cape->EvaluatePotential(target_fg, false, true, false);
222  }
223 
224  dftengine.setExternalGrid(ExtractElGrid_fromPolarsites(),ExtractNucGrid_fromPolarsites());
225 
226  if (_run_dft) {
227  dftengine.Evaluate(&orb_iter_input);
228  }
229  FILE *out;
230  out = fopen((runFolder + "/system.pdb").c_str(),"w");
231  orb_iter_input.WritePDB(out,"Full structure");
232  fclose(out);
233 
234  // Run GWBSE
235  if (_run_gwbse){
236  EvaluateGWBSE(orb_iter_input, runFolder);
237  }
238 
239 
240 
241  // COMPUTE POLARIZATION STATE WITH QM0(n+1)
242  if (_run_ape) {
243  // Update QM0 density: QM0(n) => QM0(n+1)
244  // ...
245  std::vector< ctp::QMAtom* > Atomlist;
246  for(const auto& atom:orb_iter_input.QMAtoms()){
247  if(!atom->from_environment){
248  Atomlist.push_back(atom);
249  }
250  }
251  thisIter->UpdatePosChrgFromQMAtoms(Atomlist,_job->getPolarTop()->QM0());
252  // Do not reset FGC (= only reset FU), do not use BGP state, nor apply FP fields (BG & FG)
253  _cape->EvaluateInductionQMMM(false, false, false, false, false);
254  // COMPUTE MM ENERGY
255  _cape->EvaluateEnergyQMMM();
256  _cape->ShowEnergySplitting(_log);
257  }
258 
259  // COMPILE HAMILTONIAN & CHECK FOR CONVERGENCE
260  // ...
261 
262 
263  // THIS IS A SHORT VERSION OF PEWALD3D WITH QM/MM ENERGY SPLITTING
264  _cape->ShowAgenda(_log);
265  _cape->EvaluateInductionQMMM(true, true, true, true, true);
266  _cape->EvaluateEnergyQMMM();
267  _cape->ShowEnergySplitting(_log);
268 
269  return true;
270 }

280 bool QMAPEMachine::EvaluateGWBSE(Orbitals &orb, string runFolder) {
281 
282  std::vector<int> _state_index;
283  GWBSE _gwbse(&orb);
284  _gwbse.setLogger(_log);
285 
286  if (_state > 0) {
287  CTP_LOG(ctp::logDEBUG, *_log) << "Excited state via GWBSE: " << flush;
288  CTP_LOG(ctp::logDEBUG, *_log) << " --- type: " << _type << flush;
289  CTP_LOG(ctp::logDEBUG, *_log) << " --- state: " << _state << flush;
290  if (_has_osc_filter) {
291  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: osc.str. > " << _osc_threshold << flush;
292  }
293  if (_has_dQ_filter) {
294  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: crg.trs. > " << _dQ_threshold << flush;
295  }
296  if (_has_loc_filter){
297  if (_loc_threshold>0.5){
298  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: localisation > " << _loc_threshold << flush;
299  }else{
300  CTP_LOG(ctp::logDEBUG, *_log) << " --- filter: localisation < " << _loc_threshold << flush;
301  }
302  }
303 
304  if (_has_osc_filter && _has_dQ_filter) {
305  CTP_LOG(ctp::logDEBUG, *_log) << " --- WARNING: filtering for optically active CT transition - might not make sense... " << flush;
306  }
307 
308  // define own logger for GW-BSE that is written into a runFolder logfile
309  ctp::Logger gwbse_logger(ctp::logDEBUG);
310  gwbse_logger.setMultithreading(false);
311  //gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
312  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
313  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
314  gwbse_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
315  _gwbse.Initialize(&_gwbse_options);
316  // actual GW-BSE run
317 
318  _gwbse.Evaluate();
319 
320  // write logger to log file
321  ofstream ofs;
322  string gwbse_logfile = runFolder + "/gwbse.log";
323  ofs.open(gwbse_logfile.c_str(), ofstream::out);
324  if (!ofs.is_open()) {
325  throw runtime_error("Bad file handle: " + gwbse_logfile);
326  }
327  ofs << gwbse_logger << endl;
328  ofs.close();
329 
330  // PROCESSING the GW-BSE result
331  // - find the excited state of interest
332  // oscillator strength filter
333 
334 
335  if (_has_osc_filter) {
336 
337  // go through list of singlets
338  const std::vector<double> oscs = orb.Oscillatorstrengths();
339  for (unsigned _i = 0; _i < oscs.size(); _i++) {
340 
341  double osc =oscs[_i];
342  if (osc > _osc_threshold) _state_index.push_back(_i);
343  }
344  } else {
345  const ub::vector<real_gwbse>& energies = (_type=="singlet")
346  ? orb.BSESingletEnergies() : orb.BSETripletEnergies();
347 
348  for (unsigned _i = 0; _i < energies.size(); _i++) {
349  _state_index.push_back(_i);
350  }
351  }
352 
353  // filter according to charge transfer, go through list of excitations in _state_index
354  if (_has_dQ_filter) {
355  std::vector<int> _state_index_copy;
356  const std::vector< ub::vector<double> >& dQ_frag= (_type=="singlet")
357  ? orb.getFragmentChargesSingEXC():orb.getFragmentChargesTripEXC();
358  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
359  if (std::abs(dQ_frag[_state_index[_i]](0)) > _dQ_threshold) {
360  _state_index_copy.push_back(_state_index[_i]);
361  }
362  }
363  _state_index = _state_index_copy;
364  }
365  else if (_has_loc_filter) {
366  std::vector<int> _state_index_copy;
367  const std::vector< ub::vector<double> >& popE= (_type=="singlet")
368  ? orb.getFragment_E_localisation_singlet():orb.getFragment_E_localisation_triplet();
369  const std::vector< ub::vector<double> >& popH= (_type=="singlet")
370  ? orb.getFragment_H_localisation_singlet():orb.getFragment_H_localisation_triplet();
371  if(_localiseonA){
372  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
373  if (popE[_state_index[_i]](0) > _loc_threshold && popH[_state_index[_i]](0) > _loc_threshold ) {
374  _state_index_copy.push_back(_state_index[_i]);
375  }
376  }
377  }else{
378  for (unsigned _i = 0; _i < _state_index.size(); _i++) {
379  if (popE[_state_index[_i]](1) > _loc_threshold && popH[_state_index[_i]](1) > _loc_threshold ) {
380  _state_index_copy.push_back(_state_index[_i]);
381  }
382  }
383  }
384  _state_index = _state_index_copy;
385  }
386  if (_state_index.size() < 1) {
387  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " WARNING: FILTER yielded no state. Taking lowest excitation"<< flush;
388  _state_index.push_back(0);
389  }else{
390  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Filter yielded state"<<_type<<":"<<_state_index[_state - 1]+1<< flush;
391  }
392  } // only if state >0
393 
394  std::vector< ctp::QMAtom* > Atomlist;
395  for(const auto& atom:orb.QMAtoms()){
396  if(!atom->from_environment){
397  Atomlist.push_back(atom);
398  }
399  }
400 
401 
402  // load DFT basis set (element-wise information) from xml file
403  BasisSet dftbs;
404 
405  dftbs.LoadBasisSet(orb.getDFTbasis());
406 
407  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Loaded DFT Basis Set " << orb.getDFTbasis() << flush;
408 
409  // fill DFT AO basis by going through all atoms
410  AOBasis dftbasis;
411  dftbasis.AOBasisFill(&dftbs, Atomlist );
412  // TBD: Need to switch between singlets and triplets depending on _type
413  ub::matrix<double> DMATGS = orb.DensityMatrixGroundState();
414  ub::matrix<double> DMAT_tot = DMATGS; // Ground state + hole_contribution + electron contribution
415 
416  if (_state > 0) {
417 
418  std::vector<ub::matrix<double> > DMAT = orb.DensityMatrixExcitedState(_type, _state_index[_state - 1]);
419  DMAT_tot = DMAT_tot - DMAT[0] + DMAT[1]; // Ground state + hole_contribution + electron contribution
420  }
421 
422  // fill DFT AO basis by going through all atoms
423 
424  Espfit esp = Espfit(_log);
425  if (_run_gwbse) {
426  esp.setUseECPs(true);
427  }
428  esp.Fit2Density(Atomlist, DMAT_tot, dftbasis, dftbs, "medium");
429  return true;
430 }

500 bool QMAPEMachine::hasConverged() {
501 
502  _convg_dR = false;
503  _convg_dQ = false;
504  _convg_dE_QM = false;
505  _convg_dE_MM = false;
506 
507  if (_iters.size() > 1) {
508 
509  QMMIter *iter_0 = _iters[_iters.size()-2];
510  QMMIter *iter_1 = _iters[_iters.size()-1];
511 
512  double dR = iter_1->getRMSdR();
513  double dQ = iter_1->getRMSdQ();
514  double dE_QM = iter_1->getQMEnergy() - iter_0->getQMEnergy();
515  double dE_MM = iter_1->getMMEnergy() - iter_0->getMMEnergy();
516 
517  if (dR <= _crit_dR) _convg_dR = true;
518  if (dQ <= _crit_dQ) _convg_dQ = true;
519  if (dE_QM*dE_QM <= _crit_dE_QM*_crit_dE_QM) _convg_dE_QM = true;
520  if (dE_MM*dE_MM <= _crit_dE_MM*_crit_dE_MM) _convg_dE_MM = true;
521  }
522 
523  _isConverged = ((_convg_dR && _convg_dQ) && (_convg_dE_QM && _convg_dE_MM));
524 
525  CTP_LOG(ctp::logINFO,*_log)
526  << (format("Convergence check")) << flush;
527  CTP_LOG(ctp::logINFO,*_log)
528  << format(" o Converged dR ? %s") % (_convg_dR ? "True" : "False") << flush;
529  CTP_LOG(ctp::logINFO,*_log)
530  << format(" o Converged dQ ? %s") % (_convg_dQ ? "True" : "False") << flush;
531  CTP_LOG(ctp::logINFO,*_log)
532  << format(" o Converged QM ? %s") % (_convg_dE_QM ? "True" : "False") << flush;
533  CTP_LOG(ctp::logINFO,*_log)
534  << format(" o Converged MM ? %s") % (_convg_dE_MM ? "True" : "False") << flush;
535 
536  return _isConverged;
537 }
