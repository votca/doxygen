34 QMAPEMachine::QMAPEMachine(ctp::XJob *job, ctp::Ewald3DnD *cape,
35  Property *opt, string sfx)
36  : _job(job), _cape(cape),_isConverged(false) {
37 
38  // CONVERGENCE THRESHOLDS
39  string key = sfx + ".qmmmconvg";
40  _crit_dR =opt->ifExistsReturnElseReturnDefault<double>(key + ".dR",0.01);//nm
41  _crit_dQ =opt->ifExistsReturnElseReturnDefault<double>(key + ".dQ",0.01);//e
42  _crit_dE_QM =opt->ifExistsReturnElseReturnDefault<double>(key + ".dQdE_QM",0.001);//eV
43  _crit_dE_MM =opt->ifExistsReturnElseReturnDefault<double>(key + ".dE_MM",_crit_dE_QM);//eV
44  _maxIter=opt->ifExistsReturnElseReturnDefault<int>(key + ".max_iter",32);
45 
46  // TASKS
47  key = sfx + ".tasks";
48  _run_ape = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_ape",true);
49  _run_dft = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_dft",true);
50  _run_gwbse = opt->ifExistsReturnElseReturnDefault<bool>(key+".run_gwbse",true);
51 
52 
53  if(_run_dft){
54  key=sfx+".dft";
55  string dft_xml = opt->ifExistsReturnElseThrowRuntimeError<string>(key + ".dftengine");
56  load_property_from_xml(_dft_options, dft_xml.c_str());
57  _externalgridaccuracy=opt->ifExistsReturnElseReturnDefault<string>(key+".externalgrid","medium");
58  }
59  if(_run_gwbse){
60  key = sfx + ".gwbse";
61  string gwbse_xml = opt->ifExistsReturnElseThrowRuntimeError<string>(key + ".gwbse_options");
62  load_property_from_xml(_gwbse_options, gwbse_xml.c_str());
63  std::string statestring= opt->ifExistsReturnElseReturnDefault<string>(key+".state","s1");
64  _initialstate.FromString(statestring);
65  key = sfx + ".gwbse.filter";
66  if(opt->exists(key)){
67  tools::Property filterprop=opt->get(key);
68  _filter.Initialize(filterprop);
69  _filter.setInitialState(_initialstate);
70  }
71  }
72  return;
73  }

98 void QMAPEMachine::Evaluate(ctp::XJob *job) {
99 
100  // PREPARE JOB DIRECTORY
101  string jobFolder = "qmapejob_" + boost::lexical_cast<string>(_job->getId())
102  + "_" + _job->getTag();
103  bool created = boost::filesystem::create_directory(jobFolder);
104 
105  CTP_LOG(ctp::logINFO,*_log) << flush;
106  if (created) {
107  CTP_LOG(ctp::logINFO,*_log) << "Created directory " << jobFolder << flush;
108  }
109 
110  CTP_LOG(ctp::logINFO,*_log)
111  << format("... dR %1$1.4f dQ %2$1.4f QM %3$1.4f MM %4$1.4f IT %5$d")
112  % _crit_dR % _crit_dQ % _crit_dE_QM % _crit_dE_MM % _maxIter << flush;
113 
114  // FIGURE OUT CHARGE + MULTIPLICITY
115  double dQ = 0.0;
116  for (unsigned i = 0; i < _job->getPolarTop()->QM0().size(); ++i) {
117  dQ += _job->getPolarTop()->QM0()[i]->CalcTotQ();
118  }
119  int chrg = round(dQ);
120  int spin = ( (chrg < 0) ? -chrg:chrg ) % 2 + 1;
121  CTP_LOG(ctp::logINFO,*_log) << "... Q = " << chrg << ", 2S+1 = " << spin << flush;
122 
123  if(chrg!=0){
124  throw runtime_error("Charged DFT calculations are not possible at the moment");
125  }
126 
127 
128 
129  int iterCnt = 0;
130  int iterMax = _maxIter;
131  for ( ; iterCnt < iterMax; ++iterCnt) {
132  CTP_LOG(ctp::logINFO,*_log) << "QMMM ITERATION:" << iterCnt+1<<" of "<<iterMax << flush;
133  //bool code = Iterate(jobFolder, iterCnt);
134  Iterate(jobFolder, iterCnt);
135  if (hasConverged()) {
136  CTP_LOG(ctp::logINFO,*_log) << "QMMM CONVERGED after:" << iterCnt+1<<" iterations."<< flush;
137 
138  break; }
139  }
140 
141  if (iterCnt == iterMax-1 && !_isConverged) {
142  CTP_LOG(ctp::logWARNING,*_log)
143  << format("Not converged within %1$d iterations.") % iterMax;
144  }
145 
146  return;
147 }

150 bool QMAPEMachine::Iterate(string jobFolder, int iterCnt) {
151 
152  // CREATE ITERATION OBJECT & SETUP RUN DIRECTORY
153  QMMIter *thisIter = this->CreateNewIter();
154  int iter = iterCnt;
155  string runFolder = jobFolder + "/iter_" + boost::lexical_cast<string>(iter);
156 
157  CTP_LOG(ctp::logINFO,*_log) << flush;
158  bool created = boost::filesystem::create_directory(runFolder);
159  if (created)
160  CTP_LOG(ctp::logDEBUG,*_log) << "Created directory " << runFolder << flush;
161  else
162  CTP_LOG(ctp::logWARNING,*_log) << "Could not create directory " << runFolder << flush;
163 
164  Orbitals orb_iter_input;
165 
166  qminterface.GenerateQMAtomsFromPolarSegs(_job->getPolarTop(), orb_iter_input);
167 
168  DFTEngine dftengine;
169  dftengine.Initialize(_dft_options);
170  dftengine.setLogger(_log);
171  dftengine.ConfigureExternalGrid(_externalgridaccuracy);
172  dftengine.Prepare(orb_iter_input);
173  if (iterCnt == 0) {
174  SetupPolarSiteGrids(dftengine.getExternalGridpoints(),orb_iter_input.QMAtoms());
175  }
176 
177  // COMPUTE POLARIZATION STATE WITH QM0(0)
178  if (_run_ape) {
179  if (iterCnt == 0) {
180  _cape->ShowAgenda(_log);
181  // Reset FGC, start from BGP state, apply FP fields (BG & FG)
182  _cape->EvaluateInductionQMMM(true, true, true, true, true);
183  // Add BG, do not add MM1 & QM0
184  _cape->EvaluatePotential(target_bg, true, false, false);
185  }
186  // Do not add BG & QM0, add MM1
187  _cape->EvaluatePotential(target_fg, false, true, false);
188  }
189 
190  dftengine.setExternalGrid(ExtractElGrid_fromPolarsites(),ExtractNucGrid_fromPolarsites());
191 
192  if (_run_dft) {
193  dftengine.Evaluate(orb_iter_input);
194  }
195 
196  orb_iter_input.WriteXYZ(runFolder + "/Fullstructure.xyz","Full structure");
197 
198  // Run GWBSE
199  if (_run_gwbse){
200  EvaluateGWBSE(orb_iter_input, runFolder);
201  }
202 
203  // COMPUTE POLARIZATION STATE WITH QM0(n+1)
204  if (_run_ape) {
205  // Update QM0 density: QM0(n) => QM0(n+1)
206  thisIter->UpdatePosChrgFromQMAtoms(orb_iter_input.QMAtoms(), _job->getPolarTop()->QM0());
207  // Do not reset FGC (= only reset FU), do not use BGP state, nor apply FP fields (BG & FG)
208  _cape->EvaluateInductionQMMM(false, false, false, false, false);
209  // COMPUTE MM ENERGY
210  _cape->EvaluateEnergyQMMM();
211  _cape->ShowEnergySplitting(_log);
212  }
213 
214  // THIS IS A SHORT VERSION OF PEWALD3D WITH QM/MM ENERGY SPLITTING
215  _cape->ShowAgenda(_log);
216  _cape->EvaluateInductionQMMM(true, true, true, true, true);
217  _cape->EvaluateEnergyQMMM();
218  _cape->ShowEnergySplitting(_log);
219 
220  return true;
221 }

230  bool QMAPEMachine::EvaluateGWBSE(Orbitals &orb, string runFolder) {
231 
232 
233 
234  if (_initialstate.Type() == QMStateType::Gstate) {
235  throw std::runtime_error("QMAPEMachine no GWBSE necessary for " + _initialstate.ToLongString());
236  }
237  GWBSE gwbse(orb);
238  CTP_LOG(ctp::logDEBUG, *_log) << "Excited state via GWBSE: " << flush;
239  CTP_LOG(ctp::logDEBUG, *_log) << " --- state: " << _initialstate.ToLongString() << flush;
240  // define own logger for GW-BSE that is written into a runFolder logfile
241  ctp::Logger gwbse_logger(ctp::logDEBUG);
242  gwbse_logger.setMultithreading(false);
243  //gwbse_logger.setPreface(ctp::logINFO, (format("\nGWBSE INF ...")).str());
244  gwbse_logger.setPreface(ctp::logERROR, (format("\nGWBSE ERR ...")).str());
245  gwbse_logger.setPreface(ctp::logWARNING, (format("\nGWBSE WAR ...")).str());
246  gwbse_logger.setPreface(ctp::logDEBUG, (format("\nGWBSE DBG ...")).str());
247  _filter.setLogger(&gwbse_logger);
248  gwbse.setLogger(&gwbse_logger);
249  _filter.PrintInfo();
250  gwbse.Initialize(_gwbse_options);
251  // actual GW-BSE run
252 
253  gwbse.Evaluate();
254 
255 
256 
257  // write logger to log file
258  ofstream ofs;
259  string gwbse_logfile = runFolder + "/gwbse.log";
260  ofs.open(gwbse_logfile.c_str(), ofstream::out);
261  if (!ofs.is_open()) {
262  throw runtime_error("Bad file handle: " + gwbse_logfile);
263  }
264  ofs << gwbse_logger << endl;
265  ofs.close();
266 
267  // PROCESSING the GW-BSE result
268  // - find the excited state of interest
269 
270  QMState nextState = _filter.CalcStateAndUpdate(orb);
271  // load DFT basis set (element-wise information) from xml file
272  BasisSet dftbs;
273  dftbs.LoadBasisSet(orb.getDFTbasisName());
274 
275  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Loaded DFT Basis Set " << orb.getDFTbasisName() << flush;
276 
277  // fill DFT AO basis by going through all atoms
278  AOBasis dftbasis;
279  dftbasis.AOBasisFill(dftbs, orb.QMAtoms());
280  Eigen::MatrixXd DMAT = orb.DensityMatrixFull(nextState);
281  Espfit esp = Espfit(_log);
282  esp.Fit2Density(orb.QMAtoms(), DMAT, dftbasis, "medium");
283  return true;
284  }

287 void QMAPEMachine::SetupPolarSiteGrids(const std::vector<const vec *>& gridpoints,const std::vector<QMAtom*>& atoms){
288  NumberofAtoms=0;
289  std::vector<QMAtom*>::const_iterator qmt;
290  std::vector<ctp::APolarSite*> sites1;
291  std::vector<ctp::APolarSite*> sites2;
292 
293  for(qmt=atoms.begin();qmt!=atoms.end();++qmt){
294  NumberofAtoms++;
295  sites1.push_back(qminterface.Convert(*qmt));
296  sites2.push_back(qminterface.Convert(*qmt));
297  }
298 
299 
300  std::vector<const vec *>::const_iterator grt;
301  for (grt=gridpoints.begin();grt<gridpoints.end();++grt){
302  ctp::APolarSite* site1=new ctp::APolarSite();
303  ctp::APolarSite* site2=new ctp::APolarSite();
304  tools::vec pos=*(*grt)*tools::conv::bohr2nm;
305  site1->setPos(pos);
306  site2->setPos(pos);
307  sites1.push_back(site1);
308  sites2.push_back(site2);
309  }
310 
311 
312  target_bg.push_back( new ctp::PolarSeg(0, sites1));
313  target_fg.push_back( new ctp::PolarSeg(0, sites2));
314 
315  return;
316  }

318 std::vector<double> QMAPEMachine::ExtractNucGrid_fromPolarsites() {
319  std::vector<double> gridpoints;
320  double int2hrt = tools::conv::int2V * tools::conv::ev2hrt;
321  for (unsigned i = 0; i < target_bg.size(); ++i) {
322  for (unsigned j = 0; j < NumberofAtoms; ++j) {
323  ctp::PolarSeg* seg1 = target_bg[i];
324  ctp::PolarSeg* seg2 = target_fg[i];
325  ctp::APolarSite* site1 = (*seg1)[j];
326  ctp::APolarSite* site2 = (*seg2)[j];
327  double value = (site1->getPhi() + site2->getPhi()) * int2hrt;
328  gridpoints.push_back(value);
329  }
330  }
331  return gridpoints;
332 }

353 bool QMAPEMachine::hasConverged() {
354 
355  bool convg_dR = false;
356  bool convg_dQ = false;
357  bool convg_dE_QM = false;
358  bool convg_dE_MM = false;
359 
360  if (_iters.size() > 1) {
361 
362  QMMIter *iter_0 = _iters[_iters.size()-2];
363  QMMIter *iter_1 = _iters[_iters.size()-1];
364 
365  double dR = iter_1->getRMSdR();
366  double dQ = iter_1->getRMSdQ();
367  double dE_QM = iter_1->getQMEnergy() - iter_0->getQMEnergy();
368  double dE_MM = iter_1->getMMEnergy() - iter_0->getMMEnergy();
369 
370  if (dR <= _crit_dR) convg_dR = true;
371  if (dQ <= _crit_dQ) convg_dQ = true;
372  if (dE_QM*dE_QM <= _crit_dE_QM*_crit_dE_QM) convg_dE_QM = true;
373  if (dE_MM*dE_MM <= _crit_dE_MM*_crit_dE_MM) convg_dE_MM = true;
374  }
375 
376  _isConverged = ((convg_dR && convg_dQ) && (convg_dE_QM && convg_dE_MM));
377 
378  CTP_LOG(ctp::logINFO,*_log)
379  << (format("Convergence check")) << flush;
380  CTP_LOG(ctp::logINFO,*_log)
381  << format(" o Converged dR ? %s") % (convg_dR ? "True" : "False") << flush;
382  CTP_LOG(ctp::logINFO,*_log)
383  << format(" o Converged dQ ? %s") % (convg_dQ ? "True" : "False") << flush;
384  CTP_LOG(ctp::logINFO,*_log)
385  << format(" o Converged QM ? %s") % (convg_dE_QM ? "True" : "False") << flush;
386  CTP_LOG(ctp::logINFO,*_log)
387  << format(" o Converged MM ? %s") % (convg_dE_MM ? "True" : "False") << flush;
388 
389  return _isConverged;
390 }
