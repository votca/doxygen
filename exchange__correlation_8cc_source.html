79  void ExchangeCorrelation::evalPBE_exchange(const double rho, const double drho_dX, const double drho_dY, const double drho_dZ, double& fx, double& dfx_drho, double& dfx_drho_dX, double& dfx_drho_dY, double& dfx_drho_dZ) {
80 
81  double norm = 0.0;
82  double drho_dX_normal = 0.0;
83  double drho_dY_normal = 0.0;
84  double drho_dZ_normal = 0.0;
85 
86 
87  if (rho <= 0.0) {
88  return ;
89  } else {
90  // exchange part
91  // constants
92  const double mu = 0.21951;
93  const double kappa = 0.804;
94 
95  // get PW91 exchange
96  double fx_LDA = 0.0;
97  double dfx_drho_LDA = 0.0;
98  evalPW91_exchange(rho, fx_LDA, dfx_drho_LDA);
99 
100  double s;
101  double ds_dr;
102  double ds_dnrho;
103  // prep norm of gradient and gradient normal vector
104  setNormGradient(drho_dX, drho_dY, drho_dZ, norm, drho_dX_normal, drho_dY_normal, drho_dZ_normal);
105  set_s(rho, norm, s, ds_dr, ds_dnrho);
106 
107 
108  if (std::abs(s) < 1.e-10) {
109 
110  fx = fx_LDA;
111  dfx_drho = dfx_drho_LDA;
112 
113  } else {
114 
115  double F_denom = 1.0 + mu * s * s / kappa;
116  double F = 1.0 + kappa - kappa / F_denom;
117  double dF_ds = 2.0 * mu * s / (F_denom * F_denom);
118 
119  // cout << "F_denom " << F_denom << " F " << F << " dF_ds " << dF_ds << endl;
120 
121  fx = F * fx_LDA;
122  dfx_drho = dfx_drho_LDA * F + fx_LDA * dF_ds * ds_dr;
123  dfx_drho_dX = fx_LDA * dF_ds * ds_dnrho * drho_dX_normal;
124  dfx_drho_dY = fx_LDA * dF_ds * ds_dnrho * drho_dY_normal;
125  dfx_drho_dZ = fx_LDA * dF_ds * ds_dnrho * drho_dZ_normal;
126 
127 
128  }
129 
130  } // exchange part
131  return;
132  } // evalPBE_exchange

179  void ExchangeCorrelation::evalPBE_correlation_H(const double rho, double& zeta, double& nrho, double& eps, double& deps_drho, double& deps_dzeta, double& H, double& dH_drho, double& dH_dzeta, double& dH_dnrho){
180 
181 
182  const double beta = 0.066725;
183  const double gamma = 0.031091;
184 
185  double gphi;
186  double dgphi_dzeta;
187  set_gphi(zeta,gphi,dgphi_dzeta);
188 
189 
190  double t;
191  double dt_drho;
192  double dt_dgphi;
193  double dt_dnrho;
194  set_t(rho, gphi, nrho, t, dt_drho, dt_dgphi, dt_dnrho );
195 
196 
197  if (std::abs(t) < 1e-10) {
198  H = 0.0;
199  dH_drho = 0.0;
200  dH_dzeta = 0.0;
201  dH_dnrho = 0.0;
202  return;
203  }
204 
205  // A part
206 
207 
208  double Aea = - eps / (gamma * pow(gphi,3) ); // there was some Hartree in here, but we didn't convert eps to Hartree before!
209  double Aexp = exp(Aea);
210  double Adenom;
211  if ( std::abs(Aea) < sqrt( std::numeric_limits<double>::epsilon() ) ){
212  Adenom = Aea;
213  } else {
214  Adenom = Aexp - 1.0;
215  }
216 
217  double A = beta / gamma / Adenom;
218  double dAea_deps = Aea / eps;
219  double dAea_dgphi = -3.0 * Aea / gphi;
220  double dA_dAea = -beta / gamma / pow(Adenom, 2) * Aexp;
221  double dA_deps = dA_dAea * dAea_deps;
222  double dA_dgphi = dA_dAea * dAea_dgphi;
223 
224  // --- Hb
225 
226  double Hb = A * pow(t, 2);
227  double dHb_dA = pow(t, 2);
228  double dHb_dt = 2.0 * A*t;
229 
230  // --- Ha
231 
232  double Ha_denom = 1.0 + Hb + Hb*Hb;
233  double Ha = beta / gamma * pow(t, 2) * (1.0 + Hb) / Ha_denom;
234 
235  double dHa_dHb = -beta / gamma * pow(t, 2) * Hb * (2.0 + Hb) / (Ha_denom * Ha_denom);
236  double dHa_dt_noHb = 2.0 * Ha / t;
237 
238  double dHa_dt = dHa_dt_noHb + dHa_dHb * dHb_dt;
239  double dHa_dA = dHa_dHb * dHb_dA;
240 
241  // --- H
242 
243  H = gamma * pow(gphi, 3) * log(1.0 + Ha); // there was a Hartree here
244 
245  double dH_dHa = gamma * pow(gphi, 3) / (1.0 + Ha); // there was a Hartree here
246 
247  double dH_dA = dH_dHa * dHa_dA;
248  double dH_dt = dH_dHa * dHa_dt;
249 
250  double dH_dgphi = 3.0 * H / gphi + dH_dA * dA_dgphi + dH_dt*dt_dgphi;
251  double dH_deps = dH_dA * dA_deps;
252 
253  dH_dzeta = dH_dgphi * dgphi_dzeta + dH_deps * deps_dzeta;
254  dH_drho = dH_deps * deps_drho + dH_dt * dt_drho;
255 
256  // cout << " rho " << rho << " dH_deps " << dH_deps << " deps_drho " << deps_drho << " dH_dt " << dH_dt << " dt_drho " << dt_drho << endl;
257 
258  dH_dnrho = dH_dt * dt_dnrho;
259 
260 
261 
262  return;
263  }

267  void ExchangeCorrelation::evalPW91_correlation_int(const double rho, const double zeta, double& ec, double& dec_drho, double& dec_dzeta){
268 
269  const double f0_second = 1.709920934161365447;
270 
271  // get zeta derived values ( all 0 for PBE )
272  double fzeta = 0.0 ;
273  double dfzeta_dzeta = 0.0 ;
274  if ( zeta != 0.0 ) set_fzeta(zeta,fzeta,dfzeta_dzeta);
275 
276  // get rs
277  double rs;
278  double drs_drho;
279  set_rs(rho, rs, drs_drho);
280 
281  // first evaluation of PW91 G function
282  double ec0;
283  double dec0_drs;
284  double p = 1.0;
285  evalPW91_correlation_G1(rs,ec0,dec0_drs,p);
286  double dec0_drho = dec0_drs * drs_drho;
287 
288  if ( zeta != 0.0 ){
289 
290  // second evaluation of PW91 G function
291  double ec1;
292  double dec1_drs;
293  evalPW91_correlation_G2(rs,ec1,dec1_drs,p);
294  double dec1_drho = dec1_drs * drs_drho;
295 
296  // third evaluation of PW91 G function
297  double eca;
298  double deca_drs;
299  evalPW91_correlation_G3(rs,eca,deca_drs,p);
300  eca = - eca;
301  deca_drs = - deca_drs;
302  double deca_drho = deca_drs * drs_drho;
303 
304  // now sum all contributions (weighted)
305  double afac = fzeta / f0_second * (1.0 - pow(zeta,4));
306  double dafac_dzeta = dfzeta_dzeta / f0_second * (1.0 - pow(zeta,4) ) - fzeta / f0_second * 4.0*pow(zeta,3);
307  double pfac = fzeta * pow(zeta,4);
308  double dpfac_dzeta = dfzeta_dzeta * pow(zeta,4) + fzeta * 4.0*pow(zeta,3);
309 
310  ec = ec0 + eca*afac + (ec1 - ec0)*pfac;
311  dec_drho = dec0_drho + deca_drho*afac + (dec1_drho - dec0_drho)*pfac;
312  dec_dzeta = eca*dafac_dzeta + (ec1 - ec0)*dpfac_dzeta;
313 
314  } else {
315 
316  ec = ec0;
317  dec_drho = dec0_drho;
318  dec_dzeta = 0.0;
319 
320 
321  }
322 
323  return;
324  }

328  void ExchangeCorrelation::evalPW91_correlation_G(const double rs, double& G, double& dG_drs, const double A, const double alpha1, const double beta1, const double beta2, const double beta3, const double beta4, const double p){
329 
330 
331  double rs_1 = sqrt(rs);
332  double rs_3 = rs * rs_1;
333  double rs_p = pow(rs, p);
334  double rs_p1 = rs * rs_p;
335 
336  double ln_denom = 2.0 * A * (beta1 * rs_1 + beta2 * rs + beta3 * rs_3 + beta4 * rs_p1);
337  double ln_arg = 1.0 + 1.0 / ln_denom;
338  double ln_res = log(ln_arg);
339  double pre_fac = -2.0 * A * (1.0 + alpha1 * rs);
340 
341  G = pre_fac * ln_res;
342 
343  double dlnd_rs = 2.0 * A * (0.5 * beta1 / rs_1 + beta2 + 1.5 * beta3 * rs_1 + (p + 1.0) * beta4 * rs_p);
344  dG_drs = -2.0 * A * alpha1 * ln_res + pre_fac / ln_arg * (-dlnd_rs / (ln_denom * ln_denom));
345  return;
346  } // evalPW91_correlation_G

450

452  return;

470  void ExchangeCorrelation::evalLDA_correlation(const double rho, double& fc, double& dfc_drho) {
471 
472 
473  if (rho <= 0.0) {
474  return;
475  } else {
476 
477  //const double Hartree = 2.0;
478  const double ec1_p = -0.1423; // * Hartree PZ_gamma_U
479  const double ec1_sq = 1.0529; // PZ_beta1_U
480  const double ec1_lin = 0.3334; // PX_beta2_U
481  const double ec2_p = -0.0480; // * Hartree PZ_B_U
482  const double ec2_log = 0.0311; // * Hartree PZ_A_U
483  const double ec2_lin = -0.0116; // * Hartree PZ_D_U
484  const double ec2_lig = 0.0020; // * Hartree PZ_C_U
485 
486  double ec;
487  double dec_drs;
488 
489  double rs;
490  double drs_dr;
491  set_rs(rho, rs, drs_dr);
492 
493  if (rs >= 1.0) {
494 
495  double sqrt_rs = sqrt(rs);
496  double ec_denom = 1.0 / (1.0 + ec1_sq * sqrt_rs + ec1_lin * rs);
497  ec = ec1_p * ec_denom;
498  dec_drs = -ec1_p * ec_denom * ec_denom * (0.5 * ec1_sq / sqrt_rs + ec1_lin);
499 
500  } else {
501 
502  double ln_rs = log(rs);
503  ec = ec2_p + ec2_log * ln_rs + ec2_lin * rs + ec2_lig * rs * ln_rs;
504  dec_drs = ec2_log / rs + ec2_lin + ec2_lig * (1.0 + ln_rs);
505 
506  }
507 
508 
509  double dec_drho = dec_drs * drs_dr;
510  fc = rho * ec;
511  dfc_drho = ec + rho * dec_drho;
512 
513 
514  }
515  return;
516  }

559  void ExchangeCorrelation::set_fzeta(const double zeta, double& fzeta, double& dfzeta_dzeta) {
560 
561  const double c4_3 = 4.0 / 3.0;
562  const double c1_3 = 1.0 / 3.0;
563 
564
