75  void BeginEvaluate(Topology *top, Topology *top_ref) override;
76  void EvalConfiguration(Topology *top, Topology *top_ref) override {
77  if (!_do_hybrid) {
78  // simply write the topology mapped by csgapplication class
79  if (_do_vel) {
80  top->SetHasVel(true);
81  }
82  if (_do_force) {
83  top->SetHasForce(true);
84  }
85  _writer->Write(top);
86  } else {
87  // we want to combine atomistic and coarse-grained into one topology
88  Topology hybtol;
89 
90  hybtol.setBox(top->getBox());
91  hybtol.setTime(top->getTime());
92  hybtol.setStep(top->getStep());
93 
94  // copy all residues from both
95  for (auto &residue : top_ref->Residues()) {
96  hybtol.CreateResidue(residue->getName());
97  }
98  for (auto &residue : top->Residues()) {
99  hybtol.CreateResidue(residue->getName());
100  }
101 
102  // copy all molecules and beads
103 
104  for (const auto &molecule : top_ref->Molecules()) {
105  Molecule *mi = hybtol.CreateMolecule(molecule.getName());
106  for (votca::Index i = 0; i < molecule.BeadCount(); i++) {
107  // copy atomistic beads of molecule
108  votca::Index beadid = molecule.getBead(i)->getId();
109 
110  const Bead *bi = molecule.getBead(i);
111  if (!hybtol.BeadTypeExist(bi->getType())) {
112  hybtol.RegisterBeadType(bi->getType());
113  }
114 
115  Bead *bn =
116  hybtol.CreateBead(bi->getSymmetry(), bi->getName(), bi->getType(),
117  bi->getResnr(), bi->getMass(), bi->getQ());
118  bn->setPos(bi->getPos());
119  if (bi->HasVel()) {
120  bn->setVel(bi->getVel());
121  }
122  if (bi->HasF()) {
123  bn->setF(bi->getF());
124  }
125 
126  mi->AddBead(hybtol.Beads()[beadid], molecule.getBeadName(i));
127  }
128 
129  if (mi->getId() < top->MoleculeCount()) {
130  // copy cg beads of molecule
131  Molecule *cgmol = top->getMolecule(mi->getId());
132  for (votca::Index i = 0; i < cgmol->BeadCount(); i++) {
133  Bead *bi = cgmol->getBead(i);
134  // todo: this is a bit dirty as a cg bead will always have the resid
135  // of its first parent
136  const Bead *bparent = molecule.getBead(0);
137  Bead *bn = hybtol.CreateBead(bi->getSymmetry(), bi->getName(),
138  bi->getType(), bparent->getResnr(),
139  bi->getMass(), bi->getQ());
140  bn->setPos(bi->getPos());
141  if (bi->HasVel()) {
142  bn->setVel(bi->getVel());
143  }
144  mi->AddBead(bi, bi->getName());
145  }
146  }
147  }
148  hybtol.setBox(top_ref->getBox());
149 
150  _writer->Write(&hybtol);
151  }
152  }

163 void CsgMapApp::BeginEvaluate(Topology *, Topology *) {
164  string out = OptionsMap()["out"].as<string>();
165  cout << "writing coarse-grained trajectory to " << out << endl;
166  _writer = TrjWriterFactory().Create(out);
167  if (_writer == nullptr) {
168  throw runtime_error("output format not supported: " + out);
169  }
170 
171  _do_hybrid = false;
172  if (OptionsMap().count("hybrid")) {
173  if (!_do_mapping) {
174  throw runtime_error("options hybrid and no-map not compatible");
175  }
176  cout << "Doing hybrid mapping..." << endl;
177  _do_hybrid = true;
178  }
179 
180  _do_vel = false;
181  if (OptionsMap().count("vel")) {
182  _do_vel = true;
183  }
184 
185  _do_force = false;
186  if (OptionsMap().count("force")) {
187  _do_force = true;
188  }
189 
190  _writer->Open(out);
191 }
