72  void BeginEvaluate(Topology *top, Topology *top_ref);
73  void EvalConfiguration(Topology *top, Topology *top_ref) {
74  if (!_do_hybrid) {
75  // simply write the topology mapped by csgapplication class
76  if (_do_vel) top->SetHasVel(true);
77  if (_do_force) top->SetHasForce(true);
78  _writer->Write(top);
79  } else {
80  // we want to combine atomistic and coarse-grained into one topology
81  Topology *hybtol = new Topology();
82 
83  ResidueContainer::iterator it_res;
84  MoleculeContainer::iterator it_mol;
85 
86  hybtol->setBox(top->getBox());
87  hybtol->setTime(top->getTime());
88  hybtol->setStep(top->getStep());
89 
90  // copy all residues from both
91  for (it_res = top_ref->Residues().begin();
92  it_res != top_ref->Residues().end(); ++it_res) {
93  hybtol->CreateResidue((*it_res)->getName());
94  }
95  for (it_res = top->Residues().begin(); it_res != top->Residues().end();
96  ++it_res) {
97  hybtol->CreateResidue((*it_res)->getName());
98  }
99 
100  // copy all molecules and beads
101 
102  for (it_mol = top_ref->Molecules().begin();
103  it_mol != top_ref->Molecules().end(); ++it_mol) {
104  Molecule *mi = hybtol->CreateMolecule((*it_mol)->getName());
105  for (int i = 0; i < (*it_mol)->BeadCount(); i++) {
106  // copy atomistic beads of molecule
107  int beadid = (*it_mol)->getBead(i)->getId();
108 
109  Bead *bi = (*it_mol)->getBead(i);
110  if (!hybtol->BeadTypeExist(bi->getType())) {
111  hybtol->RegisterBeadType(bi->getType());
112  }
113 
114  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(),
115  bi->getType(), bi->getResnr(),
116  bi->getMass(), bi->getQ());
117  bn->setPos(bi->getPos());
118  if (bi->HasVel()) bn->setVel(bi->getVel());
119  if (bi->HasF()) bn->setF(bi->getF());
120 
121  mi->AddBead(hybtol->Beads()[beadid], (*it_mol)->getBeadName(i));
122  }
123 
124  if (mi->getId() < top->MoleculeCount()) {
125  // copy cg beads of molecule
126  Molecule *cgmol = top->Molecules()[mi->getId()];
127  for (int i = 0; i < cgmol->BeadCount(); i++) {
128  Bead *bi = cgmol->getBead(i);
129  // todo: this is a bit dirty as a cg bead will always have the resid
130  // of its first parent
131  Bead *bparent = (*it_mol)->getBead(0);
132  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(),
133  bi->getType(), bparent->getResnr(),
134  bi->getMass(), bi->getQ());
135  bn->setPos(bi->getPos());
136  if (bi->HasVel()) bn->setVel(bi->getVel());
137  mi->AddBead(bi, bi->getName());
138  }
139  }
140  }
141  hybtol->setBox(top_ref->getBox());
142 
143  _writer->Write(hybtol);
144  }
145  }

181  _do_force = true;
182  }
