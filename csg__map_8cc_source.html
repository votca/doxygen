00099

00101                     BeadType *type = hybtol->GetOrCreateBeadType(bi->getType()->getName());

00113                     // copy cg beads of molecule

00116                         Bead *bi = cgmol->getBead(i);
00117                         // todo: this is a bit dirty as a cg bead will always have the resid of its first parent
00118                         Bead *bparent = (*it_mol)->getBead(0);
