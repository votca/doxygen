65 void EvalConfiguration(Topology *top, Topology *top_ref) {
66  if (!_do_hybrid) {
67  // simply write the topology mapped by csgapplication class
68  if (_do_vel) top->SetHasVel(true);
69  if (_do_force) top->SetHasForce(true);
70  _writer->Write(top);
71  } else {
72  // we want to combine atomistic and coarse-grained into one topology
73  Topology *hybtol = new Topology();
74 
75  BeadContainer::iterator it_bead;
76  ResidueContainer::iterator it_res;
77  MoleculeContainer::iterator it_mol;
78  InteractionContainer::iterator it_ia;
79 
80  hybtol->setBox(top->getBox());
81  hybtol->setTime(top->getTime());
82  hybtol->setStep(top->getStep());
83 
84  // copy all residues from both
85  for (it_res = top_ref->Residues().begin(); it_res != top_ref->Residues().end(); ++it_res) {
86  hybtol->CreateResidue((*it_res)->getName());
87  }
88  for (it_res = top->Residues().begin(); it_res != top->Residues().end(); ++it_res) {
89  hybtol->CreateResidue((*it_res)->getName());
90  }
91 
92  // copy all molecules and beads
93 
94  for(it_mol=top_ref->Molecules().begin();it_mol!=top_ref->Molecules().end(); ++it_mol) {
95  Molecule *mi = hybtol->CreateMolecule((*it_mol)->getName());
96  for (int i = 0; i < (*it_mol)->BeadCount(); i++) {
97  // copy atomistic beads of molecule
98  int beadid = (*it_mol)->getBead(i)->getId();
99 
100  Bead *bi = (*it_mol)->getBead(i);
101  BeadType *type = hybtol->GetOrCreateBeadType(bi->getType()->getName());
102  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(), type, bi->getResnr(), bi->getM(), bi->getQ());
103  bn->setOptions(bi->Options());
104  bn->setPos(bi->getPos());
105  if (bi->HasVel()) bn->setVel(bi->getVel());
106  if (bi->HasF()) bn->setF(bi->getF());
107 
108  mi->AddBead(hybtol->Beads()[beadid], (*it_mol)->getBeadName(i));
109 
110  }
111 
112  if (mi->getId() < top->MoleculeCount()) {
113  // copy cg beads of molecule
114  Molecule *cgmol = top->Molecules()[mi->getId()];
115  for (int i = 0; i < cgmol->BeadCount(); i++) {
116  Bead *bi = cgmol->getBead(i);
117  // todo: this is a bit dirty as a cg bead will always have the resid of its first parent
118  Bead *bparent = (*it_mol)->getBead(0);
119  BeadType *type = hybtol->GetOrCreateBeadType(bi->getType()->getName());
120  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(), type, bparent->getResnr(), bi->getM(), bi->getQ());
121  bn->setOptions(bi->Options());
122  bn->setPos(bi->getPos());
123  if (bi->HasVel()) bn->setVel(bi->getVel());
124  mi->AddBead(bi, bi->getName());
125  }
126  }
127 
128  }
129  hybtol->setBox(top_ref->getBox());
130 
131  _writer->Write(hybtol);
132  }
133  }

148 void CsgMapApp::BeginEvaluate(Topology *top, Topology *top_atom) {
149  string out = OptionsMap()["out"].as<string > ();
150  cout << "writing coarse-grained trajectory to " << out << endl;
151  _writer = TrjWriterFactory().Create(out);
152  if (_writer == NULL)
153  throw runtime_error("output format not supported: " + out);
154 
155  _do_hybrid = false;

159  cout << "Doing hybrid mapping..." << endl;
160  _do_hybrid = true;

162

167 
168  _do_force = false;
