65 void EvalConfiguration(Topology *top, Topology *top_ref) {
66  if (!_do_hybrid) {
67  // simply write the topology mapped by csgapplication class
68  if (_do_vel) top->SetHasVel(true);
69  if (_do_force) top->SetHasForce(true);
70  _writer->Write(top);
71  } else {
72  // we want to combine atomistic and coarse-grained into one topology
73  Topology *hybtol = new Topology();
74 
75  ResidueContainer::iterator it_res;
76  MoleculeContainer::iterator it_mol;
77 
78  hybtol->setBox(top->getBox());
79  hybtol->setTime(top->getTime());
80  hybtol->setStep(top->getStep());
81 
82  // copy all residues from both
83  for (it_res = top_ref->Residues().begin(); it_res != top_ref->Residues().end(); ++it_res) {
84  hybtol->CreateResidue((*it_res)->getName());
85  }
86  for (it_res = top->Residues().begin(); it_res != top->Residues().end(); ++it_res) {
87  hybtol->CreateResidue((*it_res)->getName());
88  }
89 
90  // copy all molecules and beads
91 
92  for(it_mol=top_ref->Molecules().begin();it_mol!=top_ref->Molecules().end(); ++it_mol) {
93  Molecule *mi = hybtol->CreateMolecule((*it_mol)->getName());
94  for (int i = 0; i < (*it_mol)->BeadCount(); i++) {
95  // copy atomistic beads of molecule
96  int beadid = (*it_mol)->getBead(i)->getId();
97 
98  Bead *bi = (*it_mol)->getBead(i);
99  BeadType *type = hybtol->GetOrCreateBeadType(bi->getType()->getName());
100  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(), type, bi->getResnr(), bi->getM(), bi->getQ());
101  bn->setOptions(bi->Options());
102  bn->setPos(bi->getPos());
103  if (bi->HasVel()) bn->setVel(bi->getVel());
104  if (bi->HasF()) bn->setF(bi->getF());
105 
106  mi->AddBead(hybtol->Beads()[beadid], (*it_mol)->getBeadName(i));
107 
108  }
109 
110  if (mi->getId() < top->MoleculeCount()) {
111  // copy cg beads of molecule
112  Molecule *cgmol = top->Molecules()[mi->getId()];
113  for (int i = 0; i < cgmol->BeadCount(); i++) {
114  Bead *bi = cgmol->getBead(i);
115  // todo: this is a bit dirty as a cg bead will always have the resid of its first parent
116  Bead *bparent = (*it_mol)->getBead(0);
117  BeadType *type = hybtol->GetOrCreateBeadType(bi->getType()->getName());
118  Bead *bn = hybtol->CreateBead(bi->getSymmetry(), bi->getName(), type, bparent->getResnr(), bi->getM(), bi->getQ());
119  bn->setOptions(bi->Options());
120  bn->setPos(bi->getPos());
121  if (bi->HasVel()) bn->setVel(bi->getVel());
122  mi->AddBead(bi, bi->getName());
123  }
124  }
125 
126  }
127  hybtol->setBox(top_ref->getBox());
128 
129  _writer->Write(hybtol);
130  }
131  }
