52 void Log2Mps::ParseOptions(const tools::Property &options) {
53 
54  QMPackageFactory::RegisterAll();
55 
56  _package = options.get(".package").as<std::string>();
57 
58  if (_package == "xtp") {

72 bool Log2Mps::Run() {
73 
74  // Logger (required for QM package, so we can just as well use it)
75  Logger log;
76  log.setCommonPreface("\n... ...");
77  log.setReportLevel(Log::current_level);
78  log.setMultithreading(true);
79 
80  // Set-up QM package
81  XTP_LOG(Log::error, log) << "Using package <" << _package << ">"
82  << std::flush;
83 
84  std::unique_ptr<QMPackage> qmpack = std::unique_ptr<QMPackage>(
85  QMPackageFactory::QMPackages().Create(_package));
86  qmpack->setLog(&log);
87  qmpack->setRunDir(".");
88  qmpack->setLogFileName(_logfile);
89 
90  // Create orbitals, fill with life & extract QM atoms
91 
92  StaticSegment atoms = qmpack->GetCharges();
93 
94  // Sanity checks, total charge
95 
96  if (atoms.size() < 1) {
97  throw std::runtime_error("ERROR No charges extracted from " + _logfile);
98  }
99 
100  double Q = atoms.CalcTotalQ();
101  XTP_LOG(Log::error, log) << atoms.size()
102  << " QM atoms, total charge Q = " << Q << std::flush;
103 
104  std::string tag =
105  "::LOG2MPS " + (boost::format("(log-file='%1$s' : %2$d QM atoms)") %
106  _logfile % atoms.size())
107  .str();
108  atoms.WriteMPS(_mpsfile, tag);
109  return true;
110 }
