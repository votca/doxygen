28 void Vxc_Grid::SortGridpointsintoBlocks(
29  const std::vector<std::vector<GridContainers::Cartesian_gridpoint> >&
30  grid) {
31  const double boxsize = 1; // 1 bohr
32 
33  Eigen::Array3d min =
34  Eigen::Array3d::Ones() * std::numeric_limits<double>::max();
35  Eigen::Array3d max =
36  Eigen::Array3d::Ones() * std::numeric_limits<double>::min();
37 
38  for (const auto& atom_grid : grid) {
39  for (const auto& gridpoint : atom_grid) {
40  const Eigen::Vector3d& pos = gridpoint.grid_pos;
41  max = max.max(pos.array()).eval();
42  min = min.min(pos.array()).eval();
43  }
44  }
45 
46  Eigen::Array3d molextension = max - min;
47  Eigen::Array<Index, 3, 1> numberofboxes =
48  (molextension / boxsize).ceil().cast<Index>();
49 
50  std::vector<std::vector<
51  std::vector<std::vector<const GridContainers::Cartesian_gridpoint*> > > >
52  boxes;
53  // creating temparray
54  for (Index i = 0; i < numberofboxes.x(); i++) {
55  std::vector<
56  std::vector<std::vector<const GridContainers::Cartesian_gridpoint*> > >
57  boxes_yz;
58  for (Index j = 0; j < numberofboxes.y(); j++) {
59  std::vector<std::vector<const GridContainers::Cartesian_gridpoint*> >
60  boxes_z;
61  for (Index k = 0; k < numberofboxes.z(); k++) {
62  std::vector<const GridContainers::Cartesian_gridpoint*> box;
63  box.reserve(100);
64  boxes_z.push_back(box);
65  }
66  boxes_yz.push_back(boxes_z);
67  }
68  boxes.push_back(boxes_yz);
69  }
70 
71  for (const auto& atomgrid : grid) {
72  for (const auto& gridpoint : atomgrid) {
73  Eigen::Array3d pos = gridpoint.grid_pos - min.matrix();
74  Eigen::Array<Index, 3, 1> index = (pos / boxsize).floor().cast<Index>();
75  boxes[index.x()][index.y()][index.z()].push_back(&gridpoint);
76  }
77  }
78  for (auto& boxes_xy : boxes) {
79  for (auto& boxes_z : boxes_xy) {
80  for (auto& box : boxes_z) {
81  if (box.empty()) {
82  continue;
83  }
84  GridBox gridbox;
85  for (const auto& point : box) {
86  gridbox.addGridPoint(*point);
87  }
88  _grid_boxes.push_back(gridbox);
89  }
90  }
91  }
92  return;
93 }

95 void Vxc_Grid::FindSignificantShells(const AOBasis& basis) {
96 
97 #pragma omp parallel for
98  for (Index i = 0; i < getBoxesSize(); i++) {
99  _grid_boxes[i].FindSignificantShells(basis);
100  }
101 
102  std::vector<GridBox> grid_boxes_copy;
103  // use vector of bool to indicate if a gridbox has already been merged into
104  // another
105  std::vector<bool> Merged = std::vector<bool>(_grid_boxes.size(), false);
106  for (Index i = 0; i < Index(_grid_boxes.size()); i++) {
107  if (Merged[i]) {
108  continue;
109  }
110  GridBox box = _grid_boxes[i];
111  if (box.Shellsize() < 1) {
112  continue;
113  }
114  Merged[i] = true;
115  for (Index j = i + 1; j < Index(_grid_boxes.size()); j++) {
116  if (GridBox::compareGridboxes(_grid_boxes[i], _grid_boxes[j])) {
117  Merged[j] = true;
118  box.addGridBox(_grid_boxes[j]);
119  }
120  }
121  grid_boxes_copy.push_back(box);
122  }
123 
124  _totalgridsize = 0;
125  for (auto& box : grid_boxes_copy) {
126  _totalgridsize += box.size();
127  box.PrepareForIntegration();
128  }
129  _grid_boxes = grid_boxes_copy;
130 }

144 Eigen::MatrixXd Vxc_Grid::CalcInverseAtomDist(const QMMolecule& atoms) const {
145  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(atoms.size(), atoms.size());
146 #pragma omp parallel for
147  for (Index i = 0; i < atoms.size(); ++i) {
148  const Eigen::Vector3d& pos_a = atoms[i].getPos();
149  for (Index j = 0; j < i; ++j) {
150  const Eigen::Vector3d& pos_b = atoms[j].getPos();
151  result(j, i) = 1 / (pos_a - pos_b).norm();
152  }
153  }
154  return result + result.transpose();
155 }

157 Index Vxc_Grid::UpdateOrder(LebedevGrid& sphericalgridofElement, Index maxorder,
158  std::vector<double>& PruningIntervals,
159  double r) const {
160  Index order;
161  Index maxindex = sphericalgridofElement.getIndexFromOrder(maxorder);
162  if (maxindex == 1) {
163  // smallest possible grid anyway, nothing to do
164  order = maxorder;
165  } else if (maxindex == 2) {
166  // only three intervals
167  if (r < PruningIntervals[0]) {
168  order = sphericalgridofElement.getOrderFromIndex(1); // 1;
169  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[3])) {
170  order = sphericalgridofElement.getOrderFromIndex(2);
171  } else {
172  order = sphericalgridofElement.getOrderFromIndex(1);
173  } // maxorder == 2
174  } else {
175  // five intervals
176  if (r < PruningIntervals[0]) {
177  order = sphericalgridofElement.getOrderFromIndex(2);
178  } else if ((r >= PruningIntervals[0]) && (r < PruningIntervals[1])) {
179  order = sphericalgridofElement.getOrderFromIndex(4);
180  } else if ((r >= PruningIntervals[1]) && (r < PruningIntervals[2])) {
181  order =
182  sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 4l));
183  } else if ((r >= PruningIntervals[2]) && (r < PruningIntervals[3])) {
184  order = maxorder;
185  } else {
186  order =
187  sphericalgridofElement.getOrderFromIndex(std::max(maxindex - 1, 1l));
188  }
189  }
190  return order;
191 }

193 GridContainers::Cartesian_gridpoint Vxc_Grid::CreateCartesianGridpoint(
194  const Eigen::Vector3d& atomA_pos, GridContainers::radial_grid& radial_grid,
195  GridContainers::spherical_grid& spherical_grid, Index i_rad,
196  Index i_sph) const {
197  GridContainers::Cartesian_gridpoint gridpoint;
198  double p = spherical_grid.phi[i_sph];
199  double t = spherical_grid.theta[i_sph];
200  const Eigen::Vector3d s =
201  Eigen::Vector3d{sin(p) * cos(t), sin(p) * sin(t), cos(p)};
202  double r = radial_grid.radius[i_rad];
203  gridpoint.grid_pos = atomA_pos + r * s;
204  gridpoint.grid_weight =
205  radial_grid.weight[i_rad] * spherical_grid.weight[i_sph];
206  return gridpoint;
207 }

209 Eigen::MatrixXd Vxc_Grid::CalcDistanceAtomsGridpoints(
210  const QMMolecule& atoms,
211  std::vector<GridContainers::Cartesian_gridpoint>& atomgrid) const {
212  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(atoms.size(), atomgrid.size());
213 #pragma omp parallel for
214  for (Index i = 0; i < atoms.size(); ++i) {
215  const Eigen::Vector3d& atom_pos = atoms[i].getPos();
216  for (Index j = 0; j < Index(atomgrid.size()); ++j) {
217  const auto& gridpoint = atomgrid[j];
218  result(i, j) = (atom_pos - gridpoint.grid_pos).norm();
219  }
220  }
221  return result;
222 }

224 void Vxc_Grid::SSWpartitionAtom(
225  const QMMolecule& atoms,
226  std::vector<GridContainers::Cartesian_gridpoint>& atomgrid, Index i_atom,
227  const Eigen::MatrixXd& Rij) const {
228  Eigen::MatrixXd AtomGridDist = CalcDistanceAtomsGridpoints(atoms, atomgrid);
229 
230 #pragma omp parallel for schedule(guided)
231  for (Index i_grid = 0; i_grid < Index(atomgrid.size()); i_grid++) {
232  Eigen::VectorXd p = SSWpartition(AtomGridDist.col(i_grid), Rij);
233  // check weight sum
234  double wsum = p.sum();
235  if (wsum != 0.0) {
236  // update the weight of this grid point
237  atomgrid[i_grid].grid_weight *= p[i_atom] / wsum;
238  } else {
239  std::cerr << "\nSum of partition weights of grid point " << i_grid
240  << " of atom " << i_atom << " is zero! ";
241  throw std::runtime_error("\nThis should never happen!");
242  }
243  } // partition weight for each gridpoint
244 }

246 void Vxc_Grid::GridSetup(const std::string& type, const QMMolecule& atoms,
247  const AOBasis& basis) {
248  GridContainers initialgrids;
249  // get radial grid per element
250  EulerMaclaurinGrid radialgridofElement;
251  initialgrids.radial_grids = radialgridofElement.CalculateAtomicRadialGrids(
252  basis, atoms, type); // this checks out 1:1 with NWChem results! AWESOME
253  LebedevGrid sphericalgridofElement;
254  initialgrids.spherical_grids =
255  sphericalgridofElement.CalculateSphericalGrids(atoms, type);
256 
257  // for the partitioning, we need all inter-center distances later, stored in
258  // matrix
259  Eigen::MatrixXd Rij = CalcInverseAtomDist(atoms);
260  std::vector<std::vector<GridContainers::Cartesian_gridpoint> > grid;
261 
262  for (Index i_atom = 0; i_atom < atoms.size(); ++i_atom) {
263  const QMAtom& atom = atoms[i_atom];
264 
265  const Eigen::Vector3d& atomA_pos = atom.getPos();
266  const std::string& name = atom.getElement();
267  GridContainers::radial_grid radial_grid =
268  initialgrids.radial_grids.at(name);
269  GridContainers::spherical_grid spherical_grid =
270  initialgrids.spherical_grids.at(name);
271 
272  // maximum order (= number of points) in spherical integration grid
273  Index maxorder = sphericalgridofElement.Type2MaxOrder(name, type);
274  // for pruning of integration grid, get interval boundaries for this element
275  std::vector<double> PruningIntervals =
276  radialgridofElement.CalculatePruningIntervals(name);
277  Index current_order = 0;
278  // for each radial value
279  std::vector<GridContainers::Cartesian_gridpoint> atomgrid;
280  for (Index i_rad = 0; i_rad < radial_grid.radius.size(); i_rad++) {
281  double r = radial_grid.radius[i_rad];
282 
283  // which Lebedev order for this point?
284  Index order =
285  UpdateOrder(sphericalgridofElement, maxorder, PruningIntervals, r);
286  // get new spherical grid, if order changed
287  if (order != current_order) {
288  spherical_grid = sphericalgridofElement.CalculateUnitSphereGrid(order);
289  current_order = order;
290  }
291 
292  for (Index i_sph = 0; i_sph < spherical_grid.phi.size(); i_sph++) {
293  GridContainers::Cartesian_gridpoint gridpoint =
294  CreateCartesianGridpoint(atomA_pos, radial_grid, spherical_grid,
295  i_rad, i_sph);
296  atomgrid.push_back(gridpoint);
297  } // spherical gridpoints
298  } // radial gridpoint
299 
300  SSWpartitionAtom(atoms, atomgrid, i_atom, Rij);
301  // now remove points from the grid with negligible weights
302  std::vector<GridContainers::Cartesian_gridpoint> atomgrid_cleanedup;
303  for (const auto& point : atomgrid) {
304  if (point.grid_weight > 1e-13) {
305  atomgrid_cleanedup.push_back(point);
306  }
307  }
308  grid.push_back(atomgrid_cleanedup);
309  } // atoms
310  SortGridpointsintoBlocks(grid);
311  FindSignificantShells(basis);
312  return;
313 }

315 Eigen::VectorXd Vxc_Grid::SSWpartition(const Eigen::VectorXd& rq_i,
316  const Eigen::MatrixXd& Rij) const {
317  const double ass = 0.725;
318  // initialize partition vector to 1.0
319  Eigen::VectorXd p = Eigen::VectorXd::Ones(rq_i.size());
320  const double tol_scr = 1e-10;
321  const double leps = 1e-6;
322  // go through centers
323  for (Index i = 1; i < rq_i.size(); i++) {
324  double rag = rq_i(i);
325  // through all other centers (one-directional)
326  for (Index j = 0; j < i; j++) {
327  if ((std::abs(p[i]) > tol_scr) || (std::abs(p[j]) > tol_scr)) {
328  double mu = (rag - rq_i(j)) * Rij(j, i);
329  if (mu > ass) {
330  p[i] = 0.0;
331  } else if (mu < -ass) {
332  p[j] = 0.0;
333  } else {
334  double sk;
335  if (std::abs(mu) < leps) {
336  sk = -1.88603178008 * mu + 0.5;
337  } else {
338  sk = erf1c(mu);
339  }
340  if (mu > 0.0) {
341  sk = 1.0 - sk;
342  }
343  p[j] = p[j] * sk;
344  p[i] = p[i] * (1.0 - sk);
345  }
346  }
347  }
348  }
349  return p;
350 }

352 double Vxc_Grid::erf1c(double x) const {
353  const static double alpha_erf1 = 1.0 / 0.30;
354  return 0.5 * std::erfc(std::abs(x / (1.0 - x * x)) * alpha_erf1);
