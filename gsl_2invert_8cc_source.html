00030 void linalg_invert( ub::matrix<double> &A, ub::matrix<double> &V){
00031         // matrix inversion using gsl
00032         
00033         gsl_error_handler_t *handler = gsl_set_error_handler_off();
00034         const size_t N = A.size1();
00035         // signum s (for LU decomposition)
00036         int s;
00037         //make copy of A as A is destroyed by GSL
00038         ub::matrix<double> work=A;
00039         V.resize(N, N, false);
00040         
00041         // Define all the used matrices
00042         gsl_matrix_view A_view = gsl_matrix_view_array(&work(0,0), N, N);
00043         gsl_matrix_view V_view = gsl_matrix_view_array(&V(0,0), N, N);
00044         gsl_permutation * perm = gsl_permutation_alloc (N);
00045         
00046         // Make LU decomposition of matrix A_view
00047         gsl_linalg_LU_decomp (&A_view.matrix, perm, &s);
00048 
00049         // Invert the matrix A_view
00050         (void)gsl_linalg_LU_invert (&A_view.matrix, perm, &V_view.matrix);
00051 
00052         gsl_set_error_handler(handler);
00053         
00054         // return (status != 0);
00055 }
