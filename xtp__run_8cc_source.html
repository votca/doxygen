59 bool XtpRun::EvaluateOptions() {
60 
61  std::string helpdir = "xtp/xml";
62  if (OptionsMap().count("list")) {
63  std::cout << "Available XTP calculators:\n";
64  for (const auto& calc : xtp::Calculators().getObjects()) {
65  PrintDescription(std::cout, calc.first, helpdir, Application::HelpShort);
66  }
67  StopExecution();
68  return true;
69  }
70 
71  if (OptionsMap().count("description")) {
72  CheckRequired("description", "no calculator is given");
73  tools::Tokenizer tok(OptionsMap()["description"].as<std::string>(),
74  " ,\n\t");
75  // loop over the names in the description string
76  for (const std::string& n : tok) {
77  // loop over calculators
78  bool printerror = true;
79  for (const auto& calc : xtp::Calculators().getObjects()) {
80 
81  if (n.compare(calc.first) == 0) {
82  PrintDescription(std::cout, calc.first, helpdir,
83  Application::HelpLong);
84  printerror = false;
85  break;
86  }
87  }
88  if (printerror) {
89  std::cout << "Calculator " << n << " does not exist\n";
90  }
91  }
92  StopExecution();
93  return true;
94  }
95 
96  xtp::StateApplication::EvaluateOptions();
97  CheckRequired("execute", "Nothing to do here: Abort.");
98 
99  tools::Tokenizer calcs(OptionsMap()["execute"].as<std::string>(), " ,\n\t");
100  std::vector<std::string> calc_string = calcs.ToVector();
101  if (calc_string.size() != 1) {
102  throw std::runtime_error(
103  "You can only run one calculator at the same time.");
104  }
105  bool found_calc = false;
106  for (const auto& calc : xtp::Calculators().getObjects()) {
107 
108  if (calc_string[0].compare(calc.first) == 0) {
109  xtp::StateApplication::SetCalculator(
110  xtp::Calculators().Create(calc_string[0]));
111  found_calc = true;
112  break;
113  }
114  }
115  if (!found_calc) {
116  std::cout << "Calculator " << calc_string[0] << " does not exist\n";
117  StopExecution();
118  } else {
119  _options.LoadFromXML(_op_vm["options"].as<std::string>());
120  }
121  return true;
122 }
