65 bool XtpRun::EvaluateOptions() {
66 
67  string helpdir = "xtp/xml";
68  string ctphelpdir="ctp/xml";
69  if (OptionsMap().count("list")) {
70  cout << "Available XTP calculators: \n";
71  for(xtp::Calculatorfactory::assoc_map::const_iterator iter=
72  xtp::Calculators().getObjects().begin();
73  iter != xtp::Calculators().getObjects().end(); ++iter) {
74  PrintDescription( std::cout, (iter->first), helpdir, Application::HelpShort );
75  }
76  cout << "Available (wrapped) CTP calculators: \n";
77  for(ctp::Calculatorfactory::assoc_map::const_iterator iter=
78  ctp::Calculators().getObjects().begin();
79  iter != ctp::Calculators().getObjects().end(); ++iter) {
80  PrintDescription( std::cout, (iter->first), ctphelpdir, Application::HelpShort );
81  }
82  StopExecution();
83  return true;
84  }
85 
86 
87  if (OptionsMap().count("description")) {
88  CheckRequired("description", "no calculator is given");
89  Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
90  // loop over the names in the description string
91  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
92  // loop over calculators
93  bool printerror = true;
94  for(xtp::Calculatorfactory::assoc_map::const_iterator iter=xtp::Calculators().getObjects().begin();
95  iter != xtp::Calculators().getObjects().end(); ++iter) {
96 
97  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
98  PrintDescription( std::cout, (iter->first), helpdir, Application::HelpLong );
99  printerror = false;
100  break;
101  }
102  }
103  for(ctp::Calculatorfactory::assoc_map::const_iterator iter=ctp::Calculators().getObjects().begin();
104  iter != ctp::Calculators().getObjects().end(); ++iter) {
105 
106  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
107  PrintDescription( std::cout, (iter->first), ctphelpdir, Application::HelpLong );
108  printerror = false;
109  break;
110  }
111  }
112  if ( printerror ) cout << "Calculator " << *n << " does not exist\n";
113  }
114  StopExecution();
115  return true;
116  }
117 
118  xtp::SqlApplication::EvaluateOptions();
119  CheckRequired("options", "Please provide an xml file with calculator options");
120  CheckRequired("execute", "Nothing to do here: Abort.");
121 
122  Tokenizer calcs(OptionsMap()["execute"].as<string>(), " ,\n\t");
123  Tokenizer::iterator it;
124  for (it = calcs.begin(); it != calcs.end(); it++) {
125  xtp::SqlApplication::AddCalculator(xtp::Calculators().Create((*it).c_str()));
126  }
127 
128  load_property_from_xml(_options, _op_vm["options"].as<string>());
129  return true;
130 }
