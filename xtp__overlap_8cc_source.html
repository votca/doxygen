48 void MOOApplication::Initialize()
49 {
50  // read in program options
51  namespace po = boost::program_options;

60 bool MOOApplication::EvaluateOptions()

64  _pos1 = OptionsMap()["pos1"].as<string>();

70 void MOOApplication::Run(void)
71 {
72  //cout << "Reading Crg unit Types" << endl;
73  JCalc jcalc(_conjseg);
74  //cout << "Finished reading Crg unit Types" << endl;
75 
76  ifstream in1(_pos1.c_str());
77  ifstream in2(_pos2.c_str());
78  int written=0;
79  while (in1 && in2) {
80  string name1, name2;
81  vec com1;
82  vec com2;
83  matrix or1;
84  matrix or2;
85  in1 >> name1 >> com1.x() >> com1.y() >> com1.z() >>
86  or1[0][0] >> or1[0][1] >> or1[0][2] >>
87  or1[1][0] >> or1[1][1] >> or1[1][2] >>
88  or1[2][0] >> or1[2][1] >> or1[2][2];
89  in2 >> name2 >> com2.x() >> com2.y() >> com2.z() >>
90  or2[0][0] >> or2[0][1] >> or2[0][2] >>
91  or2[1][0] >> or2[1][1] >> or2[1][2] >>
92  or2[2][0] >> or2[2][1] >> or2[2][2];
93 
94  if (!in1 || !in2) break;
95 
96  if (globals::verbose) {
97  cout << "molecule A: " << name1 << endl
98  << " translated by: " << com1 << endl
99  << " rotated by: " << endl << or1 << endl;
100 
101  cout << "molecule B: " << name2 << endl
102  << "translated by: " << com2 << endl
103  << "rotated by: " << endl << or2 << endl;
104  }
105 
106  CrgUnit * A = jcalc.CreateCrgUnit(0, name1);
107  CrgUnit * B = jcalc.CreateCrgUnit(1, name2);
108 
109  A->SetPos(0, com1);
110  A->SetNorm(0, or1[2]);
111  A->SetPlane(0, or1[1]);
112 
113  if (globals::verbose) {
114  cout << "molecule A: " << A->getName()
115  << " position: " << A->GetPos(0)
116  << " norm: " << A->GetNorm(0)
117  << " plane: " << A->GetPlane(0) << endl;
118  }
119 
120  B->SetPos(0, com2);
121  B->SetNorm(0, or2[2]);
122  B->SetPlane(0, or2[1]);
123 
124  if (globals::verbose) {
125  cout << "molecule B: " << B->getName()
126  << " position: " << B->GetPos(0)
127  << " norm: " << B->GetNorm(0)
128  << " plane: " << B->GetPlane(0) << endl;
129  }
130 
131  //write pdb file
132  mol_and_orb *molecule = ( A -> rotate_translate_beads() );
133  (*molecule).write_pdb(_pdbfile, "m1", written);
134  written += (*molecule).getN();
135  delete molecule;
136  molecule = ( B -> rotate_translate_beads() );
137  (*molecule).write_pdb(_pdbfile, "m2", written);
138  written += (*molecule).getN();
139  delete molecule;
140  ofstream fl;
141  fl.open(_pdbfile.c_str(), ios::app);
142  fl.setf(ios::fixed);
143  fl << "END" <<endl;
144 
145 
146  // cout << "Compute J" <<endl;
147  vector <double> Js = jcalc.CalcJ(*A, *B);
148  // cout << "Finished computing J" <<endl;
149  vector <double>::iterator itJ = Js.begin();
150  for (; itJ != Js.end(); ++itJ) cout << '\t' << *itJ << endl;
151  };
152 }
