26 void StateSaverSQLite::Open(ctp::Topology& qmtop, const string &file, bool lock) {
27  _sqlfile = file;
28  if (lock) this->LockStateFile();
29  _db.OpenHelper(file.c_str());
30 
31  _qmtop = &qmtop;
32  _frames.clear();
33  _topIds.clear();
34 
35  _frame = 0;
36  _current_frame = -1;
37  _was_read = false;
38 
39  // Query available frames in database
40  Statement *stmt = _db.Prepare("SELECT _id, id FROM frames;");
41  while(stmt->Step() != SQLITE_DONE) {
42  _frames.push_back(stmt->Column<int>(0));
43  _topIds.push_back(stmt->Column<int>(1));
44  }
45  delete stmt;
46  if (lock) this->UnlockStateFile();
47  return;
48 }

52 void StateSaverSQLite::WriteFrame() {
53  this->LockStateFile();
54  bool hasAlready = this->HasTopology(_qmtop);
55 
56  if ( ! hasAlready ) {
57  if (_qmtop->getDatabaseId() >= 0 ) {
58  throw runtime_error ("How was this topology generated? ");
59  }
60  _qmtop->setDatabaseId(_frames.size());
61  _topIds.push_back(_qmtop->getDatabaseId());
62  cout << "Saving ";
63  }
64  else {
65  cout << "Updating ";
66  }
67 
68  cout << "MD+QM topology ID " << _qmtop->getDatabaseId()
69  << " (step = " << _qmtop->getStep()
70  << ", time = " << _qmtop->getTime()
71  << ") to " << _sqlfile << endl;
72  cout << "... ";
73 
74  _db.BeginTransaction();
75 
76  this->WriteMeta(hasAlready);
77  this->WriteMolecules(hasAlready);
78  this->WriteSegTypes(hasAlready);
79  this->WriteSegments(hasAlready);
80  this->WriteFragments(hasAlready);
81  this->WriteAtoms(hasAlready);
82  this->WritePairs(hasAlready);
83  this->WriteSuperExchange(hasAlready);
84 
85  _db.EndTransaction();
86 
87  cout << ". " << endl;
88  this->UnlockStateFile();
89  return;
90 }

93 void StateSaverSQLite::WriteMeta(bool update) {
94 
95  Statement *stmt;
96  if ( update ) {
97  return; // Nothing to do here
98  }
99  else {
100  stmt = _db.Prepare( "INSERT INTO frames ("
101  "id, time, step, "
102  "box11, box12, box13, "
103  "box21, box22, box23, "
104  "box31, box32, box33, "
105  "canRigid )"
106  "VALUES ("
107  "?, ?, ?,"
108  "?, ?, ?,"
109  "?, ?, ?,"
110  "?, ?, ?,"
111  "?)");
112  }
113 
114  stmt->Bind(1, _qmtop->getDatabaseId());
115  stmt->Bind(2, _qmtop->getTime());
116  stmt->Bind(3, _qmtop->getStep());
117 
118  for (int i = 0; i < 3; i++) {
119  for (int j = 0; j < 3; j++) {
120  stmt->Bind(4+3*i+j, _qmtop->getBox().get(i,j));
121  }
122  }
123 
124  int canRigid = 0;
125  if (_qmtop->canRigidify()) { canRigid = 1; }
126  stmt->Bind(13, canRigid);
127 
128  stmt->InsertStep();
129  delete stmt;
130  stmt = NULL;
131 }

662 void StateSaverSQLite::ReadMeta(int topId) {
663 
664  Statement *stmt = _db.Prepare("SELECT "
665  "time, step, "
666  "box11, box12, box13, "
667  "box21, box22, box23, "
668  "box31, box32, box33, "
669  "canRigid "
670  "FROM frames WHERE "
671  "id = ?;");
672  stmt->Bind(1, topId);
673 
674  if (stmt->Step() == SQLITE_DONE) {
675  // ReadFrame should not have been called in the first place:
676  throw runtime_error("Database appears to be broken. Abort...");
677  }
678 
679  _qmtop->setTime(stmt->Column<double>(0));
680  _qmtop->setStep(stmt->Column<int>(1));
