68  template <typename MatrixReplacement>
69  void solve(const MatrixReplacement &A, Index neigen,
70  Index size_initial_guess = 0) {
71 
72  if (_max_search_space < neigen) {
73  _max_search_space = neigen * 5;
74  }
75  std::chrono::time_point<std::chrono::system_clock> start =
76  std::chrono::system_clock::now();
77  Index op_size = A.rows();
78 
79  checkOptions(op_size);
80  printOptions(op_size);
81 
82  // initial guess size
83  if (size_initial_guess == 0) {
84  size_initial_guess = 2 * neigen;
85  }
86  _restart_size = size_initial_guess;
87 
88  // get the diagonal of the operator
89  this->_Adiag = A.diagonal();
90 
91  // target the lowest diagonal element
92  ProjectedSpace proj = initProjectedSpace(neigen, size_initial_guess);
93  RitzEigenPair rep;
94  XTP_LOG(Log::error, _log)
95  << TimeStamp() << " iter\tSearch Space\tNorm" << flush;
96 
97  for (_i_iter = 0; _i_iter < _iter_max; _i_iter++) {
98 
99  updateProjection(A, proj);
100 
101  rep = getRitzEigenPairs(proj);
102 
103  bool converged = checkConvergence(rep, proj, neigen);
104 
105  printIterationData(rep, proj, neigen);
106 
107  bool last_iter = _i_iter == (_iter_max - 1);
108 
109  if (converged) {
110  storeConvergedData(rep, neigen);
111  break;
112  } else if (last_iter) {
113  storeNotConvergedData(rep, proj.root_converged, neigen);
114  break;
115  }
116  Index extension_size = extendProjection(rep, proj);
117  bool do_restart = (proj.search_space() > _max_search_space);
118 
119  if (do_restart) {
120  restart(rep, proj, extension_size);
121  }
122  }
123 
124  printTiming(start);
125  }

174  template <typename MatrixReplacement>
175  void updateProjection(const MatrixReplacement &A,
176  ProjectedSpace &proj) const {
177 
178  if (_i_iter == 0) {
179  proj.AV = A * proj.V;
180  proj.T = proj.V.transpose() * proj.AV;
181  if (_matrix_type == MATRIX_TYPE::HAM) {
182  proj.AAV = A * proj.AV;
183  proj.B = proj.V.transpose() * proj.AAV;
184  }
185 
186  } else {
187  /* if we use a Gram Schmid(GS) orthogonalisation we do not have to
188  recompute the entire projection as GS doesn't change the original
189  subspace*/
190  Index old_dim = proj.AV.cols();
191  Index new_dim = proj.V.cols();
192  Index nvec = new_dim - old_dim;
193  proj.AV.conservativeResize(Eigen::NoChange, new_dim);
194  proj.AV.rightCols(nvec) = A * proj.V.rightCols(nvec);
195 
196  proj.T.conservativeResize(new_dim, new_dim);
197  proj.T.rightCols(nvec) = proj.V.transpose() * proj.AV.rightCols(nvec);
198 
199  if (_matrix_type == MATRIX_TYPE::SYMM) {
200  proj.T.bottomLeftCorner(nvec, old_dim) =
201  proj.T.topRightCorner(old_dim, nvec).transpose();
202 
203  } else {
204  proj.T.bottomLeftCorner(nvec, old_dim) =
205  proj.V.rightCols(nvec).transpose() * proj.AV.leftCols(old_dim);
206 
207  proj.AAV.conservativeResize(Eigen::NoChange, new_dim);
208  proj.AAV.rightCols(nvec) = A * proj.AV.rightCols(nvec);
209  proj.B.conservativeResize(new_dim, new_dim);
210  proj.B.rightCols(nvec) = proj.V.transpose() * proj.AAV.rightCols(nvec);
211  proj.B.bottomLeftCorner(nvec, old_dim) =
212  proj.V.rightCols(nvec).transpose() * proj.AAV.leftCols(old_dim);
213  }
214  }
215  }
