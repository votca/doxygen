60  template <typename MatrixReplacement>
61  void solve(const MatrixReplacement &A, Index neigen,
62  Index size_initial_guess = 0) {
63 
64  if (_max_search_space < neigen) {
65  _max_search_space = neigen * 5;
66  }
67  std::chrono::time_point<std::chrono::system_clock> start =
68  std::chrono::system_clock::now();
69  Index op_size = A.rows();
70 
71  checkOptions(op_size);
72  printOptions(op_size);
73 
74  // initial guess size
75  if (size_initial_guess == 0) {
76  size_initial_guess = 2 * neigen;
77  }
78 
79  // get the diagonal of the operator
80  this->_Adiag = A.diagonal();
81 
82  // target the lowest diagonal element
83  ProjectedSpace proj = initProjectedSpace(neigen, size_initial_guess);
84  RitzEigenPair rep;
85 
86  for (_i_iter = 0; _i_iter < _iter_max; _i_iter++) {
87 
88  bool do_restart = (proj.search_space() > _max_search_space);
89 
90  if (do_restart) {
91  restart(rep, proj, size_initial_guess);
92 
93  } else {
94  updateProjection(A, proj);
95  rep = getGuessVectors(A, proj);
96  extendProjection(rep, proj);
97  }
98 
99  printIterationData(rep, proj, neigen);
100 
101  bool converged = (rep.res_norm().head(neigen) < _tol).all();
102  bool last_iter = _i_iter == (_iter_max - 1);
103 
104  if (converged) {
105  storeConvergedData(rep, neigen);
106  break;
107  } else if (last_iter) {
108  storeNotConvergedData(rep, proj.root_converged, neigen);
109  break;
110  }
111  }
112 
113  printTiming(start);
114  }

141  struct RitzEigenPair {

151  struct ProjectedSpace {

153  Eigen::MatrixXd AV; // A * V

162  template <typename MatrixReplacement>
163  void updateProjection(const MatrixReplacement &A,
164  ProjectedSpace &proj) const {
165 
166  if (_i_iter == 0 || _davidson_ortho == ORTHO::QR) {
167  /* if we use QR we need to recompute the entire projection
168  since QR will modify original subspace*/
169  proj.AV = A * proj.V;
170  proj.T = proj.V.transpose() * proj.AV;
171 
172  } else if (_davidson_ortho == ORTHO::GS) {
173  /* if we use a GS ortho we do not have to recompute
174  the entire projection as GS doesn't change the original subspace*/
175  Index old_dim = proj.T.cols();
176  Index new_dim = proj.V.cols();
177  Index nvec = new_dim - old_dim;
178  proj.AV.conservativeResize(Eigen::NoChange, new_dim);
179  proj.AV.rightCols(nvec) = A * proj.V.rightCols(nvec);
180  Eigen::MatrixXd VAV = proj.V.transpose() * proj.AV.rightCols(nvec);
181  proj.T.conservativeResize(new_dim, new_dim);
182  proj.T.rightCols(nvec) = VAV;
183  proj.T.bottomLeftCorner(nvec, old_dim) =
184  proj.T.topRightCorner(old_dim, nvec).transpose();
185  }
186  }

188  template <typename MatrixReplacement>
189  RitzEigenPair getGuessVectors(const MatrixReplacement &A,
190  const ProjectedSpace &proj) const {
191  // get the ritz vectors
192  switch (this->_matrix_type) {
193  case MATRIX_TYPE::SYMM: {
194  return getRitz(proj);
195  }
196  case MATRIX_TYPE::HAM: {
197  return getHarmonicRitz(A, proj);
198  }
199  }
200  return RitzEigenPair();
201  }

203  template <typename MatrixReplacement>
204  RitzEigenPair getHarmonicRitz(const MatrixReplacement &A,
205  const ProjectedSpace &proj) const {
206 
207  /* Compute the Harmonic Ritz vector following
208  * Computing Interior Eigenvalues of Large Matrices
209  * Ronald B Morgan
210  * LINEAR ALGEBRA AND ITS APPLICATIONS 154-156:289-309 (1991)
211  * https://cpb-us-w2.wpmucdn.com/sites.baylor.edu/dist/e/71/files/2015/05/InterEvals-1vgdz91.pdf
212  */
213 
214  RitzEigenPair rep;
215  Eigen::MatrixXd B = proj.V.transpose() * (A * proj.AV);
216 
217  bool return_eigenvectors = true;
218  Eigen::GeneralizedEigenSolver<Eigen::MatrixXd> ges(proj.T, B,
219  return_eigenvectors);
220 
221  ArrayXl idx = DavidsonSolver::argsort(ges.eigenvalues().real());
222  // smallest to largest
223  idx = idx.reverse();
224  // we need the largest values, because this is the inverse value, so reverse
225  // list
226 
227  rep.U = DavidsonSolver::extract_vectors(ges.eigenvectors().real(), idx);
228  rep.U.colwise().normalize();
229  rep.lambda = (rep.U.transpose() * proj.T * rep.U).diagonal();
230 
231  rep.q = proj.V * rep.U; // Ritz vectors
232  rep.res = proj.AV * rep.U - rep.q * rep.lambda.asDiagonal(); // residues
233  return rep;
234  }
