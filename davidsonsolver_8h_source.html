58  template <typename MatrixReplacement>
59  void solve(MatrixReplacement &A, int neigen, int size_initial_guess = 0) {
60 
61  std::chrono::time_point<std::chrono::system_clock> start, end;
62  std::chrono::duration<double> elapsed_time;
63  start = std::chrono::system_clock::now();
64 
65  CTP_LOG(ctp::logDEBUG, _log)
66  << ctp::TimeStamp() << " Davidson Solver" << flush;
67 
68  switch (this->_davidson_correction) {
69 
70  case CORR::DPR:
71  CTP_LOG(ctp::logDEBUG, _log)
72  << ctp::TimeStamp() << " DPR Correction" << flush;
73  break;
74 
75  case CORR::OLSEN:
76  CTP_LOG(ctp::logDEBUG, _log)
77  << ctp::TimeStamp() << " Olsen Correction" << flush;
78  break;
79  }
80 
81  switch (this->_davidson_ortho) {
82 
83  case ORTHO::GS:
84  CTP_LOG(ctp::logDEBUG, _log)
85  << ctp::TimeStamp() << " Gram-Schmidt Orthogonalization" << flush;
86  break;
87 
88  case ORTHO::QR:
89  CTP_LOG(ctp::logDEBUG, _log)
90  << ctp::TimeStamp() << " QR Orthogonalization" << flush;
91  break;
92  }
93 
94  CTP_LOG(ctp::logDEBUG, _log)
95  << ctp::TimeStamp() << " Tolerance : " << this->_tol << flush;
96 
97  int op_size = A.rows();
98 
99  CTP_LOG(ctp::logDEBUG, _log)
100  << ctp::TimeStamp() << " Matrix size : " << op_size << 'x' << op_size
101  << flush;
102 
103  //. search space exeeding the system size
104  if (_max_search_space > op_size) {
105  CTP_LOG(ctp::logDEBUG, _log)
106  << ctp::TimeStamp() << " Warning Max search space ("
107  << _max_search_space << ") larger than system size (" << op_size
108  << ")" << flush;
109  _max_search_space = op_size;
110  CTP_LOG(ctp::logDEBUG, _log)
111  << ctp::TimeStamp() << " Max search space set to " << op_size
112  << flush;
113  }
114 
115  // initial guess size
116  if (size_initial_guess == 0) {
117  size_initial_guess = 2 * neigen;
118  }
119  int search_space = size_initial_guess;
120  int size_restart = size_initial_guess;
121  int size_update = DavidsonSolver::get_size_update(neigen);
122  int nupdate;
123 
124  Eigen::ArrayXd res_norm = Eigen::ArrayXd::Zero(size_update);
125  Eigen::ArrayXd root_converged = Eigen::ArrayXd::Zero(size_update);
126 
127  double percent_converged;
128  bool has_converged = false;
129 
130  // initialize the guess eigenvector
131  Eigen::VectorXd Adiag = A.diagonal();
132 
133  // target the lowest diagonal element
134  Eigen::MatrixXd V =
135  DavidsonSolver::SetupInitialEigenvectors(Adiag, size_initial_guess);
136 
137  // eigenvalues and Ritz Eigenvector
138  Eigen::VectorXd lambda;
139  Eigen::MatrixXd q;
140 
141  // project the matrix on the trial subspace
142  Eigen::MatrixXd T = V.transpose() * (A * V);
143 
144  CTP_LOG(ctp::logDEBUG, _log)
145  << ctp::TimeStamp() << " iter\tSearch Space\tNorm" << flush;
146 
147  // Start of the main iteration loop
148  for (int iiter = 0; iiter < _iter_max; iiter++) {
149 
150  // diagonalize the small subspace
151  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(T);
152  lambda = es.eigenvalues();
153  Eigen::MatrixXd U = es.eigenvectors();
154 
155  // Ritz eigenvectors
156  q = V * U;
157 
158  // precompute A*Q - lambda Q
159  Eigen::MatrixXd Aq = A * q - q * lambda.asDiagonal();
160 
161  // correction vectors
162  nupdate = 0;
163  for (int j = 0; j < size_update; j++) {
164 
165  // skip the root that have already converged
166  if (root_converged[j]) {
167  continue;
168  }
169  nupdate += 1;
170 
171  // residue vector
172  Eigen::VectorXd w = Aq.col(j);
173  res_norm[j] = w.norm();
174 
175  // compute correction vector
176  switch (this->_davidson_correction) {
177  case CORR::DPR:
178  w = DavidsonSolver::dpr_correction(w, Adiag, lambda(j));
179  break;
180  case CORR::OLSEN:
181  Eigen::VectorXd tmp = q.col(j);
182  w = DavidsonSolver::olsen_correction(w, tmp, Adiag, lambda(j));
183  break;
184  }
185 
186  // append the correction vector to the search space
187  V.conservativeResize(Eigen::NoChange, V.cols() + 1);
188  V.col(V.cols() - 1) = w.normalized();
189 
190  // track converged root
191  root_converged[j] = res_norm[j] < _tol;
192  }
193 
194  // Print iteration data
195  percent_converged = 100 * root_converged.head(neigen).sum() / neigen;
196  CTP_LOG(ctp::logDEBUG, _log)
197  << ctp::TimeStamp()
198  << format(" %1$4d %2$12d \t %3$4.2e \t %4$5.2f%% converged") % iiter %
199  search_space % res_norm.head(neigen).maxCoeff() %
200  percent_converged
201  << flush;
202 
203  // update
204  search_space = V.cols();
205 
206  // break if converged
207  if ((res_norm.head(neigen) < _tol).all()) {
208  has_converged = true;
209  break;
210  }
211 
212  // check if we need to restart
213  if (search_space > _max_search_space or search_space > op_size) {
214 
215  V = q.leftCols(size_restart);
216  V.colwise().normalize();
217  search_space = size_restart;
218 
219  // recompute the projected matrix
220  T = V.transpose() * (A * V);
221  }
222 
223  // continue otherwise
224  else {
225 
226  switch (this->_davidson_ortho) {
227  case ORTHO::GS:
228  V = DavidsonSolver::gramschmidt_ortho(V, V.cols() - nupdate);
229  DavidsonSolver::update_projected_matrix<MatrixReplacement>(T, A, V);
230  break;
231  case ORTHO::QR:
232  V = DavidsonSolver::QR_ortho(V);
233  T = V.transpose() * (A * V);
234  break;
235  }
236  }
237  }
238 
239  // store the eigenvalues/eigenvectors
240  this->_eigenvalues = lambda.head(neigen);
241  this->_eigenvectors = q.leftCols(neigen);
242  this->_eigenvectors.colwise().normalize();
243 
244  CTP_LOG(ctp::logDEBUG, _log)
245  << ctp::TimeStamp() << "-----------------------------------" << flush;
246 
247  if (!has_converged) {
248  CTP_LOG(ctp::logDEBUG, _log)
249  << ctp::TimeStamp() << "- Warning : Davidson " << percent_converged
250  << "% converged after " << _iter_max << " iterations." << flush;
251 
252  for (int i = 0; i < neigen; i++) {
253  if (not root_converged[i]) {
254  this->_eigenvalues(i) = 0;
255  this->_eigenvectors.col(i) = Eigen::VectorXd::Zero(op_size);
256  }
257  }
258  } else {
259  end = std::chrono::system_clock::now();
260  elapsed_time = end - start;
261  CTP_LOG(ctp::logDEBUG, _log)
262  << ctp::TimeStamp() << "- Davidson converged in "
263  << elapsed_time.count() << "secs." << flush;
264  }
265  CTP_LOG(ctp::logDEBUG, _log)
266  << ctp::TimeStamp() << "-----------------------------------" << flush;
267  }

297  template <class MatrixReplacement>
298  void update_projected_matrix(Eigen::MatrixXd &T, MatrixReplacement &A,
299  Eigen::MatrixXd &V) const {
300  int size = V.rows();
301  int old_dim = T.cols();
302  int new_dim = V.cols();
303  int nvec = new_dim - old_dim;
304 
305  T.conservativeResize(new_dim, new_dim);
306  Eigen::MatrixXd _tmp = A * V.block(0, old_dim, size, nvec);
307  T.block(0, old_dim, new_dim, nvec) = V.transpose() * _tmp;
308  T.block(old_dim, 0, nvec, old_dim) =
309  T.block(0, old_dim, old_dim, nvec).transpose();
310  return;
311  }
