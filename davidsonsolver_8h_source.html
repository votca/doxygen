60  template <typename MatrixReplacement>
61  void solve(const MatrixReplacement &A, int neigen,
62  int size_initial_guess = 0) {
63 
64  std::chrono::time_point<std::chrono::system_clock> start =
65  std::chrono::system_clock::now();
66  int op_size = A.rows();
67 
68  checkOptions(op_size);
69  printOptions(op_size);
70 
71  // initial guess size
72  if (size_initial_guess == 0) {
73  size_initial_guess = 2 * neigen;
74  }
75 
76  // get the diagonal of the operator
77  this->_Adiag = A.diagonal();
78 
79  // target the lowest diagonal element
80  ProjectedSpace proj = initProjectedSpace(neigen, size_initial_guess);
81  RitzEigenPair rep;
82 
83  for (int iiter = 0; iiter < _iter_max; iiter++) {
84 
85  // restart or update the projection
86  if (proj.search_space > _max_search_space) {
87  restart(rep, proj, size_initial_guess);
88  } else {
89  updateProjection(A, proj, iiter);
90  }
91 
92  // get the ritz vectors
93  switch (this->_matrix_type) {
94  case MATRIX_TYPE::SYMM: {
95  rep = getRitz(proj);
96  break;
97  }
98  case MATRIX_TYPE::HAM: {
99  rep = getHarmonicRitz(A, proj);
100  }
101  }
102 
103  // etend the subspace
104  int nupdate = extendProjection(rep, proj);
105 
106  // Print iteration data
107  printIterationData(rep, proj, neigen, iiter);
108 
109  // converged
110  bool converged = (rep.res_norm.head(neigen) < _tol).all();
111  bool last_iter = iiter == (_iter_max - 1);
112 
113  // break if converged or last
114  if (converged) {
115  storeConvergedData(rep, neigen, iiter);
116  break;
117  } else if (last_iter) {
118  storeNotConvergedData(rep, proj.root_converged, neigen);
119  break;
120  }
121 
122  proj.V = orthogonalize(proj.V, nupdate);
123  }
124 
125  printTiming(start);
126  }

170  template <typename MatrixReplacement>
171  void updateProjection(const MatrixReplacement &A, ProjectedSpace &proj,
172  int iiter) const {
173 
174  if (iiter == 0 || _davidson_ortho == ORTHO::QR) {
175  /* if we use QR we ned to recompute the entire projection
176  since QR will modify original subspace*/
177  proj.AV = A * proj.V;
178  proj.T = proj.V.transpose() * proj.AV;
179 
180  } else if (_davidson_ortho == ORTHO::GS) {
181  /* if we use a GS ortho we do not have to recompute
182  the entire projection as GS doesn't change the original subspace*/
183  int size = proj.V.rows();
184  int old_dim = proj.T.cols();
185  int new_dim = proj.V.cols();
186  int nvec = new_dim - old_dim;
187  proj.AV.conservativeResize(Eigen::NoChange, new_dim);
188  proj.AV.block(0, old_dim, size, nvec) =
189  A * proj.V.block(0, old_dim, size, nvec);
190  Eigen::MatrixXd VAV =
191  proj.V.transpose() * proj.AV.block(0, old_dim, size, nvec);
192  proj.T.conservativeResize(new_dim, new_dim);
193  proj.T.block(0, old_dim, new_dim, nvec) = VAV;
194  proj.T.block(old_dim, 0, nvec, old_dim) =
195  VAV.topRows(old_dim).transpose();
196  }
197  }

199  template <typename MatrixReplacement>
200  RitzEigenPair getHarmonicRitz(const MatrixReplacement &A,
201  const ProjectedSpace &proj) const {
202 
203  /* Compute the Harmonic Ritz vector following
204  * Computing Interior Eigenvalues of Large Matrices
205  * Ronald B Morgan
206  * LINEAR ALGEBRA AND ITS APPLICATIONS 154-156:289-309 (1991)
207  * https://cpb-us-w2.wpmucdn.com/sites.baylor.edu/dist/e/71/files/2015/05/InterEvals-1vgdz91.pdf
208  */
209 
210  RitzEigenPair rep;
211  Eigen::MatrixXd B = A * proj.AV;
212  B = proj.V.transpose() * B;
213 
214  bool return_eigenvector = true;
215  Eigen::GeneralizedEigenSolver<Eigen::MatrixXd> ges(proj.T, B,
216  return_eigenvector);
217  rep.lambda = ges.eigenvalues().real();
218  rep.U = ges.eigenvectors().real();
219 
220  Eigen::ArrayXi idx = DavidsonSolver::argsort(rep.lambda);
221  idx = idx.reverse();
222 
223  rep.U = DavidsonSolver::extract_vectors(rep.U, idx);
224  rep.U.colwise().normalize();
225  rep.lambda = (rep.U.transpose() * proj.T * rep.U).diagonal();
226 
227  rep.q = proj.V * rep.U; // Ritz vectors
228  rep.res = proj.AV * rep.U - rep.q * rep.lambda.asDiagonal(); // residues
229  rep.res_norm = rep.res.colwise().norm(); // reisdues norms
230 
231  return rep;
232  }
