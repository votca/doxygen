60  template <typename MatrixReplacement>
61  void solve(const MatrixReplacement &A, int neigen,
62  int size_initial_guess = 0) {
63 
64  std::chrono::time_point<std::chrono::system_clock> start =
65  std::chrono::system_clock::now();
66  int op_size = A.rows();
67  PrintOptions(op_size);
68  //. search space exceeding the system size
69  if (_max_search_space > op_size) {
70  XTP_LOG_SAVE(logDEBUG, _log)
71  << TimeStamp() << " Warning Max search space (" << _max_search_space
72  << ") larger than system size (" << op_size << ")" << flush;
73  _max_search_space = op_size;
74  XTP_LOG_SAVE(logDEBUG, _log)
75  << TimeStamp() << " Max search space set to " << op_size << flush;
76  }
77 
78  // initial guess size
79  if (size_initial_guess == 0) {
80  size_initial_guess = 2 * neigen;
81  }
82 
83  int search_space = size_initial_guess;
84  int size_restart = size_initial_guess;
85  int size_update = get_size_update(neigen);
86 
87  std::vector<bool> root_converged = std::vector<bool>(size_update, false);
88 
89  // initialize the guess eigenvector
90  Eigen::VectorXd Adiag = A.diagonal();
91 
92  // target the lowest diagonal element
93  Eigen::MatrixXd V = SetupInitialEigenvectors(Adiag, size_initial_guess);
94  Eigen::MatrixXd q;
95  Eigen::MatrixXd U;
96  Eigen::MatrixXd AV;
97  XTP_LOG_SAVE(logDEBUG, _log)
98  << TimeStamp() << " iter\tSearch Space\tNorm" << flush;
99 
100  // Start of the main iteration loop
101  for (int iiter = 0; iiter < _iter_max; iiter++) {
102 
103  Eigen::MatrixXd T;
104 
105  // check if we need to restart
106  bool restart_required =
107  search_space > _max_search_space || search_space > op_size;
108  if (iiter == 0 || _davidson_ortho == ORTHO::QR) {
109  AV = A * V;
110  T = V.transpose() * AV;
111  } else if (restart_required) {
112  V = q.leftCols(size_restart);
113  V.colwise().normalize();
114  AV = AV * U.leftCols(size_restart); // corresponds to replacing V with
115  // q.leftCols
116  search_space = size_restart;
117  T = V.transpose() * AV;
118  } else if (_davidson_ortho == ORTHO::GS) { // GS orthogonalisation leaves
119  // old eigenvectors untouched,
120  // so we only have to update
121  // the new ones
122  int size = V.rows();
123  int old_dim = T.cols();
124  int new_dim = V.cols();
125  int nvec = new_dim - old_dim;
126  AV.conservativeResize(Eigen::NoChange, new_dim);
127  AV.block(0, old_dim, size, nvec) = A * V.block(0, old_dim, size, nvec);
128  Eigen::MatrixXd VAV = V.transpose() * AV.block(0, old_dim, size, nvec);
129  T.conservativeResize(new_dim, new_dim);
130  T.block(0, old_dim, new_dim, nvec) = VAV;
131  T.block(old_dim, 0, nvec, old_dim) = VAV.topRows(old_dim).transpose();
132  }
133 
134  // diagonalize the small subspace
135  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(T);
136  const Eigen::VectorXd &lambda = es.eigenvalues();
137  U = es.eigenvectors();
138  q = V * U; // Ritz vectors
139  Eigen::MatrixXd r =
140  AV * U - q * lambda.asDiagonal(); // compute residual=A*Q - lambda Q
141 
142  Eigen::ArrayXd res_norm = Eigen::ArrayXd::Zero(size_update);
143  int nupdate = 0;
144  for (int j = 0; j < size_update; j++) {
145  // skip the root that have already converged
146  if (root_converged[j]) {
147  continue;
148  }
149  nupdate++;
150  res_norm[j] = r.col(j).norm();
151  // residue vector
152  Eigen::VectorXd w =
153  ComputeCorrectionVector(Adiag, q.col(j), lambda(j), r.col(j));
154  // append the correction vector to the search space
155  V.conservativeResize(Eigen::NoChange, V.cols() + 1);
156  V.rightCols<1>() = w.normalized();
157  // track converged root
158  root_converged[j] = (res_norm[j] < _tol);
159  }
160 
161  // Print iteration data
162  int converged_roots = 0;
163  for (int i = 0; i < neigen; i++) {
164  converged_roots += root_converged[i];
165  }
166  double percent_converged = 100 * double(converged_roots) / double(neigen);
167  XTP_LOG_SAVE(logDEBUG, _log)
168  << TimeStamp()
169  << format(" %1$4d %2$12d \t %3$4.2e \t %4$5.2f%% converged") % iiter %
170  search_space % res_norm.head(neigen).maxCoeff() %
171  percent_converged
172  << flush;
173 
174  // update
175  search_space = V.cols();
176  bool converged = (res_norm.head(neigen) < _tol).all();
177  if (converged) {
178  _info = Eigen::ComputationInfo::Success;
179  }
180  bool last_iter = iiter == (_iter_max - 1);
181  // break if converged
182  if (converged || last_iter) {
183 
184  // store the eigenvalues/eigenvectors
185  this->_eigenvalues = lambda.head(neigen);
186  this->_eigenvectors = q.leftCols(neigen);
187  this->_eigenvectors.colwise().normalize();
188  if (last_iter && !converged) {
189  XTP_LOG_SAVE(logDEBUG, _log)
190  << TimeStamp() << "- Warning : Davidson " << percent_converged
191  << "% converged after " << _iter_max << " iterations." << flush;
192  _info = Eigen::ComputationInfo::NoConvergence;
193  for (int i = 0; i < neigen; i++) {
194  if (!root_converged[i]) {
195  _eigenvalues(i) = 0;
196  _eigenvectors.col(i) = Eigen::VectorXd::Zero(op_size);
197  }
198  }
199  }
200  break;
201  }
202 
203  switch (this->_davidson_ortho) {
204  case ORTHO::GS:
205  V = DavidsonSolver::gramschmidt_ortho(V, V.cols() - nupdate);
206  break;
207  case ORTHO::QR:
208  V = DavidsonSolver::QR_ortho(V);
209  break;
210  }
211  }
212 
213  PrintTiming(start);
214  }
