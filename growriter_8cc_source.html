37 void GROWriter::Write(Topology *conf)
38 {
39  char format[100];
40  int i,resnr,l,vpr;
41  Topology *top = conf;
42 
43  fprintf (_out,"%s\n","what a nice title");
44  fprintf (_out,"%5d\n",top->BeadCount());
45 
46  bool v = top->HasVel();
47  int pr = 3; // precision of writeout, given by the spec
48 
49  /* build format string for printing,
50  something like "%8.3f" for x and "%8.4f" for v */
51  /*if (pr<0)
52  pr=0;
53  if (pr>30)
54  pr=30;*/
55  l=pr+5;
56  vpr=pr+1;
57  if (v)
58  sprintf(format,"%%%d.%df%%%d.%df%%%d.%df%%%d.%df%%%d.%df%%%d.%df\n",
59  l,pr,l,pr,l,pr,l,vpr,l,vpr,l,vpr);
60  else
61  sprintf(format,"%%%d.%df%%%d.%df%%%d.%df\n",l,pr,l,pr,l,pr);
62 
63  for (i=0; i< top->BeadCount(); i++) {
64  resnr=top->getBead(i)->getResnr();
65  string resname = top->getResidue(resnr)->getName();
66  string atomname = top->getBead(i)->getName();
67 
68  fprintf(_out,"%5d%-5.5s%5.5s%5d",
69  (resnr+1)%100000,resname.c_str(),atomname.c_str(),(i+1)%100000);
70  /* next fprintf uses built format string */
71  vec r = conf->getBead(i)->getPos();
72 
73  if (v) {
74  vec vv = conf->getBead(i)->getVel();
75  fprintf(_out,format,
76  r.getX(), r.getY(), r.getZ(), vv.getX(), vv.getY(), vv.getZ());
77  } else {
78  fprintf(_out,format,
79  r.getX(), r.getY(), r.getZ());
80  }
81  }
82 
83  // write the boy
84  matrix box = conf->getBox();
85 
86  if (pr<5)
87  pr=5;
88  l=pr+5;
89 
90  if (box[0][1] || box[0][2] || box[1][0] || box[1][2] ||
91  box[2][0] || box[2][1]) {
92  sprintf(format,"%%%d.%df%%%d.%df%%%d.%df"
93  "%%%d.%df%%%d.%df%%%d.%df%%%d.%df%%%d.%df%%%d.%df\n",
94  l,pr,l,pr,l,pr,l,pr,l,pr,l,pr,l,pr,l,pr,l,pr);
95  fprintf(_out,format,
96  box[0][0],box[1][1],box[2][2],
97  box[1][0],box[2][0],box[0][1],
98  box[2][1],box[0][2],box[1][2]);
99  } else {
100  sprintf(format,"%%%d.%df%%%d.%df%%%d.%df\n",l,pr,l,pr,l,pr);
101  fprintf(_out,format,
102  box[0][0],box[1][1],box[2][2]);
103  }
104  fflush(_out);
105 }
