66 /*
67  * Fill the 3-center object by looping over shells of GW basis set and
68  * calling FillBlock, which calculates all 3-center overlap integrals
69  * associated to a particular shell, convoluted with the DFT orbital
70  * coefficients
71  */
72 void TCMatrix_gwbse::Fill(const AOBasis& gwbasis, const AOBasis& dftbasis,
73  const Eigen::MatrixXd& dft_orbitals) {
74  // needed for Rebuild())
75  _auxbasis = &gwbasis;
76  _dftbasis = &dftbasis;
77  _dft_orbitals = &dft_orbitals;
78 
79  // If cuda is enabled the dft orbitals are sent first to the cuda gpu
80  // and memory in the cuda gpu is allocated for the intermediate matrices
81 #if defined(USE_CUDA)
82  if (count_available_gpus() > 0) {
83  FillAllBlocksCuda(gwbasis, dftbasis, dft_orbitals);
84  } else {
85  FillAllBlocksOpenMP(gwbasis, dftbasis, dft_orbitals);
86  }
87 #else
88  FillAllBlocksOpenMP(gwbasis, dftbasis, dft_orbitals);
89 #endif
90  AOOverlap auxoverlap;
91  auxoverlap.Fill(gwbasis);
92  AOCoulomb auxcoulomb;
93  auxcoulomb.Fill(gwbasis);
94  Eigen::MatrixXd inv_sqrt = auxcoulomb.Pseudo_InvSqrt_GWBSE(auxoverlap, 5e-7);
95  _removedfunctions = auxcoulomb.Removedfunctions();
96  MultiplyRightWithAuxMatrix(inv_sqrt);
97 
98  return;
99 }

101 /*
102  * Determines the 3-center integrals for a given shell in the GW basis
103  * by calculating the 3-center overlap integral of the functions in the
104  * GW shell with ALL functions in the DFT basis set (FillThreeCenterOLBlock)
105  */
106 std::vector<Eigen::MatrixXd> TCMatrix_gwbse::ComputeSymmStorage(
107  const AOShell& auxshell, const AOBasis& dftbasis,
108  const Eigen::MatrixXd& dft_orbitals) const {
109  std::vector<Eigen::MatrixXd> symmstorage = std::vector<Eigen::MatrixXd>(
110  auxshell.getNumFunc(),
111  Eigen::MatrixXd::Zero(dftbasis.AOBasisSize(), dftbasis.AOBasisSize()));
112  const Eigen::MatrixXd dftm =
113  dft_orbitals.block(0, _mmin, dft_orbitals.rows(), _mtotal);
114  const Eigen::MatrixXd dftn =
115  dft_orbitals.block(0, _nmin, dft_orbitals.rows(), _ntotal);
116  // alpha-loop over the "left" DFT basis function
117  for (Index row = 0; row < dftbasis.getNumofShells(); row++) {
118 
119  const AOShell& shell_row = dftbasis.getShell(row);
120  const Index row_start = shell_row.getStartIndex();
121  // ThreecMatrix is symmetric, restrict explicit calculation to triangular
122  // matrix
123  for (Index col = 0; col <= row; col++) {
124  const AOShell& shell_col = dftbasis.getShell(col);
125  const Index col_start = shell_col.getStartIndex();
126 
127  Eigen::Tensor<double, 3> threec_block(auxshell.getNumFunc(),
128  shell_row.getNumFunc(),
129  shell_col.getNumFunc());
130  threec_block.setZero();
131 
132  bool nonzero =
133  FillThreeCenterRepBlock(threec_block, auxshell, shell_row, shell_col);
134  if (nonzero) {
135  for (Index aux_c = 0; aux_c < auxshell.getNumFunc(); aux_c++) {
136  for (Index row_c = 0; row_c < shell_row.getNumFunc(); row_c++) {
137  for (Index col_c = 0; col_c < shell_col.getNumFunc(); col_c++) {
138  // symmetry
139  if ((col_start + col_c) > (row_start + row_c)) {
140  break;
141  }
142  symmstorage[aux_c](row_start + row_c, col_start + col_c) =
143  threec_block(aux_c, row_c, col_c);
144  } // ROW copy
145  } // COL copy
146  } // AUX copy
147  }
148  } // gamma-loop
149  } // alpha-loop
150 
151  return symmstorage;
152 }

154 /*
155  * Convolution of the GW shell with ALL functions with the DFT orbital
156  * coefficients
157  */
158 std::vector<Eigen::MatrixXd> TCMatrix_gwbse::FillBlock(
159  const std::vector<Eigen::MatrixXd>& symmstorage,
160  const Eigen::MatrixXd& dft_orbitals) const {
161 
162  std::vector<Eigen::MatrixXd> block = std::vector<Eigen::MatrixXd>(
163  _mtotal, Eigen::MatrixXd::Zero(_ntotal, symmstorage.size()));
164  const Eigen::MatrixXd dftm =
165  dft_orbitals.block(0, _mmin, dft_orbitals.rows(), _mtotal);
166  const Eigen::MatrixXd dftn =
167  dft_orbitals.block(0, _nmin, dft_orbitals.rows(), _ntotal);
168 
169  Index dim = static_cast<Index>(symmstorage.size());
170  for (Index k = 0; k < dim; ++k) {
171  const Eigen::MatrixXd& matrix = symmstorage[k];
172  Eigen::MatrixXd threec_inMo =
173  dftn.transpose() * matrix.selfadjointView<Eigen::Lower>() * dftm;
174  for (Index i = 0; i < threec_inMo.cols(); ++i) {
175  block[i].col(k) = threec_inMo.col(i);
176  }
177  }
178  return block;
179 }

197 void TCMatrix_gwbse::FillAllBlocksOpenMP(const AOBasis& gwbasis,
198  const AOBasis& dftbasis,
199  const Eigen::MatrixXd& dft_orbitals) {
200 #pragma omp parallel for schedule(guided) // private(_block)
201  for (Index is = 0; is < gwbasis.getNumofShells(); is++) {
202  const AOShell& shell = gwbasis.getShell(is);
203 
204  // Fill block for this shell (3-center overlap with _dft_basis +
205  // multiplication with _dft_orbitals )
206  std::vector<Eigen::MatrixXd> symmstorage =
207  ComputeSymmStorage(shell, dftbasis, dft_orbitals);
208 
209  std::vector<Eigen::MatrixXd> block = FillBlock(symmstorage, dft_orbitals);
210 
211  // put into correct position
212  for (Index m_level = 0; m_level < _mtotal; m_level++) {
213  _matrix[m_level].block(0, shell.getStartIndex(), _ntotal,
214  shell.getNumFunc()) = block[m_level];
215  } // m-th DFT orbital
216  } // shells of GW basis set
217 }

219 #if defined(USE_CUDA)
220 /*
221  * The Cuda device behaves like a server that is receiving matrix-matrix
222  * multiplications from a single stream (an Nvidia queue) and handle them
223  * in an asynchronous way. It performs the following operations when recieving a
224  * request:
225  * 1. Check that there is enough space for the arrays
226  * 2. Allocate memory for each matrix
227  * 3. Copy the matrix to the allocated space
228  * 4. Perform the matrix multiplication
229  * 5. Return the result matrix
230  * The Cuda device knows to which memory address it needs to copy back the
231  * result. see: https://docs.nvidia.com/cuda/cublas/index.html#thread-safety2
232  */

234 /*
235  * Modify 3-center matrix elements consistent with use of symmetrized
236  * Coulomb interaction using OPENMP/CUDA parallelization
237  */
238 void TCMatrix_gwbse::MultiplyRightWithAuxMatrixCuda(
239  const Eigen::MatrixXd& matrix) {
240 
241  XTP_LOG(Log::info, _log)
242  << TimeStamp() << " Using CUDA/OpenMP for tensor matrix multiplication"
243  << flush;
244 
245  CudaPipeline cuda_pip;
246  const Eigen::MatrixXd& head = _matrix.front();
247  const cudaStream_t& stream = cuda_pip.get_stream();
248  CudaMatrix cuma_A{head.rows(), head.cols(), stream};
249  CudaMatrix cuma_B{matrix, stream};
250  CudaMatrix cuma_C{head.rows(), matrix.cols(), stream};
251 
252 #pragma omp parallel for schedule(dynamic)
253  for (Index i_occ = 0; i_occ < _mtotal; i_occ++) {
254  // All the GPU communication happens through a single thread that reuses all
255  // memory allocated in the GPU and it's dynamically load-balanced by OpenMP.
256  // The rest of the threads use the default CPU matrix multiplication
257  if (OPENMP::getThreadId() == 0) {
258  cuma_A.copy_to_gpu(_matrix[i_occ]);
259  cuda_pip.gemm(cuma_A, cuma_B, cuma_C);
260  _matrix[i_occ] = cuma_C;
261  } else {
262  _matrix[i_occ] *= matrix;
263  }
264  }
265 }

267 /*
268  * Convolution of the GW shell with the DFT orbital coefficients using an Nvidia
269  * GPU
270  */
271 void TCMatrix_gwbse::FillAllBlocksCuda(const AOBasis& gwbasis,
272  const AOBasis& dftbasis,
273  const Eigen::MatrixXd& dft_orbitals) {
274  CudaPipeline cuda_pip;
275  std::array<CudaMatrix, 2> cuda_matrices =
276  SendDFTMatricesToGPU(dft_orbitals, cuda_pip);
277  std::array<CudaMatrix, 3> cuda_inter_matrices =
278  CreateIntermediateCudaMatrices(dft_orbitals.rows(), cuda_pip);
279 
280  // loop over all shells in the GW basis and get _Mmn for that shell
281 #pragma omp parallel for schedule(dynamic) // private(_block)
282  for (Index is = 0; is < gwbasis.getNumofShells(); is++) {
283  const AOShell& shell = gwbasis.getShell(is);
284 
285  // Fill block for this shell (3-center overlap with _dft_basis +
286  // multiplication with _dft_orbitals )
287  std::vector<Eigen::MatrixXd> symmstorage =
288  ComputeSymmStorage(shell, dftbasis, dft_orbitals);
289 
290  // If cuda is enable all the GPU communication happens through a single
291  // thread that reuses all memory allocated in the GPU and it's dynamically
292  // load-balanced by OpenMP. The remaining threads will perform the
293  // convolution using the default CPU method.
294  std::vector<Eigen::MatrixXd> block;
295  if (OPENMP::getThreadId() == 0) {
296  block = FillBlockCUDA(symmstorage, cuda_matrices, cuda_inter_matrices,
297  cuda_pip);
298  } else {
299  block = FillBlock(symmstorage, dft_orbitals);
300  }
301  // // Otherwise the convolution is performed by Eigen
302  // block = FillBlock(symmstorage, dft_orbitals);
303  // put into correct position
304  for (Index m_level = 0; m_level < _mtotal; m_level++) {
305  _matrix[m_level].block(0, shell.getStartIndex(), _ntotal,
306  shell.getNumFunc()) = block[m_level];
307  } // m-th DFT orbital
308  } // shells of GW basis set
309 }

311 std::vector<Eigen::MatrixXd> TCMatrix_gwbse::FillBlockCUDA(
312  const std::vector<Eigen::MatrixXd>& symmstorage,
313  const std::array<CudaMatrix, 2>& cuda_matrices,
314  std::array<CudaMatrix, 3>& cuda_inter_matrices,
315  const CudaPipeline& cuda_pip) const {
316 
317  std::vector<Eigen::MatrixXd> block = std::vector<Eigen::MatrixXd>(
318  _mtotal, Eigen::MatrixXd::Zero(_ntotal, symmstorage.size()));
319 
320  try {
321  const CudaMatrix& cuma_A = cuda_matrices[0];
322  const CudaMatrix& cuma_C = cuda_matrices[1];
323  CudaMatrix& cuma_B = cuda_inter_matrices[0];
324  CudaMatrix& cuma_X = cuda_inter_matrices[1];
325  CudaMatrix& cuma_Y = cuda_inter_matrices[2];
326 
327  Index dim = static_cast<Index>(symmstorage.size());
328  for (Index k = 0; k < dim; ++k) {
329  const Eigen::MatrixXd& matrix = symmstorage[k];
330  cuma_B.copy_to_gpu(matrix.selfadjointView<Eigen::Lower>());
331  cuda_pip.gemm(cuma_A, cuma_B, cuma_X);
332  cuda_pip.gemm(cuma_X, cuma_C, cuma_Y);
333  Eigen::MatrixXd threec_inMo = cuma_Y;
334  for (Index i = 0; i < threec_inMo.cols(); ++i) {
335  block[i].col(k) = threec_inMo.col(i);
336  }
337  }
338  } catch (const std::runtime_error& error) {
339  XTP_LOG(Log::error, _log)
340  << TimeStamp() << " FillBlockCUDA failed due to: " << error.what()
341  << flush;
342  throw;
343  }
344  return block;
345 }

348  const Eigen::MatrixXd& dft_orbitals, const CudaPipeline& cuda_pip) const {
349  const Eigen::MatrixXd dftm =
