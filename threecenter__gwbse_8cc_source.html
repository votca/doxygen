84 /*
85  * Determines the 3-center integrals for a given shell in the GW basis
86  * by calculating the 3-center overlap integral of the functions in the
87  * GW shell with ALL functions in the DFT basis set (FillThreeCenterOLBlock)
88  */
89 std::vector<Eigen::MatrixXd> TCMatrix_gwbse::ComputeAO3cBlock(
90  const AOShell& auxshell, const AOBasis& dftbasis) const {
91  std::vector<Eigen::MatrixXd> ao3c = std::vector<Eigen::MatrixXd>(
92  auxshell.getNumFunc(),
93  Eigen::MatrixXd::Zero(dftbasis.AOBasisSize(), dftbasis.AOBasisSize()));
94  // alpha-loop over the "left" DFT basis function
95  for (Index row = 0; row < dftbasis.getNumofShells(); row++) {
96 
97  const AOShell& shell_row = dftbasis.getShell(row);
98  const Index row_start = shell_row.getStartIndex();
99  // ThreecMatrix is symmetric, restrict explicit calculation to triangular
100  // matrix
101  for (Index col = 0; col <= row; col++) {
102  const AOShell& shell_col = dftbasis.getShell(col);
103  const Index col_start = shell_col.getStartIndex();
104 
105  Eigen::Tensor<double, 3> threec_block(auxshell.getNumFunc(),
106  shell_row.getNumFunc(),
107  shell_col.getNumFunc());
108  threec_block.setZero();
109 
110  bool nonzero =
111  FillThreeCenterRepBlock(threec_block, auxshell, shell_row, shell_col);
112  if (nonzero) {
113  for (Index aux_c = 0; aux_c < auxshell.getNumFunc(); aux_c++) {
114  for (Index row_c = 0; row_c < shell_row.getNumFunc(); row_c++) {
115  for (Index col_c = 0; col_c < shell_col.getNumFunc(); col_c++) {
116  ao3c[aux_c](row_start + row_c, col_start + col_c) =
117  threec_block(aux_c, row_c, col_c);
118  } // ROW copy
119  } // COL copy
120  } // AUX copy
121  }
122  } // gamma-loop
123  } // alpha-loop
124 
125  for (Eigen::MatrixXd& mat : ao3c) {
126  mat.triangularView<Eigen::Upper>() =
127  mat.triangularView<Eigen::Lower>().transpose();
128  }
129  return ao3c;
130 }

132 void TCMatrix_gwbse::Fill3cMO(const AOBasis& gwbasis, const AOBasis& dftbasis,
133  const Eigen::MatrixXd& dft_orbitals) {
134 
135  const Eigen::MatrixXd dftm =
136  dft_orbitals.block(0, _mmin, dft_orbitals.rows(), _mtotal);
137  const Eigen::MatrixXd dftn =
138  dft_orbitals.block(0, _nmin, dft_orbitals.rows(), _ntotal).transpose();
139 
140  OpenMP_CUDA transform;
141  transform.setOperators(dftn, dftm);
142 #pragma omp parallel for schedule(guided)
143  for (Index is = 0; is < gwbasis.getNumofShells(); is++) {
144  const AOShell& shell = gwbasis.getShell(is);
145 
146  std::vector<Eigen::MatrixXd> ao3c = ComputeAO3cBlock(shell, dftbasis);
147 
148  // this is basically a transpose of AO3c and at the same time the ao->mo
149  // transformation
150  // we do not want to put it into _matrix straight away is because, _matrix
151  // is shared between all threads and we want a nice clean access pattern to
152  // it
153  std::vector<Eigen::MatrixXd> block = std::vector<Eigen::MatrixXd>(
154  _mtotal, Eigen::MatrixXd::Zero(_ntotal, ao3c.size()));
155 
156  Index dim = static_cast<Index>(ao3c.size());
157  for (Index k = 0; k < dim; ++k) {
158  transform.MultiplyLeftRight(ao3c[k]);
159  for (Index i = 0; i < ao3c[k].cols(); ++i) {
160  block[i].col(k) = ao3c[k].col(i);
161  }
162  }
163 
164  // put into correct position
165  for (Index m_level = 0; m_level < _mtotal; m_level++) {
166  _matrix[m_level].block(0, shell.getStartIndex(), _ntotal,
167  shell.getNumFunc()) = block[m_level];
168  } // m-th DFT orbital
169  } // shells of GW basis set
170 }
