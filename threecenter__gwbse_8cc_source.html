48

57 /*
58  * Fill the 3-center object by looping over shells of GW basis set and
59  * calling FillBlock, which calculates all 3-center overlap integrals
60  * associated to a particular shell, convoluted with the DFT orbital
61  * coefficients
62  */
63 void TCMatrix_gwbse::Fill(const AOBasis& gwbasis, const AOBasis& dftbasis,
64  const Eigen::MatrixXd& dft_orbitals) {
65  // needed for Rebuild())
66  _auxbasis = &gwbasis;
67  _dftbasis = &dftbasis;
68  _dft_orbitals = &dft_orbitals;
69 
70  // loop over all shells in the GW basis and get _Mmn for that shell
71 #pragma omp parallel for schedule(guided) // private(_block)
72  for (int is = 0; is < gwbasis.getNumofShells(); is++) {
73  const AOShell& shell = gwbasis.getShell(is);
74  std::vector<Eigen::MatrixXd> block;
75  for (int i = 0; i < _mtotal; i++) {
76  block.push_back(Eigen::MatrixXd::Zero(_ntotal, shell.getNumFunc()));
77  }
78  // Fill block for this shell (3-center overlap with _dft_basis +
79  // multiplication with _dft_orbitals )
80  FillBlock(block, shell, dftbasis, dft_orbitals);
81 
82  // put into correct position
83  for (int m_level = 0; m_level < _mtotal; m_level++) {
84  _matrix[m_level].block(0, shell.getStartIndex(), _ntotal,
85  shell.getNumFunc()) = block[m_level];
86  } // m-th DFT orbital
87  } // shells of GW basis set
88 
89  AOOverlap auxoverlap;
90  auxoverlap.Fill(gwbasis);
91  AOCoulomb auxcoulomb;
92  auxcoulomb.Fill(gwbasis);
93  Eigen::MatrixXd inv_sqrt = auxcoulomb.Pseudo_InvSqrt_GWBSE(auxoverlap, 5e-7);
94  _removedfunctions = auxcoulomb.Removedfunctions();
95  MultiplyRightWithAuxMatrix(inv_sqrt);
96  return;
97 }

100  * Determines the 3-center integrals for a given shell in the GW basis
101  * by calculating the 3-center overlap integral of the functions in the
102  * GW shell with ALL functions in the DFT basis set (FillThreeCenterOLBlock),
103  * followed by a convolution of those with the DFT orbital coefficients
104  */

106 void TCMatrix_gwbse::FillBlock(std::vector<Eigen::MatrixXd>& block,
107  const AOShell& auxshell, const AOBasis& dftbasis,
108  const Eigen::MatrixXd& dft_orbitals) {
109  std::vector<Eigen::MatrixXd> symmstorage;
110  for (int i = 0; i < auxshell.getNumFunc(); ++i) {
111  symmstorage.push_back(
112  Eigen::MatrixXd::Zero(dftbasis.AOBasisSize(), dftbasis.AOBasisSize()));
113  }
114  const Eigen::MatrixXd dftm =
115  dft_orbitals.block(0, _mmin, dft_orbitals.rows(), _mtotal);
116  const Eigen::MatrixXd dftn =
117  dft_orbitals.block(0, _nmin, dft_orbitals.rows(), _ntotal);
118  // alpha-loop over the "left" DFT basis function
119  for (int row = 0; row < dftbasis.getNumofShells(); row++) {
120 
121  const AOShell& shell_row = dftbasis.getShell(row);
122  const int row_start = shell_row.getStartIndex();
123  // ThreecMatrix is symmetric, restrict explicit calculation to triangular
124  // matrix
125  for (int col = 0; col <= row; col++) {
126  const AOShell& shell_col = dftbasis.getShell(col);
127  const int col_start = shell_col.getStartIndex();
128 
129  Eigen::Tensor<double, 3> threec_block(auxshell.getNumFunc(),
130  shell_row.getNumFunc(),
131  shell_col.getNumFunc());
132  threec_block.setZero();
133 
134  bool nonzero =
135  FillThreeCenterRepBlock(threec_block, auxshell, shell_row, shell_col);
136  if (nonzero) {
137  for (int aux_c = 0; aux_c < auxshell.getNumFunc(); aux_c++) {
138  for (int row_c = 0; row_c < shell_row.getNumFunc(); row_c++) {
139  for (int col_c = 0; col_c < shell_col.getNumFunc(); col_c++) {
140  // symmetry
141  if ((col_start + col_c) > (row_start + row_c)) {
142  break;
143  }
144  symmstorage[aux_c](row_start + row_c, col_start + col_c) =
145  threec_block(aux_c, row_c, col_c);
146  } // ROW copy
147  } // COL copy
148  } // AUX copy
149  }
150  } // gamma-loop
151  } // alpha-loop
152  for (int k = 0; k < auxshell.getNumFunc(); ++k) {
153  const Eigen::MatrixXd& matrix = symmstorage[k];
154  Eigen::MatrixXd threec_inMo =
155  dftn.transpose() * matrix.selfadjointView<Eigen::Lower>() * dftm;
156  for (int i = 0; i < threec_inMo.cols(); ++i) {
157  block[i].col(k) = threec_inMo.col(i);
158  }
159  }
160  return;
161 }
