87 double Event::Compute_event_rate(Node* fromnode, int jump_ID, CarrierType carrier_type,
88  From_step_event from_event_type, To_step_event to_event_type,
89  double from_shortrange, double to_shortrange, double from_longrange, double to_longrange,
90  Globaleventinfo* globevent){
91 
92  Node* jumptonode = fromnode->pairing_nodes[jump_ID];
93 
94  double prefactor = 1.0;
95  double charge;
96  double static_node_energy_from;
97  double static_node_energy_to;
98 
99  if(carrier_type == Electron) {
100  charge = 1.0;
101  prefactor *= globevent->electron_prefactor;
102  static_node_energy_from = fromnode->static_electron_node_energy;
103  static_node_energy_to = jumptonode->static_electron_node_energy;
104  }
105  else {
106  charge = -1.0;
107  prefactor *= globevent->hole_prefactor;
108  static_node_energy_from = fromnode->static_hole_node_energy;
109  static_node_energy_to = jumptonode->static_hole_node_energy;
110  }
111 
112  //first calculate quantum mechanical wavefunction overlap
113  myvec distancevector = fromnode->static_event_info[jump_ID].distance;
114  double distance = abs(distancevector);
115 
116  double distancefactor = exp(-1.0*globevent->alpha*distance);
117 
118  //second, calculate boltzmann factor (Coulomb interaction still have to be implemented)
119 
120  double init_energy;
121  double final_energy;
122  double selfimpot_from = fromnode->self_image_potential;
123  double selfimpot_to = jumptonode->self_image_potential;
124  double fromtype_energy = 0.0;
125  double totype_energy = 0.0;
126 
127  if(from_event_type == Injection) {
128  fromtype_energy -= globevent->injection_barrier;
129  prefactor *= globevent->injection_prefactor;
130  }
131  if(to_event_type == Recombination) {
132  totype_energy -= globevent->binding_energy;
133  prefactor *= globevent->recombination_prefactor;
134  }
135  if(to_event_type == Collection) {
136  totype_energy -= globevent->injection_barrier;
137  prefactor *= globevent->collection_prefactor;
138  }
139 
140  double coulomb_from;
141  double coulomb_to;
142 
143  if ((from_event_type != Injection)) {
144  coulomb_from = globevent->coulomb_strength*(from_shortrange+charge*from_longrange);
145  coulomb_to = globevent->coulomb_strength*(to_shortrange+charge*to_longrange);
146  }
147  else {// if(from_event_type == Injection){
148  coulomb_from = 0.0;
149  coulomb_to = charge*globevent->coulomb_strength*(jumptonode->injection_potential+to_longrange);
150  }
151 
152  init_energy = static_node_energy_from + selfimpot_from + coulomb_from;
153  final_energy = static_node_energy_to + selfimpot_to + coulomb_to;
154 
155  double energycontrib;
156  double energyfactor;
157 
158  if (globevent->formalism == "Miller") {
159  if(to_event_type == Blocking) {
160  energyfactor = 0.0; // Keep this here for eventual simulation of bipolaron formation for example
161  }
162  else if((from_event_type == Fromnotinbox)&&(to_event_type == Tonotinbox)) {
163  energyfactor = 0.0; // Keep this here for eventual simulation of one-site events (on-node generation)
164  }
165  else {
166  energycontrib = final_energy - init_energy -charge*globevent->efield*distancevector.x();
167  if (energycontrib>0.0) {
168  energyfactor = exp(-1.0*globevent->beta*energycontrib);
169  }
170  else {
171  energyfactor = 1.0;
172  }
173  }
174  } else {
175  throw runtime_error("for globevent->formalism != Miller, energyfactor is undefined");
176  }
177 
178  double jump_rate = prefactor*distancefactor*energyfactor;
179  return jump_rate;
180 }

197 To_step_event Event::Determine_non_injection_to_event_type(Carrier* carrier, int jumpID, Node* carriernode){
198 
199  To_step_event to_type;
200 
201  if(!carrier->is_in_sim_box){
202  to_type = Tonotinbox;
203  }
204  else {
205  int node_degree = carriernode->pairing_nodes.size();
206  if(jumpID < node_degree) { // hopping event exists in graph
207  Node* jumpnode = carriernode->pairing_nodes[jumpID];
208  if(jumpnode->carriers_on_node.empty()){
209  to_type = Totransfer;
210  }
211  else if(jumpnode->carriers_on_node[0]->carrier_type == carrier->carrier_type) {
212  to_type = Blocking;
213  }
214  else{// if(jumpnode->carriers_on_node[0]->carrier_type != carrier->carrier_type) {
215  to_type = Recombination;
216  }
217  }
218  else {
219  to_type = Tonotinbox;
220  }
221  }
222 
223  return to_type;
