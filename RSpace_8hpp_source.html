135  const std::vector<T>& _d, const std::vector<T>& _Q) {

137  namespace kl3 = kokkos_linalg_3d;
138 
139  // get number of particles
140  size_t N = _xyz.size() / 3;
141 
142  // store C++ vectors to Kokkos Views
143  Kokkos::View<T * [3]> xyz("positions", N);
144  Kokkos::View<T*> q("charges", N);
145  Kokkos::View<T * [3]> d("dipole moments", N);
146  Kokkos::View<T * [9]> Q("quadrupole moments", N);
147 
148  Kokkos::View<T*> pe("potential energy", N);
149 
150  Kokkos::parallel_for(N, KOKKOS_LAMBDA(const int n) {
151  pe(n) = 0.0;
152  q(n) = _q.at(n);
153  for (int i = 0; i < 3; ++i) {
154  xyz(n, i) = _xyz.at(3 * n + i);
155  d(n, i) = _d.at(3 * n + i);
156  }
157  for (int i = 0; i < 9; ++i) {
158  Q(n, i) = _Q.at(9 * n + i);
159  }
160  });
161 
162  // create Kokkos views of sufficient size to store the factors
163  f = Kokkos::View<T * [3]>("r-space force contribution", N);
164  tqe = Kokkos::View<T * [3]>("r-space torque contribution", N);
165 
166  Kokkos::parallel_for(N, KOKKOS_LAMBDA(const int n) {
167  for (int i = 0; i < 3; ++i) {
168  f(n, i) = 0.0;
169  tqe(n, i) = 0.0;
170  }
171  });
172 
173  pot_energy = 0.0;
174  virial = 0.0;
175  Kokkos::parallel_for(N, KOKKOS_LAMBDA(const int i) {
176  T thread_Virial = 0.0;
177  T thread_Potential = 0.0;
178 
179  std::array<T, 3> force_i = {0, 0, 0};
180  std::array<T, 3> torque_i = {0, 0, 0};
181  for (int j = i + 1; j < N; ++j) {
182  std::array<T, 3> dR;
183  for (int d = 0; d < 3; d++) {
184  dR[d] = xyz(i, d) - xyz(j, d);
185  dR[d] -= std::round(dR[d] / l) * l;
186  }
187  T dist2 = dR[0] * dR[0] + dR[1] * dR[1] + dR[2] * dR[2];
188  // if (dist2 > r_max * r_max) continue;
189  T inv_dist = 1.0 / std::sqrt(dist2);
190 
191  // eqn 69-71 in Smith Point Multipoles in Ewald Summation(Revisited)
192  std::array<T, 6> Bn;
193  Bn[0] = inv_dist * std::erfc(alpha / inv_dist);
194  T expfactor = std::exp(-alpha * alpha * dist2);
195  for (int b = 1; b < 6; b++) {
196  Bn[b] = inv_dist * inv_dist *
197  ((T)(2 * b - 1) * Bn[b - 1] + std::pow(2 * alpha * alpha, b) /
198  (alpha * std::sqrt(M_PI)) *
199  expfactor);
200  }
201 
202  auto Di = Kokkos::subview(d, i, Kokkos::ALL());
203  auto Dj = Kokkos::subview(d, j, Kokkos::ALL());
204  auto Qi = Kokkos::subview(Q, i, Kokkos::ALL());
205  auto Qj = Kokkos::subview(Q, j, Kokkos::ALL());
206 
207  std::array<T, 3> dixdj = kl3::cross(Di, Dj);
208  std::array<T, 3> dixR = kl3::cross(Di, dR);
209  std::array<T, 3> djxR = kl3::cross(Dj, dR);
210 
211  std::array<T, 3> QIR = kl3::gemv(Qi, dR);
212  std::array<T, 3> QJR = kl3::gemv(Qj, dR);
213  std::array<T, 3> QIQJR = kl3::gemv(Qi, QJR);
214  std::array<T, 3> QJQIR = kl3::gemv(Qj, QIR);
215  std::array<T, 3> QIXQJ = kl3::cross_matrix_product(Qi, Qj);
216 
217  std::array<T, 3> RxQIR = kl3::cross(dR, QIR);
218  std::array<T, 3> RxQJR = kl3::cross(dR, QJR);
219 
220  std::array<T, 3> RxQIJR = kl3::cross(dR, QIQJR);
221  std::array<T, 3> RxQJIR = kl3::cross(dR, QJQIR);
222 
223  std::array<T, 3> QJRXQIR = kl3::cross(QJR, QIR);
224 
225  std::array<T, 3> QIDJ = kl3::gemv(Qi, Dj);
226  std::array<T, 3> QJDI = kl3::gemv(Qj, Di);
227 
228  std::array<T, 3> DIXQJR = kl3::cross(Di, QJR);
229  std::array<T, 3> DJXQIR = kl3::cross(Dj, QIR);
230 
231  std::array<T, 3> RXQIDJ = kl3::cross(dR, QIDJ);
232 
233  std::array<T, 3> RXQJDI = kl3::cross(dR, QJDI);
234 
235  T QII = kl3::trace(Qi);
236  T QJJ = kl3::trace(Qj); // sc1
237 
238  T DD = kl3::dot(Di, Dj); // sc2
239  T DIR = kl3::dot(Di, dR); // sc3
240  T DJR = kl3::dot(Dj, dR); // sc4
241  T QIRR = kl3::dot(QIR, dR); // sc5
242  T QJRR = kl3::dot(QJR, dR); // sc6
243  T QIRDJ = kl3::dot(QIR, Dj); // sc7
244  T QJRDI = kl3::dot(QJR, Di); // sc8
245  T QJRQIR = kl3::dot(QJR, QIR); // sc9
246  T QMIQMJ = kl3::dot(Qi, Qj); // sc10
247 
248  // eqn 38-42
249  std::array<T, 9> GL;
250  GL[0] = q(i) * q(j);
251  GL[1] = q(j) * DIR - q(i) * DJR;
252  GL[2] = q(i) * QJRR + q(j) * QIRR - DIR * DJR;
253  GL[3] = DIR * QJRR - DJR * QIRR;
254  GL[4] = QIRR * QJRR;
255  GL[5] = -(QJJ * QIRR + QJRR * QII + 4 * QJRQIR);
256  GL[6] = DD - q(j) * QII - q(i) * QJJ;
257  GL[7] = DJR * QII - QJJ * DIR + 2 * (QIRDJ - QJRDI);
258  GL[8] = 2 * QMIQMJ + QII * QJJ;
259 
260  thread_Potential = Bn[0] * GL[0] + Bn[1] * (GL[1] + GL[6]) +
261  Bn[2] * (GL[2] + GL[7] + GL[8]) +
262  Bn[3] * (GL[3] + GL[5]) + Bn[4] * GL[4];
263 
264 #ifdef DEBUG_OUTPUT_ENABLED
265  std::cout << "r-space DEBUG: " << i << " <> " << j << " " << pot_energy
266  << " { " << xyz(i, 0) << ", " << xyz(i, 1) << ", " << xyz(i, 2)
267  << " } "
268  << " { " << xyz(j, 0) << ", " << xyz(j, 1) << ", " << xyz(j, 2)
269  << " } "
270  << " + " << thread_Potential << " " << dist2 << " "
271  << " " << (Bn[0] * GL[0]) << " "
272  << " " << Bn[0] << " "
273  << " " << Bn[1] << " "
274  << " " << Bn[2] << " "
275  << " " << Bn[3] << " "
276  << " " << Bn[4] << " "
277  << " | "
278  << " " << QII << " "
279  << " " << QJJ << " "
280  << " " << GL[0] << " "
281  << " " << GL[1] << " "
282  << " " << GL[2] << " "
283  << " " << GL[3] << " "
284  << " " << GL[4] << " "
285  << " " << GL[5] << " "
286  << " " << GL[6] << " "
287  << " " << GL[7] << " " << std::endl;
288 #endif
289 
290  Kokkos::atomic_add(&pe(i), 0.5 * thread_Potential);
291  Kokkos::atomic_add(&pe(j), 0.5 * thread_Potential);
292  Kokkos::atomic_add(&pot_energy, thread_Potential);
293 
294  thread_Virial += Bn[0] * GL[0] + Bn[1] * (2 * GL[1] + 3 * GL[6]) +
295  Bn[2] * (3 * GL[2] + 4 * GL[7] + 5 * GL[8]) +
296  Bn[3] * (4 * GL[3] + 5 * GL[5]) + Bn[4] * 5 * GL[4];
297 
298  std::array<T, 7> F_M_coeff;
299 
300  F_M_coeff[0] = Bn[1] * GL[0] + Bn[2] * (GL[1] + GL[6]) +
301  Bn[3] * (GL[2] + GL[7] + GL[8]) + Bn[4] * (GL[3] + GL[5]) +
302  Bn[5] * GL[4];
303  F_M_coeff[1] = -q(j) * Bn[1] + (DJR + QJJ) * Bn[2] - QJRR * Bn[3];
304  F_M_coeff[2] = q(i) * Bn[1] + (DIR - QII) * Bn[2] + QIRR * Bn[3];
305  F_M_coeff[3] = 2 * Bn[2];
306  F_M_coeff[4] = 2 * (-q(j) * Bn[2] + (QJJ + DJR) * Bn[3] - QJRR * Bn[4]);
307  F_M_coeff[5] = 2 * (-q(i) * Bn[2] + (QII - DIR) * Bn[3] - QIRR * Bn[4]);
308  F_M_coeff[6] = 4 * Bn[3];
309 
310  std::array<T, 3> thread_force = {0, 0, 0};
311  kl3::add_to(thread_force, kl3::scale_3d(F_M_coeff[0], dR));
312  kl3::add_to(thread_force, kl3::scale_3d(F_M_coeff[1], Di));
313  kl3::add_to(thread_force, kl3::scale_3d(F_M_coeff[2], Dj));
314  kl3::add_to(thread_force,
315  kl3::scale_3d(F_M_coeff[3], kl3::subtract(QJDI, QIDJ)));
316  kl3::add_to(thread_force, kl3::scale_3d(F_M_coeff[4], QIR));
317  kl3::add_to(thread_force, kl3::scale_3d(F_M_coeff[5], QJR));
318  kl3::add_to(thread_force,
319  kl3::scale_3d(F_M_coeff[6], kl3::add(QIQJR, QJQIR)));
320 
321  std::array<T, 3> thread_M_i = {0, 0, 0};
322 
323  kl3::subtract_from(thread_M_i, kl3::scale_3d(Bn[1], dixdj));
324  kl3::add_to(thread_M_i, kl3::scale_3d(F_M_coeff[1], dixR));
325  std::array<T, 3> temp_sum = kl3::add(DIXQJR, DJXQIR);
326  kl3::add_to(temp_sum, RXQIDJ);
327  kl3::subtract_from(temp_sum, kl3::scale_3d(2, QIXQJ));
328  kl3::add_to(thread_M_i, kl3::scale_3d(F_M_coeff[3], temp_sum));
329  kl3::subtract_from(thread_M_i, kl3::scale_3d(F_M_coeff[4], RxQIR));
330  kl3::subtract_from(
331  thread_M_i, kl3::scale_3d(F_M_coeff[6], kl3::add(RxQIJR, QJRXQIR)));
332 
333  std::array<T, 3> thread_M_j = {0, 0, 0};
334 
335  kl3::add_to(thread_M_j, kl3::scale_3d(Bn[1], dixdj));
336  kl3::add_to(thread_M_j, kl3::scale_3d(F_M_coeff[2], djxR));
337  temp_sum = kl3::add(DIXQJR, DJXQIR);
338  kl3::add_to(temp_sum, RXQJDI);
339  kl3::subtract_from(temp_sum, kl3::scale_3d(2, QIXQJ));
340  kl3::subtract_from(thread_M_j, kl3::scale_3d(F_M_coeff[3], temp_sum));
341  kl3::subtract_from(thread_M_j, kl3::scale_3d(F_M_coeff[5], RxQJR));
342  kl3::subtract_from(
343  thread_M_j,
344  kl3::scale_3d(F_M_coeff[6], kl3::subtract(RxQIJR, QJRXQIR)));
345 
346  kl3::subtract_from(force_i, thread_force);
347  kl3::add_to(torque_i, thread_M_i);
348 
349  for (int k = 0; k < 3; k++) {
350  Kokkos::atomic_add(&f(j, k), thread_force[k]);
351  Kokkos::atomic_add(&tqe(j, k), thread_M_j[k]);
352  }
353  }
354 
355  for (int k = 0; k < 3; k++) {
356  Kokkos::atomic_add(&f(i, k), force_i[k]);
357  Kokkos::atomic_add(&tqe(i, k), torque_i[k]);
358  }
359 
360  Kokkos::atomic_add(&virial, thread_Virial);
361  });
362 
363  // self energy correction
364  T U_self = 0.0;
365  Kokkos::parallel_reduce(N,
366  KOKKOS_LAMBDA(const int i, T& self_tmp) {
367  T tmp = -q(i) * q(i) * alpha / std::sqrt(M_PI);
368  pe(i) += tmp;
369  self_tmp += tmp;
370  },
371  U_self);
372 
373  pot_energy += U_self;
374 #ifdef QDMEWALD_DEBUG_ENABLED
375  std::cout << "DEBUG: r-space energy particle 0: " << pe(0) << " "
376  << pot_energy << std::endl;
377  std::cout << "DEBUG: r-space energy particle N/2: " << pe(N / 2) << " "
378  << pot_energy << std::endl;

380  << pot_energy << std::endl;
