83  template<class in_t>
84  void AddAtomicInput(vector<in_t> &inputs) {
85  // Apportion the vector elements equally onto all threads
86  typename vector<in_t>::iterator iit;
87  int in_idx = 0;
88  int th_idx = 0;
89  for (iit = inputs.begin(); iit != inputs.end(); ++iit, ++in_idx) {
90  th_idx = in_idx % this->size();
91  (*this)[th_idx]->AddAtomicInput(*iit);
92  }
93  // The thread with the last input can be verbose
94  for (it_t tfit = this->begin(); tfit < this->end(); ++tfit)
95  (*tfit)->setVerbose(false);
96  (*this)[th_idx]->setVerbose(true);
97  return;
98  }
