32 
33  tools::Property options =

40  ".mpsinput", _job_name + ".mps");
41 
42  _input.LoadFromFile(mps_input);

44  ".mpsoutput", _job_name + "_polar.mps");

48  bool qmpackage_exists = options.exists(".qmpackage");
49  if (target_exists && qmpackage_exists) {
50  throw std::runtime_error(
51  "Can only read either from target or qmpackage logfile");

53

59  if (target_exists) {
60

63  throw std::runtime_error(
64  "ERROR <options.molpol.target> "
65  " should have this format: pxx pxy pxz pyy pyz pzz");
66  }
67  target_vec *= std::pow(tools::conv::ang2bohr, 3);
68  _polarization_target(0, 0) = target_vec(0);
69  _polarization_target(1, 0) = target_vec(1);
70  _polarization_target(0, 1) = target_vec(1);
71  _polarization_target(2, 0) = target_vec(2);
72  _polarization_target(0, 2) = target_vec(2);
73  _polarization_target(1, 1) = target_vec(3);
74  _polarization_target(2, 1) = target_vec(4);
75  _polarization_target(1, 2) = target_vec(4);
76  _polarization_target(2, 2) = target_vec(5);
77  } else {

79  std::string log_file = options.get(".logfile").as<std::string>();
80  Logger log;
81  log.setPreface(Log::info, "\n ...");
82  log.setPreface(Log::error, "\n ...");
83  log.setPreface(Log::warning, "\n ...");

108 Eigen::Vector3d MolPol::CalcInducedDipole(
109  const PolarSegment& input, const Eigen::Vector3d& ext_field) const {
110  Logger log;
111  log.setMultithreading(false);
112  log.setCommonPreface("\n ...");
113 
114  log.setReportLevel(Log::current_level);
115 
116  PolarRegion pol(0, log);
117  pol.Initialize(_polar_options);
118  pol.push_back(input);
119 
120  PolarSegment& workmol = pol[0];
121  for (PolarSite& site : workmol) {
122  site.V() = ext_field;
123  }
124  std::vector<std::unique_ptr<Region>> empty; // pol interacts with nobody else
125  pol.Evaluate(empty);
126  Eigen::Vector3d induced_dipole = Eigen::Vector3d::Zero();
127  for (const PolarSite& site : workmol) {
128  induced_dipole += site.Induced_Dipole();
129  }
130  if (Log::current_level > 0) {
131  std::cout << log;
132  }
133  return induced_dipole;
134 }

136 Eigen::Matrix3d MolPol::CalcClassicalPol(const PolarSegment& input) const {
137 
138  double eVnm_2_hrtbohr = tools::conv::ev2hrt / tools::conv::nm2bohr;
139  double fieldstrength = (0.1 * eVnm_2_hrtbohr);
140  Eigen::Matrix3d polarization = Eigen::Matrix3d::Zero();
141  Eigen::Vector3d ext_field = fieldstrength * Eigen::Vector3d::UnitX();
142  // central differences scheme
143  Eigen::Vector3d dxplus = CalcInducedDipole(input, ext_field);
144  Eigen::Vector3d dxminus = CalcInducedDipole(input, -ext_field);
145  polarization.col(0) = dxplus - dxminus;
146  ext_field = fieldstrength * Eigen::Vector3d::UnitY();
147  Eigen::Vector3d dyplus = CalcInducedDipole(input, ext_field);
148  Eigen::Vector3d dyminus = CalcInducedDipole(input, -ext_field);
149  polarization.col(1) = dyplus - dyminus;
150  ext_field = fieldstrength * Eigen::Vector3d::UnitZ();
151  Eigen::Vector3d dzplus = CalcInducedDipole(input, ext_field);
152  Eigen::Vector3d dzminus = CalcInducedDipole(input, -ext_field);
153  polarization.col(2) = dzplus - dzminus;
154 
155  return -polarization / (2 * fieldstrength);
156 }

158 void MolPol::Printpolarization(const Eigen::Matrix3d& result) const {
159  std::cout << std::endl << "First principle polarization [A^3]" << std::flush;
160  double conversion = std::pow(tools::conv::bohr2ang, 3);
161  std::cout << std::endl << _polarization_target * conversion << std::flush;
162  std::cout << std::endl << "Scaled classical polarization [A^3]" << std::flush;
163  std::cout << std::endl << result * conversion << std::flush;
164  std::cout << std::endl
165  << "EigenValues classical polarization [A^3]" << std::flush;
166  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es2;
167  es2.computeDirect(result, Eigen::EigenvaluesOnly);
168  Eigen::Matrix3d diag = es2.eigenvalues().asDiagonal();
169  std::cout << std::endl << diag * conversion << std::flush;
170 }

172 bool MolPol::Evaluate() {
173  OPENMP::setMaxThreads(_nThreads);
174  PolarSegment polar = _input;
175  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
176  es.computeDirect(_polarization_target, Eigen::EigenvaluesOnly);
177  const double pol_volume_target = std::pow(es.eigenvalues().prod(), 1.0 / 3.0);
178  for (Index iter = 0; iter < _max_iter; iter++) {
179 
180  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es2;
181  Eigen::Matrix3d pol = CalcClassicalPol(polar);
182  es2.computeDirect(pol, Eigen::EigenvaluesOnly);
183  const double pol_volume_iter =
184  std::pow(es2.eigenvalues().prod(), 1.0 / 3.0);
185  double scale = pol_volume_target / pol_volume_iter - 1;
186  std::cout << "\nIteration " << iter + 1 << " of " << _max_iter
187  << " target:" << pol_volume_target
188  << " current:" << pol_volume_iter << std::endl;
189 
190  for (Index i = 0; i < polar.size(); i++) {
191  Eigen::Matrix3d local_pol = polar[i].getpolarization();
192  polar[i].setpolarization(local_pol * (1 + scale * _weights[i]));
193  }
194 
195  if (std::abs(scale) < _tolerance_pol) {
196  std::cout << std::endl
197  << "... ... Iterative refinement : *CONVERGED*" << std::flush;
198  std::cout << std::endl
199  << "... ... Scaling coefficient : " << scale << std::flush;
200  polar.WriteMPS(_mps_output, "MOLPOL (OPTIMIZED)");
201  Printpolarization(pol);
202  break;
203  } else if (iter == (_max_iter - 1)) {
204  std::cout << std::endl
205  << "... ... Iterative refinement : *FAILED*" << std::flush;
206  std::cout << std::endl
207  << "... ... ERROR Convergence not achieved. "
208  << "Check your input mps-file, target polarizability <target> "
209  << std::flush;
210  Printpolarization(pol);
211  }
212  }
213  return true;
214 }
