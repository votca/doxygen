1887  std::vector<PolarSeg*>::iterator sit2;
1888  PolarSeg::iterator pit1;
1889  PolarSeg::iterator pit2;

1891  LOG(logDEBUG,*_log) << "Steps I-IV" << flush;
1892  // Depolarize midground
1893  LOG(logDEBUG,*_log) << " o Depolarize midground" << flush;
1894  for (sit1 = _mg_N.begin(); sit1 != _mg_N.end(); ++sit1) {
1895  for (pit1 = (*sit1)->begin(); pit1 != (*sit1)->end(); ++pit1) {

1904  for (pit2 = (*sit2)->begin(); pit2 != (*sit2)->end(); ++pit2) {

1906  _actor.FieldPerm_At_By(*(*pit1), *(*pit2), _polar_radial_corr_epsilon);

1908  }
1909  }

1914  for (pit1 = (*sit1)->begin(); pit1 != (*sit1)->end(); ++pit1) {
1915  (*pit1)->InduceDirect();
1916  }
1917  }
1918  // Evaluate Energy
1919  LOG(logDEBUG,*_log) << " o Evaluate stabilization energy" << flush;
1920  _polar_ERC = 0.0;
1921  for (sit1 = _mg_N.begin(); sit1 != _mg_N.end(); ++sit1) {
1922  for (pit1 = (*sit1)->begin(); pit1 < (*sit1)->end(); ++pit1) {
1923  for (sit2 = _polar_qm0.begin(); sit2 != _polar_qm0.end(); ++sit2) {
1924  for (pit2 = (*sit2)->begin(); pit2 != (*sit2)->end(); ++pit2) {
1925  _actor.BiasStat(*(*pit1), *(*pit2));
1926  _polar_ERC += _actor.EnergyInter_Indu_Perm(*(*pit1), *(*pit2));
1927  }
1928  }
1929  }
1930  }
1931  _polar_ERC *= conv::int2eV;
1932  LOG(logINFO,*_log) << "=> ERC = " << _polar_ERC << "eV" << flush;
1933  return;

2027 EWD::triple<> Ewald3DnD::ConvergeRealSpaceSum(std::vector<PolarSeg*> &target) {
2028

2030

2032  std::vector<APolarSite*> ::iterator pit1;
2033  std::vector<PolarSeg*>::iterator sit2;

2040  double this_ER = 0.0;
2041  for (int i = 0; i < 1000; ++i) {
2042  double Rc = _R_co + (i-1)*dR;

2047  for (sit1 = target.begin(); sit1 < target.end(); ++sit1) {

2050  for (pit2 = (*sit2)->begin(); pit2 < (*sit2)->end(); ++pit2) {
2051  this_ER += _actor.E_QQ_ERFC(*(*pit1), *(*pit2), _alpha);
2052  }
2053  }
2054  }
2055  }
2056  double dER_rms = sqrt((this_ER-prev_ER)*(this_ER-prev_ER))*conv::int2eV;
2057  LOG(logDEBUG,*_log)
2058  << (format("Rc = %1$+1.7f |MGN| = %3$5d nm ER = %2$+1.7f eV dER(rms) = %4$+1.7f")
