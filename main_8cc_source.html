108 ExclusionList *CsgBoltzmann::CreateExclusionList(Molecule &atomistic, Molecule &cg)
109 {
110  ExclusionList *ex = new ExclusionList();
111  //exclude all with all
112  {
113  list<Bead *> excl_list;
114  for(int i=0; i<atomistic.BeadCount(); ++i) {
115  excl_list.push_back(atomistic.getBead(i));
116  }
117  ex->ExcludeList(excl_list);
118  }
119 
120  //remove exclusions from inside a mapped bead
121  Topology *at_top = atomistic.getParent();
122  for(int i=0; i<cg.BeadCount(); ++i) {
123  vector<int> &w = cg.getBead(i)->ParentBeads();
124  list<Bead *> excl_list;
125  for(std::vector<int>::iterator it = w.begin(); it != w.end(); ++it){
126  excl_list.push_back(at_top->getBead(*it));
127  }
128  ex->Remove(excl_list);
129  }
130 
131  //remove exclusion which come from atomistic topology and hence bonds and angles
132  Topology *cg_top = cg.getParent();
133  for(int i=0; i<cg.BeadCount()-1; ++i) {
134  for(int j=i+1; j<cg.BeadCount(); ++j) {
135  if (cg_top->getExclusions().IsExcluded(cg.getBead(i),cg.getBead(j))){
136  vector<int> &w = cg.getBead(i)->ParentBeads();
137  vector<int> &v = cg.getBead(j)->ParentBeads();
138  for(std::vector<int>::iterator itw = w.begin(); itw != w.end(); ++itw){
139  for(std::vector<int>::iterator itv = v.begin(); itv != v.end(); ++itv){
140  ex->RemoveExclusion(at_top->getBead(*itw),at_top->getBead(*itv));
141  }
142  }
143  }
144  }
145  }
146  return ex;
147 }

157 void CsgBoltzmann::InteractiveMode()
158 {
159  std::map<std::string, AnalysisTool *> cmds;
160  TabulatedPotential tab;
161  StdAnalysis std;
162  tab.Register(cmds);
163  std.Register(cmds);
164 
165  string help_text =
166  "Interactive mode, expecting commands:\n"
167  "help: show this help\n"
168  "q: quit\n"
169  "list: list all available bonds\n"
170  "vals <file> <selection>: write values to file\n"
171  "hist <file> <selection>: create histogram\n"
172  "tab <file> <selection>: create tabulated potential\n"
173  "autocor <file> <selection>: calculate autocorrelation, only one row allowed in selection!\n"
174  "cor <file> <selection>: calculate correlations, first row is correlated with all other rows";
175 
176  cout << help_text << endl;
177 
178  while(1) {
179  string line;
180  cout << "> ";
181  getline(cin, line);
182 
183  boost::trim(line);
184  vector<string> args;
185  Tokenizer tok(line, " \t");
186  tok.ToVector(args);
187 
188  if(args.size() == 0) continue;
189 
190  string cmd = args.front();
191  args.erase(args.begin());
192 
193 
194  if(cmd == "q") break;
195 
196  std::map<string, AnalysisTool *>::iterator tool;
197  if(cmd == "help") {
198  if(args.size() == 0) {
199  cout << help_text << endl;
200  continue;
201  }
202  cmd = args.front();
203  args.erase(args.begin());
204  tool = cmds.find(cmd);
205  if(tool == cmds.end()) {
206  cout << "error, no help item found" << endl;
207  continue;
208  }
209  tool->second->Help(cmd, args);
210  cout << endl;
211  continue;
212  }
213 
214  tool = cmds.find(cmd);
215  if(tool == cmds.end()) {
216  cout << "error, command not found" << endl;
217  continue;
218  }
219 
220  tool->second->Command(_bs, cmd, args);
221  }
222 }
