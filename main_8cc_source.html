103 ExclusionList *CsgBoltzmann::CreateExclusionList(Molecule &atomistic,
104  Molecule &cg) {
105  ExclusionList *ex = new ExclusionList();
106  // exclude all with all
107  {
108  list<Bead *> excl_list;
109  for (int i = 0; i < atomistic.BeadCount(); ++i) {
110  excl_list.push_back(atomistic.getBead(i));
111  }
112  ex->ExcludeList(excl_list);
113  }
114 
115  // remove exclusions from inside a mapped bead
116  Topology *at_top = atomistic.getParent();
117  for (int i = 0; i < cg.BeadCount(); ++i) {
118  const vector<int> &parent_beads = cg.getBead(i)->ParentBeads();
119  list<Bead *> excl_list;
120 
121  for (const int &parent_bead_id : parent_beads) {
122  excl_list.push_back(at_top->getBead(parent_bead_id));
123  }
124  ex->Remove(excl_list);
125  }
126 
127  // remove exclusion which come from atomistic topology and hence bonds and
128  // angles
129  Topology *cg_top = cg.getParent();
130  for (int i = 0; i < cg.BeadCount() - 1; ++i) {
131  for (int j = i + 1; j < cg.BeadCount(); ++j) {
132  if (cg_top->getExclusions().IsExcluded(cg.getBead(i), cg.getBead(j))) {
133  const vector<int> &parent_beads_w = cg.getBead(i)->ParentBeads();
134  const vector<int> &parent_beads_v = cg.getBead(j)->ParentBeads();
135 
136  for (const int parent_bead_id_w : parent_beads_w) {
137  for (const int parent_bead_id_v : parent_beads_v) {
138  ex->RemoveExclusion(at_top->getBead(parent_bead_id_w),
139  at_top->getBead(parent_bead_id_v));
140  }
141  }
142  }
143  }
144  }
145  return ex;
146 }

155 void CsgBoltzmann::InteractiveMode() {
156  std::map<std::string, AnalysisTool *> cmds;
157  TabulatedPotential tab;
158  StdAnalysis std;
159  tab.Register(cmds);
160  std.Register(cmds);
161 
162  string help_text =
163  "Interactive mode, expecting commands:\n"
164  "help: show this help\n"
165  "q: quit\n"
166  "list: list all available bonds\n"
167  "vals <file> <selection>: write values to file\n"
168  "hist <file> <selection>: create histogram\n"
169  "tab <file> <selection>: create tabulated potential\n"
170  "autocor <file> <selection>: calculate autocorrelation, only one row "
171  "allowed in selection!\n"
172  "cor <file> <selection>: calculate correlations, first row is correlated "
173  "with all other rows";
174 
175  cout << help_text << endl;
176 
177  while (1) {
178  string line;
179  cout << "> ";
180  getline(cin, line);
181 
182  boost::trim(line);
183  vector<string> args;
184  Tokenizer tok(line, " \t");
185  tok.ToVector(args);
186 
187  if (args.size() == 0)
188  continue;
189 
190  string cmd = args.front();
191  args.erase(args.begin());
192 
193  if (cmd == "q")
194  break;
195 
196  std::map<string, AnalysisTool *>::iterator tool;
197  if (cmd == "help") {
198  if (args.size() == 0) {
199  cout << help_text << endl;
200  continue;
201  }
202  cmd = args.front();
203  args.erase(args.begin());
204  tool = cmds.find(cmd);
205  if (tool == cmds.end()) {
206  cout << "error, no help item found" << endl;
207  continue;
208  }
209  tool->second->Help(cmd, args);
210  cout << endl;
211  continue;
212  }
213 
214  tool = cmds.find(cmd);
215  if (tool == cmds.end()) {
216  cout << "error, command not found" << endl;
217  continue;
218  }
219 
220  tool->second->Command(_bs, cmd, args);
221  }
222 }
