106 ExclusionList *CsgBoltzmann::CreateExclusionList(Topology *top_atomistic,
107  Molecule &atomistic,
108  Topology *top_cg,
109  Molecule &cg) {
110  ExclusionList *ex = new ExclusionList();
111  // exclude all with all
112  {
113  list<Bead *> excl_list;
114  for (votca::Index i = 0; i < atomistic.BeadCount(); ++i) {
115  excl_list.push_back(atomistic.getBead(i));
116  }
117  ex->ExcludeList(excl_list);
118  }
119 
120  // remove exclusions from inside a mapped bead
121  for (votca::Index i = 0; i < cg.BeadCount(); ++i) {
122  const vector<votca::Index> &parent_beads = cg.getBead(i)->ParentBeads();
123  list<Bead *> excl_list;
124 
125  for (const votca::Index &parent_bead_id : parent_beads) {
126  excl_list.push_back(top_atomistic->getBead(parent_bead_id));
127  }
128  ex->Remove(excl_list);
129  }
130 
131  // remove exclusion which come from atomistic topology and hence bonds and
132  // angles
133  for (votca::Index i = 0; i < cg.BeadCount() - 1; ++i) {
134  for (votca::Index j = i + 1; j < cg.BeadCount(); ++j) {
135  if (top_cg->getExclusions().IsExcluded(cg.getBead(i), cg.getBead(j))) {
136  const vector<votca::Index> &parent_beads_w =
137  cg.getBead(i)->ParentBeads();
138  const vector<votca::Index> &parent_beads_v =
139  cg.getBead(j)->ParentBeads();
140 
141  for (const votca::Index parent_bead_id_w : parent_beads_w) {
142  for (const votca::Index parent_bead_id_v : parent_beads_v) {
143  ex->RemoveExclusion(top_atomistic->getBead(parent_bead_id_w),
144  top_atomistic->getBead(parent_bead_id_v));
145  }
146  }
147  }
148  }
149  }
150  return ex;
151 }

161 void CsgBoltzmann::InteractiveMode() {
162  std::map<std::string, AnalysisTool *> cmds;
163  TabulatedPotential tab;
164  StdAnalysis std;
165  tab.Register(cmds);
166  std.Register(cmds);
167 
168  string help_text =
169  "Interactive mode, expecting commands:\n"
170  "help: show this help\n"
171  "q: quit\n"
172  "list: list all available bonds\n"
173  "vals <file> <selection>: write values to file\n"
174  "hist <file> <selection>: create histogram\n"
175  "tab <file> <selection>: create tabulated potential\n"
176  "autocor <file> <selection>: calculate autocorrelation, only one row "
177  "allowed in selection!\n"
178  "cor <file> <selection>: calculate correlations, first row is correlated "
179  "with all other rows";
180 
181  cout << help_text << endl;
182 
183  while (1) {
184  string line;
185  cout << "> ";
186  getline(cin, line);
187 
188  boost::trim(line);
189 
190  votca::tools::Tokenizer tok(line, " \t");
191  vector<string> args = tok.ToVector();
192 
193  if (args.size() == 0) {
194  continue;
195  }
196 
197  string cmd = args.front();
198  args.erase(args.begin());
199 
200  if (cmd == "q") {
201  break;
202  }
203 
204  std::map<string, AnalysisTool *>::iterator tool;
205  if (cmd == "help") {
206  if (args.size() == 0) {
207  cout << help_text << endl;
208  continue;
209  }
210  cmd = args.front();
211  args.erase(args.begin());
212  tool = cmds.find(cmd);
213  if (tool == cmds.end()) {
214  cout << "error, no help item found" << endl;
215  continue;
216  }
217  tool->second->Help(cmd, args);
218  cout << endl;
219  continue;
220  }
221 
222  tool = cmds.find(cmd);
223  if (tool == cmds.end()) {
224  cout << "error, command not found" << endl;
225  continue;
226  }
227 
228  tool->second->Command(_bs, cmd, args);
229  }
230 }
