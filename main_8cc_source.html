98 ExclusionList *CsgBoltzmann::CreateExclusionList(Molecule &atomistic, Molecule &cg)
99 {
100  list<int> exclude;
101 
102  ExclusionList *ex = new ExclusionList();
103  ex->ExcludeAll(atomistic.BeadCount());
104 
105  // reintroduce bead internal nonbonded interaction
106  for(int i=0; i<cg.BeadCount(); ++i) {
107  exclude.clear();
108 
109  vector<int> &v = cg.getBead(i)->ParentBeads();
110  exclude.insert(exclude.begin(), v.begin(), v.end());
111  ex->Remove(exclude);
112  }
113 
114  Topology *top_cg = cg.getParent();
115  InteractionContainer::iterator iter;
116  // reintroduce nonbonded interactions for bonded beads
117  for(iter = top_cg->BondedInteractions().begin();
118  iter!=top_cg->BondedInteractions().end(); ++iter) {
119  Interaction *ic = *iter;
120  exclude.clear();
121  for(int i=0; i<ic->BeadCount(); i++) {
122  vector<int> &v = top_cg->getBead(ic->getBeadId(i))->ParentBeads();
123  exclude.insert(exclude.end(), v.begin(), v.end());
124  }
125  ex->Remove(exclude);
126  }
127  return ex;
128 }

138 void CsgBoltzmann::InteractiveMode()
139 {
140  std::map<std::string, AnalysisTool *> cmds;
141  TabulatedPotential tab;
142  StdAnalysis std;
143  tab.Register(cmds);
144  std.Register(cmds);
145 
146  string help_text =
147  "Interactive mode, expecting commands:\n"
148  "help: show this help\n"
149  "q: quit\n"
150  "list: list all available bonds\n"
151  "vals <file> <selection>: write values to file\n"
152  "hist <file> <selection>: create histogram\n"
153  "tab <file> <selection>: create tabulated potential\n"
154  "autocor <file> <selection>: calculate autocorrelation, only one row allowed in selection!\n"
155  "cor <file> <selection>: calculate correlations, first row is correlated with all other rows";
156 
157  cout << help_text << endl;
158 
159  while(1) {
160  string line;
161  cout << "> ";
162  getline(cin, line);
163 
164  boost::trim(line);
165  vector<string> args;
166  Tokenizer tok(line, " \t");
167  tok.ToVector(args);
168 
169  if(args.size() == 0) continue;
170 
171  string cmd = args.front();
172  args.erase(args.begin());
173 
174 
175  if(cmd == "q") break;
176 
177  std::map<string, AnalysisTool *>::iterator tool;
178  if(cmd == "help") {
179  if(args.size() == 0) {
180  cout << help_text << endl;
181  continue;
182  }
183  cmd = args.front();
184  args.erase(args.begin());
185  tool = cmds.find(cmd);
186  if(tool == cmds.end()) {
187  cout << "error, no help item found" << endl;
188  continue;
189  }
190  tool->second->Help(cmd, args);
191  cout << endl;
192  continue;
193  }
194 
195  tool = cmds.find(cmd);
196  if(tool == cmds.end()) {
197  cout << "error, command not found" << endl;
198  continue;
199  }
200 
201  tool->second->Command(_bs, cmd, args);
202  }
203 }
