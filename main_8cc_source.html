00108 ExclusionList *CsgBoltzmann::CreateExclusionList(Molecule &atomistic, Molecule &cg)
00109 {
00110     ExclusionList *ex = new ExclusionList();
00111     //exclude all with all
00112     {
00113         list<Bead *> excl_list;
00114         for(int i=0; i<atomistic.BeadCount(); ++i) {
00115             excl_list.push_back(atomistic.getBead(i));
00116         }
00117         ex->ExcludeList(excl_list);
00118     }
00119 
00120     //remove exclusions from inside a mapped bead
00121     Topology *at_top = atomistic.getParent();
00122     for(int i=0; i<cg.BeadCount(); ++i) {
00123         vector<int> &w = cg.getBead(i)->ParentBeads();
00124         list<Bead *> excl_list;
00125         for(std::vector<int>::iterator it = w.begin(); it != w.end(); ++it){
00126             excl_list.push_back(at_top->getBead(*it));
00127         }
00128         ex->Remove(excl_list);
00129     }
00130 
00131     //remove exclusion which come from atomistic topology and hence bonds and angles
00132     Topology *cg_top = cg.getParent();
00133     for(int i=0; i<cg.BeadCount()-1; ++i) {
00134         for(int j=i+1; j<cg.BeadCount(); ++j) {
00135             if (cg_top->getExclusions().IsExcluded(cg.getBead(i),cg.getBead(j))){
00136               vector<int> &w = cg.getBead(i)->ParentBeads();
00137               vector<int> &v = cg.getBead(j)->ParentBeads();
00138               for(std::vector<int>::iterator itw = w.begin(); itw != w.end(); ++itw){
00139                   for(std::vector<int>::iterator itv = v.begin(); itv != v.end(); ++itv){
00140                       ex->RemoveExclusion(at_top->getBead(*itw),at_top->getBead(*itv));
00141                   }
00142               }
00143             }
00144         }
00145     }
00146     return ex;
00147 }
