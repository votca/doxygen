60 bool TrajExtractor::EvaluateFrame(Topology *top) {
61 
62  // Rigidify std::system (if possible)
63  if (!top->Rigidify()) return 0;
64 
65  // Print coordinates
66  FILE *outPDBmd = NULL;
67  FILE *outPDBqm = NULL;
68 
69  outPDBmd = fopen(_outPDBmd.c_str(), "w");
70  outPDBqm = fopen(_outPDBqm.c_str(), "w");
71 
72  fprintf(outPDBmd, "TITLE VOT CAtastrophic title \n");
73  fprintf(outPDBmd, "Model %8d \n" , 1);
74 
75  fprintf(outPDBqm, "TITLE VOT CAtastrophic title \n");
76  fprintf(outPDBqm, "Model %8d \n", 1);
77 
78  vector< Segment* > ::iterator sit;
79  for (sit = top->Segments().begin();
80  sit < top->Segments().end();
81  sit++) {
82 
83  (*sit)->WritePDB(outPDBmd, "Atoms", "MD");
84  (*sit)->WritePDB(outPDBqm, "Atoms", "QM");
85  }
86 
87  fprintf(outPDBmd, "TER\nENDMDL\n");
88  fprintf(outPDBqm, "TER\nENDMDL\n");
89 
90  fclose(outPDBmd);
91  fclose(outPDBqm);
92 
93  return true;
94 }
