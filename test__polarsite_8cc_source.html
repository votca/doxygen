57BOOST_AUTO_TEST_CASE(rotation_test) {
58 PolarSite ps(1, "ps2", Eigen::Vector3d::UnitY());
59
60 Eigen::Matrix3d R = Eigen::Matrix3d::Zero(); // Rotation around z axes
61 R << 0, -1, 0, 1, 0, 0, 0, 0, 1;
62
63 Vector9d multipoles = Vector9d::Zero(9);
64 multipoles << 1, 1, 0, 0, 0, 1, 0, 0,
65 0; // q=1, mu_x=1 and Q_21c=1 the rest is 0
66 ps.setMultipole(multipoles, 2);
67 ps.Rotate(R, Eigen::Vector3d::Zero());
68 bool equalpos = ps.getPos().isApprox(Eigen::Vector3d(-1, 0, 0), 1e-5);
69 if (!equalpos) {
70 std::cout << "Result " << std::endl;
71 std::cout << ps.getPos() << std::endl;
72 std::cout << "Reference" << std::endl;
73 std::cout << Eigen::Vector3d(-1, 0, 0) << std::endl;
74 }
75 BOOST_CHECK_EQUAL(equalpos, true);
76
77 Eigen::VectorXd rotmultipoles = Eigen::VectorXd::Zero(9);
78 rotmultipoles << 1, 0, 1, 0, 0, 0, 1, 0, 0; // q=1, mu_y=1 and Q_21s=1 is 0
79 bool equalmultipoles = rotmultipoles.isApprox(ps.Q(), 1e-5);
80 if (!equalmultipoles) {
81 std::cout << "Result " << std::endl;
82 std::cout << ps.Q() << std::endl;
83 std::cout << "Reference" << std::endl;
84 std::cout << rotmultipoles << std::endl;
85 }
86 BOOST_CHECK_EQUAL(equalmultipoles, true);
87}
