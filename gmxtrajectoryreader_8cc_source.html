42bool GMXTrajectoryReader::FirstFrame(Topology &conf) {
43 gmx_output_env_t *oenv;
44#if GMX_VERSION >= 20210000
45 gmx::TimeUnit timeunit = gmx::TimeUnit::Picoseconds;
46 XvgFormat xvg = XvgFormat::None;
47#else
48 time_unit_t timeunit = time_ps;
49 xvg_format_t xvg = exvgNONE;
50#endif
51 output_env_init(&oenv, gmx::getProgramContext(), timeunit, FALSE, xvg, 0);
52 if (!read_first_frame(oenv, &gmx_status_, (char *)filename_.c_str(),
53 &gmx_frame_, TRX_READ_X | TRX_READ_V | TRX_READ_F)) {
54 throw std::runtime_error(std::string("cannot open ") + filename_);
55 }
56 output_env_done(oenv);
57
58 Eigen::Matrix3d m;
59 for (Index i = 0; i < 3; i++) {
60 for (Index j = 0; j < 3; j++) {
61 m(i, j) = gmx_frame_.box[j][i];
62 }
63 }
64 conf.setBox(m);
65 conf.setTime(gmx_frame_.time);
66 conf.setStep(gmx_frame_.step);
67 std::cout << std::endl;
68
69 if (gmx_frame_.natoms != (Index)conf.Beads().size()) {
70 throw std::runtime_error(
71 "number of beads in trajectory do not match topology");
72 }
73
74 // conf.HasPos(true);
75 // conf.HasF( gmx_frame_.bF);
76
77 for (Index i = 0; i < gmx_frame_.natoms; i++) {
78 Eigen::Vector3d r = {gmx_frame_.x[i][XX], gmx_frame_.x[i][YY],
79 gmx_frame_.x[i][ZZ]};
80 conf.getBead(i)->setPos(r);
81 if (gmx_frame_.bF) {
82 Eigen::Vector3d f = {gmx_frame_.f[i][XX], gmx_frame_.f[i][YY],
83 gmx_frame_.f[i][ZZ]};
84 conf.getBead(i)->setF(f);
85 }
86 if (gmx_frame_.bV) {
87 Eigen::Vector3d v = {gmx_frame_.v[i][XX], gmx_frame_.v[i][YY],
88 gmx_frame_.v[i][ZZ]};
89 conf.getBead(i)->setVel(v);
90 }
91 }
92 return true;
93}
