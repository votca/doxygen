123 xml << "<initial_guess>independent</initial_guess>" << std::endl;
124 xml << "<xtpdft>" << std::endl;

126 xml << "<fock_matrix_reset>5</fock_matrix_reset>\n";
127 xml << "<convergence>" << std::endl;
128 xml << " <energy>1e-7</energy>" << std::endl;
129 xml << " <method>DIIS</method>" << std::endl;

131 xml << " <ADIIS_start>0.8</ADIIS_start>" << std::endl;
132 xml << " <DIIS_length>20</DIIS_length>" << std::endl;
133 xml << " <levelshift>0.0</levelshift>" << std::endl;
134 xml << " <levelshift_end>0.2</levelshift_end>" << std::endl;
135 xml << " <max_iterations>100</max_iterations>\n";
136 xml << " <error>1e-7</error>\n";
137 xml << " <DIIS_maxout>false</DIIS_maxout>\n";
138 xml << " <mixing>0.7</mixing>\n";
139 xml << "</convergence>" << std::endl;
140 xml << "<integration_grid>xcoarse</integration_grid>" << std::endl;
141 xml << "<max_iterations>200</max_iterations>" << std::endl;
142 xml << "</xtpdft>" << std::endl;
143 xml << "</dftpackage>" << std::endl;
144 xml.close();
145 votca::tools::Property prop;
146 prop.LoadFromXML("dftengine2.xml");
147
148 Logger log;
149 dft.setLogger(&log);
150 dft.Initialize(prop.get("dftpackage"));
151 dft.Evaluate(orb);
152
153 BOOST_CHECK_CLOSE(orb.getDFTTotalEnergy(), -75.891017293070945, 1e-5);
154
155 Eigen::VectorXd MOs_energy_ref = Eigen::VectorXd::Zero(13);
156 MOs_energy_ref << -19.0739, -1.01904, -0.520731, -0.341996, -0.27356,
157 0.118834, 0.210783, 0.953576, 1.04314, 1.46895, 1.54729, 1.67293, 2.77584;
158 bool check_eng = MOs_energy_ref.isApprox(orb.MOs().eigenvalues(), 1e-5);
159 BOOST_CHECK_EQUAL(check_eng, true);
160 if (!check_eng) {
161 std::cout << "result eng" << std::endl;
162 std::cout << orb.MOs().eigenvalues() << std::endl;
163 std::cout << "ref eng" << std::endl;
164 std::cout << MOs_energy_ref << std::endl;
165 }
166
167 Eigen::MatrixXd MOs_coeff_ref =
168 votca::tools::EigenIO_MatrixMarket::ReadMatrix(
169 std::string(XTP_TEST_DATA_FOLDER) + "/dftengine/MOs_coeff_ref.mm");
170
171 AOBasis basis = orb.getDftBasis();
172 AOOverlap overlap;
173 overlap.Fill(basis);
174 Eigen::MatrixXd proj = MOs_coeff_ref.leftCols(5).transpose() *
175 overlap.Matrix() *
176 orb.MOs().eigenvectors().leftCols(5);
177 Eigen::VectorXd norms = proj.colwise().norm();
178 bool check_coeff = norms.isApproxToConstant(1, 1e-5);
179 BOOST_CHECK_EQUAL(check_coeff, true);
180 if (!check_coeff) {
181 std::cout << "result coeff" << std::endl;
182 std::cout << orb.MOs().eigenvectors() << std::endl;
183 std::cout << "ref coeff" << std::endl;
184 std::cout << MOs_coeff_ref << std::endl;
185 }
186
187 libint2::finalize();
188}
