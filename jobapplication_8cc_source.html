74 void JobApplication::Run() {
75 
76  load_property_from_xml(_options, _op_vm["options"].as<string>());
77 
78  // EVALUATE OPTIONS
79  int nThreads = OptionsMap()["nthreads"].as<int>();
80  int nframes = OptionsMap()["nframes"].as<int>();
81  int fframe = OptionsMap()["first-frame"].as<int>();
82  if (fframe-- == 0) throw runtime_error("ERROR: First frame is 0, counting "
83  "in VOTCA::XTP starts from 1.");
84  int save = OptionsMap()["save"].as<int>();
85 
86  // STATESAVER & PROGRESS OBSERVER
87  string statefile = OptionsMap()["file"].as<string>();
88  StateSaverSQLite statsav;
89  statsav.Open(_top, statefile);
90 
91  ProgObserver< std::vector<Job*>, Job*, Job::JobResult > progObs
92  = ProgObserver< std::vector<Job*>, Job*, Job::JobResult >();
93  progObs.InitCmdLineOpts(OptionsMap());
94 
95  // INITIALIZE & RUN CALCULATORS
96  cout << "Initializing calculators " << endl;
97  BeginEvaluate(nThreads, &progObs);
98 
99  int frameId = -1;
100  int framesDone = 0;
101  while (statsav.NextFrame() && framesDone < nframes) {
102  frameId += 1;
103  if (frameId < fframe) continue;
104  cout << "Evaluating frame " << _top.getDatabaseId() << endl;
105  EvaluateFrame();
106  if (save == 1) { statsav.WriteFrame(); }
107  else { cout << "Changes have not been written to state file." << endl; }
108  framesDone += 1;
109  }
110 
111  if (framesDone == 0)
112  cout << "Input requires first frame = " << fframe+1 << ", # frames = "
113  << nframes << " => No frames processed.";
114 
115  statsav.Close();
116  EndEvaluate();
117 
118 }

140 bool JobApplication::EvaluateFrame() {
141  list< JobCalculator* > ::iterator it;
142  for (it = _calculators.begin(); it != _calculators.end(); it++) {
143  cout << "... " << (*it)->Identify() << " " << flush;
144  if (_generate_input) (*it)->WriteJobFile(&_top);
145  if (_run) (*it)->EvaluateFrame(&_top);
146  if (_import) (*it)->ReadJobFile(&_top);
147  cout << endl;
148  }
149  return true;
150 }
