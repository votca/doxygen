74 void JobApplication::Run() {
75  std::string name = ProgramName();
76  if (VersionString() != "") {
77  name = name + ", version " + VersionString();
78  }
79  xtp::HelpTextHeader(name);
80 
81  // EVALUATE OPTIONS
82  Index nThreads = OptionsMap()["nthreads"].as<Index>();
83  Index ompThreads = OptionsMap()["ompthreads"].as<Index>();
84  Index nframes = OptionsMap()["nframes"].as<Index>();
85  Index fframe = OptionsMap()["first-frame"].as<Index>();
86  bool save = OptionsMap()["save"].as<bool>();
87 
88  // STATESAVER & PROGRESS OBSERVER
89  std::string statefile = OptionsMap()["file"].as<std::string>();
90  _options.LoadFromXML(OptionsMap()["options"].as<std::string>());
91  ProgObserver<std::vector<Job>> progObs = ProgObserver<std::vector<Job>>();
92  progObs.InitCmdLineOpts(OptionsMap());
93 
94  // INITIALIZE & RUN CALCULATORS
95  std::cout << "Initializing calculator " << std::endl;
96  BeginEvaluate(nThreads, ompThreads, progObs);
97 
98  StateSaver statsav(statefile);
99 
100  std::vector<Index> frames = statsav.getFrames();
101 
102  std::cout << frames.size() << " frames in statefile, Ids are: ";
103  for (Index frame : frames) {
104  std::cout << frame << " ";
105  }
106  std::cout << std::endl;
107  if (fframe < Index(frames.size())) {
108  std::cout << "Starting at frame " << frames[fframe] << std::endl;
109  } else {
110  std::cout << "First frame:" << fframe
111  << " is larger than number of frames:" << Index(frames.size())
112  << std::endl;
113  return;
114  }
115 
116  if ((fframe + nframes) > Index(frames.size())) {
117  nframes = Index(frames.size()) - fframe;
118  }
119 
120  for (Index i = fframe; i < nframes; i++) {
121  std::cout << "Evaluating frame " << frames[i] << std::endl;
122  Topology top = statsav.ReadFrame(frames[i]);
123  EvaluateFrame(top);
124  if (save && _import) {
125  statsav.WriteFrame(top);
126  } else {
127  std::cout << "Changes have not been written to state file." << std::endl;
128  }
129  }
130 }
