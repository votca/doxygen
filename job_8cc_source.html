203 void UPDATE_JOBS(const std::vector<Job> &from, std::vector<Job> &to,
204  const std::string &thisHost) {
205  std::vector<Job>::iterator it_int;
206  std::vector<Job>::const_iterator it_ext;
207 
208  if (to.size() != from.size())
209  throw std::runtime_error("Progress file out of sync (::size), abort.");
210 
211  for (it_int = to.begin(), it_ext = from.begin(); it_int != to.end();
212  ++it_int, ++it_ext) {
213  Job &job_int = *it_int;
214  const Job &job_ext = *it_ext;
215  if (job_int.getId() != job_ext.getId())
216  throw std::runtime_error("Progress file out of sync (::id), abort.");
217  if (job_ext.hasHost() && job_ext.getHost() != thisHost)
218  job_int.UpdateFrom(job_ext);
219  }
220 
221  return;
222 }
