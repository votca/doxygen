283  }

315 void QMMIter::UpdatePosChrgFromQMAtoms(vector< QMAtom* > &qmatoms,
316  vector< PolarSeg* > &psegs) {
317 
318  double AA_to_NM = 0.1; // Angstrom to nanometer
319 
320  double dR_RMS = 0.0;
321  double dQ_RMS = 0.0;
322  double dQ_SUM = 0.0;
323 
324  for (unsigned int i = 0, qac = 0; i < psegs.size(); ++i) {
325  PolarSeg *pseg = psegs[i];
326  for (unsigned int j = 0; j < pseg->size(); ++j, ++qac) {
327 
328  // Retrieve info from QMAtom
329  QMAtom *qmatm = qmatoms[qac];
330  vec upd_r = vec(qmatm->x, qmatm->y, qmatm->z);
331  upd_r *= AA_to_NM;
332  double upd_Q00 = qmatm->charge;
333 
334  // Compare to previous r, Q00
335  APolarSite *aps = (*pseg)[j];
336  vec old_r = aps->getPos();
337  double old_Q00 = aps->getQ00();
338  double dR = abs(upd_r - old_r);
339  double dQ00 = upd_Q00 - old_Q00;
340 
341  dR_RMS += dR*dR;
342  dQ_RMS += dQ00*dQ00;
343  dQ_SUM += dQ00;
344 
345  // Forward updated r, Q00 to APS
346  aps->setPos(upd_r);
347  aps->setQ00(upd_Q00, 0);
348  }
349  }
350 
351  dR_RMS /= qmatoms.size();
352  dQ_RMS /= qmatoms.size();
353  dR_RMS = sqrt(dR_RMS);
354  dQ_RMS = sqrt(dQ_RMS);
355 
356  this->setdRdQ(dR_RMS, dQ_RMS, dQ_SUM);
357 }

360 void QMMIter::GenerateQMAtomsFromPolarSegs(PolarTop *ptop, Orbitals &orb,
361  bool split_dpl, double dpl_spacing) {
362 
363  double AA_to_NM = 0.1; // Angstrom to nanometer
364 
365  // INNER SHELL QM0
366  for (unsigned int i = 0; i < ptop->QM0().size(); ++i) {
367  PolarSeg *pseg = ptop->QM0()[i];
368  for (unsigned int j = 0; j < pseg->size(); ++j) {
369 
370  APolarSite *aps = (*pseg)[j];
371  vec pos = aps->getPos()/AA_to_NM;
372  double Q = aps->getQ00();
373  string type = "qm";
374 
375  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, false);
376 
377  }
378  }
379 
380  // MIDDLE SHELL MM1
381  for (unsigned int i = 0; i < ptop->MM1().size(); ++i) {
382  PolarSeg *pseg = ptop->MM1()[i];
383  for (unsigned int j = 0; j < pseg->size(); ++j) {
384 
385  APolarSite *aps = (*pseg)[j];
386  vec pos = aps->getPos()/AA_to_NM;
387  double Q = aps->getQ00();
388  string type = "mm";
389 
390  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
391 
392  if (split_dpl) {
393  vec tot_dpl = vec(aps->U1x,aps->U1y,aps->U1z);
394  if (aps->getRank() > 0)
395  { tot_dpl += vec(aps->Q1x,aps->Q1y,aps->Q1z); }
396  // Calculate virtual charge positions
397  double a = dpl_spacing; // this is in nm
398  double mag_d = abs(tot_dpl); // this is in e * nm
399  vec dir_d_0 = tot_dpl.normalize();
400  vec dir_d = dir_d_0.normalize();
401  vec A = pos + 0.5 * a * dir_d /AA_to_NM; // converted to AA
402  vec B = pos - 0.5 * a * dir_d /AA_to_NM;
403  double qA = mag_d / a;
404  double qB = - qA;
405  // Zero out if magnitude small [e*nm]
406  if (aps->getIsoP() < 1e-9 || mag_d < 1e-9) {
407  A = aps->getPos() + 0.1*a*vec(1,0,0); // != pos since self-energy may diverge
408  B = aps->getPos() - 0.1*a*vec(1,0,0);
409  qA = 0;
410  qB = 0;
411  }
412  orb.AddAtom("A", A.x(), A.y(), A.z(), qA, true);
413  orb.AddAtom("B", B.x(), B.y(), B.z(), qB, true);
414  }
415  }
416  }
417 
418  // OUTER SHELL MM2
419  for (unsigned int i = 0; i < ptop->MM2().size(); ++i) {
420  PolarSeg *pseg = ptop->MM2()[i];
421  for (unsigned int j = 0; j < pseg->size(); ++j) {
422 
423  APolarSite *aps = (*pseg)[j];
424  vec pos = aps->getPos()/AA_to_NM;
425  double Q = aps->getQ00();
426  string type = "mm";
427 
428  orb.AddAtom(aps->getName(), pos.x(), pos.y(), pos.z(), Q, true);
429  }
430  }
431  return;
