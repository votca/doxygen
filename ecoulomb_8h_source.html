65 bool ECoulomb::EvaluateFrame(Topology *top) {
66 
67  // Polar sites generated?
68  if (top->isEStatified() == false) {
69  cout << endl << "... ... ERROR: Estatify first by running emultipole. " << flush;
70  return 0;
71  }
72  else {
73  cout << endl << "... ... System is estatified. Proceed. " << flush;
74  }
75 
76  vector< Segment* > ::iterator sit;
77  vector< PolarSite* > ::iterator pit;
78 
79  // Reset all polar sites to neutral state
80  for (sit = top->Segments().begin();
81  sit < top->Segments().end();
82  ++sit) {
83  for (pit = (*sit)->PolarSites().begin();
84  pit < (*sit)->PolarSites().end();
85  ++pit) {
86  (*pit)->Charge(0);
87  }}
88 
89  // Calculate energies for segments
90  for (sit = top->Segments().begin();
91  sit < top->Segments().end();
92  ++sit) {
93 
94  if ((*sit)->getId() < _first) { continue; }
95  if ((*sit)->getId() == _last+1) { break; }
96 
97  cout << endl << "... ... Evaluating site " << (*sit)->getId() << flush;
98 
99  for (int state = -1; state < 2; ++state) {
100 
101  this->EvaluateSegment(top, *sit, state);
102  }}
103 
104  // Output results
105  this->Output2File(top);
106 
107  return true;
108 }

111 void ECoulomb::EvaluateSegment(Topology *top, Segment *seg, int state) {
112 
113  if (seg->hasChrgState(state)) {
114 
115  vector< Segment* > ::iterator ext;
116  vector< PolarSite* > ::iterator pit1;
117  vector< PolarSite* > ::iterator pit2;
118 
119  // ++++++++++++++++++++++++++++ //
120  // Charge segment appropriately //
121  // ++++++++++++++++++++++++++++ //
122 
123  for (pit1 = seg->PolarSites().begin();
124  pit1 < seg->PolarSites().end();
125  ++pit1) {
126 
127  (*pit1)->Charge(state);
128  }
129 
130  double E_INTER = 0.0;
131  int COUNT_EXT = 0;
132  double INT2eV = 1/(4*M_PI*8.854187817e-12)
133  * 1.602176487e-19 / 1.000e-9;
134 
135  // ++++++++++++++++++++++++++++ //
136  // Calculate interaction energy //
137  // ++++++++++++++++++++++++++++ //
138 
139  FILE *out;
140  string sphpdb = boost::lexical_cast<string>(seg->getId())
141  + "_" + boost::lexical_cast<string>(state)
142  + "_check.pdb";
143  out = fopen(sphpdb.c_str(), "w");
144  vec com_shift = -1 * seg->getPos();
145  for (pit1 = seg->PolarSites().begin();
146  pit1 < seg->PolarSites().end();
147  ++pit1) {
148  (*pit1)->PrintPDB(out, com_shift);
149  }
150 
151  for (ext = top->Segments().begin();
152  ext < top->Segments().end();
153  ++ext) {
154 
155  // Segment different from central one?
156  if ((*ext)->getId() == seg->getId()) { continue; }
157 
158  // Segment within cutoff?
159  vec R_pbc = top->PbShortestConnect((*ext)->getPos(), seg->getPos());
160  if (abs(R_pbc) > _cutoff) { continue; }
161  vec R_dir = seg->getPos() - (*ext)->getPos();
162  vec shift = R_dir - R_pbc;
163 
164 // double r12 = abs(top->PbShortestConnect((*ext)->getPos(),
165 // seg->getPos()));
166 // if (r12 > _cutoff) { continue; }
167 
168  // Check polar sites
169  // (*ext)->WritePDB(out, "Multipoles", "Charges");
170 
171  ++COUNT_EXT;
172 
173  for (pit1 = (*ext)->PolarSites().begin();
174  pit1 < (*ext)->PolarSites().end();
175  ++pit1) {
176 
177 // vec dr_pbc = top->PbShortestConnect((*pit1)->getPos(),
178 // seg->getPos());
179 // vec dr_dir = seg->getPos() - (*pit1)->getPos();
180 // vec pol_shift = vec(0,0,0);
181 // if (abs(dr_pbc - dr_dir) > 1e-8) {
182 // pol_shift = dr_dir - dr_pbc;
183 // }
184  (*pit1)->PrintPDB(out, com_shift + shift);
185 
186  for (pit2 = seg->PolarSites().begin();
187  pit2 < seg->PolarSites().end();
188  ++pit2) {
189 
190 // double R = abs(top->PbShortestConnect((*pit1)->getPos(),
191 // (*pit2)->getPos()));
192 
193  double R = abs((*pit1)->getPos() + shift - (*pit2)->getPos());
194 
195  E_INTER += (*pit1)->Q00 * 1/R * (*pit2)->Q00;
196  }}
197  }
198 
199  fclose(out);
200 
201  // ++++++++++++++++++++++++++++++ //
202  // Store result, reset to neutral //
203  // ++++++++++++++++++++++++++++++ //
204 
205  seg->setEMpoles(state, INT2eV * E_INTER);
206  _log_seg_sphere[seg->getId()] = COUNT_EXT;
207  cout << endl << "... ... ... STATE " << state << ": "
208  << " EP(~) = " << E_INTER*INT2eV << " eV"
209  << " SPH(o) = " << COUNT_EXT+1 << flush;
210 
211  for (pit1 = seg->PolarSites().begin();
212  pit1 < seg->PolarSites().end();
213  ++pit1) {
214  (*pit1)->Charge(0);
215  }
216  }
217 }
