33 Eigen::VectorXd Readcubefile(const std::string& filename) {
34 
35  std::ifstream in1;
36  in1.open(filename, std::ios::in);
37 
38  std::string result = "";
39  std::string s;
40  getline(in1, s);
41  getline(in1, s);
42  getline(in1, s);
43  std::vector<double> cube_values;
44  do {
45  votca::tools::Tokenizer tok(s, " ");
46  std::vector<double> values;
47  tok.ConvertToVector<double>(values);
48  cube_values.insert(cube_values.end(), values.begin(), values.end());
49  } while (getline(in1, s));
50  return Eigen::Map<Eigen::VectorXd>(cube_values.data(), cube_values.size());
51 }
