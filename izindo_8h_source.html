125 void IZindo::ParseOrbitalsXML(Property *opt) {
126 
127  // update options with the VOTCASHARE defaults
128  UpdateWithDefaults( opt, "ctp" );
129 
130  string key = "options.izindo";
131  string orbitalsXML = opt->get(key+".orbitalsXML").as<string> ();
132  cout << endl << "... ... Orbital data from " << orbitalsXML << ". ";
133 
134  Property alloc;
135  load_property_from_xml(alloc, orbitalsXML.c_str());
136 
137  /* --- ORBITALS.XML Structure ---
138  *
139  * <topology>
140  *
141  * <molecules>
142  * <molecule>
143  * <name></name>
144  *
145  * <segments>
146  *
147  * <segment>
148  * <name></name>
149  *
150  * <basisset></basisset>
151  * <orbitals></orbitals>
152  *
153  * <torbital_e></torbital_e>
154  * <torbital_h></torbital_h>
155  * </segment>
156  *
157  * <segment>
158  * ...
159  *
160  */
161 
162  key = "topology.molecules.molecule";
163  list<Property*> mols = alloc.Select(key);
164  list<Property*> ::iterator molit;
165  for (molit = mols.begin(); molit != mols.end(); ++molit) {
166 
167  key = "segments.segment";
168  list<Property*> segs = (*molit)->Select(key);
169  list<Property*> ::iterator segit;
170 
171  for (segit = segs.begin(); segit != segs.end(); ++segit) {
172 
173  string segName = (*segit)->get("name").as<string> ();
174  string basisName = (*segit)->get("basisset").as<string> ();
175  string orbFile = (*segit)->get("orbitals").as<string> ();
176 
177  bool has_e = false;
178  bool has_h = false;
179  vector<int> torbs_e;
180  vector<int> torbs_h;
181 
182  if ( (*segit)->exists("torbital_e") ) {
183  torbs_e = (*segit)->get("torbital_e").as< vector<int> > ();
184  has_e = (torbs_e.size()) ? true : false;
185  }
186  if ( (*segit)->exists("torbital_h") ) {
187  torbs_h = (*segit)->get("torbital_h").as< vector<int> > ();
188  has_h = (torbs_h.size()) ? true : false;
189  }
190 
191  _seg_basisName[segName] = basisName;
192  _seg_orbFile[segName] = orbFile;
193  _seg_has_e[segName] = has_e;
194  _seg_has_h[segName] = has_h;
195  _seg_torbs_e[segName] = torbs_e;
196  _seg_torbs_h[segName] = torbs_h;
197 
198  }
199  }
200 }

203 void IZindo::EvalPair(Topology *top, QMPair *qmpair, PairOperator *opThread) {
204 
205  this->LockCout();
206  cout << "\r... ... Evaluating pair " << qmpair->getId() << flush;
207  this->UnlockCout();
208 
209  string segName1 = qmpair->Seg1()->getName();
210  string segName2 = qmpair->Seg2()->getName();
211 
212  bool pair_has_e = false;
213  bool pair_has_h = false;
214 
215  try {
216  pair_has_e = _seg_has_e.at(segName1) && _seg_has_e.at(segName2);
217  pair_has_h = _seg_has_h.at(segName1) && _seg_has_h.at(segName2);
218  }
219  catch (out_of_range&) {
220  this->LockCout();
221  cout << endl << "... ... WARNING: No orbital information for pair ["
222  << segName1 << ", " << segName2 << "]. "
223  << "Skipping... " << endl;
224  this->UnlockCout();
225 
226  return;
227  }
228 
229  if (pair_has_e) {
230  this->CTP2MOO2CTP(qmpair, opThread, -1);
231  }
232  if (pair_has_h) {
233  this->CTP2MOO2CTP(qmpair, opThread, +1);
234  }
235 }

286

289 
290  // Define basis set //

294  morb2->define_bs(basis2);

299

301  morb1->init_orbitals(*orb1, orbFile1.c_str());
302  morb2->init_orbitals(*orb2, orbFile2.c_str());
303 
304  // Convert: Counting from 1 => Counting from 0
305  vector<int> torbs1_from_zero;
306  vector<int> torbs2_from_zero;
307  for (unsigned int i = 0; i < torbs1.size(); ++i) torbs1_from_zero.push_back(torbs1[i]-1);
308  for (unsigned int i = 0; i < torbs2.size(); ++i) torbs2_from_zero.push_back(torbs2[i]-1);
309 
310  orb1->strip_orbitals(torbs1_from_zero);
311  orb2->strip_orbitals(torbs2_from_zero);
312 
313  int frontier1 = torbs1.size();
314  int frontier2 = torbs2.size();
315  for (int i = 0; i < frontier1; i++) { torbNrs1.push_back(i); }
316  for (int j = 0; j < frontier2; j++) { torbNrs2.push_back(j); }
317 
318  morb1->assign_orb(orb1);
319  morb2->assign_orb(orb2);
320 
321 
322 
323 
324  // ++++++++++++++++++++++++++++++++++++++++ //
325  // Rotate + Translate to MD Frame: Mol&Orb1 //
326  // ++++++++++++++++++++++++++++++++++++++++ //
327 
328  // Rotate + Translate QM frame to MD frame, fragment-wise
329  vector< Fragment* > ::iterator fit;
330  for (fit = seg1->Fragments().begin();
331  fit < seg1->Fragments().end();
332  fit++) {
333 
334  // Centers of maps, rotation matrix MD2QM
335  Fragment *frag = *fit;
336  vec CoMD = frag->getCoMD();
337  vec CoQM = frag->getCoQM();
338  matrix rotQM2MD = frag->getRotQM2MD();
339 
340  // Fill container with atom QM indices
341  vector<int> atmIdcs;
342  vector< Atom* > ::iterator ait;
343  for (ait = frag->Atoms().begin();
344  ait < frag->Atoms().end();
345  ait++) {
346  if ( (*ait)->HasQMPart() ) {
347  atmIdcs.push_back( (*ait)->getQMId()-1 );
348  }
349  }
350 
351 
352  // Perform translation + rotation
353  const double NM2Bohr= 10/0.529189379;
354  morb1->rotate_someatoms_ctp(atmIdcs, rotQM2MD,
355  CoMD*NM2Bohr, CoQM*NM2Bohr,
356  morb1);
357 
358  // Rotate orbitals
359  for (unsigned int i = 0; i < torbNrs1.size(); i++) {
360  orb1->rotate_someatoms(atmIdcs, &rotQM2MD,
361  morb1->getorb(i), i);
362  }
363  }
364 
365  morb1->write_pdb("morbs.pdb", "MOL", 0);
366 
367  // ++++++++++++++++++++++++++++++++++++++++ //
368  // Rotate + Translate to MD Frame: Mol&Orb2 //
369  // ++++++++++++++++++++++++++++++++++++++++ //
370  for (fit = seg2->Fragments().begin();
371  fit < seg2->Fragments().end();
372  fit++) {
373 
374  // Centers of maps, rotation matrix MD2QM
375  Fragment *frag = *fit;
376  vec CoMD = frag->getCoMD();
377  vec CoQM = frag->getCoQM();
378  matrix rotQM2MD = frag->getRotQM2MD();
379  // Fill container with atom QM indices
380  vector<int> atmIdcs;
381  vector< Atom* > ::iterator ait;
382  for (ait = frag->Atoms().begin();
383  ait < frag->Atoms().end();
384  ait++) {
385  if ( (*ait)->HasQMPart() ) {
386  atmIdcs.push_back( (*ait)->getQMId()-1 );
387  }
388  }
389 
390  // Perform translation + rotation
391  const double NM2Bohr= 10/0.529189379;
392  morb2->rotate_someatoms_ctp(atmIdcs, rotQM2MD,
393  CoMD*NM2Bohr, CoQM*NM2Bohr,
394  morb2);
395  // Rotate orbitals
396  for (unsigned int i = 0; i < torbNrs2.size(); i++) {
397  orb2->rotate_someatoms(atmIdcs, &rotQM2MD,
398  morb2->getorb(i), i);
399  }
400  }
401 
402  morb2->write_pdb("morbs.pdb", "MOL", 1);
403  // ++++++++++++++++++++++++++++ //
404  // Calculate transfer integrals //
405  // ++++++++++++++++++++++++++++ //
406 
407  // Initialise Fock matrix //
408  fock12->init(*morb1, *morb2);
409 
410  vector<double> Js;
411  std::pair< int, int > torb2torb;
412 
413  for (unsigned int i = 0; i < torbNrs1.size(); i++) {
414  for (unsigned int j = 0; j < torbNrs2.size(); j++) {
415 
416  torb2torb.first = torbNrs1[i];
417  torb2torb.second = torbNrs2[j];
418 
419  _FockPath.Lock();
420  double J = fock12->calcJ(torb2torb);
421  _FockPath.Unlock();
422 
423  Js.push_back(J);
424 
425  }}
426 
427  pair->setJs(Js, state);
428  pair->setIsPathCarrier(true, state);
429  //cout << endl << "Jeff2 " << pair->getJeff2(state) << " ___ state " << state << endl;
430  //cout << "path " << pair->isPathCarrier(state) << endl;
431 
432  delete morb1;
433  delete morb2;
434  delete basis1;
435  delete basis2;
436  delete orb1;
437  delete orb2;
438  delete fock12;
439 
440  torbNrs1.clear();
441  torbNrs2.clear();
442 
443 }
