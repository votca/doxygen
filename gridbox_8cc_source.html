67  end.push_back(shell->getStartIndex() + shell->getNumFunc());

71 
72  if (start.size() > 1) {
73  startindex.push_back(start[0]);
74 
75  for (Index i = 0; i < Index(start.size()) - 1; ++i) {
76 
77  if (end[i] != start[i + 1]) {
78  startindex.push_back(start[i + 1]);
79  endindex.push_back(end[i]);
80  }
81  }
82  endindex.push_back(end[end.size() - 1]);
83  } else {
