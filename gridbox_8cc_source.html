26 void GridBox::FindSignificantShells(const AOBasis& basis) {
27 
28  for (const AOShell& store : basis) {
29  const double decay = store.getMinDecay();
30  const Eigen::Vector3d& shellpos = store.getPos();
31  for (const auto& point : grid_pos) {
32  Eigen::Vector3d dist = shellpos - point;
33  double distsq = dist.squaredNorm();
34  // if contribution is smaller than -ln(1e-10), add shell to list
35  if ((decay * distsq) < 20.7) {
36  addShell(&store);
37  break;
38  }
39  }
40  }
41 }

103 void GridBox::PrepareForIntegration() {
104  Index index = 0;
105  aoranges = std::vector<GridboxRange>(0);
106  ranges = std::vector<GridboxRange>(0);
107  inv_ranges = std::vector<GridboxRange>(0);
108  std::vector<Index> start;
109  std::vector<Index> end;
110 
111  for (const auto shell : significant_shells) {
112  GridboxRange temp;
113  temp.size = shell->getNumFunc();
114  temp.start = index;
115  aoranges.push_back(temp);
116  index += shell->getNumFunc();
117  start.push_back(shell->getStartIndex());
118  end.push_back(shell->getStartIndex() + shell->getNumFunc());
119  }
120  std::vector<Index> startindex;
121  std::vector<Index> endindex;
122 
123  if (start.size() > 1) {
124  startindex.push_back(start[0]);
125 
126  for (Index i = 0; i < Index(start.size()) - 1; ++i) {
127 
128  if (end[i] != start[i + 1]) {
129  startindex.push_back(start[i + 1]);
130  endindex.push_back(end[i]);
131  }
132  }
133  endindex.push_back(end[end.size() - 1]);
134  } else {
135  startindex = start;
136  endindex = end;
137  }
138  Index shellstart = 0;
139  for (Index i = 0; i < Index(startindex.size()); ++i) {
140  Index size = endindex[i] - startindex[i];
141  GridboxRange temp;
142  temp.size = size;
143  temp.start = startindex[i];
144  ranges.push_back(temp);
145  GridboxRange temp2;
146  temp2.size = size;
147  temp2.start = shellstart;
148  inv_ranges.push_back(temp2);
149  shellstart += size;
150  }
151  return;
152 }
