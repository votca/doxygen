33 void TCMatrix_dft::Fill(const AOBasis& auxbasis, const AOBasis& dftbasis) {
34  {
35  AOCoulomb auxAOcoulomb;
36  auxAOcoulomb.Fill(auxbasis);
37  _inv_sqrt = auxAOcoulomb.Pseudo_InvSqrt(1e-8);
38  _removedfunctions = auxAOcoulomb.Removedfunctions();
39  }
40  _matrix = std::vector<Symmetric_Matrix>(
41  auxbasis.AOBasisSize(), Symmetric_Matrix(dftbasis.AOBasisSize()));
42  Index nthreads = OPENMP::getMaxThreads();
43  std::vector<libint2::Shell> dftshells = dftbasis.GenerateLibintBasis();
44  std::vector<libint2::Shell> auxshells = auxbasis.GenerateLibintBasis();
45  std::vector<libint2::Engine> engines(nthreads);
46  engines[0] = libint2::Engine(
47  libint2::Operator::coulomb,
48  std::max(dftbasis.getMaxNprim(), auxbasis.getMaxNprim()),
49  static_cast<int>(std::max(dftbasis.getMaxL(), auxbasis.getMaxL())), 0);
50  engines[0].set(libint2::BraKet::xs_xx);
51  for (Index i = 1; i < nthreads; ++i) {
52  engines[i] = engines[0];
53  }
54 
55  std::vector<Index> shell2bf = dftbasis.getMapToBasisFunctions();
56  std::vector<Index> auxshell2bf = auxbasis.getMapToBasisFunctions();
57 
58 #pragma omp parallel for schedule(dynamic)
59  for (Index is = dftbasis.getNumofShells() - 1; is >= 0; is--) {
60 
61  libint2::Engine& engine = engines[OPENMP::getThreadId()];
62  const libint2::Engine::target_ptr_vec& buf = engine.results();
63  const libint2::Shell& dftshell = dftshells[is];
64  Index start = shell2bf[is];
65  std::vector<Eigen::MatrixXd> block(dftshell.size());
66  for (Index i = 0; i < Index(dftshell.size()); i++) {
67  Index size = start + i + 1;
68  block[i] = Eigen::MatrixXd::Zero(auxbasis.AOBasisSize(), size);
69  }
70 
71  for (Index aux = 0; aux < auxbasis.getNumofShells(); aux++) {
72  const libint2::Shell& auxshell = auxshells[aux];
73  Index aux_start = auxshell2bf[aux];
74 
75  for (Index dis = 0; dis <= is; dis++) {
76 
77  const libint2::Shell& shell_col = dftshells[dis];
78  Index col_start = shell2bf[dis];
79  engine.compute2<libint2::Operator::coulomb, libint2::BraKet::xs_xx, 0>(
80  auxshell, libint2::Shell::unit(), dftshell, shell_col);
81 
82  if (buf[0] == nullptr) {
83  continue;
84  }
85  Eigen::TensorMap<Eigen::Tensor<const double, 3, Eigen::RowMajor> const>
86  result(buf[0], auxshell.size(), dftshell.size(), shell_col.size());
87 
88  for (size_t left = 0; left < dftshell.size(); left++) {
89  for (size_t auxf = 0; auxf < auxshell.size(); auxf++) {
90  for (size_t col = 0; col < shell_col.size(); col++) {
91  // symmetry
92  if ((col_start + col) > (start + left)) {
93  break;
94  }
95  block[left](aux_start + auxf, col_start + col) =
96  result(auxf, left, col);
97  }
98  }
99  }
100  }
101  }
102 
103  for (Index i = 0; i < Index(block.size()); ++i) {
104  Eigen::MatrixXd temp = _inv_sqrt * block[i];
105  for (Index mu = 0; mu < temp.rows(); ++mu) {
106  for (Index j = 0; j < temp.cols(); ++j) {
107  _matrix[mu](i + start, j) = temp(mu, j);
108  }
109  }
110  }
111  }
112 
113  return;
114 }

146 /*
147  * Fill the 3-center object by looping over shells of GW basis set and
148  * calling FillBlock, which calculates all 3-center overlap integrals
149  * associated to a particular shell, convoluted with the DFT orbital
150  * coefficients
151  */
152 void TCMatrix_gwbse::Fill(const AOBasis& auxbasis, const AOBasis& dftbasis,
153  const Eigen::MatrixXd& dft_orbitals) {
154  // needed for Rebuild())
155  _auxbasis = &auxbasis;
156  _dftbasis = &dftbasis;
157  _dft_orbitals = &dft_orbitals;
158 
159  Fill3cMO(auxbasis, dftbasis, dft_orbitals);
160 
161  AOOverlap auxoverlap;
162  auxoverlap.Fill(auxbasis);
163  AOCoulomb auxcoulomb;
164  auxcoulomb.Fill(auxbasis);
165  Eigen::MatrixXd inv_sqrt = auxcoulomb.Pseudo_InvSqrt_GWBSE(auxoverlap, 5e-7);
166  _removedfunctions = auxcoulomb.Removedfunctions();
167  MultiplyRightWithAuxMatrix(inv_sqrt);
168 
169  return;
170 }

172 /*
173  * Determines the 3-center integrals for a given shell in the aux basis
174  * by calculating the 3-center repulsion integral of the functions in the
175  * aux shell with ALL functions in the DFT basis set
176  */
177 std::vector<Eigen::MatrixXd> TCMatrix_gwbse::ComputeAO3cBlock(
178  const libint2::Shell& auxshell, const AOBasis& dftbasis,
179  libint2::Engine& engine) const {
180  std::vector<Eigen::MatrixXd> ao3c = std::vector<Eigen::MatrixXd>(
181  auxshell.size(),
182  Eigen::MatrixXd::Zero(dftbasis.AOBasisSize(), dftbasis.AOBasisSize()));
183 
184  std::vector<libint2::Shell> dftshells = dftbasis.GenerateLibintBasis();
185  std::vector<Index> shell2bf = dftbasis.getMapToBasisFunctions();
186 
187  const libint2::Engine::target_ptr_vec& buf = engine.results();
188  // alpha-loop over the "left" DFT basis function
189  for (Index row = 0; row < Index(dftshells.size()); row++) {
190 
191  const libint2::Shell& shell_row = dftshells[row];
192  const Index row_start = shell2bf[row];
193  // ThreecMatrix is symmetric, restrict explicit calculation to triangular
194  // matrix
195  for (Index col = 0; col <= row; col++) {
196  const libint2::Shell& shell_col = dftshells[col];
197  const Index col_start = shell2bf[col];
198 
199  engine.compute2<libint2::Operator::coulomb, libint2::BraKet::xs_xx, 0>(
200  auxshell, libint2::Shell::unit(), shell_col, shell_row);
201 
202  if (buf[0] == nullptr) {
203  continue;
204  }
205  Eigen::TensorMap<Eigen::Tensor<const double, 3, Eigen::RowMajor> const>
206  result(buf[0], auxshell.size(), shell_col.size(), shell_row.size());
207 
208  for (size_t aux_c = 0; aux_c < auxshell.size(); aux_c++) {
209  for (size_t col_c = 0; col_c < shell_col.size(); col_c++) {
210  for (size_t row_c = 0; row_c < shell_row.size(); row_c++) {
211  // ao3c is col major result is row-major so this is ideal
212  ao3c[aux_c](row_start + row_c, col_start + col_c) =
213  result(aux_c, col_c, row_c);
214  } // ROW copy
215  } // COL copy
216  } // AUX copy
217 
218  } // gamma-loop
219  } // alpha-loop
220 
221  for (Eigen::MatrixXd& mat : ao3c) {
222  mat.triangularView<Eigen::Upper>() =
223  mat.triangularView<Eigen::Lower>().transpose();
224  }
225  return ao3c;
226 }

228 void TCMatrix_gwbse::Fill3cMO(const AOBasis& auxbasis, const AOBasis& dftbasis,
229  const Eigen::MatrixXd& dft_orbitals) {
230 
231  const Eigen::MatrixXd dftm =
232  dft_orbitals.block(0, _mmin, dft_orbitals.rows(), _mtotal);
233  const Eigen::MatrixXd dftn =
234  dft_orbitals.block(0, _nmin, dft_orbitals.rows(), _ntotal).transpose();
235 
236  OpenMP_CUDA transform;
237  transform.setOperators(dftn, dftm);
238  Index nthreads = OPENMP::getMaxThreads();
239 
240  std::vector<libint2::Shell> auxshells = auxbasis.GenerateLibintBasis();
241  std::vector<libint2::Engine> engines(nthreads);
242  engines[0] = libint2::Engine(
243  libint2::Operator::coulomb,
244  std::max(dftbasis.getMaxNprim(), auxbasis.getMaxNprim()),
245  static_cast<int>(std::max(dftbasis.getMaxL(), auxbasis.getMaxL())), 0);
246  engines[0].set(libint2::BraKet::xs_xx);
247  for (Index i = 1; i < nthreads; ++i) {
248  engines[i] = engines[0];
249  }
250  std::vector<Index> auxshell2bf = auxbasis.getMapToBasisFunctions();
251 
252 #pragma omp parallel for schedule(dynamic)
253  for (Index aux = 0; aux < Index(auxshells.size()); aux++) {
254  const libint2::Shell& auxshell = auxshells[aux];
255 
256  std::vector<Eigen::MatrixXd> ao3c =
257  ComputeAO3cBlock(auxshell, dftbasis, engines[OPENMP::getThreadId()]);
258 
259  // this is basically a transpose of AO3c and at the same time the ao->mo
260  // transformation
261  // we do not want to put it into _matrix straight away is because, _matrix
262  // is shared between all threads and we want a nice clean access pattern to
263  // it
264  std::vector<Eigen::MatrixXd> block = std::vector<Eigen::MatrixXd>(
265  _mtotal, Eigen::MatrixXd::Zero(_ntotal, ao3c.size()));
266 
267  Index dim = static_cast<Index>(ao3c.size());
268  for (Index k = 0; k < dim; ++k) {
269  transform.MultiplyLeftRight(ao3c[k]);
270  for (Index i = 0; i < ao3c[k].cols(); ++i) {
271  block[i].col(k) = ao3c[k].col(i);
272  }
273  }
274 
275  // put into correct position
276  for (Index m_level = 0; m_level < _mtotal; m_level++) {
277  _matrix[m_level].block(0, auxshell2bf[aux], _ntotal, auxshell.size()) =
278  block[m_level];
279  } // m-th DFT orbital
280  } // shells of GW basis set
281 }
