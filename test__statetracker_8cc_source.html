38BOOST_AUTO_TEST_CASE(osc) {
39 libint2::initialize();
40 Orbitals orb;
41 orb.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
42 "/statetracker/molecule.xyz");
43 orb.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) +
44 "/statetracker/3-21G.xml");
45 Logger log;
46 orb.setNumberOfOccupiedLevels(4);
47 orb.setBSEindices(0, 16);
48 orb.setTDAApprox(true);
49
50 Eigen::MatrixXd& MOs = orb.MOs().eigenvectors();
51
52 orb.MOs().eigenvalues() = Eigen::VectorXd::Ones(17);
53 MOs = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
54 std::string(XTP_TEST_DATA_FOLDER) + "/statetracker/MOs.mm");
55
56 Eigen::VectorXd se_ref = Eigen::VectorXd::Zero(3);
57 se_ref << 0.107455, 0.107455, 0.107455;
58 orb.BSESinglets().eigenvalues() = se_ref;
59
60 // reference coefficients
61 Eigen::MatrixXd spsi_ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
62 std::string(XTP_TEST_DATA_FOLDER) + "/statetracker/spsi_ref.mm");
63
64 orb.BSESinglets().eigenvectors() = spsi_ref;
65 orb.CalcCoupledTransition_Dipoles();
66
67 {
68 votca::tools::Property prop;
69 prop.LoadFromXML(std::string(XTP_TEST_DATA_FOLDER) +
70 "/statetracker/statetracker.xml");
71 StateTracker tracker;
72 tracker.setLogger(&log);
73 QMState s("s1");
74 tracker.setInitialState(s);
75 tracker.Initialize(prop.get("statetracker"));
76 QMState newstate = tracker.CalcState(orb);
77 BOOST_CHECK_EQUAL(newstate.Type().ToString(), "s");
78 BOOST_CHECK_EQUAL(newstate.StateIdx(), 1);
79 }
80 libint2::finalize();
81}
