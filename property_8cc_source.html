43 const Property &Property::get(const string &key) const {
44  Tokenizer tok(key, ".");
45  Tokenizer::iterator n = tok.begin();
46  if (n == tok.end()) {
47  return *this;
48  }
49 
50  const Property *p;
51  map<string, Index>::const_iterator iter;
52  if (*n == "") {
53  p = this;
54  } else {
55  iter = _map.find(*n);
56  if (iter == _map.end()) {
57  throw std::runtime_error("property not found: " + key);
58  }
59  p = &_properties[((*iter).second)];
60  }
61  ++n;
62  try {
63  for (; n != tok.end(); ++n) {
64  p = &p->get(*n);
65  }
66  } catch (std::runtime_error &) { // catch here to get full key in
67  // exception
68  throw std::runtime_error("property not found: " + key);
69  }
70 
71  return *p;
72 }

86 std::vector<const Property *> Property::Select(const string &filter) const {
87  Tokenizer tok(filter, ".");
88  std::vector<const Property *> selection;
89  if (tok.begin() == tok.end()) {
90  return selection;
91  }
92  selection.push_back(this);
93  for (const auto &n : tok) {
94  std::vector<const Property *> childs;
95  for (const Property *p : selection) {
96  for (const Property &s : p->_properties) {
97  if (wildcmp(n.c_str(), s.name().c_str())) {
98  childs.push_back(&s);
99  }
100  }
101  }
102  selection = childs;
103  }
104  return selection;
105 }

156 void Property::LoadFromXML(string filename) {
157  ifstream fl;
158  fl.open(filename);
159  if (!fl.is_open()) {
160  throw std::ios_base::failure("Error on open xml file: " + filename);
161  }
162 
163  XML_Parser parser = XML_ParserCreate(nullptr);
164  if (!parser) {
165  throw std::runtime_error("Couldn't allocate memory for xml parser");
166  }
167 
168  XML_UseParserAsHandlerArg(parser);
169  XML_SetElementHandler(parser, start_hndl, end_hndl);
170  XML_SetCharacterDataHandler(parser, char_hndl);
171 
172  stack<Property *> pstack;
173  pstack.push(this);
174 
175  XML_SetUserData(parser, (void *)&pstack);
176  while (!fl.eof()) {
177  string line;
178  getline(fl, line);
179  line = line + "\n";
180  if (line.length() > (size_t)std::numeric_limits<int>::max()) {
181  throw std::runtime_error("Property::LoadFromXML: line is too long");
182  }
183  if (!XML_Parse(parser, line.c_str(), (int)line.length(), fl.eof())) {
184  throw std::ios_base::failure(
185  filename + ": Parse error at line " +
186  boost::lexical_cast<string>(XML_GetCurrentLineNumber(parser)) + "\n" +
187  XML_ErrorString(XML_GetErrorCode(parser)));
188  }
189  }
190  fl.close();
191  XML_ParserFree(parser);
192 }

196  if ((p.value() != "") || p.HasChildren()) {
197  if (level >= start_level) {

199  out << offset << prefix << " = " << p.value() << endl;

201  } else {

204  }

207  level++;

218 void PrintNodeXML(std::ostream &out, const Property &p,
219  PropertyIOManipulator *piom, Index level = 0,
220  string offset = "") {
221  Property::const_AttributeIterator ia;
222  bool linebreak = true;
223  bool has_value = false;
224 
225  const ColorSchemeBase *color = &DEFAULT_COLORS;
226 
227  string indent("");
228  Index start_level(0);
229 
230  if (piom) {
231  start_level = piom->getLevel();
232  indent = piom->getIndentation();
233  color = piom->getColorScheme();
234  }
235 
236  string cKey = color->Magenta();
237  string cAttribute = color->Blue();
238  string cAttributeValue = color->Green();
239  string cReset = color->Reset();
240 
241  // print starting only from the start_level (the first node (level 0) can be
242  // <> </>)
243  if (level >= start_level) {
244  // print the node name
245  out << indent << offset << "<" << cKey << p.name() << cReset;
246  // print the node attributes
247  for (ia = p.firstAttribute(); ia != p.lastAttribute(); ++ia) {
248  out << " " << cAttribute << ia->first << cReset << "=\""
249  << cAttributeValue << ia->second << cReset << "\"";
250  }
251  // print node value if it is not empty
252  has_value = ((p.value()).find_first_not_of("\t\n ") != std::string::npos);
253  if (has_value || p.HasChildren()) {
254  out << ">";
255  } else {
256  out << "/>" << std::endl;
257  }
258  if (has_value) {
259  out << cAttributeValue << p.value() << cReset;
260  linebreak = false;
261  }
262 
263  // check if we need the end of the line or not
264  if (!has_value && p.HasChildren()) {
265  out << std::endl;
266  }
267  if (!has_value && !p.HasChildren()) {
268  linebreak = false;
269  }
270  }
271 
272  // continue iteratively through the rest of the nodes
273  for (const Property &prop : p) {
274  level++;
275  if (level > start_level) {
276  offset += "\t";
277  }
278  PrintNodeXML(out, prop, piom, level, offset);
279  if (level > start_level) {
280  offset.resize(offset.size() - 1);
281  }
282  level--;
283  }
284 
285  if (level >= start_level) {
286  if (linebreak) {
287  out << indent << offset << "</" << cKey << p.name() << cReset << ">"
288  << std::endl;
289  } else if (has_value) {
290  out << "</" << cKey << p.name() << cReset << ">" << std::endl;
291  }
292  }

295 void PrintNodeTEX(std::ostream &out, const Property &p,
296  PropertyIOManipulator *piom, Index level = 0,
297  string prefix = "") {
298 
299  Index start_level = 0;
300  if (piom) {
301  start_level = piom->getLevel();
302  }
303 
304  string head_name;
305  string section(""); // reference of the description section in the manual
306  string help("");
307  // if this is the head node, print the header
308  if (level == start_level) {
309 
310  string header_format(
311  "\\subsection{%1%}\n"
312  "\\label{%2%}\n%3%\n"
313  "\\rowcolors{1}{invisiblegray}{white}\n"
314  "{\\small\n "
315  "\\begin{longtable}{m{3cm}|m{2cm}|m{1cm}|m{8cm}}\n"
316  " option & default & unit & description\\\\\n\\hline\n");
317 
318  head_name = p.name();
319  string label =
320  "calc:" + head_name; // reference of the xml file in the manual
321  if (p.hasAttribute("section")) {
322  section = p.getAttribute<string>("section");
323  }
324  if (p.hasAttribute("help")) {
325  help = p.getAttribute<string>("help");
326  }
327  out << boost::format(header_format) % head_name % label % help;
328  prefix = p.name();
329  }
330 
331  if (level > start_level) {
332  // if this node has children or a value or is not the first, start recursive
333  // printing
334  if ((p.value() != "" || p.HasChildren()) && level > -1) {
335  string tex_name = boost::replace_all_copy(p.name(), "_", "\\_");
336  string defaults(""); // default value if supplied
337  if (p.hasAttribute("default")) {
338  defaults = p.getAttribute<string>("default");
339  }
340  string unit(""); // unit, if supplied
341  if (p.hasAttribute("unit")) {
342  unit = p.getAttribute<string>("unit");
343  }
344  if (p.hasAttribute("help")) {
345  help = p.getAttribute<string>("help");
346  }
347 
348  string body_format(
349  " \\hspace{%1%pt}\\hypertarget{%2%}{%3%} & %4% & %5% & %6% \\\\\n");
350 
351  out << boost::format(body_format) %
352  Index((level - start_level - 1) * 10) % prefix % tex_name %
353  defaults % unit % help;
354  }
355  }
356 
357  // continue iteratively through the rest of the nodes
358  for (const Property &pp : p) {
359  level++;
360  if (prefix == "") {
361  PrintNodeTEX(out, pp, piom, level, prefix);
362  } else {
363  PrintNodeTEX(out, pp, piom, level, prefix);
364  }
365  level--;
366  }
367 
368  // if this is the head node, print the footer
369  if (level == start_level) {
370  string footer_format(
371  "\\end{longtable}\n}\n"
372  "\\noindent Return to the description of "
373  "\\slink{%1%}{\\texttt{%2%}}.\n");
374 
375  out << boost::format(footer_format) % section % head_name;
376  }

379 void PrintNodeHLP(std::ostream &out, const Property &p,
380  const Index start_level = 0, Index level = 0,
381  const string &prefix = "", const string &offset = "") {
382 
383  using ColorRGB = Color<csRGB>; // use the RGB palette
384  ColorRGB RGB; // Instance of an RGB palette
385  string fmt = "t|%1%%|15t|" + string(RGB.Blue()) + "%2%" +
386  string(RGB.Green()) + "%|40t|%3%%|55t|" + string(RGB.Reset()) +
387  "%4%\n";
388 
389  Index leveloffset = level;
390  string help("");
391  // if this is the head node, print the header
392  if (level == start_level) {
393  string head_name = string(RGB.Magenta()) + p.name();
394  if (p.hasAttribute("help")) {
395  if (p.hasAttribute("help")) {
396  help = string(RGB.Red()) + p.getAttribute<string>("help");
397  }
398  out << boost::format(" %1%: %|18t| %2%" + string(RGB.Reset()) + "\n") %
399  head_name % help;
400  }
401  leveloffset = 0;
402  out << boost::format("%|3" + fmt) % "OPTION" % "DEFAULT" % "UNIT" %
403  "DESCRIPTION";
404  }
405 
406  if (level > start_level) {
407  string ofmt = "%|" + boost::lexical_cast<string>(leveloffset) + fmt;
408  string unit("");
409  if (p.hasAttribute("unit")) {
410  unit = p.getAttribute<string>("unit");
411  }
412  string defaults("");
413  if (p.hasAttribute("default")) {
414  defaults = p.getAttribute<string>("default");
415  }
416  if (p.hasAttribute("help")) {
417  help = p.getAttribute<string>("help");
418  }
419  if (!unit.empty()) {
420  unit = "[" + unit + "]";
421  }
422  if (!defaults.empty()) {
423  defaults = "(" + defaults + ")";
424  }
425 
426  string name = p.name();
427 
428  out << boost::format(ofmt) % name % defaults % unit % help;
429  }
430

432  level++;
433  if (prefix == "") {
434  PrintNodeHLP(out, pp, start_level, level, pp.name(), offset);
435  } else {
436  PrintNodeHLP(out, pp, start_level, level, prefix + "." + pp.name(),
437  offset);
438  }
439  level--;
440  }
441 }

443 std::ostream &operator<<(std::ostream &out, const Property &p) {
444  if (!out.good()) {
445  return out;
446  }
447 
448  std::ostream::sentry sentry(out);
449 
450  if (sentry) {
451  // get the property format object attached to the stream
452  PropertyIOManipulator *pm =
453  (PropertyIOManipulator *)out.pword(int(Property::getIOindex()));
454 
455  string indentation("");
456  Index level = 0;
457 
458  PropertyIOManipulator::Type type = PropertyIOManipulator::XML;
459  if (pm) {
460  indentation = pm->getIndentation();
461  level = pm->getLevel();
462  type = pm->getType();
463  // check if we > or >> to a file and remove color codes
464  // if ( out.tellp() != -1 ) - not suitable for pipes
465  if (!isatty(STDOUT_FILENO) || !isatty(STDERR_FILENO)) {
466  pm->setColorScheme<csDefault>();
467  }
468  }
469 
470  switch (type) {
471  default:
472  PrintNodeTXT(out, p, level);
473  break;
474  case PropertyIOManipulator::XML:
475  PrintNodeXML(out, p, pm);
476  break;
477  case PropertyIOManipulator::TXT:
478  PrintNodeTXT(out, p, level, 0, "", indentation);
479  break;
480  case PropertyIOManipulator::TEX:
481  PrintNodeTEX(out, p, pm);
482  break;
483  case PropertyIOManipulator::HLP:
484  PrintNodeHLP(out, p, level, 0, "", indentation);
485  break;
486  }
487  }
488 
489  return out;
490 }
