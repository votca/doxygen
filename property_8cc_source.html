40 Property &Property::get(const string &key)
41 {
42  Tokenizer tok(key, ".");
43  Tokenizer::iterator n;
44 
45  n = tok.begin();
46  if(n==tok.end()) return *this;
47 
48  Property *p;
49  map<string, Property*>::iterator iter;
50  if(*n=="") {
51  p = this;
52  }
53  else {
54  iter = _map.find(*n);
55 
56  if(iter == _map.end())
57  throw runtime_error("property not found: " + key);
58 
59  p = (((*iter).second));
60  }
61  ++n;
62  try {
63  for(; n!=tok.end(); ++n) {
64  p = &p->get(*n);
65  }
66  }
67  catch(string err) { // catch here to get full key in exception
68  throw runtime_error("property not found: " + key);
69  }
70 
71  return *p;
72 }

74 std::list<Property *> Property::Select(const string &filter)
75 {
76  Tokenizer tok(filter, ".");
77 
78  std::list<Property *> selection;
79 
80  if(tok.begin()==tok.end()) return selection;
81 
82  selection.push_back(this);
83 
84  for (Tokenizer::iterator n = tok.begin();
85  n != tok.end(); ++n) {
86  std::list<Property *> childs;
87  for (std::list<Property *>::iterator p = selection.begin();
88  p != selection.end(); ++p) {
89  for (list<Property>::iterator iter = (*p)->_properties.begin();
90  iter != (*p)->_properties.end(); ++iter) {
91  if (wildcmp((*n).c_str(), (*iter).name().c_str())) {
92  childs.push_back(&(*iter));
93  }
94  }
95  }
96  selection = childs;
97  }
98 
99  return selection;
100 }

132 bool load_property_from_xml(Property &p, string filename)
133 {
134  XML_Parser parser = XML_ParserCreate(NULL);
135  if (! parser)
136  throw std::runtime_error("Couldn't allocate memory for xml parser");
137 
138  XML_UseParserAsHandlerArg(parser);
139  XML_SetElementHandler(parser, start_hndl, end_hndl);
140  XML_SetCharacterDataHandler(parser, char_hndl);
141 
142  ifstream fl;
143  fl.open(filename.c_str());
144  if(!fl.is_open())
145  throw std::ios_base::failure("Error on open xml file: " + filename);
146 
147  stack<Property *> pstack;
148  pstack.push(&p);
149 
150  XML_SetUserData(parser, (void*)&pstack);
151  while(!fl.eof()) {
152  string line;
153  getline(fl, line);
154  line=line + "\n";
155  if (! XML_Parse(parser, line.c_str(), line.length(), fl.eof()))
156  throw std::ios_base::failure(filename + ": Parse error at line " +
157  boost::lexical_cast<string>(XML_GetCurrentLineNumber(parser)) + "\n" +
158  XML_ErrorString(XML_GetErrorCode(parser)));
159  }
160  fl.close();
161  return true;
162 }

193 void PrintNodeXML(std::ostream &out, Property &p, const int start_level, int level, string prefix, string offset)
194 {
195  list<Property>::iterator iter;
196  Property::AttributeIterator ia;
197  bool _endl = true;
198 
199  // print starting only from the start_level (the first node (level 0) can be <> </>)
200  if ( level >= start_level ) {
201  // print the node name
202  out << offset << "<" << p.name() ;
203  // print the node attributes
204  for(ia = p.firstAttribute(); ia!=p.lastAttribute(); ++ia)
205  out << " " << ia->first << "=\"" << ia->second << "\"" ;
206  out << ">";
207 
208  // print node value if it is not empty
209  bool has_value = ( (p.value()).find_first_not_of("\t\n ") != std::string::npos );
210  if( has_value ) { out << p.value(); _endl = false; }
211 
212  // check if we need the end of the line or not
213  if( !has_value && p.HasChilds() ) out << endl;
214  if( !has_value && !p.HasChilds() ) _endl = false;
215  }
216 
217  // continue iteratively through the rest of the nodes
218  for(iter = p.begin(); iter!=p.end(); ++iter) {
219  level++;
220  if ( level > start_level ) offset += "\t";
221  PrintNodeXML(out, (*iter), start_level, level, (*iter).name(), offset);
222  if ( level > start_level ) offset.resize(offset.size()-1);
223  level--;
224  }
225 
226  if ( level >= start_level ) {
227  if ( _endl ) {
228  out << offset << "</" << p.name() << ">" << endl;
229  } else {
230  out << "</" << p.name() << ">" << endl;
231  }
232  }

363 std::ostream &operator<<(std::ostream &out, Property& p)
364 {
365  if (!out.good())
366  return out;
367 
368  std::ostream::sentry sentry(out);
369 
370  if(sentry)
371  {
372  // get the property format object attached to the stream
373  PropertyIOManipulator *pm =
374  (PropertyIOManipulator*)out.pword(Property::getIOindex());
375 
376  string _indentation = pm->getIndentation();
377  int _level = pm->getLevel();
378  PropertyIOManipulator::Type _type = pm->getType();
379 
380  switch( _type )
381  {
382  default:
383  PrintNodeTXT(out, p, _level);
384  case PropertyIOManipulator::XML:
385  PrintNodeXML(out, p, _level, 0, "", _indentation);
386  break;
387  case PropertyIOManipulator::TXT:
388  PrintNodeTXT(out, p, _level, 0, "", _indentation);
389  break;
390  case PropertyIOManipulator::TEX:
391  PrintNodeTEX(out, p, _level);
392  break;
393  case PropertyIOManipulator::HLP:
394  PrintNodeHLP(out, p, _level, 0, "", _indentation);
395  break;
396  }
397  //out << endl;
398  }
399 
400  return out;
