00081 
00082     if(tok.begin()==tok.end()) return selection;
00083     
00084     selection.push_back(this);
00085         
00086     for (Tokenizer::iterator n = tok.begin();
00087             n != tok.end(); ++n) {
00088         std::list<Property *> childs;
00089         for (std::list<Property *>::iterator p = selection.begin();
00090                 p != selection.end(); ++p) {
00091                 for (list<Property>::iterator iter = (*p)->_properties.begin();
00092                     iter != (*p)->_properties.end(); ++iter) {
00093                     if (wildcmp((*n).c_str(), (*iter).name().c_str())) {
00094                         childs.push_back(&(*iter));
00095                     }
00096                 }
00097         }
00098         selection = childs;        
00099     }
00100 
00101     return selection;
