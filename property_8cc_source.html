43 Property &Property::get(const string &key) {
44  Tokenizer tok(key, ".");
45  Tokenizer::iterator n;
46 
47  n = tok.begin();
48  if (n == tok.end()) return *this;
49 
50  Property *p;
51  map<string, Property *>::iterator iter;
52  if (*n == "") {
53  p = this;
54  } else {
55  iter = _map.find(*n);
56 
57  if (iter == _map.end()) throw runtime_error("property not found: " + key);
58 
59  p = (((*iter).second));
60  }
61  ++n;
62  try {
63  for (; n != tok.end(); ++n) {
64  p = &p->get(*n);
65  }
66  } catch (string err) { // catch here to get full key in exception
67  throw runtime_error("property not found: " + key);
68  }
69 
70  return *p;
71 }

73 std::list<Property *> Property::Select(const string &filter) {
74  Tokenizer tok(filter, ".");
75 
76  std::list<Property *> selection;
77 
78  if (tok.begin() == tok.end()) return selection;
79 
80  selection.push_back(this);
81 
82  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
83  std::list<Property *> childs;
84  for (std::list<Property *>::iterator p = selection.begin();
85  p != selection.end(); ++p) {
86  for (list<Property>::iterator iter = (*p)->_properties.begin();
87  iter != (*p)->_properties.end(); ++iter) {
88  if (wildcmp((*n).c_str(), (*iter).name().c_str())) {
89  childs.push_back(&(*iter));
90  }
91  }
92  }
93  selection = childs;
94  }
95 
96  return selection;
97 }

125 bool load_property_from_xml(Property &p, string filename) {
126  ifstream fl;
127  fl.open(filename.c_str());
128  if (!fl.is_open())
129  throw std::ios_base::failure("Error on open xml file: " + filename);
130 
131  XML_Parser parser = XML_ParserCreate(NULL);
132  if (!parser)
133  throw std::runtime_error("Couldn't allocate memory for xml parser");
134 
135  XML_UseParserAsHandlerArg(parser);
136  XML_SetElementHandler(parser, start_hndl, end_hndl);
137  XML_SetCharacterDataHandler(parser, char_hndl);
138 
139  stack<Property *> pstack;
140  pstack.push(&p);
141 
142  XML_SetUserData(parser, (void *)&pstack);
143  while (!fl.eof()) {
144  string line;
145  getline(fl, line);
146  line = line + "\n";
147  if (!XML_Parse(parser, line.c_str(), line.length(), fl.eof()))
148  throw std::ios_base::failure(
149  filename + ": Parse error at line " +
150  boost::lexical_cast<string>(XML_GetCurrentLineNumber(parser)) + "\n" +
151  XML_ErrorString(XML_GetErrorCode(parser)));
152  }
153  fl.close();
154  XML_ParserFree(parser);
155  return true;
156 }

159  int level = 0, string prefix = "", string offset = "") {
160  if ((p.value() != "") || p.HasChilds()) {
161  if (level >= start_level) {
162  if ((p.value()).find_first_not_of("\t\n ") != std::string::npos)
163  out << offset << prefix << " = " << p.value() << endl;
164  } else {
165  prefix = "";
166  }
167  }
168 
169  for (Property &prop : p) {
170  if (prefix == "") {
171  level++;
172  PrintNodeTXT(out, prop, start_level, level, prefix + prop.name(), offset);
173  level--;
174  } else {
175  level++;
176  PrintNodeTXT(out, prop, start_level, level, prefix + "." + prop.name(),

178  level--;
179  }

183 void PrintNodeXML(std::ostream &out, Property &p, PropertyIOManipulator *piom,
184  int level = 0, string offset = "") {
185  Property::AttributeIterator ia;
186  bool linebreak = true;
187  bool has_value = false;
188 
189  const ColorSchemeBase *color = &DEFAULT_COLORS;
190 
191  string indent("");
192  int start_level(0);
193 
194  if (piom) {
195  start_level = piom->getLevel();
196  indent = piom->getIndentation();
197  color = piom->getColorScheme();
198  }
199 
200  string cKey = color->Magenta();
201  string cAttribute = color->Blue();
202  string cAttributeValue = color->Green();
203  string cReset = color->Reset();
204 
205  // print starting only from the start_level (the first node (level 0) can be
206  // <> </>)
207  if (level >= start_level) {
208  // print the node name
209  out << indent << offset << "<" << cKey << p.name() << cReset;
210  // print the node attributes
211  for (ia = p.firstAttribute(); ia != p.lastAttribute(); ++ia) {
212  out << " " << cAttribute << ia->first << cReset << "=\""
213  << cAttributeValue << ia->second << cReset << "\"";
214  }
215  // print node value if it is not empty
216  has_value = ((p.value()).find_first_not_of("\t\n ") != std::string::npos);
217  if (has_value || p.HasChilds()) {
218  out << ">";
219  } else {
220  out << "/>" << std::endl;
221  }
222  if (has_value) {
223  out << cAttributeValue << p.value() << cReset;
224  linebreak = false;
225  }
226 
227  // check if we need the end of the line or not
228  if (!has_value && p.HasChilds()) out << std::endl;
229  if (!has_value && !p.HasChilds()) linebreak = false;
230  }
231 
232  // continue iteratively through the rest of the nodes
233  for (Property &prop : p) {
234  level++;
235  if (level > start_level) offset += "\t";
236  PrintNodeXML(out, prop, piom, level, offset);
237  if (level > start_level) offset.resize(offset.size() - 1);
238  level--;
239  }
240 
241  if (level >= start_level) {
242  if (linebreak) {
243  out << indent << offset << "</" << cKey << p.name() << cReset << ">"
244  << std::endl;
245  } else if (has_value) {
246  out << "</" << cKey << p.name() << cReset << ">" << std::endl;
247  }
248  }

251 void PrintNodeTEX(std::ostream &out, Property &p, PropertyIOManipulator *piom,
252  int level = 0, string prefix = "") {
253  list<Property>::iterator iter;
254  string head_name;
255  string _label(""); // reference of the xml file in the manual
256  string _section(""); // reference of the description section in the manual
257  string _help("");
258  string _default(""); // default value if supplied
259  string _unit(""); // unit, if supplied
260 
261  int start_level(0);
262 
263  if (piom) {
264  start_level = piom->getLevel();
265  }
266 
267  string header_format(
268  "\\subsection{%1%}\n"
269  "\\label{%2%}\n%3%\n"
270  "\\rowcolors{1}{invisiblegray}{white}\n"
271  "{\\small\n "
272  "\\begin{longtable}{m{3cm}|m{2cm}|m{1cm}|m{8cm}}\n"
273  " option & default & unit & description\\\\\n\\hline\n");
274 
275  string footer_format(
276  "\\end{longtable}\n}\n"
277  "\\noindent Return to the description of \\slink{%1%}{\\texttt{%2%}}.\n");
278 
279  string body_format(
280  " \\hspace{%1%pt}\\hypertarget{%2%}{%3%} & %4% & %5% & %6% \\\\\n");
281 
282  // if this is the head node, print the header
283  if (level == start_level) {
284  head_name = p.name();
285  _label = "calc:" + head_name;
286  if (p.hasAttribute("section")) _section = p.getAttribute<string>("section");
287  if (p.hasAttribute("help")) _help = p.getAttribute<string>("help");
288  out << boost::format(header_format) % head_name % _label % _help;
289  prefix = p.name();
290  }
291 
292  if (level > start_level) {
293  // if this node has children or a value or is not the first, start recursive
294  // printing
295  if ((p.value() != "" || p.HasChilds()) && level > -1) {
296  string _tex_name = boost::replace_all_copy(p.name(), "_", "\\_");
297 
298  if (p.hasAttribute("default"))
299  _default = p.getAttribute<string>("default");
300  if (p.hasAttribute("unit")) _unit = p.getAttribute<string>("unit");
301  if (p.hasAttribute("help")) _help = p.getAttribute<string>("help");
302 
303  out << boost::format(body_format) % int((level - start_level - 1) * 10) %
304  prefix % _tex_name % _default % _unit % _help;
305  }
306  }
307 
308  // continue iteratively through the rest of the nodes
309  for (iter = p.begin(); iter != p.end(); ++iter) {
310  if (prefix == "") {
311  level++;
312  PrintNodeTEX(out, (*iter), piom, level, prefix);
313  level--;
314  } else {
315  level++;
316  PrintNodeTEX(out, (*iter), piom, level, prefix);
317  level--;
318  }
319  }
320 
321  // if this is the head node, print the footer
322  if (level == start_level)
323  out << boost::format(footer_format) % _section % head_name;
324 }

326 void PrintNodeHLP(std::ostream &out, Property &p, const int start_level = 0,
327  int level = 0, string prefix = "", string offset = "") {
328  list<Property>::iterator iter;
329  string head_name;
330  string _help("");
331  string _unit("");
332  string _default("");
333  string _name("");
334 
335  typedef Color<csRGB> ColorRGB; // use the RGB palette
336  ColorRGB RGB; // Instance of an RGB palette
337  string fmt = "t|%1%%|15t|" + string(RGB.Blue()) + "%2%" +
338  string(RGB.Green()) + "%|40t|%3%%|55t|" + string(RGB.Reset()) +
339  "%4%\n";
340 
341  int _offset = level;
342 
343  // if this is the head node, print the header
344  if (level == start_level) {
345  head_name = string(RGB.Magenta()) + p.name();
346  if (p.hasAttribute("help")) {
347  if (p.hasAttribute("help"))
348  _help = string(RGB.Red()) + p.getAttribute<string>("help");
349  out << boost::format(" %1%: %|18t| %2%" + string(RGB.Reset()) + "\n") %
350  head_name % _help;
351  }
352  _offset = 0;
353  out << boost::format("%|3" + fmt) % "OPTION" % "DEFAULT" % "UNIT" %
354  "DESCRIPTION";
355  }
356 
357  if (level > start_level) {
358  string ofmt;
359  ofmt = "%|" + boost::lexical_cast<string>(_offset) + fmt;
360  // cout << ofmt << " " << fmt << endl;
361  if (p.hasAttribute("unit")) _unit = p.getAttribute<string>("unit");
362  if (p.hasAttribute("default")) _default = p.getAttribute<string>("default");
363  if (p.hasAttribute("help")) _help = p.getAttribute<string>("help");
364  if (!_unit.empty()) _unit = "[" + _unit + "]";
365  if (!_default.empty()) _default = "(" + _default + ")";
366 
367  _name = p.name();
368 
369  out << boost::format(ofmt) % _name % _default % _unit % _help;
370  }
371 
372  for (iter = p.begin(); iter != p.end(); ++iter) {
373  if (prefix == "") {
374  _offset = level + 2;
375  level++;
376  PrintNodeHLP(out, (*iter), start_level, level, (*iter).name(), offset);
377  _offset = level - 2;
378  level--;
379  } else {
380  _offset = level + 2;
381  level++;
382  PrintNodeHLP(out, (*iter), start_level, level,
383  prefix + "." + (*iter).name(), offset);
384  _offset = level - 2;
385  level--;
386  }
387  }
388 }

390 std::ostream &operator<<(std::ostream &out, Property &p) {
391  if (!out.good()) return out;
392 
393  std::ostream::sentry sentry(out);
394 
395  if (sentry) {
396  // get the property format object attached to the stream
397  PropertyIOManipulator *pm =
398  (PropertyIOManipulator *)out.pword(Property::getIOindex());
399 
400  string _indentation("");
401  int _level = 0;
402 
403  PropertyIOManipulator::Type _type = PropertyIOManipulator::XML;
404  if (pm) {
405  _indentation = pm->getIndentation();
406  _level = pm->getLevel();
407  _type = pm->getType();
408  // check if we > or >> to a file and remove color codes
409  // if ( out.tellp() != -1 ) - not suitable for pipes
410  if (!isatty(STDOUT_FILENO) || !isatty(STDERR_FILENO)) {
411  pm->setColorScheme<csDefault>();
412  }
413  }
414 
415  switch (_type) {
416  default:
417  PrintNodeTXT(out, p, _level);
418  case PropertyIOManipulator::XML:
419  PrintNodeXML(out, p, pm);
420  break;
421  case PropertyIOManipulator::TXT:
422  PrintNodeTXT(out, p, _level, 0, "", _indentation);
423  break;
424  case PropertyIOManipulator::TEX:
425  PrintNodeTEX(out, p, pm);
426  break;
427  case PropertyIOManipulator::HLP:
428  PrintNodeHLP(out, p, _level, 0, "", _indentation);
429  break;
430  }
431  }
432 
433  return out;
434 };
