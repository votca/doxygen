68  void TCMatrix_dft::FillBlock(std::vector< Eigen::MatrixXd >& block, int shellindex, const AOBasis& dftbasis, const AOBasis& auxbasis) {
69  const AOShell* left_dftshell = dftbasis.getShell(shellindex);
70  tensor3d::extent_gen extents;
71  int start = left_dftshell->getStartIndex();
72  // alpha-loop over the aux basis function
73  for (const AOShell* shell_aux:auxbasis) {
74  int aux_start = shell_aux->getStartIndex();
75 
76 
77  for (int is = 0; is <= shellindex; is++) {
78 
79  const AOShell* shell_col = dftbasis.getShell(is);
80  int col_start=shell_col->getStartIndex();
81  tensor3d threec_block(extents[ range(0, shell_aux->getNumFunc()) ][ range(0, left_dftshell->getNumFunc()) ][ range(0, shell_col->getNumFunc())]);
82  for (int i = 0; i < shell_aux->getNumFunc(); ++i) {
83  for (int j = 0; j < left_dftshell->getNumFunc(); ++j) {
84  for (int k = 0; k < shell_col->getNumFunc(); ++k) {
85  threec_block[i][j][k] = 0.0;
86  }
87  }
88  }
89 
90  bool nonzero = FillThreeCenterRepBlock(threec_block, shell_aux, left_dftshell, shell_col);
91  if (nonzero) {
92 
93  for (int left = 0; left < left_dftshell->getNumFunc(); left++) {
94  for (int aux = 0; aux < shell_aux->getNumFunc(); aux++) {
95  for (int col = 0; col < shell_col->getNumFunc(); col++) {
96  //symmetry
97  if ((col_start + col)>(start + left)) {
98  break;
99  }
100  block[left](aux_start + aux, col_start + col) = threec_block[aux][left][col];
101  }
102  }
103  }
104  }
105  }
106  }
107  return;
108  }
