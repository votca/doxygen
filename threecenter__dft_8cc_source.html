29 void TCMatrix_dft::Fill(const AOBasis& auxbasis, const AOBasis& dftbasis) {
30 
31  AOCoulomb auxAOcoulomb;
32  auxAOcoulomb.Fill(auxbasis);
33  _inv_sqrt = auxAOcoulomb.Pseudo_InvSqrt(1e-8);
34  _removedfunctions = auxAOcoulomb.Removedfunctions();
35 
36  for (int i = 0; i < auxbasis.AOBasisSize(); i++) {
37  try {
38  _matrix.push_back(Symmetric_Matrix(dftbasis.AOBasisSize()));
39  } catch (std::bad_alloc&) {
40  throw std::runtime_error(
41  "Basisset/aux basis too large for 3c calculation. Not enough RAM.");
42  }
43  }
44 #pragma omp parallel for schedule(dynamic)
45  for (int is = dftbasis.getNumofShells() - 1; is >= 0; is--) {
46  const AOShell& dftshell = dftbasis.getShell(is);
47  std::vector<Eigen::MatrixXd> block;
48  for (int i = 0; i < dftshell.getNumFunc(); i++) {
49  int size = dftshell.getStartIndex() + i + 1;
50  block.push_back(Eigen::MatrixXd::Zero(auxbasis.AOBasisSize(), size));
51  }
52  FillBlock(block, is, dftbasis, auxbasis);
53  int offset = dftshell.getStartIndex();
54  for (unsigned i = 0; i < block.size(); ++i) {
55  Eigen::MatrixXd temp = _inv_sqrt * block[i];
56  for (int mu = 0; mu < temp.rows(); ++mu) {
57  for (int j = 0; j < temp.cols(); ++j) {
58  _matrix[mu](i + offset, j) = temp(mu, j);
59  }
60  }
61  }
62  }
63  return;
64 }

72 void TCMatrix_dft::FillBlock(std::vector<Eigen::MatrixXd>& block,
73  int shellindex, const AOBasis& dftbasis,
74  const AOBasis& auxbasis) {
75  const AOShell& left_dftshell = dftbasis.getShell(shellindex);
76  tensor3d::extent_gen extents;
77  int start = left_dftshell.getStartIndex();
78  // alpha-loop over the aux basis function
79  for (const AOShell& shell_aux : auxbasis) {
80  int aux_start = shell_aux.getStartIndex();
81 
82  for (int is = 0; is <= shellindex; is++) {
83 
84  const AOShell& shell_col = dftbasis.getShell(is);
85  int col_start = shell_col.getStartIndex();
86  tensor3d threec_block(extents[range(0, shell_aux.getNumFunc())][range(
87  0, left_dftshell.getNumFunc())][range(0, shell_col.getNumFunc())]);
88  std::fill_n(threec_block.data(), threec_block.num_elements(), 0.0);
89 
90  bool nonzero = FillThreeCenterRepBlock(threec_block, shell_aux,
91  left_dftshell, shell_col);
92  if (nonzero) {
93 
94  for (int left = 0; left < left_dftshell.getNumFunc(); left++) {
95  for (int aux = 0; aux < shell_aux.getNumFunc(); aux++) {
96  for (int col = 0; col < shell_col.getNumFunc(); col++) {
97  // symmetry
98  if ((col_start + col) > (start + left)) {
99  break;
100  }
101  block[left](aux_start + aux, col_start + col) =
102  threec_block[aux][left][col];
103  }
104  }
105  }
106  }
107  }
108  }
109  return;
110 }
