29 void TCMatrix_dft::Fill(const AOBasis& auxbasis, const AOBasis& dftbasis) {
30 
31  AOCoulomb auxAOcoulomb;
32  auxAOcoulomb.Fill(auxbasis);
33  _inv_sqrt = auxAOcoulomb.Pseudo_InvSqrt(1e-8);
34  _removedfunctions = auxAOcoulomb.Removedfunctions();
35 
36  for (Index i = 0; i < auxbasis.AOBasisSize(); i++) {
37  try {
38  _matrix.push_back(Symmetric_Matrix(dftbasis.AOBasisSize()));
39  } catch (std::bad_alloc&) {
40  throw std::runtime_error(
41  "Basisset/aux basis too large for 3c calculation. Not enough RAM.");
42  }
43  }
44 #pragma omp parallel for schedule(dynamic)
45  for (Index is = dftbasis.getNumofShells() - 1; is >= 0; is--) {
46  const AOShell& dftshell = dftbasis.getShell(is);
47  std::vector<Eigen::MatrixXd> block;
48  for (Index i = 0; i < dftshell.getNumFunc(); i++) {
49  Index size = dftshell.getStartIndex() + i + 1;
50  block.push_back(Eigen::MatrixXd::Zero(auxbasis.AOBasisSize(), size));
51  }
52  FillBlock(block, is, dftbasis, auxbasis);
53  Index offset = dftshell.getStartIndex();
54  for (Index i = 0; i < Index(block.size()); ++i) {
55  Eigen::MatrixXd temp = _inv_sqrt * block[i];
56  for (Index mu = 0; mu < temp.rows(); ++mu) {
57  for (Index j = 0; j < temp.cols(); ++j) {
58  _matrix[mu](i + offset, j) = temp(mu, j);
59  }
60  }
61  }
62  }
63  return;
64 }

72 void TCMatrix_dft::FillBlock(std::vector<Eigen::MatrixXd>& block,
73  Index shellindex, const AOBasis& dftbasis,
74  const AOBasis& auxbasis) {
75  const AOShell& left_dftshell = dftbasis.getShell(shellindex);
76 
77  Index start = left_dftshell.getStartIndex();
78  // alpha-loop over the aux basis function
79  for (const AOShell& shell_aux : auxbasis) {
80  Index aux_start = shell_aux.getStartIndex();
81 
82  for (Index is = 0; is <= shellindex; is++) {
83 
84  const AOShell& shell_col = dftbasis.getShell(is);
85  Index col_start = shell_col.getStartIndex();
86  Eigen::Tensor<double, 3> threec_block(shell_aux.getNumFunc(),
87  left_dftshell.getNumFunc(),
88  shell_col.getNumFunc());
89  threec_block.setZero();
90 
91  bool nonzero = FillThreeCenterRepBlock(threec_block, shell_aux,
92  left_dftshell, shell_col);
93  if (nonzero) {
94 
95  for (Index left = 0; left < left_dftshell.getNumFunc(); left++) {
96  for (Index aux = 0; aux < shell_aux.getNumFunc(); aux++) {
97  for (Index col = 0; col < shell_col.getNumFunc(); col++) {
98  // symmetry
99  if ((col_start + col) > (start + left)) {
100  break;
101  }
102  block[left](aux_start + aux, col_start + col) =
103  threec_block(aux, left, col);
104  }
105  }
106  }
107  }
108  }
109  }
110  return;
111 }
