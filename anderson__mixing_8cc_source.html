49 const Eigen::VectorXd Anderson::MixHistory() {
50 
51  const Index iteration = _output.size();
52  const Index used_history = iteration - 1;
53  Eigen::VectorXd OutMixed = _output.back();
54  Eigen::VectorXd InMixed = _input.back();
55 
56  if (iteration > 1 && _order > 1) {
57 
58  Eigen::VectorXd DeltaN = OutMixed - InMixed;
59 
60  // Building Linear System for Coefficients
61  Eigen::MatrixXd A = Eigen::MatrixXd::Zero(used_history, used_history);
62  Eigen::VectorXd c = Eigen::VectorXd::Zero(used_history);
63 
64  for (Index m = 1; m < iteration; m++) {
65 
66  c(m - 1) = (DeltaN - _output[used_history - m] + _input[used_history - m])
67  .dot(DeltaN);
68 
69  for (Index j = 1; j < iteration; j++) {
70  A(m - 1, j - 1) =
71  (DeltaN - _output[used_history - m] + _input[used_history - m])
72  .dot((DeltaN - _output[used_history - j] +
73  _input[used_history - j]));
74  }
75  }
76  // Solving the System to obtain coefficients
77  Eigen::VectorXd coefficients = A.fullPivHouseholderQr().solve(c);
78 
79  // Mixing the Potentials
80  for (Index n = 1; n < iteration; n++) {
81 
82  OutMixed += coefficients(n - 1) *
83  (_output[used_history - n] - _output[used_history]);
84  InMixed += coefficients(n - 1) *
85  (_input[used_history - n] - _input[used_history]);
86  }
87  }
88 
89  // Returning the linear Mix of Input and Output
90  return _alpha * OutMixed + (1 - _alpha) * InMixed;
91 }
