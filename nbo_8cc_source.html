47 
48 // make an array which stores for each atom the the starting point for all s functions, p functions, d functions etc...
49 
50 
51  //atom/shell/starting
52  std::vector< std::map< int,std::vector< int > > >sorting;
53 
54  int functionindex=0;
55  for (QMAtom* atom :_atomlist){

57  const Element& element = bs.getElement(atom->getType());
58  for (const Shell& shell:element) {
59 
60  // for loop because shells can also consist of SP shells or alike

65 
66  index.push_back(functionindex);
67 
68  functionindex+=NumFuncShell(local_shell);
69  }
70  }
71  sorting.push_back(shellsort);
72  }
73  typedef std::map< int,std::vector< int > >::iterator it_type;
74 
75  Eigen::MatrixXd transformation=Eigen::MatrixXd::Zero(P.rows(),P.cols());
76  Eigen::VectorXd occupancies=Eigen::VectorXd::Zero(P.rows());
77  // Stting up atomic transformations
78 
79  //atoms
80  for (unsigned i=0;i<sorting.size();i++){
81  //shell
82  for(it_type iterator = sorting[i].begin(); iterator != sorting[i].end(); iterator++) {
83  int ll=2*(iterator->first)+1;
84  std::vector< int >& index=iterator->second;
85  unsigned size=index.size();

88 
89  //filling P_L and S_L
90  for (unsigned i=0;i<size;i++){
91  for (unsigned j=0;j<size;j++){
92  //summing over m
93  for (int m=0;m<ll;m++){
94  P_L(i,j)+=P(index[i]+m,index[j]+m);
95  S_L(i,j)+=overlap(index[i]+m,index[j]+m);
96  }
97  }
98  }
99 
100 
101  //Setup generalized eigenproblem for each P_L/S_L
102 
103  Eigen::GeneralizedSelfAdjointEigenSolver< Eigen::MatrixXd > es(P_L,S_L);
104 
105 
106  for (unsigned i=0;i<size;i++){
107  for (int m=0;m<ll;m++){
108  occupancies(index[i]+m)=es.eigenvalues()(i);
109  }
110  for (unsigned j=0;j<size;j++){
111  for (int m=0;m<ll;m++){
112  transformation(index[i]+m,index[j]+m)=es.eigenvectors()(i,j);
113  }

115  }
