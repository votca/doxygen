39 void Graph_DF_Visitor::addEdges_(const Graph& g, int vertex) {
40  auto eds = g.getNeighEdges(vertex);
41  if (edge_list_.empty()) {
42  // If first edges to be added
43  for (auto ed : eds) {
44  int neigh_vert = ed.getOtherEndPoint(vertex);
45  if (explored_.count(neigh_vert) == 0) {
46  edge_list_.push_back(ed);
47  }
48  }
49  } else {
50  for (const auto& ed : eds) {
51  int neigh_vert = ed.getOtherEndPoint(vertex);
52  if (explored_.count(neigh_vert) == 0) {
53  edge_list_.push_back(ed);
54  } else {
55  // Check if edge has already been added earlier in the queue
56  // if so we wil move it to the end
57  list<Edge>::iterator edge_found_iterator =
58  find(edge_list_.begin(), edge_list_.end(), ed);
59  if (edge_found_iterator != edge_list_.end()) {
60  // Move the edge to the back if it was stored earlier on.
61  edge_list_.splice(edge_list_.end(), edge_list_, edge_found_iterator);
62  }
63  }
64  }
65  }
