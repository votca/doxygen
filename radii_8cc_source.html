00062 void CsgTestApp::EvalConfiguration(Topology *top, Topology *top_ref)
00063 {    
00064     // loop over all molecules
00065     for(MoleculeContainer::iterator imol=top->Molecules().begin(); imol!=top->Molecules().end(); ++imol) {
00066         Molecule *mol = *imol;
00067         // does the id match if given?
00068         if(OptionsMap().count("mol")) {
00069           if(OptionsMap()["mol"].as<int>()!=mol->getId() + 1)
00070               continue;
00071         }
00072         // otherwise does the name pattern match?
00073         else if(!wildcmp(OptionsMap()["molname"].as<string>().c_str(), mol->getName().c_str()))
00074             continue; // if not skip this molecule
00075     
00076         // Number of beads in the molecule
00077         int N = mol->BeadCount();
00078     
00079         // sqared tensor of gyration for current snapshot
00080         double r_gyr_sq = 0;
00081         // inverse hydrodynamic radius for current snapshot
00082         double inv_r_hydr = 0;
00083            
00084         // loop over all bead pairs in molecule
00085         for(int i=0; i<N; ++i) {
00086             for(int j=i+1; j<N; ++j) {
00087                 // distance between bead i and j
00088                 vec r_ij = mol->getBead(i)->getPos() -  mol->getBead(j)->getPos();
00089                 // radius of gyration squared
00090                 r_gyr_sq += r_ij*r_ij/(double)(N*N);
00091                 // hydrodynamic radius
00092                 inv_r_hydr += 2./(abs(r_ij)*(double(N*N)));            
00093             }
00094         }
00095     
00096         // add calculated values to the averages
00097         _r_gyr_sq.Process(r_gyr_sq);
00098         _inv_r_hydr.Process(inv_r_hydr);
00099         
00100         // calculate the mass weighted tensor of gyration
00101         // first calculate mass + center of mass
00102         double M=0;
00103         vec cm(0,0,0);
00104         for(int i=0; i<N; ++i) {
00105             M+=mol->getBead(i)->getM();
00106             cm+=mol->getBead(i)->getPos()*mol->getBead(i)->getM();
00107         }
00108         cm/=M;
00109         // now tensor of gyration based on cm
00110         double r_gyr_m_sq=0;
00111         for(int i=0; i<N; ++i) {
00112             vec r_ij=mol->getBead(i)->getPos() - cm;
00113             r_gyr_m_sq+=mol->getBead(i)->getM()*(r_ij*r_ij);
00114         }
00115         r_gyr_m_sq/=M;
00116 
00117         // add to average
00118         _r_gyr_m_sq.Process(r_gyr_m_sq);
00119     }
00120 }
