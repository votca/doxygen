62 void CsgTestApp::EvalConfiguration(Topology *top, Topology *top_ref)
63 {
64  // loop over all molecules
65  for(MoleculeContainer::iterator imol=top->Molecules().begin(); imol!=top->Molecules().end(); ++imol) {
66  Molecule *mol = *imol;
67  // does the id match if given?
68  if(OptionsMap().count("mol")) {
69  if(OptionsMap()["mol"].as<int>()!=mol->getId() + 1)
70  continue;
71  }
72  // otherwise does the name pattern match?
73  else if(!wildcmp(OptionsMap()["molname"].as<string>().c_str(), mol->getName().c_str()))
74  continue; // if not skip this molecule
75 
76  // Number of beads in the molecule
77  int N = mol->BeadCount();
78 
79  // sqared tensor of gyration for current snapshot
80  double r_gyr_sq = 0;
81  // inverse hydrodynamic radius for current snapshot
82  double inv_r_hydr = 0;
83 
84  // loop over all bead pairs in molecule
85  for(int i=0; i<N; ++i) {
86  for(int j=i+1; j<N; ++j) {
87  // distance between bead i and j
88  vec r_ij = mol->getBead(i)->getPos() - mol->getBead(j)->getPos();
89  // radius of gyration squared
90  r_gyr_sq += r_ij*r_ij/(double)(N*N);
91  // hydrodynamic radius
92  inv_r_hydr += 2./(abs(r_ij)*(double(N*N)));
93  }
94  }
95 
96  // add calculated values to the averages
97  _r_gyr_sq.Process(r_gyr_sq);
98  _inv_r_hydr.Process(inv_r_hydr);
99 
100  // calculate the mass weighted tensor of gyration
101  // first calculate mass + center of mass
102  double M=0;
103  vec cm(0,0,0);
104  for(int i=0; i<N; ++i) {
105  M+=mol->getBead(i)->getM();
106  cm+=mol->getBead(i)->getPos()*mol->getBead(i)->getM();
107  }
108  cm/=M;
109  // now tensor of gyration based on cm
110  double r_gyr_m_sq=0;
111  for(int i=0; i<N; ++i) {
112  vec r_ij=mol->getBead(i)->getPos() - cm;
113  r_gyr_m_sq+=mol->getBead(i)->getM()*(r_ij*r_ij);
114  }
115  r_gyr_m_sq/=M;
116 
117  // add to average
118  _r_gyr_m_sq.Process(r_gyr_m_sq);
119  }
120 }
