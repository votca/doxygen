47 void Density2Gyration::AnalyzeDensity(Orbitals& orbitals) {
48  int threads = 1;
49 #ifdef _OPENMP
50  if (_openmp_threads > 0) omp_set_num_threads(_openmp_threads);
51  threads = omp_get_max_threads();
52 #endif
53  CTP_LOG(ctp::logDEBUG, *_log)
54  << "===== Running on " << threads << " threads ===== " << flush;
55 
56  std::vector<QMAtom*> Atomlist = orbitals.QMAtoms();
57  Eigen::MatrixXd DMAT_tot;
58  BasisSet bs;
59  bs.LoadBasisSet(orbitals.getDFTbasisName());
60  AOBasis basis;
61  basis.AOBasisFill(bs, Atomlist);
62  AnalyzeGeometry(Atomlist);
63  std::vector<Eigen::MatrixXd> DMAT;
64 
65  // setup numerical integration grid
66  NumericalIntegration numway;
67  numway.GridSetup(_gridsize, Atomlist, basis);
68 
69  if (!_dostateonly) {
70  Eigen::MatrixXd DMATGS = orbitals.DensityMatrixFull(_state);
71  Gyrationtensor gyro = numway.IntegrateGyrationTensor(DMAT_tot);
72  tools::matrix::eigensystem_t system;
73  CTP_LOG(ctp::logDEBUG, *_log)
74  << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
75  gyro.gyration.SolveEigensystem(system);
76  CTP_LOG(ctp::logDEBUG, *_log)
77  << ctp::TimeStamp() << " Calculating Quaternion " << flush;
78  // Eigen::Quaterniond _quaternion = get_quaternion( system );
79  // report results
80  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
81  ReportAnalysis(_state.ToLongString(), gyro, system);
82 
83  } else {
84  // hole density first
85  std::vector<Eigen::MatrixXd> DMAT =
86  orbitals.DensityMatrixExcitedState(_state);
87  Gyrationtensor gyro_hole = numway.IntegrateGyrationTensor(DMAT[0]);
88  tools::matrix::eigensystem_t system_h;
89  CTP_LOG(ctp::logDEBUG, *_log)
90  << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
91  gyro_hole.gyration.SolveEigensystem(system_h);
92  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
93  ReportAnalysis("hole", gyro_hole, system_h);
94 
95  // electron density
96  Gyrationtensor gyro_electron = numway.IntegrateGyrationTensor(DMAT[1]);
97  tools::matrix::eigensystem_t system_e;
98  CTP_LOG(ctp::logDEBUG, *_log)
99  << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
100  gyro_electron.gyration.SolveEigensystem(system_e);
101  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
102  ReportAnalysis("electron", gyro_electron, system_e);
103  }
104  return;
105 }

107 void Density2Gyration::AnalyzeGeometry(std::vector<QMAtom*> atoms) {
108 
109  tools::Elements elements;
110  double mass = 0.0;
111  tools::vec centroid = tools::vec(0.0);
112  tools::matrix gyration = tools::matrix(0.0);
113  for (QMAtom* atom : atoms) {
114  double m = elements.getMass(atom->getType());
115  const tools::vec& pos = atom->getPos();
116  mass += m;
117  centroid += m * pos;
118  gyration += m * (pos | pos);
119  }
120  centroid /= mass;
121  gyration /= mass;
122  gyration -= (centroid | centroid);
123  Gyrationtensor gyro;
124  gyro.mass = mass;
125  gyro.centroid = centroid;
126  gyro.gyration = gyration;
127  tools::matrix::eigensystem_t system;
128  gyration.SolveEigensystem(system);
129  ReportAnalysis("geometry", gyro, system);
130 }
