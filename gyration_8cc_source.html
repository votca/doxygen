63 void Density2Gyration::AnalyzeDensity( Orbitals & _orbitals ){

65 #ifdef _OPENMP

69  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on "<< threads << " threads ===== " << flush;
70 
71  vector< ctp::QMAtom* > Atomlist =_orbitals.QMAtoms();
72  std::vector< ctp::QMAtom* >::iterator at;
73  for (at=Atomlist.begin();at<Atomlist.end();++at){
74  ctp::QMAtom * atom=new ctp::QMAtom(*(*at));
75  _Atomlist.push_back(atom);
76  }
77  ub::matrix<double> DMAT_tot;

89 
90  //basis.ReorderMOs(_orbitals.MOCoefficients(), _orbitals.getQMpackage(), "votca" );
91 
92  if(_state=="transition"){
93  DMAT_tot=_orbitals.TransitionDensityMatrix(_spin, _state_no-1);
94  }
95  else if (_state=="ground" || _state=="excited" || _state=="exciton" ){
96  CTP_LOG(ctp::logDEBUG, *_log) << "Calculating density matrix: " << _state << " No. " << _state_no << flush;
97 
98 
99 
100  ub::matrix<double> DMATGS=_orbitals.DensityMatrixGroundState();
101  DMAT_tot=DMATGS;

103

105

107  DMAT_tot=DMAT_tot-DMAT[0]+DMAT[1];

303  ub::vector<double> Density2Gyration::get_quaternion(ub::matrix<double>& eigenframe){
304 
305  // second coordinate system is assumed to be Cartesian
306 
307  ub::matrix<double> M=ub::zero_matrix<double>(3,3);
308 
309  // add all the outer products, if second coordinate system was not Cartesian
310  // is anyways, so M=eigenframe
311  //M=ub::trans(eigenframe);
312  M=eigenframe;
313 
314  ub::matrix<double> N = ub::zero_matrix<double>(4,4);
315 
316  N(0,0) = M(0,0) + M(1,1) + M(2,2); // N11=float(M[0][:,0]+M[1][:,1]+M[2][:,2]);
317  N(1,1) = M(0,0) - M(1,1) - M(2,2); // N22=float(M[0][:,0]-M[1][:,1]-M[2][:,2])
318  N(2,2) = -M(0,0) + M(1,1) - M(2,2); // N33=float(-M[0][:,0]+M[1][:,1]-M[2][:,2])
319  N(3,3) = -M(0,0) - M(1,1) + M(2,2); // N44=float(-M[0][:,0]-M[1][:,1]+M[2][:,2])
320  N(0,1) = M(1,2) - M(2,1); // N12=float(M[1][:,2]-M[2][:,1])
321  N(0,2) = M(2,0) - M(0,2); // N13=float(M[2][:,0]-M[0][:,2])
322  N(0,3) = M(0,1) - M(1,0); // N14=float(M[0][:,1]-M[1][:,0])
323  N(1,0) = N(0,1); // N21=float(N12)
324  N(1,2) = M(0,1) + M(1,0); // N23=float(M[0][:,1]+M[1][:,0])
325  N(1,3) = M(2,0) + M(0,2); // N24=float(M[2][:,0]+M[0][:,2])
326  N(2,0) = N(0,2); // N31=float(N13)
327  N(2,1) = N(1,2); // N32=float(N23)
328  N(2,3) = M(1,2) + M(2,1); // N34=float(M[1][:,2]+M[2][:,1])
329  N(3,0) = N(0,3); // N41=float(N14)
330  N(3,1) = N(1,3); // N42=float(N24)
331  N(3,2) = N(2,3); // N43=float(N34)
332 
333  ub::vector<double> eigenvalues;
334  ub::matrix<double> eigenvectors;
335  linalg_eigenvalues( N, eigenvalues, eigenvectors );
336 
337  // find max eigenvalue
338  int index=0;
339  double maxev = eigenvalues(0);
340 
341  for (unsigned i = 1; i < eigenvalues.size(); i++ ) {
342  if ( eigenvalues(i) > maxev ) {
343  maxev = eigenvalues(i);
344  index = i;
345  }
346  }
347 
348  // return this vector
349  ub::vector<double> quaternion = ub::zero_vector<double>(4);
350  quaternion(0)= eigenvectors(0,index);
351  quaternion(1)= eigenvectors(1,index);
352  quaternion(2)= eigenvectors(2,index);
353  quaternion(3)= eigenvectors(3,index);
354 
355  return quaternion;
356 
357  }
