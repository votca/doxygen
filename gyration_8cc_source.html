44  void Density2Gyration::AnalyzeDensity(Orbitals & orbitals) {
45  int threads = 1;
46 #ifdef _OPENMP
47  if (_openmp_threads > 0) omp_set_num_threads(_openmp_threads);
48  threads = omp_get_max_threads();
49 #endif
50  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on " << threads << " threads ===== " << flush;
51 
52  std::vector< QMAtom* > Atomlist = orbitals.QMAtoms();
53  Eigen::MatrixXd DMAT_tot;
54  BasisSet bs;
55  bs.LoadBasisSet(orbitals.getDFTbasis());
56  AOBasis basis;
57  basis.AOBasisFill(bs, Atomlist);
58  AnalyzeGeometry(Atomlist);
59  std::vector<Eigen::MatrixXd > DMAT;
60 
61  // setup numerical integration grid
62  NumericalIntegration numway;
63  numway.GridSetup(_gridsize, Atomlist, basis);
64 
65  if (!_dostateonly) {
66  Eigen::MatrixXd DMATGS = orbitals.DensityMatrixFull(_state);
67  Gyrationtensor gyro = numway.IntegrateGyrationTensor(DMAT_tot);
68  tools::matrix::eigensystem_t system;
69  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
70  gyro.gyration.SolveEigensystem(system);
71  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating Quaternion " << flush;
72  //Eigen::Quaterniond _quaternion = get_quaternion( system );
73  // report results
74  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
75  ReportAnalysis(_state.ToLongString(), gyro,system);
76 
77 
78  } else{
79  // hole density first
80  std::vector<Eigen::MatrixXd > DMAT=orbitals.DensityMatrixExcitedState(_state);
81  Gyrationtensor gyro_hole = numway.IntegrateGyrationTensor(DMAT[0]);
82  tools::matrix::eigensystem_t system_h;
83  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
84  gyro_hole.gyration.SolveEigensystem(system_h);
85  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
86  ReportAnalysis("hole", gyro_hole,system_h);
87 
88  // electron density
89  Gyrationtensor gyro_electron = numway.IntegrateGyrationTensor(DMAT[1]);
90  tools::matrix::eigensystem_t system_e;
91  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Converting to Eigenframe " << flush;
92  gyro_electron.gyration.SolveEigensystem(system_e);
93  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Reporting " << flush;
94  ReportAnalysis("electron", gyro_electron,system_e);
95  }
96  return;
97  }

100  void Density2Gyration::AnalyzeGeometry(std::vector<QMAtom*> atoms){
101 
102  tools::Elements elements;
103  double mass=0.0;
104  tools::vec centroid = tools::vec(0.0);
105  tools::matrix gyration = tools::matrix(0.0);
106  for (QMAtom* atom:atoms){
107  double m = elements.getMass(atom->getType());
108  const tools::vec & pos =atom->getPos();
109  mass+= m;
110  centroid+=m*pos;
111  gyration+=m*(pos|pos);
112  }
113  centroid/=mass;
114  gyration/=mass;
115  gyration-=(centroid|centroid);
116  Gyrationtensor gyro;
117  gyro.mass=mass;
118  gyro.centroid=centroid;
119  gyro.gyration=gyration;
120  tools::matrix::eigensystem_t system;
121  gyration.SolveEigensystem(system);
122  ReportAnalysis( "geometry", gyro,system );
123  }
