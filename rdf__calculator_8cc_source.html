134 RDFCalculator::interaction_t *RDFCalculator::AddInteraction(Property *p)
135 {
136  string name = p->get("name").value();
137  string group;
138 
139  group = "none";
140 
141  interaction_t *i = new interaction_t;
142  i->_index = _interactions.size();
143  _interactions[name] = i;
144  getGroup(group)->_interactions.push_back(i);
145 
146  i->_step = p->get("step").as<double>();
147  i->_min = p->get("min").as<double>();
148  i->_max = p->get("max").as<double>();
149  i->_norm = 1.0;
150  i->_p=p;
151 
152  // initialize the current and average histogram
153  int n = (int)((i->_max - i->_min) / i->_step + 1.000000001);
154 
155  i->_average.Initialize(i->_min, i->_max+i->_step, n);
156 
157  return i;
158 }

206 public:
207  IMCNBSearchHandler(HistogramNew *hist, double subvol_rad, vec boxc, bool do_vol_corr) : _hist(hist), _subvol_rad(subvol_rad), _boxc(boxc), _do_vol_corr(do_vol_corr) {}
208 
209  HistogramNew *_hist;

211  vec _boxc; // center of box
212  bool _do_vol_corr;

214 
215  bool FoundPair(Bead *b1, Bead *b2, const vec &r, const double dist) {
216 
217  if (_do_vol_corr){
218  double dr = abs(b1->Pos()-_boxc);
219  if (dist+dr> _subvol_rad)
220  // 2.0 is because everything is normalized to 4 PI
221  _hist->Process(dist, 2.0/SurfaceRatio(dist, dr));
222  else
223  _hist->Process(dist);
224 
225  }else{
226  _hist->Process(dist);
227  }
228  return false;
229  }

231  double SurfaceRatio (double dist, double r){
232  // r: distance of particle from ex center
233  // dist: distance between particles
234  return (1.0+(_subvol_rad*_subvol_rad-r*r-dist*dist)/(2.0*dist*r));
235  }

241 void RDFCalculator::Worker::DoNonbonded(Topology *top)
242 {
243  for (list<Property*>::iterator iter = _rdfcalculator->_nonbonded.begin();
244  iter != _rdfcalculator->_nonbonded.end(); ++iter) {
245  string name = (*iter)->get("name").value();
246 
247  interaction_t &i = *_rdfcalculator->_interactions[name];
248 
249  // generate the bead lists
250  BeadList beads1, beads2;
251 
252  beads1.GenerateInSphericalSubvolume(*top, (*iter)->get("type1").value(), _rdfcalculator->_boxc, _rdfcalculator->_subvol_rad);
253  beads2.GenerateInSphericalSubvolume(*top, (*iter)->get("type2").value(), _rdfcalculator->_boxc, _rdfcalculator->_subvol_rad);
254 
255 
256  _cur_beadlist_1_count = beads1.size();
257  _cur_beadlist_2_count = beads2.size();
258 
259  // same types, so put factor 1/2 because of already counted interactions
260  if ((*iter)->get("type1").value() == (*iter)->get("type2").value()){
261  _cur_beadlist_2_count/=2.0;
262  }
263 
264  // generate the neighbour list
265  NBList *nb;
266 
267  bool gridsearch=true;
268 
269  if(_rdfcalculator->_options.exists("cg.nbsearch")) {
270  if(_rdfcalculator->_options.get("cg.nbsearch").as<string>() == "grid")
271  gridsearch=true;
272  else if(_rdfcalculator->_options.get("cg.nbsearch").as<string>() == "simple")
273  gridsearch=false;
274  else throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
275  }
276  if(gridsearch)
277  nb = new NBListGrid();
278  else
279  nb = new NBList();
280 
281  nb->setCutoff(i._max + i._step);
282 
283  // clear the current histogram
284  _current_hists[i._index].Clear();
285 
286  IMCNBSearchHandler h(&(_current_hists[i._index]), _rdfcalculator->_subvol_rad,
287  _rdfcalculator->_boxc, _rdfcalculator->_do_vol_corr);
288  nb->SetMatchFunction(&h, &IMCNBSearchHandler::FoundPair);
289 
290 
291  // is it same types or different types?
292  if((*iter)->get("type1").value() == (*iter)->get("type2").value())
293  nb->Generate(beads1);
294  else
295  nb->Generate(beads1, beads2);
296 
297  // store particle number in subvolume for each interaction
298  i._avg_beadlist_1_count.Process(_cur_beadlist_1_count);
299  i._avg_beadlist_2_count.Process(_cur_beadlist_2_count);
300 
301  // process all pairs
302  /*NBList::iterator pair_iter;
303  for(pair_iter = nb->begin(); pair_iter!=nb->end();++pair_iter) {
304  _current_hists[i._index].Process((*pair_iter)->dist());
305  }*/
306 
307  delete nb;
308  }

315 {
