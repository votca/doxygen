63 void RDFCalculator::BeginEvaluate(Topology *top, Topology *) {
64  Eigen::Matrix3d box = top->getBox();
65  _boxc = box.rowwise().sum() / 2.0;
66 
67  std::cout << "Using center of box: " << _boxc << std::endl;
68  // we didn't process any frames so far
69  _nframes = 0;
70  _nblock = 0;
71  _processed_some_frames = false;
72 
73  // initialize non-bonded structures
74  for (Property *prop : _nonbonded) {
75  std::string name = prop->get("name").value();
76 
77  interaction_t &i = *_interactions[name];
78 
79  // count total species for ideal densities
80 
81  BeadList allbeads1, allbeads2;
82  allbeads1.Generate(*top, prop->get("type1").value());
83  allbeads2.Generate(*top, prop->get("type2").value());
84 
85  if (allbeads1.size() == 0) {
86  throw std::runtime_error("Topology does not have beads of type \"" +
87  prop->get("type1").value() +
88  "\"\n"
89  "This was specified in type1 of interaction \"" +
90  name + "\"");
91  }
92  if (allbeads2.size() == 0) {
93  throw std::runtime_error("Topology does not have beads of type \"" +
94  prop->get("type2").value() +
95  "\"\n"
96  "This was specified in type2 of interaction \"" +
97  name + "\"");
98  }
99  // calculate normalization factor for rdf
100 
101  if (_do_vol_corr) {
102  std::cout << "Volume correction on" << std::endl;
103  i._norm = 1. / (4.0 * votca::tools::conv::Pi * i._step);
104  } else {
105  std::cout << "Volume correction off" << std::endl;
106  i._norm = 1. / (4. * votca::tools::conv::Pi * i._step);
107  }
108  }
109 }

111 RDFCalculator::interaction_t *RDFCalculator::AddInteraction(Property *p) {
112  std::string name = p->get("name").value();
113  std::string group;
114 
115  group = "none";
116 
117  interaction_t *i = new interaction_t;
118  i->_index = _interactions.size();
119  _interactions[name] = i;
120  getGroup(group)->_interactions.push_back(i);
121 
122  i->_step = p->get("step").as<double>();
123  i->_min = p->get("min").as<double>();
124  i->_max = p->get("max").as<double>();
125  i->_norm = 1.0;
126  i->_p = p;
127 
128  // initialize the current and average histogram
129  Index n = (Index)((i->_max - i->_min) / i->_step + 1.000000001);
130 
131  i->_average.Initialize(i->_min, i->_max + i->_step, n);
132 
133  return i;
134 }

181 class IMCNBSearchHandler {
182  public:
183  IMCNBSearchHandler(HistogramNew *hist, double subvol_rad,
184  Eigen::Vector3d boxc, bool do_vol_corr)
185  : _hist(hist),
186  _subvol_rad(subvol_rad),
187  _boxc(boxc),
188  _do_vol_corr(do_vol_corr) {}
189 
190  HistogramNew *_hist;
191  double _subvol_rad;
192  Eigen::Vector3d _boxc; // center of box
193  bool _do_vol_corr;
194 
195  bool FoundPair(Bead *b1, Bead *, const Eigen::Vector3d &, const double dist) {
196 
197  if (_do_vol_corr) {
198  double dr = (b1->Pos() - _boxc).norm();
199  if (dist + dr > _subvol_rad) {
200  // 2.0 is because everything is normalized to 4 PI
201  _hist->Process(dist, 2.0 / SurfaceRatio(dist, dr));
202  } else {
203  _hist->Process(dist);
204  }
205 
206  } else {
207  _hist->Process(dist);
208  }
209  return false;
210  }
211 
212  double SurfaceRatio(double dist, double r) {
213  // r: distance of particle from ex center
214  // dist: distance between particles
215  return (1.0 + (_subvol_rad * _subvol_rad - r * r - dist * dist) /
216  (2.0 * dist * r));
217  }
218 };

220 // process non-bonded interactions for current frame
221 void RDFCalculator::Worker::DoNonbonded(Topology *top) {
222  for (Property *prop : _rdfcalculator->_nonbonded) {
223  std::string name = prop->get("name").value();
224 
225  interaction_t &i = *_rdfcalculator->_interactions[name];
226 
227  // generate the bead lists
228  BeadList beads1, beads2;
229 
230  beads1.GenerateInSphericalSubvolume(*top, prop->get("type1").value(),
231  _rdfcalculator->_boxc,
232  _rdfcalculator->_subvol_rad);
233  beads2.GenerateInSphericalSubvolume(*top, prop->get("type2").value(),
234  _rdfcalculator->_boxc,
235  _rdfcalculator->_subvol_rad);
236 
237  _cur_beadlist_1_count = (double)beads1.size();
238  _cur_beadlist_2_count = (double)beads2.size();
239 
240  // same types, so put factor 1/2 because of already counted interactions
241  if (prop->get("type1").value() == prop->get("type2").value()) {
242  _cur_beadlist_2_count /= 2.0;
243  }
244 
245  // generate the neighbour list
246  std::unique_ptr<NBList> nb;
247 
248  bool gridsearch = true;
249 
250  if (_rdfcalculator->_options.exists("cg.nbsearch")) {
251  if (_rdfcalculator->_options.get("cg.nbsearch").as<std::string>() ==
252  "grid") {
253  gridsearch = true;
254  } else if (_rdfcalculator->_options.get("cg.nbsearch")
255  .as<std::string>() == "simple") {
256  gridsearch = false;
257  } else {
258  throw std::runtime_error("cg.nbsearch invalid, can be grid or simple");
259  }
260  }
261  if (gridsearch) {
262  nb = std::make_unique<NBList>(NBListGrid());
263  } else {
264  nb = std::make_unique<NBList>(NBList());
265  }
266 
267  nb->setCutoff(i._max + i._step);
268 
269  // clear the current histogram
270  _current_hists[i._index].Clear();
271 
272  IMCNBSearchHandler h(&(_current_hists[i._index]),
273  _rdfcalculator->_subvol_rad, _rdfcalculator->_boxc,
274  _rdfcalculator->_do_vol_corr);
275  nb->SetMatchFunction(&h, &IMCNBSearchHandler::FoundPair);
276 
277  // is it same types or different types?
278  if (prop->get("type1").value() == prop->get("type2").value()) {
279  nb->Generate(beads1);
280  } else {
281  nb->Generate(beads1, beads2);
282  }
283 
284  // store particle number in subvolume for each interaction
285  i._avg_beadlist_1_count.Process(_cur_beadlist_1_count);
286  i._avg_beadlist_2_count.Process(_cur_beadlist_2_count);
287  }
288 }

291 void RDFCalculator::Worker::DoBonded(Topology *top) {
292  for (Property *prop : _rdfcalculator->_bonded) {
293  std::string name = prop->get("name").value();
294 
295  interaction_t &i = *_rdfcalculator->_interactions[name];
296 
297  // clear the current histogram
298  _current_hists[i._index].Clear();
299 
300  // now fill with new data
301  std::list<Interaction *> list = top->InteractionsInGroup(name);
302 
303  for (auto ic : list) {

305  _current_hists[i._index].Process(v);

320 // write the distribution function
321 void RDFCalculator::WriteDist(const std::string &suffix) {
322 
323  // for all interactions
324  for (auto &_interaction : _interactions) {
325  // calculate the rdf
326  Table &t = _interaction.second->_average.data();
327  Table dist(t);
328 
329  _interaction.second->_norm /=
330  (_interaction.second->_avg_beadlist_1_count.getAvg() *
331  _interaction.second->_avg_beadlist_2_count.getAvg());
332  dist.y() = _avg_vol.getAvg() * _interaction.second->_norm *
333  dist.y().cwiseQuotient(dist.x().cwiseAbs2());
334 
335  dist.Save((_interaction.first) + suffix + ".dist.new");
336  std::cout << "written " << (_interaction.first) + suffix + ".dist.new\n";
337 
338  std::cout << "Avg. number of particles in subvol for "
339  << (_interaction.first) << std::endl;
340  std::cout << "beadlist 1: "
341  << _interaction.second->_avg_beadlist_1_count.getAvg()
342  << std::endl;
343  std::cout << "beadlist 2: "
344  << _interaction.second->_avg_beadlist_2_count.getAvg()
345  << std::endl;
346  }
347 
348  std::cout << "Volume used for normalization: " << _avg_vol.getAvg()
349  << std::endl;
350 }

367 void RDFCalculator::MergeWorker(CsgApplication::Worker *worker_) {
368  _processed_some_frames = true;
369  RDFCalculator::Worker *worker =
370  dynamic_cast<RDFCalculator::Worker *>(worker_);
371  // update the average
372 
373  ++_nframes;
374 
375  _avg_vol.Process(worker->_cur_vol);
376 
377  for (auto &_interaction : _interactions) {
378  interaction_t *i = _interaction.second;
379  i->_average.data().y() =
380  (((double)_nframes - 1.0) * i->_average.data().y() +
381  worker->_current_hists[i->_index].data().y()) /
382  (double)_nframes;
383  }
384 
385  if (_write_every != 0) {
386  if ((_nframes % _write_every) == 0) {
387  _nblock++;
388  std::string suffix =
389  std::string("_") + boost::lexical_cast<std::string>(_nblock);
390  WriteDist(suffix);
391  if (_do_blocks) {
392  ClearAverages();
393  }
394  }
395  }
