157  overlap2.Fill(aobasis2);

159  overlapcombined.Fill(combined);

161  overlapcombined.Matrix().topLeftCorner(aobasis.AOBasisSize(),
162  aobasis.AOBasisSize()),
163  1e-8);
164  BOOST_CHECK_EQUAL(check, 1);
165 
166  bool check2 = overlap2.Matrix().isApprox(
167  overlapcombined.Matrix().bottomRightCorner(aobasis2.AOBasisSize(),

178  for (size_t i = 0; i < func_per_atom3.size(); i++) {
