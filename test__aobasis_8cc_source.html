39BOOST_AUTO_TEST_CASE(FillNormBasis_test) {
40 libint2::initialize();
41 Orbitals orbitals;
42 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +

45 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/aobasis/notnormalized.xml");
46 AOBasis aobasis;
47 aobasis.Fill(basis, orbitals.QMAtoms());
48 {
49 const AOShell& shell = aobasis.getShell(0);
50 std::vector<double> ref_results = {0.1831079647, 0.9155398233};
51 Index i = 0;
52 bool check_norm = true;
53 for (const AOGaussianPrimitive& gaussian : shell) {

55 check_norm = false;
56 break;
57 }
58 i++;
59 }
60
61 i = 0;
62 if (!check_norm) {

64 std::cout << "Ref:" << ref_results[i]
65 << " result:" << gaussian.getContraction() << std::endl;
66 i++;
67 }
68 }
69 BOOST_CHECK_EQUAL(check_norm, 1);

72 const AOShell& shell = aobasis.getShell(1);

97BOOST_AUTO_TEST_CASE(Serializing) {
98 libint2::initialize();
99 Orbitals orbitals;
100 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
101 "/aobasis/molecule.xyz");
102 BasisSet basis;
103 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/aobasis/3-21G.xml");
104 AOBasis aobasis;
105 aobasis.Fill(basis, orbitals.QMAtoms());
106
107 CheckpointFile ff("aobasis.hdf5");
108 CheckpointWriter ww = ff.getWriter();
109 aobasis.WriteToCpt(ww);
110
111 CheckpointReader rr = ff.getReader();
112 AOBasis aobasis2;
113 aobasis2.ReadFromCpt(rr);
114
115 bool check_equal_basisset_names = aobasis.Name() == aobasis2.Name();
116 BOOST_CHECK_EQUAL(check_equal_basisset_names, true);
117
118 // no real way to test if two aobasis are equal so we check if the matrices
119 // are the same
120 AOOverlap overlap1;
121 overlap1.Fill(aobasis);
122 AOOverlap overlap2;
123 overlap2.Fill(aobasis2);
124 bool check = overlap1.Matrix().isApprox(overlap2.Matrix(), 1e-8);
125 BOOST_CHECK_EQUAL(check, 1);
126
127 auto func_per_atom1 = aobasis.getFuncPerAtom();
128 auto func_per_atom2 = aobasis2.getFuncPerAtom();
129
130 for (size_t i = 0; i < func_per_atom1.size(); i++) {
131 BOOST_CHECK_EQUAL(func_per_atom1[i], func_per_atom2[i]);
132 }
133 libint2::finalize();
134}

136BOOST_AUTO_TEST_CASE(Adding_a_basis) {
137 libint2::initialize();
138 QMMolecule mol("a", 0);
139 mol.LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) + "/aobasis/molecule.xyz");
140
141 QMMolecule mol2 = mol;
142 Eigen::Vector3d shift = {1, 2, 3};
143 mol2.Translate(shift);
144 BasisSet basis;
145 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/aobasis/3-21G.xml");
146 AOBasis aobasis;
147 aobasis.Fill(basis, mol);
148
149 AOBasis aobasis2;
150 aobasis2.Fill(basis, mol2);
151
152 AOBasis combined = aobasis;
153 combined.add(aobasis2);
154
155 // no real way to test if two aobasis are equal so we check if the matrices
156 // are the same
157 AOOverlap overlap1;
158 overlap1.Fill(aobasis);
159 AOOverlap overlap2;
160 overlap2.Fill(aobasis2);
161 AOOverlap overlapcombined;
162 overlapcombined.Fill(combined);
163 bool check = overlap1.Matrix().isApprox(
164 overlapcombined.Matrix().topLeftCorner(aobasis.AOBasisSize(),
165 aobasis.AOBasisSize()),
166 1e-8);
167 BOOST_CHECK_EQUAL(check, 1);
168
169 bool check2 = overlap2.Matrix().isApprox(
170 overlapcombined.Matrix().bottomRightCorner(aobasis2.AOBasisSize(),
171 aobasis2.AOBasisSize()),
172 1e-8);
173 BOOST_CHECK_EQUAL(check2, 1);
174
175 auto func_per_atom1 = aobasis.getFuncPerAtom();
176 auto func_per_atom2 = aobasis2.getFuncPerAtom();
177 func_per_atom1.insert(func_per_atom1.end(), func_per_atom2.begin(),
178 func_per_atom2.end());
179 auto func_per_atom3 = combined.getFuncPerAtom();
180
181 for (size_t i = 0; i < func_per_atom3.size(); i++) {
182 BOOST_CHECK_EQUAL(func_per_atom1[i], func_per_atom3[i]);
183 }
184 libint2::finalize();
185}
