47  //_grid.printGridtoxyzfile("grid.xyz");

113 ub::vector<double> Espfit::EvalNuclearPotential(std::vector< ctp::QMAtom* >& _atoms, Grid _grid) {
114  ub::vector<double> _NucPatGrid = ub::zero_vector<double>(_grid.getsize());
115 
116  double Znuc=0.0;
117  const std::vector< vec >& _gridpoints = _grid.getGrid();
118  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating ESP of nuclei at CHELPG grid points" << flush;
119 
120  for (unsigned i = 0; i < _gridpoints.size(); i++) {
121  for (unsigned j = 0; j < _atoms.size(); j++) {
122  vec posatom=_atoms[j]->getPos()*tools::conv::ang2nm;
123  if (_ECP) {
124  Znuc = _elements.getNucCrgECP(_atoms[j]->type);
125  } else {
126  Znuc = _elements.getNucCrg(_atoms[j]->type);
127  }
128  double dist_j = tools::abs(_gridpoints[i]-posatom) * tools::conv::nm2bohr;
129  _NucPatGrid(i) += Znuc / dist_j;
130  }
131 
132  }
133  return _NucPatGrid;
134 }

136 double Espfit::getNetcharge( std::vector< ctp::QMAtom* >& _atoms, double N ){
137  double netcharge=0.0;
138  if( std::abs(N)<0.05){
139  //CTP_LOG(ctp::logDEBUG, *_log) << "Number of Electrons is "<<N<< " transitiondensity is used for fit" << flush;
140  _do_Transition=true;
141  }
142  else{
143  double Znuc_ECP = 0.0;
144  double Znuc=0.0;
145  for ( unsigned j = 0; j < _atoms.size(); j++){
146  Znuc_ECP += _elements.getNucCrgECP(_atoms[j]->type);
147  Znuc+= _elements.getNucCrg(_atoms[j]->type);
148  }
149 
150  if (_ECP){
151  if (std::abs(Znuc_ECP-N)<4){
152  CTP_LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus ECP_Nucleus charge is "<<Znuc_ECP-N<< " you use ECPs, sounds good" << flush;
153  }
154  else if (std::abs(Znuc-N)<4){
155  CTP_LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus real Nucleus charge is "<<Znuc-N<< " you are sure you want ECPs?" << flush;
156  }
157  else{
158  CTP_LOG(ctp::logDEBUG, *_log) <<"Warning: Your molecule is highly ionized and you want ECPs, sounds interesting" << flush;
159  }
160  netcharge=Znuc_ECP-N;
161  }
162  else{
163  if (std::abs(Znuc-N)<4){
164  CTP_LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus Nucleus charge is "<<Znuc-N<< " you probably do not use ECPs, if you do use ECPs please use the option. Otherwise you are fine" << flush;
165  }
166  else if(std::abs(Znuc_ECP-N)<4){
167  CTP_LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus ECP_Nucleus charge is "<<Znuc_ECP-N<< " you probably use ECPs, if you do use ECPs please use the option to switch on" << flush;
168  }
169  else{
170  CTP_LOG(ctp::logDEBUG, *_log) <<"Warning: Your molecule is highly ionized and you use real core potentials, sounds interesting" << flush;
171  }
172  }
173  _do_Transition=false;
174  }
175  netcharge=round(netcharge);
176  CTP_LOG(ctp::logDEBUG, *_log) <<"Netcharge constrained to " << netcharge<< flush;
177 
178  return netcharge;
179 }

183 void Espfit::Fit2Density_analytic(std::vector< ctp::QMAtom* >& _atomlist, ub::matrix<double> &_dmat,AOBasis &_basis) {
184  double Nm2Bohr=tools::conv::nm2bohr;
185  double A2nm=tools::conv::ang2nm;
186  // setting up grid
187  Grid _grid;
188  _grid.setAtomlist(&_atomlist);
189  _grid.setupCHELPgrid();
190 
191  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Done setting up CHELPG grid with " << _grid.getsize() << " points " << endl;
192  // Calculating nuclear potential at gridpoints
193 
194  ub::vector<double> _ESPatGrid = ub::zero_vector<double>(_grid.getsize());
195 
196  AOOverlap overlap;
197  overlap.Initialize(_basis.AOBasisSize());
198  overlap.Fill(_basis);
199  const ub::vector<double> DMATasarray=_dmat.data();
200  const ub::vector<double> AOOasarray=overlap.Matrix().data();
201  double N=0.0;
202  #pragma omp parallel for reduction(+:N)
203  for ( unsigned _i =0; _i < DMATasarray.size(); _i++ ){
204  N =N+ DMATasarray(_i)*AOOasarray(_i);
205  }
206 
207  double netcharge=getNetcharge( _atomlist,N );
208  if(!_do_Transition){
209  ub::vector<double> _NucPatGrid = EvalNuclearPotential( _atomlist, _grid);
210  _ESPatGrid += _NucPatGrid;
211  }
212 
213  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating ESP at CHELPG grid points" << flush;
214  #pragma omp parallel for
215  for ( int i = 0 ; i < _grid.getsize(); i++){
216  // AOESP matrix
217  AOESP _aoesp;
218  _aoesp.Initialize(_basis.AOBasisSize());
219  _aoesp.Fill(_basis, _grid.getGrid()[i]*Nm2Bohr);
220  const ub::vector<double> AOESPasarray=_aoesp.Matrix().data();
221 
222  for ( unsigned _i =0; _i < DMATasarray.size(); _i++ ){
223  _ESPatGrid(i) -= DMATasarray(_i)*AOESPasarray(_i);
224  }
225  }
226 
227  std::vector< tools::vec > _fitcenters;
228 
229  for ( unsigned j = 0; j < _atomlist.size(); j++){
230  tools::vec _pos=A2nm*_atomlist[j]->getPos();
231  _fitcenters.push_back(_pos);
232  }
233  std::vector<double> _charges = FitPartialCharges(_fitcenters,_grid, _ESPatGrid, netcharge);
234 
235  //Write charges to qmatoms
236  for ( unsigned _i =0 ; _i < _atomlist.size(); _i++){
237  _atomlist[_i]->charge=_charges[_i];
238  }
239  return;

242 std::vector<double> Espfit::FitPartialCharges( std::vector< tools::vec >& _fitcenters, Grid& _grid, ub::vector<double>& _potential, double& _netcharge ){
243  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Setting up Matrices for fitting of size "<< _fitcenters.size()+1 <<" x " << _fitcenters.size()+1<< flush;
244 
245  const std::vector< tools::vec >& _gridpoints=_grid.getGrid();
246 
247  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using "<< _fitcenters.size() <<" Fittingcenters and " << _gridpoints.size()<< " Gridpoints."<< flush;
248 
249  ub::matrix<double> _Amat = ub::zero_matrix<double>(_fitcenters.size()+1,_fitcenters.size()+1);
250  ub::matrix<double> _Bvec = ub::zero_matrix<double>(_fitcenters.size()+1,1);
251  //boost::progress_display show_progress( _fitcenters.size() );
252  // setting up _Amat
253  #pragma omp parallel for
254  for ( unsigned _i =0 ; _i < _Amat.size1()-1; _i++){
255  for ( unsigned _j=_i; _j<_Amat.size2()-1; _j++){
256  for ( unsigned _k=0; _k < _gridpoints.size(); _k++){
257  double dist_i = tools::abs(_fitcenters[_i]-_gridpoints[_k])*tools::conv::nm2bohr;
258  double dist_j = tools::abs(_fitcenters[_j]-_gridpoints[_k])*tools::conv::nm2bohr;
259 
260  _Amat(_i,_j) += 1.0/dist_i/dist_j;
261  }
262  _Amat(_j,_i) = _Amat(_i,_j);
263  }
264  }
265 
266  for ( unsigned _i =0 ; _i < _Amat.size1(); _i++){
267  _Amat(_i,_Amat.size1()-1) = 1.0;
268  _Amat(_Amat.size1()-1,_i) = 1.0;
269  }
270  _Amat(_Amat.size1()-1,_Amat.size1()-1) = 0.0;
271 
272  // setting up Bvec
273  #pragma omp parallel for
274  for ( unsigned _i =0 ; _i < _Bvec.size1()-1; _i++){
275  for ( unsigned _k=0; _k < _gridpoints.size(); _k++){
276  double dist_i = tools::abs(_fitcenters[_i]-_gridpoints[_k])*tools::conv::nm2bohr;
277  _Bvec(_i,0) += _potential(_k)/dist_i;
278  }
279  }
280 
281  _Bvec(_Bvec.size1()-1,0) = _netcharge; //netcharge!!!!
282  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Inverting Matrices "<< flush;
283  // invert _Amat
284  ub::matrix<double> _Amat_inverse = ub::zero_matrix<double>(_fitcenters.size()+1,_fitcenters.size()+1);
285 
286 
287 
288  if(_do_svd){
289  int notfittedatoms=linalg_invert_svd( _Amat , _Amat_inverse,_conditionnumber);
290  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << "SVD Done. "<<notfittedatoms<<" could not be fitted and are set to zero."<< flush;
291  }
292  else{
293  linalg_invert( _Amat , _Amat_inverse);
294  }
295 
296  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Inverting Matrices done."<< flush;
297  //_Amat.resize(0,0);
298 
299 
300 
301  ub::matrix<double> _charges = ub::prod(_Amat_inverse,_Bvec);
302 
303  std::vector<double> _result;
304  for ( unsigned _i = 0; _i < _charges.size1(); _i++ ){
305  _result.push_back(_charges(_i,0));
306  }
307 
308  double _sumcrg = 0.0;
309  for ( unsigned _i =0 ; _i < _fitcenters.size(); _i++){
310 
311  //CTP_LOG(ctp::logDEBUG, *_log) << " Center " << _i << " FitCharge: " << _result[_i] << "pos " << _fitcenters[_i] << flush;
312  _sumcrg += _result[_i];
313  }
314 
315  CTP_LOG(ctp::logDEBUG, *_log) << " Sum of fitted charges: " << _sumcrg << flush;
316 
317  // get RMSE
318  double _rmse = 0.0;
319  double _totalPotSq = 0.0;
320  for ( unsigned _k=0 ; _k < _gridpoints.size(); _k++ ){
321  double temp = 0.0;
322  for ( unsigned _i=0; _i < _fitcenters.size(); _i++ ){
323  double dist = tools::abs(_gridpoints[_k]-_fitcenters[_i])*tools::conv::nm2bohr;
324  temp += _result[_i]/dist;
325  }
326  _rmse += (_potential(_k) - temp)*(_potential(_k) - temp);
327  _totalPotSq += _potential(_k)*_potential(_k);
328  }
329  CTP_LOG(ctp::logDEBUG, *_log) << " RMSE of fit: " << sqrt(_rmse/_gridpoints.size()) << flush;
330  CTP_LOG(ctp::logDEBUG, *_log) << " RRMSE of fit: " << sqrt(_rmse/_totalPotSq) << flush;
331 
332  return _result;
333  }
