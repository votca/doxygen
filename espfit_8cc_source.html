35 void Espfit::EvaluateAPECharges(Grid& _targetgrid, Grid& _chargepositions){
36  std::vector<ctp::APolarSite*> charges=_chargepositions.Sites();
37  std::vector<ctp::APolarSite*> positions=_targetgrid.Sites();
38  std::vector<ctp::APolarSite*>::iterator qit;
39  std::vector<ctp::APolarSite*>::iterator pit;
40  for (pit=positions.begin();pit!=positions.end();++pit){
41  double potential=0.0;
42  vec pos=(*pit)->getPos();
43  for (qit=charges.begin();qit!=charges.end();++qit){
44  double dist=abs((*qit)->getPos()-pos);
45  potential+=((*qit)->getQ00())/dist;
46  }
47  (*pit)->setPhi(potential,0.0);
48  }
49  }

51 void Espfit::FitAPECharges(Grid& _targetgrid_fg, Grid& _targetgrid_bg, Grid& _chargepositions, double& netcharge){

58 
59  if(_chargepositions.getsize() >_targetgrid_fg.getsize()){
60  throw std::runtime_error("Fit underdetermined, change grid options");

62 
63  std::vector< ctp::APolarSite* > _charges= _chargepositions.Sites();
64  std::vector< ctp::APolarSite* > _target_fg= _targetgrid_fg.Sites();
65  std::vector< ctp::APolarSite* > _target_bg= _targetgrid_bg.Sites();

67  LOG(ctp::logDEBUG, *_log) << " Grid of size bg:" << _targetgrid_bg.getsize() << flush;
68  LOG(ctp::logDEBUG, *_log) << " Chargepositions:" << _chargepositions.getsize() << flush;
69 
70  std::vector< ub::vector<double> > _chargepos;
71  std::vector< ctp::APolarSite* >::iterator sit;
72  for (sit=_charges.begin(); sit!=_charges.end(); ++sit) {
73  ub::vector<double> temp= ((*sit)->getPos()).converttoub();
74  _chargepos.push_back(temp);
75  }
76  ub::vector<double> _potential=ub::zero_vector<double>(_targetgrid_fg.getsize());
77  for( int i=0; i<_targetgrid_fg.getsize();i++){
78  _potential(i)=Int2Hartree*(_target_fg[i]->getPhi()+_target_bg[i]->getPhi());
79  }
80 
81  LOG(ctp::logDEBUG, *_log) << " Fitting APE to Chargeshell with netcharge " << netcharge <<"e."<< flush;
82  std::vector<double>_chargesfromfit=FitPartialCharges(_chargepos,_targetgrid_fg,_potential,netcharge);

84  for (unsigned i=0; i<_charges.size();i++) {
85  _charges[i]->setQ00(_chargesfromfit[i],state);
86  }

172 ub::vector<double> Espfit::EvalNuclearPotential(std::vector< ctp::QMAtom* >& _atoms, Grid _grid) {
173  ub::vector<double> _NucPatGrid = ub::zero_vector<double>(_grid.getsize());
174 
175  double Znuc=0.0;
176  std::vector< vec >& _gridpoints = _grid.getGrid();
177  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating ESP of nuclei at CHELPG grid points" << flush;
178 
179  for (unsigned i = 0; i < _gridpoints.size(); i++) {
180  double x_k = _gridpoints[i].getX();
181  double y_k = _gridpoints[i].getY();
182  double z_k = _gridpoints[i].getZ();
183  for (unsigned j = 0; j < _atoms.size(); j++) {
184 
185  double x_j = tools::conv::ang2nm * _atoms[j]->x;
186  double y_j = tools::conv::ang2nm * _atoms[j]->y;
187  double z_j = tools::conv::ang2nm * _atoms[j]->z;
188  if (_ECP) {
189  Znuc = _elements.getNucCrgECP(_atoms[j]->type);
190  } else {
191  Znuc = _elements.getNucCrg(_atoms[j]->type);
192  }
193  double dist_j = sqrt((x_j - x_k)*(x_j - x_k) + (y_j - y_k)*(y_j - y_k) + (z_j - z_k)*(z_j - z_k)) * tools::conv::nm2bohr;
194  _NucPatGrid(i) += Znuc / dist_j;
195  }
196 
197  }
198  return _NucPatGrid;
199 }

201 double Espfit::getNetcharge( std::vector< ctp::QMAtom* >& _atoms, double N ){
202  double netcharge=0.0;
203  if( std::abs(N)<0.05){
204  //LOG(ctp::logDEBUG, *_log) << "Number of Electrons is "<<N<< " transitiondensity is used for fit" << flush;
205  _do_Transition=true;
206  }
207  else{
208  double Znuc_ECP = 0.0;
209  double Znuc=0.0;
210  for ( unsigned j = 0; j < _atoms.size(); j++){
211  Znuc_ECP += _elements.getNucCrgECP(_atoms[j]->type);
212  Znuc+= _elements.getNucCrg(_atoms[j]->type);
213  }
214 
215  if (_ECP){
216  if (std::abs(Znuc_ECP-N)<4){
217  LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus ECP_Nucleus charge is "<<Znuc_ECP-N<< " you use ECPs, sounds good" << flush;
218  }
219  else if (std::abs(Znuc-N)<4){
220  LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus real Nucleus charge is "<<Znuc-N<< " you are sure you want ECPs?" << flush;
221  }
222  else{
223  LOG(ctp::logDEBUG, *_log) <<"Warning: Your molecule is highly ionized and you want ECPs, sounds interesting" << flush;
224  }
225  netcharge=Znuc_ECP-N;
226  }
227  else{
228  if (std::abs(Znuc-N)<4){
229  LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus Nucleus charge is "<<Znuc-N<< " you probably do not use ECPs, if you do use ECPs please use the option. Otherwise you are fine" << flush;
230  }
231  else if(std::abs(Znuc_ECP-N)<4){
232  LOG(ctp::logDEBUG, *_log) <<"Number of Electrons minus ECP_Nucleus charge is "<<Znuc_ECP-N<< " you probably use ECPs, if you do use ECPs please use the option to switch on" << flush;
233  }
234  else{
235  LOG(ctp::logDEBUG, *_log) <<"Warning: Your molecule is highly ionized and you use real core potentials, sounds interesting" << flush;
236  }
237  }
238  _do_Transition=false;
239  }
240  netcharge=round(netcharge);
241  LOG(ctp::logDEBUG, *_log) <<"Netcharge constrained to " << netcharge<< flush;
242 
243  return netcharge;
244 }

248 void Espfit::Fit2Density_analytic(std::vector< ctp::QMAtom* >& _atomlist, ub::matrix<double> &_dmat,AOBasis &_basis) {
249  double Nm2Bohr=18.8972598860;
250  double A2nm=0.1;
251  // setting up grid
252  Grid _grid;
253  _grid.setAtomlist(&_atomlist);
254  _grid.setupCHELPgrid();
255  //_grid.printGridtoxyzfile("grid.xyz");
256  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Done setting up CHELPG grid with " << _grid.getsize() << " points " << endl;
257  // Calculating nuclear potential at gridpoints
258 
259  ub::vector<double> _ESPatGrid = ub::zero_vector<double>(_grid.getsize());
260 
261  AOOverlap overlap;
262  overlap.Initialize(_basis._AOBasisSize);
263  overlap.Fill(&_basis);
264  ub::vector<double> DMATasarray=_dmat.data();
265  ub::vector<double> AOOasarray=overlap._aomatrix.data();
266  double N=0.0;
267  #pragma omp parallel for reduction(+:N)
268  for ( unsigned _i =0; _i < DMATasarray.size(); _i++ ){
269  N =N+ DMATasarray(_i)*AOOasarray(_i);
270  }
271 
272  double netcharge=getNetcharge( _atomlist,N );
273  if(!_do_Transition){
274  ub::vector<double> _NucPatGrid = EvalNuclearPotential( _atomlist, _grid);
275  _ESPatGrid += _NucPatGrid;
276  }
277

279  #pragma omp parallel for
280  for ( int i = 0 ; i < _grid.getsize(); i++){
281  // AOESP matrix
282  AOESP _aoesp;
283  _aoesp.Initialize(_basis._AOBasisSize);
284  _aoesp.Fill(&_basis, _grid.getGrid()[i]*Nm2Bohr);
285  ub::vector<double> AOESPasarray=_aoesp._aomatrix.data();
286 
287  for ( unsigned _i =0; _i < DMATasarray.size(); _i++ ){
288  _ESPatGrid(i) -= DMATasarray(_i)*AOESPasarray(_i);
289  }
290  }

314 std::vector<double> Espfit::FitPartialCharges( std::vector< ub::vector<double> >& _fitcenters, Grid& _grid, ub::vector<double>& _potential, double& _netcharge ){
315  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Setting up Matrices for fitting of size "<< _fitcenters.size()+1 <<" x " << _fitcenters.size()+1<< flush;
316 
317  std::vector< vec >& _gridpoints=_grid.getGrid();
318  //cout << "x " << _gridpoints[0](0)<< " y " << _gridpoints[0](1)<< " z " << _gridpoints[0](1);
319  //cout << "x " << _fitcenters[0](0)<< " y " << _fitcenters[0](1)<< " z " << _fitcenters[0](1);
320  // Fitting atomic partial charges
321 
322  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using "<< _fitcenters.size() <<" Fittingcenters and " << _gridpoints.size()<< " Gridpoints."<< flush;
323 
324  ub::matrix<double> _Amat = ub::zero_matrix<double>(_fitcenters.size()+1,_fitcenters.size()+1);
325  ub::matrix<double> _Bvec = ub::zero_matrix<double>(_fitcenters.size()+1,1);
326  //boost::progress_display show_progress( _fitcenters.size() );
327  // setting up _Amat
328  #pragma omp parallel for
329  for ( unsigned _i =0 ; _i < _Amat.size1()-1; _i++){
330  double x_i = _fitcenters[_i](0);
331  double y_i = _fitcenters[_i](1);
332  double z_i = _fitcenters[_i](2);
333  //++show_progress;
334  for ( unsigned _j=_i; _j<_Amat.size2()-1; _j++){
335  double x_j = _fitcenters[_j](0);
336  double y_j = _fitcenters[_j](1);
337  double z_j = _fitcenters[_j](2);
338  for ( unsigned _k=0; _k < _gridpoints.size(); _k++){
339 
340  double x_k = _gridpoints[_k].getX();
341  double y_k = _gridpoints[_k].getY();
342  double z_k = _gridpoints[_k].getZ();
343 
344  double dist_i = sqrt( (x_i - x_k)*(x_i - x_k) + (y_i - y_k)*(y_i - y_k) + (z_i - z_k)*(z_i - z_k) )*tools::conv::nm2bohr;
345  double dist_j = sqrt( (x_j - x_k)*(x_j - x_k) + (y_j - y_k)*(y_j - y_k) + (z_j - z_k)*(z_j - z_k) )*tools::conv::nm2bohr;
346 
347  _Amat(_i,_j) += 1.0/dist_i/dist_j;
348  }
349  _Amat(_j,_i) = _Amat(_i,_j);
350  }
351  }
352 
353  for ( unsigned _i =0 ; _i < _Amat.size1(); _i++){
354  _Amat(_i,_Amat.size1()-1) = 1.0;
355  _Amat(_Amat.size1()-1,_i) = 1.0;
356  }
357  _Amat(_Amat.size1()-1,_Amat.size1()-1) = 0.0;
358 
359  // setting up Bvec
360  #pragma omp parallel for
361  for ( unsigned _i =0 ; _i < _Bvec.size1()-1; _i++){
362  double x_i = _fitcenters[_i](0);
363  double y_i = _fitcenters[_i](1);
364  double z_i = _fitcenters[_i](2);
365  for ( unsigned _k=0; _k < _gridpoints.size(); _k++){
366 
367  double x_k = _gridpoints[_k].getX();
368  double y_k = _gridpoints[_k].getY();
369  double z_k = _gridpoints[_k].getZ();
370 
371  double dist_i = sqrt( (x_i - x_k)*(x_i - x_k) + (y_i - y_k)*(y_i - y_k) + (z_i - z_k)*(z_i - z_k) )*tools::conv::nm2bohr;
372  _Bvec(_i,0) += _potential(_k)/dist_i;
373  }
374  }
375 
376  _Bvec(_Bvec.size1()-1,0) = _netcharge; //netcharge!!!!
377  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Inverting Matrices "<< flush;
378  // invert _Amat
379  ub::matrix<double> _Amat_inverse = ub::zero_matrix<double>(_fitcenters.size()+1,_fitcenters.size()+1);
380 
381 
382 
383  if(_do_svd){
384  int notfittedatoms=linalg_invert_svd( _Amat , _Amat_inverse,_conditionnumber);
385  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << "SVD Done. "<<notfittedatoms<<" could not be fitted and are set to zero."<< flush;
386  }
387  else{
388  linalg_invert( _Amat , _Amat_inverse);
389  }
390 
391  LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Inverting Matrices done."<< flush;
392  //_Amat.resize(0,0);
393 
394 
395 
396  ub::matrix<double> _charges = ub::prod(_Amat_inverse,_Bvec);
397 
398  std::vector<double> _result;
399  for ( unsigned _i = 0; _i < _charges.size1(); _i++ ){
400  _result.push_back(_charges(_i,0));
401  }
402 
403  double _sumcrg = 0.0;
404  for ( unsigned _i =0 ; _i < _fitcenters.size(); _i++){
405 
406  //LOG(logDEBUG, *_log) << " Center " << _i << " FitCharge: " << _result[_i] << flush;
407  _sumcrg += _result[_i];
408  }
409 
410  LOG(ctp::logDEBUG, *_log) << " Sum of fitted charges: " << _sumcrg << flush;
411 
412  // get RMSE
413  double _rmse = 0.0;
414  double _totalPotSq = 0.0;
415  for ( unsigned _k=0 ; _k < _gridpoints.size(); _k++ ){
416  double x_k = _gridpoints[_k].getX();
417  double y_k = _gridpoints[_k].getY();
418  double z_k = _gridpoints[_k].getZ();
419  double temp = 0.0;
420  for ( unsigned _i=0; _i < _fitcenters.size(); _i++ ){
421  double x_i = _fitcenters[_i](0);
422  double y_i = _fitcenters[_i](1);
423  double z_i = _fitcenters[_i](2);
424 
425  double dist = sqrt( (x_i - x_k)*(x_i - x_k) + (y_i - y_k)*(y_i - y_k) + (z_i - z_k)*(z_i - z_k) )*tools::conv::nm2bohr;
426  temp += _result[_i]/dist;
427  }
428  _rmse += (_potential(_k) - temp)*(_potential(_k) - temp);
429  _totalPotSq += _potential(_k)*_potential(_k);
430  }
431  LOG(ctp::logDEBUG, *_log) << " RMSE of fit: " << sqrt(_rmse/_gridpoints.size()) << flush;
432  LOG(ctp::logDEBUG, *_log) << " RRMSE of fit: " << sqrt(_rmse/_totalPotSq) << flush;
433 
434  return _result;
435  }
