35 void Espfit::Fit2Density(std::vector< QMAtom* >& atomlist,const Eigen::MatrixXd &dmat,const AOBasis &basis,std::string gridsize) {
36 
37 
38  // setting up grid
39  Grid grid;
40  grid.setupCHELPGGrid(atomlist);
41  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Done setting up CHELPG grid with " << grid.getsize() << " points " << flush;
42 
43  // Calculating nuclear potential at gridpoints
44  AOOverlap overlap;
45  overlap.Fill(basis);
46  double N_comp=dmat.cwiseProduct(overlap.Matrix()).sum();
47 
48  NumericalIntegration numway;
49 
50  numway.GridSetup(gridsize,atomlist,basis);
51  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Setup "<<gridsize<<" Numerical Grid with "<<numway.getGridSize()<<" gridpoints."<< flush;
52  double N=numway.IntegrateDensity(dmat);
53  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculated Densities at Numerical Grid, Number of electrons is "<< N << flush;
54 
55  if(std::abs(N-N_comp)>0.001){
56  CTP_LOG(ctp::logDEBUG, *_log) <<"=======================" << flush;
57  CTP_LOG(ctp::logDEBUG, *_log) <<"WARNING: Calculated Densities at Numerical Grid, Number of electrons "<< N <<" is far away from the the real value "<< N_comp<<", you should increase the accuracy of the integration grid."<< flush;
58  N=N_comp;
59  CTP_LOG(ctp::logDEBUG, *_log) <<"WARNING: Electronnumber set to "<< N << flush;
60  CTP_LOG(ctp::logDEBUG, *_log) <<"=======================" << flush;
61  }
62 
63  double netcharge=getNetcharge( atomlist,N );
64 
65  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating ESP at CHELPG grid points" << flush;
66  //boost::progress_display show_progress( _grid.getsize() );
67  #pragma omp parallel for
68  for ( unsigned i = 0 ; i < grid.getsize(); i++){
69  grid.getGridValues()(i)=numway.IntegratePotential(grid.getGridPositions()[i]);
70  }
71 
72  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Electron contribution calculated" << flush;
73  if (!_do_Transition){
74  EvalNuclearPotential( atomlist, grid );
75  }
76 
77  FitPartialCharges(atomlist,grid, netcharge);
78  return;
79  }

82 void Espfit::EvalNuclearPotential(const std::vector< QMAtom* >& atoms, Grid& grid) {
83 
84  const std::vector< tools::vec >& _gridpoints = grid.getGridPositions();
85  Eigen::VectorXd& _gridvalues=grid.getGridValues();
86  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Calculating ESP of nuclei at CHELPG grid points" << flush;
87 
88  for (unsigned i = 0; i < _gridpoints.size(); i++) {
89  for (unsigned j = 0; j < atoms.size(); j++) {
90  const tools::vec& posatom=atoms[j]->getPos();
91  double Znuc=atoms[j]->getNuccharge();
92  double dist_j = tools::abs(_gridpoints[i]-posatom);
93  _gridvalues(i) += Znuc / dist_j;
94  }
95  }
96  return;
97 }

99 double Espfit::getNetcharge(const std::vector< QMAtom* >& atoms, double N ){
100  double netcharge = 0.0;
101  if (std::abs(N) < 0.05) {
102  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp()<< " Number of Electrons is " << N << " transitiondensity is used for fit" << flush;
103  _do_Transition = true;
104  } else {
105  double Znuc = 0.0;
106  for (unsigned j = 0; j < atoms.size(); j++) {
107  Znuc += atoms[j]->getNuccharge();
108  }
109 
110  if (std::abs(Znuc - N) < 4) {
111  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp()<< " Number of Electrons minus Nucleus charge is " << Znuc - N << "." << flush;
112  } else {
113  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp()<< " Warning: Your molecule is highly ionized." << flush;
114  }
115  netcharge = Znuc - N;
116  _do_Transition = false;
117  }
118 
119  netcharge = std::round(netcharge);
120  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp()<< " Netcharge constrained to " << netcharge << flush;
121 
122  return netcharge;
123 }

157 void Espfit::FitPartialCharges( std::vector< QMAtom* >& atomlist,const Grid& grid,double netcharge ){
158 
159  const int NoOfConstraints=1+_regionconstraint.size()+_pairconstraint.size();
160  const int matrixSize=atomlist.size()+NoOfConstraints;
161  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Setting up Matrices for fitting of size "<< matrixSize <<" x " << matrixSize<< flush;
162 
163  const std::vector< tools::vec >& gridpoints=grid.getGridPositions();
164  const Eigen::VectorXd & potential=grid.getGridValues();
165  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Using "<< atomlist.size() <<" Fittingcenters and " << gridpoints.size()<< " Gridpoints."<< flush;
166 
167  Eigen::MatrixXd Amat = Eigen::MatrixXd::Zero(matrixSize,matrixSize);
168  Eigen::VectorXd Bvec = Eigen::VectorXd::Zero(matrixSize);
169  // setting up _Amat
170  #pragma omp parallel for
171  for ( unsigned _i =0 ; _i < atomlist.size(); _i++){
172  for ( unsigned _j=_i; _j<atomlist.size(); _j++){
173  for ( unsigned _k=0; _k < gridpoints.size(); _k++){
174  double dist_i = tools::abs(atomlist[_i]->getPos()-gridpoints[_k]);
175  double dist_j = tools::abs(atomlist[_j]->getPos()-gridpoints[_k]);
176 
177  Amat(_i,_j) += 1.0/dist_i/dist_j;
178  }
179  Amat(_j,_i) = Amat(_i,_j);
180  }
181  }
182 
183  // setting up Bvec
184  #pragma omp parallel for
185  for ( unsigned _i =0 ; _i < atomlist.size(); _i++){
186  for ( unsigned _k=0; _k < gridpoints.size(); _k++){
187  double dist_i = tools::abs(atomlist[_i]->getPos()-gridpoints[_k]);
188  Bvec(_i) += potential(_k)/dist_i;
189  }
190  }
191  //Total charge constraint
192  for ( unsigned _i =0 ; _i < atomlist.size()+1; _i++){
193  Amat(_i,atomlist.size()) = 1.0;
194  Amat(atomlist.size(),_i) = 1.0;
195  }
196  Amat(atomlist.size(),atomlist.size()) = 0.0;
197  Bvec(atomlist.size()) = netcharge; //netcharge!!!!
198 
199 
200  // Pairconstraint
201  for (unsigned i=0;i<_pairconstraint.size();i++){
202  const std::pair<int,int>& pair=_pairconstraint[i];
203  Amat(pair.first,atomlist.size()+1+i)=1.0;
204  Amat(atomlist.size()+1+i,pair.first)=1.0;
205  Amat(pair.second,atomlist.size()+1+i)=-1.0;
206  Amat(atomlist.size()+1+i,pair.second)=-1.0;
207  }
208 
209  //Regionconstraint
210  for (unsigned i=0;i<_regionconstraint.size();i++){
211  const region& reg=_regionconstraint[i];
212  for (const int& index:reg.atomindices){
213  Amat(index,atomlist.size()+i+1+_pairconstraint.size())=1.0;
214  Amat(atomlist.size()+i+1+_pairconstraint.size(),index)=1.0;
215  }
216  Bvec(atomlist.size()+i+1+_pairconstraint.size())=reg.charge;
217  }
218 
219  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Solving linear Equation "<< flush;
220  Eigen::VectorXd charges;
221  if(_do_svd){
222  Eigen::JacobiSVD<Eigen::MatrixXd> svd;
223  svd.setThreshold(_conditionnumber);
224  svd.compute(Amat,Eigen::ComputeThinU | Eigen::ComputeThinV);
225  charges=svd.solve(Bvec);
226  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " SVD Done. "<<flush;
227  if((Bvec.size()-svd.nonzeroSingularValues())!=0){
228  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() <<Bvec.size()-svd.nonzeroSingularValues()<<" Sites could not be fitted and are set to zero."<< flush;
229  }
230  }
231  else{
232  Eigen::ColPivHouseholderQR<Eigen::MatrixXd> QR(Amat);
233  if(QR.info()!=Eigen::ComputationInfo::Success){
234  throw std::runtime_error("Espfit: Solving the constrained equation failed. Maybe try SVD.");
235  }
236  charges=QR.solve(Bvec);
237  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Solved linear least square fit ."<< flush;
238  }
239  //remove constraint
240  charges.conservativeResize(atomlist.size());
241 
242  CTP_LOG(ctp::logDEBUG, *_log) << " Sum of fitted charges: " << charges.sum() << flush;
243  for (unsigned i=0;i<atomlist.size();i++){
244  atomlist[i]->setPartialcharge(charges(i));
245  }
246 
247  // get RMSE
248  double rmse = 0.0;
249  double totalPotSq = 0.0;
250  for ( unsigned _k=0 ; _k < gridpoints.size(); _k++ ){
251  double temp = 0.0;
252  for ( const QMAtom* atom:atomlist){
253  double dist = tools::abs(gridpoints[_k]-atom->getPos());
254  temp += atom->getPartialcharge()/dist;
255  }
256  rmse += (potential(_k) - temp)*(potential(_k) - temp);
257  totalPotSq += potential(_k)*potential(_k);
258  }
259  CTP_LOG(ctp::logDEBUG, *_log) << " RMSE of fit: " << sqrt(rmse/gridpoints.size()) << flush;
260  CTP_LOG(ctp::logDEBUG, *_log) << " RRMSE of fit: " << sqrt(rmse/totalPotSq) << flush;
261 
262  return;
263  }
