31 StaticSegment Espfit::Fit2Density(const Orbitals& orbitals,
32  const QMState& state, std::string gridsize) {
33 
34  const Eigen::MatrixXd dmat = orbitals.DensityMatrixFull(state);
35  // setting up grid
36  Grid grid;
37  grid.setupCHELPGGrid(orbitals.QMAtoms());
38  XTP_LOG(Log::info, _log) << TimeStamp()
39  << " Done setting up CHELPG grid with "
40  << grid.size() << " points " << flush;
41 
42  // Calculating nuclear potential at gridpoints
43  AOBasis basis = orbitals.SetupDftBasis();
44  AOOverlap overlap;
45  overlap.Fill(basis);
46  double N_comp = dmat.cwiseProduct(overlap.Matrix()).sum();
47 
48  NumericalIntegration numway;
49 
50  numway.GridSetup(gridsize, orbitals.QMAtoms(), basis);
51  XTP_LOG(Log::info, _log) << TimeStamp() << " Setup " << gridsize
52  << " Numerical Grid with " << numway.getGridSize()
53  << " gridpoints." << flush;
54  double N = numway.IntegrateDensity(dmat);
55  XTP_LOG(Log::error, _log)
56  << TimeStamp()
57  << " Calculated Densities at Numerical Grid, Number of electrons is " << N
58  << flush;
59 
60  if (std::abs(N - N_comp) > 0.001) {
61  XTP_LOG(Log::error, _log) << "=======================" << flush;
62  XTP_LOG(Log::error, _log)
63  << "WARNING: Calculated Densities at Numerical Grid, Number of "
64  "electrons "
65  << N << " is far away from the the real value " << N_comp
66  << ", you should increase the accuracy of the integration grid."
67  << flush;
68  N = N_comp;
69  XTP_LOG(Log::error, _log)
70  << "WARNING: Electronnumber set to " << N << flush;
71  XTP_LOG(Log::error, _log) << "=======================" << flush;
72  }
73 
74  XTP_LOG(Log::error, _log)
75  << TimeStamp() << " Calculating ESP at CHELPG grid points" << flush;
76 #pragma omp parallel for
77  for (Index i = 0; i < grid.size(); i++) {
78  grid.getGridValues()(i) =
79  numway.IntegratePotential(grid.getGridPositions()[i]);
80  }
81 
82  XTP_LOG(Log::info, _log) << TimeStamp() << " Electron contribution calculated"
83  << flush;
84  double netcharge = 0.0;
85  if (!state.isTransition()) {
86  EvalNuclearPotential(orbitals.QMAtoms(), grid);
87  Index Znuc = 0;
88  for (const QMAtom& atom : orbitals.QMAtoms()) {
89  Znuc += atom.getNuccharge();
90  }
91  netcharge = double(Znuc) - N;
92  }
93  netcharge = std::round(netcharge);
94  XTP_LOG(Log::error, _log)
95  << TimeStamp() << " Netcharge constrained to " << netcharge << flush;
96  return FitPartialCharges(orbitals, grid, netcharge);
97  ;
98 }

100 void Espfit::EvalNuclearPotential(const QMMolecule& atoms, Grid& grid) {
101 
102  const std::vector<Eigen::Vector3d>& gridpoints = grid.getGridPositions();
103  Eigen::VectorXd& gridvalues = grid.getGridValues();
104  XTP_LOG(Log::info, _log) << TimeStamp()
105  << " Calculating ESP of nuclei at CHELPG grid points"
106  << flush;
107 
108  for (Index i = 0; i < Index(gridpoints.size()); i++) {
109  for (Index j = 0; j < atoms.size(); j++) {
110  const Eigen::Vector3d& posatom = atoms[j].getPos();
111  Index Znuc = atoms[j].getNuccharge();
112  double dist_j = (gridpoints[i] - posatom).norm();
113  gridvalues(i) += double(Znuc) / dist_j;
114  }
115  }
116  return;
117 }

119 StaticSegment Espfit::FitPartialCharges(const Orbitals& orbitals,
120  const Grid& grid, double netcharge) {
121  const QMMolecule& atomlist = orbitals.QMAtoms();
122  const Index NoOfConstraints =
123  1 + Index(_regionconstraint.size()) + Index(_pairconstraint.size());
124  const Index matrixSize = atomlist.size() + NoOfConstraints;
125  XTP_LOG(Log::info, _log) << TimeStamp()
126  << " Setting up Matrices for fitting of size "
127  << matrixSize << " x " << matrixSize << flush;
128 
129  const std::vector<Eigen::Vector3d>& gridpoints = grid.getGridPositions();
130  const Eigen::VectorXd& potential = grid.getGridValues();
131  XTP_LOG(Log::info, _log) << TimeStamp() << " Using " << atomlist.size()
132  << " Fittingcenters and " << gridpoints.size()
133  << " Gridpoints." << flush;
134 
135  Eigen::MatrixXd Amat = Eigen::MatrixXd::Zero(matrixSize, matrixSize);
136  Eigen::VectorXd Bvec = Eigen::VectorXd::Zero(matrixSize);
137 // setting up _Amat
138 #pragma omp parallel for
139  for (Index i = 0; i < atomlist.size(); i++) {
140  for (Index j = i; j < atomlist.size(); j++) {
141  for (const auto& gridpoint : gridpoints) {
142  double dist_i = (atomlist[i].getPos() - gridpoint).norm();
143  double dist_j = (atomlist[j].getPos() - gridpoint).norm();
144 
145  Amat(i, j) += 1.0 / dist_i / dist_j;
146  }
147  Amat(j, i) = Amat(i, j);
148  }
149  }
150 
151  // setting up Bvec
152 #pragma omp parallel for
153  for (Index i = 0; i < atomlist.size(); i++) {
154  for (Index j = 0; j < Index(gridpoints.size()); j++) {
155  double dist_i = (atomlist[i].getPos() - gridpoints[j]).norm();
156  Bvec(i) += potential(j) / dist_i;
157  }
158  }
159  // Total charge constraint
160  for (Index i = 0; i < atomlist.size() + 1; i++) {
161  Amat(i, atomlist.size()) = 1.0;
162  Amat(atomlist.size(), i) = 1.0;
163  }
164  Amat(atomlist.size(), atomlist.size()) = 0.0;
165  Bvec(atomlist.size()) = netcharge; // netcharge!!!!
166 
167  // Pairconstraint
168  for (Index i = 0; i < Index(_pairconstraint.size()); i++) {
169  const std::pair<Index, Index>& pair = _pairconstraint[i];
170  Amat(pair.first, atomlist.size() + 1 + i) = 1.0;
171  Amat(atomlist.size() + 1 + i, pair.first) = 1.0;
172  Amat(pair.second, atomlist.size() + 1 + i) = -1.0;
173  Amat(atomlist.size() + 1 + i, pair.second) = -1.0;
174  }
175 
176  // Regionconstraint
177  for (Index i = 0; i < Index(_regionconstraint.size()); i++) {
178  const QMFragment<double>& reg = _regionconstraint[i];
179  for (Index index : reg) {
180  Amat(index, atomlist.size() + i + 1 + _pairconstraint.size()) = 1.0;
181  Amat(atomlist.size() + i + 1 + _pairconstraint.size(), index) = 1.0;
182  }
183  Bvec(atomlist.size() + i + 1 + _pairconstraint.size()) = reg.value();
184  }
185 
186  XTP_LOG(Log::info, _log) << TimeStamp() << " Solving linear Equation "
187  << flush;
188  Eigen::VectorXd charges;
189  if (_do_svd) {
190  Eigen::JacobiSVD<Eigen::MatrixXd> svd;
191  svd.setThreshold(_conditionnumber);
192  svd.compute(Amat, Eigen::ComputeThinU | Eigen::ComputeThinV);
193  charges = svd.solve(Bvec);
194  XTP_LOG(Log::info, _log) << TimeStamp() << " SVD Done. " << flush;
195  if ((Bvec.size() - svd.nonzeroSingularValues()) != 0) {
196  XTP_LOG(Log::error, _log)
197  << TimeStamp() << Bvec.size() - svd.nonzeroSingularValues()
198  << " Sites could not be fitted and are set to zero." << flush;
199  }
200  } else {
201  Eigen::ColPivHouseholderQR<Eigen::MatrixXd> QR(Amat);
202  if (QR.info() != Eigen::ComputationInfo::Success) {
203  throw std::runtime_error(
204  "Espfit: Solving the constrained equation failed. Maybe try SVD.");
205  }
206  charges = QR.solve(Bvec);
207  XTP_LOG(Log::info, _log)
208  << TimeStamp() << " Solved linear least square fit ." << flush;
209  }
210  // remove constraints from charges
211  charges.conservativeResize(atomlist.size());
212  StaticSegment seg =
213  StaticSegment(orbitals.QMAtoms().getType(), orbitals.QMAtoms().getId());
214 
215  XTP_LOG(Log::error, _log)
216  << " Sum of fitted charges: " << charges.sum() << flush;
217  for (Index i = 0; i < atomlist.size(); i++) {
218  seg.push_back(StaticSite(atomlist[i], charges(i)));
219  }
220  // get RMSE
221  double rmse = 0.0;
222  double totalPotSq = 0.0;
223  for (Index k = 0; k < Index(gridpoints.size()); k++) {
224  double temp = 0.0;
225  for (const StaticSite& atom : seg) {
226  double dist = (gridpoints[k] - atom.getPos()).norm();
227  temp += atom.getCharge() / dist;
228  }
229  rmse += (potential(k) - temp) * (potential(k) - temp);
230  totalPotSq += potential(k) * potential(k);
231  }
232  XTP_LOG(Log::error, _log)
233  << " RMSE of fit: " << std::sqrt(rmse / double(gridpoints.size()))
234  << flush;
235  XTP_LOG(Log::error, _log)
236  << " RRMSE of fit: " << std::sqrt(rmse / totalPotSq) << flush;
237 
238  return seg;
239 }
