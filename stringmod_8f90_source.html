65 subroutine parse(str,delims,args,nargs)
66 
67 ! Parses the string 'str' into arguments args(1), ..., args(nargs) based on
68 ! the delimiters contained in the string 'delims'. Preceding a delimiter in
69 ! 'str' by a backslash (\) makes this particular instance not a delimiter.
70 ! The integer output variable nargs contains the number of arguments found.
71 
72 character(len=*) :: str,delims
73 character(len=len_trim(str)) :: strsav
74 character(len=*),dimension(:) :: args
75 
76 strsav=str
77 call compact(str)
78 na=size(args)
79 do i=1,na
80  args(i)=' '
81 end do
82 nargs=0
83 lenstr=len_trim(str)
84 if(lenstr==0) return
85 k=0
86 
87 do
88  if(len_trim(str) == 0) exit
89  nargs=nargs+1
90  call split(str,delims,args(nargs))
91  call removebksl(args(nargs))
92 end do
93 str=strsav
94 
95 end subroutine parse

101 ! Converts multiple spaces and tabs to single spaces; deletes control characters;
102 ! removes initial spaces.
103

252 
253 ! Shifts characters in in the string 'str' n positions (positive values
254 ! denote a right shift and negative values denote a left shift). Characters
255 ! that are shifted off the end are lost. Positions opened up by the shift

267 if(n>0) str=repeat(' ',nabs)//str(:lenstr-nabs) ! shift right
268 return

342 ! convert string to upper case
343 
344 character (len=*):: str
345 character (len=len_trim(str)):: ucstr
346

348 ioffset=iachar('A')-iachar('a')

364  ucstr(i:i)=achar(iav+ioffset)
365  else

432 subroutine match(str,ipos,imatch)
433 
434 ! Sets imatch to the position in string of the delimiter matching the delimiter
435 ! in position ipos. Allowable delimiters are (), [], {}, <>.
436 
437 character(len=*) :: str
438 character :: delim1,delim2,ch
439 
440 lenstr=len_trim(str)
441 delim1=str(ipos:ipos)
442 select case(delim1)
443  case('(')
444  idelim2=iachar(delim1)+1
445  istart=ipos+1
446  iend=lenstr
447  inc=1
448  case(')')
449  idelim2=iachar(delim1)-1
450  istart=ipos-1
451  iend=1
452  inc=-1
453  case('[','{','<')
454  idelim2=iachar(delim1)+2
455  istart=ipos+1
456  iend=lenstr
457  inc=1
458  case(']','}','>')
459  idelim2=iachar(delim1)-2
460  istart=ipos-1
461  iend=1
462  inc=-1
463  case default
464  write(*,*) delim1,' is not a valid delimiter'
465  return
466 end select
467 if(istart < 1 .or. istart > lenstr) then
468  write(*,*) delim1,' has no matching delimiter'
469  return
470 end if
471 delim2=achar(idelim2) ! matching delimiter
472 
473 isum=1
474 do i=istart,iend,inc
475  ch=str(i:i)
476  if(ch /= delim1 .and. ch /= delim2) cycle
477  if(ch == delim1) isum=isum+1
478  if(ch == delim2) isum=isum-1
479  if(isum == 0) exit
480 end do
481 if(isum /= 0) then
482  write(*,*) delim1,' has no matching delimiter'
483  return
484 end if
485 imatch=i
486 
487 return
488 
489 end subroutine match

693 subroutine split(str,delims,before,sep)
694 
695 ! Routine finds the first instance of a character from 'delims' in the
696 ! the string 'str'. The characters before the found delimiter are
697 ! output in 'before'. The characters after the found delimiter are
698 ! output in 'str'. The optional output character 'sep' contains the
699 ! found delimiter. A delimiter in 'str' is treated like an ordinary
700 ! character if it is preceded by a backslash (\). If the backslash
701 ! character is desired in 'str', then precede it with another backslash.
702 
703 character(len=*) :: str,delims,before
704 character,optional :: sep
705 logical :: pres
706 character :: ch,cha
707 
708 pres=present(sep)
709 str=adjustl(str)
710 call compact(str)
711 lenstr=len_trim(str)
712 if(lenstr == 0) return ! string str is empty
713 k=0
714 ibsl=0 ! backslash initially inactive
715 before=' '
716 do i=1,lenstr
717  ch=str(i:i)
718  if(ibsl == 1) then ! backslash active
719  k=k+1
720  before(k:k)=ch
721  ibsl=0
722  cycle
723  end if
724  if(ch == '\\') then ! backslash with backslash inactive
725  k=k+1
726  before(k:k)=ch
727  ibsl=1
728  cycle
729  end if
730  ipos=index(delims,ch)
731  if(ipos == 0) then ! character is not a delimiter
732  k=k+1
733  before(k:k)=ch
734  cycle
735  end if
736  if(ch /= ' ') then ! character is a delimiter that is not a space
737  str=str(i+1:)
738  if(pres) sep=ch
739  exit
740  end if
741  cha=str(i+1:i+1) ! character is a space delimiter
742  iposa=index(delims,cha)
743  if(iposa > 0) then ! next character is a delimiter
744  str=str(i+2:)
745  if(pres) sep=cha
746  exit
747  else
748  str=str(i+1:)
749  if(pres) sep=ch
750  exit
751  end if
752 end do
753 if(i >= lenstr) str=''
754 str=adjustl(str) ! remove initial spaces
755 return
756 
757 end subroutine split

761 subroutine removebksl(str)
762 
763 ! Removes backslash (\) characters. Double backslashes (\\) are replaced
764 ! by a single backslash.
765 
766 character(len=*):: str
767 character(len=1):: ch
768 character(len=len_trim(str))::outstr
769 
770 str=adjustl(str)
771 lenstr=len_trim(str)
772 outstr=' '
773 k=0
774 ibsl=0 ! backslash initially inactive
775 
776 do i=1,lenstr
777  ch=str(i:i)
778  if(ibsl == 1) then ! backslash active
779  k=k+1
780  outstr(k:k)=ch
781  ibsl=0
782  cycle
783  end if
784  if(ch == '\\') then ! backslash with backslash inactive
785  ibsl=1
786  cycle
787  end if
788  k=k+1
789  outstr(k:k)=ch ! non-backslash with backslash inactive
790 end do

792 str=adjustl(outstr)
793 
794 end subroutine removebksl
