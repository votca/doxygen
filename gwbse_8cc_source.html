45 void GWBSE::Initialize(tools::Property *options) {
46 
47 #if (GWBSE_DOUBLE)
48  CTP_LOG(ctp::logDEBUG, *_pLog) << " Compiled with full double support"
49  << flush;
50 #else
51  CTP_LOG(ctp::logDEBUG, *_pLog)
52  << " Compiled with float/double mixture (standard)" << flush;
53 #endif
54 
55  std::string key = Identify();
56 
57  // getting level ranges
58 double qpminfactor=0;
59  double qpmaxfactor=0;
60  double rpamaxfactor=0;
61  double bseminfactor=0;
62  double bsemaxfactor=0;
63 std::string ranges = options->ifExistsReturnElseReturnDefault<std::string>(key + ".ranges",
64  "default");
65 
66  // now check validity, and get rpa, qp, and bse level ranges accordingly
67 
68  if (ranges == "factor") {
69  // get factors
70  rpamaxfactor = options->get(key + ".rpamax").as<double>();
71  qpminfactor = options->get(key + ".qpmin").as<double>();
72  qpmaxfactor = options->get(key + ".qpmax").as<double>();
73  bseminfactor = options->get(key + ".bsemin").as<double>();
74  bsemaxfactor = options->get(key + ".bsemax").as<double>();
75  } else if (ranges == "explicit") {
76  // get explicit numbers
77  _rpamax = options->get(key + ".rpamax").as<unsigned>();
78  _qpmin = options->get(key + ".qpmin").as<unsigned>();
79  _qpmax = options->get(key + ".qpmax").as<unsigned>();
80  _bse_vmin = options->get(key + ".bsemin").as<unsigned>();
81  _bse_cmax = options->get(key + ".bsemax").as<unsigned>();
82  } else if (ranges == "" || ranges == "default") {
83  ranges = "default";
84  } else if (ranges == "full") {
85  ranges = "full";
86  } else {
87  std::cerr << "\nSpecified range option " << ranges << " invalid. ";
88  throw std::runtime_error(
89  "\nValid options are: default,factor,explicit,full");
90  }
91 
92 
93  /* Preparation of calculation parameters:
94  * - number of electrons -> index of HOMO
95  * - number of levels
96  * - highest level considered in RPA
97  * - lowest and highest level considered in GWA
98  * - lowest and highest level considered in BSE
99  * - number of excitations calculates in BSE
100  */
101 
102  _homo = _orbitals->getNumberOfElectrons() - 1; // indexed from 0
103 
104  _reset_3c=options->ifExistsReturnElseReturnDefault<int>(
105  key + ".rebuild_threecenter_freq", 5);
106 
107  _rpamin = 0; // lowest index occ min(gwa%mmin, screening%nsum_low) ! always 1
108  if (ranges == "default" || ranges == "full") {
109  _rpamax = _orbitals->getNumberOfLevels() - 1; // total number of levels
110  } else if (ranges == "factor") {
111  _rpamax = rpamaxfactor * _orbitals->getNumberOfLevels() -
112  1; // total number of levels
113  }
114  if (_rpamax > _orbitals->getNumberOfLevels() - 1) {
115  _rpamax = _orbitals->getNumberOfLevels() - 1;
116  }
117  // convert _qpmin and _qpmax if needed
118  if (ranges == "default") {
119  _qpmin = 0; // indexed from 0
120  _qpmax = 2 * _homo + 1; // indexed from 0
121  } else if (ranges == "factor") {
122  if (_orbitals->getNumberOfElectrons() -
123  int(qpminfactor * _orbitals->getNumberOfElectrons()) - 1 <
124  0) {
125  _qpmin = 0;
126  } else {
127  _qpmin = _orbitals->getNumberOfElectrons() -
128  int(qpminfactor * _orbitals->getNumberOfElectrons()) - 1;
129  }
130  _qpmax = _orbitals->getNumberOfElectrons() +
131  int(qpmaxfactor * _orbitals->getNumberOfElectrons()) - 1;
132  } else if (ranges == "explicit") {
133  _qpmin -= 1;
134  _qpmax -= 1;
135  } else if (ranges == "full") {
136  _qpmin = 0;
137  _qpmax = _orbitals->getNumberOfLevels() - 1;
138  }
139  if (_qpmax > unsigned(_orbitals->getNumberOfLevels() - 1)) {
140  _qpmax = _orbitals->getNumberOfLevels() - 1;
141  }
142 
143 
144 
145  // set BSE band range indices
146  // anything else would be stupid!
147  _bse_vmax = _homo;
148  _bse_cmin = _homo + 1;
149 
150  if (ranges == "default") {
151  _bse_vmin = 0; // indexed from 0
152  _bse_cmax = 2 * _homo + 1; // indexed from 0
153  } else if (ranges == "factor") {
154  _bse_vmin = _orbitals->getNumberOfElectrons() -
155  int(bseminfactor * _orbitals->getNumberOfElectrons()) - 1;
156  if (_orbitals->getNumberOfElectrons() -
157  int(bseminfactor * _orbitals->getNumberOfElectrons()) - 1 <
158  0) {
159  _bse_vmin = 0;
160  }
161  _bse_cmax = _orbitals->getNumberOfElectrons() +
162  int(bsemaxfactor * _orbitals->getNumberOfElectrons()) - 1;
163  } else if (ranges == "explicit") {
164  _bse_vmin -= 1;
165  _bse_cmax -= 1;
166  } else if (ranges == "full") {
167  _bse_vmin = 0;
168  _bse_cmax = _orbitals->getNumberOfLevels() - 1;
169  }
170  if (_bse_cmax > unsigned(_orbitals->getNumberOfLevels() - 1)) {
171  _bse_cmax = _orbitals->getNumberOfLevels() - 1;
172  }
173 
174  bool ignore_corelevels = options->ifExistsReturnElseReturnDefault<bool>(
175  key + ".ignore_corelevels", false);
176 
177 
178  unsigned _ignored_corelevels = 0;
179  if (ignore_corelevels) {
180  if(!_orbitals->hasECP()){
181  BasisSet basis;
182  basis.LoadPseudopotentialSet("corelevels");//
183  unsigned coreElectrons=0;
184  for(const auto& atom:_orbitals->QMAtoms()){
185  coreElectrons+=basis.getElement(atom->getType())->getNcore();
186  }
187  _ignored_corelevels = coreElectrons/2;
188  }
189 
190  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Ignoring "
191  << _ignored_corelevels << " core levels "
192  << flush;
193  }
194  // autoignore core levels in QP
195  if (ignore_corelevels && (_qpmin < _ignored_corelevels)) {
196  _qpmin = _ignored_corelevels;
197  }
198  // autoignore core levels in BSE
199  if (ignore_corelevels && (_bse_vmin < _ignored_corelevels)) {
200  _bse_vmin = _ignored_corelevels;
201  }
202 
203  int bse_vtotal = _bse_vmax - _bse_vmin + 1;
204  int bse_ctotal = _bse_cmax - _bse_cmin + 1;
205  int bse_size = bse_vtotal * bse_ctotal;
206 
207 
208  // some QP - BSE consistency checks are required
209  if (_bse_vmin < _qpmin) _qpmin = _bse_vmin;
210  if (_bse_cmax > _qpmax) _qpmax = _bse_cmax;
211 
212  _qptotal = _qpmax - _qpmin + 1;
213 
214 
215  // information for hybrid DFT
216 
217  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set RPA level range ["
218  << _rpamin + 1 << ":" << _rpamax + 1 << "]"
219  << flush;
220  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set QP level range ["
221  << _qpmin + 1 << ":" << _qpmax + 1 << "]"
222  << flush;
223  CTP_LOG(ctp::logDEBUG, *_pLog)
224  << ctp::TimeStamp() << " Set BSE level range occ[" << _bse_vmin + 1 << ":"
225  << _bse_vmax + 1 << "] virt[" << _bse_cmin + 1 << ":" << _bse_cmax + 1
226  << "]" << flush;
227  CTP_LOG(ctp::logDEBUG, *_pLog)
228  << ctp::TimeStamp() << " BSE Hamiltonian has size " << bse_size << "x"
229  << bse_size << flush;
230 
231  _bse_maxeigenvectors =
232  options->ifExistsReturnElseReturnDefault<int>(key + ".exctotal", 25);
233  if (_bse_maxeigenvectors > int(bse_size) || _bse_maxeigenvectors < 0) _bse_maxeigenvectors = bse_size;
234  _fragA = options->ifExistsReturnElseReturnDefault<int>(key + ".fragment", -1);
235 
236  std::string BSEtype =
237  options->ifExistsReturnElseReturnDefault<std::string>(key + ".BSEtype", "TDA");
238 
239  if (BSEtype == "full") {
240  _do_full_BSE = true;
241  CTP_LOG(ctp::logDEBUG, *_pLog) << " BSE type: full" << flush;
242  } else {
243  _do_full_BSE = false;
244  CTP_LOG(ctp::logDEBUG, *_pLog) << " BSE type: TDA" << flush;
245  }
246 
247  _openmp_threads =
248  options->ifExistsReturnElseReturnDefault<int>(key + ".openmp", 0);
249 
250  if (options->exists(key + ".vxc")) {
251  _doVxc =
252  options->ifExistsReturnElseThrowRuntimeError<bool>(key + ".vxc.dovxc");
253  if (_doVxc) {
254  _functional = options->ifExistsReturnElseThrowRuntimeError<std::string>(
255  key + ".vxc.functional");
256  _grid = options->ifExistsReturnElseReturnDefault<std::string>(
257  key + ".vxc.grid", "medium");
258  }
259  }
260 
261  _auxbasis_name =
262  options->ifExistsReturnElseThrowRuntimeError<std::string>(key + ".gwbasis");
263  _dftbasis_name =
264  options->ifExistsReturnElseThrowRuntimeError<std::string>(key + ".dftbasis");
265 
266  _shift = options->ifExistsReturnElseThrowRuntimeError<double>(key + ".shift");
267  _g_sc_limit = options->ifExistsReturnElseReturnDefault<double>(
268  key + ".g_sc_limit",
269  0.00001); // convergence criteria for qp iteration [Hartree]]
270  _g_sc_max_iterations = options->ifExistsReturnElseReturnDefault<int>(
271  key + ".g_sc_max_iterations",
272  40); // convergence criteria for qp iteration [Hartree]]
273 
274  _gw_sc_max_iterations = options->ifExistsReturnElseReturnDefault<int>(
275  key + ".gw_sc_max_iterations",
276  20); // convergence criteria for qp iteration [Hartree]]
277 
278  _gw_sc_limit = options->ifExistsReturnElseReturnDefault<double>(
279  key + ".gw_sc_limit", 0.00001); // convergence criteria for shift it
280  _iterate_gw = false;
281  std::string _shift_type =
282  options->ifExistsReturnElseThrowRuntimeError<std::string>(key + ".shift_type");
283  if (_shift_type != "fixed") {
284  _iterate_gw = true;
285  }
286  CTP_LOG(ctp::logDEBUG, *_pLog) << " Shift: " << _shift_type << flush;
287  CTP_LOG(ctp::logDEBUG, *_pLog) << " g_sc_limit [Hartree]: " << _g_sc_limit
288  << flush;
289  if (_iterate_gw) {
290  CTP_LOG(ctp::logDEBUG, *_pLog) << " gw_sc_limit [Hartree]: " << _gw_sc_limit
291  << flush;
292  }
293  _min_print_weight = options->ifExistsReturnElseReturnDefault<double>(
294  key + ".bse_print_weight",
295  0.2); // print exciton WF composition weight larger that thin minimum
296 
297  // setting some defaults
298  _do_qp_diag = false;
299  _do_bse_singlets = false;
300  _do_bse_triplets = false;
301  // possible tasks
302  // diagQP, singlets, triplets, all, ibse
303  std::string _tasks_string =
304  options->ifExistsReturnElseThrowRuntimeError<std::string>(key + ".tasks");
305  if (_tasks_string.find("all") != std::string::npos) {
306  _do_qp_diag = true;
307  _do_bse_singlets = true;
308  _do_bse_triplets = true;
309  }
310  if (_tasks_string.find("qpdiag") != std::string::npos) _do_qp_diag = true;
311  if (_tasks_string.find("singlets") != std::string::npos)
312  _do_bse_singlets = true;
313  if (_tasks_string.find("triplets") != std::string::npos)
314  _do_bse_triplets = true;
315  _store_eh_interaction = false;
316  _do_bse_diag = true;
317  // special construction for ibse mode
318  if (_tasks_string.find("igwbse") != std::string::npos) {
319  _do_qp_diag = false; // no qp diagonalization
320  _do_bse_diag = false; // no diagonalization of BSE Hamiltonian
321  _store_eh_interaction = true;
322  }
323 
324  // possible storage
325  // qpPert, qpdiag_energies, qp_diag_coefficients, bse_singlet_energies,
326  // bse_triplet_energies, bse_singlet_coefficients, bse_triplet_coefficients
327  _store_qp_pert = true;
328 
329  _store_qp_diag = false;
330  _store_bse_triplets = false;
331  _store_bse_singlets = false;
332  std::string _store_string =
333  options->ifExistsReturnElseThrowRuntimeError<std::string>(key + ".store");
334  if ((_store_string.find("all") != std::string::npos) ||
335  (_store_string.find("") != std::string::npos)) {
336  // store according to tasks choice
337  if (_do_qp_diag) _store_qp_diag = true;
338  if (_do_bse_singlets && _do_bse_diag) _store_bse_singlets = true;
339  if (_do_bse_triplets && _do_bse_diag) _store_bse_triplets = true;
340  }
341  if (_store_string.find("qpdiag") != std::string::npos) _store_qp_diag = true;
342  if (_store_string.find("singlets") != std::string::npos)
343  _store_bse_singlets = true;
344  if (_store_string.find("triplets") != std::string::npos)
345  _store_bse_triplets = true;
346  if (_store_string.find("ehint") != std::string::npos)
347  _store_eh_interaction = true;
348 
349  CTP_LOG(ctp::logDEBUG, *_pLog) << " Tasks: " << flush;
350  if (_do_qp_diag) {
351  CTP_LOG(ctp::logDEBUG, *_pLog) << " qpdiag " << flush;
352  }
353  if (_do_bse_singlets) {
354  CTP_LOG(ctp::logDEBUG, *_pLog) << " singlets " << flush;
355  }
356  if (_do_bse_triplets) {
357  CTP_LOG(ctp::logDEBUG, *_pLog) << " triplets " << flush;
358  }
359  CTP_LOG(ctp::logDEBUG, *_pLog) << " Store: " << flush;
360  if (_store_qp_diag) {
361  CTP_LOG(ctp::logDEBUG, *_pLog) << " qpdiag " << flush;
362  }
363  if (_store_bse_singlets) {
364  CTP_LOG(ctp::logDEBUG, *_pLog) << " singlets " << flush;
365  }
366  if (_store_bse_triplets) {
367  CTP_LOG(ctp::logDEBUG, *_pLog) << " triplets " << flush;
368  }
369  if (_store_eh_interaction) {
370  CTP_LOG(ctp::logDEBUG, *_pLog) << " ehint " << flush;
371  }
372 
373  return;

438  tools::Property *_triplet_summary = &_gwbse_summary->add("triplets", "");
439  for (int state = 0; state < printlimit; ++state) {
440 
441  tools::Property *_level_summary = &_triplet_summary->add("level", "");
442  _level_summary->setAttribute("number", state + 1);
443  _level_summary->add("omega", (format("%1$+1.6f ") %
444  (_orbitals->BSETripletEnergies()(state) * hrt2ev))

446  }
447  }
448  return;
449 }

463 Eigen::MatrixXd GWBSE::CalculateVXC(const AOBasis& dftbasis){
464 
465  Eigen::MatrixXd vxc_ao;
466  if (_orbitals->hasAOVxc()) {
467  if (_doVxc) {
468  CTP_LOG(ctp::logDEBUG, *_pLog)
469  << ctp::TimeStamp()
470  << " There is already a Vxc matrix loaded from DFT, did you maybe "
471  "run a DFT code with outputVxc?\n I will take the external "
472  "implementation"
473  << flush;
474  _doVxc = false;
475  }
476  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
477  << " Loaded external Vxc matrix" << flush;
478  vxc_ao = _orbitals->AOVxc();
479  } else if (_doVxc) {
480 
481  NumericalIntegration _numint;
482  _numint.setXCfunctional(_functional);
483  double ScaHFX_temp = _numint.getExactExchange(_functional);
484  if (ScaHFX_temp != _orbitals->getScaHFX()) {
485  throw std::runtime_error(
486  (boost::format("GWBSE exact exchange a=%s differs from qmpackage "
487  "exact exchange a=%s, probably your functionals are "
488  "inconsistent") %
489  ScaHFX_temp % _orbitals->getScaHFX())
490  .str());
491  }
492  _numint.GridSetup(_grid, _orbitals->QMAtoms(),&dftbasis);
493  CTP_LOG(ctp::logDEBUG, *_pLog)
494  << ctp::TimeStamp()
495  << " Setup grid for integration with gridsize: " << _grid << " with "
496  << _numint.getGridSize() << " points, divided into "
497  << _numint.getBoxesSize() << " boxes" << flush;
498  CTP_LOG(ctp::logDEBUG, *_pLog)
499  << ctp::TimeStamp() << " Integrating Vxc in VOTCA with functional "
500  << _functional << flush;
501  Eigen::MatrixXd DMAT = _orbitals->DensityMatrixGroundState();
502 
503  vxc_ao = _numint.IntegrateVXC(DMAT);
504  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
505  << " Calculated Vxc in VOTCA" << flush;
506 
507  } else {
508  throw std::runtime_error(
509  "So your DFT data contains no Vxc, if you want to proceed use the "
510  "dovxc option.");
511  }
512 
513  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
514  << " Set hybrid exchange factor: " << _orbitals->getScaHFX()
515  << flush;
516 
517  // now get expectation values but only for those in _qpmin:_qpmax range
518  Eigen::MatrixXd _mos = _orbitals->MOCoefficients().block(0,_qpmin,_orbitals->MOCoefficients().rows(),_qptotal);
519 
520  Eigen::MatrixXd vxc =_mos.transpose()*vxc_ao*_mos;
521  CTP_LOG(ctp::logDEBUG, *_pLog)
522  << ctp::TimeStamp()
523  << " Calculated exchange-correlation expectation values " << flush;
524  return vxc;
525 }

527 void GWBSE::PrintGWA_Energies(const Eigen::MatrixXd& vxc, const Sigma& sigma,const Eigen::VectorXd& _dft_energies){
528  const Eigen::VectorXd& gwa_energies=sigma.getGWAEnergies();
529 
530  CTP_LOG(ctp::logINFO, *_pLog)
531  << (format(
532  " ====== Perturbative quasiparticle energies (Hartree) ====== "))
533  .str()
534  << flush;
535  CTP_LOG(ctp::logINFO, *_pLog)
536  << (format(" DeltaHLGap = %1$+1.6f Hartree") % _shift).str() << flush;
537 
538  for (unsigned i = 0; i < _qptotal; i++) {

605 bool GWBSE::Evaluate() {
606 
607 // set the parallelization
608 #ifdef _OPENMP
609  if (_openmp_threads > 0) {
610  omp_set_num_threads(_openmp_threads);
611  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using "
612  << omp_get_max_threads() << " threads"
613  << flush;
614  }
615 #endif
616 
617  if(tools::globals::VOTCA_MKL){
618  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
619  << " Using MKL overload for Eigen "<< flush;
620  }else{
621  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
622  << " Using native Eigen implementation, no BLAS overload "<< flush;
623  }
624 
625 
626  /* check which QC program was used for the DFT run
627  * -> implicit info about MO coefficient storage order
628  */
629  std::string _dft_package = _orbitals->getQMpackage();
630  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
631  << " DFT data was created by " << _dft_package
632  << flush;
633 
634  // store information in _orbitals for later use
635  _orbitals->setRPAindices(_rpamin, _rpamax);
636  _orbitals->setGWAindices(_qpmin, _qpmax);
637  _orbitals->setBSEindices(_bse_vmin, _bse_vmax, _bse_cmin, _bse_cmax,
638  _bse_maxeigenvectors);
639 
640 
641 
642  // load DFT basis set (element-wise information) from xml file
643  BasisSet dftbs;
644 
645  if (_dftbasis_name != _orbitals->getDFTbasis()) {
646  throw std::runtime_error(
647  "Name of the Basisset from .orb file: " + _orbitals->getDFTbasis() +
648  " and from GWBSE optionfile " + _dftbasis_name + " do not agree.");
649  }
650 
651  dftbs.LoadBasisSet(_dftbasis_name);
652  _orbitals->setDFTbasis(_dftbasis_name);
653  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Loaded DFT Basis Set "
654  << _dftbasis_name << flush;
655 
656  // fill DFT AO basis by going through all atoms
657  AOBasis dftbasis;
658  dftbasis.AOBasisFill(&dftbs, _orbitals->QMAtoms(), _fragA);
659  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
660  << " Filled DFT Basis of size "
661  << dftbasis.AOBasisSize() << flush;
662  if (dftbasis.getAOBasisFragB() > 0 && dftbasis.getAOBasisFragA()>0) {
663  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " FragmentA size "
664  << dftbasis.getAOBasisFragA() << flush;
665  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " FragmentB size "
666  << dftbasis.getAOBasisFragB()<< flush;
667  }
668 
669  if (_do_full_BSE)
670  _orbitals->setBSEtype("full");
671  else {
672  _orbitals->setBSEtype("TDA");
673  }
674 
675 
676  // process the DFT data
677  // a) form the expectation value of the XC functional in MOs
678 
679 
680  Eigen::MatrixXd vxc=CalculateVXC(dftbasis);
681 
683 
684  // load auxiliary basis set (element-wise information) from xml file
685  BasisSet auxbs;
686  auxbs.LoadBasisSet(_auxbasis_name);
687  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Loaded Auxbasis Set "
688  << _auxbasis_name << flush;
689 
690  // fill auxiliary AO basis by going through all atoms
691  AOBasis auxbasis;
692  auxbasis.AOBasisFill(&auxbs, _orbitals->QMAtoms());
693  _orbitals->setAuxbasis(_auxbasis_name);
694  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
695  << " Filled Auxbasis of size "
696  << auxbasis.AOBasisSize() << flush;
697 
698  /*
699  * for the representation of 2-point functions with the help of the
700  * auxiliary basis, its AO overlap matrix is required.
701  * cf. M. Rohlfing, PhD thesis, ch. 3
702  */
703  AOOverlap _auxoverlap;
704  // Fill overlap
705  _auxoverlap.Fill(auxbasis);
706 
707  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
708  << " Filled Aux Overlap matrix of dimension: "
709  << _auxoverlap.Matrix().rows() << flush;
710 
711  /*
712  * for the calculation of Coulomb and exchange term in the self
713  * energy and electron-hole interaction, the Coulomb interaction
714  * is represented using the auxiliary basis set.
715  * Here, we need to prepare the Coulomb matrix expressed in
716  * the AOs of the auxbasis
717  */
718 
719  // get Coulomb matrix as AOCoulomb
720  AOCoulomb _auxcoulomb;
721 
722  // Fill Coulomb matrix
723  _auxcoulomb.Fill(auxbasis);
724  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
725  << " Filled Aux Coulomb matrix of dimension: "
726  << _auxcoulomb.Matrix().rows() << flush;
727 
728 
729  Eigen::MatrixXd Coulomb_sqrtInv=_auxcoulomb.Pseudo_InvSqrt_GWBSE(_auxoverlap,1e-8);
730  _auxoverlap.FreeMatrix();
731  _auxcoulomb.FreeMatrix();
732  CTP_LOG(ctp::logDEBUG, *_pLog)
733  << ctp::TimeStamp() << " Calculated Matrix Sqrt of Aux Coulomb Matrix"
734  << flush;
735  CTP_LOG(ctp::logDEBUG, *_pLog)
736  << ctp::TimeStamp() << " Removed " << _auxcoulomb.Removedfunctions()
737  << " functions from Aux Coulomb matrix to avoid near linear dependencies" << flush;
738 
739  // container => M_mn
740  // prepare 3-center integral object
741 
742  TCMatrix_gwbse Mmn;
743  //rpamin here, because RPA needs till rpamin
744  Mmn.Initialize(auxbasis.AOBasisSize(), _rpamin, _qpmax, _rpamin, _rpamax);
745  Mmn.Fill(auxbasis, dftbasis, _orbitals->MOCoefficients());
746  CTP_LOG(ctp::logDEBUG, *_pLog)
747  << ctp::TimeStamp()
748  << " Calculated Mmn_beta (3-center-repulsion x orbitals) " << flush;
749 
750  // make _Mmn symmetric
751  Mmn.MultiplyRightWithAuxMatrix(Coulomb_sqrtInv);
752 
753  CTP_LOG(ctp::logDEBUG, *_pLog)
754  << ctp::TimeStamp()
755  << " Multiplied Mmn_beta with Coulomb Matrix " << flush;
756  PPM ppm;
757  RPA rpa;
758  rpa.configure(_homo,_rpamin,_rpamax);
759  Eigen::VectorXd screen_r=Eigen::VectorXd::Zero(1);
760  screen_r(0)=ppm.getScreening_r();
761  Eigen::VectorXd screen_i=Eigen::VectorXd::Zero(1);
762  screen_i(0)=ppm.getScreening_i();
763  rpa.setScreening(screen_r,screen_i);
764  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
765  << " Prepared RPA " << flush;
766 
767  Sigma sigma=Sigma(_pLog);
768  sigma.configure(_homo,_qpmin,_qpmax,_g_sc_max_iterations,_g_sc_limit);
769  sigma.setDFTdata(_orbitals->getScaHFX(),&vxc,&_orbitals->MOEnergies());
770 
771  // initialize _qp_energies;
772  // shift unoccupied levels by the shift
773  Eigen::VectorXd gwa_energies = Eigen::VectorXd::Zero(_orbitals->getNumberOfLevels());
774  for (int i = 0; i < gwa_energies.size(); ++i) {
775  gwa_energies(i) = _orbitals->MOEnergies()(i);
776  if (i > int(_homo)) {
777  gwa_energies(i) += _shift;
778  }
779  }
780  sigma.setGWAEnergies(gwa_energies);
781 
782  /* for automatic iteration of both G and W, we need to
783  * - make a copy of _Mmn
784  * - calculate eps
785  * - construct ppm
786  * - threecenters for sigma
787  * - sigma_x
788  * - sigma_c
789  * - test for convergence
790  *
791  */
792 
793  if (!_iterate_gw) {
794  _gw_sc_max_iterations = 1;
795  }
796 
797  const Eigen::VectorXd &_dft_energies = _orbitals->MOEnergies();
798  for (unsigned gw_iteration = 0; gw_iteration < _gw_sc_max_iterations;
799  ++gw_iteration) {
800 
801  Eigen::VectorXd _qp_old_rpa = gwa_energies;
802  if (_iterate_gw) {
803  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " GW Iteraton "
804  << gw_iteration + 1 << " of "
805  << _gw_sc_max_iterations << flush;
806  }
807 
808 
809  if(gw_iteration%_reset_3c==0 && gw_iteration!=0){
810  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Rebuilding Mmn_beta (3-center-repulsion x orbitals)" << flush;
811  Mmn.Fill(auxbasis, dftbasis, _orbitals->MOCoefficients());
812  Mmn.MultiplyRightWithAuxMatrix(Coulomb_sqrtInv);
813  }
814 
815  rpa.calculate_epsilon(gwa_energies,Mmn);
816  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
817  << " Calculated epsilon via RPA " << flush;
818  ppm.PPM_construct_parameters(rpa);
819  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
820  << " Constructed PPM parameters " << flush;
821 
822 
823 
824  Mmn.MultiplyRightWithAuxMatrix(ppm.getPpm_phi());
825  CTP_LOG(ctp::logDEBUG, *_pLog)
826  << ctp::TimeStamp() << " Prepared threecenters for sigma " << flush;
827 
828  sigma.CalcdiagElements(Mmn,ppm);
829  CTP_LOG(ctp::logDEBUG, *_pLog)
830  << ctp::TimeStamp() << " Calculated diagonal part of Sigma " << flush;
831  // iterative refinement of qp energies
832  gwa_energies=sigma.getGWAEnergies();
833  double _DFTgap = _dft_energies(_homo + 1) - _dft_energies(_homo);
834  double _QPgap = gwa_energies(_homo + 1) - gwa_energies(_homo);
835  _shift = _QPgap - _DFTgap;
836 
837  // qp energies outside the update range are simply shifted.
838  for (unsigned i = _qpmax + 1; i < _dft_energies.size(); ++i) {
839  gwa_energies(i) = _dft_energies(i) + _shift;
840  }
841 
842  if (_iterate_gw) {
843  bool _gw_converged = true;
844  Eigen::VectorXd diff = _qp_old_rpa - gwa_energies;
845  int state = 0;
846  double E_diff_max=diff.cwiseAbs().maxCoeff(&state);
847  if(E_diff_max>_gw_sc_limit){
848  _gw_converged = false;
849  }
850 
851  double alpha = 0.0;
852  gwa_energies = alpha * _qp_old_rpa + (1 - alpha) * gwa_energies;
853  sigma.setGWAEnergies(gwa_energies);
854  if (tools::globals::verbose) {
855  CTP_LOG(ctp::logDEBUG, *_pLog)
856  << ctp::TimeStamp() << " GW_Iteration: " << gw_iteration + 1
857  << " shift=" << _shift << " E_diff max=" << E_diff_max
858  << " StateNo:" << state << flush;
859  }
860 
861  if (_gw_converged) {
862  CTP_LOG(ctp::logDEBUG, *_pLog)
863  << ctp::TimeStamp() << " Converged after " << gw_iteration + 1
864  << " GW iterations" << flush;
865  break;
866  } else if (gw_iteration == _gw_sc_max_iterations - 1) {
867  // continue regardless for now, but drop WARNING
868  CTP_LOG(ctp::logDEBUG, *_pLog)
869  << ctp::TimeStamp() << " WARNING! GWA spectrum not converged after "
870  << _gw_sc_max_iterations << " iterations." << flush;
871  CTP_LOG(ctp::logDEBUG, *_pLog)
872  << ctp::TimeStamp() << " GWA level " << state
873  << " energy changed by " << E_diff_max << flush;
874  CTP_LOG(ctp::logDEBUG, *_pLog)
875  << ctp::TimeStamp()
876  << " Run continues. Inspect results carefully!" << flush;
877  break;
878  }
879  }else
880  {
881  sigma.setGWAEnergies(gwa_energies);
882  }
883  }
884 
885  ppm.FreeMatrix();
886  rpa.FreeMatrices();
887  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
888  << " Cleaned up PPM and MmnRPA Matrices" << flush;
889 
890 
891  // Output of quasiparticle energies after all is done:
892  PrintGWA_Energies(vxc, sigma, _dft_energies);
893 
894  // store perturbative QP energy data in orbitals object (DFT, S_x,S_c, V_xc,
895  // E_qp)
896  if (_store_qp_pert) {
897  Eigen::MatrixXd &_qp_energies_store = _orbitals->QPpertEnergies();
898  _qp_energies_store=Eigen::MatrixXd::Zero(_qptotal, 5);
899  for (unsigned i = 0; i < _qptotal; i++) {
900  _qp_energies_store(i, 0) = _dft_energies(i + _qpmin);
901  _qp_energies_store(i, 1) = sigma.x(i);
902  _qp_energies_store(i, 2) = sigma.c(i);
903  _qp_energies_store(i, 3) = vxc(i, i);
904  _qp_energies_store(i, 4) = gwa_energies(i + _qpmin);
905  }
906  }
907 
908 
909 
910  if (_do_qp_diag || _do_bse_singlets || _do_bse_triplets) {
911  CTP_LOG(ctp::logDEBUG, *_pLog)
912  << ctp::TimeStamp() << " Calculating offdiagonal part of Sigma " << flush;
913  sigma.CalcOffDiagElements(Mmn,ppm);
914  CTP_LOG(ctp::logDEBUG, *_pLog)
915  << ctp::TimeStamp() << " Calculated offdiagonal part of Sigma " << flush;
916  // free no longer required three-center matrices in _Mmn
917  // max required is _bse_cmax (could be smaller than _qpmax)
918  Mmn.Prune(_bse_vmin, _bse_cmax);
919  Eigen::MatrixXd Hqp=sigma.SetupFullQPHamiltonian();
920 
921  if (_do_qp_diag) {
922  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(Hqp);
923  const Eigen::VectorXd& qp_diag_energies=es.eigenvalues();
924 
925  if(es.info()==Eigen::ComputationInfo::Success){
926  CTP_LOG(ctp::logDEBUG, *_pLog)
927  << ctp::TimeStamp() << " Diagonalized QP Hamiltonian "<< flush;
928  }
929 
930  PrintQP_Energies(gwa_energies, qp_diag_energies);
931 
932  if (_store_qp_diag) {
933  _orbitals->QPdiagCoefficients()=es.eigenvectors();
934  _orbitals->QPdiagEnergies()=es.eigenvalues();
935  }
936  } // _do_qp_diag
937 
938 
939  // proceed only if BSE requested
940  if (_do_bse_singlets || _do_bse_triplets) {
941 
942  BSE bse=BSE(_orbitals,_pLog,_min_print_weight);
943  bse.setBSEindices(_homo,_bse_vmin,_bse_cmax,_bse_maxeigenvectors);
944  bse.setGWData(&Mmn,&ppm,&Hqp);
945  // calculate direct part of eh interaction, needed for singlets and triplets
946 
947 
948 
949 
950 
951  if (_do_bse_triplets && _do_bse_diag) {
952  bse.Solve_triplets();
953  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
954  << " Solved BSE for triplets " << flush;
955  // analyze and report results
956  bse.Analyze_triplets(dftbasis);
957 
958  if (!_store_bse_triplets) {
959  bse.FreeTriplets();
960  }
961 
962  } // do_triplets
963 
964  if (_do_bse_singlets && _do_bse_diag) {
965 
966  if (_do_full_BSE) {
967  bse.Solve_singlets_BTDA();
968  CTP_LOG(ctp::logDEBUG, *_pLog)
969  << ctp::TimeStamp() << " Solved full BSE for singlets " << flush;
970  } else {
971  bse.Solve_singlets();
972  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
973  << " Solved BSE for singlets " << flush;
974  }
975  bse.Analyze_singlets(dftbasis);
976  if (!_store_bse_singlets) {
977  bse.FreeSinglets();
978  }
979  }
980 
981  if (_store_eh_interaction) {
982  if (_do_bse_singlets) {
983  bse.SetupHs();
984  }
985  if (_do_bse_triplets) {
986  bse.SetupHt();
987  }
988  }
989  }
990  }
991  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
992  << " GWBSE calculation finished " << flush;
993  return true;
994 }
