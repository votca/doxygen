44 Index GWBSE::CountCoreLevels() {
45  Index ignored_corelevels = 0;
46  if (!_orbitals.hasECPName()) {
47  ECPBasisSet basis;
48  basis.Load("corelevels");
49  Index coreElectrons = 0;
50  for (const auto& atom : _orbitals.QMAtoms()) {
51  coreElectrons += basis.getElement(atom.getElement()).getNcore();
52  }
53  ignored_corelevels = coreElectrons / 2;
54  }
55  return ignored_corelevels;
56 }

79

81  rpamax = Index(rpamaxfactor * double(num_of_levels)) -
82  1; // total number of levels
83 
84  double qpminfactor = options.get(key + ".qpmin").as<double>();
85  qpmin =
86  num_of_occlevels - Index(qpminfactor * double(num_of_occlevels)) - 1;
87 
88  double qpmaxfactor = options.get(key + ".qpmax").as<double>();
89  qpmax =
90  num_of_occlevels + Index(qpmaxfactor * double(num_of_occlevels)) - 1;

92  double bseminfactor = options.get(key + ".bsemin").as<double>();
93  bse_vmin =
94  num_of_occlevels - Index(bseminfactor * double(num_of_occlevels)) - 1;
95 
96  double bsemaxfactor = options.get(key + ".bsemax").as<double>();
97  bse_cmax =
98  num_of_occlevels + Index(bsemaxfactor * double(num_of_occlevels)) - 1;
99 
100  } else if (ranges == "explicit") {
101  // get explicit numbers

103  qpmin = options.get(key + ".qpmin").as<Index>();
104  qpmax = options.get(key + ".qpmax").as<Index>();

107  } else if (ranges == "default") {
108  rpamax = num_of_levels - 1;
109  qpmin = 0;
110  qpmax = 3 * homo + 1;
111  bse_vmin = 0;
112  bse_cmax = 3 * homo + 1;
113  } else if (ranges == "full") {

128  }

450 void GWBSE::addoutput(tools::Property& summary) {
451 
452  const double hrt2ev = tools::conv::hrt2ev;
453  tools::Property& gwbse_summary = summary.add("GWBSE", "");
454  if (_do_gw) {
455  gwbse_summary.setAttribute("units", "eV");
456  gwbse_summary.setAttribute(
457  "DFTEnergy",
458  (format("%1$+1.6f ") % (_orbitals.getDFTTotalEnergy() * hrt2ev)).str());
459 
460  tools::Property& dft_summary = gwbse_summary.add("dft", "");
461  dft_summary.setAttribute("HOMO", _gwopt.homo);
462  dft_summary.setAttribute("LUMO", _gwopt.homo + 1);
463 
464  for (Index state = 0; state < _gwopt.qpmax + 1 - _gwopt.qpmin; state++) {
465  tools::Property& level_summary = dft_summary.add("level", "");
466  level_summary.setAttribute("number", state + _gwopt.qpmin);
467  level_summary.add(
468  "dft_energy",
469  (format("%1$+1.6f ") %
470  (_orbitals.MOs().eigenvalues()(state + _gwopt.qpmin) * hrt2ev))
471  .str());
472  level_summary.add(
473  "gw_energy",
474  (format("%1$+1.6f ") % (_orbitals.QPpertEnergies()(state) * hrt2ev))
475  .str());
476 
477  level_summary.add("qp_energy",
478  (format("%1$+1.6f ") %
479  (_orbitals.QPdiag().eigenvalues()(state) * hrt2ev))
480  .str());
481  }
482  }
483  if (_do_bse_singlets) {
484  tools::Property& singlet_summary = gwbse_summary.add("singlets", "");
485  for (Index state = 0; state < _bseopt.nmax; ++state) {
486  tools::Property& level_summary = singlet_summary.add("level", "");
487  level_summary.setAttribute("number", state + 1);
488  level_summary.add(
489  "omega", (format("%1$+1.6f ") %
490  (_orbitals.BSESinglets().eigenvalues()(state) * hrt2ev))
491  .str());
492  if (_orbitals.hasTransitionDipoles()) {
493 
494  const Eigen::Vector3d& dipoles = (_orbitals.TransitionDipoles())[state];
495  double f = 2 * dipoles.squaredNorm() *
496  _orbitals.BSESinglets().eigenvalues()(state) / 3.0;
497 
498  level_summary.add("f", (format("%1$+1.6f ") % f).str());
499  tools::Property& dipol_summary = level_summary.add(
500  "Trdipole", (format("%1$+1.4f %2$+1.4f %3$+1.4f") % dipoles.x() %
501  dipoles.y() % dipoles.z())
502  .str());
503  dipol_summary.setAttribute("unit", "e*bohr");
504  dipol_summary.setAttribute("gauge", "length");
505  }
506  }
507  }
508  if (_do_bse_triplets) {
509  tools::Property& triplet_summary = gwbse_summary.add("triplets", "");
510  for (Index state = 0; state < _bseopt.nmax; ++state) {
511  tools::Property& level_summary = triplet_summary.add("level", "");
512  level_summary.setAttribute("number", state + 1);
513  level_summary.add(
514  "omega", (format("%1$+1.6f ") %
515  (_orbitals.BSETriplets().eigenvalues()(state) * hrt2ev))
516  .str());
517  }
518  }
519  return;
520 }

532 Eigen::MatrixXd GWBSE::CalculateVXC(const AOBasis& dftbasis) {
533  if (_orbitals.getXCFunctionalName().empty()) {
534  _orbitals.setXCFunctionalName(_functional);
535  } else {
536  if (!(_functional == _orbitals.getXCFunctionalName())) {
537  throw std::runtime_error("Functionals from DFT " +
538  _orbitals.getXCFunctionalName() + " GWBSE " +
539  _functional + " differ!");
540  }
541  }
542 
543  Vxc_Grid grid;
544  grid.GridSetup(_grid, _orbitals.QMAtoms(), dftbasis);
545  XTP_LOG(Log::info, *_pLog)
546  << TimeStamp() << " Setup grid for integration with gridsize: " << _grid
547  << " with " << grid.getGridSize() << " points, divided into "
548  << grid.getBoxesSize() << " boxes" << flush;
549  Vxc_Potential<Vxc_Grid> vxcpotential(grid);
550  vxcpotential.setXCfunctional(_functional);
551  XTP_LOG(Log::error, *_pLog)
552  << TimeStamp() << " Integrating Vxc in VOTCA with functional "
553  << _functional << flush;
554  Eigen::MatrixXd DMAT = _orbitals.DensityMatrixGroundState();
555  Mat_p_Energy e_vxc_ao = vxcpotential.IntegrateVXC(DMAT);
556  XTP_LOG(Log::info, *_pLog)
557  << TimeStamp() << " Calculated Vxc in VOTCA" << flush;
558  XTP_LOG(Log::error, *_pLog)
559  << TimeStamp() << " Set hybrid exchange factor: " << _orbitals.getScaHFX()
560  << flush;
561  Index qptotal = _gwopt.qpmax - _gwopt.qpmin + 1;
562  Index basissize = Index(_orbitals.MOs().eigenvectors().rows());
563  Eigen::MatrixXd mos =
564  _orbitals.MOs().eigenvectors().block(0, _gwopt.qpmin, basissize, qptotal);
565 
566  Eigen::MatrixXd vxc = mos.transpose() * e_vxc_ao.matrix() * mos;
567  XTP_LOG(Log::error, *_pLog)
568  << TimeStamp() << " Calculated exchange-correlation expectation values "
569  << flush;
570 
571  return vxc;

577  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Using "
578  << OPENMP::getMaxThreads() << " threads" << flush;
579 
580  if (XTP_HAS_MKL_OVERLOAD()) {
581  XTP_LOG(Log::error, *_pLog)
582  << TimeStamp() << " Using MKL overload for Eigen " << flush;
583  } else {
584  XTP_LOG(Log::error, *_pLog)

586  << " Using native Eigen implementation, no BLAS overload " << flush;
587  }
588  Index nogpus = OpenMP_CUDA::UsingGPUs();
589  if (nogpus > 0) {
590  XTP_LOG(Log::error, *_pLog)
591  << TimeStamp() << " Using CUDA support for tensor multiplication with "
592  << nogpus << " GPUs." << flush;

597  for (QMAtom& atom : _orbitals.QMAtoms()) {
598  std::string output = (boost::format("%5d"
599  "%5s"
600  " %1.4f %1.4f %1.4f") %
601  atom.getId() % atom.getElement() %
602  (atom.getPos().x() * tools::conv::bohr2ang) %
603  (atom.getPos().y() * tools::conv::bohr2ang) %
604  (atom.getPos().z() * tools::conv::bohr2ang))
605  .str();
606 
607  XTP_LOG(Log::error, *_pLog) << output << flush;
608  }
609 
610  std::string dft_package = _orbitals.getQMpackage();
611  XTP_LOG(Log::error, *_pLog)
612  << TimeStamp() << " DFT data was created by " << dft_package << flush;
613 
614  BasisSet dftbs;
615  dftbs.Load(_dftbasis_name);
616 
617  XTP_LOG(Log::error, *_pLog)
618  << TimeStamp() << " Loaded DFT Basis Set " << _dftbasis_name << flush;
619 
620  // fill DFT AO basis by going through all atoms
621  AOBasis dftbasis;
622  dftbasis.Fill(dftbs, _orbitals.QMAtoms());
623  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Filled DFT Basis of size "
624  << dftbasis.AOBasisSize() << flush;
625 
626  // load auxiliary basis set (element-wise information) from xml file
627  BasisSet auxbs;
628  auxbs.Load(_auxbasis_name);
629  XTP_LOG(Log::error, *_pLog)
630  << TimeStamp() << " Loaded Auxbasis Set " << _auxbasis_name << flush;
631 
632  // fill auxiliary AO basis by going through all atoms
633  AOBasis auxbasis;
634  auxbasis.Fill(auxbs, _orbitals.QMAtoms());
635  _orbitals.setAuxbasisName(_auxbasis_name);
636  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Filled Auxbasis of size "
637  << auxbasis.AOBasisSize() << flush;
638 
639  if ((_do_bse_singlets || _do_bse_triplets) && _fragments.size() > 0) {
640  for (const auto& frag : _fragments) {
641  XTP_LOG(Log::error, *_pLog) << TimeStamp() << " Fragment " << frag.getId()
642  << " size:" << frag.size() << flush;
643  }
644  }
645 
646  if (!_do_gw && !_orbitals.hasQPdiag()) {
647  throw std::runtime_error(
648  "You want no GW calculation but the orb file has no QPcoefficients for "
649  "BSE");
650  }
651  TCMatrix_gwbse Mmn;
652  // rpamin here, because RPA needs till rpamin
653  Index max_3c = std::max(_bseopt.cmax, _gwopt.qpmax);
654  Mmn.Initialize(auxbasis.AOBasisSize(), _gwopt.rpamin, max_3c, _gwopt.rpamin,
655  _gwopt.rpamax);
656  XTP_LOG(Log::error, *_pLog)
657  << TimeStamp()
658  << " Calculating Mmn_beta (3-center-repulsion x orbitals) " << flush;
659  Mmn.Fill(auxbasis, dftbasis, _orbitals.MOs().eigenvectors());
660  XTP_LOG(Log::info, *_pLog)
661  << TimeStamp() << " Removed " << Mmn.Removedfunctions()
662  << " functions from Aux Coulomb matrix to avoid near linear dependencies"
663  << flush;
664  XTP_LOG(Log::error, *_pLog)
665  << TimeStamp() << " Calculated Mmn_beta (3-center-repulsion x orbitals) "
666  << flush;
667 
668  Eigen::MatrixXd Hqp;
669  if (_do_gw) {
670  Eigen::MatrixXd vxc = CalculateVXC(dftbasis);
671  GW gw = GW(*_pLog, Mmn, vxc, _orbitals.MOs().eigenvalues());
672  gw.configure(_gwopt);
673  gw.CalculateGWPerturbation();
674 
675  if (!_sigma_plot_states.empty()) {
676  gw.PlotSigma(_sigma_plot_filename, _sigma_plot_steps, _sigma_plot_spacing,
677  _sigma_plot_states);
678  }
679 
680  // store perturbative QP energy data in orbitals object (DFT, S_x,S_c, V_xc,
681  // E_qp)
682  _orbitals.QPpertEnergies() = gw.getGWAResults();
683  _orbitals.RPAInputEnergies() = gw.RPAInputEnergies();
684 
685  XTP_LOG(Log::info, *_pLog)
686  << TimeStamp() << " Calculating offdiagonal part of Sigma " << flush;
687  gw.CalculateHQP();
688  XTP_LOG(Log::error, *_pLog)
689  << TimeStamp() << " Calculated offdiagonal part of Sigma " << flush;
690 
691  Hqp = gw.getHQP();
692 
693  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es =
694  gw.DiagonalizeQPHamiltonian();
695  if (es.info() == Eigen::ComputationInfo::Success) {
696  XTP_LOG(Log::error, *_pLog)
697  << TimeStamp() << " Diagonalized QP Hamiltonian " << flush;
698  }
699 
700  _orbitals.QPdiag().eigenvectors() = es.eigenvectors();
701  _orbitals.QPdiag().eigenvalues() = es.eigenvalues();
702  } else {
703  if (_orbitals.getGWAmax() != _gwopt.qpmax ||
704  _orbitals.getGWAmin() != _gwopt.qpmin ||
705  _orbitals.getRPAmax() != _gwopt.rpamax ||
706  _orbitals.getRPAmin() != _gwopt.rpamin) {
707  throw std::runtime_error(
708  "The ranges for GW and RPA do not agree with the ranges from the "
709  ".orb file, rerun your GW calculation");
710  }
711  const Eigen::MatrixXd& qpcoeff = _orbitals.QPdiag().eigenvectors();
712 
713  Hqp = qpcoeff * _orbitals.QPdiag().eigenvalues().asDiagonal() *
714  qpcoeff.transpose();
715  }
716 
717  // proceed only if BSE requested
718  if (_do_bse_singlets || _do_bse_triplets) {
719 
720  BSE bse = BSE(*_pLog, Mmn);
721  bse.configure(_bseopt, _orbitals.RPAInputEnergies(), Hqp);
722 
723  // store the direct contribution to the static BSE results
724  Eigen::VectorXd Hd_static_contrib_triplet;
725  Eigen::VectorXd Hd_static_contrib_singlet;
726 
727  if (_do_bse_triplets) {
728  bse.Solve_triplets(_orbitals);
729  XTP_LOG(Log::error, *_pLog)
730  << TimeStamp() << " Solved BSE for triplets " << flush;
731  bse.Analyze_triplets(_fragments, _orbitals);
732  }
733 
734  if (_do_bse_singlets) {
735  bse.Solve_singlets(_orbitals);
736  XTP_LOG(Log::error, *_pLog)
737  << TimeStamp() << " Solved BSE for singlets " << flush;
738  bse.Analyze_singlets(_fragments, _orbitals);
739  }
740 
741  // do perturbative dynamical screening in BSE
742  if (_do_dynamical_screening_bse) {
743 
744  if (_do_bse_triplets) {
745  bse.Perturbative_DynamicalScreening(QMStateType(QMStateType::Triplet),
746  _orbitals);
747  }
748 
749  if (_do_bse_singlets) {
750  bse.Perturbative_DynamicalScreening(QMStateType(QMStateType::Singlet),
751  _orbitals);
752  }
753  }
754  }
755  XTP_LOG(Log::error, *_pLog)
756  << TimeStamp() << " GWBSE calculation finished " << flush;
757  return true;
758 }
