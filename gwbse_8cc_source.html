39 int GWBSE::CountCoreLevels(){
40  int ignored_corelevels=0;
41  if(!_orbitals.hasECPName()){
42  BasisSet basis;
43  basis.LoadPseudopotentialSet("corelevels");
44  int coreElectrons=0;
45  for(const auto& atom:_orbitals.QMAtoms()){
46  coreElectrons+=basis.getElement(atom->getType()).getNcore();
47  }
48  ignored_corelevels = coreElectrons/2;
49  }
50  return ignored_corelevels;
51 }

53 void GWBSE::Initialize(tools::Property& options) {
54 
55 #if (GWBSE_DOUBLE)
56  CTP_LOG(ctp::logDEBUG, *_pLog) << " Compiled with full double support"
57  << flush;
58 #else
59  CTP_LOG(ctp::logDEBUG, *_pLog)
60  << " Compiled with float/double mixture (standard)" << flush;
61 #endif
62 
63  std::string key = Identify();
64 
65  // getting level ranges
66  int rpamax = 0;
67  int rpamin = 0; //never changes
68  int qpmin = 0;
69  int qpmax = 0;
70  int bse_vmin = 0;
71  int bse_cmax = 0;
72 
73  int homo = _orbitals.getHomo(); // indexed from 0
74  int num_of_levels = _orbitals.getBasisSetSize();
75  int num_of_occlevels = _orbitals.getNumberOfAlphaElectrons();
76 
77  std::vector<std::string> range_choice = {"default", "factor", "explicit", "full"};
78  std::string ranges = options.ifExistsAndinListReturnElseThrowRuntimeError<std::string>(key + ".ranges",
79  range_choice);
80 
81  // now check validity, and get rpa, qp, and bse level ranges accordingly
82 
83  if (ranges == "factor") {
84 
85  double rpamaxfactor = options.get(key + ".rpamax").as<double>();
86  rpamax = int(rpamaxfactor * num_of_levels) - 1; // total number of levels
87 
88  double qpminfactor = options.get(key + ".qpmin").as<double>();
89  qpmin = num_of_occlevels - int(qpminfactor * num_of_occlevels) - 1;
90 
91  double qpmaxfactor = options.get(key + ".qpmax").as<double>();
92  qpmax = num_of_occlevels + int(qpmaxfactor * num_of_occlevels) - 1;
93 
94  double bseminfactor = options.get(key + ".bsemin").as<double>();
95  bse_vmin = num_of_occlevels - int(bseminfactor * num_of_occlevels) - 1;
96 
97  double bsemaxfactor = options.get(key + ".bsemax").as<double>();
98  bse_cmax = num_of_occlevels + int(bsemaxfactor * num_of_occlevels) - 1;
99 
100  } else if (ranges == "explicit") {
101  // get explicit numbers
102  rpamax = options.get(key + ".rpamax").as<int>();
103  qpmin = options.get(key + ".qpmin").as<int>();
104  qpmax = options.get(key + ".qpmax").as<int>();
105  bse_vmin = options.get(key + ".bsemin").as<int>();
106  bse_cmax = options.get(key + ".bsemax").as<int>();
107  } else if (ranges == "default") {
108  rpamax = num_of_levels - 1;
109  qpmin = 0;
110  qpmax = 2 * homo + 1;
111  bse_vmin = 0;
112  bse_cmax = 2 * homo + 1;
113  } else if (ranges == "full") {
114  rpamax = num_of_levels - 1;
115  qpmin = 0;
116  qpmax = num_of_levels - 1;
117  bse_vmin = 0;
118  bse_cmax = num_of_levels - 1;
119  }
120  std::string ignore_corelevels = options.ifExistsReturnElseReturnDefault<std::string>(
121  key + ".ignore_corelevels", "none");
122 
123  if(ignore_corelevels=="RPA" || ignore_corelevels=="GW" || ignore_corelevels=="BSE"){
124  int ignored_corelevels = CountCoreLevels();
125  if(ignore_corelevels=="RPA"){
126  rpamin=ignored_corelevels;
127  }
128  if(ignore_corelevels=="GW" || ignore_corelevels=="RPA"){
129  if(qpmin<ignored_corelevels){
130  qpmin = ignored_corelevels;
131  }
132  }
133  if(ignore_corelevels=="GW" || ignore_corelevels=="RPA" || ignore_corelevels=="BSE"){
134  if(bse_vmin<ignored_corelevels){
135  bse_vmin = ignored_corelevels;
136  }
137  }
138 
139  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Ignoring "
140  << ignored_corelevels << " core levels for "<<ignore_corelevels<< " and beyond."
141  << flush;
142  }
143 
144  //check maximum and minimum sizes
145  if (rpamax > num_of_levels) rpamax = num_of_levels - 1;
146  if (qpmax > num_of_levels) qpmax = num_of_levels - 1;
147  if (bse_cmax > num_of_levels) bse_cmax = num_of_levels - 1;
148  if (bse_vmin < 0) bse_vmin = 0;
149  if (qpmin < 0) qpmin = 0;
150 
151  // some QP - BSE consistency checks are required
152  if (bse_vmin < qpmin) qpmin = bse_vmin;
153  if (bse_cmax > qpmax) qpmax = bse_cmax;
154 
155  _gwopt.homo=homo;
156  _gwopt.qpmin=qpmin;
157  _gwopt.qpmax=qpmax;
158  _gwopt.rpamin=rpamin;
159  _gwopt.rpamax=rpamax;
160 
161  _bseopt.vmin=bse_vmin;
162  _bseopt.cmax=bse_cmax;
163  _bseopt.homo=homo;
164  _bseopt.qpmin=qpmin;
165  _bseopt.rpamin=rpamin;
166  _bseopt.rpamax=rpamax;
167 
168  _orbitals.setRPAindices(rpamin, rpamax);
169  _orbitals.setGWindices(qpmin, qpmax);
170  _orbitals.setBSEindices(bse_vmin, bse_cmax);
171 
172 
173  int bse_vmax = homo;
174  int bse_cmin = homo + 1;
175  int bse_vtotal = bse_vmax - bse_vmin + 1;
176  int bse_ctotal = bse_cmax - bse_cmin + 1;
177  int bse_size = bse_vtotal * bse_ctotal;
178 
179  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set RPA level range ["
180  << rpamin << ":" << rpamax << "]"
181  << flush;
182  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set GW level range ["
183  << qpmin << ":" << qpmax << "]"
184  << flush;
185  CTP_LOG(ctp::logDEBUG, *_pLog)
186  << ctp::TimeStamp() << " Set BSE level range occ[" << bse_vmin << ":"
187  << bse_vmax << "] virt[" << bse_cmin << ":" << bse_cmax
188  << "]" << flush;
189  CTP_LOG(ctp::logDEBUG, *_pLog)
190  << ctp::TimeStamp() << " BSE Hamiltonian has size " << bse_size << "x"
191  << bse_size << flush;
192 
193  _gwopt.reset_3c = options.ifExistsReturnElseReturnDefault<int>(
194  key + ".rebuild_threecenter_freq", _gwopt.reset_3c);
195 
196  _bseopt.nmax =
197  options.ifExistsReturnElseReturnDefault<int>(key + ".exctotal", _bseopt.nmax);
198  if (_bseopt.nmax > bse_size || _bseopt.nmax < 0) _bseopt.nmax = bse_size;
199 
200  _fragA = options.ifExistsReturnElseReturnDefault<int>(key + ".fragment", -1);
201 
202  _bseopt.useTDA =
203  options.ifExistsReturnElseReturnDefault<bool>(key + ".useTDA", _bseopt.useTDA);
204  _orbitals.setTDAApprox(_bseopt.useTDA);
205  if (!_bseopt.useTDA) {
206  CTP_LOG(ctp::logDEBUG, *_pLog) << " BSE type: full" << flush;
207  } else {
208  CTP_LOG(ctp::logDEBUG, *_pLog) << " BSE type: TDA" << flush;
209  }
210 
211  _openmp_threads =
212  options.ifExistsReturnElseReturnDefault<int>(key + ".openmp", 0);
213 
214  if (options.exists(key + ".vxc")) {
215  _doVxc =
216  options.ifExistsReturnElseThrowRuntimeError<bool>(key + ".vxc.dovxc");
217  if (_doVxc) {
218  _functional = options.ifExistsReturnElseThrowRuntimeError<std::string>(
219  key + ".vxc.functional");
220  _grid = options.ifExistsReturnElseReturnDefault<std::string>(
221  key + ".vxc.grid", "medium");
222  }
223  }
224 
225  _auxbasis_name =
226  options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".gwbasis");
227  _dftbasis_name =
228  options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".dftbasis");
229  if (_dftbasis_name != _orbitals.getDFTbasisName()) {
230  throw std::runtime_error(
231  "Name of the Basisset from .orb file: " + _orbitals.getDFTbasisName() +
232  " and from GWBSE optionfile " + _dftbasis_name + " do not agree.");
233  }
234 
235  std::vector<std::string> choices = {"G0W0", "evGW"};
236  std::string mode = options.ifExistsAndinListReturnElseThrowRuntimeError<std::string>(key +".mode", choices);
237  if (mode == "G0W0") {
238  _gwopt.gw_sc_max_iterations = 1;
239  }
240  CTP_LOG(ctp::logDEBUG, *_pLog) << " Running GW as: " << mode << flush;
241  _gwopt.ScaHFX=_orbitals.getScaHFX();
242 
243  _gwopt.shift = options.ifExistsReturnElseReturnDefault<double>(key + ".shift", _gwopt.shift);
244  _gwopt.g_sc_limit = options.ifExistsReturnElseReturnDefault<double>(
245  key + ".g_sc_limit", _gwopt.g_sc_limit); // convergence criteria for qp iteration [Hartree]]
246  _gwopt.g_sc_max_iterations = options.ifExistsReturnElseReturnDefault<int>(
247  key + ".g_sc_max_iterations",
248  _gwopt.g_sc_max_iterations); // convergence criteria for qp iteration [Hartree]]
249 
250  _gwopt.gw_sc_max_iterations = options.ifExistsReturnElseReturnDefault<int>(
251  key + ".gw_sc_max_iterations",
252  _gwopt.gw_sc_max_iterations); // convergence criteria for qp iteration [Hartree]]
253 
254  _gwopt.gw_sc_limit = options.ifExistsReturnElseReturnDefault<double>(
255  key + ".gw_sc_limit", _gwopt.gw_sc_limit); // convergence criteria for shift it
256  CTP_LOG(ctp::logDEBUG, *_pLog) << " g_sc_limit [Hartree]: " << _gwopt.g_sc_limit
257  << flush;
258  if (_gwopt.gw_sc_max_iterations > 1) {
259  CTP_LOG(ctp::logDEBUG, *_pLog) << " gw_sc_limit [Hartree]: " << _gwopt.gw_sc_limit
260  << flush;
261  }
262  _bseopt.min_print_weight = options.ifExistsReturnElseReturnDefault<double>(
263  key + ".bse_print_weight", _bseopt.min_print_weight);
264  // print exciton WF composition weight larger than minimum
265 
266  // setting some defaults
267  _do_qp_diag = false;
268  _do_bse_singlets = false;
269  _do_bse_triplets = false;
270  // possible tasks
271  // diagQP, singlets, triplets, all, ibse
272  std::string _tasks_string =
273  options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".tasks");
274  if (_tasks_string.find("all") != std::string::npos) {
275  _do_qp_diag = true;
276  _do_bse_singlets = true;
277  _do_bse_triplets = true;
278  }
279  if (_tasks_string.find("qpdiag") != std::string::npos) _do_qp_diag = true;
280  if (_tasks_string.find("singlets") != std::string::npos)
281  _do_bse_singlets = true;
282  if (_tasks_string.find("triplets") != std::string::npos)
283  _do_bse_triplets = true;
284  _store_eh_interaction = false;
285  _do_bse_diag = true;
286  // special construction for ibse mode
287  if (_tasks_string.find("iqm") != std::string::npos) {
288  _do_qp_diag = false; // no qp diagonalization
289  _do_bse_diag = false; // no diagonalization of BSE Hamiltonian
290  _store_eh_interaction = true;
291  }
292  _store_qp_pert = true;
293 
294  _store_qp_diag = false;
295  _store_bse_triplets = false;
296  _store_bse_singlets = false;

298  options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".store");
299  if ((_store_string.find("all") != std::string::npos) ||

301  // store according to tasks choice
302  if (_do_qp_diag) _store_qp_diag = true;
303  if (_do_bse_singlets && _do_bse_diag) _store_bse_singlets = true;
304  if (_do_bse_triplets && _do_bse_diag) _store_bse_triplets = true;
305  }
306  if (_store_string.find("qpdiag") != std::string::npos) _store_qp_diag = true;
307  if (_store_string.find("singlets") != std::string::npos)
308  _store_bse_singlets = true;
309  if (_store_string.find("triplets") != std::string::npos)
310  _store_bse_triplets = true;
311  if (_store_string.find("ehint") != std::string::npos)
312  _store_eh_interaction = true;
313 
314  CTP_LOG(ctp::logDEBUG, *_pLog) << " Tasks: " << flush;
315  if (_do_qp_diag) {
316  CTP_LOG(ctp::logDEBUG, *_pLog) << " qpdiag " << flush;
317  }
318  if (_do_bse_singlets) {
319  CTP_LOG(ctp::logDEBUG, *_pLog) << " singlets " << flush;
320  }
321  if (_do_bse_triplets) {
322  CTP_LOG(ctp::logDEBUG, *_pLog) << " triplets " << flush;
323  }
324  CTP_LOG(ctp::logDEBUG, *_pLog) << " Store: " << flush;
325  if (_store_qp_diag) {
326  CTP_LOG(ctp::logDEBUG, *_pLog) << " qpdiag " << flush;
327  }
328  if (_store_bse_singlets) {
329  CTP_LOG(ctp::logDEBUG, *_pLog) << " singlets " << flush;
330  }
331  if (_store_bse_triplets) {
332  CTP_LOG(ctp::logDEBUG, *_pLog) << " triplets " << flush;
333  }
334  if (_store_eh_interaction) {
335  CTP_LOG(ctp::logDEBUG, *_pLog) << " ehint " << flush;
336  }
337 
338  return;
339 }

354

364 
365  if (_do_qp_diag) {
366  level_summary.add(
367  "qp_energy",

370  }
371  }
372 
373  if (_do_bse_singlets) {
374  tools::Property &singlet_summary = gwbse_summary.add("singlets", "");
375  for (int state = 0; state < _bseopt.nmax; ++state) {

390  .str());
391  dipol_summary.setAttribute("unit", "e*bohr");
392  dipol_summary.setAttribute("gauge", "length");
393  }
394  }
395  }
396  if (_do_bse_triplets) {
397  tools::Property &triplet_summary = gwbse_summary.add("triplets", "");
398  for (int state = 0; state < _bseopt.nmax; ++state) {
399  tools::Property &level_summary = triplet_summary.add("level", "");
400  level_summary.setAttribute("number", state + 1);
401  level_summary.add("omega", (format("%1$+1.6f ") %
402  (_orbitals.BSETripletEnergies()(state) * hrt2ev))
403  .str());
404  }
405  }
406  return;
407 }

419 Eigen::MatrixXd GWBSE::CalculateVXC(const AOBasis& dftbasis){
420 
421  Eigen::MatrixXd vxc_ao;
422  if (_orbitals.hasAOVxc()) {
423  if (_doVxc) {
424  if(_orbitals.getQMpackage()=="xtp"){
425  CTP_LOG(ctp::logDEBUG, *_pLog)
426  << ctp::TimeStamp()
427  << " Taking VXC from xtp DFT run."
428  << flush;
429  }else{
430  CTP_LOG(ctp::logDEBUG, *_pLog)
431  << ctp::TimeStamp()
432  << " There is already a Vxc matrix loaded from DFT, did you maybe "
433  "run a DFT code with outputVxc?\n I will take the external "
434  "implementation"
435  << flush;
436  }
437  _doVxc = false;
438  }
439  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
440  << " Loaded external Vxc matrix" << flush;
441  vxc_ao = _orbitals.AOVxc();
442  } else if (_doVxc) {
443 
444  NumericalIntegration numint;
445  numint.setXCfunctional(_functional);
446  double ScaHFX_temp = numint.getExactExchange(_functional);
447  if (ScaHFX_temp != _orbitals.getScaHFX()) {
448  throw std::runtime_error(
449  (boost::format("GWBSE exact exchange a=%s differs from qmpackage "
450  "exact exchange a=%s, probably your functionals are "
451  "inconsistent") %
452  ScaHFX_temp % _orbitals.getScaHFX())
453  .str());
454  }
455  numint.GridSetup(_grid, _orbitals.QMAtoms(),dftbasis);
456  CTP_LOG(ctp::logDEBUG, *_pLog)
457  << ctp::TimeStamp()
458  << " Setup grid for integration with gridsize: " << _grid << " with "
459  << numint.getGridSize() << " points, divided into "
460  << numint.getBoxesSize() << " boxes" << flush;
461  CTP_LOG(ctp::logDEBUG, *_pLog)
462  << ctp::TimeStamp() << " Integrating Vxc in VOTCA with functional "
463  << _functional << flush;
464  Eigen::MatrixXd DMAT = _orbitals.DensityMatrixGroundState();
465 
466  vxc_ao = numint.IntegrateVXC(DMAT);
467  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
468  << " Calculated Vxc in VOTCA" << flush;
469 
470  } else {
471  throw std::runtime_error(
472  "So your DFT data contains no Vxc, if you want to proceed use the "
473  "dovxc option.");
474  }
475 
476  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
477  << " Set hybrid exchange factor: " << _orbitals.getScaHFX()
478  << flush;
479  int qptotal=_gwopt.qpmax-_gwopt.qpmin+1;
480  int basissize=_orbitals.MOCoefficients().rows();
481  Eigen::MatrixXd mos = _orbitals.MOCoefficients().block(0,_gwopt.qpmin,basissize,qptotal);
482 
483  Eigen::MatrixXd vxc =mos.transpose()*vxc_ao*mos;
484  CTP_LOG(ctp::logDEBUG, *_pLog)
485  << ctp::TimeStamp()
486  << " Calculated exchange-correlation expectation values " << flush;
487 
488  return vxc;
489 }

491 bool GWBSE::Evaluate() {
492 
493 // set the parallelization
494 #ifdef _OPENMP
495  if (_openmp_threads > 0) {
496  omp_set_num_threads(_openmp_threads);
497  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using "
498  << omp_get_max_threads() << " threads"
499  << flush;
500  }
501 #endif
502 
503  if(tools::globals::VOTCA_MKL){
504  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
505  << " Using MKL overload for Eigen "<< flush;
506  }else{
507  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp()
508  << " Using native Eigen implementation, no BLAS overload "<< flush;
509  }
510 
511  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Molecule Coordinates [A] " << flush;
512  for (QMAtom* atom:_orbitals.QMAtoms()) {
513  std::string output=(boost::format(" %1$s"
514  " %2$+1.4f %3$+1.4f %4$+1.4f")
515  %atom->getType() %(atom->getPos().getX()*tools::conv::bohr2ang)
516  %(atom->getPos().getY()*tools::conv::bohr2ang)
517  %(atom->getPos().getZ()*tools::conv::bohr2ang) ).str();
518 
519  CTP_LOG(ctp::logDEBUG, *_pLog) <<output<< flush;
