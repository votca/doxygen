197  CTP_LOG(ctp::logDEBUG, *_pLog) << " triplets " << flush;

200  CTP_LOG(ctp::logDEBUG, *_pLog) << " ehint " << flush;

203  return;

206  void GWBSE::addoutput(Property *_summary) {
207 
208  const double hrt2ev = tools::conv::hrt2ev;
209  Property *_gwbse_summary = &_summary->add("GWBSE", "");
210  _gwbse_summary->setAttribute("units", "eV");
211  _gwbse_summary->setAttribute("DeltaHLGap", (format("%1$+1.6f ") % (_shift * hrt2ev)).str());
212 
213  _gwbse_summary->setAttribute("DFTEnergy", (format("%1$+1.6f ") % _orbitals->getQMEnergy()).str());
214  int printlimit = _bse_nprint; //I use this to determine how much is printed, I do not want another option to pipe through
215 
216  Property *_dft_summary = &_gwbse_summary->add("dft", "");
217  _dft_summary->setAttribute("HOMO", _homo);
218  _dft_summary->setAttribute("LUMO", _homo + 1);
219  int begin = _homo - printlimit;
220  int end = _homo + printlimit + 1;
221  if (begin < 0) {
222  begin = 0;
223  end = 2 * _homo + 1;
224  }
225  for (int state = begin; state < end; state++) {
226 
227  Property *_level_summary = &_dft_summary->add("level", "");
228  _level_summary->setAttribute("number", state);
229  _level_summary->add("dft_energy", (format("%1$+1.6f ") % ((_orbitals->MOEnergies())(_qpmin + state) * hrt2ev)).str());
230  _level_summary->add("gw_energy", (format("%1$+1.6f ") % (_qp_energies(_qpmin + state) * hrt2ev)).str());
231 
232  if (_do_qp_diag) {
233  //cout << "_do_qp_diag" <<_do_qp_diag<<endl;
234  _level_summary->add("qp_energy", (format("%1$+1.6f ") % (_qp_diag_energies(_qpmin + state) * hrt2ev)).str());
235  }
236  }
237 
238  if (_do_bse_singlets) {
239  Property *_singlet_summary = &_gwbse_summary->add("singlets", "");
240  for (int state = 0; state < printlimit; ++state) {
241  Property *_level_summary = &_singlet_summary->add("level", "");
242  _level_summary->setAttribute("number", state + 1);
243  _level_summary->add("omega", (format("%1$+1.6f ") % (_bse_singlet_energies(state) * hrt2ev)).str());
244  if (_orbitals->hasTransitionDipoles()) {
245 
246  const tools::vec & dipoles = (_orbitals->TransitionDipoles())[state];
247  double f =2* dipoles*dipoles * _bse_singlet_energies(state) / 3.0;
248 
249  _level_summary->add("f", (format("%1$+1.6f ") % f).str());
250  Property *_dipol_summary = &_level_summary->add("Trdipole", (format("%1$+1.4f %2$+1.4f %3$+1.4f") % dipoles.getX() % dipoles.getY() % dipoles.getZ()).str());
251  _dipol_summary->setAttribute("unit", "e*bohr");
252  _dipol_summary->setAttribute("gauge", "length");
253  }
254  }
255  }
256  if (_do_bse_triplets) {
257  Property *_triplet_summary = &_gwbse_summary->add("triplets", "");
258  for (int state = 0; state < printlimit; ++state) {
259 
260  Property *_level_summary = &_triplet_summary->add("level", "");
261  _level_summary->setAttribute("number", state + 1);
262  _level_summary->add("omega", (format("%1$+1.6f ") % (_bse_triplet_energies(state) * hrt2ev)).str());
263  }
264  }
265  return;
266  }

280  bool GWBSE::Evaluate() {
281 
282  // set the parallelization
283 #ifdef _OPENMP
284  if (_openmp_threads > 0){
285  omp_set_num_threads(_openmp_threads);
286  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Using "<< omp_get_max_threads()<<" threads" << flush;
287  }
288 #endif
289  /* check which QC program was used for the DFT run
290  * -> implicit info about MO coefficient storage order
291  */
292  string _dft_package = _orbitals->getQMpackage();
293  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " DFT data was created by " << _dft_package << flush;
294 
295  std::vector<ctp::QMAtom*> _atoms;
296  for(const auto& atom:_orbitals->QMAtoms()){
297  if(!atom->from_environment){
298  _atoms.push_back(atom);
299  }
300  }
301  // load DFT basis set (element-wise information) from xml file
302  BasisSet dftbs;
303 
304  if (_dftbasis_name != _orbitals->getDFTbasis()) {
305  throw std::runtime_error("Name of the Basisset from .orb file: " + _orbitals->getDFTbasis()
306  + " and from GWBSE optionfile " + _dftbasis_name + " do not agree.");
307  }
308 
309  dftbs.LoadBasisSet(_dftbasis_name);
310  _orbitals->setDFTbasis(_dftbasis_name);
311  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Loaded DFT Basis Set " << _dftbasis_name << flush;
312 
313  // fill DFT AO basis by going through all atoms
314 
315  _dftbasis.AOBasisFill(&dftbs, _atoms, _fragA);
316  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Filled DFT Basis of size " << _dftbasis.AOBasisSize() << flush;
317  if (_dftbasis._AOBasisFragB > 0) {
318  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " FragmentA size " << _dftbasis._AOBasisFragA << flush;
319  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " FragmentB size " << _dftbasis._AOBasisFragB << flush;
320  }
321 
322  /* Preparation of calculation parameters:
323  * - number of electrons -> index of HOMO
324  * - number of levels
325  * - highest level considered in RPA
326  * - lowest and highest level considered in GWA
327  * - lowest and highest level considered in BSE
328  * - number of excitations calculates in BSE
329  */
330 
331  // convert _rpamax if needed
332  _homo = _orbitals->getNumberOfElectrons() - 1; // indexed from 0
333  _rpamin = 0; // lowest index occ min(gwa%mmin, screening%nsum_low) ! always 1
334  if (_ranges == "default" || _ranges=="full") {
335  _rpamax = _orbitals->getNumberOfLevels() - 1; // total number of levels
336  } else if (_ranges == "factor") {
337  _rpamax = _rpamaxfactor * _orbitals->getNumberOfLevels() - 1; // total number of levels
338  }
339  if(_rpamax>_orbitals->getNumberOfLevels() - 1){
340  _rpamax=_orbitals->getNumberOfLevels() - 1;
341  }
342  // convert _qpmin and _qpmax if needed
343  if (_ranges == "default") {
344  _qpmin = 0; // indexed from 0
345  _qpmax = 2 * _homo + 1; // indexed from 0
346  } else if (_ranges == "factor") {
347  if(_orbitals->getNumberOfElectrons() - int( _qpminfactor * _orbitals->getNumberOfElectrons()) - 1<0){
348  _qpmin=0;
349  }
350  else{
351  _qpmin = _orbitals->getNumberOfElectrons() - int( _qpminfactor * _orbitals->getNumberOfElectrons()) - 1;
352  }
353  _qpmax = _orbitals->getNumberOfElectrons() + int( _qpmaxfactor * _orbitals->getNumberOfElectrons()) - 1;
354  } else if (_ranges == "explicit") {
355  _qpmin -= 1;
356  _qpmax -= 1;
357  }else if(_ranges=="full"){
358  _qpmin = 0;
359  _qpmax= _orbitals->getNumberOfLevels() - 1;
360  }
361  if(_qpmax>unsigned(_orbitals->getNumberOfLevels() - 1)){
362  _qpmax=_orbitals->getNumberOfLevels() - 1;
363  }
364 
365  // set BSE band range indices
366  // anything else would be stupid!
367  _bse_vmax = _homo;
368  _bse_cmin = _homo + 1;
369 
370  if (_ranges == "default") {
371  _bse_vmin = 0; // indexed from 0
372  _bse_cmax = 2 * _homo + 1; // indexed from 0
373  } else if (_ranges == "factor") {
374  _bse_vmin = _orbitals->getNumberOfElectrons() - int( _bseminfactor * _orbitals->getNumberOfElectrons()) - 1;
375  if(_orbitals->getNumberOfElectrons() - int( _bseminfactor * _orbitals->getNumberOfElectrons()) - 1<0){
376  _bse_vmin = 0;
377  }
378  _bse_cmax = _orbitals->getNumberOfElectrons() + int( _bsemaxfactor * _orbitals->getNumberOfElectrons()) - 1;
379  } else if (_ranges == "explicit") {
380  _bse_vmin -= 1;
381  _bse_cmax -= 1;
382  } else if(_ranges=="full"){
383  _bse_vmin = 0;
384  _bse_cmax = _orbitals->getNumberOfLevels() - 1;
385  }
386  if(_bse_cmax>unsigned(_orbitals->getNumberOfLevels() - 1)){
387  _bse_cmax=_orbitals->getNumberOfLevels() - 1;
388  }
389 
390  _bse_vtotal = _bse_vmax - _bse_vmin + 1;
391  _bse_ctotal = _bse_cmax - _bse_cmin + 1;
392  _bse_size = _bse_vtotal * _bse_ctotal;
393 
394  // indexing info BSE vector index to occupied/virtual orbital
395  for (unsigned _v = 0; _v < _bse_vtotal; _v++) {
396  for (unsigned _c = 0; _c < _bse_ctotal; _c++) {
397  _index2v.push_back(_bse_vmin + _v);
398  _index2c.push_back(_bse_cmin + _c);
399  }
400  }
401 
402  // some QP - BSE consistency checks are required
403  if (_bse_vmin < _qpmin) _qpmin = _bse_vmin;
404  if (_bse_cmax > _qpmax) _qpmax = _bse_cmax;
405 
406 
407  _qptotal = _qpmax - _qpmin + 1;
408  if (_bse_nmax > int(_bse_size) || _bse_nmax < 0) _bse_nmax = int(_bse_size);
409  if (_bse_nprint > _bse_nmax) _bse_nprint = _bse_nmax;
410 
411  // store information in _orbitals for later use
412  _orbitals->setRPAindices(_rpamin, _rpamax);
413  _orbitals->setGWAindices(_qpmin, _qpmax);
414  _orbitals->setBSEindices(_bse_vmin, _bse_vmax, _bse_cmin, _bse_cmax, _bse_nmax);
415 
416  // information for hybrid DFT
417 
418  _ScaHFX = -1;
419 
420 
421  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set RPA level range [" << _rpamin + 1 << ":" << _rpamax + 1 << "]" << flush;
422  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set QP level range [" << _qpmin + 1 << ":" << _qpmax + 1 << "]" << flush;
423  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set BSE level range occ[" << _bse_vmin + 1 << ":" << _bse_vmax + 1 << "] virt[" << _bse_cmin + 1 << ":" << _bse_cmax + 1 << "]" << flush;
424  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " BSE Hamiltonian has size " <<_bse_size<<"x"<<_bse_size << flush;
425 
426 
427  // process the DFT data
428  // a) form the expectation value of the XC functional in MOs
429  // we do not link them because we have to reorder them afterwards and orbitals file has them in the order of the qmpackage
430  //make copy of orbitals MOs because we reorder them
431  _dft_orbitals = _orbitals->MOCoefficients(); //
432  _ScaHFX = _orbitals->getScaHFX();
433  {// this bracket is there so that _vx_ao falls out of scope, like it more than resize
434  ub::matrix<double> _vxc_ao;
435  if (_orbitals->hasAOVxc()) {
436  if (_doVxc) {
437  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << "There is already a Vxc matrix loaded from DFT, did you maybe run a DFT code with outputVxc?\n I will take the external implementation" << flush;
438  _doVxc = false;
439  }
440  if (_dft_package == "gaussian") {
441  // we have to do some cartesian -> spherical transformation for Gaussian
442  const ub::matrix<double>& vxc_cart = _orbitals->AOVxc();
443  //cout<< vxc_cart.size1()<<"x"<<vxc_cart.size2()<<endl;
444  ub::matrix<double> _carttrafo;
445  _dftbasis.getTransformationCartToSpherical(_dft_package, _carttrafo);
446  //cout<< _carttrafo.size1()<<"x"<<_carttrafo.size2()<<endl;
447 
448  ub::matrix<double> _temp = ub::prod(_carttrafo, vxc_cart);
449  _vxc_ao = ub::prod(_temp, ub::trans(_carttrafo));
450 
451  } else {
452  _vxc_ao = _orbitals->AOVxc();
453  }
454  } else if (_doVxc) {
455 
456  NumericalIntegration _numint;
457  _numint.setXCfunctional(_functional);
458  double ScaHFX_temp = _numint.getExactExchange(_functional);
459  if (ScaHFX_temp != _ScaHFX) {
460  throw std::runtime_error((boost::format("GWBSE exact exchange a=%s differs from qmpackage exact exchange a=%s, probably your functionals are inconsistent") % ScaHFX_temp % _ScaHFX).str());
461  }
462  _numint.GridSetup(_grid, &dftbs, _atoms,&_dftbasis);
463  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Setup grid for integration with gridsize: " << _grid <<" with "<<_numint.getGridSize()<< " points, divided into "<<_numint.getBoxesSize()<<" boxes"<<flush;
464  _dftbasis.ReorderMOs(_dft_orbitals, _dft_package, "xtp");
465 
466  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Converted DFT orbital coefficient order from " << _dft_package << " to XTP" << flush;
467  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Integrating Vxc in VOTCA with functional " << _functional << flush;
468  ub::matrix<double> DMAT = _orbitals->DensityMatrixGroundState(_dft_orbitals);
469 
470  _vxc_ao = _numint.IntegrateVXC(DMAT);
471  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated Vxc in VOTCA" << flush;
472 
473  } else {
474  throw std::runtime_error("So your DFT data contains no Vxc, if you want to proceed use the dovxc option.");
475  }
476 
477  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Set hybrid exchange factor: " << _ScaHFX << flush;
478 
479  // now get expectation values but only for those in _qpmin:_qpmax range
480  ub::matrix<double> _mos = ub::project(_dft_orbitals, ub::range(_qpmin, _qpmax + 1), ub::range(0, _dftbasis.AOBasisSize()));
481 
482  ub::matrix<double> _temp = ub::prod(_vxc_ao, ub::trans(_mos));
483  _vxc = ub::prod(_mos, _temp);
484  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated exchange-correlation expectation values " << flush;
485 
486  // b) reorder MO coefficients depending on the QM package used to obtain the DFT data
487  if (_dft_package != "xtp" && !_doVxc) {
488  _dftbasis.ReorderMOs(_dft_orbitals, _dft_package, "xtp");
489  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Converted DFT orbital coefficient order from " << _dft_package << " to XTP" << flush;
490  }
491  }
492 
494 
495 
496  // load auxiliary GW basis set (element-wise information) from xml file
497  BasisSet gwbs;
498  gwbs.LoadBasisSet(_gwbasis_name);
499  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Loaded GW Basis Set " << _gwbasis_name << flush;
500 
501  // fill auxiliary GW AO basis by going through all atoms
502  AOBasis gwbasis;
503  gwbasis.AOBasisFill(&gwbs, _atoms);
504  _orbitals->setGWbasis(_gwbasis_name);
505  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Filled GW Basis of size " << gwbasis.AOBasisSize() << flush;
506 
507  /*
508  * for the representation of 2-point functions with the help of the
509  * auxiliary GW basis, its AO overlap matrix is required.
510  * cf. M. Rohlfing, PhD thesis, ch. 3
511  */
512  AOOverlap _gwoverlap;
513  // initialize overlap matrix
514  _gwoverlap.Initialize(gwbasis.AOBasisSize());
515  // Fill overlap
516  _gwoverlap.Fill(gwbasis);
517 
518 
519  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Filled GW Overlap matrix of dimension: " << _gwoverlap.Matrix().size1() << flush;
520 
521  // check eigenvalues of overlap matrix, if too small basis might have linear dependencies
522  ub::vector<double> _eigenvalues;
523  ub::matrix<double> _eigenvectors;
524  linalg_eigenvalues(_gwoverlap.Matrix(), _eigenvalues, _eigenvectors);
525  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Smallest eigenvalue of GW Overlap matrix : " << _eigenvalues[0] << flush;
526 
527 
528  /*
529  * for the calculation of Coulomb and exchange term in the self
530  * energy and electron-hole interaction, the Coulomb interaction
531  * is represented using the auxiliary GW basis set.
532  * Here, we need to prepare the Coulomb matrix expressed in
533  * the AOs of the GW basis
534  */
535 
536  // get Coulomb matrix as AOCoulomb
537  AOCoulomb _gwcoulomb;
538  // initialize Coulomb matrix
539  _gwcoulomb.Initialize(gwbasis.AOBasisSize());
540  // Fill Coulomb matrix
541  _gwcoulomb.Fill(gwbasis);
542  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Filled GW Coulomb matrix of dimension: " << _gwcoulomb.Matrix().size1() << flush;
543 
544 
545  // PPM is symmetric, so we need to get the sqrt of the Coulomb matrix
546  ub::matrix<double> _gwoverlap_inverse; // will also be needed in PPM itself
547  ub::matrix<double> _gwoverlap_cholesky_inverse; // will also be needed in PPM itself
548 
549  _gwcoulomb.Symmetrize(_gwoverlap, gwbasis, _gwoverlap_inverse, _gwoverlap_cholesky_inverse);
550  ub::matrix<double> _gwoverlap_cholesky_inverse_trans=ub::trans(_gwoverlap_cholesky_inverse);// for performance reasons
551  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Prepared GW Coulomb matrix for symmetric PPM " << flush;
552 
553  /* calculate 3-center integrals, convoluted with DFT eigenvectors
554  *
555  * M_mn(beta) = \int{ \psi^DFT_m(r) \phi^GW_beta(r) \psi^DFT_n d3r }
556  * = \sum_{alpha,gamma} { c_m,alpha c_n,gamma \int {\phi^DFT_alpha(r) \phi^GW_beta(r) \phi^DFT_gamma(r) d3r} }
557  *
558  * cf. M. Rohlfing, PhD thesis, ch. 3.2
559  *
560  */
561 
562  // --- prepare a vector (gwdacay) of matrices (orbitals, orbitals) as container => M_mn
563  // prepare 3-center integral object
564 
565  TCMatrix _Mmn;
566  _Mmn.Initialize(gwbasis.AOBasisSize(), _rpamin, _qpmax, _rpamin, _rpamax);
567  _Mmn.Fill(gwbasis, _dftbasis, _dft_orbitals);
568  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated Mmn_beta (3-center-overlap x orbitals) " << flush;
569 
570  // for use in RPA, make a copy of _Mmn with dimensions (1:HOMO)(gwabasissize,LUMO:nmax)
571  TCMatrix _Mmn_RPA;
572  _Mmn_RPA.Initialize(gwbasis.AOBasisSize(), _rpamin, _homo, _homo + 1, _rpamax);
573  RPA_prepare_threecenters(_Mmn_RPA, _Mmn, gwbasis, _gwoverlap, _gwoverlap_inverse);
574  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Prepared Mmn_beta for RPA " << flush;
575 
576 
577  // make _Mmn_RPA symmetric for use in RPA
578  _Mmn_RPA.Symmetrize(_gwcoulomb.Matrix());
579  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Symmetrize Mmn_beta for RPA " << flush;
580 
581  // make _Mmn symmetric for use in self-energy calculation
582  _Mmn.Symmetrize(_gwcoulomb.Matrix());
583  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Symmetrize Mmn_beta for self-energy " << flush;
584 
585  // fix the frequencies for PPM
586  _screening_freq = ub::zero_matrix<double>(2, 2); // two frequencies
587  //first one
588  _screening_freq(0, 0) = 0.0; // real part
589  _screening_freq(0, 1) = 0.0; // imaginary part
590  //second one
591  _screening_freq(1, 0) = 0.0; // real part
592  _screening_freq(1, 1) = 0.5; // imaginary part //hartree
593 
594  // one entry to epsilon for each frequency
595  _epsilon.resize(_screening_freq.size1());
596 
597  /* for automatic iteration of _shift, we need to
598  * - make a copy of _Mmn
599  * - calculate eps
600  * - construct ppm
601  * - threecenters for sigma
602  * - sigma_x
603  * - sigma_c
604  * - test for convergence
605  *
606  */
607 
608  //initialize _qp_energies;
609  //shift unoccupied levels by the shift
610  _qp_energies=ub::zero_vector<double>(_orbitals->getNumberOfLevels());
611  for (size_t i=0;i<_qp_energies.size();++i){
612  _qp_energies(i)=_orbitals->MOEnergies()(i);
613  if(i>_homo){
614  _qp_energies(i)+=_shift;
615  }
616  }
617 
618 
619  TCMatrix _Mmn_backup;
620  if (_iterate_qp) {
621 
622  // make copy of _Mmn, memory++
623 
624  _Mmn_backup.Initialize(gwbasis.AOBasisSize(), _rpamin, _qpmax, _rpamin, _rpamax);
625  int _mnsize = _Mmn_backup.get_mtot();
626  for (int _i = 0; _i < _mnsize; _i++) {
627  _Mmn_backup[ _i ] = _Mmn[ _i ];
628  }
629  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Made backup of _Mmn " << flush;
630  }
631  _qp_converged=false;
632  _qp_iteration=0;
633  while (!_qp_converged) {
634  _qp_iteration++;
635  // for symmetric PPM, we can initialize _epsilon with the overlap matrix!
636  for (unsigned _i_freq = 0; _i_freq < _screening_freq.size1(); _i_freq++) {
637  _epsilon[ _i_freq ] = _gwoverlap.Matrix();
638  }
639 
640 
641  // determine epsilon from RPA
642  RPA_calculate_epsilon(_Mmn_RPA);
643  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated epsilon via RPA " << flush;
644 
645 
646 
647  // construct PPM parameters
648  PPM_construct_parameters(_gwoverlap_cholesky_inverse,_gwoverlap_cholesky_inverse_trans);
649  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Constructed PPM parameters " << flush;
650  if (!_iterate_qp){
651  _qp_converged = true;
652  _gwoverlap.Matrix().resize(0, 0);
653  _gwoverlap_cholesky_inverse.resize(0,0);
654  _gwoverlap_cholesky_inverse_trans.resize(0,0);
655  _Mmn_RPA.Cleanup();
656  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Cleaned up Overlap and MmnRPA" << flush;
657  }
658  // prepare threecenters for Sigma
659  sigma_prepare_threecenters(_Mmn);
660  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Prepared threecenters for sigma " << flush;
661 
662  // calculate exchange part of sigma
663  sigma_x_setup(_Mmn);
664  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated exchange part of Sigma " << flush;
665 
666 
667  // calculate correlation part of sigma
668  sigma_c_setup(_Mmn);
669  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Calculated correlation part of Sigma " << flush;
670 
671  // restore _Mmn, if shift has not converged
672  if (_iterate_qp && !_qp_converged) {
673  int _mnsize = _Mmn_backup.get_mtot();
674  for (int _i = 0; _i < _mnsize; _i++) {
675  _Mmn[ _i ] = _Mmn_backup[ _i ];
676  }
677 
678  }
679 
680  }
681 
682  // free unused variable if shift is iterated
683  if (_iterate_qp) {
684  _gwoverlap.Matrix().resize(0, 0);
685  _gwoverlap_cholesky_inverse.resize(0,0);
686  _gwoverlap_cholesky_inverse_trans.resize(0,0);
687  _Mmn_RPA.Cleanup();
688  _Mmn_backup.Cleanup();
689  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Cleaned up Overlap, MmnRPA and Mmn_backup " << flush;
690  }
691  // free no longer required three-center matrices in _Mmn
692  // max required is _bse_cmax (could be smaller than _qpmax)
693  _Mmn.Prune(gwbasis.AOBasisSize(), _bse_vmin, _bse_cmax);
694 
695 
696  const ub::vector<double>& _dft_energies=_orbitals->MOEnergies();
697  // Output of quasiparticle energies after all is done:
698 
699  CTP_LOG(ctp::logINFO, *_pLog) << (format(" ====== Perturbative quasiparticle energies (Hartree) ====== ")).str() << flush;
700  CTP_LOG(ctp::logINFO, *_pLog) << (format(" DeltaHLGap = %1$+1.6f Hartree") % _shift).str() << flush;
701 
702  for (unsigned _i = 0; _i < _qptotal; _i++) {
703  if ((_i + _qpmin) == _homo) {
704  CTP_LOG(ctp::logINFO, *_pLog) << (format(" HOMO = %1$4d DFT = %2$+1.4f VXC = %3$+1.4f S-X = %4$+1.4f S-C = %5$+1.4f GWA = %6$+1.4f")
705  % (_i + _qpmin + 1) % _dft_energies(_i + _qpmin) % _vxc(_i, _i) % _sigma_x(_i, _i) % _sigma_c(_i, _i) % _qp_energies(_i + _qpmin)).str() << flush;
706  } else if ((_i + _qpmin) == _homo + 1) {
707  CTP_LOG(ctp::logINFO, *_pLog) << (format(" LUMO = %1$4d DFT = %2$+1.4f VXC = %3$+1.4f S-X = %4$+1.4f S-C = %5$+1.4f GWA = %6$+1.4f")
708  % (_i + _qpmin + 1) % _dft_energies(_i + _qpmin) % _vxc(_i, _i) % _sigma_x(_i, _i) % _sigma_c(_i, _i) % _qp_energies(_i + _qpmin)).str() << flush;
709 
710  } else {
711  CTP_LOG(ctp::logINFO, *_pLog) << (format(" Level = %1$4d DFT = %2$+1.4f VXC = %3$+1.4f S-X = %4$+1.4f S-C = %5$+1.4f GWA = %6$+1.4f")
712  % (_i + _qpmin + 1) % _dft_energies(_i + _qpmin) % _vxc(_i, _i) % _sigma_x(_i, _i) % _sigma_c(_i, _i) % _qp_energies(_i + _qpmin)).str() << flush;
713  }
714  }
715 
716 
717 
718 
719  // store perturbative QP energy data in orbitals object (DFT, S_x,S_c, V_xc, E_qp)
720  if (_store_qp_pert) {
721  ub::matrix<double>& _qp_energies_store = _orbitals->QPpertEnergies();
722  _qp_energies_store.resize(_qptotal, 5);
723  for (unsigned _i = 0; _i < _qptotal; _i++) {
724  _qp_energies_store(_i, 0) = _dft_energies(_i + _qpmin);
725  _qp_energies_store(_i, 1) = _sigma_x(_i, _i);
726  _qp_energies_store(_i, 2) = _sigma_c(_i, _i);
727  _qp_energies_store(_i, 3) = _vxc(_i, _i);
728  _qp_energies_store(_i, 4) = _qp_energies(_i + _qpmin);
729  }
730  }
731 
732 
733  // constructing full quasiparticle Hamiltonian and diagonalize, if requested
734  if (_do_qp_diag || _do_bse_singlets || _do_bse_triplets) {
735  FullQPHamiltonian();
736  if (_do_qp_diag) {
737  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Full quasiparticle Hamiltonian " << flush;
738  CTP_LOG(ctp::logINFO, *_pLog) << (format(" ====== Diagonalized quasiparticle energies (Hartree) ====== ")).str() << flush;
739  for (unsigned _i = 0; _i < _qptotal; _i++) {
740  if ((_i + _qpmin) == _homo) {
741  CTP_LOG(ctp::logINFO, *_pLog) << (format(" HOMO = %1$4d PQP = %2$+1.4f DQP = %3$+1.4f ") % (_i + _qpmin + 1) % _qp_energies(_i + _qpmin) % _qp_diag_energies(_i)).str() << flush;
742  } else if ((_i + _qpmin) == _homo + 1) {
743  CTP_LOG(ctp::logINFO, *_pLog) << (format(" LUMO = %1$4d PQP = %2$+1.4f DQP = %3$+1.4f ") % (_i + _qpmin + 1) % _qp_energies(_i + _qpmin) % _qp_diag_energies(_i)).str() << flush;
744 
745  } else {
746  CTP_LOG(ctp::logINFO, *_pLog) << (format(" Level = %1$4d PQP = %2$+1.4f DQP = %3$+1.4f ") % (_i + _qpmin + 1) % _qp_energies(_i + _qpmin) % _qp_diag_energies(_i)).str() << flush;
747  }
748  }
749 
750  // free memory
751 
752  if (!_store_qp_diag) {
753  _qp_diag_coefficients.resize(0, 0);
754  _qp_diag_energies.resize(0);
755  }
756  } // _do_qp_diag
757  } // constructing full quasiparticle Hamiltonian
758 
759 
760  // proceed only if BSE requested
761  if (_do_bse_singlets || _do_bse_triplets) {
762 
763  // calculate direct part of eh interaction, needed for singlets and triplets
764  BSE_d_setup(_Mmn);
765  // add qp part to Eh_d
766  BSE_Add_qp2H(_eh_d);
767  // Only if verbose do we use this really
768  if (votca::tools::globals::verbose) {
769  BSE_qp_setup();
770  }
771 
772  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Direct part of e-h interaction " << flush;
773 
774  if (_do_full_BSE) {
775  BSE_d2_setup(_Mmn);
776  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Direct part of e-h interaction RARC " << flush;
777  }
778 
779 
780  if (_do_bse_triplets && _do_bse_diag) {
781  BSE_solve_triplets();
782  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Solved BSE for triplets " << flush;
783  // analyze and report results
784  BSE_analyze_triplets();
785 
786  } // do_triplets
787 
788 
789 
790  // constructing electron-hole interaction for BSE
791  if (_do_bse_singlets) {
792  // calculate exchange part of eh interaction, only needed for singlets
793  BSE_x_setup(_Mmn);
794  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Exchange part of e-h interaction " << flush;
795  }
796 
797  if (_do_bse_singlets && _do_bse_diag) {
798 
799  if (_do_full_BSE) {
800  BSE_solve_singlets_BTDA();
801  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Solved full BSE for singlets " << flush;
802  BSE_analyze_singlets_BTDA();
803  } else {
804  BSE_solve_singlets();
805  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " Solved BSE for singlets " << flush;
806  BSE_analyze_singlets();
807  }
808 
809  if (!_store_eh_interaction) {
810  _eh_d.resize(0, 0);
811  _eh_x.resize(0, 0);
812  }
813 
814  }
815  }
816  CTP_LOG(ctp::logDEBUG, *_pLog) << ctp::TimeStamp() << " GWBSE calculation finished " << flush;
817  return true;
818  }
819  }
