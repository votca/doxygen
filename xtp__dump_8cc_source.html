63 bool XtpDump::EvaluateOptions() {
64 
65  if (OptionsMap().count("list")) {
66  cout << "Available extractors: \n";
67  for(ExtractorFactory::assoc_map::const_iterator iter=
68  Extractors().getObjects().begin();
69  iter != Extractors().getObjects().end(); ++iter) {
70  PrintDescription(std::cout, iter->first, "xtp/xml", Application::HelpShort );
71  }
72  StopExecution();
73  return true;
74  }
75 
76 
77  if (OptionsMap().count("description")) {
78  CheckRequired("description", "no extractor is given");
79  Tokenizer tok(OptionsMap()["description"].as<string>(), " ,\n\t");
80  // loop over the names in the description string
81  for (Tokenizer::iterator n = tok.begin(); n != tok.end(); ++n) {
82  // loop over calculators
83  bool printerror = true;
84  for(ExtractorFactory::assoc_map::const_iterator iter=Extractors().getObjects().begin();
85  iter != Extractors().getObjects().end(); ++iter) {
86 
87  if ( (*n).compare( (iter->first).c_str() ) == 0 ) {
88  PrintDescription(std::cout, iter->first, "xtp/xml", Application::HelpLong );
89  printerror = false;
90  break;
91  }
92  }
93  if ( printerror ) cout << "Extractor " << *n << " does not exist\n";
94  }
95  StopExecution();
96  return true;
97  }
98 
99  SqlApplication::EvaluateOptions();
100  CheckRequired("extract", "Nothing to do here: Abort.");
101 
102  Tokenizer calcs(OptionsMap()["extract"].as<string>(), " ,\n\t");
103  Tokenizer::iterator it;
104  for (it = calcs.begin(); it != calcs.end(); it++) {
105  SqlApplication::AddCalculator(Extractors().Create((*it).c_str()));
106  }
107  return true;
108 }
