39BOOST_AUTO_TEST_CASE(Serializing) {
40 libint2::initialize();
41 Orbitals orbitals;
42
43 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
44 "/ecpaobasis/molecule.xyz");
45
46 QMMolecule mol = orbitals.QMAtoms();
47
48 ECPBasisSet ecp;
49 ecp.Load(std::string(XTP_TEST_DATA_FOLDER) + "/ecpaobasis/ecp.xml");
50 ECPAOBasis aoecp;
51 aoecp.Fill(ecp, orbitals.QMAtoms());
52
53 CheckpointFile ff("aoecp.hdf5");
54 CheckpointWriter ww = ff.getWriter();
55 aoecp.WriteToCpt(ww);
56
57 CheckpointReader rr = ff.getReader();
58 ECPAOBasis aoecp2;
59 aoecp2.ReadFromCpt(rr);
60
61 BasisSet s;
62 s.Load(std::string(XTP_TEST_DATA_FOLDER) + "/ecpaobasis/3-21G.xml");
63 AOBasis b;
64 b.Fill(s, orbitals.QMAtoms());
65
66 // no real way to test if two ecps are equal so we check if the matrices
67 // are the same
68
69 AOECP ecpmat1;
70 ecpmat1.FillPotential(b, aoecp);
71 AOECP ecpmat2;
72 ecpmat2.FillPotential(b, aoecp2);
73 bool check = ecpmat1.Matrix().isApprox(ecpmat2.Matrix(), 1e-8);
74 BOOST_CHECK_EQUAL(check, 1);
75 if (!check) {
76 std::cout << "res" << std::endl;
77 std::cout << aoecp2 << std::endl;
78 std::cout << "ref" << std::endl;
79 std::cout << aoecp << std::endl;
80 }
81
82 aoecp2.AddECPChargeToMolecule(mol);
83
84 for (Index i = 0; i < mol.size(); i++) {
85 BOOST_CHECK_EQUAL(mol[i].getNuccharge(),
86 orbitals.QMAtoms()[i].getNuccharge());
87 }
88
89 libint2::finalize();
90}
