132 void EDFT::WriteJobFile(ctp::Topology *top) {
133 
134  cout << endl << "... ... Writing job file: " << flush;
135  ofstream ofs;
136  ofs.open(_jobfile.c_str(), ofstream::out);
137  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
138 
139  ofs << "<jobs>" << endl;
140 
141  ctp::QMNBList::iterator pit;
142  ctp::QMNBList &nblist = top->NBList();
143 
144 
145  int jobCount = 0;
146  if (nblist.size() == 0) {
147  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
148  return;
149  }
150 
151  // regenerate the list of bridging segments for every pair
152  // (Donor - Bridge1 - Bridge2 - ... - Acceptor) type
153  nblist.GenerateSuperExchange();
154 
155  map< int,ctp::Segment* > segments;
156  map< int,ctp::Segment* >::iterator sit;
157 
158  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
159 
160  int id1 = (*pit)->Seg1()->getId();
161  int id2 = (*pit)->Seg2()->getId();
162  segments[id1] = (*pit)->Seg1();
163  segments[id2] = (*pit)->Seg2();
164 
165  /* loop over bridging segments if any and add them to the map
166  this in principle is not needed since all pairs between
167  donors, acceptors, and bridges are already in the list
168  */
169  vector<ctp::Segment*> bridges = (*pit)->getBridgingSegments();
170  for ( vector<ctp::Segment*>::const_iterator bsit = bridges.begin(); bsit != bridges.end(); bsit++ ) {
171  //cout << "Bridging segment " << (*bsit)->getId() << " : " << (*bsit)->getName() << endl;
172  segments[ (*bsit)->getId() ] = (*bsit);
173  }
174 
175  }
176 
177 
178 
179  for (sit = segments.begin(); sit != segments.end(); ++sit) {
180 
181  int id = ++jobCount;
182  string tag = "";
183 
184  Property Input;
185  Property *pInput = &Input.add("input","");
186  Property *pSegment = &pInput->add("segment" , (format("%1$s") % sit->first).str() );
187  pSegment->setAttribute<string>("type", sit->second->getName() );
188  pSegment->setAttribute<int>("id", sit->second->getId() );
189  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE );
190  job.ToStream(ofs,"xml");
191  }
192 
193 
194  // CLOSE STREAM
195  ofs << "</jobs>" << endl;
196  ofs.close();
197 
198  cout << jobCount << " jobs" << flush;
199 
200 }

203 ctp::Job::JobResult EDFT::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *opThread) {
204 
205  string output;
206 
207  bool _run_status;
208  bool _parse_log_status;
209  bool _parse_orbitals_status;
210 
211 
212 
213  Orbitals _orbitals;
214  ctp::Job::JobResult jres = ctp::Job::JobResult();
215  Property _job_input = job->getInput();
216  list<Property*> lSegments = _job_input.Select( "segment" );
217  vector < ctp::Segment* > segments;
218  int segId = lSegments.front()->getAttribute<int>( "id" );
219  string segType = lSegments.front()->getAttribute<string>( "type" );
220 
221  ctp::Segment *seg = top->getSegment( segId );
222  assert( seg->getName() == segType );
223  segments.push_back( seg );
224  ctp::Logger* pLog = opThread->getLogger();
225  CTP_LOG(ctp::logINFO,*pLog) << ctp::TimeStamp() << " Evaluating site " << seg->getId() << flush;
226 
227  // log, com, and orbital files will be stored in ORB_FILES/package_name/frame_x/mol_ID/
228  // extracted information will be stored in ORB_FILES/molecules/frame_x/molecule_ID.orb
229 
230  string edft_work_dir = "OR_FILES";
231  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
232  string ID = boost::lexical_cast<string>( seg->getId() );
233 
234  // get the corresponding object from the QMPackageFactory
235  QMPackage *_qmpackage = QMPackages().Create( _package );
236 
237  _qmpackage->setLog( pLog );
238  _qmpackage->Initialize( &_package_options );
239 
240 
241 
242 
243  string orbitals_storage_dir = "molecules";
244  string qmpackage_work_dir = edft_work_dir + "/" + _package + "/" + frame_dir + "/mol_" + ID;
245 
246 
247  string ORB_DIR = edft_work_dir + "/" + orbitals_storage_dir + "/" + frame_dir;
248  boost::filesystem::create_directories( qmpackage_work_dir );
249  boost::filesystem::create_directories(ORB_DIR);
250 
251  _qmpackage->setRunDir( qmpackage_work_dir );
252 
253 
254  // if asked, prepare the input files
255  if ( _do_input ) {
256  _qmpackage->WriteInputFile( segments );
257  }
258 
259  // Run the executable
260  if ( _do_run ) {
261  _run_status = _qmpackage->Run( );
262  if ( !_run_status ) {
263  output += "run failed; " ;
264  CTP_LOG(ctp::logERROR,*pLog) << _package << " run failed" << flush;
265  jres.setOutput( output );
266  jres.setStatus(ctp::Job::FAILED);
267  delete _qmpackage;
268  return jres;
269  } else {
270  output += "run completed; " ;
271  }
272  }
273 
274  // Parse log files
275  if ( _do_parse ) {
276  _parse_log_status = _qmpackage->ParseLogFile( &_orbitals );
277  if ( !_parse_log_status ) {
278  output += "log incomplete; ";
279  CTP_LOG(ctp::logERROR,*pLog) << "QM log incomplete" << flush;
280  jres.setOutput( output );
281  jres.setStatus(ctp::Job::FAILED);
282  delete _qmpackage;
283  return jres;
284  } else {
285  output += "log parsed; " ;
286  }
287 
288  // Parse orbitals file
289  _parse_orbitals_status = _qmpackage->ParseOrbitalsFile( &_orbitals );
290  if ( !_parse_orbitals_status ) {
291  output += "orbitals failed; " ;
292  CTP_LOG(ctp::logERROR,*pLog) << "QM orbitals not parsed" << flush;
293  jres.setOutput( output );
294  jres.setStatus(ctp::Job::FAILED);
295  delete _qmpackage;
296  return jres;
297  } else {
298  output += "orbitals parsed; " ;
299  }
300  }
301 
302  // Trim virtual orbitals
303  if ( _do_trim ) {
304 
305  int factor = 2;
306  if ( !_do_parse ) { // orbitals must be loaded from a file
307  boost::filesystem::path arg_path;
308  string ORB_FILE = ( arg_path / ORB_DIR / (format("molecule_%1%.orb") % ID ).str() ).c_str() ;
309  CTP_LOG(ctp::logDEBUG,*pLog) << "Loading orbitals from " << ORB_FILE << flush;
310  try{
311  _orbitals.ReadFromCpt(ORB_FILE);
312  }
313  catch(std::runtime_error& error){
314  CTP_LOG(ctp::logERROR,*pLog) << "Failed loading orbitals from " << ORB_FILE << flush;
315  output += "failed loading " + ORB_FILE;
316  jres.setOutput( output );
317  jres.setStatus(ctp::Job::FAILED);
318  delete _qmpackage;
319  return jres;
320  }
321 
322  }
323 
324  _orbitals.Trim(factor);
325  CTP_LOG(ctp::logDEBUG,*pLog) << "Trimming virtual orbitals from "
326  << _orbitals.getNumberOfLevels() - _orbitals.getNumberOfElectrons() << " to "
327  << _orbitals.getNumberOfElectrons()*factor << flush;
328  output += "orbitals trimmed; " ;
329  }
330 
331 
332  if ( _do_parse ){
333  // save orbitals
334 
335  string ORB_FILE = "molecule_" + ID + ".orb";
336  CTP_LOG(ctp::logDEBUG,*pLog) << "Serializing to " << ORB_FILE << flush;
337  _orbitals.WriteToCpt(ORB_DIR+"/"+ORB_FILE);
338  // ofs.close();
339 
340  if(_qmpackage->getPackageName()=="orca"){
341  CTP_LOG(ctp::logINFO,*pLog) << "Copying monomer .gbw file to orb folder" << flush;
342  string qmpackage_gbw_dir = edft_work_dir + "/" + _package + "/" + frame_dir + "/mol_" + ID+"/system.gbw";
343  string gbwFile = ORB_DIR+"/"+(format("%1%_%2%%3%") % "molecule" % ID % ".gbw").str();
344  boost::filesystem::copy_file(qmpackage_gbw_dir, gbwFile,boost::filesystem::copy_option::overwrite_if_exists);
345  }
346 
347  CTP_LOG(ctp::logDEBUG,*pLog) << "Done serializing " << ORB_FILE << flush;
348  }
349 
350 
351 
352  // Clean run
353  _qmpackage->CleanUp();
354  delete _qmpackage;
355 
356  CTP_LOG(ctp::logINFO,*pLog) << ctp::TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
357 
358  Property _job_summary;
359  Property *_output_summary = &_job_summary.add("output","");
360  Property *_segment_summary = &_output_summary->add("segment","");
361  string segName = seg->getName();
362  segId = seg->getId();
363  _segment_summary->setAttribute("id", segId);
364  _segment_summary->setAttribute("type", segName);
365  _segment_summary->setAttribute("homo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() ));
366  _segment_summary->setAttribute("lumo", _orbitals.getEnergy( _orbitals.getNumberOfElectrons() + 1 ));
367 
368  // output of the JOB
369  jres.setOutput( _job_summary );
370  jres.setStatus(ctp::Job::COMPLETE);
371 
372  // dump the LOG
373  //cout << *pLog;
374 
375  return jres;
376 
377 }
