96  std::map<std::string, QMState> IEXCITON::FillParseMaps(const string& Mapstring) {
97  Tokenizer split_options(Mapstring, ", \t\n");
98  std::map<std::string, QMState> type2level;
99  for (const string& substring : split_options) {
100  std::vector<string> segmentpnumber;
101  Tokenizer tok(substring, ":");
102  tok.ToVector(segmentpnumber);
103  if (segmentpnumber.size() != 2) {
104  throw runtime_error("Parser iqm: Segment and exciton labels:" + substring + "are not separated properly");
105  }
106  QMState state=QMState(segmentpnumber[1]);
107  if(!state.isTransition()){
108  throw std::runtime_error("State to calculate must be a transition state");
109  }
110  string segmentname = segmentpnumber[0];
111  type2level[segmentname] = state;
112  }
113  return type2level;
114  }

125  ctp::Job::JobResult IEXCITON::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *opThread) {
126 
127  // report back to the progress observer
128  ctp::Job::JobResult jres = ctp::Job::JobResult();
129 
130  // get the logger from the thread
131  ctp::Logger* pLog = opThread->getLogger();
132 
133  // get the information about the job executed by the thread
134  int job_ID = job->getId();
135  Property job_input = job->getInput();
136  list<Property*> segment_list = job_input.Select("segment");
137  int ID_A = segment_list.front()->getAttribute<int>("id");
138  string type_A = segment_list.front()->getAttribute<string>("type");
139  string mps_fileA = segment_list.front()->getAttribute<string>("mps_file");
140  int ID_B = segment_list.back()->getAttribute<int>("id");
141  string type_B = segment_list.back()->getAttribute<string>("type");
142  string mps_fileB = segment_list.back()->getAttribute<string>("mps_file");
143 
144  ctp::Segment *seg_A = top->getSegment(ID_A);
145  ctp::Segment *seg_B = top->getSegment(ID_B);
146 
147  CTP_LOG(ctp::logINFO, *pLog) << ctp::TimeStamp() << " Evaluating pair "
148  << job_ID << " [" << ID_A << ":" << ID_B << "]" << flush;
149 
150  vector<ctp::APolarSite*> seg_A_raw = ctp::APS_FROM_MPS(mps_fileA, 0, opThread);
151  vector<ctp::APolarSite*> seg_B_raw = ctp::APS_FROM_MPS(mps_fileB, 0, opThread);
152 
153  ctp::PolarSeg seg_A_polar = *(_mps_mapper.MapPolSitesToSeg(seg_A_raw, seg_A));
154  ctp::PolarSeg seg_B_polar = *(_mps_mapper.MapPolSitesToSeg(seg_B_raw, seg_B));
155 
156  double JAB = EvaluatePair(top, &seg_A_polar, &seg_B_polar, pLog);
157 
158  for (ctp::APolarSite* site : seg_A_raw) {
159  delete site;
160  }
161  seg_A_raw.clear();
162  for (ctp::APolarSite* site : seg_B_raw) {
163  delete site;
164  }
165  seg_B_raw.clear();
166 
167  Property job_summary;
168  Property& job_output = job_summary.add("output", "");
169  Property& pair_summary = job_output.add("pair", "");
170  string nameA = seg_A->getName();
171  string nameB = seg_B->getName();
172  pair_summary.setAttribute("idA", ID_A);
173  pair_summary.setAttribute("idB", ID_B);
174  pair_summary.setAttribute("typeA", nameA);
175  pair_summary.setAttribute("typeB", nameB);
176  Property& coupling_summary = pair_summary.add("Coupling", "");
177  coupling_summary.setAttribute("jABstatic", JAB);
178 
179  jres.setOutput(job_summary);
180  jres.setStatus(ctp::Job::COMPLETE);
181 
182  return jres;
183  }

201  double IEXCITON::EvaluatePair(ctp::Topology *top, ctp::PolarSeg* Seg1, ctp::PolarSeg* Seg2, ctp::Logger* pLog) {
202 
203  ctp::XInteractor actor;
204  actor.ResetEnergy();
205  Seg1->CalcPos();
206  Seg2->CalcPos();
207  vec s = top->PbShortestConnect(Seg1->getPos(), Seg2->getPos()) + Seg1->getPos() - Seg2->getPos();
208 
209  double E = 0.0;
210  for (ctp::APolarSite* site1 : *Seg1) {
211  for (ctp::APolarSite* site2 : *Seg2) {
212  actor.BiasIndu(*site1, *site2, s);
213  site1->Depolarize();
214  site2->Depolarize();
215  E += actor.E_f(*site1, *site2);
216  }
217  }
218 
219  if (_cutoff >= 0) {
220  if (abs(s) > _cutoff) {
221  E = E / _epsilon;
222  }
223  }
224  return E * conv::int2eV;
225  }

240  ofs << "<jobs>" << endl;

247  int id2 = pair->Seg2()->getId();

251  QMState state2=GetElementFromMap(name2);
252 
253  string mps_file1 = (boost::format("MP_FILES/%s_%s.mps") % name1 % state1.ToString()).str();

260  pSegment1.setAttribute<int>("id", id1);
261  pSegment1.setAttribute<string>("mps_file", mps_file1);
262  Property& pSegment2 = pInput.add("segment", boost::lexical_cast<string>(id2));
263  pSegment2.setAttribute<string>("type", name2);
264  pSegment2.setAttribute<int>("id", id2);
265  pSegment2.setAttribute<string>("mps_file", mps_file2);
266 
267  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE);
268  job.ToStream(ofs, "xml");
269  }

272  // CLOSE STREAM
273  ofs << "</jobs>" << endl;
274  ofs.close();

276  cout << endl << "... ... In total " << jobCount << " jobs" << flush;

280  void IEXCITON::ReadJobFile(ctp::Topology *top) {
281 
282  Property xml;
283  vector<Property*> records;
284  // gets the neighborlist from the topology
285  ctp::QMNBList &nblist = top->NBList();
286  int number_of_pairs = nblist.size();
287  int current_pairs = 0;
288  ctp::Logger log;
289  log.setReportLevel(ctp::logINFO);
290 
291  // load the QC results in a vector indexed by the pair ID
292  load_property_from_xml(xml, _jobfile);
293  list<Property*> jobProps = xml.Select("jobs.job");
294  records.resize(number_of_pairs + 1);
295 
296  //to skip pairs which are not in the jobfile
297  for (unsigned i = 0; i < records.size(); i++) {
298  records[i] = NULL;
299  }
300  // loop over all jobs = pair records in the job file
301  for (Property* prop:jobProps) {
302  // if job produced an output, then continue with analysis
303  if (prop->exists("output") && prop->exists("output.pair")) {
304  current_pairs++;
305  Property& poutput = prop->get("output.pair");
306  int idA = poutput.getAttribute<int>("idA");
307  int idB = poutput.getAttribute<int>("idB");
308  ctp::Segment *segA = top->getSegment(idA);
309  ctp::Segment *segB = top->getSegment(idB);
310  ctp::QMPair *qmp = nblist.FindPair(segA, segB);
311 
312  if (qmp == NULL) {
313  CTP_LOG_SAVE(ctp::logINFO, log) << "No pair " << idA << ":" << idB << " found in the neighbor list. Ignoring" << flush;
314  } else {
315  records[qmp->getId()] = &(prop->get("output.pair"));
316  }
317  } else {
318  Property thebadone = prop->get("id");
319  throw runtime_error("\nERROR: Job file incomplete.\n Job with id " + thebadone.as<string>() + " is not finished. Check your job file for FAIL, AVAILABLE, or ASSIGNED. Exiting\n");
320  }
321  } // finished loading from the file
322 
323 
324  // loop over all pairs in the neighbor list
325  CTP_LOG_SAVE(ctp::logINFO, log) << "Neighborlist size " << top->NBList().size() << flush;
326  for (ctp::QMPair *pair: top->NBList()) {
327 
328  if (records[ pair->getId() ] == NULL) continue; //skip pairs which are not in the jobfile
329  double Jeff2 = 0.0;
330  double jAB = 0.0;
331  if (pair->getType() == ctp::QMPair::Excitoncl) {
332  Property* pair_property = records[ pair->getId() ];
333  list<Property*> pCoupling = pair_property->Select("Coupling");
334  for (Property* coup:pCoupling) {
335  jAB = coup->getAttribute<double>("jABstatic");
336  }
337  Jeff2 = jAB*jAB;
338  pair->setJeff2(Jeff2, 2);
339  pair->setIsPathCarrier(true, 2);
340  }
341  }
342  CTP_LOG_SAVE(ctp::logINFO, log) << "Pairs [total:updated] " << number_of_pairs << ":" << current_pairs << flush;
343  cout << log;
344  }
345 
346 
347 
348  }
