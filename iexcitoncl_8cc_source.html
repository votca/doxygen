54 std::map<std::string, QMState> IEXCITON::FillParseMaps(
55  const string& Mapstring) {
56  Tokenizer split_options(Mapstring, ", \t\n");
57  std::map<std::string, QMState> type2level;
58  for (const string& substring : split_options) {
59  std::vector<string> segmentpnumber;
60  Tokenizer tok(substring, ":");
61  tok.ToVector(segmentpnumber);
62  if (segmentpnumber.size() != 2) {
63  throw runtime_error("Parser iqm: Segment and exciton labels:" +
64  substring + "are not separated properly");
65  }
66  QMState state = QMState(segmentpnumber[1]);
67  if (!state.isTransition()) {
68  throw std::runtime_error("State to calculate must be a transition state");
69  }
70  string segmentname = segmentpnumber[0];
71  type2level[segmentname] = state;
72  }
73  return type2level;
74 }

76 Job::JobResult IEXCITON::EvalJob(const Topology& top, Job& job,
77  QMThread& opThread) {
78 
79  // report back to the progress observer
80  Job::JobResult jres = Job::JobResult();
81 
82  // get the logger from the thread
83  Logger& pLog = opThread.getLogger();
84 
85  // get the information about the job executed by the thread
86  int job_ID = job.getId();
87  Property job_input = job.getInput();
88  vector<Property*> segment_list = job_input.Select("segment");
89  int ID_A = segment_list.front()->getAttribute<int>("id");
90  string type_A = segment_list.front()->getAttribute<string>("type");
91  string mps_fileA = segment_list.front()->getAttribute<string>("mps_file");
92  int ID_B = segment_list.back()->getAttribute<int>("id");
93  string type_B = segment_list.back()->getAttribute<string>("type");
94  string mps_fileB = segment_list.back()->getAttribute<string>("mps_file");
95 
96  const Segment& seg_A = top.getSegment(ID_A);
97  if (type_A != seg_A.getType()) {
98  throw runtime_error("SegmentA: type " + seg_A.getType() +
99  " and type in jobfile " + type_A +
100  " do not agree for ID:" + std::to_string(ID_A));
101  }
102  const Segment& seg_B = top.getSegment(ID_B);
103  if (type_B != seg_B.getType()) {
104  throw runtime_error("SegmentB: type " + seg_B.getType() +
105  " and type in jobfile " + type_B +
106  " do not agree for ID:" + std::to_string(ID_B));
107  }
108  const QMNBList& nblist = top.NBList();
109  const QMPair* pair = nblist.FindPair(&seg_A, &seg_B);
110  if (pair == nullptr) {
111  throw std::runtime_error(
112  "pair between segments " + std::to_string(seg_A.getId()) + ":" +
113  std::to_string(seg_B.getId()) + " not found in neighborlist ");
114  }
115 
116  XTP_LOG_SAVE(logINFO, pLog) << TimeStamp() << " Evaluating pair " << job_ID
117  << " [" << ID_A << ":" << ID_B << "]" << flush;
118 
119  StaticMapper map(pLog);
120  map.LoadMappingFile(_mapfile);
121  StaticSegment seg1 = map.map(*(pair->Seg1()), mps_fileA);
122  StaticSegment seg2 = map.map(pair->Seg2PbCopy(), mps_fileB);
123  eeInteractor actor;
124  double JAB = actor.CalcStaticEnergy(seg1, seg2);
125  _cutoff = 0;
126  Property job_summary;
127  Property& job_output = job_summary.add("output", "");
128  Property& pair_summary = job_output.add("pair", "");
129  string nameA = seg_A.getType();
130  string nameB = seg_B.getType();
131  pair_summary.setAttribute("idA", ID_A);
132  pair_summary.setAttribute("idB", ID_B);
133  pair_summary.setAttribute("typeA", nameA);
134  pair_summary.setAttribute("typeB", nameB);
135  Property& coupling_summary = pair_summary.add("Coupling", "");
136  coupling_summary.setAttribute("jABstatic", JAB * tools::conv::hrt2ev);
137 
138  jres.setOutput(job_summary);
139  jres.setStatus(Job::COMPLETE);
140 
141  return jres;
142 }

144 QMState IEXCITON::GetElementFromMap(const std::string& elementname) const {
145  QMState state;
146  try {
147  state = _statemap.at(elementname);
148  } catch (std::out_of_range&) {
149  std::string errormessage =
150  "Map does not have segment of type: " + elementname;
151  errormessage += "\n segments in map are:";
152  for (const auto& s : _statemap) {

156  }

160 void IEXCITON::WriteJobFile(const Topology& top) {
161 
162  cout << endl << "... ... Writing job file " << _jobfile << flush;
163  std::ofstream ofs;
164  ofs.open(_jobfile, std::ofstream::out);
165  if (!ofs.is_open()) {
166  throw runtime_error("\nERROR: bad file handle: " + _jobfile);
167  }
168  const QMNBList& nblist = top.NBList();
169  int jobCount = 0;
170  if (nblist.size() == 0) {
171  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
172  return;
173  }
174 
175  ofs << "<jobs>" << endl;
176  string tag = "";
177 
178  for (const QMPair* pair : nblist) {
179  if (pair->getType() == QMPair::PairType::Excitoncl) {
180  int id1 = pair->Seg1()->getId();
181  string name1 = pair->Seg1()->getType();
182  int id2 = pair->Seg2()->getId();
183  string name2 = pair->Seg2()->getType();
184  int id = jobCount;
185  QMState state1 = GetElementFromMap(name1);
186  QMState state2 = GetElementFromMap(name2);
187 
188  string mps_file1 =
189  (boost::format("MP_FILES/%s_%s.mps") % name1 % state1.ToString())
190  .str();
191  string mps_file2 =
192  (boost::format("MP_FILES/%s_%s.mps") % name1 % state2.ToString())
193  .str();
194 
195  Property Input;
196  Property& pInput = Input.add("input", "");
197  Property& pSegment1 =
198  pInput.add("segment", boost::lexical_cast<string>(id1));
199  pSegment1.setAttribute<string>("type", name1);
200  pSegment1.setAttribute<int>("id", id1);
201  pSegment1.setAttribute<string>("mps_file", mps_file1);
202  Property& pSegment2 =
203  pInput.add("segment", boost::lexical_cast<string>(id2));
204  pSegment2.setAttribute<string>("type", name2);
205  pSegment2.setAttribute<int>("id", id2);
206  pSegment2.setAttribute<string>("mps_file", mps_file2);
207 
208  Job job(id, tag, Input, Job::AVAILABLE);
209  job.ToStream(ofs);
210  jobCount++;
211  }
212  }
213 
214  // CLOSE STREAM
215  ofs << "</jobs>" << endl;
216  ofs.close();
217 
218  cout << endl << "... ... In total " << jobCount << " jobs" << flush;
219 }

221 void IEXCITON::ReadJobFile(Topology& top) {
222 
223  Property xml;
224  vector<Property*> records;
225  // gets the neighborlist from the topology
226  QMNBList& nblist = top.NBList();
227  int number_of_pairs = nblist.size();
228  int current_pairs = 0;
229  Logger log;
230  log.setReportLevel(logINFO);
231 
232  // load the QC results in a vector indexed by the pair ID
233  xml.LoadFromXML(_jobfile);
234  vector<Property*> jobProps = xml.Select("jobs.job");
235  records.resize(number_of_pairs + 1);
236 
237  // to skip pairs which are not in the jobfile
238  for (auto& record : records) {
239  record = nullptr;
240  }
241  // loop over all jobs = pair records in the job file
242  for (Property* prop : jobProps) {
243  // if job produced an output, then continue with analysis
244  if (prop->exists("output") && prop->exists("output.pair")) {
245  current_pairs++;
246  Property& poutput = prop->get("output.pair");
247  int idA = poutput.getAttribute<int>("idA");
248  int idB = poutput.getAttribute<int>("idB");
249  Segment& segA = top.getSegment(idA);
250  Segment& segB = top.getSegment(idB);
251  QMPair* qmp = nblist.FindPair(&segA, &segB);
252 
253  if (qmp == nullptr) {
254  XTP_LOG_SAVE(logINFO, log)
255  << "No pair " << idA << ":" << idB
256  << " found in the neighbor list. Ignoring" << flush;
257  } else {
258  records[qmp->getId()] = &(prop->get("output.pair"));
259  }
260  } else {
261  Property thebadone = prop->get("id");
262  throw runtime_error("\nERROR: Job file incomplete.\n Job with id " +
263  thebadone.as<string>() +
264  " is not finished. Check your job file for FAIL, "
265  "AVAILABLE, or ASSIGNED. Exiting\n");
266  }
267  } // finished loading from the file
268 
269  // loop over all pairs in the neighbor list
270  XTP_LOG_SAVE(logINFO, log)
271  << "Neighborlist size " << top.NBList().size() << flush;
272  for (QMPair* pair : top.NBList()) {
273 
274  if (records[pair->getId()] == nullptr) {
275  continue; // skip pairs which are not in the jobfile
276  }
277  double Jeff2 = 0.0;
278  double jAB = 0.0;
279  if (pair->getType() == QMPair::Excitoncl) {
280  Property* pair_property = records[pair->getId()];
281  vector<Property*> pCoupling = pair_property->Select("Coupling");
282  for (Property* coup : pCoupling) {
283  jAB = coup->getAttribute<double>("jABstatic");
284  }
285  Jeff2 = jAB * jAB * tools::conv::ev2hrt * tools::conv::ev2hrt;
286  pair->setJeff2(Jeff2, QMStateType::Singlet);
287  }
288  }
289  XTP_LOG_SAVE(logINFO, log) << "Pairs [total:updated] " << number_of_pairs
290  << ":" << current_pairs << flush;
291  cout << log;
292 }
