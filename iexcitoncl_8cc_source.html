138 
139  // report back to the progress observer
140  ctp::Job::JobResult jres = ctp::Job::JobResult();

142  // get the logger from the thread
143  ctp::Logger* pLog = opThread->getLogger();
144 
145  // get the information about the job executed by the thread
146  int _job_ID = job->getId();
147  Property _job_input = job->getInput();
148  list<Property*> segment_list = _job_input.Select( "segment" );
149  int ID_A = segment_list.front()->getAttribute<int>( "id" );
150  string type_A = segment_list.front()->getAttribute<string>( "type" );
151  string mps_fileA = segment_list.front()->getAttribute<string>( "mps_file" );
152  int ID_B = segment_list.back()->getAttribute<int>( "id" );
153  string type_B = segment_list.back()->getAttribute<string>( "type" );
154  string mps_fileB = segment_list.back()->getAttribute<string>( "mps_file" );
155 
156 
157 
158  ctp::Segment *seg_A = top->getSegment( ID_A );
159  assert( seg_A->getName() == type_A );
160 
161  ctp::Segment *seg_B = top->getSegment( ID_B );
162  assert( seg_B->getName() == type_B );
163 
164  CTP_LOG(ctp::logINFO,*pLog) << ctp::TimeStamp() << " Evaluating pair "
165  << _job_ID << " [" << ID_A << ":" << ID_B << "]" << flush;
166 
167  vector<ctp::APolarSite*> seg_A_raw=ctp::APS_FROM_MPS(mps_fileA,0,opThread);
168  vector<ctp::APolarSite*> seg_B_raw=ctp::APS_FROM_MPS(mps_fileB,0,opThread);
169 
170  ctp::PolarSeg* seg_A_polar=_mps_mapper.MapPolSitesToSeg(seg_A_raw,seg_A);
171  ctp::PolarSeg* seg_B_polar=_mps_mapper.MapPolSitesToSeg(seg_B_raw,seg_B);
172 
173 
174 
175  double JAB=EvaluatePair(top,seg_A_polar,seg_B_polar, pLog);
176 
177  std::vector< ctp::APolarSite* >::iterator it;
178 
179  for (it = seg_A_raw.begin() ; it !=seg_A_raw.end(); ++it){
180  delete *it;
181  }
182  seg_A_raw.clear();
183 
184  for (it = seg_B_raw.begin() ; it !=seg_B_raw.end(); ++it){
185  delete *it;
186  }
187  seg_B_raw.clear();
188 
189 
190  delete seg_A_polar;
191  delete seg_B_polar;
192 
193 
194 
195 
196  Property _job_summary;
197 
198 
199  Property *_job_output = &_job_summary.add("output","");
200  Property *_pair_summary = &_job_output->add("pair","");
201  string nameA = seg_A->getName();
202  string nameB = seg_B->getName();
203  _pair_summary->setAttribute("idA", ID_A);
204  _pair_summary->setAttribute("idB", ID_B);
205  _pair_summary->setAttribute("typeA", nameA);
206  _pair_summary->setAttribute("typeB", nameB);
207  Property *_coupling_summary = &_pair_summary->add("Coupling","");
208  _coupling_summary->setAttribute("jABstatic", JAB);
209 
210 
211 
212 
213 
214 
215  jres.setOutput( _job_summary );
216  jres.setStatus(ctp::Job::COMPLETE);
217 
218  return jres;

221 double IEXCITON::EvaluatePair(ctp::Topology *top,ctp::PolarSeg* Seg1,ctp::PolarSeg* Seg2, ctp::Logger* pLog ){
222 
223  ctp::XInteractor actor;
224  actor.ResetEnergy();
225  Seg1->CalcPos();
226  Seg2->CalcPos();
227  vec s=top->PbShortestConnect(Seg1->getPos(),Seg2->getPos())+Seg1->getPos()-Seg2->getPos();
228  //CTP_LOG(logINFO, *pLog) << "Evaluate pair for debugging " << Seg1->getId() << ":" <<Seg2->getId() << " Distance "<< abs(s) << flush;
229  ctp::PolarSeg::iterator pit1;
230  ctp::PolarSeg::iterator pit2;
231  double E=0.0;
232  for (pit1=Seg1->begin();pit1<Seg1->end();++pit1){
233  for (pit2=Seg2->begin();pit2<Seg2->end();++pit2){
234  actor.BiasIndu(*(*pit1), *(*pit2),s);
235  (*pit1)->Depolarize();
236  (*pit2)->Depolarize();
237 
238 
239  E += actor.E_f(*(*pit1), *(*pit2));
240 
241  }
242  }
243 
244  if(_cutoff>=0){
245  if(abs(s)>_cutoff){
246  E=E/_epsilon;
247  }
248  }
249 
250 
251  return E*conv::int2eV;
252 }

255 void IEXCITON::WriteJobFile(ctp::Topology *top) {
256 
257  cout << endl << "... ... Writing job file " << flush;
258  std::ofstream ofs;
259  ofs.open(_jobfile.c_str(), std::ofstream::out);
260  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
261 
262 
263  ctp::QMNBList::iterator pit;
264  ctp::QMNBList &nblist = top->NBList();
265 
266  int jobCount = 0;
267  if (nblist.size() == 0) {
268  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
269  return;
270  }
271 
272  ofs << "<jobs>" << endl;
273  string tag = "";
274 
275  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
276  if ((*pit)->getType()==3){
277  int id1 = (*pit)->Seg1()->getId();
278  string name1 = (*pit)->Seg1()->getName();
279  int id2 = (*pit)->Seg2()->getId();
280  string name2 = (*pit)->Seg2()->getName();
281 
282  int id = ++jobCount;
283 
284 
285  string mps_file1=(boost::format("MP_FILES/%s_n2s%d.mps") % name1 % _statenumber).str();
286  string mps_file2=(boost::format("MP_FILES/%s_n2s%d.mps") % name1 % _statenumber).str();
287 
288  Property Input;
289  Property *pInput = &Input.add("input","");
290  Property *pSegment = &pInput->add("segment" , boost::lexical_cast<string>(id1) );
291  pSegment->setAttribute<string>("type", name1 );
292  pSegment->setAttribute<int>("id", id1 );
293  pSegment->setAttribute<string>("mps_file",mps_file1);
294  pSegment = &pInput->add("segment" , boost::lexical_cast<string>(id2) );
295  pSegment->setAttribute<string>("type", name2 );
296  pSegment->setAttribute<int>("id", id2 );
297  pSegment->setAttribute<string>("mps_file",mps_file2);
298 
299  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE );
300  job.ToStream(ofs,"xml");
301  }
302  }
303 
304  // CLOSE STREAM
305  ofs << "</jobs>" << endl;
306  ofs.close();
307 
308  cout << endl << "... ... In total " << jobCount << " jobs" << flush;
309

312 void IEXCITON::ReadJobFile(ctp::Topology *top) {
313 
314  Property xml;
315 
316  vector<Property*> records;
317 
318  // gets the neighborlist from the topology
319  ctp::QMNBList &nblist = top->NBList();
320  int _number_of_pairs = nblist.size();
321  int _current_pairs=0;
322 
323  // output using logger
324  ctp::Logger _log;
325  _log.setReportLevel(ctp::logINFO);
326 
327 
328  // load the QC results in a vector indexed by the pair ID
329  load_property_from_xml(xml, _jobfile);
330  list<Property*> jobProps = xml.Select("jobs.job");
331  records.resize( _number_of_pairs + 1 );
332 
333  //to skip pairs which are not in the jobfile
334  for (unsigned i=0;i<records.size();i++){
335  records[i]=NULL;
336  }
337  // loop over all jobs = pair records in the job file
338  for (list<Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
339  // if job produced an output, then continue with analysis
340  if ( (*it)->exists("output") && (*it)->exists("output.pair") ) {
341  _current_pairs++;
342  // get the output records
343  Property& poutput = (*it)->get("output.pair");
344  // id's of two segments of a pair
345  int idA = poutput.getAttribute<int>("idA");
346  int idB = poutput.getAttribute<int>("idB");
347  // segments which correspond to these ids
348  ctp::Segment *segA = top->getSegment(idA);
349  ctp::Segment *segB = top->getSegment(idB);
350  // pair that corresponds to the two segments
351  ctp::QMPair *qmp = nblist.FindPair(segA,segB);
352 
353  if (qmp == NULL) { // there is no pair in the neighbor list with this name
354  CTP_LOG_SAVE(ctp::logINFO, _log) << "No pair " << idA << ":" << idB << " found in the neighbor list. Ignoring" << flush;
355  } else {
356  //CTP_LOG(logINFO, _log) << "Store in record: " << idA << ":" << idB << flush;
357  records[qmp->getId()] = & ((*it)->get("output.pair"));
358  }
359  } else {
360  Property thebadone = (*it)->get("id");
361  throw runtime_error("\nERROR: Job file incomplete.\n Job with id "+thebadone.as<string>()+" is not finished. Check your job file for FAIL, AVAILABLE, or ASSIGNED. Exiting\n");
362  }
363  } // finished loading from the file
364 
365 
366  // loop over all pairs in the neighbor list
367  CTP_LOG_SAVE(ctp::logINFO, _log) << "Neighborlist size " << top->NBList().size() << flush;
368  for (ctp::QMNBList::iterator ipair = top->NBList().begin(); ipair != top->NBList().end(); ++ipair) {
369 
370  ctp::QMPair *pair = *ipair;
371 
372  if (records[ pair->getId() ]==NULL) continue; //skip pairs which are not in the jobfile
373 
374  //Segment* segmentA = pair->Seg1();
375  //Segment* segmentB = pair->Seg2();
376 
377  double Jeff2 = 0.0;
378  double jAB=0.0;
379 
380  //cout << "\nProcessing pair " << segmentA->getId() << ":" << segmentB->getId() << flush;
381 
382 
383  if ( pair->getType() == ctp::QMPair::Excitoncl){
384  Property* pair_property = records[ pair->getId() ];
385 
386  list<Property*> pCoupling = pair_property->Select("Coupling");
387 
388  for (list<Property*> ::iterator itCoupling = pCoupling.begin(); itCoupling != pCoupling.end(); ++itCoupling) {
389  jAB = (*itCoupling)->getAttribute<double>("jABstatic");
390  }
391  Jeff2 = jAB*jAB;
392 
393 
394 
395  pair->setJeff2(Jeff2, 2);
396  pair->setIsPathCarrier(true, 2);
397 
398 
399 
400  }
401  }
402 
403  CTP_LOG_SAVE(ctp::logINFO, _log) << "Pairs [total:updated] " << _number_of_pairs << ":" << _current_pairs << flush;
404  cout << _log;
405 }
