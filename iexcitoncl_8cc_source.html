120 ctp::Job::JobResult IEXCITON::EvalJob(ctp::Topology *top, ctp::Job *job, ctp::QMThread *opThread) {
121 
122  // report back to the progress observer
123  ctp::Job::JobResult jres = ctp::Job::JobResult();
124 
125  // get the logger from the thread
126  ctp::Logger* pLog = opThread->getLogger();
127 
128  // get the information about the job executed by the thread
129  int _job_ID = job->getId();
130  Property _job_input = job->getInput();
131  list<Property*> segment_list = _job_input.Select( "segment" );
132  int ID_A = segment_list.front()->getAttribute<int>( "id" );
133  string type_A = segment_list.front()->getAttribute<string>( "type" );
134  string mps_fileA = segment_list.front()->getAttribute<string>( "mps_file" );
135  int ID_B = segment_list.back()->getAttribute<int>( "id" );
136  string type_B = segment_list.back()->getAttribute<string>( "type" );
137  string mps_fileB = segment_list.back()->getAttribute<string>( "mps_file" );
138 
139 
140 
141  ctp::Segment *seg_A = top->getSegment( ID_A );
142  assert( seg_A->getName() == type_A );
143 
144  ctp::Segment *seg_B = top->getSegment( ID_B );
145  assert( seg_B->getName() == type_B );
146 
147  LOG(ctp::logINFO,*pLog) << ctp::TimeStamp() << " Evaluating pair "
148  << _job_ID << " [" << ID_A << ":" << ID_B << "]" << flush;
149 
150  vector<ctp::APolarSite*> seg_A_raw=_mps_mapper.GetOrCreateRawSites(mps_fileA);
151  vector<ctp::APolarSite*> seg_B_raw=_mps_mapper.GetOrCreateRawSites(mps_fileB);
152 
153  ctp::PolarSeg* seg_A_polar=_mps_mapper.MapPolSitesToSeg(seg_A_raw,seg_A);
154  ctp::PolarSeg* seg_B_polar=_mps_mapper.MapPolSitesToSeg(seg_B_raw,seg_B);
155 
156  ctp::PolarTop cleanup;
157  vector<ctp::PolarSeg*> cleanup2;
158  cleanup2.push_back(seg_A_polar);
159  cleanup2.push_back(seg_B_polar);
160  cleanup.setMM1(cleanup2);
161 
162 
163  double JAB=EvaluatePair(top,seg_A_polar,seg_B_polar, pLog);
164 
165 
166 
167 
168 
169 
170  Property _job_summary;
171 
172 
173  Property *_job_output = &_job_summary.add("output","");
174  Property *_pair_summary = &_job_output->add("pair","");
175  string nameA = seg_A->getName();
176  string nameB = seg_B->getName();
177  _pair_summary->setAttribute("idA", ID_A);
178  _pair_summary->setAttribute("idB", ID_B);
179  _pair_summary->setAttribute("typeA", nameA);
180  _pair_summary->setAttribute("typeB", nameB);
181  Property *_coupling_summary = &_pair_summary->add("Coupling","");
182  _coupling_summary->setAttribute("jABstatic", JAB);
183 
184 
185 
186 
187 
188 
189  jres.setOutput( _job_summary );
190  jres.setStatus(ctp::Job::COMPLETE);
191 
192  return jres;
193 }

195 double IEXCITON::EvaluatePair(ctp::Topology *top,ctp::PolarSeg* Seg1,ctp::PolarSeg* Seg2, ctp::Logger* pLog ){
196 
197  ctp::XInteractor actor;
198  actor.ResetEnergy();
199  Seg1->CalcPos();
200  Seg2->CalcPos();
201  vec s=top->PbShortestConnect(Seg1->getPos(),Seg2->getPos())+Seg1->getPos()-Seg2->getPos();
202  //LOG(logINFO, *pLog) << "Evaluate pair for debugging " << Seg1->getId() << ":" <<Seg2->getId() << " Distance "<< abs(s) << flush;
203  ctp::PolarSeg::iterator pit1;
204  ctp::PolarSeg::iterator pit2;
205  double E=0.0;
206  for (pit1=Seg1->begin();pit1<Seg1->end();++pit1){
207  for (pit2=Seg2->begin();pit2<Seg2->end();++pit2){
208  actor.BiasIndu(*(*pit1), *(*pit2),s);
209  (*pit1)->Depolarize();
210  (*pit2)->Depolarize();
211 
212 
213  E += actor.E_f(*(*pit1), *(*pit2));
214 
215  }
216  }
217 
218 
219  return E*conv::int2eV;
220 }

223 void IEXCITON::WriteJobFile(ctp::Topology *top) {
224 
225  cout << endl << "... ... Writing job file " << flush;
226  std::ofstream ofs;
227  ofs.open(_jobfile.c_str(), std::ofstream::out);
228  if (!ofs.is_open()) throw runtime_error("\nERROR: bad file handle: " + _jobfile);
229 
230 
231  ctp::QMNBList::iterator pit;
232  ctp::QMNBList &nblist = top->NBList();
233 
234  int jobCount = 0;
235  if (nblist.size() == 0) {
236  cout << endl << "... ... No pairs in neighbor list, skip." << flush;
237  return;
238  }
239 
240  ofs << "<jobs>" << endl;
241  string tag = "";
242 
243  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
244  if ((*pit)->getType()==3){
245  int id1 = (*pit)->Seg1()->getId();
246  string name1 = (*pit)->Seg1()->getName();
247  int id2 = (*pit)->Seg2()->getId();
248  string name2 = (*pit)->Seg2()->getName();
249 
250  int id = ++jobCount;
251 
252 
253  string mps_file1=(boost::format("MP_FILES/%s_n2s%d.mps") % name1 % _statenumber).str();
254  string mps_file2=(boost::format("MP_FILES/%s_n2s%d.mps") % name1 % _statenumber).str();
255 
256  Property Input;
257  Property *pInput = &Input.add("input","");
258  Property *pSegment = &pInput->add("segment" , boost::lexical_cast<string>(id1) );
259  pSegment->setAttribute<string>("type", name1 );
260  pSegment->setAttribute<int>("id", id1 );
261  pSegment->setAttribute<string>("mps_file",mps_file1);
262  pSegment = &pInput->add("segment" , boost::lexical_cast<string>(id2) );
263  pSegment->setAttribute<string>("type", name2 );
264  pSegment->setAttribute<int>("id", id2 );
265  pSegment->setAttribute<string>("mps_file",mps_file2);
266 
267  ctp::Job job(id, tag, Input, ctp::Job::AVAILABLE );
268  job.ToStream(ofs,"xml");
269  }
270  }
271 
272  // CLOSE STREAM
273  ofs << "</jobs>" << endl;
274  ofs.close();
275 
276  cout << endl << "... ... In total " << jobCount << " jobs" << flush;
277 
278 }

280 void IEXCITON::ReadJobFile(ctp::Topology *top) {
281 
282  Property xml;
283 
284  vector<Property*> records;
285 
286  // gets the neighborlist from the topology
287  ctp::QMNBList &nblist = top->NBList();
288  int _number_of_pairs = nblist.size();
289  int _current_pairs=0;
290 
291  // output using logger
292  ctp::Logger _log;
293  _log.setReportLevel(ctp::logINFO);
294 
295 
296  // load the QC results in a vector indexed by the pair ID
297  load_property_from_xml(xml, _jobfile);
298  list<Property*> jobProps = xml.Select("jobs.job");
299  records.resize( _number_of_pairs + 1 );
300 
301  //to skip pairs which are not in the jobfile
302  for (unsigned i=0;i<records.size();i++){
303  records[i]=NULL;
304  }
305  // loop over all jobs = pair records in the job file
306  for (list<Property*> ::iterator it = jobProps.begin(); it != jobProps.end(); ++it) {
307  // if job produced an output, then continue with analysis
308  if ( (*it)->exists("output") && (*it)->exists("output.pair") ) {
309  _current_pairs++;
310  // get the output records
311  Property& poutput = (*it)->get("output.pair");
312  // id's of two segments of a pair
313  int idA = poutput.getAttribute<int>("idA");
314  int idB = poutput.getAttribute<int>("idB");
315  // segments which correspond to these ids
316  ctp::Segment *segA = top->getSegment(idA);
317  ctp::Segment *segB = top->getSegment(idB);
318  // pair that corresponds to the two segments
319  ctp::QMPair *qmp = nblist.FindPair(segA,segB);
320 
321  if (qmp == NULL) { // there is no pair in the neighbor list with this name
322  LOG_SAVE(ctp::logINFO, _log) << "No pair " << idA << ":" << idB << " found in the neighbor list. Ignoring" << flush;
323  } else {
324  //LOG(logINFO, _log) << "Store in record: " << idA << ":" << idB << flush;
325  records[qmp->getId()] = & ((*it)->get("output.pair"));
326  }
327  } else {
328  Property thebadone = (*it)->get("id");
329  throw runtime_error("\nERROR: Job file incomplete.\n Job with id "+thebadone.as<string>()+" is not finished. Check your job file for FAIL, AVAILABLE, or ASSIGNED. Exiting\n");
330  }
331  } // finished loading from the file
332 
333 
334  // loop over all pairs in the neighbor list
335  LOG_SAVE(ctp::logINFO, _log) << "Neighborlist size " << top->NBList().size() << flush;
336  for (ctp::QMNBList::iterator ipair = top->NBList().begin(); ipair != top->NBList().end(); ++ipair) {
337 
338  ctp::QMPair *pair = *ipair;
339 
340  if (records[ pair->getId() ]==NULL) continue; //skip pairs which are not in the jobfile
341 
342  //Segment* segmentA = pair->Seg1();
343  //Segment* segmentB = pair->Seg2();
344 
345  double Jeff2 = 0.0;
346  double jAB=0.0;
347 
348  //cout << "\nProcessing pair " << segmentA->getId() << ":" << segmentB->getId() << flush;
349 
350 
351  if ( pair->getType() == ctp::QMPair::Excitoncl){
352  Property* pair_property = records[ pair->getId() ];
353 
354  list<Property*> pCoupling = pair_property->Select("Coupling");
355 
356  for (list<Property*> ::iterator itCoupling = pCoupling.begin(); itCoupling != pCoupling.end(); ++itCoupling) {
357  jAB = (*itCoupling)->getAttribute<double>("jABstatic");
358  }
359  Jeff2 = jAB*jAB;
360 
361 
362 
363  pair->setJeff2(Jeff2, 2);
364  pair->setIsPathCarrier(true, 2);
365 
366 
367 
368  }
369  }
370 
371  LOG_SAVE(ctp::logINFO, _log) << "Pairs [total:updated] " << _number_of_pairs << ":" << _current_pairs << flush;
372  cout << _log;
373 }
