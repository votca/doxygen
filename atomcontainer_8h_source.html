92  // calculates the lowest and highest point in the cube, sorrounding the
93  // molecule
94  std::pair<Eigen::Vector3d, Eigen::Vector3d> CalcSpatialMinMax() const {
95  std::pair<Eigen::Vector3d, Eigen::Vector3d> result;
96  Eigen::Vector3d min =
97  std::numeric_limits<double>::max() * Eigen::Vector3d::Ones();
98  Eigen::Vector3d max =
99  std::numeric_limits<double>::min() * Eigen::Vector3d::Ones();
100  for (const T& atom : _atomlist) {
101  const Eigen::Vector3d& pos = atom.getPos();
102  if (pos.x() < min.x()) {
103  min.x() = pos.x();
104  }
105  if (pos.x() > max.x()) {
106  max.x() = pos.x();
107  }
108  if (pos.y() < min.y()) {
109  min.y() = pos.y();
110  }
111  if (pos.y() > max.y()) {
112  max.y() = pos.y();
113  }
114  if (pos.z() < min.z()) {
115  min.z() = pos.z();
116  }
117  if (pos.z() > max.z()) {
118  max.z() = pos.z();
119  }
120  }
121  result.first = min;
122  result.second = max;
123  return result;
124  }

165  virtual void ReadFromCpt(CheckpointReader& r) {
166  r(_type, "type");
167  r(_id, "id");
168  int size = 0;
169  r(size, "size");
170  if (size == 0) {
171  return;
172  }
173  T element(0, "H", Eigen::Vector3d::Zero()); // dummy element to get
174  // .identify for type
175  CptTable table = r.openTable(element.identify() + "s", element);
176  _atomlist.clear();
177  _atomlist.reserve(table.numRows());
178  std::vector<typename T::data> dataVec(table.numRows());
179  table.read(dataVec);
180  for (std::size_t i = 0; i < table.numRows(); ++i) {
181  _atomlist.push_back(T(dataVec[i]));
182  }
183  calcPos();
184  }

186  void calcPos() {
187  tools::Elements element;
188  Eigen::Vector3d pos = Eigen::Vector3d::Zero();
189  double totalmass = 0.0;
190  for (const T& atom : _atomlist) {
191  double mass = element.getMass(atom.getElement());
192  totalmass += mass;
193  pos += mass * atom.getPos();
194  }
195  _pos = pos / totalmass;
196  }
