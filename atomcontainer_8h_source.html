91  // calculates the lowest and highest point in the cube, sorrounding the
92  // molecule
93  std::pair<Eigen::Vector3d, Eigen::Vector3d> CalcSpatialMinMax() const {
94  std::pair<Eigen::Vector3d, Eigen::Vector3d> result;
95  Eigen::Vector3d min =
96  std::numeric_limits<double>::max() * Eigen::Vector3d::Ones();
97  Eigen::Vector3d max =
98  std::numeric_limits<double>::min() * Eigen::Vector3d::Ones();
99  for (const T& atom : _atomlist) {
100  const Eigen::Vector3d& pos = atom.getPos();
101  if (pos.x() < min.x()) {
102  min.x() = pos.x();
103  }
104  if (pos.x() > max.x()) {
105  max.x() = pos.x();
106  }
107  if (pos.y() < min.y()) {
108  min.y() = pos.y();
109  }
110  if (pos.y() > max.y()) {
111  max.y() = pos.y();
112  }
113  if (pos.z() < min.z()) {
114  min.z() = pos.z();
115  }
116  if (pos.z() > max.z()) {
117  max.z() = pos.z();
118  }
119  }
120  result.first = min;
121  result.second = max;
122  return result;
123  }

164  virtual void ReadFromCpt(CheckpointReader& r) {
165  r(_type, "type");
166  r(_id, "id");
167  Index size = 0;
168  r(size, "size");
169  if (size == 0) {
170  return;
171  }
172  T element(0, "H", Eigen::Vector3d::Zero()); // dummy element to get
173  // .identify for type
174  CptTable table = r.openTable(element.identify() + "s", element);
175  _atomlist.clear();
176  _atomlist.reserve(table.numRows());
177  std::vector<typename T::data> dataVec(table.numRows());
178  table.read(dataVec);
179  for (std::size_t i = 0; i < table.numRows(); ++i) {
180  _atomlist.push_back(T(dataVec[i]));
181  }
182  calcPos();
183  }

185  void calcPos() {
186  tools::Elements element;
187  Eigen::Vector3d pos = Eigen::Vector3d::Zero();
188  double totalmass = 0.0;
189  for (const T& atom : _atomlist) {
190  double mass = element.getMass(atom.getElement());
191  totalmass += mass;
192  pos += mass * atom.getPos();
193  }
194  _pos = pos / totalmass;
195  }
