66 void linalg_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::vector<double> *residual)
67 {
68 #ifdef NOGSL
69  throw std::runtime_error("linalg_qrsolve is not compiled-in due to disabling of GSL - recompile Votca Tools with GSL support");
70 #else
71  // check matrix for zero column
72  int nonzero_found = 0;
73  for(size_t j=0; j<A.size2(); j++) {
74  nonzero_found = 0;
75  for(size_t i=0; i<A.size1(); i++) {
76  if(fabs(A(i,j))>0) {
77  nonzero_found = 1;
78  }
79  }
80  if(nonzero_found==0) {
81  throw "qrsolve_zero_column_in_matrix";
82  }
83  }
84 
85  gsl_matrix_view m
86  = gsl_matrix_view_array (&A(0,0), A.size1(), A.size2());
87 
88  gsl_vector_view gb
89  = gsl_vector_view_array (&b(0), b.size());
90 
91  gsl_vector *gsl_x = gsl_vector_alloc (x.size());
92  gsl_vector *tau = gsl_vector_alloc (x.size());
93  gsl_vector *gsl_residual = gsl_vector_alloc (b.size());
94 
95  gsl_linalg_QR_decomp (&m.matrix, tau);
96 
97  gsl_linalg_QR_lssolve (&m.matrix, tau, &gb.vector, gsl_x, gsl_residual);
98 
99  for (size_t i =0 ; i < x.size(); i++)
100  x(i) = gsl_vector_get(gsl_x, i);
101 
102  if(residual)
103  for (size_t i =0 ; i < residual->size(); i++)
104  (*residual)(i) = gsl_vector_get(gsl_residual, i);
105 
106  gsl_vector_free (gsl_x);
107  gsl_vector_free (tau);
108  gsl_vector_free (gsl_residual);
109 #endif
110 }

112 void linalg_constrained_qrsolve(ub::vector<double> &x, ub::matrix<double> &A, ub::vector<double> &b, ub::matrix<double> &constr)
113 {
114 #ifdef NOGSL
115  throw std::runtime_error("linalg_constrained_qrsolve is not compiled-in due to disabling of GSL - recompile Votca Tools with GSL support");
116 #else
117  // check matrix for zero column
118  int nonzero_found = 0;
119  for(size_t j=0; j<A.size2(); j++) {
120  nonzero_found = 0;
121  for(size_t i=0; i<A.size1(); i++) {
122  if(fabs(A(i,j))>0) {
123  nonzero_found = 1;
124  }
125  }
126  if(nonzero_found==0) {
127  throw std::runtime_error("constrained_qrsolve_zero_column_in_matrix");
128  }
129  }
130 
131  // Transpose constr:
132  constr = trans(constr);
133 
134  const int N = b.size();
135  const int ngrid = x.size()/2;
136 
137  // temporary variables
138  ub::matrix<double> Q(2*ngrid, 2*ngrid); // Q matrix: QR decomposition of trans(B)
139  ub::matrix<double> Q_k(2*ngrid, 2*ngrid);
140  ub::identity_matrix<double> I (2*ngrid);
141  ub::vector<double> v(2*ngrid);
142 
143  Q = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
144  Q_k = ub::zero_matrix<double>(2*ngrid, 2*ngrid);
145  v = ub::zero_vector<double>(2*ngrid);
146 
147  double *tmp = & constr(0,0);
148  gsl_matrix_view gsl_constr
149  = gsl_matrix_view_array (tmp, constr.size1(), constr.size2());
150 
151  tmp = &b(0);
152  gsl_vector_view gsl_b
153  = gsl_vector_view_array (tmp, b.size());
154 
155 
156  gsl_vector *tau_qr = gsl_vector_alloc (ngrid);
157 
158  gsl_linalg_QR_decomp (&gsl_constr.matrix, tau_qr);
159 
160  Q = I;
161 
162  for (int k = ngrid; k > 0 ; k--) {
163 
164  for (int icout = 0; icout < k - 1; icout++) {
165  v(icout) = 0;
166  }
167  v(k - 1) = 1.0;
168 
169  for (int icout = k; icout < 2*ngrid; icout++) {
170  v(icout) = gsl_matrix_get(&gsl_constr.matrix, icout, k - 1 );
171  }
172 
173  Q_k = I - gsl_vector_get(tau_qr, k - 1 ) * outer_prod ( v, v );
174  Q = prec_prod(Q, Q_k);
175 
176  }
177 
178  Q = trans(Q);
179  gsl_vector_free (tau_qr);
180 
181  // Calculate A * Q and store the result in A
182  A = prec_prod(A, Q);
183 
184 
185  // A = [A1 A2], so A2 is just a block of A
186  ub::matrix<double> A2 = ub::matrix_range<ub::matrix<double> >(A,
187  ub::range (0, N), ub::range (ngrid, 2*ngrid)
188  );
189 
190  tmp = &A2(0,0);
191  gsl_matrix_view gsl_A2
192  = gsl_matrix_view_array (tmp, A2.size1(), A2.size2());
193 
194 
195  gsl_vector *z = gsl_vector_alloc (ngrid);
196  gsl_vector *tau_solve = gsl_vector_alloc (ngrid); // already done!
197  gsl_vector *residual = gsl_vector_alloc (N);
198 
199  gsl_linalg_QR_decomp (&gsl_A2.matrix, tau_solve);
200  gsl_linalg_QR_lssolve (&gsl_A2.matrix, tau_solve, &gsl_b.vector, z, residual);
201 
202  // Next two cycles assemble vector from y (which is zero-vector) and z
203  // (which we just got by gsl_linalg_QR_lssolve)
204 
205  for (int i = 0; i < ngrid; i++ ) {
206  x[i] = 0.0;
207  }
208 
209  for (int i = ngrid; i < 2 * ngrid; i++ ) {
210  x[i] = gsl_vector_get(z, i - ngrid);
211  }
212 
213  // To get the final answer this vector should be multiplied by matrix Q
214  // TODO: here i changed the sign, check again! (victor)
215  x = -prec_prod( Q, x );
216 
217  gsl_vector_free (z);
218  gsl_vector_free (tau_solve);
219  gsl_vector_free (residual);
220 #endif
221 }
