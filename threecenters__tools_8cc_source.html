77  /*
78  * Fill the 3-center object by looping over shells of GW basis set and
79  * calling FillBlock, which calculates all 3-center overlap integrals
80  * associated to a particular shell, convoluted with the DFT orbital
81  * coefficients
82  */
83  void TCMatrix::Fill(const AOBasis& _gwbasis,const AOBasis& _dftbasis,const ub::matrix<double>& _dft_orbitals) {
84 
85  // loop over all shells in the GW basis and get _Mmn for that shell
86  #pragma omp parallel for //private(_block)
87  for ( unsigned _is= 0; _is < _gwbasis.getNumofShells() ; _is++ ){
88  const AOShell* _shell = _gwbasis.getShell(_is);
89  int _start = _shell->getStartIndex();
90  // each element is a shell_size-by-n matrix, initialize to zero
91  std::vector< ub::matrix<double> > _block(this->get_mtot());
92  for (int i = 0; i < this->get_mtot(); i++) {
93  _block[i] = ub::zero_matrix<double>(_shell->getNumFunc(), this->get_ntot());
94  }
95  // Fill block for this shell (3-center overlap with _dft_basis + multiplication with _dft_orbitals )
96  FillBlock(_block, _shell, _dftbasis, _dft_orbitals);
97 
98  // put into correct position
99  for (int _m_level = 0; _m_level < this->get_mtot(); _m_level++) {
100  for (int _i_gw = 0; _i_gw < _shell->getNumFunc(); _i_gw++) {
101  for (int _n_level = 0; _n_level < this->get_ntot(); _n_level++) {
102 
103  _matrix[_m_level](_start + _i_gw, _n_level) = _block[_m_level](_i_gw, _n_level);
104 
105  } // n-th DFT orbital
106  } // GW basis function in shell
107  } // m-th DFT orbital
108  } // shells of GW basis set
109  return;
110  }

113  /*
114  * Determines the 3-center integrals for a given shell in the GW basis
115  * by calculating the 3-center overlap integral of the functions in the
116  * GW shell with ALL functions in the DFT basis set (FillThreeCenterOLBlock),
117  * followed by a convolution of those with the DFT orbital coefficients
118  */

120  void TCMatrix::FillBlock(std::vector< ub::matrix<double> >& _block,const AOShell* _shell,const AOBasis& dftbasis,const ub::matrix<double>& _dft_orbitals) {
121 
122  // prepare local storage for 3-center overlap x m-orbitals
123  ub::matrix<double> _imstore = ub::zero_matrix<double>(mtotal * _shell->getNumFunc(), dftbasis.AOBasisSize());
124 
125 
126  // alpha-loop over the "left" DFT basis function
127  for (AOBasis::AOShellIterator _row = dftbasis.firstShell(); _row != dftbasis.lastShell(); ++_row) {
128  const AOShell* _shell_row = dftbasis.getShell(_row);
129  int _row_start = _shell_row->getStartIndex();
130  int _row_end = _row_start + _shell_row->getNumFunc();
131 
132  // get slice of _dft_orbitals for m-summation, belonging to this shell
133  const ub::matrix_range< const ub::matrix<double> > _m_orbitals = ub::subrange(_dft_orbitals, mmin, mmax + 1, _row_start, _row_end);
134 
135  // gamma-loop over the "right" DFT basis function
136  for (AOBasis::AOShellIterator _col = dftbasis.firstShell(); _col != dftbasis.lastShell(); ++_col) {
137  const AOShell* _shell_col = dftbasis.getShell(_col);
138  int _col_start = _shell_col->getStartIndex();
139  int _col_end = _col_start + _shell_col->getNumFunc();
140 
141  // get 3-center overlap directly as _subvector
142  ub::matrix<double> _subvector = ub::zero_matrix<double>(_shell_row->getNumFunc(), _shell->getNumFunc() * _shell_col->getNumFunc());
143  bool nonzero = FillThreeCenterOLBlock(_subvector, _shell, _shell_row, _shell_col);
144 
145  // if this contributes, multiply _subvector with _dft_orbitals and place in _imstore
146  if (nonzero) {
147 
148  ub::matrix<double> _temp = ub::prod(_m_orbitals, _subvector);
149 
150  // put _temp into _imstore
151  for (unsigned _m_level = 0; _m_level < _temp.size1(); _m_level++) {
152  for (int _i_gw = 0; _i_gw < _shell->getNumFunc(); _i_gw++) {
153  int _ridx = _shell->getNumFunc() * (_m_level - this->mmin) + _i_gw;
154 
155  for (int _cidx = _col_start; _cidx < _col_end; _cidx++) {
156  int _tidx = _shell_col->getNumFunc() * (_i_gw) + _cidx - _col_start;
157  _imstore(_ridx, _cidx) += _temp(_m_level, _tidx);
158  } // index magic
159  } // GW basis function in shell
160  } // m-level
161  } // IF: adding contribution
162  } // gamma-loop
163  } // alpha-loop
164 
165 
166  // get transposed slice of _dft_orbitals
167  const ub::matrix<double> _n_orbitals = ub::trans(ub::subrange(_dft_orbitals, nmin, nmax + 1, 0, _dft_orbitals.size2()));
168 
169  // Now, finally multiply _imstore with _n_orbitals
170  ub::matrix<double> _temp = ub::prod(_imstore, _n_orbitals);
171  //ub::matrix<real_gwbse> _temp = ub::prod(_imstore, _n_orbitals);
172 
173  // and put it into the block it belongs to
174  for (int _m_level = 0; _m_level < mtotal; _m_level++) {
175  for (int _i_gw = 0; _i_gw < _shell->getNumFunc(); _i_gw++) {
176  int _midx = _shell->getNumFunc() *(_m_level - mmin) + _i_gw;
177  for (int _n_level = 0; _n_level < ntotal; _n_level++) {
178  _block[_m_level](_i_gw, _n_level) = _temp(_midx, _n_level);
179  } // n-level
180  } // GW basis function in shell
181  } // m-level
182  return;
183  } // TCMatrix::FillBlock
