168 void CsgFluctuations::EvalConfiguration(Topology *conf, Topology*conf_atom = 0)
169 {
170  vec eR;
171  double r;
172  int rbin;
173 
174 
175 
176  if (_refmol != ""){
177  for(BeadContainer::iterator iter = conf->Beads().begin();
178  iter!=conf->Beads().end();++iter) {
179  Bead *bead = *iter;
180  if(wildcmp(_refmol.c_str(), bead->getName().c_str())){
181  _ref = bead->getPos();
182  cout << " Solute pos " << _ref << endl;
183  }
184 
185  }
186  }
187 
188  for (int i=0; i< _nbins; i++){
189  N[i]=0;
190  }
191 
192  /* check how many molecules are in each bin*/
193  for (BeadContainer::iterator iter = conf->Beads().begin();
194  iter != conf->Beads().end(); ++iter) {
195  Bead *bead = *iter;
196  if (!wildcmp(_filter.c_str(), bead->getName().c_str())) continue;
197 
198  if (_do_spherical){
199  eR = bead->getPos() - _ref;
200  r = abs(eR);
201  }else{
202  eR = bead->getPos();
203  if (_dim == 0) r = eR.getX();
204  else if (_dim == 1) r = eR.getY();
205  else if (_dim == 2) r = eR.getZ();
206  }
207  if (r > _rmin && r < _rmax) {
208  rbin = (int)_nbins*(double)((r - _rmin)/(_rmax - _rmin));
209  N[rbin]++;
210  }
211  }
212 
213  /* update averages*/
214  for (int i=0; i< _nbins; i++){
215  _N_avg[i] += N[i];
216  _N_sq_avg[i] += N[i]*N[i];
217  }
218 
219  _nframes++;
220 }

224 void CsgFluctuations::EndEvaluate()
225 {
226  cout << "Writing results to " << _outfilename << endl;
227  _outfile << "# radius number_fluct avg_number" << endl;
228 
229  for (int i=0; i< _nbins; i++){
230  _N_avg[i] /=_nframes;
231  _N_sq_avg[i] /= _nframes;
232  }
233  for (int i=0; i< _nbins; i++){
234  _outfile << _rmin+i*(_rmax-_rmin)/_nbins << " ";

236  _outfile << _N_avg[i] << endl;
237  }
238 }
