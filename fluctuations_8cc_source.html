160 void CsgFluctuations::EvalConfiguration(Topology *conf, Topology *) {
161 
162  if (_refmol != "") {
163  for (Bead *bead : conf->Beads()) {
164  if (votca::tools::wildcmp(_refmol, bead->getName())) {
165  _ref = bead->getPos();
166  cout << " Solute pos " << _ref << endl;
167  }
168  }
169  }
170 
171  votca::tools::HistogramNew hist;
172  hist.Initialize(_rmin, _rmax, _nbins);
173 
174  /* check how many molecules are in each bin*/
175  for (Bead *bead : conf->Beads()) {
176  if (!votca::tools::wildcmp(_filter, bead->getName())) {
177  continue;
178  }
179  double r = 0;
180  if (_do_spherical) {
181  Eigen::Vector3d eR = bead->getPos() - _ref;
182  r = eR.norm();
183  } else {
184  Eigen::Vector3d eR = bead->getPos();
185  if (_dim == 0) {
186  r = eR.x();
187  } else if (_dim == 1) {
188  r = eR.y();
189  } else if (_dim == 2) {
190  r = eR.z();
191  }
192  }
193  hist.Process(r);
194  }
195 
196  /* update averages*/
197  _N_avg += hist.data().y();
198  _N_sq_avg += hist.data().y().cwiseAbs2();
199 
200  _nframes++;
201 }

203 // output everything when processing frames is done
204 void CsgFluctuations::EndEvaluate() {
205  cout << "Writing results to " << _outfilename << endl;
206  _outfile << "# radius number_fluct avg_number" << endl;
207 
208  for (votca::Index i = 0; i < _nbins; i++) {
209  _N_avg[i] /= (double)_nframes;
210  _N_sq_avg[i] /= (double)_nframes;
211  }
212  for (votca::Index i = 0; i < _nbins; i++) {
213  _outfile << _rmin + (double)i * (_rmax - _rmin) / (double)_nbins << " ";
214  _outfile << (_N_sq_avg[i] - _N_avg[i] * _N_avg[i]) / _N_avg[i]
215  << " "; // fluctuation
216  _outfile << _N_avg[i] << endl;
217  }
218 }
