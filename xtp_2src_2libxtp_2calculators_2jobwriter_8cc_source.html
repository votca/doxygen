31 void JobWriter::Initialize(Property *options) {
32 
33  // update options with the VOTCASHARE defaults
34  UpdateWithDefaults( options, "xtp" );
35 
36  // REGISTER FUNCTIONS
37  _key_funct["mps.dimer"] = &JobWriter::mps_dimer;
38  _key_funct["mps.monomer"] = &JobWriter::mps_monomer;
39  _key_funct["mps.background"] = &JobWriter::mps_background;
40 
41  // SPLIT KEYS
42  string key="options."+Identify();
43  std::string keys = options->get(key+".keys").as<string>();
44  Tokenizer tok_keys(keys, " ,\t\n");
45  tok_keys.ToVector(_keys);
46 
47  // VALIDATE KEYS
48  for (const std::string& key:_keys) {
49  std::map<std::string,WriteFunct>::iterator it = _key_funct.find(key);
50  if (it == _key_funct.end()) {
51  cout << endl << "... ... ERROR: Bad key '" << key << "'" << endl;
52  cout << endl << "Allowed keys: " << flush;
53  for (auto& keyfunc:_key_funct)
54  cout << endl << "... " << keyfunc.first << flush;
55  cout << endl << endl;
56  throw runtime_error("No match for input key.");
57  }
58  }
59 
60  _options = options;
61  return;
62 }

76 void JobWriter::mps_monomer(ctp::Topology *top) {
77 
78  // SET UP FILE STREAM
79  ofstream ofs;
80  std::string jobFile = Identify()+".mps.monomer.xml";
81  ofs.open(jobFile.c_str(), ofstream::out);
82  if (!ofs.is_open()) throw runtime_error("Bad file handle: " + jobFile);
83 
84  ofs << "<jobs>" << endl;
85 
86  int jobCount = 0;
87 
88  std::string str_states = _options->ifExistsReturnElseReturnDefault<string>("options."+Identify()+".states","n e h");
89  std::string seg_pattern = _options->ifExistsReturnElseReturnDefault<string>("options."+Identify()+".pattern","*");
90 
91  Tokenizer tok_states(str_states, " ,\t\n");
92  std::vector<string> statesstring;
93  tok_states.ToVector(statesstring);
94  std::vector<QMState> states;
95  for(const std::string& s:statesstring){
96  states.push_back(QMState(s));
97  }
98 
99 
100  std::vector<string> seg_patterns;
101  Tokenizer tok_pattern(seg_pattern, " ,\t\n");
102  tok_pattern.ToVector(seg_patterns);
103 
104 
105  // CREATE JOBS FOR ALL SEGMENTS AND STATES
106  cout << endl;
107  for (ctp::Segment*seg1:top->Segments()) {
108  int id1 = seg1->getId();
109  std::string name1 = seg1->getName();
110 
111  bool do_continue=true;
112  for (auto & pattern : seg_patterns){
113  if (votca::tools::wildcmp(pattern.c_str(), name1.c_str())){
114  do_continue=false;
115  break;
116  }
117  }
118  if(do_continue){continue;}
119 
120  for (QMState& state: states) {
121  int id = ++jobCount;
122  std::string tag = (format("%1$d:%3$s:%2$s") % id1 % state.ToString() % name1).str();
123  std::string input = (format("%1$d:%2$s:MP_FILES/%2$s_%3$s.mps")
124  % id1 % name1 % state.ToString()).str();
125  std::string stat = "AVAILABLE";
126  ctp::Job job(id, tag, input, stat);
127  job.ToStream(ofs,"xml");
128  cout << "\r... ... # = " << jobCount << flush;
129  }
130  }
131  // CLOSE STREAM
132  ofs << "</jobs>" << endl;
133  ofs.close();
134 }

153 
154  std::string str_states = _options->ifExistsReturnElseReturnDefault<string>("options."+Identify()+".states","nn eh");
155  std::string seg_pattern = _options->ifExistsReturnElseReturnDefault<string>("options."+Identify()+".pattern","*");
156  std::vector<string> statesstring;
157 
158  Tokenizer tok_states(str_states, " ,\t\n");
159  tok_states.ToVector(statesstring);
160  std::vector<string> seg_patterns;
161  Tokenizer tok_pattern(seg_pattern, " ,\t\n");
162  tok_pattern.ToVector(seg_patterns);
163 
164 
165  // DEFINE PAIR CHARGE STATES
166  std::map< string, vector<string> > state1_state2;
167 
168  for (auto & state : statesstring){
169  if(state=="nn"){
170  state1_state2["n"] = vector<string>(1,"n");
171  }
172  else if ( state=="eh" || state=="he"){
173  state1_state2["h"] = vector<string>(1,"e");
174  state1_state2["e"] = vector<string>(1,"h");
175  }
176  }
177 
178  // CREATE JOBS FOR ALL PAIRS AND STATES
179  cout << endl;
180  for (ctp::QMPair* pair:nblist) {
181 
182  int id1 = pair->Seg1()->getId();
183  string name1 = pair->Seg1()->getName();
184  int id2 = pair->Seg2()->getId();
185  string name2 = pair->Seg2()->getName();
186 
187 
188  bool do_continue1=true;
189  for (auto & pattern : seg_patterns){
190  if (votca::tools::wildcmp(pattern.c_str(), name1.c_str())){
191  do_continue1=false;
192  break;
193  }
194  }
195  bool do_continue2=true;
196  for (auto & pattern : seg_patterns){
197  if (votca::tools::wildcmp(pattern.c_str(), name2.c_str())){
198  do_continue2=false;
199  break;
200  }
201  }
202  if(do_continue1 || do_continue2){continue;}
203 
204  for (auto& state1:state1_state2) {
205  for (auto& state2:state1.second) {
206  int id = ++jobCount;
207  string s1 = state1.first;
208  string s2 = state2;
209  string tag = (format("%1$d%2$s:%3$d%4$s") % id1 % s1 % id2 % s2).str();
210  string input = (format("%1$d:%2$s:MP_FILES/%2$s_%3$s.mps "
211  "%4$d:%5$s:MP_FILES/%5$s_%6$s.mps")
212  % id1 % name1 % s1 % id2 % name2 % s2).str();
213  string stat = "AVAILABLE";
214 
215  ctp::Job job(id, tag, input, stat);
216  job.ToStream(ofs,"xml");
217 
218  cout << "\r... ... # = " << jobCount << flush;
219  }
220  }
221  }
222 
223  // CLOSE STREAM
224  ofs << "</jobs>" << endl;
225  ofs.close();
226 }
