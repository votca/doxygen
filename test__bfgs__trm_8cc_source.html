39BOOST_AUTO_TEST_CASE(parabola_test) {
40 class parabola : public Optimiser_costfunction {
41
42 double EvaluateCost(const Eigen::VectorXd& parameters) override {
43 Eigen::VectorXd value = parameters;
44 value(0) -= 2;
45 double cost = value.cwiseAbs2().sum();
46 return cost;
47 }
48
49 Eigen::VectorXd EvaluateGradient(
50 const Eigen::VectorXd& parameters) override {
51 Eigen::VectorXd gradient = 2 * parameters;
52 gradient(0) -= 4;
53 return gradient;
54 }
55
56 bool Converged(const Eigen::VectorXd&, double,
57 const Eigen::VectorXd& gradient) override {
58 if (gradient.cwiseAbs().maxCoeff() < 1e-8) {
59 return true;
60 } else {
61 return false;
62 }
63 }
64
65 Index NumParameters() const override { return 5; }
66 };
67
68 parabola p5;
69 BFGSTRM bfgstrm(p5);
70 Logger log;
71 bfgstrm.setLog(&log);
72 bfgstrm.setNumofIterations(100);
73 bfgstrm.setTrustRadius(0.1);
74 bfgstrm.Optimize(5 * Eigen::VectorXd::Ones(5));
75
76 Eigen::VectorXd ref = Eigen::VectorXd::Zero(5);
77 ref(0) = 2;
78 bool equal = bfgstrm.getParameters().isApprox(ref, 0.00001);

145BOOST_AUTO_TEST_CASE(adiis_test) {
146 Index size = 5;
147
148 Eigen::VectorXd DiF = Eigen::VectorXd::Zero(size);
149 DiF << 0.679243, 0.562675, 0.39399, -0.0258519, 0;
150 Eigen::MatrixXd DiFj = Eigen::MatrixXd::Zero(size, size);
151 DiFj << 0.613998, 0.192684, 0.0326914, -0.193661, 0, 0.192371, 0.0653754,
152 0.0131825, -0.0631915, 0, 0.032739, 0.0131883, 0.0038493, -0.0110991, 0,
153 -0.192873, -0.0631203, -0.0111063, 0.0633221, 0, 0, 0, 0, 0, 0;
154
155 Logger log;
156
157 ADIIS_costfunction a_cost = ADIIS_costfunction(DiF, DiFj);
158 BFGSTRM optimizer = BFGSTRM(a_cost);
159 optimizer.setNumofIterations(1000);
160 optimizer.setTrustRadius(0.01);
161 optimizer.setLog(&log);
162 // Starting point: equal weights on all matrices
163 Eigen::VectorXd coeffs = Eigen::VectorXd::Constant(size, 1.0 / (double)size);
164 optimizer.Optimize(coeffs);
165 bool success = optimizer.Success();
166 coeffs = optimizer.getParameters().cwiseAbs2();
167 double xnorm = coeffs.sum();
168 coeffs /= xnorm;
169
170 BOOST_CHECK_EQUAL(success, true);
171 Eigen::VectorXd ref = Eigen::VectorXd::Zero(size);
172 ref << 0, 0, 0, 0.40826075912352, 0.59173924087648;
173 bool equal = coeffs.isApprox(ref, 0.00001);
174 if (!equal) {
175 cout << "minimum found:" << endl;
176 cout << coeffs << endl;
177 cout << "minimum ref:" << endl;
178 cout << ref << endl;
179 } else {
180 cout << optimizer.getIteration() << endl;
181 }
182 BOOST_CHECK_EQUAL(equal, 1);
183}
