80 void Events::On_execute(Event* event, Graph* graph, State* state, Globaleventinfo* globevent) {
81 
82  if(event->fromtype == Fromtransfer) {
83  Carrier* carrier = event->carrier;
84  Node* fromnode = graph->nodes[carrier->carrier_node_ID];
85  Node* tonode = fromnode->pairing_nodes[event->tonode_ID];
86  Add_remove_carrier(Remove,carrier,graph,fromnode,state,globevent);
87 
88  if(event->totype == Totransfer) {
89  Add_remove_carrier(Add,carrier,graph,tonode,state,globevent);
90  }
91  else if(event->totype == Recombination) {
92  Carrier* recombined_carrier = tonode->carriers_on_node[0];
93  Add_remove_carrier(Remove, recombined_carrier, graph, tonode,state,globevent);
94  if(carrier->carrier_type == Electron) {
95  state->Sell(state->electrons, state->electron_reservoir, carrier->carrier_ID);
96  state->Sell(state->holes, state->hole_reservoir, recombined_carrier->carrier_ID);
97  }
98  else if(carrier->carrier_type == Hole) {
99  state->Sell(state->holes, state->hole_reservoir, carrier->carrier_ID);
100  state->Sell(state->electrons, state->electron_reservoir, recombined_carrier->carrier_ID);
101  }
102  }
103  else if(event->totype == Collection) {
104  if(carrier->carrier_type == Electron) {
105  state->Sell(state->electrons, state->electron_reservoir, carrier->carrier_ID);
106  }
107  else if(carrier->carrier_type == Hole) {
108  state->Sell(state->holes, state->hole_reservoir, carrier->carrier_ID);
109  }
110  }
111  }
112  else if(event->fromtype == Injection) {
113  Node* tonode = event->electrode->pairing_nodes[event->tonode_ID];
114 
115  if(event->totype == Totransfer) {
116  int carrier_ID;
117  if(event->inject_cartype == Electron) {
118  if(state->electron_reservoir.empty()){
119  state->Grow(state->electrons, state->electron_reservoir, globevent->state_grow_size, graph->max_pair_degree);
120  Grow_non_injection_eventvector(globevent->state_grow_size, state->electrons, El_non_injection_events,graph->max_pair_degree);
121  El_non_injection_rates->resize(El_non_injection_events.size());
122  }
123  carrier_ID = state->Buy(state->electrons, state->electron_reservoir);
124  Add_remove_carrier(Add,state->electrons[carrier_ID],graph,tonode,state,globevent);
125  }
126  else if (event->inject_cartype == Hole) {
127  if(state->hole_reservoir.empty()){
128  state->Grow(state->holes, state->hole_reservoir, globevent->state_grow_size, graph->max_pair_degree);
129  Grow_non_injection_eventvector(globevent->state_grow_size, state->holes, Ho_non_injection_events,graph->max_pair_degree);
130  Ho_non_injection_rates->resize(Ho_non_injection_events.size());
131  }
132  carrier_ID = state->Buy(state->holes, state->hole_reservoir);
133  Add_remove_carrier(Add,state->holes[carrier_ID],graph,tonode,state,globevent);
134  }
135  }
136  else if(event->totype == Recombination) {
137  Carrier* recombined_carrier = tonode->carriers_on_node[0];
138  Add_remove_carrier(Remove,recombined_carrier,graph,tonode,state,globevent);
139  if(event->inject_cartype == Electron) {
140  state->Sell(state->holes, state->hole_reservoir, recombined_carrier->carrier_ID);
141  }
142  else if(event->inject_cartype == Hole) {
143  state->Sell(state->electrons, state->electron_reservoir, recombined_carrier->carrier_ID);
144  }
145  }
146  }
147 }

149 void Events::Add_remove_carrier(action AR, Carrier* carrier,Graph* graph, Node* action_node, State* state, Globaleventinfo* globevent){
150 
151  if(AR == Add) {
152  carrier->carrier_node_ID = action_node->node_ID;
153  action_node->carriers_on_node.push_back(carrier);
154 
155  if (carrier->carrier_type == Hole) {
156  nholes++;
157  }
158  else if (carrier->carrier_type == Electron) {
159  nelectrons++;
160  }
161  ncarriers++;
162 
163  state->Add_to_coulomb_mesh(graph, carrier, globevent);
164  }
165  else if(AR == Remove) {
166  action_node->carriers_on_node.pop_back();
167  // Remove existing carrier from lattice
168  if (carrier->carrier_type == Hole) {
169  nholes--;
170  }
171  else if (carrier->carrier_type == Electron) {
172  nelectrons--;
173  }
174  ncarriers--;
175  }
176 
177  Effect_potential_and_non_injection_rates(AR,carrier,graph,state, globevent);
178 
179  // check proximity to left electrode
180  if(globevent->device){
181  double dist_to_left_electrode = action_node->node_position.x();
182  if(dist_to_left_electrode<graph->hopdist){
183  Effect_injection_rates(AR,graph,carrier,dist_to_left_electrode,graph->left_electrode,globevent);
184  }
185 
186  // check proximity to right electrode
187  double dist_to_right_electrode = graph->sim_box_size.x() - action_node->node_position.x();
188  if(dist_to_right_electrode<graph->hopdist){
189  Effect_injection_rates(AR,graph,carrier,dist_to_right_electrode,graph->right_electrode, globevent);
190  }
191  }
192 
193  if(AR == Remove){
194  state->Remove_from_coulomb_mesh(graph, carrier, globevent);
195  }

199 void Events::Effect_potential_and_non_injection_rates(action AR, Carrier* carrier, Graph* graph, State* state,
200  Globaleventinfo* globevent) {
201 
202  int interact_sign;
203 
204  if(AR == Add) {interact_sign = 1;}
205  if(AR == Remove) {interact_sign = -1;}
206  if(carrier->carrier_type == Electron) {interact_sign *= -1;}
207  if(carrier->carrier_type == Hole) {interact_sign *=1;}
208 
209  Node* carnode = graph->nodes[carrier->carrier_node_ID];
210 
211  //calculate the change to the longrange cache
212 
213  if(globevent->device){
214  int layer_index = carnode->layer_index;
215  longrange->layercharge[layer_index] += interact_sign;
216  }
217 
218  myvec carpos = carnode->node_position;
219 
220  // Define cubic boundaries in non-periodic coordinates
221  double ix1 = carpos.x()-globevent->coulcut-graph->hopdist; double ix2 = carpos.x()+globevent->coulcut+graph->hopdist;
222  double iy1 = carpos.y()-globevent->coulcut-graph->hopdist; double iy2 = carpos.y()+globevent->coulcut+graph->hopdist;
223  double iz1 = carpos.z()-globevent->coulcut-graph->hopdist; double iz2 = carpos.z()+globevent->coulcut+graph->hopdist;
224 
225  // Break periodicity in x-direction
226  if(globevent->device) {
227  if (ix1<0.0) ix1 = 0.0;
228  if (ix2>=graph->sim_box_size.x()) ix2 = graph->sim_box_size.x();
229  }
230 
231  // Translate cubic boundaries to sublattice boundaries in non-periodic coordinates
232  int sx1 = floor(ix1/globevent->coulcut);
233  int sx2 = floor(ix2/globevent->coulcut);
234  int sy1 = floor(iy1/globevent->coulcut);
235  int sy2 = floor(iy2/globevent->coulcut);
236  int sz1 = floor(iz1/globevent->coulcut);
237  int sz2 = floor(iz2/globevent->coulcut);
238 
239  // Now visit all relevant sublattices
240  for (int isz=sz1; isz<=sz2; isz++) {
241  int r_isz = isz;
242  while (r_isz < 0) r_isz += state->meshsizeZ;
243  while (r_isz >= state->meshsizeZ) r_isz -= state->meshsizeZ;
244  for (int isy=sy1; isy<=sy2; isy++) {
245  int r_isy = isy;
246  while (r_isy < 0) r_isy += state->meshsizeY;
247  while (r_isy >= state->meshsizeY) r_isy -= state->meshsizeY;
248  for (int isx=sx1; isx<=sx2; isx++) {
249  int r_isx = isx;
250  while (r_isx < 0) r_isx += state->meshsizeX;
251  while (r_isx >= state->meshsizeX) r_isx -= state->meshsizeX;
252  for (int icartype = 0;icartype <2;icartype++) {
253 
254  // Ask a list of all charges in this sublattice
255  list<int>::iterator li1,li2,li3;
256  list<int> *carrierList = &state->coulomb_mesh[r_isx][r_isy][r_isz][icartype];
257  li1 = carrierList->begin();
258  li2 = carrierList->end();
259  for (li3=li1; li3!=li2; li3++) {
260  int probecarrier_ID = *li3;
261  Carrier* probecarrier;
262  if(icartype == 0) {probecarrier = state->electrons[probecarrier_ID];}
263  if(icartype == 1) {probecarrier = state->holes[probecarrier_ID];}
264  Node* probenode = graph->nodes[probecarrier->carrier_node_ID];
265  myvec probepos = probenode->node_position;
266  int probecharge;
267  if(icartype == 0) {
268  probecharge = -1;
269  }
270  else {
271  probecharge = 1;
272  }
273 
274  interact_sign *= probecharge;
275 
276  // Compute coordinates in non-periodic lattice
277  myvec periodic_convert = myvec((isx-r_isx)*globevent->coulcut,(isy-r_isy)*globevent->coulcut,(isz-r_isz)*globevent->coulcut);
278  myvec np_probepos = probepos + periodic_convert;
279  myvec distance = np_probepos-carpos;
280 
281  double distancesqr = abs(distance)*abs(distance);
282 
283  if (probecarrier_ID!=carrier->carrier_ID) {
284  if((carnode->node_ID!=probenode->node_ID)&&(distancesqr<=globevent->coulcut*globevent->coulcut)) {
285 
286  // Charge interacting with its own images, taken care off in graph.h
287  // In case multiple charges are on the same node, coulomb calculation on the same spot is catched
288 
289  //First we take the direction sr interactions into account
290  if (AR==Add) carrier->srfrom +=interact_sign*Compute_Coulomb_potential(np_probepos.x(),-1.0*distance,
291  graph->sim_box_size,globevent);
292  probecarrier->srfrom += interact_sign*Compute_Coulomb_potential(carpos.x(),distance,
293  graph->sim_box_size,globevent);
294  }
295  if (AR==Add) {
296 
297  // Adjust Coulomb potential for neighbours of the added carrier
298  for (unsigned int jump=0; jump < carnode->pairing_nodes.size(); jump++) {
299  myvec jumpdistancevector = carnode->static_event_info[jump].distance;
300  myvec jumpcarrierpos = carnode->node_position + jumpdistancevector;
301  myvec jumpdistance = np_probepos - jumpcarrierpos;
302  double distancejumpsqr = abs(jumpdistance)*abs(jumpdistance);
303 
304  if(distancejumpsqr <= globevent->coulcut*globevent->coulcut) {
305 
306  carrier->srto[jump] += interact_sign*Compute_Coulomb_potential(np_probepos.x(),jumpdistance,
307  graph->sim_box_size, globevent);
308  }
309  }
310  }
311  else if (AR==Remove) {
312 
313  // Reset Coulomb potential for carrier1 and its neighbours
314  carrier->srfrom = 0.0;
315  for (unsigned int jump=0; jump < carnode->pairing_nodes.size(); jump++) {
316  carrier->srto[jump] = 0.0;
317  }
318  }
319 
320  // Adjust Coulomb potential and event rates for neighbours of carrier2
321  for (unsigned int jump=0; jump < probenode->pairing_nodes.size(); jump++) {
322  myvec jumpdistancevector = probenode->static_event_info[jump].distance;
323  myvec jumpprobepos = np_probepos+jumpdistancevector;
324  myvec jumpdistance = carpos-jumpprobepos;
325  double distsqr = abs(jumpdistance)*abs(jumpdistance);
326  int event_ID = probecarrier->carrier_ID*graph->max_pair_degree+jump;
327 
328  double fromlongrange;
329  double tolongrange;
330  if(globevent->device) {
331  fromlongrange = longrange->Get_cached_longrange(probenode->layer_index);
332  if(probenode->pairing_nodes[jump]->node_type == Normal) {
333  tolongrange = longrange->Get_cached_longrange(probenode->pairing_nodes[jump]->layer_index);
334  }
335  else { // collection
336  tolongrange = 0.0;
337  }
338  }
339  else {
340  fromlongrange = 0.0;
341  tolongrange = 0.0;
342  }
343 
344  if(distsqr <= globevent->coulcut*globevent->coulcut) {
345  if(probecarrier->carrier_type==Electron) {
346  probecarrier->srto[jump] +=
347  interact_sign*Compute_Coulomb_potential(carpos.x(),jumpdistance,
348  graph->sim_box_size, globevent);
349 
350  El_non_injection_events[event_ID]->Set_non_injection_event(graph->nodes, probecarrier, jump, fromlongrange, tolongrange, globevent);
351  El_non_injection_rates->setrate(event_ID, El_non_injection_events[event_ID]->rate);
352  el_dirty = true;
353  }
354  else if(probecarrier->carrier_type==Hole) {
355  probecarrier->srto[jump] +=
356  interact_sign*Compute_Coulomb_potential(carpos.x(),jumpdistance,
357  graph->sim_box_size, globevent);
358  Ho_non_injection_events[event_ID]->Set_non_injection_event(graph->nodes, probecarrier, jump, fromlongrange, tolongrange, globevent);
359  Ho_non_injection_rates->setrate(event_ID, Ho_non_injection_events[event_ID]->rate);
360  ho_dirty = true;
361  }
362  }
363  }
364  }
365  }
366  }
367  }
368  }
369  }
370 
371  // update event rates for carrier 1 , done after all carriers within radius coulcut are checked
372  for (unsigned int jump=0; jump < carnode->pairing_nodes.size(); jump++) {
373  int event_ID = carrier->carrier_ID*graph->max_pair_degree+jump;
374 
375  double fromlongrange;
376  double tolongrange;
377  if(globevent->device) {
378  fromlongrange = longrange->Get_cached_longrange(carnode->layer_index);
379  if(carnode->pairing_nodes[jump]->node_type == Normal) {
380  tolongrange = longrange->Get_cached_longrange(carnode->pairing_nodes[jump]->layer_index);
381  }
382  else { // collection
383  tolongrange = 0.0;
384  }
385  }
386  else {
387  fromlongrange = 0.0;
388  tolongrange = 0.0;
389  }
390 
391  if(carrier->carrier_type==Electron) {
392  if(AR == Add) {
393  El_non_injection_events[event_ID]->Set_non_injection_event(graph->nodes, carrier, jump, fromlongrange, tolongrange, globevent);
394  El_non_injection_rates->setrate(event_ID, El_non_injection_events[event_ID]->rate);
395  el_dirty = true;
396  }
397  else {
398  El_non_injection_events[event_ID]->fromtype = Fromnotinbox;
399  El_non_injection_events[event_ID]->totype = Tonotinbox;
400  El_non_injection_events[event_ID]->rate = 0.0;
401  El_non_injection_rates->setrate(event_ID, 0.0);
402  el_dirty = true;
403  }
404  }
405  else if(carrier->carrier_type==Hole) {
406  if(AR == Add) {
407  Ho_non_injection_events[event_ID]->Set_non_injection_event(graph->nodes, carrier, jump, fromlongrange, tolongrange, globevent);
408  Ho_non_injection_rates->setrate(event_ID, Ho_non_injection_events[event_ID]->rate);
409  ho_dirty = true;
410  }
411  else {
412  Ho_non_injection_events[event_ID]->fromtype = Fromnotinbox;
413  Ho_non_injection_events[event_ID]->totype = Tonotinbox;
414  Ho_non_injection_events[event_ID]->rate = 0.0;
415  Ho_non_injection_rates->setrate(event_ID, 0.0);
416  ho_dirty = true;
417  }
418  }
419  }

422 void Events::Effect_injection_rates(action AR, Graph* graph, Carrier* carrier,
423  double dist_to_electrode, Node* electrode,
424  Globaleventinfo* globevent) {
425 
426  int interact_sign;
427  int x_mesh;
428 
429  if(AR == Add) {interact_sign = 1;}
430  if(AR == Remove) {interact_sign = -1;}
431  if(carrier->carrier_type == Electron) {interact_sign *= -1;}
432  if(carrier->carrier_type == Hole) {interact_sign *= 1;}
433  if(electrode->node_type == LeftElectrode) {x_mesh = 0;}
434  if(electrode->node_type == RightElectrode) {x_mesh = graph->nodemeshsizeX-1;}
435 
436  Node* carnode = graph->nodes[carrier->carrier_node_ID];
437  myvec carpos = carnode->node_position;
438 
439  double bound = sqrt(double(globevent->coulcut*globevent->coulcut - dist_to_electrode*dist_to_electrode));
440 
441  // Define cubic boundaries in non-periodic coordinates
442  double iy1 = carpos.y()-bound-graph->hopdist; double iy2 = carpos.y()+bound+graph->hopdist;
443  double iz1 = carpos.z()-bound-graph->hopdist; double iz2 = carpos.z()+bound+graph->hopdist;
444 
445  // Translate cubic boundaries to sublattice boundaries in non-periodic coordinates
446  int sy1 = floor(iy1/graph->hopdist);
447  int sy2 = floor(iy2/graph->hopdist);
448  int sz1 = floor(iz1/graph->hopdist);
449  int sz2 = floor(iz2/graph->hopdist);
450 
451  // Now visit all relevant sublattices
452  for (int isz=sz1; isz<=sz2; isz++) {
453  int r_isz = isz;
454  while (r_isz < 0) r_isz += graph->nodemeshsizeZ;
455  while (r_isz >= graph->nodemeshsizeZ) r_isz -= graph->nodemeshsizeZ;
456  for (int isy=sy1; isy<=sy2; isy++) {
457  int r_isy = isy;
458  while (r_isy < 0) r_isy += graph->nodemeshsizeY;
459  while (r_isy >= graph->nodemeshsizeY) r_isy -= graph->nodemeshsizeY;
460 
461  // Ask a list of all nodes in this sublattice
462  list<Node*>::iterator li1,li2,li3;
463  list<Node*> *nodemesh = &graph->node_mesh[x_mesh][r_isy][r_isz];
464  li1 = nodemesh->begin();
465  li2 = nodemesh->end();
466  for (li3=li1; li3!=li2; li3++) {
467  Node* probenode = *li3;
468  myvec probepos = probenode->node_position;
469 
470  // Compute coordinates in non-periodic lattice
471 
472  myvec periodic_convert = myvec(0.0,(isy-r_isy)*graph->hopdist,(isz-r_isz)*graph->hopdist);
473  myvec np_probepos = probepos + periodic_convert;
474  myvec distance = np_probepos-carpos;
475 
476  double distancesqr = abs(distance)*abs(distance);
477 
478  if ((probenode->node_ID!=carnode->node_ID)&&(distancesqr <= globevent->coulcut*globevent->coulcut)) { // calculated for holes, multiply interact_sign with -1 for electrons
479  probenode->injection_potential +=interact_sign*Compute_Coulomb_potential(carpos.x(),distance,graph->sim_box_size,globevent);
480  int event_ID;
481  int injector_ID;
482  double tolongrange;
483 
484  if(probenode->node_type == Normal){
485  tolongrange = longrange->Get_cached_longrange(probenode->layer_index);
486  }
487  else { // collection (in this case injection to collection)
488  tolongrange = 0.0;
489  }
490 
491  if(electrode->node_type == LeftElectrode) {
492  injector_ID = probenode->left_injector_ID;
493  event_ID = injector_ID;
494  if(globevent->left_injection[1]){
495  Ho_injection_events[event_ID]->Set_injection_event(electrode, injector_ID,
496  Hole, 0.0, tolongrange, globevent);
497  Ho_injection_rates->setrate(event_ID, Ho_injection_events[event_ID]->rate);
498  ho_dirty = true;
499  }
500  if(globevent->left_injection[0]) {
501  El_injection_events[event_ID]->Set_injection_event(electrode, injector_ID,
502  Electron, 0.0, tolongrange, globevent);
503  El_injection_rates->setrate(event_ID, El_injection_events[event_ID]->rate);
504  el_dirty = true;
505  }
506  }
507  else if(electrode->node_type == RightElectrode) {
508  injector_ID = probenode->right_injector_ID;
509  if(globevent->right_injection[1]) {
510  event_ID = injector_ID;
511  if(globevent->left_injection[1]) event_ID += graph->nr_left_injector_nodes;
512  Ho_injection_events[event_ID]->Set_injection_event(electrode, injector_ID,
513  Hole, 0.0, tolongrange, globevent);
514  Ho_injection_rates->setrate(event_ID, Ho_injection_events[event_ID]->rate);
515  ho_dirty = true;
516  }
517  if(globevent->right_injection[0]) {
518  event_ID = injector_ID;
519  if(globevent->left_injection[0]) event_ID += graph->nr_left_injector_nodes;
520  El_injection_events[event_ID]->Set_injection_event(electrode, injector_ID,
521  Electron, 0.0, tolongrange, globevent);
522  El_injection_rates->setrate(event_ID, El_injection_events[event_ID]->rate);

532 double Events::Compute_Coulomb_potential(double startx, myvec dif, myvec sim_box_size, Globaleventinfo* globevent) {
533 
534  double coulpot;
535  double RC = globevent->coulcut;
536  double RCSQR = RC*RC;
537 
538  if(!globevent->device) {
539  coulpot = 1.0/abs(dif)-1.0/RC;
540  }
541  else {
542  coulpot = 1.0/abs(dif)-1.0/RC; // self image potential is taken into account elsewhere
543 
544  double L = sim_box_size.x();
545  double distsqr_planar = dif.y()*dif.y() + dif.z()*dif.z();
546  //double distsqr = dif.x()*dif.x() + distsqr_planar;
547 
548  int sign;
549  double distx_1;
550  double distx_2;
551  double distancesqr_1;
552  double distancesqr_2;
553  bool outside_cut_off1 = false;
554  bool outside_cut_off2 = false;
555 
556  while(!(outside_cut_off1&&outside_cut_off2)) {
557  for (int i=0;i<globevent->nr_sr_images; i++) {
558  if (div(i,2).rem==0) { // even generation
559  sign = -1;
560  distx_1 = i*L + 2*startx + dif.x();
561  distx_2 = (i+2)*L - 2*startx - dif.x();
562  }
563  else {
564  sign = 1;
565  distx_1 = (i+1)*L + dif.x();
566  distx_2 = (i+1)*L - dif.x();
567  }
568  distancesqr_1 = distx_1*distx_1 + distsqr_planar;
569  if (distancesqr_1<=RCSQR) {
570  coulpot += sign*1.0/sqrt(distancesqr_1)-1.0/(RC);
571  }
572  else {
573  outside_cut_off1 = true;
574  }
575  distancesqr_2 = distx_2*distx_2 + distsqr_planar;
576  if (distancesqr_2<=RCSQR) {
577  coulpot += sign*1.0/sqrt(distancesqr_2)-1.0/(RC);
578  }
579  else {
580  outside_cut_off2 = true;
581  }
582  }
583  }
584  }
585  return coulpot;
586 }
