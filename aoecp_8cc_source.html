46  void AOECP::FillBlock(ub::matrix_range< ub::matrix<double> >& _matrix, const AOShell* _shell_row, const AOShell* _shell_col, AOBasis* ecp) {

48  /*
49  *
50  * Currently supported:
51  *
52  * S, P, D, F, G functions in DFT basis and non-local ECPs with l = 0, 1, 2, 3
53  *
54  * or
55  *
56  * S, P, D, F functions in DFT basis and non-local ECPs with l = 0, 1, 2, 3, 4
57  *
58  */

61 
62  // get shell positions
63 
64  int _lmax_row = _shell_row->getLmax();
65  std::vector<double> _contractions_row_full((_lmax_row + 1)*(_lmax_row + 1));
66 
67 
68  int _lmax_col = _shell_col->getLmax();

70 
71 
72 
73  const vec& _pos_row = _shell_row->getPos();
74  const vec& _pos_col = _shell_col->getPos();
75  const vec _diff = _pos_row - _pos_col;
76  // initialize some helper
77  double _distsq = _diff*_diff;
78 
79 
80  // iterate over Gaussians in this _shell_row
81  for (AOShell::GaussianIterator itr = _shell_row->firstGaussian(); itr != _shell_row->lastGaussian(); ++itr) {
82  // iterate over Gaussians in this _shell_col
83  // get decay constant
84  const double _decay_row = itr->getDecay();
85 
86  //if ( _decay_row > 0.08 ) continue;
87 
88  const std::vector<double>& _contractions_row = itr->getContraction();
89  // shitty magic
90  for (int L = 0; L <= _lmax_row; L++) {
91  for (int M = L*L; M < (L + 1)*(L + 1); M++) {
92  _contractions_row_full[M] = _contractions_row[L];
93  }
94  }
95 
96 
97  for (AOShell::GaussianIterator itc = _shell_col->firstGaussian(); itc != _shell_col->lastGaussian(); ++itc) {
98  //get decay constant
99  const double _decay_col = itc->getDecay();
100  const double _fak = 0.5 / (_decay_row + _decay_col);
101  const double _fak2 = 2.0 * _fak;
102 
103 
104  double _exparg = _fak2 * _decay_row * _decay_col *_distsq;
105 
106  // check if distance between postions is big, then skip step
107 
108  if ( _exparg > 30.0 ) {
109  continue;
110  }
111 
112  const std::vector<double>& _contractions_col = itc->getContraction();
113  for (int L = 0; L <= _lmax_col; L++) {
114  for (int M = L * L; M < (L + 1)*(L + 1); M++) {
115  _contractions_col_full[M] = _contractions_col[L];
116  }
117  }
118  // for each atom and its pseudopotential, get a matrix
119  int _atomidx = 0;
120 
121  ub::matrix<int> _power_matrix = ub::zero_matrix<int>(4, 5); // 4 fit components, non-local ECPs l = 0, 1, 2, 3, 4
122  ub::matrix<double> _decay_matrix = ub::zero_matrix<double>(4, 5);
123  ub::matrix<double> _coef_matrix = ub::zero_matrix<double>(4, 5);
124 
125  AOBasis::AOShellIterator final_iter = ecp->lastShell();
126  --final_iter;
127  vec _ecp_eval_pos = vec(0.0);
128  int _lmax_ecp = 0;
129  for (AOBasis::AOShellIterator _ecp = ecp->firstShell(); _ecp != ecp->lastShell(); ++_ecp) {
130 
131  const AOShell* _shell_ecp = ecp->getShell(_ecp);
132  const vec& _ecp_pos = _shell_ecp->getPos();
133 
134  int this_atom = _shell_ecp->getIndex();
135 
136  const int _ecp_l = _shell_ecp->getOffset(); // angular momentum l is stored in offset for ECP
137 
138  // only do the non-local parts
139  if (_ecp_l < _shell_ecp->getNumFunc()) {
140  int _lmax_ecp_old = _lmax_ecp;
141  _lmax_ecp = _shell_ecp->getNumFunc() - 1;
142  int i_fit = -1;
143  for (AOShell::GaussianIterator itecp = _shell_ecp->firstGaussian(); itecp != _shell_ecp->lastGaussian(); ++itecp) {
144  i_fit++;
145 
146  // get info for this angular momentum shell
147  const int _power_ecp = itecp->getPower();
148  const double _decay_ecp = itecp->getDecay();
149  const double _contraction_ecp = itecp->getContraction()[0];
150 
151  // collect atom ECP
152  if (this_atom == _atomidx) {
153  _ecp_eval_pos = _ecp_pos;
154  _power_matrix(i_fit, _ecp_l ) = _power_ecp;
155  _decay_matrix(i_fit, _ecp_l ) = _decay_ecp;
156  _coef_matrix(i_fit, _ecp_l ) = _contraction_ecp;
157  }
158 
159  if ((this_atom != _atomidx ) || ( _ecp == final_iter)) {
160  if (this_atom != _atomidx) {
161  _lmax_ecp = _lmax_ecp_old;
162  }
163 
164  // evaluate collected data, returns a (10x10) matrix of already normalized matrix elements
165  ub::matrix<double> VNL_ECP = calcVNLmatrix(_lmax_ecp, _ecp_eval_pos, *itr, *itc, _power_matrix ,_decay_matrix, _coef_matrix);
166 
167  // consider contractions
168  // cut out block that is needed. sum
169  // cout << "_matrix.size1,2() " << _matrix.size1() << " " << _matrix.size2() << endl;
170  for ( unsigned i = 0; i < _matrix.size1(); i++ ) {
171  for (unsigned j = 0; j < _matrix.size2(); j++) {
172  _matrix(i,j) += VNL_ECP(i+_shell_row->getOffset(),j+_shell_col->getOffset()) * _contractions_row_full[i+_shell_row->getOffset()]* _contractions_col_full[j+_shell_col->getOffset()];
173  }
174  }
175 
176 
177  // reset atom ECP containers
178  _power_matrix = ub::zero_matrix<int>(4, 5); // 4 fit components, non-local ECPs l = 0, 1, 2, 3, 4
179  _decay_matrix = ub::zero_matrix<double>(4, 5);
180  _coef_matrix = ub::zero_matrix<double>(4, 5);
181  _atomidx++;
182  i_fit = 0;
183  //cout << "setting new matrix " << i_fit << " l " << _ecp_l << " alpha " << _decay_ecp << " pref " << _contraction_ecp << endl;
184  _power_matrix(i_fit, _ecp_l ) = _power_ecp;
185  _decay_matrix(i_fit, _ecp_l ) = _decay_ecp;
186  _coef_matrix(i_fit, _ecp_l ) = _contraction_ecp;
187  } // evaluate if new atom is found
188 
189  } // all Gaussians in ecp_shell
190  } // only for non local parts
191 
192  } // all ecp_shells
193 
194  }// _shell_col Gaussians
195  }// _shell_row Gaussians
196 
197  return;
198  }

200  ub::matrix<double> AOECP::calcVNLmatrix(int _lmax_ecp, const vec& posC, const AOGaussianPrimitive& _g_row, const AOGaussianPrimitive& _g_col,const ub::matrix<int>& _power_ecp, const ub::matrix<double>& _gamma_ecp,const ub::matrix<double>& _pref_ecp) {
201 
202  /* calculate the contribution of the nonlocal
203  * ECP of atom at posC with
204  * decay constants in _gamma_ecp
205  * coefficients in _pref_ecp
206  * with angular momentum of max 4
207  *
208  * to DFT basis shell pair
209  * with decay alpha at posA
210  * decay beta at posB
211  */
212 
213  const double conv = 1.e-9; // 1.e-8
214  const int NMAX = 41;
215  const double PI = boost::math::constants::pi<double>();
216  double SQPI = sqrt(PI);
217  double SQ2 = sqrt(2.);
218  double SQ3 = sqrt(3.);
219  double SQ5 = sqrt(5.);
220  double SQ7 = sqrt(7.);
221 
222  double alpha = _g_row.getDecay();
223  double beta = _g_col.getDecay();
224  const vec& posA = _g_row.getShell()->getPos();
225  const vec& posB = _g_col.getShell()->getPos();
226  int _lmax_row = _g_row.getShell()->getLmax();
227  int _lmax_col = _g_col.getShell()->getLmax();
228  int _lmin = std::min({_lmax_row, _lmax_col, _lmax_ecp});
229  int _lmax = std::max({_lmax_row, _lmax_col, _lmax_ecp});
230  int _nsph_row = (_lmax_row + 1) * (_lmax_row + 1);
231  int _nsph_col = (_lmax_col + 1) * (_lmax_col + 1);
232 
233  vec AVS = posA - posC;
234  vec BVS = posB - posC;
235  double AVS2 = AVS * AVS;
236  double BVS2 = BVS * BVS;
237 
238  int INULL = 0;
239  if (AVS2 > 0.01) INULL = 2;
240  if (BVS2 > 0.01) INULL++;
241 
242 
243  ub::matrix<double> matrix = ub::zero_matrix<double>(_nsph_row,_nsph_col);
244  const int nnonsep = _gamma_ecp.size1();
245  int nmax;
246  if (INULL == 0) {
247  nmax = 2 * _lmin;
248  } else if (INULL == 3) {
249  nmax = 2 * NMAX + _lmax_row + _lmax_col;
250  } else {
251  nmax = NMAX + 2 * _lmax;
252  }
253  ub::matrix<double> XI = ub::zero_matrix<double>(_lmax_ecp + 1, nmax + 1);
254 
255  double f_even_r0 = .5 * SQPI;
256  double f_even_r1 = .5;
257  double f_even_r2 = .25 * SQPI;
258 
259  for (int N = 0; N <= nmax; N++) {
260 
261  if ( (N % 2) == 0 ) { // N even, XI(L, odd N) never needed
262 
263  if (N > 0) {
264  f_even_r0 = f_even_r2; // !!
265  f_even_r1 = f_even_r1 * double(N / 2);
266  f_even_r2 = .5 * f_even_r0 * double(N + 1);
267  }
268 
269  double DFAK_r0 = .5 * double(N + 1);
270  double DFAK_r1 = .5 * double(N + 2);
271  double DFAK_r2 = .5 * double(N + 3);
272 
273  for (int L = 0; L <= _lmax_ecp; L++) {
274 
275  for (int I = 0; I < nnonsep; I++) {
276  int power = _power_ecp(I, L);
277  double DLI = (alpha + beta + _gamma_ecp(I, L));
278  if (power == 2) {
279  XI(L, N) += f_even_r2 * _pref_ecp(I, L) / pow(DLI, DFAK_r2); // r^2 terms
280  } else if (power == 0) {
281  XI(L, N) += f_even_r0 * _pref_ecp(I, L) / pow(DLI, DFAK_r0); // r^0 terms
282  } else if (power == 1) {
283  XI(L, N) += f_even_r1 * _pref_ecp(I, L) / pow(DLI, DFAK_r1); // r^1 terms
284  }
285  }
286 
287  }
288 
289  } // end if ( (N % 2) == 0 )
290 
291  }

304  double AVSSQ = 0.;
305  int NMAX1 = 0;
306  if (AVS2 > 0.01) {
307 
308  G1 = exp(-alpha * AVS2);
309  AVSSQ = sqrt(AVS2);
310  double AMAX = 0.0;
311  double fak = 2.0 * alpha * AVSSQ;
312  double Pow = 1.;
313  double factorialNN = 1;

328  case 1:
329  AMAX = AF * std::max({XI(0, ii), XI(1, ii)});
330  break;
331  case 2:
332  AMAX = AF * std::max({XI(0, ii), XI(1, ii), XI(2, ii)});
333  break;
334  case 3:
335  AMAX = AF * std::max({XI(0, ii), XI(1, ii), XI(2, ii), XI(3, ii)});
336  break;
337  case 4:
338  AMAX = AF * std::max({XI(0, ii), XI(1, ii), XI(2, ii), XI(3, ii), XI(4, ii)});
339  break;
340  }
341 
342  if (NMAX1 == 0 && AMAX <= conv) NMAX1 = NN;
343  if (NMAX1 != 0 && AMAX > conv ) NMAX1 = 0;
344  }

379  break;
380  case 2:

382  break;

384  BMAX = BF * std::max({XI(0, ii), XI(1, ii), XI(2, ii), XI(3, ii)});

389  }
390

392  if (NMAX2 != 0 && BMAX > conv) NMAX2 = 0;
393  }

870  ub::vector<double> AOECP::CalcNorms(double decay, int size) {
871  ub::vector<double> Norms = ub::vector<double>(size);
872  const double PI = boost::math::constants::pi<double>();
873  double SQ2, SQ3, SQ5;
874 
875  double Norm_S = pow(2.0 * decay / PI, 0.75);
876  double Norm_P=0.0;
877  double Norm_D=0.0;
878  Norms[0] = Norm_S; // Y 00
879 
880  if (size > 1) {
881  Norm_P = 2.0 * sqrt(decay) * Norm_S;
882  Norms[1] = Norm_P; // Y 10
883  Norms[2] = Norm_P; // Y 1-1
884  Norms[3] = Norm_P; // Y 11
885  }
886 
887  if (size > 4) {
888  SQ3 = sqrt(3.);
889  double Norm_D = 4.00 * decay * Norm_S;
890  Norms[4] = .5 * Norm_D / SQ3; // Y 20
891  Norms[5] = Norm_D; // Y 2-1
892  Norms[6] = Norm_D; // Y 21
893  Norms[7] = Norm_D; // Y 2-2
894  Norms[8] = .5 * Norm_D; // Y 22
895  }
896 
897  if (size > 9) {
898  SQ2 = sqrt(2.);
899  SQ5 = sqrt(5.);
900  double Norm_F = 4.00 * decay * Norm_P;
901  double Norm_F_1 = .5 * Norm_F / (SQ2 * SQ5);
902  double Norm_F_3 = .5 * Norm_F / (SQ2 * SQ3);
903  Norms[9] = .5 * Norm_F / (SQ3 * SQ5); // Y 30

909  Norms[15] = Norm_F_3; // Y 33
910  }

912  if (size > 16) {
913  double SQ7 = sqrt(7.);

916  double Norm_G_m2 = .5 * Norm_G / (SQ3 * SQ7);

928  }
