27 Segment::Segment(int id, string name)
28  : _id(id), _name(name),
29  _has_e(false), _has_h(false),_has_s(false), _has_t(false)
30  { _eMpoles.resize(5); }
31 
32 // This constructor creates a copy of the stencil segment, without
33 // adding it to any containers further up in the hierarchy; i.e. the topology
34 // and molecules will neither know about the existence of this segment, nor
35 // be able to access it. Used for creating the ghost in PB corrected pairs.
36 Segment::Segment(Segment *stencil)
37  : _id(stencil->getId()), _name(stencil->getName()+"_ghost"),
38  _typ(stencil->getType()), _top(NULL), _mol(NULL),
39  _CoM(stencil->getPos()),
40  _has_e(false), _has_h(false),_has_s(false), _has_t(false)
41  { _eMpoles.resize(5);
42 
43  std::vector<Fragment*> ::iterator fit;
44  for (fit = stencil->Fragments().begin();
45  fit < stencil->Fragments().end();
46  fit++) {
47 
48  Fragment *newFrag = new Fragment(*fit);
49  this->AddFragment(newFrag);
50 
51  std::vector<Atom*> ::iterator ait;
52  for (ait = newFrag->Atoms().begin();
53  ait < newFrag->Atoms().end();
54  ait++) {
55  this->AddAtom(*ait);
56  }
57  }
58 }

356 void Segment::calcApproxSize(){
357  _approxsize=0.0;
358 
359  tools::vec min=vec( std::numeric_limits<const double>::max() );
360  tools::vec max=vec(std::numeric_limits<const double>::min());
361  std::vector< ctp::Fragment* > ::iterator fragit1;
362  for (fragit1 = Fragments().begin();
363  fragit1 < Fragments().end();
364  fragit1 ++) {
365  const tools::vec& pos= (*fragit1)->getPos();
366  if(pos.getX()>max.getX()){
367  max.x()=pos.getX();
368  }
369  else if(pos.getX()<min.getX()){
370  min.x()=pos.getX();
371  }
372  if(pos.getY()>max.getY()){
373  max.y()=pos.getY();
374  }
375  else if(pos.getY()<min.getY()){
376  min.y()=pos.getY();
377  }
378  if(pos.getZ()>max.getZ()){
379  max.z()=pos.getZ();
380  }
381  else if(pos.getZ()<min.getZ()){
382  min.z()=pos.getZ();
383  }
384  }
385 
386  _approxsize=abs(max-min);
387 
388  return;
389 }
