26 void matrix::RandomRotation()
27 {
28  matrix &M=(*this);
29  double theta = drand48() * PITIMES2; /* Rotation about the pole (Z). */
30  double phi = drand48() * PITIMES2; /* For direction of pole deflection. */
31  double z = drand48() * 2.0; /* For magnitude of pole deflection. */
32 
33  /* Compute a vector V used for distributing points over the sphere */
34  /* via the reflection I - V Transpose(V). This formulation of V */
35  /* will guarantee that if x[1] and x[2] are uniformly distributed, */
36  /* the reflected points will be uniform on the sphere. Note that V */
37  /* has length sqrt(2) to eliminate the 2 in the Householder matrix. */
38 
39  double r = sqrt( z );
40  double Vx = sin( phi ) * r;
41  double Vy = cos( phi ) * r;
42  double Vz = sqrt( 2.0 - z );
43 
44  /* Compute the row vector S = Transpose(V) * R, where R is a simple */
45  /* rotation by theta about the z-axis. No need to compute Sz since */
46  /* it's just Vz. */
47 
48  double st = sin( theta );
49  double ct = cos( theta );
50  double Sx = Vx * ct - Vy * st;
51  double Sy = Vx * st + Vy * ct;
52 
53  /* Construct the rotation matrix ( V Transpose(V) - I ) R, which */
54  /* is equivalent to V S - R. */
55 
56  M[0][0] = Vx * Sx - ct;
57  M[0][1] = Vx * Sy - st;
58  M[0][2] = Vx * Vz;
59 
60  M[1][0] = Vy * Sx + st;
61  M[1][1] = Vy * Sy - ct;
62  M[1][2] = Vy * Vz;
63 
64  M[2][0] = Vz * Sx;
65  M[2][1] = Vz * Sy;
66  M[2][2] = 1.0 - z; /* This equals Vz * Vz - 1.0 */
67 }

69 int cjcbi(matrix &a, matrix &v, double eps, int jt)
70 {
71  int n=3;
72  int i,j,p,q,l;
73  double fm,cn,sn,omega,x,y,d;
74  l=1;
75 
76  v.UnitMatrix();
77 
78  while (true) {
79  fm=0.0;
80  for (i=1; i<=n-1; i++)
81  for (j=0; j<=i-1; j++) {
82  d=fabs(a[i][j]);
83  if ((i!=j)&&(d>fm))
84  { fm=d; p=i; q=j;}
85  }
86  if (fm<eps) return(1);
87  if (l>jt) return(-1);
88  l=l+1;
89  x=-a[p][q]; y=(a[q][q]-a[p][p])/2.0;
90  omega=x/sqrt(x*x+y*y);
91  if (y<0.0) omega=-omega;
92  sn=1.0+sqrt(1.0-omega*omega);
93  sn=omega/sqrt(2.0*sn);
94  cn=sqrt(1.0-sn*sn);
95  fm=a[p][p];
96  a[p][p]=fm*cn*cn+a[q][q]*sn*sn+a[p][q]*omega;
97  a[q][q]=fm*sn*sn+a[q][q]*cn*cn-a[p][q]*omega;
98  a[p][q]=0.0; a[q][p]=0.0;
99  for (j=0; j<=n-1; j++)
100  if ((j!=p)&&(j!=q))
101  { //u=p*n+j; w=q*n+j;
102  fm=a[p][j];
103  a[p][j]=fm*cn+a[q][j]*sn;
104  a[q][j]=-fm*sn+a[q][j]*cn;
105  }
106  for (i=0; i<=n-1; i++)
107  if ((i!=p)&&(i!=q))
108  { //u=i*n+p; w=i*n+q;
109  fm=a[i][p];
110  a[i][p]=fm*cn+a[i][q]*sn;
111  a[i][q]=-fm*sn+a[i][q]*cn;
112  }
113  for (i=0; i<=n-1; i++)
114  { //u=i*n+p; w=i*n+q;
115  fm=v[i][p];
116  v[i][p]=fm*cn+v[i][q]*sn;
117  v[i][q]=-fm*sn+v[i][q]*cn;
118  }
119  }
120  return(1);
121 }

123 void matrix::SolveEigensystem(eigensystem_t &out)
124 {
125  matrix m(*this);
126  matrix v;
127  cjcbi(m, v);
128 
129  for(int i=0; i<3; ++i) {
130  out.eigenvalues[i] = m[i][i];
131  out.eigenvecs[i] = vec(v[0][i], v[1][i], v[2][i]);
132  out.eigenvecs[i].normalize();
133  }
134 
135  //cout << v << endl;

151 void matrix::Invert()
152 {
153  matrix mi;
154  matrix &m=*this;
155  mi[0][0] = m[1][1]*m[2][2] - m[1][2]*m[2][1];
156  mi[1][0] = m[1][2]*m[2][0] - m[1][0]*m[2][2];
157  mi[2][0] = m[1][0]*m[2][1] - m[1][1]*m[2][0];
158  mi[0][1] = m[0][2]*m[2][1] - m[0][1]*m[2][2];
159  mi[1][1] = m[0][0]*m[2][2] - m[0][2]*m[2][0];
160  mi[2][1] = m[0][1]*m[2][0] - m[0][0]*m[2][1];
161  mi[0][2] = m[0][1]*m[1][2] - m[0][2]*m[1][1];
162  mi[1][2] = m[0][2]*m[1][0] - m[0][0]*m[1][2];
163  mi[2][2] = m[0][0]*m[1][1] - m[0][1]*m[1][0];
164 
166  double D;
167 
168  D = m[0][0]*mi[0][0] + m[0][1]*mi[1][0] + m[0][2]*mi[2][0];
169 
170  mi /= D;
171  *this = mi;
172 }
