00026 void matrix::RandomRotation()
00027 {
00028     matrix &M=(*this);
00029     double theta = drand48() * PITIMES2; /* Rotation about the pole (Z).      */
00030     double phi   = drand48() * PITIMES2; /* For direction of pole deflection. */
00031     double z     = drand48() * 2.0;      /* For magnitude of pole deflection. */
00032 
00033     /* Compute a vector V used for distributing points over the sphere  */
00034     /* via the reflection I - V Transpose(V).  This formulation of V    */
00035     /* will guarantee that if x[1] and x[2] are uniformly distributed,  */
00036     /* the reflected points will be uniform on the sphere.  Note that V */
00037     /* has length sqrt(2) to eliminate the 2 in the Householder matrix. */
00038 
00039     double r  = sqrt( z );
00040     double Vx = sin( phi ) * r;
00041     double Vy = cos( phi ) * r;
00042     double Vz = sqrt( 2.0 - z );    
00043 
00044     /* Compute the row vector S = Transpose(V) * R, where R is a simple */
00045     /* rotation by theta about the z-axis.  No need to compute Sz since */
00046     /* it's just Vz.                                                    */
00047 
00048     double st = sin( theta );
00049     double ct = cos( theta );
00050     double Sx = Vx * ct - Vy * st;
00051     double Sy = Vx * st + Vy * ct;
00052 
00053     /* Construct the rotation matrix  ( V Transpose(V) - I ) R, which   */
00054     /* is equivalent to V S - R.                                        */
00055 
00056     M[0][0] = Vx * Sx - ct;
00057     M[0][1] = Vx * Sy - st;
00058     M[0][2] = Vx * Vz;
00059 
00060     M[1][0] = Vy * Sx + st;
00061     M[1][1] = Vy * Sy - ct;
00062     M[1][2] = Vy * Vz;
00063 
00064     M[2][0] = Vz * Sx;
00065     M[2][1] = Vz * Sy;
00066     M[2][2] = 1.0 - z;   /* This equals Vz * Vz - 1.0 */   
00067 }

00069 int cjcbi(matrix &a, matrix &v, double eps, int jt)
00070 { 
00071     int n=3;
00072     int i,j,p,q,l;
00073     double fm,cn,sn,omega,x,y,d;
00074     l=1;
00075     
00076     v.UnitMatrix();
00077     
00078     while (true) {
00079         fm=0.0;
00080         for (i=1; i<=n-1; i++)
00081             for (j=0; j<=i-1; j++) {
00082                 d=fabs(a[i][j]);
00083                 if ((i!=j)&&(d>fm))
00084                     { fm=d; p=i; q=j;}
00085             }
00086         if (fm<eps)  return(1);
00087         if (l>jt)  return(-1);
00088         l=l+1;
00089         x=-a[p][q]; y=(a[q][q]-a[p][p])/2.0;
00090         omega=x/sqrt(x*x+y*y);
00091         if (y<0.0) omega=-omega;
00092         sn=1.0+sqrt(1.0-omega*omega);
00093         sn=omega/sqrt(2.0*sn);
00094         cn=sqrt(1.0-sn*sn);
00095         fm=a[p][p];
00096         a[p][p]=fm*cn*cn+a[q][q]*sn*sn+a[p][q]*omega;
00097         a[q][q]=fm*sn*sn+a[q][q]*cn*cn-a[p][q]*omega;
00098         a[p][q]=0.0; a[q][p]=0.0;
00099         for (j=0; j<=n-1; j++)
00100         if ((j!=p)&&(j!=q))
00101           { //u=p*n+j; w=q*n+j;
00102             fm=a[p][j];
00103             a[p][j]=fm*cn+a[q][j]*sn;
00104             a[q][j]=-fm*sn+a[q][j]*cn;
00105           }
00106         for (i=0; i<=n-1; i++)
00107           if ((i!=p)&&(i!=q))
00108             { //u=i*n+p; w=i*n+q;
00109               fm=a[i][p];
00110               a[i][p]=fm*cn+a[i][q]*sn;
00111               a[i][q]=-fm*sn+a[i][q]*cn;
00112             }
00113         for (i=0; i<=n-1; i++)
00114           { //u=i*n+p; w=i*n+q;
00115             fm=v[i][p];
00116             v[i][p]=fm*cn+v[i][q]*sn;
00117             v[i][q]=-fm*sn+v[i][q]*cn;
00118           }
00119       }
00120     return(1);
00121 }

00123 void matrix::SolveEigensystem(eigensystem_t &out)
00124 {
00125     matrix m(*this);
00126     matrix v;
00127     cjcbi(m, v);
00128 
00129     for(int i=0; i<3; ++i) {
00130         out.eigenvalues[i] = m[i][i];
00131         out.eigenvecs[i] = vec(v[0][i], v[1][i], v[2][i]);
00132         out.eigenvecs[i].normalize();
00133     }
00134 
00135     //cout << v << endl;
00136     // sort by eigenvalues
00137     if(out.eigenvalues[0] > out.eigenvalues[1]) {
00138         std::swap(out.eigenvalues[0], out.eigenvalues[1]);
00139         std::swap(out.eigenvecs[0], out.eigenvecs[1]);
00140     }
00141     if(out.eigenvalues[1] > out.eigenvalues[2]) {
00142         std::swap(out.eigenvalues[1], out.eigenvalues[2]);
00143         std::swap(out.eigenvecs[1], out.eigenvecs[2]);
00144     }
00145     if(out.eigenvalues[0] > out.eigenvalues[1]) {
00146         std::swap(out.eigenvalues[0], out.eigenvalues[1]);

00151 void matrix::Invert()
00152 {
00153     matrix mi;
00154     matrix &m=*this;
00155     mi[0][0] = m[1][1]*m[2][2] - m[1][2]*m[2][1];
00156     mi[1][0] = m[1][2]*m[2][0] - m[1][0]*m[2][2];
00157     mi[2][0] = m[1][0]*m[2][1] - m[1][1]*m[2][0];
00158     mi[0][1] = m[0][2]*m[2][1] - m[0][1]*m[2][2];
00159     mi[1][1] = m[0][0]*m[2][2] - m[0][2]*m[2][0];
00160     mi[2][1] = m[0][1]*m[2][0] - m[0][0]*m[2][1];
00161     mi[0][2] = m[0][1]*m[1][2] - m[0][2]*m[1][1];
00162     mi[1][2] = m[0][2]*m[1][0] - m[0][0]*m[1][2];
00163     mi[2][2] = m[0][0]*m[1][1] - m[0][1]*m[1][0];
00164 
00166     double D;
00167 
00168     D = m[0][0]*mi[0][0] + m[0][1]*mi[1][0] + m[0][2]*mi[2][0];
00169     
00170     mi /= D;
00171     *this = mi;
00172 }
