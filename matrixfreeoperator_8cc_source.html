33 Eigen::VectorXd MatrixFreeOperator::diagonal() const {
34  Eigen::VectorXd D = Eigen::VectorXd::Zero(_size);
35  if (useRow()) {
36 #pragma omp parallel for schedule(guided)
37  for (Index i = 0; i < _size; i++) {
38  Eigen::RowVectorXd row_data = this->OperatorRow(i);
39  D(i) = row_data(i);
40  }
41  }
42 
43  if (useBlock()) {
44  Index blocksize = getBlocksize();
45  if (size() % blocksize != 0) {
46  throw std::runtime_error("blocksize is not a multiple of matrix size");
47  }
48  Index blocks = size() / blocksize;
49 
50 // this is inefficient if blocks<num_ofthreads
51 #pragma omp parallel for schedule(guided)
52  for (Index i = 0; i < blocks; i++) {
53  Eigen::MatrixXd block = OperatorBlock(i, i);
54  D.segment(i * blocksize, blocksize) += block.diagonal();
55  }
56  }
57 
58  return D;
59 }
