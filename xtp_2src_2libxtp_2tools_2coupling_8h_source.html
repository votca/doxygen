98 bool Coupling::Evaluate() {
99 
100  _log.setReportLevel( ctp::logDEBUG );
101  _log.setMultithreading( true );
102 
103  _log.setPreface(ctp::logINFO, "\n... ...");
104  _log.setPreface(ctp::logERROR, "\n... ...");
105  _log.setPreface(ctp::logWARNING, "\n... ...");
106  _log.setPreface(ctp::logDEBUG, "\n... ...");
107 
108  // get the corresponding object from the QMPackageFactory
109  QMPackage *_qmpackage = QMPackages().Create( _package );
110  _qmpackage->setLog( &_log );
111  _qmpackage->Initialize( &_package_options );
112  _qmpackage->setRunDir(".");
113  Orbitals _orbitalsA, _orbitalsB, _orbitalsAB;
114 
115 
116 
117  _qmpackage->setLogFileName( _logA );
118  bool _parse_logA_status = _qmpackage->ParseLogFile( &_orbitalsA );
119  if ( !_parse_logA_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule A" << std::flush; }
120 
121  _qmpackage->setLogFileName( _logB );
122  bool _parse_logB_status = _qmpackage->ParseLogFile( &_orbitalsB );
123  if ( !_parse_logB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule B" << std::flush; }
124 
125  _qmpackage->setLogFileName( _logAB );
126  bool _parse_logAB_status = _qmpackage->ParseLogFile( &_orbitalsAB );
127  if ( !_parse_logAB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule AB" << std::flush; }
128 
129  _qmpackage->setOrbitalsFileName( _orbA );
130  bool _parse_orbitalsA_status = _qmpackage->ParseOrbitalsFile( &_orbitalsA );
131  if ( !_parse_orbitalsA_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of molecule A" << std::flush; }
132 
133  _qmpackage->setOrbitalsFileName( _orbB );
134  bool _parse_orbitalsB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsB );
135  if ( !_parse_orbitalsB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of molecule B" << std::flush; }
136 
137  _qmpackage->setOrbitalsFileName( _orbAB );
138  bool _parse_orbitalsAB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
139  if ( !_parse_orbitalsAB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of dimer AB" << std::flush; }
140 
141  int _degAH = 1;
142  int _degAL = 1;
143  int _degBH = 1;
144  int _degBL = 1;
145 
146  // trim monomers A and B to one level
147  if ((_trimA < 0) || (_trimB <0) ) { // any -1 overrides the specification of the other
148 
149  // find degeneracy of HOMOs and LUMOs
150  std::vector<int> list_levelsAH = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons()-1, _degeneracy ));
151  _degAH = list_levelsAH.size();
152  std::vector<int> list_levelsAL = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons(), _degeneracy ));
153  _degAL = list_levelsAL.size();
154 
155  std::vector<int> list_levelsBH = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons()-1, _degeneracy ));
156  _degBH = list_levelsBH.size();
157  std::vector<int> list_levelsBL = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons(), _degeneracy ));
158  _degBL = list_levelsBL.size();
159 
160  _orbitalsA.Trim(_degAH,_degAL);
161  _orbitalsB.Trim(_degBH,_degBL);
162 
163  // trim by the factors (_trimA-1) and (_trimB-1)
164  } else {
165 
166  if ( _orbitalsA.getNumberOfElectrons()*(_trimA-1) < int(_orbitalsA.getNumberOfLevels()) - _orbitalsA.getNumberOfElectrons() ) {
167  CTP_LOG(ctp::logDEBUG,_log) << "Trimming virtual orbitals A:"
168  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
169  << _orbitalsA.getNumberOfElectrons()*(_trimA-1) << std::flush;
170  _orbitalsA.Trim(_trimA);
171  }
172 
173  if ( _orbitalsB.getNumberOfElectrons()*(_trimB-1) < int(_orbitalsB.getNumberOfLevels()) - _orbitalsB.getNumberOfElectrons() ) {
174  CTP_LOG(ctp::logDEBUG,_log) << "Trimming virtual orbitals B:"
175  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
176  << _orbitalsB.getNumberOfElectrons()*(_trimB-1) << std::flush;
177  _orbitalsB.Trim(_trimB);
178  }
179 
180  }
181 
182  DFTcoupling dftcoupling;
183  dftcoupling.setLogger(&_log);
184 
185  ub::matrix<double> _JAB;
186  bool _calculate_integrals = dftcoupling.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
187  if ( !_calculate_integrals ) { CTP_LOG(ctp::logERROR,_log) << "Failed to evaluate integrals" << std::flush; }
188 
189  std::cout << _log;
190 
191 
192  // output the results
193  tools::Property _summary;
194  tools::Property *_job_output = &_summary.add("output","");
195  tools::Property *_pair_summary = &_job_output->add("pair","");
196  int HOMO_A = _orbitalsA.getNumberOfElectrons();
197  int HOMO_B = _orbitalsB.getNumberOfElectrons();
198  int LUMO_A = HOMO_A + 1;
199  int LUMO_B = HOMO_B + 1;
200  _pair_summary->setAttribute("homoA", HOMO_A);
201  _pair_summary->setAttribute("homoB", HOMO_B);
202 
203  if ( (_trimA <0) || (_trimB <0) ) {
204 
205  // HOMO-HOMO coupling
206  double JAB = dftcoupling.getCouplingElement(_degAH, _degBH , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
207  tools::Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<std::string>(JAB));
208  double energyA = _orbitalsA.getEnergy(_degAH);
209  double energyB = _orbitalsB.getEnergy(_degBH);
210  _overlap_summary->setAttribute("orbA", HOMO_A);
211  _overlap_summary->setAttribute("orbB", HOMO_B);
212  //_overlap_summary->setAttribute("jAB", JAB);
213  _overlap_summary->setAttribute("eA", energyA);
214  _overlap_summary->setAttribute("eB", energyB);
215 
216  // LUMO-LUMO coupling
217  JAB = dftcoupling.getCouplingElement(_degAH+1, _degBH+1 , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
218  _overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<std::string>(JAB));
219  energyA = _orbitalsA.getEnergy(_degAH +1);
220  energyB = _orbitalsB.getEnergy(_degBH +1);
221  _overlap_summary->setAttribute("orbA", LUMO_A);
222  _overlap_summary->setAttribute("orbB", LUMO_B);
223  //_overlap_summary->setAttribute("jAB", JAB);
224  _overlap_summary->setAttribute("eA", energyA);
225  _overlap_summary->setAttribute("eB", energyB);
226 
227  } else {
228 
229  for (int levelA = HOMO_A - _levA +1; levelA <= LUMO_A + _levA - 1; ++levelA ) {
230  for (int levelB = HOMO_B - _levB + 1; levelB <= LUMO_B + _levB -1 ; ++levelB ) {
231  double JAB = dftcoupling.getCouplingElement( levelA , levelB, &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy );
232  tools::Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<std::string>(JAB));
233  double energyA = _orbitalsA.getEnergy( levelA );
234  double energyB = _orbitalsB.getEnergy( levelB );
235  _overlap_summary->setAttribute("orbA", levelA);
236  _overlap_summary->setAttribute("orbB", levelB);
237  //_overlap_summary->setAttribute("jAB", JAB);
238  _overlap_summary->setAttribute("eA", energyA);
239  _overlap_summary->setAttribute("eB", energyB);
240  }
241  }
242  }
243 
244  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
245 
246  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
247  ofs << *_job_output;
248  ofs.close();
249 
250  return true;
251 }
