92 bool Coupling::Evaluate() {
93 
94  _log.setReportLevel( ctp::logDEBUG );
95  _log.setMultithreading( true );
96 
97  _log.setPreface(ctp::logINFO, "\n... ...");
98  _log.setPreface(ctp::logERROR, "\n... ...");

101 
102  // get the corresponding object from the QMPackageFactory
103  QMPackage *qmpackage = QMPackages().Create( _package );
104  qmpackage->setLog( &_log );
105  qmpackage->Initialize( _package_options );
106  qmpackage->setRunDir(".");
107  Orbitals orbitalsA, orbitalsB, orbitalsAB;
108 
109 
110 
111  qmpackage->setLogFileName( _logA );
112  bool _parse_logA_status = qmpackage->ParseLogFile( orbitalsA );
113  if ( !_parse_logA_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule A" << std::flush; }
114 
115  qmpackage->setLogFileName( _logB );
116  bool _parse_logB_status = qmpackage->ParseLogFile( orbitalsB );
117  if ( !_parse_logB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule B" << std::flush; }
118 
119  qmpackage->setLogFileName( _logAB );
120  bool _parse_logAB_status = qmpackage->ParseLogFile( orbitalsAB );
121  if ( !_parse_logAB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read log of molecule AB" << std::flush; }
122 
123  qmpackage->setOrbitalsFileName( _orbA );
124  bool _parse_orbitalsA_status = qmpackage->ParseOrbitalsFile( orbitalsA );
125  if ( !_parse_orbitalsA_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of molecule A" << std::flush; }
126 
127  qmpackage->setOrbitalsFileName( _orbB );
128  bool _parse_orbitalsB_status = qmpackage->ParseOrbitalsFile( orbitalsB );
129  if ( !_parse_orbitalsB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of molecule B" << std::flush; }
130 
131  qmpackage->setOrbitalsFileName( _orbAB );
132  bool _parse_orbitalsAB_status = qmpackage->ParseOrbitalsFile( orbitalsAB );
133  if ( !_parse_orbitalsAB_status ) { CTP_LOG(ctp::logERROR,_log) << "Failed to read orbitals of dimer AB" << std::flush; }
134 
135 
136 
137  DFTcoupling dftcoupling;
138  dftcoupling.setLogger(&_log);
139  dftcoupling.Initialize(_dftcoupling_options);
140 
141  dftcoupling.CalculateCouplings( orbitalsA, orbitalsB, orbitalsAB);
142  std::cout << _log;
143 
144 
145  // output the results
146  tools::Property summary;
147  tools::Property &job_output = summary.add("output","");
148  tools::Property &pair_summary = job_output.add("pair","");
149  dftcoupling.Addoutput(pair_summary,orbitalsA,orbitalsB);
150 
151  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
152 
153  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
154  ofs << job_output;
155  ofs.close();
156 
157  return true;
158 }
