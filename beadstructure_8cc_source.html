52 void BeadStructure::AddBead(BaseBead *bead) {
53  if (beads_.count(bead->getId())) {
54  string err = "Cannot add bead with Id ";
55  err += to_string(bead->getId());
56  err += " because each bead must have a unique Id and a bead with that Id ";

60  auto numberOfBeads = beads_.size();
61  beads_[bead->getId()] = bead;

85 void BeadStructure::InitializeGraph_() {
86  cerr << "Graph up to Date " << graphUpToDate << endl;
87  if (!graphUpToDate) {
88  vector<Edge> connections_vector;
89  for (auto edge : connections_) {
90  connections_vector.push_back(edge);
91  }
92 
93  unordered_map<int, GraphNode> graphnodes;
94  for (auto id_bead_ptr_pair : beads_) {
95  graphnodes_[id_bead_ptr_pair.first] =
96  BaseBeadToGraphNode(id_bead_ptr_pair.second);
97  graphnodes[id_bead_ptr_pair.first] =
98  *(graphnodes_[id_bead_ptr_pair.first]);
99  }
100  graph_ = make_shared<Graph>(Graph(connections_vector, graphnodes));
101  graphUpToDate = true;
102  }

114 bool BeadStructure::isSingleMolecule() {
115 
116  InitializeGraph_();
117  auto vertices = graph_->getVertices();
118  if (vertices.size() == 0)
119  return false;
120 
121  // Choose first vertex that is actually in the graph as the starting vertex
122  Graph_BF_Visitor gv_breadth_first;
123  gv_breadth_first.setStartingVertex(vertices.at(0));
124 
125  if (!singleNetwork(*graph_, gv_breadth_first))
126  return false;
127  if (beads_.size() == 0)
128  return false;
129  if (vertices.size() != beads_.size())
130  return false;
131  return true;
132 }

158 vector<shared_ptr<BeadStructure>> BeadStructure::breakIntoMolecules() {
159  vector<shared_ptr<BeadStructure>> structures;
160  if (!graphUpToDate)
161  InitializeGraph_();
162  auto sub_graphs = decoupleIsolatedSubGraphs(*(this->graph_));
163  for (auto sub_graph_it = sub_graphs.begin(); sub_graph_it != sub_graphs.end();
164  ++sub_graph_it) {
165  auto sub_graph_edges = (*sub_graph_it)->getEdges();
166  auto sub_graph_vertices = (*sub_graph_it)->getVertices();
167  BeadStructure beadstructure;
168  for (auto vertex : sub_graph_vertices)
169  beadstructure.AddBead(beads_[vertex]);
170  for (auto edge : sub_graph_edges)
171  beadstructure.ConnectBeads(edge.getV1(), edge.getV2());
172  structures.push_back(make_shared<BeadStructure>(beadstructure));
173  }
174  return structures;
175 }
