56 int main(int argc, char **argv) {
57  string top_file, trj_file, ptypes_file, out_file, grid, comment,
58  coordinate = "z";
59  double min, max, step, coord;
60 
61  votca::Index first_frame(0), last_frame(-1);
62 
63  // Load topology+trajectory formats
64  TopologyReader::RegisterPlugins();
65  TrajectoryReader::RegisterPlugins();
66 
67  // read program options
68  namespace po = boost::program_options;
69 
70  // Declare the supported options.
71  po::options_description desc("Allowed options");
72 
73  // let cg_engine add some program options
74  desc.add_options()("top", po::value<string>(&top_file), "topology file")(
75  "trj", po::value<string>(&trj_file), "trajectory file")(
76  "grid", po::value<string>(&grid), "output grid spacing (min:step:max)")(
77  "out", po::value<string>(&out_file),
78  "output particle distribution table")(
79  "ptypes", po::value<string>(&ptypes_file),
80  "particle types to include in the analysis\n"
81  " arg: file - particle types separated by space"
82  "\n default: all particle types")("first_frame",
83  po::value<votca::Index>(&first_frame),
84  "first frame considered for analysis")(
85  "last_frame", po::value<votca::Index>(&last_frame),
86  "last frame considered for analysis")(
87  "coord", po::value<string>(&coordinate),
88  "coordinate analyzed ('x', 'y', or 'z' (default))")(
89  "shift_com", "shift center of mass to zero")(
90  "comment", po::value<string>(&comment),
91  "store a comment in the output table")("help",
92  "produce this help message");
93 
94  // now read in the command line
95  po::variables_map vm;
96  try {
97  po::store(po::parse_command_line(argc, argv, desc), vm);
98  po::notify(vm);
99  } catch (po::error &err) {
100  cout << "error parsing command line: " << err.what() << endl;
101  return -1;
102  }
103 
104  // does the user want help?
105  if (vm.count("help")) {
106  help_text();
107  cout << desc << endl;
108  return 0;
109  }
110 
111  check_option(desc, vm, "top");
112  check_option(desc, vm, "trj");
113  check_option(desc, vm, "out");
114  check_option(desc, vm, "grid");
115 
116  if (coordinate.compare("x") != 0 && coordinate.compare("y") != 0 &&
117  coordinate.compare("z") != 0) {
118  cout << "Bad format for coordinate: " << coordinate << endl;
119  return 1;
120  }
121 
122  // Parse grid
123  Tokenizer tok(grid, ":");
124  vector<string> toks;
125  tok.ToVector(toks);
126  if (toks.size() != 3) {
127  cout << "Wrong range format, use min:step:max\n";
128  return 1;
129  }
130  min = std::stod(toks[0]);
131  step = std::stod(toks[1]);
132  max = std::stod(toks[2]);
133  // Calculate number of bins
134  votca::Index n_bins = (votca::Index)((max - min) / (1. * step) + 1);
135  vector<votca::Index> ptypes;
136  Topology top;
137  Eigen::MatrixXi p_occ;
138  votca::Index analyzed_frames(0);
139  try {
140 
141  // Load topology
142  std::unique_ptr<TopologyReader> reader = std::unique_ptr<TopologyReader>(
143  TopReaderFactory().Create(vm["top"].as<string>()));
144  if (reader == nullptr) {
145  throw std::runtime_error("input format not supported: " +
146  vm["top"].as<string>());
147  }
148 
149  reader->ReadTopology(vm["top"].as<string>(), top);
150 
151  // Read the particle types file and save to variable ptypes
152  if (vm.count("ptypes")) {
153  ifstream fl_ptypes;
154  fl_ptypes.open(vm["ptypes"].as<string>());
155  if (!fl_ptypes.is_open()) {
156  throw std::runtime_error("can't open " + vm["ptypes"].as<string>());
157  }
158  // build the list of particle types
159  if (fl_ptypes.eof()) {
160  throw std::runtime_error("file " + vm["ptypes"].as<string>() +
161  " is empty");
162  }
163  while (!fl_ptypes.eof()) {
164  // Not very elegant, but makes sure we don't count the same element
165  // twice
166  std::string string_tmp = "__";
167  fl_ptypes >> string_tmp;
168  if (string_tmp != "" && string_tmp != "__") {
169  ptypes.push_back(std::stol(string_tmp));
170  }
171  }
172  fl_ptypes.close();
173 
174  } else {
175  // Include all particle types
176  for (auto &mol : top.Molecules()) {
177  for (votca::Index i = 0; i < mol->BeadCount(); ++i) {
178  bool flag_found = false;
179  votca::Index part_type = std::stol(mol->getBead(i)->getType());
180  for (votca::Index ptype : ptypes) {
181  if (part_type == ptype) {
182  flag_found = true;
183  }
184  }
185  if (!flag_found) {
186  ptypes.push_back(part_type);
187  }
188  }
189  }
190  }
191 
192  p_occ = Eigen::MatrixXi::Zero(ptypes.size(), ptypes.size());
193 
194  // If we need to shift the center of mass, calculate the number of
195  // particles (only the ones that belong to the particle type index
196  // ptypes)
197  votca::Index n_part = 0;
198  if (vm.count("shift_com")) {
199  for (auto &mol : top.Molecules()) {
200  for (votca::Index i = 0; i < mol->BeadCount(); ++i) {
201  votca::Index part_type = std::stol(mol->getBead(i)->getType());
202  for (votca::Index ptype : ptypes) {
203  if (part_type == ptype) {
204  ++n_part;
205  }
206  }
207  }
208  }
209  }
210 
211  // Now load trajectory
212  std::unique_ptr<TrajectoryReader> trajreader =
213  std::unique_ptr<TrajectoryReader>(
214  TrjReaderFactory().Create(vm["trj"].as<string>()));
215  if (trajreader == nullptr) {
216  throw std::runtime_error("input format not supported: " +
217  vm["trj"].as<string>());
218  }
219  trajreader->Open(vm["trj"].as<string>());
220 
221  // Read the trajectory. Analyze each frame to obtain
222  // particle occupancy as a function of coordinate z.
223  bool moreframes = true;
224  bool not_the_last = true;
225  while (moreframes) {
226  // Read frame
227  votca::Index frame_id = 0;
228  if (frame_id == 0) {
229  moreframes = trajreader->FirstFrame(top);
230  } else {
231  moreframes = trajreader->NextFrame(top);
232  }
233 
234  // Was this the last frame we read?
235  if (last_frame == -1) {
236  not_the_last = 1;
237  } else {
238  if (frame_id <= last_frame) {
239  not_the_last = 1;
240  } else {
241  not_the_last = 0;
242  }
243  }
244 
245  // Calculate new center of mass position in the direction of 'coordinate'
246  double com = 0.;
247  if (vm.count("shift_com")) {
248  for (auto &mol : top.Molecules()) {
249  for (votca::Index i = 0; i < mol->BeadCount(); ++i) {
250  votca::Index part_type = std::stol(mol->getBead(i)->getType());
251  for (votca::Index ptype : ptypes) {
252  if (part_type == ptype) {
253  if (coordinate.compare("x") == 0) {
254  com += mol->getBead(i)->getPos().x();
255  } else if (coordinate.compare("y") == 0) {
256  com += mol->getBead(i)->getPos().y();
257  } else {
258  com += mol->getBead(i)->getPos().z();
259  }
260  }
261  }
262  }
263  }
264  com /= (double)n_part;
265  }
266 
267  // Analyze frame
268  if (moreframes && frame_id >= first_frame && not_the_last) {
269  ++analyzed_frames;
270  // Loop over each atom property
271  for (auto &mol : top.Molecules()) {
272  for (votca::Index i = 0; i < mol->BeadCount(); ++i) {
273  votca::Index part_type = std::stol(mol->getBead(i)->getType());
274  for (votca::Index j = 0; j < votca::Index(ptypes.size()); ++j) {
275  if (part_type == ptypes[j]) {
276  if (coordinate.compare("x") == 0) {
277  coord = mol->getBead(i)->getPos().x();
278  } else if (coordinate.compare("y") == 0) {
279  coord = mol->getBead(i)->getPos().y();
280  } else {
281  coord = mol->getBead(i)->getPos().z();
282  }
283 
284  if (coord - com > min && coord - com < max) {
285  p_occ(j,
286  (votca::Index)std::floor((coord - com - min) / step))++;
287  }
288  }
289  }
290  }
291  }
292  }
293  ++frame_id;
294  }
295 
296  trajreader->Close();
297 
298  } catch (std::exception &error) {
299  cerr << "An error occured!" << endl << error.what() << endl;
300  }
301 
302  // Output particle occupancy
303  try {
304  ofstream fl_out;
305  fl_out.open(vm["out"].as<string>());
306  if (!fl_out.is_open()) {
307  throw std::runtime_error("can't open " + vm["out"].as<string>());
308  }
309 
310  fl_out << "#z\t" << flush;
311  for (votca::Index ptype : ptypes) {
312  fl_out << "type " << ptype << "\t" << flush;
313  }
314  fl_out << endl;
315  for (votca::Index k = 0; k < n_bins; ++k) {
316  fl_out << min + (double)k * step << "\t" << flush;
317  for (votca::Index j = 0; j < votca::Index(ptypes.size()); ++j) {
318  if (p_occ(j, k) == 0) {
319  fl_out << 0 << "\t" << flush;
320  } else {
321  fl_out << p_occ(j, k) / (1. * (double)analyzed_frames) << "\t"
322  << flush;
323  }
324  }
325  fl_out << endl;
326  }
327  fl_out.close();
328  } catch (std::exception &error) {
329  cerr << "An error occured!" << endl << error.what() << endl;
330  }
331  cout << "The table was written to " << vm["out"].as<string>() << endl;
332 
333  return 0;
