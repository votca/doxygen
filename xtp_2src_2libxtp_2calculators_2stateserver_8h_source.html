60 void StateServer::Initialize(Property *opt) {
61

63  UpdateWithDefaults(opt, "xtp");
64 
65  string tag = "options." + Identify();
66 
67  // Tabular output
68  if (opt->exists(tag + ".out")) {
69  _outfile = opt->get(tag + ".out").as<string>();

71  _outfile = "stateinfo.out";

73  // PDB output
74  if (opt->exists(tag + ".pdb")) {

79

242  fprintf(out, " Time %2.3f \n", top->getTime());
243

247  int N_nbs = top->NBList().size();

249  fprintf(out, " # Molecules %7d \n", N_mol);

251  fprintf(out, " # Atoms %7d \n", N_atm);
252  fprintf(out, " # Pairs %7d \n", N_nbs);
253  return;

282 void StateServer::DownloadPairs(FILE *out, ctp::Topology *top) {
283  ctp::QMNBList::iterator nit;
284 
285  for (nit = top->NBList().begin(); nit != top->NBList().end(); nit++) {
286 
287  ctp::QMPair *pair = *nit;
288 
289  int ghost = (pair->HasGhost()) ? 1 : 0;
290 
291  fprintf(out,
292  "ID %5d SEG1 %4d %1s SEG2 %4d %1s dR %2.4f PBC %1d "
293  "dE(-1) %4.7f dE(+1) %4.7f "
294  "L(-1) %1.4f L(+1) %1.4f J2(-1) %1.8e J2(+1) %1.8e "
295  "R12(-1) %2.4e R12(+1) %2.4e R21(-1) %2.4e R21(-1) %2.4e "
296  "dE(+2) %4.7f dE(+3) %4.7f "
297  "L(+2) %1.4f L(+3) %1.4f J2(+2) %1.8e J2(+3) %1.8e "
298  "R12(+2) %2.4e R12(+3) %2.4e R21(+2) %2.4e R21(+3) %2.4e\n",
299  pair->getId(), pair->first->getId(), pair->first->getName().c_str(),
300  pair->second->getId(), pair->second->getName().c_str(),
301  pair->Dist(), ghost, pair->getdE12(-1), pair->getdE12(+1),
302  pair->getLambdaO(-1), pair->getLambdaO(+1), pair->getJeff2(-1),
303  pair->getJeff2(+1), pair->getRate12(-1), pair->getRate12(+1),
304  pair->getRate21(-1), pair->getRate21(+1), pair->getdE12(+2),
305  pair->getdE12(+3), pair->getLambdaO(+2), pair->getLambdaO(+3),
306  pair->getJeff2(+2), pair->getJeff2(+3), pair->getRate12(+2),
307  pair->getRate12(+3), pair->getRate21(+2), pair->getRate21(+3));
308  }
309  return;
