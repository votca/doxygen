63 void Coupling::Initialize(Property* options)
64 {
65 
66  // update options with the VOTCASHARE defaults
67  UpdateWithDefaults( options, "ctp" );
68  std::string key = "options." + Identify();
69 
70  _degeneracy = options->get(key + ".degeneracy").as<double> ();
71  _package = options->get(key + ".package").as<string> ();
72 
73  _orbA = options->get(key + ".moleculeA.orbitals").as<string> ();
74  _orbB = options->get(key + ".moleculeB.orbitals").as<string> ();
75  _orbAB = options->get(key + ".dimerAB.orbitals").as<string> ();
76 
77  _logA = options->get(key + ".moleculeA.log").as<string> ();
78  _logB = options->get(key + ".moleculeB.log").as<string> ();
79  _logAB = options->get(key + ".dimerAB.log").as<string> ();
80 
81  _levA = options->get(key + ".moleculeA.levels").as<int> ();
82  _levB = options->get(key + ".moleculeB.levels").as<int> ();
83 
84  _trimA = options->get(key + ".moleculeA.trim").as<int> ();
85  _trimB = options->get(key + ".moleculeB.trim").as<int> ();
86 
87  _output_file = options->get(key + ".output").as<string> ();
88 
89  // get the path to the shared folders with xml files
90  char *votca_share = getenv("VOTCASHARE");
91  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
92  string xmlFile = string(getenv("VOTCASHARE")) + string("/ctp/qmpackages/") + _package + string("_idft_pair.xml");
93  load_property_from_xml( _package_options, xmlFile );
94 
95  // register all QM packages (Gaussian, TURBOMOLE, etc)
96  QMPackageFactory::RegisterAll();

100 bool Coupling::Evaluate() {
101 
102  _log.setReportLevel( logDEBUG );
103  _log.setMultithreading( true );
104 
105  _log.setPreface(logINFO, "\n... ...");
106  _log.setPreface(logERROR, "\n... ...");
107  _log.setPreface(logWARNING, "\n... ...");
108  _log.setPreface(logDEBUG, "\n... ...");
109

113  _qmpackage->Initialize( &_package_options );
114 
115  Orbitals _orbitalsA, _orbitalsB, _orbitalsAB;
116 
117  _qmpackage->setOrbitalsFileName( _orbA );
118  //int _parse_orbitalsA_status =
119  (void)_qmpackage->ParseOrbitalsFile( &_orbitalsA );
120 
121  _qmpackage->setOrbitalsFileName( _orbB );
122  //int _parse_orbitalsB_status =
123  (void)_qmpackage->ParseOrbitalsFile( &_orbitalsB );
124 
125  _qmpackage->setOrbitalsFileName( _orbAB );
126  //int _parse_orbitalsAB_status =
127  (void)_qmpackage->ParseOrbitalsFile( &_orbitalsAB );
128 
129  _qmpackage->setLogFileName( _logA );
130  //int _parse_logA_status =
131  (void)_qmpackage->ParseLogFile( &_orbitalsA );
132 
133  _qmpackage->setLogFileName( _logB );
134  //int _parse_logB_status =
135  (void)_qmpackage->ParseLogFile( &_orbitalsB );
136 
137  _qmpackage->setLogFileName( _logAB );
138  //int _parse_logAB_status =
139  (void)_qmpackage->ParseLogFile( &_orbitalsAB );
140 
141  if ( _orbitalsA.getNumberOfElectrons()*(_trimA-1) < _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() ){
142  LOG(logDEBUG,_log) << "Trimming virtual orbitals A:"
143  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
144  << _orbitalsA.getNumberOfElectrons()*(_trimA-1) << std::flush;
145  _orbitalsA.Trim(_trimA);
146  }
147 
148  if ( _orbitalsB.getNumberOfElectrons()*(_trimB-1) < _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() ){
149  LOG(logDEBUG,_log) << "Trimming virtual orbitals B:"
150  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
151  << _orbitalsB.getNumberOfElectrons()*(_trimB-1) << std::flush;
152  _orbitalsB.Trim(_trimB);
153  }
154  Overlap _overlap;
155  _overlap.setLogger(&_log);
156 
157  ub::matrix<double> _JAB;
158  //bool _calculate_integrals =
159  (void)_overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
160  std::cout << _log;
161 
162 
163  // output the results
164  Property _summary;
165  Property *_job_output = &_summary.add("output","");
166  Property *_pair_summary = &_job_output->add("pair","");
167  int HOMO_A = _orbitalsA.getNumberOfElectrons();
168  int HOMO_B = _orbitalsB.getNumberOfElectrons();
169  int LUMO_A = HOMO_A + 1;
170  int LUMO_B = HOMO_B + 1;
171  _pair_summary->setAttribute("homoA", HOMO_A);
172  _pair_summary->setAttribute("homoB", HOMO_B);
173  for (int levelA = HOMO_A - _levA +1; levelA <= LUMO_A + _levA - 1; ++levelA ) {
174  for (int levelB = HOMO_B - _levB + 1; levelB <= LUMO_B + _levB -1 ; ++levelB ) {
175  double JAB = _overlap.getCouplingElement( levelA , levelB, &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy );
176  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
177  double energyA = _orbitalsA.getEnergy( levelA );
178  double energyB = _orbitalsB.getEnergy( levelB );
179  _overlap_summary->setAttribute("orbA", levelA);
180  _overlap_summary->setAttribute("orbB", levelB);
181  //_overlap_summary->setAttribute("jAB", JAB);
182  _overlap_summary->setAttribute("eA", energyA);
183  _overlap_summary->setAttribute("eB", energyB);
184  }
185  }
186 
187  votca::tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 1, "");
188 
189  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
190  ofs << *_job_output;
191  ofs.close();
192 
193  return true;
194 }
