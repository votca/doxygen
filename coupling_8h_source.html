63 void Coupling::Initialize(Property* options)
64 {
65 
66  // update options with the VOTCASHARE defaults
67  UpdateWithDefaults( options, "xtp" );
68  std::string key = "options." + Identify();
69 
70  _degeneracy = options->get(key + ".degeneracy").as<double> ();
71  _package = options->get(key + ".dftpackage").as<string> ();
72 
73  _orbA = options->get(key + ".moleculeA.orbitals").as<string> ();
74  _orbB = options->get(key + ".moleculeB.orbitals").as<string> ();
75  _orbAB = options->get(key + ".dimerAB.orbitals").as<string> ();
76 
77  _logA = options->get(key + ".moleculeA.log").as<string> ();
78  _logB = options->get(key + ".moleculeB.log").as<string> ();
79  _logAB = options->get(key + ".dimerAB.log").as<string> ();
80 
81  _levA = options->get(key + ".moleculeA.levels").as<int> ();
82  _levB = options->get(key + ".moleculeB.levels").as<int> ();
83  if ( options->exists(key+".moleculeA.trim")){
84  _trimA = options->get(key + ".moleculeA.trim").as<int> ();
85  }
86  else{
87  _trimA=-1;
88  }
89  if ( options->exists(key+".moleculeB.trim")){
90  _trimB = options->get(key + ".moleculeB.trim").as<int> ();
91  }
92  else{
93  _trimB=-1;
94  }
95 
96  _output_file = options->get(key + ".output").as<string> ();
97 
98  // get the path to the shared folders with xml files
99  char *votca_share = getenv("VOTCASHARE");
100  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
101  string xmlFile = string(getenv("VOTCASHARE")) + string("/xtp/packages/") + _package + string("_idft_pair.xml");
102  load_property_from_xml( _package_options, xmlFile );
103 
104  // register all QM packages (Gaussian, TURBOMOLE, etc)
105  QMPackageFactory::RegisterAll();
106 
107 }

109 bool Coupling::Evaluate() {
110 
111  _log.setReportLevel( logDEBUG );
112  _log.setMultithreading( true );
113 
114  _log.setPreface(logINFO, "\n... ...");
115  _log.setPreface(logERROR, "\n... ...");
116  _log.setPreface(logWARNING, "\n... ...");
117  _log.setPreface(logDEBUG, "\n... ...");
118 
119  // get the corresponding object from the QMPackageFactory
120  QMPackage *_qmpackage = QMPackages().Create( _package );
121  _qmpackage->setLog( &_log );
122  _qmpackage->Initialize( &_package_options );
123 
124  Orbitals _orbitalsA, _orbitalsB, _orbitalsAB;
125 
126  _qmpackage->setOrbitalsFileName( _orbA );
127  //int _parse_orbitalsA_status = _qmpackage->ParseOrbitalsFile( &_orbitalsA );
128  _qmpackage->ParseOrbitalsFile( &_orbitalsA );
129 
130  _qmpackage->setOrbitalsFileName( _orbB );
131  //int _parse_orbitalsB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsB );
132  _qmpackage->ParseOrbitalsFile( &_orbitalsB );
133 
134  _qmpackage->setOrbitalsFileName( _orbAB );
135  //int _parse_orbitalsAB_status = _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
136  _qmpackage->ParseOrbitalsFile( &_orbitalsAB );
137 
138  _qmpackage->setLogFileName( _logA );
139  //int _parse_logA_status = _qmpackage->ParseLogFile( &_orbitalsA );
140  _qmpackage->ParseLogFile( &_orbitalsA );
141 
142  _qmpackage->setLogFileName( _logB );
143  //int _parse_logB_status = _qmpackage->ParseLogFile( &_orbitalsB );
144  _qmpackage->ParseLogFile( &_orbitalsB );
145 
146  _qmpackage->setLogFileName( _logAB );
147  //int _parse_logAB_status = _qmpackage->ParseLogFile( &_orbitalsAB );
148  _qmpackage->ParseLogFile( &_orbitalsAB );
149 
150 
151  // trim monomer A to minimal?
152  int _degAH = 1;
153  int _degAL = 1;
154  int _degBH = 1;
155  int _degBL = 1;
156 
157  if ((_trimA == -1) || (_trimB == -1) ) { // any -1 overrides the specification of the other
158 
159 
160  // find degeneracy of HOMOs and LUMOs
161  std::vector<int> list_levelsAH = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons()-1, _degeneracy ));
162  _degAH = list_levelsAH.size();
163  std::vector<int> list_levelsAL = (*_orbitalsA.getDegeneracy( _orbitalsA.getNumberOfElectrons(), _degeneracy ));
164  _degAL = list_levelsAL.size();
165 
166  std::vector<int> list_levelsBH = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons()-1, _degeneracy ));
167  _degBH = list_levelsBH.size();
168  std::vector<int> list_levelsBL = (*_orbitalsB.getDegeneracy( _orbitalsB.getNumberOfElectrons(), _degeneracy ));
169  _degBL = list_levelsBL.size();
170 
171  LOG(logDEBUG,_log) << "Trimming orbitals to minimal set " << flush;
172  LOG(logDEBUG,_log) << "HOMO(A)-" << _degAH << " to " << "LUMO(A)+" << _degAL << flush;
173  _orbitalsA.Trim(_degAH,_degAL);
174  LOG(logDEBUG,_log) << "HOMO(B)-" << _degBH << " to " << "LUMO(B)+" << _degBL << flush;
175  _orbitalsB.Trim(_degBH,_degBL);
176 
177 
178 
179  } else {
180  // standard trim
181 
182  if (_orbitalsA.getNumberOfElectrons()*(_trimA - 1) < _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons()) {
183 
184 
185  LOG(logDEBUG, _log) << "Trimming virtual orbitals A:"
186  << _orbitalsA.getNumberOfLevels() - _orbitalsA.getNumberOfElectrons() << "->"
187  << _orbitalsA.getNumberOfElectrons()*(_trimA - 1) << std::flush;
188  _orbitalsA.Trim(_trimA);
189  }
190  LOG(logDEBUG, _log) << "DONE!" << std::flush;
191  if (_orbitalsB.getNumberOfElectrons()*(_trimB - 1) < _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons()) {
192 
193  LOG(logDEBUG, _log) << "Trimming virtual orbitals B:"
194  << _orbitalsB.getNumberOfLevels() - _orbitalsB.getNumberOfElectrons() << "->"
195  << _orbitalsB.getNumberOfElectrons()*(_trimB - 1) << std::flush;
196  _orbitalsB.Trim(_trimB);
197  }
198 
199 
200 
201  }
202  Overlap _overlap;
203  _overlap.setLogger(&_log);
204 
205  ub::matrix<double> _JAB;
206  //bool _calculate_integrals = _overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
207  _overlap.CalculateIntegrals( &_orbitalsA, &_orbitalsB, &_orbitalsAB, &_JAB );
208  std::cout << _log;
209 
210 
211 
212  // output the results
213  Property _summary;
214  Property *_job_output = &_summary.add("output","");
215  Property *_pair_summary = &_job_output->add("pair","");
216 
217  int HOMO_A = _orbitalsA.getNumberOfElectrons();
218  int HOMO_B = _orbitalsB.getNumberOfElectrons();
219  int LUMO_A = HOMO_A + 1;
220  int LUMO_B = HOMO_B + 1;
221  _pair_summary->setAttribute("homoA", HOMO_A);
222  _pair_summary->setAttribute("homoB", HOMO_B);
223 
224  if ( (_trimA == -1) || (_trimB == -1) ) {
225 
226  // HOMO-HOMO coupling
227  double JAB = _overlap.getCouplingElement(_degAH, _degBH , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
228  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
229  double energyA = _orbitalsA.getEnergy(_degAH);
230  double energyB = _orbitalsB.getEnergy(_degBH);
231  _overlap_summary->setAttribute("orbA", HOMO_A);
232  _overlap_summary->setAttribute("orbB", HOMO_B);
233  //_overlap_summary->setAttribute("jAB", JAB);
234  _overlap_summary->setAttribute("eA", energyA);
235  _overlap_summary->setAttribute("eB", energyB);
236 
237  // LUMO-LUMO coupling
238  JAB = _overlap.getCouplingElement(_degAH+1, _degBH+1 , &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
239  _overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
240  energyA = _orbitalsA.getEnergy(_degAH +1);
241  energyB = _orbitalsB.getEnergy(_degBH +1);
242  _overlap_summary->setAttribute("orbA", LUMO_A);
243  _overlap_summary->setAttribute("orbB", LUMO_B);
244  //_overlap_summary->setAttribute("jAB", JAB);
245  _overlap_summary->setAttribute("eA", energyA);
246  _overlap_summary->setAttribute("eB", energyB);
247 
248  } else {
249 
250  for (int levelA = HOMO_A - _levA + 1; levelA <= LUMO_A + _levA - 1; ++levelA) {
251  for (int levelB = HOMO_B - _levB + 1; levelB <= LUMO_B + _levB - 1; ++levelB) {
252  double JAB = _overlap.getCouplingElement(levelA, levelB, &_orbitalsA, &_orbitalsB, &_JAB, _degeneracy);
253  Property *_overlap_summary = &_pair_summary->add("overlap", boost::lexical_cast<string>(JAB));
254  double energyA = _orbitalsA.getEnergy(levelA);
255  double energyB = _orbitalsB.getEnergy(levelB);
256  _overlap_summary->setAttribute("orbA", levelA);
257  _overlap_summary->setAttribute("orbB", levelB);
258  //_overlap_summary->setAttribute("jAB", JAB);
259  _overlap_summary->setAttribute("eA", energyA);
260  _overlap_summary->setAttribute("eB", energyB);
261  }
262  }
263 
264  }
265 
266  votca::tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 1, "");
267 
268  std::ofstream ofs (_output_file.c_str(), std::ofstream::out);
269  ofs << *_job_output;
270  ofs.close();
271 
272  return true;
273 }
