97 string Job::ConvertStatus(JobStatus status) const {
98 
99  string converted;
100  switch(status) {
101  case AVAILABLE: converted = "AVAILABLE"; break;
102  case ASSIGNED: converted = "ASSIGNED"; break;
103  case FAILED: converted = "FAILED"; break;
104  case COMPLETE: converted = "COMPLETE"; break;
105  default: throw runtime_error("Incomprehensible status (enum)");
106  }
107  return converted;
108 }

111 Job::JobStatus Job::ConvertStatus(string status) const {
112  JobStatus converted;

131 void Job::ToStream(ofstream &ofs, string fileformat) {
132 
133  votca::tools::PropertyIOManipulator iomXML(votca::tools::PropertyIOManipulator::XML, 0, "\t\t");
134 
135  if (fileformat == "xml") {
136  string tab = "\t";
137 
138  ofs << tab << "<job>\n";
139  ofs << tab << tab << (format("<id>%1$d</id>\n") % _id).str();
140  ofs << tab << tab << (format("<tag>%1$s</tag>\n") % _tag).str();
141  //PropertyFormat::PrintNodeXML(ofs, _input, 0, 0, "", "\t\t");
142  ofs << iomXML << _input;
143  //ofs << tab << tab << (format("<input>%1$s</input>\n") % _input).str();
144  ofs << tab << tab << (format("<status>%1$s</status>\n") % ConvertStatus(_status)).str();
145 
146  if (_has_sqlcmd)
147  ofs << tab << tab << (format("<sqlcmd>%1$s</sqlcmd>\n")
148  % _sqlcmd).str();
149  if (_has_host)
150  ofs << tab << tab << (format("<host>%1$s</host>\n")
151  % _host).str();
152  if (_has_time)
153  ofs << tab << tab << (format("<time>%1$s</time>\n")
154  % _time).str();
155  if (_has_output)
156  //PropertyFormat::PrintNodeXML(ofs, _output, 0, 0, "", "\t\t");
157  ofs << iomXML << _output;
158  if (_has_error)
159  ofs << tab << tab << (format("<error>%1$s</error>\n")
160  % _error).str();
161  ofs << tab << "</job>\n";
162  }
163  else if (fileformat == "tab") {
164  string time = _time;
165  if (!_has_time) time = "__:__";
166  string host = _host;
167  if (!_has_host) host = "__:__";
168  string status = ConvertStatus(_status);
169  ofs << (format("%4$10s %5$20s %6$10s %1$5d %2$10s %3$30s %7$s %8$s\n")
170  % _id % _tag % _input % status % host
171  % time % _error % _output).str();
172  }
173  else {
174  assert(false);
175  }
176 
177  return;
178 }
