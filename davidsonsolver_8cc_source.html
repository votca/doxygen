135 Eigen::VectorXd DavidsonSolver::olsen_correction(Eigen::VectorXd &r,
136  Eigen::VectorXd &x,
137  Eigen::VectorXd &D,
138  double lambda) const {
139  /* \brief Compute the olsen correction :
140 
141  \delta = (D-\lambda)^{-1} (-r + \epsilon x)
142 
143  */
144  int size = r.rows();
145  Eigen::VectorXd delta = Eigen::VectorXd::Zero(size);
146  delta = DavidsonSolver::dpr_correction(r, D, lambda);
147  double num = -x.transpose() * delta;
148  double denom = -x.transpose() * DavidsonSolver::dpr_correction(x, D, lambda);
149  double eps = num / denom;
150  delta += eps * x;
151  return delta;
152 }

154 Eigen::MatrixXd DavidsonSolver::QR_ortho(const Eigen::MatrixXd &A) const {
155 
156  int nrows = A.rows();
157  int ncols = A.cols();
158  ncols = std::min(nrows, ncols);
159 
160  Eigen::HouseholderQR<Eigen::MatrixXd> qr(A);
161  Eigen::MatrixXd result =
162  qr.householderQ() * Eigen::MatrixXd::Identity(nrows, ncols);
163  return result;
164 }

166 Eigen::MatrixXd DavidsonSolver::gramschmidt_ortho(const Eigen::MatrixXd &A,
167  int nstart) const {
168  Eigen::MatrixXd Q = A;
169  for (int j = nstart; j < A.cols(); ++j) {
170  Q.col(j) -= Q.leftCols(j) * (Q.leftCols(j).transpose() * A.col(j));
171  if (Q.col(j).norm() <= 1E-12 * A.col(j).norm()) {
172  throw std::runtime_error(
173  "Linear dependencies in Gram-Schmidt. Switch to QR");
174  }
175  Q.col(j).normalize();
176  }
177  return Q;
178 }
