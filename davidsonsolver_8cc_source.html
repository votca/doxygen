182 Eigen::VectorXd DavidsonSolver::dpr_correction(const Eigen::VectorXd &r,
183  const Eigen::VectorXd &D,
184  double lambda) const {
185  /* \brief Compute the diagonal preconditoned residue : delta = - (D -
186  * lambda)^{-1} r
187  */
188  Eigen::VectorXd delta = r.array() / (lambda - D.array());
189  return delta;
190 }

192 Eigen::VectorXd DavidsonSolver::olsen_correction(const Eigen::VectorXd &r,
193  const Eigen::VectorXd &x,
194  const Eigen::VectorXd &D,
195  double lambda) const {
196  /* \brief Compute the olsen correction :
197 
198  \delta = (D-\lambda)^{-1} (-r + \epsilon x)
199 
200  */
201  int size = r.rows();
202  Eigen::VectorXd delta = Eigen::VectorXd::Zero(size);
203  delta = DavidsonSolver::dpr_correction(r, D, lambda);
204  double num = -x.transpose() * delta;
205  double denom = -x.transpose() * dpr_correction(x, D, lambda);
206  double eps = num / denom;
207  delta += eps * x;
208  return delta;
209 }

211 Eigen::MatrixXd DavidsonSolver::QR_ortho(const Eigen::MatrixXd &A) const {
212 
213  int nrows = A.rows();
214  int ncols = A.cols();
215  ncols = std::min(nrows, ncols);
216 
217  Eigen::HouseholderQR<Eigen::MatrixXd> qr(A);
218  Eigen::MatrixXd result = qr.householderQ();
219  result.conservativeResize(nrows, ncols);
220  return result;
221 }

223 Eigen::MatrixXd DavidsonSolver::gramschmidt_ortho(const Eigen::MatrixXd &A,
224  int nstart) {
225  Eigen::MatrixXd Q = A;
226  for (int j = nstart; j < A.cols(); ++j) {
227  Q.col(j) -= Q.leftCols(j) * (Q.leftCols(j).transpose() * A.col(j));
228  if (Q.col(j).norm() <= 1E-12 * A.col(j).norm()) {
229  _info = Eigen::ComputationInfo::NumericalIssue;
230  throw std::runtime_error(
231  "Linear dependencies in Gram-Schmidt. Switch to QR");
232  }
233  Q.col(j).normalize();
234  }
235  return Q;
236 }
