183 Index DavidsonSolver::getSizeUpdate(Index neigen) const {
184  Index size_update;
185  switch (this->_davidson_update) {
186  case UPDATE::MIN:
187  size_update = neigen;
188  break;
189  case UPDATE::SAFE:
190  if (neigen < 20) {
191  size_update = static_cast<Index>(1.5 * double(neigen));
192  } else {
193  size_update = neigen + 10;
194  }
195  break;
196  case UPDATE::MAX:
197  size_update = 2 * neigen;
198  break;
199  default:
200  size_update = 2 * neigen;
201  break;
202  }
203  return size_update;
204 }

214 Eigen::MatrixXd DavidsonSolver::setupInitialEigenvectors(
215  Index size_initial_guess) const {
216 
217  Eigen::MatrixXd guess =
218  Eigen::MatrixXd::Zero(_Adiag.size(), size_initial_guess);
219  ArrayXl idx = DavidsonSolver::argsort(_Adiag);
220 
221  switch (this->_matrix_type) {
222  case MATRIX_TYPE::SYMM:
223  /* \brief Initialize the guess eigenvector so that they 'target' the
224  * smallest diagonal elements */
225  for (Index j = 0; j < size_initial_guess; j++) {
226  guess(idx(j), j) = 1.0;
227  }
228  break;
229 
230  case MATRIX_TYPE::HAM:
231  /* Initialize the guess eigenvector so that they 'target' the lowest
232  * positive diagonal elements */
233  Index ind0 = _Adiag.size() / 2;
234  for (Index j = 0; j < size_initial_guess; j++) {
235  guess(idx(ind0 + j), j) = 1.0;
236  }
237  break;
238  }
239  return guess;
240 }

242 DavidsonSolver::RitzEigenPair DavidsonSolver::getRitz(
243  const DavidsonSolver::ProjectedSpace &proj) const {
244 
245  DavidsonSolver::RitzEigenPair rep;
246  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(proj.T);
247  rep.lambda = es.eigenvalues();
248  rep.U = es.eigenvectors();
249 
250  rep.q = proj.V * rep.U; // Ritz vectors
251  rep.res = proj.AV * rep.U - rep.q * rep.lambda.asDiagonal(); // residues
252  return rep;
253 }

269 bool DavidsonSolver::checkConvergence(const DavidsonSolver::RitzEigenPair &rep,
270  DavidsonSolver::ProjectedSpace &proj,
271  Index neigen) {
272 
273  Eigen::ArrayXd res_norm = rep.res_norm();
274  bool converged = true;
275  for (Index j = 0; j < proj.size_update; j++) {
276  proj.root_converged[j] = (res_norm[j] < _tol);
277  if (j < neigen) {
278  converged &= (res_norm[j] < _tol);
279  }
280  }
281  return converged;
282 }

284 void DavidsonSolver::extendProjection(const DavidsonSolver::RitzEigenPair &rep,
285  DavidsonSolver::ProjectedSpace &proj) {
286 
287  Index nupdate = 0;
288  for (Index j = 0; j < proj.size_update; j++) {
289 
290  // skip the roots that have already converged
291  if (proj.root_converged[j]) {
292  continue;
293  }
294 
295  nupdate++;
296 
297  // residue vector
298  Eigen::VectorXd w =
299  computeCorrectionVector(rep.q.col(j), rep.lambda(j), rep.res.col(j));
300 
301  // append the correction vector to the search space
302  proj.V.conservativeResize(Eigen::NoChange, proj.V.cols() + 1);
303  proj.V.rightCols<1>() = w.normalized();
304  }
305 
306  proj.V = orthogonalize(proj.V, nupdate);
307 }

318 Eigen::VectorXd DavidsonSolver::computeCorrectionVector(
319  const Eigen::VectorXd &qj, double lambdaj,
320  const Eigen::VectorXd &Aqj) const {
321 
322  /* compute correction vector with either DPR or OLSEN CORRECTION
323  * For details on the method see :
324  * Systematic Study of Selected Diagonalization Methods
325  * for Configuration Interaction Matrices
326  * M.L. Leininger et al .
327  * Journal of Computational Chemistry Vol 22, No. 13 1574-1589 (2001)
328  */
329 
330  switch (this->_davidson_correction) {
331  case CORR::DPR: {
332  return dpr(Aqj, lambdaj);
333  }
334  case CORR::OLSEN: {
335  return olsen(Aqj, qj, lambdaj);
336  }
337  }
338  return Eigen::VectorXd::Zero(0);
339 }

341 Eigen::VectorXd DavidsonSolver::dpr(const Eigen::VectorXd &r,
342  double lambda) const {
343  /* \brief Compute the diagonal preconditoned residue :
344  \delta = -r/(D - lambda)
345  */
346  return (-r.array() / (_Adiag.array() - lambda));
347 }

349 Eigen::VectorXd DavidsonSolver::olsen(const Eigen::VectorXd &r,
350  const Eigen::VectorXd &x,
351  double lambda) const {
352  /* \brief Compute the olsen correction :
353  \delta = (D-\lambda)^{-1} (-r + \epsilon x)
354  */
355  Eigen::VectorXd delta = DavidsonSolver::dpr(r, lambda);
356  double num = -x.transpose() * delta;
357  double denom = -x.transpose() * dpr(x, lambda);
358  double eps = num / denom;
359  delta += eps * x;
360  return delta;
361 }

376 Eigen::MatrixXd DavidsonSolver::qr(const Eigen::MatrixXd &A) const {
377 
378  Index nrows = A.rows();
379  Index ncols = A.cols();
380  ncols = std::min(nrows, ncols);
381  Eigen::MatrixXd I = Eigen::MatrixXd::Identity(nrows, ncols);
382  Eigen::HouseholderQR<Eigen::MatrixXd> qr(A);
383  return qr.householderQ() * I;
384 }

386 Eigen::MatrixXd DavidsonSolver::gramschmidt(const Eigen::MatrixXd &A,
387  Index nstart) {
388  Eigen::MatrixXd Q = A;
389  for (Index j = nstart; j < A.cols(); ++j) {
390  Q.col(j) -= Q.leftCols(j) * (Q.leftCols(j).transpose() * A.col(j));
391  Q.col(j).normalize();
392  // two is enough GS
393  // http://stoppels.blog/posts/orthogonalization-performance
394  Q.col(j) -= Q.leftCols(j) * (Q.leftCols(j).transpose() * Q.col(j));
395  if (Q.col(j).norm() <= 1E-12 * A.col(j).norm()) {
396  _info = Eigen::ComputationInfo::NumericalIssue;
397  throw std::runtime_error(
398  "Linear dependencies in Gram-Schmidt. Switch to QR");
399  }
400  Q.col(j).normalize();
401  }
402  return Q;
403 }

406  DavidsonSolver::ProjectedSpace &proj,
407  Index size_restart) const {
