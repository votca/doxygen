213 template<class Norm, class V>
214 inline bool VectorSort<Norm,V>::operator() (const V &v1,
215  const V &v2) {
216  bool smaller = false;
217  // LEVEL 1: MAGNITUDE
218  double V1 = _norm(v1);
219  double V2 = _norm(v2);
220  if (MatchDouble(V1,V2)) {
221  // LEVEL 2: X
222  double X1 = v1.getX();
223  double X2 = v2.getX();
224  if (MatchDouble(X1,X2)) {
225  // LEVEL 3: Y
226  double Y1 = v1.getY();
227  double Y2 = v2.getY();
228  if (MatchDouble(Y1,Y2)) {
229  // LEVEL 4: Z
230  double Z1 = v1.getZ();
231  double Z2 = v2.getZ();
232  if (MatchDouble(Z1,Z2)) smaller = true;
233  else smaller = (Z1 < Z2) ? true : false;
234  }
235  else smaller = (Y1 < Y2) ? true : false;
236  }
237  else smaller = (X1 < X2) ? true : false;
238  }
239  else smaller = (V1 < V2) ? true : false;
240  return smaller;
241 }
