25 void HistogramNew::Initialize_() {
26  if (_periodic) {
27  _step = (_max - _min) / double(_nbins);
28  } else {
29  _step = (_max - _min) / (double(_nbins) - 1.0);
30  }
31 
32  if (_nbins == 1) {
33  _step = 1;
34  }
35  _data.resize(_nbins);
36  double v = _min;
37  for (Index i = 0; i < _nbins; v += _step, ++i) {
38  _data.x(i) = v;
39  }
40  _data.y() = Eigen::VectorXd::Zero(_nbins);
41  _data.yerr() = Eigen::VectorXd::Zero(_nbins);
42  _data.flags() = std::vector<char>(_nbins, 'i');
43 }

52 void HistogramNew::Process(const double &v, double scale) {
53 
54  Index i = (Index)floor((v - _min) / _step + 0.5);
55  if (i < 0 || i >= _nbins) {
56  if (_periodic) {
57  if (i < 0) {
58  i = _nbins - ((-i) % _nbins);
59  } else {
60  i = i % _nbins;
61  }
62  } else {
63  return;
64  }
65  }
66  _data.y(i) += scale;
67 }
