28 void HistogramNew::Initialize_() {
29  if (_periodic) {
30  _step = (_max - _min) / double(_nbins);
31  } else {
32  _step = (_max - _min) / (double(_nbins) - 1.0);
33  }
34 
35  if (_nbins == 1) {
36  _step = 1;
37  }
38  _data.resize(_nbins);
39  double v = _min;
40  for (Index i = 0; i < _nbins; v += _step, ++i) {
41  _data.x(i) = v;
42  }
43  _data.y() = Eigen::VectorXd::Zero(_nbins);
44  _data.yerr() = Eigen::VectorXd::Zero(_nbins);
45  _data.flags() = std::vector<char>(_nbins, 'i');
46 }

55 void HistogramNew::Process(const double &v, double scale) {
56 
57  Index i = (Index)floor((v - _min) / _step + 0.5);
58  if (i < 0 || i >= _nbins) {
59  if (_periodic) {
60  if (i < 0) {
61  i = _nbins - ((-i) % _nbins);
62  } else {
63  i = i % _nbins;
64  }
65  } else {
66  return;
67  }
68  }
69  _data.y(i) += scale;
70 }
