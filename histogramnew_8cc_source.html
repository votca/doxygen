34 void HistogramNew::Initialize(double min, double max, int nbins)
35 {
36  _min = min; _max = max;
37  _step = (_max - _min)/nbins;
38  _weight = 1.;
39  _data.resize(nbins);
40  _nbins = nbins;
41 
42  for(double v=_min, i=0; i<nbins; v+=_step,++i)
43  _data.x(i)=v;
44 
45  _data.y()=ub::zero_vector<double>(_nbins);
46  _data.yerr()=ub::zero_vector<double>(_nbins);
47  _data.flags()=ub::scalar_vector<char>(_nbins, 'i');
48 }
49 
50 void HistogramNew::Process(const double &v, double scale)
51 {
52  int i = (int)floor((v - _min) / _step + 0.5);
53 
54  if (i < 0 || i >= _nbins) {
55  if(!_periodic) return;
56  if(i<0) i = _nbins - ((-i) % _nbins);
57  else i = i % _nbins;
58  }
59  _data.y(i) += _weight * scale;
60 }

62 void HistogramNew::Normalize()

66  area=ub::norm_1(_data.y()) * _step;
67 
68  double scale = 1./area;

70  _data.y() *= scale;
