78 void HistogramNew::ProcessP_(const double &v, double scale) {
79  int i = (int)floor((v - _min) / _step_p + 0.5);
80  if (i < 0 || i >= _nbins) {
81  if (i < 0)
82  i = _nbins - ((-i) % _nbins);

84  i = i % _nbins;

86  _data_p.y(i) += _weight * scale;

104  double value = static_cast<double>(bin);
105  if (_periodic) {
106  bounds.first = value * _step_p + _min;
