00034 void HistogramNew::Initialize(double min, double max, int nbins)
00035 {
00036     _min = min; _max = max;
00037     _step = (_max - _min)/nbins;
00038     _weight = 1.;
00039     _data.resize(nbins);  
00040     _nbins = nbins;
00041

00050 void HistogramNew::Process(const double &v, double scale)
00051 {
00052     int i = (int) ((v - _min) / _step + 0.5);
00053     
00054     if (i < 0 || i >= _nbins) {
00055         if(!_periodic) return;
00056         if(i<0) i = _nbins - ((-i) % _nbins);
00057         else i = i % _nbins;        
00058     }
00059     _data.y(i) += _weight * scale;
00060 }
