84  void Run(void) {
85 
86  if (_switch_energy_induce == 1) {
87  while (_forker->NextChunkTodo(_id,1)) {

170  void InterEnergy() {
171 
172  double e_f_12_21 = 0.0;
173  double e_m_12 = 0.0;
174  double e_m_21 = 0.0;
175 
176  for (int i = _nx1; i < _nx2; ++i) {
177  for (int j = (i >= _ny1) ? i+1 : _ny1; j < _ny2; ++j) {
178 
179  bool i_inCenter
180  = _forker->_job->isInCenter((*_qmm)[i]->getId());
181  bool j_inCenter
182  = _forker->_job->isInCenter((*_qmm)[j]->getId());
183 
184  // Pair-non-pair interaction
185  if ( i_inCenter ^ j_inCenter ) {
186 
187  for (pit1 = (*_qmm)[i]->begin();
188  pit1 < (*_qmm)[i]->end();
189  ++pit1) {
190  for (pit2 = (*_qmm)[j]->begin();
191  pit2 < (*_qmm)[j]->end();
192  ++pit2) {
193 
194  _actor.BiasIndu(*(*pit1),*(*pit2));
195  e_f_12_21 = _actor.E_f(*(*pit1),*(*pit2));
196  e_m_12 = _actor.E_m(*(*pit1),*(*pit2));
197  _actor.RevBias();
198  e_m_21 = _actor.E_m(*(*pit2),*(*pit1));
199 
200  _E_Pair_Sph1 += e_f_12_21 + e_m_12 + e_m_21;
201 
202  _E_f_C_non_C += e_f_12_21;
203  if ( i_inCenter ) {
204  _E_m_C += e_m_12;
205  _E_m_non_C += e_m_21;
206  }
207  else {
208  _E_m_C += e_m_21;
209  _E_m_non_C += e_m_12;
210  }
211  }}
212  }
213 
214  // Pair-pair interaction
215  else if ( i_inCenter && j_inCenter ) {
216 
217  for (pit1 = (*_qmm)[i]->begin();
218  pit1 < (*_qmm)[i]->end();
219  ++pit1) {
220  for (pit2 = (*_qmm)[j]->begin();
221  pit2 < (*_qmm)[j]->end();
222  ++pit2) {
223 
224  _actor.BiasIndu(*(*pit1),*(*pit2));
225  e_f_12_21 = _actor.E_f(*(*pit1),*(*pit2));
226  e_m_12 = _actor.E_m(*(*pit1),*(*pit2));
227  _actor.RevBias();
228  e_m_21 = _actor.E_m(*(*pit2),*(*pit1));
229 
230  _E_Pair_Pair += e_f_12_21 + e_m_12 + e_m_21;
231 
232  _E_f_C_C += e_f_12_21;
233  _E_m_C += e_m_12;
234  _E_m_C += e_m_21;
235 
236  }}
237  }
238 
239  // Non-pair-non-pair interaction
240  else {
241  for (pit1 = (*_qmm)[i]->begin();
242  pit1 < (*_qmm)[i]->end();
243  ++pit1) {
244  for (pit2 = (*_qmm)[j]->begin();
245  pit2 < (*_qmm)[j]->end();
246  ++pit2) {
247 
248  _actor.BiasIndu(*(*pit1),*(*pit2));
249  e_f_12_21 = _actor.E_f(*(*pit1),*(*pit2));
250  e_m_12 = _actor.E_m(*(*pit1),*(*pit2));
251  _actor.RevBias();
252  e_m_21 = _actor.E_m(*(*pit2),*(*pit1));
253 
254  _E_Sph1_Sph1 += e_f_12_21 + e_m_12 + e_m_21;
255 
256  _E_f_non_C_non_C += e_f_12_21;
257  _E_m_non_C += e_m_12;
258  _E_m_non_C += e_m_21;
259  }}
260  }

310  void InitChunks() {
311 
312  _nx1.clear();
313  _nx2.clear();
314  _ny1.clear();
315  _ny2.clear();
316 
317  _xy_done.clear();
318  _chunks_avail.clear();
319 
320  int T = this->_subthreads; // Threads
321  int C = T * 2; // Chunks
322  int N = _qmm.size(); // Elements
323  int nr = N % C; // Rest size
324  int nt = (N-nr) / C; // Chunk size
325 
326  assert (N == C*nt + nr);
327 
328  for (int id = 0; id < C+1; ++id) {
329  _nx1.push_back( std::vector<int>(C+1,0) );
330  _nx2.push_back( std::vector<int>(C+1,0) );
331  _ny1.push_back( std::vector<int>(C+1,0) );
332  _ny2.push_back( std::vector<int>(C+1,0) );
333 
334  _xy_done.push_back( std::vector<bool>(C+1,false) );
335  _chunks_avail.push_back(true);
336  }
337 
338  for (int col = 0; col < C+1; ++col) {
339  for (int row = 0; row < C+1; ++row) {
340 
341  if (col < row) {
342  _nx1[row][col] = 0; _ny1[row][col] = 0;
343  _nx2[row][col] = 0; _ny2[row][col] = 0;
344  }
345  else if (col == C && row == C) {
346  _nx1[row][col] = C*nt;
347  _nx2[row][col] = C*nt + nr;
348  _ny1[row][col] = C*nt;
349  _ny2[row][col] = C*nt + nr;
350  }
351  else if (col == C && row < C) {
352  _nx1[row][col] = row*nt;
353  _nx2[row][col] = (row+1)*nt;
354  _ny1[row][col] = C*nt;
355  _ny2[row][col] = C*nt + nr;
356  }
357  else {
358  _nx1[row][col] = row*nt;
359  _nx2[row][col] = (row+1)*nt;
360  _ny1[row][col] = col*nt;
361  _ny2[row][col] = (col+1)*nt;
362  }
363  }
364  }
365 
366  if (T > 1 && _maverick) {
367  printf("\n\nTHREAD MESH LOAD: "
368  "NST%1d C%1d N%1d nt%1d nr%1d\n", T, C, N, nt, nr);
369  for (int id = 0; id < C+1; ++id) {
370  for (int run = 0; run < C+1; ++run) {
371  printf("--------+");
372  }
373  printf("\n");
374  for (int run = 0; run < C+1; ++run) {
375  printf("%3d %3d |", _nx1[id][run], _ny1[id][run]);
376  }
377  printf("\n");
378  for (int run = 0; run < C+1; ++run) {
379  printf("%3d %3d |", _nx2[id][run], _ny2[id][run]);
380  }
381  printf("\n");
382  }
383  for (int run = 0; run < C+1; ++run) {
384  printf("--------+");
385  }
386  printf("\n");
387  }
388  }

395  bool NextChunkTodo(int indu_id, int switch_energy_induce) {
396 
397  _alloc_chunk.Lock();
398 
399  bool todo = false;
400 
401  while (true) {
402 
403  for (unsigned int i = 0; i < _xy_done.size(); ++i) {
404  for (unsigned int j = i; j < _xy_done[i].size(); ++j) {
405  if (!_xy_done[i][j]) {
406  todo = true;
407  if (_chunks_avail[i] && _chunks_avail[j]) {
408 
409  if (switch_energy_induce == 1) {
410  _chunks_avail[i] = false;
411  _chunks_avail[j] = false;
412  }
413 
414  _xy_done[i][j] = true;
415  _indus[indu_id]->Setc1c2(i,j);
416  _indus[indu_id]->Setnx12ny12(_nx1[i][j],
417  _nx2[i][j],
418  _ny1[i][j],
419  _ny2[i][j]);
420 
421  _alloc_chunk.Unlock();
422  return todo;
423  }
424  else { ; }
425  }
426  }}
427 
428  if (!todo) { break; }
429 
430  }
431 
432  _alloc_chunk.Unlock();
433  return todo;
434  }
