102 bool MapChecker::Evaluate(Topology& top) {
103  std::cout << std::endl;
104  std::string filename = AddSteptoFilename(_segmentfile, top.getStep());
105  std::cout << "Writing segments to " << filename << std::endl;
106  top.WriteToPdb(filename);
107 
108  Logger log;
109  log.setReportLevel(Log::current_level);
110  log.setCommonPreface("\n... ...");
111 
112  QMMapper map(log);
113 
114  for (QMState state : _qmstates) {
115  map.LoadMappingFile(_mapfile);
116  std::string filename_qm = AddStatetoFilename(_qmfile, state);
117  csg::PDBWriter qmwriter;
118  std::string filename_qm_state =
119  AddSteptoFilename(filename_qm, top.getStep());
120  std::cout << "Writing qmmolecules to " << filename_qm_state << std::endl;
121  qmwriter.Open(filename_qm_state, false);
122  qmwriter.WriteHeader("QMFrame:" + std::to_string(top.getStep()) +
123  " state:" + state.ToString());
124  qmwriter.WriteBox(top.getBox() * tools::conv::bohr2ang);
125  for (const Segment& seg : top.Segments()) {
126  QMMolecule mol = map.map(seg, state);
127  qmwriter.WriteContainer(mol);
128  }
129  qmwriter.Close();
130  }
131 
132  PolarMapper mp(log);
133  for (QMState state : _mdstates) {
134  mp.LoadMappingFile(_mapfile);
135  std::string filename_mp = AddStatetoFilename(_mpfile, state);
136  csg::PDBWriter mpwriter;
137  std::string filename_mp_state =
138  AddSteptoFilename(filename_mp, top.getStep());
139  std::cout << "Writing polarsegments to " << filename_mp_state << std::endl;
140  mpwriter.Open(filename_mp_state, false);
141  mpwriter.WriteHeader("MPFrame:" + std::to_string(top.getStep()) +
142  " state:" + state.ToString());
143  mpwriter.WriteBox(top.getBox() * tools::conv::bohr2ang);
144  for (const Segment& seg : top.Segments()) {
145  PolarSegment mol = mp.map(seg, state);
146  mpwriter.WriteContainer(mol);
147  }
148  mpwriter.Close();
149  }
150 
151  return true;
152 }
