81 std::vector<QMState> MapChecker::StringToStates(

100 bool MapChecker::EvaluateFrame(Topology& top) {
101  std::cout << std::endl;
102  std::string filename = AddSteptoFilename(_segmentfile, top.getStep());
103  std::cout << "Writing segments to " << filename << std::endl;
104  top.WriteToPdb(filename);
105 
106  Logger log;
107  log.setReportLevel(Log::current_level);
108  log.setCommonPreface("\n... ...");
109 
110  QMMapper map(log);
111 
112  for (QMState state : _qmstates) {
113  map.LoadMappingFile(_mapfile);
114  std::string filename_qm = AddStatetoFilename(_qmfile, state);
115  csg::PDBWriter qmwriter;
116  std::string filename_qm_state =
117  AddSteptoFilename(filename_qm, top.getStep());
118  std::cout << "Writing qmmolecules to " << filename_qm_state << std::endl;
119  qmwriter.Open(filename_qm_state, false);
120  qmwriter.WriteHeader("QMFrame:" + std::to_string(top.getStep()) +
121  " state:" + state.ToString());
122  qmwriter.WriteBox(top.getBox() * tools::conv::bohr2ang);
123  for (const Segment& seg : top.Segments()) {
124  QMMolecule mol = map.map(seg, state);
125  qmwriter.WriteContainer(mol);
126  }
127  qmwriter.Close();
128  }
129 
130  PolarMapper mp(log);
131  for (QMState state : _mdstates) {
132  mp.LoadMappingFile(_mapfile);
133  std::string filename_mp = AddStatetoFilename(_mpfile, state);
134  csg::PDBWriter mpwriter;
135  std::string filename_mp_state =
136  AddSteptoFilename(filename_mp, top.getStep());
137  std::cout << "Writing polarsegments to " << filename_mp_state << std::endl;
138  mpwriter.Open(filename_mp_state, false);
139  mpwriter.WriteHeader("MPFrame:" + std::to_string(top.getStep()) +
140  " state:" + state.ToString());
141  mpwriter.WriteBox(top.getBox() * tools::conv::bohr2ang);
142  for (const Segment& seg : top.Segments()) {
143  PolarSegment mol = mp.map(seg, state);
144  mpwriter.WriteContainer(mol);
145  }
146  mpwriter.Close();
147  }
148 
149  return true;
150 }
