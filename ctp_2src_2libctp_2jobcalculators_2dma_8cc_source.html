50 void DMA::Initialize(Property *options) {
51 
52  _do_input = false;
53  _do_orbitals = false;
54  _do_dma = false;
55 
56  _maverick = (_nThreads == 1) ? true : false;
57 
58  string key = "options." + Identify();
59  string _package_xml = options->get(key+".package").as<string> ();
60  _chkFile = options->get(key+".chk").as<string> ();
61  _executable = options->get(key+".executable").as<string> ();
62 
63  _density = options->get(key+".density").as<string> ();
64  _limit = options->get(key+".multipoles.limit").as<int> ();
65  _radius = options->get(key+".multipoles.radius").as<double> ();
66  _switch = options->get(key+".multipoles.switch").as<int> ();
67  _outFile = options->get(key+".output").as<string> ();
68 
69  string _tasks_string = options->get(key+".tasks").as<string> ();
70  if (_tasks_string.find("input") != std::string::npos) _do_input = true;
71  if (_tasks_string.find("orbitals") != std::string::npos) _do_orbitals = true;
72  if (_tasks_string.find("dma") != std::string::npos) _do_dma = true;
73 
74  key = "options." + Identify() +".job";
75  _jobfile = options->get(key + ".file").as<string>();
76 
77  load_property_from_xml( _package_options, _package_xml.c_str() );
78  key = "package";
79  _package = _package_options.get(key+".name").as<string> ();
80  if ( _package != "gaussian" ) { throw runtime_error("\nERROR: Package: " + _package + "not supported. Use Gaussian for GDMA"); }
81 
82 
83  // register all QM packages (Gaussian, Turbomole, NWChem))
84  QMPackageFactory::RegisterAll();
85 
86 }

128 Job::JobResult DMA::EvalJob(Topology *top, Job *job, QMThread *opThread) {
129 
130  string output;
131 
132  bool _orbitals_status;
133  //bool _formchk_status;
134  bool _dma_status;
135 
136  Job::JobResult jres = Job::JobResult();
137  Property _job_input = job->getInput();
138  list<Property*> lSegments = _job_input.Select( "segment" );
139 
140  vector < Segment* > segments;
141  int segId = lSegments.front()->getAttribute<int>( "id" );
142  string segType = lSegments.front()->getAttribute<string>( "type" );
143 
144  Segment *seg = top->getSegment( segId );
145  assert( seg->getName() == segType );
146  segments.push_back( seg );
147  Logger* pLog = opThread->getLogger();
148  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Evaluating site " << seg->getId() << flush;
149 
150  // log, com, and orbital files will be stored in ORB_FILES/package_name/frame_x/mol_ID/
151  // extracted information will be stored in ORB_FILES/molecules/frame_x/molecule_ID.orb
152 
153  string dma_work_dir = "DMA_FILES";
154  string frame_dir = "frame_" + boost::lexical_cast<string>(top->getDatabaseId());
155  string ID = boost::lexical_cast<string>( seg->getId() );
156 
157  // get the corresponding object from the QMPackageFactory
158  QMPackage *_qmpackage = QMPackages().Create( _package );
159 
160  _qmpackage->setLog( pLog );
161  _qmpackage->Initialize( &_package_options );
162 
163  string qmpackage_work_dir;
164 
165  qmpackage_work_dir = dma_work_dir + "/" + _package + "/" + frame_dir + "/mol_" + ID;
166  boost::filesystem::create_directories( qmpackage_work_dir );
167 
168  _qmpackage->setRunDir( qmpackage_work_dir );
169 
170  // if asked, prepare the input files
171  if ( _do_input ) {
172  _qmpackage->WriteInputFile( segments );
173  }
174 
175  // Run the Gaussian executable
176  if ( _do_orbitals ) {
177  _orbitals_status = _qmpackage->Run( );
178  if ( !_orbitals_status ) {
179  output += "run failed; " ;
180  CTP_LOG(logERROR,*pLog) << _package << " run failed" << flush;
181  jres.setOutput( output );
182  jres.setStatus(Job::FAILED);
183  delete _qmpackage;
184  return jres;
185  } else {
186  output += "orbitals completed; " ;
187  }
188  }
189 
190  // Parse log files
191  if ( _do_dma ) {
192 
193 
194  CTP_LOG(logDEBUG,*pLog) << "DMA: running formcheck on [" << _chkFile << "]" << flush;
195 
196  if (system(NULL)) {
197  string _command;
198  _command = "cd " + qmpackage_work_dir + "; formchk " + _chkFile;
199 
200  int i = system ( _command.c_str() );
201 
202  // prepare the GDMA input file
203  //Title ""
204  // File AAE007_neutral.fchk
205  //Multipoles
206  // Limit 2
207  // Limit 1 H
208  // Radius H 0.35
209  // Punch workout_neutral_6-311+gdp.mps
210  //Start
211  //Finish
212 
213  std::fstream fs;
214  string _dma_input_file = qmpackage_work_dir + "/gdma.in";
215 
216  cout << _dma_input_file << std::endl;
217 
218  fs.open (_dma_input_file.c_str(), std::fstream::out );
219  if (fs.is_open()) {
220  fs << "Title " << "GDMA\n";
221  fs << "Density " << _density << std::endl;
222 
223  // chk -> fchk
224  string _fchkFile = _chkFile;
225  boost::algorithm::replace_all(_fchkFile, ".chk", ".fchk");
226 
227  fs << "File " << _fchkFile << std::endl;
228  fs << "Multipoles\n" ;
229  fs << "Switch " << _switch << std::endl;
230  fs << "Limit " << _limit << std::endl;
231  fs << "Radius " << _radius << std::endl;
232  fs << "Punch " << _outFile << std::endl ;
233  fs << "Start\n" ;
234  fs << "Finish\n" ;
235  fs.close();
236 
237  _command = "cd " + qmpackage_work_dir + "; " + _executable + " < gdma.in > gdma.out";
238 
239  CTP_LOG(logINFO,*pLog) << _command << flush;
240 
241  int j = system ( _command.c_str() );
242 
243  _dma_status = (i == 0) && (j == 0);
244 
245  }
246  else {
247  CTP_LOG(logDEBUG,*pLog) << "Error opening file " << _dma_input_file << flush;
248  _dma_status = false;
249  }
250 
251  if ( !_dma_status ) {
252  output += "DMA job incomplete; ";
253  CTP_LOG(logERROR,*pLog) << "DMA job incomplete" << flush;
254  jres.setOutput( output );
255  jres.setStatus(Job::FAILED);
256  return jres;
257  } else {
258  output += "DMA complete; " ;
259  }
260  }
261 
262 
263  }
264 
265 
266  // Clean run
267  _qmpackage->CleanUp();
268  delete _qmpackage;
269 
270  CTP_LOG(logINFO,*pLog) << TimeStamp() << " Finished evaluating site " << seg->getId() << flush;
271 
272  Property _job_summary;
273  Property *_output_summary = &_job_summary.add("output","");
274  Property *_segment_summary = &_output_summary->add("segment","");
275  string segName = seg->getName();
276  segId = seg->getId();
277  _segment_summary->setAttribute("id", segId);
278 
279  // output of the JOB
280  jres.setOutput( _job_summary );
281  jres.setStatus(Job::COMPLETE);
282 
283 
284  return jres;
