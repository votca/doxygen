138 bool EAnalyze::EvaluateFrame(Topology *top) {
139 
140  // Short-list segments according to pattern
141  std::vector<Segment*>::iterator sit;
142  for (sit=top->Segments().begin(); sit!=top->Segments().end(); ++sit) {
143  std::string seg_name = (*sit)->getName();
144  if (votca::tools::wildcmp(_seg_pattern.c_str(), seg_name.c_str())) {
145  _seg_shortlist.push_back(*sit);
146  }
147  }
148  std::cout << std::endl << "... ... Short-listed " << _seg_shortlist.size()
149  << " segments (pattern='" << _seg_pattern << "')" << std::flush;
150  std::cout << std::endl << "... ... ... NOTE Statistics of site energies and spatial"
151  << " correlations thereof are based on the short-listed segments only. "
152  << std::flush;
153  std::cout << std::endl << "... ... ... "
154  << "Statistics of site-energy differences operate on the full list."
155  << std::flush;
156 
157  // Calculate
158  // ... Site-energy histogram, mean, width
159  // ... Pair-energy histogram, mean, width
160  // ... Site-energy correlation
161 
162  QMNBList &nblist = top->NBList();
163 
164  for (unsigned i = 0; i < _states.size(); ++i) {
165 
166  int state = _states[i];
167  cout << std::endl << "... ... Charge state " << state << std::flush;
168 
169  if (!_seg_shortlist.size()) {
170  std::cout << std::endl << "... ... ... No segments short-listed. Skip ... "
171  << std::flush;
172  }
173  else {
174  // Site-energy histogram <> DOS
175  if (_skip_sites) {
176  std::cout << std::endl << "... ... ... Skip site-energy hist." << std::flush;
177  }
178  else {
179  SiteHist(top, state);
180  }
181 
182  // Site-energy correlation function
183  if (_skip_corr) {
184  std::cout << std::endl << "... ... ... Skip correlation ..." << std::flush;
185  }
186  else {
187  SiteCorr(top, state);
188  }
189  }
190 
191  if (!nblist.size()) {
192  std::cout << std::endl << "... ... ... No pairs in topology. Skip ... "
193  << std::flush;
194  }
195  else {
196  // Site-energy-difference histogram <> Pair DOS
197  if (_skip_pairs) {
198  std::cout << std::endl << "... ... ... Skip pair-energy hist." << std::flush;
199  }
200  else {
201  PairHist(top, state);
202  }
203  }
204  }
205 
206  return true;

209 void EAnalyze::SiteHist(Topology *top, int state) {
210 
211  vector< double > Es;
212  Es.reserve(_seg_shortlist.size());
213 
214  double MIN = _seg_shortlist[0]->getSiteEnergy(state);
215  double MAX = _seg_shortlist[0]->getSiteEnergy(state);
216  double AVG = 0.0;
217  double VAR = 0.0;
218  double STD = 0.0;
219 
220  // Collect energies from segments, calc AVG
221  vector< Segment* > ::iterator sit;
222  for (sit = _seg_shortlist.begin();
223  sit < _seg_shortlist.end();
224  ++sit) {
225 
226  double E = (*sit)->getSiteEnergy(state);
227 
228  MIN = (E < MIN) ? E : MIN;
229  MAX = (E > MAX) ? E : MAX;
230  AVG += E / _seg_shortlist.size();
231 
232  Es.push_back(E);
233  }
234 
235  _site_avg = AVG;
236 
237  // Prepare bins
238  int BIN = int( (MAX-MIN)/_resolution_sites + 0.5 ) + 1;
239  vector< vector<double> > histE;
240  histE.resize(BIN);
241 
242  // Execute binning, calc VAR
243  vector< double > ::iterator eit;
244  for (eit = Es.begin(); eit < Es.end(); ++eit) {
245 
246  int bin = int( (*eit-MIN)/_resolution_sites + 0.5 );
247  histE[bin].push_back(*eit);
248  VAR += ((*eit) - AVG)*((*eit) - AVG) / _seg_shortlist.size();
249  }
250 
251  vector< int > histN;
252  histN.resize(BIN);
253  for (int bin = 0; bin < BIN; ++bin) {
254  histN[bin] = histE[bin].size();
255  }
256 
257  STD = sqrt(VAR);
258 
259  FILE *out;
260  std::string statename;
261  if (state==-1){
262  statename="e";
263  }
264  else if (state==1){
265  statename="h";
266  }
267  else if (state==2){
268  statename="s";
269  }
270  else if (state==3){
271  statename="t";
272  }
273 
274 
275  std::string tag = boost::lexical_cast<std::string>("eanalyze.sitehist_") +statename+ ".out";
276  out = fopen(tag.c_str(), "w");
277 
278  fprintf(out, "# EANALYZE: SITE-ENERGY HISTOGRAM \n");
279  fprintf(out, "# AVG %4.7f STD %4.7f MIN %4.7f MAX %4.7f \n",
280  AVG, STD, MIN, MAX);
281 
282  for (int bin = 0; bin < BIN; ++bin) {
283  double E = MIN + bin*_resolution_sites;
284  fprintf(out, "%4.7f %4d \n", E, histN[bin]);
285  }
286  fclose(out);
287 
288  // Write "seg x y z energy" with atomic {x,y,z}
289  if (_do_atomic_xyze) {
290  tag = (state == -1) ? "eanalyze.landscape_e.out" : "eanalyze.landscape_h.out";
291  out = fopen(tag.c_str(), "w");
292 
293  for (sit = _seg_shortlist.begin();
294  sit < _seg_shortlist.end();
295  ++sit) {
296 
297  if ((*sit)->getId() < _atomic_first) { continue; }
298  if ((*sit)->getId() == _atomic_last) { break; }
299  double E = (*sit)->getSiteEnergy(state);
300 
301  vector< Atom* > ::iterator ait;
302  for (ait = (*sit)->Atoms().begin();
303  ait < (*sit)->Atoms().end();
304  ++ait) {
305 
306  Atom *atm = *ait;
307 
308  fprintf(out, "%3s %4.7f %4.7f %4.7f %4.7f\n",
309  (*sit)->getName().c_str(),
310  atm->getPos().getX(),
311  atm->getPos().getY(),
312  atm->getPos().getZ(),
313  E);
314  }
315  }
316  fclose(out);
317  }

321 void EAnalyze::PairHist(Topology *top, int state) {
322 
323  QMNBList &nblist = top->NBList();
324  QMNBList::iterator pit;
325 
326  double MIN = nblist.front()->Seg1()->getSiteEnergy(state)
327  - nblist.front()->Seg2()->getSiteEnergy(state);
328  double MAX = nblist.front()->Seg1()->getSiteEnergy(state)
329  - nblist.front()->Seg2()->getSiteEnergy(state);
330  double AVG = 0.0;
331  double VAR = 0.0;
332  double STD = 0.0;
333 
334  FILE *out_dEs;
335  std::string tag_dEs = boost::lexical_cast<std::string>("eanalyze.pairlist_") + ( (state == -1) ? "e" : "h" ) + ".out";
336  out_dEs = fopen(tag_dEs.c_str(), "w");
337 
338  // Collect site-energy differences from neighbourlist
339  vector< double > dEs;
340  for (pit = nblist.begin(); pit != nblist.end(); ++pit) {
341 
342  Segment *seg1 = (*pit)->Seg1();
343  Segment *seg2 = (*pit)->Seg2();
344 
345  double dE = seg2->getSiteEnergy(state) - seg1->getSiteEnergy(state);
346 
347  MIN = (dE < MIN) ? dE : MIN;
348  MIN = (-dE < MIN) ? -dE : MIN;
349  MAX = (dE > MAX) ? dE : MAX;
350  MAX = (-dE > MAX) ? -dE : MAX;
351  AVG += dE / nblist.size();
352 
353  dEs.push_back(dE);
354  dEs.push_back(-dE);
355 
356  fprintf(out_dEs, "%5d %5d %4.7f \n", seg1->getId(), seg2->getId(), dE);
357  }
358  fclose(out_dEs);
359 
360  // Prepare bins
361  int BIN = int( (MAX-MIN)/_resolution_pairs + 0.5 ) + 1;
362  vector< vector<double> > histE;
363  histE.resize(BIN);
364 
365  // Execute binning, calc VAR
366  vector< double > ::iterator eit;
367  for (eit = dEs.begin(); eit < dEs.end(); ++eit) {
368 
369  int bin = int( ((*eit)-MIN)/_resolution_pairs + 0.5 );
370 
371  histE[bin].push_back((*eit));
372  VAR += ((*eit) - AVG)*((*eit) - AVG) / nblist.size();
373  }
374 
375  vector< int > histN;
376  histN.resize(BIN);
377  for (int bin = 0; bin < BIN; ++bin) {
378  histN[bin] = histE[bin].size();
379  }
380 
381  STD = sqrt(VAR);
382 
383  FILE *out;
384  std::string statename;
385  if (state==-1){
386  statename="e";
387  }
388  else if (state==1){
389  statename="h";
390  }
391  else if (state==2){
392  statename="s";
393  }
394  else if (state==3){
395  statename="t";
396  }
397  std::string tag = boost::lexical_cast<std::string>("eanalyze.pairhist_") + statename + ".out";
398  out = fopen(tag.c_str(), "w");
399 
400  fprintf(out, "# EANALYZE: PAIR-ENERGY HISTOGRAM \n");
401  fprintf(out, "# AVG %4.7f STD %4.7f MIN %4.7f MAX %4.7f \n",
402  AVG, STD, MIN, MAX);
403 
404  for (int bin = 0; bin < BIN; ++bin) {
405  double E = MIN + bin*_resolution_pairs;
406  fprintf(out, "%4.7f %4d \n", E, histN[bin]);
407  }
408  fclose(out);
409 }

412 void EAnalyze::SiteCorr(Topology *top, int state) {
413 
414  double AVG = 0.0;
415  double AVGESTATIC = 0.0;
416  double VAR = 0.0;
417  double STD = 0.0;
418 
419  vector< Segment* > ::iterator sit1;
420  vector< Segment* > ::iterator sit2;
421 
422  // Calculate mean site energy
423  vector< double > Es;
424 
425  vector< Segment* > ::iterator sit;
426  for (sit = _seg_shortlist.begin();
427  sit < _seg_shortlist.end();
428  ++sit) {
429 
430  double E = (*sit)->getSiteEnergy(state);
431  AVG += E / _seg_shortlist.size();
432 
433  AVGESTATIC += (*sit)->getEMpoles(state) / top->Segments().size();
434 
435  Es.push_back(E);
436  }
437 
438  // Calculate variance
439  vector< double > ::iterator eit;
440  for (eit = Es.begin(); eit < Es.end(); ++eit) {
441 
442  VAR += ((*eit) - AVG)*((*eit) - AVG) / _seg_shortlist.size();
443  }
444 
445  STD = sqrt(VAR);
446 
447  // Collect inter-site distances, correlation product
448  vector< double > Rs;
449  vector< double > Cs;
450 
451  double MIN = +1e15;
452  double MAX = -1e15;
453 
454  vector< Fragment* > ::iterator fit1;
455  vector< Fragment* > ::iterator fit2;
456 
457  std::cout << std::endl;
458 
459  FILE *corr_out;
460  std::string statename;
461  if (state==-1){
462  statename="e";
463  }
464  else if (state==1){
465  statename="h";
466  }
467  else if (state==2){
468  statename="s";
469  }
470  else if (state==3){
471  statename="t";
472  }
473  std::string corrfile = boost::lexical_cast<std::string>("eanalyze.sitecorr.atomic_") + statename+ ".out";
474  corr_out = fopen(corrfile.c_str(), "w");
475 
476  for (sit1 = _seg_shortlist.begin(); sit1 < _seg_shortlist.end(); ++sit1) {
477 
478  std::cout << "\r... ... ..." << " Correlating segment ID = "
479  << (*sit1)->getId() << std::flush;
480 
481  for (sit2 = sit1 + 1; sit2 < _seg_shortlist.end(); ++sit2) {
482 
483  double R = abs(top->PbShortestConnect((*sit1)->getPos(),
484  (*sit2)->getPos()));
485 
486  if(_distancemode == "segment"){
487  for (fit1 = (*sit1)->Fragments().begin();
488  fit1 < (*sit1)->Fragments().end();
489  ++fit1) {
490  for (fit2 = (*sit2)->Fragments().begin();
491  fit2 < (*sit2)->Fragments().end();
492  ++fit2) {
493 
494  double R_FF = abs(top->PbShortestConnect((*fit1)->getPos(),
495  (*fit2)->getPos()));
496 
497  if (R_FF < R) { R = R_FF; }
498  }}
499  }
500 
501 
502  MIN = (R < MIN) ? R : MIN;
503  MAX = (R > MAX) ? R : MAX;
504 
505  double C = ((*sit1)->getSiteEnergy(state) - AVG)
506  * ((*sit2)->getSiteEnergy(state) - AVG);
507 
508  Rs.push_back(R);
509  Cs.push_back(C);
510 
511  fprintf(corr_out, "%+1.7f %+1.7f\n", R, C);
512 
513  }}
514 
515  fclose(corr_out);
516 
517  // Prepare bins
518  int BIN = int( (MAX-MIN)/_resolution_space + 0.5 ) + 1;
519  vector< vector<double> > histCs;
520  histCs.resize(BIN);
521 
522  for (unsigned i = 0; i < Rs.size(); ++i) {
523 
524  int bin = int((Rs[i] - MIN)/_resolution_space + 0.5);
525  histCs[bin].push_back(Cs[i]);
526  }
527 
528  // Calculate spatial correlation
529  vector< double > histC;
530  vector< double > histC_error;
531  histC.resize(BIN);
532  histC_error.resize(BIN);
533  for (int bin = 0; bin < BIN; ++bin) {
534 
535  double corr = 0.0;
536  double dcorr2 = 0.0;
537  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
538  corr += histCs[bin][i] / VAR;
539  //corr2 += (histCs[bin][i] / VAR)*(histCs[bin][i] / VAR);
540  }
541 
542  corr = corr / histCs[bin].size();
543  //corr2 = corr2 / histCs[bin].size();
544 
545  for (unsigned i = 0; i < histCs[bin].size(); ++i) {
546  dcorr2 += (histCs[bin][i]/VAR/histCs[bin].size() - corr)*(histCs[bin][i]/VAR/histCs[bin].size() - corr);
547  }
548 
549 
550  histC[bin] = corr;
551 
552  // error on mean value
553  dcorr2 = dcorr2 / histCs[bin].size() / (histCs[bin].size()-1);
554  histC_error[bin] = sqrt(dcorr2);
555  }
556 
557  FILE *out;
558  std::string tag = boost::lexical_cast<std::string>("eanalyze.sitecorr_") + ( (state == -1) ? "e" : "h" ) + ".out";
559  out = fopen(tag.c_str(), "w");
560 
561  fprintf(out, "# EANALYZE: SPATIAL SITE-ENERGY CORRELATION \n");
562  fprintf(out, "# AVG %4.7f STD %4.7f MIN_R %4.7f MAX_R %4.7f AVGESTATIC %4.7f\n",
563  AVG, STD, MIN, MAX, AVGESTATIC);
564 
565  for (int bin = 0; bin < BIN; ++bin) {
566  double R = MIN + bin*_resolution_space;
567  fprintf(out, "%4.7f %4.7f %4.7f\n", R, histC[bin], histC_error[bin]);
568  }
569  fclose(out);
570 }
