611 vector<PolarSite*> EMultipole_StdAl::ParseGdmaFile(string filename, int state) {
612 
613  int poleCount = 1;
614  double Q0_total = 0.0;
615  string units = "";
616  bool useDefaultPs = true;
617 
618  vector<PolarSite*> poles;
619  PolarSite *thisPole = NULL;
620 
621  vector<double> Qs; // <- multipole moments
622  double P1; // <- dipole polarizability
623 
624  std::string line;
625  std::ifstream intt;
626  intt.open(filename.c_str());
627 
628  if (intt.is_open() ) {
629  while ( intt.good() ) {
630 
631  std::getline(intt, line);
632  vector<string> split;
633  Tokenizer toker(line, " ");
634  toker.ToVector(split);
635 
636  if ( !split.size() ||
637  split[0] == "!" ||
638  split[0].substr(0,1) == "!" ) { continue; }
639 
640  // ! Interesting information here, e.g.
641  // ! DCV2T opt
642  // ! SP RB3LYP 6-311+G(d,p)
643  // Units bohr
644  //
645  // C -4.2414603400 -3.8124751600 0.0017575736 Rank 2
646  // -0.3853409355
647  // -0.0002321905 0.2401559510 0.6602334308
648  // -0.7220625314 0.0004894995 -0.0003833545 0.4526409813 -0.50937399
649  // P 1.75
650 
651 
652  // Units used
653  if ( split[0] == "Units") {
654  units = split[1];
655  if (units != "bohr" && units != "angstrom") {
656  throw std::runtime_error( "Unit " + units + " in file "
657  + filename + " not supported.");
658  }
659  }
660 
661  // element, position, rank limit
662  else if ( split.size() == 6 ) {
663 
664  Qs.clear();
665  P1 = -1.;
666 
667  int id = poleCount++; // <- starts from 1
668  string name = split[0];
669 
670  double BOHR2NM = 0.0529189379;
671  double ANGSTROM2NM = 0.1;
672  double x, y, z;
673 
674  if (units == "bohr") {
675  x = BOHR2NM * boost::lexical_cast<double>(split[1]);
676  y = BOHR2NM * boost::lexical_cast<double>(split[2]);
677  z = BOHR2NM * boost::lexical_cast<double>(split[3]);
678  }
679  else if (units == "angstrom") {
680  x = ANGSTROM2NM * boost::lexical_cast<double>(split[1]);
681  y = ANGSTROM2NM * boost::lexical_cast<double>(split[2]);
682  z = ANGSTROM2NM * boost::lexical_cast<double>(split[3]);
683  }
684  else {
685  throw std::runtime_error( "Unit " + units + " in file "
686  + filename + " not supported.");
687  }
688 
689  vec pos = vec(x,y,z);
690 
691  int rank = boost::lexical_cast<int>(split[5]);
692 
693  PolarSite *newPole = new PolarSite(id, name);
694  newPole->setRank(rank);
695  newPole->setPos(pos);
696  poles.push_back(newPole);
697  thisPole = newPole;
698 
699  }
700 
701  // 'P', dipole polarizability
702  else if ( split[0] == "P" && split.size() == 2 ) {
703  P1 = 1e-3 * boost::lexical_cast<double>(split[1]);
704  thisPole->setPs(P1, state);
705  useDefaultPs = false;
706  }
707 
708  // multipole line
709  else {
710 
711  int lineRank = int( sqrt(thisPole->getQs(state).size()) + 0.5 );
712 
713  if (lineRank == 0) {
714  Q0_total += boost::lexical_cast<double>(split[0]);
715  }
716 
717  for (unsigned int i = 0; i < split.size(); i++) {
718 
719  double qXYZ = boost::lexical_cast<double>(split[i]);
720 
721  // Convert e*(a_0)^k to e*(nm)^k where k = rank
722  double BOHR2NM = 0.0529189379;
723  qXYZ *= pow(BOHR2NM, lineRank); // OVERRIDE
724 
725  Qs.push_back(qXYZ);
726 
727  }
728  thisPole->setQs(Qs, state);
729  }
730 
731  } /* Exit loop over lines */
732  }
733  else { cout << endl << "ERROR: No such file " << filename << endl; }
734 
735  if (useDefaultPs) {
736 
737  cout << endl << "... ... NOTE Using default Thole polarizabilities "
738  << "for charge state " << state << ". ";
739 
740  vector< PolarSite* > ::iterator pol;
741  for (pol = poles.begin(); pol < poles.end(); ++pol) {
742  string elem = (*pol)->getName();
743  double alpha = 0.0;
744  // Original set of Thole polarizabilites
745  if (elem == "C") { alpha = 1.75e-3; } // <- conversion from
746  else if (elem == "H") { alpha = 0.696e-3; } // A³ to nm³ = 10⁻³
747  else if (elem == "N") { alpha = 1.073e-3; }
748  else if (elem == "O") { alpha = 0.837e-3; }
749  else if (elem == "S") { alpha = 2.926e-3; }
750  // Different set of Thole polarizabilities
751  //if (elem == "C") { alpha = 1.334e-3; } // <- conversion from
752  //else if (elem == "H") { alpha = 0.496e-3; } // A³ to nm³ = 10⁻³
753  //else if (elem == "N") { alpha = 1.073e-3; }
754  //else if (elem == "O") { alpha = 0.837e-3; }
755  //else if (elem == "S") { alpha = 3.300e-3; }
756  else { throw runtime_error("No polarizability given "
757  "for polar site type " + elem + ". "); }
758  (*pol)->setPs(alpha, state);
759  }
760  }
761 
762  cout << endl << "... ... Reading " << filename <<
763  ": Q0(Total) = " << Q0_total << flush;
764 
765  return poles;
