00036 string DLPOLYTopologyReader::_NextKeyline(ifstream &fs, const char* wspace)
00037 // function to find and read the next line starting with a keyword/directive (skipping comments starting with "#" or ";")
00038 // NOTE: the line is returned case-preserved, not to alter molecule/atom names (consider if --no-map is used)
00039 {
00040   string line;
00041   size_t i_nws=0;
00042 
00043   do {
00044     getline(fs, line);
00045 
00046     if (fs.eof())
00047       throw std::runtime_error("Error: unexpected end of dlpoly file '" + _fname + "'");
00048 
00049     i_nws = line.find_first_not_of(wspace);
00050   } while ( line.substr(i_nws,1) == "#" || line.substr(i_nws,1) == ";" );
00051 
00052   return line.substr(i_nws,line.size()-i_nws);
00053 }

00055 string DLPOLYTopologyReader::_NextKeyInt(ifstream &fs, const char* wspace, const string &word, int &ival)
00056 // function to read the next line containing only a given keyword and an integer value after it (only skipping comments!)
00057 // NOTE: this function must only be called when the next directive line has to contain the given keyword and an integer value
00058 {
00059   stringstream sl(_NextKeyline(fs,wspace));
00060   string line,sval;
00061 
00062   sl >> line; // allow user not to bother about the case
00063   boost::to_upper(line);
00064 
00065   if( line.substr(0,word.size()) != word )
00066     throw std::runtime_error("Error: unexpected line from dlpoly file '" + _fname + "', expected '" + word + "' but got '" + line + "'");
00067 
00068   sl >> sval;
00069 
00070   size_t i_num = sval.find_first_of("0123456789"); // assume integer number straight after the only keyword
00071 
00072   if( i_num>0 ) 
00073     throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
00074 
00075   ival = boost::lexical_cast<int>(sval);

00080 bool DLPOLYTopologyReader::_isKeyInt(const string &line, const char* wspace, const string &word, int &ival)
00081 // function to check if the given (last read) directive line starts with a given keyword and has an integer value at the end
00082 // NOTE: comments are allowed beyond the integer (anything after the first integer is ignored)
00083 {
00084   // split directives consisting of a few words: the sought keyword must be the first one!
00085   Tokenizer tok(line,wspace); 
00086   vector<string> fields;
00087   tok.ToVector(fields);
00088 
00089   ival = 0;
00090 
00091   if( fields.size() < 2 ) return false;
00092 
00093   boost::to_upper(fields[0]);
00094 
00095   if (fields[0].substr(0,word.size()) != word )
00096     throw std::runtime_error("Error: unexpected directive from dlpoly file '" + _fname + "', expected keyword '"+ word +"' but got '" + fields[0] + "'");
00097 
00098   size_t i_num=string::npos;
00099 
00100   unsigned int i=1;
00101   do { // find integer number in the field with the lowest index (closest to the keyword)
00102     i_num = fields[i++].find_first_of("0123456789");
00103   } while ( i_num>0 && i<fields.size() );
00104 
00105   if( i_num>0 ) return false;
00106 
00107   ival = boost::lexical_cast<int>(fields[i-1]);
00108 
00109   return true;
00110 }
