40 string DLPOLYTopologyReader::_NextKeyline(ifstream &fs, const char *wspace)
41 // function to find and read the next line starting with a keyword/directive
42 // (skipping comments starting with "#" or ";") NOTE: the line is returned
43 // case-preserved, not to alter molecule/atom names (consider if --no-map is
44 // used)
45 {
46  string line;
47  size_t i_nws = 0;
48 
49  do {
50  getline(fs, line);
51 
52  if (fs.eof()) {
53  throw std::runtime_error("Error: unexpected end of dlpoly file '" +
54  _fname + "'");
55  }
56 
57  i_nws = line.find_first_not_of(wspace);
58  } while (line.substr(i_nws, 1) == "#" || line.substr(i_nws, 1) == ";");
59 
60  return line.substr(i_nws, line.size() - i_nws);
61 }

63 string DLPOLYTopologyReader::_NextKeyInt(ifstream &fs, const char *wspace,
64  const string &word, Index &ival)
65 // function to read the next line containing only a given keyword and an integer
66 // value after it (only skipping comments!) NOTE: this function must only be
67 // called when the next directive line has to contain the given keyword and an
68 // integer value
69 {
70  stringstream sl(_NextKeyline(fs, wspace));
71  string line, sval;
72 
73  sl >> line; // allow user not to bother about the case
74  boost::to_upper(line);
75 
76  if (line.substr(0, word.size()) != word) {
77  throw std::runtime_error("Error: unexpected line from dlpoly file '" +
78  _fname + "', expected '" + word + "' but got '" +
79  line + "'");
80  }
81 
82  sl >> sval;
83 
84  size_t i_num = sval.find_first_of(
85  "0123456789"); // assume integer number straight after the only keyword
86 
87  if (i_num > 0) {
88  throw std::runtime_error("Error: missing integer number in directive '" +
89  line + "' in topology file '" + _fname + "'");
90  }
91 
92  ival = boost::lexical_cast<Index>(sval);
93 
94  return sl.str();
95 }

97 bool DLPOLYTopologyReader::_isKeyInt(const string &line, const char *wspace,
98  const string &word, Index &ival)
99 // function to check if the given (last read) directive line starts with a given
100 // keyword and has an integer value at the end NOTE: comments are allowed beyond
101 // the integer (anything after the first integer is ignored)
102 {
103  // split directives consisting of a few words: the sought keyword must be the
104  // first one!
105  Tokenizer tok(line, wspace);
106  vector<string> fields;
107  tok.ToVector(fields);
108 
109  ival = 0;
110 
111  if (fields.size() < 2) {
112  return false;
113  }
114 
115  boost::to_upper(fields[0]);
116 
117  if (fields[0].substr(0, word.size()) != word) {
118  throw std::runtime_error("Error: unexpected directive from dlpoly file '" +
119  _fname + "', expected keyword '" + word +
120  "' but got '" + fields[0] + "'");
121  }
122 
123  size_t i_num = string::npos;
124 
125  Index i = 1;
126  do { // find integer number in the field with the lowest index (closest to
127  // the keyword)
128  i_num = fields[i++].find_first_of("0123456789");
129  } while (i_num > 0 && i < Index(fields.size()));
130 
131  if (i_num > 0) {
132  return false;
133  }
134 
135  ival = boost::lexical_cast<Index>(fields[i - 1]);
136 
137  return true;
138 }

140 bool DLPOLYTopologyReader::ReadTopology(string file, Topology &top) {
141 
142  Index natoms = 0;
143 
144  boost::filesystem::path filepath(file.c_str());
145 
146  string line;
147 
148  // TODO: fix residue naming / assignment - DL_POLY has no means to recognise
149  // residues!
150  Residue *res = top.CreateResidue("no");
151 
152  if (boost::filesystem::basename(filepath).size() == 0) {
153  if (filepath.parent_path().string().size() == 0) {
154  _fname = "FIELD"; // DL_POLY uses fixed file names in current/working
155  // directory
156  } else {
157  _fname = filepath.parent_path().string() + "/FIELD";
158  }
159  } else {
160  _fname = file;
161  }
162  std::ifstream fl;
163  fl.open(_fname);
164 
165  if (!fl.is_open()) {
166  throw std::runtime_error("Error on opening dlpoly file '" + _fname + "'");
167  } else {
168  const char *WhiteSpace = " \t";
169  _NextKeyline(fl, WhiteSpace); // read title line and skip it
170  line = _NextKeyline(fl, WhiteSpace); // read next directive line
171  boost::to_upper(line);
172 
173  if (line.substr(0, 4) == "UNIT") { // skip 'unit' line
174  line = _NextKeyline(fl, WhiteSpace); // read next directive line
175  boost::to_upper(line);
176  }
177 
178  if (line.substr(0, 4) == "NEUT") { // skip 'neutral groups' line (DL_POLY
179  // Classic FIELD format)
180  line = _NextKeyline(fl, WhiteSpace); // look for next directive line
181  boost::to_upper(line);
182  }
183 
184  Index nmol_types;
185 
186  if (!_isKeyInt(line, WhiteSpace, "MOLEC", nmol_types)) {
187  throw std::runtime_error("Error: missing integer number in directive '" +
188  line + "' in topology file '" + _fname + "'");
189  }
190 
191 #ifdef DEBUG
192  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - "
193  << nmol_types << endl;
194 #endif
195 
196  string mol_name;
197 
198  for (Index nmol_type = 0; nmol_type < nmol_types; nmol_type++) {
199 
200  mol_name = _NextKeyline(fl, WhiteSpace);
201  Molecule *mi = top.CreateMolecule(mol_name);
202 
203  Index nreplica = 1;
204  line = _NextKeyInt(fl, WhiteSpace, "NUMMOL", nreplica);
205 
206 #ifdef DEBUG
207  cout << "Read from dlpoly file '" << _fname << "' : '" << mol_name
208  << "' - '" << line << "' - " << nreplica << endl;
209 #endif
210 
211  line = _NextKeyInt(fl, WhiteSpace, "ATOMS", natoms);
212 
213 #ifdef DEBUG
214  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - "
215  << natoms << endl;
216 #endif
217  std::vector<Index> id_map(natoms);
218  for (Index i = 0; i < natoms;) { // i is altered in repeater loop
219  stringstream sl(_NextKeyline(fl, WhiteSpace));
220 
221 #ifdef DEBUG
222  cout << "Read atom specs from dlpoly topology : '" << sl.str() << "'"
223  << endl;
224 #endif
225  string beadtype;
226  sl >> beadtype;
227  if (!top.BeadTypeExist(beadtype)) {
228  top.RegisterBeadType(beadtype);
229  }
230  double mass;
231  sl >> mass;
232  double charge;
233  sl >> charge;
234 
235  line = " ";
236  sl >> line; // rest of the atom line
237 
238  Tokenizer tok(line, WhiteSpace);
239  vector<string> fields;
240  tok.ToVector(fields);
241 
242 #ifdef DEBUG
243  cout << "Rest atom specs from dlpoly topology : '" << line << "'"
244  << endl;
245 #endif
246 
247  Index repeater = 1;
248  if (fields.size() > 1) {
249  repeater = boost::lexical_cast<Index>(fields[0]);
250  }
251 
252  for (Index j = 0; j < repeater; j++) {
253 
254  string beadname = beadtype + "#" + boost::lexical_cast<string>(i + 1);
255  Bead *bead = top.CreateBead(Bead::spherical, beadname, beadtype,
256  res->getId(), mass, charge);
257 
258  stringstream nm;
259  nm << bead->getResnr() + 1 << ":"
260  << top.getResidue(bead->getResnr())->getName() << ":"
261  << bead->getName();
262  mi->AddBead(bead, nm.str());
263  id_map[i] = bead->getId();
264  i++;
265 #ifdef DEBUG
266  cout << "Atom identification in maps '" << nm.str() << "'" << endl;
267 #endif
268  }
269  }
270 
271  while (line != "FINISH") {
272 
273  stringstream nl(_NextKeyline(fl, WhiteSpace));
274  nl >> line;
275 
276 #ifdef DEBUG
277  cout << "Read unit type# from dlpoly topology : '" << nl.str() << "'"
278  << endl;
279 #endif
280 
281  boost::to_upper(line);
282  line = line.substr(0, 6);
283  if ((line == "BONDS") || (line == "ANGLES") || (line == "DIHEDR")) {
284  string type = line;
285  Index count;
286  nl >> count;
287  for (Index i = 0; i < count; i++) {
288 
289  stringstream sl(_NextKeyline(fl, WhiteSpace));
290 #ifdef DEBUG
291  cout << "Read unit specs from dlpoly topology : '" << sl.str()
292  << "'" << endl;
293 #endif
294  sl >> line; // internal dlpoly bond/angle/dihedral function types
295  // are merely skipped (ignored)
296  Index ids[4];
297  Interaction *ic = nullptr;
298  sl >> ids[0];
299  sl >> ids[1];
300  if (type == "BONDS") {
301  ic = new IBond(id_map[ids[0] - 1],
302  id_map[ids[1] - 1]); // -1 due to fortran vs c
303  } else if (type == "ANGLES") {
304  sl >> ids[2];
305  ic = new IAngle(id_map[ids[0] - 1], id_map[ids[1] - 1],
306  id_map[ids[2] - 1]); // -1 due to fortran vs c
307  } else if (type.substr(0, 6) == "DIHEDR") {
308  type = "DIHEDRALS";
309  sl >> ids[2];
310  sl >> ids[3];
311  ic = new IDihedral(id_map[ids[0] - 1], id_map[ids[1] - 1],
312  id_map[ids[2] - 1],
313  id_map[ids[3] - 1]); // -1 due to fortran vs c
314  } else {
315  throw std::runtime_error(
316  "Error: type should be BONDS, ANGLES or DIHEDRALS");
317  }
318  // could one use bond/angle/dihedral function types for 1:1 mapping?
319  // (CG map overwrites ic->Group anyway)
320  // ic->setGroup(line);
321  ic->setGroup(type);
322  ic->setIndex(i);
323  ic->setMolecule(mi->getId());
324  top.AddBondedInteraction(ic);
325  mi->AddInteraction(ic);
326  }
327  }
328  }
329 
330 #ifdef DEBUG
331  cout << "Read from dlpoly file '" << _fname << "' : '" << line
332  << "' - done with '" << mol_name << "'" << endl;
333 #endif
334 
335  // replicate molecule
336  for (Index replica = 1; replica < nreplica; replica++) {
337  Molecule *mi_replica = top.CreateMolecule(mol_name);
338  for (Index i = 0; i < mi->BeadCount(); i++) {
339  Bead *bead = mi->getBead(i);
340  string type = bead->getType();
341  Bead *bead_replica =
342  top.CreateBead(Bead::spherical, bead->getName(), type,
343  res->getId(), bead->getMass(), bead->getQ());
344  mi_replica->AddBead(bead_replica, bead->getName());
345  }
346  InteractionContainer ics = mi->Interactions();
347  for (auto &ic : ics) {
348  Interaction *ic_replica = nullptr;
349  Index offset =
350  mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
351  if (ic->BeadCount() == 2) {
352  ic_replica =
353  new IBond(ic->getBeadId(0) + offset, ic->getBeadId(1) + offset);
354  } else if (ic->BeadCount() == 3) {
355  ic_replica =
356  new IAngle(ic->getBeadId(0) + offset, ic->getBeadId(1) + offset,
357  ic->getBeadId(2) + offset);
358  } else if (ic->BeadCount() == 4) {
359  ic_replica = new IDihedral(
360  ic->getBeadId(0) + offset, ic->getBeadId(1) + offset,
361  ic->getBeadId(2) + offset, ic->getBeadId(3) + offset);
362  } else {
363  throw std::runtime_error("Error: BeadCount not equal 2, 3 or 4");
364  }
365  ic_replica->setGroup(ic->getGroup());
366  ic_replica->setIndex(ic->getIndex());
367  ic_replica->setMolecule(mi_replica->getId());
368  top.AddBondedInteraction(ic_replica);
369  mi_replica->AddInteraction(ic_replica);
370  }
371  }
372  }
373  top.RebuildExclusions();
374  }
375 
376 #ifdef DEBUG
377  getline(fl, line); // is "close" found?
378  if (line == "close") {
379  cout << "Read from dlpoly file '" << _fname << "' : '" << line
380  << "' - done with topology" << endl;
381  } else {
382  cout << "Read from dlpoly file '" << _fname
383  << "' : 'EOF' - done with topology (directive 'close' not read!)"
384  << endl;
385  }
386 #endif
387 
388  // we don't need the rest
389  fl.close();
390 
391  return true;
392 }
