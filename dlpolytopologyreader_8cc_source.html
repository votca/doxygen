36 string DLPOLYTopologyReader::_NextKeyline(ifstream &fs, const char* wspace)
37 // function to find and read the next line starting with a keyword/directive (skipping comments starting with "#" or ";")
38 // NOTE: the line is returned case-preserved, not to alter molecule/atom names (consider if --no-map is used)
39 {
40  string line;
41  size_t i_nws=0;
42 
43  do {
44  getline(fs, line);
45 
46  if (fs.eof())
47  throw std::runtime_error("Error: unexpected end of dlpoly file '" + _fname + "'");
48 
49  i_nws = line.find_first_not_of(wspace);
50  } while ( line.substr(i_nws,1) == "#" || line.substr(i_nws,1) == ";" );
51 
52  return line.substr(i_nws,line.size()-i_nws);
53 }

55 string DLPOLYTopologyReader::_NextKeyInt(ifstream &fs, const char* wspace, const string &word, int &ival)
56 // function to read the next line containing only a given keyword and an integer value after it (only skipping comments!)
57 // NOTE: this function must only be called when the next directive line has to contain the given keyword and an integer value
58 {
59  stringstream sl(_NextKeyline(fs,wspace));
60  string line,sval;
61 
62  sl >> line; // allow user not to bother about the case
63  boost::to_upper(line);
64 
65  if( line.substr(0,word.size()) != word )
66  throw std::runtime_error("Error: unexpected line from dlpoly file '" + _fname + "', expected '" + word + "' but got '" + line + "'");
67 
68  sl >> sval;
69 
70  size_t i_num = sval.find_first_of("0123456789"); // assume integer number straight after the only keyword
71 
72  if( i_num>0 )
73  throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
74 
75  ival = boost::lexical_cast<int>(sval);
76 
77  return sl.str();
78 }

80 bool DLPOLYTopologyReader::_isKeyInt(const string &line, const char* wspace, const string &word, int &ival)
81 // function to check if the given (last read) directive line starts with a given keyword and has an integer value at the end
82 // NOTE: comments are allowed beyond the integer (anything after the first integer is ignored)
83 {
84  // split directives consisting of a few words: the sought keyword must be the first one!
85  Tokenizer tok(line,wspace);
86  vector<string> fields;
87  tok.ToVector(fields);
88 
89  ival = 0;
90 
91  if( fields.size() < 2 ) return false;
92 
93  boost::to_upper(fields[0]);
94 
95  if (fields[0].substr(0,word.size()) != word )
96  throw std::runtime_error("Error: unexpected directive from dlpoly file '" + _fname + "', expected keyword '"+ word +"' but got '" + fields[0] + "'");
97 
98  size_t i_num=string::npos;
99 
100  unsigned int i=1;
101  do { // find integer number in the field with the lowest index (closest to the keyword)
102  i_num = fields[i++].find_first_of("0123456789");
103  } while ( i_num>0 && i<fields.size() );
104 
105  if( i_num>0 ) return false;
106 
107  ival = boost::lexical_cast<int>(fields[i-1]);
108 
109  return true;
110 }

113 bool DLPOLYTopologyReader::ReadTopology(string file, Topology &top)
114 {
115  const char *WhiteSpace=" \t";
116 
117  int matoms=0;
118  int natoms=0;
119 
120  std::ifstream fl;
121  boost::filesystem::path filepath(file.c_str());
122 
123  string line;
124 
125  // TODO: fix residue naming / assignment - DL_POLY has no means to recognise residues!
126  Residue *res = top.CreateResidue("no");
127 
128  if ( boost::filesystem::basename(filepath).size() == 0 ) {
129  if (filepath.parent_path().string().size() == 0) {
130  _fname="FIELD"; // DL_POLY uses fixed file names in current/working directory
131  } else {
132  _fname=filepath.parent_path().string() + "/FIELD";
133  }
134  } else {
135  _fname=file;
136  }
137 
138  fl.open(_fname.c_str());
139 
140  if (!fl.is_open()){
141  throw std::runtime_error("Error on opening dlpoly file '" + _fname + "'");
142  } else {
143 
144  line = _NextKeyline(fl,WhiteSpace); //read title line and skip it
145  line = _NextKeyline(fl,WhiteSpace); //read next directive line
146  boost::to_upper(line);
147 
148  if( line.substr(0,4)=="UNIT" ) { //skip 'unit' line
149  line = _NextKeyline(fl,WhiteSpace); //read next directive line
150  boost::to_upper(line);
151  }
152 
153  if( line.substr(0,4)=="NEUT" ) { //skip 'neutral groups' line (DL_POLY Classic FIELD format)
154  line = _NextKeyline(fl,WhiteSpace); //look for next directive line
155  boost::to_upper(line);
156  }
157 
158  int nmol_types;
159 
160  if( !_isKeyInt(line,WhiteSpace,"MOLEC",nmol_types) )
161  throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
162 
163 #ifdef DEBUG
164  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - " << nmol_types << endl;
165 #endif
166 
167  string mol_name;
168 
169  for (int nmol_type=0;nmol_type<nmol_types; nmol_type++) {
170 
171  mol_name = _NextKeyline(fl,WhiteSpace);
172  Molecule *mi = top.CreateMolecule(mol_name);
173 
174  int nreplica = 1;
175  line = _NextKeyInt(fl,WhiteSpace,"NUMMOL",nreplica);
176 
177 #ifdef DEBUG
178  cout << "Read from dlpoly file '" << _fname << "' : '" << mol_name << "' - '" << line << "' - " << nreplica << endl;
179 #endif
180 
181  line = _NextKeyInt(fl,WhiteSpace,"ATOMS",natoms);
182 
183 #ifdef DEBUG
184  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - " << natoms << endl;
185 #endif
186 
187  //read molecule
188  int id_map[natoms];
189  for (int i=0;i<natoms;){//i is altered in repeater loop
190  stringstream sl(_NextKeyline(fl,WhiteSpace));
191 
192 #ifdef DEBUG
193  cout << "Read atom specs from dlpoly topology : '" << sl.str() << "'" << endl;
194 #endif
195  string beadtype;
196  sl >> beadtype;
197  BeadType *type = top.GetOrCreateBeadType(beadtype);
198  double mass;
199  sl >> mass;
200  double charge;
201  sl >> charge;
202 
203  line = " ";
204  sl >> line; //rest of the atom line
205 
206  Tokenizer tok(line, WhiteSpace);
207  vector<string> fields;
208  tok.ToVector(fields);
209 
210 #ifdef DEBUG
211  cout << "Rest atom specs from dlpoly topology : '" << line << "'" << endl;
212 #endif
213 
214  int repeater=1;
215  if (fields.size() > 1) repeater=boost::lexical_cast<int>(fields[0]);
216 
217  for(int j=0;j<repeater;j++){
218 
219  string beadname = beadtype + "#" + boost::lexical_cast<string>(i+1);
220  Bead *bead = top.CreateBead(1, beadname, type, res->getId(), mass, charge);
221 
222  stringstream nm;
223  nm << bead->getResnr() + 1 << ":" << top.getResidue(bead->getResnr())->getName() << ":" << bead->getName();
224  mi->AddBead(bead, nm.str());
225  id_map[i]=bead->getId();
226  i++;
227 #ifdef DEBUG
228  cout << "Atom identification in maps '" << nm.str() << "'" << endl;
229 #endif
230  }
231  matoms += repeater;
232  }
233 
234  while (line != "FINISH"){
235 
236  stringstream nl(_NextKeyline(fl,WhiteSpace));
237  nl >> line;
238 
239 #ifdef DEBUG
240  cout << "Read unit type# from dlpoly topology : '" << nl.str() << "'" << endl;
241 #endif
242 
243  boost::to_upper(line);
244  line=line.substr(0,6);
245  if ((line == "BONDS")||(line == "ANGLES")||(line == "DIHEDR")) {
246  string type = line;
247  int count;
248  nl >> count;
249  for (int i=0;i<count;i++){
250 
251  stringstream sl(_NextKeyline(fl,WhiteSpace));
252 #ifdef DEBUG
253  cout << "Read unit specs from dlpoly topology : '" << sl.str() << "'" << endl;
254 #endif
255  sl >> line; //internal dlpoly bond/angle/dihedral function types are merely skipped (ignored)
256  int ids[4];
257  Interaction *ic=NULL;
258  sl >> ids[0];
259  sl >> ids[1];
260  if (type == "BONDS"){
261  ic = new IBond(id_map[ids[0]-1],id_map[ids[1]-1]); // -1 due to fortran vs c
262  } else if (type == "ANGLES"){
263  sl >> ids[2];
264  ic = new IAngle(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1]); // -1 due to fortran vs c
265  } else if (type.substr(0,6) == "DIHEDR"){
266  type = "DIHEDRALS";
267  sl >> ids[2];
268  sl >> ids[3];
269  ic = new IDihedral(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1],id_map[ids[3]-1]); // -1 due to fortran vs c
270  } else {
271  throw std::runtime_error("Error: type should be BONDS, ANGLES or DIHEDRALS");
272  }
273  // could one use bond/angle/dihedral function types for 1:1 mapping? (CG map overwrites ic->Group anyway)
274  //ic->setGroup(line);
275  ic->setGroup(type);
276  ic->setIndex(i);
277  ic->setMolecule(mi->getId());
278  top.AddBondedInteraction(ic);
279  mi->AddInteraction(ic);
280  }
281  }
282  }
283 
284 #ifdef DEBUG
285  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - done with '" << mol_name << "'" << endl;
286 #endif
287 
288  //replicate molecule
289  for (int replica=1;replica<nreplica;replica++){
290  Molecule *mi_replica = top.CreateMolecule(mol_name);
291  for(int i=0;i<mi->BeadCount();i++){
292  Bead *bead=mi->getBead(i);
293  BeadType *type = top.GetOrCreateBeadType(bead->Type()->getName());
294  string beadname=mi->getBeadName(i);
295  Bead *bead_replica = top.CreateBead(1, bead->getName(), type, res->getId(), bead->getM(), bead->getQ());
296  mi_replica->AddBead(bead_replica,beadname);
297  }
298  matoms+=mi->BeadCount();
299  InteractionContainer ics=mi->Interactions();
300  for(vector<Interaction *>::iterator ic=ics.begin(); ic!=ics.end(); ++ic) {
301  Interaction *ic_replica=NULL;
302  int offset = mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
303  if ((*ic)->BeadCount() == 2) {
304  ic_replica = new IBond((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset);
305  } else if ((*ic)->BeadCount() == 3) {
306  ic_replica = new IAngle((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset);
307  } else if ((*ic)->BeadCount() == 4) {
308  ic_replica = new IDihedral((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset,(*ic)->getBeadId(3)+offset);
309  } else {
310  throw std::runtime_error("Error: BeadCount not equal 2, 3 or 4");
311  }
312  ic_replica->setGroup((*ic)->getGroup());
313  ic_replica->setIndex((*ic)->getIndex());
314  ic_replica->setMolecule(mi_replica->getId());
315  top.AddBondedInteraction(ic_replica);
316  mi_replica->AddInteraction(ic_replica);
317  }
318  }
319  }
320  top.RebuildExclusions();
321  }
322 
323 #ifdef DEBUG
324  getline(fl, line); //is "close" found?
325  if(line=="close") {
326  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - done with topology" << endl;
327  }
328  else {
329  cout << "Read from dlpoly file '" << _fname << "' : 'EOF' - done with topology (directive 'close' not read!)" << endl;
330  }
331 #endif
332 
333  //we don't need the rest
334  fl.close();
335 
336  return true;
337 }
