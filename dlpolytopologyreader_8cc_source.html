36 string DLPOLYTopologyReader::_NextKeyline(ifstream &fs, const char* wsp)
37 {
38  string line;
39  size_t i_nws=0;
40 
41  do {
42  getline(fs, line);
43 
44  if (fs.eof())
45  throw std::runtime_error("Error: unexpected end of dlpoly file '" + _fname + "'");
46 
47  i_nws = line.find_first_not_of(wsp);
48  } while ( line.substr(i_nws,1) == "#" || line.substr(i_nws,1) == ";" );
49 
50  return line.substr(i_nws,line.size()-i_nws);
51 }

53 string DLPOLYTopologyReader::_NextKeyInt(ifstream &fs, const char* wsp, const string &word, int &ival)
54 {
55  stringstream sl(_NextKeyline(fs,wsp));
56  string line,sval;
57 
58  sl >> line; // allow user not to bother about the case
59  boost::to_upper(line);
60 
61  if( line.substr(0,word.size()) != word )
62  throw std::runtime_error("Error: unexpected line in dlpoly file " + _fname + ", expected '" + word + "' but got '" + line + "'");
63 
64  sl >> sval;
65 
66  size_t i_num = sval.find_first_of("0123456789"); // assume integer number straight after the only keyword
67 
68  if( i_num>0 )
69  throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
70 
71  ival = boost::lexical_cast<int>(sval);
72 
73  return sl.str();
74 }

76 bool DLPOLYTopologyReader::_isKeyInt(const string &line, const char* wsp, const string &word, int &ival)
77 {
78  Tokenizer tok(line,wsp); // split directives consisting of a few words - the keyword is the first one!
79  vector<string> fields;
80  tok.ToVector(fields);
81 
82  ival = 0;
83 
84  if( fields.size() < 2 ) return false;
85  //throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
86 
87  boost::to_upper(fields[0]);
88 
89  if (fields[0].substr(0,word.size()) != word )
90  throw std::runtime_error("Error: unexpected directive in dlpoly file '" + _fname + "', expected keyword '"+ word +"' but got '" + fields[0] + "'");
91 
92  size_t i_num=string::npos;
93 
94  int i=1;
95  do { // find integer number in the field with the lowest index (closest to the keyword)
96  i_num = fields[i++].find_first_of("0123456789");
97  } while ( i_num>0 && i<fields.size() );
98 
99  if( i_num>0 ) return false;
100  //throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
101 
102  ival = boost::lexical_cast<int>(fields[i-1]);
103 
104  return true;
105 }

112  int mavecs=0;

116  bool hasVs=false;
117  bool hasFs=false;

122  boost::filesystem::path filepath(file.c_str());
123 
124  string line;
125 
126  // TODO: fix residue naming / assignment - DL_POLY has no means to recognise residues!
127  Residue *res = top.CreateResidue("no");
128 
129  if ( boost::filesystem::basename(filepath).size() == 0 ) {
130  if (filepath.parent_path().string().size() == 0) {
131  _fname="FIELD"; // DL_POLY uses fixed file names in current/working directory
132  } else {
133  _fname=filepath.parent_path().string() + "/FIELD";
134  }
135  } else {
136  _fname=file;
137  }
138 
139  fl.open(_fname.c_str());
140 
141  if (!fl.is_open()){
142  throw std::runtime_error("Error on opening dlpoly file '" + _fname + "'");
143  } else {
144 
145  line = _NextKeyline(fl,WSP); //read title line and skip it
146  line = _NextKeyline(fl,WSP); //read next directive line
147  boost::to_upper(line);
148 
149  if( line.substr(0,4)=="UNIT" ) { //skip 'unit' line
150  line = _NextKeyline(fl,WSP); //read next directive line
151  boost::to_upper(line);
152  }
153 
154  if( line.substr(0,4)=="NEUT" ) { //skip 'neutral groups' line (DL_POLY Classic FIELD format)
155  line = _NextKeyline(fl,WSP); //look for next directive line
156  boost::to_upper(line);
157  }
158 
159  int nmol_types;
160 
161  if( !_isKeyInt(line,WSP,"MOLEC",nmol_types) )
162  throw std::runtime_error("Error: missing integer number in directive '" + line + "' in topology file '"+ _fname +"'");
163 
164 #ifdef DEBUG
165  cout << "Read from topology file " << _fname << " : '" << line << "' - " << nmol_types << endl;
166 #endif
167 
168  string mol_name;
169 
170  int id=0;
171  for (int nmol_type=0;nmol_type<nmol_types; nmol_type++) {
172 
173  mol_name = _NextKeyline(fl,WSP);
174  Molecule *mi = top.CreateMolecule(mol_name);
175 
176  int nreplica = 1;
177  line = _NextKeyInt(fl,WSP,"NUMMOL",nreplica);
178 
179 #ifdef DEBUG
180  cout << "Read from topology file " << _fname << " : '" << mol_name << "' - '" << line << "' - " << nreplica << endl;
181 #endif
182 
183  line = _NextKeyInt(fl,WSP,"ATOMS",natoms);
184 
185 #ifdef DEBUG
186  cout << "Read from topology file " << _fname << " : '" << line << "' - " << natoms << endl;
187 #endif
188 
189  //read molecule
190  int id_map[natoms];
191  for (int i=0;i<natoms;){//i is altered in repeater loop
192  string beadtype;
193  fl >> beadtype;
194  BeadType *type = top.GetOrCreateBeadType(beadtype);
195  double mass;
196  fl >> mass;
197  double charge;
198  fl >> charge;
199 
200  getline(fl,line); //rest of the atom line
201 
202  Tokenizer tok(line, WSP);
203  vector<string> fields;
204  tok.ToVector(fields);
205 
206  int repeater=1;
207  if (fields.size() > 1) repeater=boost::lexical_cast<int>(fields[0]);
208 
209  for(int j=0;j<repeater;j++){
210 
211  string beadname = beadtype + "#" + boost::lexical_cast<string>(i+1);
212  Bead *bead = top.CreateBead(1, beadname, type, res->getId(), mass, charge);
213 
214  stringstream nm;
215  nm << bead->getResnr() + 1 << ":" << top.getResidue(bead->getResnr())->getName() << ":" << bead->getName();
216  mi->AddBead(bead, nm.str());
217  id_map[i]=bead->getId();
218  i++;
219 #ifdef DEBUG
220  cout << "Atom identification in maps '" << nm.str() << "'" << endl;
221 #endif
222  }
223  matoms += repeater;
224  }
225 
226  while (line != "FINISH"){
227  if ((line == "BONDS")||(line == "ANGLES")||(line == "DIHEDRALS")) {
228  string type = line;
229  int count;
230  fl >> count;
231  for (int i=0;i<count;i++){
232  fl >> line; //bond/angle/dih type not used
233  int ids[4];
234  Interaction *ic;
235  fl >> ids[0]; fl>>ids[1];
236  if (type == "BONDS"){
237  ic = new IBond(id_map[ids[0]-1],id_map[ids[1]-1]); // -1 due to fortran vs c
238  } else if (type == "ANGLES"){
239  fl >> ids[2];
240  ic = new IAngle(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1]); // -1 due to fortran vs c
241  } else if (type == "DIHEDRALS"){
242  fl >> ids[2]; fl >> ids[3];
243  ic = new IDihedral(id_map[ids[0]-1],id_map[ids[1]-1],id_map[ids[2]-1],id_map[ids[3]-1]); // -1 due to fortran vs c
244  }
245  ic->setGroup(type);
246  ic->setIndex(i);
247  ic->setMolecule(mi->getId());
248  top.AddBondedInteraction(ic);
249  mi->AddInteraction(ic);
250  getline(fl,line);
251  }
252  }
253  fl >> line;
254  boost::to_upper(line);
255  if (fl.eof())
256  throw std::runtime_error("Error: unexpected end of dlpoly file " + _fname + " while scanning for kerword 'finish'");
257  }
258 
259 #ifdef DEBUG
260  cout << "Read from topology file " << _fname << " : '" << line << "' - done with '" << mol_name << "'" << endl;
261 #endif
262 
263  getline(fl, line); //rest of the FINISH line
264 
265  //replicate molecule
266  for (int replica=1;replica<nreplica;replica++){
267  Molecule *mi_replica = top.CreateMolecule(mol_name);
268  for(int i=0;i<mi->BeadCount();i++){
269  Bead *bead=mi->getBead(i);
270  BeadType *type = top.GetOrCreateBeadType(bead->Type()->getName());
271  string beadname=mi->getBeadName(i);
272  Bead *bead_replica = top.CreateBead(1, bead->getName(), type, res->getId(), bead->getM(), bead->getQ());
273  mi_replica->AddBead(bead_replica,beadname);
274  }
275  matoms+=mi->BeadCount();
276  InteractionContainer ics=mi->Interactions();
277  for(vector<Interaction *>::iterator ic=ics.begin(); ic!=ics.end(); ++ic) {
278  Interaction *ic_replica;
279  int offset = mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
280  if ((*ic)->BeadCount() == 2) {
281  ic_replica = new IBond((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset);
282  } else if ((*ic)->BeadCount() == 3) {
283  ic_replica = new IAngle((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset);
284  } else if ((*ic)->BeadCount() == 4) {
285  ic_replica = new IDihedral((*ic)->getBeadId(0)+offset,(*ic)->getBeadId(1)+offset,(*ic)->getBeadId(2)+offset,(*ic)->getBeadId(3)+offset);
286  }
287  ic_replica->setGroup((*ic)->getGroup());
288  ic_replica->setIndex((*ic)->getIndex());
289  ic_replica->setMolecule(mi_replica->getId());
290  top.AddBondedInteraction(ic_replica);
291  mi_replica->AddInteraction(ic_replica);
292  }
293  }
294  }
295  top.RebuildExclusions();
296  }
297 
298 #ifdef DEBUG
299  getline(fl, line); //is "close" found?
300  if(line=="close") {
301  cout << "Read from topology file " << _fname << " : '" << line << "' - done with topology" << endl;
302  }
303  else {
304  cout << "Read from topology file " << _fname << " : 'EOF' - done with topology (directive 'close' not read!)" << endl;
305  }
306 #endif
307 
308  //we don't need the rest
309  fl.close();
310 
311  string filename;
312 
313  if ( boost::filesystem::basename(filepath).size() == 0 ) {
314  if (filepath.parent_path().string().size() == 0) {
315  filename="CONFIG";
316  } else {
317  filename=filepath.parent_path().string() + "/CONFIG";
318  }
319  } else {
320  filename=filepath.parent_path().string()+boost::filesystem::basename(filepath)+".dlpc";
321  cout << "NOTE: explicit dlpoly topology file name given '" << file << "', so trying to read boundary conditions from CONFIG file named '" << filename << "'" << endl;
322  }
323 
324  fl.open(filename.c_str());
325 
326  if(fl.is_open()) {
327  string line;
328 
329  getline(fl, line); //title
330 
331 #ifdef DEBUG
332  cout << "Read from initial configuration file : '" << line << "' - header" << endl;
333 #endif
334 
335  getline(fl, line); // 2nd header line
336 
337 #ifdef DEBUG
338  cout << "Read from initial configuration file : '" << line << "' - directives line" << endl;
339 #endif
340 
341  Tokenizer tok(line, WSP);
342  vector<string> fields;
343  tok.ToVector(fields);
344 
345  if( fields.size() < 3 )
346  throw std::runtime_error("Error: too few directive switches (<3) in the initial configuration (check its 2-nd line)");
347 
348  mavecs = boost::lexical_cast<int>(fields[0]);
349  mpbct = boost::lexical_cast<int>(fields[1]);
350  natoms = boost::lexical_cast<int>(fields[2]);
351 
352  hasVs = (mavecs > 0); // 1 or 2 => in DL_POLY frame velocity vector follows coords for each atom/bead
353  hasFs = (mavecs > 1); // 2 => in DL_POLY frame force vector follows velocities for each atom/bead
354 
355  top.SetHasVel(hasVs);
356  top.SetHasForce(hasFs);
357 
358  if(natoms != matoms)
359 #ifdef DEBUG
360  cout << "Warning: N of atoms/beads in initial configuration & topology differ: " << natoms << " =?= " << matoms << endl;
361 #else
362  throw std::runtime_error("Error: N of atoms/beads in initial configuration & topology differ " +
363  boost::lexical_cast<string>(natoms) + " vs " + boost::lexical_cast<string>(matoms));
364 #endif
365 
366  vec box_vectors[3];
367  for (int i=0;i<3;i++){ // read 3 box lines
368  getline(fl, line);
369 
370 #ifdef DEBUG
371  cout << "Read from initial configuration file : '" << line << "' - box vector # " << i+1 << " (Angs)" << endl;
372 #endif
373 
374  if(fl.eof())
375  throw std::runtime_error("Error: unexpected EOF in dlpoly file " + filename +", when reading box vector " +
376  boost::lexical_cast<string>(i));
377 
378  Tokenizer tok(line, WSP);
379  vector<double> fields;
380  tok.ConvertToVector<double>(fields);
381  box_vectors[i]=vec(fields[0],fields[1],fields[2]);
382 
383 #ifdef DEBUG
384  cout << "Read from initial configuration file : '" << fixed << setprecision(10) << setw(20) << fields[0] << setw(20) << fields[1] << setw(20) << fields[2] << "' - box vector # " << i+1 << " (Angs)" << endl;
385 #endif
386  }
387  matrix box(box_vectors[0],box_vectors[1],box_vectors[2]);
388 
389  if(mpbct == 0) {
390  pbc_type=BoundaryCondition::typeOpen;
391  }
392  else if(mpbct == 1 || mpbct == 2 ) {
393  pbc_type=BoundaryCondition::typeOrthorhombic;
394  }
395  else if(mpbct == 3) {
396  pbc_type=BoundaryCondition::typeTriclinic;

402

405 #endif
406  }
407  else {
408  cout << "NOTE: could not open dlpoly file " << filename << ", so no PBC set in topology - assuming 'open box'" << endl;
409  }
410 
411  return true;
412 }
