48string DLPOLYTopologyReader::NextKeyline_(ifstream &fs, const char *wspace)
49// function to find and read the next line starting with a keyword/directive
50// (skipping comments starting with "#" or ";") NOTE: the line is returned
51// case-preserved, not to alter molecule/atom names (consider if --no-map is
52// used)
53{
54 string line;
55 size_t i_nws = 0;
56
57 do {
58 tools::getline(fs, line);
59
60 if (fs.eof()) {
61 throw std::runtime_error("Error: unexpected end of dlpoly file '" +
62 fname_ + "'");
63 }
64
65 i_nws = line.find_first_not_of(wspace);
66 } while (line.substr(i_nws, 1) == "#" || line.substr(i_nws, 1) == ";");
67
68 return line.substr(i_nws, line.size() - i_nws);
69}

71string DLPOLYTopologyReader::NextKeyInt_(ifstream &fs, const char *wspace,
72 const string &word, Index &ival)
73// function to read the next line containing only a given keyword and an integer
74// value after it (only skipping comments!) NOTE: this function must only be
75// called when the next directive line has to contain the given keyword and an
76// integer value
77{
78 stringstream sl(NextKeyline_(fs, wspace));
79 string line, sval;
80
81 sl >> line; // allow user not to bother about the case
82 boost::to_upper(line);
83
84 if (line.substr(0, word.size()) != word) {
85 throw std::runtime_error("Error: unexpected line from dlpoly file '" +
86 fname_ + "', expected '" + word + "' but got '" +
87 line + "'");
88 }
89
90 sl >> sval;
91
92 size_t i_num = sval.find_first_of(
93 "0123456789"); // assume integer number straight after the only keyword
94
95 if (i_num > 0) {
96 throw std::runtime_error("Error: missing integer number in directive '" +
97 line + "' in topology file '" + fname_ + "'");
98 }
99
100 ival = boost::lexical_cast<Index>(sval);
101
102 return sl.str();
103}

105bool DLPOLYTopologyReader::isKeyInt_(const string &line, const char *wspace,
106 const string &word, Index &ival)
107// function to check if the given (last read) directive line starts with a given
108// keyword and has an integer value at the end NOTE: comments are allowed beyond
109// the integer (anything after the first integer is ignored)
110{
111 // split directives consisting of a few words: the sought keyword must be the
112 // first one!
113
114 vector<string> fields = Tokenizer(line, wspace).ToVector();
115
116 ival = 0;
117
118 if (fields.size() < 2) {
119 return false;
120 }
121
122 boost::to_upper(fields[0]);
123
124 if (fields[0].substr(0, word.size()) != word) {
125 throw std::runtime_error("Error: unexpected directive from dlpoly file '" +
126 fname_ + "', expected keyword '" + word +
127 "' but got '" + fields[0] + "'");
128 }
129
130 size_t i_num = string::npos;
131
132 Index i = 1;
133 do { // find integer number in the field with the lowest index (closest to
134 // the keyword)
135 i_num = fields[i++].find_first_of("0123456789");
136 } while (i_num > 0 && i < Index(fields.size()));
137
138 if (i_num > 0) {
139 return false;
140 }
141
142 ival = boost::lexical_cast<Index>(fields[i - 1]);
143
144 return true;
145}

147bool DLPOLYTopologyReader::ReadTopology(string file, Topology &top) {
148
149 Index natoms = 0;
150
151 boost::filesystem::path filepath(file.c_str());
152
153 string line;
154
155 // TODO: fix residue naming / assignment - DL_POLY has no means to recognise
156 // residues!
157 const Residue &res = top.CreateResidue("no");
158
159 if (boost::filesystem::basename(filepath).size() == 0) {
160 if (filepath.parent_path().string().size() == 0) {
161 fname_ = "FIELD"; // DL_POLY uses fixed file names in current/working
162 // directory
163 } else {
164 fname_ = filepath.parent_path().string() + "/FIELD";
165 }
166 } else {
167 fname_ = file;
168 }
169 std::ifstream fl;
170 fl.open(fname_);
171
172 if (!fl.is_open()) {
173 throw std::runtime_error("Error on opening dlpoly file '" + fname_ + "'");
174 } else {
175 const char *WhiteSpace = " \t";
176 NextKeyline_(fl, WhiteSpace); // read title line and skip it
177 line = NextKeyline_(fl, WhiteSpace); // read next directive line
178 boost::to_upper(line);
179
180 if (line.substr(0, 4) == "UNIT") { // skip 'unit' line
181 line = NextKeyline_(fl, WhiteSpace); // read next directive line
182 boost::to_upper(line);
183 }
184
185 if (line.substr(0, 4) == "NEUT") { // skip 'neutral groups' line (DL_POLY
186 // Classic FIELD format)
187 line = NextKeyline_(fl, WhiteSpace); // look for next directive line
188 boost::to_upper(line);
189 }
190
191 Index nmol_types;
192
193 if (!isKeyInt_(line, WhiteSpace, "MOLEC", nmol_types)) {
194 throw std::runtime_error("Error: missing integer number in directive '" +
195 line + "' in topology file '" + fname_ + "'");
196 }
197
198#ifndef NDEBUG
199 cout << "Read from dlpoly file '" << fname_ << "' : '" << line << "' - "
200 << nmol_types << endl;
201#endif
202
203 string mol_name;
204
205 for (Index nmol_type = 0; nmol_type < nmol_types; nmol_type++) {
206
207 mol_name = NextKeyline_(fl, WhiteSpace);
208 Molecule *mi = top.CreateMolecule(mol_name);
209
210 Index nreplica = 1;
211 line = NextKeyInt_(fl, WhiteSpace, "NUMMOL", nreplica);
212
213#ifndef NDEBUG
214 cout << "Read from dlpoly file '" << fname_ << "' : '" << mol_name
215 << "' - '" << line << "' - " << nreplica << endl;
216#endif
217
218 line = NextKeyInt_(fl, WhiteSpace, "ATOMS", natoms);
219
220#ifndef NDEBUG
221 cout << "Read from dlpoly file '" << fname_ << "' : '" << line << "' - "
222 << natoms << endl;
223#endif
224 std::vector<Index> id_map(natoms);
225 for (Index i = 0; i < natoms;) { // i is altered in repeater loop
226 stringstream sl(NextKeyline_(fl, WhiteSpace));
227
228#ifndef NDEBUG
229 cout << "Read atom specs from dlpoly topology : '" << sl.str() << "'"
230 << endl;
231#endif
232 string beadtype;
233 sl >> beadtype;
234 if (!top.BeadTypeExist(beadtype)) {
235 top.RegisterBeadType(beadtype);
236 }
237 double mass;
238 sl >> mass;
239 double charge;
240 sl >> charge;
241
242 line = " ";
243 sl >> line; // rest of the atom line
244
245 vector<string> fields = Tokenizer(line, WhiteSpace).ToVector();
246
247#ifndef NDEBUG
248 cout << "Rest atom specs from dlpoly topology : '" << line << "'"
249 << endl;
250#endif
251
252 Index repeater = 1;
253 if (fields.size() > 1) {
254 repeater = boost::lexical_cast<Index>(fields[0]);
255 }
256
257 for (Index j = 0; j < repeater; j++) {
258
259 string beadname = beadtype + "#" + boost::lexical_cast<string>(i + 1);
260 Bead *bead = top.CreateBead(Bead::spherical, beadname, beadtype,
261 res.getId(), mass, charge);
262
263 stringstream nm;
264 nm << bead->getResnr() + 1 << ":"
265 << top.getResidue(bead->getResnr()).getName() << ":"
266 << bead->getName();
267 mi->AddBead(bead, nm.str());
268 id_map[i] = bead->getId();
269 i++;
270#ifndef NDEBUG
271 cout << "Atom identification in maps '" << nm.str() << "'" << endl;
272#endif
273 }
274 }
275
276 while (line != "FINISH") {
277
278 stringstream nl(NextKeyline_(fl, WhiteSpace));
279 nl >> line;
280
281#ifndef NDEBUG
282 cout << "Read unit type# from dlpoly topology : '" << nl.str() << "'"
283 << endl;
284#endif
285
286 boost::to_upper(line);
287 line = line.substr(0, 6);
288 if ((line == "BONDS") || (line == "ANGLES") || (line == "DIHEDR")) {
289 string type = line;
290 Index count;
291 nl >> count;
292 for (Index i = 0; i < count; i++) {
293
294 stringstream sl(NextKeyline_(fl, WhiteSpace));
295#ifndef NDEBUG
296 cout << "Read unit specs from dlpoly topology : '" << sl.str()
297 << "'" << endl;
298#endif
299 sl >> line; // internal dlpoly bond/angle/dihedral function types
300 // are merely skipped (ignored)
301 Index ids[4];
302 Interaction *ic = nullptr;
303 sl >> ids[0];
304 sl >> ids[1];
305 if (type == "BONDS") {
306 ic = new IBond(id_map[ids[0] - 1],
307 id_map[ids[1] - 1]); // -1 due to fortran vs c
308 } else if (type == "ANGLES") {
309 sl >> ids[2];
310 ic = new IAngle(id_map[ids[0] - 1], id_map[ids[1] - 1],
311 id_map[ids[2] - 1]); // -1 due to fortran vs c
312 } else if (type.substr(0, 6) == "DIHEDR") {
313 type = "DIHEDRALS";
314 sl >> ids[2];
315 sl >> ids[3];
316 ic = new IDihedral(id_map[ids[0] - 1], id_map[ids[1] - 1],
317 id_map[ids[2] - 1],
318 id_map[ids[3] - 1]); // -1 due to fortran vs c
319 } else {
320 throw std::runtime_error(
321 "Error: type should be BONDS, ANGLES or DIHEDRALS");
322 }
323 // could one use bond/angle/dihedral function types for 1:1 mapping?
324 // (CG map overwrites ic->Group anyway)
325 // ic->setGroup(line);
326 ic->setGroup(type);
327 ic->setIndex(i);
328 ic->setMolecule(mi->getId());
329 top.AddBondedInteraction(ic);
330 mi->AddInteraction(ic);
331 }
332 }
333 }
334
335#ifndef NDEBUG
336 cout << "Read from dlpoly file '" << fname_ << "' : '" << line
337 << "' - done with '" << mol_name << "'" << endl;
338#endif
339
340 // replicate molecule
341 for (Index replica = 1; replica < nreplica; replica++) {
342 Molecule *mi_replica = top.CreateMolecule(mol_name);
343 for (Index i = 0; i < mi->BeadCount(); i++) {
344 Bead *bead = mi->getBead(i);
345 string type = bead->getType();
346 Bead *bead_replica =
347 top.CreateBead(Bead::spherical, bead->getName(), type,
348 res.getId(), bead->getMass(), bead->getQ());
349 mi_replica->AddBead(bead_replica, bead->getName());
350 }
351 InteractionContainer ics = mi->Interactions();
352 for (auto &ic : ics) {
353 Interaction *ic_replica = nullptr;
354 Index offset =
355 mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
356 if (ic->BeadCount() == 2) {
357 ic_replica =
358 new IBond(ic->getBeadId(0) + offset, ic->getBeadId(1) + offset);
359 } else if (ic->BeadCount() == 3) {
360 ic_replica =
361 new IAngle(ic->getBeadId(0) + offset, ic->getBeadId(1) + offset,
362 ic->getBeadId(2) + offset);
363 } else if (ic->BeadCount() == 4) {
364 ic_replica = new IDihedral(
365 ic->getBeadId(0) + offset, ic->getBeadId(1) + offset,
366 ic->getBeadId(2) + offset, ic->getBeadId(3) + offset);
367 } else {
368 throw std::runtime_error("Error: BeadCount not equal 2, 3 or 4");
369 }
370 ic_replica->setGroup(ic->getGroup());
371 ic_replica->setIndex(ic->getIndex());
372 ic_replica->setMolecule(mi_replica->getId());
373 top.AddBondedInteraction(ic_replica);
374 mi_replica->AddInteraction(ic_replica);
375 }
376 }
377 }
378 top.RebuildExclusions();
379 }
380
381#ifndef NDEBUG
382 tools::getline(fl, line); // is "close" found?
383 if (line == "close") {
384 cout << "Read from dlpoly file '" << fname_ << "' : '" << line
385 << "' - done with topology" << endl;
386 } else {
387 cout << "Read from dlpoly file '" << fname_
388 << "' : 'EOF' - done with topology (directive 'close' not read!)"
389 << endl;
390 }
391#endif
392
393 // we don't need the rest
394 fl.close();
395
396 return true;
397}
