40 string DLPOLYTopologyReader::_NextKeyline(ifstream &fs, const char *wspace)
41 // function to find and read the next line starting with a keyword/directive
42 // (skipping comments starting with "#" or ";") NOTE: the line is returned
43 // case-preserved, not to alter molecule/atom names (consider if --no-map is
44 // used)
45 {
46  string line;
47  size_t i_nws = 0;
48 
49  do {
50  getline(fs, line);
51 
52  if (fs.eof())
53  throw std::runtime_error("Error: unexpected end of dlpoly file '" +
54  _fname + "'");
55 
56  i_nws = line.find_first_not_of(wspace);
57  } while (line.substr(i_nws, 1) == "#" || line.substr(i_nws, 1) == ";");
58 
59  return line.substr(i_nws, line.size() - i_nws);
60 }

62 string DLPOLYTopologyReader::_NextKeyInt(ifstream &fs, const char *wspace,
63  const string &word, int &ival)
64 // function to read the next line containing only a given keyword and an integer
65 // value after it (only skipping comments!) NOTE: this function must only be
66 // called when the next directive line has to contain the given keyword and an
67 // integer value
68 {
69  stringstream sl(_NextKeyline(fs, wspace));
70  string line, sval;
71 
72  sl >> line; // allow user not to bother about the case
73  boost::to_upper(line);
74 
75  if (line.substr(0, word.size()) != word)
76  throw std::runtime_error("Error: unexpected line from dlpoly file '" +
77  _fname + "', expected '" + word + "' but got '" +
78  line + "'");
79 
80  sl >> sval;
81 
82  size_t i_num = sval.find_first_of(
83  "0123456789"); // assume integer number straight after the only keyword
84 
85  if (i_num > 0)
86  throw std::runtime_error("Error: missing integer number in directive '" +
87  line + "' in topology file '" + _fname + "'");
88 
89  ival = boost::lexical_cast<int>(sval);
90 
91  return sl.str();
92 }

94 bool DLPOLYTopologyReader::_isKeyInt(const string &line, const char *wspace,
95  const string &word, int &ival)
96 // function to check if the given (last read) directive line starts with a given
97 // keyword and has an integer value at the end NOTE: comments are allowed beyond
98 // the integer (anything after the first integer is ignored)
99 {
100  // split directives consisting of a few words: the sought keyword must be the
101  // first one!
102  Tokenizer tok(line, wspace);
103  vector<string> fields;
104  tok.ToVector(fields);
105 
106  ival = 0;
107 
108  if (fields.size() < 2) return false;
109 
110  boost::to_upper(fields[0]);
111 
112  if (fields[0].substr(0, word.size()) != word)
113  throw std::runtime_error("Error: unexpected directive from dlpoly file '" +
114  _fname + "', expected keyword '" + word +
115  "' but got '" + fields[0] + "'");
116 
117  size_t i_num = string::npos;
118 
119  unsigned int i = 1;
120  do { // find integer number in the field with the lowest index (closest to
121  // the keyword)
122  i_num = fields[i++].find_first_of("0123456789");
123  } while (i_num > 0 && i < fields.size());
124 
125  if (i_num > 0) return false;
126 
127  ival = boost::lexical_cast<int>(fields[i - 1]);
128 
129  return true;
130 }

132 bool DLPOLYTopologyReader::ReadTopology(string file, Topology &top) {
133  const char *WhiteSpace = " \t";
134 
135  int matoms = 0;
136  int natoms = 0;
137 
138  std::ifstream fl;
139  boost::filesystem::path filepath(file.c_str());
140 
141  string line;
142 
143  // TODO: fix residue naming / assignment - DL_POLY has no means to recognise
144  // residues!
145  Residue *res = top.CreateResidue("no");
146 
147  if (boost::filesystem::basename(filepath).size() == 0) {
148  if (filepath.parent_path().string().size() == 0) {
149  _fname = "FIELD"; // DL_POLY uses fixed file names in current/working
150  // directory
151  } else {
152  _fname = filepath.parent_path().string() + "/FIELD";
153  }
154  } else {
155  _fname = file;
156  }
157 
158  fl.open(_fname.c_str());
159 
160  if (!fl.is_open()) {
161  throw std::runtime_error("Error on opening dlpoly file '" + _fname + "'");
162  } else {
163 
164  line = _NextKeyline(fl, WhiteSpace); // read title line and skip it
165  line = _NextKeyline(fl, WhiteSpace); // read next directive line
166  boost::to_upper(line);
167 
168  if (line.substr(0, 4) == "UNIT") { // skip 'unit' line
169  line = _NextKeyline(fl, WhiteSpace); // read next directive line
170  boost::to_upper(line);
171  }
172 
173  if (line.substr(0, 4) == "NEUT") { // skip 'neutral groups' line (DL_POLY
174  // Classic FIELD format)
175  line = _NextKeyline(fl, WhiteSpace); // look for next directive line
176  boost::to_upper(line);
177  }
178 
179  int nmol_types;
180 
181  if (!_isKeyInt(line, WhiteSpace, "MOLEC", nmol_types))
182  throw std::runtime_error("Error: missing integer number in directive '" +
183  line + "' in topology file '" + _fname + "'");
184 
185 #ifdef DEBUG
186  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - "
187  << nmol_types << endl;
188 #endif
189 
190  string mol_name;
191 
192  for (int nmol_type = 0; nmol_type < nmol_types; nmol_type++) {
193 
194  mol_name = _NextKeyline(fl, WhiteSpace);
195  Molecule *mi = top.CreateMolecule(mol_name);
196 
197  int nreplica = 1;
198  line = _NextKeyInt(fl, WhiteSpace, "NUMMOL", nreplica);
199 
200 #ifdef DEBUG
201  cout << "Read from dlpoly file '" << _fname << "' : '" << mol_name
202  << "' - '" << line << "' - " << nreplica << endl;
203 #endif
204 
205  line = _NextKeyInt(fl, WhiteSpace, "ATOMS", natoms);
206 
207 #ifdef DEBUG
208  cout << "Read from dlpoly file '" << _fname << "' : '" << line << "' - "
209  << natoms << endl;
210 #endif
211 
212  // read molecule
213  int id_map[natoms];
214  for (int i = 0; i < natoms;) { // i is altered in repeater loop
215  stringstream sl(_NextKeyline(fl, WhiteSpace));
216 
217 #ifdef DEBUG
218  cout << "Read atom specs from dlpoly topology : '" << sl.str() << "'"
219  << endl;
220 #endif
221  string beadtype;
222  sl >> beadtype;
223  if (!top.BeadTypeExist(beadtype)) {
224  top.RegisterBeadType(beadtype);
225  }
226  double mass;
227  sl >> mass;
228  double charge;
229  sl >> charge;
230 
231  line = " ";
232  sl >> line; // rest of the atom line
233 
234  Tokenizer tok(line, WhiteSpace);
235  vector<string> fields;
236  tok.ToVector(fields);
237 
238 #ifdef DEBUG
239  cout << "Rest atom specs from dlpoly topology : '" << line << "'"
240  << endl;
241 #endif
242 
243  int repeater = 1;
244  if (fields.size() > 1) repeater = boost::lexical_cast<int>(fields[0]);
245 
246  for (int j = 0; j < repeater; j++) {
247 
248  string beadname = beadtype + "#" + boost::lexical_cast<string>(i + 1);
249  Bead *bead =
250  top.CreateBead(1, beadname, beadtype, res->getId(), mass, charge);
251 
252  stringstream nm;
253  nm << bead->getResnr() + 1 << ":"
254  << top.getResidue(bead->getResnr())->getName() << ":"
255  << bead->getName();
256  mi->AddBead(bead, nm.str());
257  id_map[i] = bead->getId();
258  i++;
259 #ifdef DEBUG
260  cout << "Atom identification in maps '" << nm.str() << "'" << endl;
261 #endif
262  }
263  matoms += repeater;
264  }
265 
266  while (line != "FINISH") {
267 
268  stringstream nl(_NextKeyline(fl, WhiteSpace));
269  nl >> line;
270 
271 #ifdef DEBUG
272  cout << "Read unit type# from dlpoly topology : '" << nl.str() << "'"
273  << endl;
274 #endif
275 
276  boost::to_upper(line);
277  line = line.substr(0, 6);
278  if ((line == "BONDS") || (line == "ANGLES") || (line == "DIHEDR")) {
279  string type = line;
280  int count;
281  nl >> count;
282  for (int i = 0; i < count; i++) {
283 
284  stringstream sl(_NextKeyline(fl, WhiteSpace));
285 #ifdef DEBUG
286  cout << "Read unit specs from dlpoly topology : '" << sl.str()
287  << "'" << endl;
288 #endif
289  sl >> line; // internal dlpoly bond/angle/dihedral function types
290  // are merely skipped (ignored)
291  int ids[4];
292  Interaction *ic = NULL;
293  sl >> ids[0];
294  sl >> ids[1];
295  if (type == "BONDS") {
296  ic = new IBond(id_map[ids[0] - 1],
297  id_map[ids[1] - 1]); // -1 due to fortran vs c
298  } else if (type == "ANGLES") {
299  sl >> ids[2];
300  ic = new IAngle(id_map[ids[0] - 1], id_map[ids[1] - 1],
301  id_map[ids[2] - 1]); // -1 due to fortran vs c
302  } else if (type.substr(0, 6) == "DIHEDR") {
303  type = "DIHEDRALS";
304  sl >> ids[2];
305  sl >> ids[3];
306  ic = new IDihedral(id_map[ids[0] - 1], id_map[ids[1] - 1],
307  id_map[ids[2] - 1],
308  id_map[ids[3] - 1]); // -1 due to fortran vs c
309  } else {
310  throw std::runtime_error(
311  "Error: type should be BONDS, ANGLES or DIHEDRALS");
312  }
313  // could one use bond/angle/dihedral function types for 1:1 mapping?
314  // (CG map overwrites ic->Group anyway)
315  // ic->setGroup(line);
316  ic->setGroup(type);
317  ic->setIndex(i);
318  ic->setMolecule(mi->getId());
319  top.AddBondedInteraction(ic);
320  mi->AddInteraction(ic);
321  }
322  }
323  }
324 
325 #ifdef DEBUG
326  cout << "Read from dlpoly file '" << _fname << "' : '" << line
327  << "' - done with '" << mol_name << "'" << endl;
328 #endif
329 
330  // replicate molecule
331  for (int replica = 1; replica < nreplica; replica++) {
332  Molecule *mi_replica = top.CreateMolecule(mol_name);
333  for (int i = 0; i < mi->BeadCount(); i++) {
334  Bead *bead = mi->getBead(i);
335  string type = bead->getType();
336  Bead *bead_replica =
337  top.CreateBead(1, bead->getName(), type, res->getId(),
338  bead->getMass(), bead->getQ());
339  mi_replica->AddBead(bead_replica, bead->getName());
340  }
341  matoms += mi->BeadCount();
342  InteractionContainer ics = mi->Interactions();
343  for (vector<Interaction *>::iterator ic = ics.begin(); ic != ics.end();
344  ++ic) {
345  Interaction *ic_replica = NULL;
346  int offset =
347  mi_replica->getBead(0)->getId() - mi->getBead(0)->getId();
348  if ((*ic)->BeadCount() == 2) {
349  ic_replica = new IBond((*ic)->getBeadId(0) + offset,
350  (*ic)->getBeadId(1) + offset);
351  } else if ((*ic)->BeadCount() == 3) {
352  ic_replica = new IAngle((*ic)->getBeadId(0) + offset,
353  (*ic)->getBeadId(1) + offset,
354  (*ic)->getBeadId(2) + offset);
355  } else if ((*ic)->BeadCount() == 4) {
356  ic_replica = new IDihedral(
357  (*ic)->getBeadId(0) + offset, (*ic)->getBeadId(1) + offset,
358  (*ic)->getBeadId(2) + offset, (*ic)->getBeadId(3) + offset);
359  } else {
360  throw std::runtime_error("Error: BeadCount not equal 2, 3 or 4");
361  }
362  ic_replica->setGroup((*ic)->getGroup());
363  ic_replica->setIndex((*ic)->getIndex());
364  ic_replica->setMolecule(mi_replica->getId());
365  top.AddBondedInteraction(ic_replica);
366  mi_replica->AddInteraction(ic_replica);
367  }
368  }
369  }
370  top.RebuildExclusions();
371  }
372 
373 #ifdef DEBUG
374  getline(fl, line); // is "close" found?
375  if (line == "close") {
376  cout << "Read from dlpoly file '" << _fname << "' : '" << line
377  << "' - done with topology" << endl;
378  } else {
379  cout << "Read from dlpoly file '" << _fname
380  << "' : 'EOF' - done with topology (directive 'close' not read!)"
381  << endl;
382  }
383 #endif
384 
385  // we don't need the rest
386  fl.close();
387 
388  return true;
389 }
