66  if (this->get("name") != "xtp") {
67  this->check_mandatory_keyword("executable");

69  for (const auto& x : keywords) {
70  this->check_mandatory_keyword(x);

72  std::stringstream stream;
73  // Check that the input keys are valid

76  this->_general_properties.cend(), pair.first);
77 
78  if (it == this->_general_properties.cend()) {

80  << "Keywords must be one of the following:\n";
81  for (const std::string& key : this->_general_properties) {
