141 void Diffusion::LoadGraph() {
142 
143  Database db;
144  db.Open( _filename );
145  cout << " Loading graph from " << _filename << endl;
146  Statement *stmt = db.Prepare("SELECT _id, name FROM segments;");
147 
148  while (stmt->Step() != SQLITE_DONE) {
149  int id = stmt->Column<int>(0);
150  string name = stmt->Column<string>(1);
151  node_t *n =new node_t(id);
152  _nodes.push_back(n);
153  _nodes_lookup[id] = _nodes.back();
154  if (wildcmp(_injection_name.c_str(), name.c_str())) {
155  _injection.push_back(n);
156  _injection_lookup[id] = _injection.back();
157  }
158  }
159  //delete stmt;
160  cout << " -Nodes: " << _nodes.size() << endl;
161  cout << " -Injection Points: " << _injection.size() << endl;
162 
163  delete stmt;
164 
165 
166  int links = 0;
167  stmt = db.Prepare("SELECT seg1, seg2, rate12e, rate21e, drX, drY, drZ FROM pairs;");
168  while (stmt->Step() != SQLITE_DONE) {
169  node_t *n1 = _nodes_lookup[stmt->Column<int>(0)];
170  node_t *n2 = _nodes_lookup[stmt->Column<int>(1)];
171  double rate12 = stmt->Column<double>(2);
172  double rate21 = stmt->Column<double>(3);
173  vec r = vec(stmt->Column<double>(4), stmt->Column<double>(5), stmt->Column<double>(6));
174  n1->AddEvent(new link_t(n2, rate12, r));
175  n2->AddEvent(new link_t(n1, rate21, -r));
176  links += 2;
177  }
178  delete stmt;
179  cout << " -Links: " << links << endl;
180 }

182 void Diffusion::RunKMC(void)
183 {
184  double t = 0;
185  number_of_points = 0;
186  //put zeros in diffusion tensor matrix
187  diffusion.ZeroMatrix();
188 
189  srand(_seed);
190  Random::init(rand(), rand(), rand(), rand());
191 
192 
193  current=_injection[Random::rand_uniform_int(_injection.size())];
194  cout <<" Starting simulation at node: "<<current->_id-1<<endl;
195  double next_output = _dt;
196  int i=0;
197  while(t<_runtime) {
198  t+=current->WaitingTime();
199  current->onExecute();
200  if(t > next_output) {
201  next_output = t + _dt;
202  number_of_points++;
203 
204 
205  // calculation of diffusion tensor
206  diffusion += r|r;
207 
208  cout << t << ": " << r << endl;
209  }
210  }
211  _runtime = t;
212  WriteOcc();
213  cout << std::scientific << "\nKMC run finished\nAverage velocity (m/s): " << r/t*1e-9 << endl;
214 
215  // calculating diffusion and printing matrix
216  diffusion /= number_of_points*2*t;
217  diffusion *= 1e-18;
218  cout<<endl<<"Diffusion tensor:"<<endl<<diffusion<<endl;
219 
220 
221  //solve eigensystem
222  diffusion.SolveEigensystem(diff_tensor_eigensystem);
223  //cout<<diff_tensor_eigensystem<<endl;
224 
225 
226  cout<<endl<<"Eigenvalues: "<<endl<<endl;
