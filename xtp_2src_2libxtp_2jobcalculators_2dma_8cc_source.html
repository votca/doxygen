47  string key = "options." + Identify();

55  _switch = options->get(key+".multipoles.switch").as<int> ();
56  _outFile = options->get(key+".output").as<string> ();
57 
58  string _tasks_string = options->get(key+".tasks").as<string> ();
59  if (_tasks_string.find("input") != std::string::npos) _do_input = true;
60  if (_tasks_string.find("orbitals") != std::string::npos) _do_orbitals = true;
61  if (_tasks_string.find("dma") != std::string::npos) _do_dma = true;
62 
63  key = "options." + Identify() +".job";
64  _jobfile = options->get(key + ".file").as<string>();
65 
66  load_property_from_xml( _package_options, _package_xml.c_str() );
67  key = "package";
68  _package = _package_options.get(key+".name").as<string> ();
69  if ( _package != "gaussian" ) { throw runtime_error("\nERROR: Package: " + _package + "not supported. Use Gaussian for GDMA"); }
70 
71 
72  // register all QM packages (Gaussian, Turbomole, NWChem))
73  QMPackageFactory::RegisterAll();
