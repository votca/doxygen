69 bool TDump::EvaluateFrame(Topology *top) {
70 
71  if (_framesWritten > _framesToWrite) { return 1; }
72 
73  // Rigidify std::system (if possible)
74  bool isRigid = top->Rigidify();
75  if (!isRigid) {
76  return 0;
77  }
78 
79  // Print coordinates
80  FILE *outPDBmd = NULL;
81  FILE *outPDBqm = NULL;
82 
83  outPDBmd = fopen(_outPDBmd.c_str(), "a");
84  outPDBqm = fopen(_outPDBqm.c_str(), "a");
85 
86  fprintf(outPDBmd, "TITLE VOT CAtastrophic title \n");
87  fprintf(outPDBmd, "Model %8d \n" , _framesWritten);
88 
89  fprintf(outPDBqm, "TITLE VOT CAtastrophic title \n");
90  fprintf(outPDBqm, "Model %8d \n", _framesWritten);
91 
92  vector< Segment* > ::iterator sit;
93  for (sit = top->Segments().begin();
94  sit < top->Segments().end();
95  sit++) {
96 
97  (*sit)->WritePDB(outPDBmd, "Atoms", "MD");
98  (*sit)->WritePDB(outPDBqm, "Atoms", "QM");
99  }
100 
101  fprintf(outPDBmd, "TER\nENDMDL\n");
102  fprintf(outPDBqm, "TER\nENDMDL\n");
103 
104  fclose(outPDBmd);
105  fclose(outPDBqm);
106 
107  _framesWritten++;
108 
109  return true;
110 }
