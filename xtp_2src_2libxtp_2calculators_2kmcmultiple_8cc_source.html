31 void KMCMultiple::Initialize(tools::Property* options) {
32  std::string key = "options." + Identify();
33 
34  _runtime =
35  options->ifExistsReturnElseThrowRuntimeError<double>(key + ".runtime");
36  _seed = options->ifExistsReturnElseThrowRuntimeError<int>(key + ".seed");
37  _numberofcharges = options->ifExistsReturnElseThrowRuntimeError<int>(
38  key + ".numberofcharges");
39  _injection_name = options->ifExistsReturnElseThrowRuntimeError<std::string>(
40  key + ".injectionpattern");
41  _intermediateoutput_frequency = options->ifExistsReturnElseReturnDefault<int>(
42  key + ".intermediateoutput", 1E9);
43 
44  _maxrealtime = options->ifExistsReturnElseReturnDefault<double>(
45  key + ".maxrealtime", 1E10);
46  _trajectoryfile = options->ifExistsReturnElseReturnDefault<std::string>(
47  key + ".trajectoryfile", "trajectory.csv");
48  _temperature = options->ifExistsReturnElseReturnDefault<double>(
49  key + ".temperature", 300);
50  _rates = options->ifExistsReturnElseReturnDefault<std::string>(key + ".rates",
51  "statefile");
52 
53  _injectionmethod = options->ifExistsReturnElseReturnDefault<std::string>(
54  key + ".injectionmethod", "random");
55 
56  if (_injectionmethod != "random") {
57  cout << "WARNING in kmcmultiple: Unknown injection method. It will be set "
58  "to random injection."
59  << endl;
60  _injectionmethod = "random";
61  }
62  _field = options->ifExistsReturnElseReturnDefault<tools::vec>(
63  key + ".field", tools::vec(0, 0, 0));
64  double mtonm = 1E9;
65  _field /= mtonm; // Converting from V/m to V/nm
66 
67  _outputtime =
68  options->ifExistsReturnElseReturnDefault<double>(key + ".outputtime", 0);
69  _timefile = options->ifExistsReturnElseReturnDefault<std::string>(
70  key + ".timefile", "timedependence.csv");
71 
72  std::string carriertype =
73  options->ifExistsReturnElseReturnDefault<std::string>(
74  key + ".carriertype", "e");
75  _carriertype = StringtoCarriertype(carriertype);
76 
77  lengthdistribution = options->ifExistsReturnElseReturnDefault<double>(
78  key + ".jumplengthdist", 0);
79  if (lengthdistribution > 0) {
80  dolengthdistributon = true;
81  }
82 
83  return;
84 }

112  cout << "output frequency: ";
113  cout << "every " << _outputtime << " seconds." << endl;
114  }
115  }

123  "ERROR in kmcmultiple: runtime was specified in steps (>100) and "
124  "outputtime in seconds (not an integer). Please use the same units for "
125  "both input parameters.");
126  }
127

146  traj << "'carrier" << i + 1 << "_y'\t";
147  traj << "'carrier" << i + 1 << "_z";
148  if (i < _numberofcharges - 1) {
149  traj << "'\t";
150  }
151  }
152  traj << endl;
153 
154  cout << "Writing time dependence of energy and mobility to " << _timefile

157  tfile << "time[s]\t "
158  "steps\tenergy_per_carrier[eV]\tmobility[nm**2/"
159  "Vs]\tdistance_fielddirection[nm]\tdistance_absolute[nm]"
160  << endl;
161  }
162 
163  double absolute_field = tools::abs(_field);
164 
165  RandomlyCreateCharges();
166  vector<tools::vec> startposition(_numberofcharges, tools::vec(0.0));
167  for (unsigned int i = 0; i < _numberofcharges; i++) {
168  startposition[i] = _carriers[i]->getCurrentPosition();
169  }
170 
171  traj << 0 << "\t";
172  traj << 0 << "\t";
173  for (unsigned int i = 0; i < _numberofcharges; i++) {
174  traj << startposition[i].getX() << "\t";
175  traj << startposition[i].getY() << "\t";
176  traj << startposition[i].getZ();
177  if (i < _numberofcharges - 1) {
178  traj << "\t";
179  } else {
180  traj << endl;
181  }
182  }
183 
184  vector<int> forbiddennodes;
185  vector<int> forbiddendests;
186 
187  tools::matrix avgdiffusiontensor;
188  avgdiffusiontensor.ZeroMatrix();
189 
190  unsigned long diffusionresolution = 1000;
191  double simtime = 0.0;
192  unsigned long step = 0;
193 
194  while (((stopontime && simtime < _runtime) ||
195  (!stopontime && step < maxsteps))) {
196 
197  if ((time(NULL) - realtime_start) > _maxrealtime * 60. * 60.) {
198  cout << endl
199  << "Real time limit of " << _maxrealtime << " hours ("
200  << int(_maxrealtime * 60 * 60 + 0.5)
201  << " seconds) has been reached. Stopping here." << endl
202  << endl;
203  break;
204  }
205 
206  double cumulated_rate = 0;
207  for (unsigned int i = 0; i < _numberofcharges; i++) {
208  cumulated_rate += _carriers[i]->getCurrentEscapeRate();
209  }
210  if (cumulated_rate == 0) { // this should not happen: no possible jumps
211  // defined for a node
212  throw runtime_error(
213  "ERROR in kmcmultiple: Incorrect rates in the database file. All the "
214  "escape rates for the current setting are 0.");
215  }
216 
217  double dt = Promotetime(cumulated_rate);
218 
219  simtime += dt;
220  step++;
221  if (tools::globals::verbose) {
222  cout << "simtime += " << dt << endl << endl;
223  }
224 
225  for (unsigned int i = 0; i < _numberofcharges; i++) {
226  _carriers[i]->updateOccupationtime(dt);
227  }
228 
229  ResetForbiddenlist(forbiddennodes);
230  bool level1step = true;
231  while (level1step) {
232 
233  // determine which electron will escape
234 
235  GNode* newnode = NULL;
236  Chargecarrier* affectedcarrier = ChooseAffectedCarrier(cumulated_rate);
237 
238  if (CheckForbidden(affectedcarrier->getCurrentNodeId(), forbiddennodes)) {
239  continue;
240  }
241 
242  // determine where it will jump to
243  ResetForbiddenlist(forbiddendests);
244  while (true) {
245  // LEVEL 2
246  if (tools::globals::verbose) {
247  cout << "There are "
248  << affectedcarrier->getCurrentNode()->events.size()
249  << " possible jumps for this charge:";
250  }
251 
252  GLink* event = ChooseHoppingDest(affectedcarrier->getCurrentNode());
253  newnode = _nodes[event->destination];
254  if (newnode == affectedcarrier->getCurrentNode()) {
255  cout << event->dr << endl;
256  }
257 
258  if (newnode == NULL) {
259  if (tools::globals::verbose) {
260  cout << endl
261  << "Node " << affectedcarrier->getCurrentNodeId() + 1
262  << " is SURROUNDED by forbidden destinations and zero rates. "
263  "Adding it to the list of forbidden nodes. After that: "
264  "selection of a new escape node."
265  << endl;
266  }
267  AddtoForbiddenlist(affectedcarrier->getCurrentNodeId(),
268  forbiddennodes);
269  break; // select new escape node (ends level 2 but without setting
270  // level1step to 1)
271  }
272  if (tools::globals::verbose) {
273  cout << endl << "Selected jump: " << newnode->id + 1 << endl;
274  }
275 
276  // check after the event if this was allowed
277  if (CheckForbidden(newnode->id, forbiddendests)) {
278  if (tools::globals::verbose) {
279  cout << "Node " << newnode->id + 1
280  << " is FORBIDDEN. Now selection new hopping destination."
281  << endl;
282  }
283  continue;
284  }
285 
286  // if the new segment is unoccupied: jump; if not: add to forbidden list
287  // and choose new hopping destination
288  if (newnode->occupied) {
289  if (CheckSurrounded(affectedcarrier->getCurrentNode(),
290  forbiddendests)) {
291  if (tools::globals::verbose) {
292  cout << "Node " << affectedcarrier->getCurrentNodeId() + 1
293  << " is SURROUNDED by forbidden destinations. "
294  "Adding it to the list of forbidden nodes. After that: "
295  "selection of a new escape node."
296  << endl;
297  }
298  AddtoForbiddenlist(affectedcarrier->getCurrentNodeId(),
299  forbiddennodes);
300  break; // select new escape node (ends level 2 but without setting
301  // level1step to 1)
302  }
303  if (tools::globals::verbose) {
304  cout << "Selected segment: " << newnode->id + 1
305  << " is already OCCUPIED. Added to forbidden list." << endl
306  << endl;
307  }
308  AddtoForbiddenlist(newnode->id, forbiddendests);
309  if (tools::globals::verbose) {
310  cout << "Now choosing different hopping destination." << endl;
311  }
312  continue; // select new destination
313  } else {
314  affectedcarrier->jumpfromCurrentNodetoNode(newnode);
315  affectedcarrier->dr_travelled += event->dr;
316  AddtoJumplengthdistro(event, dt);
317  level1step = false;
318  if (tools::globals::verbose) {
319  cout << "Charge has jumped to segment: " << newnode->id + 1 << "."
320  << endl;
321  }
322 
323  break; // this ends LEVEL 2 , so that the time is updated and the
324  // next MC step started
325  }
326 
327  if (tools::globals::verbose) {
328  cout << "." << endl;
329  }
330  // END LEVEL 2
331  }
332  // END LEVEL 1
333  }
334 
335  // outputstuff
336 
337  if (step % diffusionresolution == 0) {
338  for (unsigned int i = 0; i < _numberofcharges; i++) {
339  avgdiffusiontensor +=
340  (_carriers[i]->dr_travelled) | (_carriers[i]->dr_travelled);
341  }
342  }
343 
344  if (step != 0 && step % _intermediateoutput_frequency == 0) {
345 
346  if (absolute_field == 0) {
347  unsigned long diffusionsteps = step / diffusionresolution;
348  tools::matrix result =
349  avgdiffusiontensor /
350  (diffusionsteps * 2 * simtime * _numberofcharges);
351  cout << endl
352  << "Step: " << step
353  << " Diffusion tensor averaged over all carriers (nm^2/s):" << endl
354  << result << endl;
355  } else {
356  double average_mobility = 0;
357  cout << endl << "Mobilities (nm^2/Vs): " << endl;
358  for (unsigned int i = 0; i < _numberofcharges; i++) {
359  tools::vec velocity = _carriers[i]->dr_travelled / simtime;
360  cout << std::scientific << " charge " << i + 1 << ": mu="
361  << (velocity * _field) / (absolute_field * absolute_field)
362  << endl;
363  average_mobility +=
364  (velocity * _field) / (absolute_field * absolute_field);
365  }
366  average_mobility /= _numberofcharges;
367  cout << std::scientific
368  << " Overall average mobility in field direction <mu>="
369  << average_mobility << " nm^2/Vs " << endl;
370  }
371  }
372 
373  if (checkifoutput) {
374  bool outputsteps = (!stopontime && step % outputstep == 0);
375  bool outputtime = (stopontime && simtime > nexttrajoutput);
376  if (outputsteps || outputtime) {
377  // write to trajectory file
378  nexttrajoutput = simtime + _outputtime;
379  traj << simtime << "\t";
380  traj << step << "\t";
381  for (unsigned int i = 0; i < _numberofcharges; i++) {
382  traj << startposition[i].getX() + _carriers[i]->dr_travelled.getX()
383  << "\t";
384  traj << startposition[i].getY() + _carriers[i]->dr_travelled.getY()
385  << "\t";
386  traj << startposition[i].getZ() + _carriers[i]->dr_travelled.getZ();
387  if (i < _numberofcharges - 1) {
388  traj << "\t";
389  } else {
390  traj << endl;
391  }
392  }
393 
394  double currentenergy = 0;
395  double currentmobility = 0;
396  tools::vec dr_travelled_current = tools::vec(0, 0, 0);
397  double dr_travelled_field = 0.0;
398  tools::vec avgvelocity_current = tools::vec(0, 0, 0);
399  if (absolute_field != 0) {
400  for (unsigned int i = 0; i < _numberofcharges; i++) {
401  dr_travelled_current += _carriers[i]->dr_travelled;
402  currentenergy += _carriers[i]->getCurrentEnergy();
403  }
404  dr_travelled_current /= _numberofcharges;
405  currentenergy /= _numberofcharges;
406  avgvelocity_current = dr_travelled_current / simtime;
407  currentmobility =
408  (avgvelocity_current * _field) / absolute_field / absolute_field;
409  dr_travelled_field = (dr_travelled_current * _field) / absolute_field;
410  }
411 
412  tfile << simtime << "\t" << step << "\t" << currentenergy << "\t"
413  << currentmobility << "\t" << dr_travelled_field << "\t"
414  << tools::abs(dr_travelled_current) << "\t" << endl;
415  }
416  }
417 
418  } // KMC
419 
420  if (checkifoutput) {
421  traj.close();
422  tfile.close();
423  }
424 
425  vector<ctp::Segment*>& seg = top->Segments();
426  for (unsigned i = 0; i < seg.size(); i++) {
427  double occupationprobability = _nodes[i]->occupationtime / simtime;
428  seg[i]->setOcc(occupationprobability, _carriertype);
429  }
430 
431  cout << endl << "finished KMC simulation after " << step << " steps." << endl;
432  cout << "simulated time " << simtime << " seconds." << endl;
433  cout << "runtime: ";
434  cout << endl << endl;
435 
436  tools::vec avg_dr_travelled = tools::vec(0, 0, 0);
437  for (unsigned int i = 0; i < _numberofcharges; i++) {
438  cout << std::scientific << " charge " << i + 1 << ": "
439  << _carriers[i]->dr_travelled / simtime << endl;
440  avg_dr_travelled += _carriers[i]->dr_travelled;
441  }
442  avg_dr_travelled /= _numberofcharges;
443 
444  tools::vec avgvelocity = avg_dr_travelled / simtime;
445  cout << std::scientific
446  << " Overall average velocity (nm/s): " << avgvelocity << endl;
447 
448  cout << endl << "Distances travelled (nm): " << endl;
449  for (unsigned int i = 0; i < _numberofcharges; i++) {
450  cout << std::scientific << " charge " << i + 1 << ": "
451  << _carriers[i]->dr_travelled << endl;
452  }
453 
454  // calculate mobilities
455 
456  if (absolute_field != 0) {
457  double average_mobility = 0;
458  cout << endl << "Mobilities (nm^2/Vs): " << endl;
459  for (unsigned int i = 0; i < _numberofcharges; i++) {
460  tools::vec velocity = _carriers[i]->dr_travelled / simtime;
461  cout << std::scientific << " charge " << i + 1
462  << ": mu=" << (velocity * _field) / (absolute_field * absolute_field)
463  << endl;
464  average_mobility +=
465  (velocity * _field) / (absolute_field * absolute_field);
466  }
467  average_mobility /= _numberofcharges;
468  cout << std::scientific
469  << " Overall average mobility in field direction <mu>="
470  << average_mobility << " nm^2/Vs " << endl;
471  }
472  cout << endl;
473 
474  // calculate diffusion tensor
475  unsigned long diffusionsteps = step / diffusionresolution;
476  avgdiffusiontensor /= (diffusionsteps * 2 * simtime * _numberofcharges);
477  cout << endl
478  << "Diffusion tensor averaged over all carriers (nm^2/s):" << endl
479  << avgdiffusiontensor << endl;
480 
481  tools::matrix::eigensystem_t diff_tensor_eigensystem;
482  cout << endl << "Eigenvalues: " << endl << endl;
483  avgdiffusiontensor.SolveEigensystem(diff_tensor_eigensystem);
484  for (int i = 0; i <= 2; i++) {
485  cout << "Eigenvalue: " << diff_tensor_eigensystem.eigenvalues[i] << endl
486  << "Eigenvector: ";
487 
488  cout << diff_tensor_eigensystem.eigenvecs[i].x() << " ";
489  cout << diff_tensor_eigensystem.eigenvecs[i].y() << " ";
490  cout << diff_tensor_eigensystem.eigenvecs[i].z() << endl << endl;
491  }
492 
493  // calculate average mobility from the Einstein relation
494  if (absolute_field == 0) {
495  cout << "The following value is calculated using the Einstein relation and "
496  "assuming an isotropic medium"
497  << endl;
498  double avgD = 1. / 3. *
499  (diff_tensor_eigensystem.eigenvalues[0] +
500  diff_tensor_eigensystem.eigenvalues[1] +
501  diff_tensor_eigensystem.eigenvalues[2]);
502  double average_mobility = std::abs(avgD / tools::conv::kB / _temperature);
503  cout << std::scientific
504  << " Overall average mobility <mu>=" << average_mobility
505  << " nm^2/Vs " << endl;
506  }
507 
508  PrintJumplengthdistro();
509 
510  return;
