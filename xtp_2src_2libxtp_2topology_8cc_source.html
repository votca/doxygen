153 void Topology::setBox(const matrix &box,
154  CSG::BoundaryCondition::eBoxtype boxtype) {
155 
156  // Determine box type automatically in case boxtype == typeAuto
157  if (boxtype == CSG::BoundaryCondition::typeAuto) {
158  boxtype = AutoDetectBoxType(box);
159  }
160 
161  if (_hasPb) {
162  if (votca::tools::globals::verbose) {
163  cout << "Removing periodic box. Creating new... " << endl;
164  }
165  delete _bc;
166  }
167 
168  switch (boxtype) {
169  case CSG::BoundaryCondition::typeTriclinic:
170  _bc = new CSG::TriclinicBox();
171  break;
172  case CSG::BoundaryCondition::typeOrthorhombic:
173  _bc = new CSG::OrthorhombicBox();
174  break;
175  default:
176  _bc = new CSG::OpenBox();
177  break;
178  }
179 
180  _bc->setBox(box);
181  _hasPb = true;
182 }

184 CSG::BoundaryCondition::eBoxtype Topology::AutoDetectBoxType(
185  const matrix &box) {
186 
187  // Set box type to OpenBox in case "box" is the zero matrix,
188  // to OrthorhombicBox in case "box" is a diagonal matrix,
189  // or to TriclinicBox otherwise
190 
191  if (box.get(0, 0) == 0 && box.get(0, 1) == 0 && box.get(0, 2) == 0 &&
192  box.get(1, 0) == 0 && box.get(1, 1) == 0 && box.get(1, 2) == 0 &&
193  box.get(2, 0) == 0 && box.get(2, 1) == 0 && box.get(2, 2) == 0) {
194 
195  cout << "WARNING: No box vectors specified in trajectory."
196  "Using open-box boundary conditions. "
197  << endl;
198  return CSG::BoundaryCondition::typeOpen;
199  }
200 
201  else if (box.get(0, 1) == 0 && box.get(0, 2) == 0 && box.get(1, 0) == 0 &&
202  box.get(1, 2) == 0 && box.get(2, 0) == 0 && box.get(2, 1) == 0) {
203 
204  return CSG::BoundaryCondition::typeOrthorhombic;
205  }
206 
207  else {
208  return CSG::BoundaryCondition::typeTriclinic;
209  }
210 
211  return CSG::BoundaryCondition::typeOpen;
212 }

218 bool Topology::Rigidify() {
219 
220  if (!_canRigidify) {
221  cout << endl
222  << "... ... ABORT: Request to rigidify system, but no QM "
223  "coordinates provided. ";
224  return 0;
225  } else {
226 
227  // Rigidify segments
228  vector<Segment *>::iterator sit;
229  for (sit = _segments.begin(); sit < _segments.end(); sit++) {
230 
231  (*sit)->Rigidify();
232  }
233 
234  cout << endl
235  << "... ... Rigidified " << _segments.size() << " segments. " << flush;
236 
237  if (this->NBList().size() > 0) {
238 
239  // Rigidify pairs
240  // [ Why this is needed: Orientation matrices for fragments are not
241  // not written to the state file, and hence lost whenever a frame
242  // is saved and reloaded. When reading in a new frame from the
243  // database, pairs are created from scratch based on two segment
244  // IDs each. Then the pair constructor is called to check whether
245  // the pair is formed across the periodic boundary. If so, it
246  // creates a ghost from the second partner. This ghost is a new
247  // segment which is just accessible from within the pair, but not
248  // from the topology; i.e. it is not stored in any segment containers.
249  // Since at this point, fragments do not store rotation matrices yet,
250  // the ghost - very much deconnected from its originator - does not,
251  // either. Therefore it should not be forgotten here. --- A way out
252  // would be to rigidify the topology within StateSaver::ReadFrame,
253  // after atoms have been created, but before pairs are created. ]
254 
255  QMNBList &nblist = this->NBList();
256 
257  QMNBList::iterator pit;
258  int count = 0;
259  for (pit = nblist.begin(); pit != nblist.end(); pit++) {
260 
261  QMPair *qmpair = *pit;
262  if (qmpair->HasGhost()) {
263  count++;
264 
265  qmpair->Seg2PbCopy()->Rigidify();
266  }
267  }
268 
269  cout << endl << "... ... Rigidified " << count << " ghosts. " << flush;
270  }
271 
272  _isRigid = true;
273  return 1;
274  }
275 }
