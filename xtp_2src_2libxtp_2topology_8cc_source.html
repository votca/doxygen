69 void Topology::setBox(const Eigen::Matrix3d &box,
70  csg::BoundaryCondition::eBoxtype boxtype) {
71 
72  // Determine box type automatically in case boxtype == typeAuto
73  if (boxtype == csg::BoundaryCondition::typeAuto) {
74  boxtype = AutoDetectBoxType(box);
75  }
76 
77  if (_bc != nullptr) {
78  if (Log::verbose()) {
79  std::cout << "Removing periodic box. Creating new... " << std::endl;
80  }
81  }
82  _bc.reset(nullptr);
83  switch (boxtype) {
84  case csg::BoundaryCondition::typeTriclinic:
85  _bc.reset(new csg::TriclinicBox());
86  break;
87  case csg::BoundaryCondition::typeOrthorhombic:
88  _bc.reset(new csg::OrthorhombicBox());
89  break;
90  default:
91  _bc.reset(new csg::OpenBox());
92  break;
93  }
94 
95  _bc->setBox(box);
96 }

98 csg::BoundaryCondition::eBoxtype Topology::AutoDetectBoxType(
99  const Eigen::Matrix3d &box) {
100 
101  // Set box type to OpenBox in case "box" is the zero matrix,
102  // to OrthorhombicBox in case "box" is a diagonal matrix,
103  // or to TriclinicBox otherwise
104 
105  if (box.isApproxToConstant(0)) {
106  std::cout << "WARNING: No box vectors specified in trajectory."
107  "Using open-box boundary conditions. "
108  << std::endl;
109  return csg::BoundaryCondition::typeOpen;
110  }
111 
112  else if ((box - Eigen::Matrix3d(box.diagonal().asDiagonal()))
113  .isApproxToConstant(0)) {
114  return csg::BoundaryCondition::typeOrthorhombic;
115  }
116 
117  else {
118  return csg::BoundaryCondition::typeTriclinic;
119  }
120 
121  return csg::BoundaryCondition::typeOpen;
122 }

129 double Topology::GetShortestDist(const Segment &seg1,
130  const Segment &seg2) const {
131  double R2 = std::numeric_limits<double>::max();
132  for (const Atom &atom1 : seg1) {
133  for (const Atom &atom2 : seg2) {
134  double R2_test =
135  PbShortestConnect(atom1.getPos(), atom2.getPos()).squaredNorm();
136  if (R2_test < R2) {
137  R2 = R2_test;
138  }
139  }
140  }
141  return std::sqrt(R2);
142 }

144 std::vector<const Segment *> Topology::FindAllSegmentsOnMolecule(
145  const Segment &seg1, const Segment &seg2) const {
146  const std::vector<Index> &ids1 = seg1.getMoleculeIds();
147  const std::vector<Index> &ids2 = seg2.getMoleculeIds();
148  std::vector<Index> common_elements;
149  std::set_intersection(ids1.begin(), ids1.end(), ids2.begin(), ids2.end(),
150  std::back_inserter(common_elements));
151  std::vector<const Segment *> results;
152  if (common_elements.empty() || common_elements.size() > 1) {
153  return results;
154  }
155  Index molid = common_elements[0];
156 
157  for (const Segment &seg : _segments) {
158  if (std::find(seg.getMoleculeIds().begin(), seg.getMoleculeIds().end(),
159  molid) != seg.getMoleculeIds().end()) {
160  results.push_back(&seg);
161  }
162  }
163  return results;
164 }

191 void Topology::ReadFromCpt(CheckpointReader &r) {
192  r(_time, "time");
193  r(_step, "step");
194  Eigen::Matrix3d box;
195  r(box, "box");
196  setBox(box);
197  CheckpointReader v = r.openChild("segments");
198  _segments.clear();
199  Index count = v.getNumDataSets();
200  _segments.reserve(count);
201  for (Index i = 0; i < count; i++) {
202  CheckpointReader w = v.openChild("segment" + std::to_string(i));
203  _segments.push_back(Segment(w));
204  }
205  CheckpointReader rr = r.openChild("neighborlist");
206  _nblist.ReadFromCpt(rr, _segments);
207 }
