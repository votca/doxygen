152 void Topology::setBox(const matrix &box,
153  CSG::BoundaryCondition::eBoxtype boxtype) {
154 
155  // Determine box type automatically in case boxtype == typeAuto
156  if(boxtype == CSG::BoundaryCondition::typeAuto) {
157  boxtype = AutoDetectBoxType(box);
158  }
159 
160  if(_hasPb) {
161  if (votca::tools::globals::verbose) {
162  cout << "Removing periodic box. Creating new... " << endl;
163  }
164  delete _bc;
165  }
166 
167  switch(boxtype) {
168  case CSG::BoundaryCondition::typeTriclinic:
169  _bc = new CSG::TriclinicBox();
170  break;
171  case CSG::BoundaryCondition::typeOrthorhombic:
172  _bc = new CSG::OrthorhombicBox();
173  break;
174  default:
175  _bc = new CSG::OpenBox();
176  break;
177  }
178 
179  _bc->setBox(box);
180  _hasPb = true;
181 }

184 CSG::BoundaryCondition::eBoxtype Topology::AutoDetectBoxType(const matrix &box){
185 
186  // Set box type to OpenBox in case "box" is the zero matrix,
187  // to OrthorhombicBox in case "box" is a diagonal matrix,
188  // or to TriclinicBox otherwise
189 
190  if(box.get(0,0)==0 && box.get(0,1)==0 && box.get(0,2)==0 &&
191  box.get(1,0)==0 && box.get(1,1)==0 && box.get(1,2)==0 &&
192  box.get(2,0)==0 && box.get(2,1)==0 && box.get(2,2)==0) {
193 
194  cout << "WARNING: No box vectors specified in trajectory."
195  "Using open-box boundary conditions. " << endl;
196  return CSG::BoundaryCondition::typeOpen;
197  }
198 
199  else if(box.get(0,1)==0 && box.get(0,2)==0 &&
200  box.get(1,0)==0 && box.get(1,2)==0 &&
201  box.get(2,0)==0 && box.get(2,1)==0) {
202 
203  return CSG::BoundaryCondition::typeOrthorhombic;
204  }
205 
206  else {
207  return CSG::BoundaryCondition::typeTriclinic;
208  }
209 
210  return CSG::BoundaryCondition::typeOpen;
211 }

221 bool Topology::Rigidify() {
222 
223  if (!_canRigidify) {
224  cout << endl
225  << "... ... ABORT: Request to rigidify system, but no QM "
226  "coordinates provided. ";
227  return 0;
228  }
229  else {
230 
231  // Rigidify segments
232  std::vector<Segment*> ::iterator sit;
233  for (sit = _segments.begin();
234  sit < _segments.end();
235  sit++) {
236 
237  (*sit)->Rigidify();
238  }
239 
240  cout << endl
241  << "... ... Rigidified " << _segments.size() << " segments. "
242  << flush;
243 
244  if (this->NBList().size() > 0) {
245 
246  // Rigidify pairs
247  // [ Why this is needed: Orientation matrices for fragments are not
248  // not written to the state file, and hence lost whenever a frame
249  // is saved and reloaded. When reading in a new frame from the
250  // database, pairs are created from scratch based on two segment
251  // IDs each. Then the pair constructor is called to check whether
252  // the pair is formed across the periodic boundary. If so, it
253  // creates a ghost from the second partner. This ghost is a new
254  // segment which is just accessible from within the pair, but not
255  // from the topology; i.e. it is not stored in any segment containers.
256  // Since at this point, fragments do not store rotation matrices yet,
257  // the ghost - very much deconnected from its originator - does not,
258  // either. Therefore it should not be forgotten here. --- A way out
259  // would be to rigidify the topology within StateSaver::ReadFrame,
260  // after atoms have been created, but before pairs are created. ]
261 
262  QMNBList &nblist = this->NBList();
263 
264  QMNBList::iterator pit;
265  int count = 0;
266  for (pit = nblist.begin(); pit != nblist.end(); pit++) {
267 
268  QMPair *qmpair = *pit;
269  if (qmpair->HasGhost()) {
270  count++;
271 
272 
273 
274  qmpair->Seg2PbCopy()->Rigidify();
275  }
276  }
277 
278  cout << endl
279  << "... ... Rigidified " << count << " ghosts. "
280  << flush;
281  }
282 
283  _isRigid = true;
284  return 1;
285  }

359  name.resize(3);

361  resname.resize(3);
362  int resnr = (*seg)->getMolecule()->getId();
363  vec position = (*seg)->getPos();
364 
365  fprintf(out, "ATOM %5d %4s%1s%3s %1s%4d%1s %8.3f%8.3f%8.3f%6.2f%6.2f %4s%2s%2s\n",
366  id, // Atom serial number %5d
367  name.c_str(), // Atom name %4s
368  " ", // alternate location indicator.%1s
369  resname.c_str(), // Residue name. %3s
370  "A", // Chain identifier %1s
371  resnr, // Residue sequence number %4d
372  " ", // Insertion of residues. %1s
373  position.getX()*10, // X in Angstroms %8.3f
374  position.getY()*10, // Y in Angstroms %8.3f
375  position.getZ()*10, // Z in Angstroms %8.3f
376  1.0, // Occupancy %6.2f
377  0.0, // Temperature factor %6.2f
378  " ", // Segment identifier %4s
379  name.c_str(), // Element symbol %2s
380  " " // Charge on the atom. %2s

383  }
