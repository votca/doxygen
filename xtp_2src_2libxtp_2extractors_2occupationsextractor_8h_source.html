51 bool OccupationsExtractor::EvaluateFrame(ctp::Topology *top) {
52 
53  std::string xmlfile = Identify() + ".xml";
54 
55  tools::Property state("state", "", "");
56  tools::Property &segs = state.add("segments","");
57  tools::Property *next = NULL;
58 
59 
60  using boost::format;
61 
62  // SEGMENTS
63  std::vector<ctp::Segment*> ::iterator sit;
64  next = &segs;
65  for (sit = top->Segments().begin(); sit < top->Segments().end(); ++sit) {
66  ctp::Segment *seg = *sit;
67  tools::Property &segprop = next->add("segment", "");
68  segprop.add("id", (format("%1$d") % seg->getId()).str());
69  segprop.add("name", seg->getName());
70  segprop.add("xyz", (format("%1$+1.4f %2$+1.4f %3$+1.4f")
71  % seg->getPos().getX() % seg->getPos().getY() % seg->getPos().getZ()).str());
72  if (seg->hasChrgState(+1)) {
73  tools::Property &channel = segprop.add("channel","");
74  channel.setAttribute("type", "hole");
75  channel.add("occupation_h", (format("%1$+1.7e") % seg->getOcc(+1)).str());
76  }
77  if (seg->hasChrgState(-1)) {
78  tools::Property &channel = segprop.add("channel","");
79  channel.setAttribute("type", "electron");
80  channel.add("occupation_e", (format("%1$+1.7e") % seg->getOcc(-1)).str());
81  }
82  }
83 
84  std::ofstream ofs;
85  ofs.open(xmlfile.c_str(), std::ofstream::out);
86  if (!ofs.is_open()) {
87  throw std::runtime_error("Bad file handle: " + xmlfile);
88  }
89  ofs << tools::XML << state;
90  ofs.close();
91 
92  return true;
93 }
