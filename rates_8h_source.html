91 complex<double> ccgamma(complex<double> z,int OPT)
92 {
93  complex<double> g,z0,z1;
94  double x0,q1,q2,x,y,th,th1,th2,g0,gr,gi,gr1,gi1;
95  double na=0,t,x1=0,y1,sr,si;
96  //int i,j,k;
97  int j,k;
98 
99  static double a[] = {
100  8.333333333333333e-02,
101  -2.777777777777778e-03,
102  7.936507936507937e-04,
103  -5.952380952380952e-04,
104  8.417508417508418e-04,
105  -1.917526917526918e-03,
106  6.410256410256410e-03,
107  -2.955065359477124e-02,
108  1.796443723688307e-01,
109  -1.39243221690590};
110 
111  x = real(z);
112  y = imag(z);
113  if (x > 171) return complex<double>(1e308,0);
114  if ((y == 0.0) && (x == (int)x) && (x <= 0.0))
115  return complex<double>(1e308,0);
116  else if (x < 0.0) {
117  x1 = x;
118  y1 = y;
119  x = -x;
120  y = -y;
121  }
122  x0 = x;
123  if (x <= 7.0) {
124  na = (int)(7.0-x);
125  x0 = x+na;
126  }
127  q1 = sqrt(x0*x0+y*y);
128  th = atan(y/x0);
129  gr = (x0-0.5)*log(q1)-th*y-x0+0.5*log(2.0*M_PI);
130  gi = th*(x0-0.5)+y*log(q1)-y;
131  for (k=0;k<10;k++){
132  t = pow(q1,-1.0-2.0*k);
133  gr += (a[k]*t*cos((2.0*k+1.0)*th));
134  gi -= (a[k]*t*sin((2.0*k+1.0)*th));
135  }
136  if (x <= 7.0) {
137  gr1 = 0.0;
138  gi1 = 0.0;
139  for (j=0;j<na;j++) {
140  gr1 += (0.5*log((x+j)*(x+j)+y*y));
141  gi1 += atan(y/(x+j));
142  }
143  gr -= gr1;
144  gi -= gi1;
145  }
146  if (x1 <= 0.0) {
147  q1 = sqrt(x*x+y*y);
148  th1 = atan(y/x);
149  sr = -sin(M_PI*x)*cosh(M_PI*y);
150  si = -cos(M_PI*x)*sinh(M_PI*y);
151  q2 = sqrt(sr*sr+si*si);
152  th2 = atan(si/sr);
153  if (sr < 0.0) th2 += M_PI;
154  gr = log(M_PI/(q1*q2))-gr;
155  gi = -th1-th2-gi;
156  x = x1;
157  y = y1;
158  }
159  if (OPT == 0) {
160  g0 = exp(gr);
161  gr = g0*cos(gi);
162  gi = g0*sin(gi);
163  }
164  g = complex<double>(gr,gi);
165  return g;
166 }
