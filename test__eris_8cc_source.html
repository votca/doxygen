34BOOST_AUTO_TEST_CASE(fourcenter) {
35 libint2::initialize();
36 Orbitals orbitals;
37 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
38 "/eris/molecule.xyz");
39 BasisSet basis;
40 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/eris/3-21G.xml");
41
42 AOBasis aobasis;
43 aobasis.Fill(basis, orbitals.QMAtoms());
44
45 Eigen::MatrixXd dmat = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
46 std::string(XTP_TEST_DATA_FOLDER) + "/eris/dmat.mm");
47
48 ERIs eris;
49 eris.Initialize_4c(aobasis);
50 Eigen::MatrixXd erissmall = eris.CalculateERIs_4c(dmat, 1e-20);
51
52 Eigen::MatrixXd eris_ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
53 std::string(XTP_TEST_DATA_FOLDER) + "/eris/eris_ref.mm");
54
55 bool eris_check = erissmall.isApprox(eris_ref, 0.00001);
56 if (!eris_check) {
57 std::cout << "result eri" << std::endl;
58 std::cout << erissmall << std::endl;
59 std::cout << "ref eri" << std::endl;
60 std::cout << eris_ref << std::endl;
61 std::cout << " quotient" << std::endl;
62 std::cout << erissmall.cwiseQuotient(eris_ref);
63 }
64 BOOST_CHECK_EQUAL(eris_check, 1);
65
66 std::array<Eigen::MatrixXd, 2> both = eris.CalculateERIs_EXX_4c(dmat, 1e-20);
67 const Eigen::MatrixXd& exx_small = both[1];
68 Eigen::MatrixXd exx_ref = -votca::tools::EigenIO_MatrixMarket::ReadMatrix(
69 std::string(XTP_TEST_DATA_FOLDER) + "/eris/exx_ref.mm");
70
71 Eigen::MatrixXd sum = both[0] + both[1];
72 Eigen::MatrixXd sum_ref = exx_ref + eris_ref;
73 bool sum_check = sum.isApprox(sum_ref, 1e-5);
74 BOOST_CHECK_EQUAL(sum_check, 1);
75 if (!sum_check) {
76 std::cout << "result sum" << std::endl;
77 std::cout << sum << std::endl;
78 std::cout << "ref sum" << std::endl;
79 std::cout << sum_ref << std::endl;
80 }
81
82 bool exxs_check = exx_small.isApprox(exx_ref, 0.00001);
83 if (!eris_check) {
84 std::cout << "result exx" << std::endl;
85 std::cout << exx_small << std::endl;
86 std::cout << "ref exx" << std::endl;
87 std::cout << exx_ref << std::endl;
88 std::cout << "quotient" << std::endl;
89 std::cout << exx_small.cwiseQuotient(exx_ref);
90 }
91 BOOST_CHECK_EQUAL(exxs_check, 1);
92
93 libint2::finalize();
94}

118 eris.CalculateERIs_EXX_3c(mos.block(0, 0, 17, 4), dmat)[1];
119
120 bool compare_exx = exx_mo.isApprox(exx_dmat, 1e-4);
121 BOOST_CHECK_EQUAL(compare_exx, true);
122
123 Eigen::MatrixXd exx_ref = -votca::tools::EigenIO_MatrixMarket::ReadMatrix(
124 std::string(XTP_TEST_DATA_FOLDER) + "/eris/exx_ref2.mm");

128 std::cout << "result exx" << std::endl;
129 std::cout << exx_mo << std::endl;
130 std::cout << "ref exx" << std::endl;
131 std::cout << exx_ref << std::endl;
132 }
133 BOOST_CHECK_EQUAL(compare_exx_ref, true);
134
135 Eigen::MatrixXd eri = eris.CalculateERIs_3c(dmat);
136
137 Eigen::MatrixXd eris_ref = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
138 std::string(XTP_TEST_DATA_FOLDER) + "/eris/eris_ref2.mm");
