75 void Equilibrate::Initialize(Property *opt) {
76 
77  string key = "options.equilibrate";
78  _topfile = opt->get(key+".topfile").as<string>();
79  _trjfile = opt->get(key+".trjfile").as<string>();
80 
81  CSG::TrajectoryWriter::RegisterPlugins();
82  CSG::TrajectoryReader::RegisterPlugins();
83  CSG::TopologyReader::RegisterPlugins();
84 
85  cout << endl;
86 
87  // READ TOPOLOGY
88  _topread = CSG::TopReaderFactory().Create(_topfile);
89 
90  if (_topread == NULL) {
91  throw runtime_error( string("Input format not supported: ")
92  + _topfile );
93  }
94 
95  _topread->ReadTopology(_topfile, this->_mdtopol);
96  if (votca::tools::globals::verbose) {
97  cout << "Read MD topology from " << _topfile << ": Found "
98  << _mdtopol.BeadCount() << " atoms in "
99  << _mdtopol.MoleculeCount() << " molecules. "
100  << endl;
101  }
102 
103  // READ CONFIGURATION
104  _trjread = CSG::TrjReaderFactory().Create(_trjfile);
105 
106  if (_trjread == NULL) {
107  throw runtime_error( string("Input format not supported: ")
108  + _trjfile );
109  }
110  _trjread->Open(_trjfile);
111  _trjread->FirstFrame(this->_mdtopol);
112 
113  return;
114 }
