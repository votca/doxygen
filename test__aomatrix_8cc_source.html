151BOOST_AUTO_TEST_CASE(aocoulomb_inv_test) {
152 libint2::initialize();
153 QMMolecule mol = Methane();
154 BasisSet basis;
155 basis.Load(std::string(XTP_TEST_DATA_FOLDER) + "/aomatrix/3-21G.xml");
156 AOBasis aobasis;
157 aobasis.Fill(basis, mol);
158
159 AOCoulomb cou;
160 cou.Fill(aobasis);
161
162 Eigen::MatrixXd PseudoInvSqrt = cou.Pseudo_InvSqrt(1e-7);
163
164 Eigen::MatrixXd Reformed = PseudoInvSqrt * PseudoInvSqrt * cou.Matrix();
165
166 bool check_inv = Reformed.isApprox(Eigen::MatrixXd::Identity(17, 17), 0.0001);
167 if (!check_inv) {
168 std::cout << "reformed" << endl;
169 std::cout << Reformed << endl;
170 }
171 BOOST_CHECK_EQUAL(check_inv, 1);
172 libint2::finalize();
173}
