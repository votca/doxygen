00061 void XMLTopologyReader::ParseTopology(const string &el, map<string, string> &attr)
00062 {
00063     if(el == "molecules") {
00064          _parser.NextHandler(this, &XMLTopologyReader::ParseMolecules);
00065     }
00066     else if(el == "box") {
00067         matrix m;
00068         m.ZeroMatrix();
00069         string xx = attr["xx"];
00070         string yy = attr["yy"];
00071         string zz = attr["zz"];
00072         if (xx == "" || yy == "" || zz == "")
00073             throw runtime_error("invalid box tag - missing xx, yy or zz");
00074         try {
00075            m[0][0] = boost::lexical_cast<double>(xx);
00076         } catch(boost::bad_lexical_cast &) {
00077            throw std::runtime_error("Cannot convert xx='"+ xx + "' to double");
00078         }
00079         try {
00080            m[1][1] = boost::lexical_cast<double>(yy);
00081         } catch(boost::bad_lexical_cast &) {
00082            throw std::runtime_error("Cannot convert yy='"+ yy + "' to double");
00083         }
00084         try {
00085            m[2][2] = boost::lexical_cast<double>(zz);
00086         } catch(boost::bad_lexical_cast &) {
00087            throw std::runtime_error("Cannot convert zz='"+ zz + "' to double");
00088         }
00089         _top->setBox(m);
00090         _parser.NextHandler(this, &XMLTopologyReader::ParseTopology);
00091     }
00092     else if(el == "beadtypes") {
00093          _parser.NextHandler(this, &XMLTopologyReader::ParseBeadTypes);
00094     } else if (el == "h5md_particle_group") {
00095         _top->setParticleGroup(attr["name"]);
00096         _parser.NextHandler(this, &XMLTopologyReader::ParseTopology);
00097     } else {
00098         throw runtime_error("unknown tag: "+ el);
00099     }
00100 }
