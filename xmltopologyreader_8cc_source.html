61 void XMLTopologyReader::ParseTopology(const string &el, map<string, string> &attr)
62 {
63  if(el == "molecules") {
64  _parser.NextHandler(this, &XMLTopologyReader::ParseMolecules);
65  }
66  else if(el == "box") {
67  matrix m;
68  m.ZeroMatrix();
69  string xx = attr["xx"];
70  string yy = attr["yy"];
71  string zz = attr["zz"];
72  if (xx == "" || yy == "" || zz == "")
73  throw runtime_error("invalid box tag - missing xx, yy or zz");
74  try {
75  m[0][0] = boost::lexical_cast<double>(xx);
76  } catch(boost::bad_lexical_cast &) {
77  throw std::runtime_error("Cannot convert xx='"+ xx + "' to double");
78  }
79  try {
80  m[1][1] = boost::lexical_cast<double>(yy);
81  } catch(boost::bad_lexical_cast &) {
82  throw std::runtime_error("Cannot convert yy='"+ yy + "' to double");
83  }
84  try {
85  m[2][2] = boost::lexical_cast<double>(zz);
86  } catch(boost::bad_lexical_cast &) {
87  throw std::runtime_error("Cannot convert zz='"+ zz + "' to double");
88  }
89  _top->setBox(m);
90  _parser.NextHandler(this, &XMLTopologyReader::ParseTopology);
91  }
92  else if(el == "beadtypes") {
93  _parser.NextHandler(this, &XMLTopologyReader::ParseBeadTypes);
94  } else if (el == "h5md_particle_group") {
95  _top->setParticleGroup(attr["name"]);
96  _parser.NextHandler(this, &XMLTopologyReader::ParseTopology);
97  } else {
98  throw runtime_error("unknown tag: "+ el);
99  }
100 }

102 void XMLTopologyReader::ParseMolecules(const string &el, map<string, string> &attr)
103 {
104  map<string, string>::iterator iter;
105  if (el == "clear") {
106  _top->ClearMoleculeList();
107  _parser.IgnoreChilds();
108  }
109  else if (el == "rename") {
110  string molname = attr["name"];
111  string range = attr["range"];
112  if (molname == "" || range == "")
113  throw runtime_error("invalid rename tag");
114  _top->RenameMolecules(range, molname);
115  _parser.IgnoreChilds();
116  }
117  if (el == "define") {
118  string molname = attr["name"];
119  string first = attr["first"];
120  string nbeads = attr["nbeads"];
121  string nmols = attr["nmols"];
122  if (molname == "" || first == "" || nbeads == "" || nmols == "")
123  throw runtime_error("invalid define tag - missing molname, first, nbeads or nmols");
124  int first_int, nbeads_int, nmols_int;
125  try {
126  first_int = boost::lexical_cast<int>(first);
127  } catch(boost::bad_lexical_cast &) {
128  throw std::runtime_error("Cannot convert first='"+ first + "' to int");
129  }
130  if (first_int < 1) {
131  throw std::runtime_error("Attribute first is supose to be > 0, but found " + boost::lexical_cast<string>(first_int));
132  }
133  try {
134  nbeads_int = boost::lexical_cast<int>(nbeads);
135  } catch(boost::bad_lexical_cast &) {
136  throw std::runtime_error("Cannot convert nbeads='"+ nbeads + "' to int");
137  }
138  if (nbeads_int < 1) {
139  throw std::runtime_error("Attribute nbeads is supose to be > 0, but found " + boost::lexical_cast<string>(nbeads_int));
140  }
141  try {
142  nmols_int = boost::lexical_cast<int>(nmols);
143  } catch(boost::bad_lexical_cast &) {
144  throw std::runtime_error("Cannot convert nmols='"+ nmols + "' to int");
145  }
146  if (nmols_int < 1) {
147  throw std::runtime_error("Attribute nmols is supose to be > 0, but found " + boost::lexical_cast<string>(nmols_int));
148  }
149  _top->CreateMoleculesByRange(molname,first_int, nbeads_int, nmols_int);
150  _parser.IgnoreChilds();
151  }
152 }
