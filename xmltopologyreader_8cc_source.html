61 void XMLTopologyReader::ParseTopology(const string &el, map<string, string> &attr)
62 {
63  if(el == "molecules") {
64  _parser.NextHandler(this, &XMLTopologyReader::ParseMolecules);
65  }
66  else if(el == "box") {
67  matrix m;
68  m.ZeroMatrix();
69  m[0][0] = boost::lexical_cast<double>(attr["xx"]);
70  m[1][1] = boost::lexical_cast<double>(attr["yy"]);
71  m[2][2] = boost::lexical_cast<double>(attr["zz"]);
72  _top->setBox(m);
73  _parser.NextHandler(this, &XMLTopologyReader::ParseTopology);
74  }
75  else if(el == "beadtypes") {
76  _parser.NextHandler(this, &XMLTopologyReader::ParseBeadTypes);
77  }
78  else {
79  throw runtime_error("unknown tag: "+ el);
80  }
81 }

83 void XMLTopologyReader::ParseMolecules(const string &el, map<string, string> &attr)
84 {
85  map<string, string>::iterator iter;
86  if (el == "clear") {
87  _top->ClearMoleculeList();
88  _parser.IgnoreChilds();
89  }
90  else if (el == "rename") {
91  string molname = attr["name"];
92  string range = attr["range"];
93  if (molname == "" || range == "")
94  throw runtime_error("invalid rename tag");
95  _top->RenameMolecules(range, molname);
96  _parser.IgnoreChilds();
97  }
98  if (el == "define") {
99  string molname = attr["name"];
100  string first = attr["first"];
101  string nbeads = attr["nbeads"];
102  string nmols = attr["nmols"];
103  if (molname == "" && first == "" && nbeads == "" && nmols == "")
104  throw runtime_error("invalid define tag");
105  _top->CreateMoleculesByRange(molname,
106  boost::lexical_cast<int>(first),
107  boost::lexical_cast<int>(nbeads),
108  boost::lexical_cast<int>(nmols));
109  _parser.IgnoreChilds();
110  }
111 }
