50 void XMLTopologyReader::ParseRoot(Property &property) {
51  _has_base_topology = false;
52  if (property.hasAttribute("base")) {
53  ReadTopolFile(property.getAttribute<string>("base"));
54  _has_base_topology = true;
55  }
56 
57  // Iterate over keys at first level.
58  for (Property::iterator it = property.begin(); it != property.end(); ++it) {
59  if (it->name() == "h5md_particle_group") {
60  _top->setParticleGroup(it->getAttribute<string>("name"));
61  } else if (it->name() == "molecules") {
62  _mol_index = 1;
63  _bead_index = 0;
64  ParseMolecules(*it);
65  } else if (it->name() == "bonded") {
66  ParseBonded(*it);
67  } else if (it->name() == "box") {
68  ParseBox(*it);
69  } else if (it->name() == "beadtypes") {
70  ParseBeadTypes(*it);
71  } else {
72  throw runtime_error("unknown tag: topology."+ it->name());
73  }
74  }
75 }

85 void XMLTopologyReader::ParseMolecules(Property &p) {
86  for (Property::iterator it = p.begin(); it != p.end(); ++it) {
87  if (it->name() == "clear") {
88  _top->ClearMoleculeList();
89  } else if (it->name() == "rename") {
90  string molname = it->getAttribute<string>("name");
91  string range = it->getAttribute<string>("range");
92  _top->RenameMolecules(range, molname);
93  } else if (it->name() == "define" || it->name() == "molecule") {
94  string molname = it->getAttribute<string>("name");
95  int first = 0;
96  if (it->name() == "define")
97  first = it->getAttribute<int>("first");
98  int nbeads = it->getAttribute<int>("nbeads");
99  int nmols = it->getAttribute<int>("nmols");
100  if (it->name() == "define" && first < 1)
101  throw std::runtime_error("Attribute first is suppose to be > 0, but found " +
102  boost::lexical_cast<string>(it->getAttribute<string>("first")));
103  if (nbeads < 1)
104  throw std::runtime_error("Attribute nbeads is suppose to be > 0, but found " +
105  boost::lexical_cast<string>(it->getAttribute<string>("nbeads")));
106  if (nmols < 1)
107  throw std::runtime_error("Attribute nmols is suppose to be > 0, but found " +
108  boost::lexical_cast<string>(it->getAttribute<string>("nmols")));
109  if (it->name() == "define") {
110  _top->CreateMoleculesByRange(molname, first, nbeads, nmols);
111  } else {
112  if (_has_base_topology)
113  throw std::runtime_error("The defined list of beads only works for pure xml topology, without 'base' attribute.");
114  ParseMolecule(*it, molname, nbeads, nmols);
115  }
116  }
117  }
118 }

120 void XMLTopologyReader::ParseMolecule(Property &p, string molname, int nbeads, int nmols) {
121  vector<XMLBead*> xmlBeads;
122  for (Property::iterator it=p.begin(); it != p.end(); ++it) {
123  if (it->name() == "bead") {
124  string atname = it->getAttribute<string>("name");
125  string attype = it->getAttribute<string>("type");
126  double atmass, atq;
127  try {
128  atmass = it->getAttribute<double>("mass");
129  } catch (runtime_error) {
130  atmass = 1.0;
131  }
132  try {
133  atq = it->getAttribute<double>("q");
134  } catch (runtime_error) {
135  atq = 0.0;
136  }
137  XMLBead *xmlBead = new XMLBead(atname, attype, atmass, atq);
138  xmlBeads.push_back(xmlBead);
139  } else {
140  throw std::runtime_error(
141  "Wrong element under topology.molecules.molecule: " + it->name());
142  }
143  }
144  // Create molecule in topology. Replicate data.
145  int resnr = _top->ResidueCount();
146  for (int mn = 0; mn < nmols; mn++) {
147  Molecule *mi = _top->CreateMolecule(molname);
148  XMLMolecule *xmlMolecule = new XMLMolecule(molname, nmols);
149  xmlMolecule->pid = mi->getId();
150  xmlMolecule->mi = mi;
151  _molecules.insert(make_pair(molname, xmlMolecule));
152  for (vector<XMLBead*>::iterator itb = xmlBeads.begin(); itb != xmlBeads.end(); ++itb) {
153  stringstream bname;
154  XMLBead &b = **itb;
155  _top->CreateResidue(molname, resnr);
156  BeadType *type = _top->GetOrCreateBeadType(b.type);
157  Bead *bead = _top->CreateBead(1, b.name, type, resnr, b.mass, b.q);
158  bname << _mol_index << ":" << molname << ":" << b.name;
159  mi->AddBead(bead, bname.str());
160 
161  // Data for bonded terms.
162  XMLBead *b_rep = new XMLBead(b);
163  b_rep->pid = _bead_index;
164  if (xmlMolecule->name2beads.count(b.name) != 0)
165  throw std::runtime_error(
166  "Atom " + b.name + " in molecule " + molname + " already exists.");
167  xmlMolecule->name2beads.insert(make_pair(b.name, b_rep));
168  _bead_index++;
169  }
170  resnr++;
171  }
172  _mol_index++;
173 
174  // clean up
175  for (std::vector<XMLBead*>::iterator itb = xmlBeads.begin(); itb != xmlBeads.end(); ++itb) {
176  delete (*itb);
177  }
178 }

212 void XMLTopologyReader::ParseBond(Property &p) {
213  string name = p.get("name").as<string>();
214  string beads = p.get("beads").as<string>();
215  Tokenizer tok(beads, " ");
216  vector<string> bead_list;
217  tok.ToVector(bead_list);
218  if (bead_list.size() % 2 == 1)
219  throw runtime_error("Wrong number of beads in bond: " + name);
220  Interaction *ic = NULL;
221  typedef pair<MoleculesMap::iterator, MoleculesMap::iterator> MRange;
222  int b_index = 0;
223  for (vector<string>::iterator it = bead_list.begin(); it != bead_list.end(); ) {
224  BondBead b1(*(it++));
225  BondBead b2(*(it++));
226  if (b1.molname == b2.molname) {
227  // Iterates over molecules and gets atom pids.
228  MRange mRange = _molecules.equal_range(b1.molname);
229  for (MoleculesMap::iterator itm = mRange.first;
230  itm != mRange.second; ++itm) {
231  XMLMolecule &xmlMolecule = *itm->second;
232  XMLBead &xmlBead1 = *xmlMolecule.name2beads[b1.atname];
233  XMLBead &xmlBead2 = *xmlMolecule.name2beads[b2.atname];
234  ic = new IBond(xmlBead1.pid, xmlBead2.pid);
235  ic->setGroup(name);
236  ic->setIndex(b_index);
237  ic->setMolecule(xmlMolecule.pid);
238  xmlMolecule.mi->AddInteraction(ic);
239  _top->AddBondedInteraction(ic);
240  b_index++;
241  }
242  } else {
243  throw std::runtime_error(
244  "Beads from different molecules, not supported!");
245  }
246  }
247 }
