50 void XMLTopologyReader::ParseRoot(Property &property) {
51  _has_base_topology = false;
52  if (property.hasAttribute("base")) {
53  ReadTopolFile(property.getAttribute<string>("base"));
54  _has_base_topology = true;
55  }
56 
57  // Iterate over keys at first level.
58  for (Property::iterator it = property.begin(); it != property.end(); ++it) {
59  if (it->name() == "h5md_particle_group") {
60  _top->setParticleGroup(it->getAttribute<string>("name"));
61  } else if (it->name() == "molecules") {
62  _mol_index = 1;
63  _bead_index = 0;
64  ParseMolecules(*it);
65  } else if (it->name() == "bonded") {
66  ParseBonded(*it);
67  } else if (it->name() == "box") {
68  ParseBox(*it);
69  } else if (it->name() == "beadtypes") {
70  ParseBeadTypes(*it);
71  } else {
72  throw runtime_error("unknown tag: topology."+ it->name());
73  }
74  }
75 }

85 void XMLTopologyReader::ParseMolecules(Property &p) {
86  for (Property::iterator it = p.begin(); it != p.end(); ++it) {
87  if (it->name() == "clear") {
88  _top->ClearMoleculeList();
89  } else if (it->name() == "rename") {
90  string molname = it->getAttribute<string>("name");
91  string range = it->getAttribute<string>("range");
92  _top->RenameMolecules(range, molname);
93  } else if (it->name() == "define" || it->name() == "molecule") {
94  string molname = it->getAttribute<string>("name");
95  int first = 0;
96  if (it->name() == "define")
97  first = it->getAttribute<int>("first");
98  int nbeads = it->getAttribute<int>("nbeads");
99  int nmols = it->getAttribute<int>("nmols");
100  if (it->name() == "define" && first < 1)
101  throw std::runtime_error("Attribute first is suppose to be > 0, but found " +
102  boost::lexical_cast<string>(it->getAttribute<string>("first")));
103  if (nbeads < 1)
104  throw std::runtime_error("Attribute nbeads is suppose to be > 0, but found " +
105  boost::lexical_cast<string>(it->getAttribute<string>("nbeads")));
106  if (nmols < 1)
107  throw std::runtime_error("Attribute nmols is suppose to be > 0, but found " +
108  boost::lexical_cast<string>(it->getAttribute<string>("nmols")));
109  if (it->name() == "define") {
110  _top->CreateMoleculesByRange(molname, first, nbeads, nmols);
111  } else {
112  if (_has_base_topology)
113  throw std::runtime_error("The defined list of beads only works for pure xml topology, without 'base' attribute.");
114  ParseMolecule(*it, molname, nbeads, nmols);
115  }
116  }
117  }

120 void XMLTopologyReader::ParseMolecule(Property &p, string molname, int nbeads, int nmols) {
121  vector<XMLBead*> xmlBeads;
122  vector<int> xmlResidues;
123  for (Property::iterator it=p.begin(); it != p.end(); ++it) {
124  if (it->name() == "bead") {
125  string atname = it->getAttribute<string>("name");
126  string attype = it->getAttribute<string>("type");
127  double atmass, atq;
128  int resid;
129  try {
130  atmass = it->getAttribute<double>("mass");
131  } catch (runtime_error&) {
132  atmass = 1.0;
133  }
134  try {
135  atq = it->getAttribute<double>("q");
136  } catch (runtime_error&) {
137  atq = 0.0;
138  }
139  try {
140  resid = it->getAttribute<int>("resid");
141  if (resid<=0) {
142  throw std::invalid_argument(
143  "Residue count for beads in topology.molecules.molecule has to be greater than zero");
144  }
145  } catch (runtime_error&) {
146  resid = -1;
147  }
148  if (!xmlResidues.empty()) {
149  if (xmlResidues.back() != resid && xmlResidues.back() != resid-1) {
150  throw std::invalid_argument(
151  "Residue count for beads in topology.molecules.molecule does not increase monotonically in steps of 1");
152  }
153  if (xmlResidues.back() != -1 && resid == -1) {
154  throw std::invalid_argument(
155  "Residue count for beads in topology.molecules.molecule has to be declared for all beads or for none");
156  }
157  }
158  xmlResidues.push_back(resid);
159  XMLBead *xmlBead = new XMLBead(atname, attype, atmass, atq);
160  xmlBeads.push_back(xmlBead);
161  } else {
162  throw std::runtime_error(
163  "Wrong element under topology.molecules.molecule: " + it->name());
164  }
165  }
166  if (xmlResidues.size() != xmlBeads.size())
167  throw std::runtime_error(
168  "Number of elements in bead-vector and residue-vector are not identical");
169  // Create molecule in topology. Replicate data.
170  int resnr = _top->ResidueCount();
171  if (!xmlResidues.empty()) {
172  if (xmlResidues.front() != resnr+1 && xmlResidues.front() != -1) {
173  throw std::runtime_error(
174  "Residue count for beads in topology.molecules.molecule has to be greater than the number of residues already in the topology");
175  }
176  }
177  for (int mn = 0; mn < nmols; mn++) {
178  Molecule *mi = _top->CreateMolecule(molname);
179  XMLMolecule *xmlMolecule = new XMLMolecule(molname, nmols);
180  xmlMolecule->pid = mi->getId();
181  xmlMolecule->mi = mi;
182  _molecules.insert(make_pair(molname, xmlMolecule));
183  vector<int>::iterator resit = xmlResidues.begin();
184  for (vector<XMLBead*>::iterator itb = xmlBeads.begin(); itb != xmlBeads.end(); ++itb, ++resit) {
185  stringstream bname;
186  XMLBead &b = **itb;
187  if (*resit != -1) {
188  if (_top->ResidueCount() < *resit) {
189  resnr = *resit - 1;
190  _top->CreateResidue(molname, resnr);
191  }
192  } else {
193  _top->CreateResidue(molname, resnr);
194  }
195  BeadType *type = _top->GetOrCreateBeadType(b.type);
196  Bead *bead = _top->CreateBead(1, b.name, type, resnr, b.mass, b.q);
197  bname << _mol_index << ":" << molname << ":" << b.name;
198  mi->AddBead(bead, bname.str());
199 
200  // Data for bonded terms.
201  XMLBead *b_rep = new XMLBead(b);
202  b_rep->pid = _bead_index;
203  if (xmlMolecule->name2beads.count(b.name) != 0)
204  throw std::runtime_error(
205  "Atom " + b.name + " in molecule " + molname + " already exists.");
206  xmlMolecule->name2beads.insert(make_pair(b.name, b_rep));
207  _bead_index++;
208  }
209  resnr++;
210  }
211  _mol_index++;
212 
213  // clean up
214  for (std::vector<XMLBead*>::iterator itb = xmlBeads.begin(); itb != xmlBeads.end(); ++itb) {
215  delete (*itb);
216  }
217 }

251 void XMLTopologyReader::ParseBond(Property &p) {
252  string name = p.get("name").as<string>();
253  string beads = p.get("beads").as<string>();
254  Tokenizer tok(beads, " \n\t");
255  vector<string> bead_list;
256  tok.ToVector(bead_list);
257  if (bead_list.size() % 2 == 1)
258  throw runtime_error("Wrong number of beads in bond: " + name);
259  Interaction *ic = NULL;
260  typedef pair<MoleculesMap::iterator, MoleculesMap::iterator> MRange;
261  int b_index = 0;
262  for (vector<string>::iterator it = bead_list.begin(); it != bead_list.end(); ) {
263  BondBead b1(*(it++));
264  BondBead b2(*(it++));
265  if (b1.molname == b2.molname) {
266  // Iterates over molecules and gets atom pids.
267  MRange mRange = _molecules.equal_range(b1.molname);
268  for (MoleculesMap::iterator itm = mRange.first;
269  itm != mRange.second; ++itm) {
270  XMLMolecule &xmlMolecule = *itm->second;
271  XMLBead &xmlBead1 = *xmlMolecule.name2beads[b1.atname];
272  XMLBead &xmlBead2 = *xmlMolecule.name2beads[b2.atname];
273  ic = new IBond(xmlBead1.pid, xmlBead2.pid);
274  ic->setGroup(name);
275  ic->setIndex(b_index);
276  ic->setMolecule(xmlMolecule.pid);
277  xmlMolecule.mi->AddInteraction(ic);
278  _top->AddBondedInteraction(ic);
279  b_index++;
280  }
281  } else {
282  throw std::runtime_error(
283  "Beads from different molecules, not supported!");
284  }
285  }
286 }
