54 void XMLTopologyReader::ParseRoot(tools::Property &property) {
55  _has_base_topology = false;
56  if (property.hasAttribute("base")) {
57  ReadTopolFile(property.getAttribute<string>("base"));
58  _has_base_topology = true;
59  }
60 
61  // Iterate over keys at first level.
62  for (auto &it : property) {
63  if (it.name() == "h5md_particle_group") {
64  _top->setParticleGroup(it.getAttribute<string>("name"));
65  } else if (it.name() == "molecules") {
66  _mol_index = 1;
67  _bead_index = 0;
68  ParseMolecules(it);
69  } else if (it.name() == "bonded") {
70  ParseBonded(it);
71  } else if (it.name() == "box") {
72  ParseBox(it);
73  } else if (it.name() == "beadtypes") {
74  ParseBeadTypes(it);
75  } else {
76  throw runtime_error("unknown tag: topology." + it.name());
77  }
78  }
79 }

89 void XMLTopologyReader::ParseMolecules(tools::Property &p) {
90  for (auto &it : p) {
91  if (it.name() == "clear") {
92  _top->ClearMoleculeList();
93  } else if (it.name() == "rename") {
94  string molname = it.getAttribute<string>("name");
95  string range = it.getAttribute<string>("range");
96  _top->RenameMolecules(range, molname);
97  } else if (it.name() == "define" || it.name() == "molecule") {
98  string molname = it.getAttribute<string>("name");
99  Index first = 0;
100  if (it.name() == "define") {
101  first = it.getAttribute<Index>("first");
102  }
103  Index nbeads = it.getAttribute<Index>("nbeads");
104  Index nmols = it.getAttribute<Index>("nmols");
105  if (it.name() == "define" && first < 1) {
106  throw std::runtime_error(
107  "Attribute first is suppose to be > 0, but found " +
108  boost::lexical_cast<string>(it.getAttribute<string>("first")));
109  }
110  if (nbeads < 1) {
111  throw std::runtime_error(
112  "Attribute nbeads is suppose to be > 0, but found " +
113  boost::lexical_cast<string>(it.getAttribute<string>("nbeads")));
114  }
115  if (nmols < 1) {
116  throw std::runtime_error(
117  "Attribute nmols is suppose to be > 0, but found " +
118  boost::lexical_cast<string>(it.getAttribute<string>("nmols")));
119  }
120  if (it.name() == "define") {
121  _top->CreateMoleculesByRange(molname, first, nbeads, nmols);
122  } else {
123  if (_has_base_topology) {
124  throw std::runtime_error(
125  "The defined list of beads only works for pure xml topology, "
126  "without 'base' attribute.");
127  }
128  ParseMolecule(it, molname, nmols);
129  }
130  }
131  }

134 void XMLTopologyReader::ParseMolecule(tools::Property &p, string molname,
135  Index nmols) {
136  vector<XMLBead *> xmlBeads;
137  vector<Index> xmlResidues;
138  for (auto &it : p) {
139  if (it.name() == "bead") {
140  string atname = it.getAttribute<string>("name");
141  string attype = it.getAttribute<string>("type");
142  double atmass, atq;
143  Index resid;
144  try {
145  atmass = it.getAttribute<double>("mass");
146  } catch (runtime_error &) {
147  atmass = 1.0;
148  }
149  try {
150  atq = it.getAttribute<double>("q");
151  } catch (runtime_error &) {
152  atq = 0.0;
153  }
154  try {
155  resid = it.getAttribute<Index>("resid");
156  if (resid <= 0) {
157  throw std::invalid_argument(
158  "Residue count for beads in topology.molecules.molecule has to "
159  "be greater than zero");
160  }
161  } catch (runtime_error &) {
162  resid = -1;
163  }
164  if (!xmlResidues.empty()) {
165  if (xmlResidues.back() != resid && xmlResidues.back() != resid - 1) {
166  throw std::invalid_argument(
167  "Residue count for beads in topology.molecules.molecule does not "
168  "increase monotonically in steps of 1");
169  }
170  if (xmlResidues.back() != -1 && resid == -1) {
171  throw std::invalid_argument(
172  "Residue count for beads in topology.molecules.molecule has to "
173  "be declared for all beads or for none");
174  }
175  }
176  xmlResidues.push_back(resid);
177  XMLBead *xmlBead = new XMLBead(atname, attype, atmass, atq);
178  xmlBeads.push_back(xmlBead);
179  } else {
180  throw std::runtime_error(
181  "Wrong element under topology.molecules.molecule: " + it.name());
182  }
183  }
184  if (xmlResidues.size() != xmlBeads.size()) {
185  throw std::runtime_error(
186  "Number of elements in bead-vector and residue-vector are not "
187  "identical");
188  }
189  // Create molecule in topology. Replicate data.
190  Index resnr = _top->ResidueCount();
191  if (!xmlResidues.empty()) {
192  if (xmlResidues.front() != resnr + 1 && xmlResidues.front() != -1) {
193  throw std::runtime_error(
194  "Residue count for beads in topology.molecules.molecule has to be "
195  "greater than the number of residues already in the topology");
196  }
197  }
198  for (Index mn = 0; mn < nmols; mn++) {
199  Molecule *mi = _top->CreateMolecule(molname);
200  XMLMolecule *xmlMolecule = new XMLMolecule(molname, nmols);
201  xmlMolecule->pid = mi->getId();
202  xmlMolecule->mi = mi;
203  _molecules.insert(make_pair(molname, xmlMolecule));
204  vector<Index>::iterator resit = xmlResidues.begin();
205  for (vector<XMLBead *>::iterator itb = xmlBeads.begin();
206  itb != xmlBeads.end(); ++itb, ++resit) {
207  stringstream bname;
208  XMLBead &b = **itb;
209  if (*resit != -1) {
210  if (_top->ResidueCount() < *resit) {
211  resnr = *resit - 1;
212  _top->CreateResidue(molname, resnr);
213  }
214  } else {
215  _top->CreateResidue(molname, resnr);
216  }
217 
218  if (!_top->BeadTypeExist(b.type)) {
219  _top->RegisterBeadType(b.type);
220  }
221  Bead *bead =
222  _top->CreateBead(Bead::spherical, b.name, b.type, resnr, b.mass, b.q);
223  bname << _mol_index << ":" << molname << ":" << b.name;
224  mi->AddBead(bead, bname.str());
225 
226  // Data for bonded terms.
227  XMLBead *b_rep = new XMLBead(b);
228  b_rep->pid = _bead_index;
229  if (xmlMolecule->name2beads.count(b.name) != 0) {
230  throw std::runtime_error("Atom " + b.name + " in molecule " + molname +
231  " already exists.");
232  }
233  xmlMolecule->name2beads.insert(make_pair(b.name, b_rep));
234  _bead_index++;
235  }
236  resnr++;
237  }
238 
239  // clean up
240  for (auto &xmlBead : xmlBeads) {
241  delete xmlBead;
242  }
243 }

278 void XMLTopologyReader::ParseBond(tools::Property &p) {
279  string name = p.get("name").as<string>();
280  string beads = p.get("beads").as<string>();
281  tools::Tokenizer tok(beads, " \n\t");
282  vector<string> bead_list = tok.ToVector();
283  if (bead_list.size() % 2 == 1) {
284  throw runtime_error("Wrong number of beads in bond: " + name);
285  }
286  Interaction *ic = nullptr;
287  typedef pair<MoleculesMap::iterator, MoleculesMap::iterator> MRange;
288  Index b_index = 0;
289  for (vector<string>::iterator it = bead_list.begin();
290  it != bead_list.end();) {
291  BondBead b1(*(it++));
292  BondBead b2(*(it++));
293  if (b1.molname == b2.molname) {
294  // Iterates over molecules and gets atom pids.
295  MRange mRange = _molecules.equal_range(b1.molname);
296  for (MoleculesMap::iterator itm = mRange.first; itm != mRange.second;
297  ++itm) {
298  XMLMolecule &xmlMolecule = *itm->second;
299  XMLBead &xmlBead1 = *xmlMolecule.name2beads[b1.atname];
300  XMLBead &xmlBead2 = *xmlMolecule.name2beads[b2.atname];
301  ic = new IBond(xmlBead1.pid, xmlBead2.pid);
302  ic->setGroup(name);
303  ic->setIndex(b_index);
304  ic->setMolecule(xmlMolecule.pid);
305  xmlMolecule.mi->AddInteraction(ic);
306  _top->AddBondedInteraction(ic);
307  b_index++;
308  }
309  } else {
310  throw std::runtime_error(
311  "Beads from different molecules, not supported!");
312  }
313  }
314 }
