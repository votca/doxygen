66 void Neighborlist::Initialize(Property *options) {
67 
68  // update options with the VOTCASHARE defaults
69  UpdateWithDefaults( options,"ctp" );
70  std::string key = "options." + Identify();
71 
72  list< Property* > segs = options->Select(key+".segments");
73  list< Property* > ::iterator segsIt;
74 
75  for (segsIt = segs.begin();
76  segsIt != segs.end();
77  segsIt++) {
78 
79  string types = (*segsIt)->get("type").as<string>();
80  double cutoff = (*segsIt)->get("cutoff").as<double>();
81 
82  Tokenizer tok(types, " ");
83  vector< string > names;
84  tok.ToVector(names);
85 
86  if (names.size() != 2) {
87  cout << "ERROR: Faulty pair definition for cut-off's: "
88  << "Need two segment names separated by a space" << endl;
89  throw std::runtime_error("Error in options file.");
90  }
91 
92  _cutoffs[names[0]][names[1]] = cutoff;
93  _cutoffs[names[1]][names[0]] = cutoff;
94 
95  }
96 
97  if (options->exists(key+".constant")) {
98  _useConstantCutoff = true;
99  _constantCutoff = options->get(key+".constant").as< double >();
100  }
101  else {
102  _useConstantCutoff = false;
103  }
104  if (options->exists(key+".generate_from")) {
105  _generate_from_file = true;
106  _generate_from = options->get(key+".generate_from").as< string >();
107  }
108  else {
109  _generate_from_file = false;
110  _generate_from = "nofile";
111  }
112  if (options->exists(key+".generate_unsafe")) {
113  _generate_unsafe = true;
114  }
115  else {
116  _generate_unsafe = false;
117  }
118 
119  // if superexchange is given
120  if (options->exists(key + ".superexchange")) {
121  list< Property* > _se = options->Select(key + ".superexchange");
122  list< Property* > ::iterator seIt;
123 
124  for (seIt = _se.begin(); seIt != _se.end(); seIt++) {
125  string types = (*seIt)->get("type").as<string>();
126  QMNBList::SuperExchangeType* _su = new QMNBList::SuperExchangeType(types);
127  _superexchange.push_back(_su);
128  }
129  }

133 bool Neighborlist::EvaluateFrame(Topology *top) {
134 
135  top->NBList().Cleanup();
136 
137  if (_generate_from_file) {
138  this->GenerateFromFile(top, _generate_from);
139  }
140 
141  else {
142 
143  vector< Segment* > ::iterator segit1;
144  vector< Segment* > ::iterator segit2;
145  vector< Fragment* > ::iterator fragit1;
146  vector< Fragment* > ::iterator fragit2;
147 
148  double cutoff;
149  vec r1;
150  vec r2;
151 
152  if (TOOLS::globals::verbose) {
153  cout << endl << "... ..." << flush;
154  }
155 
156  for (segit1 = top->Segments().begin();
157  segit1 < top->Segments().end();
158  segit1++) {
159 
160  Segment *seg1 = *segit1;
161  if (TOOLS::globals::verbose) {
162  cout << "\r ... ... NB List Seg " << seg1->getId() << flush;
163  }
164 
165  for (segit2 = segit1 + 1;
166  segit2 < top->Segments().end();
167  segit2++) {
168 
169  Segment *seg2 = *segit2;
170 
171  if (!_useConstantCutoff) {
172  // Find cut-off
173  try {
174  cutoff = _cutoffs.at(seg1->getName())
175  .at(seg2->getName());
176  }
177  catch (out_of_range) {
178  cout << "ERROR: No cut-off specified for segment pair "
179  << seg1->getName() << " | " << seg2->getName()
180  << ". " << endl;
181  throw std::runtime_error("Missing input in options.");
182  }
183  }
184 
185  else { cutoff = _constantCutoff; }
186 
187 
188  bool stopLoop = false;
189  for (fragit1 = seg1->Fragments().begin();
190  fragit1 < seg1->Fragments().end();
191  fragit1 ++) {
192 
193  if (stopLoop) { break; }
194 
195  for (fragit2 = seg2->Fragments().begin();
196  fragit2 < seg2->Fragments().end();
197  fragit2++) {
198 
199 
200  r1 = (*fragit1)->getPos();
201  r2 = (*fragit2)->getPos();
202  if( abs( top->PbShortestConnect(r1, r2) ) > cutoff ) {
203  continue;
204  }
205  else {
206  top->NBList().Add(seg1, seg2);
207  stopLoop = true;
208  break;
209  }
210 
211  } /* exit loop frag2 */
212  } /* exit loop frag1 */
213  } /* exit loop seg2 */
214 
215  // break;
216  } /* exit loop seg1 */
217 
218  }
219 
220  cout << endl << " ... ... Created " << top->NBList().size() << " direct pairs.";
221 
222  // add superexchange pairs
223  top->NBList().setSuperExchangeTypes(_superexchange);
224  top->NBList().GenerateSuperExchange();
225 
226  // DEBUG output
227  if (votca::tools::globals::verbose) {
228 
229  Property bridges_summary;
230  Property *_bridges = &bridges_summary.add("bridges","");
231 
232  cout << "Bridged Pairs \n [idA:idB] com distance" << endl;
233  for (QMNBList::iterator ipair = top->NBList().begin(); ipair != top->NBList().end(); ++ipair) {
234  QMPair *pair = *ipair;
235  Segment* segment1 = pair->Seg1PbCopy();
236  Segment* segment2 = pair->Seg2PbCopy();
237 
238  cout << " [" << segment1->getId() << ":" << segment2->getId()<< "] "
239  << pair->Dist()<< " bridges: "
240  << (pair->getBridgingSegments()).size()
241  << " type: "
242  << pair->getType()
243  << " | " << flush;
244 
245  vector<Segment*> bsegments = pair->getBridgingSegments();
246 
247  Property *_pair_property = &_bridges->add("pair","");
248 
249  _pair_property->setAttribute("id1", segment1->getId());
250  _pair_property->setAttribute("id2", segment2->getId());
251  _pair_property->setAttribute("name1", segment1->getName());
252  _pair_property->setAttribute("name2", segment2->getName());
253  _pair_property->setAttribute("r12", pair->Dist());
254 
255  Property *_bridge_property = &_pair_property->add("bridge","");
256 
257  for ( vector<Segment*>::iterator itb = bsegments.begin(); itb != bsegments.end(); itb++ ) {
258  cout << (*itb)->getId() << " " ;
259  _bridge_property->setAttribute("id", (*itb)->getId());
260  }
261 
262  cout << endl;
263  }
264  //cout << bridges_summary;
265  }
266 
267  return true;
268 }
