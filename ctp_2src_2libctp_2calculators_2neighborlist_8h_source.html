72 void Neighborlist::Initialize(Property *options) {
73 
74  // update options with the VOTCASHARE defaults
75  UpdateWithDefaults( options,"ctp" );
76  std::string key = "options." + Identify();
77 
78  // properties of the logger
79  _log.setPreface(logINFO, "\n... ...");
80  _log.setPreface(logERROR, "\n... ...");
81  _log.setPreface(logWARNING, "\n... ...");
82  _log.setPreface(logDEBUG, "\n... ...");
83 
84  if (TOOLS::globals::verbose) {
85  _log.setReportLevel( logDEBUG );
86  } else {
87  _log.setReportLevel( logINFO );
88  }
89 
90  list< Property* > segs = options->Select(key+".segments");
91  list< Property* > ::iterator segsIt;
92 
93  for (segsIt = segs.begin(); segsIt != segs.end(); segsIt++) {
94 
95  double cutoff = (*segsIt)->get("cutoff").as<double>();
96 
97  // get pairs of segment types
98  string types = (*segsIt)->get("type").as<string>();
99 
100  Tokenizer tok(types, " ");
101  vector< string > names;
102  tok.ToVector(names);
103 
104  if (names.size() != 2) {
105  CTP_LOG(logERROR,_log) << "Faulty pair definition for cut-off's: "
106  << "Need two segment names separated by a space" << std::flush;
107  throw std::runtime_error("Error in options file.");
108  }
109 
110  _cutoffs[names[0]][names[1]] = cutoff;
111  _cutoffs[names[1]][names[0]] = cutoff;
112 
113  // get active fragments for each pair of segments
114 
115  std::string fragments;
116  _use_active_fragments = false;
117  if ((*segsIt)->exists("fragments")) {
118  // if * is provided, do not use fragments filtering
119  fragments = (*segsIt)->get("fragments").as<string>();
120 
121  std::size_t found = fragments.find("*");
122  if (found == std::string::npos) {
123  _use_active_fragments = true;
124  CTP_LOG(logWARNING,_log) << "Using filtering by fragments: "
125  << fragments << std::flush;
126  }
127 
128  } else {
129  CTP_LOG(logWARNING,_log)
130  << "From v1.6 you can use a list of fragments (or '*')."
131  << std::flush;
132  }
133 
134  _active_fragments[names[0]][names[1]] = fragments;
135  _active_fragments[names[1]][names[0]] = fragments;
136 
137  }
138 
139  if (options->exists(key+".constant")) {
140  _useConstantCutoff = true;
141  _constantCutoff = options->get(key+".constant").as< double >();
142  }
143  else {
144  _useConstantCutoff = false;
145  }
146 
147  _generate_from_file = false;
148  if (options->exists(key+".file")) {
149  _file_name = options->get(key+".file").as< string >();
150  if ( _file_name.size() != 0 ) _generate_from_file = true;
151  }
152 
153  // if superexchange is given
154  if (options->exists(key + ".superexchange")) {
155  list< Property* > _se = options->Select(key + ".superexchange");
156  list< Property* > ::iterator seIt;
157 
158  for (seIt = _se.begin(); seIt != _se.end(); seIt++) {
159  string types = (*seIt)->get("type").as<string>();
160  QMNBList::SuperExchangeType* _su = new QMNBList::SuperExchangeType(types);
161  _superexchange.push_back(_su);
162  }
163  }
164 
165 }

167 bool Neighborlist::EvaluateFrame(Topology *top) {
168 
169  top->NBList().Cleanup();
170 
171  if (_generate_from_file) {
172  this->GenerateFromFile(top, _file_name);
173  }
174  else {
175 
176  vector< Segment* > ::iterator segit1;
177  vector< Segment* > ::iterator segit2;
178  vector< Fragment* > ::iterator fragit1;
179  vector< Fragment* > ::iterator fragit2;
180 
181  double cutoff;
182  std::string fragment_list;
183 
184  vec r1;
185  vec r2;
186 
187  for (segit1 = top->Segments().begin();
188  segit1 < top->Segments().end();
189  segit1++) {
190 
191  Segment *seg1 = *segit1;
192  CTP_LOG(logDEBUG,_log) << "NB List Seg " << seg1->getId() << std::flush;
193 
194  for (segit2 = segit1 + 1;
195  segit2 < top->Segments().end();
196  segit2++) {
197 
198  Segment *seg2 = *segit2;
199 
200  if (!_useConstantCutoff) {
201  // Find cut-off
202  try {
203  cutoff = _cutoffs.at(seg1->getName())
204  .at(seg2->getName());
205 
206  fragment_list = _active_fragments.at(seg1->getName())
207  .at(seg2->getName());
208  }
209  catch (out_of_range&) {
210  CTP_LOG(logERROR,_log) << "ERROR: No cut-off specified for segment pair "
211  << seg1->getName() << " | " << seg2->getName()
212  << ". " << std::flush;
213  throw std::runtime_error("Missing input in options.");
214  }
215  }
216 
217  else { cutoff = _constantCutoff; }
218 
219 
220  bool stopLoop = false;
221  for (fragit1 = seg1->Fragments().begin();
222  fragit1 < seg1->Fragments().end();
223  fragit1 ++) {
224 
225  // check if this fragment is active
226  std::string _frag1name = (*fragit1)->getName();
227  std::size_t found1 = fragment_list.find(_frag1name);
228  if ( _use_active_fragments && found1 == std::string::npos) { continue; }
229 
230  if (stopLoop) { break; }
231 
232  for (fragit2 = seg2->Fragments().begin();
233  fragit2 < seg2->Fragments().end();
234  fragit2++) {
235 
236  // check if this fragment is active
237  std::string _frag2name = (*fragit1)->getName();
238  std::size_t found2 = fragment_list.find(_frag2name);
239  if (_use_active_fragments && found2 == std::string::npos) { continue; }
240 
241  r1 = (*fragit1)->getPos();
242  r2 = (*fragit2)->getPos();
243  if( abs( top->PbShortestConnect(r1, r2) ) > cutoff ) {
244  continue;
245  }
246  else {
247  top->NBList().Add(seg1, seg2);
248  stopLoop = true;
249  break;
250  }
251 
252  } /* exit loop frag2 */
253  } /* exit loop frag1 */
254  } /* exit loop seg2 */
255 
256  // break;
257  } /* exit loop seg1 */
258 
259  }
260 
261  // add superexchange pairs
262  top->NBList().setSuperExchangeTypes(_superexchange);
263  top->NBList().GenerateSuperExchange();
264 
265  // short summary at the end
266  std::map<int, int> npairs;
267  for ( QMNBList::iterator pit = top->NBList().begin(); pit != top->NBList().end(); pit++ ) {
268  QMPair *pair = (*pit);
269  npairs[pair->getType()] += 1;
270  }
271 
272  CTP_LOG(logINFO,_log) << "Created " << top->NBList().size() << " pairs." << std::flush;
273  CTP_LOG(logINFO,_log) << "Hopping only pairs: " << npairs[QMPair::Hopping] << std::flush;
274  CTP_LOG(logINFO,_log) << "Superexchange pairs: " << npairs[QMPair::SuperExchange] << std::flush;
275  CTP_LOG(logINFO,_log) << "Superexchange and hopping pairs: " << npairs[QMPair::SuperExchangeAndHopping] << std::flush;
276 
277  // DEBUG output
278  if (votca::tools::globals::verbose) {
279 
280  Property bridges_summary;
281  Property *_bridges = &bridges_summary.add("bridges","");
282 
283  CTP_LOG(logDEBUG,_log) << "Bridged Pairs \n [idA:idB] com distance" << std::flush;
284  for (QMNBList::iterator ipair = top->NBList().begin(); ipair != top->NBList().end(); ++ipair) {
285  QMPair *pair = *ipair;
286  Segment* segment1 = pair->Seg1PbCopy();
287  Segment* segment2 = pair->Seg2PbCopy();
288 
289  CTP_LOG(logDEBUG,_log) << " [" << segment1->getId() << ":" << segment2->getId()<< "] "
290  << pair->Dist()<< " bridges: "
291  << (pair->getBridgingSegments()).size()
292  << " type: "
293  << pair->getType()
294  << " | " << std::flush;
295 
296  vector<Segment*> bsegments = pair->getBridgingSegments();
297 
298  Property *_pair_property = &_bridges->add("pair","");
299 
300  _pair_property->setAttribute("id1", segment1->getId());
301  _pair_property->setAttribute("id2", segment2->getId());
302  _pair_property->setAttribute("name1", segment1->getName());
303  _pair_property->setAttribute("name2", segment2->getName());
304  _pair_property->setAttribute("r12", pair->Dist());
305 
306  Property *_bridge_property = &_pair_property->add("bridge","");
307 
308  for ( vector<Segment*>::iterator itb = bsegments.begin(); itb != bsegments.end(); itb++ ) {
309  CTP_LOG(logDEBUG,_log) << (*itb)->getId() << " " ;
310  _bridge_property->setAttribute("id", (*itb)->getId());
311  }
312 
313  CTP_LOG(logDEBUG,_log) << std::flush;
314  }
315  }
316 
317  std::cout << _log;
318  return true;
319 }

326 void Neighborlist::GenerateFromFile(Topology *top, string filename) {
327 
328  std::string line;
329  std::ifstream intt;
330  intt.open(filename.c_str());
331 
332  if (intt.is_open() ) {
333  while ( intt.good() ) {
334 
335  std::getline(intt, line);
336  vector<string> split;
337  Tokenizer toker(line, " \t");
338  toker.ToVector(split);
339 
340  if ( !split.size() ||
341  split[0] == "!" ||
342  split[0].substr(0,1) == "!" ) { continue; }
343 
344  int seg1id = boost::lexical_cast<int>(split[1]);
345  int seg2id = boost::lexical_cast<int>(split[2]);
346 
347  Segment* seg1 = top->getSegment(seg1id);
348  Segment* seg2 = top->getSegment(seg2id);
349 
350  string seg1name = boost::lexical_cast<string>(split[3]);
351  string seg2name = boost::lexical_cast<string>(split[4]);
352  assert(seg1->getName() == seg1name);
353  assert(seg2->getName() == seg2name);
354 
355  (void)top->NBList().Add(seg1,seg2);
356 
357  } /* Exit loop over lines */
358  }
359  else {
360  CTP_LOG(logERROR,_log) << "ERROR: No such file " << filename << std::flush;
361  throw std::runtime_error("Supply input file.");
362  }
363 }
