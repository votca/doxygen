27 std::vector<std::vector<double>> CubeFile_Writer::CalculateValues(
28  const Orbitals& orb, QMState state, bool dostateonly,
29  const Regular_Grid& grid) const {
30 
31  bool do_amplitude = (state.Type().isSingleParticleState());
32  if (do_amplitude) {
33  Eigen::VectorXd amplitude;
34  if (state.Type() == QMStateType::DQPstate) {
35  Index amplitudeindex = state.StateIdx() - orb.getGWAmin();
36  amplitude = orb.CalculateQParticleAORepresentation().col(amplitudeindex);
37  } else {
38  amplitude = orb.MOs().eigenvectors().col(state.StateIdx());
39  }
40  AmplitudeIntegration<Regular_Grid> ampl(grid);
41  return ampl.IntegrateAmplitude(amplitude);
42  } else {
43  Eigen::MatrixXd mat;
44  if (state.Type().isExciton() && dostateonly) {
45  std::array<Eigen::MatrixXd, 2> DMAT =
46  orb.DensityMatrixExcitedState(state);
47  mat = DMAT[1] - DMAT[0];
48  } else {
49  mat = orb.DensityMatrixFull(state);
50  }
51  DensityIntegration<Regular_Grid> density(grid);
52  density.IntegrateDensity(mat);
53  return density.getDensities();
54  }
55 }

57 std::vector<double> FlattenValues(
58  const std::vector<std::vector<double>>& gridvalues) {
59  Index size = 0;
60  for (const auto& box : gridvalues) {
61  size += Index(box.size());
62  }
63  std::vector<double> result;
64  result.reserve(size);
65  for (const auto& box : gridvalues) {
66  result.insert(result.end(), box.begin(), box.end());
67  }
68  return result;
69 }

71 void CubeFile_Writer::WriteFile(const std::string& filename,
72  const Orbitals& orb, QMState state,
73  bool dostateonly) const {
74 
75  Regular_Grid grid;
76  Eigen::Array3d padding = Eigen::Array3d::Ones() * _padding;
77  AOBasis basis = orb.SetupDftBasis();
78  XTP_LOG(Log::info, _log) << " Loaded DFT Basis Set " << orb.getDFTbasisName()
79  << std::flush;
80  grid.GridSetup(_steps, padding, orb.QMAtoms(), basis);
81  XTP_LOG(Log::info, _log) << " Calculating Gridvalues " << std::flush;
82  auto temp = CalculateValues(orb, state, dostateonly, grid);
83  std::vector<double> grid_values = FlattenValues(temp);
84  XTP_LOG(Log::info, _log) << " Calculated Gridvalues " << std::flush;
85  bool do_amplitude = (state.Type().isSingleParticleState());
86  std::ofstream out(filename);
87  if (!out.is_open()) {
88  throw std::runtime_error("Bad file handle: " + filename);
89  }
90 
91  // write cube header
92  if (state.isTransition()) {
93  out << boost::format("Transition state: %1$s \n") % state.ToString();
94  } else if (do_amplitude) {
95  out << boost::format("%1$s with energy %2$f eV \n") % state.ToString() %
96  (orb.getExcitedStateEnergy(state) * tools::conv::hrt2ev);
97  } else {
98  if (dostateonly) {
99  out << boost::format(
100  "Difference electron density of excited state %1$s \n") %
101  state.ToString();
102  } else {
103  out << boost::format("Total electron density of %1$s state\n") %
104  state.ToString();
105  }
106  }
107  Eigen::Vector3d start = grid.getStartingPoint();
108  out << "Created by VOTCA-XTP \n";
109  if (do_amplitude) {
110  out << boost::format("-%1$lu %2$f %3$f %4$f \n") % orb.QMAtoms().size() %
111  start.x() % start.y() % start.z();
112  } else {
113  out << boost::format("%1$lu %2$f %3$f %4$f \n") % orb.QMAtoms().size() %
114  start.x() % start.y() % start.z();
115  }
116  Eigen::Array<Index, 3, 1> steps = grid.getSteps();
117  Eigen::Array3d stepsizes = grid.getStepSizes();
118  out << boost::format("%1$d %2$f 0.0 0.0 \n") % steps.x() % stepsizes.x();
119  out << boost::format("%1$d 0.0 %2$f 0.0 \n") % steps.y() % stepsizes.y();
120  out << boost::format("%1$d 0.0 0.0 %2$f \n") % steps.z() % stepsizes.z();
121  tools::Elements elements;
122  for (const QMAtom& atom : orb.QMAtoms()) {
123  double x = atom.getPos().x();
124  double y = atom.getPos().y();
125  double z = atom.getPos().z();
126  std::string element = atom.getElement();
127  Index atnum = elements.getEleNum(element);
128  Index crg = atom.getNuccharge();
129  out << boost::format("%1$d %2$d %3$f %4$f %5$f\n") % atnum % crg % x % y %
130  z;
131  }
132 
133  if (do_amplitude) {
134  out << boost::format(" 1 %1$d \n") % (state.StateIdx() + 1);
135  }
136 
137  Eigen::TensorMap<Eigen::Tensor<double, 3>> gridvalues(
138  grid_values.data(), steps.z(), steps.y(), steps.x());
139  for (Index ix = 0; ix < steps.x(); ix++) {
140  for (Index iy = 0; iy < steps.y(); iy++) {
141  Index Nrecord = 0;
142  for (Index iz = 0; iz < steps.z(); iz++) {
143  out << boost::format("%1$E ") % gridvalues(iz, iy, ix);
144  Nrecord++;
145  if (Nrecord == 6 || iz == (steps.z() - 1)) {
146  out << "\n";
147  Nrecord = 0;
148  }
149  }
150  }
151  }
152  out.close();
153 }
