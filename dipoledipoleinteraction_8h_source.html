56  DipoleDipoleInteraction(const eeInteractor& interactor,
57  const std::vector<PolarSegment>& segs)
58  : _interactor(interactor) {
59  _size = 0;
60  for (const PolarSegment& seg : segs) {
61  _size += 3 * seg.size();
62  }
63  _sites.reserve(_size / 3);
64  for (const PolarSegment& seg : segs) {

106  // this is not a fast method
107  const double& operator()(const Index i, const Index j) const {
108  Index seg1id = Index(i / 3);
109  Index xyz1 = Index(i % 3);
110  Index seg2id = Index(j / 3);
111  Index xyz2 = Index(j % 3);
112 
113  if (seg1id == seg2id) {
114  return _sites[seg1id]->getPInv()(xyz1, xyz2);
115  } else {
116  const PolarSite& site1 = *_sites[seg1id];
117  const PolarSite& site2 = *_sites[seg2id];
118  return _interactor.FillTholeInteraction(site1, site2)(xyz1, xyz2);
119  }

122  Eigen::VectorXd multiply(const Eigen::VectorXd& v) const {
123  assert(v.size() == _size &&
124  "input vector has the wrong size for multiply with operator");
125  const Index segment_size = Index(_sites.size());
126  std::vector<Eigen::VectorXd> thread_result(OPENMP::getMaxThreads(),
127  Eigen::VectorXd::Zero(_size));
128 #pragma omp parallel for schedule(dynamic)
129  for (Index i = 0; i < segment_size; i++) {
130  const PolarSite& site1 = *_sites[i];
131  for (Index j = i + 1; j < segment_size; j++) {
132  const PolarSite& site2 = *_sites[j];
133  Eigen::Matrix3d block = _interactor.FillTholeInteraction(site1, site2);
134  thread_result[OPENMP::getThreadId()].segment<3>(3 * i) +=
135  block * v.segment<3>(3 * j);
136  thread_result[OPENMP::getThreadId()].segment<3>(3 * j) +=
137  block.transpose() * v.segment<3>(3 * i);
138  }
139  }
140 #pragma omp parallel for schedule(dynamic)
141  for (Index i = 0; i < segment_size; i++) {
142  const PolarSite& site = *_sites[i];
143  thread_result[OPENMP::getThreadId()].segment<3>(3 * i) +=
144  site.getPInv() * v.segment<3>(3 * i);
145  }
146 
147  return std::accumulate(thread_result.begin(), thread_result.end(),
148  Eigen::VectorXd::Zero(_size).eval());
149  }
150 
151  private:
152  const eeInteractor& _interactor;
153  std::vector<const PolarSite*> _sites;
154  Index _size;
155 };
