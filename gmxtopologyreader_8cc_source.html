42 bool GMXTopologyReader::ReadTopology(string file, Topology &top)
43 {
44  gmx_mtop_t mtop;
45 
46  int natoms;
47  // cleanup topology to store new data
48  top.Cleanup();
49 
50  t_inputrec ir;
51  ::matrix gbox;
52 
53 #if GMX == 52
54  (void)read_tpx((char *)file.c_str(),&ir,gbox,&natoms,NULL,NULL,&mtop);
55 #else
56  (void)read_tpx((char *)file.c_str(),&ir,gbox,&natoms,NULL,NULL,NULL,&mtop);
57 #endif
58 
59  int count=0;
60  for(int iblock=0; iblock<mtop.nmolblock; ++iblock)
61  count+=mtop.molblock[iblock].nmol;
62 
63  if(count != mtop.mols.nr ) {
64  throw runtime_error("gromacs topology contains inconsistency in molecule definitons\n\n"
65  "A possible reason is an outdated .tpr file. Please rerun grompp to generate a new tpr file.\n"
66  "If the problem remains or "
67  "you're missing the files to rerun grompp,\n contact the votca mailing list for a solution.");
68  }
69 
70  int ifirstatom = 0;
71  for(int iblock=0; iblock<mtop.nmolblock; ++iblock) {
72  gmx_moltype_t *mol
73  = &(mtop.moltype[mtop.molblock[iblock].type]);
74 
75  string molname = *(mol->name);
76 
77  int res_offset = top.ResidueCount();
78 
79  t_atoms *atoms=&(mol->atoms);
80 
81  for(int i=0; i < atoms->nres; i++) {
82  top.CreateResidue(*(atoms->resinfo[i].name));
83  }
84 
85  for(int imol=0; imol<mtop.molblock[iblock].nmol; ++imol) {
86  Molecule *mi = top.CreateMolecule(molname);
87 
88  // read the atoms
89  for(int iatom=0; iatom<mtop.molblock[iblock].natoms_mol; iatom++) {
90  t_atom *a = &(atoms->atom[iatom]);
91 
92  BeadType *type = top.GetOrCreateBeadType(*(atoms->atomtype[iatom]));
93  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), type, a->resind + res_offset, a->m, a->q);
94 
95  stringstream nm;
96  nm << bead->getResnr() + 1 - res_offset << ":" << top.getResidue(bead->getResnr())->getName() << ":" << bead->getName();
97  mi->AddBead(bead, nm.str());
98  }
99 
100  // add exclusions
101  for(int iatom=0; iatom<mtop.molblock[iblock].natoms_mol; iatom++) {
102  // read exclusions
103  t_blocka * excl = &(mol->excls);
104  // insert exclusions
105  list<Bead *> excl_list;
106  for(int k=excl->index[iatom]; k<excl->index[iatom+1]; k++) {
107  excl_list.push_back(top.getBead(excl->a[k]+ifirstatom));
108  }
109  top.InsertExclusion(top.getBead(iatom+ifirstatom), excl_list);
110  }
111  ifirstatom+=mtop.molblock[iblock].natoms_mol;
112  }
113  }
114 
115  matrix m;
116  for(int i=0; i<3; i++)
117  for(int j=0; j<3; j++)
118  m[i][j] = gbox[j][i];
119  top.setBox(m);
120 
121  return true;
122 }
