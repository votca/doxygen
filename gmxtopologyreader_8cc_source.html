60 bool GMXTopologyReader::ReadTopology(string file, Topology &top)
61 {
62  gmx_mtop_t mtop;
63 
64  int natoms;
65  // cleanup topology to store new data
66  top.Cleanup();
67  set_program_name("VOTCA");
68 
69 #if GMX == 50
70  t_inputrec ir;
71  ::matrix gbox;
72 
73  (void)read_tpx((char *)file.c_str(),&ir,gbox,&natoms,NULL,NULL,NULL,&mtop);
74 #elif GMX == 45
75  t_inputrec ir;
76  ::matrix gbox;
77  (void)read_tpx((char *)file.c_str(),&ir,gbox,&natoms,NULL,NULL,NULL,&mtop);
78 #elif GMX == 40
79  int sss; // wtf is this
80  ::real ttt,lll; // wtf is this
81  (void)read_tpx((char *)file.c_str(),&sss,&ttt,&lll,NULL,NULL,&natoms,NULL,NULL,NULL,&mtop);
82 #else
83 #error Unsupported GMX version
84 #endif
85 
86  int count=0;
87  for(int iblock=0; iblock<mtop.nmolblock; ++iblock)
88  count+=mtop.molblock[iblock].nmol;
89 
90  if(count != mtop.mols.nr ) {
91  throw runtime_error("gromacs topology contains inconsistency in molecule definitons\n\n"
92  "A possible reason is an outdated .tpr file. Please rerun grompp to generate a new tpr file.\n"
93  "If the problem remains or "
94  "you're missing the files to rerun grompp,\n contact the votca mailing list for a solution.");
95  }
96 
97  for(int iblock=0; iblock<mtop.nmolblock; ++iblock) {
98  gmx_moltype_t *mol
99  = &(mtop.moltype[mtop.molblock[iblock].type]);
100 
101  string molname = *(mol->name);
102 
103  int res_offset = top.ResidueCount();
104 
105  t_atoms *atoms=&(mol->atoms);
106 
107  for(int i=0; i < atoms->nres; i++) {
108 #if GMX == 50
109  top.CreateResidue(*(atoms->resinfo[i].name));
110 #elif GMX == 45
111  top.CreateResidue(*(atoms->resinfo[i].name));
112 #elif GMX == 40
113  top.CreateResidue(*(atoms->resname[i]));
114 #else
115 #error Unsupported GMX version
116 #endif
117  }
118 
119  int ifirstatom = 0;
120  for(int imol=0; imol<mtop.molblock[iblock].nmol; ++imol) {
121  Molecule *mi = top.CreateMolecule(molname);
122 
123  // read the atoms
124  for(int iatom=0; iatom<mtop.molblock[iblock].natoms_mol; iatom++) {
125  t_atom *a = &(atoms->atom[iatom]);
126 
127  // read exclusions
128  t_blocka * excl = &(mol->excls);
129  // insert exclusions
130  list<int> excl_list;
131  for(int k=excl->index[iatom]; k<excl->index[iatom+1]; k++) {
132  excl_list.push_back(excl->a[k]+ifirstatom);
133  }
134  top.InsertExclusion(iatom+ifirstatom, excl_list);
135 
136  BeadType *type = top.GetOrCreateBeadType(*(atoms->atomtype[iatom]));
137 #if GMX == 50
138  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), type, a->resind, a->m, a->q);
139 #elif GMX == 45
140  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), type, a->resind, a->m, a->q);
141 #elif GMX == 40
142  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), type, a->resnr, a->m, a->q);
143 #else
144 #error Unsupported GMX version
145 #endif
146 
147  stringstream nm;
148  nm << bead->getResnr() + 1 << ":" << top.getResidue(res_offset + bead->getResnr())->getName() << ":" << bead->getName();
149  mi->AddBead(bead, nm.str());
150  }
151  ifirstatom+=mtop.molblock[iblock].natoms_mol;
152  }
153  }
154 
155  return true;
156 }
