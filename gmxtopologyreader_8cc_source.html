42 bool GMXTopologyReader::ReadTopology(std::string file, Topology &top) {
43  gmx_mtop_t mtop;
44 
45  int natoms;
46  // cleanup topology to store new data
47  top.Cleanup();
48 
49  t_inputrec ir;
50  ::matrix gbox;
51 
52  (void)read_tpx((char *)file.c_str(), &ir, gbox, &natoms, nullptr, nullptr,
53  &mtop);
54 
55  size_t ifirstatom = 0;
56 
57 #if GMX_VERSION >= 20190000
58  size_t nmolblock = mtop.molblock.size();
59 #else
60  size_t nmolblock = mtop.nmolblock;
61 #endif
62 
63  for (size_t iblock = 0; iblock < nmolblock; ++iblock) {
64  gmx_moltype_t *mol = &(mtop.moltype[mtop.molblock[iblock].type]);
65 
66  std::string molname = *(mol->name);
67 
68  Index res_offset = top.ResidueCount();
69 
70  t_atoms *atoms = &(mol->atoms);
71 
72  for (Index i = 0; i < atoms->nres; i++) {
73  top.CreateResidue(*(atoms->resinfo[i].name));
74  }
75 
76  for (Index imol = 0; imol < mtop.molblock[iblock].nmol; ++imol) {
77  Molecule *mi = top.CreateMolecule(molname);
78 
79 #if GMX_VERSION >= 20190000
80  size_t natoms_mol = mtop.moltype[mtop.molblock[iblock].type].atoms.nr;
81 #else
82  size_t natoms_mol = mtop.molblock[iblock].natoms_mol;
83 #endif
84  // read the atoms
85  for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
86  t_atom *a = &(atoms->atom[iatom]);
87 
88  std::string bead_type = *(atoms->atomtype[iatom]);
89  if (!top.BeadTypeExist(bead_type)) {
90  top.RegisterBeadType(bead_type);
91  }
92  Bead *bead =
93  top.CreateBead(Bead::spherical, *(atoms->atomname[iatom]),
94  bead_type, a->resind + res_offset, a->m, a->q);
95 
96  std::stringstream nm;
97  nm << bead->getResnr() + 1 - res_offset << ":"
98  << top.getResidue(bead->getResnr()).getName() << ":"
99  << bead->getName();
100  mi->AddBead(bead, nm.str());
101  }
102 
103  // add exclusions
104  for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
105  std::list<Bead *> excl_list;
106 #if GMX_VERSION >= 20210000
107  gmx::ListOfLists<int> &excl = mol->excls;
108  for (const Index k : excl[iatom]) {
109  excl_list.push_back(top.getBead(k + ifirstatom));
110  }
111 #else
112  t_blocka *excl = &(mol->excls);
113  for (Index k = excl->index[iatom]; k < excl->index[iatom + 1]; k++) {
114  excl_list.push_back(top.getBead(excl->a[k] + ifirstatom));
115  }
116 #endif
117  top.InsertExclusion(top.getBead(iatom + ifirstatom), excl_list);
118  }
119  ifirstatom += natoms_mol;
120  }
121  }
122 
123  Eigen::Matrix3d m;
124  for (Index i = 0; i < 3; i++) {
125  for (Index j = 0; j < 3; j++) {
126  m(i, j) = gbox[j][i];
127  }
128  }
129  top.setBox(m);
130 
131  return true;
132 }
