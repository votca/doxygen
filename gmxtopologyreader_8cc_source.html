35 bool GMXTopologyReader::ReadTopology(string file, Topology &top)
36 {
37  gmx_mtop_t mtop;
38 
39  int natoms;
40  // cleanup topology to store new data
41  top.Cleanup();
42 
43  t_inputrec ir;
44  ::matrix gbox;
45 
46  (void)read_tpx((char *)file.c_str(),&ir,gbox,&natoms,NULL,NULL,&mtop);
47 
48  size_t ifirstatom = 0;
49 
50 #if GROMACS_VERSION >= 20190000
51  size_t nmolblock=mtop.molblock.size();
52 #else
53  size_t nmolblock=mtop.nmolblock;
54 #endif
55 
56  for(size_t iblock=0; iblock<nmolblock; ++iblock) {
57  gmx_moltype_t *mol
58  = &(mtop.moltype[mtop.molblock[iblock].type]);
59 
60  string molname = *(mol->name);
61 
62  int res_offset = top.ResidueCount();
63 
64  t_atoms *atoms=&(mol->atoms);
65 
66  for(int i=0; i < atoms->nres; i++) {
67  top.CreateResidue(*(atoms->resinfo[i].name));
68  }
69 
70  for(int imol=0; imol<mtop.molblock[iblock].nmol; ++imol) {
71  Molecule *mi = top.CreateMolecule(molname);
72 
73 #if GROMACS_VERSION >= 20190000
74  size_t natoms_mol = mtop.moltype[mtop.molblock[iblock].type].atoms.nr;
75 #else
76  size_t natoms_mol = mtop.molblock[iblock].natoms_mol;
77 #endif
78  // read the atoms
79  for(size_t iatom=0; iatom<natoms_mol; iatom++) {
80  t_atom *a = &(atoms->atom[iatom]);
81 
82  BeadType *type = top.GetOrCreateBeadType(*(atoms->atomtype[iatom]));
83  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), type, a->resind + res_offset, a->m, a->q);
84 
85  stringstream nm;
86  nm << bead->getResnr() + 1 - res_offset << ":" << top.getResidue(bead->getResnr())->getName() << ":" << bead->getName();
87  mi->AddBead(bead, nm.str());
88  }
89 
90  // add exclusions
91  for(size_t iatom=0; iatom<natoms_mol; iatom++) {
92  // read exclusions
93  t_blocka * excl = &(mol->excls);
94  // insert exclusions
95  list<Bead *> excl_list;
96  for(int k=excl->index[iatom]; k<excl->index[iatom+1]; k++) {
97  excl_list.push_back(top.getBead(excl->a[k]+ifirstatom));
98  }
99  top.InsertExclusion(top.getBead(iatom+ifirstatom), excl_list);
100  }
101  ifirstatom+=natoms_mol;
102  }
103  }
104 
105  matrix m;
106  for(int i=0; i<3; i++)
107  for(int j=0; j<3; j++)
108  m[i][j] = gbox[j][i];
109  top.setBox(m);
110 
111  return true;
112 }
