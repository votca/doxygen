42bool GMXTopologyReader::ReadTopology(std::string file, Topology &top) {
43 gmx_mtop_t mtop;
44
45 int natoms;
46 // cleanup topology to store new data
47 top.Cleanup();
48
49 t_inputrec ir;
50 ::matrix gbox;
51
52 (void)read_tpx((char *)file.c_str(), &ir, gbox, &natoms, nullptr, nullptr,
53 &mtop);
54
55 size_t ifirstatom = 0;
56
57 size_t nmolblock = mtop.molblock.size();
58
59 for (size_t iblock = 0; iblock < nmolblock; ++iblock) {
60 gmx_moltype_t *mol = &(mtop.moltype[mtop.molblock[iblock].type]);
61
62 std::string molname = *(mol->name);
63
64 Index res_offset = top.ResidueCount();
65
66 t_atoms *atoms = &(mol->atoms);
67
68 for (Index i = 0; i < atoms->nres; i++) {
69 top.CreateResidue(*(atoms->resinfo[i].name));
70 }
71
72 for (Index imol = 0; imol < mtop.molblock[iblock].nmol; ++imol) {
73 Molecule *mi = top.CreateMolecule(molname);
74
75 size_t natoms_mol = mtop.moltype[mtop.molblock[iblock].type].atoms.nr;
76 // read the atoms
77 for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
78 t_atom *a = &(atoms->atom[iatom]);
79
80 std::string bead_type = *(atoms->atomtype[iatom]);
81 if (!top.BeadTypeExist(bead_type)) {
82 top.RegisterBeadType(bead_type);
83 }
84 Bead *bead =
85 top.CreateBead(Bead::spherical, *(atoms->atomname[iatom]),
86 bead_type, a->resind + res_offset, a->m, a->q);
87
88 std::stringstream nm;
89 nm << bead->getResnr() + 1 - res_offset << ":"
90 << top.getResidue(bead->getResnr()).getName() << ":"
91 << bead->getName();
92 mi->AddBead(bead, nm.str());
93 }
94
95 // add exclusions
96 for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
97 std::list<Bead *> excl_list;
98#if GMX_VERSION >= 20210000
99 gmx::ListOfLists<int> &excl = mol->excls;
100 for (const Index k : excl[iatom]) {
101 excl_list.push_back(top.getBead(k + ifirstatom));
102 }
103#else
104 t_blocka *excl = &(mol->excls);
105 for (Index k = excl->index[iatom]; k < excl->index[iatom + 1]; k++) {
106 excl_list.push_back(top.getBead(excl->a[k] + ifirstatom));
107 }
108#endif
109 top.InsertExclusion(top.getBead(iatom + ifirstatom), excl_list);
110 }
111 ifirstatom += natoms_mol;
112 }
113 }
114
115 Eigen::Matrix3d m;
116 for (Index i = 0; i < 3; i++) {
117 for (Index j = 0; j < 3; j++) {
118 m(i, j) = gbox[j][i];
119 }
120 }
121 top.setBox(m);
122
123 return true;
124}
