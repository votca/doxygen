43 bool GMXTopologyReader::ReadTopology(string file, Topology &top) {
44  gmx_mtop_t mtop;
45 
46  int natoms;
47  // cleanup topology to store new data
48  top.Cleanup();
49 
50  t_inputrec ir;
51  ::matrix gbox;
52 
53  (void)read_tpx((char *)file.c_str(), &ir, gbox, &natoms, NULL, NULL, &mtop);
54 
55  size_t ifirstatom = 0;
56 
57 #if GROMACS_VERSION >= 20190000
58  size_t nmolblock = mtop.molblock.size();
59 #else
60  size_t nmolblock = mtop.nmolblock;
61 #endif
62 
63  for (size_t iblock = 0; iblock < nmolblock; ++iblock) {
64  gmx_moltype_t *mol = &(mtop.moltype[mtop.molblock[iblock].type]);
65 
66  string molname = *(mol->name);
67 
68  int res_offset = top.ResidueCount();
69 
70  t_atoms *atoms = &(mol->atoms);
71 
72  for (int i = 0; i < atoms->nres; i++) {
73  top.CreateResidue(*(atoms->resinfo[i].name));
74  }
75 
76  for (int imol = 0; imol < mtop.molblock[iblock].nmol; ++imol) {
77  Molecule *mi = top.CreateMolecule(molname);
78 
79 #if GROMACS_VERSION >= 20190000
80  size_t natoms_mol = mtop.moltype[mtop.molblock[iblock].type].atoms.nr;
81 #else
82  size_t natoms_mol = mtop.molblock[iblock].natoms_mol;
83 #endif
84  // read the atoms
85  for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
86  t_atom *a = &(atoms->atom[iatom]);
87 
88  string bead_type = *(atoms->atomtype[iatom]);
89  if (!top.BeadTypeExist(bead_type)) {
90  top.RegisterBeadType(bead_type);
91  }
92  Bead *bead = top.CreateBead(1, *(atoms->atomname[iatom]), bead_type,
93  a->resind + res_offset, a->m, a->q);
94 
95  stringstream nm;
96  nm << bead->getResnr() + 1 - res_offset << ":"
97  << top.getResidue(bead->getResnr())->getName() << ":"
98  << bead->getName();
99  mi->AddBead(bead, nm.str());
100  }
101 
102  // add exclusions
103  for (size_t iatom = 0; iatom < natoms_mol; iatom++) {
104  // read exclusions
105  t_blocka *excl = &(mol->excls);
106  // insert exclusions
107  list<Bead *> excl_list;
108  for (int k = excl->index[iatom]; k < excl->index[iatom + 1]; k++) {
109  excl_list.push_back(top.getBead(excl->a[k] + ifirstatom));
110  }
111  top.InsertExclusion(top.getBead(iatom + ifirstatom), excl_list);
112  }
113  ifirstatom += natoms_mol;
114  }
115  }
116 
117  Eigen::Matrix3d m;
118  for (int i = 0; i < 3; i++) {
119  for (int j = 0; j < 3; j++) {
120  m(i, j) = gbox[j][i];
121  }
122  }
123  top.setBox(m);
124 
125  return true;
126 }
