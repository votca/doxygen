69 double PolarRegion::StaticInteraction() {
70 
71  eeInteractor eeinteractor;
72  double e = 0.0;
73 #pragma omp parallel for reduction(+ : e)
74  for (Index i = 0; i < size(); ++i) {
75  for (Index j = 0; j < size(); ++j) {
76  if (i == j) {
77  continue;
78  }
79  PolarSegment& seg1 = _segments[i];
80  const PolarSegment& seg2 = _segments[j];
81  e += eeinteractor.ApplyStaticField<PolarSegment, Estatic::noE_V>(seg2,
82  seg1);
83  }
84  }
85 
86  return 0.5 * e;
87 }

89 eeInteractor::E_terms PolarRegion::PolarEnergy() const {
90 #pragma omp declare reduction(CustomPlus \
91  : eeInteractor::E_terms \
92  : omp_out += omp_in)
93 
94  eeInteractor eeinteractor(_exp_damp);
95 
96  eeInteractor::E_terms terms;
97 
98 #pragma omp parallel for reduction(CustomPlus : terms)
99  for (Index i = 0; i < size(); ++i) {
100  for (Index j = 0; j < i; ++j) {
101  terms += eeinteractor.CalcPolarEnergy(_segments[i], _segments[j]);
102  }
103  }
104 
105 #pragma omp parallel for reduction(CustomPlus : terms)
106  for (Index i = 0; i < size(); ++i) {
107  terms.E_indu_indu() +=
108  eeinteractor.CalcPolarEnergy_IntraSegment(_segments[i]);
109  }
110 
111 #pragma omp parallel for reduction(CustomPlus : terms)
112  for (Index i = 0; i < size(); ++i) {
113  for (const PolarSite& site : _segments[i]) {
114  terms.E_internal() += site.InternalEnergy();
115  }
116  }
117  return terms;
118 }

192 Eigen::VectorXd PolarRegion::CalcInducedDipolesViaPCG(
193  const Eigen::VectorXd& initial_guess) {
194  Eigen::VectorXd b = Eigen::VectorXd::Zero(initial_guess.size());
195  Index index = 0;
196  for (PolarSegment& seg : _segments) {
197  for (const PolarSite& site : seg) {
198  auto V = site.V() + site.V_noE();
199  b.segment<3>(index) = -V;
200  index += 3;
201  }
202  }
203  eeInteractor interactor(_exp_damp);
204  DipoleDipoleInteraction A(interactor, _segments);
205  Eigen::ConjugateGradient<DipoleDipoleInteraction, Eigen::Lower | Eigen::Upper,
206  Eigen::DiagonalPreconditioner<double>>
207  cg;
208  cg.setMaxIterations(_max_iter);
209  cg.setTolerance(_deltaD);
210  cg.compute(A);
211  Eigen::VectorXd x = cg.solveWithGuess(b, initial_guess);
212 
213  XTP_LOG(Log::error, _log)
214  << TimeStamp() << " CG: #iterations: " << cg.iterations()
215  << ", estimated error: " << cg.error() << std::flush;
216 
217  if (cg.info() == Eigen::ComputationInfo::NoConvergence) {
218  _info = false;
219  _errormsg = "PCG iterations did not converge";
220  }
221 
222  if (cg.info() == Eigen::ComputationInfo::NumericalIssue) {
223  _info = false;
224  _errormsg = "PCG had a numerical issue";
225  }
226  return x;
227 }

229 void PolarRegion::Evaluate(std::vector<std::unique_ptr<Region>>& regions) {
230 
231  std::vector<double> energies = ApplyInfluenceOfOtherRegions(regions);
232  Energy_terms e_contrib;
233  e_contrib.E_static_ext() =
234  std::accumulate(energies.begin(), energies.end(), 0.0);
235  XTP_LOG(Log::info, _log) << TimeStamp()
236  << " Calculated static-static and polar-static "
237  "interaction with other regions"
238  << std::flush;
239  e_contrib.E_static_static() = StaticInteraction();
240  XTP_LOG(Log::info, _log) << TimeStamp()
241  << " Calculated static interaction in region "
242  << std::flush;
243  Index dof_polarization = CalcPolDoF();
244  XTP_LOG(Log::error, _log)
245  << TimeStamp() << " Starting Solving for classical polarization with "
246  << dof_polarization << " degrees of freedom." << std::flush;
247 
248  Eigen::VectorXd initial_induced_dipoles;
249  if (!_E_hist.filled() || _segments.size() == 1) {
250  initial_induced_dipoles = CalcInducedDipoleInsideSegments();
251  } else {
252  initial_induced_dipoles = ReadInducedDipolesFromLastIteration();
253  }
254 
255  Eigen::VectorXd x; // if only one segment
256  // it is solved exactly through the initial guess
257  if (_segments.size() != 1) {
258  x = CalcInducedDipolesViaPCG(initial_induced_dipoles);
259  if (!_info) {
260  return;
261  }
262  } else {
263  x = initial_induced_dipoles;
264  }
265 
266  WriteInducedDipolesToSegments(x);
267 
268  e_contrib.addInternalPolarContrib(PolarEnergy());
269  XTP_LOG(Log::info, _log) << TimeStamp()
270  << " Calculated polar interaction in region"
271  << std::flush;
272  e_contrib.E_polar_ext() = PolarEnergy_extern();
273  XTP_LOG(Log::info, _log) << TimeStamp()
274  << " Calculated polar interaction with other regions"
275  << std::flush;
276 
277  XTP_LOG(Log::info, _log) << std::setprecision(10)
278  << " Internal static energy [hrt]= "
279  << e_contrib.E_static_static() << std::flush;
280  XTP_LOG(Log::info, _log) << std::setprecision(10)
281  << " External static energy [hrt]= "
282  << e_contrib.E_static_ext() << std::flush;
283  XTP_LOG(Log::error, _log)
284  << std::setprecision(10)
285  << " Total static energy [hrt]= " << e_contrib.Estatic() << std::flush;
286 
287  XTP_LOG(Log::info, _log) << std::setprecision(10)
288  << " internal dQ-dQ energy [hrt]= "
289  << e_contrib.E_indu_indu() << std::flush;
290 
291  XTP_LOG(Log::info, _log) << std::setprecision(10)
292  << " internal Q-dQ energy [hrt]= "
293  << e_contrib.E_indu_stat() << std::flush;
294 
295  XTP_LOG(Log::info, _log) << std::setprecision(10)
296  << " Internal energy [hrt]= "
297  << e_contrib.E_internal() << std::flush;
298 
299  XTP_LOG(Log::info, _log) << std::setprecision(10)
300  << " External polar energy [hrt]= "
301  << e_contrib.E_polar_ext() << std::flush;
302 
303  XTP_LOG(Log::error, _log)
304  << std::setprecision(10)
305  << " Total polar energy [hrt]= " << e_contrib.Epolar() << std::flush;
306 
307  XTP_LOG(Log::error, _log)
308  << std::setprecision(10) << " Total energy [hrt]= " << e_contrib.Etotal()
309  << std::flush;
310  _E_hist.push_back(e_contrib);
311  return;
312 }

319 double PolarRegion::InteractwithPolarRegion(const PolarRegion& region) {
320  bool noE_V = true;
321  if (this->getId() < region.getId()) {
322  noE_V = false;
323  }
324 
325  double e = 0;
326 #pragma omp parallel for reduction(+ : e)
327  for (Index i = 0; i < Index(_segments.size()); i++) {
328  PolarSegment& pseg1 = _segments[i];
329  double e_thread = 0.0;
330  eeInteractor ee;
331  for (const PolarSegment& pseg2 : region) {
332  if (noE_V) {
333  ee.ApplyStaticField<PolarSegment, Estatic::noE_V>(pseg2, pseg1);
334  ee.ApplyInducedField<Estatic::noE_V>(pseg2, pseg1);
335  } else {
336  e_thread += ee.ApplyStaticField<PolarSegment, Estatic::V>(pseg2, pseg1);
337  e_thread += ee.ApplyInducedField<Estatic::V>(pseg2, pseg1);
338  }
339  e += e_thread;
340  }
341  }
342  return e;
343 }
