48 bool PolarRegion::Converged() const {
49 
50  if (!_E_hist.filled()) {
51  return false;
52  }
53  double Echange = _E_hist.getDiff().Etotal();
54  std::string info = "not converged";
55  bool converged = false;
56  if (std::abs(Echange) < _deltaE) {
57  info = "converged";
58  converged = true;
59  }
60  XTP_LOG_SAVE(logINFO, _log)
61  << TimeStamp() << " Region:" << this->identify() << " " << this->getId()
62  << " is " << info << " deltaE=" << Echange << std::flush;
63  return converged;

66 double PolarRegion::StaticInteraction() {
67 
68  eeInteractor eeinteractor;
69  double e = 0.0;
70 #pragma omp parallel for reduction(+ : e)
71  for (int i = 0; i < size(); ++i) {
72  for (int j = 0; j < size(); ++j) {
73  if (i == j) {
74  continue;
75  }
76  PolarSegment& seg1 = _segments[i];
77  const PolarSegment& seg2 = _segments[j];
78  e += eeinteractor.ApplyStaticField<PolarSegment, Estatic::noE_V>(seg2,
79  seg1);
80  }
81  }
82 
83  return e / 2.0;
84 }

86 eeInteractor::E_terms PolarRegion::PolarEnergy() const {
87 #pragma omp declare reduction(CustomPlus \
88  : eeInteractor::E_terms \
89  : omp_out += omp_in)
90 
91  eeInteractor eeinteractor(_exp_damp);
92 
93  eeInteractor::E_terms terms;
94 
95 #pragma omp parallel for reduction(CustomPlus : terms)
96  for (int i = 0; i < size(); ++i) {
97  for (int j = 0; j < i; ++j) {
98  terms += eeinteractor.CalcPolarEnergy(_segments[i], _segments[j]);
99  }
100  }
101 
102 #pragma omp parallel for reduction(CustomPlus : terms)
103  for (int i = 0; i < size(); ++i) {
104  terms.E_indu_indu() +=
105  eeinteractor.CalcPolarEnergy_IntraSegment(_segments[i]);
106  }
107 
108 #pragma omp parallel for reduction(CustomPlus : terms)
109  for (int i = 0; i < size(); ++i) {
110  for (const PolarSite& site : _segments[i]) {
111  terms.E_internal() += site.InternalEnergy();
112  }
113  }
114  return terms;
115 }

143 void PolarRegion::Evaluate(std::vector<std::unique_ptr<Region> >& regions) {
144 
145  std::vector<double> energies = ApplyInfluenceOfOtherRegions(regions);
146  Energy_terms e_contrib;
147  e_contrib.E_static_ext() =
148  std::accumulate(energies.begin(), energies.end(), 0.0);
149  XTP_LOG_SAVE(logINFO, _log) << TimeStamp()
150  << " Calculated static-static and polar-static "
151  "interaction with other regions"
152  << std::flush;
153  e_contrib.E_static_static() = StaticInteraction();
154  XTP_LOG_SAVE(logINFO, _log)
155  << TimeStamp() << " Calculated static interaction in region "
156  << std::flush;
157  int dof_polarisation = 0;
158  for (const PolarSegment& seg : _segments) {
159  dof_polarisation += seg.size() * 3;
160  }
161  XTP_LOG_SAVE(logINFO, _log)
162  << TimeStamp() << " Starting Solving for classical polarisation with "
163  << dof_polarisation << " degrees of freedom." << std::flush;
164 
165  Eigen::VectorXd initial_induced_dipoles =
166  Eigen::VectorXd::Zero(dof_polarisation);
167 
168  if (!_E_hist.filled()) {
169  eeInteractor interactor(_exp_damp);
170  int index = 0;
171  for (PolarSegment& seg : _segments) {
172  initial_induced_dipoles.segment(index, 3 * seg.size()) =
173  interactor.Cholesky_IntraSegment(seg);
174  index += 3 * seg.size();
175  }
176  } else {
177  int index = 0;
178  for (PolarSegment& seg : _segments) {
179  for (const PolarSite& site : seg) {
180  initial_induced_dipoles.segment<3>(index) = site.Induced_Dipole();
181  index += 3;
182  }
183  }
184  }
185  Eigen::VectorXd b = Eigen::VectorXd::Zero(dof_polarisation);
186  int index = 0;
187  for (PolarSegment& seg : _segments) {
188  for (const PolarSite& site : seg) {
189  const Eigen::Vector3d V = site.V() + site.V_noE();
190  b.segment<3>(index) = -V;
191  index += 3;
192  }
193  }
194 
195  eeInteractor interactor(_exp_damp);
196  DipoleDipoleInteraction A(interactor, _segments);
197  Eigen::ConjugateGradient<DipoleDipoleInteraction, Eigen::Lower | Eigen::Upper>
198  cg;
199  cg.setMaxIterations(_max_iter);
200  cg.setTolerance(_deltaD);
201  cg.compute(A);
202  Eigen::VectorXd x = cg.solveWithGuess(b, initial_induced_dipoles);
203 
204  XTP_LOG_SAVE(logINFO, _log)
205  << TimeStamp() << " CG: #iterations: " << cg.iterations()
206  << ", estimated error: " << cg.error() << std::flush;
207 
208  if (cg.info() == Eigen::ComputationInfo::NoConvergence) {
209  _info = false;
210  _errormsg = "PCG iterations did not converge";
211  return;
212  }
213 
214  index = 0;
215  for (PolarSegment& seg : _segments) {
216  for (PolarSite& site : seg) {
217  site.setInduced_Dipole(x.segment<3>(index));
218  index += 3;
219  }
220  }
221 
222  e_contrib.addInternalPolarContrib(PolarEnergy());
223  XTP_LOG_SAVE(logINFO, _log)
224  << TimeStamp() << " Calculated polar interaction in region" << std::flush;
225  e_contrib.E_polar_ext() = PolarEnergy_extern();
226  XTP_LOG_SAVE(logINFO, _log)
227  << TimeStamp() << " Calculated polar interaction with other regions"
228  << std::flush;
229 
230  XTP_LOG_SAVE(logINFO, _log)
231  << std::setprecision(10)
232  << " Internal static energy [hrt]= " << e_contrib.E_static_static()
233  << std::flush;
234  XTP_LOG_SAVE(logINFO, _log)
235  << std::setprecision(10)
236  << " External static energy [hrt]= " << e_contrib.E_static_ext()
237  << std::flush;
238  XTP_LOG_SAVE(logINFO, _log)
239  << std::setprecision(10)
240  << " Total static energy [hrt]= " << e_contrib.Estatic() << std::flush;
241 
242  XTP_LOG_SAVE(logINFO, _log)
243  << std::setprecision(10)
244  << " internal dQ-dQ energy [hrt]= " << e_contrib.E_indu_indu()
245  << std::flush;
246 
247  XTP_LOG_SAVE(logINFO, _log)
248  << std::setprecision(10)
249  << " internal Q-dQ energy [hrt]= " << e_contrib.E_indu_stat()
250  << std::flush;
251 
252  XTP_LOG_SAVE(logINFO, _log)
253  << std::setprecision(10)
254  << " Internal energy [hrt]= " << e_contrib.E_internal() << std::flush;
255 
256  XTP_LOG_SAVE(logINFO, _log)
257  << std::setprecision(10)
258  << " External polar energy [hrt]= " << e_contrib.E_polar_ext()
259  << std::flush;
260 
261  XTP_LOG_SAVE(logINFO, _log)
262  << std::setprecision(10)
263  << " Total polar energy [hrt]= " << e_contrib.Epolar() << std::flush;
264 
265  XTP_LOG_SAVE(logINFO, _log)
266  << std::setprecision(10) << " Total energy [hrt]= " << e_contrib.Etotal()
267  << std::flush;
268  _E_hist.push_back(e_contrib);
269  return;
270 }

277 double PolarRegion::InteractwithPolarRegion(const PolarRegion& region) {
278  bool noE_V = true;
279  if (this->getId() < region.getId()) {
280  noE_V = false;
281  }
282 
283  double e = 0;
284 #pragma omp parallel for reduction(+ : e)
285  for (unsigned i = 0; i < _segments.size(); i++) {
286  PolarSegment& pseg1 = _segments[i];
287  double e_thread = 0.0;
288  eeInteractor ee;
289  for (const PolarSegment& pseg2 : region) {
290  if (noE_V) {
291  ee.ApplyStaticField<PolarSegment, Estatic::noE_V>(pseg2, pseg1);
292  ee.ApplyInducedField<Estatic::noE_V>(pseg2, pseg1);
293  } else {
294  e_thread += ee.ApplyStaticField<PolarSegment, Estatic::V>(pseg2, pseg1);
295  e_thread += ee.ApplyInducedField<Estatic::V>(pseg2, pseg1);
296  }
297  e += e_thread;
298  }
299  }
300  return e;
301 }
