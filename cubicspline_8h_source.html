194 template<typename matrix_type>
195 inline void CubicSpline::AddBCToFitMatrix(matrix_type &M,
196  int offset1, int offset2)
197 {
198  for(size_t i=0; i<_r.size() - 2; ++i) {
199  M(offset1+i+1, offset2 + i) = A_prime_l(i);
200  M(offset1+i+1, offset2 + i+1) = B_prime_l(i) - A_prime_r(i);
201  M(offset1+i+1, offset2 + i+2) = -B_prime_r(i);
202 
203  M(offset1+i+1, offset2 + _r.size() + i) = C_prime_l(i);
204  M(offset1+i+1, offset2 + _r.size() + i+1) = D_prime_l(i) - C_prime_r(i);
205  M(offset1+i+1, offset2 + _r.size() + i+2) = -D_prime_r(i);
206  }
207  // currently only natural boundary conditions:
208  switch(_boundaries) {
209  case splineNormal:
210  M(offset1, offset2 + _r.size()) = 1;
211  M(offset1 + _r.size() - 1, offset2 + 2*_r.size()-1) = 1;
212  break;
213  case splineDerivativeZero:
214  // y
215  M(offset1+0, offset2 + 0) = -1*A_prime_l(0);
216  M(offset1+0, offset2 + 1) = -1*B_prime_l(0);
217 
218  M(offset1+ _r.size()-1, offset2 + _r.size()-2) = A_prime_l(_r.size()-2);
219  M(offset1+ _r.size()-1, offset2 + _r.size()-1) = B_prime_l(_r.size()-2);
220 
221  // y''
222  M(offset1+0, offset2 + _r.size() + 0) = D_prime_l(0);
223  M(offset1+0, offset2 + _r.size() + 1) = C_prime_l(0);
224 
225  M(offset1+ _r.size()-1, offset2 + 2*_r.size()-2) = C_prime_l(_r.size()-2);
226  M(offset1+ _r.size()-1, offset2 + 2*_r.size()-1) = D_prime_l(_r.size()-2);
227  break;
228 
229  case splinePeriodic:
230  M(offset1, offset2) = 1;
231  M(offset1, offset2 + _r.size()-1) = -1;
232  M(offset1 + _r.size() - 1, offset2 + _r.size()) = 1;
233  M(offset1 + _r.size() - 1, offset2 + 2*_r.size()-1) = -1;
234  break;
235  }
