172 inline double CubicSpline::Calculate(const double &r) {
173  int interval = getInterval(r);
174  return A(r) * _f[interval] + B(r) * _f[interval + 1] + C(r) * _f2[interval] +
175  D(r) * _f2[interval + 1];
176 }

184 template <typename matrix_type>
185 inline void CubicSpline::AddToFitMatrix(matrix_type &M, double x, int offset1,
186  int offset2, double scale) {
187  int spi = getInterval(x);
188  M(offset1, offset2 + spi) += A(x) * scale;
189  M(offset1, offset2 + spi + 1) += B(x) * scale;
190  M(offset1, offset2 + spi + _r.size()) += C(x) * scale;
191  M(offset1, offset2 + spi + _r.size() + 1) += D(x) * scale;
192 }

232 template <typename matrix_type>
233 inline void CubicSpline::AddBCToFitMatrix(matrix_type &M, int offset1,
234  int offset2) {
235  for (int i = 0; i < _r.size() - 2; ++i) {
236  M(offset1 + i + 1, offset2 + i) = A_prime_l(i);
237  M(offset1 + i + 1, offset2 + i + 1) = B_prime_l(i) - A_prime_r(i);
238  M(offset1 + i + 1, offset2 + i + 2) = -B_prime_r(i);
239 
240  M(offset1 + i + 1, offset2 + _r.size() + i) = C_prime_l(i);
241  M(offset1 + i + 1, offset2 + _r.size() + i + 1) =
242  D_prime_l(i) - C_prime_r(i);
243  M(offset1 + i + 1, offset2 + _r.size() + i + 2) = -D_prime_r(i);
244  }
245  // currently only natural boundary conditions:
246  switch (_boundaries) {
247  case splineNormal:
248  M(offset1, offset2 + _r.size()) = 1;
249  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = 1;
250  break;
251  case splineDerivativeZero:
252  // y
253  M(offset1 + 0, offset2 + 0) = -1 * A_prime_l(0);
254  M(offset1 + 0, offset2 + 1) = -1 * B_prime_l(0);
255 
256  M(offset1 + _r.size() - 1, offset2 + _r.size() - 2) =
257  A_prime_l(_r.size() - 2);
258  M(offset1 + _r.size() - 1, offset2 + _r.size() - 1) =
259  B_prime_l(_r.size() - 2);
260 
261  // y''
262  M(offset1 + 0, offset2 + _r.size() + 0) = D_prime_l(0);
263  M(offset1 + 0, offset2 + _r.size() + 1) = C_prime_l(0);
264 
265  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 2) =
266  C_prime_l(_r.size() - 2);
267  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) =
268  D_prime_l(_r.size() - 2);
269  break;
270 
271  case splinePeriodic:
272  M(offset1, offset2) = 1;
273  M(offset1, offset2 + _r.size() - 1) = -1;
274  M(offset1 + _r.size() - 1, offset2 + _r.size()) = 1;
275  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = -1;
276  break;
277  }

298 inline double CubicSpline::C(const double &r) {
299  double xxi, h;
300  xxi = r - _r[getInterval(r)];
301  h = _r[getInterval(r) + 1] - _r[getInterval(r)];
302 
303  return (0.5 * xxi * xxi - (1.0 / 6.0) * xxi * xxi * xxi / h -
304  (1.0 / 3.0) * xxi * h);
305 }

315 inline double CubicSpline::D(const double &r) {
316  double xxi, h;
317  xxi = r - _r[getInterval(r)];
318  h = _r[getInterval(r) + 1] - _r[getInterval(r)];
