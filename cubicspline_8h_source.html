170 template <typename matrix_type>
171 inline void CubicSpline::AddToFitMatrix(matrix_type &M, double x, Index offset1,
172  Index offset2, double scale) {
173  Index spi = getInterval(x);
174  M(offset1, offset2 + spi) += A(x) * scale;
175  M(offset1, offset2 + spi + 1) += B(x) * scale;
176  M(offset1, offset2 + spi + _r.size()) += C(x) * scale;
177  M(offset1, offset2 + spi + _r.size() + 1) += D(x) * scale;
178 }

215 template <typename matrix_type>
216 inline void CubicSpline::AddBCToFitMatrix(matrix_type &M, Index offset1,
217  Index offset2) {
218  for (Index i = 0; i < _r.size() - 2; ++i) {
219  M(offset1 + i + 1, offset2 + i) = A_prime_l(i);
220  M(offset1 + i + 1, offset2 + i + 1) = B_prime_l(i) - A_prime_r(i);
221  M(offset1 + i + 1, offset2 + i + 2) = -B_prime_r(i);
222 
223  M(offset1 + i + 1, offset2 + _r.size() + i) = C_prime_l(i);
224  M(offset1 + i + 1, offset2 + _r.size() + i + 1) =
225  D_prime_l(i) - C_prime_r(i);
226  M(offset1 + i + 1, offset2 + _r.size() + i + 2) = -D_prime_r(i);
227  }
228  switch (_boundaries) {
229  case splineNormal:
230  M(offset1, offset2 + _r.size()) = 1;
231  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = 1;
232  break;
233  case splineDerivativeZero:
234  // y
235  M(offset1 + 0, offset2 + 0) = -1 * A_prime_l(0);
236  M(offset1 + 0, offset2 + 1) = -1 * B_prime_l(0);
237 
238  M(offset1 + _r.size() - 1, offset2 + _r.size() - 2) =
239  A_prime_l(_r.size() - 2);
240  M(offset1 + _r.size() - 1, offset2 + _r.size() - 1) =
241  B_prime_l(_r.size() - 2);
242 
243  // y''
244  M(offset1 + 0, offset2 + _r.size() + 0) = D_prime_l(0);
245  M(offset1 + 0, offset2 + _r.size() + 1) = C_prime_l(0);
246 
247  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 2) =
248  C_prime_l(_r.size() - 2);
249  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) =
250  D_prime_l(_r.size() - 2);
251  break;
252  case splinePeriodic:
253  M(offset1, offset2) = 1;
254  M(offset1, offset2 + _r.size() - 1) = -1;
255  M(offset1 + _r.size() - 1, offset2 + _r.size()) = 1;
256  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = -1;
257  break;
258  }
259 }
