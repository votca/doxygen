171 inline double CubicSpline::Calculate(const double &r) {
172  long int interval = getInterval(r);
173  return A(r) * _f[interval] + B(r) * _f[interval + 1] + C(r) * _f2[interval] +
174  D(r) * _f2[interval + 1];
175 }

184 inline void CubicSpline::AddToFitMatrix(matrix_type &M, double x,
185  long int offset1, long int offset2,
186  double scale) {
187  long int spi = getInterval(x);
188  M(offset1, offset2 + spi) += A(x) * scale;
189  M(offset1, offset2 + spi + 1) += B(x) * scale;
190  M(offset1, offset2 + spi + _r.size()) += C(x) * scale;
191  M(offset1, offset2 + spi + _r.size() + 1) += D(x) * scale;
192 }

233 template <typename matrix_type>
234 inline void CubicSpline::AddBCToFitMatrix(matrix_type &M, long int offset1,
235  long int offset2) {
236  for (int i = 0; i < _r.size() - 2; ++i) {
237  M(offset1 + i + 1, offset2 + i) = A_prime_l(i);
238  M(offset1 + i + 1, offset2 + i + 1) = B_prime_l(i) - A_prime_r(i);
239  M(offset1 + i + 1, offset2 + i + 2) = -B_prime_r(i);
240 
241  M(offset1 + i + 1, offset2 + _r.size() + i) = C_prime_l(i);
242  M(offset1 + i + 1, offset2 + _r.size() + i + 1) =
243  D_prime_l(i) - C_prime_r(i);
244  M(offset1 + i + 1, offset2 + _r.size() + i + 2) = -D_prime_r(i);
245  }
246  // currently only natural boundary conditions:
247  switch (_boundaries) {
248  case splineNormal:
249  M(offset1, offset2 + _r.size()) = 1;
250  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = 1;
251  break;
252  case splineDerivativeZero:
253  // y
254  M(offset1 + 0, offset2 + 0) = -1 * A_prime_l(0);
255  M(offset1 + 0, offset2 + 1) = -1 * B_prime_l(0);
256 
257  M(offset1 + _r.size() - 1, offset2 + _r.size() - 2) =
258  A_prime_l(_r.size() - 2);
259  M(offset1 + _r.size() - 1, offset2 + _r.size() - 1) =
260  B_prime_l(_r.size() - 2);
261 
262  // y''
263  M(offset1 + 0, offset2 + _r.size() + 0) = D_prime_l(0);
264  M(offset1 + 0, offset2 + _r.size() + 1) = C_prime_l(0);
265 
266  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 2) =
267  C_prime_l(_r.size() - 2);
268  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) =
269  D_prime_l(_r.size() - 2);
270  break;
271 
272  case splinePeriodic:
273  M(offset1, offset2) = 1;
274  M(offset1, offset2 + _r.size() - 1) = -1;
275  M(offset1 + _r.size() - 1, offset2 + _r.size()) = 1;
276  M(offset1 + _r.size() - 1, offset2 + 2 * _r.size() - 1) = -1;
277  break;
278  }

299 inline double CubicSpline::C(const double &r) {
300  double xxi, h;
301  xxi = r - _r[getInterval(r)];
302  h = _r[getInterval(r) + 1] - _r[getInterval(r)];
303 
304  return (0.5 * xxi * xxi - (1.0 / 6.0) * xxi * xxi * xxi / h -
305  (1.0 / 3.0) * xxi * h);
306 }
