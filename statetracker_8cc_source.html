63 std::vector<Index> StateTracker::ComparePairofVectors(
64  std::vector<Index>& vec1, std::vector<Index>& vec2) const {
65  std::vector<Index> result(std::min(vec1, vec2));
66  std::sort(vec1.begin(), vec1.end());
67  std::sort(vec2.begin(), vec2.end());
68  std::vector<Index>::iterator it = std::set_intersection(
69  vec1.begin(), vec1.end(), vec2.begin(), vec2.end(), result.begin());
70  result.resize(it - result.begin());
71  return result;
72 }

87 QMState StateTracker::CalcState(const Orbitals& orbitals) const {
88 
89  if (_filters.empty()) {
90  return _statehist[0];
91  }
92 
93  std::vector<std::vector<Index> > results;
94  for (const auto& filter : _filters) {
95  if (_statehist.size() < 2 && filter->NeedsInitialState()) {
96  XTP_LOG(Log::error, *_log)
97  << "Filter " << filter->Identify()
98  << " not used in first iteration as it needs a reference state"
99  << flush;
100  continue;
101  }
102  results.push_back(filter->CalcIndeces(orbitals, _statehist[0].Type()));
103  }
104 
105  std::vector<Index> result = CollapseResults(results);
106  QMState state;
107  if (result.size() < 1) {
108  state = _statehist.back();
109  XTP_LOG(Log::error, *_log)
110  << "No State found by tracker using last state: " << state.ToString()
111  << flush;
112  } else {
113  state = QMState(_statehist.back().Type(), result[0], false);
114  XTP_LOG(Log::error, *_log)
115  << "Next State is: " << state.ToString() << flush;
116  }
117  return state;
118 }
