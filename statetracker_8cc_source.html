28 void StateTracker::Initialize(tools::Property& options) {
29  if (options.exists("oscillator_strength")) {
30  _use_osctracker = true;
31  _oscthreshold = options.ifExistsReturnElseThrowRuntimeError<double>(
32  "oscillator_strength");
33  }
34  if (options.exists("overlap")) {
35  _use_overlaptracker = true;
36  _overlapthreshold =
37  options.ifExistsReturnElseReturnDefault<double>("overlap", 0.0);
38  }
39  if (options.exists("localization")) {
40  _use_localizationtracker = true;
41  std::string indices =
42  options.ifExistsReturnElseThrowRuntimeError<std::string>(
43  "localization.fragment");
44  _fragment_loc = QMFragment<double>(0, indices);
45  _fragment_loc.value() = options.ifExistsReturnElseThrowRuntimeError<double>(
46  "localization.threshold");
47  }
48 
49  if (options.exists("charge_transfer")) {
50  _use_dQtracker = true;
51  std::string indices =
52  options.ifExistsReturnElseThrowRuntimeError<std::string>(
53  "charge_transfer.fragment");
54  _fragment_dQ = QMFragment<double>(0, indices);
55  _fragment_dQ.value() = options.ifExistsReturnElseThrowRuntimeError<double>(
56  "charge_transfer.threshold");
57  }
58  if (_use_dQtracker && _use_localizationtracker) {
59  throw std::runtime_error(
60  "Cannot use localization and charge_transfer tracker at the same "
61  "time.");
62  }
63 }

67  << flush;

84  }
85  }
86  if (_use_localizationtracker) {
87  XTP_LOG(logDEBUG, *_log)
88  << "Using localization tracker for " << _fragment_loc << flush;
89  }
90  if (_use_dQtracker) {
91  XTP_LOG(logDEBUG, *_log)
92  << "Using Delta Q tracker for fragment " << _fragment_dQ << flush;
93  }
94  if (_use_osctracker && _use_dQtracker) {
95  XTP_LOG(logDEBUG, *_log) << "WARNING: trackering for optically active CT "
96  "transition - might not make sense... "
97  << flush;
98  }
99  if (_use_dQtracker + _use_osctracker + _use_localizationtracker +
100  _use_osctracker <
101  1) {
102  XTP_LOG(logDEBUG, *_log) << "WARNING: No tracker is used " << flush;
103  }

106 std::vector<Index> StateTracker::ComparePairofVectors(
107  std::vector<Index>& vec1, std::vector<Index>& vec2) const {
108  std::vector<Index> result(std::min(vec1, vec2));
109  std::vector<Index>::iterator it;
110  std::sort(vec1.begin(), vec1.end());
111  std::sort(vec2.begin(), vec2.end());
112  it = std::set_intersection(vec1.begin(), vec1.end(), vec2.begin(), vec2.end(),
113  result.begin());
114  result.resize(it - result.begin());
115  return result;
116 }

118 std::vector<Index> StateTracker::CollapseResults(
119  std::vector<std::vector<Index> >& results) const {
120  if (results.size() == 1) {

122  } else {

131 QMState StateTracker::CalcState(const Orbitals& orbitals) const {
132 
133  if (_use_dQtracker + _use_osctracker + _use_localizationtracker +
134  _use_osctracker <
135  1) {
136  return _statehist[0];
137  }
138 
139  std::vector<std::vector<Index> > results;
140  if (_use_osctracker) {
141  results.push_back(OscTracker(orbitals));
142  }
143  if (_use_localizationtracker) {
144  results.push_back(LocTracker(orbitals));
145  }
146  if (_use_overlaptracker) {
147  results.push_back(OverlapTracker(orbitals));
148  }
149  if (_use_dQtracker) {
150  results.push_back(DeltaQTracker(orbitals));
151  }
152 
153  std::vector<Index> result = CollapseResults(results);
154  QMState state;
155  if (result.size() < 1) {
156  state = _statehist.back();
157  XTP_LOG(logDEBUG, *_log)
158  << "No State found by tracker using last state: " << state.ToString()
159  << flush;
160  } else {
161  state = QMState(_statehist.back().Type(), result[0], false);
162  XTP_LOG(logDEBUG, *_log) << "Next State is: " << state.ToString() << flush;
163  }
164  return state;
165 }

187 std::vector<Index> StateTracker::LocTracker(const Orbitals& orbitals) const {
188  std::vector<Index> indexes;
189  Lowdin low;
190  QMFragment<BSE_Population> frag;
191  frag.copy_withoutvalue(_fragment_loc);
192  std::vector<QMFragment<BSE_Population> > loc = {frag};
193  low.CalcChargeperFragment(loc, orbitals, _statehist[0].Type());
194  const Eigen::VectorXd& popE = loc[0].value().E;
195  const Eigen::VectorXd& popH = loc[0].value().H;
196  for (Index i = 0; i < popE.size(); i++) {
197  if (popE[i] > _fragment_loc.value() && popH[i] > _fragment_loc.value()) {
198  indexes.push_back(i);
199  }
200  }
201  return indexes;
202 }

204 std::vector<Index> StateTracker::DeltaQTracker(const Orbitals& orbitals) const {
205  std::vector<Index> indexes;
206  Lowdin low;
207  QMFragment<BSE_Population> frag;
208  frag.copy_withoutvalue(_fragment_dQ);
209  std::vector<QMFragment<BSE_Population> > loc = {frag};
210  low.CalcChargeperFragment(loc, orbitals, _statehist[0].Type());
211  Eigen::VectorXd dq = (loc[0].value().H - loc[0].value().E).cwiseAbs();
212 
213  for (Index i = 0; i < dq.size(); i++) {
214  if (dq[i] > _fragment_dQ.value()) {
215  indexes.push_back(i);
216  }
217  }
218  return indexes;

221 Eigen::VectorXd StateTracker::CalculateOverlap(const Orbitals& orbitals) const {
222  Eigen::MatrixXd coeffs = CalcOrthoCoeffs(orbitals);
223  Eigen::VectorXd overlap = (coeffs * _laststatecoeff).cwiseAbs2();
224  return overlap;

227 Eigen::MatrixXd StateTracker::CalcOrthoCoeffs(const Orbitals& orbitals) const {
228  QMStateType type = _statehist[0].Type();
229  Eigen::MatrixXd coeffs;
230  if (type.isSingleParticleState()) {
231  if (type == QMStateType::DQPstate) {
232  coeffs = orbitals.CalculateQParticleAORepresentation();
233  } else {
234  coeffs = orbitals.MOs().eigenvectors();
235  }
236  } else {
237  throw std::runtime_error("Overlap for excitons not implemented yet");
238  }
239  return coeffs;

242 void StateTracker::UpdateLastCoeff(const Orbitals& orbitals) {
243  Eigen::MatrixXd ortho_coeffs = CalcOrthoCoeffs(orbitals);
244  Index offset = 0;
245  if (_statehist[0].Type() == QMStateType::DQPstate) {
246  offset = orbitals.getGWAmin();
247  }
248  _laststatecoeff = ortho_coeffs.col(_statehist.back().StateIdx() - offset);
249 }

251 std::vector<Index> StateTracker::OverlapTracker(
252  const Orbitals& orbitals) const {
253  std::vector<Index> indexes;
254  if (_statehist.size() <= 1) {
255  indexes = std::vector<Index>{_statehist[0].StateIdx()};
256  return indexes;
257  }
258 
259  Eigen::VectorXd Overlap = CalculateOverlap(orbitals);
260  Index validelements = Index(Overlap.size());
261  for (Index i = 0; i < Index(Overlap.size()); i++) {
262  if (Overlap(i) < _overlapthreshold) {
263  validelements--;
264  }
265  }
266 
267  std::vector<Index> index = std::vector<Index>(Overlap.size());
268  std::iota(index.begin(), index.end(), 0);
269  std::stable_sort(index.begin(), index.end(), [&Overlap](Index i1, Index i2) {
270  return Overlap[i1] > Overlap[i2];
271  });
272 
273  Index offset = 0;
274  if (_statehist[0].Type().isGWState()) {
275  offset = orbitals.getGWAmin();
276  }
277 
278  for (Index i : index) {
279  if (int(indexes.size()) == validelements) {
280  break;
281  }
282  indexes.push_back(i + offset);
283  }
284  return indexes;
285 }
