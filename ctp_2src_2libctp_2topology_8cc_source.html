152 void Topology::setBox(const matrix &box,
153  CSG::BoundaryCondition::eBoxtype boxtype) {
154 
155  // Determine box type automatically in case boxtype == typeAuto
156  if(boxtype == CSG::BoundaryCondition::typeAuto) {
157  boxtype = AutoDetectBoxType(box);
158  }
159 
160  if(_hasPb) {
161  if (votca::tools::globals::verbose) {
162  cout << "Removing periodic box. Creating new... " << endl;
163  }
164  delete _bc;
165  }
166 
167  switch(boxtype) {
168  case CSG::BoundaryCondition::typeTriclinic:
169  _bc = new CSG::TriclinicBox();
170  break;
171  case CSG::BoundaryCondition::typeOrthorhombic:
172  _bc = new CSG::OrthorhombicBox();
173  break;
174  default:
175  _bc = new CSG::OpenBox();
176  break;
177  }
178 
179  _bc->setBox(box);
180  _hasPb = true;
181 }

184 CSG::BoundaryCondition::eBoxtype Topology::AutoDetectBoxType(const matrix &box){
185 
186  // Set box type to OpenBox in case "box" is the zero matrix,
187  // to OrthorhombicBox in case "box" is a diagonal matrix,
188  // or to TriclinicBox otherwise
189 
190  if(box.get(0,0)==0 && box.get(0,1)==0 && box.get(0,2)==0 &&
191  box.get(1,0)==0 && box.get(1,1)==0 && box.get(1,2)==0 &&
192  box.get(2,0)==0 && box.get(2,1)==0 && box.get(2,2)==0) {
193 
194  cout << "WARNING: No box vectors specified in trajectory."
195  "Using open-box boundary conditions. " << endl;
196  return CSG::BoundaryCondition::typeOpen;
197  }
198 
199  else if(box.get(0,1)==0 && box.get(0,2)==0 &&
200  box.get(1,0)==0 && box.get(1,2)==0 &&
201  box.get(2,0)==0 && box.get(2,1)==0) {
202 
203  return CSG::BoundaryCondition::typeOrthorhombic;
204  }
205 
206  else {
207  return CSG::BoundaryCondition::typeTriclinic;
208  }
209 
210  return CSG::BoundaryCondition::typeOpen;
211 }

220 bool Topology::Rigidify() {
221 
222  if (!_canRigidify) {
223  cout << endl
224  << "... ... ABORT: Request to rigidify system, but no QM "
225  "coordinates provided. ";
226  return 0;
227  }
228  else {
229 
230  // Rigidify segments
231  vector<Segment*> ::iterator sit;
232  for (sit = _segments.begin();
233  sit < _segments.end();
234  sit++) {
235 
236  (*sit)->Rigidify();
237  }
238 
239  cout << endl
240  << "... ... Rigidified " << _segments.size() << " segments. "
241  << flush;
242 
243  if (this->NBList().size() > 0) {
244 
245  // Rigidify pairs
246  // [ Why this is needed: Orientation matrices for fragments are not
247  // not written to the state file, and hence lost whenever a frame
248  // is saved and reloaded. When reading in a new frame from the
249  // database, pairs are created from scratch based on two segment
250  // IDs each. Then the pair constructor is called to check whether
251  // the pair is formed across the periodic boundary. If so, it
252  // creates a ghost from the second partner. This ghost is a new
253  // segment which is just accessible from within the pair, but not
254  // from the topology; i.e. it is not stored in any segment containers.
255  // Since at this point, fragments do not store rotation matrices yet,
256  // the ghost - very much deconnected from its originator - does not,
257  // either. Therefore it should not be forgotten here. --- A way out
258  // would be to rigidify the topology within StateSaver::ReadFrame,
259  // after atoms have been created, but before pairs are created. ]
260 
261  QMNBList &nblist = this->NBList();
262 
263  QMNBList::iterator pit;
264  int count = 0;
265  for (pit = nblist.begin(); pit != nblist.end(); pit++) {
266 
267  QMPair *qmpair = *pit;
268  if (qmpair->HasGhost()) {
269  count++;
270 
271 
272 
273  qmpair->Seg2PbCopy()->Rigidify();
274  }
275  }
276 
277  cout << endl
278  << "... ... Rigidified " << count << " ghosts. "
279  << flush;
280  }
281 
282  _isRigid = true;
283  return 1;
284  }
285 }
