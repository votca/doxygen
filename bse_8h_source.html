67  void setBSEindices(int homo,int vmin, int cmax, int nmax) {
68  _homo=homo;
69  _bse_vmin = vmin;
70  _bse_vmax = homo;
71  _bse_cmin = homo+1;
72  _bse_cmax = cmax;
73  _bse_nmax = nmax;
74  _bse_vtotal = _bse_vmax - _bse_vmin + 1;
75  _bse_ctotal = _bse_cmax - _bse_cmin + 1;
76  _bse_size = _bse_vtotal * _bse_ctotal;
77  // indexing info BSE vector index to occupied/virtual orbital
78  for (int v = 0; v < _bse_vtotal; v++) {
79  for (int c = 0; c < _bse_ctotal; c++) {
80  _index2v.push_back(_bse_vmin + v);
81  _index2c.push_back(_bse_cmin + c);
82  }
83  }
84  return;
85  }
