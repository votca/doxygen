57 void TrajForce::BeginEvaluate(Topology *top, Topology *top_atom)
58 {
59  _top_force.CopyTopologyData(top);
60  _trjreader_force = TrjReaderFactory().Create(_op_vm["trj-force"].as<string>());
61  if(_trjreader_force == NULL)
62  throw runtime_error(string("input format not supported: ") + _op_vm["trj-force"].as<string>());
63  // open the trajectory
64  _trjreader_force->Open(_op_vm["trj-force"].as<string>());
65  // read in first frame
66  _trjreader_force->FirstFrame(_top_force);
67 
68  // output trajectory file
69  _trjwriter = TrjWriterFactory().Create(_op_vm["out"].as<string>());
70  if(_trjwriter == NULL)
71  throw runtime_error(string("output trajectory format not supported: ") + _op_vm["out"].as<string>());
72  bool append = true;
73  _trjwriter->Open(_op_vm["out"].as<string>(),append);
74 }
