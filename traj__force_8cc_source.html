54 void TrajForce::BeginEvaluate(Topology *top, Topology *) {

58  if (_trjreader_force == nullptr) {

60  _op_vm["trj-force"].as<string>());

62  // open the trajectory
63  _trjreader_force->Open(_op_vm["trj-force"].as<string>());
64  // read in first frame
65  _trjreader_force->FirstFrame(_top_force);
66 
67  // output trajectory file
68  _trjwriter = TrjWriterFactory().Create(_op_vm["out"].as<string>());
69  if (_trjwriter == nullptr) {
70  throw runtime_error(string("output trajectory format not supported: ") +
71  _op_vm["out"].as<string>());
72  }
73  bool append = true;
74  _trjwriter->Open(_op_vm["out"].as<string>(), append);

87 void TrajForce::EvalConfiguration(Topology *conf, Topology *) {
88  if (conf->BeadCount() != _top_force.BeadCount()) {
89  throw std::runtime_error(
90  "number of beads in topology and reference force topology does not "
91  "match");
92  }
93  for (votca::Index i = 0; i < conf->BeadCount(); ++i) {
94 
95  // \todo check why "conf" HasForce() is false
96  // Since "conf" topology Force is set to false
97  // for now using _top_force to store resultant output forces
98 
99  _top_force.getBead(i)->F() =
100  conf->getBead(i)->getF() + _scale * _top_force.getBead(i)->getF();
101  Eigen::Vector3d d =
102  conf->getBead(i)->getPos() - _top_force.getBead(i)->getPos();
103  if (d.norm() > 1e-6) {
104  throw std::runtime_error(
105  "One or more bead positions in trajectory and reference force "
106  "trajectory differ by more than 1e-6");
107  }
108  }
109 
110  _trjwriter->Write(&_top_force);
111  _trjreader_force->NextFrame(_top_force);
112 }
