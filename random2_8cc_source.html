28 void Random2::init( int nA1, int nA2, int nA3, int nB1 ) {
29 
30  nA1 = nA1 % 178 + 1;
31  nA2 = nA2 % 178 + 1;
32  nA3 = nA3 % 178 + 1;
33  nB1 = nB1 % 169;
34 
35  if ((nA1 == 1) && (nA2 == 1) && (nA3 == 1)) {
36  // Should not all be unity
37  cout << flush
38  << "WARNING: MARSAGLIA RNG INITIALISED INCORRECTLY. "
39  << "ADAPTING SEEDS APPROPRIATELY."
40  << endl;
41  nA1 += nB1;
42  }
43 
44  cout << flush
45  << "INITIALIZED MARSFIELD WITH "
46  << nA1 << " " << nA2 << " " << nA3 << " " << nB1
47  << endl;
48 
49  /*
50  initializes the global data of the
51  MARSAGLIA pseudo random number generator
52  */
53 
54  int mA1, mA2, mA3, mANEW, mB1, mHELP ;
55  int i1, i2 ;
56  double varS, varT ;
57 
58  MARSarray = (double*)malloc(MARS_FIELD_SIZE*sizeof(double));
59 
60  mA1 = nA1 ; mA2 = nA2 ; mA3 = nA3 ;
61  mB1 = nB1 ;
62  MARSi = 97 ; MARSj = 33 ;
63 
64  for( i1 = 1 ; i1 < 98 ; i1++ )
65  {
66  varS = 0.0 ;
67  varT = 0.5 ;
68  for( i2 = 1 ; i2 < 25 ; i2++ )
69  {
70  mANEW = ((( mA1 * mA2 ) % 179 )*mA3 ) % 179 ;
71  mA1 = mA2 ;
72  mA2 = mA3 ;
73  mA3 = mANEW ;
74  mB1 = ( 53*mB1 + 1 ) % 169 ;
75  mHELP = ( mB1 * mANEW ) % 64 ;
76  if( mHELP > 31 )
77  varS += varT ;
78  varT *= 0.5 ;
79  }
80 
81  MARSarray[ i1 ] = varS ;
82  }
83 
84  MARSc = 362436.0 / 16777216.0 ;
85  MARScd = 7654321.0 / 16777216.0 ;
86  MARScm = 16777213.0 / 16777216.0 ;
87 
88  return;
89 }

160 double Random2::rand_uniform( void ) {
161 
162  /*
163  generates a pseudo random number 0 .. +1
164  following the proposal of MARSAGLIA
165  */
166 
167  double ranMARS ;
168 
169  ranMARS = MARSarray[ MARSi ] - MARSarray[ MARSj ] ;
170  if( ranMARS < 0.0 )
171  ranMARS += 1.0 ;
172 
173  MARSarray[ MARSi ] = ranMARS ;
174 
175  MARSi-- ;
176  if( MARSi < 1 )
177  MARSi = 97 ;
178 
179  MARSj-- ;
180  if( MARSj < 1 )
181  MARSj = 97 ;
182 
183  MARSc -= MARScd ;
184  if( MARSc < 0.0 )
185  MARSc += MARScm ;
186 
187  ranMARS -= MARSc ;
188  if( ranMARS < 0.0 )
189  ranMARS += 1.0 ;
190 
191  return ranMARS ;
192 }
