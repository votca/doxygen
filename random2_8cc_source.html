00156     /*
00157      generates a pseudo random number 0 .. +1
00158      following the proposal of MARSAGLIA
00159     */
00160 
00161     double ranMARS ;
00162 
00163     ranMARS = MARSarray[ MARSi ] - MARSarray[ MARSj ] ;
00164     if( ranMARS < 0.0 )
00165     ranMARS += 1.0 ;
00166 
00167     MARSarray[ MARSi ] = ranMARS ;
00168 
00169     MARSi-- ;
00170     if( MARSi < 1 )
00171     MARSi = 97 ;
00172 
00173     MARSj-- ;
00174     if( MARSj < 1 )
00175     MARSj = 97 ;
00176 
00177     MARSc -= MARScd ;
00178     if( MARSc < 0.0 )
00179     MARSc += MARScm ;
00180 
00181     ranMARS -= MARSc ;
