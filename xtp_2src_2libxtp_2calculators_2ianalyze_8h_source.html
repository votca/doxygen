52 void IAnalyze::Initialize(tools::Property *opt) {
53  _do_pairtype=false;
54  _do_IRdependence=false;
55  // update options with the VOTCASHARE defaults
56  UpdateWithDefaults( opt, "xtp" );
57  std::string key = "options." + Identify();
58  _states = opt->get(key+".states").as< std::vector<int> >();
59  _resolution_logJ2 = opt->get(key+".resolution_logJ2").as< double >();
60 
61  if ( opt->exists(key+".pairtype")) {
62  _do_pairtype=true;
63  std::string _store_stdstring = opt->get(key+".pairtype").as<std::string> ();
64  if (_store_stdstring.find("Hopping") != std::string::npos) _pairtype.push_back(ctp::QMPair::Hopping);
65  if (_store_stdstring.find("SuperExchange") != std::string::npos) _pairtype.push_back(ctp::QMPair::SuperExchange);
66  if (_store_stdstring.find("SuperExchangeAndHopping") != std::string::npos) _pairtype.push_back(ctp::QMPair::SuperExchangeAndHopping);
67  if (_store_stdstring.find("Excitoncl") != std::string::npos) _pairtype.push_back(ctp::QMPair::Excitoncl);
68  if (!_pairtype.size()){
69  std::cout << std::endl << "... ... No pairtypes recognized will output all pairs. ";
70  _do_pairtype=false;
71  }
72  //std::cout <<_pairtype.size()<<std::endl;
73  }
74  if ( opt->exists(key+".resolution_space")) {
75 
76  _resolution_space = opt->get(key+".resolution_space").as< double >();
77  if (_resolution_space!=0.0) _do_IRdependence=true;
78  }

83 bool IAnalyze::EvaluateFrame(ctp::Topology *top) {
84 
85  ctp::QMNBList &nblist = top->NBList();
86 
87  if (!nblist.size()) {
88  std::cout << std::endl << "... ... No pairs in topology. Skip...";
89  return 0;
90  }
91 
92  if (_do_pairtype){
93  bool pairs_exist=false;
94  ctp::QMNBList::iterator nit;
95  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
96  ctp::QMPair::PairType pairtype=(*nit)->getType();
97  if(std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end()) {
98  pairs_exist=true;
99  break;
100  }
101  }
102  if (!pairs_exist) {
103  std::cout << std::endl << "... ... No pairs of given pairtypes in topology. Skip...";
104  return 0;
105  }
106  }
107 
108  for (unsigned int i = 0; i < _states.size(); ++i) {
109 
110  this->IHist(top, _states[i]);
111  if (_do_IRdependence){
112  this->IRdependence(top, _states[i]);
113  }
114  }
115 
116  return true;
117 }

120 void IAnalyze::IHist(ctp::Topology *top, int state) {
121 
122  ctp::QMNBList &nblist = top->NBList();
123  ctp::QMNBList::iterator nit;
124 
125 
126  double MIN = std::numeric_limits<double>::max();
127  double MAX = 0.0;
128 
129  // Collect J2s from pairs
130  std::vector< double > J2s;
131  //J2s.reserve(nblist.size()); //does not make a difference
132 
133  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
134  if(_do_pairtype){
135  ctp::QMPair::PairType pairtype=(*nit)->getType();
136  if(!(std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end())){
137  continue;
138  }
139  }
140  double test = (*nit)->getJeff2(state);
141  double J2=0.0;
142 
143  if(test <= 0) {continue;} // avoid -inf in output
144  J2=log10(test);
145 
146 
147  MIN = (J2 < MIN) ? J2 : MIN;
148  MAX = (J2 > MAX) ? J2 : MAX;
149  J2s.push_back(J2);
150 
151  }
152 
153  if(J2s.size()<1){
154  std::cerr <<"Error: Couplings are all zero. You have not yet imported them! "<<std::endl;
155  std::exit(0);
156  }
157 
158  // Prepare bins
159  int BIN = ( (MAX-MIN)/_resolution_logJ2 + 0.5 ) + 1;
160  std::vector< std::vector<double> > histJ2;
161 
162  histJ2.resize(BIN);
163 
164  // Execute binning
165  std::vector< double > ::iterator jit;
166  for (jit = J2s.begin(); jit < J2s.end(); ++jit) {
167 
168  int bin = int( (*jit-MIN)/_resolution_logJ2 + 0.5 );
169  histJ2[bin].push_back(*jit);
170  }
171 
172  std::vector< int > histN;
173  histN.resize(BIN);
174  for (int bin = 0; bin < BIN; ++bin) {
175  histN[bin] = histJ2[bin].size();
176  }
177  FILE *out;
178  std::string name;
179  if (state==-1) name="e";
180  else if (state==1) name="h";
181  else if (state==2) name="s";
182  else if (state==3) name="t";
183  std::string tag = boost::lexical_cast<std::string>("ianalyze.ihist_") +name + ".out";
184  out = fopen(tag.c_str(), "w");
185  std::cout << std::endl << "... Printing to "<<tag.c_str()<<std::endl;
186  fprintf(out, "# IANALYZE: PAIR-INTEGRAL J2 HISTOGRAM\n");
187  fprintf(out, "# STATE %1d\n", state);
188 
189  for (int bin = 0; bin < BIN; ++bin) {
190  double J2 = MIN + bin*_resolution_logJ2;
191  fprintf(out, "%4.7f %4d \n", J2, histN[bin]);
192  }
193  fclose(out);
194 }

196 void IAnalyze::IRdependence(ctp::Topology *top, int state) {
197 
198  ctp::QMNBList &nblist = top->NBList();
199  ctp::QMNBList::iterator nit;
200 
201  double MIN = log10(nblist.front()->getJeff2(state));
202  double MAX = log10(nblist.front()->getJeff2(state));
203  double MINR = abs(nblist.front()->getR());
204  double MAXR = abs(nblist.front()->getR());
205 
206  // Collect J2s from pairs
207  std::vector< double > J2s;
208  J2s.reserve(nblist.size());
209  std::vector< double > distances;
210  distances.reserve(nblist.size());
211 
212  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
213  double J2 = log10((*nit)->getJeff2(state));
214 
215  MIN = (J2 < MIN) ? J2 : MIN;
216  MAX = (J2 > MAX) ? J2 : MAX;
217 
218  double distance = abs((*nit)->getR());
219 
220  MINR = (distance < MINR) ? distance : MINR;
221  MAXR = (distance > MAXR) ? distance : MAXR;
222 
223  distances.push_back(distance);
224  J2s.push_back(J2);
225  }
226 
227  // Prepare R bins
228  int _pointsR = (MAXR-MINR)/_resolution_space;
229  std::vector< std::vector<double> > rJ2;
230  rJ2.resize(_pointsR);
231 
232 
233  // Loop over distance
234  for (int i = 0; i< _pointsR; ++i){
235  double thisMINR = MINR + i*_resolution_space;
236  double thisMAXR = MINR + (i+1)*_resolution_space;
237 
238  // now count Js that lie within this R range, calculate mean and sigma
239  //double meanJ2 = 0;
240  //double sigmaJ2 = 0;
241  //int noJ2 = 0;
242 
243  std::vector< double > ::iterator jit;
244  int j = 0;
245  for (jit = J2s.begin(); jit < J2s.end(); ++jit) {
246  if(thisMINR < distances[j] && distances[j] < thisMAXR){
247  rJ2[i].push_back(*jit);
248  }
249  j++;
250  }
251  }
252 
253  // make plot values
254  std::vector< double > avgJ2;
255  for (std::vector< std::vector<double> > ::iterator it = rJ2.begin() ; it != rJ2.end(); ++it){
256  double thisavgJ2 = 0;
257  for(unsigned i=0; i < (*it).size(); i++){
258  thisavgJ2 += (*it)[i];
259  }
260  thisavgJ2 /= (*it).size();
261  avgJ2.push_back(thisavgJ2);
262  }
263  std::vector< double > errJ2;
264  int j = 0;
265  for (std::vector< std::vector<double> > ::iterator it = rJ2.begin() ; it != rJ2.end(); ++it){
266  double thiserrJ2 = 0;
267  for(unsigned i=0; i < (*it).size(); i++){
268  thiserrJ2 += ((*it)[i]-avgJ2[j])*((*it)[i]-avgJ2[j]);
269  }
270  thiserrJ2 /= (*it).size();
271  thiserrJ2 = sqrt(thiserrJ2);
272  errJ2.push_back(thiserrJ2);
273  j++;
274  }
275 
276 
277  // print to file
278  FILE *out;
279  std::string name;
280  if (state==-1) name="e";
281  else if (state==1) name="h";
282  else if (state==2) name="s";
283  else if (state==3) name="t";
284  std::string tag = boost::lexical_cast<std::string>("ianalyze.ispatial_") + name + ".out";
285  out = fopen(tag.c_str(), "w");
286 
287  fprintf(out, "# IANALYZE: SPATIAL DEPENDENCE OF log10(J2) [r,log10(J),error]\n");
288  fprintf(out, "# STATE %1d\n", state);
289 
290 
291  for (int i = 0; i < _pointsR; ++i) {
292  double thisR = MINR + (i+0.5)*_resolution_space;
293  fprintf(out, "%4.7f %4.7f %4.7f \n", thisR, avgJ2[i], errJ2[i]);
294  }
295  fclose(out);
296 
297 
298 }
