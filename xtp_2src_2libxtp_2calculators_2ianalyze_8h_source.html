56  void IAnalyze::Initialize(tools::Property *opt) {
57  std::cout<<std::endl;
58  _do_pairtype = false;
59  _do_IRdependence = false;
60  // update options with the VOTCASHARE defaults
61  UpdateWithDefaults(opt, "xtp");
62  std::string key = "options." + Identify();
63  if (opt->exists(key + ".states")) {
64  std::string statestrings = opt->get(key + ".states").as< std::string>();
65  tools::Tokenizer tok(statestrings, ",\n\t ");
66  std::vector<std::string> string_vec;
67  tok.ToVector(string_vec);
68  for (std::string& state : string_vec) {
69  _states.push_back(QMStateType(state));
70  }
71  } else {
72  _states.push_back(QMStateType(QMStateType::Electron));
73  _states.push_back(QMStateType(QMStateType::Hole));
74  }
75 
76  _resolution_logJ2 = opt->get(key + ".resolution_logJ2").as< double >();
77  if (opt->exists(key + ".pairtype")) {
78  _do_pairtype = true;
79  std::string _store_stdstring = opt->get(key + ".pairtype").as<std::string> ();
80  if (_store_stdstring.find("Hopping") != std::string::npos) _pairtype.push_back(ctp::QMPair::Hopping);
81  if (_store_stdstring.find("Excitoncl") != std::string::npos) _pairtype.push_back(ctp::QMPair::Excitoncl);
82  if (!_pairtype.size()) {
83  std::cout << std::endl << "... ... No pairtypes recognized will output all pairs. ";
84  _do_pairtype = false;
85  }
86  }
87  if (opt->exists(key + ".resolution_space")) {
88  _resolution_space = opt->get(key + ".resolution_space").as< double >();
89  if (_resolution_space != 0.0) _do_IRdependence = true;
90  }
91  }

94  std::cout<<std::endl;
95  ctp::QMNBList &nblist = top->NBList();
96  if (!nblist.size()) {

100  if (_do_pairtype) {
101  bool pairs_exist = false;
102  for (ctp::QMPair* pair:nblist) {
103  ctp::QMPair::PairType pairtype = pair->getType();
104  if (std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end()) {
105  pairs_exist = true;
106  break;
107  }
108  }
109  if (!pairs_exist) {
110  std::cout << std::endl << "... ... No pairs of given pairtypes in topology. Skip...";
111  return 0;
112  }
113  }
114  for (QMStateType state:_states) {
115  std::cout<<"Calculating for state "<<state.ToString()<<" now."<<std::endl;
116  this->IHist(top, state);
117  if (_do_IRdependence) {
118  this->IRdependence(top, state);
119  }
120  }
121  return true;
122  }

124  void IAnalyze::IHist(ctp::Topology *top, QMStateType state) {
125  ctp::QMNBList &nblist = top->NBList();
126 
127  // Collect J2s from pairs
128  std::vector< double > J2s;
129  for (ctp::QMPair* pair:nblist) {
130  if (_do_pairtype) {
131  ctp::QMPair::PairType pairtype =pair->getType();
132  if (!(std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end())) {
133  continue;
134  }
135  }
136  double test = pair->getJeff2(state.ToCTPIndex());
137  if (test <= 0) {
138  continue;
139  } // avoid -inf in output
140  double J2 = std::log10(test);
141  J2s.push_back(J2);
142  }
143 
144  if (J2s.size() < 1) {
145  std::cout<<"WARNING:"+state.ToLongString()+" Couplings are all zero. You have not yet imported them! "<<std::endl;
146  return;
147  }
148 
149  double MAX = *std::max_element(J2s.begin(), J2s.end());
150  double MIN = *std::min_element(J2s.begin(), J2s.end());
151  double sum = std::accumulate(J2s.begin(), J2s.end(), 0.0);
152  double AVG = sum / J2s.size();
153  double sq_sum = std::inner_product(J2s.begin(), J2s.end(), J2s.begin(), 0.0);
154  double STD = std::sqrt(sq_sum / J2s.size() - AVG * AVG);
155  // Prepare bins
156  int BIN = ((MAX - MIN) / _resolution_logJ2 + 0.5) + 1;
157 
158  tools::HistogramNew hist;
159  hist.Initialize(MIN, MAX, BIN);
160  hist.ProcessRange<std::vector<double>::iterator>(J2s.begin(), J2s.end());
161  tools::Table& tab = hist.data();
162  std::string comment = (boost::format("IANALYZE: PAIR-INTEGRAL J2 HISTOGRAM \n # AVG %1$4.7f STD %2$4.7f MIN %3$4.7f MAX %4$4.7f") % AVG % STD % MIN % MAX).str();
163  std::string filename = "ianalyze.ihist_" + state.ToString() + ".out";
164  tab.set_comment(comment);
165  tab.flags()= std::vector<char>(tab.size(), ' ');
166  tab.Save(filename);
167 
168  }

170  void IAnalyze::IRdependence(ctp::Topology *top, QMStateType state) {
171 
172  ctp::QMNBList &nblist = top->NBList();
173 
174  // Collect J2s from pairs
175  std::vector< double > J2s;
176  J2s.reserve(nblist.size());
177  std::vector< double > distances;
178  distances.reserve(nblist.size());
179 
180  for (ctp::QMPair* pair:nblist) {
181  double J2 = std::log10(pair->getJeff2(state.ToCTPIndex()));
182  double distance = tools::abs(pair->getR());
183  distances.push_back(distance);
184  J2s.push_back(J2);
185  }
186 
187  double MAXR = *std::max_element(distances.begin(), distances.end());
188  double MINR = *std::min_element(distances.begin(), distances.end());
189 
190  // Prepare R bins
191  int pointsR = (MAXR - MINR) / _resolution_space;
192  std::vector< std::vector<double> > rJ2;
193  rJ2.resize(pointsR);
194 
195  // Loop over distance
196  for (int i = 0; i < pointsR; ++i) {
197  double thisMINR = MINR + i*_resolution_space;
198  double thisMAXR = MINR + (i + 1) * _resolution_space;
199  // now count Js that lie within this R range
200  for (unsigned j=0;j<J2s.size();++j) {
201  if (thisMINR < distances[j] && distances[j] < thisMAXR) {
202  rJ2[i].push_back(J2s[j]);
203  }
204  }
205  }
206 
207  tools::Table tab;
208  tab.SetHasYErr(true);
209  tab.resize(pointsR);
210 
211  // make plot values
212  for (unsigned i=0;i<rJ2.size();i++) {
213  const std::vector<double>& vec=rJ2[i];
214  double sum = std::accumulate(vec.begin(), vec.end(), 0.0);
215  double AVG = sum / vec.size();
216  double thisR = MINR + (i + 0.5) * _resolution_space;
217  double sq_sum = std::inner_product(vec.begin(), vec.end(), vec.begin(), 0.0);
218  double STD = std::sqrt(sq_sum / vec.size() - AVG * AVG);
219  tab.set(i,thisR,AVG,' ',STD);
220  }
221  std::string filename = "ianalyze.ispatial_" + state.ToString() + ".out";
222  std::string comment ="# IANALYZE: SPATIAL DEPENDENCE OF log10(J2) [r,log10(J),error]";
223  tab.setErrorDetails(comment);
224  tab.Save(filename);
225 
226  }
227 
228  }
