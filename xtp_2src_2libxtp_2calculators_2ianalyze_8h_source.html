52 void IAnalyze::Initialize(tools::Property *opt) {
53  _do_pairtype=false;
54  _do_IRdependence=false;
55  // update options with the VOTCASHARE defaults
56  UpdateWithDefaults( opt, "xtp" );
57  std::string key = "options." + Identify();
58  _states = opt->get(key+".states").as< std::vector<int> >();
59  _resolution_logJ2 = opt->get(key+".resolution_logJ2").as< double >();
60 
61  if ( opt->exists(key+".pairtype")) {
62  _do_pairtype=true;
63  std::string _store_stdstring = opt->get(key+".pairtype").as<std::string> ();
64  if (_store_stdstring.find("Hopping") != std::string::npos) _pairtype.push_back(ctp::QMPair::Hopping);
65  if (_store_stdstring.find("SuperExchange") != std::string::npos) _pairtype.push_back(ctp::QMPair::SuperExchange);
66  if (_store_stdstring.find("SuperExchangeAndHopping") != std::string::npos) _pairtype.push_back(ctp::QMPair::SuperExchangeAndHopping);
67  if (_store_stdstring.find("Excitoncl") != std::string::npos) _pairtype.push_back(ctp::QMPair::Excitoncl);
68  if (!_pairtype.size()){
69  std::cout << std::endl << "... ... No pairtypes recognized will output all pairs. ";
70  _do_pairtype=false;
71  }
72  //std::cout <<_pairtype.size()<<std::endl;
73  }
74  if ( opt->exists(key+".resolution_space")) {
75 
76  _resolution_space = opt->get(key+".resolution_space").as< double >();
77  if (_resolution_space!=0.0) _do_IRdependence=true;
78  }

80 }

83 bool IAnalyze::EvaluateFrame(ctp::Topology *top) {
84 
85  ctp::QMNBList &nblist = top->NBList();
86 
87  if (!nblist.size()) {
88  std::cout << std::endl << "... ... No pairs in topology. Skip...";
89  return 0;
90  }
91 
92  if (_do_pairtype){
93  bool pairs_exist=false;
94  ctp::QMNBList::iterator nit;
95  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
96  ctp::QMPair::PairType pairtype=(*nit)->getType();
97  if(std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end()) {
98  pairs_exist=true;
99  break;
100  }
101  }
102  if (!pairs_exist) {
103  std::cout << std::endl << "... ... No pairs of given pairtypes in topology. Skip...";
104  return 0;
105  }
106  }
107 
108  for (unsigned int i = 0; i < _states.size(); ++i) {
109 
110  this->IHist(top, _states[i]);
111  if (_do_IRdependence){
112  this->IRdependence(top, _states[i]);
113  }
114  }
115 
116  return true;
117 }

120 void IAnalyze::IHist(ctp::Topology *top, int state) {
121 
122  ctp::QMNBList &nblist = top->NBList();
123  ctp::QMNBList::iterator nit;
124 
125 
126  double MIN = std::numeric_limits<double>::max();
127  double MAX = 0.0;
128 
129  // Collect J2s from pairs
130  std::vector< double > J2s;
131  //J2s.reserve(nblist.size()); //does not make a difference
132 
133  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
134  if(_do_pairtype){
135  ctp::QMPair::PairType pairtype=(*nit)->getType();
136  if(!(std::find(_pairtype.begin(), _pairtype.end(), pairtype) != _pairtype.end())){
137  continue;
138  }
139  }
140  double test = (*nit)->getJeff2(state);
141  double J2=0.0;
142 
143  if(test <= 0) {continue;} // avoid -inf in output
144  J2=log10(test);
145 
146 
147  MIN = (J2 < MIN) ? J2 : MIN;
148  MAX = (J2 > MAX) ? J2 : MAX;
149  J2s.push_back(J2);
150 
151  }
152 
153  if(J2s.size()<1){
154  std::cerr <<"Error: Couplings are all zero. You have not yet imported them! "<<std::endl;
155  std::exit(0);
156  }
157 
158  // Prepare bins
159  int BIN = ( (MAX-MIN)/_resolution_logJ2 + 0.5 ) + 1;
160  std::cout<<BIN<<std::endl;
161  std::vector< std::vector<double> > histJ2;
162 
163  histJ2.resize(BIN);
164 
165  // Execute binning
166  std::vector< double > ::iterator jit;
167  for (jit = J2s.begin(); jit < J2s.end(); ++jit) {
168 
169  int bin = int( (*jit-MIN)/_resolution_logJ2 + 0.5 );
170  histJ2[bin].push_back(*jit);
171  }
172 
173  std::vector< int > histN;
174  histN.resize(BIN);
175  for (int bin = 0; bin < BIN; ++bin) {
176  histN[bin] = histJ2[bin].size();
177  }
178  FILE *out;
179  std::string name;
180  if (state==-1) name="e";
181  else if (state==1) name="h";
182  else if (state==2) name="s";
183  else if (state==3) name="t";
184  std::string tag = boost::lexical_cast<std::string>("ianalyze.ihist_") +name + ".out";
185  out = fopen(tag.c_str(), "w");
186  std::cout << std::endl << "... Printing to tag.c_str()";
187  fprintf(out, "# IANALYZE: PAIR-INTEGRAL J2 HISTOGRAM\n");
188  fprintf(out, "# STATE %1d\n", state);
189 
190  for (int bin = 0; bin < BIN; ++bin) {
191  double J2 = MIN + bin*_resolution_logJ2;
192  fprintf(out, "%4.7f %4d \n", J2, histN[bin]);
193  }
194  fclose(out);
195 }

197 void IAnalyze::IRdependence(ctp::Topology *top, int state) {
198 
199  ctp::QMNBList &nblist = top->NBList();
200  ctp::QMNBList::iterator nit;
201 
202  double MIN = log10(nblist.front()->getJeff2(state));
203  double MAX = log10(nblist.front()->getJeff2(state));
204  double MINR = abs(nblist.front()->getR());
205  double MAXR = abs(nblist.front()->getR());
206 
207  // Collect J2s from pairs
208  std::vector< double > J2s;
209  J2s.reserve(nblist.size());
210  std::vector< double > distances;
211  distances.reserve(nblist.size());
212 
213  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
214  double J2 = log10((*nit)->getJeff2(state));
215 
216  MIN = (J2 < MIN) ? J2 : MIN;
217  MAX = (J2 > MAX) ? J2 : MAX;
218 
219  double distance = abs((*nit)->getR());
220 
221  MINR = (distance < MINR) ? distance : MINR;
222  MAXR = (distance > MAXR) ? distance : MAXR;
223 
224  distances.push_back(distance);
225  J2s.push_back(J2);
226  }
227 
228  // Prepare R bins
229  int _pointsR = (MAXR-MINR)/_resolution_space;
230  std::vector< std::vector<double> > rJ2;
231  rJ2.resize(_pointsR);
232 
233 
234  // Loop over distance
235  for (int i = 0; i< _pointsR; ++i){
236  double thisMINR = MINR + i*_resolution_space;
237  double thisMAXR = MINR + (i+1)*_resolution_space;
238 
239  // now count Js that lie within this R range, calculate mean and sigma
240  //double meanJ2 = 0;
241  //double sigmaJ2 = 0;
242  //int noJ2 = 0;
243 
244  std::vector< double > ::iterator jit;
245  int j = 0;
246  for (jit = J2s.begin(); jit < J2s.end(); ++jit) {
247  if(thisMINR < distances[j] && distances[j] < thisMAXR){
248  rJ2[i].push_back(*jit);
249  }
250  j++;
251  }
252  }
253 
254  // make plot values
255  std::vector< double > avgJ2;
256  for (std::vector< std::vector<double> > ::iterator it = rJ2.begin() ; it != rJ2.end(); ++it){
257  double thisavgJ2 = 0;
258  for(unsigned i=0; i < (*it).size(); i++){
259  thisavgJ2 += (*it)[i];
260  }
261  thisavgJ2 /= (*it).size();
262  avgJ2.push_back(thisavgJ2);
263  }
264  std::vector< double > errJ2;
265  int j = 0;
266  for (std::vector< std::vector<double> > ::iterator it = rJ2.begin() ; it != rJ2.end(); ++it){
267  double thiserrJ2 = 0;
268  for(unsigned i=0; i < (*it).size(); i++){
269  thiserrJ2 += ((*it)[i]-avgJ2[j])*((*it)[i]-avgJ2[j]);
270  }
271  thiserrJ2 /= (*it).size();
272  thiserrJ2 = sqrt(thiserrJ2);
273  errJ2.push_back(thiserrJ2);
274  j++;
275  }
276 
277 
278  // print to file
279  FILE *out;
280  std::string name;
281  if (state==-1) name="e";
282  else if (state==1) name="h";
283  else if (state==2) name="s";
284  else if (state==3) name="t";
285  std::string tag = boost::lexical_cast<std::string>("ianalyze.ispatial_") + name + ".out";
286  out = fopen(tag.c_str(), "w");
287 
288  fprintf(out, "# IANALYZE: SPATIAL DEPENDENCE OF log10(J2) [r,log10(J),error]\n");
289  fprintf(out, "# STATE %1d\n", state);
290 
291 
292  for (int i = 0; i < _pointsR; ++i) {
293  double thisR = MINR + (i+0.5)*_resolution_space;
294  fprintf(out, "%4.7f %4.7f %4.7f \n", thisR, avgJ2[i], errJ2[i]);
295  }
296  fclose(out);
297 
298 
299 }
