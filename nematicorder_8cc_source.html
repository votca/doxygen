25 void NematicOrder::Process(Topology &top, const string &filter) {
26  _mu = Eigen::Matrix3d::Zero();
27  _mv = Eigen::Matrix3d::Zero();
28  _mw = Eigen::Matrix3d::Zero();
29  int N = 0;
30  bool bU, bV, bW;
31  bU = bV = bW = false;
32 
33  for (BeadContainer::iterator iter = top.Beads().begin();
34  iter != top.Beads().end(); ++iter) {
35 
36  Bead *bead = *iter;
37 
38  if (!wildcmp(filter.c_str(), bead->getName().c_str())) continue;
39 
40  if (bead->getSymmetry() == 1) continue;
41 
42  if (bead->HasU()) {
43  _mu += bead->getU() * bead->getU().transpose();
44  _mu.diagonal().array() -= 1. / 3.;
45  bU = true;
46  }
47 
48  if (bead->HasV()) {
49  _mu += bead->getV() * bead->getV().transpose();
50  _mu.diagonal().array() -= 1. / 3.;
51  bV = true;
52  }
53 
54  if (bead->HasW()) {
55  _mu += bead->getW() * bead->getW().transpose();
56  _mu.diagonal().array() -= 1. / 3.;
57  bW = true;
58  }
59  N++;
60  }
61 
62  double f = 1. / (double)N * 3. / 2.;
63  _mu = f * _mu;
64  _mv = f * _mv;
65  _mw = f * _mw;
66 
67  if (bU) _nemat_u.computeDirect(_mu);
68  if (bV) _nemat_v.computeDirect(_mv);
69  if (bW) _nemat_w.computeDirect(_mw);
70 }
