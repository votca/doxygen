25 void NematicOrder::Process(Topology &top, const string &filter)
26 {
27  _mu.ZeroMatrix();
28  _mv.ZeroMatrix();
29  _mw.ZeroMatrix();
30  int N=0;
31  bool bU, bV, bW;
32  bU=bV=bW = false;
33 
34  for(BeadContainer::iterator iter = top.Beads().begin();
35  iter!=top.Beads().end();++iter) {
36 
37  Bead *bead = *iter;
38 
39  if(!wildcmp(filter.c_str(), bead->getName().c_str())) continue;
40 
41  if( bead->getSymmetry() ==1 )
42  continue;
43 
44  if(bead->HasU()) {
45  _mu += bead->getU()|bead->getU();
46  _mu[0][0] -= 1./3.;
47  _mu[1][1] -= 1./3.;
48  _mu[2][2] -= 1./3.;
49  bU = true;
50  }
51 
52  if(bead->HasV()) {
53  _mv += bead->getV()|bead->getV();
54  _mv[0][0] -= 1./3.;
55  _mv[1][1] -= 1./3.;
56  _mv[2][2] -= 1./3.;
57  bV = true;
58  }
59 
60  if(bead->HasW()) {
61  _mw += bead->getW()|bead->getW();
62  _mw[0][0] -= 1./3.;
63  _mw[1][1] -= 1./3.;
64  _mw[2][2] -= 1./3.;
65  bW = true;
66  }
67  N++;
68  }
69 
70  double f = 1./(double)N*3./2.;
71  _mu = f*_mu;_mv = f*_mv;_mw = f*_mw;
72 
73  if(bU)
74  _mu.SolveEigensystem(_nemat_u);
75  if(bV)
76  _mv.SolveEigensystem(_nemat_v);
77  if(bW)
78  _mw.SolveEigensystem(_nemat_w);
79 }
