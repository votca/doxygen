89  void setupgrid(const vector< QMAtom* >& Atomlist){
90  Elements _elements;
91  double xmin=1000;
92  double ymin=1000;
93  double zmin=1000;
94 
95  double xmax=-1000;
96  double ymax=-1000;
97  double zmax=-1000;

99  //setup one polarsite and use copy constructor later
100 
101 
102  for (vector<QMAtom* >::const_iterator atom = Atomlist.begin(); atom != Atomlist.end(); ++atom ) {
103  xtemp=(*atom)->x;
104  ytemp=(*atom)->y;
105  ztemp=(*atom)->z;
106  if (xtemp<xmin)
107  xmin=xtemp;
108  if (xtemp>xmax)
109  xmax=xtemp;
110  if (ytemp<ymin)
111  ymin=ytemp;
112  if (ytemp>ymax)
113  ymax=ytemp;
114  if (ztemp<zmin)
115  zmin=ztemp;
116  if (ztemp>zmax)
117  zmax=ztemp;
118 
119  }
120 
121  //[-Wunused-variable]
122  //double boxdimx=xmax-xmin+2*_padding;
123 
124 
125  double x=xmin-_padding;
126 
127 
128  ub::vector<double> temppos= ub::zero_vector<double>(3);
129  while(x< xmax+_padding){
130  double y=ymin-_padding;
131  while(y< ymax+_padding){
132  double z=zmin-_padding;
133  while(z< zmax+_padding){
134  bool _is_valid = false;
135  for (vector<QMAtom* >::const_iterator atom = Atomlist.begin(); atom != Atomlist.end(); ++atom ) {
136  //cout << "Punkt " << x <<":"<< y << ":"<<z << endl;
137  xtemp=(*atom)->x;
138  ytemp=(*atom)->y;
139  ztemp=(*atom)->z;
140  double distance2=pow((x-xtemp),2)+pow((y-ytemp),2)+pow((z-ztemp),2);
141  double VdW=_elements.getVdWChelpG((*atom)->type);
142  //cout << "Punkt " << x <<":"<< y << ":"<<z << ":"<< distance2 << ":"<< (*atom)->type <<":"<<pow(VdW,2)<< endl;
143  if ( _outsidemolecule && distance2<pow(VdW,2)){
144  //cout << "Punkt" << x <<":"<< y << ":"<<z << "rejected" << endl;
145 
146  _is_valid = false;
147  break;
148  }
149  else if (distance2<pow(_cutoff,2)){
150  //cout << "hier" << endl;
151  _is_valid = true;
152  }
153 
154 
155 
156  }
157  if (_is_valid){
158  temppos(0)=x;
159  temppos(1)=y;
160  temppos(2)=z;
161  _gridpoints.push_back(temppos);
162  if(_createpolarsites){
163  vec temp=vec(x,y,z);
164  APolarSite apolarsite;
165  apolarsite.setPos(temp);
166  _gridsites.push_back(apolarsite);
167  }
168  }
169  z+=_gridspacing;
170  }
171  y+=_gridspacing;
172  //cout << "Punkt " << x <<":"<< xmax+padding <<":"<< y << ":"<<z << endl;
173  }
174  x+=_gridspacing;
175  //cout << (x<xmax+padding) << endl;
176  }
