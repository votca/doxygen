43 void BSE::configure(const options& opt, const Eigen::VectorXd& RPAInputEnergies,
44  const Eigen::MatrixXd& Hqp_in) {
45  _opt = opt;
46  _bse_vmax = _opt.homo;
47  _bse_cmin = _opt.homo + 1;
48  _bse_vtotal = _bse_vmax - _opt.vmin + 1;
49  _bse_ctotal = _opt.cmax - _bse_cmin + 1;
50  _bse_size = _bse_vtotal * _bse_ctotal;
51  _max_dyn_iter = _opt.max_dyn_iter;
52  _dyn_tolerance = _opt.dyn_tolerance;
53  if (_opt.use_Hqp_offdiag) {
54  _Hqp = AdjustHqpSize(Hqp_in, RPAInputEnergies);
55  } else {
56  _Hqp = AdjustHqpSize(Hqp_in, RPAInputEnergies).diagonal().asDiagonal();
57  }
58  SetupDirectInteractionOperator(RPAInputEnergies, 0.0);
59 }

61 Eigen::MatrixXd BSE::AdjustHqpSize(const Eigen::MatrixXd& Hqp,
62  const Eigen::VectorXd& RPAInputEnergies) {
63 
64  Index hqp_size = _bse_vtotal + _bse_ctotal;
65  Index gwsize = _opt.qpmax - _opt.qpmin + 1;
66  Index RPAoffset = _opt.vmin - _opt.rpamin;
67  Eigen::MatrixXd Hqp_BSE = Eigen::MatrixXd::Zero(hqp_size, hqp_size);
68 
69  if (_opt.vmin >= _opt.qpmin) {
70  Index start = _opt.vmin - _opt.qpmin;
71  if (_opt.cmax <= _opt.qpmax) {
72  Hqp_BSE = Hqp.block(start, start, hqp_size, hqp_size);
73  } else {
74  Index virtoffset = gwsize - start;
75  Hqp_BSE.topLeftCorner(virtoffset, virtoffset) =
76  Hqp.block(start, start, virtoffset, virtoffset);
77 
78  Index virt_extra = _opt.cmax - _opt.qpmax;
79  Hqp_BSE.diagonal().tail(virt_extra) =
80  RPAInputEnergies.segment(RPAoffset + virtoffset, virt_extra);
81  }
82  }
83 
84  if (_opt.vmin < _opt.qpmin) {
85  Index occ_extra = _opt.qpmin - _opt.vmin;
86  Hqp_BSE.diagonal().head(occ_extra) =
87  RPAInputEnergies.segment(RPAoffset, occ_extra);
88 
89  Hqp_BSE.block(occ_extra, occ_extra, gwsize, gwsize) = Hqp;
90 
91  if (_opt.cmax > _opt.qpmax) {
92  Index virtoffset = occ_extra + gwsize;
93  Index virt_extra = _opt.cmax - _opt.qpmax;
94  Hqp_BSE.diagonal().tail(virt_extra) =
95  RPAInputEnergies.segment(RPAoffset + virtoffset, virt_extra);
96  }
97  }
98 
99  return Hqp_BSE;
100 }

102 void BSE::SetupDirectInteractionOperator(
103  const Eigen::VectorXd& RPAInputEnergies, double energy) {
104  RPA rpa = RPA(_log, _Mmn);
105  rpa.configure(_opt.homo, _opt.rpamin, _opt.rpamax);
106  rpa.setRPAInputEnergies(RPAInputEnergies);
107 
108  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(
109  rpa.calculate_epsilon_r(energy));
110  _Mmn.MultiplyRightWithAuxMatrix(es.eigenvectors());
111 
112  _epsilon_0_inv = Eigen::VectorXd::Zero(es.eigenvalues().size());
113  for (Index i = 0; i < es.eigenvalues().size(); ++i) {
114  if (es.eigenvalues()(i) > 1e-8) {
115  _epsilon_0_inv(i) = 1 / es.eigenvalues()(i);
116  }
117  }
118 }

180 template <typename BSE_OPERATOR>
181 tools::EigenSystem BSE::solve_hermitian(BSE_OPERATOR& h) const {
182 
183  std::chrono::time_point<std::chrono::system_clock> start, end;
184  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
185  std::chrono::duration<double> elapsed_time;
186  start = std::chrono::system_clock::now();
187 
188  tools::EigenSystem result;
189  if (_opt.davidson) {
190 
191  DavidsonSolver DS(_log);
192 
193  DS.set_correction(_opt.davidson_correction);
194  DS.set_tolerance(_opt.davidson_tolerance);
195  DS.set_ortho(_opt.davidson_ortho);
196  DS.set_size_update(_opt.davidson_update);
197  DS.set_iter_max(_opt.davidson_maxiter);
198  DS.set_max_search_space(10 * _opt.nmax);
199 
200  if (_opt.matrixfree) {
201  XTP_LOG(Log::error, _log)
202  << TimeStamp() << " Using matrix free method" << flush;
203  DS.solve(h, _opt.nmax);
204  } else {
205  XTP_LOG(Log::error, _log)
206  << TimeStamp() << " Using full matrix method" << flush;
207 
208  // get the full matrix
209  hstart = std::chrono::system_clock::now();
210  Eigen::MatrixXd hfull = h.get_full_matrix();
211  hend = std::chrono::system_clock::now();
212 
213  elapsed_time = hend - hstart;
214 
215  XTP_LOG(Log::info, _log) << TimeStamp() << " Full matrix assembled in "
216  << elapsed_time.count() << " secs" << flush;
217 
218  // solve theeigenalue problem
219  hstart = std::chrono::system_clock::now();
220  DS.solve(hfull, _opt.nmax);
221  hend = std::chrono::system_clock::now();
222 
223  elapsed_time = hend - hstart;
224  XTP_LOG(Log::info, _log) << TimeStamp() << " Davidson solve done in "
225  << elapsed_time.count() << " secs" << flush;
226  }
227  result.eigenvalues() = DS.eigenvalues();
228  result.eigenvectors() = DS.eigenvectors();
229 
230  } else {
231 
232  XTP_LOG(Log::error, _log)
233  << TimeStamp() << " Lapack Diagonalization" << flush;
234 
235  hstart = std::chrono::system_clock::now();
236  Eigen::MatrixXd hfull = h.get_full_matrix();
237  hend = std::chrono::system_clock::now();
238 
239  elapsed_time = hend - hstart;
240  XTP_LOG(Log::info, _log) << TimeStamp() << " Full matrix assembled in "
241  << elapsed_time.count() << " secs" << flush;
242  hstart = std::chrono::system_clock::now();
243  result = tools::linalg_eigenvalues(hfull, _opt.nmax);
244  hend = std::chrono::system_clock::now();
245  elapsed_time = hend - hstart;
246  XTP_LOG(Log::info, _log) << TimeStamp() << " Lapack solve done in "
247  << elapsed_time.count() << " secs" << flush;
248  }
249  end = std::chrono::system_clock::now();
250  elapsed_time = end - start;
251 
252  XTP_LOG(Log::info, _log) << TimeStamp() << " Diagonalization done in "
253  << elapsed_time.count() << " secs" << flush;
254 
255  return result;
256 }

300 template <typename BSE_OPERATOR_ApB, typename BSE_OPERATOR_AmB>
301 tools::EigenSystem BSE::Solve_nonhermitian(BSE_OPERATOR_ApB& apb,
302  BSE_OPERATOR_AmB& amb) const {
303 
304  // For details of the method, see EPL,78(2007)12001,
305  // Nuclear Physics A146(1970)449, Nuclear Physics A163(1971)257.
306  // setup resonant (A) and RARC blocks (B)
307  // corresponds to
308  // _ApB = (_eh_d +_eh_qp + _eh_d2 + 4.0 * _eh_x);
309  // _AmB = (_eh_d +_eh_qp - _eh_d2);
310 
311  std::chrono::time_point<std::chrono::system_clock> start, end;
312  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
313  std::chrono::duration<double> elapsed_time;
314  start = std::chrono::system_clock::now();
315 
316  Eigen::MatrixXd ApB = apb.get_full_matrix();
317  Eigen::MatrixXd AmB = amb.get_full_matrix();
318  XTP_LOG(Log::error, _log)
319  << TimeStamp() << " Setup singlet hamiltonian " << flush;
320  XTP_LOG(Log::error, _log)
321  << TimeStamp() << " Lapack Diagonalization" << flush;
322 
323  // calculate Cholesky decomposition of A-B = LL^T. It throws an error if not
324  // positive definite
325  //(A-B) is not needed any longer and can be overwritten
326  XTP_LOG(Log::info, _log) << TimeStamp()
327  << " Trying Cholesky decomposition of KAA-KAB"
328  << flush;
329  Eigen::LLT<Eigen::Ref<Eigen::MatrixXd> > L(AmB);
330 
331  for (Index i = 0; i < AmB.rows(); ++i) {
332  for (Index j = i + 1; j < AmB.cols(); ++j) {
333  AmB(i, j) = 0;
334  }
335  }
336  if (L.info() != Eigen::ComputationInfo::Success) {
337  XTP_LOG(Log::error, _log)
338  << TimeStamp()
339  << " Cholesky decomposition of KAA-KAB was unsucessful. Try a smaller "
340  "basisset. This can indicate a triplet instability."
341  << flush;
342  throw std::runtime_error("Cholesky decompostion failed");
343  } else {
344  XTP_LOG(Log::info, _log)
345  << TimeStamp() << " Cholesky decomposition of KAA-KAB was successful"
346  << flush;
347  }
348 
349  ApB = AmB.transpose() * ApB * AmB;
350 
351  XTP_LOG(Log::info, _log) << TimeStamp() << " Calculated H = L^T(A+B)L "
352  << flush;
353 
354  XTP_LOG(Log::error, _log) << TimeStamp() << " Solving for first " << _opt.nmax
355  << " eigenvectors" << flush;
356 
357  hstart = std::chrono::system_clock::now();
358  tools::EigenSystem eigensys = tools::linalg_eigenvalues(ApB, _opt.nmax);
359  hend = std::chrono::system_clock::now();
360  elapsed_time = hend - hstart;
361  XTP_LOG(Log::info, _log) << TimeStamp() << " Lapack solve done in "
362  << elapsed_time.count() << " secs" << flush;
363 
364  if (eigensys.info() != Eigen::ComputationInfo::Success) {
365  XTP_LOG(Log::error, _log)
366  << TimeStamp() << " Could not solve problem" << flush;
367  } else {
368  XTP_LOG(Log::info, _log)
369  << TimeStamp() << " Solved HR_l = eps_l^2 R_l " << flush;
370  }
371  ApB.resize(0, 0);
372  if ((eigensys.eigenvalues().array() < 0).any()) {
373  throw std::runtime_error("Negative eigenvalues in BTDA");
374  }
375  Eigen::VectorXd energies =
376  eigensys.eigenvalues().cwiseSqrt(); // has to stay otherwise mkl
377  // complains
378 
379  tools::EigenSystem result;
380  result.eigenvalues() = energies;
381  // reconstruct real eigenvectors X_l = 1/2 [sqrt(eps_l) (L^T)^-1 +
382  // 1/sqrt(eps_l)L ] R_l
383  // Y_l = 1/2 [sqrt(eps_l) (L^T)^-1 -
384  // 1/sqrt(eps_l)L ] R_l
385  // 1/sqrt(eps_l)L ] R_l
386  // determine inverse of L^T
387  Eigen::MatrixXd LmT = AmB.inverse().transpose();
388  Index dim = LmT.rows();
389  result.eigenvectors().resize(dim, _opt.nmax); // resonant part (_X_evec)
390  result.eigenvectors2().resize(dim,
391  _opt.nmax); // anti-resonant part (_Y_evec)
392  for (Index level = 0; level < _opt.nmax; level++) {
393  double sqrt_eval = std::sqrt(energies(level));
394  // get l-th reduced EV
395  result.eigenvectors().col(level) =
396  (0.5 / sqrt_eval * (energies(level) * LmT + AmB) *
397  eigensys.eigenvectors().col(level));
398  result.eigenvectors2().col(level) =
399  (0.5 / sqrt_eval * (energies(level) * LmT - AmB) *
400  eigensys.eigenvectors().col(level));
401  }
402 
403  end = std::chrono::system_clock::now();
404  elapsed_time = end - start;
405 
406  XTP_LOG(Log::info, _log) << TimeStamp() << " Diagonalization done in "
407  << elapsed_time.count() << " secs" << flush;
408 
409  return result;
410 }

412 template <typename BSE_OPERATOR_A, typename BSE_OPERATOR_B>
413 tools::EigenSystem BSE::Solve_nonhermitian_Davidson(BSE_OPERATOR_A& Aop,
414  BSE_OPERATOR_B& Bop) const {
415 
416  std::chrono::time_point<std::chrono::system_clock> start, end;
417  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
418  std::chrono::duration<double> elapsed_time;
419  start = std::chrono::system_clock::now();
420 
421  // operator
422  HamiltonianOperator<BSE_OPERATOR_A, BSE_OPERATOR_B> Hop(Aop, Bop);
423 
424  // Davidson solver
425  DavidsonSolver DS(_log);
426  DS.set_correction(_opt.davidson_correction);
427  DS.set_tolerance(_opt.davidson_tolerance);
428  DS.set_ortho(_opt.davidson_ortho);
429  DS.set_size_update(_opt.davidson_update);
430  DS.set_iter_max(_opt.davidson_maxiter);
431  DS.set_max_search_space(10 * _opt.nmax);
432  DS.set_matrix_type("HAM");
433 
434  if (_opt.matrixfree) {
435  XTP_LOG(Log::error, _log)
436  << TimeStamp() << " Using matrix free method" << flush;
437  DS.solve(Hop, _opt.nmax);
438 
439  } else {
440  XTP_LOG(Log::error, _log)
441  << TimeStamp() << " Using full matrix method" << flush;
442  // get the full matrix
443  hstart = std::chrono::system_clock::now();
444  Eigen::MatrixXd hfull = Hop.get_full_matrix();
445  hend = std::chrono::system_clock::now();
446 
447  elapsed_time = hend - hstart;
448 
449  XTP_LOG(Log::info, _log) << TimeStamp() << " Full matrix assembled in "
450  << elapsed_time.count() << " secs" << flush;
451 
452  // solve theeigenalue problem
453  hstart = std::chrono::system_clock::now();
454  DS.solve(hfull, _opt.nmax);
455  hend = std::chrono::system_clock::now();
456 
457  elapsed_time = hend - hstart;
458  XTP_LOG(Log::info, _log) << TimeStamp() << " Davidson solve done in "
459  << elapsed_time.count() << " secs" << flush;
460  }
461 
462  // results
463  tools::EigenSystem result;
464  result.eigenvalues() = DS.eigenvalues();
465  Eigen::MatrixXd _tmpX = DS.eigenvectors().topRows(Aop.size());
466  Eigen::MatrixXd _tmpY = DS.eigenvectors().bottomRows(Aop.size());
467 
468  // // normalization so that eigenvector^2 - eigenvector2^2 = 1
469  Eigen::VectorXd normX = _tmpX.colwise().squaredNorm();
470  Eigen::VectorXd normY = _tmpY.colwise().squaredNorm();
471 
472  Eigen::ArrayXd sqinvnorm = (normX - normY).array().inverse().cwiseSqrt();
473 
474  result.eigenvectors() = _tmpX * sqinvnorm.matrix().asDiagonal();
475  result.eigenvectors2() = _tmpY * sqinvnorm.matrix().asDiagonal();
476 
477  end = std::chrono::system_clock::now();
478  elapsed_time = end - start;
479 
480  XTP_LOG(Log::info, _log) << TimeStamp() << " Diagonalization done in "
481  << elapsed_time.count() << " secs" << flush;
482 
483  return result;
484 }

517 void BSE::Analyze_singlets(std::vector<QMFragment<BSE_Population> > fragments,
518  const Orbitals& orb) const {
519 
520  Interaction act;
521  QMStateType singlet = QMStateType(QMStateType::Singlet);
522 
523  Eigen::VectorXd oscs = orb.Oscillatorstrengths();
524 
525  act = Analyze_eh_interaction(singlet, orb);
526 
527  if (fragments.size() > 0) {
528  Lowdin low;
529  low.CalcChargeperFragment(fragments, orb, singlet);
530  }
531 
532  const Eigen::VectorXd& energies = orb.BSESinglets().eigenvalues();
533 
534  double hrt2ev = tools::conv::hrt2ev;
535  XTP_LOG(Log::error, _log)
536  << " ====== singlet energies (eV) ====== " << flush;
537  for (Index i = 0; i < _opt.nmax; ++i) {
538  Eigen::VectorXd weights =
539  orb.BSESinglets().eigenvectors().col(i).cwiseAbs2();
540  if (!orb.getTDAApprox()) {
541  weights -= orb.BSESinglets().eigenvectors2().col(i).cwiseAbs2();
542  }
543 
544  double osc = oscs[i];
545  XTP_LOG(Log::error, _log)

616 template <typename BSE_OPERATOR>
617 BSE::ExpectationValues BSE::ExpectationValue_Operator(
618  const QMStateType& type, const Orbitals& orb, const BSE_OPERATOR& H) const {
619 
620  const tools::EigenSystem& BSECoefs =
621  (type == QMStateType::Singlet) ? orb.BSESinglets() : orb.BSETriplets();
622 
623  ExpectationValues expectation_values;
624 
625  expectation_values.direct_term =
626  BSECoefs.eigenvectors()
627  .cwiseProduct((H * BSECoefs.eigenvectors()))
628  .colwise()
629  .sum()
630  .transpose();
631 
632  if (!orb.getTDAApprox()) {
633  expectation_values.direct_term +=
634  BSECoefs.eigenvectors2()
635  .cwiseProduct((H * BSECoefs.eigenvectors2()))
636  .colwise()
637  .sum()
638  .transpose();
639 
640  expectation_values.cross_term =
641  2 * BSECoefs.eigenvectors2()
642  .cwiseProduct((H * BSECoefs.eigenvectors()))
643  .colwise()
644  .sum()
645  .transpose();
646  } else {
647  expectation_values.cross_term = Eigen::VectorXd::Zero(0);
648  }
649 
650  return expectation_values;
651 }

653 template <typename BSE_OPERATOR>
654 BSE::ExpectationValues BSE::ExpectationValue_Operator_State(
655  const QMStateType& type, const Orbitals& orb, const BSE_OPERATOR& H,
656  const Index state) const {
657 
658  const tools::EigenSystem& BSECoefs =
659  (type == QMStateType::Singlet) ? orb.BSESinglets() : orb.BSETriplets();
660 
661  ExpectationValues expectation_values;
662 
663  const Eigen::MatrixXd& BSECoefs_state = BSECoefs.eigenvectors().col(state);
664 
665  expectation_values.direct_term =
666  BSECoefs_state.cwiseProduct((H * BSECoefs_state))
667  .colwise()
668  .sum()
669  .transpose();
670 
671  if (!orb.getTDAApprox()) {
672  const Eigen::MatrixXd& BSECoefs2_state =
673  BSECoefs.eigenvectors2().col(state);
674 
675  expectation_values.direct_term +=
676  BSECoefs2_state.cwiseProduct((H * BSECoefs2_state))
677  .colwise()
678  .sum()
679  .transpose();
680 
681  expectation_values.cross_term =
682  2 * BSECoefs2_state.cwiseProduct((H * BSECoefs_state))
683  .colwise()
684  .sum()
685  .transpose();
686  } else {
687  expectation_values.cross_term = Eigen::VectorXd::Zero(0);
688  }
689 
690  return expectation_values;
691 }

693 // Composition of the excitation energy in terms of QP, direct (screened),
694 // and exchance contributions in the BSE
695 // Full BSE:
696 //
697 // | A* | | H K | | A |
698 // | -B* | | -K -H | | B | = A*.H.A + B*.H.B + 2A*.K.B
699 //
700 // with: H = H_qp + H_d + eta.H_x
701 // K = H_d2 + eta.H_x
702 //
703 // reports composition for FULL BSE as
704 // <FT> = A*.H_qp.A + B*.H_qp.B
705 // <Kx> = eta.(A*.H_x.A + B*.H_x.B + 2A*.H_x.B)
706 // <Kd> = A*.H_d.A + B*.H_d.B + 2A*.H_d2.B
707 BSE::Interaction BSE::Analyze_eh_interaction(const QMStateType& type,
708  const Orbitals& orb) const {
709  Interaction analysis;
710  ExpectationValues expectation_values;
711 
712  HqpOperator hqp(_epsilon_0_inv, _Mmn, _Hqp);
713  configureBSEOperator(hqp);
714  expectation_values = ExpectationValue_Operator(type, orb, hqp);
715  analysis.qp_contrib = expectation_values.direct_term;
716 
717  HdOperator hd(_epsilon_0_inv, _Mmn, _Hqp);
718  configureBSEOperator(hd);
719  expectation_values = ExpectationValue_Operator(type, orb, hd);
720  analysis.direct_contrib = expectation_values.direct_term;
721 
722  if (!orb.getTDAApprox()) {
723  Hd2Operator hd2(_epsilon_0_inv, _Mmn, _Hqp);
724  configureBSEOperator(hd2);
725  expectation_values = ExpectationValue_Operator(type, orb, hd2);
726  analysis.direct_contrib += expectation_values.cross_term;
727  }
728 
729  if (type == QMStateType::Singlet) {
730  HxOperator hx(_epsilon_0_inv, _Mmn, _Hqp);
731  configureBSEOperator(hx);
732  expectation_values = ExpectationValue_Operator(type, orb, hx);
733  analysis.exchange_contrib = 2.0 * expectation_values.direct_term;
734  if (!orb.getTDAApprox()) {
735  analysis.exchange_contrib += 2.0 * expectation_values.cross_term;
736  }
737  } else {
738  analysis.exchange_contrib = Eigen::VectorXd::Zero(0);
739  }
740 
741  return analysis;
742 }

744 // Dynamical Screening in BSE as perturbation to static excitation energies
745 // as in Phys. Rev. B 80, 241405 (2009) for the TDA case
746 void BSE::Perturbative_DynamicalScreening(const QMStateType& type,
747  Orbitals& orb) {
748 
749  const tools::EigenSystem& BSECoefs =
750  (type == QMStateType::Singlet) ? orb.BSESinglets() : orb.BSETriplets();
751 
752  const Eigen::VectorXd& RPAInputEnergies = orb.RPAInputEnergies();
753 
754  ExpectationValues expectation_values;
755 
756  // static case as reference
757  SetupDirectInteractionOperator(RPAInputEnergies, 0.0);
758  HdOperator Hd_static(_epsilon_0_inv, _Mmn, _Hqp);
759  configureBSEOperator(Hd_static);
760  expectation_values = ExpectationValue_Operator(type, orb, Hd_static);
761  Eigen::VectorXd Hd_static_contribution = expectation_values.direct_term;
762  if (!orb.getTDAApprox()) {
763  Hd2Operator Hd2_static(_epsilon_0_inv, _Mmn, _Hqp);
764  configureBSEOperator(Hd2_static);
765  expectation_values = ExpectationValue_Operator(type, orb, Hd2_static);
766  Hd_static_contribution += expectation_values.cross_term;
767  }
768 
769  const Eigen::VectorXd& BSEenergies = BSECoefs.eigenvalues();
770 
771  // initial copy of static BSE energies to dynamic
772  Eigen::VectorXd BSEenergies_dynamic = BSEenergies;
773 
774  // recalculate Hd at the various energies
775  for (Index i_exc = 0; i_exc < BSEenergies.size(); i_exc++) {
776  XTP_LOG(Log::info, _log) << "Dynamical Screening BSE, Excitation " << i_exc
777  << " static " << BSEenergies(i_exc) << flush;
778 
779  for (Index iter = 0; iter < _max_dyn_iter; iter++) {
780 
781  // setup the direct operator with the last energy as screening frequency
782  double old_energy = BSEenergies_dynamic(i_exc);
783  SetupDirectInteractionOperator(RPAInputEnergies, old_energy);
784  HdOperator Hd_dyn(_epsilon_0_inv, _Mmn, _Hqp);
785  configureBSEOperator(Hd_dyn);
786 
787  // get the contribution of Hd for the dynamic case
788  expectation_values =
789  ExpectationValue_Operator_State(type, orb, Hd_dyn, i_exc);
790  Eigen::VectorXd Hd_dynamic_contribution = expectation_values.direct_term;
791  if (!orb.getTDAApprox()) {
792  Hd2Operator Hd2_dyn(_epsilon_0_inv, _Mmn, _Hqp);
793  configureBSEOperator(Hd2_dyn);
794  expectation_values =
795  ExpectationValue_Operator_State(type, orb, Hd2_dyn, i_exc);
796  Hd_dynamic_contribution += expectation_values.cross_term;
797  }
798 
799  // new energy perturbatively
800  BSEenergies_dynamic(i_exc) = BSEenergies(i_exc) +
801  Hd_static_contribution(i_exc) -
802  Hd_dynamic_contribution(0);
803 
804  XTP_LOG(Log::info, _log)
805  << "Dynamical Screening BSE, excitation " << i_exc << " iteration "
806  << iter << " dynamic " << BSEenergies_dynamic(i_exc) << flush;
807 
808  // check tolerance
809  if (std::abs(BSEenergies_dynamic(i_exc) - old_energy) < _dyn_tolerance) {
810  break;
811  }
812  }
813  }
814 
815  double hrt2ev = tools::conv::hrt2ev;
816 
817  if (type == QMStateType::Singlet) {
818  orb.BSESinglets_dynamic() = BSEenergies_dynamic;
819  XTP_LOG(Log::error, _log) << " ====== singlet energies with perturbative "
820  "dynamical screening (eV) ====== "
821  << flush;
822  Eigen::VectorXd oscs = orb.Oscillatorstrengths();
823  for (Index i = 0; i < _opt.nmax; ++i) {
824  double osc = oscs[i];
825  XTP_LOG(Log::error, _log)
826  << format(
827  " S(dynamic) = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f "
828  "nm f "
829  "= %4$+1.4f") %
830  (i + 1) % (hrt2ev * BSEenergies_dynamic(i)) %
831  (1240.0 / (hrt2ev * BSEenergies_dynamic(i))) %
832  (osc * BSEenergies_dynamic(i) / BSEenergies(i))
833  << flush;
834  }
835 
836  } else {
837  orb.BSETriplets_dynamic() = BSEenergies_dynamic;
838  XTP_LOG(Log::error, _log) << " ====== triplet energies with perturbative "
839  "dynamical screening (eV) ====== "
840  << flush;
841  for (Index i = 0; i < _opt.nmax; ++i) {
842  XTP_LOG(Log::error, _log)
843  << format(
844  " T(dynamic) = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f "
845  "nm ") %
846  (i + 1) % (hrt2ev * BSEenergies_dynamic(i)) %
847  (1240.0 / (hrt2ev * BSEenergies_dynamic(i)))
848  << flush;
849  }
850  }
851 }
