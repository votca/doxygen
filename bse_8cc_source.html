40 void BSE::SetupDirectInteractionOperator() {
41  RPA rpa = RPA(_Mmn);
42  rpa.configure(_opt.homo, _opt.rpamin, _opt.rpamax);
43  rpa.UpdateRPAInputEnergies(_orbitals.MOEnergies(), _Hqp.diagonal(),
44  _opt.qpmin);
45 
46  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(rpa.calculate_epsilon_r(0));
47  _Mmn.MultiplyRightWithAuxMatrix(es.eigenvectors());
48 
49  _epsilon_0_inv = Eigen::VectorXd::Zero(es.eigenvalues().size());
50  for (int i = 0; i < es.eigenvalues().size(); ++i) {
51  if (es.eigenvalues()(i) > 1e-8) {
52  _epsilon_0_inv(i) = 1 / es.eigenvalues()(i);
53  }
54  }
55 }

133 template <typename BSE_OPERATOR>
134 void BSE::solve_hermitian(BSE_OPERATOR& h, Eigen::VectorXd& energies,
135  Eigen::MatrixXd& coefficients) {
136 
137  std::chrono::time_point<std::chrono::system_clock> start, end;
138  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
139  std::chrono::duration<double> elapsed_time;
140  start = std::chrono::system_clock::now();
141 
142  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " Solving for first "
143  << _opt.nmax << " eigenvectors" << flush;
144 
145  if (_opt.davidson) {
146 
147  DavidsonSolver DS(_log);
148 
149  DS.set_correction(_opt.davidson_correction);
150  DS.set_tolerance(_opt.davidson_tolerance);
151  DS.set_ortho(_opt.davidson_ortho);
152  DS.set_size_update(_opt.davidson_update);
153  DS.set_iter_max(_opt.davidson_maxiter);
154  DS.set_max_search_space(10 * _opt.nmax);
155 
156  if (_opt.matrixfree) {
157  CTP_LOG(ctp::logDEBUG, _log)
158  << ctp::TimeStamp() << " Using matrix free method" << flush;
159  DS.solve(h, _opt.nmax);
160  }
161 
162  else {
163  CTP_LOG(ctp::logDEBUG, _log)
164  << ctp::TimeStamp() << " Using full matrix method" << flush;
165 
166  // get the full matrix
167  hstart = std::chrono::system_clock::now();
168  Eigen::MatrixXd hfull = h.get_full_matrix();
169  hend = std::chrono::system_clock::now();
170 
171  elapsed_time = hend - hstart;
172 
173  CTP_LOG(ctp::logDEBUG, _log)
174  << ctp::TimeStamp() << " Full matrix assembled in "
175  << elapsed_time.count() << " secs" << flush;
176 
177  // solve theeigenalue problem
178  hstart = std::chrono::system_clock::now();
179  DS.solve(hfull, _opt.nmax);
180  hend = std::chrono::system_clock::now();
181 
182  elapsed_time = hend - hstart;
183  CTP_LOG(ctp::logDEBUG, _log)
184  << ctp::TimeStamp() << " Davidson solve done in "
185  << elapsed_time.count() << " secs" << flush;
186  }
187 
188  energies = DS.eigenvalues();
189  coefficients = DS.eigenvectors();
190 
191  }
192 
193  else {
194 
195  CTP_LOG(ctp::logDEBUG, _log)
196  << ctp::TimeStamp() << " Lapack Diagonalization" << flush;
197 
198  hstart = std::chrono::system_clock::now();
199  Eigen::MatrixXd hfull = h.get_full_matrix();
200  hend = std::chrono::system_clock::now();
201 
202  elapsed_time = hend - hstart;
203  CTP_LOG(ctp::logDEBUG, _log)
204  << ctp::TimeStamp() << " Full matrix assembled in "
205  << elapsed_time.count() << " secs" << flush;
206 
207  hstart = std::chrono::system_clock::now();
208  tools::linalg_eigenvalues(hfull, energies, coefficients, _opt.nmax);
209  hend = std::chrono::system_clock::now();
210 
211  elapsed_time = hend - hstart;
212  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " Lapack solve done in "
213  << elapsed_time.count() << " secs" << flush;
214  }
215 
216  end = std::chrono::system_clock::now();
217  elapsed_time = end - start;
218 
219  CTP_LOG(ctp::logDEBUG, _log)
220  << ctp::TimeStamp() << " Diagonalization done in " << elapsed_time.count()
221  << " secs" << flush;
222 
223  return;
224 }

249 template <typename BSE_OPERATOR_ApB, typename BSE_OPERATOR_AmB>
250 void BSE::Solve_antihermitian(BSE_OPERATOR_ApB& apb, BSE_OPERATOR_AmB& amb,
251  Eigen::VectorXd& energies,
252  Eigen::MatrixXd& coefficients,
253  Eigen::MatrixXd& coefficients_AR) {
254 
255  // For details of the method, see EPL,78(2007)12001,
256  // Nuclear Physics A146(1970)449, Nuclear Physics A163(1971)257.
257  // setup resonant (A) and RARC blocks (B)
258  // corresponds to
259  // _ApB = (_eh_d +_eh_qp + _eh_d2 + 4.0 * _eh_x);
260  // _AmB = (_eh_d +_eh_qp - _eh_d2);
261 
262  Eigen::MatrixXd ApB = apb.get_full_matrix();
263  Eigen::MatrixXd AmB = amb.get_full_matrix();
264 
265  // calculate Cholesky decomposition of A-B = LL^T. It throws an error if not
266  // positive definite
267  //(A-B) is not needed any longer and can be overwritten
268  CTP_LOG(ctp::logDEBUG, _log)
269  << ctp::TimeStamp() << " Trying Cholesky decomposition of KAA-KAB"
270  << flush;
271  Eigen::LLT<Eigen::Ref<Eigen::MatrixXd> > L(AmB);
272 
273  for (int i = 0; i < AmB.rows(); ++i) {
274  for (int j = i + 1; j < AmB.cols(); ++j) {
275  AmB(i, j) = 0;
276  }
277  }
278  if (L.info() != 0) {
279  CTP_LOG(ctp::logDEBUG, _log)
280  << ctp::TimeStamp()
281  << " Cholesky decomposition of KAA-KAB was unsucessful. Try a smaller "
282  "basisset. This can indicate a triplet instability."
283  << flush;
284  throw std::runtime_error("Cholesky decompostion failed");
285  } else {
286  CTP_LOG(ctp::logDEBUG, _log)
287  << ctp::TimeStamp()
288  << " Cholesky decomposition of KAA-KAB was successful" << flush;
289  }
290 
291  Eigen::MatrixXd temp = ApB * AmB;
292  ApB.noalias() = AmB.transpose() * temp;
293  temp.resize(0, 0);
294  CTP_LOG(ctp::logDEBUG, _log)
295  << ctp::TimeStamp() << " Calculated H = L^T(A+B)L " << flush;
296  Eigen::VectorXd eigenvalues;
297  Eigen::MatrixXd eigenvectors;
298  CTP_LOG(ctp::logDEBUG, _log) << ctp::TimeStamp() << " Solving for first "
299  << _opt.nmax << " eigenvectors" << flush;
300  bool success_diag =
301  tools::linalg_eigenvalues(ApB, eigenvalues, eigenvectors, _opt.nmax);
302  if (!success_diag) {
303  CTP_LOG(ctp::logDEBUG, _log)
304  << ctp::TimeStamp() << " Could not solve problem" << flush;
305  } else {
306  CTP_LOG(ctp::logDEBUG, _log)
307  << ctp::TimeStamp() << " Solved HR_l = eps_l^2 R_l " << flush;
308  }
309  ApB.resize(0, 0);
310  if ((eigenvalues.array() < 0).any()) {
311  throw std::runtime_error("Negative eigenvalues in BTDA");
312  }
313  Eigen::VectorXd tempvec =
314  eigenvalues.cwiseSqrt(); // has to stay otherwise mkl complains
315  energies = tempvec;
316  // reconstruct real eigenvectors X_l = 1/2 [sqrt(eps_l) (L^T)^-1 +
317  // 1/sqrt(eps_l)L ] R_l
318  // Y_l = 1/2 [sqrt(eps_l) (L^T)^-1 -
319  // 1/sqrt(eps_l)L ] R_l
320  // determine inverse of L^T
321  Eigen::MatrixXd LmT = AmB.inverse().transpose();
322  int dim = LmT.rows();
323  coefficients.resize(dim, _opt.nmax); // resonant part (_X_evec)
324  coefficients_AR.resize(dim, _opt.nmax); // anti-resonant part (_Y_evec)
325  for (int level = 0; level < _opt.nmax; level++) {
326  double sqrt_eval = std::sqrt(energies(level));
327  // get l-th reduced EV
328  coefficients.col(level) = (0.5 / sqrt_eval * (energies(level) * LmT + AmB) *
329  eigenvectors.col(level));
330  coefficients_AR.col(level) =
331  (0.5 / sqrt_eval * (energies(level) * LmT - AmB) *
332  eigenvectors.col(level));
333  }
334  return;
335 }

355 void BSE::PrintWeight(int i, int i_bse, QMStateType state) {
356 
357  const Eigen::MatrixXd& BSECoefs_AR = (state == QMStateType::Singlet)
358  ? _bse_singlet_coefficients_AR
359  : _bse_triplet_coefficients_AR;
360  const Eigen::MatrixXd& BSECoefs = (state == QMStateType::Singlet)
361  ? _bse_singlet_coefficients
362  : _bse_triplet_coefficients;
363  double weight = std::pow(BSECoefs(i_bse, i), 2);
364  if (!_opt.useTDA) {
365  weight -= std::pow(BSECoefs_AR(i_bse, i), 2);
366  }
367  vc2index vc = vc2index(_opt.vmin, _bse_cmin, _bse_ctotal);
368  if (weight > _opt.min_print_weight) {
369  CTP_LOG(ctp::logINFO, _log)
370  << format(" HOMO-%1$-3d -> LUMO+%2$-3d : %3$3.1f%%") %
371  (_opt.homo - vc.v(i_bse)) % (vc.c(i_bse) - _opt.homo - 1) %
372  (100.0 * weight)
373  << flush;
374  }
375  return;
376 }

378 void BSE::Analyze_singlets(const AOBasis& dftbasis) {
379 
380  Interaction act;
381  Population pop;
382  QMStateType singlet = QMStateType(QMStateType::Singlet);
383  std::vector<tools::vec> transition_dipoles =
384  CalcCoupledTransition_Dipoles(dftbasis);
385  _orbitals.TransitionDipoles() = transition_dipoles;
386  std::vector<double> oscs = _orbitals.Oscillatorstrengths();
387 
388  if (tools::globals::verbose) {
389  act = Analyze_eh_interaction(singlet);
390  }
391  if (dftbasis.getAOBasisFragA() > 0 && dftbasis.getAOBasisFragB() > 0) {
392  pop = FragmentPopulations(singlet, dftbasis);
393  _orbitals.setFragmentChargesSingEXC(pop.Crgs);
394  _orbitals.setFragment_E_localisation_singlet(pop.popE);
395  _orbitals.setFragment_H_localisation_singlet(pop.popH);
396  _orbitals.setFragmentChargesGS(pop.popGs);
397  }
398 
399  double hrt2ev = tools::conv::hrt2ev;
400  CTP_LOG(ctp::logINFO, _log)
401  << " ====== singlet energies (eV) ====== " << flush;
402  for (int i = 0; i < _opt.nmax; ++i) {
403  const tools::vec& trdip = transition_dipoles[i];
404  double osc = oscs[i];
405  if (tools::globals::verbose) {
406  CTP_LOG(ctp::logINFO, _log)
407  << format(
408  " S = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f nm <FT> "
409  "= %4$+1.4f <K_x> = %5$+1.4f <K_d> = %6$+1.4f") %
410  (i + 1) % (hrt2ev * _bse_singlet_energies(i)) %
411  (1240.0 / (hrt2ev * _bse_singlet_energies(i))) %
412  (hrt2ev * act.qp_contrib(i)) %
413  (hrt2ev * act.exchange_contrib(i)) %
414  (hrt2ev * act.direct_contrib(i))
415  << flush;
416  } else {
417  CTP_LOG(ctp::logINFO, _log)
418  << format(" S = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f nm") %
419  (i + 1) % (hrt2ev * _bse_singlet_energies(i)) %
420  (1240.0 / (hrt2ev * _bse_singlet_energies(i)))
421  << flush;
422  }
423  CTP_LOG(ctp::logINFO, _log)
424  << format(
425  " TrDipole length gauge[e*bohr] dx = %1$+1.4f dy = "
426  "%2$+1.4f dz = %3$+1.4f |d|^2 = %4$+1.4f f = %5$+1.4f") %
427  trdip.getX() % trdip.getY() % trdip.getZ() % (trdip * trdip) %
428  osc
429  << flush;
430  for (int i_bse = 0; i_bse < _bse_size; ++i_bse) {
431  // if contribution is larger than 0.2, print
432  PrintWeight(i, i_bse, singlet);
433  }
434  // results of fragment population analysis
435  if (dftbasis.getAOBasisFragA() > 0 && dftbasis.getAOBasisFragB() > 0) {
436  printFragInfo(pop, i);
437  }
438 
439  CTP_LOG(ctp::logINFO, _log) << flush;
440  }
441  return;
442 }

494 template <typename BSE_OPERATOR>
495 Eigen::VectorXd BSE::Analyze_IndividualContribution(const QMStateType& type,
496  const BSE_OPERATOR& H) {
497  Eigen::VectorXd contrib = Eigen::VectorXd::Zero(_opt.nmax);
498  const Eigen::MatrixXd& BSECoefs_AR = (type == QMStateType::Singlet)
499  ? _bse_singlet_coefficients_AR
500  : _bse_triplet_coefficients_AR;
501  const Eigen::MatrixXd& BSECoefs = (type == QMStateType::Singlet)
502  ? _bse_singlet_coefficients
503  : _bse_triplet_coefficients;
504 
505  for (int i_exc = 0; i_exc < _opt.nmax; i_exc++) {
506  Eigen::MatrixXd slice_R = BSECoefs.block(0, i_exc, _bse_size, 1);
507  contrib(i_exc) = (slice_R.transpose() * (H * slice_R)).value();
508  if (!_opt.useTDA) {
509  Eigen::MatrixXd slice_AR = BSECoefs_AR.block(0, i_exc, _bse_size, 1);
510  // get anti-resonant contribution from direct Keh
511  contrib(i_exc) -= (slice_AR.transpose() * (H * slice_AR)).value();
512  }
513  }
514  return contrib;
515 }

517 BSE::Interaction BSE::Analyze_eh_interaction(const QMStateType& type) {
518  Interaction analysis;
519 
520  HqpOperator hqp(_epsilon_0_inv, _Mmn, _Hqp);
521  configureBSEOperator(hqp);
522  analysis.qp_contrib = Analyze_IndividualContribution(type, hqp);
523 
524  HdOperator hd(_epsilon_0_inv, _Mmn, _Hqp);
525  configureBSEOperator(hd);
526  analysis.direct_contrib = Analyze_IndividualContribution(type, hd);
527 
528  if (type == QMStateType::Singlet) {
529  HxOperator hx(_epsilon_0_inv, _Mmn, _Hqp);
530  configureBSEOperator(hx);

539 BSE::Population BSE::FragmentPopulations(const QMStateType& type,
540  const AOBasis& dftbasis) {
541  Population pop;
542  // Mulliken fragment population analysis
543  AOOverlap dftoverlap;
544  dftoverlap.Fill(dftbasis);
545  CTP_LOG(ctp::logDEBUG, _log)
546  << ctp::TimeStamp() << " Filled DFT Overlap matrix of dimension: "
547  << dftoverlap.Matrix().rows() << flush;
548  // ground state populations
549  Eigen::MatrixXd DMAT = _orbitals.DensityMatrixGroundState();
550  Eigen::VectorXd nuccharges =
551  _orbitals.FragmentNuclearCharges(dftbasis.getAOBasisFragA());
552  Eigen::VectorXd pops = _orbitals.LoewdinPopulation(
553  DMAT, dftoverlap.Matrix(), dftbasis.getAOBasisFragA());
554  pop.popGs = nuccharges - pops;
555  // population to electron charges and add nuclear charges
556  for (int i_state = 0; i_state < _opt.nmax; i_state++) {
557  QMState state = QMState(type, i_state, false);
558  // checking Density Matrices
559  std::vector<Eigen::MatrixXd> DMAT =
560  _orbitals.DensityMatrixExcitedState(state);
561  // hole part
562  Eigen::VectorXd popsH = _orbitals.LoewdinPopulation(
563  DMAT[0], dftoverlap.Matrix(), dftbasis.getAOBasisFragA());
564  pop.popH.push_back(popsH);
565  // electron part
566  Eigen::VectorXd popsE = _orbitals.LoewdinPopulation(
567  DMAT[1], dftoverlap.Matrix(), dftbasis.getAOBasisFragA());
568  pop.popE.push_back(popsE);
569  // update effective charges
570  Eigen::VectorXd diff = popsH - popsE;
571  pop.Crgs.push_back(diff);
572  }
573  CTP_LOG(ctp::logDEBUG, _log)
574  << ctp::TimeStamp() << " Ran Excitation fragment population analysis "
575  << flush;
576 
577  return pop;
578 }

580 std::vector<Eigen::MatrixXd> BSE::CalcFreeTransition_Dipoles(
581  const AOBasis& dftbasis) {
582  const Eigen::MatrixXd& dft_orbitals = _orbitals.MOCoefficients();
583  // Testing electric dipole AOMatrix
584  AODipole dft_dipole;
585  dft_dipole.Fill(dftbasis);
586 
587  // now transition dipole elements for free interlevel transitions
588  std::vector<Eigen::MatrixXd> interlevel_dipoles;
589 
590  Eigen::MatrixXd empty =
591  dft_orbitals.block(0, _bse_cmin, dftbasis.AOBasisSize(), _bse_ctotal);
592  Eigen::MatrixXd occ =
593  dft_orbitals.block(0, _opt.vmin, dftbasis.AOBasisSize(), _bse_vtotal);
594  for (int i_comp = 0; i_comp < 3; i_comp++) {
595  interlevel_dipoles.push_back(occ.transpose() * dft_dipole.Matrix()[i_comp] *
596  empty);
597  }
598  return interlevel_dipoles;
599 }

601 std::vector<tools::vec> BSE::CalcCoupledTransition_Dipoles(
602  const AOBasis& dftbasis) {
603  std::vector<Eigen::MatrixXd> interlevel_dipoles =
604  CalcFreeTransition_Dipoles(dftbasis);
605  vc2index vc = vc2index(0, 0, _bse_ctotal);
606  std::vector<tools::vec> dipols;
607  const double sqrt2 = sqrt(2.0);
608  for (int i_exc = 0; i_exc < _opt.nmax; i_exc++) {
609  tools::vec tdipole = tools::vec(0, 0, 0);
610  for (int c = 0; c < _bse_ctotal; c++) {
611  for (int v = 0; v < _bse_vtotal; v++) {
612  int index_vc = vc.I(v, c);
613  double factor = _bse_singlet_coefficients(index_vc, i_exc);
614  if (_bse_singlet_coefficients_AR.rows() > 0) {
615  factor += _bse_singlet_coefficients_AR(index_vc, i_exc);
616  }
617  // The Transition dipole is sqrt2 bigger because of the spin, the
618  // excited state is a linear combination of 2 slater determinants, where
619  // either alpha or beta spin electron is excited
620  tdipole.x() += factor * interlevel_dipoles[0](v, c);
621  tdipole.y() += factor * interlevel_dipoles[1](v, c);
622  tdipole.z() += factor * interlevel_dipoles[2](v, c);
623  }
624  }
625 
626  dipols.push_back(-sqrt2 * tdipole); //- because electrons are negative
627  }
628  return dipols;
629 }
