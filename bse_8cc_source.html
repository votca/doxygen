39 void BSE::configure(const options& opt, const Eigen::VectorXd& DFTenergies) {
40  _opt = opt;
41  _bse_vmax = _opt.homo;
42  _bse_cmin = _opt.homo + 1;
43  _bse_vtotal = _bse_vmax - _opt.vmin + 1;
44  _bse_ctotal = _opt.cmax - _bse_cmin + 1;
45  _bse_size = _bse_vtotal * _bse_ctotal;
46  SetupDirectInteractionOperator(DFTenergies);
47 }

49 void BSE::SetupDirectInteractionOperator(const Eigen::VectorXd& DFTenergies) {
50  RPA rpa = RPA(_Mmn);
51  rpa.configure(_opt.homo, _opt.rpamin, _opt.rpamax);
52  rpa.UpdateRPAInputEnergies(DFTenergies, _Hqp.diagonal(), _opt.qpmin);
53 
54  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> es(rpa.calculate_epsilon_r(0));
55  _Mmn.MultiplyRightWithAuxMatrix(es.eigenvectors());
56 
57  _epsilon_0_inv = Eigen::VectorXd::Zero(es.eigenvalues().size());
58  for (int i = 0; i < es.eigenvalues().size(); ++i) {
59  if (es.eigenvalues()(i) > 1e-8) {
60  _epsilon_0_inv(i) = 1 / es.eigenvalues()(i);
61  }
62  }
63 }

125 template <typename BSE_OPERATOR>
126 tools::EigenSystem BSE::solve_hermitian(BSE_OPERATOR& h) const {
127 
128  std::chrono::time_point<std::chrono::system_clock> start, end;
129  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
130  std::chrono::duration<double> elapsed_time;
131  start = std::chrono::system_clock::now();
132 
133  tools::EigenSystem result;
134  if (_opt.davidson) {
135 
136  DavidsonSolver DS(_log);
137 
138  DS.set_correction(_opt.davidson_correction);
139  DS.set_tolerance(_opt.davidson_tolerance);
140  DS.set_ortho(_opt.davidson_ortho);
141  DS.set_size_update(_opt.davidson_update);
142  DS.set_iter_max(_opt.davidson_maxiter);
143  DS.set_max_search_space(10 * _opt.nmax);
144 
145  if (_opt.matrixfree) {
146  XTP_LOG_SAVE(logDEBUG, _log)
147  << TimeStamp() << " Using matrix free method" << flush;
148  DS.solve(h, _opt.nmax);
149  } else {
150  XTP_LOG_SAVE(logDEBUG, _log)
151  << TimeStamp() << " Using full matrix method" << flush;
152 
153  // get the full matrix
154  hstart = std::chrono::system_clock::now();
155  Eigen::MatrixXd hfull = h.get_full_matrix();
156  hend = std::chrono::system_clock::now();
157 
158  elapsed_time = hend - hstart;
159 
160  XTP_LOG_SAVE(logDEBUG, _log)
161  << TimeStamp() << " Full matrix assembled in " << elapsed_time.count()
162  << " secs" << flush;
163 
164  // solve theeigenalue problem
165  hstart = std::chrono::system_clock::now();
166  DS.solve(hfull, _opt.nmax);
167  hend = std::chrono::system_clock::now();
168 
169  elapsed_time = hend - hstart;
170  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Davidson solve done in "
171  << elapsed_time.count() << " secs" << flush;
172  }
173  result.eigenvalues() = DS.eigenvalues();
174  result.eigenvectors() = DS.eigenvectors();
175 
176  } else {
177 
178  XTP_LOG_SAVE(logDEBUG, _log)
179  << TimeStamp() << " Lapack Diagonalization" << flush;
180 
181  hstart = std::chrono::system_clock::now();
182  Eigen::MatrixXd hfull = h.get_full_matrix();
183  hend = std::chrono::system_clock::now();
184 
185  elapsed_time = hend - hstart;
186  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Full matrix assembled in "
187  << elapsed_time.count() << " secs" << flush;
188  hstart = std::chrono::system_clock::now();
189  result = tools::linalg_eigenvalues(hfull, _opt.nmax);
190  hend = std::chrono::system_clock::now();
191  elapsed_time = hend - hstart;
192  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Lapack solve done in "
193  << elapsed_time.count() << " secs" << flush;
194  }
195  end = std::chrono::system_clock::now();
196  elapsed_time = end - start;
197 
198  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Diagonalization done in "
199  << elapsed_time.count() << " secs" << flush;
200 
201  return result;
202 }

246 template <typename BSE_OPERATOR_ApB, typename BSE_OPERATOR_AmB>
247 tools::EigenSystem BSE::Solve_nonhermitian(BSE_OPERATOR_ApB& apb,
248  BSE_OPERATOR_AmB& amb) const {
249 
250  // For details of the method, see EPL,78(2007)12001,
251  // Nuclear Physics A146(1970)449, Nuclear Physics A163(1971)257.
252  // setup resonant (A) and RARC blocks (B)
253  // corresponds to
254  // _ApB = (_eh_d +_eh_qp + _eh_d2 + 4.0 * _eh_x);
255  // _AmB = (_eh_d +_eh_qp - _eh_d2);
256 
257  std::chrono::time_point<std::chrono::system_clock> start, end;
258  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
259  std::chrono::duration<double> elapsed_time;
260  start = std::chrono::system_clock::now();
261 
262  Eigen::MatrixXd ApB = apb.get_full_matrix();
263  Eigen::MatrixXd AmB = amb.get_full_matrix();
264  XTP_LOG_SAVE(logDEBUG, _log)
265  << TimeStamp() << " Setup singlet hamiltonian " << flush;
266  XTP_LOG_SAVE(logDEBUG, _log)
267  << TimeStamp() << " Lapack Diagonalization" << flush;
268 
269  // calculate Cholesky decomposition of A-B = LL^T. It throws an error if not
270  // positive definite
271  //(A-B) is not needed any longer and can be overwritten
272  XTP_LOG_SAVE(logDEBUG, _log)
273  << TimeStamp() << " Trying Cholesky decomposition of KAA-KAB" << flush;
274  Eigen::LLT<Eigen::Ref<Eigen::MatrixXd> > L(AmB);
275 
276  for (int i = 0; i < AmB.rows(); ++i) {
277  for (int j = i + 1; j < AmB.cols(); ++j) {
278  AmB(i, j) = 0;
279  }
280  }
281  if (L.info() != Eigen::ComputationInfo::Success) {
282  XTP_LOG_SAVE(logDEBUG, _log)
283  << TimeStamp()
284  << " Cholesky decomposition of KAA-KAB was unsucessful. Try a smaller "
285  "basisset. This can indicate a triplet instability."
286  << flush;
287  throw std::runtime_error("Cholesky decompostion failed");
288  } else {
289  XTP_LOG_SAVE(logDEBUG, _log)
290  << TimeStamp() << " Cholesky decomposition of KAA-KAB was successful"
291  << flush;
292  }
293 
294  ApB = AmB.transpose() * ApB * AmB;
295 
296  XTP_LOG_SAVE(logDEBUG, _log)
297  << TimeStamp() << " Calculated H = L^T(A+B)L " << flush;
298 
299  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Solving for first "
300  << _opt.nmax << " eigenvectors" << flush;
301 
302  hstart = std::chrono::system_clock::now();
303  tools::EigenSystem eigensys = tools::linalg_eigenvalues(ApB, _opt.nmax);
304  hend = std::chrono::system_clock::now();
305  elapsed_time = hend - hstart;
306  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Lapack solve done in "
307  << elapsed_time.count() << " secs" << flush;
308 
309  if (eigensys.info() != Eigen::ComputationInfo::Success) {
310  XTP_LOG_SAVE(logDEBUG, _log)
311  << TimeStamp() << " Could not solve problem" << flush;
312  } else {
313  XTP_LOG_SAVE(logDEBUG, _log)
314  << TimeStamp() << " Solved HR_l = eps_l^2 R_l " << flush;
315  }
316  ApB.resize(0, 0);
317  if ((eigensys.eigenvalues().array() < 0).any()) {
318  throw std::runtime_error("Negative eigenvalues in BTDA");
319  }
320  Eigen::VectorXd energies =
321  eigensys.eigenvalues().cwiseSqrt(); // has to stay otherwise mkl
322  // complains
323 
324  tools::EigenSystem result;
325  result.eigenvalues() = energies;
326  // reconstruct real eigenvectors X_l = 1/2 [sqrt(eps_l) (L^T)^-1 +
327  // 1/sqrt(eps_l)L ] R_l
328  // Y_l = 1/2 [sqrt(eps_l) (L^T)^-1 -
329  // 1/sqrt(eps_l)L ] R_l
330  // 1/sqrt(eps_l)L ] R_l
331  // determine inverse of L^T
332  Eigen::MatrixXd LmT = AmB.inverse().transpose();
333  int dim = LmT.rows();
334  result.eigenvectors().resize(dim, _opt.nmax); // resonant part (_X_evec)
335  result.eigenvectors2().resize(dim,
336  _opt.nmax); // anti-resonant part (_Y_evec)
337  for (int level = 0; level < _opt.nmax; level++) {
338  double sqrt_eval = std::sqrt(energies(level));
339  // get l-th reduced EV
340  result.eigenvectors().col(level) =
341  (0.5 / sqrt_eval * (energies(level) * LmT + AmB) *
342  eigensys.eigenvectors().col(level));
343  result.eigenvectors2().col(level) =
344  (0.5 / sqrt_eval * (energies(level) * LmT - AmB) *
345  eigensys.eigenvectors().col(level));
346  }
347 
348  end = std::chrono::system_clock::now();
349  elapsed_time = end - start;
350 
351  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Diagonalization done in "
352  << elapsed_time.count() << " secs" << flush;
353 
354  return result;
355 }

357 template <typename BSE_OPERATOR_A, typename BSE_OPERATOR_B>
358 tools::EigenSystem BSE::Solve_nonhermitian_Davidson(BSE_OPERATOR_A& Aop,
359  BSE_OPERATOR_B& Bop) const {
360 
361  std::chrono::time_point<std::chrono::system_clock> start, end;
362  std::chrono::time_point<std::chrono::system_clock> hstart, hend;
363  std::chrono::duration<double> elapsed_time;
364  start = std::chrono::system_clock::now();
365 
366  // operator
367  HamiltonianOperator<BSE_OPERATOR_A, BSE_OPERATOR_B> Hop(Aop, Bop);
368 
369  // Davidson solver
370  DavidsonSolver DS(_log);
371  DS.set_correction(_opt.davidson_correction);
372  DS.set_tolerance(_opt.davidson_tolerance);
373  DS.set_ortho("QR");
374  DS.set_size_update(_opt.davidson_update);
375  DS.set_iter_max(_opt.davidson_maxiter);
376  DS.set_max_search_space(10 * _opt.nmax);
377  DS.set_matrix_type("HAM");
378 
379  if (_opt.matrixfree) {
380  XTP_LOG_SAVE(logDEBUG, _log)
381  << TimeStamp() << " Using matrix free method" << flush;
382  DS.solve(Hop, _opt.nmax);
383 
384  } else {
385  XTP_LOG_SAVE(logDEBUG, _log)
386  << TimeStamp() << " Using full matrix method" << flush;
387  // get the full matrix
388  hstart = std::chrono::system_clock::now();
389  Eigen::MatrixXd hfull = Hop.get_full_matrix();
390  hend = std::chrono::system_clock::now();
391 
392  elapsed_time = hend - hstart;
393 
394  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Full matrix assembled in "
395  << elapsed_time.count() << " secs" << flush;
396 
397  // solve theeigenalue problem
398  hstart = std::chrono::system_clock::now();
399  DS.solve(hfull, _opt.nmax);
400  hend = std::chrono::system_clock::now();
401 
402  elapsed_time = hend - hstart;
403  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Davidson solve done in "
404  << elapsed_time.count() << " secs" << flush;
405  }
406 
407  // results
408  tools::EigenSystem result;
409  result.eigenvalues() = DS.eigenvalues();
410  Eigen::MatrixXd _tmpX = DS.eigenvectors().topRows(Aop.size());
411  Eigen::MatrixXd _tmpY = DS.eigenvectors().bottomRows(Aop.size());
412 
413  // // normalization so that eigenvector^2 - eigenvector2^2 = 1
414  Eigen::VectorXd normX = _tmpX.colwise().squaredNorm();
415  Eigen::VectorXd normY = _tmpY.colwise().squaredNorm();
416 
417  Eigen::ArrayXd sqinvnorm = (normX - normY).array().inverse().cwiseSqrt();
418 
419  result.eigenvectors() = _tmpX * sqinvnorm.matrix().asDiagonal();
420  result.eigenvectors2() = _tmpY * sqinvnorm.matrix().asDiagonal();
421 
422  end = std::chrono::system_clock::now();
423  elapsed_time = end - start;
424 
425  XTP_LOG_SAVE(logDEBUG, _log) << TimeStamp() << " Diagonalization done in "
426  << elapsed_time.count() << " secs" << flush;
427 
428  return result;
429 }

431 void BSE::printFragInfo(const std::vector<QMFragment<BSE_Population> >& frags,
432  int state) const {
433  for (const QMFragment<BSE_Population>& frag : frags) {
434  double dq = frag.value().H[state] + frag.value().E[state];
435  double qeff = dq + frag.value().Gs;
436  XTP_LOG_SAVE(logINFO, _log)
437  << format(
438  " Fragment %1$4d%% -- hole: %2$5.1f%% electron: "
439  "%3$5.1f%% dQ: %4$+5.2f Qeff: %5$+5.2f") %
440  frag.getId() % (100.0 * frag.value().H[state]) %
441  (-100.0 * frag.value().E[state]) % dq % qeff
442  << flush;
443  }
444  return;

462 void BSE::Analyze_singlets(std::vector<QMFragment<BSE_Population> > fragments,
463  const Orbitals& orb) const {
464 
465  Interaction act;
466  QMStateType singlet = QMStateType(QMStateType::Singlet);
467 
468  Eigen::VectorXd oscs = orb.Oscillatorstrengths();
469  if (tools::globals::verbose) {
470  act = Analyze_eh_interaction(singlet, orb);
471  }
472  if (fragments.size() > 0) {
473  Lowdin low;
474  low.CalcChargeperFragment(fragments, orb, singlet);
475  }
476 
477  const Eigen::VectorXd& energies = orb.BSESinglets().eigenvalues();
478 
479  double hrt2ev = tools::conv::hrt2ev;
480  XTP_LOG_SAVE(logINFO, _log)
481  << " ====== singlet energies (eV) ====== " << flush;
482  for (int i = 0; i < _opt.nmax; ++i) {
483  Eigen::VectorXd weights =
484  orb.BSESinglets().eigenvectors().col(i).cwiseAbs2();
485  if (!orb.getTDAApprox()) {
486  weights -= orb.BSESinglets().eigenvectors2().col(i).cwiseAbs2();
487  }
488 
489  double osc = oscs[i];
490  if (tools::globals::verbose) {
491  XTP_LOG_SAVE(logINFO, _log)
492  << format(
493  " S = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f nm <FT> "
494  "= %4$+1.4f <K_x> = %5$+1.4f <K_d> = %6$+1.4f") %
495  (i + 1) % (hrt2ev * energies(i)) %
496  (1240.0 / (hrt2ev * energies(i))) %
497  (hrt2ev * act.qp_contrib(i)) %
498  (hrt2ev * act.exchange_contrib(i)) %
499  (hrt2ev * act.direct_contrib(i))
500  << flush;
501  } else {
502  XTP_LOG_SAVE(logINFO, _log)
503  << format(" S = %1$4d Omega = %2$+1.12f eV lamdba = %3$+3.2f nm") %
504  (i + 1) % (hrt2ev * energies(i)) %
505  (1240.0 / (hrt2ev * energies(i)))
506  << flush;
507  }
508  const Eigen::Vector3d& trdip = orb.TransitionDipoles()[i];
509  XTP_LOG_SAVE(logINFO, _log)
510  << format(
511  " TrDipole length gauge[e*bohr] dx = %1$+1.4f dy = "
512  "%2$+1.4f dz = %3$+1.4f |d|^2 = %4$+1.4f f = %5$+1.4f") %
513  trdip[0] % trdip[1] % trdip[2] % (trdip.squaredNorm()) % osc
514  << flush;
515 
516  PrintWeights(weights);
517  if (fragments.size() > 0) {
518  printFragInfo(fragments, i);
519  }
520 
521  XTP_LOG_SAVE(logINFO, _log) << flush;
522  }
523  return;
524 }

526 void BSE::Analyze_triplets(std::vector<QMFragment<BSE_Population> > fragments,
527  const Orbitals& orb) const {
528 
529  Interaction act;
530  QMStateType triplet = QMStateType(QMStateType::Triplet);
531  if (tools::globals::verbose) {
532  act = Analyze_eh_interaction(triplet, orb);
533  }
534  if (fragments.size() > 0) {
535  Lowdin low;
536  low.CalcChargeperFragment(fragments, orb, triplet);
537  }
538 
539  const Eigen::VectorXd& energies = orb.BSETriplets().eigenvalues();
540  XTP_LOG_SAVE(logINFO, _log)

578  const QMStateType& type, const Orbitals& orb, const BSE_OPERATOR& H) const {
579

581  (type == QMStateType::Singlet) ? orb.BSESinglets() : orb.BSETriplets();
582 
583  Eigen::VectorXd contrib = BSECoefs.eigenvectors()
584  .cwiseProduct((H * BSECoefs.eigenvectors()))
585  .colwise()
586  .sum()
587  .transpose();

589  contrib -= BSECoefs.eigenvectors2()
590  .cwiseProduct((H * BSECoefs.eigenvectors2()))
591  .colwise()
592  .sum()
593  .transpose();
594  }
