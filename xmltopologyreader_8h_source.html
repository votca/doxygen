36  std::vector<std::string> tmp_vec;
37  tok.ToVector(tmp_vec);
38  if (tmp_vec.size() != 2)
39  throw std::runtime_error("Wrong number of elements in bead: " + line);
40  molname = tmp_vec[0];
41  atname = tmp_vec[1];
42  molname.erase(molname.find_last_not_of(" \n\r\t") + 1);
43  atname.erase(atname.find_last_not_of(" \n\r\t") + 1);
44  }

48 };
