37  vector<string> tmp_vec;
38  tok.ToVector(tmp_vec);
39  if (tmp_vec.size() != 2)
40  throw runtime_error("Wrong number of elements in bead: " + line);
41  molname = tmp_vec[0];
42  atname = tmp_vec[1];

45  }
