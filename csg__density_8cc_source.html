82 void CsgDensityApp::BeginEvaluate(Topology *top, Topology *top_atom) {
83 
84  matrix box = top->getBox();
85  vec a = box.getCol(0);
86  vec b = box.getCol(1);
87  vec c = box.getCol(2);
88 
89  _dist.setPeriodic(true);
90  _axis=vec(0,0,0);
91  _area=0;
92  if(_axisname=="x") {
93  _axis.setX(1);
94  _rmax = abs(a);
95  _area= abs(b^c);
96  }
97  else if(_axisname=="y") {
98  _axis.setY(1);
99  _rmax = abs(b);
100  _area= abs(a^c);
101  }
102  else if(_axisname=="z") {
103  _axis.setZ(1);
104  _rmax = abs(c);
105  _area= abs(a^b);
106  }
107  else if(_axisname=="r") {
108  _dist.setPeriodic(false);
109  _rmax = min(min(abs(a/2), abs(b/2)), abs(c/2));
110  } else {
111  throw std::runtime_error("unknown axis type");
112  }
113 
114  if(OptionsMap().count("rmax"))
115  _rmax = OptionsMap()["rmax"].as<double>();
116 
117  if(OptionsMap().count("block-length")){
118  _block_length=OptionsMap()["block-length"].as<int>();
119  } else {
120  _block_length=0;
121  }
122 
123 
124  if (_axisname=="r") {
125  if(!OptionsMap().count("ref"))
126  _ref = a/2+b/2+c/2;
127  cout << "Using referece point: " << _ref << endl;
128  }
129  else if(OptionsMap().count("ref"))
130  throw std::runtime_error("reference center can only be used in case of spherical density");
131 
132  _nbin=(int)floor(_rmax/_step);
133  _dist.Initialize(0, _rmax, _nbin);
134 
135  cout << "rmax: " << _rmax << endl;
136  cout << "axis: " << _axisname << endl;
137  cout << "Bins: " << _nbin << endl;
138  _frames=0;
139  _nblock=0;
140 }

142 void CsgDensityApp::EvalConfiguration(Topology *top, Topology *top_ref)
143 {
144  // loop over all molecules
145  bool did_something = false;
146  for(MoleculeContainer::iterator imol=top->Molecules().begin(); imol!=top->Molecules().end(); ++imol) {
147  Molecule *mol = *imol;
148  if(!wildcmp(_molname.c_str(),mol->getName().c_str()))
149  continue;
150  int N = mol->BeadCount();
151  for(int i=0; i<N; i++) {
152  Bead *b = mol->getBead(i);
153  if(!wildcmp(_filter.c_str(), b->getName().c_str()))
154  continue;
155  double r;
156  if (_axisname=="r") {
157  r = abs(top->BCShortestConnection(_ref, b->getPos()));
158  } else {
159  r = b->getPos() * _axis;
160  }
161  _dist.Process(r, b->getM());
162  did_something = true;
163  }
164  }
165  _frames++;
166  if (!did_something) throw std::runtime_error("No molecule in selection");
167  if(_block_length != 0) {
168  if((_nframes % _block_length)==0) {
169  _nblock++;
170  string suffix = string("_") + boost::lexical_cast<string>(_nblock);
171  WriteDensity(_block_length,suffix);
172  _dist.Clear();
173  }
174  }
175 }

178 // output everything when processing frames is done
179 void CsgDensityApp::WriteDensity(int nframes, const string &suffix)
180 {
181  if (_axisname=="r") {
182  _dist.data().y() = _scale/(nframes*_rmax/(double)_nbin *4*M_PI) * element_div( _dist.data().y(),
183  element_prod(_dist.data().x(), _dist.data().x()));
184  } else {
185  _dist.data().y() = _scale/((double)nframes * _area * _rmax/ (double)_nbin ) *_dist.data().y();
186  }
187  _dist.data().Save(_out + suffix);
188 }
