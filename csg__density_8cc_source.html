77 void CsgDensityApp::BeginEvaluate(Topology *top, Topology *) {
78 
79  Eigen::Matrix3d box = top->getBox();
80  Eigen::Vector3d a = box.col(0);
81  Eigen::Vector3d b = box.col(1);
82  Eigen::Vector3d c = box.col(2);
83 
84  _dist.setPeriodic(true);
85  _axis = Eigen::Vector3d::Zero();
86  _area = 0;
87  if (_axisname == "x") {
88  _axis = Eigen::Vector3d::UnitX();
89  _rmax = a.norm();
90  _area = b.cross(c).norm();
91  } else if (_axisname == "y") {
92  _axis = Eigen::Vector3d::UnitY();
93  _rmax = b.norm();
94  _area = a.cross(c).norm();
95  } else if (_axisname == "z") {
96  _axis = Eigen::Vector3d::UnitZ();
97  _rmax = c.norm();
98  _area = a.cross(b).norm();
99  } else if (_axisname == "r") {
100  _dist.setPeriodic(false);
101  _rmax = min(min((a / 2).norm(), (b / 2).norm()), (c / 2).norm());
102  } else {
103  throw std::runtime_error("unknown axis type");
104  }
105 
106  if (OptionsMap().count("rmax")) {
107  _rmax = OptionsMap()["rmax"].as<double>();
108  }
109 
110  if (OptionsMap().count("block-length")) {
111  _block_length = OptionsMap()["block-length"].as<votca::Index>();
112  } else {
113  _block_length = 0;
114  }
115 
116  if (_axisname == "r") {
117  if (!OptionsMap().count("ref")) {
118  _ref = a / 2 + b / 2 + c / 2;
119  }
120  cout << "Using referece point: " << _ref << endl;
121  } else if (OptionsMap().count("ref")) {
122  throw std::runtime_error(
123  "reference center can only be used in case of spherical density");
124  }
125 
126  _nbin = (votca::Index)floor(_rmax / _step);
127  _dist.Initialize(0, _rmax, _nbin);
128 
129  cout << "rmax: " << _rmax << endl;
130  cout << "axis: " << _axisname << endl;
131  cout << "Bins: " << _nbin << endl;
132  _frames = 0;
133  _nblock = 0;
134 }

136 void CsgDensityApp::EvalConfiguration(Topology *top, Topology *) {
137  // loop over all molecules
138  bool did_something = false;
139  for (Molecule *mol : top->Molecules()) {
140  if (!votca::tools::wildcmp(_molname, mol->getName())) {
141  continue;
142  }
143  votca::Index N = mol->BeadCount();
144  for (votca::Index i = 0; i < N; i++) {
145  Bead *b = mol->getBead(i);
146  if (!votca::tools::wildcmp(_filter, b->getName())) {
147  continue;
148  }
149  double r;
150  if (_axisname == "r") {
151  r = (top->BCShortestConnection(_ref, b->getPos()).norm());
152  } else {
153  r = b->getPos().dot(_axis);
154  }
155  if (_dens_type == "mass") {
156  _dist.Process(r, b->getMass());
157  } else {
158  _dist.Process(r, 1.0);
159  }
160  did_something = true;
161  }
162  }
163  _frames++;
164  if (!did_something) {
165  throw std::runtime_error("No molecule in selection");
166  }
167  if (_block_length != 0) {
168  if ((_nframes % _block_length) == 0) {
169  _nblock++;
170  string suffix = string("_") + boost::lexical_cast<string>(_nblock);
171  WriteDensity(_block_length, suffix);
172  _dist.Clear();
173  }
174  }
175 }

177 // output everything when processing frames is done
178 void CsgDensityApp::WriteDensity(votca::Index nframes, const string &suffix) {
179  if (_axisname == "r") {
180  _dist.data().y() =
181  _scale /
182  (double(nframes) * _rmax / (double)_nbin * 4 * votca::tools::conv::Pi) *
183  _dist.data().y().cwiseQuotient(_dist.data().x().cwiseAbs2());
184 
185  } else {
186  _dist.data().y() = _scale /
187  ((double)nframes * _area * _rmax / (double)_nbin) *
188  _dist.data().y();
189  }
190  _dist.data().Save(_out + suffix);
191 }

193 namespace Eigen {
194 std::istream &operator>>(std::istream &in, Vector3d &v) {
195  char c;
196  in.get(c);
197  if (c != '[') {
198  throw std::runtime_error("error, invalid character in vector string");
199  }
200 
201  std::string str;
202  while (in.good()) {
203  in.get(c);
204  if (c == ']') { // found end of vector
205  votca::tools::Tokenizer tok(str, ",");
206  std::vector<double> d;
207  tok.ConvertToVector(d);
208  if (d.size() != 3) {
209  throw std::runtime_error("error, invalid number of entries in vector");
210  }
211  v.x() = d[0];
212  v.y() = d[1];
213  v.z() = d[2];
214  return in;
215  }
216  str += c;
217  }
218  throw std::runtime_error(
219  "did not find closing bracket in string to vec conversion");
220 
221  return in;
222 }
