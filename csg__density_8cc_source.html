00083 void CsgDensityApp::BeginEvaluate(Topology *top, Topology *top_atom) {
00084 
00085      matrix box = top->getBox();
00086      vec a = box.getCol(0);
00087      vec b = box.getCol(1);
00088      vec c = box.getCol(2);
00089     
00090     _dist.setPeriodic(true);
00091     _axis=vec(0,0,0);
00092     _area=0;
00093     if(_axisname=="x") {
00094       _axis.setX(1);
00095       _rmax = abs(a);
00096       _area= abs(b^c);
00097     }
00098     else if(_axisname=="y") {
00099       _axis.setY(1);
00100        _rmax = abs(b);
00101       _area= abs(a^c);
00102     }
00103     else if(_axisname=="z") {
00104       _axis.setZ(1);
00105       _rmax = abs(c);
00106       _area= abs(a^b);
00107     }
00108     else if(_axisname=="r") {
00109       _dist.setPeriodic(false);
00110       _rmax = min(min(abs(a/2), abs(b/2)), abs(c/2));
00111     } else {
00112       throw std::runtime_error("unknown axis type");
00113     }
00114 
00115     if(OptionsMap().count("rmax"))
00116       _rmax = OptionsMap()["rmax"].as<double>();
00117     
00118     if(OptionsMap().count("block-length")){
00119        _block_length=OptionsMap()["block-length"].as<int>();
00120     } else {
00121       _block_length=0;
00122     }
00123 
00124 
00125     if (_axisname=="r") {
00126       if(!OptionsMap().count("ref"))
00127          _ref = a/2+b/2+c/2;
00128        cout << "Using referece point: " << _ref << endl;
00129     } 
00130     else if(OptionsMap().count("ref"))
00131         throw std::runtime_error("reference center can only be used in case of spherical density");
00132     
00133     _nbin=(int)floor(_rmax/_step);
00134     _dist.Initialize(0, _rmax, _nbin);
00135     
00136     cout << "rmax: " << _rmax << endl;

00143 void CsgDensityApp::EvalConfiguration(Topology *top, Topology *top_ref)
00144 {
00145     // loop over all molecules
00146     bool did_something = false;
00147     for(MoleculeContainer::iterator imol=top->Molecules().begin(); imol!=top->Molecules().end(); ++imol) {
00148         Molecule *mol = *imol;
00149         if(!wildcmp(_molname.c_str(),mol->getName().c_str())) 
00150             continue;
00151         int N = mol->BeadCount();
00152             for(int i=0; i<N; i++) {
00153                 Bead *b = mol->getBead(i);        
00154                 if(!wildcmp(_filter.c_str(), b->getName().c_str()))
00155                     continue;
00156                 double r;
00157                 if (_axisname=="r") {
00158                     r = abs(top->BCShortestConnection(_ref, b->getPos()));
00159                 } else {
00160                     r = b->getPos() *  _axis;
00161                 }
00162                 if (_dens_type=="mass") {
00163                   _dist.Process(r, b->getM());
00164                 } else {
00165                   _dist.Process(r, 1.0);
00166                 }
00167                 did_something = true;
00168             }
00169     }
00170     _frames++;
00171     if (!did_something) throw std::runtime_error("No molecule in selection");
00172     if(_block_length != 0) {
00173         if((_nframes % _block_length)==0) {
00174             _nblock++;
00175             string suffix = string("_") + boost::lexical_cast<string>(_nblock);
00176             WriteDensity(_block_length,suffix);
00177             _dist.Clear();
00178         }
00179     }
00180 }

00184 void CsgDensityApp::WriteDensity(int nframes, const string &suffix)
00185 {
00186   if (_axisname=="r") {
00187     _dist.data().y() = _scale/(nframes*_rmax/(double)_nbin *4*M_PI) * element_div( _dist.data().y(),
00188                    element_prod(_dist.data().x(), _dist.data().x()));
00189   } else {
00190     _dist.data().y() = _scale/((double)nframes * _area * _rmax/ (double)_nbin ) *_dist.data().y();
00191   }
00192   _dist.data().Save(_out + suffix);    
00193 }
