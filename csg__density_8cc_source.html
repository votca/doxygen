77 void CsgDensityApp::BeginEvaluate(Topology *top, Topology *top_atom) {
78 
79  Eigen::Matrix3d box = top->getBox();
80  Eigen::Vector3d a = box.col(0);
81  Eigen::Vector3d b = box.col(1);
82  Eigen::Vector3d c = box.col(2);
83 
84  _dist.setPeriodic(true);
85  _axis = Eigen::Vector3d::Zero();
86  _area = 0;
87  if (_axisname == "x") {
88  _axis = Eigen::Vector3d::UnitX();
89  _rmax = a.norm();
90  _area = b.cross(c).norm();
91  } else if (_axisname == "y") {
92  _axis = Eigen::Vector3d::UnitY();
93  _rmax = b.norm();
94  _area = a.cross(c).norm();
95  } else if (_axisname == "z") {
96  _axis = Eigen::Vector3d::UnitZ();
97  _rmax = c.norm();
98  _area = a.cross(b).norm();
99  } else if (_axisname == "r") {
100  _dist.setPeriodic(false);
101  _rmax = min(min((a / 2).norm(), (b / 2).norm()), (c / 2).norm());
102  } else {
103  throw std::runtime_error("unknown axis type");
104  }
105 
106  if (OptionsMap().count("rmax")) _rmax = OptionsMap()["rmax"].as<double>();
107 
108  if (OptionsMap().count("block-length")) {
109  _block_length = OptionsMap()["block-length"].as<int>();
110  } else {
111  _block_length = 0;
112  }
113 
114  if (_axisname == "r") {
115  if (!OptionsMap().count("ref")) _ref = a / 2 + b / 2 + c / 2;
116  cout << "Using referece point: " << _ref << endl;
117  } else if (OptionsMap().count("ref"))
118  throw std::runtime_error(
119  "reference center can only be used in case of spherical density");
120 
121  _nbin = (int)floor(_rmax / _step);
122  _dist.Initialize(0, _rmax, _nbin);
123 
124  cout << "rmax: " << _rmax << endl;
125  cout << "axis: " << _axisname << endl;
126  cout << "Bins: " << _nbin << endl;
127  _frames = 0;
128  _nblock = 0;
129 }

131 void CsgDensityApp::EvalConfiguration(Topology *top, Topology *top_ref) {
132  // loop over all molecules
133  bool did_something = false;
134  for (MoleculeContainer::iterator imol = top->Molecules().begin();
135  imol != top->Molecules().end(); ++imol) {
136  Molecule *mol = *imol;
137  if (!votca::tools::wildcmp(_molname.c_str(), mol->getName().c_str()))
138  continue;
139  int N = mol->BeadCount();
140  for (int i = 0; i < N; i++) {
141  Bead *b = mol->getBead(i);
142  if (!votca::tools::wildcmp(_filter.c_str(), b->getName().c_str()))
143  continue;
144  double r;
145  if (_axisname == "r") {
146  r = (top->BCShortestConnection(_ref, b->getPos()).norm());
147  } else {
148  r = b->getPos().dot(_axis);
149  }
150  if (_dens_type == "mass") {
151  _dist.Process(r, b->getMass());
152  } else {
153  _dist.Process(r, 1.0);
154  }
155  did_something = true;
156  }
157  }
158  _frames++;
159  if (!did_something) throw std::runtime_error("No molecule in selection");
160  if (_block_length != 0) {
161  if ((_nframes % _block_length) == 0) {
162  _nblock++;
163  string suffix = string("_") + boost::lexical_cast<string>(_nblock);
164  WriteDensity(_block_length, suffix);
165  _dist.Clear();
166  }
167  }
168 }

170 // output everything when processing frames is done
171 void CsgDensityApp::WriteDensity(int nframes, const string &suffix) {
172  if (_axisname == "r") {
173  _dist.data().y() =
174  _scale /
175  (nframes * _rmax / (double)_nbin * 4 * votca::tools::conv::Pi) *
176  _dist.data().y().cwiseQuotient(_dist.data().x().cwiseAbs2());
177 
178  } else {
179  _dist.data().y() = _scale /
180  ((double)nframes * _area * _rmax / (double)_nbin) *
181  _dist.data().y();
182  }
183  _dist.data().Save(_out + suffix);
184 }

186 namespace Eigen {
187 std::istream &operator>>(std::istream &in, Vector3d &v) {
188  char c;
189  in.get(c);
190  if (c != '[') {
191  throw std::runtime_error("error, invalid character in vector string");
192  }
193 
194  std::string str;
195  while (in.good()) {
196  in.get(c);
197  if (c == ']') { // found end of vector
198  votca::tools::Tokenizer tok(str, ",");
199  std::vector<double> d;
200  tok.ConvertToVector(d);
201  if (d.size() != 3)
202  throw std::runtime_error("error, invalid number of entries in vector");
203  v.x() = d[0];
204  v.y() = d[1];
205  v.z() = d[2];
206  return in;
207  }
208  str += c;
209  }
210  throw std::runtime_error(
211  "did not find closing bracket in string to vec conversion");
212 
213  return in;
214 }
