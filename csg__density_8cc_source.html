83 void CsgDensityApp::BeginEvaluate(Topology *top, Topology *top_atom) {
84 
85  matrix box = top->getBox();
86  vec a = box.getCol(0);
87  vec b = box.getCol(1);
88  vec c = box.getCol(2);
89 
90  _dist.setPeriodic(true);
91  _axis=vec(0,0,0);
92  _area=0;
93  if(_axisname=="x") {
94  _axis.setX(1);
95  _rmax = abs(a);
96  _area= abs(b^c);
97  }
98  else if(_axisname=="y") {
99  _axis.setY(1);
100  _rmax = abs(b);
101  _area= abs(a^c);
102  }
103  else if(_axisname=="z") {
104  _axis.setZ(1);
105  _rmax = abs(c);
106  _area= abs(a^b);
107  }
108  else if(_axisname=="r") {
109  _dist.setPeriodic(false);
110  _rmax = min(min(abs(a/2), abs(b/2)), abs(c/2));
111  } else {
112  throw std::runtime_error("unknown axis type");
113  }
114 
115  if(OptionsMap().count("rmax"))
116  _rmax = OptionsMap()["rmax"].as<double>();
117 
118  if(OptionsMap().count("block-length")){
119  _block_length=OptionsMap()["block-length"].as<int>();
120  } else {
121  _block_length=0;
122  }
123 
124 
125  if (_axisname=="r") {
126  if(!OptionsMap().count("ref"))
127  _ref = a/2+b/2+c/2;
128  cout << "Using referece point: " << _ref << endl;
129  }
130  else if(OptionsMap().count("ref"))
131  throw std::runtime_error("reference center can only be used in case of spherical density");
132 
133  _nbin=(int)floor(_rmax/_step);
134  _dist.Initialize(0, _rmax, _nbin);
135 
136  cout << "rmax: " << _rmax << endl;
137  cout << "axis: " << _axisname << endl;
138  cout << "Bins: " << _nbin << endl;
139  _frames=0;
140  _nblock=0;
141 }

143 void CsgDensityApp::EvalConfiguration(Topology *top, Topology *top_ref)
144 {
145  // loop over all molecules
146  bool did_something = false;
147  for(MoleculeContainer::iterator imol=top->Molecules().begin(); imol!=top->Molecules().end(); ++imol) {
148  Molecule *mol = *imol;
149  if(!wildcmp(_molname.c_str(),mol->getName().c_str()))
150  continue;
151  int N = mol->BeadCount();
152  for(int i=0; i<N; i++) {
153  Bead *b = mol->getBead(i);
154  if(!wildcmp(_filter.c_str(), b->getName().c_str()))
155  continue;
156  double r;
157  if (_axisname=="r") {
158  r = abs(top->BCShortestConnection(_ref, b->getPos()));
159  } else {
160  r = b->getPos() * _axis;
161  }
162  if (_dens_type=="mass") {
163  _dist.Process(r, b->getM());
164  } else {
165  _dist.Process(r, 1.0);
166  }
167  did_something = true;
168  }
169  }
170  _frames++;
171  if (!did_something) throw std::runtime_error("No molecule in selection");
172  if(_block_length != 0) {
173  if((_nframes % _block_length)==0) {
174  _nblock++;
175  string suffix = string("_") + boost::lexical_cast<string>(_nblock);
176  WriteDensity(_block_length,suffix);
177  _dist.Clear();
178  }
179  }
180 }

183 // output everything when processing frames is done
184 void CsgDensityApp::WriteDensity(int nframes, const string &suffix)
185 {
186  if (_axisname=="r") {
187  _dist.data().y() = _scale/(nframes*_rmax/(double)_nbin *4*M_PI) * element_div( _dist.data().y(),
188  element_prod(_dist.data().x(), _dist.data().x()));
189  } else {
190  _dist.data().y() = _scale/((double)nframes * _area * _rmax/ (double)_nbin ) *_dist.data().y();
191  }
192  _dist.data().Save(_out + suffix);
193 }
