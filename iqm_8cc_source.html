44 void IQM::ParseSpecificOptions(const tools::Property& options) {
45 
46  QMPackageFactory::RegisterAll();
47 
48  // job tasks
49  std::string tasks_string = options.get(".tasks").as<std::string>();
50 
51  // We split either on a space or a comma
52  tools::Tokenizer tokenizedTasks(tasks_string, " ,");
53  std::vector<std::string> tasks = tokenizedTasks.ToVector();
54 
55  // Check which tasks exist and set them to true
56  do_dft_input_ = std::find(tasks.begin(), tasks.end(), "input") != tasks.end();
57  do_dft_run_ = std::find(tasks.begin(), tasks.end(), "dft") != tasks.end();
58  do_dft_parse_ = std::find(tasks.begin(), tasks.end(), "parse") != tasks.end();
59  do_dftcoupling_ =
60  std::find(tasks.begin(), tasks.end(), "dftcoupling") != tasks.end();
61  do_gwbse_ = std::find(tasks.begin(), tasks.end(), "gwbse") != tasks.end();
62  do_bsecoupling_ =
63  std::find(tasks.begin(), tasks.end(), "bsecoupling") != tasks.end();
64 
65  // storage options
66  std::string store_string = options.get(".store").as<std::string>();
67  if (store_string.find("dft") != std::string::npos) {
68  store_dft_ = true;
69  }
70  if (store_string.find("gw") != std::string::npos) {
71  store_gw_ = true;
72  }
73 
74  dftpackage_options_ = options.get(".dftpackage");
75  gwbse_options_ = options.get("gwbse");
76  dftcoupling_options_ = options.get(".dftcoupling");
77  bsecoupling_options_ = options.get("bsecoupling");
78 
79  // read linker groups
80  std::string linker =
81  options.ifExistsReturnElseReturnDefault<std::string>(".linker_names", "");
82 
83  for (const std::string& link :
84  tools::Tokenizer(linker, ", \t\n").ToVector()) {
85  tools::Tokenizer toker2(link, ":");
86  std::vector<std::string> link_split = toker2.ToVector();
87  if (link_split.size() != 2) {
88  throw std::runtime_error(
89  "Linker molecule has to be defined NAME:STATEGEO .e.g. DCV5T:n");
90  }
91  linkers_[link_split[0]] = QMState(link_split[1]);
92  }
93 
94  // options for parsing data into state file
95  std::string key_read = ".readjobfile";
96  if (options.exists(key_read + ".singlet")) {
97  std::string parse_string_s =
98  options.get(key_read + ".singlet").as<std::string>();
99  singlet_levels_ = FillParseMaps(parse_string_s);
100  }
101  if (options.exists(key_read + ".triplet")) {
102  std::string parse_string_t =
103  options.get(key_read + ".triplet").as<std::string>();
104  triplet_levels_ = FillParseMaps(parse_string_t);
105  }
106 
107  if (options.exists(key_read + ".hole")) {
108  std::string parse_string_h =
109  options.get(key_read + ".hole").as<std::string>();
110  hole_levels_ = FillParseMaps(parse_string_h);
111  }
112  if (options.exists(key_read + ".electron")) {
113  std::string parse_string_e =
114  options.get(key_read + ".electron").as<std::string>();
115  electron_levels_ = FillParseMaps(parse_string_e);
116  }
117 }

119 std::map<std::string, QMState> IQM::FillParseMaps(
120  const std::string& Mapstring) {
121  std::map<std::string, QMState> type2level;
122  for (const std::string& substring : tools::Tokenizer(Mapstring, ", \t\n")) {
123  std::vector<std::string> segmentpnumber =
124  tools::Tokenizer(substring, ":").ToVector();
125  if (segmentpnumber.size() != 2) {
126  throw std::runtime_error("Parser iqm: Segment and exciton labels:" +
127  substring + "are not separated properly");
128  }
129  QMState state = QMState(segmentpnumber[1]);
130  std::string segmentname = segmentpnumber[0];
131  type2level[segmentname] = state;
132  }
133  return type2level;
134 }

136 void IQM::addLinkers(std::vector<const Segment*>& segments,
137  const Topology& top) {
138  const Segment* seg1 = segments[0];
139  const Segment* seg2 = segments[1];
140  std::vector<const Segment*> segmentsInMolecule =
141  top.FindAllSegmentsOnMolecule(*seg1, *seg2);
142 
143  for (const Segment* segment : segmentsInMolecule) {
144  Index idIterator = segment->getId();
145  if (idIterator != seg1->getId() && idIterator != seg2->getId() &&
146  isLinker(segment->getType())) {
147  segments.push_back(segment);
148  }
149  }

174 Job::JobResult IQM::EvalJob(const Topology& top, Job& job, QMThread& opThread) {
175 
176  // report back to the progress observer
177  Job::JobResult jres = Job::JobResult();
178 
179  std::string iqm_work_dir = "OR_FILES";
180  std::string eqm_work_dir = "OR_FILES";
181  std::string frame_dir =
182  "frame_" + boost::lexical_cast<std::string>(top.getStep());
183 
184  Logger& pLog = opThread.getLogger();
185 
186  QMMapper mapper(pLog);
187  mapper.LoadMappingFile(mapfile_);
188 
189  // get the information about the job executed by the thread
190  Index job_ID = job.getId();
191  tools::Property job_input = job.getInput();
192  std::vector<tools::Property*> segment_list = job_input.Select("segment");
193  Index ID_A = segment_list.front()->getAttribute<Index>("id");
194  std::string type_A = segment_list.front()->getAttribute<std::string>("type");
195  Index ID_B = segment_list.back()->getAttribute<Index>("id");
196  std::string type_B = segment_list.back()->getAttribute<std::string>("type");
197 
198  std::string qmgeo_state_A = "n";
199  if (segment_list.front()->exists("qm_geometry")) {
200  qmgeo_state_A =
201  segment_list.front()->getAttribute<std::string>("qm_geometry");
202  }
203 
204  std::string qmgeo_state_B = "n";
205  if (segment_list.back()->exists("qm_geometry")) {
206  qmgeo_state_B =
207  segment_list.back()->getAttribute<std::string>("qm_geometry");
208  }
209  QMState stateA(qmgeo_state_A);
210  QMState stateB(qmgeo_state_B);
211 
212  // set the folders
213  std::string pair_dir =
214  (format("%1%%2%%3%%4%%5%") % "pair" % "_" % ID_A % "_" % ID_B).str();
215 
216  boost::filesystem::path arg_path, arg_pathA, arg_pathB, arg_pathAB;
217 
218  std::string orbFileA =
219  (arg_pathA / eqm_work_dir / "molecules" / frame_dir /
220  (format("%1%_%2%%3%") % "molecule" % ID_A % ".orb").str())
221  .generic_string();
222  ;
223  std::string orbFileB =
224  (arg_pathB / eqm_work_dir / "molecules" / frame_dir /
225  (format("%1%_%2%%3%") % "molecule" % ID_B % ".orb").str())
226  .generic_string();
227  ;
228  std::string orbFileAB =
229  (arg_pathAB / iqm_work_dir / "pairs_iqm" / frame_dir /
230  (format("%1%%2%%3%%4%%5%") % "pair_" % ID_A % "_" % ID_B % ".orb").str())
231  .generic_string();
232  ;
233  std::string orb_dir =
234  (arg_path / iqm_work_dir / "pairs_iqm" / frame_dir).generic_string();
235  ;
236 
237  const Segment& seg_A = top.getSegment(ID_A);
238  const Segment& seg_B = top.getSegment(ID_B);
239  const QMNBList& nblist = top.NBList();
240  const QMPair* pair = nblist.FindPair(&seg_A, &seg_B);
241 
242  XTP_LOG(Log::error, pLog)
243  << TimeStamp() << " Evaluating pair " << job_ID << " [" << ID_A << ":"
244  << ID_B << "] out of " << (top.NBList()).size() << std::flush;
245 
246  std::string package_append = "workdir_" + Identify();
247  std::vector<const Segment*> segments;
248  segments.push_back(&seg_A);
249  segments.push_back(&seg_B);
250  std::string work_dir =
251  (arg_path / iqm_work_dir / package_append / frame_dir / pair_dir)
252  .generic_string();
253 
254  if (linkers_.size() > 0) {
255  addLinkers(segments, top);
256  }
257  Orbitals orbitalsAB;
258  // if a pair object is available and is not linked take into account PBC,
259  // otherwise write as is
260  if (pair == nullptr || segments.size() > 2) {
261  if (pair == nullptr) {
262  XTP_LOG(Log::warning, pLog)
263  << "PBCs are not taken into account when writing the coordinate file!"
264  << std::flush;
265  }
266 
267  orbitalsAB.QMAtoms() = mapper.map(*(segments[0]), stateA);
268  orbitalsAB.QMAtoms().AddContainer(mapper.map(*(segments[1]), stateB));
269 
270  for (Index i = 2; i < Index(segments.size()); i++) {
271  QMState linker_state = linkers_.at(segments[i]->getType());
272  orbitalsAB.QMAtoms().AddContainer(
273  mapper.map(*(segments[i]), linker_state));
274  }
275 
276  } else {
277  const Segment* seg1 = pair->Seg1();
278  orbitalsAB.QMAtoms() = mapper.map(*seg1, stateA);
279  Segment seg2 = pair->Seg2PbCopy();
280  orbitalsAB.QMAtoms().AddContainer(mapper.map(seg2, stateB));
281  }
282 
283  if (do_dft_input_ || do_dft_run_ || do_dft_parse_) {
284  std::string qmpackage_work_dir =
285  (arg_path / iqm_work_dir / package_append / frame_dir / pair_dir)
286  .generic_string();
287 
288  Logger dft_logger(Log::current_level);
289  dft_logger.setMultithreading(false);
290  dft_logger.setPreface(Log::info, (format("\nDFT INF ...")).str());
291  dft_logger.setPreface(Log::error, (format("\nDFT ERR ...")).str());
292  dft_logger.setPreface(Log::warning, (format("\nDFT WAR ...")).str());
293  dft_logger.setPreface(Log::debug, (format("\nDFT DBG ...")).str());
294  std::string package = dftpackage_options_.get("name").as<std::string>();
295  std::unique_ptr<QMPackage> qmpackage =
296  QMPackageFactory::QMPackages().Create(package);
297  qmpackage->setLog(&dft_logger);
298  qmpackage->setRunDir(qmpackage_work_dir);
299  qmpackage->Initialize(dftpackage_options_);
300 
301  // if asked, prepare the input files
302  if (do_dft_input_) {
303  boost::filesystem::create_directories(qmpackage_work_dir);
304  if (qmpackage->GuessRequested()) {
305  if (linkers_.size() > 0) {
306  throw std::runtime_error(
307  "Error: You are using a linker and want "
308  "to use a monomer guess for the dimer. These are mutually "
309  "exclusive.");
310  }
311 
312  XTP_LOG(Log::error, pLog)
313  << "Guess requested, reading molecular orbitals" << std::flush;
314 
315  if (qmpackage->getPackageName() == "orca") {
316  XTP_LOG(Log::info, pLog)
317  << "Copying monomer .gbw files to pair folder" << std::flush;
318  std::string gbwFileA =
319  (arg_pathA / eqm_work_dir / "molecules" / frame_dir /
320  (format("%1%_%2%%3%") % "molecule" % ID_A % ".gbw").str())
321  .generic_string();
322  ;
323  std::string gbwFileB =
324  (arg_pathB / eqm_work_dir / "molecules" / frame_dir /
325  (format("%1%_%2%%3%") % "molecule" % ID_B % ".gbw").str())
326  .generic_string();
327  ;
328  std::string gbwFileA_workdir =
329  (boost::filesystem::path(qmpackage_work_dir) / "molA.gbw")
330  .generic_string();
331  ;
332  std::string gbwFileB_workdir =
333  (boost::filesystem::path(qmpackage_work_dir) / "molB.gbw")
334  .generic_string();
335  ;
336  boost::filesystem::copy_file(
337  gbwFileA, gbwFileA_workdir,
338  boost::filesystem::copy_option::overwrite_if_exists);
339  boost::filesystem::copy_file(
340  gbwFileB, gbwFileB_workdir,
341  boost::filesystem::copy_option::overwrite_if_exists);
342  } else {
343  Orbitals orbitalsB;
344  Orbitals orbitalsA;
345 
346  try {
347  XTP_LOG(Log::error, pLog)
348  << "Reading MoleculeA from " << orbFileA << std::flush;
349  orbitalsA.ReadFromCpt(orbFileA);
350  } catch (std::runtime_error&) {
351  SetJobToFailed(
352  jres, pLog,
353  "Do input: failed loading orbitals from " + orbFileA);
354  return jres;
355  }
356 
357  try {
358  XTP_LOG(Log::error, pLog)
359  << "Reading MoleculeB from " << orbFileB << std::flush;
360  orbitalsB.ReadFromCpt(orbFileB);
361  } catch (std::runtime_error&) {
362  SetJobToFailed(
363  jres, pLog,
364  "Do input: failed loading orbitals from " + orbFileB);
365  return jres;
366  }
367  XTP_LOG(Log::info, pLog)
368  << "Constructing the guess for dimer orbitals" << std::flush;
369  orbitalsAB.PrepareDimerGuess(orbitalsA, orbitalsB);
370  }
371  } else {
372  XTP_LOG(Log::info, pLog)
373  << "No Guess requested, starting from DFT starting Guess"
374  << std::flush;
375  }
376  qmpackage->WriteInputFile(orbitalsAB);
377  }
378 
379  if (do_dft_run_) {
380  XTP_LOG(Log::error, pLog) << "Running DFT" << std::flush;
381  bool run_dft_status_ = qmpackage->Run();
382  if (!run_dft_status_) {
383  SetJobToFailed(jres, pLog, qmpackage->getPackageName() + " run failed");
384  WriteLoggerToFile(work_dir + "/dft.log", dft_logger);
385  return jres;
386  }
387  }
388 
389  if (do_dft_parse_) {
390  bool parse_log_status = qmpackage->ParseLogFile(orbitalsAB);
391  if (!parse_log_status) {
392  SetJobToFailed(jres, pLog, "LOG parsing failed");
393  return jres;
394  }
395 
396  bool parse_orbitals_status = qmpackage->ParseMOsFile(orbitalsAB);
397 
398  if (!parse_orbitals_status) {
399  SetJobToFailed(jres, pLog, "Orbitals parsing failed");
400  return jres;
401  }
402 
403  } // end of the parse orbitals/log
404  qmpackage->CleanUp();
405  WriteLoggerToFile(work_dir + "/dft.log", dft_logger);
406  } else {
407  try {
408  orbitalsAB.ReadFromCpt(orbFileAB);
409  } catch (std::runtime_error&) {
410  SetJobToFailed(jres, pLog,
411  "Do input: failed loading orbitals from " + orbFileAB);
412  return jres;
413  }
414  }
415  tools::Property job_summary;
416  tools::Property& job_output = job_summary.add("output", "");
417  if (do_dftcoupling_) {
418  DFTcoupling dftcoupling;
419  dftcoupling.setLogger(&pLog);
420  dftcoupling.Initialize(dftcoupling_options_);
421  Orbitals orbitalsB;
422  Orbitals orbitalsA;
423 
424  try {
425  orbitalsA.ReadFromCpt(orbFileA);
426  } catch (std::runtime_error&) {
427  SetJobToFailed(jres, pLog,
428  "Do input: failed loading orbitals from " + orbFileA);
429  return jres;
430  }
431 
432  try {
433  orbitalsB.ReadFromCpt(orbFileB);
434  } catch (std::runtime_error&) {
435  SetJobToFailed(jres, pLog,
436  "Do input: failed loading orbitals from " + orbFileB);
437  return jres;
438  }
439  try {
440  dftcoupling.CalculateCouplings(orbitalsA, orbitalsB, orbitalsAB);
441  dftcoupling.Addoutput(job_output, orbitalsA, orbitalsB);
442  } catch (std::runtime_error& error) {
443  std::string errormessage(error.what());
444  SetJobToFailed(jres, pLog, errormessage);
445  return jres;
446  }
447  }
448 
449  // do excited states calculation
450  if (do_gwbse_) {
451  try {
452  XTP_LOG(Log::error, pLog) << "Running GWBSE" << std::flush;
453  Logger gwbse_logger(Log::current_level);
454  gwbse_logger.setMultithreading(false);
455  gwbse_logger.setPreface(Log::info, (format("\nGWBSE INF ...")).str());
456  gwbse_logger.setPreface(Log::error, (format("\nGWBSE ERR ...")).str());
457  gwbse_logger.setPreface(Log::warning, (format("\nGWBSE WAR ...")).str());
458  gwbse_logger.setPreface(Log::debug, (format("\nGWBSE DBG ...")).str());
459  GWBSE gwbse = GWBSE(orbitalsAB);
460  gwbse.setLogger(&gwbse_logger);
461  gwbse.Initialize(gwbse_options_);
462  gwbse.Evaluate();
463  WriteLoggerToFile(work_dir + "/gwbse.log", gwbse_logger);
464  } catch (std::runtime_error& error) {
465  std::string errormessage(error.what());
466  SetJobToFailed(jres, pLog, errormessage);
467  return jres;
468  }
469 
470  } // end of excited state calculation, exciton data is in orbitalsAB_
471 
472  // calculate the coupling
473 
474  if (do_bsecoupling_) {
475  XTP_LOG(Log::error, pLog) << "Running BSECoupling" << std::flush;
476  BSECoupling bsecoupling;
477  // orbitals must be loaded from a file
478  if (!do_gwbse_) {
479  try {
480  orbitalsAB.ReadFromCpt(orbFileAB);
481  } catch (std::runtime_error&) {
482  SetJobToFailed(jres, pLog,
483  "Do input: failed loading orbitals from " + orbFileAB);
484  return jres;
485  }
486  }
487 
488  Orbitals orbitalsB;
489  Orbitals orbitalsA;
490 
491  try {
492  orbitalsA.ReadFromCpt(orbFileA);
493  } catch (std::runtime_error&) {
494  SetJobToFailed(jres, pLog,
495  "Do input: failed loading orbitals from " + orbFileA);
496  return jres;
497  }
498 
499  try {
500  orbitalsB.ReadFromCpt(orbFileB);
501  } catch (std::runtime_error&) {
502  SetJobToFailed(jres, pLog,
503  "Do input: failed loading orbitals from " + orbFileB);
504  return jres;
505  }
506 
507  try {
508  Logger bsecoupling_logger(Log::current_level);
509  bsecoupling_logger.setMultithreading(false);
510  bsecoupling_logger.setPreface(Log::info,
511  (format("\nBSECOU INF ...")).str());
512  bsecoupling_logger.setPreface(Log::error,
513  (format("\nBSECOU ERR ...")).str());
514  bsecoupling_logger.setPreface(Log::warning,
515  (format("\nBSECOU WAR ...")).str());
516  bsecoupling_logger.setPreface(Log::debug,
517  (format("\nBSECOU DBG ...")).str());
518  bsecoupling.setLogger(&bsecoupling_logger);
519  bsecoupling.Initialize(bsecoupling_options_);
520  bsecoupling.CalculateCouplings(orbitalsA, orbitalsB, orbitalsAB);
521  bsecoupling.Addoutput(job_output, orbitalsA, orbitalsB);
522  WriteLoggerToFile(work_dir + "/bsecoupling.log", bsecoupling_logger);
523  } catch (std::runtime_error& error) {
524  std::string errormessage(error.what());
525  SetJobToFailed(jres, pLog, errormessage);
526  return jres;
527  }
528  }
529 
530  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
531  std::stringstream sout;
532  sout << iomXML << job_summary;
533  XTP_LOG(Log::error, pLog) << TimeStamp() << " Finished evaluating pair "
534  << ID_A << ":" << ID_B << std::flush;
535  if (store_dft_ || store_gw_) {
536  boost::filesystem::create_directories(orb_dir);
537  XTP_LOG(Log::error, pLog)
538  << "Saving orbitals to " << orbFileAB << std::flush;
539  if (!store_dft_) {
540  orbitalsAB.MOs().clear();
541  }
542  if (!store_gw_) {
543  orbitalsAB.QPdiag().clear();
544  orbitalsAB.QPpertEnergies().resize(0);
545  }
546  orbitalsAB.WriteToCpt(orbFileAB);
547  } else {
548  XTP_LOG(Log::error, pLog)
549  << "Orb file is not saved according to options " << std::flush;
550  }
551 
552  jres.setOutput(job_summary);
553  jres.setStatus(Job::COMPLETE);
554 
555  return jres;
556 }

558 void IQM::WriteJobFile(const Topology& top) {
559 
560  std::cout << std::endl
561  << "... ... Writing job file " << jobfile_ << std::flush;
562  std::ofstream ofs;
563  ofs.open(jobfile_, std::ofstream::out);
564  if (!ofs.is_open()) {
565  throw std::runtime_error("\nERROR: bad file handle: " + jobfile_);
566  }
567 
568  const QMNBList& nblist = top.NBList();
569 
570  Index jobCount = 0;
571  if (nblist.size() == 0) {
572  std::cout << std::endl
573  << "... ... No pairs in neighbor list, skip." << std::flush;
574  return;
575  }
576 
577  ofs << "<jobs>" << std::endl;
578  std::string tag = "";
579 
580  for (const QMPair* pair : nblist) {
581  if (pair->getType() == QMPair::Excitoncl) {
582  continue;
583  }
584  Index id1 = pair->Seg1()->getId();
585  std::string name1 = pair->Seg1()->getType();
586  Index id2 = pair->Seg2()->getId();
587  std::string name2 = pair->Seg2()->getType();
588  Index id = jobCount;
589  tools::Property Input;
590  tools::Property& pInput = Input.add("input", "");

670 void IQM::ReadJobFile(Topology& top) {
671  // gets the neighborlist from the topology
672  QMNBList& nblist = top.NBList();
673  Index number_of_pairs = nblist.size();
674  Index dft_h = 0;
675  Index dft_e = 0;
676  Index bse_s = 0;
677  Index bse_t = 0;
678  Index incomplete_jobs = 0;
679  Logger log;
680  log.setReportLevel(Log::current_level);
681 
682  tools::Property xml;
683  // load the QC results in a vector indexed by the pair ID
684  xml.LoadFromXML(jobfile_);
685 
686  // loop over all jobs = pair records in the job file
687  for (tools::Property* job : xml.Select("jobs.job")) {
688  if (!job->exists("status")) {
689  throw std::runtime_error(
690  "Jobfile is malformed. <status> tag missing on job.");
691  }
692  if (job->get("status").as<std::string>() != "COMPLETE" ||
693  !job->exists("output")) {
694  incomplete_jobs++;
695  continue;
696  }
697 
698  // job file is stupid, because segment ids are only in input have to get
699  // them out l
700  std::vector<Index> id;
701  for (tools::Property* segment : job->Select("input.segment")) {
702  id.push_back(segment->getAttribute<Index>("id"));
703  }
704  if (id.size() != 2) {
705  throw std::runtime_error(
706  "Getting pair ids from jobfile failed, check jobfile.");
707  }
708 
709  // segments which correspond to these ids
710  Segment& segA = top.getSegment(id[0]);
711  Segment& segB = top.getSegment(id[1]);
712  // pair that corresponds to the two segments
713  QMPair* qmp = nblist.FindPair(&segA, &segB);
714 
715  if (qmp == nullptr) { // there is no pair in the neighbor list with this
716  // name
717  XTP_LOG(Log::error, log)
718  << "No pair " << id[0] << ":" << id[1]
719  << " found in the neighbor list. Ignoring" << std::flush;
720  continue;
721  }
722  if (qmp->getType() != QMPair::PairType::Hopping) {
723  XTP_LOG(Log::error, log) << "WARNING Pair " << qmp->getId()
724  << " is not of any of the "
725  "Hopping type. Skipping pair"
726  << std::flush;
727  continue;
728  }
729 
730  const tools::Property& pair_property = job->get("output");
731 
732  if (pair_property.exists("dftcoupling")) {
733  const tools::Property& dftprop = pair_property.get("dftcoupling");
734  Index homoA = dftprop.getAttribute<Index>("homoA");
735  Index homoB = dftprop.getAttribute<Index>("homoB");
736  QMStateType hole = QMStateType(QMStateType::Hole);
737  if (dftprop.exists(hole.ToLongString())) {
738  const tools::Property& holes = dftprop.get(hole.ToLongString());
739  QMState stateA = GetElementFromMap(hole_levels_, segA.getType());
740  QMState stateB = GetElementFromMap(hole_levels_, segB.getType());
741  Index levelA = homoA - stateA.StateIdx(); // h1 is is homo;
742  Index levelB = homoB - stateB.StateIdx();
743  double J2 = GetDFTCouplingFromProp(holes, levelA, levelB);
744  if (J2 >= 0) {
745  qmp->setJeff2(J2, hole);
746  dft_h++;
747  }
748  }
749  QMStateType electron = QMStateType(QMStateType::Electron);
750  if (dftprop.exists(electron.ToLongString())) {
751  const tools::Property& electrons = dftprop.get(electron.ToLongString());
752  QMState stateA = GetElementFromMap(electron_levels_, segA.getType());
753  QMState stateB = GetElementFromMap(electron_levels_, segB.getType());
754  Index levelA = homoA + 1 + stateA.StateIdx(); // e1 is lumo;
755  Index levelB = homoB + 1 + stateB.StateIdx();
756  double J2 = GetDFTCouplingFromProp(electrons, levelA, levelB);
757  if (J2 >= 0) {
758  qmp->setJeff2(J2, electron);
759  dft_e++;
760  }
761  }
762  }
763  if (pair_property.exists("bsecoupling")) {
764  const tools::Property& bseprop = pair_property.get("bsecoupling");
765  QMStateType singlet = QMStateType(QMStateType::Singlet);
766  if (bseprop.exists(singlet.ToLongString())) {
767  const tools::Property& singlets = bseprop.get(singlet.ToLongString());
768  QMState stateA = GetElementFromMap(singlet_levels_, segA.getType());
769  QMState stateB = GetElementFromMap(singlet_levels_, segB.getType());
770  double J2 = GetBSECouplingFromProp(singlets, stateA, stateB);
771  if (J2 >= 0) {
772  qmp->setJeff2(J2, singlet);
773  bse_s++;
774  }
775  }
776  QMStateType triplet = QMStateType(QMStateType::Triplet);
777  if (bseprop.exists(triplet.ToLongString())) {
778  const tools::Property& triplets = bseprop.get(triplet.ToLongString());
779  QMState stateA = GetElementFromMap(triplet_levels_, segA.getType());
780  QMState stateB = GetElementFromMap(triplet_levels_, segB.getType());
781  double J2 = GetBSECouplingFromProp(triplets, stateA, stateB);
782  if (J2 >= 0) {
783  qmp->setJeff2(J2, triplet);
784  bse_t++;
785  }
786  }
787  }
788  }
789  XTP_LOG(Log::error, log) << "Pairs [total:updated(e,h,s,t)] "
790  << number_of_pairs << ":(" << dft_e << "," << dft_h
791  << "," << bse_s << "," << bse_t
792  << ") Incomplete jobs: " << incomplete_jobs << "\n"
793  << std::flush;
794  std::cout << log;
795  return;
796 }
