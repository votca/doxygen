34BOOST_AUTO_TEST_CASE(mapping_test) {
35
36 Logger log;
37 QMMapper mapper = QMMapper(log);
38 mapper.LoadMappingFile(std::string(XTP_TEST_DATA_FOLDER) +
39 "/segmentmapper/ch4.xml");
40 Segment seg("Methane", 1);
41 Atom atm1(1, "CB", 5, Eigen::Vector3d::Zero(), "C");
42 Atom atm2(1, "HB1", 6, Eigen::Vector3d::UnitX(), "H");
43 Atom atm3(1, "HB2", 7, Eigen::Vector3d::UnitY(), "H");
44 Atom atm4(1, "HB3", 8, -Eigen::Vector3d::UnitX(), "H");
45 Atom atm5(1, "HB4", 9, -Eigen::Vector3d::UnitY(), "H");
46 seg.push_back(atm1);
47 seg.push_back(atm2);
48 seg.push_back(atm3);
49 seg.push_back(atm4);
50 seg.push_back(atm5);
51
52 QMMolecule qmmol = mapper.map(
53 seg, std::string(XTP_TEST_DATA_FOLDER) + "/segmentmapper/molecule.xyz");
54 std::vector<std::string> name_ref = {"C", "H", "H", "H", "H"};
55 std::vector<Index> id_ref = {0, 1, 2, 3, 4};
56 std::vector<Eigen::Vector3d> pos_ref;
57 Eigen::Vector3d pos1 = {-0.026627, -0.0672429, 8.10559e-19};
58 pos_ref.push_back(pos1);
59 Eigen::Vector3d pos2 = {2.03254, -0.0672429, 3.24336e-17};
60 pos_ref.push_back(pos2);
61 Eigen::Vector3d pos3 = {-0.713016, 1.87416, -4.21603e-17};
62 pos_ref.push_back(pos3);
63 Eigen::Vector3d pos4 = {-1, 0, 0};
64 pos_ref.push_back(pos4);
65 Eigen::Vector3d pos5 = {0, -1, 0};
66 pos_ref.push_back(pos5);
67 BOOST_CHECK_EQUAL(qmmol.getId(), 1);
68 BOOST_CHECK_EQUAL(qmmol.getType(), "Methane");
69 Eigen::Vector3d ref = {0.000140384, 0.000354522, 0};
70 BOOST_CHECK_EQUAL(qmmol.getPos().isApprox(ref, 1e-5), true);
71 if (!qmmol.getPos().isApprox(ref, 1e-5)) {
72 std::cout << "qmmolpos" << std::endl;
73 std::cout << qmmol.getPos() << std::endl;
74 std::cout << "qmmolref" << std::endl;
75 std::cout << ref << std::endl;
76 }
77 for (Index i = 0; i < qmmol.size(); i++) {
78 BOOST_CHECK_EQUAL(qmmol[i].getElement(), name_ref[i]);
79 bool pos_equal = qmmol[i].getPos().isApprox(pos_ref[i], 1e-5);
80 if (!pos_equal) {
81 std::cout << "Atom " << i << std::endl;
82 std::cout << "pos" << std::endl;
83 std::cout << qmmol[i].getPos() << std::endl;
84 std::cout << "ref" << std::endl;
85 std::cout << pos_ref[i] << std::endl;
86 }
87 BOOST_CHECK_EQUAL(qmmol[i].getId(), id_ref[i]);
88 }
89}
