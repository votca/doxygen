31 void DftGwBse::Initialize(tools::Property& options) {
32 
33  _do_optimize = false;
34  std::string key = "options." + Identify();
35 
36  if (options.exists(key + ".mpsfile")) {

39  } else {
40  _do_external = false;
41  }
42 
43  if (options.exists(key + ".guess")) {
44  _do_guess = true;
45  _guess_file = options.get(key + ".guess").as<string>();
46 
47  } else {
48  _do_guess = false;
49  }
50 
51  _archive_file = options.ifExistsReturnElseReturnDefault<string>(
52  key + ".archive", "system.orb");
53  _reporting = options.ifExistsReturnElseReturnDefault<string>(
54  key + ".reporting", "default");
55 
56  // job tasks
57  std::vector<string> choices = {"optimize", "energy"};
58  string mode = options.ifExistsAndinListReturnElseThrowRuntimeError<string>(
59  key + ".mode", choices);
60  if (mode == "optimize") _do_optimize = true;
61 
62  // GWBSEENGINE options
63  _gwbseengine_options = options.get(key + ".gwbse_engine");
64 
65  // options for dft package
66  string _package_xml = options.get(key + ".dftpackage").as<string>();
67  load_property_from_xml(_package_options, _package_xml);
68  _package = _package_options.get("package.name").as<string>();
69 
70  // MOLECULE properties
71  _xyzfile =
72  options.ifExistsReturnElseThrowRuntimeError<string>(key + ".molecule");
73 
74  // XML OUTPUT
75  _xml_output = options.ifExistsReturnElseReturnDefault<string>(
76  key + ".output", "dftgwbse.out.xml");
77 
78  // if optimization is chosen, get options for geometry_optimizer
79  if (_do_optimize)
80  _geoopt_options = options.get(key + ".geometry_optimization");
81 
82  // register all QM packages (Gaussian, NWCHEM, etc)
83  QMPackageFactory::RegisterAll();

86 bool DftGwBse::Evaluate() {
87  OPENMP::setMaxThreads(_nThreads);
88  if (_reporting == "silent")
89  _log.setReportLevel(logERROR); // only output ERRORS, GEOOPT info, and
90  // excited state info for trial geometry
91  if (_reporting == "noisy")
92  _log.setReportLevel(logDEBUG); // OUTPUT ALL THE THINGS
93  if (_reporting == "default") _log.setReportLevel(logINFO); //
94 
95  _log.setMultithreading(true);
96  _log.setPreface(logINFO, "\n... ...");
97  _log.setPreface(logERROR, "\n... ...");
98  _log.setPreface(logWARNING, "\n... ...");
99  _log.setPreface(logDEBUG, "\n... ...");
100 
101  // Get orbitals object
102  Orbitals orbitals;
103 
104  if (_do_guess) {
105  XTP_LOG_SAVE(logDEBUG, _log)
106  << "Reading guess from " << _guess_file << flush;
107  orbitals.ReadFromCpt(_guess_file);
108  } else {
109  XTP_LOG_SAVE(logDEBUG, _log)
110  << "Reading structure from " << _xyzfile << flush;
111  orbitals.QMAtoms().LoadFromFile(_xyzfile);
112  }
113 
114  std::unique_ptr<QMPackage> qmpackage =
115  std::unique_ptr<QMPackage>(QMPackages().Create(_package));
116  qmpackage->setLog(&_log);
117  qmpackage->Initialize(_package_options);
118  qmpackage->setRunDir(".");
119 
120  if (_do_external) {
121  StaticRegion region(0, _log);
122  StaticSegment seg = StaticSegment("", 0);
123  seg.LoadFromFile(_mpsfile);
124  region.push_back(seg);
125  qmpackage->AddRegion(region);
126  }
127 
128  GWBSEEngine gwbse_engine;
129  gwbse_engine.setLog(&_log);
130  gwbse_engine.setQMPackage(qmpackage.get());
131  gwbse_engine.Initialize(_gwbseengine_options, _archive_file);
132 
133  if (_do_optimize) {
134  GeometryOptimization geoopt(gwbse_engine, orbitals);
135  geoopt.setLog(&_log);
136  geoopt.Initialize(_geoopt_options);
137  geoopt.Evaluate();
138  } else {
139  gwbse_engine.ExcitationEnergies(orbitals);
140  }
141 
142  XTP_LOG_SAVE(logDEBUG, _log) << "Saving data to " << _archive_file << flush;
143  orbitals.WriteToCpt(_archive_file);
144 
145  tools::Property summary = gwbse_engine.ReportSummary();
146  if (summary.exists("output")) { // only do gwbse summary output if we
147  // actually did gwbse
148  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1,
149  "");
150  XTP_LOG_SAVE(logDEBUG, _log)
151  << "Writing output to " << _xml_output << flush;
152  std::ofstream ofout(_xml_output, std::ofstream::out);
153  ofout << (summary.get("output"));
154  ofout.close();
155  }
156  return true;
157 }
