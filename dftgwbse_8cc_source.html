64

67  .add("basisset", options.get("basisset").as<std::string>());
68  _gwbseengine_options.get("gwbse_options.gwbse")
69  .add("auxbasisset", options.get("auxbasisset").as<std::string>());
70  _gwbseengine_options.get("gwbse_options.gwbse.vxc")
71  .add("functional", options.get("functional").as<std::string>());

100 bool DftGwBse::Evaluate() {
101  OPENMP::setMaxThreads(_nThreads);
102 
103  _log.setReportLevel(Log::current_level);
104 
105  _log.setMultithreading(true);
106  _log.setCommonPreface("\n... ...");
107 
108  // Get orbitals object
109  Orbitals orbitals;
110 
111  if (_do_guess) {
112  XTP_LOG(Log::error, _log)
113  << "Reading guess from " << _guess_file << std::flush;
114  orbitals.ReadFromCpt(_guess_file);
115  } else {
116  XTP_LOG(Log::error, _log)
117  << "Reading structure from " << _xyzfile << std::flush;
118  orbitals.QMAtoms().LoadFromFile(_xyzfile);
119  }
120 
121  std::unique_ptr<QMPackage> qmpackage =
122  std::unique_ptr<QMPackage>(QMPackages().Create(_package));
123  qmpackage->setLog(&_log);
124  qmpackage->Initialize(_package_options);
125  qmpackage->setRunDir(".");
126 
127  if (_do_external) {
128  StaticRegion region(0, _log);
129  StaticSegment seg = StaticSegment("", 0);
130  seg.LoadFromFile(_mpsfile);
131  region.push_back(seg);
132  qmpackage->AddRegion(region);
133  }
134 
135  GWBSEEngine gwbse_engine;
136  gwbse_engine.setLog(&_log);
137  gwbse_engine.setQMPackage(qmpackage.get());
138  gwbse_engine.Initialize(_gwbseengine_options, _archive_file);
139 
140  if (_do_optimize) {
141  GeometryOptimization geoopt(gwbse_engine, orbitals);
142  geoopt.setLog(&_log);
143  geoopt.Initialize(_geoopt_options);
144  geoopt.Evaluate();
145  } else {
146  gwbse_engine.ExcitationEnergies(orbitals);
147  }
148 
149  XTP_LOG(Log::error, _log) << "Saving data to " << _archive_file << std::flush;
150  orbitals.WriteToCpt(_archive_file);
151 
152  tools::Property summary = gwbse_engine.ReportSummary();
153  if (summary.exists("output")) { // only do gwbse summary output if we
154  // actually did gwbse
155  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1,
156  "");
157  XTP_LOG(Log::error, _log)
158  << "Writing output to " << _xml_output << std::flush;
159  std::ofstream ofout(_xml_output, std::ofstream::out);
160  ofout << (summary.get("output"));
161  ofout.close();
162  }
163  return true;
164 }
