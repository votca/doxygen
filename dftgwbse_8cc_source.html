27 void DftGwBse::Initialize(tools::Property* options) {
28 
29  _do_optimize = false;
30  std::string key = "options." + Identify();
31 
32  if (options->exists(key + ".mpsfile")) {
33  _do_external = true;
34  _mpsfile = options->get(key + ".mpsfile").as<string>();
35  _dipole_spacing = options->get(key + ".multipolespacing").as<double>();
36  } else {
37  _do_external = false;
38  }
39 
40  if (options->exists(key + ".guess")) {
41  _do_guess = true;
42  _guess_file = options->get(key + ".guess").as<string>();
43 
44  } else {
45  _do_guess = false;
46  }
47 
48  _archive_file = options->ifExistsReturnElseReturnDefault<string>(
49  key + ".archive", "system.orb");
50  _reporting = options->ifExistsReturnElseReturnDefault<string>(
51  key + ".reporting", "default");
52 
53  // job tasks
54  std::vector<string> choices = {"optimize", "energy"};
55  string mode = options->ifExistsAndinListReturnElseThrowRuntimeError<string>(
56  key + ".mode", choices);
57  if (mode == "optimize") _do_optimize = true;
58 
59  // GWBSEENGINE options
60  _gwbseengine_options = options->get(key + ".gwbse_engine");
61 
62  // options for dft package
63  string _package_xml = options->get(key + ".dftpackage").as<string>();
64  load_property_from_xml(_package_options, _package_xml.c_str());
65  _package = _package_options.get("package.name").as<string>();
66 
67  // MOLECULE properties
68  _xyzfile =
69  options->ifExistsReturnElseThrowRuntimeError<string>(key + ".molecule");
70 
71  // XML OUTPUT
72  _xml_output = options->ifExistsReturnElseReturnDefault<string>(
73  key + ".output", "dftgwbse.out.xml");
74 
75  // if optimization is chosen, get options for geometry_optimizer
76  if (_do_optimize)
77  _geoopt_options = options->get(key + ".geometry_optimization");
78 
79  // register all QM packages (Gaussian, NWCHEM, etc)
80  QMPackageFactory::RegisterAll();
81 }

83 bool DftGwBse::Evaluate() {
84 
85  if (_reporting == "silent")
86  _log.setReportLevel(ctp::logERROR); // only output ERRORS, GEOOPT info, and
87  // excited state info for trial
88  // geometry
89  if (_reporting == "noisy")
90  _log.setReportLevel(ctp::logDEBUG); // OUTPUT ALL THE THINGS
91  if (_reporting == "default") _log.setReportLevel(ctp::logINFO); //
92 
93  _log.setMultithreading(true);
94  _log.setPreface(ctp::logINFO, "\n... ...");
95  _log.setPreface(ctp::logERROR, "\n... ...");
96  _log.setPreface(ctp::logWARNING, "\n... ...");
97  _log.setPreface(ctp::logDEBUG, "\n... ...");
98 
99  // Get orbitals object
100  Orbitals orbitals;
101 
102  if (_do_guess) {
103  CTP_LOG(ctp::logDEBUG, _log)
104  << "Reading guess from " << _guess_file << flush;
105  orbitals.ReadFromCpt(_guess_file);
106  } else {
107  CTP_LOG(ctp::logDEBUG, _log)
108  << "Reading structure from " << _xyzfile << flush;
109  orbitals.LoadFromXYZ(_xyzfile);
110  }
111 
112  QMPackage* qmpackage = QMPackages().Create(_package);
113  qmpackage->setLog(&_log);
114  qmpackage->Initialize(_package_options);
115  qmpackage->setRunDir(".");
116  std::vector<std::shared_ptr<ctp::PolarSeg> > polar_segments;
117  if (_do_external) {
118  vector<ctp::APolarSite*> sites = ctp::APS_FROM_MPS(_mpsfile, 0);
119  std::shared_ptr<ctp::PolarSeg> newPolarSegment(new ctp::PolarSeg(0, sites));
120  polar_segments.push_back(newPolarSegment);
121  qmpackage->setMultipoleBackground(polar_segments);
122  qmpackage->setDipoleSpacing(_dipole_spacing);
123  qmpackage->setWithPolarization(true);
124  }
125 
126  GWBSEEngine gwbse_engine;
127  gwbse_engine.setLog(&_log);
128  gwbse_engine.setQMPackage(qmpackage);
129  gwbse_engine.Initialize(_gwbseengine_options, _archive_file);
130 
131  if (_do_optimize) {
132  GeometryOptimization geoopt(gwbse_engine, orbitals);
133  geoopt.setLog(&_log);
134  geoopt.Initialize(_geoopt_options);
135  geoopt.Evaluate();
136  } else {
137  gwbse_engine.ExcitationEnergies(orbitals);
138  }
139 
140  CTP_LOG(ctp::logDEBUG, _log) << "Saving data to " << _archive_file << flush;
141  orbitals.WriteToCpt(_archive_file);
142 
143  tools::Property summary = gwbse_engine.ReportSummary();
144  if (summary.exists("output")) { // only do gwbse summary output if we
145  // actually did gwbse
146  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1,
147  "");
148  CTP_LOG(ctp::logDEBUG, _log)
149  << "Writing output to " << _xml_output << flush;
150  std::ofstream ofout(_xml_output.c_str(), std::ofstream::out);
151  ofout << (summary.get("output"));
152  ofout.close();
153  }
154  delete qmpackage;
155  return true;
156 }
