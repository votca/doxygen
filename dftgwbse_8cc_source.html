34 void DftGwBse::ParseOptions(const tools::Property& options) {
35 
36  // molecule coordinates
37  xyzfile_ = job_name_ + ".xyz";
38 
39  // options for dft package
40  package_options_ = options.get(".dftpackage");
41 
42  // GWBSEENGINE options
43  gwbseengine_options_ = options;
44 
45  // lets get the archive file name from the xyz file name
46  archive_file_ = job_name_ + ".orb";
47 
48  // XML OUTPUT
49  xml_output_ = job_name_ + "_summary.xml";
50 
51  if (options.exists(".mpsfile")) {
52  mpsfile_ = options.get(".mpsfile").as<std::string>();
53  }
54 
55  // check if guess is requested
56  if (options.exists(".guess")) {
57  guess_file_ = options.get(".guess").as<std::string>();
58  }
59 
60  // if optimization is chosen, get options for geometry_optimizer
61  if (options.exists(".geometry_optimization")) {
62  do_optimize_ = true;
63  geoopt_options_ = options.get(".geometry_optimization");
64  }
65 
66  // register all QM packages
67  QMPackageFactory::RegisterAll();
68 }

70 bool DftGwBse::Run() {
71 
72  log_.setReportLevel(Log::current_level);
73 
74  log_.setMultithreading(true);
75  log_.setCommonPreface("\n... ...");
76 
77  // Get orbitals object
78  Orbitals orbitals;
79 
80  if (!guess_file_.empty()) {
81  XTP_LOG(Log::error, log_)
82  << "Reading guess from " << guess_file_ << std::flush;
83  orbitals.ReadFromCpt(guess_file_);
84  } else {
85  XTP_LOG(Log::error, log_)
86  << "Reading structure from " << xyzfile_ << std::flush;
87  orbitals.QMAtoms().LoadFromFile(xyzfile_);
88  }
89 
90  std::unique_ptr<QMPackage> qmpackage =
91  std::unique_ptr<QMPackage>(QMPackageFactory::QMPackages().Create(
92  package_options_.get("name").as<std::string>()));
93  qmpackage->setLog(&log_);
94  qmpackage->Initialize(package_options_);
95  qmpackage->setRunDir(".");
96 
97  if (!mpsfile_.empty()) {
98  StaticRegion region(0, log_);
99  StaticSegment seg = StaticSegment("", 0);
100  seg.LoadFromFile(mpsfile_);
101  region.push_back(seg);
102  qmpackage->AddRegion(region);
103  }
104 
105  GWBSEEngine gwbse_engine;
106  gwbse_engine.setLog(&log_);
107  gwbse_engine.setQMPackage(qmpackage.get());
108  gwbse_engine.Initialize(gwbseengine_options_, archive_file_);
109 
110  if (do_optimize_) {
111  GeometryOptimization geoopt(gwbse_engine, orbitals);
112  geoopt.setLog(&log_);
113  geoopt.Initialize(geoopt_options_);
114  geoopt.Evaluate();
115  } else {
116  gwbse_engine.ExcitationEnergies(orbitals);
117  }
118 
119  XTP_LOG(Log::error, log_) << "Saving data to " << archive_file_ << std::flush;
120  orbitals.WriteToCpt(archive_file_);
121 
122  tools::Property summary = gwbse_engine.ReportSummary();
123  if (summary.exists("output")) { // only do gwbse summary output if we
124  // actually did gwbse
125  XTP_LOG(Log::error, log_)
126  << "Writing output to " << xml_output_ << std::flush;
127  std::ofstream ofout(xml_output_, std::ofstream::out);
128  ofout << (summary.get("output"));
129  ofout.close();
130  }
131  return true;
132 }
