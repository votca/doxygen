63bool DftGwBse::Run() {
64
65 log_.setReportLevel(Log::current_level);
66
67 log_.setMultithreading(true);
68 log_.setCommonPreface("\n... ...");
69
70 // Get orbitals object
71 Orbitals orbitals;
72
73 if (!guess_file_.empty()) {
74 XTP_LOG(Log::error, log_)
75 << "Reading guess from " << guess_file_ << std::flush;
76 orbitals.ReadFromCpt(guess_file_);
77 } else {
78 XTP_LOG(Log::error, log_)
79 << "Reading structure from " << xyzfile_ << std::flush;
80 orbitals.QMAtoms().LoadFromFile(xyzfile_);
81 }
82
83 std::unique_ptr<QMPackage> qmpackage =
84 std::unique_ptr<QMPackage>(QMPackageFactory().Create(
85 package_options_.get("name").as<std::string>()));
86 qmpackage->setLog(&log_);
87 qmpackage->Initialize(package_options_);
88 qmpackage->setRunDir(".");
89
90 if (!mpsfile_.empty()) {
91 StaticRegion region(0, log_);
92 StaticSegment seg = StaticSegment("", 0);
93 seg.LoadFromFile(mpsfile_);
94 region.push_back(seg);
95 qmpackage->AddRegion(region);
96 }
97
98 GWBSEEngine gwbse_engine;
99 gwbse_engine.setLog(&log_);
100 gwbse_engine.setQMPackage(qmpackage.get());
101 gwbse_engine.Initialize(gwbseengine_options_, archive_file_);
102
103
104 QMMolecule fullMol = orbitals.QMAtoms();
105 gwbse_engine.ExcitationEnergies(orbitals);
106 // If truncation was enabled then rewrite full basis/aux-basis, MOs in full
107 // basis and full QMAtoms
108 if (orbitals.getCalculationType() == "Truncated") {
109 orbitals.QMAtoms().clearAtoms();
110 orbitals.QMAtoms() = fullMol;
111 orbitals.MOs().eigenvectors() = orbitals.getTruncMOsFullBasis();
112 orbitals.SetupDftBasis(orbitals.getDftBasis().Name());
113 if (orbitals.hasAuxbasisName()) {
114 orbitals.SetupAuxBasis(orbitals.getAuxBasis().Name());
115 }
116 }
117
118 XTP_LOG(Log::error, log_) << "Saving data to " << archive_file_ << std::flush;
119 orbitals.WriteToCpt(archive_file_);
120
121 tools::Property summary = gwbse_engine.ReportSummary();
122 if (summary.exists("output")) { // only do gwbse summary output if we
123 // actually did gwbse
124 XTP_LOG(Log::error, log_)
125 << "Writing output to " << xml_output_ << std::flush;
126 std::ofstream ofout(xml_output_, std::ofstream::out);
127 ofout << (summary.get("output"));
128 ofout.close();
129 }
130 return true;
131}
