513 void PolarBackground::RThread::FU_FieldCalc() {
514  vector<PolarSeg*>::iterator sit1;
515  vector<APolarSite*> ::iterator pit1;
516  vector<PolarSeg*>::iterator sit2;
517  vector<APolarSite*> ::iterator pit2;
518  vector<PolarNb*>::iterator nit;
519 
520  // SET-UP NB CONTAINER
521  if (_do_setup_nbs) {
522 
523  // CLEAR POLAR NEIGHBOR-LIST BEFORE SET-UP
524  if (_verbose)
525  LOG(logDEBUG,*(_master->_log))
526  << " - Clearing polar nb-list" << endl;
527  for (sit1 = _part_bg_P.begin(); sit1 < _part_bg_P.end(); ++sit1) {
528  (*sit1)->ClearPolarNbs();
529  }
530 
531  double R_co_sum = 0.0;
532  int R_co_sum_count = 0;
533 
534  for (sit1 = _part_bg_P.begin(); sit1 < _part_bg_P.end(); ++sit1) {
535  PolarSeg *pseg1 = *sit1;
536  if (_verbose)
537  LOG(logDEBUG,*(_master->_log))
538  << "\rMST DBG - Progress " << pseg1->getId()
539  << "/" << _full_bg_P.size() << flush;
540 
541  // GENERATE NEIGHBOUR SHELLS
542  double dR_shell = 0.5;
543  double R_co_max = 2*_master->_R_co;
544  int N_shells = int(R_co_max/dR_shell)+1;
545  vector< vector<PolarNb*> > shelled_nbs;
546  shelled_nbs.resize(N_shells);
547  int allocated_count = 0;
548  int deleted_count = 0;
549 
550  for (sit2 = _full_bg_P.begin(); sit2 < _full_bg_P.end(); ++sit2) {
551  PolarSeg *pseg2 = *sit2;
552  // Active segment?
553  if (!pseg2->IsCharged() && !pseg2->IsPolarizable()) continue;
554  for (int na = -_master->_na_max; na < _master->_na_max+1; ++na) {
555  for (int nb = -_master->_nb_max; nb < _master->_nb_max+1; ++nb) {
556  for (int nc = -_master->_nc_max; nc < _master->_nc_max+1; ++nc) {
557  // Identical?
558  if (na == 0 && nb == 0 && nc == 0 && pseg1 == pseg2) continue;
559  if (na == 0 && nb == 0 && nc == 0 && pseg1->getId() == pseg2->getId()) assert(false);
560  // Apply periodic-boundary correction, check c/o, shift
561  vec dr12_pbc = _master->_top->PbShortestConnect(pseg1->getPos(), pseg2->getPos());
562  vec dr12_dir = pseg2->getPos() - pseg1->getPos();
563  // Image box correction
564  vec L = na*_master->_a + nb*_master->_b + nc*_master->_c;
565  vec dr12_pbc_L = dr12_pbc + L;
566  vec s22x_L = dr12_pbc_L - dr12_dir;
567  double R = votca::tools::abs(dr12_pbc_L);
568  if (R > R_co_max) continue;
569  // Add to shell
570  int shell_idx = int(R/dR_shell);
571  shelled_nbs[shell_idx].push_back(new PolarNb(pseg2, dr12_pbc_L, s22x_L));
572  allocated_count += 1;
573  }}}
574  }
575 
576  // SUM OVER CONSECUTIVE SHELLS & STORE NBS FOR REUSE
577  bool converged = false;
578  int polarizable_nbs_count = 0;
579  int shell_idx = 0;
580  double shell_R = 0;
581  for (int sidx = 0; sidx < N_shells; ++sidx) {
582  // Figure out shell parameters
583  shell_idx = sidx;
584  shell_R = (sidx+1)*dR_shell;
585  vector<PolarNb*> &nb_shell = shelled_nbs[sidx];
586  if (nb_shell.size() < 1) continue;
587  double shell_rms = 0.0;
588  int shell_rms_count = 0;
589  // Interact ...
590  for (nit = nb_shell.begin(); nit < nb_shell.end(); ++nit) {
591  PolarSeg *pseg2 = (*nit)->getNb();
592  // Add neighbour for later use
593  pseg1->AddPolarNb(*nit);
594  if (!pseg2->IsPolarizable()) continue;
595  polarizable_nbs_count += 1;
596  if (votca::tools::abs((*nit)->getR()) > R_co_max) assert(false);
597  // Interact taking into account shift
598  for (pit1 = pseg1->begin(); pit1 < pseg1->end(); ++pit1) {
599  for (pit2 = pseg2->begin(); pit2 < pseg2->end(); ++pit2) {
600  shell_rms += _ewdactor.FU12_ERFC_At_By(*(*pit1), *(*pit2), (*nit)->getS());
601  shell_rms_count += 1;
602  }
603  }
604  }
605  // Determine convergence - measure is the energy of a dipole
606  // of size 0.1*e*nm summed over the shell in an rms manner
607  shell_rms = sqrt(shell_rms/shell_rms_count)*EWD::int2V_m;
608  double e_measure = shell_rms*1e-10*shell_rms_count;
609  if (shell_rms_count > 0 && e_measure <= _master->_crit_dE && shell_R >= _master->_R_co) {
610  converged = true;
611  break;
612  }
613  }
614  if (!converged && polarizable_nbs_count > 0) {
615  _not_converged_count += 1;
616  }
617  if (polarizable_nbs_count > 0) {
618  R_co_sum += shell_R;
619  R_co_sum_count += 1;
620  }
621 
622  // DELETE ALL NEIGHBOURS THAT WERE NOT NEEDED TO CONVERGE SUM
623  for (int sidx = shell_idx+1; sidx < N_shells; ++sidx) {
624  vector<PolarNb*> &nb_shell = shelled_nbs[sidx];
625  for (nit = nb_shell.begin(); nit < nb_shell.end(); ++nit) {
626  delete *nit;
627  deleted_count += 1;
628  }
629  }
630  shelled_nbs.clear();
631  assert(pseg1->PolarNbs().size()+deleted_count == allocated_count);
632  }
633  if (R_co_sum_count == 0) _avg_R_co = 0.0;
634  else _avg_R_co = R_co_sum/R_co_sum_count;
635  }
636  // REUSE NB CONTAINER
637  else {
638  double rms = 0.0;
639  int rms_count = 0;
640  for (sit1 = _part_bg_P.begin(); sit1 < _part_bg_P.end(); ++sit1) {
641  PolarSeg *pseg1 = *sit1;
642  if (_verbose)
643  LOG(logDEBUG,*(_master->_log))
644  << "\rMST DBG - Progress " << pseg1->getId()
645  << "/" << _full_bg_P.size() << flush;
646  for (nit = pseg1->PolarNbs().begin(); nit < pseg1->PolarNbs().end(); ++nit) {
647  PolarSeg *pseg2 = (*nit)->getNb();
648  if (!pseg2->IsPolarizable()) continue;
649  // Interact taking into account shift
650  for (pit1 = pseg1->begin(); pit1 < pseg1->end(); ++pit1) {
651  for (pit2 = pseg2->begin(); pit2 < pseg2->end(); ++pit2) {
652  rms += _ewdactor.FU12_ERFC_At_By(*(*pit1), *(*pit2), (*nit)->getS());
653  rms_count += 1;
654  }
655  }
656  }
657  }
658  rms = sqrt(rms/rms_count)*EWD::int2V_m;
659  }
660  return;

1023 void PolarBackground::GenerateKVectors(vector<PolarSeg*> &ps1,

1027  // All three components non-zero
1028  // S(kx)*S(ky)*S(kz)
