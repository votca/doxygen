179  template <typename T>
180  void WriteData(const CptLoc& loc, const Eigen::MatrixBase<T>& matrix,
181  const std::string& name) const {
182 
183  hsize_t matRows = hsize_t(matrix.rows());
184  hsize_t matCols = hsize_t(matrix.cols());
185 
186  hsize_t dims[2] = {matRows, matCols}; // eigen vectors are n,1 matrices
187 
188  if (dims[1] == 0) {
189  dims[1] = 1;
190  }
191 
192  H5::DataSpace dp(2, dims);
193  const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
194  H5::DataSet dataset;
195  try {
196  dataset = loc.createDataSet(name.c_str(), *dataType, dp);
197  } catch (H5::GroupIException&) {
198  dataset = loc.openDataSet(name.c_str());
199  }
200 
201  hsize_t matColSize = matrix.derived().outerStride();
202 
203  hsize_t fileRows = matCols;
204 
205  hsize_t fStride[2] = {1, fileRows};
206  hsize_t fCount[2] = {1, 1};
207  hsize_t fBlock[2] = {1, fileRows};
208 
209  hsize_t mStride[2] = {matColSize, 1};
210  hsize_t mCount[2] = {1, 1};
211  hsize_t mBlock[2] = {matCols, 1};
212 
213  hsize_t mDim[2] = {matCols, matColSize};
214  H5::DataSpace mspace(2, mDim);
215 
216  for (hsize_t i = 0; i < matRows; i++) {
217  hsize_t fStart[2] = {i, 0};
218  hsize_t mStart[2] = {0, i};
219  dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
220  mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
221  dataset.write(matrix.derived().data(), *dataType, mspace, dp);
222  }
223  }
