154  template <typename T>
155  void WriteData(const CptLoc& loc, const Eigen::MatrixBase<T>& matrix,
156  const std::string& name) {
157 
158  hsize_t matRows = hsize_t(matrix.rows());
159  hsize_t matCols = hsize_t(matrix.cols());
160 
161  hsize_t dims[2] = {matRows, matCols}; // eigen vectors are n,1 matrices
162 
163  if (dims[1] == 0) dims[1] = 1;
164 
165  H5::DataSpace dp(2, dims);
166  const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
167  H5::DataSet dataset;
168  try {
169  dataset = loc.createDataSet(name.c_str(), *dataType, dp);
170  } catch (H5::GroupIException& error) {
171  dataset = loc.openDataSet(name.c_str());
172  }
173 
174  hsize_t matColSize = matrix.derived().outerStride();
175 
176  hsize_t fileRows = matCols;
177 
178  hsize_t fStride[2] = {1, fileRows};
179  hsize_t fCount[2] = {1, 1};
180  hsize_t fBlock[2] = {1, fileRows};
181 
182  hsize_t mStride[2] = {matColSize, 1};
183  hsize_t mCount[2] = {1, 1};
184  hsize_t mBlock[2] = {matCols, 1};
185 
186  hsize_t mDim[2] = {matCols, matColSize};
187  H5::DataSpace mspace(2, mDim);
188 
189  for (hsize_t i = 0; i < matRows; i++) {
190  hsize_t fStart[2] = {i, 0};
191  hsize_t mStart[2] = {0, i};
192  dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
193  mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
194  dataset.write(matrix.derived().data(), *dataType, mspace, dp);
195  }
196  }
