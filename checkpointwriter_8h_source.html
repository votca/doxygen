127 template <typename T>
128 CptTable openTable(const std::string& name, std::size_t nRows,
129 bool compact = false) {
130 CptTable table;
131 try {
132 table = CptTable(name, sizeof(typename T::data), loc_);
133 T::SetupCptTable(table);
134 } catch (H5::Exception&) {
135 try {
136 table = CptTable(name, sizeof(typename T::data), nRows);
137 T::SetupCptTable(table);
138 table.initialize(loc_, compact);
139 } catch (H5::Exception&) {
140 std::stringstream message;

142 << loc_.getFileName() << ":" << path_ << std::endl;
143 throw std::runtime_error(message.str());
144 }

153 template <typename T>
154 void WriteScalar(const CptLoc& loc, const T& value,
155 const std::string& name) const {
156
157 hsize_t dims[1] = {1};
158 H5::DataSpace dp(1, dims);
159 const H5::DataType* dataType = InferDataType<T>::get();
160 H5::Attribute attr;
161 try {
162 attr = loc.createAttribute(name, *dataType, dp);
163 } catch (H5::AttributeIException&) {
164 attr = loc.openAttribute(name);

166 attr.write(*dataType, &value);

187 template <typename T>
188 void WriteData(const CptLoc& loc, const Eigen::MatrixBase<T>& matrix,
189 const std::string& name) const {
190
191 hsize_t matRows = hsize_t(matrix.rows());
192 hsize_t matCols = hsize_t(matrix.cols());
193
194 hsize_t dims[2] = {matRows, matCols}; // eigen vectors are n,1 matrices
195
196 if (dims[1] == 0) {
197 dims[1] = 1;
198 }
199
200 H5::DataSpace dp(2, dims);
201 const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
202 H5::DataSet dataset;
203 try {
204 dataset = loc.createDataSet(name.c_str(), *dataType, dp);
205 } catch (H5::GroupIException&) {
206 dataset = loc.openDataSet(name.c_str());
207 }
208
209 hsize_t matColSize = matrix.derived().outerStride();
210
211 hsize_t fileRows = matCols;
212
213 hsize_t fStride[2] = {1, fileRows};
214 hsize_t fCount[2] = {1, 1};
215 hsize_t fBlock[2] = {1, fileRows};
216
217 hsize_t mStride[2] = {matColSize, 1};
218 hsize_t mCount[2] = {1, 1};
219 hsize_t mBlock[2] = {matCols, 1};
220
221 hsize_t mDim[2] = {matCols, matColSize};
222 H5::DataSpace mspace(2, mDim);
223
224 for (hsize_t i = 0; i < matRows; i++) {
225 hsize_t fStart[2] = {i, 0};
226 hsize_t mStart[2] = {0, i};
227 dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
228 mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
229 dataset.write(matrix.derived().data(), *dataType, mspace, dp);
230 }
231 }

252

271 void WriteData(const CptLoc& loc, const std::vector<Eigen::Vector3d>& v,
272 const std::string& name) const {
273
274 size_t c = 0;
275 std::string r;
276 CptLoc parent;
277 try {
278 parent = loc.createGroup(name);
279 } catch (H5::GroupIException&) {
280 parent = loc.openGroup(name);
281 }
282 for (auto const& x : v) {
283 r = std::to_string(c);
284 WriteData(parent, x, "ind" + r);
285 ++c;
286 }
287 }

289 void WriteData(const CptLoc& loc, const tools::EigenSystem& sys,
290 const std::string& name) const {
291
292 CptLoc parent;
293 try {
294 parent = loc.createGroup(name);
295 } catch (H5::GroupIException&) {
296 parent = loc.openGroup(name);
297 }
298
299 WriteData(parent, sys.eigenvalues(), "eigenvalues");
300 WriteData(parent, sys.eigenvectors(), "eigenvectors");
301 WriteData(parent, sys.eigenvectors2(), "eigenvectors2");
302 WriteScalar(parent, Index(sys.info()), "info");
303 }

305 template <typename T1, typename T2>
306 void WriteData(const CptLoc& loc, const std::map<T1, std::vector<T2>> map,
307 const std::string& name) const {
308
309 size_t c = 0;
310 std::string r;
311 // Iterate over the map and write map as a number of vectors with T1 as
312 // index
313 for (auto const& x : map) {
314 r = std::to_string(c);
315 CptLoc tempGr;
316 try {
317 tempGr = loc.createGroup(name);
318 } catch (H5::GroupIException&) {
319 tempGr = loc.openGroup(name);
320 }
321 WriteData(tempGr, x.second, "index" + r);
322 ++c;
323 }
324 }
