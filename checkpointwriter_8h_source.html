84  template <typename T>
85  void WriteData(const CptLoc& loc, const Eigen::MatrixBase<T>& matrix,
86  const std::string& name) {
87 
88  hsize_t matRows = hsize_t(matrix.rows());
89  hsize_t matCols = hsize_t(matrix.cols());
90 
91  hsize_t dims[2] = {matRows, matCols}; // eigen vectors are n,1 matrices
92 
93  if (dims[1] == 0) dims[1] = 1;
94 
95  H5::DataSpace dp(2, dims);
96  const H5::DataType* dataType = InferDataType<typename T::Scalar>::get();
97  H5::DataSet dataset;
98  dataset = loc.createDataSet(name.c_str(), *dataType, dp);
99 
100  hsize_t matColSize = matrix.derived().outerStride();
101 
102  hsize_t fileRows = matCols;
103 
104  hsize_t fStride[2] = {1, fileRows};
105  hsize_t fCount[2] = {1,1};
106  hsize_t fBlock[2] = {1, fileRows};
107 
108  hsize_t mStride[2] = {matColSize, 1};
109  hsize_t mCount[2] = {1,1};
110  hsize_t mBlock[2] = {matCols, 1};
111 
112  hsize_t mDim[2] = {matCols, matColSize};
113  H5::DataSpace mspace(2, mDim);
114 
115  for (hsize_t i = 0; i < matRows; i++){
116  hsize_t fStart[2] = {i, 0};
117  hsize_t mStart[2] = {0, i};
118  dp.selectHyperslab(H5S_SELECT_SET, fCount, fStart, fStride, fBlock);
119  mspace.selectHyperslab(H5S_SELECT_SET, mCount, mStart, mStride, mBlock);
120  dataset.write(matrix.derived().data(), *dataType, mspace, dp);
121  }
122  }
