111 void KMCSingle::LoadGraph() {
112 
113  Database db;
114  db.Open( _filename );
115  cout << " Loading graph from " << _filename << endl;
116  Statement *stmt = db.Prepare("SELECT _id, name FROM segments;");
117 
118  while (stmt->Step() != SQLITE_DONE) {
119  int id = stmt->Column<int>(0);
120  string name = stmt->Column<string>(1);
121  node_t *n =new node_t(id);
122  _nodes.push_back(n);
123  _nodes_lookup[id] = _nodes.back();
124  if (wildcmp(_injection_name.c_str(), name.c_str())) {
125  _injection.push_back(n);
126  _injection_lookup[id] = _injection.back();
127  }
128  }
129  //delete stmt;
130  cout << " -Nodes: " << _nodes.size() << endl;
131  cout << "seed:" << _seed << endl;
132  cout << " -Injection Points: " << _injection.size() << endl;
133 
134  delete stmt;
135 
136  int links = 0;
137  stmt = db.Prepare("SELECT seg1, seg2, rate12e, rate21e, drX, drY, drZ FROM pairs;"); // electron rates, kmcsingle still lacks capability to treat holes
138  while (stmt->Step() != SQLITE_DONE) {
139  node_t *n1 = _nodes_lookup[stmt->Column<int>(0)];
140  node_t *n2 = _nodes_lookup[stmt->Column<int>(1)];
141  double rate12 = stmt->Column<double>(2);
142  double rate21 = stmt->Column<double>(3);
143  vec r = vec(stmt->Column<double>(4), stmt->Column<double>(5), stmt->Column<double>(6));
144  n1->AddEvent(new link_t(n2, rate12, r));
145  n2->AddEvent(new link_t(n1, rate21, -r));
146  links += 2;
147 
148  if ( votca::tools::globals::verbose ) {
149  cout << "rate12=" << rate12 << endl;
150  cout << "rate21=" << rate21 << endl;
151  cout << "r=" << r << endl;
152  }
153  }
154  delete stmt;
155  cout << " -Links: " << links << endl;
156 
157 }

159 void KMCSingle::RunKMC(void)
160 {
161 
162  double t = 0;
163 
164  srand(_seed);
165  votca::tools::Random::init(rand(), rand(), rand(), rand());
166 
167  // cout << " seed:size:site " << _seed << ":" << _injection.size() << ":" << Random::rand_uniform_int(_injection.size()) << endl;
168  current=_injection[Random::rand_uniform_int(_injection.size())];
169  cout <<" Starting simulation at node: "<<current->_id-1<<endl;
170  if (current->Rate() == 0) {
171  int id = current->_id;
172  throw std::runtime_error("Injected on an unconnected site, id = "+boost::lexical_cast<string>(id));
173  }
174  double next_output = _dt;
175  int i=0;
176  while(t<_runtime) {
177  t+=current->WaitingTime();
178  current->onExecute(); // this line causes a Segmentation fault
179  if(t > next_output) {
180  next_output = t + _dt;
181  cout << t << ": " << r << endl;
182  }
183  }
184  _runtime = t;
185  WriteOcc();
186  cout << std::scientific << "\nKMC run finished\nAverage velocity (m/s): " << r/t*1e-9 << endl;
