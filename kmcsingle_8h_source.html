42 /***
43  * \brief groups events based on variable step size method
44  *
45  * This class groups the events based on the variable step size method. The waiting time is calculated
46  * from the total rate. The probability to choose an event is P_i = \omega_i / \omega_tot
47  *
48  * The template argument is a class which has to implement the functions Rate, which should return the rate of the event
49  * and onExecute, which is called if the event occurs. VSSMGroup implements these methods it's own and therefore
50  * can be used as an event itself. This allows hierarchical grouping of events.
51  *
52  * \todo add a refresh function if rates have changed
53  */
54 template<typename event_t>
55 class VSSMGroup {
56 public:
57  VSSMGroup() {
58  _acc_rate.push_back(0);
59  }
60 
61  /***
62  * \brief add event to list of possible events
63  */
64  void AddEvent(event_t *event) {
65  _events.push_back(event);
66  _acc_rate.push_back(_acc_rate.back() + event->Rate());
67  UpdateWaitingTime();
68  }
69 
73  double Rate() { return _acc_rate.back(); };
74 
78  void onExecute();
82  double WaitingTime() { return _waiting_time; }
83 
87  void UpdateWaitingTime() {
88  _waiting_time = -log( 1.0 - Random::rand_uniform() ) / Rate();
89  }
90 
91  event_t *getEvent(int i) { return _events[i]; }
92 
93 protected:
97  event_t *SelectEvent_LinearSearch();
98 
102  event_t *SelectEvent_BinarySearch();
103 
104  std::vector<event_t*> _events;
105  std::vector<double> _acc_rate;
106  double _waiting_time;

108 
109 };

118 template<typename event_t>
119 inline event_t *VSSMGroup<event_t>::SelectEvent_LinearSearch()
120 {
121  double u = (1.-Random::rand_uniform())*Rate();
122  event_t *event;
123  // find the biggest event for that u < \sum omega_i
124  for(typename std::vector<event_t*>::iterator e=_events.begin();
125  e!=_events.end(); ++e) {
126  u-=(*e)->Rate();
127  if(u<=0) return *e;
128  }
129  // should never happen
130  return _events.back();
131 }

133 template<typename event_t>
134 event_t *VSSMGroup<event_t>::SelectEvent_BinarySearch()
135 {
136 
137  double u = 1.-Random::rand_uniform();
138  u=u*Rate();
139  //double max = Rate();
140  // to a binary search in accumulated events
141  int imin=0;
142  int imax=_acc_rate.size();
143 
144  while(imax - imin > 1) {
145  int imid=(int)((imin+imax)*0.5);
146  //std::cout << u << " " << _acc_rate[imid] << std::endl;
147  if(u<=_acc_rate[imid])
148  imax=imid;
149  else
150  imin=imid;
151  }
152  // std::cout << imin << " " << Rate() <<" " << u << std::endl;
153  return _events[imin];
154 }

269  Statement *stmt = db.Prepare("SELECT _id, name FROM segments;");

286 
287  delete stmt;
288 
289  int links = 0;

291  while (stmt->Step() != SQLITE_DONE) {

308  cout << " -Links: " << links << endl;
