88 template <typename Vtype>
89 struct generic_product_impl<votca::xtp::MatrixFreeOperator, Vtype, DenseShape,

95  typedef
96  typename Product<votca::xtp::MatrixFreeOperator, Vtype>::Scalar Scalar;
97 
98  template <typename Dest>
99  static void scaleAndAddTo(Dest& dst, const votca::xtp::MatrixFreeOperator& op,
100  const Vtype& v, const Scalar& alpha) {
101  // returns dst = alpha * op * v
102  // alpha must be 1 here
103  assert(alpha == Scalar(1) && "scaling is not implemented");
104  EIGEN_ONLY_USED_FOR_DEBUG(alpha);
105  if (op.useRow()) {
106 // make the mat vect product
107 #pragma omp parallel for schedule(guided)
108  for (Index i = 0; i < op.rows(); i++) {
109  dst(i) = op.OperatorRow(i) * v;
110  }
111  }
112 
113  if (op.useBlock()) {
114  Index blocksize = op.getBlocksize();
115  if (op.size() % blocksize != 0) {
116  throw std::runtime_error("blocksize is not a multiple of matrix size");
117  }
118  Index blocks = op.size() / blocksize;
119 
120 // this is inefficient if blocks<num_ofthreads
121 #pragma omp parallel for schedule(guided)
122  for (Index i_row = 0; i_row < blocks; i_row++) {
123  for (Index i_col = 0; i_col < blocks; i_col++) {
124  dst.segment(i_row * blocksize, blocksize) +=
125  op.OperatorBlock(i_row, i_col) *
126  v.segment(i_col * blocksize, blocksize);
127  }
128  }
129  }
130  }
131 };

133 // replacement of the mat*mat operation
134 template <typename Mtype>
135 struct generic_product_impl<votca::xtp::MatrixFreeOperator, Mtype, DenseShape,
136  DenseShape, GemmProduct>
137  : generic_product_impl_base<
138  votca::xtp::MatrixFreeOperator, Mtype,
139  generic_product_impl<votca::xtp::MatrixFreeOperator, Mtype>> {
140 
141  typedef
142  typename Product<votca::xtp::MatrixFreeOperator, Mtype>::Scalar Scalar;
143 
144  template <typename Dest>
145  static void scaleAndAddTo(Dest& dst, const votca::xtp::MatrixFreeOperator& op,
146  const Mtype& m, const Scalar& alpha) {
147  // returns dst = alpha * op * v
148  // alpha must be 1 here
149  assert(alpha == Scalar(1) && "scaling is not implemented");
150  EIGEN_ONLY_USED_FOR_DEBUG(alpha);
151 
152  // make the mat mat product
153  if (op.useRow()) {
154 #pragma omp parallel for
155  for (Index i = 0; i < op.rows(); i++) {
156  const Eigen::RowVectorXd row = op.OperatorRow(i) * m;
157  dst.row(i) = row;
158  }
159  }
160 
161  if (op.useBlock()) {
162  Index blocksize = op.getBlocksize();
163  if (op.size() % blocksize != 0) {
164  throw std::runtime_error("blocksize is not a multiple of matrix size");
165  }
166  Index blocks = op.size() / blocksize;
167  // this uses the fact that all our matrices are symmetric, i.e we can
168  // reuse half the blocks
169  Eigen::MatrixXd result = Eigen::MatrixXd::Zero(dst.rows(), dst.cols());
170 
171 #pragma omp declare reduction (+: Eigen::MatrixXd: omp_out=omp_out+omp_in)\
172  initializer(omp_priv=Eigen::MatrixXd::Zero(omp_orig.rows(),omp_orig.cols()))
173 #pragma omp parallel for schedule(guided) reduction(+ : result)
174  for (Index i_row = 0; i_row < blocks; i_row++) {
175  for (Index i_col = i_row; i_col < blocks; i_col++) {
176  const Eigen::MatrixXd blockmat = op.OperatorBlock(i_row, i_col);
177  result.block(i_row * blocksize, 0, blocksize, dst.cols()) +=
178  blockmat * m.block(i_col * blocksize, 0, blocksize, m.cols());
179  if (i_row != i_col) {
180  result.block(i_col * blocksize, 0, blocksize, dst.cols()) +=
181  blockmat.transpose() *
182  m.block(i_row * blocksize, 0, blocksize, m.cols());
183  }
184  }
185  }
186 
187  dst += result;
188  }
189  }
190 };
