45 /*
46  * The Cuda device behaves like a server that is receiving matrix-matrix
47  * multiplications from a single stream (an Nvidia queue) and handle them
48  * in an asynchronous way. It performs the following operations when recieving
49  * a request:
50  * 1. Check that there is enough space for the arrays
51  * 2. Allocate memory for each matrix
52  * 3. Copy the matrix to the allocated space
53  * 4. Perform the matrix multiplication
54  * 5. Return the result matrix
55  * The Cuda device knows to which memory address it needs to copy back the
56  * result. see: https://docs.nvidia.com/cuda/cublas/index.html#thread-safety2
57  */
58 void OpenMP_CUDA::MultiplyRight(Eigen::MatrixXd& tensor) {
59 
60  // All the GPU communication happens through a single thread that reuses
61  // all memory allocated in the GPU and it's dynamically load-balanced by
62  // OpenMP. The rest of the threads use the default CPU matrix
63  // multiplication
64 #ifdef USE_CUDA
65  if (OPENMP::getThreadId() == 0) {
66  A->copy_to_gpu(tensor);
67  cuda_pip_.gemm(*A, *B, *C);
68  tensor = *C;
69  } else {
70  tensor *= *rightoperator_;
71  }
72 #else
73  tensor *= *rightoperator_;
74 #endif
75  return;
76 }

104  } else {

106  }
