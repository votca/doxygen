75 /*
76  * The Cuda device behaves like a server that is receiving matrix-matrix
77  * multiplications from a single stream (an Nvidia queue) and handle them
78  * in an asynchronous way. It performs the following operations when recieving
79  * a request:
80  * 1. Check that there is enough space for the arrays
81  * 2. Allocate memory for each matrix
82  * 3. Copy the matrix to the allocated space
83  * 4. Perform the matrix multiplication
84  * 5. Return the result matrix
85  * The Cuda device knows to which memory address it needs to copy back the
86  * result. see: https://docs.nvidia.com/cuda/cublas/index.html#thread-safety2
87  */
88 void OpenMP_CUDA::MultiplyRight(Eigen::MatrixXd& tensor) {
89 
90  // All the GPU communication happens through a single thread that reuses
91  // all memory allocated in the GPU and it's dynamically load-balanced by
92  // OpenMP. The rest of the threads use the default CPU matrix
93  // multiplication
94 #ifdef USE_CUDA
95 
96  Index threadid =
97  inside_Parallel_region_ ? threadID_parent_ : OPENMP::getThreadId();
98  if (isInVector(threadid, gpuIDs_)) {
99  if (inside_Parallel_region_) {
100  threadid = 0;
101  }
102  checkCuda(cudaSetDevice(cuda_pips_[threadid]->getDeviceId()));
103  temp_[threadid].A->copy_to_gpu(tensor);
104  cuda_pips_[threadid]->gemm(*temp_[threadid].A, *temp_[threadid].B,
105  *temp_[threadid].C);
106  tensor = *temp_[threadid].C;
107  } else {
108  tensor *= *rightoperator_;
109  }
110 #else
111  tensor *= *rightoperator_;
112 #endif
113  return;
114 }

185 void OpenMP_CUDA::A_TDA(const Eigen::MatrixXd& matrix,
186  const Eigen::VectorXd& vec) {
187  Index threadid =
188  inside_Parallel_region_ ? threadID_parent_ : OPENMP::getThreadId();
189 #ifdef USE_CUDA
190  if (isInVector(threadid, gpuIDs_)) {
191  if (inside_Parallel_region_) {
192  threadid = 0;
193  }
194  checkCuda(cudaSetDevice(cuda_pips_[threadid]->getDeviceId()));
195 
196  temp_[threadid].A->copy_to_gpu(vec);
197  temp_[threadid].B->copy_to_gpu(matrix);
198  cuda_pips_[threadid]->diag_gemm(*temp_[threadid].B, *temp_[threadid].A,
199  *temp_[threadid].C);
200  cuda_pips_[threadid]->gemm(*temp_[threadid].B, *temp_[threadid].C,
201  *temp_[threadid].D, true, false, 1.0);
202  } else {
203  if (inside_Parallel_region_) {
204  threadid = 0;
205  }
206  reduction_[threadid] += matrix.transpose() * vec.asDiagonal() * matrix;
207  }
208 #else
209  if (inside_Parallel_region_) {
210  threadid = 0;
211  }
212  reduction_[threadid] += matrix.transpose() * vec.asDiagonal() * matrix;
213 #endif
214 }
