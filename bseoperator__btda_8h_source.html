116  template <typename Dest>
117  static void scaleAndAddTo(Dest& dst,
118  const votca::xtp::HamiltonianOperator<
119  MatrixReplacementA, MatrixReplacementB>& op,
120  const Mtype& m, const Scalar& alpha) {
121  // returns dst = alpha * op * v
122  // alpha must be 1 here
123  assert(alpha == Scalar(1) && "scaling is not implemented");
124  EIGEN_ONLY_USED_FOR_DEBUG(alpha);
125 
126  Index half = op.rows() / 2;
136  Map<const MatrixXd> m_reshaped(m.data(), m.rows() / 2, m.cols() * 2);
137  MatrixXd temp = op._A * m_reshaped;
138  Map<const MatrixXd> temp_unshaped(temp.data(), m.rows(), m.cols());
139  dst.topRows(half) = temp_unshaped.topRows(half);
140  dst.bottomRows(half) = -temp_unshaped.bottomRows(half);
141  temp = op._B * m_reshaped;
142 
143  // create a second map because temp may have moved so temp_reshaped may
144  // point to invalid memory
145  Map<const MatrixXd> temp_unshaped2(temp.data(), m.rows(), m.cols());
146  dst.topRows(half) += temp_unshaped2.bottomRows(half);
147  dst.bottomRows(half) -= temp_unshaped2.topRows(half);
148  }
