41 template <typename MatrixReplacementA, typename MatrixReplacementB>
42 class HamiltonianOperator
43  : public Eigen::EigenBase<
44  HamiltonianOperator<MatrixReplacementA, MatrixReplacementB>> {
45  public:
46  // Required typedefs, constants, and method:
47  using Scalar = double;
48  using RealScalar = double;
49  using StorageIndex = long;
50  enum {
51  ColsAtCompileTime = Eigen::Dynamic,
52  MaxColsAtCompileTime = Eigen::Dynamic,
53  IsRowMajor = false
54  };
55 
56  HamiltonianOperator(const MatrixReplacementA& A, const MatrixReplacementB& B)
57  : _A(A), _B(B) {
58  _size = 2 * A.cols();
59  _diag = get_diagonal();
60  };
61 
62  Eigen::Index rows() const { return this->_size; }
63  Eigen::Index cols() const { return this->_size; }
64 
65  template <typename Vtype>
66  Eigen::Product<HamiltonianOperator, Vtype, Eigen::AliasFreeProduct> operator*(
67  const Eigen::MatrixBase<Vtype>& x) const {
68  return Eigen::Product<HamiltonianOperator, Vtype, Eigen::AliasFreeProduct>(
69  *this, x.derived());
70  }
71 
72  Eigen::VectorXd get_diagonal() const {
73  Eigen::VectorXd diag = Eigen::VectorXd::Zero(_size);
74  Index half = _size / 2;
75  diag.head(half) = _A.diagonal();
76  diag.tail(half) = -diag.head(half);
77  return diag;
78  }
79 
80  Eigen::VectorXd diagonal() const { return _diag; }
81 
82  // get the full matrix if we have to
83  Eigen::MatrixXd get_full_matrix() const {
84  Eigen::MatrixXd matrix = Eigen::MatrixXd::Zero(_size, _size);
85  Index half = _size / 2;
86  matrix.topLeftCorner(half, half) = _A.get_full_matrix();
87  matrix.topRightCorner(half, half) = _B.get_full_matrix();
88  matrix.bottomLeftCorner(half, half) = -matrix.topRightCorner(half, half);
89  matrix.bottomRightCorner(half, half) = -matrix.topLeftCorner(half, half);
90  return matrix;
91  }
92 
93  const MatrixReplacementA& _A;
94  const MatrixReplacementB& _B;
95 
96  private:
97  Index _size;
98  Eigen::VectorXd _diag;

124  template <typename Dest>
125  static void scaleAndAddTo(Dest& dst,
126  const votca::xtp::HamiltonianOperator<
127  MatrixReplacementA, MatrixReplacementB>& op,
128  const Vtype& v, const Scalar& alpha) {
129  // returns dst = alpha * op * v

132  EIGEN_ONLY_USED_FOR_DEBUG(alpha);

135  dst.head(half) = op._A * v.head(half) + op._B * v.tail(half);
136  dst.tail(half) = -(op._B * v.head(half) + op._A * v.tail(half));
137  }
