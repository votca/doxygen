55 std::string findStructureId(Graph& g) {
56 
57  // Determine the highest degree in the graph
58  int maxD = g.getMaxDegree();
59  // Get the vertices with this degree
60  auto verts = g.getVerticesDegree(maxD);
61 
62  // Get the nodes and determine which node has the greatest stringID
63  // When compared using compare function
64  std::string str_id = "";
65  std::vector<int> gn_ids;
66  for (auto v : verts) {
67  auto gn = g.getNode(v);
68  int comp_int = str_id.compare(gn.getStringId());
69  if (comp_int > 0) {
70  str_id = gn.getStringId();
71  gn_ids.clear();
72  gn_ids.push_back(v);
73  } else if (comp_int == 0) {
74  gn_ids.push_back(v);
75  }
76  }
77 
78  // If the str_id is empty it means the nodes are empty and we will
79  // simply have to rely on the degree to choose the vertices to explore from
80  if (str_id.compare("") == 0) {
81  gn_ids = verts;
82  }
83 
84  // If two or more graph nodes are found to be equal then
85  // they must all be explored
86  std::string chosenId = "";
87  Graph g_chosen = g;
88 
89  for (auto v : gn_ids) {
90  GV gv;
91  Graph g_temp = g;
92  exploreGraph(g_temp, gv, v);
93  std::string temp_struct_id = g_temp.getId();
94  if (chosenId.compare(temp_struct_id) < 0) {
95  chosenId = temp_struct_id;
96  g_chosen = g_temp;
97  }
98  }
99 
100  g = g_chosen;
101  return chosenId;
102 }
103 }
