160 template <typename GV>
161 std::string findStructureId(Graph& graph) {
162 
163  // Determine the highest degree in the graph
164  int maxD = graph.getMaxDegree();
165  // Get the vertices with this degree
166  std::vector<int> vertices = graph.getVerticesDegree(maxD);
167 
168  // Get the nodes and determine which node has the greatest stringID
169  // When compared using compare function
170  std::string str_id = "";
171  std::vector<int> graph_node_ids;
172  for (const int& vertex : vertices) {
173  auto graph_node = graph.getNode(vertex);
174  int comp_int = str_id.compare(graph_node.getStringId());
175  if (comp_int > 0) {
176  str_id = graph_node.getStringId();
177  graph_node_ids.clear();
178  graph_node_ids.push_back(vertex);
179  } else if (comp_int == 0) {
180  graph_node_ids.push_back(vertex);
181  }
182  }
183 
184  // If the str_id is empty it means the nodes are empty and we will
185  // simply have to rely on the degree to choose the vertices to explore from
186  if (str_id.compare("") == 0) {
187  graph_node_ids = vertices;
188  }
189 
190  // If two or more graph nodes are found to be equal then
191  // they must all be explored
192  std::string chosenId = "";
193  Graph graph_chosen = graph;
194 
195  for (const int& vertex : graph_node_ids) {
196  GV graph_visitor;
197  graph_visitor.setStartingVertex(vertex);
198  Graph graph_temp = graph;
199  exploreGraph(graph_temp, graph_visitor);
200  std::string temp_struct_id = graph_temp.getId();
201  if (chosenId.compare(temp_struct_id) < 0) {
202  chosenId = temp_struct_id;
203  graph_chosen = graph_temp;
204  }
205  }
206 
207  graph = graph_chosen;
208  return chosenId;
209 }
