94 std::string findStructureId(Graph& g) {
95 
96  // Determine the highest degree in the graph
97  int maxD = g.getMaxDegree();
98  // Get the vertices with this degree
99  auto verts = g.getVerticesDegree(maxD);
100 
101  // Get the nodes and determine which node has the greatest stringID
102  // When compared using compare function
103  std::string str_id = "";
104  std::vector<int> gn_ids;
105  for (auto v : verts) {
106  auto gn = g.getNode(v);
107  int comp_int = str_id.compare(gn.getStringId());
108  if (comp_int > 0) {
109  str_id = gn.getStringId();
110  gn_ids.clear();
111  gn_ids.push_back(v);
112  } else if (comp_int == 0) {
113  gn_ids.push_back(v);
114  }
115  }
116 
117  // If the str_id is empty it means the nodes are empty and we will
118  // simply have to rely on the degree to choose the vertices to explore from
119  if (str_id.compare("") == 0) {
120  gn_ids = verts;
121  }
122 
123  // If two or more graph nodes are found to be equal then
124  // they must all be explored
125  std::string chosenId = "";
126  Graph g_chosen = g;
127 
128  for (auto v : gn_ids) {
129  GV gv;
130  gv.setStartingVertex(v);
131  Graph g_temp = g;
132  exploreGraph(g_temp, gv);
133  std::string temp_struct_id = g_temp.getId();
134  if (chosenId.compare(temp_struct_id) < 0) {
135  chosenId = temp_struct_id;
136  g_chosen = g_temp;
137  }
138  }
139 
140  g = g_chosen;
141  return chosenId;
142 }
143 }
144 }
