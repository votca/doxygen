121 template <typename GV>
122 std::string findStructureId(Graph& graph) {
123 
124  // Determine the highest degree in the graph
125  int maxD = graph.getMaxDegree();
126  // Get the vertices with this degree
127  std::vector<int> vertices = graph.getVerticesDegree(maxD);
128 
129  // Get the nodes and determine which node has the greatest stringID
130  // When compared using compare function
131  std::string str_id = "";
132  std::vector<int> graph_node_ids;
133  for (const int& vertex : vertices) {
134  auto graph_node = graph.getNode(vertex);
135  int comp_int = str_id.compare(graph_node.getStringId());
136  if (comp_int > 0) {
137  str_id = graph_node.getStringId();
138  graph_node_ids.clear();
139  graph_node_ids.push_back(vertex);
140  } else if (comp_int == 0) {
141  graph_node_ids.push_back(vertex);
142  }
143  }
144 
145  // If the str_id is empty it means the nodes are empty and we will
146  // simply have to rely on the degree to choose the vertices to explore from
147  if (str_id.compare("") == 0) {
148  graph_node_ids = vertices;
149  }
150 
151  // If two or more graph nodes are found to be equal then
152  // they must all be explored
153  std::string chosenId = "";
154  Graph graph_chosen = graph;
155 
156  for (const int& vertex : graph_node_ids) {
157  GV graph_visitor;
158  graph_visitor.setStartingVertex(vertex);
159  Graph graph_temp = graph;
160  exploreGraph(graph_temp, graph_visitor);
161  std::string temp_struct_id = graph_temp.getId();
162  if (chosenId.compare(temp_struct_id) < 0) {
163  chosenId = temp_struct_id;
164  graph_chosen = graph_temp;
165  }
166  }
167 
168  graph = graph_chosen;
169  return chosenId;
170 }
