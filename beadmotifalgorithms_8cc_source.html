74 Index determineEdgesOfBranchesWithSingleConnectionToJunction_(
75  Index junction, ReducedGraph& reduced_graph,
76  unordered_map<Edge, bool>& remove_edges,
77  unordered_map<Index, vector<Edge>>
78  contracted_branch_edges_connected_to_junction) {
79 
80  Index number_branches_with_more_than_one_bond = 0;
81  // Cycle the branches
82  for (pair<Index, vector<Edge>> branch_and_contracted_edges :
83  contracted_branch_edges_connected_to_junction) {
84  bool edge_is_single_connection_to_junction = false;
85  if (branch_and_contracted_edges.second.size() == 1) {
86  // Make sure the branch is not a loop as well
87  Edge contracted_edge = branch_and_contracted_edges.second.at(0);
88  if (contracted_edge.loop() == false) {
89  vector<vector<Edge>> expanded_edges =
90  reduced_graph.expandEdge(contracted_edge);
91  // Determine which of these edges are conneted to the vertex
92  if (expanded_edges.size() == 1) {
93  for (Edge& edge : expanded_edges.at(0)) {
94  if (edge.contains(junction)) {
95  remove_edges[edge] = true;
96  edge_is_single_connection_to_junction = true;
97  break;
98  }
99  }
100  }
101  }
102  }
103  if (edge_is_single_connection_to_junction == false) {
104  ++number_branches_with_more_than_one_bond;
105  }
106  }
107 
108  return number_branches_with_more_than_one_bond;
109 }

118 void calculateEdgesToRemove_(Graph& full_graph,
119  unordered_map<Edge, bool>& remove_edges) {
120 
121  ReducedGraph reduced_graph = reduceGraph(full_graph);
122 
123  vector<Index> junctions = reduced_graph.getJunctions();
124 
125  for (Index& junction : junctions) {
126  vector<Edge> neighboring_edges = reduced_graph.getNeighEdges(junction);
127  unordered_map<Edge, bool> contracted_edge_explored;
128  for (Edge& edge : neighboring_edges) {
129  contracted_edge_explored[edge] = false;
130  }
131 
132  Index branch_index = 0;
133  unordered_map<Index, vector<Edge>>
134  contracted_branch_edges_connected_to_junction;
135  for (Edge& branch_starting_edge : neighboring_edges) {
136  if (contracted_edge_explored[branch_starting_edge] == false) {
137  // The starting edge could be redundant but I guess it doesn't matter
138  // redundant edges would mean they are part of the same branch anyway
139  set<Edge> branch_edges =
140  exploreBranch(reduced_graph, junction, branch_starting_edge);
141 
142  vector<Edge> starting_edges_explored_by_branch;
143  for (Edge& branch_starting_edge2 : neighboring_edges) {
144  if (branch_edges.count(branch_starting_edge2)) {
145  contracted_edge_explored[branch_starting_edge2] = true;
146  starting_edges_explored_by_branch.push_back(branch_starting_edge2);
147  }
148  }
149  contracted_branch_edges_connected_to_junction[branch_index] =
150  starting_edges_explored_by_branch;
151  ++branch_index;
152  }
153  } // for(Edge & brach_starting_edge : neighboring_edges)
154 
155  // If there is more than one branch
156  if (branch_index > 1) {
157  // We now know how many edges are connected to the starting node from each
158  // branch We can use the full graph to decouple the edges we can begin by
159  // snipping branches with only a single edge connecting them to starting
160  // nodes.
161  Index number_branches_with_more_than_one_bond =
162  determineEdgesOfBranchesWithSingleConnectionToJunction_(
163  junction, reduced_graph, remove_edges,
164  contracted_branch_edges_connected_to_junction);
165 
166  // If more than one branch claims the junction, and each branch has more
167  // than one connetion to the junction neigther branch can claim the
168  // junction, in this case all connections to the junction will be cut.
169  if (number_branches_with_more_than_one_bond > 1) {
170  removeAllEdgesConnectedToVertex_(junction, full_graph, remove_edges);
171  }
172  }
173  } // for(Index & junction : junctions)
174 }

299 void MotifDeconstructor_::deconstructComplexSingleStructures(
300  BeadMotifConnector& connector) {
301  list<BeadMotif> split_motifs;
302  for (IdMotif& id_and_bead_motif : motifs_complex_single_structure_) {
303  Graph full_graph = id_and_bead_motif.second.getGraph();
304  vector<Edge> all_edges = full_graph.getEdges();
305  unordered_map<Edge, bool> remove_edges;
306  for (Edge& edge : all_edges) {
307  remove_edges[edge] = false;
308  }
309 
310  calculateEdgesToRemove_(full_graph, remove_edges);
311 
312  vector<Index> all_vertices = full_graph.getVertices();
313 
314  BeadStructure<BaseBead> new_beadstructure;
315  for (Index& vertex : all_vertices) {
316  new_beadstructure.AddBead(id_and_bead_motif.second.getBead(vertex));
317  }
318 
319  for (pair<const Edge, bool>& edge_and_remove : remove_edges) {
320  if (edge_and_remove.second == false) {
321  Edge edge = edge_and_remove.first;
322  new_beadstructure.ConnectBeads(edge.getEndPoint1(),
323  edge.getEndPoint2());
324  } else {
325  bead_edges_removed_.push_back(edge_and_remove.first);
326  }
327  }
328 
329  list<BeadMotif> split_motifs_temp =
330  breakIntoMotifs<list<BeadMotif>>(new_beadstructure);
331 
332  split_motifs.splice(split_motifs.end(), split_motifs_temp);
333  }
334 
335  motifs_complex_single_structure_.clear();
336  sortMotifsAndAssignIdsToSimpleMotifs_(split_motifs);
337  determineMotifConnections_(connector);
338 }

340 void MotifDeconstructor_::determineMotifConnections_(
341  BeadMotifConnector& connector) {
342 
343  // Cycle the edges
344  list<Edge>::iterator edge_iterator = bead_edges_removed_.begin();
345  while (edge_iterator != bead_edges_removed_.end()) {
346  assert(edge_iterator->loop() == false);
347  Index bead_id1 = edge_iterator->getEndPoint1();
348  Index bead_id2 = edge_iterator->getEndPoint2();
349  // Cycle the motifs
350  list<IdMotif>::iterator motif_bead1_iterator;
351  for (motif_bead1_iterator = motifs_simple_.begin();
352  motif_bead1_iterator != motifs_simple_.end(); ++motif_bead1_iterator) {
353 
354  if (motif_bead1_iterator->second.BeadExist(bead_id1)) {
355  break;
356  }
357  }
358 
359  list<IdMotif>::iterator motif_bead2_iterator;
360  for (motif_bead2_iterator = motifs_simple_.begin();
361  motif_bead2_iterator != motifs_simple_.end(); ++motif_bead2_iterator) {
362 
363  if (motif_bead2_iterator->second.BeadExist(bead_id2)) {
364  break;
365  }
366  }
367 
368  // We remove the edge from the list and add it as a connection
369  if (motif_bead2_iterator != motifs_simple_.end() &&
370  motif_bead1_iterator != motifs_simple_.end()) {
371  Edge motif_edge(motif_bead1_iterator->first, motif_bead2_iterator->first);
372  connector.AddMotifAndBeadEdge(motif_edge, *edge_iterator);
373  edge_iterator = bead_edges_removed_.erase(edge_iterator);
374  } else {
375  ++edge_iterator;
376  }
377  } // while( edge_iterator != bead_edges.end() )
378 }

380 void MotifDeconstructor_::sortMotifsAndAssignIdsToSimpleMotifs_(
381  list<BeadMotif>& bead_motifs) {
382 
383  list<BeadMotif>::iterator bead_motif_iter = bead_motifs.begin();
384  while (bead_motif_iter != bead_motifs.end()) {
385 
386  Index new_motif_id = unassigned_id;
387  if (bead_motif_iter->isMotifSimple()) {
388  new_motif_id = motif_index;
389  ++motif_index;
390  }
391  pair<Index, BeadMotif> id_and_motif(new_motif_id, move(*bead_motif_iter));
392  AddMotif(id_and_motif);
393  bead_motif_iter = bead_motifs.erase(bead_motif_iter);
394  }
395 }
