137 
138 
139 
140  timesteps_update_longrange = options->get("options.device.timesteps_update_longrange").as<int>();
141 
142  if(advanced) {
143  // default value is equal to 1
144  injection_prefactor = options->get("options.device.injection_prefactor").as<double>();
145  collection_prefactor = options->get("options.device.collection_prefactor").as<double>();
146 
147  // default value is equal to 0.5
148  self_image_prefactor = options->get("options.device.self_image_prefactor").as<double>();
149 
150  // default value is integer floor of size of device
151  number_of_layers = options->get("options.device.number_of_layers").as<int>();
152 
153  // default value is 10 images
154  number_short_range_images = options->get("options.device.number_short_range_images").as<int>();
155  number_long_range_images = options->get("options.device.number_long_range_images").as<int>();
156 
157  // default slab-based longrange calculations (true)
158  longrange_slab = options->get("options.device.longrange_slab").as<bool>();
159 
160  // default is to write profiles (true))
161  write_charge_profile = options->get("options.device.write_charge_profile").as<bool>();
162  write_potential_profile = options->get("options.device.write_potential_profile").as<bool>();
163 
164  }
165  else {
166  injection_prefactor = 1.0;
167  collection_prefactor = 1.0;
168 
169  self_image_prefactor = 0.5;
170 
171  number_of_layers = floor(size_x);
172 
173  number_short_range_images = 10;
174  number_long_range_images = 10;
175 
176  longrange_slab = true;
177 
178  write_charge_profile = true;
179  write_potential_profile = true;
180  }
