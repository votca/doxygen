48 void AOBasis::Fill(const BasisSet& bs, const QMMolecule& atoms) {
49  _AOBasisSize = 0;
50  _aoshells.clear();
51  _FuncperAtom.clear();
52  // loop over atoms
53  for (const QMAtom& atom : atoms) {
54  Index atomfunc = 0;
55  const std::string& name = atom.getElement();
56  const Element& element = bs.getElement(name);
57  for (const Shell& shell : element) {
58  Index numfuncshell = NumFuncShell(shell.getL());
59  AOShell& aoshell = addShell(shell, atom, _AOBasisSize);
60  _AOBasisSize += numfuncshell;
61  atomfunc += numfuncshell;
62  for (const GaussianPrimitive& gaussian : shell) {
63  aoshell.addGaussian(gaussian);
64  }
65  aoshell.CalcMinDecay();
66  aoshell.normalizeContraction();
67  }
68  _FuncperAtom.push_back(atomfunc);
69  }
70  return;
71 }
