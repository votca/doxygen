44 
45  // cout << " Reordering MOs from " << start << " to " << target << endl;
46 
47  if (start == target) {
48  return;
49  }
50 
51  if(target=="orca" || target=="nwchem"){
52  std::vector<int> multiplier = getMultiplierVector(target,start);
53  // and reorder rows of _orbitals->_mo_coefficients() accordingly
54  MultiplyMOs(v, multiplier);
55  }
56 
57  // get reordering vector _start -> target
58 
59  std::vector<int> order = getReorderVector(start, target);
60 
61  // Sanity check
62  if (v.rows() != int(order.size())) {
63  std::cerr << "Size mismatch in ReorderMOs" << v.rows() << ":" << order.size() << std::endl;
64  throw std::runtime_error("Abort!");
65  }
66 
67  // actual swapping of coefficients
68  for (unsigned _i_orbital = 0; _i_orbital < v.cols(); _i_orbital++) {
69  for (unsigned s = 1, d; s < order.size(); ++s) {
70  for (d = order[s]; d < s; d = order[d]) {
71  ;
72  }
73  if (d == s) while (d = order[d], d != s) std::swap(v(s,_i_orbital), v(d,_i_orbital));
74  }
75  }
76 
77  // NWChem has some strange minus in d-functions
78  if (start == "nwchem" || start == "orca") {
79 
80  std::vector<int> multiplier = getMultiplierVector(start, target);
81  // and reorder rows of _orbitals->_mo_coefficients() accordingly
82  MultiplyMOs(v, multiplier);
83 
84  }
85 
86 
87  return;

90 void AOBasis::ReorderMatrix(Eigen::MatrixXd &v,const std::string& start,const std::string& target ){
91  if (start==target){
92  return;
93  }
94  std::vector<int> order = getReorderVector(start, target);
95  std::vector<int> multiplier=getMultiplierVector(start,target);
96 
97  if (v.cols() != int(order.size())) {
98  std::cerr << "Size mismatch in ReorderMatrix" << v.cols() << ":" << order.size() << std::endl;
99  throw std::runtime_error("Abort!");
100  }
101 
102  if (start != "xtp") {
103  std::vector<int>newmultiplier=std::vector<int>(multiplier.size());
104  for(unsigned i=0;i<newmultiplier.size();i++){
105  newmultiplier[i]=multiplier[order[i]];
106  }
107  multiplier=newmultiplier;
108  }
109 
110  Eigen::MatrixXd temp=v;
111  for(unsigned i=0;i<temp.cols();i++){
112  int i_index=order[i];
113  for(unsigned j=0;j<temp.rows();j++){
114  int j_index=order[j];
115  v(i_index,j_index)=multiplier[i]*multiplier[j]*temp(i,j);
116  }
117  }
118 
119 
120  return;
121 }

124 void AOBasis::MultiplyMOs(Eigen::MatrixXd &v, std::vector<int> const &multiplier ) {
125  // Sanity check
126  if ( v.cols() != int(multiplier.size()) ) {
127  std::cerr << "Size mismatch in MultiplyMOs" << v.cols() << ":" << multiplier.size() << std::endl;
128  throw std::runtime_error( "Abort!");
129  }
130  for ( unsigned _i_basis = 0; _i_basis < v.cols(); _i_basis++ ){
131  for ( unsigned _i_orbital = 0; _i_orbital < v.rows(); _i_orbital++ ){
132  v(_i_basis ,_i_orbital) = multiplier[_i_basis] * v(_i_basis,_i_orbital );
133  }
134  }
135  return;

180 void AOBasis::addTrafoCartShell( const AOShell* shell , Eigen::Block<Eigen::MatrixXd>& _submatrix ){
181 
182 
183  // fill _local according to _lmax;
184  int _lmax = shell->getLmax();
185  std::string _type = shell->getType();
186 
187  int _sph_size =NumFuncShell( _type ) + OffsetFuncShell( _type );
188  int _cart_size = NumFuncShell_cartesian( _type ) + OffsetFuncShell_cartesian( _type ) ;
189 
190  // cout << " local size : " << _sph_size << " : " << _cart_size << endl;
191 
192  Eigen::MatrixXd _local = Eigen::MatrixXd::Zero(_sph_size,_cart_size);
193 
194  // s-functions
195  _local(0,0) = 1.0; // s
196 
197  // p-functions
198  if ( _lmax > 0 ){
199  _local(1,1) = 1.0;
200  _local(2,2) = 1.0;
201  _local(3,3) = 1.0;
202  }
203 
204  // d-functions
205  if ( _lmax > 1 ){
206  _local(4,4) = -0.5; // d3z2-r2 (dxx)
207  _local(4,5) = -0.5; // d3z2-r2 (dyy)
208  _local(4,6) = 1.0; // d3z2-r2 (dzz)
209  _local(5,8) = 1.0; // dxz
210  _local(6,9) = 1.0; // dyz
211  _local(7,4) = 0.5*sqrt(3.0); // dx2-y2 (dxx)
212  _local(7,5) = -_local(7,4); // dx2-y2 (dyy)
213  _local(8,7) = 1.0; // dxy
214  }
215 
216  if ( _lmax > 2 ){
217  std::cerr << " Gaussian input with f- functions or higher not yet supported!" << std::endl;
218  exit(1);
219  }
220 
221  // now copy to _trafo
222  for ( int _i_sph = 0 ; _i_sph < NumFuncShell( _type ) ; _i_sph++ ){
223  for ( int _i_cart = 0 ; _i_cart < NumFuncShell_cartesian( _type ) ; _i_cart++ ){
224 
225 
226  _submatrix( _i_sph , _i_cart ) = _local( _i_sph + OffsetFuncShell( _type ) , _i_cart + OffsetFuncShell_cartesian( _type ) );
227 
228  }
229  }
230  return;
231 }

400  void AOBasis::addReorderShell(const std::string& start, const std::string& target, const std::string& shell_type, std::vector<int>& order) {
401  //Reordering is given by email from gaussian, orca output MOs, and http://www.nwchem-sw.org/index.php/Release66:Basis for nwchem
402 
403  // current length of vector
404 
405  int _cur_pos = order.size() - 1;
406 
407  if (target == "xtp") {
408 
409  // single type shells defined here
410  if (shell_type.length() == 1) {
411  if (shell_type == "S") {
412  order.push_back(_cur_pos + 1);
413  }//for S
414 
415 
416  //votca order is z,y,x e.g. Y1,0 Y1,-1 Y1,1
417  else if (shell_type == "P") {
418  if (start == "orca") {
419  //orca order is z,x,y Y1,0,Y1,1,Y1,-1
420  order.push_back(_cur_pos + 1);
421  order.push_back(_cur_pos + 3);
422  order.push_back(_cur_pos + 2);
423  } else if (start == "gaussian" || start == "nwchem") {
424  //nwchem gaussian x,y,z Y1,1 Y1,-1 Y1,0
425  order.push_back(_cur_pos + 3);
426  order.push_back(_cur_pos + 2);
427  order.push_back(_cur_pos + 1);
428  } else if (start == "votca") {//for usage with old orb files
429  //old votca x,y,z Y1,1 Y1,-1 Y1,0
430  order.push_back(_cur_pos + 3);
431  order.push_back(_cur_pos + 2);
432  order.push_back(_cur_pos + 1);
433  } else {
434  std::cerr << "Tried to reorder p-functions from package " << start << ".";
435  throw std::runtime_error("Reordering not implemented yet!");
436  }
437  }//for P
438  //votca order is d3z2-r2 dyz dxz dxy dx2-y2 e.g. Y2,0 Y2,-1 Y2,1 Y2,-2 Y2,2
439  else if (shell_type == "D") {
440  //orca order is d3z2-r2 dxz dyz dx2-y2 dxy e.g. Y2,0 Y2,1 Y2,-1 Y2,2 Y2,-2
441  if (start == "gaussian" || start == "orca") {
442  order.push_back(_cur_pos + 1);
443  order.push_back(_cur_pos + 3);
444  order.push_back(_cur_pos + 2);
445  order.push_back(_cur_pos + 5);
446  order.push_back(_cur_pos + 4);

451  order.push_back(_cur_pos + 1);
452  order.push_back(_cur_pos + 3);
453  order.push_back(_cur_pos + 5);
454  } else if (start == "votca") { //for usage with old orb files
455  order.push_back(_cur_pos + 3);

549 void AOBasis::AOBasisFill(BasisSet* bs , std::vector<QMAtom* > _atoms, int _fragbreak ) {
550  tools::Elements elementinfo;
551  std::vector<QMAtom* > :: iterator ait;
552 
553  _AOBasisSize =0;
554  _AOBasisFragA=0;
555  _AOBasisFragB=0;
556  // loop over atoms
557  for (ait = _atoms.begin(); ait < _atoms.end(); ++ait) {
558 
559  const tools::vec& pos=(*ait)->getPos();
560 
561  // get element type of the atom
562  std::string name = (*ait)->getType();
563  //assign its nuclear charge
564  (*ait)->nuccharge=elementinfo.getNucCrg(name);
565  // get the basis set entry for this element
566  Element* element = bs->getElement(name);
567 
568  // and loop over all shells
569  for (Element::ShellIterator its = element->firstShell(); its != element->lastShell(); its++) {
570  Shell* shell = (*its);
571  int numfuncshell=NumFuncShell(shell->getType());
572  AOShell* aoshell = addShell(shell->getType(), shell->getLmax(), shell->getLmin(), shell->getScale(),
573  numfuncshell, _AOBasisSize, OffsetFuncShell(shell->getType()), pos, name, (*ait)->getAtomID());
574  _AOBasisSize += numfuncshell;
575  for (Shell::GaussianIterator itg = shell->firstGaussian(); itg != shell->lastGaussian(); itg++) {
576  GaussianPrimitive* gaussian = *itg;
577  aoshell->addGaussian(gaussian->decay, gaussian->contraction);
578  }
579  aoshell->CalcMinDecay();
580  aoshell->normalizeContraction();
581 
582  }
583  if ( (*ait)->getAtomID() < _fragbreak ) _AOBasisFragA = _AOBasisSize;
584 
585 
586 }
587
