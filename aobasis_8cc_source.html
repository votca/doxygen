48 void AOBasis::ReorderMOs(Eigen::MatrixXd &v, const std::string& start, const std::string& target) {
49 
50  if (start == target) {
51  return;
52  }
53 
54  if(target=="orca" || target=="nwchem"){
55  std::vector<int> multiplier = getMultiplierVector(target,start);
56  // and reorder rows of _orbitals->_mo_coefficients() accordingly
57  MultiplyMOs(v, multiplier);
58  }
59 
60  // get reordering vector _start -> target
61  std::vector<int> order = getReorderVector(start, target);
62 
63  // Sanity check
64  if (v.rows() != int(order.size())) {
65  std::cerr << "Size mismatch in ReorderMOs" << v.rows() << ":" << order.size() << std::endl;
66  throw std::runtime_error("Abort!");
67  }
68 
69  // actual swapping of coefficients
70  for (unsigned _i_orbital = 0; _i_orbital < v.cols(); _i_orbital++) {
71  for (unsigned s = 1, d; s < order.size(); ++s) {
72  for (d = order[s]; d < s; d = order[d]) {
73  ;
74  }
75  if (d == s) while (d = order[d], d != s) std::swap(v(s,_i_orbital), v(d,_i_orbital));
76  }
77  }
78 
79  // NWChem has some strange minus in d-functions
80  if (start == "nwchem" || start == "orca") {
81  std::vector<int> multiplier = getMultiplierVector(start, target);
82  // and reorder rows of _orbitals->_mo_coefficients() accordingly
83  MultiplyMOs(v, multiplier);
84  }
85  return;
86 }

88 void AOBasis::ReorderMatrix(Eigen::MatrixXd &v,const std::string& start,const std::string& target ){
89  if (start==target){
90  return;
91  }
92  std::vector<int> order = getReorderVector(start, target);
93  std::vector<int> multiplier=getMultiplierVector(start,target);
94 
95  if (v.cols() != int(order.size())) {
96  std::cerr << "Size mismatch in ReorderMatrix" << v.cols() << ":" << order.size() << std::endl;
97  throw std::runtime_error("Abort!");
98  }
99 
100  if (start != "xtp") {
101  std::vector<int>newmultiplier=std::vector<int>(multiplier.size());
102  for(unsigned i=0;i<newmultiplier.size();i++){
103  newmultiplier[i]=multiplier[order[i]];
104  }
105  multiplier=newmultiplier;
106  }
107 
108  Eigen::MatrixXd temp=v;
109  for(unsigned i=0;i<temp.cols();i++){
110  int i_index=order[i];
111  for(unsigned j=0;j<temp.rows();j++){
112  int j_index=order[j];
113  v(i_index,j_index)=multiplier[i]*multiplier[j]*temp(i,j);
114  }
115  }
116 
117  return;
118 }

121 void AOBasis::MultiplyMOs(Eigen::MatrixXd &v, std::vector<int> const &multiplier ) {
122  // Sanity check
123  if ( v.cols() != int(multiplier.size()) ) {
124  std::cerr << "Size mismatch in MultiplyMOs" << v.cols() << ":" << multiplier.size() << std::endl;
125  throw std::runtime_error( "Abort!");
126  }
127  for ( unsigned _i_basis = 0; _i_basis < v.cols(); _i_basis++ ){
128  for ( unsigned _i_orbital = 0; _i_orbital < v.rows(); _i_orbital++ ){
129  v(_i_basis ,_i_orbital) = multiplier[_i_basis] * v(_i_basis,_i_orbital );
130  }
131  }
132  return;
133  }

171 void AOBasis::addTrafoCartShell( const AOShell* shell , Eigen::Block<Eigen::MatrixXd>& submatrix ){
172 
173  // fill _local according to _lmax;
174  int lmax = shell->getLmax();
175  std::string type = shell->getType();
176 
177  int sph_size =NumFuncShell( type ) + OffsetFuncShell( type );
178  int cart_size = NumFuncShell_cartesian( type ) + OffsetFuncShell_cartesian( type ) ;
179  Eigen::MatrixXd local = Eigen::MatrixXd::Zero(sph_size,cart_size);
180 
181  // s-functions
182  local(0,0) = 1.0; // s
183  // p-functions
184  if ( lmax > 0 ){
185  local(1,1) = 1.0;
186  local(2,2) = 1.0;
187  local(3,3) = 1.0;
188  }
189  // d-functions
190  if ( lmax > 1 ){
191  local(4,4) = -0.5; // d3z2-r2 (dxx)
192  local(4,5) = -0.5; // d3z2-r2 (dyy)
193  local(4,6) = 1.0; // d3z2-r2 (dzz)
194  local(5,8) = 1.0; // dxz
195  local(6,9) = 1.0; // dyz
196  local(7,4) = 0.5*sqrt(3.0); // dx2-y2 (dxx)
197  local(7,5) = -local(7,4); // dx2-y2 (dyy)
198  local(8,7) = 1.0; // dxy
199  }
200  if ( lmax > 2 ){
201  std::cerr << " Gaussian input with f- functions or higher not yet supported!" << std::endl;
202  exit(1);
203  }
204  // now copy to _trafo
205  for ( int i_sph = 0 ; i_sph < NumFuncShell( type ) ; i_sph++ ){
206  for ( int i_cart = 0 ; i_cart < NumFuncShell_cartesian( type ) ; i_cart++ ){
207  submatrix( i_sph , i_cart ) = local( i_sph + OffsetFuncShell( type ) , i_cart + OffsetFuncShell_cartesian( type ) );
208  }
209  }
210  return;
211 }

361  void AOBasis::addReorderShell(const std::string& start, const std::string& target, const std::string& shell_type, std::vector<int>& order) {
362  //Reordering is given by email from gaussian, orca output MOs, and http://www.nwchem-sw.org/index.php/Release66:Basis for nwchem
363 
364  // current length of vector
365 
366  int cur_pos = order.size() - 1;
367 
368  if (target == "xtp") {
369  // single type shells defined here
370  if (shell_type.length() == 1) {
371  if (shell_type == "S") {
372  order.push_back(cur_pos + 1);
373  }//for S
374 
375  //votca order is z,y,x e.g. Y1,0 Y1,-1 Y1,1
376  else if (shell_type == "P") {
377  if (start == "orca") {
378  //orca order is z,x,y Y1,0,Y1,1,Y1,-1
379  order.push_back(cur_pos + 1);
380  order.push_back(cur_pos + 3);
381  order.push_back(cur_pos + 2);
382  } else if (start == "gaussian" || start == "nwchem") {
383  //nwchem gaussian x,y,z Y1,1 Y1,-1 Y1,0
384  order.push_back(cur_pos + 3);
385  order.push_back(cur_pos + 2);
386  order.push_back(cur_pos + 1);
387  } else if (start == "votca") {//for usage with old orb files
388  //old votca x,y,z Y1,1 Y1,-1 Y1,0
389  order.push_back(cur_pos + 3);
390  order.push_back(cur_pos + 2);
391  order.push_back(cur_pos + 1);
392  } else {
393  std::cerr << "Tried to reorder p-functions from package " << start << ".";
394  throw std::runtime_error("Reordering not implemented yet!");
395  }
396  }//for P
397  //votca order is d3z2-r2 dyz dxz dxy dx2-y2 e.g. Y2,0 Y2,-1 Y2,1 Y2,-2 Y2,2
398  else if (shell_type == "D") {
399  //orca order is d3z2-r2 dxz dyz dx2-y2 dxy e.g. Y2,0 Y2,1 Y2,-1 Y2,2 Y2,-2
400  if (start == "gaussian" || start == "orca") {
401  order.push_back(cur_pos + 1);
402  order.push_back(cur_pos + 3);
403  order.push_back(cur_pos + 2);
404  order.push_back(cur_pos + 5);
405  order.push_back(cur_pos + 4);
406  } else if (start == "nwchem") {
407  // nwchem order is dxy dyz d3z2-r2 -dxz dx2-y2, e.g. Y2,-2 Y2,-1 Y2,0 Y2,1 Y2,2
408  order.push_back(cur_pos + 4);
409  order.push_back(cur_pos + 2);
410  order.push_back(cur_pos + 1);
411  order.push_back(cur_pos + 3);
412  order.push_back(cur_pos + 5);
413  } else if (start == "votca") { //for usage with old orb files
414  order.push_back(cur_pos + 3);
415  order.push_back(cur_pos + 2);
416  order.push_back(cur_pos + 4);
417  order.push_back(cur_pos + 1);
418  order.push_back(cur_pos + 5);
419  } else {
420  std::cerr << "Tried to reorder d-functions from package " << start << ".";
421  throw std::runtime_error("Reordering not implemented yet!");
422  }
423  } else if (shell_type == "F") {
424  //ordering for votca is Yl,0 Yl,-1 Yl,1 ......Yl,-m Yl,m

426  //ordering for gaussian and orca is Yl,0 Yl,1 Yl,-1 ......Yl,m Yl,-m

428  order.push_back(cur_pos + 3);

431  order.push_back(cur_pos + 4);
432  order.push_back(cur_pos + 7);

434  } else if (start == "nwchem") {
435  //ordering for nwchem is fxxy-yyy, fxyz,fyzz-xxy-yyy,fzzz-xxz-yyz,f-xzz+xxx+xyy,fxxz-yyz,fxyy-xxx

438  order.push_back(cur_pos + 4);

497  void AOBasis::AOBasisFill(const BasisSet& bs, std::vector<QMAtom* >& atoms, int fragbreak) {
498  tools::Elements elementinfo;

500  _AOBasisFragA = 0;
501  _AOBasisFragB = 0;

504  for (QMAtom* atom : atoms) {
505  int atomfunc=0;
506  const std::string& name = atom->getType();
507  atom->_nuccharge = elementinfo.getNucCrg(name);

521  _FuncperAtom.push_back(atomfunc);
522  }
523 
524  if (fragbreak < 0) {
525  _AOBasisFragA = _AOBasisSize;
526  _AOBasisFragB = 0;
527  } else {
528  _AOBasisFragB = _AOBasisSize - _AOBasisFragA;
529  }

533  void AOBasis::ECPFill(const BasisSet& bs, std::vector<QMAtom* >& atoms) {
534  _FuncperAtom=std::vector<int>(0);
535  _AOBasisSize = 0;
536  for (QMAtom* atom : atoms) {
537  int atomfunc=0;
538  std::string name = atom->getType();
539  if (name == "H" || name == "He") {
540  _FuncperAtom.push_back(0);
541  continue;
542  }
543  const Element& element = bs.getElement(name);
544  atom->_ecpcharge = element.getNcore();
545  int lmax = element.getLmax();
546  for (const Shell& shell:element) {
547  if (shell.getType().size() > 1) {
548  throw std::runtime_error("In ecps no combined shells e.g. SP are allowed");
549  }
550  //Local part is with L=Lmax
551  bool nonlocal = false;
552  if (shell.getLmax() < lmax) {
553  nonlocal = true;
554  }
555  AOShell* aoshell = addECPShell(shell, *atom, _AOBasisSize, nonlocal);
556  _AOBasisSize += NumFuncShell(shell.getType());
557  atomfunc+=NumFuncShell(shell.getType());
558  for (const GaussianPrimitive& gaussian:shell) {
559  aoshell->addGaussian(gaussian);
560  }
561  aoshell->CalcMinDecay();
562  }
563  _FuncperAtom.push_back(atomfunc);
564  }
565  return;
