33 void AOBasis::ReorderMOs(Eigen::MatrixXd& v, const std::string& start,
34  const std::string& target) const {
35 
36  if (start == target) {
37  return;
38  }
39 
40  if (target == "orca" || target == "nwchem") {
41  std::vector<Index> multiplier = getMultiplierVector(target, start);
42  // and reorder rows of _orbitals->_mo_coefficients() accordingly
43  MultiplyMOs(v, multiplier);
44  }
45 
46  // get reordering vector _start -> target
47  std::vector<Index> order = getReorderVector(start, target);
48 
49  // Sanity check
50  if (v.rows() != Index(order.size())) {
51  throw std::runtime_error("Size mismatch in ReorderMOs " +
52  std::to_string(v.rows()) + ":" +
53  std::to_string(order.size()));
54  }
55 
56  // actual swapping of coefficients
57  for (Index s = 1, d; s < (Index)order.size(); ++s) {
58  for (d = order[s]; d < s; d = order[d]) {
59  ;
60  }
61  if (d == s) {
62  while (d = order[d], d != s) {
63  v.row(s).swap(v.row(d));
64  }
65  }
66  }
67  // NWChem has some strange minus in d-functions
68  if (start == "nwchem" || start == "orca") {
69  std::vector<Index> multiplier = getMultiplierVector(start, target);
70  // and reorder rows of _orbitals->_mo_coefficients() accordingly
71  MultiplyMOs(v, multiplier);
72  }
73  return;
74 }

76 void AOBasis::MultiplyMOs(Eigen::MatrixXd& v,
77  const std::vector<Index>& multiplier) const {
78  // Sanity check

80  std::cerr << "Size mismatch in MultiplyMOs" << v.cols() << ":"
81  << multiplier.size() << std::endl;
82  throw std::runtime_error("Abort!");
83  }
84  for (Index i_basis = 0; i_basis < v.cols(); i_basis++) {
85  v.row(i_basis) = multiplier[i_basis] * v.row(i_basis);
86  }

233 void AOBasis::addReorderShell(const std::string& start,
234  const std::string& target,
235  const std::string& shell_type,
236  std::vector<Index>& order) const {
237  // Reordering is given by email from gaussian, orca output MOs, and
238  // http://www.nwchem-sw.org/index.php/Release66:Basis for nwchem
239 
240  // current length of vector
241 
242  Index cur_pos = Index(order.size()) - 1;
243 
244  if (target == "xtp") {
245  // single type shells defined here
246  if (shell_type.length() == 1) {
247  if (shell_type == "S") {
248  order.push_back(cur_pos + 1);
249  } // for S
250 
251  // votca order is z,y,x e.g. Y1,0 Y1,-1 Y1,1
252  else if (shell_type == "P") {
253  if (start == "orca") {
254  // orca order is z,x,y Y1,0,Y1,1,Y1,-1
255  order.push_back(cur_pos + 1);
256  order.push_back(cur_pos + 3);
257  order.push_back(cur_pos + 2);
258  } else if (start == "gaussian" || start == "nwchem") {
259  // nwchem gaussian x,y,z Y1,1 Y1,-1 Y1,0
260  order.push_back(cur_pos + 3);
261  order.push_back(cur_pos + 2);
262  order.push_back(cur_pos + 1);
263  } else {
264  std::cerr << "Tried to reorder p-functions from package " << start
265  << ".";
266  throw std::runtime_error("Reordering not implemented yet!");
267  }
268  } // for P
269  // votca order is d3z2-r2 dyz dxz dxy dx2-y2 e.g. Y2,0 Y2,-1 Y2,1 Y2,-2
270  // Y2,2
271  else if (shell_type == "D") {
272  // orca order is d3z2-r2 dxz dyz dx2-y2 dxy e.g. Y2,0 Y2,1 Y2,-1 Y2,2
273  // Y2,-2
274  if (start == "gaussian" || start == "orca") {
275  order.push_back(cur_pos + 1);
276  order.push_back(cur_pos + 3);
277  order.push_back(cur_pos + 2);
278  order.push_back(cur_pos + 5);
279  order.push_back(cur_pos + 4);
280  } else if (start == "nwchem") {
281  // nwchem order is dxy dyz d3z2-r2 -dxz dx2-y2, e.g. Y2,-2 Y2,-1 Y2,0
282  // Y2,1 Y2,2
283  order.push_back(cur_pos + 4);
284  order.push_back(cur_pos + 2);
285  order.push_back(cur_pos + 1);
286  order.push_back(cur_pos + 3);
287  order.push_back(cur_pos + 5);
288  } else {
289  std::cerr << "Tried to reorder d-functions from package " << start
290  << ".";
291  throw std::runtime_error("Reordering not implemented yet!");
292  }
293  } else if (shell_type == "F") {
294  // ordering for votca is Yl,0 Yl,-1 Yl,1 ......Yl,-m Yl,m
295  if (start == "gaussian" || start == "orca") {
296  // ordering for gaussian and orca is Yl,0 Yl,1 Yl,-1 ......Yl,m Yl,-m
297  order.push_back(cur_pos + 1);
298  order.push_back(cur_pos + 3);
299  order.push_back(cur_pos + 2);
300  order.push_back(cur_pos + 5);
301  order.push_back(cur_pos + 4);
302  order.push_back(cur_pos + 7);
303  order.push_back(cur_pos + 6);
304  } else if (start == "nwchem") {
305  // ordering for nwchem is fxxy-yyy,
306  // fxyz,fyzz-xxy-yyy,fzzz-xxz-yyz,f-xzz+xxx+xyy,fxxz-yyz,fxyy-xxx
307  // e.g. Y3,-3 Y3,-2 Y3,-1 Y3,0 Y3,1 Y3,2 Y3,3
308  order.push_back(cur_pos + 6);
309  order.push_back(cur_pos + 4);
310  order.push_back(cur_pos + 2);
311  order.push_back(cur_pos + 1);
312  order.push_back(cur_pos + 3);
313  order.push_back(cur_pos + 5);
314  order.push_back(cur_pos + 7);
315  } else {
316  std::cerr << "Tried to reorder f-functions from package " << start
317  << ".";
318  throw std::runtime_error("Reordering not implemented yet!");
319  }
320  } else if (shell_type == "G") {
321  // ordering for votca is Yl,0 Yl,-1 Yl,1 ......Yl,-m Yl,m
322  if (start == "gaussian" || start == "orca") {
323  // ordering for gaussian and orca is Yl,0 Yl,1 Yl,-1 ......Yl,m Yl,-m
324  order.push_back(cur_pos + 1);
325  order.push_back(cur_pos + 3);
326  order.push_back(cur_pos + 2);
327  order.push_back(cur_pos + 5);
328  order.push_back(cur_pos + 4);
329  order.push_back(cur_pos + 7);
330  order.push_back(cur_pos + 6);
331  order.push_back(cur_pos + 9);
332  order.push_back(cur_pos + 8);
333  } else {
334  std::cerr << "Tried to reorder g-functions from package " << start
335  << ".";
336  throw std::runtime_error("Reordering not implemented");
337  }
338  } else {
339  std::cerr << "Tried to reorder functions of shell type " << shell_type
340  << std::endl;
341  throw std::runtime_error("Reordering not implemented");
342  }
343  } else {
344  // for combined shells, iterate over all contributions
345  //_nbf = 0;
346  for (Index i = 0; i < Index(shell_type.length()); ++i) {
347  std::string local_shell = std::string(shell_type, i, 1);
348  this->addReorderShell(start, target, local_shell, order);
349  }
350  }
351 
352  } else {
353  std::cerr << "Tried to reorder functions (neworder) from " << start
354  << " to " << target << std::endl;
355  throw std::runtime_error("Reordering not implemented yet!");
356  }
357  return;
358 }

370 void AOBasis::Fill(const BasisSet& bs, const QMMolecule& atoms) {
371  _AOBasisSize = 0;
372  _aoshells.clear();
373  _FuncperAtom.clear();
374  // loop over atoms
375  for (const QMAtom& atom : atoms) {
376  Index atomfunc = 0;
377  const std::string& name = atom.getElement();
378  const Element& element = bs.getElement(name);
379  for (const Shell& shell : element) {
380  Index numfuncshell = NumFuncShell(shell.getType());
381  AOShell& aoshell = addShell(shell, atom, _AOBasisSize);
382  _AOBasisSize += numfuncshell;
383  atomfunc += numfuncshell;
384  for (const GaussianPrimitive& gaussian : shell) {
385  aoshell.addGaussian(gaussian);
386  }
387  aoshell.CalcMinDecay();
388  aoshell.normalizeContraction();
389  }
390  _FuncperAtom.push_back(atomfunc);
391  }
392  return;
393 }
