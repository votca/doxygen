85 void AOBasis::Fill(const BasisSet& bs, const QMMolecule& atoms) {
86  clear();
87  name_ = bs.Name();
88  // loop over atoms
89  for (const QMAtom& atom : atoms) {
90  Index atomfunc = 0;
91  const std::string& name = atom.getElement();
92  const Element& element = bs.getElement(name);
93  for (const Shell& shell : element) {
94  Index numfuncshell = NumFuncShell(shell.getL());
95  AOShell& aoshell = addShell(shell, atom, AOBasisSize_);
96  AOBasisSize_ += numfuncshell;
97  atomfunc += numfuncshell;
98  for (const GaussianPrimitive& gaussian : shell) {
99  aoshell.addGaussian(gaussian);
100  }
101  aoshell.CalcMinDecay();
102  aoshell.normalizeContraction();
103  }
104  }
105  FillFuncperAtom();
106  return;
107 }

143 void AOBasis::WriteToCpt(CheckpointWriter& w) const {
144  w(name_, "name");
145  w(AOBasisSize_, "basissize");
146  Index numofprimitives = 0;
147  for (const auto& shell : aoshells_) {
148  numofprimitives += shell.getSize();
149  }
150 
151  CptTable table =
152  w.openTable<AOGaussianPrimitive>("Contractions", numofprimitives);
153 
154  std::vector<AOGaussianPrimitive::data> dataVec(numofprimitives);
155  Index i = 0;
156  for (const auto& shell : aoshells_) {
157  for (const auto& gaussian : shell) {
158  gaussian.WriteData(dataVec[i], shell);
159  i++;
160  }
161  }
162 
163  table.write(dataVec);
164 }

166 void AOBasis::ReadFromCpt(CheckpointReader& r) {
167  clear();
168  r(name_, "name");
169  r(AOBasisSize_, "basissize");
170  if (AOBasisSize_ > 0) {
171 
172  CptTable table = r.openTable<AOGaussianPrimitive>("Contractions");
173  std::vector<AOGaussianPrimitive::data> dataVec(table.numRows());
174  table.read(dataVec);
175  Index laststartindex = -1;
176  for (std::size_t i = 0; i < table.numRows(); ++i) {
177  if (dataVec[i].startindex != laststartindex) {
178  aoshells_.push_back(AOShell(dataVec[i]));
179  laststartindex = dataVec[i].startindex;
180  } else {
181  aoshells_.back().gaussians_.push_back(AOGaussianPrimitive(dataVec[i]));
182  }
183  }
184 
185  FillFuncperAtom();
186  }
187 }
