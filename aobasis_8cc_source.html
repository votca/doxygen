87 void AOBasis::Fill(const BasisSet& bs, const QMMolecule& atoms) {
88  clear();
89  _name = bs.Name();
90  // loop over atoms
91  for (const QMAtom& atom : atoms) {
92  Index atomfunc = 0;
93  const std::string& name = atom.getElement();
94  const Element& element = bs.getElement(name);
95  for (const Shell& shell : element) {
96  Index numfuncshell = NumFuncShell(shell.getL());
97  AOShell& aoshell = addShell(shell, atom, _AOBasisSize);
98  _AOBasisSize += numfuncshell;
99  atomfunc += numfuncshell;
100  for (const GaussianPrimitive& gaussian : shell) {
101  aoshell.addGaussian(gaussian);
102  }
103  aoshell.CalcMinDecay();
104  aoshell.normalizeContraction();
105  }
106  }
107  FillFuncperAtom();
108  return;
109 }

132 std::vector<std::vector<Index>> AOBasis::ComputeShellPairs(
133  double threshold) const {
134 
135  Index nthreads = OPENMP::getMaxThreads();
136 
137  std::vector<libint2::Shell> shells = GenerateLibintBasis();
138 
139  // construct the 2-electron repulsion integrals engine
140  std::vector<libint2::Engine> engines;
141  engines.reserve(nthreads);
142  engines.emplace_back(libint2::Operator::overlap, getMaxNprim(), getMaxL(), 0);
143  for (Index i = 1; i != nthreads; ++i) {
144  engines.push_back(engines[0]);
145  }
146 
147  std::vector<std::vector<Index>> pairs(shells.size());
148 
149 #pragma omp parallel for schedule(dynamic)
150  for (Index s1 = 0; s1 < Index(shells.size()); ++s1) {
151  Index thread_id = OPENMP::getThreadId();
152 
153  libint2::Engine& engine = engines[thread_id];
154  const libint2::Engine::target_ptr_vec& buf = engine.results();
155  Index n1 = shells[s1].size();
156 
157  for (Index s2 = 0; s2 <= s1; ++s2) {
158  bool on_same_center = (shells[s1].O == shells[s2].O);
159  bool significant = on_same_center;
160  if (!on_same_center) {
161  Index n2 = shells[s2].size();
162  engine.compute(shells[s1], shells[s2]);
163  Eigen::Map<const Eigen::MatrixXd> buf_mat(buf[0], n1, n2);
164  significant = (buf_mat.norm() >= threshold);
165  }
166  if (significant) {
167  pairs[s1].push_back(s2);
168  }
169  }
170  std::sort(pairs[s1].begin(), pairs[s1].end());
171  }
172  return pairs;
173 }

188 void AOBasis::WriteToCpt(CheckpointWriter& w) const {
189  w(_name, "name");
190  w(_AOBasisSize, "basissize");
191  Index numofprimitives = 0;
192  for (const auto& shell : _aoshells) {
193  numofprimitives += shell.getSize();
194  }
195 
196  // this is all to make dummy AOGaussian
197  Shell s(L::S, 0);
198  GaussianPrimitive d(0.1, 0.1);
199  QMAtom dummy(0, "H", Eigen::Vector3d::Zero());
200  AOShell s1(s, dummy, 0);
201  s1.addGaussian(d);
202  const AOGaussianPrimitive& dummy2 = *s1.begin();
203 
204  CptTable table = w.openTable("Contractions", dummy2, numofprimitives);
205 
206  std::vector<AOGaussianPrimitive::data> dataVec(numofprimitives);
207  Index i = 0;
208  for (const auto& shell : _aoshells) {
209  for (const auto& gaussian : shell) {
210  gaussian.WriteData(dataVec[i]);
211  i++;
212  }
213  }
214 
215  table.write(dataVec);

218 void AOBasis::ReadFromCpt(CheckpointReader& r) {
219  clear();
220  r(_name, "name");
221  r(_AOBasisSize, "basissize");
222  if (_AOBasisSize > 0) {
223  // this is all to make dummy AOGaussian
224  Shell s(L::S, 0);
225  GaussianPrimitive d(0.1, 0.1);
226  QMAtom dummy(0, "H", Eigen::Vector3d::Zero());
227  AOShell s1(s, dummy, 0);
228  s1.addGaussian(d);
229  const AOGaussianPrimitive& dummy2 = *s1.begin();
230 
231  CptTable table = r.openTable("Contractions", dummy2);
232  std::vector<AOGaussianPrimitive::data> dataVec(table.numRows());
233  table.read(dataVec);
234  Index laststartindex = -1;
235  for (std::size_t i = 0; i < table.numRows(); ++i) {
236  if (dataVec[i].startindex != laststartindex) {
237  _aoshells.push_back(AOShell(dataVec[i]));
238  laststartindex = dataVec[i].startindex;
239  } else {
240  _aoshells.back()._gaussians.push_back(
241  AOGaussianPrimitive(dataVec[i], _aoshells.back()));
242  }
243  }
244 
245  FillFuncperAtom();
246  }
