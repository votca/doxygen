44 void AOBasis::Fill(const BasisSet& bs, const QMMolecule& atoms) {
45  _AOBasisSize = 0;
46  _aoshells.clear();
47  _FuncperAtom.clear();
48  // loop over atoms
49  for (const QMAtom& atom : atoms) {
50  Index atomfunc = 0;
51  const std::string& name = atom.getElement();
52  const Element& element = bs.getElement(name);
53  for (const Shell& shell : element) {
54  Index numfuncshell = NumFuncShell(shell.getType());
55  AOShell& aoshell = addShell(shell, atom, _AOBasisSize);
56  _AOBasisSize += numfuncshell;
57  atomfunc += numfuncshell;
58  for (const GaussianPrimitive& gaussian : shell) {
59  aoshell.addGaussian(gaussian);
60  }
61  aoshell.CalcMinDecay();
62  aoshell.normalizeContraction();
63  }
64  _FuncperAtom.push_back(atomfunc);
65  }
66  return;
67 }
