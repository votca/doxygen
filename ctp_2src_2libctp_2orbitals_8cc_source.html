150 bool Orbitals::CheckDegeneracy( double _energy_difference ) {
151 
152  ub::vector<double>::iterator it1 = _mo_energies.begin();
153  bool _degenerate = false;
154 
155  if ( tools::globals::verbose ) cout << endl <<"... ... Checking level degeneracy " << endl;
156 
157  _level_degeneracy.clear();
158 
159  while ( it1 !=_mo_energies.end() ) {
160 
161  // in all containers counters start with 0; real life - with 1
162  int _level1 = std::distance(_mo_energies.begin(), it1) + 1;
163 
164  // add the level itself - it is easier to loo over all levels later
165  _level_degeneracy[_level1].push_back( _level1 );
166 
167  ub::vector<double>::iterator it2 = it1;
168  it2++;
169 
170  while ( it2 !=_mo_energies.end() ) {
171  //cout << _level1 << ":" << *it1 << ":" << *it2 << endl;
172  double energy1 = *it1;
173  double energy2 = *it2;
174 
175  // in all containers counters start with 0; real life - with 1
176  int _level2 = std::distance(_mo_energies.begin(), it2) + 1;
177 
178  if ( std::abs(energy1 - energy2)*tools::globals::conversion::Hrt2eV < _energy_difference ) {
179  _level_degeneracy[_level1].push_back( _level2 );
180  _level_degeneracy[_level2].push_back( _level1 );
181  _degenerate = true;
182  }
183  it2++;
184  }
185  it1++;
186  }
187 
188  if ( tools::globals::verbose ){
189 
190  if ( _degenerate ) {
191  cout << "... ... Some levels are degenerate" << endl;
192  for (std::map<int, std::vector<int> >::iterator it = _level_degeneracy.begin();
193  it != _level_degeneracy.end();
194  ++it) {
195  // output only degenerate levels
196  if ( (it->second).size() > 1 ) {
197  std::cout << "... ... level "<< it->first << " : ";
198  for (vector<int>::iterator lev = (it->second).begin(); lev != (it->second).end(); lev++)
199  cout << *lev << " ";
200  cout << endl;
201  }
202  }
203  } else {
204  cout << "... ... No degeneracy found" << endl;
205  }
206 
207  cout << "... ... Done checking level degeneracy" << endl;
208 
209  }
210 
211  // _has_level_degeneracy = true;
212  return _degenerate;
213 
214 }
