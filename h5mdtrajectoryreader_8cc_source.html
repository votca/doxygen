40 bool H5MDTrajectoryReader::Open(const string &file) {
41  // Checks if we deal with hdf5 file.
42  if (!H5Fis_hdf5(file.c_str())) {
43  cout << file << " is not recognise as HDF5 file format" << endl;
44  return false;
45  }
46  file_id_ = H5Fopen(file.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
47  file_opened_ = true;
48 
49  // Check the version of the file.
50  hid_t g_h5md = H5Gopen(file_id_, "h5md", H5P_DEFAULT);
51  CheckError(g_h5md, "Unable to open /h5md group.");
52 
53  hid_t at_version = H5Aopen(g_h5md, "version", H5P_DEFAULT);
54  CheckError(at_version, "Unable to read version attribute.");
55  int version[2];
56  H5Aread(at_version, H5Aget_type(at_version), &version);
57  if (version[0] != 1 || (version[0] == 1 && version[1] > 0)) {
58  cout << "Major version " << version[0] << endl;
59  cout << "Minor version " << version[1] << endl;
60  throw ios_base::failure("Wrong version of H5MD file.");
61  }
62 
63  // Clean up.
64  H5Aclose(at_version);
65  H5Gclose(g_h5md);
66 
67  // Checks if the particles group exists and what is the number of members.
68  particle_group_ = H5Gopen(file_id_, "particles", H5P_DEFAULT);
69  CheckError(particle_group_, "Unable to open /particles group.");
70  hsize_t num_obj = 0;
71  H5Gget_num_objs(particle_group_, &num_obj);
72  if (num_obj == 0) {
73  throw ios_base::failure("The particles group is empty.");
74  }
75 
76  first_frame_ = true;
77 
78  // Handle errors by internal check up.
79  H5Eset_auto2(H5E_DEFAULT, NULL, NULL);
80 
81  return true;
82 }

91 void H5MDTrajectoryReader::Initialize(Topology &top) {
92  string *particle_group_name_ = new string(top.getParticleGroup());
93  if(*particle_group_name_ == "")
94  throw ios_base::failure(
95  "Missing particle group in topology. Please set `h5md_particle_group` tag with `name` attribute set to the particle group.");
96  string *position_group_name = new string(*particle_group_name_ + "/position");
97  atom_position_group_ = H5Gopen(particle_group_, position_group_name->c_str(), H5P_DEFAULT);
98  CheckError(atom_position_group_, "Unable to open " + *position_group_name + " group");
99 
100  idx_frame_ = -1;
101  ds_atom_position_ = H5Dopen(atom_position_group_, "value", H5P_DEFAULT);
102  CheckError(ds_atom_position_, "Unable to open " + *position_group_name + "/value dataset");
103 
104  // Reads the box information.
105  string *box_gr_name = new string(*particle_group_name_ + "/box");
106  hid_t g_box = H5Gopen(particle_group_, box_gr_name->c_str(), H5P_DEFAULT);
107  CheckError(g_box, "Unable to open " + *box_gr_name + " group");
108  hid_t at_box_dimension = H5Aopen(g_box, "dimension", H5P_DEFAULT);
109  CheckError(at_box_dimension, "Unable to open dimension attribute.");
110  int dimension;
111  H5Aread(at_box_dimension, H5Aget_type(at_box_dimension), &dimension);
112  if (dimension != 3 ) {
113  throw ios_base::failure(
114  "Wrong dimension " + boost::lexical_cast<string>(dimension));
115  }
116  // TODO: check if boundary is periodic.
117  string *box_edges_name = new string(*particle_group_name_ + "/box/edges");
118  if (GroupExists(particle_group_, *box_edges_name)) {
119  g_box = H5Gopen(particle_group_, box_gr_name->c_str(), H5P_DEFAULT);
120  edges_group_ = H5Gopen(g_box, "edges", H5P_DEFAULT);
121  ds_edges_group_ = H5Dopen(edges_group_, "value", H5P_DEFAULT);
122  cout << "H5MD: has /box/edges" << endl;
123  has_box_ = H5MDTrajectoryReader::TIMEDEPENDENT;
124  } else {
125  cout << "H5MD: static box" << endl;
126  hid_t ds_edges = H5Dopen(g_box, "edges", H5P_DEFAULT);
127  CheckError(ds_edges, "Unable to open /box/edges");
128  double *box = new double[3];
129  ReadStaticData(ds_edges, H5T_NATIVE_DOUBLE, box);
130  cout << "H5MD: Found box " << box[0] << " x " << box[1] <<
131  " x " << box[2] << endl;
132  // Sets box size.
133  m.ZeroMatrix();
134  m[0][0] = box[0];
135  m[1][1] = box[1];
136  m[2][2] = box[2];
137  top.setBox(m);
138  has_box_ = H5MDTrajectoryReader::STATIC;
139  }
140  H5Gclose(g_box);
141 
142  // Gets the force group.
143  string *force_group_name = new string(*particle_group_name_ + "/force");
144  if (GroupExists(particle_group_, *force_group_name)) {
145  atom_force_group_ = H5Gopen(particle_group_, force_group_name->c_str(), H5P_DEFAULT);
146  ds_atom_force_ = H5Dopen(atom_force_group_, "value", H5P_DEFAULT);
147  has_force_ = H5MDTrajectoryReader::TIMEDEPENDENT;
148  cout << "H5MD: has /force" << endl;
149  } else {
150  has_force_ = H5MDTrajectoryReader::NONE;
151  }
152 
153  // Gets the velocity group.
154  string *velocity_group_name = new string(*particle_group_name_ + "/velocity");
155  if (GroupExists(particle_group_, *velocity_group_name)) {
156  atom_velocity_group_ = H5Gopen(particle_group_,
157  velocity_group_name->c_str(),
158  H5P_DEFAULT);
159  ds_atom_velocity_ = H5Dopen(atom_velocity_group_, "value", H5P_DEFAULT);
160  has_velocity_ = H5MDTrajectoryReader::TIMEDEPENDENT;
161  cout << "H5MD: has /velocity" << endl;
162  } else {
163  has_velocity_ = H5MDTrajectoryReader::NONE;
164  }
165 
166  // Gets the id group so that the atom id is taken from this group.
167  string *id_group_name = new string(*particle_group_name_ + "/id");
168  if (GroupExists(particle_group_, *id_group_name)) {
169  atom_id_group_ = H5Gopen(particle_group_, id_group_name->c_str(), H5P_DEFAULT);
170  ds_atom_id_ = H5Dopen(atom_id_group_, "value", H5P_DEFAULT);
171  has_id_group_ = H5MDTrajectoryReader::TIMEDEPENDENT;
172  cout << "H5MD: has /id group" << endl;
173  } else {
174  has_id_group_ = H5MDTrajectoryReader::NONE;
175  }
176 
177  // Gets number of particles and dimensions.
178  hid_t fs_atom_position_ = H5Dget_space(ds_atom_position_);
179  CheckError(fs_atom_position_, "Unable to open atom position space.");
180  hsize_t dims[3];
181  rank_ = H5Sget_simple_extent_dims(fs_atom_position_, dims, NULL);
182  N_particles_ = dims[1];
183  vec_components_ = dims[2];
184  max_idx_frame_ = dims[0] - 1;
185 
186  // TODO: reads mass, charge and particle type.
187 
188  if (has_id_group_ == H5MDTrajectoryReader::NONE && top.BeadCount() > 0 && N_particles_ != top.BeadCount()) {
189  cout << "Warning: The number of beads (" << N_particles_ << ")";
190  cout << " in the trajectory is different than defined in the topology ("
191  << top.BeadCount() << ")" << endl;
192  cout << "The number of beads from topology will be used!" << endl;
193  N_particles_ = top.BeadCount();
194  }
195 
196  delete id_group_name;
197  delete velocity_group_name;
198  delete force_group_name;
199  delete box_edges_name;
200  delete particle_group_name_;
201 }

213 bool H5MDTrajectoryReader::NextFrame(Topology &top) { // NOLINT const reference
214  // Reads the position row.
215  idx_frame_++;
216  if (idx_frame_ > max_idx_frame_)
217  return false;
218 
219  // Set volume of box because top on workers somehow does not have this information.
220  top.setBox(m);
221 
222  cout << '\r' << "Reading frame: " << idx_frame_;
223  cout.flush();
224  double *positions;
225  double *forces = NULL;
226  double *velocities = NULL;
227  int *ids = NULL;
228 
229  try {
230  positions = ReadVectorData<double>(ds_atom_position_, H5T_NATIVE_DOUBLE, idx_frame_);
231  } catch (const runtime_error &e) {
232  return false;
233  }
234 
235  if (has_velocity_ != H5MDTrajectoryReader::NONE) {
236  velocities = ReadVectorData<double>(ds_atom_velocity_, H5T_NATIVE_DOUBLE, idx_frame_);
237  }
238 
239  if (has_force_ != H5MDTrajectoryReader::NONE) {
240  forces = ReadVectorData<double>(ds_atom_force_, H5T_NATIVE_DOUBLE, idx_frame_);
241  }
242 
243  if (has_id_group_ != H5MDTrajectoryReader::NONE) {
244  ids = ReadScalarData<int>(ds_atom_id_, H5T_NATIVE_INT, idx_frame_);
245  }
246 
247  //Process atoms.
248  for (int at_idx = 0; at_idx < N_particles_; at_idx++) {
249  double x, y, z;
250  int array_index = at_idx*vec_components_;
251  x = positions[array_index];
252  y = positions[array_index + 1];
253  z = positions[array_index + 2];
254  // Set atom id, or it is an index of a row in dataset or from id dataset.
255  int atom_id = at_idx;
256  if (has_id_group_ != H5MDTrajectoryReader::NONE) {
257  if (ids[at_idx] == -1) // ignore values where id == -1
258  continue;
259  atom_id = ids[at_idx] - 1;
260  }
261 
262  // Topology has to be defined in the xml file or in other
263  // topology files. The h5md only stores the trajectory data.
264  Bead *b = top.getBead(atom_id);
265  if (b == NULL)
266  throw runtime_error("Bead not found: " + boost::lexical_cast<string>(atom_id));
267 
268  b->setPos(vec(x, y, z));
269  if (has_velocity_ == H5MDTrajectoryReader::TIMEDEPENDENT) {
270  double vx, vy, vz;
271  vx = velocities[array_index];
272  vy = velocities[array_index + 1];
273  vz = velocities[array_index + 2];
274  b->setVel(vec(vx, vy, vz));
275  }
276 
277  if (has_force_ == H5MDTrajectoryReader::TIMEDEPENDENT) {
278  double fx, fy, fz;
279  fx = forces[array_index];
280  fy = forces[array_index + 1];
281  fz = forces[array_index + 2];
282  b->setF(vec(fx, fy, fz));
283  }
284  }
285 
286  // Clean up pointers.
287  delete[] positions;
288  if (has_force_ == H5MDTrajectoryReader::TIMEDEPENDENT)
289  delete[] forces;
290  if (has_velocity_ == H5MDTrajectoryReader::TIMEDEPENDENT)
291  delete[] velocities;
292  if (has_id_group_ == H5MDTrajectoryReader::TIMEDEPENDENT)
293  delete[] ids;
294 
295  return true;
296 }
