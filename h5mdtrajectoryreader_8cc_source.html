56 bool H5MDTrajectoryReader::Open(const string &file) {
57  // Turn on exceptions instead of various messages.
58  H5::Exception::dontPrint();
59 
60  // Checks if we deal with hdf5 file.
61  if (! H5::H5File::isHdf5(file) )
62  throw std::ios_base::failure("Error on open trajectory file: " + file);
63 
64  h5file_ = new H5::H5File(file, H5F_ACC_RDONLY);
65 
66  // Check the version of the file.
67  H5::Group *g_h5md = new H5::Group(h5file_->openGroup("h5md"));
68  H5::Attribute *at_version = new H5::Attribute(g_h5md->openAttribute("version"));
69  int version[2];
70  at_version->read(at_version->getDataType(), &version);
71  if (version[0] != 1 || (version[0] == 1 && version[1] > 0)) {
72  h5file_->close();
73  std::cout << "Major version " << version[0] << std::endl;
74  std::cout << "Minor version " << version[1] << std::endl;
75  throw std::ios_base::failure("Wrong version of H5MD file.");
76  }
77 
78  // Clean up.
79  g_h5md->close();
80  delete at_version;
81  delete g_h5md;
82 
83  // Checks if the particles group exists and what is the number of members.
84  particle_group_ = new H5::Group(h5file_->openGroup("particles"));
85  if (particle_group_->getNumObjs() == 0) {
86  h5file_->close();
87  throw std::ios_base::failure("The particles group is empty.");
88  }
89 
90  has_velocity_ = false;
91  has_force_ = false;
92  has_static_box_ = true;
93  first_frame_ = true;
94 
95  return true;
96 }

102 bool H5MDTrajectoryReader::FirstFrame(Topology &top) { // NOLINT const reference
103  // First frame has access to topology so here we will check if the particle_group exists.
104  if (first_frame_) {
105  first_frame_ = false;
106 
107  std::string *particle_group_name = new std::string(top.getParticleGroup());
108  if(*particle_group_name == "")
109  throw std::ios_base::failure("Missing particle group in topology. Please set h5md_particle_group tag with name \
110  attribute set to the particle group.");
111 
112  particle_group_ = new H5::Group(h5file_->openGroup("particles"));
113  try {
114  atom_position_group_ = new H5::Group(particle_group_->openGroup(
115  *particle_group_name + "/position"));
116  } catch (H5::GroupIException not_found_error) {
117  std::cout << "Error: The path '/particles/" << *particle_group_name << "/position' not found. Please check the name ";
118  std::cout << "of particle group defined in topology" << std::endl;
119  throw not_found_error;
120  }
121 
122  idx_frame_ = -1;
123 
124  ds_atom_position_ = new H5::DataSet(atom_position_group_->openDataSet("value"));
125  ds_step_ = new H5::DataSet(atom_position_group_->openDataSet("step"));
126  ds_time_ = new H5::DataSet(atom_position_group_->openDataSet("time"));
127 
128  // Gets number of particles and dimensions.
129  H5::DataSpace fs_atom_position_ = ds_atom_position_->getSpace();
130  hsize_t dims[3];
131  rank_ = fs_atom_position_.getSimpleExtentDims(dims);
132  N_particles_ = dims[1];
133  variables_ = dims[2];
134 
135  if(N_particles_ != top.BeadCount()) {
136  std::cout << "Warning: The number of beads (" << N_particles_ << ")";
137  std::cout << " in the trajectory is different than defined in the topology (" << top.BeadCount() << ")" << std::endl;
138  std::cout << "The number of beads from topology will be used!" << std::endl;
139  N_particles_ = top.BeadCount();
140  }
141  chunk_rows_[0] = 1;
142  chunk_rows_[1] = N_particles_;
143  chunk_rows_[2] = variables_;
144 
145  // Reads the time set and step set.
146  H5::DataSet *ds_pos_step = new H5::DataSet(atom_position_group_->openDataSet("step"));
147  H5::DataSet *ds_pos_time = new H5::DataSet(atom_position_group_->openDataSet("time"));
148  H5::DataSpace dsp_step = H5::DataSpace(ds_pos_step->getSpace());
149  H5::DataSpace dsp_time = H5::DataSpace(ds_pos_time->getSpace());
150  hsize_t step_dims[3];
151  dsp_step.getSimpleExtentDims(step_dims);
152  time_set_ = new double[step_dims[0]];
153  step_set_ = new int[step_dims[0]];
154  ds_pos_step->read(step_set_, H5::PredType::NATIVE_INT, dsp_step, dsp_step);
155  ds_pos_time->read(time_set_, H5::PredType::NATIVE_DOUBLE, dsp_time, dsp_time);
156 
157  delete ds_pos_step;
158  delete ds_pos_time;
159 
160  // Reads the box information.
161  H5::Group g_box = H5::Group(particle_group_->openGroup(
162  *particle_group_name + "/box"));
163 
164  H5::Attribute at_box_dimension = H5::Attribute(g_box.openAttribute("dimension"));
165  int dimension;
166  at_box_dimension.read(at_box_dimension.getDataType(), &dimension);
167  if (dimension != 3 ) {
168  throw std::ios_base::failure("Wrong dimension " + boost::lexical_cast<string>(dimension));
169  }
170  // TODO: check also attribute periodic
171  try {
172  edges_group_ = new H5::Group(g_box.openGroup("edges"));
173  has_static_box_ = false;
174  throw std::runtime_error("The box is not static during the simulation. Is it NVT ensemble?");
175  } catch (H5::GroupIException not_found_error) {
176  ds_edges_ = new H5::DataSet(g_box.openDataSet("edges"));
177  has_static_box_ = true;
178  }
179 
180  // Gets the force group.
181  try {
182  std::string force_group_name = *particle_group_name + "/force";
183  atom_force_group_ = new H5::Group(particle_group_->openGroup(force_group_name));
184  ds_atom_force_ = new H5::DataSet(atom_force_group_->openDataSet("value"));
185  } catch (H5::GroupIException not_found) {
186  has_force_ = false;
187  }
188  // Gets the velocity group.
189  try {
190  std::string velocity_group_name = *particle_group_name + "/velocity";
191  atom_velocity_group_ = new H5::Group(particle_group_->openGroup(velocity_group_name));
192  ds_atom_velocity_ = new H5::DataSet(atom_velocity_group_->openDataSet("value"));
193  } catch (H5::GroupIException not_found) {
194  has_velocity_ = false;
195  }
196  delete particle_group_name;
197  }
198  NextFrame(top);
199  return true;
200 }

202 double* H5MDTrajectoryReader::ReadData(H5::DataSet *ds, int row){
203  hsize_t offset[3] = {row, 0, 0};
204  H5::DataSpace dsp = H5::DataSpace(ds->getSpace());
205  dsp.selectHyperslab(H5S_SELECT_SET, chunk_rows_, offset);
206  H5::DataSpace mspace1(variables_, chunk_rows_);
207  double *data_out = new double[N_particles_ * variables_];
208  ds->read(data_out, H5::PredType::NATIVE_DOUBLE, mspace1, dsp);
209  return data_out;
210 }

213 bool H5MDTrajectoryReader::NextFrame(Topology &top) { // NOLINT const reference
214  // Reads the position row.
215  H5::Exception::dontPrint();
216  idx_frame_++;
217  std::cout << '\r' << "Reading frame: " << idx_frame_;
218  std::cout.flush();
219  hsize_t offset[3] = {idx_frame_, 0, 0};
220  double *positions;
221  try {
222  positions = ReadData(ds_atom_position_, idx_frame_);
223  } catch (H5::DataSetIException not_found) {
224  return false;
225  }
226 
227  double *forces = NULL;
228  double *velocities = NULL;
229  if (has_velocity_) {
230  velocities = ReadData(ds_atom_velocity_, idx_frame_);
231  }
232 
233  if (has_force_) {
234  forces = ReadData(ds_atom_force_, idx_frame_);
235  }
236 
237  //Process atoms.
238  for (int at_idx = 0; at_idx < N_particles_; at_idx++) {
239  double x, y, z;
240  int array_index = at_idx*variables_;
241  x = positions[array_index];
242  y = positions[array_index + 1];
243  z = positions[array_index + 2];
244  // Topology has to be defined in the xml file or in other
245  // topology files. The h5md only stores the trajectory data.
246  Bead *b = top.getBead(at_idx);
247  b->setPos(vec(x, y, z));
248  if (has_velocity_) {
249  double vx, vy, vz;
250  vx = velocities[array_index];
251  vy = velocities[array_index + 1];
252  vz = velocities[array_index + 2];
253  b->setVel(vec(vx, vy, vz));
254  }
255 
256  if (has_force_) {
257  double fx, fy, fz;
258  fx = forces[array_index];
259  fy = forces[array_index + 1];
260  fz = forces[array_index + 2];
261  b->setF(vec(fx, fy, fz));
262  }
263  }
264 
265  // Clean up pointers.
266  delete[] positions;
267  delete[] forces;
268  delete[] velocities;
269 
270  return true;
271 }
