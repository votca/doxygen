00040 bool H5MDTrajectoryReader::Open(const string &file) {
00041   // Checks if we deal with hdf5 file.
00042   if (!H5Fis_hdf5(file.c_str())) {
00043     cout << file << " is not recognise as HDF5 file format" << endl;
00044     return false;
00045   }
00046   file_id_ = H5Fopen(file.c_str(), H5F_ACC_RDONLY, H5P_DEFAULT);
00047   file_opened_ = true;
00048 
00049   // Check the version of the file.
00050   hid_t g_h5md = H5Gopen(file_id_, "h5md", H5P_DEFAULT);
00051   CheckError(g_h5md, "Unable to open /h5md group.");
00052 
00053   hid_t at_version = H5Aopen(g_h5md, "version", H5P_DEFAULT);
00054   CheckError(at_version, "Unable to read version attribute.");
00055   int version[2];
00056   H5Aread(at_version, H5Aget_type(at_version), &version);
00057   if (version[0] != 1 || (version[0] == 1 && version[1] > 0)) {
00058     cout << "Major version " << version[0] << endl;
00059     cout << "Minor version " << version[1] << endl;
00060     throw ios_base::failure("Wrong version of H5MD file.");
00061   }
00062 
00063   // Clean up.
00064   H5Aclose(at_version);
00065   H5Gclose(g_h5md);
00066 
00067   // Checks if the particles group exists and what is the number of members.
00068   particle_group_ = H5Gopen(file_id_, "particles", H5P_DEFAULT);
00069   CheckError(particle_group_, "Unable to open /particles group.");
00070   hsize_t num_obj = 0;
00071   H5Gget_num_objs(particle_group_, &num_obj);
00072   if (num_obj == 0) {
00073     throw ios_base::failure("The particles group is empty.");
00074   }
00075 
00076   first_frame_ = true;
00077 
00078   // Handle errors by internal check up.
00079   H5Eset_auto2(H5E_DEFAULT, NULL, NULL);
00080 
00081   return true;
00082 }

00213 bool H5MDTrajectoryReader::NextFrame(Topology &top) {  // NOLINT const reference
00214   // Reads the position row.
00215   idx_frame_++;
00216   if (idx_frame_ > max_idx_frame_)
00217     return false;
00218 
00219   // Set volume of box because top on workers somehow does not have this information.
00220   top.setBox(m);
00221 
00222   cout << '\r' << "Reading frame: " << idx_frame_;
00223   cout.flush();
00224   double *positions;
00225   double *forces = NULL;
00226   double *velocities = NULL;
00227   int *ids = NULL;
00228 
00229   try {
00230     positions = ReadVectorData<double>(ds_atom_position_, H5T_NATIVE_DOUBLE, idx_frame_);
00231   } catch (const runtime_error &e) {
00232     return false;
00233   }
00234 
00235   if (has_velocity_ != H5MDTrajectoryReader::NONE) {
00236     velocities = ReadVectorData<double>(ds_atom_velocity_, H5T_NATIVE_DOUBLE, idx_frame_);
00237   }
00238 
00239   if (has_force_ != H5MDTrajectoryReader::NONE) {
00240     forces = ReadVectorData<double>(ds_atom_force_, H5T_NATIVE_DOUBLE, idx_frame_);
00241   }
00242 
00243   if (has_id_group_ != H5MDTrajectoryReader::NONE) {
00244     ids = ReadScalarData<int>(ds_atom_id_, H5T_NATIVE_INT, idx_frame_);
00245   }
00246 
00247   //Process atoms.
00248   for (int at_idx = 0; at_idx < N_particles_; at_idx++) {
00249     double x, y, z;
00250     int array_index = at_idx*vec_components_;
00251     x = positions[array_index];
00252     y = positions[array_index + 1];
00253     z = positions[array_index + 2];
00254     // Set atom id, or it is an index of a row in dataset or from id dataset.
00255     int atom_id = at_idx;
00256     if (has_id_group_ != H5MDTrajectoryReader::NONE) {
00257       if (ids[at_idx] == -1)  // ignore values where id == -1
00258         continue;
00259       atom_id = ids[at_idx] - 1;
00260     }
00261 
00262     // Topology has to be defined in the xml file or in other
00263     // topology files. The h5md only stores the trajectory data.
00264     Bead *b = top.getBead(atom_id);
00265     if (b == NULL)
00266       throw runtime_error("Bead not found: " + boost::lexical_cast<string>(atom_id));
00267 
00268     b->setPos(vec(x, y, z));
00269     if (has_velocity_ == H5MDTrajectoryReader::TIMEDEPENDENT) {
00270       double vx, vy, vz;
00271       vx = velocities[array_index];
00272       vy = velocities[array_index + 1];
00273       vz = velocities[array_index + 2];
00274       b->setVel(vec(vx, vy, vz));
00275     }
00276 
00277     if (has_force_ == H5MDTrajectoryReader::TIMEDEPENDENT) {
00278       double fx, fy, fz;
00279       fx = forces[array_index];
00280       fy = forces[array_index + 1];
00281       fz = forces[array_index + 2];
00282       b->setF(vec(fx, fy, fz));
00283     }
00284   }
00285 
00286   // Clean up pointers.
00287   delete[] positions;

00289     delete[] forces;

00292   if (has_id_group_ == H5MDTrajectoryReader::TIMEDEPENDENT)
