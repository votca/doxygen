59 bool H5MDTrajectoryReader::Open(const string &file) {
60  // Turn on exceptions instead of various messages.
61  H5::Exception::dontPrint();
62 
63  // Checks if we deal with hdf5 file.
64  if (! H5::H5File::isHdf5(file) )
65  throw std::ios_base::failure("Error on open trajectory file: " + file);
66 
67  h5file_ = new H5::H5File(file, H5F_ACC_RDONLY);
68 
69  // Check the version of the file.
70  H5::Group *g_h5md = new H5::Group(h5file_->openGroup("h5md"));
71  H5::Attribute *at_version = new H5::Attribute(g_h5md->openAttribute("version"));
72  int version[2];
73  at_version->read(at_version->getDataType(), &version);
74  if (version[0] != 1 || (version[0] == 1 && version[1] > 0)) {
75  h5file_->close();
76  std::cout << "Major version " << version[0] << std::endl;
77  std::cout << "Minor version " << version[1] << std::endl;
78  throw std::ios_base::failure("Wrong version of H5MD file.");
79  }
80 
81  // Clean up.
82  g_h5md->close();
83  delete at_version;
84  delete g_h5md;
85 
86  // Checks if the particles group exists and what is the number of members.
87  particle_group_ = new H5::Group(h5file_->openGroup("particles"));
88  if (particle_group_->getNumObjs() == 0) {
89  h5file_->close();
90  throw std::ios_base::failure("The particles group is empty.");
91  }
92 
93  has_velocity_ = false;
94  has_force_ = false;
95  first_frame_ = true;
96 
97  return true;
98 }

104 bool H5MDTrajectoryReader::FirstFrame(Topology &top) { // NOLINT const reference
105  // First frame has access to topology so here we will check if the particle_group exists.
106  if (first_frame_) {
107  first_frame_ = false;
108 
109  std::string *particle_group_name = new std::string(top.getParticleGroup());
110  if(*particle_group_name == "")
111  throw std::ios_base::failure("Missing particle group in topology. Please set h5md_particle_group tag with name \
112  attribute set to the particle group.");
113 
114  //particle_group_ = new H5::Group(h5file_->openGroup("particles"));
115  try {
116  atom_position_group_ = new H5::Group(particle_group_->openGroup(
117  *particle_group_name + "/position"));
118  } catch (H5::GroupIException not_found_error) {
119  std::cout << "Error: The path '/particles/" << *particle_group_name << "/position' not found. Please check the name ";
120  std::cout << "of particle group defined in topology" << std::endl;
121  throw not_found_error;
122  }
123 
124  idx_frame_ = -1;
125 
126  ds_atom_position_ = new H5::DataSet(atom_position_group_->openDataSet("value"));
127  ds_step_ = new H5::DataSet(atom_position_group_->openDataSet("step"));
128  ds_time_ = new H5::DataSet(atom_position_group_->openDataSet("time"));
129 
130  // Reads the time set and step set.
131  H5::DataSet *ds_pos_step = new H5::DataSet(atom_position_group_->openDataSet("step"));
132  H5::DataSet *ds_pos_time = new H5::DataSet(atom_position_group_->openDataSet("time"));
133  H5::DataSpace dsp_step = H5::DataSpace(ds_pos_step->getSpace());
134  H5::DataSpace dsp_time = H5::DataSpace(ds_pos_time->getSpace());
135  hsize_t step_dims[3];
136  dsp_step.getSimpleExtentDims(step_dims);
137  time_set_ = new double[step_dims[0]];
138  step_set_ = new int[step_dims[0]];
139  ds_pos_step->read(step_set_, H5::PredType::NATIVE_INT, dsp_step, dsp_step);
140  ds_pos_time->read(time_set_, H5::PredType::NATIVE_DOUBLE, dsp_time, dsp_time);
141 
142  delete ds_pos_step;
143  delete ds_pos_time;
144 
145  // Reads the box information.
146  H5::Group g_box = H5::Group(particle_group_->openGroup(
147  *particle_group_name + "/box"));
148 
149  H5::Attribute at_box_dimension = H5::Attribute(g_box.openAttribute("dimension"));
150  int dimension;
151  at_box_dimension.read(at_box_dimension.getDataType(), &dimension);
152  if (dimension != 3 ) {
153  throw std::ios_base::failure("Wrong dimension " + boost::lexical_cast<string>(dimension));
154  }
155 
156  // Gets the force group.
157  try {
158  std::string force_group_name = *particle_group_name + "/force";
159  atom_force_group_ = new H5::Group(particle_group_->openGroup(force_group_name));
160  ds_atom_force_ = new H5::DataSet(atom_force_group_->openDataSet("value"));
161  has_force_ = true;
162  std::cout << "H5MD: has /force" << std::endl;
163  } catch (H5::GroupIException not_found) {
164  has_force_ = false;
165  }
166  // Gets the velocity group.
167  try {
168  std::string velocity_group_name = *particle_group_name + "/velocity";
169  atom_velocity_group_ = new H5::Group(particle_group_->openGroup(velocity_group_name));
170  ds_atom_velocity_ = new H5::DataSet(atom_velocity_group_->openDataSet("value"));
171  has_velocity_ = true;
172  std::cout << "H5MD: has /velocity" << std::endl;
173  } catch (H5::GroupIException not_found) {
174  has_velocity_ = false;
175  }
176  // Gets the id group so that the atom id is taken from this group.
177  try {
178  std::string id_group_name = *particle_group_name + "/id";
179  atom_id_group_ = new H5::Group(particle_group_->openGroup(id_group_name));
180  ds_atom_id_ = new H5::DataSet(atom_id_group_->openDataSet("value"));
181  has_id_group_ = true;
182  std::cout << "H5MD: has /id group" << std::endl;
183  } catch (H5::GroupIException not_found) {
184  has_id_group_ = false;
185  }
186 
187  // Gets number of particles and dimensions.
188  H5::DataSpace fs_atom_position_ = ds_atom_position_->getSpace();
189  hsize_t dims[3];
190  rank_ = fs_atom_position_.getSimpleExtentDims(dims);
191  N_particles_ = dims[1];
192  variables_ = dims[2];
193 
194  if (!has_id_group_ && N_particles_ != top.BeadCount()) {
195  std::cout << "Warning: The number of beads (" << N_particles_ << ")";
196  std::cout << " in the trajectory is different than defined in the topology (" << top.BeadCount() << ")" << std::endl;
197  std::cout << "The number of beads from topology will be used!" << std::endl;
198  N_particles_ = top.BeadCount();
199  }
200  chunk_rows_[0] = 1;
201  chunk_rows_[1] = N_particles_;
202  chunk_rows_[2] = variables_;
203 
204  delete particle_group_name;
205  }
206  NextFrame(top);
207  return true;
208 }

211 bool H5MDTrajectoryReader::NextFrame(Topology &top) { // NOLINT const reference
212  // Reads the position row.
213  H5::Exception::dontPrint();
214  idx_frame_++;
215  std::cout << '\r' << "Reading frame: " << idx_frame_;
216  std::cout.flush();
217  double *positions;
218  try {
219  positions = ReadVectorData<double>(ds_atom_position_, H5::PredType::NATIVE_DOUBLE, idx_frame_);
220  } catch (H5::DataSetIException not_found) {
221  return false;
222  }
223 
224  double *forces = NULL;
225  double *velocities = NULL;
226  int *ids = NULL;
227  if (has_velocity_) {
228  velocities = ReadVectorData<double>(ds_atom_velocity_, H5::PredType::NATIVE_DOUBLE, idx_frame_);
229  }
230 
231  if (has_force_) {
232  forces = ReadVectorData<double>(ds_atom_force_, H5::PredType::NATIVE_DOUBLE, idx_frame_);
233  }
234 
235  if (has_id_group_) {
236  ids = ReadScalarData<int>(ds_atom_id_, H5::PredType::NATIVE_INT, idx_frame_);
237  }
238 
239  //Process atoms.
240  for (int at_idx = 0; at_idx < N_particles_; at_idx++) {
241  double x, y, z;
242  int array_index = at_idx*variables_;
243  x = positions[array_index];
244  y = positions[array_index + 1];
245  z = positions[array_index + 2];
246  // Set atom id, or it is an index of a row in dataset or from id dataset.
247  int atom_id = at_idx;
248  if (has_id_group_) {
249  if (ids[at_idx] == -1) // ignore values where id == -1
250  continue;
251  atom_id = ids[at_idx] - 1;
252  }
253 
254  // Topology has to be defined in the xml file or in other
255  // topology files. The h5md only stores the trajectory data.
256  Bead *b = top.getBead(atom_id);
257  b->setPos(vec(x, y, z));
258  if (has_velocity_) {
259  double vx, vy, vz;
260  vx = velocities[array_index];
261  vy = velocities[array_index + 1];
262  vz = velocities[array_index + 2];
263  b->setVel(vec(vx, vy, vz));
264  }
265 
266  if (has_force_) {
267  double fx, fy, fz;
268  fx = forces[array_index];
269  fy = forces[array_index + 1];
270  fz = forces[array_index + 2];
271  b->setF(vec(fx, fy, fz));
272  }
273  }
274 
275  // Clean up pointers.
276  delete[] positions;
277  delete[] forces;
278  delete[] velocities;
279  delete[] ids;
280 
281  return true;
282 }
