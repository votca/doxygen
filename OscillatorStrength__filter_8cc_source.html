38std::vector<Index> OscillatorStrength_filter::CalcIndeces(
39 const Orbitals& orb, QMStateType type) const {
40
41 if (type != QMStateType::Singlet) {
42 throw std::runtime_error(
43 "OscillatorStrength filter only works for singlets.");
44 }
45 Eigen::VectorXd oscs = orb.Oscillatorstrengths();
46 std::vector<Index> indexes;
47 for (Index i = 0; i < oscs.size(); i++) {
48 if (oscs[i] > threshold_) {
49 indexes.push_back(i);
50 }
51 }
52 return indexes;
53}
