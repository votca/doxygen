35 void LAMMPSDumpWriter::Write(Topology *conf) {

37  Eigen::Matrix3d box = conf->getBox();
38  fprintf(_out, "ITEM: TIMESTEP\n%ld\n", top->getStep());

40  fprintf(_out, "ITEM: BOX BOUNDS pp pp pp\n");
41  fprintf(_out, "0 %f\n0 %f\n0 %f\n", box(0, 0) * conv::nm2ang,
42  box(1, 1) * conv::nm2ang, box(2, 2) * conv::nm2ang);
43 
44  fprintf(_out, "ITEM: ATOMS id type x y z");
45  bool v = top->HasVel();

47  fprintf(_out, " vx vy vz");
48  }

51  fprintf(_out, " fx fy fz");
52  }
53  fprintf(_out, "\n");
54 
55  for (BeadContainer::iterator iter = conf->Beads().begin();
56  iter != conf->Beads().end(); ++iter) {
57  Bead *bi = *iter;
58 
59  Index type_id = conf->getBeadTypeId(bi->getType());
60 
61  fprintf(_out, "%ld %li", bi->getId() + 1, type_id);
62  fprintf(_out, " %f %f %f", bi->getPos().x() * conv::nm2ang,
63  bi->getPos().y() * conv::nm2ang, bi->getPos().z() * conv::nm2ang);
64  if (v) {
65  fprintf(_out, " %f %f %f", bi->getVel().x() * conv::nm2ang,
66  bi->getVel().y() * conv::nm2ang, bi->getVel().z() * conv::nm2ang);

70  bi->getF().y() * conv::kj2kcal / conv::nm2ang,
71  bi->getF().z() * conv::kj2kcal / conv::nm2ang);
72  }
73  fprintf(_out, "\n");
74  }
