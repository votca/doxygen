81 void Visualisation::Update_visualisation(Event* event)
82 {
83  Link* viz_link = event->link();
84  Node* node1 = viz_link->node1(); Node* node2 = viz_link->node2();
85  votca::tools::vec node1_pos = node1->position(); votca::tools::vec node2_pos = node2->position();
86 
87  if(node1->type() == (int) NormalNode && node2->type() == (int) NormalNode) {
88  int mesh1_pos_x = floor(node1_pos.x()/viz_size_x); int mesh1_pos_y = floor(node1_pos.y()/viz_size_y); int mesh1_pos_z = floor(node1_pos.z()/viz_size_z);
89  int mesh2_pos_x = floor(node2_pos.x()/viz_size_x); int mesh2_pos_y = floor(node2_pos.y()/viz_size_y); int mesh2_pos_z = floor(node2_pos.z()/viz_size_z);
90 
91  if(node1_pos.z() < node2_pos.z()) {
92  if(!(((mesh1_pos_x == mesh2_pos_x)&&(mesh1_pos_y == mesh2_pos_y))&&(mesh1_pos_z == mesh2_pos_z))) {
93  viz_mesh[mesh1_pos_x][mesh1_pos_y][mesh1_pos_z]+=1.0;
94  viz_mesh[mesh2_pos_x][mesh2_pos_y][mesh2_pos_z]+=1.0;
95  layer_cur[mesh1_pos_z] += 1.0;
96  layer_cur[mesh2_pos_z] += 1.0;
97  }
98  }
99  if(node1_pos.z() > node2_pos.z()) {
100  if(!(((mesh1_pos_x == mesh2_pos_x)&&(mesh1_pos_y == mesh2_pos_y))&&(mesh1_pos_z == mesh2_pos_z))) {
101  viz_mesh[mesh1_pos_x][mesh1_pos_y][mesh1_pos_z]-=1.0;
102  viz_mesh[mesh2_pos_x][mesh2_pos_y][mesh2_pos_z]-=1.0;
103  layer_cur[mesh1_pos_z] -= 1.0;
104  layer_cur[mesh2_pos_z] -= 1.0;
105  }
106  }
107  }
108 }

110 void Visualisation::Print_visualisation()
111 {
112  for(int ix = 0; ix < viz_meshnr_x; ix++) {
113  for(int iy=0; iy < viz_meshnr_y; iy++) {
114  for(int iz=0; iz < viz_meshnr_z; iz++) {
115 
116  if(num_mesh[ix][iy][iz] == 0) num_mesh[ix][iy][iz] = 1;
117  if(layer_num[iz] == 0) layer_num[iz] = 1;
118 
119  double value;
120  double layer_value;
121 
122  value = viz_mesh[ix][iy][iz]/(1.0*num_mesh[ix][iy][iz]);
123  layer_value = layer_cur[iz]/(1.0*layer_num[iz]);
124  if(layer_value == 0) layer_value = 1;
125 
126  double current = value/layer_value;
127  viz_stream << ix << " " << iy << " " << iz << " " << current << "\n";
128  }
129  }
130  }
131  viz_stream.flush();
132 }

162 void Visualisation::Init_node_numbers(GraphKMC* graph)
163 {
164  for(int it = 0; it != graph->Numberofnodes(); it++)
165  {
166  Node* node = graph->GetNode(it);
167  votca::tools::vec position = node->position();
168  if(node->type() == (int) NormalNode) {
169  int mesh_pos_x = floor(position.x()/viz_size_x);
170  int mesh_pos_y = floor(position.y()/viz_size_y);
171  int mesh_pos_z = floor(position.z()/viz_size_z);
172  if (mesh_pos_x == viz_meshnr_x) mesh_pos_x--;
173  if (mesh_pos_y == viz_meshnr_y) mesh_pos_y--;
174  if (mesh_pos_z == viz_meshnr_z) mesh_pos_z--;
175 
176  num_mesh[mesh_pos_x][mesh_pos_y][mesh_pos_z]++;
177  layer_num[mesh_pos_z]++;
178  }
179  }
