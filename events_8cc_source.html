461  coulpot = 1.0/abs(dif) - 1.0/sqrt(RCSQR);
462  }
463  else {
464  coulpot = 0.0;
465  }
466 
467  if(eventinfo->device) {
468 
469  double L = simboxsize.z();
470  double distsqr_planar = dif.x()*dif.x() + dif.y()*dif.y();
471 
472  double sign;
473  double distz_1;
474  double distz_2;
475  double distancesqr_1;
476  double distancesqr_2;
477  // bool outside_cut_off1 = false;
478  //bool outside_cut_off2 = false;
479 
480  for (int i=0;i<eventinfo->number_short_range_images; i++) {
481  if (div(i,2).rem==0) { // even generation
482  sign = -1.0;
483  distz_1 = i*L + 2*startz + dif.z();
484  distz_2 = (i+2)*L - 2*startz - dif.z();
485  }
486  else {
487  sign = 1.0;
488  distz_1 = (i+1)*L + dif.z();
489  distz_2 = (i+1)*L - dif.z();
490  }
491  distancesqr_1 = distz_1*distz_1 + distsqr_planar;
492  if(distancesqr_1 <= RCSQR) coulpot += sign*(1.0/sqrt(distancesqr_1) - 1.0/sqrt(RCSQR));
493  distancesqr_2 = distz_2*distz_2 + distsqr_planar;
494  if(distancesqr_2 <= RCSQR) coulpot += sign*(1.0/sqrt(distancesqr_2) - 1.0/sqrt(RCSQR));
495  }
496 
497  }
498  return coulpot;
499 }
