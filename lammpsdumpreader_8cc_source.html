72 bool LAMMPSDumpReader::NextFrame(Topology &top) {
73  string line;
74  tools::getline(_fl, line);
75  boost::algorithm::trim(line);
76  while (!_fl.eof()) {
77  if (line.substr(0, 5) != "ITEM:") {
78  throw std::ios_base::failure("unexpected line in lammps file:\n" + line);
79  }
80  if (line.substr(6, 8) == "TIMESTEP") {
81  ReadTimestep(top);
82  } else if (line.substr(6, 15) == "NUMBER OF ATOMS") {
83  ReadNumAtoms(top);
84  } else if (line.substr(6, 10) == "BOX BOUNDS") {
85  ReadBox(top);
86  } else if (line.substr(6, 5) == "ATOMS") {
87  ReadAtoms(top, line);
88  break;
89  }
90 
91  else {
92  throw std::ios_base::failure("unknown item lammps file : " +
93  line.substr(6));
94  }
95  tools::getline(_fl, line);
96  boost::algorithm::trim(line);
97  }
98  if (_topology) {
99  cout << "WARNING: topology created from .dump file, masses, charges, "
100  "types, residue names are wrong!\n";
101  }
102  return !_fl.eof();
103  ;
104 }

114 void LAMMPSDumpReader::ReadBox(Topology &top) {
115  string s;
116 
117  Eigen::Matrix3d m = Eigen::Matrix3d::Zero();
118 
119  for (Index i = 0; i < 3; ++i) {
120  tools::getline(_fl, s);
121  boost::algorithm::trim(s);
122 
123  tools::Tokenizer tok(s, " ");
124  vector<double> v;
125  tok.ConvertToVector(v);
126  if (v.size() != 2) {
127  throw std::ios_base::failure("invalid box format");
128  }
129  m(i, i) = v[1] - v[0];
130  }
131  top.setBox(m * tools::conv::ang2nm);
132 }

144 void LAMMPSDumpReader::ReadAtoms(Topology &top, string itemline) {
145  if (_topology) {
146  top.CreateResidue("dum");
147  if (!top.BeadTypeExist("no")) {
148  top.RegisterBeadType("no");
149  }
150  for (Index i = 0; i < _natoms; ++i) {
151  (void)top.CreateBead(Bead::spherical, "no", "no", 0, 0, 0);
152  }
153  }
154 
155  bool pos = false;
156  bool force = false;
157  bool vel = false;
158  Index id = -1;
159 
160  vector<string> fields;
161 
162  {
163  tools::Tokenizer tok(itemline.substr(12), " ");
164  tok.ToVector(fields);
165  Index j = 0;
166  for (tools::Tokenizer::iterator i = tok.begin(); i != tok.end(); ++i, ++j) {
167  if (*i == "x" || *i == "y" || *i == "z") {
168  pos = true;
169  } else if (*i == "xu" || *i == "yu" || *i == "zu") {
170  pos = true;
171  } else if (*i == "xs" || *i == "ys" || *i == "zs") {
172  pos = true;
173  } else if (*i == "vx" || *i == "vy" || *i == "vz") {
174  vel = true;
175  } else if (*i == "fx" || *i == "fy" || *i == "fz") {
176  force = true;
177  } else if (*i == "id") {
178  id = j;
179  }
180  }
181  }
182  if (id < 0) {
183  throw std::runtime_error(
184  "error, id not found in any column of the atoms section");
185  }
186 
187  for (Index i = 0; i < _natoms; ++i) {
188  string s;
189  tools::getline(_fl, s);
190  boost::algorithm::trim(s);
191  if (_fl.eof()) {
192  throw std::runtime_error("Error: unexpected end of lammps file '" +
193  _fname + "' only " +
194  boost::lexical_cast<string>(i) + " atoms of " +
195  boost::lexical_cast<string>(_natoms) + " read.");
196  }
197 
198  tools::Tokenizer tok(s, " ");
199  tools::Tokenizer::iterator itok = tok.begin();
200  vector<string> fields2 = tok.ToVector();
201  // internal numbering begins with 0
202  Index atom_id = boost::lexical_cast<Index>(fields2[id]);
203  if (atom_id > _natoms) {
204  throw std::runtime_error(
205  "Error: found atom with id " + boost::lexical_cast<string>(atom_id) +
206  " but only " + boost::lexical_cast<string>(_natoms) +
207  " atoms defined in header of file '" + _fname + "'");
208  }
209  Bead *b = top.getBead(atom_id - 1);
210  b->HasPos(pos);
211  b->HasF(force);
212  b->HasVel(vel);
213  Eigen::Matrix3d m = top.getBox();
214 
215  for (size_t j = 0; itok != tok.end(); ++itok, ++j) {
216  if (j == fields.size()) {
217  throw std::runtime_error(
218  "error, wrong number of columns in atoms section");
219  } else if (fields[j] == "x") {
220  b->Pos().x() = stod(*itok) * tools::conv::ang2nm;
221  } else if (fields[j] == "y") {
222  b->Pos().y() = stod(*itok) * tools::conv::ang2nm;
223  } else if (fields[j] == "z") {
224  b->Pos().z() = stod(*itok) * tools::conv::ang2nm;
225  } else if (fields[j] == "xu") {
226  b->Pos().x() = stod(*itok) * tools::conv::ang2nm;
227  } else if (fields[j] == "yu") {
228  b->Pos().y() = stod(*itok) * tools::conv::ang2nm;
229  } else if (fields[j] == "zu") {
230  b->Pos().z() = stod(*itok) * tools::conv::ang2nm;
231  } else if (fields[j] == "xs") {
232  b->Pos().x() = stod(*itok) * m(0, 0); // box is already in nm
233  } else if (fields[j] == "ys") {
234  b->Pos().y() = stod(*itok) * m(1, 1); // box is already in nm
235  } else if (fields[j] == "zs") {
236  b->Pos().z() = stod(*itok) * m(2, 2); // box is already in nm
237  } else if (fields[j] == "vx") {
238  b->Vel().x() = stod(*itok) * tools::conv::ang2nm;
239  } else if (fields[j] == "vy") {
240  b->Vel().y() = stod(*itok) * tools::conv::ang2nm;
241  } else if (fields[j] == "vz") {
242  b->Vel().z() = stod(*itok) * tools::conv::ang2nm;
243  } else if (fields[j] == "fx") {
244  b->F().x() = stod(*itok) * tools::conv::kcal2kj / tools::conv::ang2nm;
245  } else if (fields[j] == "fy") {
246  b->F().y() = stod(*itok) * tools::conv::kcal2kj / tools::conv::ang2nm;
247  } else if (fields[j] == "fz") {
248  b->F().z() = stod(*itok) * tools::conv::kcal2kj / tools::conv::ang2nm;
249  } else if ((fields[j] == "type") && _topology) {
250  if (!top.BeadTypeExist(*itok)) {
251  top.RegisterBeadType(*itok);
252  }
253  b->setType(*itok);
254  }
255  }
256  }
