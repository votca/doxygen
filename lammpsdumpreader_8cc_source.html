66 bool LAMMPSDumpReader::NextFrame(Topology &top)
67 {
68  string line;
69  getline(_fl, line);
70  while(!_fl.eof()) {
71  if(line.substr(0, 5) != "ITEM:")
72  throw std::ios_base::failure("unexpected line in lammps file:\n"+line);
73  if(line.substr(6,8) == "TIMESTEP") {
74  ReadTimestep(top, line);
75  }
76  else if(line.substr(6,15) == "NUMBER OF ATOMS") {
77  ReadNumAtoms(top, line);
78  }
79  else if(line.substr(6,10) == "BOX BOUNDS") {
80  ReadBox(top, line);
81  }
82  else if(line.substr(6, 5) == "ATOMS") {
83  ReadAtoms(top, line);
84  break;
85  }
86 
87  else {
88  throw std::ios_base::failure("unknown item lammps file : " + line.substr(6));
89  }
90  getline(_fl, line);
91  }
92  if (_topology) {
93  cout << "WARNING: topology created from .dump file, masses, charges, types, residue names are wrong!\n";
94  }
95  return !_fl.eof();;
96 }

106 void LAMMPSDumpReader::ReadBox(Topology &top, string itemline)
107 {

113  for(int i=0; i<3; ++i) {
114  getline(_fl, s);

117  tok.ConvertToVector(v);
118  if(v.size()!=2)
119  throw std::ios_base::failure("invalid box format");
120  m[i][i] = v[1] - v[0];
121  }
122  top.setBox(m );

134 void LAMMPSDumpReader::ReadAtoms(Topology &top, string itemline) {
135  if(_topology){
136  top.CreateResidue("dum");
137  for(int i=0; i<_natoms; ++i) {
138  (void)top.CreateBead(1, "no", (top.GetOrCreateBeadType("no")), 0, 0, 0);
139  }
140  }
141 
142  bool pos=false;
143  bool force=false;
144  bool vel=false;
145  int id=-1;
146 
147  vector<string> fields;
148 
149  {
150  Tokenizer tok(itemline.substr(12), " ");
151  tok.ToVector(fields);
152  int j=0;
153  for(Tokenizer::iterator i=tok.begin(); i!=tok.end(); ++i,++j) {
154  if(*i == "x" || *i == "y" || *i == "z")
155  pos =true;
156  else if(*i == "xu" || *i == "yu" || *i == "zu")
157  pos =true;
158  else if(*i == "xs" || *i == "ys" || *i == "zs")
159  pos =true;
160  else if(*i == "vx" || *i == "vy" || *i == "vz")
161  vel =true;
162  else if(*i == "fx" || *i == "fy" || *i == "fz")
163  force=true;
164  else if(*i == "id")
165  id=j;
166  }
167  }
168  if (id<0){
169  throw std::runtime_error("error, id not found in any column of the atoms section");
170  }
171 
172  for(int i=0; i<_natoms; ++i) {
173  string s;
174  getline(_fl, s);
175  if (_fl.eof())
176  throw std::runtime_error("Error: unexpected end of lammps file '" + _fname + "' only " + boost::lexical_cast<string>(i) + " atoms of " + boost::lexical_cast<string>(_natoms) + " read.");
177 
178  Tokenizer tok(s, " ");
179  Tokenizer::iterator itok= tok.begin();
180  vector<string> fields2;
181  tok.ToVector(fields2);
182  // internal numbering begins with 0
183  int atom_id = boost::lexical_cast<int>(fields2[id]);
184  if (atom_id > _natoms)
185  throw std::runtime_error("Error: found atom with id "+ boost::lexical_cast<string>(atom_id) + " but only "+ boost::lexical_cast<string>(_natoms) + " atoms defined in header of file '" + _fname + "'");
186  Bead *b = top.getBead(atom_id-1);
187  b->HasPos(pos);
188  b->HasF(force);
189  b->HasVel(vel);
190  matrix m=top.getBox();
191 
192  for(size_t j=0; itok!=tok.end(); ++itok, ++j) {
193  if(j == fields.size())
194  throw std::runtime_error("error, wrong number of columns in atoms section");
195  else if(fields[j] == "x")
196  b->Pos().x() = boost::lexical_cast<double>(*itok);
197  else if(fields[j] == "y")
198  b->Pos().y() = boost::lexical_cast<double>(*itok);
199  else if(fields[j] == "z")
200  b->Pos().z() = boost::lexical_cast<double>(*itok);
201  else if(fields[j] == "xu")
202  b->Pos().x() = boost::lexical_cast<double>(*itok);
203  else if(fields[j] == "yu")
204  b->Pos().y() = boost::lexical_cast<double>(*itok);
205  else if(fields[j] == "zu")
206  b->Pos().z() = boost::lexical_cast<double>(*itok);
207  else if(fields[j] == "xs")
208  b->Pos().x() = boost::lexical_cast<double>(*itok)*m[0][0];
209  else if(fields[j] == "ys")
210  b->Pos().y() = boost::lexical_cast<double>(*itok)*m[1][1];
211  else if(fields[j] == "zs")
212  b->Pos().z() = boost::lexical_cast<double>(*itok)*m[2][2];
213  else if(fields[j] == "vx")
214  b->Vel().x() = boost::lexical_cast<double>(*itok);
215  else if(fields[j] == "vy")
216  b->Vel().y() = boost::lexical_cast<double>(*itok);
217  else if(fields[j] == "vz")
218  b->Vel().z() = boost::lexical_cast<double>(*itok);
219  else if(fields[j] == "fx")
220  b->F().x() = boost::lexical_cast<double>(*itok);
