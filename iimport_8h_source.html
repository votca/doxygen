122 void IImport::List2PairsTI(Topology *top, string &ti_file) {
123 
124  QMNBList &nblist = top->NBList();
125  int pair_count = 0;
126 
127  std::string line;
128  std::ifstream intt;
129  intt.open(ti_file.c_str());
130 
131  if (intt.is_open()) {
132  while (intt.good()) {
133 
134  std::getline(intt, line);
135  vector<string> split;
136  Tokenizer toker(line, " \t");
137  toker.ToVector(split);
138 
139  if (split.size() == 0) { continue; }
140 
141  if (split.size() != 4 && split.size() != 6) {
142  cout << endl << "... ... Invalid line: " << line << flush;
143  continue;
144  }
145 
146  int seg1id = boost::lexical_cast<int>(split[0]);
147  int seg2id = boost::lexical_cast<int>(split[1]);
148 
149  Segment *seg1 = top->getSegment(seg1id);
150  Segment *seg2 = top->getSegment(seg2id);
151 
152  QMPair *qmp = nblist.FindPair(seg1,seg2);
153  if (qmp == NULL) {
154  cout << endl
155  << "... ... ERROR: " << line
156  << flush;
157  cout << endl
158  << "... ... Line is not compatible with neighborlist. "
159  << flush;
160  throw std::runtime_error("Forgot to run -e neighborlist?");
161  }
162 
163  int e_h_1 = (split[2] == "e") ? -1 : +1;
164  double j2_1 = boost::lexical_cast<double>(split[3]);
165 
166  if (split.size() == 6) {
167 
168  int e_h_2 = (split[4] == "e") ? -1 : +1;
169  if (e_h_1 == e_h_2) {
170  cout << endl << "... ... Invalid line: " << line << flush;
171  continue;
172  }
173  double j2_2 = boost::lexical_cast<double>(split[5]);
174 
175  qmp->setJeff2(j2_2, e_h_2);
176  qmp->setIsPathCarrier(true, e_h_2);
177  }
178 
179  qmp->setJeff2(j2_1, e_h_1);
180  qmp->setIsPathCarrier(true, e_h_1);
181  ++pair_count;
182 
183  }
184  }
185  else {
186  cout << endl
187  << "ERROR: No such file" << ti_file << ". "
188  << flush;
189  throw std::runtime_error("Missing input file in IImport calculator.");
190  }
191 
192  cout << endl
193  << "... ... Set transfer integrals for " << pair_count << " pairs. "
194  << flush;
195 }

198 void IImport::XML2PairTI(QMPair *qmpair, string &xmlDirFile) {

200  printf("\n... ... Import TIs for pair %5d (ID1 %4d ID2 %4d) ",

217 
218  if (split.size() < 3) {
219  continue;

221 
222 // cout << split.size() << " " << flush;

224 // for (sit = split.begin(); sit < split.end(); ++sit) {
225 // cout << (*sit) << " " << flush;
226 // }

241  }
242  else if (_TI_tag == "J_sq_degen" || _TI_tag == "J_sq_boltz") {
243  double J2 = boost::lexical_cast<double>(split[1]);
244  qmpair->setJeff2(J2, STATE);
245  qmpair->setIsPathCarrier(1, STATE);
246  printf("\n... ... ... J2(State = %+1d) = %4.7e",
247  STATE, qmpair->getJeff2(STATE));
248  }
249  }
250  } /* Exit loop over lines */
251  }
252  else { cout << endl
253  << "... ... ... ERROR: No TI.xml in pair folder. Skip..."
254  << flush;
255  }
256 
257  intt.close();

317  int orbB = (*itOverlap)->getAttribute<double>("orbB");

319  if ( orbA == homoA && orbB == homoB ) {
320  qmp->setJeff2(overlapAB*overlapAB, 1);
321  qmp->setIsPathCarrier(true, 1);
322  }
323 
324  if ( orbA == homoA+1 && orbB == homoB+1 ) {
325  qmp->setJeff2(overlapAB*overlapAB, -1);
326  qmp->setIsPathCarrier(true, -1);
327  }
328  }
329  }
330 
331  } else {
332  _incomplete_jobs++;
333  LOG(logINFO, _log) << "Job " << (*it)->get( "id" ).as<string>() << " is " << (*it)->get( "status" ).as<string>() << endl;
334  }
335  }
336

338  cout << _log;
