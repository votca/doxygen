228 double IImport::StochasticMakeJ(double thisdistance, vector<double> distances, vector<double> means, vector<double> sigmas, votca::tools::Random2 *RandomVariable){
229  double offset = 0.5*(distances[distances.size()-1] - distances[0])/distances.size();
230  thisdistance -= offset;
231  double MINR = distances[0];
232  double MAXR = distances[distances.size()-1];
233  double thismean=0.0;
234  double thissigma=0.0;
235  if(thisdistance <= 0){
236  return 0;
237  }
238  else if(thisdistance <= MINR){
239  thismean = means[0];
240  thissigma = sigmas[0];
241  }
242  else if(thisdistance > MAXR){
243  return 0;
244  }
245  else{
246  for(unsigned i = 0; i<distances.size()-2; i++){
247  if(distances[i] < thisdistance && thisdistance <= distances[i+1]){
248  // linear interpolations
249  thismean = (means[i+1]-means[i])/(distances[i+1]-distances[i])*(thisdistance-distances[i])+means[i];
250  thissigma = (sigmas[i+1]-sigmas[i])/(distances[i+1]-distances[i])*(thisdistance-distances[i])+sigmas[i];
251  break;
252  }
253  }
254  }
255  if(thismean && thissigma){
256  double log10J2 = thismean+RandomVariable->rand_gaussian(thissigma);
257  double J2;
258  J2 = pow(10,log10J2);
259  return J2;
260  }
261  else{
262  cout << "WARNING: skipping uparametrized distance " << thisdistance << " nm." << endl;
263  return 0;
264  }

268 void IImport::StochasticTI(Topology *top, string &filename, int state) {
269  if(state == 1){
270  cout << endl << "... ... calculating stochastic hole TI." << endl;
271  }
272  else if(state == -1){
273  cout << endl << "... ... calculating stochastic electron TI." << endl;
274  }
275  // read in probability function
276  cout << "... ... provided means and sigmas (for log10(J^2)):" << endl;
277  vector<double> distances;
278  vector<double> means;
279  vector<double> sigmas;
280 
281  std::string line;
282  std::ifstream intt;
283  intt.open(filename.c_str());
284  int linenumber = 0;
285  if (intt.is_open() ) {
286  while ( intt.good() ) {
287 
288  std::getline(intt, line);
289  if (linenumber > 1){
290  vector<string> split;
291  Tokenizer toker(line, " \t");
292  toker.ToVector(split);
293 
294  if ( !split.size() ||
295  split[0] == "!" ||
296  split[0].substr(0,1) == "!" ) { continue; }
297 
298  double distance = stod(split[0]);
299  double mean = stod(split[1]);
300  double sigma = stod(split[2]);
301  cout << " " << distance << " nm: " << mean << " +/- " << sigma << endl;
302  distances.push_back(distance);
303  means.push_back(mean);
304  sigmas.push_back(sigma);
305  }
306  linenumber++;
307  }
308  }
309  else { cout << endl << "ERROR: No such file " << filename << endl;
310  throw std::runtime_error("Supply input probability file."); }
311 
312 
313  // Initialise random number generator
314  if(votca::tools::globals::verbose) { cout << endl << "Initialising random number generator" << endl; }
315  srand(12345);
316  votca::tools::Random2 *RandomVariable = new votca::tools::Random2();
317  RandomVariable->init(rand(), rand(), rand(), rand());
318 
319 
320  // loop over neighbor list
321  QMNBList &nblist = top->NBList();
322  QMNBList::iterator nit;
323 
324  for (nit = nblist.begin(); nit != nblist.end(); ++nit) {
325  Segment *seg1 = (*nit)->Seg1();
326  Segment *seg2 = (*nit)->Seg2();
327  double thisdistance = abs((*nit)->getR());
328 
329  double J2 = StochasticMakeJ(thisdistance, distances, means, sigmas, RandomVariable);
330 
331  QMPair *qmpair = nblist.FindPair(seg1,seg2);
332  if (qmpair == NULL) {
333  cout << endl
334  << "... ... ERROR: " << line
335  << flush;
336  cout << endl
337  << "... ... Line is not compatible with neighborlist. "
338  << flush;
339  throw std::runtime_error("Forgot to run -e neighborlist?");
340  }
341 
342  qmpair->setJeff2(J2, state);
343  qmpair->setIsPathCarrier(1, state);
344 
345  }
