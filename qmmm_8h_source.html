233 XJob QMMM::ProcessInputString(Job *job, Topology *top, QMThread *thread) {
234 
235  string input = job->getInput().as<string>();
236  vector<Segment*> qmSegs;

238  vector<string> split;
239  Tokenizer toker(input, " \t\n");
240  toker.ToVector(split);
241 
242  for (unsigned int i = 0; i < split.size(); ++i) {
243 
244  string id_seg_mps = split[i];
245  vector<string> split_id_seg_mps;
246  Tokenizer toker(id_seg_mps, ":");
247  toker.ToVector(split_id_seg_mps);
248 
249  int segId = boost::lexical_cast<int>(split_id_seg_mps[0]);
250  string segName = split_id_seg_mps[1];
251  string mpsFile = split_id_seg_mps[2];
252 
253  Segment *seg = top->getSegment(segId);
254  if (seg->getName() != segName) {
255  LOG(logERROR,*(thread->getLogger()))
256  << "ERROR: Seg " << segId << ":" << seg->getName() << " "
257  << " maltagged as " << segName << ". Skip job ..." << flush;
258  throw std::runtime_error("Input does not match topology.");
259  }
260 
261  qmSegs.push_back(seg);
262  qmSegMps.push_back(mpsFile);
263  }
264 
265  return XJob(job->getId(), job->getTag(), qmSegs, qmSegMps, top);
