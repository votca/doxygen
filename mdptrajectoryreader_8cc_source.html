00038 bool MDPTrajectoryReader::FirstFrame(Configuration &conf)
00039 {
00040     FILE *fl = _fl;
00041     long I4;
00042     float R4;
00043     double R8;
00044         
00045     conf.HasPos(true);
00046 /* 
00047  *   The header line looks as follows (in fortran binary out, so with record blocks):
00048  *   
00049  *   I4    0/1 has velocities?
00050  *   R8    dt
00051  *   3*R8  box size (Angstr)
00052  *   R8    unit in m (10^-10 for Angstr)
00053  *   I4    number if mol types
00054  *   for each molecule:
00055  *   I4    nspec (number of molecules of this type)
00056  *   I4    nsites (number of atoms in molecule)
00057  *   L4    true ??
00058  *
00059  * */
00060 
00061 // read record begin
00062     fread(&I4, sizeof(I4), 1, fl);
00063     
00064     // read ivel
00065     fread(&I4, sizeof(I4), 1, fl);
00066     
00067     double dt, unit;
00068     vec box;
00069     
00070     fread(&dt, sizeof(dt), 1, fl);
00071     fread(&box, sizeof(box), 1, fl);
00072     fread(&unit, sizeof(unit), 1, fl);
00073     fread(&_moltypes, sizeof(_moltypes), 1, fl);
00074     
00075             
00076     for(int i=0; i<_moltypes; ++i) {
00077         fread(&I4, sizeof(I4), 1, fl);
00078         _nmols.push_back(I4);
00079         fread(&I4, 1, sizeof(I4), fl);
00080         _natoms.push_back(I4);
00081         fread(&I4, sizeof(I4), 1, fl);
00082     }
00083 
00084     // read record end
00085     fread(&I4, sizeof(I4), 1, fl);
00086     // now read in the molecule types        
00087     /*
00088      *  for each type 
00089      *    for each atom
00090      *      R8       mass
00091      *  I4 ?         List (0: has coordinates, 1: no coordinates)
00092      */
00093       
00094     fread(&I4, sizeof(I4), 1, fl);
00095      
00096     for(int type=0; type<_moltypes; ++type) {
00097         for(int atom=0; atom<_natoms[type]; ++atom) {
00098             fread(&R8, sizeof(R8), 1, fl);
00099             cout << R8 << endl;
00100         }
00101 
00102         // is molecule read in trajectory?
00103         fread(&I4, sizeof(I4), 1, fl);
00104         cout << I4 << endl;
00105     }
00106     
00107     // read record end
00108     fread(&I4, sizeof(I4), 1, fl);
00109     
00110     return NextFrame(conf);
00111 }
