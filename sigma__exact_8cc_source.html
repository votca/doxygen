38 double Sigma_Exact::CalcCorrelationDiagElement(Index gw_level,
39  double frequency) const {
40  const double eta2 = _opt.eta * _opt.eta;
41  const Index lumo = _opt.homo + 1;
42  const Index n_occ = lumo - _opt.rpamin;
43  const Index n_unocc = _opt.rpamax - _opt.homo;
44  double sigma = 0.0;
45  for (Index s = 0; s < _rpa_solution.omega.size(); s++) {
46  const double eigenvalue = _rpa_solution.omega(s);
47  const Eigen::ArrayXd res_12 = _residues[gw_level].col(s).cwiseAbs2();
48  Eigen::ArrayXd temp = -_rpa.getRPAInputEnergies().array() + frequency;
49  temp.segment(0, n_occ) += eigenvalue;
50  temp.segment(n_occ, n_unocc) -= eigenvalue;
51  const Eigen::ArrayXd denom = temp.abs2() + eta2;
52  sigma += (res_12 * temp / denom).sum();
53  }
54  return 2 * sigma;
55 }

57 double Sigma_Exact::CalcCorrelationDiagElementDerivative(
58  Index gw_level, double frequency) const {
59  const double eta2 = _opt.eta * _opt.eta;
60  const Index lumo = _opt.homo + 1;
61  const Index n_occ = lumo - _opt.rpamin;
62  const Index n_unocc = _opt.rpamax - _opt.homo;
63  double dsigma_domega = 0.0;
64  for (Index s = 0; s < _rpa_solution.omega.size(); s++) {
65  const double eigenvalue = _rpa_solution.omega(s);
66  const Eigen::ArrayXd res_12 = _residues[gw_level].col(s).cwiseAbs2();
67  Eigen::ArrayXd temp = -_rpa.getRPAInputEnergies().array() + frequency;
68  temp.segment(0, n_occ) += eigenvalue;
69  temp.segment(n_occ, n_unocc) -= eigenvalue;
70  const Eigen::ArrayXd denom = temp.abs2() + eta2;
71  dsigma_domega += ((eta2 - temp.abs2()) * res_12 / denom.abs2()).sum();
72  }
73  return 2 * dsigma_domega;
74 }

76 double Sigma_Exact::CalcCorrelationOffDiagElement(Index gw_level1,
77  Index gw_level2,
78  double frequency1,
79  double frequency2) const {
80  const double eta2 = _opt.eta * _opt.eta;
81  const Index lumo = _opt.homo + 1;
82  const Index n_occ = lumo - _opt.rpamin;
83  const Index n_unocc = _opt.rpamax - _opt.homo;
84  const Index rpasize = _rpa_solution.omega.size();
85  double sigma_c = 0.0;
86  for (Index s = 0; s < rpasize; s++) {
87  const double eigenvalue = _rpa_solution.omega(s);
88  const Eigen::VectorXd& res1 = _residues[gw_level1].col(s);
89  const Eigen::VectorXd& res2 = _residues[gw_level2].col(s);
90  const Eigen::VectorXd res_12 = res1.cwiseProduct(res2);
91  Eigen::ArrayXd temp1 = -_rpa.getRPAInputEnergies().array();
92  temp1.segment(0, n_occ) += eigenvalue;
93  temp1.segment(n_occ, n_unocc) -= eigenvalue;
94  const Eigen::ArrayXd temp2 = temp1 + frequency2;
95  temp1 += frequency1;
96  const Eigen::ArrayXd numer1 = res_12.array() * temp1;
97  const Eigen::ArrayXd numer2 = res_12.array() * temp2;
98  const Eigen::ArrayXd denom1 = temp1.abs2() + eta2;
99  const Eigen::ArrayXd denom2 = temp2.abs2() + eta2;
100  sigma_c += 0.5 * ((numer1 / denom1) + (numer2 / denom2)).sum();
101  }
102  // Multiply with factor 2.0 to sum over both (identical) spin states
103  return 2.0 * sigma_c;
104 }

106 Eigen::MatrixXd Sigma_Exact::CalcResidues(Index gw_level) const {
107  const Index lumo = _opt.homo + 1;
108  const Index n_occ = lumo - _opt.rpamin;
109  const Index n_unocc = _opt.rpamax - _opt.homo;
110  const Index rpasize = n_occ * n_unocc;
111  const Index qpoffset = _opt.qpmin - _opt.rpamin;
112  const Index auxsize = _Mmn.auxsize();
113  vc2index vc = vc2index(0, 0, n_unocc);
114  const Eigen::MatrixXd& Mmn_i = _Mmn[gw_level + qpoffset];
115  Eigen::MatrixXd res = Eigen::MatrixXd::Zero(_rpatotal, rpasize);
116  for (Index v = 0; v < n_occ; v++) { // Sum over v
117  auto Mmn_v = _Mmn[v].block(n_occ, 0, n_unocc, auxsize);
118  const Eigen::MatrixXd fc = Mmn_v * Mmn_i.transpose(); // Sum over chi
119  auto XpY_v = _rpa_solution.XpY.block(vc.I(v, 0), 0, n_unocc, rpasize);
120  res += fc.transpose() * XpY_v; // Sum over c
121  }
122  return res;
123 }
