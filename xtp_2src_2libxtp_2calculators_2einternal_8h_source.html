84 void XEInternal::ParseEnergiesXML(Property *opt) {
85 
86  // update options with the VOTCASHARE defaults
87  UpdateWithDefaults( opt, "xtp" );
88  std::string key = "options." + Identify();
89 
90  std::string energiesXML = opt->get(key+".energiesXML").as<std::string> ();
91 
92  cout << endl
93  << "... ... Site, reorg. energies from " << energiesXML << ". "
94  << flush;
95 
96  Property alloc;
97  load_property_from_xml(alloc, energiesXML.c_str());
98 
99  /* --- ENERGIES.XML Structure ---
100  *
101  * <topology>
102  *
103  * <molecules>
104  * <molecule>
105  * <name></name>
106  *
107  * <segments>
108  *
109  * <segment>
110  * <name></name>
111  *
112  * <!-- U_sG_sG, s->state, G->geometry !-->
113  *
114  * <U_cC_nN_e></U_cC_nN_e>
115  * <U_cC_nN_h></U_cC_nN_h>
116  *
117  * <U_nC_nN_e></U_nC_nN_e>
118  * <U_nC_nN_h></U_nC_nN_h>
119  *
120  * <U_cN_cC_e></U_cN_cC_e>
121  * <U_cN_cC_h></U_cN_cC_h>
122  *
123  * </segment>
124  *
125  * <segment>
126  * ...
127  *
128  */
129 
130  key = "topology.molecules.molecule";
131  list<Property*> mols = alloc.Select(key);
132  list<Property*> ::iterator molit;
133  for (molit = mols.begin(); molit != mols.end(); ++molit) {
134 
135  key = "segments.segment";
136  list<Property*> segs = (*molit)->Select(key);
137  list<Property*> ::iterator segit;
138 
139  for (segit = segs.begin(); segit != segs.end(); ++segit) {
140 
141  std::string segName = (*segit)->get("name").as<std::string> ();
142 
143  bool has_seg = true;
144  bool has_e = false;
145  bool has_h = false;
146  bool has_s = false;
147  bool has_t = false;
148 
149  double U_cC_nN_e = 0.0;
150  double U_cC_nN_h = 0.0;
151  double U_nC_nN_e = 0.0;
152  double U_nC_nN_h = 0.0;
153  double U_cN_cC_e = 0.0;
154  double U_cN_cC_h = 0.0;
155 
156  double U_xX_nN_s = 0.0;
157  double U_xX_nN_t = 0.0;
158  double U_nX_nN_s = 0.0;
159  double U_nX_nN_t = 0.0;
160  double U_xN_xX_s = 0.0;
161  double U_xN_xX_t = 0.0;
162 
163 
164  if ( (*segit)->exists("U_cC_nN_e") &&
165  (*segit)->exists("U_nC_nN_e") &&
166  (*segit)->exists("U_cN_cC_e") ) {
167 
168  U_cC_nN_e = (*segit)->get("U_cC_nN_e").as< double > ();
169  U_nC_nN_e = (*segit)->get("U_nC_nN_e").as< double > ();
170  U_cN_cC_e = (*segit)->get("U_cN_cC_e").as< double > ();
171 
172  has_e = true;
173  }
174 
175  if ( (*segit)->exists("U_cC_nN_h") &&
176  (*segit)->exists("U_nC_nN_h") &&
177  (*segit)->exists("U_cN_cC_h") ) {
178 
179  U_cC_nN_h = (*segit)->get("U_cC_nN_h").as< double > ();
180  U_nC_nN_h = (*segit)->get("U_nC_nN_h").as< double > ();
181  U_cN_cC_h = (*segit)->get("U_cN_cC_h").as< double > ();
182 
183  has_h = true;
184  }

186  if ( (*segit)->exists("U_xX_nN_s") &&
187  (*segit)->exists("U_nX_nN_s") &&
188  (*segit)->exists("U_xN_xX_s") ) {
189 
190  U_xX_nN_s = (*segit)->get("U_xX_nN_s").as< double > ();
191  U_nX_nN_s = (*segit)->get("U_nX_nN_s").as< double > ();
192  U_xN_xX_s = (*segit)->get("U_xN_xX_s").as< double > ();
193 
194  has_s = true;
195  }
196  if ( (*segit)->exists("U_xX_nN_t") &&

198  (*segit)->exists("U_xN_xX_t") ) {
199

203 
204  has_t = true;
205  }
206  //cout << U_xX_nN_s << U_nX_nN_s << U_xN_xX_s << endl;
207  _seg_U_cC_nN_e[segName] = U_cC_nN_e;

233 bool XEInternal::EvaluateFrame(Topology *top) {
234 
235  std::vector< Segment* > ::iterator sit;
236  int count = 0;
237  for (sit = top->Segments().begin(); sit < top->Segments().end(); ++sit) {
238 
239  std::string segName = (*sit)->getName();
240 
241  try {
242  //bool has_seg = _has_seg.at(segName);
243  }
244  catch (out_of_range) {
245  cout << endl << "... ... WARNING: No energy information for seg ["
246  << segName << "]. Skipping... ";
247  continue;
248  }
249 
250  ++count;
251 
252  if (_seg_has_e[segName]) {
253 
254  double u = _seg_U_cC_nN_e[segName];
255  double l1 = _seg_U_nC_nN_e[segName];
256  double l2 = _seg_U_cN_cC_e[segName];
257  bool has_e = true;
258 
259  (*sit)->setU_cC_nN(u, -1);
260  (*sit)->setU_nC_nN(l1, -1);
261  (*sit)->setU_cN_cC(l2, -1);
262  (*sit)->setHasState(has_e, -1);
263  }
264 
265  if (_seg_has_h[segName]) {
266 
267  double u = _seg_U_cC_nN_h[segName];
268  double l1 = _seg_U_nC_nN_h[segName];
269  double l2 = _seg_U_cN_cC_h[segName];
270  bool has_h = true;
271 
272  (*sit)->setU_cC_nN(u, +1);
273  (*sit)->setU_nC_nN(l1, +1);
274  (*sit)->setU_cN_cC(l2, +1);
275  (*sit)->setHasState(has_h, +1);
276  }
277  if (_seg_has_s[segName]) {
278 
279  double u = _seg_U_xX_nN_s[segName];
280  double l1 = _seg_U_nX_nN_s[segName];
281  double l2 = _seg_U_xN_xX_s[segName];
282  bool has_s = true;
283  //cout << u << l1 << l2<<endl;
284  (*sit)->setU_xX_nN(u, +2);
285  (*sit)->setU_nX_nN(l1, +2);
286  (*sit)->setU_xN_xX(l2, +2);
287  (*sit)->setHasState(has_s, +2);
288  }
289  if (_seg_has_t[segName]) {
290 
291  double u = _seg_U_xX_nN_t[segName];
292  double l1 = _seg_U_nX_nN_t[segName];
293  double l2 = _seg_U_xN_xX_t[segName];
294  bool has_t = true;
295 
296  (*sit)->setU_xX_nN(u, +3);
297  (*sit)->setU_nX_nN(l1, +3);
298  (*sit)->setU_xN_xX(l2, +3);
299  (*sit)->setHasState(has_t, +3);
300  }
301  }
302 
303  cout << endl
304  << "... ... Read in site, reorg. energies for "
305  << count << " segments. " << flush;
306 
307 
308  return 1;
309 }
