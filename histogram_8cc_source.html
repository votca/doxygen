00038 void Histogram::ProcessData(DataCollection<double>::selection *data)
00039 {
00040     DataCollection<double>::selection::iterator array;
00041     DataCollection<double>::array::iterator iter;
00042     int ii;
00043     int ndata = 0;
00044     
00045     _pdf.assign(_options._n, 0);
00046         
00047     if(_options._auto_interval) {
00048         _min = numeric_limits<double>::max();
00049         _max = numeric_limits<double>::min();
00050         _options._extend_interval = true;
00051     }
00052     else {
00053         _min = _options._min;
00054         _max = _options._max;
00055     }
00056     
00057     for(array = data->begin(); array!=data->end(); ++array) {
00058         ndata+=(*array)->size();
00059         if(_options._extend_interval || _options._auto_interval) {
00060             for(iter=(*array)->begin(); iter!=(*array)->end(); ++iter) {
00061                 _min = min(*iter, _min);
00062                 _max = max(*iter, _max);            
00063             }
00064         }
00065     }
00066     
00067     // make that the highes value fits into interval
00068     //if(_options._auto_interval || _max!=_options._max)
00069     //    _max = _max + 0.5*(_max - _min)/(double)(_options._n);
00070     
00071     _interval = (_max - _min)/(double)(_options._n-1);
00072 
00073     double v = 1.;
00074     for(array = data->begin(); array!=data->end(); ++array) {
00075         for(iter=(*array)->begin(); iter!=(*array)->end(); ++iter) {
00076             ii = (int)( (*iter - _min) / _interval + 0.5); // the interval should be centered around the sampling point              
00077             if(ii< 0 || ii >= _options._n) {
00078                 if(_options._periodic) {
00079                     while(ii<0) ii+=_options._n;
00080                     ii = ii % _options._n;
00081                 }
00082                 else { continue; } //cout << "[histogram.cc]: out of bounds" << endl; continue;}
00083             }
00084             _pdf[ii]+= v;
00085         }
00086     }        
00087     
00088     //cout << _pdf.size() << " " << _options._periodic << endl;
00089     if(_options._scale == "bond") {
00090         for(size_t i=0; i<_pdf.size(); ++i) {
00091             double r = _min + _interval*(double)i;
00092             if(abs(r) < 1e-10) {
00093                 r = _min + _interval*(double)(i+1);
00094                 _pdf[i] = _pdf[i+1];
00095             }
00096             _pdf[i] /= (r*r);
00097         }
00098     }
00099     else if(_options._scale == "angle") {
00100         for(size_t i=0; i<_pdf.size(); ++i) {
00101             double alpha = _min + _interval*(double)i;
00102             double sa = sin(alpha); 
00103             if(abs(sa) < 1e-5) {
00104                 if(i<_pdf.size()-1) {
00105                     alpha = _min + _interval*(double)(i+1);
00106                     _pdf[i]=_pdf[i+1]/sin(alpha);
00107                 }
00108                 else {
00109                     _pdf[i]=_pdf[i-1];
00110                 }
00111                 
00112             }
00113             else _pdf[i] /= sa;
00114         }        
00115     }
00116     
00117     if(_options._periodic) {
00118         _pdf[0] = (_pdf[0] + _pdf[_options._n-1]);
00119         _pdf[_options._n-1] = _pdf[0];
00120     }
00121     if(_options._normalize)
00122         Normalize();
00123 }
