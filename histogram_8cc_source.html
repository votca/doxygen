38 void Histogram::ProcessData(DataCollection<double>::selection *data)
39 {
40  DataCollection<double>::selection::iterator array;
41  DataCollection<double>::array::iterator iter;
42  int ii;
43  int ndata = 0;
44 
45  _pdf.assign(_options._n, 0);
46 
47  if(_options._auto_interval) {
48  _min = numeric_limits<double>::max();
49  _max = numeric_limits<double>::min();
50  _options._extend_interval = true;
51  }
52  else {
53  _min = _options._min;
54  _max = _options._max;
55  }
56 
57  for(array = data->begin(); array!=data->end(); ++array) {
58  ndata+=(*array)->size();
59  if(_options._extend_interval || _options._auto_interval) {
60  for(iter=(*array)->begin(); iter!=(*array)->end(); ++iter) {
61  _min = min(*iter, _min);
62  _max = max(*iter, _max);
63  }
64  }
65  }
66 
67  // make that the highes value fits into interval
68  //if(_options._auto_interval || _max!=_options._max)
69  // _max = _max + 0.5*(_max - _min)/(double)(_options._n);
70 
71  _interval = (_max - _min)/(double)(_options._n-1);
72 
73  double v = 1.;
74  for(array = data->begin(); array!=data->end(); ++array) {
75  for(iter=(*array)->begin(); iter!=(*array)->end(); ++iter) {
76  ii = (int)floor( (*iter - _min) / _interval + 0.5); // the interval should be centered around the sampling point
77  if(ii< 0 || ii >= _options._n) {
78  if(_options._periodic) {
79  while(ii<0) ii+=_options._n;
80  ii = ii % _options._n;
81  }
82  else { continue; } //cout << "[histogram.cc]: out of bounds" << endl; continue;}
83  }
84  _pdf[ii]+= v;
85  }
86  }
87 
88  //cout << _pdf.size() << " " << _options._periodic << endl;
89  if(_options._scale == "bond") {
90  for(size_t i=0; i<_pdf.size(); ++i) {
91  double r = _min + _interval*(double)i;
92  if(abs(r) < 1e-10) {
93  r = _min + _interval*(double)(i+1);
94  _pdf[i] = _pdf[i+1];
95  }
96  _pdf[i] /= (r*r);
97  }
98  }
99  else if(_options._scale == "angle") {
100  for(size_t i=0; i<_pdf.size(); ++i) {
101  double alpha = _min + _interval*(double)i;
102  double sa = sin(alpha);
103  if(abs(sa) < 1e-5) {
104  if(i<_pdf.size()-1) {
105  alpha = _min + _interval*(double)(i+1);
106  _pdf[i]=_pdf[i+1]/sin(alpha);
107  }
108  else {
109  _pdf[i]=_pdf[i-1];
110  }
111 
112  }
113  else _pdf[i] /= sa;
114  }
115  }
116 
117  if(_options._periodic) {
118  _pdf[0] = (_pdf[0] + _pdf[_options._n-1]);
119  _pdf[_options._n-1] = _pdf[0];
120  }
121  if(_options._normalize)
122  Normalize();
123 }
