95 void Fragment::calcPos(string tag) {
96  vec pos = vec(0,0,0);
97  double totWeight = 0.0;
98  for (unsigned int i = 0; i< _atoms.size(); i++) {
99  if (tag == "MD") {
100  pos += _atoms[i]->getPos() * _atoms[i]->getWeight();
101  }
102  else if (tag == "QM") {
103  if (_atoms[i]->HasQMPart())
104  pos += _atoms[i]->getQMPos() * _atoms[i]->getWeight();
105  else
106  assert(_atoms[i]->getWeight() < 1e-6);
107  }
108  totWeight += _atoms[i]->getWeight();
109  }
110 
111  if (tag == "MD") {
112  _CoMD = pos / totWeight;
113  }
114  else if (tag == "QM") {
115  _CoQM = pos / totWeight;
116  }

123 void Fragment::Rigidify(bool Auto) {
124 
125  // +++++++++++++++++++++++++++++++++++++++++ //
126  // Establish reference atoms for local frame //
127  // +++++++++++++++++++++++++++++++++++++++++ //
128 
129  vector<Atom*> trihedron;
130 
131  if (Auto) {
132 
133  // Automated search for suitable atoms
134  bool enough = false;
135  vector< Atom* > ::iterator ait = this->Atoms().begin();
136 
137  while (!enough) {
138 
139  if (ait == this->Atoms().end()) { break; }
140  Atom *atm = *ait;
141  ait++;
142 
143  // When specifying mapping weights for rigid fragments,
144  // mobile atoms (whose position can vary significantly with
145  // respect to the rigid plane, for instance) should have a weight 0.
146  // We do not want to pick those to establish the local frame.
147  if (atm->getWeight() == 0.0) { continue; }
148  else {
149  trihedron.push_back(atm);
150  if (trihedron.size() == 3) { enough = true; }
151  }
152  }
153  }
154  else {
155 
156  // Take atoms specified in mapping file <= DEFAULT
157  vector<Atom*> ::iterator ait;
158  for (ait = _atoms.begin(); ait < _atoms.end(); ait++) {
159  if ( ! (*ait)->HasQMPart() ) { continue; }
160 
161  if ( (*ait)->getQMId() == _trihedron[0] ||
162  (*ait)->getQMId() == _trihedron[1] ||
163  (*ait)->getQMId() == _trihedron[2] ) {
164  Atom *atm = *ait;
165  trihedron.push_back(atm);
166  }
167  }
168  }
169 
170  _symmetry = trihedron.size();
171  // [-Wlogical-not-parentheses]
172  if ( trihedron.size() != _trihedron.size() ) {
173  cout << endl << "ERROR Local frame ill-defined" << flush;
174  }
175 
176  // +++++++++++++++++++++++ //
177  // Construct trihedra axes //
178  // +++++++++++++++++++++++ //
179 
180  vec xMD, yMD, zMD;
181  vec xQM, yQM, zQM;
182 
183  if (_symmetry == 3) {
184 
185  vec r1MD = trihedron[0]->getPos();
186  vec r2MD = trihedron[1]->getPos();
187  vec r3MD = trihedron[2]->getPos();
188  vec r1QM = trihedron[0]->getQMPos();
189  vec r2QM = trihedron[1]->getQMPos();
190  vec r3QM = trihedron[2]->getQMPos();
191 
192  xMD = r2MD - r1MD;
193  yMD = r3MD - r1MD;
194  xQM = r2QM - r1QM;
195  yQM = r3QM - r1QM;
196 
197  zMD = xMD ^ yMD;
198  zQM = xQM ^ yQM;
199 
200  yMD = zMD ^ xMD;
201  yQM = zQM ^ xQM;
202 
203  xMD = xMD.normalize();
204  yMD = yMD.normalize();
205  zMD = zMD.normalize();
206  xQM = xQM.normalize();
207  yQM = yQM.normalize();
208  zQM = zQM.normalize();
209 
210  }
211 
212  else if (_symmetry == 2) {
213 
214  vec r1MD = trihedron[0]->getPos();
215  vec r2MD = trihedron[1]->getPos();
216  vec r1QM = trihedron[0]->getQMPos();
217  vec r2QM = trihedron[1]->getQMPos();
218 
219  xMD = r2MD - r1MD;
220  xQM = r2QM - r1QM;
221 
222  // Normalising not necessary, but recommendable, when doing...
223  xMD = xMD.normalize();
224  xQM = xQM.normalize();
225 
226  vec yMDtmp = vec(0,0,0);
227  vec yQMtmp = vec(0,0,0);
228 
229  // ... this: Check whether one of the components is equal or close to
230  // zero. If so, this easily gives a second leg for the trihedron.
231  if ( xMD.getX()*xMD.getX() < 1e-6 ) { yMDtmp = vec(1,0,0); }
232  else if ( xMD.getY()*xMD.getY() < 1e-6 ) { yMDtmp = vec(0,1,0); }
233  else if ( xMD.getZ()*xMD.getZ() < 1e-6 ) { yMDtmp = vec(0,0,1); }
234  if ( xQM.getX()*xQM.getX() < 1e-6 ) { yQMtmp = vec(1,0,0); }
235  else if ( xQM.getY()*xQM.getY() < 1e-6 ) { yQMtmp = vec(0,1,0); }
236  else if ( xQM.getZ()*xQM.getZ() < 1e-6 ) { yQMtmp = vec(0,0,1); }
237 
238  if ( abs(yMDtmp) < 0.5 ) {
239  // All components of xMD are unequal to zero => division is safe.
240  // Choose vector from plane with xMD * inPlaneVec = 0:
241  double tmp_x = 1.;
242  double tmp_y = 1.;
243  double tmp_z = 1/xMD.getZ() * (-xMD.getX()*tmp_x - xMD.getY()*tmp_y);
244  yMDtmp = vec(tmp_x, tmp_y, tmp_z);
245  yMDtmp.normalize();
246  }
247  if ( abs(yQMtmp) < 0.5 ) {
248  double tmp_x = 1.;
249  double tmp_y = 1.;
250  double tmp_z = 1/xQM.getZ() * (-xQM.getX()*tmp_x - xQM.getY()*tmp_y);
251  yQMtmp = vec(tmp_x, tmp_y, tmp_z);
252  yQMtmp.normalize();
253  }
254 
255  // Now proceed as for symmetry 3
256  zMD = xMD ^ yMDtmp;
257  yMD = zMD ^ xMD;
258  zQM = xQM ^ yQMtmp;
259  yQM = zQM ^ xQM;
260 
261  xMD.normalize();
262  yMD.normalize();
263  zMD.normalize();
264  xQM.normalize();
265  yQM.normalize();
266  zQM.normalize();
267 
268  }
269 
270  else if (_symmetry == 1) {
271 
272  xMD = vec(1,0,0);
273  yMD = vec(0,1,0);
274  zMD = vec(0,0,1);
275  xQM = vec(1,0,0);
276  yQM = vec(0,1,0);
277  zQM = vec(0,0,1);
278 
279  }
280 
281  else {
282  cout << endl
283  << "NOTE: Invalid definition of local frame in fragment "
284  << this->getName();
285  cout << ". Assuming point particle for mapping. "
286  << endl;
287 
288  xMD = vec(1,0,0);
289  yMD = vec(0,1,0);
290  zMD = vec(0,0,1);
291  xQM = vec(1,0,0);
292  yQM = vec(0,1,0);
293  zQM = vec(0,0,1);
294 
295  }
296 
297  // +++++++++++++++++ //
298  // Rotation matrices //
299  // +++++++++++++++++ //
300 
301  matrix rotMD = matrix(xMD, yMD, zMD);
302  matrix rotQM = matrix(xQM, yQM, zQM);
303 
304  matrix rotateQM2MD = rotMD * rotQM.Transpose();
305  _rotateQM2MD = rotateQM2MD;
306 
307 
308  // ++++++++++++++++++ //
309  // Transform fragment //
310  // ++++++++++++++++++ //
311 
312  this->calcPos("QM");
313  this->RotTransQM2MD();
314  _translateQM2MD = _CoMD - _CoQM;
315 
316 }
