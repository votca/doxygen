92 libint2::Shell AOShell::LibintShell() const {
93  libint2::svector<libint2::Shell::real_t> decays;
94  libint2::svector<libint2::Shell::Contraction> contractions;
95  const Eigen::Vector3d& pos = getPos();
96  libint2::Shell::Contraction contr;
97  contr.l = static_cast<int>(getL());
98  contr.pure = true;
99  for (const auto& primitive : _gaussians) {
100  decays.push_back(primitive.getDecay());
101  contr.coeff.push_back(primitive.getContraction());
102  }
103  contractions.push_back(contr);
104  std::array<libint2::Shell::real_t, 3> libintpos = {pos[0], pos[1], pos[2]};
105  return libint2::Shell(decays, contractions, libintpos);

118 void AOShell::EvalAOspace(Eigen::VectorBlock<Eigen::VectorXd>& AOvalues,
119  Eigen::Block<Eigen::MatrixX3d>& gradAOvalues,
120  const Eigen::Vector3d& grid_pos) const {
121 
122  // need position of shell
123  const Eigen::Vector3d center = (grid_pos - _pos);
124  const double distsq = center.squaredNorm();
125 
126  // iterate over Gaussians in this shell
127  for (const AOGaussianPrimitive& gaussian : _gaussians) {
128 
129  const double alpha = gaussian.getDecay();
130  const double contraction = gaussian.getContraction();
131 
132  const double expofactor =
133  gaussian.getPowfactor() * std::exp(-alpha * distsq);
134  const Eigen::Vector3d second_term = -2.0 * alpha * center;
135 
136  switch (_l) {
137  case L::S: {
138  double AOvalue = contraction * expofactor;
139  AOvalues(0) += AOvalue; // s-function
140  gradAOvalues.row(0) += second_term * AOvalue; // gradient of s-function
141  } break;
142  case L::P: {
143  const double factor = 2. * sqrt(alpha) * contraction * expofactor;
144 
145  double AOvalue = factor * center.y(); // Y 1,-1
146  AOvalues(0) += AOvalue;
147  gradAOvalues.row(0) += second_term * AOvalue;
148  gradAOvalues(0, 1) += factor;
149 
150  AOvalue = factor * center.z(); // Y 1,0
151  AOvalues(1) += AOvalue;
152  gradAOvalues.row(1) += second_term * AOvalue;
153  gradAOvalues(1, 2) += factor;
154 
155  AOvalue = factor * center.x(); // Y 1,1
156  AOvalues(2) += AOvalue;
157  gradAOvalues(2, 0) += factor;
158  gradAOvalues.row(2) += second_term * AOvalue; // y gradient
159  } break;
160  case L::D: {
161  const double factor = 2. * alpha * contraction * expofactor;
162  const double factor_1 = factor / sqrt(3.);
163 
164  double AOvalue = 2. * factor * (center.x() * center.y()); // Y 2,-2
165  AOvalues(0) += AOvalue;
166  Eigen::Array3d coeff = {2 * center.y(), 2 * center.x(), 0};
167  gradAOvalues.row(0) += factor * coeff.matrix() + second_term * AOvalue;
168 
169  AOvalue = 2. * factor * (center.y() * center.z()); // Y 2,-1
170  AOvalues(1) += AOvalue;
171  coeff = {0, 2 * center.z(), 2 * center.y()};
172  gradAOvalues.row(1) += factor * coeff.matrix() + second_term * AOvalue;
173 
174  AOvalue = factor_1 * (3. * center.z() * center.z() - distsq); // Y 2,0
175  AOvalues(2) += AOvalue;
176  coeff = {-2, -2, 4};
177  gradAOvalues.row(2) += (factor_1 * coeff * center.array()).matrix() +
178  second_term * AOvalue;
179 
180  AOvalue = 2. * factor * (center.x() * center.z()); // Y 2,1
181  AOvalues(3) += AOvalue;
182  coeff = {2 * center.z(), 0, 2 * center.x()};
183  gradAOvalues.row(3) += factor * coeff.matrix() + second_term * AOvalue;
184 
185  AOvalue = factor *
186  (center.x() * center.x() - center.y() * center.y()); // Y 2,2
187  AOvalues(4) += AOvalue;
188  coeff = {2 * center.x(), -2 * center.y(), 0};
189  gradAOvalues.row(4) += factor * coeff.matrix() + second_term * AOvalue;
190  } break;
191  case L::F: {
192  const double factor = 2. * pow(alpha, 1.5) * contraction * expofactor;
193  const double factor_1 = factor * 2. / sqrt(15.);
194  const double factor_2 = factor * sqrt(2.) / sqrt(5.);
195  const double factor_3 = factor * sqrt(2.) / sqrt(3.);
196  AxA c(center);
197 
198  double AOvalue =
199  factor_3 * center.y() * (3. * c.xx() - c.yy()); // Y 3,-3
200  AOvalues(0) += AOvalue;
201  Eigen::Array3d coeff = {6. * c.xy(), 3. * (c.xx() - c.yy()), 0};
202  gradAOvalues.row(0) +=
203  factor_3 * coeff.matrix() + second_term * AOvalue;
204 
205  AOvalue = 4. * factor * center.x() * center.y() * center.z(); // Y 3,-2
206  AOvalues(1) += AOvalue;
207  coeff = {c.yz(), c.xz(), c.xy()};
208  gradAOvalues.row(1) +=
209  4 * factor * coeff.matrix() + second_term * AOvalue;
210 
211  AOvalue = factor_2 * center.y() * (5. * c.zz() - distsq); // Y 3,-1
212  AOvalues(2) += AOvalue;
213  coeff = {-2. * c.xy(), 4. * c.zz() - c.xx() - 3. * c.yy(), 8. * c.yz()};
214  gradAOvalues.row(2) +=
215  factor_2 * coeff.matrix() + second_term * AOvalue;
216 
217  AOvalue = factor_1 * center.z() * (5. * c.zz() - 3. * distsq); // Y 3,0
218  AOvalues(3) += AOvalue;
219  coeff = {-6. * c.xz(), -6. * c.yz(), 3. * (3. * c.zz() - distsq)};
220  gradAOvalues.row(3) +=
221  factor_1 * coeff.matrix() + second_term * AOvalue;
222 
223  AOvalue = factor_2 * center.x() * (5. * c.zz() - distsq); // Y 3,1
224  AOvalues(4) += AOvalue;
225  coeff = {4. * c.zz() - c.yy() - 3. * c.xx(), -2. * c.xy(), 8. * c.xz()};
226  gradAOvalues.row(4) +=
227  factor_2 * coeff.matrix() + second_term * AOvalue;
228 
229  AOvalue = 2. * factor * center.z() * (c.xx() - c.yy()); // Y 3,2
230  AOvalues(5) += AOvalue;
231  coeff = {2. * c.xz(), -2. * c.yz(), c.xx() - c.yy()};
232  gradAOvalues.row(5) +=
233  2 * factor * coeff.matrix() + second_term * AOvalue;
234 
235  AOvalue = factor_3 * center.x() * (c.xx() - 3. * c.yy()); // Y 3,3
236  AOvalues(6) += AOvalue;
237  coeff = {3. * (c.xx() - c.yy()), -6. * c.xy(), 0};
238  gradAOvalues.row(6) +=
239  factor_3 * coeff.matrix() + second_term * AOvalue;
240  } break;
241  case L::G: {
242  const double factor =
243  2. / sqrt(3.) * alpha * alpha * contraction * expofactor;
244  const double factor_1 = factor / sqrt(35.);
245  const double factor_2 = factor * 4. / sqrt(14.);
246  const double factor_3 = factor * 2. / sqrt(7.);
247  const double factor_4 = factor * 2. * sqrt(2.);
248  AxA c(center);
249 
250  double AOvalue = 4. * factor * c.xy() * (c.xx() - c.yy()); // Y 4,-4
251  AOvalues(0) += AOvalue;
252  Eigen::Array3d coeff = {center.y() * (3. * c.xx() - c.yy()),
253  center.x() * (c.xx() - 3. * c.yy()), 0};
254  gradAOvalues.row(0) +=
255  4 * factor * coeff.matrix() + second_term * AOvalue;
256 
257  AOvalue = factor_4 * c.yz() * (3. * c.xx() - c.yy()); // Y 4,-3
258  AOvalues(1) += AOvalue;
259  coeff = {6. * center.x() * c.yz(), 3. * center.z() * (c.xx() - c.yy()),
260  center.y() * (3. * c.xx() - c.yy())};
261  gradAOvalues.row(1) +=
262  factor_4 * coeff.matrix() + second_term * AOvalue;
263 
264  AOvalue = 2. * factor_3 * c.xy() * (7. * c.zz() - distsq); // Y 4,-2
265  AOvalues(2) += AOvalue;
266  coeff = {center.y() * (6. * c.zz() - 3. * c.xx() - c.yy()),
267  center.x() * (6. * c.zz() - c.xx() - 3. * c.yy()),
268  12. * center.z() * c.xy()};
269  gradAOvalues.row(2) +=
270  2 * factor_3 * coeff.matrix() + second_term * AOvalue;
271 
272  AOvalue = factor_2 * c.yz() * (7. * c.zz() - 3. * distsq); // Y 4,-1
273  AOvalues(3) += AOvalue;
274  coeff = {(-6. * center.x() * c.yz()),
275  center.z() * (4. * c.zz() - 3. * c.xx() - 9. * c.yy()),
276  3. * center.y() * (5. * c.zz() - distsq)};
277  gradAOvalues.row(3) +=
278  factor_2 * coeff.matrix() + second_term * AOvalue;
279 
280  AOvalue = factor_1 * (35. * c.zz() * c.zz() - 30. * c.zz() * distsq +
281  3. * distsq * distsq); // Y 4,0
282  AOvalues(4) += AOvalue;
283  coeff = {12. * center.x() * (distsq - 5. * c.zz()),
284  12. * center.y() * (distsq - 5. * c.zz()),
285  16. * center.z() * (5. * c.zz() - 3. * distsq)};
286 
287  gradAOvalues.row(4) +=
288  factor_1 * coeff.matrix() + second_term * AOvalue;
289 
290  AOvalue = factor_2 * c.xz() * (7. * c.zz() - 3. * distsq); // Y 4,1
291  AOvalues(5) += AOvalue;
292  coeff = {center.z() * (4. * c.zz() - 9. * c.xx() - 3. * c.yy()),
293  (-6. * center.y() * c.xz()),
294  3. * center.x() * (5. * c.zz() - distsq)};
295  gradAOvalues.row(5) +=
296  factor_2 * coeff.matrix() + second_term * AOvalue;
297 
298  AOvalue =
299  factor_3 * (c.xx() - c.yy()) * (7. * c.zz() - distsq); // Y 4,2
300  AOvalues(6) += AOvalue;
301  coeff = {4. * center.x() * (3. * c.zz() - c.xx()),
302  4. * center.y() * (c.yy() - 3. * c.zz()),
303  12. * center.z() * (c.xx() - c.yy())};
304  gradAOvalues.row(6) +=
305  factor_3 * coeff.matrix() + second_term * AOvalue;
306 
307  AOvalue = factor_4 * c.xz() * (c.xx() - 3. * c.yy()); // Y 4,3
308  AOvalues(7) += AOvalue;
309  coeff = {3. * center.z() * (c.xx() - c.yy()),
310  (-6. * center.y() * c.xz()),
311  center.x() * (c.xx() - 3. * c.yy())};
312  gradAOvalues.row(7) +=
313  factor_4 * coeff.matrix() + second_term * AOvalue;
314 
315  AOvalue = factor * (c.xx() * c.xx() - 6. * c.xx() * c.yy() +
316  c.yy() * c.yy()); // Y 4,4
317  AOvalues(8) += AOvalue;
318  coeff = {center.x() * (c.xx() - 3. * c.yy()),
319  center.y() * (c.yy() - 3. * c.xx()), 0};
320  gradAOvalues.row(8) +=
321  4 * factor * coeff.matrix() + second_term * AOvalue;
322  } break;
323  default:
324  throw std::runtime_error("Shell type:" + EnumToString(_l) +
325  " not known");
326  break;
327  }
328  } // contractions
329  return;
