111 void Fragment::calcPos(string tag) {
112  vec pos = vec(0,0,0);
113  double totWeight = 0.0;
114 
115  for (unsigned int i = 0; i< _atoms.size(); i++) {
116  if (tag == "MD") {
117  pos += _atoms[i]->getPos() * _atoms[i]->getWeight();
118  }
119  else if (tag == "QM") {
120  if (_atoms[i]->HasQMPart())
121  pos += _atoms[i]->getQMPos() * _atoms[i]->getWeight();
122  else
123  assert(_atoms[i]->getWeight() < 1e-6);
124  }
125  totWeight += _atoms[i]->getWeight();
126  }
127 
128  if (tag == "MD") {
129  _CoMD = pos / totWeight;
130  }
131  else if (tag == "QM") {
132  _CoQM = pos / totWeight;
133  }

140 void Fragment::Rigidify(bool Auto) {
141 
142  // +++++++++++++++++++++++++++++++++++++++++ //
143  // Establish reference atoms for local frame //
144  // +++++++++++++++++++++++++++++++++++++++++ //
145 
146  vector<Atom*> trihedron;
147 
148  if (Auto) {
149 
150  // Automated search for suitable atoms
151  bool enough = false;
152  vector< Atom* > ::iterator ait = this->Atoms().begin();
153 
154  while (!enough) {
155 
156  if (ait == this->Atoms().end()) { break; }
157  Atom *atm = *ait;
158  ait++;
159 
160  // When specifying mapping weights for rigid fragments,
161  // mobile atoms (whose position can vary significantly with
162  // respect to the rigid plane, for instance) should have a weight 0.
163  // We do not want to pick those to establish the local frame.
164  if (atm->getWeight() == 0.0) { continue; }
165  else {
166  trihedron.push_back(atm);
167  if (trihedron.size() == 3) { enough = true; }
168  }
169  }
170  }
171  else {
172 
173  // Take atoms specified in mapping file <= DEFAULT
174  vector<Atom*> ::iterator ait;
175  for (ait = _atoms.begin(); ait < _atoms.end(); ait++) {
176  if ( ! (*ait)->HasQMPart() ) { continue; }
177 
178  if ( (*ait)->getQMId() == _trihedron[0] ||
179  (*ait)->getQMId() == _trihedron[1] ||
180  (*ait)->getQMId() == _trihedron[2] ) {
181  Atom *atm = *ait;
182  trihedron.push_back(atm);
183  }
184  }
185  }
186 
187  _symmetry = trihedron.size();
188  // [-Wlogical-not-parentheses]
189  if ( trihedron.size() != _trihedron.size() ) {
190  cout << endl << "ERROR Local frame ill-defined" << flush;
191  }
192 
193  // +++++++++++++++++++++++ //
194  // Construct trihedra axes //
195  // +++++++++++++++++++++++ //
196 
197  vec xMD, yMD, zMD;
198  vec xQM, yQM, zQM;
199 
200  if (_symmetry == 3) {
201 
202  vec r1MD = trihedron[0]->getPos();
203  vec r2MD = trihedron[1]->getPos();
204  vec r3MD = trihedron[2]->getPos();
205  vec r1QM = trihedron[0]->getQMPos();
206  vec r2QM = trihedron[1]->getQMPos();
207  vec r3QM = trihedron[2]->getQMPos();
208 
209  xMD = r2MD - r1MD;
210  yMD = r3MD - r1MD;
211  xQM = r2QM - r1QM;
212  yQM = r3QM - r1QM;
213 
214  zMD = xMD ^ yMD;
215  zQM = xQM ^ yQM;
216 
217  yMD = zMD ^ xMD;
218  yQM = zQM ^ xQM;
219 
220  xMD = xMD.normalize();
221  yMD = yMD.normalize();
222  zMD = zMD.normalize();
223  xQM = xQM.normalize();
224  yQM = yQM.normalize();
225  zQM = zQM.normalize();
226 
227  }
228 
229  else if (_symmetry == 2) {
230 
231  vec r1MD = trihedron[0]->getPos();
232  vec r2MD = trihedron[1]->getPos();
233  vec r1QM = trihedron[0]->getQMPos();
234  vec r2QM = trihedron[1]->getQMPos();
235 
236  xMD = r2MD - r1MD;
237  xQM = r2QM - r1QM;
238 
239  // Normalising not necessary, but recommendable, when doing...
240  xMD = xMD.normalize();
241  xQM = xQM.normalize();
242 
243  vec yMDtmp = vec(0,0,0);
244  vec yQMtmp = vec(0,0,0);
245 
246  // ... this: Check whether one of the components is equal or close to
247  // zero. If so, this easily gives a second leg for the trihedron.
248  if ( xMD.getX()*xMD.getX() < 1e-6 ) { yMDtmp = vec(1,0,0); }
249  else if ( xMD.getY()*xMD.getY() < 1e-6 ) { yMDtmp = vec(0,1,0); }
250  else if ( xMD.getZ()*xMD.getZ() < 1e-6 ) { yMDtmp = vec(0,0,1); }
251  if ( xQM.getX()*xQM.getX() < 1e-6 ) { yQMtmp = vec(1,0,0); }
252  else if ( xQM.getY()*xQM.getY() < 1e-6 ) { yQMtmp = vec(0,1,0); }
253  else if ( xQM.getZ()*xQM.getZ() < 1e-6 ) { yQMtmp = vec(0,0,1); }
254 
255  if ( abs(yMDtmp) < 0.5 ) {
256  // All components of xMD are unequal to zero => division is safe.
257  // Choose vector from plane with xMD * inPlaneVec = 0:
258  double tmp_x = 1.;
259  double tmp_y = 1.;
260  double tmp_z = 1/xMD.getZ() * (-xMD.getX()*tmp_x - xMD.getY()*tmp_y);
261  yMDtmp = vec(tmp_x, tmp_y, tmp_z);
262  yMDtmp.normalize();
263  }
264  if ( abs(yQMtmp) < 0.5 ) {
265  double tmp_x = 1.;
266  double tmp_y = 1.;
267  double tmp_z = 1/xQM.getZ() * (-xQM.getX()*tmp_x - xQM.getY()*tmp_y);
268  yQMtmp = vec(tmp_x, tmp_y, tmp_z);
269  yQMtmp.normalize();
270  }
271 
272  // Now proceed as for symmetry 3
273  zMD = xMD ^ yMDtmp;
274  yMD = zMD ^ xMD;
275  zQM = xQM ^ yQMtmp;
276  yQM = zQM ^ xQM;
277 
278  xMD.normalize();
279  yMD.normalize();
280  zMD.normalize();
281  xQM.normalize();
282  yQM.normalize();
283  zQM.normalize();
284 
285  }
286 
287  else if (_symmetry == 1) {
288 
289  xMD = vec(1,0,0);
290  yMD = vec(0,1,0);
291  zMD = vec(0,0,1);
292  xQM = vec(1,0,0);
293  yQM = vec(0,1,0);
294  zQM = vec(0,0,1);
295 
296  }
297 
298  else {
299  cout << endl
300  << "NOTE: Invalid definition of local frame in fragment "
301  << this->getName();
302  cout << ". Assuming point particle for mapping. "
303  << endl;
304 
305  xMD = vec(1,0,0);
306  yMD = vec(0,1,0);
307  zMD = vec(0,0,1);
308  xQM = vec(1,0,0);
309  yQM = vec(0,1,0);
310  zQM = vec(0,0,1);
311 
312  }
313 
314  // +++++++++++++++++ //
315  // Rotation matrices //
316  // +++++++++++++++++ //
317 
318  matrix rotMD = matrix(xMD, yMD, zMD);
319  matrix rotQM = matrix(xQM, yQM, zQM);
320 
321  matrix rotateQM2MD = rotMD * rotQM.Transpose();
322  _rotateQM2MD = rotateQM2MD;
323 
324 
325  // ++++++++++++++++++ //
326  // Transform fragment //
327  // ++++++++++++++++++ //
328 
329  this->calcPos("QM");
330  this->RotTransQM2MD();
331  _translateQM2MD = _CoMD - _CoQM;
332 
333 }
