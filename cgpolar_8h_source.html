59 void CgPolar::Initialize(Property *opt) {
60 
61  _options = opt;
62  std::cout << std::endl
63  << "... ... Initialized with " << _nThreads << " threads. "
64  << std::flush;
65 
66  std::string key = "options.cgpolar";
67  if (opt->exists(key+".multipoles")) {
68  _xml_file = opt->get(key+".multipoles").as< std::string >();
69  }
70  else {
71  std::cout << std::endl;
72  throw std::runtime_error("No multipole mapping file provided");
73  }
74  key = "options.cgpolar.control";
75  if (opt->exists(key+".mps_table")) {
76  _mps_table = opt->get(key+".mps_table").as<std::string>();
77  }
78  else {
79  _mps_table = opt->get(key+".emp_file").as<std::string>();
80  }
81  if (opt->exists(key+".pdb_check")) {
82  _pdb_check = opt->get(key+".pdb_check").as<bool>();
83  }
84  else { _pdb_check = false; }
85  if (opt->exists(key+".mps_check")) {
86  _mps_check = opt->get(key+".mps_check").as<bool>();
87  }
88  else { _mps_check = false; }
89  if (opt->exists(key+".load_ptop_from")) {
90  _load_ptop_archfile = opt->get(key+".load_ptop_from").as<std::string>();
91  }
92  else { _load_ptop_archfile = ""; }
93 
94  key = "options.cgpolar.coarsegrain";
95  if (opt->exists(key+".cg_anisotropic")) {
96  _cg_anisotropic = opt->get(key+".cg_anisotropic").as<bool>();
97  }
98  else { _cg_anisotropic = false; }
99 
100  return;
101 }

106 bool CgPolar::EvaluateFrame(Topology *top) {
107 
108  // GENERATE POLAR TOPOLOGY
109  PolarTop ptop(top);
110  if (_load_ptop_archfile == "") {
111  std::cout << std::endl << "... ... Create polar topology from map" << std::flush;
112  std::cout << std::endl << "... ... ... Initialize MPS-mapper: " << std::flush;
113  _mps_mapper.GenerateMap(_xml_file, _mps_table, top);
114  _mps_mapper.Gen_BGN(top, &ptop, NULL);
115  }
116  else {
117  std::cout << std::endl << "... ... Load polar topology from hard drive: '"
118  << _load_ptop_archfile << "'" << std::flush;
119  ptop.LoadFromDrive(_load_ptop_archfile);
120  }
121  std::vector<PolarSeg*> bgn = ptop.BGN();
122 
123  if (_mps_check) {
124  std::string atomistic = "mps_mapped";
125  if (!boost::filesystem::exists(atomistic)) {
126  boost::filesystem::create_directory(atomistic);
127  }
128  for (std::vector<PolarSeg*>::iterator sit = bgn.begin();
129  sit < bgn.end(); ++sit) {
130  std::string mpsfile = atomistic + "/" + boost::lexical_cast<std::string>((*sit)->getId()) + ".mps";
131  (*sit)->WriteMPS(mpsfile, "<cgpolar> (atomistic)");
132  }
133  }
134 
135  // VERIFY INPUT: PDB, PTOP, XML
136  if (_pdb_check) ptop.PrintPDB("cgpolar.fine.pdb");
137  ptop.SaveToDrive("cgpolar.fine.ptop");
138  std::ofstream ofs;
139  ofs.open("cgpolar.fine.xml", std::ofstream::out);
140  for (vector<PolarSeg*>::iterator sit = bgn.begin();
141  sit < bgn.end(); ++sit) {
142  for (PolarSeg::iterator pit = (*sit)->begin();
143  pit < (*sit)->end(); ++pit) {
144  (*pit)->WriteXmlLine(ofs);
145  }
146  }
147  ofs.close();
148 
149  MolPolEngine engine = MolPolEngine(0.39, 0.30, 1024, 0.0001);
150 
151  for (std::vector<PolarSeg*>::iterator sit=bgn.begin();
152  sit<bgn.end(); ++sit) {
153  //matrix p0 = engine.CalculateMolPol(*(*sit), true);
154  engine.CalculateMolPol(*(*sit), true);
155  }
156 
157  // COARSE-GRAIN
158  std::cout << std::endl;
159  for (std::vector<PolarSeg*>::iterator sit=bgn.begin();
160  sit<bgn.end(); ++sit) {
161  //MolPolEngine engine = MolPolEngine();
162  //matrix p0 = engine.CalculateMolPol(*(*sit), true);
163  engine.CalculateMolPol(*(*sit), true);
164  //(*sit)->WriteMPS("cgpolar.fine.mps", "FINE");
165  std::cout << "\rCoarse-grain ID = " << (*sit)->getId() << std::flush;
166  (*sit)->Coarsegrain(_cg_anisotropic);
167  //(*sit)->WriteMPS("cgpolar.coarse.mps", "COARSE");
168  //matrix p1 = engine.CalculateMolPol(*(*sit), true);
169  //int a; cin >> a;
170  }
171 
172  if (_mps_check) {
173  std::string coarse = "mps_coarse";
174  if (!boost::filesystem::exists(coarse)) {
175  boost::filesystem::create_directory(coarse);
176  }
177  for (std::vector<PolarSeg*>::iterator sit = bgn.begin();
178  sit < bgn.end(); ++sit) {
179  std::string mpsfile = coarse + "/" + boost::lexical_cast<std::string>((*sit)->getId()) + ".mps";
180  (*sit)->WriteMPS(mpsfile, "<cgpolar> (coarse)");
181  }
182  }
183 
184  // VERIFY OUTPUT: PDB, PTOP, XML
185  if (_pdb_check) ptop.PrintPDB("cgpolar.coarse.pdb");
186  ptop.SaveToDrive("cgpolar.coarse.ptop");
187  ofs.open("cgpolar.coarse.xml", std::ofstream::out);
188  for (vector<PolarSeg*>::iterator sit = bgn.begin();
189  sit < bgn.end(); ++sit) {
190  for (PolarSeg::iterator pit = (*sit)->begin();
191  pit < (*sit)->end(); ++pit) {
192  (*pit)->WriteXmlLine(ofs);
193  }
194  }
195  ofs.close();
196 
197  return true;
198 }
