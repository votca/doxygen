71  std::vector<std::vector<double> > QMPackage::SplitMultipoles(ctp::APolarSite* aps){
72 
73  std::vector< std::vector<double> > multipoles_split;
74 
75  const tools::vec pos = aps->getPos() * tools::conv::nm2ang;
76  tools::vec tot_dpl = tools::vec(0.0);
77  if (_with_polarization) {
78  tot_dpl += aps->getU1();
79  }
80  if (aps->getRank() > 0) {
81  tot_dpl += aps->getQ1();
82  }
83  // Calculate virtual charge positions
84  double a = _dpl_spacing; // this is in nm
85  double mag_d = abs(tot_dpl); // this is in e * nm
86  if (mag_d > 1e-9) {
87  tools::vec dir_d = tot_dpl.normalize();
88  tools::vec A = pos + 0.5 * a * dir_d * tools::conv::nm2ang; // converted to AA
89  tools::vec B = pos - 0.5 * a * dir_d * tools::conv::nm2ang;
90  double qA = mag_d / a;
91  double qB = -qA;
92  multipoles_split.push_back({A.getX(),A.getY(),A.getZ(),qA});
93  multipoles_split.push_back({B.getX(),B.getY(),B.getZ(),qB});
94  }
95 
96 
97  if (aps->getRank() > 1 ) {
98  tools::matrix components = aps->getQ2cartesian();
99  tools::matrix::eigensystem_t system;
100  components.SolveEigensystem(system);
101  double a = 2*_dpl_spacing;
102  //string Atomnameplus[] = {"X", "Y", "Z"};
103  //string Atomnameminus[] = {"X", "Y", "Z"};
104  for (unsigned i = 0; i < 3; i++) {
105 
106  double q = system.eigenvalues[i] / (a * a);
107  if (std::abs(q) < 1e-9) {
108  continue;
109  }
110  tools::vec vec1 = pos + 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
111  tools::vec vec2 = pos - 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
112 
113  multipoles_split.push_back({vec1.getX(),vec1.getY(),vec1.getZ(),q});
114  multipoles_split.push_back({vec2.getX(),vec2.getY(),vec2.getZ(),q});
115 
116  }
117 
118  }
119 
120  return multipoles_split;
121  }

125  bool QMPackage::WriteInputFilePBC(ctp::QMPair* pair, Orbitals* orbitals) {
126 
127  //std::cout << "IDFT writes input with PBC" << std::endl;
128 
129  ctp::Segment* seg1 = pair->Seg1();
130  ctp::Segment* seg2 = pair->Seg2();
131  ctp::Segment* ghost = NULL;
132 
133  ctp::Topology* _top = seg1->getTopology();
134 
135  ctp::vec r1 = seg1->getPos();
136  ctp::vec r2 = seg2->getPos();
137 
138  ctp::vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
139 
140  // Check whether pair formed across periodic boundary
141  if (abs(r2 - r1 - _R) > 1e-8) {
142  ghost = new ctp::Segment(seg2);
143  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
144  std::vector<ctp::Atom*>::iterator ait;
145  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
146  (*ait)->setQMPos((*ait)->getQMPos() + r1 - r2 + _R);
147  }
148  }
149 
150  std::vector< ctp::Segment* > segments;
151  segments.push_back(seg1);
152 
153  if (ghost) {
154  segments.push_back(ghost);
155  } else {
156  segments.push_back(seg2);
157  }
158 
159  WriteInputFile(segments, orbitals);
160 
161  delete ghost;
162  return true;
163  }
