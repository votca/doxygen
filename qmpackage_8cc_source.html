60 void QMPackage::ReorderOutput(Orbitals& orbitals) const {
61  if (!orbitals.hasQMAtoms()) {
62  throw std::runtime_error("Orbitals object has no QMAtoms");
63  }
64 
65  AOBasis dftbasis = orbitals.SetupDftBasis();
66  // necessary to update nuclear charges on qmatoms
67  if (orbitals.hasECPName()) {
68  ECPBasisSet ecps;
69  ecps.Load(orbitals.getECPName());
70  ECPAOBasis ecpbasis;
71  ecpbasis.Fill(ecps, orbitals.QMAtoms());
72  }
73 
74  if (orbitals.hasMOs()) {
75  ReorderMOsToXTP(orbitals.MOs().eigenvectors(), dftbasis);
76  XTP_LOG(Log::info, *_pLog) << "Reordered MOs" << flush;
77  }
78 
79  return;
80 }

92 std::vector<QMPackage::MinimalMMCharge> QMPackage::SplitMultipoles(
93  const StaticSite& aps) const {
94 
95  std::vector<QMPackage::MinimalMMCharge> multipoles_split;
96  // Calculate virtual charge positions
97  double a = _settings.get<double>("dipole_spacing"); // this is in a0
98  double mag_d = aps.getDipole().norm(); // this is in e * a0
99  const Eigen::Vector3d dir_d = aps.getDipole().normalized();
100  const Eigen::Vector3d A = aps.getPos() + 0.5 * a * dir_d;
101  const Eigen::Vector3d B = aps.getPos() - 0.5 * a * dir_d;
102  double qA = mag_d / a;
103  double qB = -qA;
104  if (std::abs(qA) > 1e-12) {
105  multipoles_split.push_back(MinimalMMCharge(A, qA));
106  multipoles_split.push_back(MinimalMMCharge(B, qB));
107  }
108 
109  if (aps.getRank() > 1) {
110  const Eigen::Matrix3d components = aps.CalculateCartesianMultipole();
111  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
112  es.computeDirect(components);
113  double a2 = 2 * a;
114  for (Index i = 0; i < 3; i++) {
115  double q = es.eigenvalues()[i] / (a2 * a2);
116  const Eigen::Vector3d vec1 =
117  aps.getPos() + 0.5 * a2 * es.eigenvectors().col(i);
118  const Eigen::Vector3d vec2 =
119  aps.getPos() - 0.5 * a2 * es.eigenvectors().col(i);
120  multipoles_split.push_back(MinimalMMCharge(vec1, q));
121  multipoles_split.push_back(MinimalMMCharge(vec2, q));
122  }
123  }
124  return multipoles_split;
125 }

143 void QMPackage::ReorderMOs(Eigen::MatrixXd& v,
144  const std::vector<Index>& order) const {
145  // Sanity check
146  if (v.rows() != Index(order.size())) {
147  throw std::runtime_error("Size mismatch in ReorderMOs " +
148  std::to_string(v.rows()) + ":" +
149  std::to_string(order.size()));
150  }
151  // actual swapping of coefficients
152  for (Index s = 1, d; s < (Index)order.size(); ++s) {
153  for (d = order[s]; d < s; d = order[d]) {
154  ;
155  }
156  if (d == s) {
157  while (d = order[d], d != s) {
158  v.row(s).swap(v.row(d));
159  }
160  }
161  }
162 }

175 void QMPackage::ReorderMOsToNative(Eigen::MatrixXd& v,
176  const AOBasis& basis) const {
177 
178  std::vector<Index> multiplier = getMultiplierVector(basis);
179  MultiplyMOs(v, multiplier);

181  std::vector<Index> reverseorder = invertOrder(order);

186 void QMPackage::MultiplyMOs(Eigen::MatrixXd& v,
187  const std::vector<Index>& multiplier) const {
188  // Sanity check
189  if (v.cols() != Index(multiplier.size())) {
190  std::cerr << "Size mismatch in MultiplyMOs" << v.cols() << ":"
191  << multiplier.size() << std::endl;
192  throw std::runtime_error("Abort!");
193  }
194  for (Index i = 0; i < v.cols(); i++) {
195  v.row(i) = multiplier[i] * v.row(i);
196  }
197  return;
198 }

200 std::vector<Index> QMPackage::getMultiplierVector(const AOBasis& basis) const {
201  std::vector<Index> multiplier;
202  multiplier.reserve(basis.AOBasisSize());
203  // go through basisset
204  for (const AOShell& shell : basis) {
205  std::vector<Index> shellmultiplier = getMultiplierShell(shell);
206  multiplier.insert(multiplier.end(), shellmultiplier.begin(),
207  shellmultiplier.end());
208  }

221 std::vector<Index> QMPackage::getReorderVector(const AOBasis& basis) const {
222  std::vector<Index> reorder;
223  reorder.reserve(basis.AOBasisSize());
224  // go through basisset
225  for (const AOShell& shell : basis) {
226  std::vector<Index> shellreorder = getReorderShell(shell);
227  std::for_each(shellreorder.begin(), shellreorder.end(),
228  [&reorder](Index& i) { i += Index(reorder.size()); });
229  reorder.insert(reorder.end(), shellreorder.begin(), shellreorder.end());
230  }
231  return reorder;
232 }

234 std::vector<Index> QMPackage::getReorderShell(const AOShell& shell) const {
235  std::vector<Index> reorder;

237  // Shellreorder tells how much a certain funktion has to be shifted with

240  }

244 std::vector<Index> QMPackage::invertOrder(
245  const std::vector<Index>& order) const {
246 
247  std::vector<Index> neworder = std::vector<Index>(order.size());
248  for (Index i = 0; i < Index(order.size()); i++) {
