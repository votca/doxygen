28  void QMPackage::ReorderOutput(Orbitals& orbitals) {
29  BasisSet dftbasisset;
30  dftbasisset.LoadBasisSet(_basisset_name);
31  if (!orbitals.hasQMAtoms()) {
32  throw std::runtime_error("Orbitals object has no QMAtoms");
33  }
34 
35  AOBasis dftbasis;
36  dftbasis.AOBasisFill(dftbasisset, orbitals.QMAtoms());
37  //necessary to update nuclear charges on qmatoms
38  if (_write_pseudopotentials) {
39  BasisSet ecps;
40  ecps.LoadPseudopotentialSet(_ecp_name);
41  AOBasis ecpbasis;
42  ecpbasis.ECPFill(ecps, orbitals.QMAtoms());
43  }
44 
45  if (orbitals.hasAOOverlap()) {
46  dftbasis.ReorderMatrix(orbitals.AOOverlap(), getPackageName(), "xtp");
47  CTP_LOG(ctp::logDEBUG, *_pLog) << "Reordered Overlap matrix" << flush;
48  }
49  if (orbitals.hasAOVxc()) {
50  dftbasis.ReorderMatrix(orbitals.AOVxc(), getPackageName(), "xtp");
51  CTP_LOG(ctp::logDEBUG, *_pLog) << "Reordered VXC matrix" << flush;
52  }
53  if (orbitals.hasMOCoefficients()) {
54  dftbasis.ReorderMOs(orbitals.MOCoefficients(), getPackageName(), "xtp");
55  CTP_LOG(ctp::logDEBUG, *_pLog) << "Reordered MOs" << flush;
56  }
57 
58  return;

73  std::vector<std::vector<double> > QMPackage::SplitMultipoles(ctp::APolarSite* aps) {
74 
75  std::vector< std::vector<double> > multipoles_split;
76 
77  const tools::vec pos = aps->getPos() * tools::conv::nm2ang;
78  tools::vec tot_dpl = tools::vec(0.0);
79  if (_with_polarization) {
80  tot_dpl += aps->getU1();
81  }
82  if (aps->getRank() > 0) {
83  tot_dpl += aps->getQ1();
84  }
85  // Calculate virtual charge positions
86  double a = _dpl_spacing; // this is in nm
87  double mag_d = abs(tot_dpl); // this is in e * nm
88  if (mag_d > 1e-9) {
89  tools::vec dir_d = tot_dpl.normalize();
90  tools::vec A = pos + 0.5 * a * dir_d * tools::conv::nm2ang; // converted to AA
91  tools::vec B = pos - 0.5 * a * dir_d * tools::conv::nm2ang;
92  double qA = mag_d / a;
93  double qB = -qA;
94  multipoles_split.push_back({A.getX(), A.getY(), A.getZ(), qA});
95  multipoles_split.push_back({B.getX(), B.getY(), B.getZ(), qB});
96  }
97 
98  if (aps->getRank() > 1) {
99  tools::matrix components = aps->getQ2cartesian();
100  tools::matrix::eigensystem_t system;
101  components.SolveEigensystem(system);
102  double a = 2 * _dpl_spacing;
103  for (unsigned i = 0; i < 3; i++) {
104 
105  double q = system.eigenvalues[i] / (a * a);
106  if (std::abs(q) < 1e-9) {
107  continue;
108  }
109  tools::vec vec1 = pos + 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
110  tools::vec vec2 = pos - 0.5 * a * system.eigenvecs[i] * tools::conv::nm2ang;
111 
112  multipoles_split.push_back({vec1.getX(), vec1.getY(), vec1.getZ(), q});
113  multipoles_split.push_back({vec2.getX(), vec2.getY(), vec2.getZ(), q});
114 
115  }
116  }
117  return multipoles_split;
118  }
