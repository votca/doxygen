51  _spin = options.ifExistsReturnElseThrowRuntimeError<int>(key + ".spin");
52  _cleanup =

75 void QMPackage::ReorderOutput(Orbitals& orbitals) const {
76  if (!orbitals.hasQMAtoms()) {
77  throw std::runtime_error("Orbitals object has no QMAtoms");
78  }
79 
80  AOBasis dftbasis = orbitals.SetupDftBasis();
81  // necessary to update nuclear charges on qmatoms
82  if (orbitals.hasECPName()) {
83  ECPBasisSet ecps;
84  ecps.Load(orbitals.getECPName());
85  ECPAOBasis ecpbasis;
86  ecpbasis.Fill(ecps, orbitals.QMAtoms());
87  }
88 
89  if (orbitals.hasMOs()) {
90  dftbasis.ReorderMOs(orbitals.MOs().eigenvectors(), getPackageName(), "xtp");
91  XTP_LOG(logDEBUG, *_pLog) << "Reordered MOs" << flush;
92  }
93 
94  return;
95 }

110 std::vector<QMPackage::MinimalMMCharge> QMPackage::SplitMultipoles(
111  const StaticSite& aps) const {
112 
113  std::vector<QMPackage::MinimalMMCharge> multipoles_split;
114  // Calculate virtual charge positions
115  double a = _dpl_spacing; // this is in a0
116  double mag_d = aps.getDipole().norm(); // this is in e * a0
117  const Eigen::Vector3d dir_d = aps.getDipole().normalized();
118  const Eigen::Vector3d A = aps.getPos() + 0.5 * a * dir_d;
119  const Eigen::Vector3d B = aps.getPos() - 0.5 * a * dir_d;
120  double qA = mag_d / a;
121  double qB = -qA;
122  if (std::abs(qA) > 1e-12) {
123  multipoles_split.push_back(MinimalMMCharge(A, qA));
124  multipoles_split.push_back(MinimalMMCharge(B, qB));
125  }
126 
127  if (aps.getRank() > 1) {
128  const Eigen::Matrix3d components = aps.CalculateCartesianMultipole();
129  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
130  es.computeDirect(components);
131  double a = 2 * _dpl_spacing;
132  for (int i = 0; i < 3; i++) {
133  double q = es.eigenvalues()[i] / (a * a);
134  const Eigen::Vector3d vec1 =
135  aps.getPos() + 0.5 * a * es.eigenvectors().col(i);
136  const Eigen::Vector3d vec2 =
137  aps.getPos() - 0.5 * a * es.eigenvectors().col(i);
138  multipoles_split.push_back(MinimalMMCharge(vec1, q));
139  multipoles_split.push_back(MinimalMMCharge(vec2, q));
140  }
141  }
142  return multipoles_split;
143 }
