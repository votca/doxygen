63 void QMPackage::ReorderOutput(Orbitals& orbitals) const {
64  if (!orbitals.hasQMAtoms()) {
65  throw std::runtime_error("Orbitals object has no QMAtoms");
66  }
67 
68  AOBasis dftbasis = orbitals.SetupDftBasis();
69  // necessary to update nuclear charges on qmatoms
70  if (orbitals.hasECPName()) {
71  ECPBasisSet ecps;
72  ecps.Load(orbitals.getECPName());
73  ECPAOBasis ecpbasis;
74  ecpbasis.Fill(ecps, orbitals.QMAtoms());
75  }
76 
77  if (orbitals.hasMOs()) {
78  OrbReorder reorder(ShellReorder(), ShellMulitplier());
79  reorder.reorderOrbitals(orbitals.MOs().eigenvectors(), dftbasis);
80  XTP_LOG(Log::info, *_pLog) << "Reordered MOs" << flush;
81  }
82 
83  return;
84 }

98 std::vector<QMPackage::MinimalMMCharge> QMPackage::SplitMultipoles(
99  const StaticSite& aps) const {
100 
101  std::vector<QMPackage::MinimalMMCharge> multipoles_split;
102  // Calculate virtual charge positions
103  double a = _settings.get<double>("dipole_spacing"); // this is in a0
104  double mag_d = aps.getDipole().norm(); // this is in e * a0
105  const Eigen::Vector3d dir_d = aps.getDipole().normalized();
106  const Eigen::Vector3d A = aps.getPos() + 0.5 * a * dir_d;
107  const Eigen::Vector3d B = aps.getPos() - 0.5 * a * dir_d;
108  double qA = mag_d / a;
109  double qB = -qA;
110  if (std::abs(qA) > 1e-12) {
111  multipoles_split.push_back(MinimalMMCharge(A, qA));
112  multipoles_split.push_back(MinimalMMCharge(B, qB));
113  }
114 
115  if (aps.getRank() > 1) {
116  const Eigen::Matrix3d components = aps.CalculateCartesianMultipole();
117  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> es;
118  es.computeDirect(components);
119  double a2 = 2 * a;
120  for (Index i = 0; i < 3; i++) {
121  double q = es.eigenvalues()[i] / (a2 * a2);
122  const Eigen::Vector3d vec1 =
123  aps.getPos() + 0.5 * a2 * es.eigenvectors().col(i);
124  const Eigen::Vector3d vec2 =
125  aps.getPos() - 0.5 * a2 * es.eigenvectors().col(i);
126  multipoles_split.push_back(MinimalMMCharge(vec1, q));
127  multipoles_split.push_back(MinimalMMCharge(vec2, q));
128  }
129  }
130  return multipoles_split;
131 }
