54 int main(int argc, char** argv)
55 {
56  string name_in, name_out, name_unpack;
57  // program options
58  po::options_description desc("Allowed options");
59 
60  desc.add_options()
61  ("in", po::value<string>(&name_in), "files to read")
62  ("out", po::value<string>(&name_out), "files to write")
63  ("unpack", po::value<string>(&name_unpack), "extract all tables from this file")
64  ("help", "display help message");
65 
66  po::variables_map vm;
67  try {
68  po::store(po::parse_command_line(argc, argv, desc), vm);
69  po::notify(vm);
70  }
71  catch(po::error err) {
72  cout << "error parsing command line: " << err.what() << endl;
73  return -1;
74  }
75 
76  // does the user want help?
77  if (vm.count("help")) {
78  help_text();
79  cout << desc << endl;
80  return 0;
81  }
82 
83  check_option(desc, vm, "in");
84 
85  ub::vector<double> r;
86  ub::vector<double> dS;
87  ub::matrix<double> gmc;
88  vector<string> names;
89  vector<RangeParser> ranges;
90 
91  imcio_read_index(name_in + ".idx", names, ranges);
92 
93  if(vm.count("unpack")) {
94  RangeParser *cur_rp;
95  Table tbl_in;
96  tbl_in.Load(name_unpack);
97  vector<string>::iterator iter_name = names.begin();
98  vector<RangeParser>::iterator iter_range = ranges.begin();
99 
100  while(iter_name != names.end()) {
101  cur_rp = &(*iter_range);
102  Table tbl;
103  for(RangeParser::iterator ir=cur_rp->begin(); ir!=cur_rp->end(); ++ir) {
104  tbl.push_back(tbl_in.x(*ir-1), tbl_in.y(*ir-1), 'i');
105  }
106  tbl.Save(*iter_name + ".dpot.imc");
107  ++iter_name;
108  ++iter_range;
109  }
110  } else {
111  check_option(desc, vm, "out");
112  RangeParser *cur_rp;
113 
114  vector<string>::iterator iter_name = names.begin();
115  vector<RangeParser>::iterator iter_range = ranges.begin();
116  int beg=1;
117  int end=0;
118  list<int> list;
119 
120  imcio_read_dS(name_in + ".imc", r, dS);
121  imcio_read_matrix(name_in + ".gmc", gmc);
122 
123  while(iter_name != names.end()) {
124  cur_rp = &(*iter_range);
125  for(RangeParser::iterator ir=cur_rp->begin(); ir!=cur_rp->end(); ++ir) {
126  for(size_t i=0; i<gmc.size1(); ++i)
127  if(fabs(gmc(i,*ir-1)) > 1e-8) {
128  list.push_back(*ir-1);
129  end++;
130  break;
131  }
132  }
133  RangeParser new_rp;
134  new_rp.Add(beg, end);
135  beg=end+1;
136  *iter_range = new_rp;
137  ++iter_name;
138  ++iter_range;
139  }
140  imcio_write_dS(name_out + ".imc", r, dS, &list);
141  imcio_write_matrix(name_out + ".gmc", gmc, &list);
142  imcio_write_index(name_out + ".idx", names, ranges);
143  }
144 
145 }
