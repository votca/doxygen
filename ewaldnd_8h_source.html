35 // NOTE: This is not a conventional 3D Ewald summation, so use carefully
36 // (tuned for the purpose of cluster energy calculations)

41 // foreground charge density (FGC, FGN).

111  // FOREGROUND TRACKER
112  class ForegroundTable
113  {
114  public:
115  typedef boost::multi_array<bool,4> fgtable_t;
116  typedef fgtable_t::index idx_t;
117 
118  ForegroundTable(int n_segs_cell, int na_max, int nb_max, int nc_max)
119  : _na_max(na_max), _nb_max(nb_max), _nc_max(nc_max),
120  _id_na_nb_nc__inFg(
121  fgtable_t(boost::extents [n_segs_cell]
122  [2*na_max+1]
123  [2*nb_max+1]
124  [2*nc_max+1]))
125  {
126  for (idx_t i = 0; i < n_segs_cell; ++i) {
127  for (idx_t a = 0; a < 2*na_max+1; ++a) {
128  for (idx_t b = 0; b < 2*nb_max+1; ++b) {
129  for (idx_t c = 0; c < 2*nc_max+1; ++c) {
130  _id_na_nb_nc__inFg[i][a][b][c] = false;
131  }}}}
132  }
133 
134  void AddToForeground(int segid, int na, int nb, int nc) {
135  idx_t i = segid-1;
136  idx_t a = na + _na_max;
137  idx_t b = nb + _nb_max;
138  idx_t c = nc + _nc_max;
139  assert (_id_na_nb_nc__inFg[i][a][b][c] == false);
140  _id_na_nb_nc__inFg[i][a][b][c] = true;
141  }
142 
143  bool IsInForeground(int segid, int na, int nb, int nc) {
144  bool is_in_fg;
145  if (std::abs(na) > _na_max
146  || std::abs(nb) > _nb_max
147  || std::abs(nc) > _nc_max) {
148  is_in_fg = false;
149  }
150  else {
151  idx_t i = segid-1;
152  idx_t a = na + _na_max;
153  idx_t b = nb + _nb_max;
154  idx_t c = nc + _nc_max;
155  is_in_fg = _id_na_nb_nc__inFg[i][a][b][c];
156  }
157  return is_in_fg;
158  }

165  int _nc_max;
166  fgtable_t _id_na_nb_nc__inFg;
