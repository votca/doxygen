26template <bool T>
27StaticSegment Populationanalysis<T>::CalcChargeperAtom(
28 const Orbitals& orbitals, const QMState& state) const {
29
30 AOBasis basis = orbitals.getDftBasis();
31 Eigen::MatrixXd dmat = orbitals.DensityMatrixFull(state);
32 AOOverlap overlap;
33 overlap.Fill(basis);
34 Eigen::VectorXd charges = -CalcElecChargeperAtom(dmat, overlap, basis);
35 if (!state.isTransition()) {
36 charges += CalcNucChargeperAtom(orbitals.QMAtoms());
37 }
38
39 StaticSegment seg =
40 StaticSegment(orbitals.QMAtoms().getType(), orbitals.QMAtoms().getId());
41 for (Index i = 0; i < orbitals.QMAtoms().size(); ++i) {
42 seg.push_back(StaticSite(orbitals.QMAtoms()[i], charges(i)));
43 }
44 return seg;
45}

47template <bool T>
48void Populationanalysis<T>::CalcChargeperFragment(
49 std::vector<QMFragment<BSE_Population> >& frags, const Orbitals& orbitals,
50 QMStateType type) const {
51 if (!type.isExciton()) {
52 throw std::runtime_error(
53 "CalcChargeperFragment: QmStateType must be an exciton");
54 }
55 AOBasis basis = orbitals.getDftBasis();
56 AOOverlap overlap;
57 overlap.Fill(basis);
58 Eigen::VectorXd nuccharges = CalcNucChargeperAtom(orbitals.QMAtoms());
59 Eigen::MatrixXd dmatgs = orbitals.DensityMatrixGroundState();
60 Eigen::VectorXd gscharges =
61 nuccharges - CalcElecChargeperAtom(dmatgs, overlap, basis);
62 Index numofstates = orbitals.NumberofStates(type);
63 for (auto& frag : frags) {
64 frag.value().Initialize(numofstates);
65 frag.value().Gs = frag.ExtractFromVector(gscharges);
66 }
67 for (Index i_state = 0; i_state < numofstates; i_state++) {
68 QMState state(type, i_state, false);
69 std::array<Eigen::MatrixXd, 2> dmat_ex =
70 orbitals.DensityMatrixExcitedState(state);
71 Eigen::VectorXd atom_h = CalcElecChargeperAtom(dmat_ex[0], overlap, basis);
72 Eigen::VectorXd atom_e = -CalcElecChargeperAtom(dmat_ex[1], overlap, basis);
73 for (auto& frag : frags) {
74 frag.value().E(i_state) = frag.ExtractFromVector(atom_e);
75 frag.value().H(i_state) = frag.ExtractFromVector(atom_h);
76 }
77 }
78}

80template <bool T>
81Eigen::VectorXd Populationanalysis<T>::CalcNucChargeperAtom(
82 const QMMolecule& mol) const {
83 Eigen::VectorXd result = Eigen::VectorXd::Zero(mol.size());
84 for (Index i = 0; i < mol.size(); i++) {
85 result(i) = double(mol[i].getNuccharge());
86 }
87 return result;
88}

90template <bool T>
91Eigen::VectorXd Populationanalysis<T>::CalcElecChargeperAtom(
92 const Eigen::MatrixXd& dmat, AOOverlap& overlap,
93 const AOBasis& basis) const {
94 Eigen::MatrixXd prodmat;
95 if (T) {
96 Eigen::MatrixXd Smsqrt = overlap.Sqrt();
97 prodmat = Smsqrt * dmat * Smsqrt;
98 } else {
99 prodmat = dmat * overlap.Matrix();
100 }
101 std::vector<Index> funcperatom = basis.getFuncPerAtom();
102 Index noofatoms = Index(funcperatom.size());
103 Eigen::VectorXd charges = Eigen::VectorXd::Zero(noofatoms);
104 Index start = 0;
105 for (Index i = 0; i < charges.size(); ++i) {
106 Index nofunc = funcperatom[i];
107 charges(i) = prodmat.diagonal().segment(start, nofunc).sum();
108 start += nofunc;
109 }
110 return charges;
111}
