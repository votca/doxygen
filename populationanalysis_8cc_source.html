25 template <bool T>
26 StaticSegment Populationanalysis<T>::CalcChargeperAtom(
27  const Orbitals& orbitals, const QMState& state) const {
28 
29  AOBasis basis = orbitals.SetupDftBasis();
30  Eigen::MatrixXd dmat = orbitals.DensityMatrixFull(state);
31  AOOverlap overlap;
32  overlap.Fill(basis);
33  Eigen::VectorXd charges = -CalcElecChargeperAtom(dmat, overlap, basis);
34  if (!state.isTransition()) {
35  charges += CalcNucChargeperAtom(orbitals.QMAtoms());
36  }
37 
38  StaticSegment seg =
39  StaticSegment(orbitals.QMAtoms().getType(), orbitals.QMAtoms().getId());
40  for (int i = 0; i < orbitals.QMAtoms().size(); ++i) {
41  seg.push_back(StaticSite(orbitals.QMAtoms()[i], charges(i)));
42  }
43  return seg;
44 }

46 template <bool T>
47 void Populationanalysis<T>::CalcChargeperFragment(
48  std::vector<QMFragment<BSE_Population> >& frags, const Orbitals& orbitals,
49  QMStateType type) const {
50  if (!type.isExciton()) {
51  throw std::runtime_error(
52  "CalcChargeperFragment: QmStateType must be an exciton");
53  }
54  AOBasis basis = orbitals.SetupDftBasis();
55  AOOverlap overlap;
56  overlap.Fill(basis);
57  Eigen::VectorXd nuccharges = CalcNucChargeperAtom(orbitals.QMAtoms());
58  Eigen::MatrixXd dmatgs = orbitals.DensityMatrixGroundState();
59  Eigen::VectorXd gscharges =
60  nuccharges - CalcElecChargeperAtom(dmatgs, overlap, basis);
61  int numofstates = orbitals.NumberofStates(type);
62  for (auto& frag : frags) {
63  frag.value().Initialize(numofstates);
64  frag.value().Gs = frag.ExtractFromVector(gscharges);
65  }
66  for (int i_state = 0; i_state < numofstates; i_state++) {
67  QMState state(type, i_state, false);
68  std::array<Eigen::MatrixXd, 2> dmat_ex =
69  orbitals.DensityMatrixExcitedState(state);
70  Eigen::VectorXd atom_h = CalcElecChargeperAtom(dmat_ex[0], overlap, basis);
71  Eigen::VectorXd atom_e = -CalcElecChargeperAtom(dmat_ex[1], overlap, basis);
72  for (auto& frag : frags) {
73  frag.value().E(i_state) = frag.ExtractFromVector(atom_e);
74  frag.value().H(i_state) = frag.ExtractFromVector(atom_h);
75  }
76  }
77 }

79 template <bool T>

81  const QMMolecule& mol) const {

83  for (int i = 0; i < mol.size(); i++) {

89 template <bool T>
90 Eigen::VectorXd Populationanalysis<T>::CalcElecChargeperAtom(
91  const Eigen::MatrixXd& dmat, AOOverlap& overlap,
92  const AOBasis& basis) const {
93  Eigen::MatrixXd prodmat;
94  if (T) {
95  Eigen::MatrixXd Smsqrt = overlap.Sqrt();
96  prodmat = Smsqrt * dmat * Smsqrt;
97  } else {
98  prodmat = dmat * overlap.Matrix();
99  }
100  int noofatoms = basis.getFuncPerAtom().size();
101  Eigen::VectorXd charges = Eigen::VectorXd::Zero(noofatoms);
102  int start = 0;
103  for (int i = 0; i < charges.size(); ++i) {
104  int nofunc = basis.getFuncPerAtom()[i];
105  charges(i) = prodmat.diagonal().segment(start, nofunc).sum();
106  start += nofunc;
107  }
108  return charges;
109 }
