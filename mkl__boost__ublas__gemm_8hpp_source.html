50  typename matrix_matrix_binary_traits<typename E1::value_type, E1,
51  typename E2::value_type, E2>::result_type
52  gemm(const matrix_expression<E1> &e1,
53  const matrix_expression<E2> &e2)
54  {
55  typedef typename matrix_matrix_binary_traits<typename E1::value_type, E1,
56  typename E2::value_type, E2>::storage_category storage_category;
57  typedef typename matrix_matrix_binary_traits<typename E1::value_type, E1,
58  typename E2::value_type, E2>::orientation_category orientation_category;
59  return prod (e1, e2, storage_category (), orientation_category ());
60  } // For unsupported matrix types.

215  template<class E1, class E2, class T, class F, class A>
216  MKL_BOOST_UBLAS_INLINE
217  void
218  gemm(const CBLAS_TRANSPOSE transa, const CBLAS_TRANSPOSE transb,
219  const matrix_expression<E1> &a, const matrix_expression<E2> &b,
220  matrix<T,F,A> &c)
221  {
222  typedef T value_type;
223  typedef typename F::orientation_category orientation_category;
224  if(supported_type(value_type()) > 0) {
225  MKL_INT m = a().size1();
226  MKL_INT k = a().size2();
227  MKL_INT n = b().size2();
228  MKL_INT lda = leading_dimension(a(),orientation_category());
229  MKL_INT ldb = leading_dimension(b(),orientation_category());
230  MKL_INT ldc = leading_dimension(c,orientation_category());
231  value_type alpha = OneElement(value_type()); // boost/numeric/ublas/detail/concepts.hpp
232  value_type beta = ZeroElement(value_type()); // boost/numeric/ublas/detail/concepts.hpp
233  CBLAS_ORDER layout = storage_layout(orientation_category());
234  T *pa = pointer(a());
235  T *pb = pointer(b());
236  T *pc = pointer(c);
237  gemm(layout, transa, transb, m, n, k, alpha, pa, lda, pb, ldb, beta, pc, ldc);
238  } else {
239  c = gemm(a,b);
240  }
241  }
