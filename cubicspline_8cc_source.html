00030 void CubicSpline::Interpolate(ub::vector<double> &x, ub::vector<double> &y)
00031 {    
00032     if(x.size() != y.size())
00033         throw std::invalid_argument("error in CubicSpline::Interpolate : sizes of vectors x and y do not match");
00034     
00035     if(x.size()<3)
00036         throw std::invalid_argument("error in CubicSpline::Interpolate : vectors x and y have to contain at least 3 points");
00037 
00038     const int N = x.size();
00039     
00040     // adjust the grid
00041     _r.resize(N);
00042     _f.resize(N);
00043     _f2.resize(N);
00044     
00045     // create vector proxies to individually access f and f''
00046 
00047     // copy the grid points into f
00048     _r = x;
00049     _f = y;
00050     _f2 = ub::zero_vector<double>(N);
00051     
00052     // not calculate the f''
00053     ub::matrix<double> A(N, N);
00054     A = ub::zero_matrix<double>(N,N);
00055     
00056     for(int i=0; i<N - 2; ++i) {
00057             _f2(i+1) = -( A_prime_l(i)*_f(i)
00058             + (B_prime_l(i) - A_prime_r(i)) * _f(i+1)
00059             -B_prime_r(i) * _f(i+2));
00060 
00061             A(i+1, i) = C_prime_l(i);
00062             A(i+1, i+1) = D_prime_l(i) - C_prime_r(i);
00063             A(i+1, i+2) = -D_prime_r(i);
00064     }
00065     
00066     switch(_boundaries) {
00067         case splineNormal:
00068             A(0, 0) = 1;
00069             A(N - 1, N-1) = 1;
00070             break;
00071         case splinePeriodic:
00072             A(0,0) = 1; A(0,N-1) = -1;
00073             A(N-1,0) = 1; A(N-1,N-1) = -1;
00074             break;
00075         case splineDerivativeZero:
00076             throw std::runtime_error("erro in CubicSpline::Interpolate: case splineDerivativeZero not implemented yet");
00077             break;
00078     }
00079 
00080     votca::tools::linalg_qrsolve(_f2, A, _f2);
00081 }

00083 void CubicSpline::Fit(ub::vector<double> &x, ub::vector<double> &y)
00084 {
00085     if(x.size() != y.size())
00086         throw std::invalid_argument("error in CubicSpline::Fit : sizes of vectors x and y do not match");
00087     
00088     const int N = x.size();
00089     const int ngrid = _r.size();
00090     
00091     // construct the equation
00092     // A*u = b
00093     // where u = { {f[i]}, {f''[i]} }
00094     // and b[i] = y[i] for 0<=i<N
00095     // and b[i]=0 for i>=N (for smoothing condition)
00096     // A[i,j] contains the data fitting + the spline smoothing conditions
00097     
00098     ub::matrix<double> A(N, 2*ngrid);
00099     ub::vector<double> b(N);    
00100     ub::matrix<double> B_constr(ngrid, 2*ngrid);  // Matrix with smoothing conditions
00101 
00102     A = ub::zero_matrix<double>(N, 2*ngrid);
00103     b  = ub::zero_vector<double>(N);
00104     B_constr = ub::zero_matrix<double>(ngrid, 2*ngrid);
00105     
00106     // Construct smoothing matrix
00107     AddBCToFitMatrix(B_constr, 0);
00108 
00109     // construct the matrix to fit the points and the vector b
00110     AddToFitMatrix(A, x, 0);
00111     b = -y; // why is it -y?
00112 
00113     // now do a constrained qr solve
00114     ub::vector<double> sol(2*ngrid);
00115     votca::tools::linalg_constrained_qrsolve(sol, A, b, B_constr);
00116 
00117     // check vector "sol" for nan's
00118     for(int i=0; i<2*ngrid; i++) {
00119         if( (isinf(sol(i))) || (isnan(sol(i))) ) {
00120             throw std::runtime_error("error in CubicSpline::Fit : value nan occurred due to wrong fitgrid boundaries");
00121         }
00122     }
00123 
00124     _f = ub::vector_range<ub::vector<double> >(sol, ub::range (0, ngrid));
00125     _f2 = ub::vector_range<ub::vector<double> >(sol, ub::range (ngrid, 2*ngrid));
00126 }
