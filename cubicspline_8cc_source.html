30 void CubicSpline::Interpolate(ub::vector<double> &x, ub::vector<double> &y)
31 {
32  if(x.size() != y.size())
33  throw std::invalid_argument("error in CubicSpline::Interpolate : sizes of vectors x and y do not match");
34 
35  if(x.size()<3)
36  throw std::invalid_argument("error in CubicSpline::Interpolate : vectors x and y have to contain at least 3 points");
37 
38  const int N = x.size();
39 
40  // adjust the grid
41  _r.resize(N);
42  _f.resize(N);
43  _f2.resize(N);
44 
45  // create vector proxies to individually access f and f''
46 
47  // copy the grid points into f
48  _r = x;
49  _f = y;
50  _f2 = ub::zero_vector<double>(N);
51 
52  // not calculate the f''
53  ub::matrix<double> A(N, N);
54  A = ub::zero_matrix<double>(N,N);
55 
56  for(int i=0; i<N - 2; ++i) {
57  _f2(i+1) = -( A_prime_l(i)*_f(i)
58  + (B_prime_l(i) - A_prime_r(i)) * _f(i+1)
59  -B_prime_r(i) * _f(i+2));
60 
61  A(i+1, i) = C_prime_l(i);
62  A(i+1, i+1) = D_prime_l(i) - C_prime_r(i);
63  A(i+1, i+2) = -D_prime_r(i);
64  }
65 
66  switch(_boundaries) {
67  case splineNormal:
68  A(0, 0) = 1;
69  A(N - 1, N-1) = 1;
70  break;
71  case splinePeriodic:
72  A(0,0) = 1; A(0,N-1) = -1;
73  A(N-1,0) = 1; A(N-1,N-1) = -1;
74  break;
75  case splineDerivativeZero:
76  throw std::runtime_error("erro in CubicSpline::Interpolate: case splineDerivativeZero not implemented yet");
77  break;
78  }
79 
80  votca::tools::linalg_qrsolve(_f2, A, _f2);
81 }

83 void CubicSpline::Fit(ub::vector<double> &x, ub::vector<double> &y)
84 {
85  if(x.size() != y.size())
86  throw std::invalid_argument("error in CubicSpline::Fit : sizes of vectors x and y do not match");
87 
88  const int N = x.size();
89  const int ngrid = _r.size();
90 
91  // construct the equation
92  // A*u = b
93  // where u = { {f[i]}, {f''[i]} }
94  // and b[i] = y[i] for 0<=i<N
95  // and b[i]=0 for i>=N (for smoothing condition)
96  // A[i,j] contains the data fitting + the spline smoothing conditions
97 
98  ub::matrix<double> A(N, 2*ngrid);
99  ub::vector<double> b(N);
100  ub::matrix<double> B_constr(ngrid, 2*ngrid); // Matrix with smoothing conditions
101 
102  A = ub::zero_matrix<double>(N, 2*ngrid);
103  b = ub::zero_vector<double>(N);
104  B_constr = ub::zero_matrix<double>(ngrid, 2*ngrid);
105 
106  // Construct smoothing matrix
107  AddBCToFitMatrix(B_constr, 0);
108 
109  // construct the matrix to fit the points and the vector b
110  AddToFitMatrix(A, x, 0);
111  b = -y; // why is it -y?
112 
113  // now do a constrained qr solve
114  ub::vector<double> sol(2*ngrid);
115  votca::tools::linalg_constrained_qrsolve(sol, A, b, B_constr);
116 
117  // check vector "sol" for nan's
118  for(int i=0; i<2*ngrid; i++) {
119  if( (std::isinf(sol(i))) || (std::isnan(sol(i))) ) {
120  throw std::runtime_error("error in CubicSpline::Fit : value nan occurred due to wrong fitgrid boundaries");
121  }
122  }
123 
124  _f = ub::vector_range<ub::vector<double> >(sol, ub::range (0, ngrid));
125  _f2 = ub::vector_range<ub::vector<double> >(sol, ub::range (ngrid, 2*ngrid));
126 }
