28 void CubicSpline::Interpolate(Eigen::VectorXd &x, Eigen::VectorXd &y) {
29  if (x.size() != y.size())
30  throw std::invalid_argument(
31  "error in CubicSpline::Interpolate : sizes of vectors x and y do not "
32  "match");
33 
34  if (x.size() < 3)
35  throw std::invalid_argument(
36  "error in CubicSpline::Interpolate : vectors x and y have to contain "
37  "at least 3 points");
38 
39  const int N = x.size();
40 
41  // copy the grid points into f
42  _r = x;
43  _f = y;
44  _f2 = Eigen::VectorXd::Zero(N);
45 
46  // not calculate the f''
47  Eigen::MatrixXd A = Eigen::MatrixXd::Zero(N, N);
48 
49  for (int i = 0; i < N - 2; ++i) {
50  _f2(i + 1) =
51  -(A_prime_l(i) * _f(i) + (B_prime_l(i) - A_prime_r(i)) * _f(i + 1) -
52  B_prime_r(i) * _f(i + 2));
53 
54  A(i + 1, i) = C_prime_l(i);
55  A(i + 1, i + 1) = D_prime_l(i) - C_prime_r(i);
56  A(i + 1, i + 2) = -D_prime_r(i);
57  }
58 
59  switch (_boundaries) {
60  case splineNormal:
61  A(0, 0) = 1;
62  A(N - 1, N - 1) = 1;
63  break;
64  case splinePeriodic:
65  A(0, 0) = 1;
66  A(0, N - 1) = -1;
67  A(N - 1, 0) = 1;
68  A(N - 1, N - 1) = -1;
69  break;
70  case splineDerivativeZero:
71  throw std::runtime_error(
72  "erro in CubicSpline::Interpolate: case splineDerivativeZero not "
73  "implemented yet");
74  break;
75  }
76 
77  Eigen::HouseholderQR<Eigen::MatrixXd> QR(A);
78  _f2 = QR.solve(_f2);
79 }

81 void CubicSpline::Fit(Eigen::VectorXd &x, Eigen::VectorXd &y) {
82  if (x.size() != y.size())
83  throw std::invalid_argument(
84  "error in CubicSpline::Fit : sizes of vectors x and y do not match");
85 
86  const int N = x.size();
87  const int ngrid = _r.size();
88 
89  // construct the equation
90  // A*u = b
91  // where u = { {f[i]}, {f''[i]} }
92  // and b[i] = y[i] for 0<=i<N
93  // and b[i]=0 for i>=N (for smoothing condition)
94  // A[i,j] contains the data fitting + the spline smoothing conditions
95 
96  Eigen::MatrixXd A = Eigen::MatrixXd::Zero(N, 2 * ngrid);
97  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(
98  ngrid, 2 * ngrid); // Matrix with smoothing conditions
99 
100  // Construct smoothing matrix
101  AddBCToFitMatrix(B, 0);
102  // construct the matrix to fit the points and the vector b
103  AddToFitMatrix(A, x, 0);
104  // now do a constrained qr solve
105  Eigen::VectorXd sol = Eigen::VectorXd::Zero(2 * ngrid);
106  linalg_constrained_qrsolve(sol, A, y, B);
107 
108  // check vector "sol" for nan's
109  for (int i = 0; i < 2 * ngrid; i++) {
110  if ((std::isinf(sol(i))) || (std::isnan(sol(i)))) {
111  throw std::runtime_error(
112  "error in CubicSpline::Fit : value nan occurred due to wrong fitgrid "
113  "boundaries");
114  }
115  }
116 
117  _f = sol.segment(0, ngrid);
118  _f2 = sol.segment(ngrid, ngrid);
119 }
