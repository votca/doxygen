28 void CubicSpline::Interpolate(const Eigen::VectorXd &x,
29  const Eigen::VectorXd &y) {
30  if (x.size() != y.size()) {
31  throw std::invalid_argument(
32  "error in CubicSpline::Interpolate : sizes of vectors x and y do not "
33  "match");
34  }
35 
36  if (x.size() < 3) {
37  throw std::invalid_argument(
38  "error in CubicSpline::Interpolate : vectors x and y have to contain "
39  "at least 3 points");
40  }
41 
42  const Index N = x.size();
43 
44  // copy the grid points into f
45  _r = x;
46  _f = y;
47  Eigen::VectorXd temp = Eigen::VectorXd::Zero(N);
48 
49  Eigen::MatrixXd A = Eigen::MatrixXd::Zero(N, N);
50 
51  for (Index i = 0; i < N - 2; ++i) {
52  temp(i + 1) =
53  -(A_prime_l(i) * _f(i) + (B_prime_l(i) - A_prime_r(i)) * _f(i + 1) -
54  B_prime_r(i) * _f(i + 2));
55 
56  A(i + 1, i) = C_prime_l(i);
57  A(i + 1, i + 1) = D_prime_l(i) - C_prime_r(i);
58  A(i + 1, i + 2) = -D_prime_r(i);
59  }
60 
61  switch (_boundaries) {
62  case splineNormal:
63  A(0, 0) = 1;
64  A(N - 1, N - 1) = 1;
65  break;
66  case splinePeriodic:
67  A(0, 0) = 1;
68  A(0, N - 1) = -1;
69  A(N - 1, 0) = 1;
70  A(N - 1, N - 1) = -1;
71  break;
72  case splineDerivativeZero:
73  throw std::runtime_error(
74  "erro in CubicSpline::Interpolate: case splineDerivativeZero not "
75  "implemented yet");
76  break;
77  }
78 
79  Eigen::HouseholderQR<Eigen::MatrixXd> QR(A);
80  _f2 = QR.solve(temp);
81 }

83 void CubicSpline::Fit(const Eigen::VectorXd &x, const Eigen::VectorXd &y) {
84  if (x.size() != y.size()) {
85  throw std::invalid_argument(
86  "error in CubicSpline::Fit : sizes of vectors x and y do not match");
87  }
88 
89  const Index N = x.size();
90  const Index ngrid = _r.size();
91 
92  // construct the equation
93  // A*u = b
94  // where u = { {f[i]}, {f''[i]} }
95  // and b[i] = y[i] for 0<=i<N
96  // and b[i]=0 for i>=N (for smoothing condition)
97  // A[i,j] contains the data fitting + the spline smoothing conditions
98 
99  Eigen::MatrixXd A = Eigen::MatrixXd::Zero(N, 2 * ngrid);
100  Eigen::MatrixXd B = Eigen::MatrixXd::Zero(
101  ngrid, 2 * ngrid); // Matrix with smoothing conditions
102 
103  // Construct smoothing matrix
104  AddBCToFitMatrix(B, 0);
105  // construct the matrix to fit the points and the vector b
106  AddToFitMatrix(A, x, 0);
107  // now do a constrained qr solve
108  Eigen::VectorXd sol = linalg_constrained_qrsolve(A, y, B);
109 
110  // check vector "sol" for nan's
111  for (Index i = 0; i < 2 * ngrid; i++) {
112  if ((std::isinf(sol(i))) || (std::isnan(sol(i)))) {
113  throw std::runtime_error(
114  "error in CubicSpline::Fit : value nan occurred due to wrong fitgrid "
115  "boundaries");
116  }
117  }
118 
119  _f = sol.segment(0, ngrid);
120  _f2 = sol.segment(ngrid, ngrid);
121 }

153 double CubicSpline::C(double r) {
154 
155  double xxi = r - _r[getInterval(r)];
156  double h = _r[getInterval(r) + 1] - _r[getInterval(r)];
157 
158  return (0.5 * xxi * xxi - (1.0 / 6.0) * xxi * xxi * xxi / h -
159  (1.0 / 3.0) * xxi * h);
