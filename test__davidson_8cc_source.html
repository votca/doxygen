34Eigen::MatrixXd symm_matrix(Index N, double eps) {
35 Eigen::MatrixXd matrix;
36 matrix = eps * Eigen::MatrixXd::Random(N, N);
37 Eigen::MatrixXd tmat = matrix.transpose();
38 matrix = matrix + tmat;
39 return matrix;
40}
41
42Eigen::MatrixXd init_matrix(Index N, double eps) {
43 Eigen::MatrixXd matrix = Eigen::MatrixXd::Zero(N, N);
44 for (Index i = 0; i < N; i++) {
45 for (Index j = i; j < N; j++) {
46 if (i == j) {
47 matrix(i, i) = std::sqrt(static_cast<double>(1 + i));
48 } else {
49 matrix(i, j) = eps / std::pow(static_cast<double>(j - i), 2);
50 matrix(j, i) = eps / std::pow(static_cast<double>(j - i), 2);
51 }
52 }
53 }
54 return matrix;
55}

131 Eigen::MatrixXd matmul(const Eigen::MatrixXd &input) const {
132 Eigen::MatrixXd result = Eigen::MatrixXd::Zero(input.rows(), input.cols());
133 for (Index i = 0; i < this->size(); i++) {
134 result.row(i) = OperatorRow(i) * input;
135 }
136 return result;
137 }
138
139 Eigen::VectorXd diagonal() const {
140 Index size = this->size();
141 Eigen::VectorXd diag = Eigen::VectorXd::Zero(size);
142 for (Index i = 0; i < size; i++) {
143 Eigen::RowVectorXd row = OperatorRow(i);
144 diag(i) = row(i);
145 }
146 return diag;
147 }
148
149 private:
150 Eigen::RowVectorXd OperatorRow(Index index) const;
151};

153// get a col of the operator
154Eigen::RowVectorXd TestOperator::OperatorRow(Index index) const {
155 Index lsize = this->size();
156 Eigen::RowVectorXd row_out = Eigen::RowVectorXd::Zero(lsize);
157 for (Index j = 0; j < lsize; j++) {
158 if (j == index) {
159 row_out(j) = std::sqrt(static_cast<double>(index + 1));
160 } else {
161 row_out(j) = 0.01 / std::pow(static_cast<double>(j - index), 2);
162 }
163 }

231Eigen::ArrayXi index_eval(Eigen::VectorXd ev, Index neigen) {
232
233 Index nev = ev.rows();
234 Index npos = nev / 2;
235
236 Eigen::ArrayXi idx = Eigen::ArrayXi::Zero(npos);
237 Index nstored = 0;
238
239 // get only positives
240 for (Index i = 0; i < nev; i++) {
241 if (ev(i) > 0) {
242 idx(nstored) = int(i);
243 nstored++;
244 }
245 }
246
247 // sort the epos eigenvalues
248 std::sort(idx.data(), idx.data() + idx.size(),
249 [&](Index i1, Index i2) { return ev[i1] < ev[i2]; });
250 return idx.head(neigen);
251}

253Eigen::MatrixXd extract_eigenvectors(const Eigen::MatrixXd &V,
254 const Eigen::ArrayXi &idx) {
255 Eigen::MatrixXd W = Eigen::MatrixXd::Zero(V.rows(), idx.size());
256 for (Index i = 0; i < idx.size(); i++) {
257 W.col(i) = V.col(idx(i));
258 }
259 return W;
260}

281BOOST_AUTO_TEST_CASE(davidson_hamiltonian_matrix_free) {
282
283 Index size = 60;
284 Index neigen = 5;
285 Logger log;
286
287 // Create Operator
288 HermitianBlockOperator Rop;
289 Rop.set_size(size);
290 Eigen::MatrixXd rmat = init_matrix(size, 0.01);
291 Rop.attach_matrix(rmat);
292
293 HermitianBlockOperator Cop;
294 Cop.set_size(size);
295 Eigen::MatrixXd cmat = symm_matrix(size, 0.01);
296 Cop.attach_matrix(cmat);
297
298 // create Hamiltonian operator
299 HamiltonianOperator<HermitianBlockOperator, HermitianBlockOperator> Hop(Rop,
300 Cop);
301
302 DavidsonSolver DS(log);
303 DS.set_tolerance("normal");
304 DS.set_size_update("max");
305 DS.set_matrix_type("HAM");
306 DS.solve(Hop, neigen);
307 auto lambda = DS.eigenvalues().real();
308 std::sort(lambda.data(), lambda.data() + lambda.size());
309 Eigen::MatrixXd identity = Eigen::MatrixXd::Identity(Hop.rows(), Hop.cols());
310 Eigen::MatrixXd H = Hop * identity;
311
312 Eigen::EigenSolver<Eigen::MatrixXd> es(H);
313 Eigen::ArrayXi idx = index_eval(es.eigenvalues().real(), neigen);
314 Eigen::VectorXd lambda_ref = idx.unaryExpr(es.eigenvalues().real());
315
316 bool check_eigenvalues = lambda.isApprox(lambda_ref.head(neigen), 1E-6);
317 if (!check_eigenvalues) {
318 std::cout << "Davidson not converged after " << DS.num_iterations()
319 << " iterations" << std::endl;
320 std::cout << "Reference eigenvalues" << std::endl;
321 std::cout << lambda_ref.head(neigen) << std::endl;
322 std::cout << "Davidson eigenvalues" << std::endl;
323 std::cout << lambda << std::endl;
324 }
325 BOOST_CHECK_EQUAL(check_eigenvalues, 1);
326
327 Eigen::MatrixXd evect_dav = DS.eigenvectors().real();
328 Eigen::MatrixXd evect = es.eigenvectors().real();
329 Eigen::MatrixXd evect_ref = extract_eigenvectors(evect, idx);
330
331 bool check_eigenvectors =
332 evect_ref.cwiseAbs2().isApprox(evect_dav.cwiseAbs2(), 0.001);
333 BOOST_CHECK_EQUAL(check_eigenvectors, 1);
334}
