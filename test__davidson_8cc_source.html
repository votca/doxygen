136 Eigen::ArrayXi index_eval(Eigen::VectorXd ev, int neigen) {
137 
138  int nev = ev.rows();
139  int npos = nev / 2;
140 
141  Eigen::ArrayXi idx = Eigen::ArrayXi::Zero(npos);
142  int nstored = 0;
143 
144  // get only positives
145  for (int i = 0; i < nev; i++) {
146  if (ev(i) > 0) {
147  idx(nstored) = i;
148  nstored++;
149  }
150  }
151 
152  // sort the epos eigenvalues
153  std::sort(idx.data(), idx.data() + idx.size(),
154  [&](int i1, int i2) { return ev[i1] < ev[i2]; });
