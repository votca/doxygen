164 void Numoutput::Convergence_check(double simtime, Eventinfo* eventinfo) {
165 
166  if (fabs(_vel_z/simtime-_vz_old)/_vz_old < 0.08 && _vel_z > 0.0) { _direct_iv_counter++; } else { _direct_iv_counter = 0; }
167  if (fabs(_nrecombinations/simtime-_reco_old)/_reco_old < 0.08) { _direct_reco_counter++;} else { _direct_reco_counter = 0;}
168 
169  if(_direct_iv_counter >= eventinfo->number_direct_conv_iv) {_direct_iv_convergence = true;}
170  if(_direct_reco_counter >= eventinfo->number_direct_conv_reco) {_direct_reco_convergence = true;}
171 
172  _vz_old = _vel_z/simtime;
173  _reco_old = _nrecombinations/simtime;
174 }

176 void Numoutput::Update(Event* event, double simtime, double timestep) {
177 
178  if(event->init_type() == (int) Injection) { // Injection events
179 
180  _ninjections++;
181  if(event->link()->node1()->type() == LeftElectrodeNode) { _nleftinjections++; } else { _nrightinjections++; }
182 
183  if(event->final_type() == (int) TransferTo) {
184  _ncarriers++;
185  if(event->carrier_type() == (int) Electron) { _nelectrons++; } else { _nholes++;}
186  }
187  else if(event->final_type() == (int) Recombination) {
188  _nrecombinations++;
189  _ninject_to_recombination++;
190  _ncarriers--;
191  if(event->carrier_type() == (int) Electron) {
192  // hole gets recombined away
193  _nholes--;
194  }
195  else {
196  // electron gets recombined away
197  _nelectrons--;
198  }
199  }
200  else if(event->final_type() == (int) Collection) {
201  _ncollections++;
202  if(event->link()->node2()->type() == LeftElectrodeNode) { _nleftcollections++;} else { _nrightcollections++;}
203  }
204  }
205  else if(event->init_type() == (int) TransferFrom) { // Normal transfer
206 
207  if(event->final_type() == (int) Collection) {
208  _ncollections++;
209  if(event->link()->node2()->type() == LeftElectrodeNode) { _nleftcollections++;} else { _nrightcollections++;}
210 
211  _ncarriers--;
212  if(event->carrier_type() == (int) Electron) { _nelectrons--; } else { _nholes--;}
213  }
214  if(event->final_type() == (int) Recombination) { //injection to recombination (sink of charges here)
215  _nrecombinations++;
216  _nelectrons--; _ncarriers--;
217  _nholes--; _ncarriers--;
218  }
219  else if(event->final_type() == (int) TransferTo) {
220  _nplaintransfer++;
221  }
222  }
223 
224  _ninjectionrate = _ninjections/simtime;
225  _ncollectionrate = _ncollections/simtime;
226  _nrecombinationrate = _nrecombinations/simtime;
227 
228  Node* node1 = event->link()->node1();
229  Node* node2 = event->link()->node2();
230  //votca::tools::vec nodepos1 = node1->position();
231  //votca::tools::vec nodepos2 = node2->position();
232  int node1_layer = dynamic_cast<NodeDevice*>(node1)->layer();
233  int node2_layer = dynamic_cast<NodeDevice*>(node2)->layer();
234 
235  votca::tools::vec travelvec = event->link()->r12();
236  double direction;
237  double dirx; double diry; double dirz;
238 
239  if(event->carrier_type() == (int) Electron) { direction = -1.0; } else { direction = 1.0;}
240  if(travelvec.x() > 0) {dirx = 1.0;} else {if(travelvec.x() == 0) {dirx = 0.0;} else {dirx = -1.0;}}
241  if(travelvec.y() > 0) {diry = 1.0;} else {if(travelvec.y() == 0) {diry = 0.0;} else {diry = -1.0;}}
242  if(travelvec.z() > 0) {dirz = 1.0;} else {if(travelvec.z() == 0) {dirz = 0.0;} else {dirz = -1.0;}}
243 
244  if(node1->type() == (int) NormalNode) {
245  if(node1_layer != node2_layer) layercurrent[node1_layer] += 0.5*dirz;
246  _intvel_x += 0.5*dirx;
247  _intvel_y += 0.5*diry;
248  _intvel_z += 0.5*dirz;
249  }
250 
251  if(node2->type() == (int) NormalNode) {
252  if(node1_layer != node2_layer) layercurrent[node2_layer] += 0.5*dirz;
253  _intvel_x += 0.5*dirx;
254  _intvel_y += 0.5*diry;
255  _intvel_z += 0.5*dirz;
256  }
257 
258  _vel_x += direction*travelvec.x();
259  _vel_y += direction*travelvec.y();
260  _vel_z += direction*travelvec.z();
261 
262  if(event->carrier_type() == (int) Electron) {
263  _electron_vel_x += direction*travelvec.x();
264  _electron_vel_y += direction*travelvec.y();
265  _electron_vel_z += direction*travelvec.z();
266  }
267 
268  if(event->carrier_type() == (int) Hole) {
269  _hole_vel_x += direction*travelvec.x();
270  _hole_vel_y += direction*travelvec.y();
271  _hole_vel_z += direction*travelvec.z();
272  }
273 
274 }
