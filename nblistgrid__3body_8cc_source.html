119 void NBListGrid_3Body::InitializeGrid(const matrix &box)
120 {
121  _box_a = box.getCol(0);
122  _box_b = box.getCol(1);
123  _box_c = box.getCol(2);
124 
125  // create plane normals
126  _norm_a = _box_b ^ _box_c;
127  _norm_b = _box_c ^ _box_a;
128  _norm_c = _box_a ^ _box_b;
129 
130  _norm_a.normalize();
131  _norm_b.normalize();
132  _norm_c.normalize();
133 
134  double la = _box_a * _norm_a;
135  double lb = _box_b * _norm_b;
136  double lc = _box_c * _norm_c;
137 
138  // calculate grid size, each grid has to be at least size of cut-off
139  _box_Na = max((int)(fabs(la/_cutoff)), 1);
140  _box_Nb = max((int)(fabs(lb/_cutoff)), 1);
141  _box_Nc = max((int)(fabs(lc/_cutoff)), 1);
142 
143  _norm_a = _norm_a / (_box_a*_norm_a)*(double)_box_Na;
144  _norm_b = _norm_b / (_box_b*_norm_b)*(double)_box_Nb;
145  _norm_c = _norm_c / (_box_c*_norm_c)*(double)_box_Nc;
146 
147  //cout << "grid size: " << _box_Na << "x" << _box_Nb << "x" << _box_Nc << endl;
148  _grid.resize(_box_Na*_box_Nb*_box_Nc);
149 
150  int a1,a2,b1,b2,c1,c2;
151 
152  a1 = b1 = c1 = -1;
153  a2 = b2 = c2 = 1;
154 
155  if(_box_Na < 3) a2 = 0;
156  if(_box_Nb < 3) b2 = 0;
157  if(_box_Nc < 3) c2 = 0;
158 
159  if(_box_Na < 2) a1 = 0;
160  if(_box_Nb < 2) b1 = 0;
161  if(_box_Nc < 2) c1 = 0;
162 
163  // wow, setting up the neighbours is an ugly for construct!
164  // loop from N..2*N to avoid if and only use %
165  for(int a=_box_Na; a<2*_box_Na; ++a)
166  for(int b=_box_Nb; b<2*_box_Nb; ++b)
167  for(int c=_box_Nc; c<2*_box_Nc; ++c) {
168  cell_t &cell = getCell(a%_box_Na, b%_box_Nb, c%_box_Nc);
169  for(int aa=a+a1; aa<=a+a2; ++aa)
170  for(int bb=b+b1; bb<=b+b2; ++bb)
171  for(int cc=c+c1; cc<=c+c2; ++cc) {
172  //test: for 3body algorithm: each cell is a neighbor of its own !!!
173  cell._neighbours.push_back(
174  &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc)
175  );
176  }
177  }
178 }

180 NBListGrid_3Body::cell_t &NBListGrid_3Body::getCell(const vec &r)
181 {
182  int a = (int)floor(r*_norm_a);
183  int b = (int)floor(r*_norm_b);
184  int c = (int)floor(r*_norm_c);
185 
186  if(a<0) a = _box_Na + a%_box_Na;
187  a %= _box_Na;
188 
189  if(b<0)
190  b = _box_Nb + b%_box_Nb;
191  b %= _box_Nb;
192 
193  if(c<0) c = _box_Nc + c%_box_Nc;
194  c %= _box_Nc;
195 
196  return getCell(a,b,c);
197  }
