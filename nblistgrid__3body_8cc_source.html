117 void NBListGrid_3Body::InitializeGrid(const matrix &box)
118 {
119  _box_a = box.getCol(0);
120  _box_b = box.getCol(1);
121  _box_c = box.getCol(2);
122 
123  // create plane normals
124  _norm_a = _box_b ^ _box_c;
125  _norm_b = _box_c ^ _box_a;
126  _norm_c = _box_a ^ _box_b;
127 
128  _norm_a.normalize();
129  _norm_b.normalize();
130  _norm_c.normalize();
131 
132  double la = _box_a * _norm_a;
133  double lb = _box_b * _norm_b;
134  double lc = _box_c * _norm_c;
135 
136  // calculate grid size, each grid has to be at least size of cut-off
137  _box_Na = max((int)(fabs(la/_cutoff)), 1);
138  _box_Nb = max((int)(fabs(lb/_cutoff)), 1);
139  _box_Nc = max((int)(fabs(lc/_cutoff)), 1);
140 
141  _norm_a = _norm_a / (_box_a*_norm_a)*(double)_box_Na;
142  _norm_b = _norm_b / (_box_b*_norm_b)*(double)_box_Nb;
143  _norm_c = _norm_c / (_box_c*_norm_c)*(double)_box_Nc;
144 
145  //cout << "grid size: " << _box_Na << "x" << _box_Nb << "x" << _box_Nc << endl;
146  _grid.resize(_box_Na*_box_Nb*_box_Nc);
147 
148  int a1,a2,b1,b2,c1,c2;
149 
150  a1 = b1 = c1 = -1;
151  a2 = b2 = c2 = 1;
152 
153  if(_box_Na < 3) a2 = 0;
154  if(_box_Nb < 3) b2 = 0;
155  if(_box_Nc < 3) c2 = 0;
156 
157  if(_box_Na < 2) a1 = 0;
158  if(_box_Nb < 2) b1 = 0;
159  if(_box_Nc < 2) c1 = 0;
160 
161  // wow, setting up the neighbours is an ugly for construct!
162  // loop from N..2*N to avoid if and only use %
163  for(int a=_box_Na; a<2*_box_Na; ++a)
164  for(int b=_box_Nb; b<2*_box_Nb; ++b)
165  for(int c=_box_Nc; c<2*_box_Nc; ++c) {
166  cell_t &cell = getCell(a%_box_Na, b%_box_Nb, c%_box_Nc);
167  for(int aa=a+a1; aa<=a+a2; ++aa)
168  for(int bb=b+b1; bb<=b+b2; ++bb)
169  for(int cc=c+c1; cc<=c+c2; ++cc) {
170  //test: for 3body algorithm: each cell is a neighbor of its own !!!
171  cell._neighbours.push_back(
172  &getCell(aa%_box_Na, bb%_box_Nb, cc%_box_Nc)
173  );
174  }
175  }
176 }

178 NBListGrid_3Body::cell_t &NBListGrid_3Body::getCell(const vec &r)
179 {
180  int a = (int)floor(r*_norm_a);
181  int b = (int)floor(r*_norm_b);
182  int c = (int)floor(r*_norm_c);
183 
184  if(a<0) a = _box_Na + a%_box_Na;
185  a %= _box_Na;
186 
187  if(b<0)
188  b = _box_Nb + b%_box_Nb;
189  b %= _box_Nb;
190 
191  if(c<0) c = _box_Nc + c%_box_Nc;
192  c %= _box_Nc;
193 
194  return getCell(a,b,c);
195  }

207 
208

210  continue;
211  }
212 
213 
214  //loop again over all neighbors (this now includes the cell itself!) to iterate over all beads of type3 of the cell and its neighbors

216  for(iter3=(*(*iterc3))._beads3.begin(); iter3!=(*(*iterc3))._beads3.end(); ++iter3) {
