290 vector<Node*> KMCMultiple::LoadGraph()
291 {
292  vector<Node*> node;
293 
294  // Load nodes
295  votca::tools::Database db;
296  db.Open( _filename );
297  if(votca::tools::globals::verbose) {cout << "LOADING GRAPH" << endl << "database file: " << _filename << endl; }
298  votca::tools::Statement *stmt = db.Prepare("SELECT _id-1, name, posX, posY, posZ, UnCnN"+_carriertype+", UcNcC"+_carriertype+",eAnion,eNeutral,eCation,ucCnN"+_carriertype+" FROM segments;");
299 
300  int i=0;
301  while (stmt->Step() != SQLITE_DONE)
302  {
303  Node *newNode = new Node();
304  node.push_back(newNode);
305 
306  int newid = stmt->Column<int>(0);
307  string name = stmt->Column<string>(1);
308  node[i]->id = newid;
309  myvec nodeposition = myvec(stmt->Column<double>(2)*1E-9, stmt->Column<double>(3)*1E-9, stmt->Column<double>(4)*1E-9); // converted from nm to m
310  node[i]->position = nodeposition;
311  node[i]->reorg_intorig = stmt->Column<double>(5); // UnCnN
312  node[i]->reorg_intdest = stmt->Column<double>(6); // UcNcC
313  double eAnion = stmt->Column<double>(7);
314  //double eNeutral = stmt->Column<double>(8);
315  double eCation = stmt->Column<double>(9);
316  double internalenergy = stmt->Column<double>(10); // UcCnN
317  double siteenergy = 0;
318  if(_carriertype == "e")
319  {
320  siteenergy = eAnion + internalenergy;
321  }
322  else if(_carriertype == "h")
323  {
324  siteenergy = eCation + internalenergy;
325  }
326  node[i]->siteenergy = siteenergy;
327  if (votca::tools::wildcmp(_injection_name.c_str(), name.c_str()))
328  {
329  node[i]->injectable = 1;
330  }
331  else
332  {
333  node[i]->injectable = 0;
334  }
335  i++;
336  }
337  delete stmt;
338  if(votca::tools::globals::verbose) { cout << "segments: " << node.size() << endl; }
339 
340  // Load pairs and rates
341  int numberofpairs = 0;
342  stmt = db.Prepare("SELECT seg1-1 AS 'segment1', seg2-1 AS 'segment2', rate12"+_carriertype+" AS 'rate', drX, drY, drZ, Jeff2"+_carriertype+", lO"+_carriertype+" FROM pairs UNION SELECT seg2-1 AS 'segment1', seg1-1 AS 'segment2', rate21"+_carriertype+" AS 'rate', -drX AS 'drX', -drY AS 'drY', -drZ AS 'drZ', Jeff2"+_carriertype+", lO"+_carriertype+" FROM pairs ORDER BY segment1;");
343  while (stmt->Step() != SQLITE_DONE)
344  {
345  int seg1 = stmt->Column<int>(0);
346  int seg2 = stmt->Column<int>(1);
347 
348  double rate12 = stmt->Column<double>(2);
349 
350  myvec dr = myvec(stmt->Column<double>(3)*1E-9, stmt->Column<double>(4)*1E-9, stmt->Column<double>(5)*1E-9); // converted from nm to m
351  double Jeff2 = stmt->Column<double>(6);
352  double reorg_out = stmt->Column<double>(7);
353  node[seg1]->AddEvent(seg2,rate12,dr,Jeff2,reorg_out);
354  numberofpairs ++;
355  }
356  delete stmt;
357 
358  if(votca::tools::globals::verbose) { cout << "pairs: " << numberofpairs/2 << endl; }
359 
360  // Calculate initial escape rates !!!THIS SHOULD BE MOVED SO THAT IT'S NOT DONE TWICE IN CASE OF COULOMB INTERACTION!!!
361  for(unsigned int i=0; i<node.size(); i++)
362  {
363  node[i]->InitEscapeRate();
364  }
365  return node;
366 }

499 void KMCMultiple::InitialRates(vector<Node*> node)
500 {
501  cout << endl <<"Calculating initial Marcus rates." << endl;
502  cout << " Temperature T = " << _temperature << " K." << endl;
503  cout << " Field: (" << _fieldX << ", " << _fieldY << ", " << _fieldZ << ") V/m" << endl;
504  if (_carriertype == "e")
505  {
506  _q = -1.0;
507  }
508  else if (_carriertype == "h")
509  {
510  _q = 1.0;
511  }
512  cout << " carriertype: " << _carriertype << " (charge: " << _q << " eV)" << endl;
513  unsigned int numberofsites = node.size();
514  cout << " Rates for "<< numberofsites << " sites are computed." << endl;
515  double maxreldiff = 0;
516  int totalnumberofrates = 0;
517  for(unsigned int i=0; i<numberofsites; i++)
518  {
519  unsigned int numberofneighbours = node[i]->event.size();
520  for(unsigned int j=0; j<numberofneighbours; j++)
521  {
522  double dX = node[i]->event[j].dr.x();
523  double dY = node[i]->event[j].dr.y();
524  double dZ = node[i]->event[j].dr.z();
525  double dG_Field = _q * (dX*_fieldX + dY*_fieldY + dZ*_fieldZ);
526 
527  double destindex = node[i]->event[j].destination;
528  double reorg = node[i]->reorg_intorig + node[destindex]->reorg_intdest + node[i]->event[j].reorg_out;
529 
530  double dG_Site = node[destindex]->siteenergy - node[i]->siteenergy;
531 
532  double J2 = node[i]->event[j].Jeff2;
533 
534  double dG = dG_Site - dG_Field;
535 
536  double rate = 2*Pi/hbar * J2/sqrt(4*Pi*reorg*kB*_temperature) * exp(-(dG+reorg)*(dG+reorg) / (4*reorg*kB*_temperature));
537 
538  // calculate relative difference compared to values in the table
539  double reldiff = (node[i]->event[j].rate - rate) / node[i]->event[j].rate;
540  if (reldiff > maxreldiff) {maxreldiff = reldiff;}
541  reldiff = (node[i]->event[j].rate - rate) / rate;
542  if (reldiff > maxreldiff) {maxreldiff = reldiff;}
543 
544  // set rates to calculated values
545  node[i]->event[j].rate = rate;
546  node[i]->event[j].initialrate = rate;
547 
548  totalnumberofrates ++;
549 
550  }
551 
552  // Initialise escape rates
553  for(unsigned int i=0; i<node.size(); i++)
554  {
555  node[i]->InitEscapeRate();
556  }
557 
558  }
559  cout << " " << totalnumberofrates << " rates have been calculated." << endl;
560  if(maxreldiff < 0.01)
561  {
562  cout << " Good agreement with rates in the state file. Maximal relative difference: " << maxreldiff*100 << " %"<< endl;
563  }
564  else
565  {
566  cout << " WARNING: Rates differ from those in the state file up to " << maxreldiff*100 << " %." << " If the rates in the state file are calculated for a different temperature/field or if they are not Marcus rates, this is fine. Otherwise something might be wrong here." << endl;
567  }
568 }

570 void KMCMultiple::RateUpdateCoulomb(vector<Node*> &node, vector< Chargecarrier* > &carrier, CoulombMap &coulomb)
571 {
572  // Calculate new rates for all possible events, i.e. for the possible hoppings of all occupied nodes
573  //cout << "Updating Coulomb interaction part of rates." << endl;
574  //#pragma omp parallel for
575  for(unsigned int cindex=0; cindex<carrier.size(); cindex++)
576  {
577  //cout << " carrier no. "<< cindex+1 << endl;
578  Node *node_i = carrier[cindex]->node;
579  double escaperate = 0;
580  //#pragma omp parallel for
581  for(unsigned int destindex=0; destindex<node_i->event.size(); destindex++)
582  {
583  int destid = node_i->event[destindex].destination;
584  Node *node_j = node[destid];
585  if(node_j->occupied == 1)
586  { // in principal shouldn't be needed:
587  node_i->event[destindex].rate = 0;
588  // escaperate += node_i->event[destindex].rate;
589  }
590  else
591  {
592  //cout << " event i->j: " << node_i->id+1 << " -> " << node_j->id+1 << endl;
593  // getting the correction factor for the rate i->j (node_i -> node_j)
594  // now calculating the contribution for all neighbouring charges
595  double coulombsum = 0;
596 
597  int dimension = node.size();
598  //#pragma omp parallel for reduction(+:coulombsum)
599  for(unsigned int ncindex=0; ncindex<carrier.size(); ncindex++)
600  {
601  if(_explicitcoulomb == 2) // raw Coulomb
602  {

628  distancevec = carrier[ncindex]->node->position - node_j->position;
629  dX = std::abs(distancevec.x());
