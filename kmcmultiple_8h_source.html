366  votca::tools::Database db;
367  db.Open( _filename );
368  if(votca::tools::globals::verbose) {cout << "LOADING GRAPH" << endl << "database file: " << _filename << endl; }
369  votca::tools::Statement *stmt=NULL;
370  if (_carriertype=="h" || _carriertype=="e"){
371  stmt= db.Prepare("SELECT _id-1, name, posX, posY, posZ, UnCnN"+_carriertype+", UcNcC"+_carriertype+",eAnion,eNeutral,eCation,UcCnN"+_carriertype+" FROM segments;");
372  }

374  stmt= db.Prepare("SELECT _id-1, name, posX, posY, posZ, UnXnN"+_carriertype+", UxNxX"+_carriertype+",eSinglet,eNeutral,eTriplet,UxXnN"+_carriertype+" FROM segments;");
375  }
376 
377  int i=0;
378  while (stmt->Step() != SQLITE_DONE)
379  {
380  GNode *newNode = new GNode();
381  node.push_back(newNode);
382 
383  int newid = stmt->Column<int>(0);
384  string name = stmt->Column<string>(1);
385  node[i]->id = newid;
386  vec nodeposition = vec(stmt->Column<double>(2)*1E-9, stmt->Column<double>(3)*1E-9, stmt->Column<double>(4)*1E-9); // converted from nm to m
387  node[i]->position = nodeposition;
388  node[i]->reorg_intorig = stmt->Column<double>(5); // UnCnN or UnXnN
389  node[i]->reorg_intdest = stmt->Column<double>(6); // UcNcC or UxNxX
390  double eAnion = stmt->Column<double>(7);
391  //double eNeutral = stmt->Column<double>(8);
392  double eCation = stmt->Column<double>(9);
393  double internalenergy = stmt->Column<double>(10); // UcCnN or UxXnN
394  double siteenergy = 0;
395  if(_carriertype == "e" || _carriertype=="s")
396  {
397  siteenergy = eAnion + internalenergy;
398  }
399  else if(_carriertype == "h" || _carriertype=="t")
400  {
401  siteenergy = eCation + internalenergy;
402  }
403 
404  node[i]->siteenergy = siteenergy;
405  if (votca::tools::wildcmp(_injection_name.c_str(), name.c_str()))
406  {
407  node[i]->injectable = 1;
408  }
409  else
410  {
411  node[i]->injectable = 0;
412  }
413  i++;
414  }
415  delete stmt;
416  if(votca::tools::globals::verbose) { cout << "segments: " << node.size() << endl; }
417 
418  // Load pairs and rates

420  stmt = db.Prepare("SELECT seg1-1 AS 'segment1', seg2-1 AS 'segment2', rate12"+_carriertype+" AS 'rate', drX, drY, drZ, Jeff2"+_carriertype+", lO"+_carriertype+" FROM pairs UNION SELECT seg2-1 AS 'segment1', seg1-1 AS 'segment2', rate21"+_carriertype+" AS 'rate', -drX AS 'drX', -drY AS 'drY', -drZ AS 'drZ', Jeff2"+_carriertype+", lO"+_carriertype+" FROM pairs ORDER BY segment1;");

425 
426  double rate12 = stmt->Column<double>(2);

441  node[i]->InitEscapeRate();
442  }
