74 void JobApplication::Run() {
75  std::string name = ProgramName();
76  if (VersionString() != "") name = name + ", version " + VersionString();
77  votca::xtp::HelpTextHeader(name);
78 
79  load_property_from_xml(_options, _op_vm["options"].as<string>());
80 
81  // EVALUATE OPTIONS
82  int nThreads = OptionsMap()["nthreads"].as<int>();
83  int nframes = OptionsMap()["nframes"].as<int>();
84  int fframe = OptionsMap()["first-frame"].as<int>();
85  if (fframe-- == 0) throw runtime_error("ERROR: First frame is 0, counting "
86  "in VOTCA::XTP starts from 1.");
87  int save = OptionsMap()["save"].as<int>();
88 
89  // STATESAVER & PROGRESS OBSERVER
90  string statefile = OptionsMap()["file"].as<string>();
91  StateSaverSQLite statsav;
92  statsav.Open(_top, statefile);
93 
94  ctp::ProgObserver< std::vector<ctp::Job*>, ctp::Job*, ctp::Job::JobResult > progObs
95  = ctp::ProgObserver< std::vector<ctp::Job*>, ctp::Job*, ctp::Job::JobResult >();
96  progObs.InitCmdLineOpts(OptionsMap());
97 
98  // INITIALIZE & RUN CALCULATORS
99  cout << "Initializing calculators " << endl;
100  BeginEvaluate(nThreads, &progObs);
101 
102  int frameId = -1;
103  int framesDone = 0;
104  while (statsav.NextFrame() && framesDone < nframes) {
105  frameId += 1;
106  if (frameId < fframe) continue;
107  cout << "Evaluating frame " << _top.getDatabaseId() << endl;
108  EvaluateFrame();
109  if (save == 1) { statsav.WriteFrame(); }
110  else { cout << "Changes have not been written to state file." << endl; }
111  framesDone += 1;
112  }
113 
114  if (framesDone == 0)
115  cout << "Input requires first frame = " << fframe+1 << ", # frames = "
116  << nframes << " => No frames processed.";
117 
118  statsav.Close();
119  EndEvaluate();
120 
121 }

143 bool JobApplication::EvaluateFrame() {
144  for (ctp::JobCalculator* calculator:_calculators) {
145  cout << "... " << calculator->Identify() << " " << flush;
146  if (_generate_input) calculator->WriteJobFile(&_top);
147  if (_run) calculator->EvaluateFrame(&_top);

149  cout << endl;
