74 void Graph::Load_Graph(std::string filename) {
75 
76  std::cout << "Loading the graph from " << filename << std::endl;
77 
78  // initialising the database file
79  votca::tools::Database db;
80  db.Open( filename );
81 
82  votca::tools::Statement *stmt = db.Prepare("SELECT id-1, posX, posY, posZ, UnCnNe, UnCnNh, UcNcCe, UcNcCh, eAnion, eNeutral, eCation, UcCnNe, UcCnNh FROM segments;");
83 
84  while (stmt->Step() != SQLITE_DONE)
85  {
86  BNode *node = AddNode();
87  int id = stmt->Column<int>(0);
88  //used so node id starts at 1
89  node->id = id+1;
90 
91  // position in nm (converted to m for the final output in state.h)
92  double x = stmt->Column<double>(1);
93  double y = stmt->Column<double>(2);
94  double z = stmt->Column<double>(3);
95 
96  myvec position = myvec(x, y, z);
97  node->position = position;
98 
99  //node has all energetic information read from state file (for rate calculation if required)
100  //Energies for each node - for electron and hole
101  //UnCnNe & UnCnNh
102  node->reorg_intorig_electron = stmt->Column<double>(4);
103  node->reorg_intorig_hole = stmt->Column<double>(5);
104 
105  //UcNcCe & UcNcCh
106  node->reorg_intdest_electron = stmt->Column<double>(6);
107  node->reorg_intdest_hole = stmt->Column<double>(7);
108 
109  node->eAnion = stmt->Column<double>(8);
110  node->eNeutral = stmt->Column<double>(9);
111  node->eCation = stmt->Column<double>(10);
112 
113  //UcCnNe & UcCnNh
114  node->internal_energy_electron = stmt->Column<double>(11);
115  node->internal_energy_hole = stmt->Column<double>(12);
116 
117  node->site_energy_electron = node->eAnion + node->internal_energy_electron;
118  node->site_energy_hole = node->eCation + node->internal_energy_hole;
119 
120  //node->PrintNode();
121  }
122 
123  delete stmt;
124 }

127 void Graph::Load_Rates(std::string filename){
128 
129  votca::tools::Database db;
130  db.Open( filename );
131 
132  //List of neighbours (from neighbour list for charge transfer - state.sql "pairs" table)
133  //List of rates for electron transfer from seg 1 to seg 2 and vice versa
134 
135  votca::tools::Statement *stmt = db.Prepare("SELECT seg1, seg2, drX, drY, drZ, rate12e, rate21e, rate12h, rate21h FROM pairs ;");
136  while (stmt->Step() != SQLITE_DONE)
137  {
138  int seg1 = stmt->Column<int>(0);
139  int seg2 = stmt->Column<int>(1);
140 
141  BNode* node1 = GetNode( seg1 );
142  BNode* node2 = GetNode( seg2 );
143 
144  double dx_pbc = stmt->Column<double>(2);
145  double dy_pbc = stmt->Column<double>(3);
146  double dz_pbc = stmt->Column<double>(4);
147 
148  double rate12e = stmt->Column<double>(5); // 1 -> 2
149  double rate21e = stmt->Column<double>(6); // 2 -> 1
150 
151  double rate12h = stmt->Column<double>(7); // 1 -> 2
152  double rate21h = stmt->Column<double>(8); // 2 -> 1
153 
154  //std::cout << "seg1: " << seg1 << " seg2: " << seg2 << " rate12 " << rate12e << std::endl;
155 
156  votca::tools::vec distance_pbc(dx_pbc, dy_pbc, dz_pbc);
157  Edge* edge12 = new Edge(node1, node2, distance_pbc, rate12e, rate12h);
158  Edge* edge21 = new Edge(node2, node1, -distance_pbc, rate21e, rate21h);
159 
160  edges.push_back( edge12 );
161  edges.push_back( edge21 );
162 
163  // make sure no duplicates are added
164  node1->AddEdge( edge12 );
165  node2->AddEdge( edge21 );
166 
167 
168  }
169  delete stmt;
170 }

172 // Calculates rates if they are not read from the state file
173 void Graph::Rates_Calculation(std::string filename, int nelectrons, int nholes, double fieldX, double fieldY, double fieldZ, double temperature){
174 
175  double kB = 8.617332478E-5; // eV/K
176  double hbar = 6.5821192815E-16; // eV*s
177  double Pi = 3.14159265358979323846;
178 
179  double charge_e = -1.0;
180  double charge_h = 1.0;
181 
182  //Calculation of the initial Marcus rates - allows the rates to be calculated with varying field and temperature
183  std::cout << "Calculating initial Marcus rates." << std::endl;
184  std::cout << "Temperature T = " << temperature << " K." << std::endl;
185  std::cout << "Field: (" << fieldX << ", " << fieldY << ", " << fieldZ << ") V/m" << std::endl;
186 
187  votca::tools::Database db;
188  db.Open( filename );
189 
190  votca::tools::Statement *stmt = db.Prepare("SELECT seg1, seg2, drX, drY, drZ, Jeff2e, Jeff2h, lOe, lOh FROM pairs;");
191 
192  while (stmt->Step() != SQLITE_DONE)
193  {
194  int seg1 = stmt->Column<int>(0);
195  int seg2 = stmt->Column<int>(1);
196 
197  BNode* node1 = GetNode( seg1 );
198  BNode* node2 = GetNode( seg2 );
199 
200  //distance (nm) from node 1 to node 2) - keep in (nm) as all other objects use (nm)
201  double dX = stmt->Column<double>(2);
202  double dY = stmt->Column<double>(3);
203  double dZ = stmt->Column<double>(4);
204 
205  votca::tools::vec distance(dX, dY, dZ);
206 
207  //distance (nm) from node 1 to node 2) converted to (m) for dG field calculation
208  double _dX = dX*1E-9;
209  double _dY = dY*1E-9;
210  double _dZ = dZ*1E-9;
211 
212  //Effect of field on charge carrier (electron and hole)
213  double dG_Field_e = charge_e * (_dX*fieldX + _dY*fieldY + _dZ*fieldZ);
214  double dG_Field_h = charge_h * (_dX*fieldX + _dY*fieldY + _dZ*fieldZ);
215 
216  double Jeff2e = stmt->Column<double>(5);
217  double Jeff2h = stmt->Column<double>(6);
218 
219  double reorg_out_e = stmt->Column<double>(7);
220  double reorg_out_h = stmt->Column<double>(8);
221 
222  //reorganisation energy difference for nodes 1 and 2 (for electrons and holes)
223  //1->2
224  double reorg_e12 = node1->reorg_intorig_electron + node2->reorg_intdest_electron + reorg_out_e;
225  double reorg_h12 = node1->reorg_intorig_hole + node2->reorg_intdest_hole + reorg_out_h;
226  //2->1
227  double reorg_e21 = node2->reorg_intorig_electron + node1->reorg_intdest_electron + reorg_out_e;
228  double reorg_h21 = node2->reorg_intorig_hole + node1->reorg_intdest_hole + reorg_out_h;
229 
230  //Site energy difference for nodes 1 and 2 (for electrons and holes)
231  //1->2
232  double dG_Site_e12 = node2->site_energy_electron - node1->site_energy_electron;
233  double dG_Site_h12 = node2->site_energy_hole - node1->site_energy_hole;
234  //2->1
235  double dG_Site_e21 = node1->site_energy_electron - node2->site_energy_electron;
236  double dG_Site_h21 = node1->site_energy_hole - node2->site_energy_hole;
237 
238  //Site energy difference and field difference for charge carrier hop
239  //1->2
240  double dG_e12 = dG_Site_e12 - dG_Field_e;
241  double dG_h12 = dG_Site_h12 - dG_Field_h;
242  //2->1
243  double dG_e21 = dG_Site_e21 - dG_Field_e;
244  double dG_h21 = dG_Site_h21 - dG_Field_h;
245 
246  //KMC calculated Marcus rates
247  //1->2
248  double cal_rate12e = 2*Pi/hbar * Jeff2e/sqrt(4*Pi*reorg_e12*kB*temperature) * exp(-(dG_e12+reorg_e12)*(dG_e12+reorg_e12) / (4*reorg_e12*kB*temperature));
249  double cal_rate12h = 2*Pi/hbar * Jeff2h/sqrt(4*Pi*reorg_h12*kB*temperature) * exp(-(dG_h12+reorg_h12)*(dG_h12+reorg_h12) / (4*reorg_h12*kB*temperature));
250  //2->1
251  double cal_rate21e = 2*Pi/hbar * Jeff2e/sqrt(4*Pi*reorg_e21*kB*temperature) * exp(-(dG_e21+reorg_e21)*(dG_e21+reorg_e21) / (4*reorg_e21*kB*temperature));
252  double cal_rate21h = 2*Pi/hbar * Jeff2h/sqrt(4*Pi*reorg_h21*kB*temperature) * exp(-(dG_h21+reorg_h21)*(dG_h21+reorg_h21) / (4*reorg_h21*kB*temperature));
253 
254  //std::cout << "Rate 12e " << cal_rate12e << " Rate 21e: " << cal_rate21e << std::endl;
255 
256  //add the new edge with calculated rate, to 1->2 and from 2->1
257  Edge* edge12 = new Edge(node1, node2, distance, cal_rate12e, cal_rate12h);
258  Edge* edge21 = new Edge(node2, node1, -distance, cal_rate21e, cal_rate21h);
259 
260  node1->AddEdge(edge12);
261  node2->AddEdge( edge21 );
262 
263  edges.push_back( edge12 );
264  edges.push_back( edge21 );
265 
266 
267  }
268 
269  delete stmt;
270 }
