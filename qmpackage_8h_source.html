106 inline bool QMPackage::WriteInputFilePBC( QMPair* pair, Orbitals* orbitals) {
107 
108  //std::cout << "IDFT writes input with PBC" << std::endl;
109 
110  Segment* seg1 = pair->Seg1();
111  Segment* seg2 = pair->Seg2();
112  Segment* ghost = NULL;
113 
114  Topology* _top = seg1->getTopology();
115 
116  vec r1 = seg1->getPos();
117  vec r2 = seg2->getPos();
118 
119  vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
120 
121  // Check whether pair formed across periodic boundary
122  if ( abs(r2 - r1 - _R) > 1e-8 ) {
123  ghost = new Segment(seg2);
124  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
125  vector<Atom*>::iterator ait;
126  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
127  (*ait)->setQMPos((*ait)->getQMPos()+r1-r2+_R);
128  }
129  }
130 
131  vector< Segment* > segments;
132  segments.push_back(seg1);
133 
134  if ( ghost ) {
135  segments.push_back(ghost);
136  } else {
137  segments.push_back(seg2);
138  }
139 
140  WriteInputFile( segments, orbitals);
141 
142  delete ghost;
143  return true;
144 }
