62  using Segmenttype =

64  using Sitetype = typename std::iterator_traits<

125  // each qmpackage has its own ordering of the individual functions in each
126  // shell
127  // i.e. VOTCA uses z,y,x e.g. Y1,0 Y1,-1 Y1,1 for the p shell
128  // d3z2-r2 dyz dxz dxy dx2-y2 e.g. Y2,0 Y2,-1 Y2,1 Y2,-2 for the d shell and
129  // so forth. ORCA uses z,x,y for the p shell
130  // these methods reorder the MOs to that format using the
131  // ShellReorder() and ShellMulitplier() which specify the order for each
