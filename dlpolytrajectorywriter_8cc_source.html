28 void DLPOLYTrajectoryWriter::Open(string file, bool bAppend)
29 // open/create a dlpoly configuration or trajectory file
30 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention:
31 // ".dlpc"="CONFIG_CGV", ".dlph"="HISTORY_CGV")
32 {
33  if (bAppend) {
34  throw std::runtime_error(
35  "Error: appending to dlpoly files not implemented");
36  }
37 
38  boost::filesystem::path filepath(file.c_str());
39  string out_name = "HISTORY_CGV";
40 
41  if (boost::filesystem::extension(filepath).size() == 0) {
42 
43  throw std::ios_base::failure("Error on creating dlpoly file '" + file +
44  "' - extension is expected, .dlph or .dlpc");
45 
46  } else if (boost::filesystem::extension(filepath) == ".dlpc") {
47 
48  _isConfig = true;
49  out_name = "CONFIG_CGV";
50 
51  } else if (boost::filesystem::extension(filepath) == ".dlph") {
52 
53  _isConfig = false;
54 
55  } else {
56  throw std::ios_base::failure("Error on creating dlpoly file '" + file +
57  "' - wrong extension, use .dlph or .dlpc");
58  }
59 
60  if (boost::filesystem::basename(filepath).size() == 0) {
61  if (filepath.parent_path().string().size() == 0) {
62  _fname = out_name;
63  } else {
64  _fname = filepath.parent_path().string() + "/" + out_name;
65  }
66  } else {
67  _fname = file;
68  }
69 
70  _fl.open(_fname);
71  if (!_fl.is_open()) {
72  throw std::ios_base::failure("Error on creating dlpoly file '" + _fname +
73  "'");
74  }
75 }

79 void DLPOLYTrajectoryWriter::Write(Topology *conf) {
80  static Index nstep = 1;
81  const double scale = 10.0; // nm -> A factor
82  Index mavecs = 0;
83  Index mpbct = 0;
84 
85  if (conf->HasForce() && conf->HasVel()) {
86  mavecs = 2;
87  } else if (conf->HasVel()) {
88  mavecs = 1;
89  }
90 
91  if (conf->getBoxType() == BoundaryCondition::typeOrthorhombic) {
92  mpbct = 2;
93  }
94  if (conf->getBoxType() == BoundaryCondition::typeTriclinic) {
95  mpbct = 3;
96  }
97 
98  if (_isConfig) {
99  double energy = 0.0;
100  _fl << "From VOTCA with love" << endl;
101  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10)
102  << conf->BeadCount() << setw(20) << energy << endl;
103  Eigen::Matrix3d m = conf->getBox();
104  for (Index i = 0; i < 3; i++) {
105  _fl << fixed << setprecision(10) << setw(20) << m(i, 0) * scale
106  << setw(20) << m(i, 1) * scale << setw(20) << m(i, 2) * scale << endl;
107  }
108 
109  } else {
110  static double dstep = 0.0;
111  if (nstep == 1) {
112  _fl << "From VOTCA with love" << endl;
113  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10)
114  << conf->BeadCount() << endl;
115  dstep = conf->getTime() / (double)(conf->getStep());
116  }
117 
118  _fl << "timestep" << setprecision(9) << setw(10) << conf->getStep()
119  << setw(10) << conf->BeadCount() << setw(10) << mavecs << setw(10)
120  << mpbct;
121  _fl << setprecision(9) << setw(12) << dstep << setw(12) << conf->getTime()
122  << endl;
123 
124  Eigen::Matrix3d m = conf->getBox();
125  for (Index i = 0; i < 3; i++) {
126  _fl << setprecision(12) << setw(20) << m(i, 0) * scale << setw(20)
127  << m(i, 1) * scale << setw(20) << m(i, 2) * scale << endl;
128  }
129  }
130 
131  for (Index i = 0; i < conf->BeadCount(); i++) {
132  Bead *bead = conf->getBead(i);
133 
134  // AB: DL_POLY needs bead TYPE, not name!
135 
136  if (_isConfig) {
137  _fl << setw(8) << left << bead->getType() << right << setw(10) << i + 1
138  << endl;
139  } else {
140  _fl << setw(8) << left << bead->getType() << right << setw(10) << i + 1;
141  _fl << setprecision(6) << setw(12) << bead->getMass() << setw(12)
142  << bead->getQ() << setw(12) << " 0.0" << endl;
143  }
144 
145  // alternative with atom NAME & fixed floating point format (in case the
146  // need arises)
147  //_fl << setw(8) << left << bead->getName() << right << setw(10) << i+1;
148  //_fl << fixed << setprecision(6) << setw(12) << bead->getMass() << setw(12)
149  //<< bead->getQ() << " 0.0" << endl;
150 
151  // nm -> Angs
152  _fl << resetiosflags(std::ios::fixed) << setprecision(12) << setw(20)
153  << bead->getPos().x() * scale;
154  _fl << setw(20) << bead->getPos().y() * scale << setw(20)
155  << bead->getPos().z() * scale << endl;
156 
157  if (mavecs > 0) {
158  if (!bead->HasVel()) {
159  throw std::ios_base::failure(
160  "Error: dlpoly frame is supposed to contain velocities, but bead "
161  "does not have v-data");
162  }
163 
164  // nm -> Angs
165  _fl << setprecision(12) << setw(20) << bead->getVel().x() * scale
166  << setw(20);
167  _fl << bead->getVel().y() * scale << setw(20)
168  << bead->getVel().z() * scale << endl;
169 
170  if (mavecs > 1) {
171  if (!bead->HasF()) {
172  throw std::ios_base::failure(
173  "Error: dlpoly frame is supposed to contain forces, but bead "
174  "does not have f-data");
175  }
176 
177  // nm -> Angs
178  _fl << setprecision(12) << setw(20) << bead->getF().x() * scale
179  << setw(20);
180  _fl << bead->getF().y() * scale << setw(20) << bead->getF().z() * scale
181  << endl;
182  }
183  }
184  }
185  nstep++;
186 }
