25 void DLPOLYTrajectoryWriter::Open(string file, bool bAppend)
26 // open/create a dlpoly configuration or trajectory file
27 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention: ".dlpc"="CONFIG_CGV", ".dlph"="HISTORY_CGV")
28 {
29  if (bAppend)
30  throw std::runtime_error("Error: appending to dlpoly files not implemented");
31 
32  boost::filesystem::path filepath(file.c_str());
33  string out_name="HISTORY_CGV";
34 
35  if ( boost::filesystem::extension(filepath).size() == 0 ) {
36 
37  throw std::ios_base::failure("Error on creating dlpoly file '" + file + "' - extension is expected, .dlph or .dlpc");
38 
39  } else if( boost::filesystem::extension(filepath)==".dlpc" ) {
40 
41  _isConfig=true;
42  out_name="CONFIG_CGV";
43 
44  } else if( boost::filesystem::extension(filepath)==".dlph" ) {
45 
46  _isConfig=false;
47 
48  } else {
49  throw std::ios_base::failure("Error on creating dlpoly file '" + file + "' - wrong extension, use .dlph or .dlpc");
50  }
51 
52  if ( boost::filesystem::basename(filepath).size() == 0 ) {
53  if (filepath.parent_path().string().size() == 0) {
54  _fname=out_name;
55  } else {
56  _fname=filepath.parent_path().string() + "/" + out_name;
57  }
58  } else {
59  _fname=file;
60  }
61 
62  _fl.open(_fname.c_str());
63  if(!_fl.is_open())
64  throw std::ios_base::failure("Error on creating dlpoly file '"+ _fname + "'");
65 }

72 void DLPOLYTrajectoryWriter::Write(Topology *conf)
73 {
74  static int nstep = 1;
75  static double dstep = 0.0;
76  const double scale = 10.0; // nm -> A factor
77  int mavecs = 0;
78  int mpbct = 0;
79  double energy = 0.0;
80 
81  if( conf->HasForce() && conf->HasVel() ) { mavecs=2; }
82  else if ( conf->HasVel() ) { mavecs=1; }
83 
84  if (conf->getBoxType()==BoundaryCondition::typeOrthorhombic) mpbct=2;
85  if (conf->getBoxType()==BoundaryCondition::typeTriclinic) mpbct=3;
86 
87  if( _isConfig ) {
88 
89  _fl << "From VOTCA with love" << endl;
90  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10) << conf->BeadCount() << setw(20) << energy << endl;
91  matrix m=conf->getBox();
92  for (int i=0;i<3;i++)
93  _fl << fixed << setprecision(10) << setw(20) << m[i][0]*scale << setw(20) << m[i][1]*scale << setw(20) << m[i][2]*scale << endl;
94 
95  } else {
96 
97  if (nstep==1) {
98  _fl << "From VOTCA with love" << endl;
99  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10) << conf->BeadCount() << endl;
100  dstep = conf->getTime()/(double)(conf->getStep());
101  }
102 
103  _fl << "timestep" << setprecision(9) << setw(10) << conf->getStep() << setw(10) << conf->BeadCount()
104  << setw(10) << mavecs << setw(10) << mpbct;
105  _fl << setprecision(9) << setw(12) << dstep << setw(12) <<conf->getTime() << endl;
106 
107  matrix m=conf->getBox();
108  for (int i=0;i<3;i++)
109  _fl << setprecision(12) << setw(20) << m[i][0]*scale << setw(20) << m[i][1]*scale << setw(20) << m[i][2]*scale << endl;
110 
111  }
112 
113  for (int i=0;i<conf->BeadCount(); i++){
114  Bead *bead=conf->getBead(i);
115 
116  // AB: DL_POLY needs bead TYPE, not name!
117 
118  if( _isConfig) {
119 
120  _fl << setw(8) << left << bead->getType()->getName() << right << setw(10) << i+1 << endl;
121 
122  } else {
123 
124  _fl << setw(8) << left << bead->getType()->getName() << right << setw(10) << i+1;
125  _fl << setprecision(6) << setw(12) << bead->getM() << setw(12) << bead->getQ() << setw(12) << " 0.0" << endl;
126 
127  }
128 
129  // alternative with atom NAME & fixed floating point format (in case the need arises)
130  //_fl << setw(8) << left << bead->getName() << right << setw(10) << i+1;
131  //_fl << fixed << setprecision(6) << setw(12) << bead->getM() << setw(12) << bead->getQ() << " 0.0" << endl;
132 
133  //nm -> Angs
134  _fl << resetiosflags(std::ios::fixed) << setprecision(12) << setw(20) << bead->getPos().getX()*scale;
135  _fl << setw(20) << bead->getPos().getY()*scale << setw(20) << bead->getPos().getZ()*scale << endl;
136 
137  if (mavecs>0) {
138  if (!bead->HasVel())
139  throw std::ios_base::failure("Error: dlpoly frame is supposed to contain velocities, but bead does not have v-data");
140 
141  //nm -> Angs
142  _fl << setprecision(12) << setw(20) << bead->getVel().getX()*scale << setw(20);
143  _fl << bead->getVel().getY()*scale << setw(20) << bead->getVel().getZ()*scale << endl;
144 
145  if (mavecs>1) {
146  if (!bead->HasF())
147  throw std::ios_base::failure("Error: dlpoly frame is supposed to contain forces, but bead does not have f-data");
148 
149  //nm -> Angs

151  _fl << bead->getF().getY()*scale << setw(20) << bead->getF().getZ()*scale << endl;
152  }
153  }
154  }
155  nstep++;
156 }
