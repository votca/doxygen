33 void DLPOLYTrajectoryWriter::Open(string file, bool bAppend)
34 // open/create a dlpoly configuration or trajectory file
35 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention:
36 // ".dlpc"="CONFIG_CGV", ".dlph"="HISTORY_CGV")
37 {
38  if (bAppend) {
39  throw std::runtime_error(
40  "Error: appending to dlpoly files not implemented");
41  }
42 
43  boost::filesystem::path filepath(file.c_str());
44  string out_name = "HISTORY_CGV";
45 
46  if (boost::filesystem::extension(filepath).size() == 0) {
47 
48  throw std::ios_base::failure("Error on creating dlpoly file '" + file +
49  "' - extension is expected, .dlph or .dlpc");
50 
51  } else if (boost::filesystem::extension(filepath) == ".dlpc") {
52 
53  _isConfig = true;
54  out_name = "CONFIG_CGV";
55 
56  } else if (boost::filesystem::extension(filepath) == ".dlph") {
57 
58  _isConfig = false;
59 
60  } else {
61  throw std::ios_base::failure("Error on creating dlpoly file '" + file +
62  "' - wrong extension, use .dlph or .dlpc");
63  }
64 
65  if (boost::filesystem::basename(filepath).size() == 0) {
66  if (filepath.parent_path().string().size() == 0) {
67  _fname = out_name;
68  } else {
69  _fname = filepath.parent_path().string() + "/" + out_name;
70  }
71  } else {
72  _fname = file;
73  }
74 
75  _fl.open(_fname);
76  if (!_fl.is_open()) {
77  throw std::ios_base::failure("Error on creating dlpoly file '" + _fname +
78  "'");
79  }
80 }

84 void DLPOLYTrajectoryWriter::Write(Topology *conf) {
85  static Index nstep = 1;
86  const double scale = 10.0; // nm -> A factor
87  Index mavecs = 0;
88  Index mpbct = 0;
89 
90  if (conf->HasForce() && conf->HasVel()) {
91  mavecs = 2;
92  } else if (conf->HasVel()) {
93  mavecs = 1;
94  }
95 
96  if (conf->getBoxType() == BoundaryCondition::typeOrthorhombic) {
97  mpbct = 2;
98  }
99  if (conf->getBoxType() == BoundaryCondition::typeTriclinic) {
100  mpbct = 3;
101  }
102 
103  if (_isConfig) {
104  double energy = 0.0;
105  _fl << "From VOTCA with love" << endl;
106  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10)
107  << conf->BeadCount() << setw(20) << energy << endl;
108  Eigen::Matrix3d m = conf->getBox();
109  for (Index i = 0; i < 3; i++) {
110  _fl << fixed << setprecision(10) << setw(20) << m(i, 0) * scale
111  << setw(20) << m(i, 1) * scale << setw(20) << m(i, 2) * scale << endl;
112  }
113 
114  } else {
115  static double dstep = 0.0;
116  if (nstep == 1) {
117  _fl << "From VOTCA with love" << endl;
118  _fl << setw(10) << mavecs << setw(10) << mpbct << setw(10)
119  << conf->BeadCount() << endl;
120  dstep = conf->getTime() / (double)(conf->getStep());
121  }
122 
123  _fl << "timestep" << setprecision(9) << setw(10) << conf->getStep()
124  << setw(10) << conf->BeadCount() << setw(10) << mavecs << setw(10)
125  << mpbct;
126  _fl << setprecision(9) << setw(12) << dstep << setw(12) << conf->getTime()
127  << endl;
128 
129  Eigen::Matrix3d m = conf->getBox();
130  for (Index i = 0; i < 3; i++) {
131  _fl << setprecision(12) << setw(20) << m(i, 0) * scale << setw(20)
132  << m(i, 1) * scale << setw(20) << m(i, 2) * scale << endl;
133  }
134  }
135 
136  for (Index i = 0; i < conf->BeadCount(); i++) {
137  Bead *bead = conf->getBead(i);
138 
139  // AB: DL_POLY needs bead TYPE, not name!
140 
141  if (_isConfig) {
142  _fl << setw(8) << left << bead->getType() << right << setw(10) << i + 1
143  << endl;
144  } else {
145  _fl << setw(8) << left << bead->getType() << right << setw(10) << i + 1;
146  _fl << setprecision(6) << setw(12) << bead->getMass() << setw(12)
147  << bead->getQ() << setw(12) << " 0.0" << endl;
148  }
149 
150  // alternative with atom NAME & fixed floating point format (in case the
151  // need arises)
152  //_fl << setw(8) << left << bead->getName() << right << setw(10) << i+1;
153  //_fl << fixed << setprecision(6) << setw(12) << bead->getMass() << setw(12)
154  //<< bead->getQ() << " 0.0" << endl;
155 
156  // nm -> Angs
157  _fl << resetiosflags(std::ios::fixed) << setprecision(12) << setw(20)
158  << bead->getPos().x() * scale;
159  _fl << setw(20) << bead->getPos().y() * scale << setw(20)
160  << bead->getPos().z() * scale << endl;
161 
162  if (mavecs > 0) {
163  if (!bead->HasVel()) {
164  throw std::ios_base::failure(
165  "Error: dlpoly frame is supposed to contain velocities, but bead "
166  "does not have v-data");
167  }
168 
169  // nm -> Angs
170  _fl << setprecision(12) << setw(20) << bead->getVel().x() * scale
171  << setw(20);
172  _fl << bead->getVel().y() * scale << setw(20)
173  << bead->getVel().z() * scale << endl;
174 
175  if (mavecs > 1) {
176  if (!bead->HasF()) {
177  throw std::ios_base::failure(
178  "Error: dlpoly frame is supposed to contain forces, but bead "
179  "does not have f-data");
180  }
181 
182  // nm -> Angs
183  _fl << setprecision(12) << setw(20) << bead->getF().x() * scale
184  << setw(20);
185  _fl << bead->getF().y() * scale << setw(20) << bead->getF().z() * scale
186  << endl;
187  }
188  }
189  }
190  nstep++;
191 }
