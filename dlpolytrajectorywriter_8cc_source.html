25 void DLPOLYTrajectoryWriter::Open(string file, bool bAppend)
26 // open/create a dlpoly configuration or trajectory file
27 // NOTE: allowed file naming - <name>.dlpc or <name>.dlph (convention: ".dlpc"="CONFIG_CGV", ".dlph"="HISTORY_CGV")
28 {
29  if (bAppend)
30  throw std::runtime_error("Error: appending to dlpoly files not implemented");
31 
32  boost::filesystem::path filepath(file.c_str());
33  string out_name="HISTORY_CGV";
34 
35  if ( boost::filesystem::extension(filepath).size() == 0 ) {
36 
37  throw std::ios_base::failure("Error on creating dlpoly file '" + file + "' - extension is expected, .dlph or .dlpc");
38 
39  } else if( boost::filesystem::extension(filepath)==".dlpc" ) {
40 
41  _isConfig=true;
42  out_name="CONFIG_CGV";
43 
44  } else if( boost::filesystem::extension(filepath)==".dlph" ) {
45 
46  _isConfig=false;
47 
48  } else {
49  throw std::ios_base::failure("Error on creating dlpoly file '" + file + "' - wrong extension, use .dlph or .dlpc");
50  }
51 
52  if ( boost::filesystem::basename(filepath).size() == 0 ) {
53  if (filepath.parent_path().string().size() == 0) {
54  _fname=out_name;
55  } else {
56  _fname=filepath.parent_path().string() + "/" + out_name;
57  }
58  } else {
59  _fname=file;
60  }
61 
62  _fl.open(_fname.c_str());
63  if(!_fl.is_open())
64  throw std::ios_base::failure("Error on creating dlpoly file '"+ _fname + "'");
65 }
