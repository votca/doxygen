103  : _gw_level(gw_level), _offset(offset), _sigma_c_func(sigma){};
104  std::pair<double, double> operator()(double frequency) const {
105  std::pair<double, double> value;
106  value.first =
107  _sigma_c_func.CalcCorrelationDiagElement(_gw_level, frequency);
108  value.second = _sigma_c_func.CalcCorrelationDiagElementDerivative(
109  _gw_level, frequency);
110  value.first += (_offset - frequency);
111  value.second -= 1.0;
112  return value;
113  }
