105  // small class which calculates f(w) with and df/dw(w)
106  // f=Sigma_c(w)+offset-w
107  // offset= e_dft+Sigma_x-Vxc
108  class QPFunc {
109  public:
110  QPFunc(Index gw_level, const Sigma_base& sigma, double offset)
111  : _gw_level(gw_level), _offset(offset), _sigma_c_func(sigma){};
112  std::pair<double, double> operator()(double frequency) const {
113  std::pair<double, double> value;
114  value.first =
115  _sigma_c_func.CalcCorrelationDiagElement(_gw_level, frequency);
116  value.second = _sigma_c_func.CalcCorrelationDiagElementDerivative(
117  _gw_level, frequency);
118  value.first += (_offset - frequency);
119  value.second -= 1.0;
120  return value;
121  }
122  double value(double frequency) const {
123  return _sigma_c_func.CalcCorrelationDiagElement(_gw_level, frequency) +
124  _offset - frequency;
125  }
126  double deriv(double frequency) const {
127  return _sigma_c_func.CalcCorrelationDiagElementDerivative(_gw_level,
128  frequency) +
129  _offset - frequency;
130  }
131 
132  private:
133  Index _gw_level;
134  double _offset;
135  const Sigma_base& _sigma_c_func;
136  };
