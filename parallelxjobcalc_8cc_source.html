31 template <typename JobContainer>
32 bool ParallelXJobCalc<JobContainer>::EvaluateFrame(const Topology &top) {
33 
34  // INITIALIZE PROGRESS OBSERVER
35  std::string progFile = _jobfile;
36  std::unique_ptr<JobOperator> master = std::unique_ptr<JobOperator>(
37  new JobOperator(-1, top, *this, _openmp_threads));
38  master->getLogger().setReportLevel(Log::current_level);
39  master->getLogger().setMultithreading(true);
40  master->getLogger().setPreface(Log::info, "\nMST INF");
41  master->getLogger().setPreface(Log::error, "\nMST ERR");
42  master->getLogger().setPreface(Log::warning, "\nMST WAR");
43  master->getLogger().setPreface(Log::debug, "\nMST DBG");
44  _progObs->InitFromProgFile(progFile, *(master.get()));
45 
46  // CREATE + EXECUTE THREADS (XJOB HANDLERS)
47  std::vector<std::unique_ptr<JobOperator>> jobOps;
48 
49  for (Index id = 0; id < _nThreads; id++) {
50  jobOps.push_back(std::unique_ptr<JobOperator>(
51  new JobOperator(id, top, *this, _openmp_threads)));
52  }
53 
54  for (Index id = 0; id < _nThreads; ++id) {
55  CustomizeLogger(*jobOps[id]);
56  }
57 
58  if (!_maverick) {
59  std::cout << std::endl; // REQUIRED FOR PROGRESS BAR IN OBSERVER
60  }
61 
62  for (Index id = 0; id < _nThreads; id++) {
63  jobOps[id]->Start();
64  }
65 
66  for (Index id = 0; id < _nThreads; id++) {
67  jobOps[id]->WaitDone();
68  }
69 
70  if (!_maverick) {
71  for (Index id = 0; id < _nThreads; id++) {
72  std::cout << std::endl << (jobOps[id]->getLogger()) << std::flush;
73  }
74  }
75 
76  jobOps.clear();
77 
78  // SYNC REMAINING COMPLETE JOBS
79  _progObs->SyncWithProgFile(*(master.get()));
80 
81  return true;
82 }

84 template <typename JobContainer>
85 void ParallelXJobCalc<JobContainer>::JobOperator::Run() {
86  OPENMP::setMaxThreads(_openmp_threads);
87  while (true) {
88  Job *job = _master._progObs->RequestNextJob(*this);
89 
90  if (job == nullptr) {
91  break;
92  } else {
93  Result res = this->_master.EvalJob(_top, *job, *this);
94  this->_master._progObs->ReportJobDone(*job, res, *this);
95  }
96  }
97 }
