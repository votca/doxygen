59  bool Exciton::Evaluate() {
60 
61 
62  if (_reporting == "silent") _log.setReportLevel(ctp::logERROR); // only output ERRORS, GEOOPT info, and excited state info for trial geometry
63  if (_reporting == "noisy") _log.setReportLevel(ctp::logDEBUG); // OUTPUT ALL THE THINGS
64  if (_reporting == "default") _log.setReportLevel(ctp::logINFO); //
65 
66  _log.setMultithreading(true);
67  _log.setPreface(ctp::logINFO, "\n... ...");
68  _log.setPreface(ctp::logERROR, "\n... ...");
69  _log.setPreface(ctp::logWARNING, "\n... ...");
70  _log.setPreface(ctp::logDEBUG, "\n... ...");
71 
72  // Get orbitals object
73  Orbitals _orbitals;
74 
75  // Read molecular geometry from xyz file and store in a segment (WHY SEGMENT?)
76  std::vector <ctp::Segment* > _segments;
77  ctp::Segment _segment(0, "mol");
78  CTP_LOG(ctp::logDEBUG, _log) << "Reading molecular coordinates from " << _xyzfile << flush;
79  ReadXYZ(&_segment, _xyzfile);
80  _segments.push_back(&_segment);
81 
82  // Get and initialize QMPackage for DFT ground state
83  QMPackage *_qmpackage = QMPackages().Create(_package);
84  _qmpackage->setLog(&_log);
85  _qmpackage->Initialize(&_package_options);
86  _qmpackage->setRunDir(".");
87 
88  // Get GWBSEENGINE Object and initialize
89  GWBSEENGINE _gwbse_engine;
90  _gwbse_engine.setLog(&_log);
91  _gwbse_engine.Initialize(&_gwbseengine_options, _archive_file);
92 
93  if ( _do_optimize ) {
94  // Run Geometry Optimization
95  GeometryOptimization _geoopt(_gwbse_engine,_qmpackage, _segments, &_orbitals);
96  _geoopt.setLog(&_log);
97  _geoopt.Initialize(&_geoopt_options);
98  _geoopt.Evaluate();
99  } else {
100  // Run GWBSE
101  _gwbse_engine.ExcitationEnergies(_qmpackage, _segments, &_orbitals);
102  }
103 
104  CTP_LOG(ctp::logDEBUG, _log) << "Saving data to " << _archive_file << flush;
105  _orbitals.Save(_archive_file);
106 
107  return true;
108  }

124

126 
127  if (in.is_open()) {
128  while (in.good()) {

131  std::vector< string > split;

135  split.size() != 4 ||
136  split[0] == "#" ||
137  split[0].substr(0, 1) == "#") {
138  continue;
139  }
140 
141  // Interesting information written here: e.g. 'C 0.000 0.000 0.000'
142  atomCount++;
143  string element = split[0];
144  double x = boost::lexical_cast<double>(split[1]) / 10.; //Â°A to NM
145  double y = boost::lexical_cast<double>(split[2]) / 10.;
146  double z = boost::lexical_cast<double>(split[3]) / 10.;
147  vec Pos = vec(x, y, z);
148  ctp::Atom *pAtom = new ctp::Atom(atomCount, element);
149  pAtom->setPos(Pos);
150  pAtom->setQMPart(atomCount, Pos);
151  pAtom->setElement(element);
152  _segment->AddAtom(pAtom);
153 
154  }
155  } else {
156  throw std::runtime_error("No such file: '" + filename + "'.");
157  }
