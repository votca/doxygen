62  bool Exciton::Evaluate() {
63 
64 
65  if (_reporting == "silent") _log.setReportLevel(ctp::logERROR); // only output ERRORS, GEOOPT info, and excited state info for trial geometry
66  if (_reporting == "noisy") _log.setReportLevel(ctp::logDEBUG); // OUTPUT ALL THE THINGS
67  if (_reporting == "default") _log.setReportLevel(ctp::logINFO); //
68 
69  _log.setMultithreading(true);
70  _log.setPreface(ctp::logINFO, "\n... ...");
71  _log.setPreface(ctp::logERROR, "\n... ...");
72  _log.setPreface(ctp::logWARNING, "\n... ...");
73  _log.setPreface(ctp::logDEBUG, "\n... ...");
74 
75  // Get orbitals object
76  Orbitals _orbitals;
77 
78  // Read molecular geometry from xyz file and store in a segment (WHY SEGMENT?)
79  std::vector <ctp::Segment* > _segments;
80  ctp::Segment _segment(0, "mol");
81  CTP_LOG(ctp::logDEBUG, _log) << "Reading molecular coordinates from " << _xyzfile << flush;
82  ReadXYZ(&_segment, _xyzfile);
83  _segments.push_back(&_segment);
84 
85  // Get and initialize QMPackage for DFT ground state
86  QMPackage *_qmpackage = QMPackages().Create(_package);
87  _qmpackage->setLog(&_log);
88  _qmpackage->Initialize(&_package_options);
89  _qmpackage->setRunDir(".");
90 
91  // Get GWBSEENGINE Object and initialize
92  GWBSEENGINE _gwbse_engine;
93  _gwbse_engine.setLog(&_log);
94  _gwbse_engine.Initialize(&_gwbseengine_options, _archive_file);
95 
96  if ( _do_optimize ) {
97  // Run Geometry Optimization
98  GeometryOptimization _geoopt(_gwbse_engine,_qmpackage, _segments, &_orbitals);
99  _geoopt.setLog(&_log);
100  _geoopt.Initialize(&_geoopt_options);
101  _geoopt.Evaluate();
102  } else {
103  // Run GWBSE
104  _gwbse_engine.ExcitationEnergies(_qmpackage, _segments, &_orbitals);
105  }
106 
107  CTP_LOG(ctp::logDEBUG, _log) << "Saving data to " << _archive_file << flush;
108  _orbitals.Save(_archive_file);
109 
110  Property _summary = _gwbse_engine.ReportSummary();
111  tools::PropertyIOManipulator iomXML(tools::PropertyIOManipulator::XML, 1, "");
112  CTP_LOG(ctp::logDEBUG, _log) << "Writing output to " << _xml_output << flush;
113  std::ofstream ofout(_xml_output.c_str(), std::ofstream::out);
114  ofout << (_summary.get("output"));
115  ofout.close();
116 
117  return true;
118  }

120  void Exciton::ReadXYZ(ctp::Segment* _segment, string filename) {
121 
122  string line;
123  std::ifstream in;
124 
125 
126  string label, type;
127  vec pos;
128 
129 
130  in.open(filename.c_str(), std::ios::in);
131  if (!in) throw runtime_error(string("Error reading coordinates from: ")
132  + filename);
133 
134 
135  int atomCount = 1;
136 
137  if (in.is_open()) {
138  while (in.good()) {
139  std::getline(in, line);
140 
141  std::vector< string > split;
142  Tokenizer toker(line, " \t");
143  toker.ToVector(split);
144  if (!split.size() ||
145  split.size() != 4 ||
146  split[0] == "#" ||
147  split[0].substr(0, 1) == "#") {
148  continue;
149  }
150 
151  // Interesting information written here: e.g. 'C 0.000 0.000 0.000'
152  atomCount++;
153  string element = split[0];
154  double x = boost::lexical_cast<double>(split[1]) / 10.; //°A to NM
155  double y = boost::lexical_cast<double>(split[2]) / 10.;
156  double z = boost::lexical_cast<double>(split[3]) / 10.;
157  vec Pos = vec(x, y, z);
158  ctp::Atom *pAtom = new ctp::Atom(atomCount, element);
159  pAtom->setPos(Pos);
160  pAtom->setQMPart(atomCount, Pos);
161  pAtom->setElement(element);
162  _segment->AddAtom(pAtom);
163 
164  }
165  } else {
166  throw std::runtime_error("No such file: '" + filename + "'.");
167  }
168 
169  return;
170  }
