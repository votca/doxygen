44 std::pair<double, double> CalcStatistics(const std::vector<double>& v) {
45  double mean = std::reduce(v.begin(), v.end()) / ((double)v.size());
46  double sq_sum = std::inner_product(v.begin(), v.end(), v.begin(), 0.0);
47  double stdev = std::sqrt(sq_sum / ((double)v.size()) - mean * mean);
48  return {mean, stdev};
49 }

51 template <class T>
52 tools::Property RunPart(T&& payload, const std::string& name,
53  Index repetitions) {
54  std::vector<double> individual_timings;
55  individual_timings.reserve(repetitions);
56  Index count = 0;
57  std::cout << name << std::endl;
58  for (Index i = 0; i < repetitions; i++) {
59  std::chrono::time_point<std::chrono::steady_clock> start =
60  std::chrono::steady_clock::now();
61  count += payload();
62  std::chrono::time_point<std::chrono::steady_clock> end =
63  std::chrono::steady_clock::now();
64  individual_timings.push_back(
65  std::chrono::duration_cast<std::chrono::duration<double> >(end - start)
66  .count());
67  }
68  // this is more to prevent optimisation of the whole loop if the payload is
69  // small
70  if (count != repetitions) {
71  std::cout << "sth went wrong with the count" << std::endl;
72  }
73  auto [mean1, std1] = CalcStatistics(individual_timings);
74  std::cout << "avg:" << mean1 << " std:" << std1 << std::endl;
75  tools::Property output(name, "", "");
76  output.add("avg", std::to_string(mean1));
77  output.add("std", std::to_string(std1));
78  tools::Property& runs = output.add("runs", "");
79  for (double time : individual_timings) {
80  runs.add("timing", std::to_string(time));
81  }
82  return output;
83 }

85 bool GPUBenchmark::Run() {
86 
87  Orbitals orb;
88  orb.ReadFromCpt(job_name_ + ".orb");
89  std::cout << "\n\nCreating benchmark for " << job_name_ << ".orb"
90  << std::endl;
91  AOBasis basis = orb.getDftBasis();
92  AOBasis auxbasis = orb.getAuxBasis();
93 
94  std::cout << "Repetitions:" << repetitions_ << std::endl;
95  std::cout << "Number of CPUs:" << OPENMP::getMaxThreads()
96  << " \nNumber gpus:" << OpenMP_CUDA::UsingGPUs() << std::endl;
97  std::cout << "Using MKL for Eigen overload:" << XTP_HAS_MKL_OVERLOAD()
98  << std::endl;
99 
100  TCMatrix_gwbse Mmn;
101  Index max_3c = std::max(orb.getBSEcmax(), orb.getGWAmax());
102  Mmn.Initialize(auxbasis.AOBasisSize(), orb.getRPAmin(), max_3c,
103  orb.getRPAmin(), orb.getRPAmax());
104  std::cout << "BasisSet:" << basis.Name() << " size:" << basis.AOBasisSize()
105  << std::endl;
106  std::cout << "AuxBasisSet:" << auxbasis.Name()
107  << " size:" << auxbasis.AOBasisSize() << std::endl;
108  std::cout << "rpamin:" << orb.getRPAmin() << " rpamax:" << orb.getRPAmax()
109  << std::endl;
110 
111  tools::Property output("GPU_Benchmark", "", "");
112  output.add("Repetitions", std::to_string(repetitions_));
113  output.add("CPUs", std::to_string(OPENMP::getMaxThreads()));
114 
115  output.add("GPUs", std::to_string(OpenMP_CUDA::UsingGPUs()));
116  output.add("MKL_overload", std::to_string(XTP_HAS_MKL_OVERLOAD()));
117  output.add("Basisset", basis.Name());
118  output.add("Basissetsize", std::to_string(basis.AOBasisSize()));
119  output.add("AuxBasisset", auxbasis.Name());
120  output.add("AuxBasissetsize", std::to_string(auxbasis.AOBasisSize()));
121  output.add("rpamin", std::to_string(orb.getRPAmin()));
122  output.add("rpamax", std::to_string(orb.getRPAmax()));
123  output.add(RunPart(
124  [&]() {
125  Mmn.Fill(auxbasis, basis, orb.MOs().eigenvectors());
126  return 1;
127  },
128  "Filling_ThreeCenter", repetitions_));
129 
130  Eigen::MatrixXd op =
131  Eigen::MatrixXd::Random(auxbasis.AOBasisSize(), auxbasis.AOBasisSize());
132  output.add(RunPart(
133  [&]() {
134  Mmn.MultiplyRightWithAuxMatrix(op);
135  return 1;
136  },
137  "Multiplication_of_tensor_with_matrix", repetitions_));
138  Mmn.Rebuild();
139  Logger log;
140  RPA rpa(log, Mmn);
141  rpa.configure(orb.getHomo(), orb.getRPAmin(), orb.getRPAmax());
142  Index rpatotal = orb.getRPAmax() - orb.getRPAmin() + 1;
143  rpa.setRPAInputEnergies(
144  orb.MOs().eigenvalues().segment(orb.getRPAmin(), rpatotal));
145  double frequency = 0.5;
146  Eigen::MatrixXd result =
147  Eigen::MatrixXd::Zero(auxbasis.AOBasisSize(), auxbasis.AOBasisSize());
148  output.add(RunPart(
149  [&]() {
150  result += rpa.calculate_epsilon_i(frequency);
151  result += rpa.calculate_epsilon_r(-frequency);
152  return 1;
153  },
154  "RPA_evaluation", repetitions_));
155  frequency += result.sum();
156  Index hqp_size = orb.getBSEcmax() - orb.getBSEvmin() + 1;
157  Eigen::MatrixXd Hqp_fake = Eigen::MatrixXd::Random(hqp_size, hqp_size);
158  Eigen::VectorXd epsilon_inv_fake = result.diagonal();
159  BSEOperator_Options opt;
160  opt.cmax = orb.getBSEcmax();
161  opt.homo = orb.getHomo();
162  opt.qpmin = orb.getGWAmin();
163  opt.rpamin = orb.getRPAmin();
164  opt.vmin = orb.getBSEvmin();
165  SingletOperator_TDA s_op(epsilon_inv_fake, Mmn, Hqp_fake);
166  s_op.configure(opt);
167  TripletOperator_TDA t_op(epsilon_inv_fake, Mmn, Hqp_fake);
168  t_op.configure(opt);
169  SingletOperator_BTDA_B sbtda_op(epsilon_inv_fake, Mmn, Hqp_fake);
170  sbtda_op.configure(opt);
171  HxOperator hx_op(epsilon_inv_fake, Mmn, Hqp_fake);
172  hx_op.configure(opt);
173  Index spacesize = 100; // A searchspace of 100 is pretty decent
174  Eigen::MatrixXd state = Eigen::MatrixXd::Random(s_op.size(), spacesize);
175  Eigen::MatrixXd result_op = Eigen::MatrixXd::Zero(s_op.size(), spacesize);
176 
177  output.add(RunPart(
178  [&]() {
179  result_op += s_op * state;
180  return 1;
181  },
182  "SingletOperator_TDA", repetitions_));
183 
184  output.add(RunPart(
185  [&]() {
186  result_op += t_op * state;
187  return 1;
188  },
189  "TripletOperator_TDA", repetitions_));
190 
191  output.add(RunPart(
192  [&]() {
193  result_op += sbtda_op * state;
194  return 1;
195  },
196  "SingletOperator_BTDA_B", repetitions_));
197 
198  output.add(RunPart(
199  [&]() {
200  result_op += hx_op * state;
201  return 1;
202  },
203  "HxOperator", repetitions_));
204  std::ofstream outputfile;
205  outputfile.open(outputfile_);
206  outputfile << output << std::endl;
207  outputfile.close();
208  frequency += result_op.sum();
209  return true;
210 }
