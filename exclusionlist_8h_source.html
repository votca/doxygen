114  ;Bead *bead2 = *i;

145 inline void ExclusionList::RemoveExclusion(Bead *bead1, Bead *bead2) {
146  if (bead2->getId() < bead1->getId()) swap(bead1, bead2);
147  if(bead1==bead2) return;
148  if(!IsExcluded(bead1, bead2)) return;
149  list<exclusion_t*>::iterator ex;
150  for(ex=_exclusions.begin(); ex!=_exclusions.end(); ++ex)
151  if((*ex)->_atom == bead1) break;
152  if(ex==_exclusions.end()) return;
153  (*ex)->_exclude.remove(bead2);
154  if((*ex)->_exclude.empty()) {
155  (*ex)=NULL;
156  _exclusions.erase(ex);
157  }
158  _exclusions.remove(NULL);
159 }
