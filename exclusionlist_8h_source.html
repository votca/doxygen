158 inline void ExclusionList::RemoveExclusion(Bead *bead1, Bead *bead2) {
159  if (bead2->getId() < bead1->getId()) {
160  std::swap(bead1, bead2);
161  }
162 
163  if (bead1 == bead2) {
164  return;
165  }
166 
167  if (!IsExcluded(bead1, bead2)) {
168  return;
169  }
170 
171  std::list<exclusion_t *>::iterator ex;
172  for (ex = _exclusions.begin(); ex != _exclusions.end(); ++ex) {
173  if ((*ex)->_atom == bead1) {
174  break;
175  }
176 
177  if (ex == _exclusions.end()) {
178  return;
179  }
180 
181  (*ex)->_exclude.remove(bead2);
182  if ((*ex)->_exclude.empty()) {
183  (*ex) = nullptr;
184  _exclusions.erase(ex);
185  }
186  _exclusions.remove(nullptr);
187  }
188 }
