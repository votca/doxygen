109  Bead *bead1 = bead1_;
110  ;
111  Bead *bead2 = *i;
112  if (bead2->getId() < bead1->getId()) std::swap(bead1, bead2);

142 inline void ExclusionList::RemoveExclusion(Bead *bead1, Bead *bead2) {
143  if (bead2->getId() < bead1->getId()) std::swap(bead1, bead2);
144  if (bead1 == bead2) return;
145  if (!IsExcluded(bead1, bead2)) return;
146  std::list<exclusion_t *>::iterator ex;
147  for (ex = _exclusions.begin(); ex != _exclusions.end(); ++ex)
148  if ((*ex)->_atom == bead1) break;
149  if (ex == _exclusions.end()) return;
150  (*ex)->_exclude.remove(bead2);
151  if ((*ex)->_exclude.empty()) {
152  (*ex) = NULL;
153  _exclusions.erase(ex);
154  }
155  _exclusions.remove(NULL);
156 }
