40std::pair<std::string, std::vector<double>> readFchkFile(std::string filename,
41 Index skipLines = 0) {
42 std::ifstream inFile(filename);
43 if (inFile) {
44 std::ostringstream ss;
45 std::string line;
46 for (int i = 0; i < skipLines; ++i) {
47 std::getline(inFile, line);
48 }
49 while (!inFile.eof()) {
50 std::getline(inFile, line);
51 ss << line << "\n";
52 if (line.find("Total SCF Density") != std::string::npos) {
53 break;
54 }
55 }
56 std::vector<double> density;
57 double dummy;
58 inFile >> dummy;
59 while (!inFile.eof()) {
60 density.push_back(dummy);
61 inFile >> dummy;
62 }
63 return std::pair<std::string, std::vector<double>>(ss.str(), density);
64 } else {
65 throw std::runtime_error("Could not open file: " + filename + "\n");
66 }
67}

69BOOST_AUTO_TEST_CASE(gaussianwriter_test) {
70 libint2::initialize();
71
72 // 1. Setup Orbitals Object
73 Orbitals orbitals;
74 orbitals.QMAtoms().LoadFromFile(std::string(XTP_TEST_DATA_FOLDER) +
75 "/gaussianwriter/methane.xyz");
76 orbitals.SetupDftBasis(std::string(XTP_TEST_DATA_FOLDER) +
77 "/gaussianwriter/def2-tzvp.xml");
78
79 orbitals.MOs().eigenvalues() = votca::tools::EigenIO_MatrixMarket::ReadMatrix(
80 std::string(XTP_TEST_DATA_FOLDER) + "/gaussianwriter/orb_energies.mm");
81 orbitals.MOs().eigenvectors() =
82 votca::tools::EigenIO_MatrixMarket::ReadMatrix(
83 std::string(XTP_TEST_DATA_FOLDER) + "/gaussianwriter/orb_coeffs.mm");
84 orbitals.setQMEnergy(-40.53758914883583);
85 orbitals.setNumberOfAlphaElectrons(5);
86 orbitals.setNumberOfOccupiedLevels(5);
87
88 // 2. Write object to gaussian checkpoint file (.fchk)
89 Logger log;
90 GaussianWriter writer(log);
91 bool diff2gs = false;
92 writer.WriteFile("methane", orbitals, QMState("n"), diff2gs);
93
94 // 3.1 Check everything except density matrix
95 Index linesToSkip = 2; // basisset names will be different due to test folder
96 std::pair<std::string, std::vector<double>> testFile =
97 readFchkFile("methane.fchk", linesToSkip);
98 std::pair<std::string, std::vector<double>> refFile = readFchkFile(
99 std::string(XTP_TEST_DATA_FOLDER) + "/gaussianwriter/fchkRefMethane.fchk",
100 linesToSkip);
101
102 bool filesAreEqual = testFile.first == refFile.first;
103 if (!filesAreEqual) {
104 std::cout << "GENERATED FILE: " << std::endl;
105 std::cout << testFile.first << std::endl;
106 std::cout << "REFERENCE FILE: " << std::endl;
107 std::cout << refFile.first << std::endl;
108 }
109 BOOST_CHECK(filesAreEqual);
110
111 // 3.2 Check Density Matrix
112 bool densityMatrixIsEqual = true;
113 for (Index i = 0; i < static_cast<Index>(testFile.second.size()); ++i) {
114 densityMatrixIsEqual =
115 densityMatrixIsEqual &&
116 (std::abs(testFile.second[i] - refFile.second[i]) < 1e-10);
117 }
118 if (!densityMatrixIsEqual) {
119 std::cout << "GENERATED FILE vs REFERENCE: " << std::endl;
120 for (Index i = 0; i < static_cast<Index>(testFile.second.size()); ++i) {
121 std::cout << testFile.second[i] << " " << refFile.second[i]
122 << std::endl;
123 }
124 }
125 BOOST_CHECK(densityMatrixIsEqual);
126
127 libint2::finalize();
128}
