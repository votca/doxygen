55 bool Energy_costfunction::Converged(const Eigen::VectorXd& delta_parameters,
56  double delta_cost,
57  const Eigen::VectorXd& gradient) {
58  _iteration++;
59  bool energy_converged = false;
60  bool RMSForce_converged = false;
61  bool MaxForce_converged = false;
62  bool RMSStep_converged = false;
63  bool MaxStep_converged = false;
64  Energy_costfunction::conv_paras convval;
65  convval.deltaE = delta_cost;
66  convval.RMSForce =
67  std::sqrt(gradient.cwiseAbs2().sum()) / double(gradient.size());
68  convval.MaxForce = gradient.cwiseAbs().maxCoeff(&convval.maxforceindex);
69  convval.RMSStep = std::sqrt(delta_parameters.cwiseAbs2().sum()) /
70  double(delta_parameters.size());
71  convval.MaxStep = delta_parameters.cwiseAbs().maxCoeff(&convval.maxstepindex);
72 
73  if (std::abs(convval.deltaE) < _convpara.deltaE) {
74  energy_converged = true;
75  }
76  if (convval.RMSForce < _convpara.RMSForce) {
77  RMSForce_converged = true;
78  }
79  if (convval.MaxForce < _convpara.MaxForce) {
80  MaxForce_converged = true;
81  }
82  if (convval.RMSStep < _convpara.RMSStep) {
83  RMSStep_converged = true;
84  }
85  if (convval.MaxStep < _convpara.MaxStep) {
86  MaxStep_converged = true;
87  }
88  Report(convval);
89  if (energy_converged && RMSForce_converged && MaxForce_converged &&
90  RMSStep_converged && MaxStep_converged) {
91  return true;
92  }
93  return false;
94 }

159 Eigen::VectorXd Energy_costfunction::Write3XMatrixToVector(
160  const Eigen::MatrixX3d& matrix) {
161  Eigen::VectorXd vec = Eigen::VectorXd::Zero(matrix.rows() * matrix.cols());
162  for (Index i_cart = 0; i_cart < 3; i_cart++) {
163  for (Index i_atom = 0; i_atom < matrix.rows(); i_atom++) {
164  Index idx = 3 * i_atom + i_cart;
165  vec(idx) = matrix(i_atom, i_cart);
166  }
167  }
168  return vec;
169 }
