54 bool Energy_costfunction::Converged(const Eigen::VectorXd& delta_parameters,
55  double delta_cost,
56  const Eigen::VectorXd& gradient) {
57  _iteration++;
58  bool energy_converged = false;
59  bool RMSForce_converged = false;
60  bool MaxForce_converged = false;
61  bool RMSStep_converged = false;
62  bool MaxStep_converged = false;
63  Energy_costfunction::conv_paras convval;
64  convval.deltaE = delta_cost;
65  convval.RMSForce = std::sqrt(gradient.cwiseAbs2().sum()) / gradient.size();
66  convval.MaxForce = gradient.cwiseAbs().maxCoeff(&convval.maxforceindex);
67  convval.RMSStep =
68  std::sqrt(delta_parameters.cwiseAbs2().sum()) / delta_parameters.size();
69  convval.MaxStep = delta_parameters.cwiseAbs().maxCoeff(&convval.maxstepindex);
70 
71  if (std::abs(convval.deltaE) < _convpara.deltaE) energy_converged = true;
72  if (convval.RMSForce < _convpara.RMSForce) RMSForce_converged = true;
73  if (convval.MaxForce < _convpara.MaxForce) MaxForce_converged = true;
74  if (convval.RMSStep < _convpara.RMSStep) RMSStep_converged = true;
75  if (convval.MaxStep < _convpara.MaxStep) MaxStep_converged = true;
76  Report(convval);
77  if (energy_converged && RMSForce_converged && MaxForce_converged &&
78  RMSStep_converged && MaxStep_converged) {
79  return true;
80  }
81  return false;
82 }

84 void Energy_costfunction::Report(const Energy_costfunction::conv_paras& val) {
85  const Energy_costfunction::conv_paras& paras = getConvParas();
86  CTP_LOG(ctp::logINFO, *_pLog)

89  .str()

100  .str()
101  << std::flush;

149 Eigen::VectorXd Energy_costfunction::Write3XMatrixToVector(
150  const Eigen::MatrixX3d& matrix) {
151  Eigen::VectorXd vec = Eigen::VectorXd::Zero(matrix.rows() * matrix.cols());
152  for (int i_cart = 0; i_cart < 3; i_cart++) {
153  for (int i_atom = 0; i_atom < matrix.rows(); i_atom++) {
154  int idx = 3 * i_atom + i_cart;
155  vec(idx) = matrix(i_atom, i_cart);
156  }
157  }
158  return vec;
159 }
