28double Energy_costfunction::EvaluateCost(const Eigen::VectorXd& parameters) {
29 Vector2QMAtoms(parameters, orbitals_.QMAtoms());
30 gwbse_engine_.setLoggerFile("");
31 std::string logger_file;
32 if (iteration_ == 0) {
33 logger_file =
34 "gwbse_iteration_" + (boost::format("%1%") % iteration_).str() + ".log";
35 iteration_++;

37 logger_file =
38 "gwbse_iteration_" + (boost::format("%1%") % iteration_).str() + ".log";
39 }
40 gwbse_engine_.setLoggerFile(logger_file);
41 gwbse_engine_.ExcitationEnergies(orbitals_);
42 energy_ = orbitals_.getTotalStateEnergy(

55bool Energy_costfunction::Converged(const Eigen::VectorXd& delta_parameters,
56 double delta_cost,
57 const Eigen::VectorXd& gradient) {
58 iteration_++;
59 bool energy_converged = false;
60 bool RMSForce_converged = false;
61 bool MaxForce_converged = false;
62 bool RMSStep_converged = false;
63 bool MaxStep_converged = false;
64 Energy_costfunction::conv_paras convval;
65 convval.deltaE = delta_cost;
66 convval.RMSForce =
67 std::sqrt(gradient.cwiseAbs2().sum()) / double(gradient.size());
68 convval.MaxForce = gradient.cwiseAbs().maxCoeff(&convval.maxforceindex);
69 convval.RMSStep = std::sqrt(delta_parameters.cwiseAbs2().sum()) /
70 double(delta_parameters.size());
71 convval.MaxStep = delta_parameters.cwiseAbs().maxCoeff(&convval.maxstepindex);
72
73 if (std::abs(convval.deltaE) < convpara_.deltaE) {
74 energy_converged = true;
75 }
76 if (convval.RMSForce < convpara_.RMSForce) {
77 RMSForce_converged = true;
78 }
79 if (convval.MaxForce < convpara_.MaxForce) {
80 MaxForce_converged = true;
81 }
82 if (convval.RMSStep < convpara_.RMSStep) {
83 RMSStep_converged = true;
84 }
85 if (convval.MaxStep < convpara_.MaxStep) {
86 MaxStep_converged = true;
87 }
88 Report(convval);
89 if (energy_converged && RMSForce_converged && MaxForce_converged &&
90 RMSStep_converged && MaxStep_converged) {
91 return true;
92 }
93 return false;
94}

139void Energy_costfunction::Vector2QMAtoms(const Eigen::VectorXd& pos,
140 QMMolecule& atoms) {
141 for (Index i = 0; i < atoms.size(); i++) {
142 Eigen::Vector3d pos_displaced;
143 pos_displaced << pos(3 * i), pos(3 * i + 1), pos(3 * i + 2);
144 atoms[i].setPos(pos_displaced);
145 }
146}

148Eigen::VectorXd Energy_costfunction::QMAtoms2Vector(QMMolecule& atoms) {
149 Eigen::VectorXd result = Eigen::VectorXd::Zero(3 * atoms.size());
150
151 for (Index i = 0; i < atoms.size(); i++) {
152 result(3 * i) = atoms[i].getPos().x();
153 result(3 * i + 1) = atoms[i].getPos().y();
154 result(3 * i + 2) = atoms[i].getPos().z();
155 }
156 return result;
157}

159Eigen::VectorXd Energy_costfunction::Write3XMatrixToVector(
160 const Eigen::MatrixX3d& matrix) {
161 Eigen::VectorXd vec = Eigen::VectorXd::Zero(matrix.rows() * matrix.cols());
162 for (Index i_cart = 0; i_cart < 3; i_cart++) {
163 for (Index i_atom = 0; i_atom < matrix.rows(); i_atom++) {
164 Index idx = 3 * i_atom + i_cart;
165 vec(idx) = matrix(i_atom, i_cart);
166 }
167 }
168 return vec;
169}
