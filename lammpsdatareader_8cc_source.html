42 vector<string> TrimCommentsFrom_(vector<string> fields) {
43  vector<string> tempFields;
44  for (const auto &field : fields) {
45  if (field.at(0) == '#') {
46  break;
47  }
48  tempFields.push_back(field);
49  }
50  return tempFields;
51 }

115 bool LAMMPSDataReader::NextFrame(Topology &top) {
116 
117  string header;
118  getline(fl_, header);
119  string line;
120  getline(fl_, line);
121  while (!fl_.eof()) {
122 
123  tools::Tokenizer tok(line, " ");
124  vector<string> fields = tok.ToVector();
125  fields = TrimCommentsFrom_(fields);
126  // If not check the size of the vector and parse according
127  // to the number of fields
128  if (fields.size() == 1) {
129  MatchOneFieldLabel_(fields, top);
130  } else if (fields.size() == 2) {
131  MatchTwoFieldLabels_(fields, top);
132  } else if (fields.size() == 3) {
133  MatchThreeFieldLabels_(fields);
134  } else if (fields.size() == 4) {
135  MatchFourFieldLabels_(fields, top);
136  } else if (fields.size() != 0) {
137  string err = "Unrecognized line in lammps .data file:\n" + line;
138  throw runtime_error(err);
139  }
140  getline(fl_, line);
141  }
142  return !fl_.eof();
143 }

149 void LAMMPSDataReader::RenameMolecule(Molecule &mol) const {
150  if (mol.getName() == "UNKNOWN") {
151  std::map<std::string, Index> molname_map;
152  for (const Bead *atom : mol.Beads()) {
153  std::string atomname = atom->getName();
154  std::string element = atomname.substr(0, 1);
155  if (std::islower(atomname[1])) {
156  element += atomname[1];
157  }
158  molname_map[element]++;
159  }
160 
161  std::string molname = "";
162  for (auto const &pair : molname_map) {
163  molname += (pair.first + std::to_string(pair.second));
164  }
165  mol.setName(molname);
166  }
167 }

169 bool LAMMPSDataReader::MatchOneFieldLabel_(vector<string> fields,
170  Topology &top) {
171 
172  if (fields.at(0) == "Masses") {
173  SortIntoDataGroup_("Masses");
174  InitializeAtomAndBeadTypes_();
175  } else if (fields.at(0) == "Atoms") {
176  ReadAtoms_(top);
177  } else if (fields.at(0) == "Bonds") {
178  ReadBonds_(top);
179  } else if (fields.at(0) == "Angles") {
180  ReadAngles_(top);
181  } else if (fields.at(0) == "Dihedrals") {
182  ReadDihedrals_(top);
183  } else if (fields.at(0) == "Impropers") {
184  cout << endl;
185  cout << "WARNING Impropers are not currently supported, skipping." << endl;
186  cout << endl;
187  // Impropers are not yet supported
188  SkipImpropers_();
189  } else {
190  return false;
191  }
192  return true;
193 }

253 void LAMMPSDataReader::InitializeAtomAndBeadTypes_() {
254  if (!data_.count("Masses")) {
255  string err = "Masses must first be parsed before the atoms can be read.";
256  throw runtime_error(err);
257  }
258 
259  Index index = 0;
260  tools::Elements elements;
261  for (auto mass : data_["Masses"]) {
262  // Determine the mass associated with the atom
263  double mass_atom_bead = stod(mass.at(1));
264  string baseName;
265  if (elements.isMassAssociatedWithElement(mass_atom_bead, 0.01)) {
266  baseName = elements.getEleShortClosestInMass(mass_atom_bead, 0.01);
267  } else {
268  baseName = "Bead";
269  cout << "Unable to associate mass " << mass.at(1)
270  << " with element assuming pseudo atom, assigning name "
271  << "Bead" << mass.at(0) << " ." << endl;
272  }
273  atomtypes_[index] = baseName + mass.at(0);
274  ++index;
275  }
276 }

278 void LAMMPSDataReader::ReadBox_(vector<string> fields, Topology &top) {
279  Eigen::Matrix3d m = Eigen::Matrix3d::Zero();
280  m(0, 0) = stod(fields.at(1)) - stod(fields.at(0));
281 
282  for (Index i = 1; i < 3; ++i) {
283  string line;
284  getline(fl_, line);
285  tools::Tokenizer tok(line, " ");
286  tok.ConvertToVector(fields);
287  if (fields.size() != 4) {
288  throw runtime_error("invalid box format in the lammps data file");
289  }
290 
291  m(i, i) = stod(fields.at(1)) - stod(fields.at(0));
292  }
293  top.setBox(m * tools::conv::ang2nm);
294 }

340 LAMMPSDataReader::lammps_format LAMMPSDataReader::determineDataFileFormat_(
341  string line) {
342 
343  tools::Tokenizer tok(line, " ");
344  vector<string> fields = tok.ToVector();
345  lammps_format format;
346  if (fields.size() == 5 || fields.size() == 8) {
347  format = style_atomic;
348  } else if (fields.size() == 6 || fields.size() == 9) {
349  format = style_angle_bond_molecule;
350  } else if (fields.size() == 7 || fields.size() == 10) {
351  format = style_full;
352  } else {
353  throw runtime_error(
354  "You have submitted a lammps data file with an "
355  "unsupported format.");
356  }
357  return format;
358 }

360 void LAMMPSDataReader::ReadAtoms_(Topology &top) {
361 
362  if (data_.count("Masses") == 0) {
363  string err =
364  "You are attempting to read in the atom block before the masses, or "
365  "you have failed to include the masses in the data file.";
366  throw runtime_error(err);
367  }
368 
369  string line;
370  getline(fl_, line);
371  getline(fl_, line);
372 
373  lammps_format format = determineDataFileFormat_(line);
374  bool chargeRead = false;
375  bool moleculeRead = false;
376  if (format == style_angle_bond_molecule) {
377  moleculeRead = true;
378  }
379  if (format == style_full) {
380  moleculeRead = true;
381  chargeRead = true;
382  }
383 
384  map<Index, string> sorted_file;
385  Index startingIndex;
386  Index startingIndexMolecule = 0;
387  istringstream issfirst(line);
388  issfirst >> startingIndex;
389  if (moleculeRead) {
390  issfirst >> startingIndexMolecule;
391  }
392  sorted_file[startingIndex] = line;
393  getline(fl_, line);
394  boost::trim(line);
395 
396  Index atomId = 0;
397  Index moleculeId = 0;
398  while (!line.empty()) {
399  istringstream iss(line);
400  iss >> atomId;
401  if (moleculeRead) {
402  iss >> moleculeId;
403  }
404  sorted_file[atomId] = line;
405  getline(fl_, line);
406  boost::trim(line);
407  if (atomId < startingIndex) {
408  startingIndex = atomId;
409  }
410  if (moleculeId < startingIndexMolecule) {
411  startingIndexMolecule = moleculeId;
412  }
413  }
414 
415  for (Index atomIndex = startingIndex;
416  static_cast<size_t>(atomIndex - startingIndex) < sorted_file.size();
417  ++atomIndex) {
418 
419  Index atomTypeId;
420  double charge = 0;
421  double x, y, z;
422 
423  istringstream iss(sorted_file[atomIndex]);
424  iss >> atomId;
425  if (moleculeRead) {
426  iss >> moleculeId;
427  } else {
428  moleculeId = atomId;
429  }
430  iss >> atomTypeId;
431  if (chargeRead) {
432  iss >> charge;
433  }
434  iss >> x;
435  iss >> y;
436  iss >> z;
437 
438  // Exclusion list assumes beads start with ids of 0
439  --atomId;
440  --atomTypeId;
441  moleculeId -= startingIndexMolecule;
442 
443  // We want to start with an index of 0 not 1
444  // atomId;
445  Bead *b;
446  if (topology_) {
447 
448  atomIdToIndex_[atomId] = atomIndex - startingIndex;
449  atomIdToMoleculeId_[atomId] = moleculeId;
450  Molecule *mol;
451  if (!molecules_.count(moleculeId)) {
452  mol = top.CreateMolecule("UNKNOWN");
453  molecules_[moleculeId] = mol;
454  } else {
455  mol = molecules_[moleculeId];
456  }
457 
458  double mass = stod(data_["Masses"].at(atomTypeId).at(1));
459 
460  if (Index(data_.at("Masses").size()) <= atomTypeId) {
461  string err =
462  "The atom block contains an atom of type " + to_string(atomTypeId) +
463  " however, the masses are only specified for atoms up to type " +
464  to_string(data_.at("Masses").size() - 1);
465  throw runtime_error(err);
466  }
467 
468  Index residue_index = moleculeId;
469  if (residue_index >= top.ResidueCount()) {
470  while ((residue_index - 1) >= top.ResidueCount()) {
471  top.CreateResidue("DUM");
472  }
473  top.CreateResidue("DUM");
474  }
475 
476  if (atomtypes_.count(atomTypeId) == 0) {
477  string err =
478  "Unrecognized atomTypeId, the atomtypes map "
479  "may be uninitialized";
480  throw runtime_error(err);
481  }
482 
483  string bead_type_name = atomtypes_[atomTypeId];
484  if (!top.BeadTypeExist(bead_type_name)) {
485  top.RegisterBeadType(bead_type_name);
486  }
487 
488  b = top.CreateBead(Bead::spherical, bead_type_name, bead_type_name,
489  residue_index, mass, charge);
490 
491  mol->AddBead(b, bead_type_name);
492  b->setMoleculeId(mol->getId());
493 
494  } else {
495  b = top.getBead(atomIndex - startingIndex);
496  }
497 
498  Eigen::Vector3d xyz_pos(x, y, z);
499  b->setPos(xyz_pos);
500  }
501 
502  if (top.BeadCount() != numberOf_["atoms"]) {
503  string err =
504  "The number of atoms read in is not equivalent to the "
505  "number of atoms indicated to exist in the lammps data file. \n"
506  "Number of atoms that should exist " +
507  to_string(numberOf_["atoms"]) + "\nNumber of atoms that were read in " +
508  to_string(top.BeadCount()) + "\n";
509  throw runtime_error(err);
510  }
511 }

513 void LAMMPSDataReader::ReadBonds_(Topology &top) {
514  string line;
515  getline(fl_, line);
516  getline(fl_, line);
517  boost::trim(line);
518 
519  Index bondId;
520  Index bondTypeId;
521  Index atom1Id, atom2Id;
522 
523  Index bond_count = 0;
524  while (!line.empty()) {
525 
526  if (topology_) {
527  istringstream iss(line);
528  iss >> bondId;
529  iss >> bondTypeId;
530  iss >> atom1Id;
531  iss >> atom2Id;
532 
533  --atom1Id;
534  --atom2Id;
535  --bondTypeId;
536  --bondId;
537 
538  Index atom1Index = atomIdToIndex_[atom1Id];
539  Index atom2Index = atomIdToIndex_[atom2Id];
540 
541  Interaction *ic = new IBond(atom1Index, atom2Index);
542  ic->setGroup("BONDS");
543  ic->setIndex(bondId);
544  auto b = top.getBead(atom1Index);
545  auto mi = top.getMolecule(b->getMoleculeId());
546  ic->setMolecule(atomIdToMoleculeId_[atom1Index]);
547  top.AddBondedInteraction(ic);
548  mi->AddInteraction(ic);
549  }
550 
551  ++bond_count;
552  getline(fl_, line);
553  boost::trim(line);
554  }
555 
556  if (bond_count != numberOf_["bonds"]) {
557  string err =
558  "The number of bonds read in is not equivalent to the "
559  "number of bonds indicated to exist in the lammps data file. \n"
560  "Number of bonds that should exist " +
561  to_string(numberOf_["bonds"]) + "\nNumber of bonds that were read in " +
562  to_string(bond_count) + "\n";
563  throw runtime_error(err);
564  }
