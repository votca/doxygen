66 bool LAMMPSDataReader::NextFrame(Topology &top) {
67 
68  string line;
69  getline(fl_, line);
70  while (!fl_.eof()) {
71 
72  bool labelMatched = false;
73  Tokenizer tok(line, " ");
74  vector<string> fields;
75  tok.ToVector(fields);
76  fields = TrimCommentsFrom_(fields);
77 
78  // If not check the size of the vector and parse according
79  // to the number of fields
80  if (fields.size() == 1) {
81  labelMatched = MatchOneFieldLabel_(fields, top);
82  } else if (fields.size() == 2) {
83  labelMatched = MatchTwoFieldLabels_(fields, top);
84  } else if (fields.size() == 3) {
85  labelMatched = MatchThreeFieldLabels_(fields, top);
86  } else if (fields.size() == 4) {
87  labelMatched = MatchFourFieldLabels_(fields, top);
88  } else if (fields.size() != 0) {
89 
90  // See if the line is the lammps .data header/info line
91  labelMatched = MatchFieldsTimeStepLabel_(fields, top);
92 
93  if (!labelMatched) {
94  string err = "Unrecognized line in lammps .data file:\n" + line;
95  throw runtime_error(err);
96  }
97  }
98  getline(fl_, line);
99  }
100  return !fl_.eof();
101 }

211 // The purpose of this function is to take lammps output where there are more
212 // than a single atom type of the same element. For instance there may be 4
213 // atom types with mass of 12.01. Well this means that they are all carbon but
214 // are treated differently in lammps. It makes since to keep track of this. So
215 // When creating the atom names we will take into account so say we have the
216 // following masses in the lammps .data file:
217 // Masses
218 //
219 // 1 1.0
220 // 2 12.01
221 // 3 12.01
222 // 4 16.0
223 // 5 12.01
224 //
225 // Then we would translate this to the following atom names
226 // 1 H
227 // 2 C1
228 // 3 C2
229 // 4 O
230 // 5 C3
231 //
232 // Note that we do not append a number if it is singular, in such cases the
233 // element and the atom name is the same.
234 void LAMMPSDataReader::InitializeAtomTypes_() {
235  if (!data_.count("Masses")) {
236  string err = "Masses must first be parsed before the atoms can be read.";
237  throw runtime_error(err);
238  }
239 
240  std::map<std::string, int> countAtomsOfSameElement;
241  Elements elements;
242 
243  for (auto mass : data_["Masses"]) {
244  // Determine the mass associated with the atom
245  double mass_atom = boost::lexical_cast<double>(mass.at(1));
246  // Determine the element (Symbol) by looking at the mass
247  // Second argument is the tolerance
248  string eleShort = elements.getEleShortClosestInMass(mass_atom, 0.01);
249  // Count the number of atoms
250  if (!countAtomsOfSameElement.count(eleShort)) {
251  countAtomsOfSameElement[eleShort] = 1;
252  }
253  countAtomsOfSameElement[eleShort]++;
254  }
255 
256  vector<int> indices(0, countAtomsOfSameElement.size());
257  // If there is more than one atom type of the same element append a number
258  // to the atom type name
259  map<string, int> eleShortIndices;
260  int index = 0;
261  for (auto mass : data_["Masses"]) {
262  // Determine the mass associated with the atom
263  double mass_atom = boost::lexical_cast<double>(mass.at(1));
264  // Determine the element (Symbol) by looking at the mass
265  // Second argument is the tolerance
266  string eleShort = elements.getEleShortClosestInMass(mass_atom, 0.01);
267  string label = eleShort;
268  if (countAtomsOfSameElement[eleShort] > 1) {
269  if (eleShortIndices.count(eleShort) == 0) {
270  label += "1";
271  eleShortIndices[eleShort] = 1;
272  } else {
273  eleShortIndices[eleShort]++;
274  label += to_string(eleShortIndices[eleShort]);
275  }
276  }
277  atomtypes_[index].push_back(eleShort);
278  atomtypes_[index].push_back(label);
279  ++index;
280  }
281 }

283 void LAMMPSDataReader::ReadBox_(vector<string> fields, Topology &top) {
284  matrix m;
285  m.ZeroMatrix();
286  m[0][0] = boost::lexical_cast<double>(fields.at(1)) -
287  boost::lexical_cast<double>(fields.at(0));
288 
289  for (int i = 1; i < 3; ++i) {
290  string line;
291  getline(fl_, line);
292  Tokenizer tok(line, " ");
293  tok.ConvertToVector(fields);
294  if (fields.size() != 4) {
295  throw runtime_error("invalid box format in the lammps data file");
296  }
297 
298  m[i][i] = boost::lexical_cast<double>(fields.at(1)) -
299  boost::lexical_cast<double>(fields.at(0));
300  }
301  top.setBox(m);
302 }

350 void LAMMPSDataReader::ReadAtoms_(Topology &top) {
351 
352  string line;
353  getline(fl_, line);
354  getline(fl_, line);
355 
356  int atomId;
357  int moleculeId;
358  int atomTypeId;
359 
360  double charge = 0;
361  double x, y, z;
362 
363  map<int, string> sorted_file;
364 
365  while (!line.empty()) {
366 
367  istringstream iss(line);
368  iss >> atomId;
369  --atomId;
370  sorted_file[atomId] = line;
371  getline(fl_, line);
372  }
373 
374  for (int atomIndex = 0;
375  boost::lexical_cast<size_t>(atomIndex) < sorted_file.size();
376  ++atomIndex) {
377 
378  istringstream iss(sorted_file[atomIndex]);
379  iss >> atomId;
380  iss >> moleculeId;
381  iss >> atomTypeId;
382  iss >> x;
383  iss >> y;
384  iss >> z;
385 
386  // We want to start with an index of 0 not 1
387  atomId--;
388  moleculeId--;
389  atomTypeId--;
390 
391  Bead *b;
392  if (topology_) {
393 
394  atomIdToIndex_[atomId] = atomIndex;
395  atomIdToMoleculeId_[atomId] = moleculeId;
396 
397  Molecule *mol;
398  if (!molecules_.count(moleculeId)) {
399  mol = top.CreateMolecule("Unknown");
400  molecules_[moleculeId] = mol;
401  } else {
402  mol = molecules_[moleculeId];
403  }
404 
405  int symmetry = 1; // spherical
406  double mass =
407  boost::lexical_cast<double>(data_["Masses"].at(atomTypeId).at(1));
408  // Will use the molecule id as the resnum for lack of a better option
409  int resnr = moleculeId;
410 
411  string bead_type_name = atomtypes_[atomTypeId].at(1);
412  BeadType *bead_type = top.GetOrCreateBeadType(bead_type_name);
413  if (atomtypes_.count(atomTypeId) == 0) {
414  string err = "Unrecognized atomTypeId, the atomtypes map "
415  "may be uninitialized";
416  throw runtime_error(err);
417  }
418 
419  b = top.CreateBead(symmetry, bead_type_name, bead_type, resnr, mass,
420  charge);
421 
422  mol->AddBead(b, bead_type_name);
423  b->setMolecule(mol);
424 
425  } else {
426  b = top.getBead(atomIndex);
427  }
428 
429  vec xyz_pos(x, y, z);
430  b->setPos(xyz_pos);
431  }
432 
433  if (top.BeadCount() != numberOf_["atoms"]) {
434  string err =
435  "The number of atoms read in is not equivalent to the "
436  "number of atoms indicated to exist in the lammps data file. \n"
437  "Number of atoms that should exist " +
438  to_string(numberOf_["atoms"]) + "\nNumber of atoms that were read in " +
439  to_string(top.BeadCount()) + "\n";
440  throw runtime_error(err);
441  }
442 }
