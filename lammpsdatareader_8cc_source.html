32 vector<string> TrimCommentsFrom_(vector<string> fields) {
33  vector<string> tempFields;
34  for (auto field : fields) {
35  if (field.at(0) == '#') return tempFields;
36  tempFields.push_back(field);
37  }
38  return tempFields;
39 }

41 // trim from start (in place)
42 static inline void ltrim_(std::string &s) {
43  s.erase(s.begin(), std::find_if(s.begin(), s.end(),
44  [](int ch) { return !std::isspace(ch); }));
45 }
46 
47 // trim from end (in place)
48 static inline void rtrim_(std::string &s) {
49  s.erase(std::find_if(s.rbegin(), s.rend(),
50  [](int ch) { return !std::isspace(ch); })
51  .base(),
52  s.end());
53 }
54 
55 // trim from both ends (in place)
56 static inline void trim_(std::string &s) {
57  ltrim_(s);
58  rtrim_(s);
59 }

61 bool withinTolerance_(double value1, double value2, double tolerance) {
62  return abs(value1 - value2) / min(value1, value2) < tolerance;
63 }

65 string getStringGivenDoubleAndMap_(double value, map<string, double> nameValue,
66  double tolerance) {
67 
68  for (auto string_value_pair : nameValue) {
69  if (withinTolerance_(value, string_value_pair.second, tolerance)) {
70  return string_value_pair.first;
71  }
72  }
73  throw runtime_error(
74  "getStringGivenDoubleAndMap_ function fails. This method "
75  "is meant to be passed a double that is to be matched within a tolerance"
76  " with a double in a map<string,double> and then return the string. It is"
77  " likely that none of the doubles were a close enough match.");
78 }

135 bool LAMMPSDataReader::NextFrame(Topology &top) {
136 
137  string line;
138  getline(fl_, line);
139  while (!fl_.eof()) {
140 
141  bool labelMatched = false;
142  Tokenizer tok(line, " ");
143  vector<string> fields;
144  tok.ToVector(fields);
145  fields = TrimCommentsFrom_(fields);
146  // If not check the size of the vector and parse according
147  // to the number of fields
148  if (fields.size() == 1) {
149  labelMatched = MatchOneFieldLabel_(fields, top);
150  } else if (fields.size() == 2) {
151  labelMatched = MatchTwoFieldLabels_(fields, top);
152  } else if (fields.size() == 3) {
153  labelMatched = MatchThreeFieldLabels_(fields, top);
154  } else if (fields.size() == 4) {
155  labelMatched = MatchFourFieldLabels_(fields, top);
156  } else if (fields.size() != 0) {
157 
158  // See if the line is the lammps .data header/info line
159  labelMatched = MatchFieldsTimeStepLabel_(fields, top);
160 
161  if (!labelMatched) {
162  string err = "Unrecognized line in lammps .data file:\n" + line;
163  throw runtime_error(err);
164  }
165  }
166  getline(fl_, line);
167  }
168  return !fl_.eof();
169 }

176  Topology &top) {
177 
178  if (fields.at(0) == "Masses") {
179  SortIntoDataGroup_("Masses");

186  ReadAngles_(top);
187  } else if (fields.at(0) == "Dihedrals") {
188  ReadDihedrals_(top);
189  } else if (fields.at(0) == "Impropers") {
190  } else {
191  return false;
192  }
193  return true;

268 void LAMMPSDataReader::InitializeAtomAndBeadTypes_() {
269  if (!data_.count("Masses")) {
270  string err = "Masses must first be parsed before the atoms can be read.";
271  throw runtime_error(err);
272  }
273 
274  auto baseNamesMasses = determineBaseNameAssociatedWithMass_();
275  auto baseNamesCount = determineAtomAndBeadCountBasedOnMass_(baseNamesMasses);
276 
277  // If there is more than one atom type of the same element append a number
278  // to the atom type name
279  map<string, int> baseNameIndices;
280  int index = 0;
281 
282  for (auto mass : data_["Masses"]) {
283  // Determine the mass associated with the atom
284  double mass_atom_bead = boost::lexical_cast<double>(mass.at(1));
285 
286  auto baseName =
287  getStringGivenDoubleAndMap_(mass_atom_bead, baseNamesMasses, 0.01);
288 
289  string label = baseName;
290  if (baseNamesCount[baseName] > 1) {
291  if (baseNameIndices.count(baseName) == 0) {
292  label += label + " Type 1";
293  baseNameIndices[baseName] = 1;
294  } else {
295  baseNameIndices[baseName]++;
296  label += label + " Type " + to_string(baseNameIndices[baseName]);
297  }
298  }
299  atomtypes_[index].push_back(baseName);
300  atomtypes_[index].push_back(label);
301  ++index;
302  }
303 }

305 map<string, double> LAMMPSDataReader::determineBaseNameAssociatedWithMass_() {
306  Elements elements;
307  map<string, double> baseNamesAndMasses;
308  int bead_index_type = 1;
309  for (auto mass : data_["Masses"]) {
310  double mass_atom_bead = boost::lexical_cast<double>(mass.at(1));
311  string beadElementName;
312  if (elements.isMassAssociatedWithElement(mass_atom_bead, 0.01)) {
313  beadElementName = elements.getEleShortClosestInMass(mass_atom_bead, 0.01);
314  } else {
315  beadElementName = "Bead" + to_string(bead_index_type);
316  ++bead_index_type;
317  }
318  baseNamesAndMasses[beadElementName] = mass_atom_bead;
319  }
320  return baseNamesAndMasses;
321 }

341 void LAMMPSDataReader::ReadBox_(vector<string> fields, Topology &top) {
342  matrix m;
343  m.ZeroMatrix();
344  m[0][0] = boost::lexical_cast<double>(fields.at(1)) -
345  boost::lexical_cast<double>(fields.at(0));
346 
347  for (int i = 1; i < 3; ++i) {
348  string line;
349  getline(fl_, line);
350  Tokenizer tok(line, " ");
351  tok.ConvertToVector(fields);
352  if (fields.size() != 4) {
353  throw runtime_error("invalid box format in the lammps data file");
354  }
355 
356  m[i][i] = boost::lexical_cast<double>(fields.at(1)) -
357  boost::lexical_cast<double>(fields.at(0));
358  }
359  top.setBox(m);
360 }

363  string line;

367  vector<vector<string>> group;

412 LAMMPSDataReader::lammps_format LAMMPSDataReader::determineDataFileFormat_(
413  string line) {
414 
415  Tokenizer tok(line, " ");
416  vector<string> fields;
417  tok.ConvertToVector(fields);
418  lammps_format format;
419  if (fields.size() == 5 || fields.size() == 8) {
420  format = style_atomic;
421  } else if (fields.size() == 6 || fields.size() == 9) {
422  format = style_angle_bond_molecule;
423  } else if (fields.size() == 7 || fields.size() == 10) {
424  format = style_full;
425  } else {
426  throw runtime_error(
427  "You have submitted a lammps data file with an "
428  "unsupported format.");
429  }
430  return format;
431 }

433 void LAMMPSDataReader::ReadAtoms_(Topology &top) {
434 
435  string line;
436  getline(fl_, line);
437  getline(fl_, line);
438 
439  lammps_format format = determineDataFileFormat_(line);
440  bool chargeRead = false;
441  bool moleculeRead = false;
442  if (format == style_angle_bond_molecule) moleculeRead = true;
443  if (format == style_full) {
444  moleculeRead = true;
445  chargeRead = true;
446  }
447 
448  map<int, string> sorted_file;
449  int startingIndex;
450  int startingIndexMolecule = 0;
451  istringstream issfirst(line);
452  issfirst >> startingIndex;
453  if (moleculeRead) {
454  issfirst >> startingIndexMolecule;
455  }
456  sorted_file[startingIndex] = line;
457  getline(fl_, line);
458  trim_(line);
459 
460  int atomId = 0;
461  int moleculeId = 0;
462  while (!line.empty()) {
463  istringstream iss(line);
464  iss >> atomId;
465  if (moleculeRead) {
466  iss >> moleculeId;
467  }
468  sorted_file[atomId] = line;
469  getline(fl_, line);
470  trim_(line);
471  if (atomId < startingIndex) startingIndex = atomId;
472  if (moleculeId < startingIndexMolecule) startingIndexMolecule = moleculeId;
473  }
474 
475  for (int atomIndex = startingIndex;
476  static_cast<size_t>(atomIndex - startingIndex) < sorted_file.size();
477  ++atomIndex) {
478 
479  int atomTypeId;
480  double charge = 0;
481  double x, y, z;
482 
483  istringstream iss(sorted_file[atomIndex]);
484  iss >> atomId;
485  if (moleculeRead) {
486  iss >> moleculeId;
487  } else {
488  moleculeId = atomId;
489  }
490  iss >> atomTypeId;
491  if (chargeRead) iss >> charge;
492  iss >> x;
493  iss >> y;
494  iss >> z;
495 
496  // Exclusion list assumes beads start with ids of 0
497  --atomId;
498  --atomTypeId;
499  moleculeId -= startingIndexMolecule;
500 
501  // We want to start with an index of 0 not 1
502  // atomId;
503  Bead *b;
504  if (topology_) {
505 
506  atomIdToIndex_[atomId] = atomIndex - startingIndex;
507  atomIdToMoleculeId_[atomId] = moleculeId;
508  Molecule *mol;
509  if (!molecules_.count(moleculeId)) {
510  mol = top.CreateMolecule("UNKNOWN");
511  molecules_[moleculeId] = mol;
512  } else {
513  mol = molecules_[moleculeId];
514  }
515  int symmetry = 1; // spherical
516  double mass =
517  boost::lexical_cast<double>(data_["Masses"].at(atomTypeId).at(1));
518 
519  int residue_index = moleculeId;
520  if (residue_index >= top.ResidueCount()) {
521  while ((residue_index - 1) >= top.ResidueCount()) {
522  top.CreateResidue("DUM");
523  }
524  top.CreateResidue("DUM");
525  }
526 
527  string bead_type_name = to_string(atomTypeId + 1);
528  if (!top.BeadTypeExist(bead_type_name)) {
529  top.RegisterBeadType(bead_type_name);
530  }
531  if (atomtypes_.count(atomTypeId) == 0) {
532  string err =
533  "Unrecognized atomTypeId, the atomtypes map "
534  "may be uninitialized";
535  throw runtime_error(err);
536  }
537 
538  b = top.CreateBead(symmetry, bead_type_name, bead_type_name,
539  residue_index, mass, charge);
540 
541  mol->AddBead(b, bead_type_name);
542  b->setMolecule(mol);
543 
544  } else {
545  b = top.getBead(atomIndex - startingIndex);
546  }
547 
548  vec xyz_pos(x, y, z);
549  b->setPos(xyz_pos);
550  }
551 
552  if (top.BeadCount() != numberOf_["atoms"]) {
553  string err =
554  "The number of atoms read in is not equivalent to the "
555  "number of atoms indicated to exist in the lammps data file. \n"
556  "Number of atoms that should exist " +
557  to_string(numberOf_["atoms"]) + "\nNumber of atoms that were read in " +
558  to_string(top.BeadCount()) + "\n";
559  throw runtime_error(err);
560  }
561 }

567  trim_(line);
568 
569  int bondId;
570  int bondTypeId;
571  int atom1Id, atom2Id;
572 
573  int bond_count = 0;
574  while (!line.empty()) {
575 
576  if (topology_) {
577  istringstream iss(line);

579  iss >> bondTypeId;
