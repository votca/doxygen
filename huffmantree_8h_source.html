31  void makeTree() {
32  if (!events) {
33  throw std::runtime_error(
34  "Error in Huffmantree::makeTree : Pointer to Events not set!");
35  }
36 
37  // queue of the nodes, sorted by probability
38  auto compare = [](huffmanNode<T> *n1, huffmanNode<T> *n2) {
39  return n1->probability > n2->probability;
40  };
41 
42  // priority queues, because the algorithm always needs the element with the
43  // smallest probability. Also, it keep adding nodes to it, so it would we
44  // very inefficient to sort it in every iteration.
45  std::priority_queue<huffmanNode<T> *, std::vector<huffmanNode<T> *>,
46  decltype(compare)>
47  queue(compare);
48 
49  htree = std::vector<huffmanNode<T>>(
50  events->size() % 2 ? events->size() : events->size() - 1);
51 
52  auto comp2 = [](T *e1, T *e2) { return e1->getValue() > e2->getValue(); };
53  std::priority_queue<T *, std::vector<T *>, decltype(comp2)> eventQueue(
54  comp2);
55  sum_of_values = 0.0;
56 
57  Index firstEmptyFieldIndex = 0;
58  for (T &e : *events) {
59  eventQueue.push(&e);
60  sum_of_values += e.getValue();
61  }
62  while (eventQueue.size() > 1) {
63  htree[firstEmptyFieldIndex].isOnLastLevel = true;
64  htree[firstEmptyFieldIndex].leftLeaf = eventQueue.top();
65  eventQueue.pop();
66  htree[firstEmptyFieldIndex].rightLeaf = eventQueue.top();
67  eventQueue.pop();
68  htree[firstEmptyFieldIndex].probability =
69  (htree[firstEmptyFieldIndex].leftLeaf->getValue() +
70  htree[firstEmptyFieldIndex].rightLeaf->getValue()) /
71  sum_of_values;
72  queue.push(&(htree[firstEmptyFieldIndex]));
73  firstEmptyFieldIndex++;
74  }
75  if (!eventQueue.empty()) {
76  htree[firstEmptyFieldIndex].isOnLastLevel = true;
77  htree[firstEmptyFieldIndex].rightLeaf = eventQueue.top();
78  htree[firstEmptyFieldIndex].leftLeaf = eventQueue.top();
79  htree[firstEmptyFieldIndex].probability =
80  htree[firstEmptyFieldIndex].leftLeaf->getValue() / sum_of_values;
81  queue.push(&(htree[firstEmptyFieldIndex]));
82  firstEmptyFieldIndex++;
83  }
84 
85  // now connect the hnodes, making a new one for every connection:
86  // always take the two nodes with the smallest probability and "combine"
87  // them, repeat, until just one node (the root) is left.
88  huffmanNode<T> *h1;
89  huffmanNode<T> *h2;
90  while (queue.size() > 1) {
91  h1 = queue.top();
92  queue.pop();
93  h2 = queue.top();
94  queue.pop();
95  htree[firstEmptyFieldIndex].probability =
96  h1->probability + h2->probability;
97  htree[firstEmptyFieldIndex].leftChild = h1;
98  htree[firstEmptyFieldIndex].rightChild = h2;
99  queue.push(&(htree[firstEmptyFieldIndex]));
100  firstEmptyFieldIndex++;
101  }
102  // reorganize the probabilities: in every node, add the probability of one
103  // subtree ("small") to all nodes of the other subtree.
104  addProbabilityFromRightSubtreeToLeftSubtree(&htree[htree.size() - 1], 0);
105  moveProbabilitiesFromRightSubtreesOneLevelUp(&htree[htree.size() - 1]);
106  treeIsMade = true;
107  }
108 
109  T *findHoppingDestination(double p) const {
110  if (!treeIsMade) {
111  throw std::runtime_error(
112  "Tried to find Hopping Destination without initializing the "
113  "Huffmantree first!");
114  }
115  const huffmanNode<T> *node = &htree.back();
116  while (!node->isOnLastLevel) {
117  if (p > node->probability) {
118  node = node->leftChild;
119  } else {
120  node = node->rightChild;
121  }
122  }
123  return (p > node->probability ? node->leftLeaf : node->rightLeaf);
124  }

141  void addProbabilityFromRightSubtreeToLeftSubtree(huffmanNode<T> *n,
142  double add) {
143  // for each node, adds the probability of the right childnode to the left
144  // childnode and every node under it. if the Tree would look like this (with
145  // the Numbers representing the probability of every node) before calling
146  // this function
147 
148  // 1.0
149  // ____||____
150  // | |
151  // 0.4 0.6
152  // _||_ _||_
153  // | | | |
154  // 0.25 0.15 0.35 0.25
155  // _||_
156  // | |
157  // 0.1 0.05
158  // then it would look like this after calling it
159  // 1.0
160  // ____||____
161  // | |
162  // 1.0 0.6
163  // _||_ _||_
164  // | | | |
165  // 1.0 0.75 0.6 0.25
166  // _||_
167  // | |
168  // 0.75 0.65
169  // now the tree could be traversed with "while (!n.isLeaf())
170  // n=p>n.right.p?n.left:n.right" so in the function
171  // moveProbabilitiesFromRightSubtreesOneLevelUp the numbers are moved one
172  // level up to call n.p instead of n.right.p
173 
174  // adds "add" to the probability, then calls itself recursively.
175  // this calculates the probabilities needed to traverse the tree quickly
176  n->probability += add;
177  // if leftId=-1 (=> node is leaf), returns
178  if (n->isOnLastLevel) {
179  return;
180  }
181 
182  addProbabilityFromRightSubtreeToLeftSubtree(
183  n->leftChild, add + n->rightChild->probability);
184  addProbabilityFromRightSubtreeToLeftSubtree(n->rightChild, add);
185  }
186 
187  void moveProbabilitiesFromRightSubtreesOneLevelUp(huffmanNode<T> *n) {
188  // moves the Probabilities on the right subtrees one level up.
189  // if the Tree would look like this (with the Numbers representing the
190  // probability of every node) before calling this function
191  // 1.0
192  // ____||____
193  // | |
194  // 1.0 0.6
195  // _||_ _||_
196  // | | | |
197  // 1.0 0.75 0.6 0.25
198  // _||_
199  // | |
200  // 0.75 0.65
201  // then it would look like this after calling it
202  // 0.
203  // ____||____
204  // | |
205  // 0.75 0.25
206  // _||_ _||_
207  // | | | |
208  // 1.0 0.65 0.6 0.25
209  // _||_
210  // | |
211  // 0.75 0.65
212  // note, that now the probabilities on the leaf level are not needed anymore
213  // to traverse the tree; the algorithm now is "while (!n.isLeaf())
214  // n=p>n.p?n.left:n.right"
215  if (n->isOnLastLevel) {
216  n->probability -= n->leftLeaf->getValue() / sum_of_values;
217  } else {
218  n->probability = n->rightChild->probability;
219  moveProbabilitiesFromRightSubtreesOneLevelUp(n->rightChild);
220  moveProbabilitiesFromRightSubtreesOneLevelUp(n->leftChild);
221  }
222  }

225  bool treeIsMade = false;

227  std::vector<T> *events = nullptr;
