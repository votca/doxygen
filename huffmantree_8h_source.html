35  void makeTree(){
36  if (!events) throw runtime_error("Error in Huffmantree::makeTree : Pointer to Events not set!");
37 
38  //queue of the nodes, sorted by probability
39  auto compare = [](huffmanNode<T> * n1, huffmanNode<T> * n2)
40  { return n1->probability>n2->probability;};
41 
42  //priority queues, because the algorithm always needs the element with the smallest probability.
43  //Also, it keep adding nodes to it, so it would we very inefficient to sort it in every iteration.
44  priority_queue<huffmanNode<T> *,vector<huffmanNode<T> *>, decltype(compare)> queue(compare);
45 
46  htree=vector<huffmanNode<T>>(events->size()%2?events->size():events->size()-1);
47 
48  auto comp2 = [](T * e1, T * e2){
49  return e1->rate>e2->rate;
50  };
51  priority_queue<T *, vector<T *>, decltype(comp2)> eventQueue(comp2);
52  sum_of_values=0.0;
53 
54  int firstEmptyFieldIndex=0;
55  for (T &e:*events){
56  eventQueue.push(&e);
57  sum_of_values+=e.getValue();
58  }
59  while (eventQueue.size()>1){
60  htree[firstEmptyFieldIndex].isOnLastLevel=true;
61  htree[firstEmptyFieldIndex].leftLeaf=eventQueue.top();
62  eventQueue.pop();
63  htree[firstEmptyFieldIndex].rightLeaf=eventQueue.top();
64  eventQueue.pop();
65  htree[firstEmptyFieldIndex].probability=(htree[firstEmptyFieldIndex].leftLeaf->rate+
66  htree[firstEmptyFieldIndex].rightLeaf->rate)/sum_of_values;
67  queue.push(&(htree[firstEmptyFieldIndex]));
68  firstEmptyFieldIndex++;
69  }
70  if (!eventQueue.empty()){
71  htree[firstEmptyFieldIndex].isOnLastLevel=true;
72  htree[firstEmptyFieldIndex].rightLeaf=eventQueue.top();
73  htree[firstEmptyFieldIndex].leftLeaf=eventQueue.top();
74  htree[firstEmptyFieldIndex].probability=htree[firstEmptyFieldIndex].leftLeaf->rate/sum_of_values;
75  queue.push(&(htree[firstEmptyFieldIndex]));
76  firstEmptyFieldIndex++;
77  }
78 
79  //now connect the hnodes, making a new one for every connection:
80  //always take the two nodes with the smallest probability and "combine" them, repeat, until just one node (the root) is left.
81  huffmanNode<T> * h1;
82  huffmanNode<T> * h2;
83  while (queue.size()>1){
84  h1=queue.top();
85  queue.pop();
86  h2=queue.top();
87  queue.pop();
88  htree[firstEmptyFieldIndex].probability=h1->probability+h2->probability;
89  htree[firstEmptyFieldIndex].leftChild=h1;
90  htree[firstEmptyFieldIndex].rightChild=h2;
91  queue.push(&(htree[firstEmptyFieldIndex]));
92  firstEmptyFieldIndex++;
93  }
94  //reorganize the probabilities: in every node, add the probability of one subtree ("small")
95  //to all nodes of the other subtree.
96  addProbabilityFromRightSubtreeToLeftSubtree(&htree[htree.size()-1],0);
97  moveProbabilitiesFromRightSubtreesOneLevelUp(&htree[htree.size()-1]);
98  treeIsMade=true;
99 
100  }

107  else node=node->rightChild;
108  }

127  void addProbabilityFromRightSubtreeToLeftSubtree(huffmanNode<T> * n,double add){
128  //for each node, adds the probability of the right childnode to the left childnode and every node under it.
129  //if the Tree would look like this (with the Numbers representing the probability of every node) before calling this function
130 
131  // 1.0
132  // ____||____
133  // | |
134  // 0.4 0.6
135  // _||_ _||_
136  // | | | |
137  // 0.25 0.15 0.35 0.25
138  // _||_
139  // | |
140  // 0.1 0.05
141  //then it would look like this after calling it
142  // 1.0
143  // ____||____
144  // | |
145  // 1.0 0.6
146  // _||_ _||_
147  // | | | |
148  // 1.0 0.75 0.6 0.25
149  // _||_
150  // | |
151  // 0.75 0.65
152  //now the tree could be traversed with "while (!n.isLeaf()) n=p>n.right.p?n.left:n.right"
153  //so in the function moveProbabilitiesFromRightSubtreesOneLevelUp the numbers are moved one level up to call n.p instead of n.right.p
154 
155 
156  //adds "add" to the probability, then calls itself recursively.
157  //this calculates the probabilities needed to traverse the tree quickly
158  n->probability+=add;
159  //if leftId=-1 (=> node is leaf), returns
160  if (n->isOnLastLevel){
161  return;
162  }
163 
164  addProbabilityFromRightSubtreeToLeftSubtree(n->leftChild,add+n->rightChild->probability);
165  addProbabilityFromRightSubtreeToLeftSubtree(n->rightChild,add);
166  }

168  void moveProbabilitiesFromRightSubtreesOneLevelUp(huffmanNode<T> * n){
169  //moves the Probabilities on the right subtrees one level up.
170  //if the Tree would look like this (with the Numbers representing the probability of every node) before calling this function
171  // 1.0
172  // ____||____
173  // | |
174  // 1.0 0.6
175  // _||_ _||_
176  // | | | |
177  // 1.0 0.75 0.6 0.25
178  // _||_
179  // | |
180  // 0.75 0.65
181  //then it would look like this after calling it
182  // 0.
183  // ____||____
184  // | |
185  // 0.75 0.25
186  // _||_ _||_
187  // | | | |
188  // 1.0 0.65 0.6 0.25
189  // _||_
190  // | |
191  // 0.75 0.65
192  //note, that now the probabilities on the leaf level are not needed anymore to traverse the tree; the algorithm now is "while (!n.isLeaf()) n=p>n.p?n.left:n.right"
193  if (n->isOnLastLevel){
194  n->probability-=n->leftLeaf->rate/sum_of_values;
195  }
196  else{
197  n->probability=n->rightChild->probability;
198  moveProbabilitiesFromRightSubtreesOneLevelUp(n->rightChild);
199  moveProbabilitiesFromRightSubtreesOneLevelUp(n->leftChild);
200  }
201  }
