832 vector<PolarSite*> XMultipole::ParseGdmaFile(string filename, int state) {
833 
834  int poleCount = 1;
835  double Q0_total = 0.0;
836  string units = "";
837  bool useDefaultPs = true;
838 
839  vector<PolarSite*> poles;
840  PolarSite *thisPole = NULL;
841 
842  vector<double> Qs; // <- multipole moments
843  double P1; // <- dipole polarizability
844 
845  std::string line;
846  std::ifstream intt;
847  intt.open(filename.c_str());
848 
849  if (intt.is_open() ) {
850  while ( intt.good() ) {
851 
852  std::getline(intt, line);
853  vector<string> split;
854  Tokenizer toker(line, " ");
855  toker.ToVector(split);
856 
857  if ( !split.size() ||
858  split[0] == "!" ||
859  split[0].substr(0,1) == "!" ) { continue; }
860 
861  // ! Interesting information here, e.g.
862  // ! DCV2T opt
863  // ! SP RB3LYP 6-311+G(d,p)
864  // Units bohr
865  //
866  // C -4.2414603400 -3.8124751600 0.0017575736 Rank 2
867  // -0.3853409355
868  // -0.0002321905 0.2401559510 0.6602334308
869  // -0.7220625314 0.0004894995 -0.0003833545 0.4526409813 -0.50937399
870  // P 1.75
871 
872 
873  // Units used
874  if ( split[0] == "Units") {
875  units = split[1];
876  if (units != "bohr" && units != "angstrom") {
877  throw std::runtime_error( "Unit " + units + " in file "
878  + filename + " not supported.");
879  }
880  }
881 
882  // element, position, rank limit
883  else if ( split.size() == 6 ) {
884 
885  Qs.clear();
886  P1 = -1.;
887 
888  int id = poleCount++; // <- starts from 1
889  string name = split[0];
890 
891  double BOHR2NM = 0.0529189379;
892  double ANGSTROM2NM = 0.1;
893  double x, y, z;
894 
895  if (units == "bohr") {
896  x = BOHR2NM * boost::lexical_cast<double>(split[1]);
897  y = BOHR2NM * boost::lexical_cast<double>(split[2]);
898  z = BOHR2NM * boost::lexical_cast<double>(split[3]);
899  }
900  else if (units == "angstrom") {
901  x = ANGSTROM2NM * boost::lexical_cast<double>(split[1]);
902  y = ANGSTROM2NM * boost::lexical_cast<double>(split[2]);
903  z = ANGSTROM2NM * boost::lexical_cast<double>(split[3]);
904  }
905  else {
906  throw std::runtime_error( "Unit " + units + " in file "
907  + filename + " not supported.");
908  }
909 
910  vec pos = vec(x,y,z);
911 
912  int rank = boost::lexical_cast<int>(split[5]);
913 
914  PolarSite *newPole = new PolarSite(id, name);
915  newPole->setRank(rank);
916  newPole->setPos(pos);
917  poles.push_back(newPole);
918  thisPole = newPole;
919 
920  }
921 
922  // 'P', dipole polarizability
923  else if ( split[0] == "P" && split.size() == 2 ) {
924  P1 = 1e-3 * boost::lexical_cast<double>(split[1]);
925  thisPole->setPs(P1, state);
926  useDefaultPs = false;
927  }
928
