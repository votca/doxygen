32 void EInternal::ParseEnergies() {
33 
34  std::cout << "\n... ... Site, reorg. energies from " << _energiesXML << ".\n";
35 
36  tools::Property alloc;
37  alloc.LoadFromXML(_energiesXML);
38 
39  std::string key = "topology.molecules.molecule";
40  std::vector<tools::Property *> mols = alloc.Select(key);
41  for (tools::Property *molprop : mols) {
42 
43  key = "segments.segment";
44  std::vector<tools::Property *> segs = molprop->Select(key);
45  for (tools::Property *segprop : segs) {
46 
47  std::string segName = segprop->get("name").as<std::string>();
48 
49  bool has_seg = true;
50 
51  QMStateCarrierStorage<double> U_xX_nN;
52  QMStateCarrierStorage<double> U_nX_nN;
53  QMStateCarrierStorage<double> U_xN_xX;
54  QMStateCarrierStorage<bool> has_state;
55  double eV2hrt = tools::conv::ev2hrt;
56 
57  std::vector<QMStateType> types = {QMStateType::Electron,
58  QMStateType::Hole, QMStateType::Singlet,
59  QMStateType::Triplet};
60  for (QMStateType type : types) {
61  std::string u_xX_nN = "U_xX_nN_" + type.ToString();
62  std::string u_nX_nN = "U_nX_nN_" + type.ToString();
63  std::string u_xN_xX = "U_xN_xX_" + type.ToString();
64  if (segprop->exists(u_xX_nN) && segprop->exists(u_nX_nN) &&
65  segprop->exists(u_xN_xX)) {
66  U_xX_nN.setValue(segprop->get(u_xX_nN).as<double>() * eV2hrt, type);
67  U_nX_nN.setValue(segprop->get(u_nX_nN).as<double>() * eV2hrt, type);
68  U_xN_xX.setValue(segprop->get(u_xN_xX).as<double>() * eV2hrt, type);
69  has_state.setValue(true, type);
70  }
71  }
72  _seg_has_state[segName] = has_state;
73  _seg_U_xX_nN[segName] = U_xX_nN;
74  _seg_U_nX_nN[segName] = U_nX_nN;
75  _seg_U_xN_xX[segName] = U_xN_xX;
76 
77  _has_seg[segName] = has_seg;
78  }
79  }
80 }

82 bool EInternal::Evaluate(Topology &top) {
83 
84  ParseEnergies();
85 
86  Index count = 0;
87  for (Segment &seg : top.Segments()) {
88 
89  std::string segName = seg.getType();
90 
91  if (!_has_seg.count(segName)) {
92  std::cout << std::endl
93  << "... ... WARNING: No energy information for seg [" << segName
94  << "]. Skipping... ";
95  continue;
96  }
97 
98  ++count;
99 
100  std::vector<QMStateType> types = {QMStateType::Electron, QMStateType::Hole,
101  QMStateType::Singlet,
102  QMStateType::Triplet};
103  for (QMStateType type : types) {
104 
105  if (_seg_has_state[segName].getValue(type.Type())) {
106  seg.setU_xX_nN(_seg_U_xX_nN[segName].getValue(type.Type()),
107  type.Type());
108  seg.setU_nX_nN(_seg_U_nX_nN[segName].getValue(type.Type()),
109  type.Type());
110  seg.setU_xN_xX(_seg_U_xN_xX[segName].getValue(type.Type()),
111  type.Type());
112  }
113  }
114  }
115 
116  std::cout << std::endl
117  << "... ... Read in site, reorg. energies for " << count
118  << " segments. " << std::flush;
119 
120  return true;
121 }
