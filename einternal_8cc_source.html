34 void EInternal::ParseEnergies() {
35 
36  std::cout << std::endl
37  << "... ... Site, reorg. energies from " << _energiesXML << ". "
38  << std::flush;
39 
40  tools::Property alloc;
41  alloc.LoadFromXML(_energiesXML);
42 
43  std::string key = "topology.molecules.molecule";
44  std::vector<tools::Property *> mols = alloc.Select(key);
45  for (tools::Property *molprop : mols) {
46 
47  key = "segments.segment";
48  std::vector<tools::Property *> segs = molprop->Select(key);
49  for (tools::Property *segprop : segs) {
50 
51  std::string segName = segprop->get("name").as<std::string>();
52 
53  bool has_seg = true;
54 
55  QMStateCarrierStorage<double> U_xX_nN;
56  QMStateCarrierStorage<double> U_nX_nN;
57  QMStateCarrierStorage<double> U_xN_xX;
58  QMStateCarrierStorage<bool> has_state;
59  double eV2hrt = tools::conv::ev2hrt;
60 
61  std::vector<QMStateType> types = {QMStateType::Electron,
62  QMStateType::Hole, QMStateType::Singlet,
63  QMStateType::Triplet};
64  for (QMStateType type : types) {
65  std::string u_xX_nN = "U_xX_nN_" + type.ToString();
66  std::string u_nX_nN = "U_nX_nN_" + type.ToString();
67  std::string u_xN_xX = "U_xN_xX_" + type.ToString();
68  if (segprop->exists(u_xX_nN) && segprop->exists(u_nX_nN) &&
69  segprop->exists(u_xN_xX)) {
70  U_xX_nN.setValue(segprop->get(u_xX_nN).as<double>() * eV2hrt, type);
71  U_nX_nN.setValue(segprop->get(u_nX_nN).as<double>() * eV2hrt, type);
72  U_xN_xX.setValue(segprop->get(u_xN_xX).as<double>() * eV2hrt, type);
73  has_state.setValue(true, type);
74  }
75  }
76  _seg_has_state[segName] = has_state;
77  _seg_U_xX_nN[segName] = U_xX_nN;
78  _seg_U_nX_nN[segName] = U_nX_nN;
79  _seg_U_xN_xX[segName] = U_xN_xX;
80 
81  _has_seg[segName] = has_seg;
82  }
83  }
84 }

86 bool EInternal::EvaluateFrame(Topology &top) {
87 
88  ParseEnergies();
89 
90  int count = 0;
91  for (Segment &seg : top.Segments()) {
92 
93  std::string segName = seg.getType();
94 
95  if (!_has_seg.count(segName)) {
96  std::cout << std::endl
97  << "... ... WARNING: No energy information for seg [" << segName
98  << "]. Skipping... ";
99  continue;
100  }
101 
102  ++count;
103 
104  std::vector<QMStateType> types = {QMStateType::Electron, QMStateType::Hole,
105  QMStateType::Singlet,
106  QMStateType::Triplet};
107  for (QMStateType type : types) {
108 
109  if (_seg_has_state[segName].getValue(type.Type())) {
110  seg.setU_xX_nN(_seg_U_xX_nN[segName].getValue(type.Type()),
111  type.Type());
112  seg.setU_nX_nN(_seg_U_nX_nN[segName].getValue(type.Type()),
113  type.Type());
114  seg.setU_xN_xX(_seg_U_xN_xX[segName].getValue(type.Type()),
115  type.Type());
116  }
117  }
118  }
119 
120  std::cout << std::endl
121  << "... ... Read in site, reorg. energies for " << count
122  << " segments. " << std::flush;
123 
124  return true;
125 }
