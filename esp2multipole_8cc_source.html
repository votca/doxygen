27 void Esp2multipole::Initialize(Property* options) {
28  string key = Identify();
29  _use_ecp=false;
30  _do_svd=false;
31 
32  _use_mulliken=false;
33  _use_CHELPG=false;
34  _use_bulkESP=false;
35  _use_GDMA=false;
36  _use_CHELPG_SVD=false;
37  _use_lowdin=false;
38  _use_NBO=false;
39 
40  _state = options->get(key + ".state").as<string> ();
41  _state_no = options->get(key + ".statenumber").as<int> ();
42  _spin = options->get(key + ".spin").as<string> ();
43  if ( options->exists(key+".ecp")) {
44  _use_ecp=options->get(key + ".ecp").as<bool> ();
45  }
46  if ( options->exists(key+".method")) {
47  _method = options->get(key + ".method").as<string> ();
48  if (_method=="Mulliken" || _method=="mulliken")_use_mulliken=true;
49  else if(_method=="loewdin" || _method=="Loewdin") _use_lowdin=true;
50  else if(_method=="CHELPG")_use_CHELPG=true;
51  else if(_method=="bulkESP")_use_bulkESP=true;
52  else if(_method=="GDMA") throw std::runtime_error("GDMA not implemented yet");
53  else if(_method=="CHELPG_SVD") throw std::runtime_error("CHELPG_SVD not implemented yet");
54  else if(_method=="NBO") _use_NBO=true;
55  else throw std::runtime_error("Method not recognized. Mulliken, Lowdin and CHELPG implemented");
56  }
57  else _use_CHELPG=true;
58 
59 
60 
61 
62  if (_use_CHELPG){
63  _integrationmethod = options->get(key + ".integrationmethod").as<string> ();
64  }
65 
66 
67  if (!(_integrationmethod=="numeric" || _integrationmethod=="analytic")){
68  std::runtime_error("Method not recognized. Only numeric and analytic available");
69  }
70  if ( options->exists(key+".gridsize")) {
71  _gridsize = options->get(key+".gridsize").as<string>();
72  }
73  else _gridsize="medium";
74  if ( options->exists(key+".openmp")) {
75  _openmp_threads = options->get(key+".openmp").as<int>();
76  }
77  else _openmp_threads=0;
78  if ( options->exists(key+".svd")) {
79  _do_svd = options->get(key+".svd.do_svd").as<bool>();
80  _conditionnumber = options->get(key+".svd.conditionnumber").as<double>();
81  }

88  char *votca_share = getenv("VOTCASHARE");

94 string Esp2multipole::GetIdentifier(){
95  string identifier;
96  if(_state=="transition" && _spin=="singlet" ){
97  identifier=(boost::format("n2s%i") % _state_no).str();
98  }
99  else if(_state=="transition" && _spin=="triplet" ){
100  identifier=(boost::format("n2t%i") % _state_no).str();
101  }
102  else if(_state=="excited" && _spin=="triplet" ){
103  identifier=(boost::format("t%i") % _state_no).str();
104  }
105  else if(_state=="excited" && _spin=="singlet" ){
106  identifier=(boost::format("s%i") % _state_no).str();
107  }
108  else if(_state=="excited" && _spin=="singlet" ){
109  identifier=(boost::format("s%i") % _state_no).str();
110  }
111  else if(_state=="ground" ){
112  identifier="n";
113  }
114  else{
115  throw std::runtime_error("Esp2multipole::GetIdentifier did not recognize config");
116  }
117  return identifier;
118 }

120 void Esp2multipole::WritetoFile(string _output_file, string identifier){
121  _use_mps=false;
122  _use_pdb=false;
123  string data_format = boost::filesystem::extension(_output_file);
124  //cout << data_format << endl;
125  if (data_format==".mps")_use_mps=true;
126  else if(data_format==".pdb")_use_pdb=true;
127  else throw std::runtime_error("Outputfile format not recognized. Export only to .pdb and .mps");
128  string tag="TOOL:"+Identify()+"_"+_state+"_"+_spin+boost::lexical_cast<string>(_state_no);
129  if(_use_mps){
130  QMMInterface Converter;
131  ctp::PolarSeg result=Converter.Convert(_Atomlist);
132 
133  result.WriteMPS(_output_file,tag);
134  }
135  else if(_use_pdb){
136  FILE *out;
137  Orbitals _orbitals;
138  std::vector< ctp::QMAtom* >::iterator at;
139  for (at=_Atomlist.begin();at<_Atomlist.end();++at){
140 
141  _orbitals.AddAtom(*(*at));
142  }
143  out = fopen(_output_file.c_str(), "w");
144  _orbitals.WritePDB(out, tag);
145  }
146 }

151 void Esp2multipole::Extractingcharges( Orbitals & _orbitals ){
152  int threads=1;
153 #ifdef _OPENMP
154  if ( _openmp_threads > 0 ) omp_set_num_threads(_openmp_threads);
155  threads=omp_get_max_threads();
156 #endif
157  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on "<< threads << " threads ===== " << flush;
158 
159  vector< ctp::QMAtom* > Atomlist =_orbitals.QMAtoms();
160  std::vector< ctp::QMAtom* >::iterator at;
161  for (at=Atomlist.begin();at<Atomlist.end();++at){
162  ctp::QMAtom * atom=new ctp::QMAtom(*(*at));
163  _Atomlist.push_back(atom);
164  }
165  ub::matrix<double> DMAT_tot;
166  BasisSet bs;
167  bs.LoadBasisSet(_orbitals.getDFTbasis());
168  AOBasis basis;
169  basis.AOBasisFill(&bs, _Atomlist );
170 
171 
172  ub::matrix<double> _MO_Coefficients = _orbitals.MOCoefficients();
173 
174  //basis.ReorderMOs(_orbitals.MOCoefficients(), _orbitals.getQMpackage(), "votca" );
175  basis.ReorderMOs(_MO_Coefficients, _orbitals.getQMpackage(), "xtp" );
176  bool _do_transition=false;
177  if(_state=="transition"){
178  _do_transition=true;
179  if (_spin=="singlet"){
180  //DMAT_tot=_orbitals.TransitionDensityMatrix(_orbitals.MOCoefficients() , _orbitals.BSESingletCoefficients(), _state_no-1);
181  DMAT_tot=_orbitals.TransitionDensityMatrix(_MO_Coefficients, _orbitals.BSESingletCoefficients(), _state_no-1);
182  }
183  else if (_spin=="triplet"){
184  //DMAT_tot=_orbitals.TransitionDensityMatrix(_orbitals.MOCoefficients() , _orbitals.BSETripletCoefficients(), _state_no-1);
185  DMAT_tot=_orbitals.TransitionDensityMatrix(_MO_Coefficients, _orbitals.BSETripletCoefficients(), _state_no-1);
186  }
187  else throw std::runtime_error("Spin entry not recognized");
188  }
189  else if (_state=="ground" || _state=="excited"){
190 
191 
192  //ub::matrix<double> &DMATGS=_orbitals.DensityMatrixGroundState(_orbitals.MOCoefficients());
193  ub::matrix<double> DMATGS=_orbitals.DensityMatrixGroundState(_MO_Coefficients);
194  DMAT_tot=DMATGS;
195  if ( _state_no > 0 && _state=="excited"){
196  std::vector<ub::matrix<double> > DMAT;
197  if (_spin=="singlet"){
198  //DMAT = _orbitals.DensityMatrixExcitedState( _orbitals.MOCoefficients() , _orbitals.BSESingletCoefficients(), _state_no-1);
199  DMAT = _orbitals.DensityMatrixExcitedState( _MO_Coefficients , _orbitals.BSESingletCoefficients(), _state_no-1);
200  }
201  else if (_spin=="triplet"){
202  //DMAT = _orbitals.DensityMatrixExcitedState( _orbitals.MOCoefficients() , _orbitals.BSETripletCoefficients(), _state_no-1);
203  DMAT = _orbitals.DensityMatrixExcitedState( _MO_Coefficients , _orbitals.BSETripletCoefficients(), _state_no-1);
204  }
205  else throw std::runtime_error("Spin entry not recognized");
206  DMAT_tot=DMAT_tot-DMAT[0]+DMAT[1];
207  }
208  // Ground state + hole_contribution + electron contribution
209  }
210  else throw std::runtime_error("State entry not recognized");
211 
212 
213  if (_use_mulliken) {
214  Mulliken mulliken;
215  mulliken.setUseECPs(_use_ecp);
216  mulliken.EvaluateMulliken(_Atomlist, DMAT_tot, basis, bs, _do_transition);
217 
218  }
219  else if (_use_lowdin) {
220  Lowdin lowdin;
221  lowdin.setUseECPs(_use_ecp);
222  lowdin.EvaluateLowdin(_Atomlist, DMAT_tot, basis, bs, _do_transition);
223  }
224  else if (_use_CHELPG){
225  Espfit esp=Espfit(_log);
226  esp.setUseECPs(_use_ecp);
227  if(_do_svd){
228  esp.setUseSVD(_do_svd,_conditionnumber);
229  }
230  if (_integrationmethod=="numeric") {
231  esp.Fit2Density(_Atomlist, DMAT_tot, basis,bs,_gridsize);
232  }
233  else if (_integrationmethod=="analytic") esp.Fit2Density_analytic(_Atomlist,DMAT_tot,basis);
234  }
235  else if (_use_bulkESP){
236  Espfit esp=Espfit(_log);
237  esp.setUseECPs(_use_ecp);
238  if(_do_svd){
239  esp.setUseSVD(_do_svd,_conditionnumber);
240  }
241  if (_integrationmethod=="numeric") {
242  esp.Fit2Density(_Atomlist, DMAT_tot, basis,bs,_gridsize);
243  }
244  else if (_integrationmethod=="analytic") esp.Fit2Density_analytic(_Atomlist,DMAT_tot,basis);
245  }
246  else if(_use_NBO){
247 
248  std::cout<<"WARNING: NBO analysis isn't fully implemented yet."<<std::endl;
249  //CTP_LOG(logDEBUG, _log) << "Initializing NBO" << flush;
250  NBO nbo=NBO(_log);
251  nbo.setUseECPs(_use_ecp);
252  //nbo.LoadMatrices("", "");
253  nbo.EvaluateNBO(_Atomlist, DMAT_tot, basis,bs);
254  }
255  else{
256  std::cout<<"Method not recognized."<<std::endl;
257  }
258 }
