33 void Esp2multipole::Initialize(tools::Property& options) {
34  std::string key = Identify();
35  _do_svd = false;
36 
37  _use_mulliken = false;
38  _use_CHELPG = false;
39  _use_lowdin = false;
40 
41  std::string statestring = options.get(key + ".state").as<std::string>();
42  _state.FromString(statestring);
43 
44  _method = options.get(key + ".method").as<std::string>();
45 
46  if (_method == "mulliken") {
47  _use_mulliken = true;
48  } else if (_method == "loewdin") {
49  _use_lowdin = true;
50  } else if (_method == "CHELPG") {
51  _use_CHELPG = true;
52  }
53 
54  if (options.exists(key + ".constraints")) {
55  if (options.exists(key + ".constraints.regions")) {
56  std::vector<tools::Property*> prop_region =
57  options.Select(key + ".constraints.regions.region");
58  Index index = 0;
59  for (tools::Property* prop : prop_region) {
60  std::string indices = prop->get("indices").as<std::string>();
61  QMFragment<double> reg = QMFragment<double>(index, indices);
62  index++;
63  reg.value() = prop->get("charge").as<double>();
64  _regionconstraint.push_back(reg);
65  XTP_LOG(Log::error, _log) << "Fit constrained by Region" << flush;
66  XTP_LOG(Log::error, _log) << reg;
67  }
68  }
69  if (options.exists(key + ".constraints.pairs")) {
70  std::vector<tools::Property*> prop_pair =
71  options.Select(key + ".constraints.pairs.pair");
72  for (tools::Property* prop : prop_pair) {
73  std::string pairstring = prop->as<std::string>();
74  tools::Tokenizer tok(pairstring, "\n\t ,");
75  std::vector<Index> pairvec;
76  tok.ConvertToVector<Index>(pairvec);
77  std::pair<Index, Index> pair;
78  pair.first = pairvec[0];
79  pair.second = pairvec[1];
80  _pairconstraint.push_back(pair);
81  XTP_LOG(Log::error, _log)
82  << "Charges " << pair.first << " " << pair.second
83  << " constrained to be equal." << flush;
84  }
85  }
86  }
87 
88  _gridsize = options.ifExistsReturnElseReturnDefault<std::string>(
89  key + ".gridsize", "medium");
90 
91  if (options.exists(key + ".svd")) {
92  _do_svd = options.get(key + ".svd.do_svd").as<bool>();
93  _conditionnumber = options.get(key + ".svd.conditionnumber").as<double>();
94  }
95 
96  return;
97 }

119 StaticSegment Esp2multipole::Extractingcharges(const Orbitals& orbitals) const {
120  XTP_LOG(Log::error, _log) << "===== Running on " << OPENMP::getMaxThreads()
121  << " threads ===== " << flush;
122  StaticSegment result("result", 0);
123  if (_use_mulliken) {
124  Mulliken mulliken;
125  result = mulliken.CalcChargeperAtom(orbitals, _state);
126  } else if (_use_lowdin) {
127  Lowdin lowdin;
128  result = lowdin.CalcChargeperAtom(orbitals, _state);
129  } else if (_use_CHELPG) {
130  Espfit esp = Espfit(_log);
131  if (_pairconstraint.size() > 0) {
132  esp.setPairConstraint(_pairconstraint);
133  }
134  if (_regionconstraint.size() > 0) {
135  esp.setRegionConstraint(_regionconstraint);
136  }
137 
138  if (_do_svd) {
139  esp.setUseSVD(_conditionnumber);
140  }
141  result = esp.Fit2Density(orbitals, _state, _gridsize);
142  }
143 
144  PrintDipoles(orbitals, result);
145  return result;
146 }
