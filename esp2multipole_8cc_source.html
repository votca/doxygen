28  void Esp2multipole::Initialize(Property* options) {
29  string key = Identify();
30  _use_ecp = false;
31  _do_svd = false;
32 
33  _use_mulliken = false;
34  _use_CHELPG = false;
35  _use_bulkESP = false;
36  _use_GDMA = false;
37  _use_CHELPG_SVD = false;
38  _use_lowdin = false;
39  _use_NBO = false;
40 
41  _state = options->get(key + ".state").as<string> ();
42  _state_no = options->get(key + ".statenumber").as<int> ();
43  _spin = options->get(key + ".spin").as<string> ();
44  if (options->exists(key + ".ecp")) {
45  _use_ecp = options->get(key + ".ecp").as<bool> ();
46  }
47  if (options->exists(key + ".method")) {
48  _method = options->get(key + ".method").as<string> ();
49  if (_method == "Mulliken" || _method == "mulliken")_use_mulliken = true;
50  else if (_method == "loewdin" || _method == "Loewdin") _use_lowdin = true;
51  else if (_method == "CHELPG")_use_CHELPG = true;
52  else if (_method == "GDMA") throw std::runtime_error("GDMA not implemented yet");
53  else if (_method == "CHELPG_SVD") throw std::runtime_error("CHELPG_SVD not implemented yet");
54  else if (_method == "NBO") _use_NBO = true;
55  else throw std::runtime_error("Method not recognized. Mulliken, Lowdin and CHELPG implemented");
56  } else _use_CHELPG = true;
57 
58  if (_use_CHELPG) {
59  _integrationmethod = options->get(key + ".integrationmethod").as<string> ();
60  }
61 
62  if (options->exists(key + ".constraints")) {
63  if (options->exists(key + ".constraints.regions")) {
64  std::list<Property*> prop_region = options->Select(key + ".constraints.regions.region");
65  for (std::list<Property*> ::iterator it = prop_region.begin(); it != prop_region.end(); ++it) {
66  std::string indices=(*it)->get("indices").as<std::string>();
67  tools::Tokenizer tok(indices,"\n\t ,");
68  Espfit::region reg;
69  tok.ConvertToVector<int>(reg.atomindices);
70  reg.charge=(*it)->get("charge").as<double>();
71  _regionconstraint.push_back(reg);
72  CTP_LOG(ctp::logDEBUG, *_log) << "Fit constrained by SUM(";
73  for(int i:reg.atomindices){
74  CTP_LOG(ctp::logDEBUG, *_log)<<i<<" ";
75  }
76  CTP_LOG(ctp::logDEBUG, *_log)<<")="<<reg.charge<< flush;
77  }
78  }
79  if (options->exists(key + ".constraints.pairs")) {
80  std::list<Property*> prop_pair = options->Select(key + ".constraints.pairs.pair");
81  for (std::list<Property*> ::iterator it = prop_pair.begin(); it != prop_pair.end(); ++it) {
82  std::string pairstring=(*it)->as<std::string>();
83  tools::Tokenizer tok(pairstring,"\n\t ,");
84  std::vector<int> pairvec;
85  tok.ConvertToVector<int>(pairvec);
86  std::pair<int,int> pair;
87  pair.first=pairvec[0];
88  pair.second=pairvec[1];
89  _pairconstraint.push_back(pair);
90  CTP_LOG(ctp::logDEBUG, *_log) << "Charge "<<pair.first<<" "<<pair.second<<" constrained to be equal."<<flush;
91  }
92  }
93  }
94 
95 
96  if (!(_integrationmethod == "numeric" || _integrationmethod == "analytic")) {
97  std::runtime_error("Method not recognized. Only numeric and analytic available");
98  }
99  if (options->exists(key + ".gridsize")) {
100  _gridsize = options->get(key + ".gridsize").as<string>();
101  } else _gridsize = "medium";
102  if (options->exists(key + ".openmp")) {
103  _openmp_threads = options->get(key + ".openmp").as<int>();
104  } else _openmp_threads = 0;
105  if (options->exists(key + ".svd")) {
106  _do_svd = options->get(key + ".svd.do_svd").as<bool>();
107  _conditionnumber = options->get(key + ".svd.conditionnumber").as<double>();
108  }
109 
110  // get the path to the shared folders with xml files
111  char *votca_share = getenv("VOTCASHARE");
112  if (votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
113  return;
114  }

116  string Esp2multipole::GetIdentifier() {
117  string identifier;
118  if (_state == "transition" && _spin == "singlet") {
119  identifier = (boost::format("n2s%i") % _state_no).str();
120  }
121  else if (_state == "transition" && _spin == "triplet") {
122  identifier = (boost::format("n2t%i") % _state_no).str();
123  }
124  else if (_state == "excited" && _spin == "triplet") {
125  identifier = (boost::format("t%i") % _state_no).str();
126  }
127  else if (_state == "excited" && _spin == "singlet") {
128  identifier = (boost::format("s%i") % _state_no).str();
129  }
130  else if (_state == "excited" && _spin == "singlet") {
131  identifier = (boost::format("s%i") % _state_no).str();
132  }
133  else if (_state == "ground") {
134  identifier = "n";
135  }
136  else {
137  throw std::runtime_error("Esp2multipole::GetIdentifier did not recognize config");
138  }
139  return identifier;
140  }

157  void Esp2multipole::Extractingcharges(Orbitals & _orbitals) {
158  int threads = 1;
159 #ifdef _OPENMP
160  if (_openmp_threads > 0) omp_set_num_threads(_openmp_threads);
161  threads = omp_get_max_threads();
162 #endif
163  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on " << threads << " threads ===== " << flush;
164 
165  _Atomlist = _orbitals.QMAtoms();
166  Eigen::MatrixXd DMAT_tot;
167  BasisSet bs;
168  bs.LoadBasisSet(_orbitals.getDFTbasis());
169  AOBasis basis;
170  basis.AOBasisFill(&bs, _Atomlist);
171 
172  bool _do_transition = false;
173  if (_state == "transition") {
174 
175  _do_transition = true;
176  DMAT_tot = _orbitals.TransitionDensityMatrix(_spin, _state_no - 1);
177 
178  } else if (_state == "ground" || _state == "excited") {
179  Eigen::MatrixXd DMATGS = _orbitals.DensityMatrixGroundState();
180  DMAT_tot = DMATGS;
181  if (_state_no > 0 && _state == "excited") {
182  std::vector<Eigen::MatrixXd > DMAT = _orbitals.DensityMatrixExcitedState(_spin, _state_no - 1);
183  DMAT_tot = DMAT_tot - DMAT[0] + DMAT[1];
184  }
185  // Ground state + hole_contribution + electron contribution
186  } else throw std::runtime_error("State entry not recognized");
187 
188  if (_use_mulliken) {
189  Mulliken mulliken;
190  mulliken.EvaluateMulliken(_Atomlist, DMAT_tot, basis, _do_transition);
191  }
192  else if (_use_lowdin) {
193  Lowdin lowdin;
194  lowdin.EvaluateLowdin(_Atomlist, DMAT_tot, basis, _do_transition);
195  } else if (_use_CHELPG) {
196  Espfit esp = Espfit(_log);
197  if(_pairconstraint.size()>0){
198  esp.setPairConstraint(_pairconstraint);
199  }
200  if(_regionconstraint.size()>0){
201  esp.setRegionConstraint(_regionconstraint);
202  }
203 
204  if (_do_svd) {
205  esp.setUseSVD(_conditionnumber);
206  }
207  if (_integrationmethod == "numeric") {
208  esp.Fit2Density(_Atomlist, DMAT_tot, basis, _gridsize);
209  } else if (_integrationmethod == "analytic") esp.Fit2Density_analytic(_Atomlist, DMAT_tot, basis);
210  } else if (_use_NBO) {
211  std::cout << "WARNING: NBO analysis isn't fully implemented yet." << std::endl;
212  //CTP_LOG(logDEBUG, _log) << "Initializing NBO" << flush;
213  NBO nbo = NBO(_log);
214  nbo.EvaluateNBO(_Atomlist, DMAT_tot, basis, bs);
215  } else {
216  std::cout << "Method not recognized." << std::endl;
217  }
218  }
219 
220  }
