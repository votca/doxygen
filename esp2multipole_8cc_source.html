27 void Esp2multipole::Initialize(Property* options) {
28  string key = Identify();
29  _use_ecp=false;
30  _do_svd=false;
31 
32  _use_mulliken=false;
33  _use_CHELPG=false;
34  _use_bulkESP=false;
35  _use_GDMA=false;
36  _use_CHELPG_SVD=false;
37  _use_lowdin=false;
38  _use_NBO=false;
39 
40  _state = options->get(key + ".state").as<string> ();
41  _state_no = options->get(key + ".statenumber").as<int> ();
42  _spin = options->get(key + ".spin").as<string> ();
43  if ( options->exists(key+".ecp")) {
44  _use_ecp=options->get(key + ".ecp").as<bool> ();
45  }
46  if ( options->exists(key+".method")) {
47  _method = options->get(key + ".method").as<string> ();
48  if (_method=="Mulliken" || _method=="mulliken")_use_mulliken=true;
49  else if(_method=="loewdin" || _method=="Loewdin") _use_lowdin=true;
50  else if(_method=="CHELPG")_use_CHELPG=true;
51  else if(_method=="GDMA") throw std::runtime_error("GDMA not implemented yet");
52  else if(_method=="CHELPG_SVD") throw std::runtime_error("CHELPG_SVD not implemented yet");
53  else if(_method=="NBO") _use_NBO=true;
54  else throw std::runtime_error("Method not recognized. Mulliken, Lowdin and CHELPG implemented");
55  }
56  else _use_CHELPG=true;
57 
58  if (_use_CHELPG){
59  _integrationmethod = options->get(key + ".integrationmethod").as<string> ();
60  }
61 
62 
63  if (!(_integrationmethod=="numeric" || _integrationmethod=="analytic")){
64  std::runtime_error("Method not recognized. Only numeric and analytic available");
65  }
66  if ( options->exists(key+".gridsize")) {
67  _gridsize = options->get(key+".gridsize").as<string>();
68  }
69  else _gridsize="medium";
70  if ( options->exists(key+".openmp")) {
71  _openmp_threads = options->get(key+".openmp").as<int>();
72  }
73  else _openmp_threads=0;
74  if ( options->exists(key+".svd")) {
75  _do_svd = options->get(key+".svd.do_svd").as<bool>();
76  _conditionnumber = options->get(key+".svd.conditionnumber").as<double>();
77  }
78 
79  // get the path to the shared folders with xml files
80  char *votca_share = getenv("VOTCASHARE");
81  if(votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
82  return;
83 }

85 string Esp2multipole::GetIdentifier(){
86  string identifier;
87  if(_state=="transition" && _spin=="singlet" ){
88  identifier=(boost::format("n2s%i") % _state_no).str();
89  }
90  else if(_state=="transition" && _spin=="triplet" ){
91  identifier=(boost::format("n2t%i") % _state_no).str();
92  }
93  else if(_state=="excited" && _spin=="triplet" ){
94  identifier=(boost::format("t%i") % _state_no).str();
95  }
96  else if(_state=="excited" && _spin=="singlet" ){
97  identifier=(boost::format("s%i") % _state_no).str();
98  }
99  else if(_state=="excited" && _spin=="singlet" ){
100  identifier=(boost::format("s%i") % _state_no).str();
101  }
102  else if(_state=="ground" ){
103  identifier="n";
104  }
105  else{
106  throw std::runtime_error("Esp2multipole::GetIdentifier did not recognize config");
107  }
108  return identifier;
109 }

127 void Esp2multipole::Extractingcharges( Orbitals & _orbitals ){
128  int threads=1;
129 #ifdef _OPENMP
130  if ( _openmp_threads > 0 ) omp_set_num_threads(_openmp_threads);
131  threads=omp_get_max_threads();
132 #endif
133  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on "<< threads << " threads ===== " << flush;
134 
135  _Atomlist =_orbitals.QMAtoms();
136 
137  ub::matrix<double> DMAT_tot;
138  BasisSet bs;
139  bs.LoadBasisSet(_orbitals.getDFTbasis());
140  AOBasis basis;
141  basis.AOBasisFill(&bs, _Atomlist );
142 
143 
144  bool _do_transition=false;
145  if(_state=="transition"){
146  _do_transition=true;
147  DMAT_tot=_orbitals.TransitionDensityMatrix(_spin, _state_no-1);
148 
149  }
150  else if (_state=="ground" || _state=="excited"){
151 
152 
153  //ub::matrix<double> &DMATGS=_orbitals.DensityMatrixGroundState(_orbitals.MOCoefficients());
154  ub::matrix<double> DMATGS=_orbitals.DensityMatrixGroundState();
155  DMAT_tot=DMATGS;
156  if ( _state_no > 0 && _state=="excited"){
157  std::vector<ub::matrix<double> > DMAT;
158  DMAT = _orbitals.DensityMatrixExcitedState( _spin , _state_no-1);
159  DMAT_tot=DMAT_tot-DMAT[0]+DMAT[1];
160  }
161  // Ground state + hole_contribution + electron contribution
162  }
163  else throw std::runtime_error("State entry not recognized");
164 
165 
166  if (_use_mulliken) {
167  Mulliken mulliken;
168  mulliken.EvaluateMulliken(_Atomlist, DMAT_tot, basis, _do_transition);
169 
170  }
171  else if (_use_lowdin) {
172  Lowdin lowdin;
173  lowdin.EvaluateLowdin(_Atomlist, DMAT_tot, basis, _do_transition);
174  }
175  else if (_use_CHELPG){
176  Espfit esp=Espfit(_log);
177  if(_do_svd){
178  esp.setUseSVD(_do_svd,_conditionnumber);
179  }
180  if (_integrationmethod=="numeric") {
181  esp.Fit2Density(_Atomlist, DMAT_tot, basis,_gridsize);
182  }
183  else if (_integrationmethod=="analytic") esp.Fit2Density_analytic(_Atomlist,DMAT_tot,basis);
184  }
185  else if(_use_NBO){
186 
187  std::cout<<"WARNING: NBO analysis isn't fully implemented yet."<<std::endl;
188  //CTP_LOG(logDEBUG, _log) << "Initializing NBO" << flush;
189  NBO nbo=NBO(_log);
190  nbo.EvaluateNBO(_Atomlist, DMAT_tot, basis,bs);
191  }
192  else{
193  std::cout<<"Method not recognized."<<std::endl;
194  }
195 }
196 
197 }}
