30  void Esp2multipole::Initialize(tools::Property& options) {
31  std::string key = Identify();
32  _do_svd = false;
33 
34  _use_mulliken = false;
35  _use_CHELPG = false;
36  _use_lowdin = false;
37 
38  std::string statestring = options.ifExistsReturnElseThrowRuntimeError<std::string>(key + ".state");
39  _state.FromString(statestring);
40 
41  std::vector<std::string> choices={"mulliken","loewdin","CHELPG"};
42  _method=options.ifExistsAndinListReturnElseThrowRuntimeError(key+".method",choices);
43 
44  if (_method == "mulliken")_use_mulliken = true;
45  else if (_method == "loewdin" ) _use_lowdin = true;
46  else if (_method == "CHELPG")_use_CHELPG = true;
47 
48 
49  if (_use_CHELPG) {
50  _integrationmethod = options.ifExistsReturnElseReturnDefault<std::string>(key + ".integrationmethod","numeric");
51  }
52  if (!(_integrationmethod == "numeric" || _integrationmethod == "analytic")) {
53  std::runtime_error("Method not recognized. Only numeric and analytic available");
54  }
55 
56  if (options.exists(key + ".constraints")) {
57  if (options.exists(key + ".constraints.regions")) {
58  std::list<tools::Property*> prop_region = options.Select(key + ".constraints.regions.region");
59  for (tools::Property* prop:prop_region) {
60  std::string indices=prop->get("indices").as<std::string>();
61  tools::Tokenizer tok(indices,"\n\t ,");
62  Espfit::region reg;
63  tok.ConvertToVector<int>(reg.atomindices);
64  reg.charge=prop->get("charge").as<double>();
65  _regionconstraint.push_back(reg);
66  CTP_LOG(ctp::logDEBUG, *_log) << "Fit constrained by SUM(";
67  for(int i:reg.atomindices){
68  CTP_LOG(ctp::logDEBUG, *_log)<<i<<" ";
69  }
70  CTP_LOG(ctp::logDEBUG, *_log)<<")="<<reg.charge<< flush;
71  }
72  }
73  if (options.exists(key + ".constraints.pairs")) {
74  std::list<tools::Property*> prop_pair = options.Select(key + ".constraints.pairs.pair");
75  for (tools::Property* prop:prop_pair) {
76  std::string pairstring=prop->as<std::string>();
77  tools::Tokenizer tok(pairstring,"\n\t ,");
78  std::vector<int> pairvec;
79  tok.ConvertToVector<int>(pairvec);
80  std::pair<int,int> pair;
81  pair.first=pairvec[0];
82  pair.second=pairvec[1];
83  _pairconstraint.push_back(pair);
84  CTP_LOG(ctp::logDEBUG, *_log) << "Charge "<<pair.first<<" "<<pair.second<<" constrained to be equal."<<flush;
85  }
86  }
87  }
88 
89  _gridsize =options.ifExistsReturnElseReturnDefault<std::string>(key + ".gridsize","medium");
90  _openmp_threads = options.ifExistsReturnElseReturnDefault<int>(key + ".openmp",1);
91 
92  if (options.exists(key + ".svd")) {
93  _do_svd = options.get(key + ".svd.do_svd").as<bool>();
94  _conditionnumber = options.get(key + ".svd.conditionnumber").as<double>();
95  }
96 
97  // get the path to the shared folders with xml files
98  char *votca_share = getenv("VOTCASHARE");
99  if (votca_share == NULL) throw std::runtime_error("VOTCASHARE not set, cannot open help files.");
100  return;
101  }

120  void Esp2multipole::PrintDipoles(Orbitals& orbitals){
121  Eigen::Vector3d CoM =orbitals.CalcCoM();
122 
123  Eigen::Vector3d classical_dip = Eigen::Vector3d::Zero();
124  for (QMAtom* atom : _atomlist) {
125  classical_dip += (atom->getPos().toEigen() - CoM) * atom->getPartialcharge();
126  }
127  CTP_LOG(ctp::logDEBUG, *_log) << "El Dipole from fitted charges [e*bohr]:\n\t\t"
128  << boost::format(" dx = %1$+1.4f dy = %2$+1.4f dz = %3$+1.4f |d|^2 = %4$+1.4f")
129  % classical_dip[0] % classical_dip[1] % classical_dip[2] % classical_dip.squaredNorm()<< flush;
130  Eigen::Vector3d qm_dip=orbitals.CalcElDipole(_state);
131  CTP_LOG(ctp::logDEBUG, *_log) << "El Dipole from exact qm density [e*bohr]:\n\t\t"
132  << boost::format(" dx = %1$+1.4f dy = %2$+1.4f dz = %3$+1.4f |d|^2 = %4$+1.4f")
133  % qm_dip[0] % qm_dip[1] % qm_dip[2] % qm_dip.squaredNorm()<< flush;
134  }

136  void Esp2multipole::Extractingcharges(Orbitals & orbitals) {
137  int threads = 1;
138 #ifdef _OPENMP
139  if (_openmp_threads > 0) omp_set_num_threads(_openmp_threads);
140  threads = omp_get_max_threads();
141 #endif
142  CTP_LOG(ctp::logDEBUG, *_log) << "===== Running on " << threads << " threads ===== " << flush;
143 
144  _atomlist = orbitals.QMAtoms();
145  BasisSet bs;
146  bs.LoadBasisSet(orbitals.getDFTbasisName());
147  AOBasis basis;
148  basis.AOBasisFill(bs, _atomlist);
149  Eigen::MatrixXd DMAT=orbitals.DensityMatrixFull(_state);
150 
151 
152  if (_use_mulliken) {
153  Mulliken mulliken;
154  mulliken.EvaluateMulliken(_atomlist, DMAT, basis, _state.isTransition());
155  }
156  else if (_use_lowdin) {
157  Lowdin lowdin;
158  lowdin.EvaluateLowdin(_atomlist, DMAT, basis, _state.isTransition());
159  } else if (_use_CHELPG) {
160  Espfit esp = Espfit(_log);
161  if(_pairconstraint.size()>0){
162  esp.setPairConstraint(_pairconstraint);
163  }
164  if(_regionconstraint.size()>0){
165  esp.setRegionConstraint(_regionconstraint);
166  }
167 
168  if (_do_svd) {
169  esp.setUseSVD(_conditionnumber);
170  }
171  if (_integrationmethod == "numeric") {
172  esp.Fit2Density(_atomlist, DMAT, basis, _gridsize);
173  } else if (_integrationmethod == "analytic") esp.Fit2Density_analytic(_atomlist, DMAT, basis);
174  }
175 
176  PrintDipoles(orbitals);
177 
178  }
179 
180  }
