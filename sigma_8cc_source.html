44  void Sigma::X_diag(const TCMatrix_gwbse& Mmn){
45  int gwsize = Mmn.getAuxDimension(); // size of the GW basis
46  _sigma_x=Eigen::MatrixXd::Zero(_qptotal,_qptotal);
47  #pragma omp parallel for
48  for (int gw_level = 0; gw_level < _qptotal; gw_level++) {
49  const MatrixXfd & Mmn1 = Mmn[ gw_level + _qpmin ];
50  double sigma_x=-Mmn1.block(0,0,_homo+1,gwsize).cwiseAbs2().sum();
51  _sigma_x(gw_level, gw_level) = (1.0 - _ScaHFX) * sigma_x;
52  }

55  void Sigma::C_diag(const TCMatrix_gwbse& Mmn, const PPM& ppm, const Eigen::VectorXd& qp_old){
56  int levelsum = Mmn.get_ntot(); // total number of bands
57  int gwsize = Mmn.getAuxDimension(); // size of the GW basis
58  int lumo=_homo+1;
59  // loop over all GW levels
60 #pragma omp parallel for
61  for (int gw_level = 0; gw_level < _qptotal; gw_level++) {
62  const double qpmin = qp_old(gw_level + _qpmin);
63  double sigma_c = 0.0;
64  // loop over all functions in GW basis
65  for (int i_gw = 0; i_gw < gwsize; i_gw++) {
66  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc PPM_construct_parameters
67  if (ppm.getPpm_weight()(i_gw) < 1.e-9) {
68  continue;
69  }
70 
71 #if (GWBSE_DOUBLE)
72  const Eigen::VectorXd Mmn2=Mmn[ gw_level + _qpmin ].col(i_gw).cwiseAbs2();
73 #else
74  const Eigen::VectorXd Mmn2=Mmn[ gw_level + _qpmin ].col(i_gw).cwiseAbs2().cast<double>();
75 #endif
76  const double ppm_freq = ppm.getPpm_freq()(i_gw);
77  const double fac = 0.5*ppm.getPpm_weight()(i_gw) * ppm_freq;
78  Eigen::ArrayXd denom=qpmin-qp_old.array();
79  denom.segment(0,lumo)+= ppm.getPpm_freq()(i_gw);
80  denom.segment(lumo,levelsum-lumo)-= ppm.getPpm_freq()(i_gw);
81  Stabilize(denom);
82  sigma_c += fac*(denom.inverse()*Mmn2.array()).sum();
83 
84  }// GW functions
85  _sigma_c(gw_level, gw_level) = sigma_c;
86  // update _gwa_energies
87  _gwa_energies(gw_level + _qpmin) = (*_dftenergies)(gw_level + _qpmin) + sigma_c + _sigma_x(gw_level, gw_level) - (*_vxc)(gw_level, gw_level);
88  }// all bands
89  }

91  void Sigma::CalcdiagElements(const TCMatrix_gwbse& Mmn, const PPM & ppm) {
92  X_diag(Mmn);
93  if(_gwa_energies.size()<1){
94  throw std::runtime_error("Sigma gwa_energies not set!");
95  }
96  _sigma_c=Eigen::MatrixXd::Zero(_qptotal,_qptotal);
97 
98  // initial _qp_energies are dft energies
99  Eigen::VectorXd qp_old = _gwa_energies;
100  // only diagonal elements except for in final iteration
101  for (int g_iter = 0; g_iter < _g_sc_max_iterations; g_iter++) {
102 
103  C_diag(Mmn, ppm, qp_old);
104  Eigen::VectorXd diff = qp_old - _gwa_energies;
105  bool energies_converged = true;
106 
107  int state = 0;
108  double diff_max = diff.cwiseAbs().maxCoeff(&state);
109  if (diff_max > _g_sc_limit) {
110  energies_converged = false;
111  }
112 
113  if (tools::globals::verbose) {
114  double _DFTgap = (*_dftenergies)(_homo + 1) - (*_dftenergies)(_homo);
115  double _QPgap = _gwa_energies(_homo + 1) - _gwa_energies(_homo);
116  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " G_Iteration: " << g_iter + 1 << " shift=" << _QPgap - _DFTgap << " E_diff max=" << diff_max << " StateNo:" << state << std::flush;
117  }
118  double alpha = 0.0;
119  _gwa_energies = (1 - alpha) * _gwa_energies + alpha*qp_old;
120 
121  if (energies_converged) {
122  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " Converged after " << g_iter + 1 << " G iterations." << std::flush;
123  break;
124  } else if (g_iter == _g_sc_max_iterations - 1) {
125  CTP_LOG(ctp::logDEBUG, *_log) << ctp::TimeStamp() << " G-self-consistency cycle not converged after " << _g_sc_max_iterations << " iterations." << std::flush;
126  break;
127 
128  } else {
129  qp_old = _gwa_energies;
130  }
131  } // iterations
132  return;
133  }

135  void Sigma::X_offdiag(const TCMatrix_gwbse& Mmn){
136  int gwsize = Mmn.getAuxDimension();
137  #pragma omp parallel for schedule(dynamic)
138  for (int gw_level1 = 0; gw_level1 < _qptotal; gw_level1++) {
139  const MatrixXfd& Mmn1 = Mmn[ gw_level1 + _qpmin ];
140  for (int gw_level2 = gw_level1+1; gw_level2 < _qptotal; gw_level2++) {
141  const MatrixXfd & Mmn2 = Mmn[ gw_level2 + _qpmin ];
142  double sigma_x=-Mmn1.block(0,0,_homo+1,gwsize).cwiseProduct(Mmn2.block(0,0,_homo+1,gwsize)).sum();
143  _sigma_x(gw_level1, gw_level2) = (1.0 - _ScaHFX) * sigma_x;
144  _sigma_x(gw_level2, gw_level1) = (1.0 - _ScaHFX) * sigma_x;
145  }
146  }

150  void Sigma::Stabilize(Eigen::ArrayXd& denom){
151  const double fourpi = 4*boost::math::constants::pi<double>();
152  for(int i=0;i<denom.size();++i){
153  if (std::abs(denom[i]) < 0.25) {
154  denom[i]=denom[i]/(0.5 * (1.0 - std::cos(fourpi * denom[i])));
155  }
156  }

159  void Sigma::C_offdiag(const TCMatrix_gwbse& Mmn, const PPM& ppm){
160 
161  #pragma omp parallel
162  {
163  int lumo=_homo+1;
164  const int levelsum = Mmn.get_ntot(); // total number of bands
165  const int gwsize = Mmn.getAuxDimension(); // size of the GW basis
166  const Eigen::VectorXd ppm_weight=ppm.getPpm_weight();
167  const Eigen::VectorXd ppm_freqs=ppm.getPpm_freq();
168  const Eigen::VectorXd fac=0.25*ppm_weight.cwiseProduct(ppm_freqs);
169 
170  const Eigen::VectorXd gwa_energies=_gwa_energies;
171  #pragma omp for schedule(dynamic)
172  for (int gw_level1 = 0; gw_level1 < _qptotal; gw_level1++) {
173  const MatrixXfd& Mmn1=Mmn[ gw_level1 + _qpmin ];
174  const double qpmin1 = gwa_energies(gw_level1 + _qpmin);
175  for (int gw_level2 = gw_level1+1; gw_level2 < _qptotal; gw_level2++) {
176  const MatrixXfd& Mmn2=Mmn[ gw_level2 + _qpmin ];
177  const double qpmin2 = gwa_energies(gw_level2 + _qpmin);
178  double sigma_c=0;
179  for (int i_gw = 0; i_gw < gwsize; i_gw++) {
180  // the ppm_weights smaller 1.e-5 are set to zero in rpa.cc PPM_construct_parameters
181  if (ppm_weight(i_gw) < 1.e-9) {
182  continue;
183  }
184  #if (GWBSE_DOUBLE)
185  const Eigen::VectorXd Mmn1xMmn2=Mmn1.col(i_gw).cwiseProduct(Mmn2.col(i_gw));
186 #else
187  const Eigen::VectorXd Mmn1xMmn2=(Mmn1.col(i_gw).cwiseProduct(Mmn2.col(i_gw))).cast<double>();
188 #endif
189 
190  Eigen::ArrayXd denom1=gwa_energies;
191 
192  denom1.segment(0,lumo)-=ppm_freqs(i_gw);
193  denom1.segment(lumo,levelsum-lumo)+=ppm_freqs(i_gw);
194 
195  Eigen::ArrayXd denom2=(qpmin2-denom1);
196  Stabilize(denom2);
197  denom1=(qpmin1-denom1);
198  Stabilize(denom1);
199  sigma_c += fac(i_gw)*((denom1.inverse()+denom2.inverse())*Mmn1xMmn2.array()).sum();
200  }
201  _sigma_c(gw_level1, gw_level2) = sigma_c;
202  _sigma_c(gw_level2, gw_level1) = sigma_c;
203  }// GW row
204  }//GW col
205  }
206 
207  return;
208  }
