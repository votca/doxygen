381 bool NWChem::ParseOrbitalsFile( Orbitals* _orbitals )
382 {
383  std::map <int, std::vector<double> > _coefficients;
384  std::map <int, double> _energies;
385  std::map <int, double> _occ;
386 
387  std::string _line;
388  unsigned _levels = 0;
389  //unsigned _level;
390  unsigned _basis_size = 0;
391  int _number_of_electrons = 0;
392 
393  /* maybe we DO need to convert from fortran binary to ASCII first to avoid
394  compiler-dependent issues */
395  string _orb_file_name_bin = _run_dir + "/" + _orb_file_name;
396  string _command;
397  _command = "cd " + _run_dir + "; mov2asc 10000 system.movecs system.mos > convert.log";
398  int i = system ( _command.c_str() );
399  if ( i == 0 ) {
400  LOG(logDEBUG, *_pLog) << "Converted MO file from binary to ascii"<< flush;
401  } else {
402  LOG( logERROR, *_pLog ) << "Conversion of binary MO file to ascii failed. " << flush;
403  return false;
404  }
405 
406  // opening the ascii MO file
407  string _orb_file_name_full = _run_dir + "/" + "system.mos" ;
408  std::ifstream _input_file( _orb_file_name_full.c_str() );
409 
410  if (_input_file.fail()) {
411  LOG( logERROR, *_pLog ) << "File " << _orb_file_name << " with molecular orbitals is not found " << flush;
412  return false;
413  } else {
414  LOG(logDEBUG, *_pLog) << "Reading MOs from " << _orb_file_name << flush;
415  }
416 
417  // the first 12 lines are garbage info
418  for ( i=1; i<13; i++) {
419  getline(_input_file, _line);
420  }
421  // next line has basis set size
422  _input_file >> _basis_size ;
423  LOG( logDEBUG, *_pLog ) << "Basis set size: " << _basis_size << flush;
424 
425 
426  // next line has number of stored MOs
427  _input_file >> _levels ;
428  LOG( logDEBUG, *_pLog ) << "Energy levels: " << _levels << flush;
429 
430  /* next lines contain information about occupation of the MOs
431  * - each line has 3 numbers
432  * - from _occ we can determine the number of electrons/2 */
433  int _n_lines = (( _levels - 1 ) / 3);
434  int _n_rest = _levels - 3*_n_lines;
435  // read in the data
436  int _imo = 0;
437  for ( i=1; i<=_n_lines; i++ ) {
438  for ( int j=0; j<3; j++ ){
439  _input_file >> _occ[ _imo ];
440  if ( _occ[ _imo ] == 2.0 ) {
441  _number_of_electrons++;
442  }
443  _imo++;
444  }
445  }
446  if ( _n_rest != 0 ) {
447  for ( i=0; i<_n_rest; i++ ){
448  _input_file >> _occ[ _imo ];
449  _imo++;
450  }
451  }
452  LOG(logDEBUG,*_pLog) << "Alpha electrons: " << _number_of_electrons << flush ;
453 
454  int _occupied_levels = _number_of_electrons;
455  int _unoccupied_levels = _levels - _occupied_levels;
456  LOG(logDEBUG,*_pLog) << "Occupied levels: " << _occupied_levels << flush;
457  LOG(logDEBUG,*_pLog) << "Unoccupied levels: " << _unoccupied_levels << flush;
458 
459  // reset index and read MO energies the same way
460  _imo = 0;
461  for ( i=1; i<=_n_lines; i++ ) {
462  for ( int j=0; j<3; j++ ){
463  _input_file >> _energies[ _imo ];
464  _imo++;
465  }
466  }
467  if ( _n_rest != 0 ) {
468  for ( i=0; i<_n_rest; i++ ){
469  _input_file >> _energies[ _imo ];
470  _imo++;
471  }
472  }
473 
474 
475  // Now, the same for the coefficients
476  double coef;
477  for ( unsigned int _imo=0; _imo < _levels ; _imo++ ){
478  for ( i=1; i<=_n_lines; i++ ) {
479  for ( int j=0; j<3; j++ ){
480  _input_file >> coef;
481  _coefficients[ _imo ].push_back( coef );
482  }
483  }
484  if ( _n_rest != 0 ) {
485  for ( i=0; i<_n_rest; i++ ){
486  _input_file >> coef;
487  _coefficients[ _imo ].push_back( coef );
488  }
489  }
490  }
491 
492 
493 
494 
495  // copying information to the orbitals object
496  _orbitals->_basis_set_size = _basis_size;
497  _orbitals->_has_basis_set_size = true;
498  _orbitals->_number_of_electrons = _number_of_electrons ;
499  _orbitals->_has_number_of_electrons = true;
500  _orbitals->_has_mo_coefficients = true;
501  _orbitals->_has_mo_energies = true;
502  _orbitals->_has_occupied_levels = true;
503  _orbitals->_has_unoccupied_levels = true;
504  _orbitals->_occupied_levels = _occupied_levels;
505  _orbitals->_unoccupied_levels = _unoccupied_levels;
506 
507  // copying energies to a matrix
508  _orbitals->_mo_energies.resize( _levels );
509  //_level = 1;
510  for(size_t i=0; i < _orbitals->_mo_energies.size(); i++) {
511  _orbitals->_mo_energies[i] = _energies[ i ];
512  }
513 
514 
515  // copying orbitals to the matrix
516  (_orbitals->_mo_coefficients).resize( _levels, _basis_size );
517  for(size_t i = 0; i < _orbitals->_mo_coefficients.size1(); i++) {
518  for(size_t j = 0 ; j < _orbitals->_mo_coefficients.size2(); j++) {
519  _orbitals->_mo_coefficients(i,j) = _coefficients[i][j];
520  //cout << i << " " << j << endl;
521  }
522  }
523 
524 
525 
526 
527  //cout << _mo_energies << endl;
528  //cout << _mo_coefficients << endl;
529 
530  // cleanup
531  _coefficients.clear();
532  _energies.clear();
533  _occ.clear();
534 
535  // when all is done, we can trash the ascii file
536  string file_name = _run_dir + "/system.mos";
537  remove ( file_name.c_str() );
538 
539 
540 
541  LOG(logDEBUG, *_pLog) << "Done reading MOs" << flush;
542 
543  return true;
544 }

546 bool NWChem::CheckLogFile() {
547 
548  // check if the log file exists
549  char ch;
550  ifstream _input_file((_run_dir + "/" + _log_file_name).c_str());
551 
552  if (_input_file.fail()) {
553  LOG(logERROR,*_pLog) << "NWChem LOG is not found" << flush;
554  return false;
555  };
556 
557 
558  /* Checking the log file is a pain in the *** since NWChem throws an error
559  * for our 'iterations 1' runs (even though it outputs the required data
560  * correctly. The only way that works for both scf and noscf runs is to
561  * check for "Total DFT energy" near the end of the log file.
562  */
563 
564  _input_file.seekg(0,ios_base::end); // go to the EOF
565 
566  std::string::size_type total_energy_pos = std::string::npos;
567  std::string::size_type diis_pos = std::string::npos;
568  do {
569  // get the beginning of the line
570  do {
571  _input_file.seekg(-2,ios_base::cur);
572  _input_file.get(ch);
573  //cout << "\nNext Char: " << ch << " TELL G " << (int)_input_file.tellg() << endl;
574  } while ( ch != '\n' || (int)_input_file.tellg() == -1 );
575 
576  string _line;
577  getline(_input_file,_line); // Read the current line
578  // cout << "\nResult: " << _line << '\n'; // Display it
579  total_energy_pos = _line.find("Total DFT energy");
580  diis_pos = _line.find("diis");
581  // whatever is found first, determines the completeness of the file
582  if (total_energy_pos != std::string::npos) {
583  return true;
584  } else if (diis_pos != std::string::npos) {
585  LOG(logERROR,*_pLog) << "NWChem LOG is incomplete" << flush;
586  return false;
587  } else {
588  // go to previous line
589  //_input_file.get(ch);
590  do {
591  _input_file.seekg(-2,ios_base::cur);
592  _input_file.get(ch);
593  //cout << "\nChar: " << ch << endl;
594  } while ( ch != '\n' || (int)_input_file.tellg() == -1);
595  }
596  } while ( total_energy_pos == std::string::npos && diis_pos == std::string::npos );
597 
598 
599  _input_file.close();
600  return true;
601 }
