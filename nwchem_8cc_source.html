66 Index NWChem::WriteBackgroundCharges(ofstream& nw_file) {
67 
68  Index numberofcharges = 0;
69  boost::format fmt("%1$+1.7f %2$+1.7f %3$+1.7f %4$+1.7f");
70  for (const std::unique_ptr<StaticSite>& site : _externalsites) {
71  Eigen::Vector3d pos = site->getPos() * tools::conv::bohr2ang;
72  string sitestring =
73  boost::str(fmt % pos.x() % pos.y() % pos.z() % site->getCharge());
74  if (site->getCharge() != 0.0) {
75  nw_file << sitestring << endl;
76  numberofcharges++;
77  }
78  std::vector<MinimalMMCharge> split_multipoles = SplitMultipoles(*site);
79  for (const auto& mpoles : split_multipoles) {
80  Eigen::Vector3d pos2 = mpoles._pos * tools::conv::bohr2ang;
81  string multipole =
82  boost::str(fmt % pos2.x() % pos2.y() % pos2.z() % mpoles._q);
83  nw_file << multipole << endl;
84  numberofcharges++;
85  }
86  }
87  nw_file << endl;
88  return numberofcharges;
89 }

91 bool NWChem::WriteGuess(const Orbitals& orbitals) {
92  ofstream orb_file;
93  // get name of temporary ascii file and open it
94  std::string filebase = tools::filesystem::GetFileBase(_mo_file_name);
95  std::string orb_file_name_ascii = _run_dir + "/" + filebase + ".mos";
96  orb_file.open(orb_file_name_ascii);
97 
98  // header
99  orb_file << "#generated by VOTCA\nbasisum\ngeomsum\n\nscf\nFri Sep 13 "
100  "00:00:00 2013\nscf\n1\n\n8\nao basis\n1\n"
101  << flush;
102  Index size_of_basis = orbitals.getBasisSetSize();
103  orb_file << size_of_basis << endl;
104  orb_file << size_of_basis << endl;
105  Eigen::MatrixXd MOs = ReorderMOsBack(orbitals);
106  Index level = 1;
107  Index ncolumns = 3;
108  // write occupations as double in three columns
109  // occupied levels
110  Index column = 1;
111  for (Index i = 0; i < orbitals.getNumberOfAlphaElectrons(); i++) {
112  orb_file << FortranFormat(2.0);
113  if (column == ncolumns) {
114  orb_file << endl;
115  column = 0;
116  }
117  column++;
118  }
119  // unoccupied levels
120  for (Index i = orbitals.getNumberOfAlphaElectrons(); i < size_of_basis; i++) {
121  orb_file << FortranFormat(0.0);
122  if (column == ncolumns) {
123  orb_file << endl;
124  column = 0;
125  }
126  column++;
127  }
128  // extra endl
129  if (column != 1) {
130  orb_file << endl;
131  }
132 
133  // write all energies in same format
134  column = 1;
135  for (Index i = 0; i < orbitals.MOs().eigenvalues().size(); ++i) {
136  double energy = orbitals.MOs().eigenvalues()[i];
137  orb_file << FortranFormat(energy);
138  if (column == ncolumns) {
139  orb_file << endl;
140  column = 0;
141  }
142  column++;
143  }
144  if (column != 1) {
145  orb_file << endl;
146  }
147 
148  // write coefficients in same format
149  for (Index i = 0; i < MOs.cols(); ++i) {
150  Eigen::VectorXd mr = MOs.col(i);
151  column = 1;
152  for (Index j = 0; j < mr.size(); ++j) {
153  orb_file << FortranFormat(mr[j]);
154  if (column == ncolumns) {
155  orb_file << endl;
156  column = 0;
157  }
158  column++;
159  }
160  level++;
161  if (column != 1) {
162  orb_file << endl;
163  }
164  }
165  orb_file << " 0.0000 0.0000" << endl;
166  orb_file.close();
167  // now convert this ascii file to binary
168  std::string command = "cd " + _run_dir +
169  "; asc2mov 5000 system.mos system.movecs > convert.log";
170  Index i = std::system(command.c_str());
171  if (i == 0) {
172  XTP_LOG(Log::info, *_pLog)
173  << "Converted MO file from ascii to binary" << flush;
174  } else {
175  XTP_LOG(Log::error, *_pLog)
176  << "Conversion of binary MO file to binary failed. " << flush;
177  return false;
178  }
179  return true;
180 }

186 bool NWChem::WriteInputFile(const Orbitals& orbitals) {
187 
188  std::string temp_suffix = "/id";
189  std::string scratch_dir_backup = _scratch_dir;
190  std::ofstream nw_file;
191  std::ofstream crg_file;
192 
193  std::string nw_file_name_full = _run_dir + "/" + _input_file_name;
194  std::string crg_file_name_full = _run_dir + "/background.crg";
195 
196  nw_file.open(nw_file_name_full);
197  // header
198  nw_file << "geometry noautoz noautosym" << endl;
199 
200  const QMMolecule& qmatoms = orbitals.QMAtoms();
201 
202  for (const QMAtom& atom : qmatoms) {
203  Eigen::Vector3d pos = atom.getPos() * tools::conv::bohr2ang;
204  nw_file << setw(3) << atom.getElement() << setw(12)
205  << setiosflags(ios::fixed) << setprecision(5) << pos.x() << setw(12)
206  << setiosflags(ios::fixed) << setprecision(5) << pos.y() << setw(12)
207  << setiosflags(ios::fixed) << setprecision(5) << pos.z() << endl;
208  }
209  nw_file << "end\n";
210  if (_write_charges) {
211  // part for the MM charge coordinates
212  crg_file.open(crg_file_name_full);
213  Index numberofcharges = WriteBackgroundCharges(crg_file);
214  crg_file << endl;
215  crg_file.close();
216  nw_file << endl;
217  nw_file << "set bq:max_nbq " << numberofcharges << endl;
218  nw_file << "bq background" << endl;
219  nw_file << "load background.crg format 1 2 3 4" << endl;
220  nw_file << "end\n" << endl;
221  }
222 
223  if (_write_basis_set) {
224  WriteBasisset(nw_file, qmatoms);
225  }
226 
227  if (_write_pseudopotentials) {
228  WriteECP(nw_file, qmatoms);
229  }
230 
231  // write charge of the molecule
232  nw_file << "\ncharge " << _charge << "\n";
233 
234  // writing scratch_dir info
235  if (_scratch_dir != "") {
236  std::string _temp("scratch_dir " + _scratch_dir + temp_suffix + "\n");
237  nw_file << _temp;
238  }
239  if (_charge != 0) {
240  std::string dft = "dft";
241  if (_options.find(dft) != std::string::npos) {
242  Index dftpos = Index(_options.find(dft));
243  dftpos += Index(dft.size());
244  std::string openshell =
245  "\nodft\n" + (boost::format("mult %1%\n") % _spin).str();
246  _options.insert(dftpos, openshell, 0, openshell.size());
247  } else {
248  throw runtime_error("NWCHEM: dft input data missing");
249  }
250  }
251 
252  nw_file << _options << "\n";
253  if (_write_guess) {
254  bool worked = WriteGuess(orbitals);
255  if (!worked) {
256  return false;
257  }
258  }
259 
260  nw_file << endl;
261  nw_file.close();
262  // and now generate a shell script to run both jobs, if neccessary
263  XTP_LOG(Log::info, *_pLog)
264  << "Setting the scratch dir to " << _scratch_dir + temp_suffix << flush;
265 
266  _scratch_dir = scratch_dir_backup + temp_suffix;
267 
268  WriteShellScript();
269  _scratch_dir = scratch_dir_backup;
270 
271  return true;
272 }

296 bool NWChem::Run() {
297 
298  XTP_LOG(Log::error, *_pLog) << "Running NWChem job" << flush;
299 
300  if (std::system(nullptr)) {
301 
302  // NWChem overrides input information, if *.db and *.movecs files are
303  // present better trash the old version
304  std::string file_name = _run_dir + "/system.db";
305  remove(file_name.c_str());
306  file_name = _run_dir + "/" + _log_file_name;
307  remove(file_name.c_str());
308 
309  std::string command = "cd " + _run_dir + "; sh " + _shell_file_name;
310 
311  Index check = std::system(command.c_str());
312  if (check == -1) {
313  XTP_LOG(Log::error, *_pLog)
314  << _input_file_name << " failed to start" << flush;
315  return false;
316  }
317 
318  if (CheckLogFile()) {
319  XTP_LOG(Log::error, *_pLog) << "Finished NWChem job" << flush;
320  /* maybe we DO need to convert from fortran binary to ASCII first to avoid
321  compiler-dependent issues */
322  std::string command2;
323  ascii_mo_file_name =
324  tools::filesystem::GetFileBase(_mo_file_name) + ".mos";
325  command2 = "cd " + _run_dir + "; mov2asc 10000 " + _mo_file_name + " " +
326  ascii_mo_file_name + "> convert.log";
327  Index i = std::system(command2.c_str());
328  if (i == 0) {
329  XTP_LOG(Log::info, *_pLog)
330  << "Converted MO file from binary to ascii" << flush;
331  } else {
332  XTP_LOG(Log::error, *_pLog)
333  << "Conversion of binary MO file to ascii failed. " << flush;
334  return false;
335  }
336 
337  return true;
338  } else {
339  XTP_LOG(Log::error, *_pLog) << "NWChem job failed" << flush;
340  return false;
341  }
342  } else {
343  XTP_LOG(Log::error, *_pLog)
344  << _input_file_name << " failed to start" << flush;
345  return false;
346  }
347 
348  return true;

395 bool NWChem::ParseMOsFile(Orbitals& orbitals) {
396  std::map<Index, std::vector<double> > coefficients;
397  std::map<Index, double> energies;
398  std::map<Index, double> occupancy;
399 
400  std::string line;
401  Index levels = 0;
402  // Index _level;
403  Index basis_size = 0;
404  Index number_of_electrons = 0;
405 
406  // opening the ascii MO file
407  std::string orb_file_name_full = _run_dir + "/" + ascii_mo_file_name;
408  std::ifstream input_file(orb_file_name_full);
409 
410  if (input_file.fail()) {
411  XTP_LOG(Log::error, *_pLog)
412  << "File " << orb_file_name_full
413  << " with molecular orbitals is not found " << flush;
414  return false;
415  } else {
416  XTP_LOG(Log::error, *_pLog)
417  << "Reading MOs from " << orb_file_name_full << flush;
418  }
419 
420  // the first 12 lines are garbage info
421  for (Index i = 0; i < 12; i++) {
422  tools::getline(input_file, line);
423  }
424  // next line has basis set size
425  input_file >> basis_size;
426  XTP_LOG(Log::error, *_pLog) << "Basis set size: " << basis_size << flush;
427 
428  // next line has number of stored MOs
429  input_file >> levels;
430  XTP_LOG(Log::info, *_pLog) << "Energy levels: " << levels << flush;
431 
432  /* next lines contain information about occupation of the MOs
433  * - each line has 3 numbers
434  * - from _occ we can determine the number of electrons/2 */
435  Index n_lines = ((levels - 1) / 3);
436  Index n_rest = levels - 3 * n_lines;
437  // read in the data
438  Index imo = 0;
439  for (Index i = 0; i < n_lines; i++) {
440  for (Index j = 0; j < 3; j++) {
441  input_file >> occupancy[imo];
442  if (occupancy[imo] == 2.0) {
443  number_of_electrons++;
444  }
445  imo++;
446  }
447  }
448  if (n_rest != 0) {
449  for (Index i = 0; i < n_rest; i++) {
450  input_file >> occupancy[imo];
451  imo++;
452  }
453  }
454 
455  XTP_LOG(Log::error, *_pLog)
456  << "Alpha electrons: " << number_of_electrons << flush;
457 
458  Index occupied_levels = number_of_electrons;
459  Index unoccupied_levels = levels - occupied_levels;
460  XTP_LOG(Log::info, *_pLog) << "Occupied levels: " << occupied_levels << flush;
461  XTP_LOG(Log::info, *_pLog)
462  << "Unoccupied levels: " << unoccupied_levels << flush;
463 
464  // reset index and read MO energies the same way
465  imo = 0;
466  for (Index i = 0; i < n_lines; i++) {
467  for (Index j = 0; j < 3; j++) {
468  input_file >> energies[imo];
469  imo++;
470  }
471  }
472  if (n_rest != 0) {
473  for (Index i = 0; i < n_rest; i++) {
474  input_file >> energies[imo];
475  imo++;
476  }
477  }
478 
479  // Now, the same for the coefficients
480  double coef;
481  for (Index imo2 = 0; imo2 < levels; imo2++) {
482  for (Index i = 0; i < n_lines; i++) {
483  for (Index j = 0; j < 3; j++) {
484  input_file >> coef;
485  coefficients[imo2].push_back(coef);
486  }
487  }
488  if (n_rest != 0) {
489  for (Index i = 0; i < n_rest; i++) {
490  input_file >> coef;
491  coefficients[imo2].push_back(coef);
492  }
493  }
494  }
495 
496  // copying information to the orbitals object
497  orbitals.setBasisSetSize(basis_size);
498  orbitals.setNumberOfAlphaElectrons(number_of_electrons);
499  orbitals.setNumberOfOccupiedLevels(occupied_levels);
500  // copying energies to a matrix
501  orbitals.MOs().eigenvalues().resize(levels);
502  //_level = 1;
503  for (Index i = 0; i < levels; i++) {
504  orbitals.MOs().eigenvalues()[i] = energies[i];
505  }
506 
507  // copying orbitals to the matrix
508  orbitals.MOs().eigenvectors().resize(basis_size, levels);
509  for (Index i = 0; i < levels; i++) {
510  for (Index j = 0; j < basis_size; j++) {
511  orbitals.MOs().eigenvectors()(j, i) = coefficients[i][j];
512  }
513  }
514 
515  ReorderOutput(orbitals);
516  XTP_LOG(Log::error, *_pLog) << "Done reading MOs" << flush;
517 
518  return true;
519 }

521 StaticSegment NWChem::GetCharges() const {
522  StaticSegment result("charges", 0);
523  if (!CheckLogFile()) {
524  throw std::runtime_error("logfile not correctly formatted");
525  }
526  std::string line;
527  ifstream input_file((_run_dir + "/" + _log_file_name));
528  bool has_charges = false;
529  while (input_file) {
530  tools::getline(input_file, line);
531  boost::trim(line);
532 
533  std::string::size_type charge_pos = line.find("ESP");
534  if (charge_pos != std::string::npos) {
535  XTP_LOG(Log::error, *_pLog) << "Getting charges" << flush;
536  has_charges = true;
537  // two empty lines
538  tools::getline(input_file, line);
539  tools::getline(input_file, line);
540 
541  // now starts the data in format
542  // _id type x y z q
543 
544  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
545  Index nfields = Index(row.size());
546  while (nfields == 6) {
547  Index atom_id = boost::lexical_cast<Index>(row.at(0)) - 1;
548  std::string atom_type = row.at(1);
549  double x = boost::lexical_cast<double>(row.at(2));
550  double y = boost::lexical_cast<double>(row.at(3));
551  double z = boost::lexical_cast<double>(row.at(4));
552  Eigen::Vector3d pos = {x, y, z};
553  pos *= tools::conv::ang2bohr;
554  double atom_charge = boost::lexical_cast<double>(row.at(5));
555  row = GetLineAndSplit(input_file, "\t ");
556  nfields = row.size();
557 
558  StaticSite temp = StaticSite(atom_id, atom_type, pos);
559  temp.setCharge(atom_charge);
560  result.push_back(temp);
561  }
562  }
563  }
564  if (!has_charges) {
565  throw std::runtime_error("Could not find charges in logfile");
566  }
567  return result;
568 }

570 Eigen::Matrix3d NWChem::GetPolarizability() const {
571  if (!CheckLogFile()) {
572  throw std::runtime_error("logfile not correctly formatted");
573  }
574  std::string line;
575  ifstream input_file((_run_dir + "/" + _log_file_name));
576  bool has_pol = false;
577 
578  Eigen::Matrix3d pol = Eigen::Matrix3d::Zero();
579  while (input_file) {
580  tools::getline(input_file, line);
581  boost::trim(line);
582 
583  std::string::size_type pol_pos =
584  line.find("DFT Linear Response polarizability / au");
585  if (pol_pos != std::string::npos) {
586  XTP_LOG(Log::error, *_pLog) << "Getting polarizability" << flush;
587  tools::getline(input_file, line);
588  tools::Tokenizer tok(line, " ");
589  std::vector<std::string> line_split = tok.ToVector();
590  double frequency = std::stod(line_split[2]);
591  if (std::abs(frequency) > 1e-9) {
592  XTP_LOG(Log::error, *_pLog)
593  << "Warning: Polarizability was calculated for frequency "
594  << frequency << " normally f=0 for static polarizability" << flush;
595  }
596  tools::getline(input_file, line);
597  tools::getline(input_file, line);
598  tools::getline(input_file, line);
599  for (Index i = 0; i < 3; i++) {
600  tools::getline(input_file, line);
601  tools::Tokenizer tok2(line, " ");
602  std::vector<std::string> values = tok2.ToVector();
603  if (values.size() != 4) {
604  throw std::runtime_error("Polarisation line " + line +
605  " cannot be parsed");
606  }
607  Eigen::Vector3d row;
608  row << std::stod(values[1]), std::stod(values[2]), std::stod(values[3]);
609  pol.row(i) = row;
610  }
611 
612  has_pol = true;
613  }
614  }
615  if (!has_pol) {
616  throw std::runtime_error("Could not find polarisation in logfile");
617  }
618  return pol;
619 }

621 bool NWChem::CheckLogFile() const {
622 
623  // check if the log file exists
624 
625  ifstream input_file((_run_dir + "/" + _log_file_name));
626 
627  if (input_file.fail()) {
628  XTP_LOG(Log::error, *_pLog) << "NWChem LOG is not found" << flush;
629  return false;
630  };
631 
632  if (input_file.peek() == std::ifstream::traits_type::eof()) {
633  XTP_LOG(Log::error, *_pLog)
634  << "NWChem run failed. Check OpenMPI version!" << flush;
635  return false;
636  };
637 
638  /* Checking the log file is a pain in the *** since NWChem throws an error
639  * for our 'iterations 1' runs (even though it outputs the required data
640  * correctly. The only way that works for both scf and noscf runs is to
641  * check for "Total DFT energy" near the end of the log file.
642  */
643 
644  input_file.seekg(0, ios_base::end); // go to the EOF
645  char ch;
646  std::string::size_type total_energy_pos = std::string::npos;

687 bool NWChem::ParseLogFile(Orbitals& orbitals) {
688 
689  std::string line;
690  std::vector<std::string> results;
691 
692  XTP_LOG(Log::error, *_pLog) << "Parsing " << _log_file_name << flush;
693  std::string log_file_name_full = _run_dir + "/" + _log_file_name;
694  // check if LOG file is complete
695  if (!CheckLogFile()) {
696  return false;
697  }
698 
699  // save qmpackage name
700  orbitals.setQMpackage(getPackageName());
701  orbitals.setDFTbasisName(_basisset_name);
702  if (_write_pseudopotentials) {
703  orbitals.setECPName(_ecp_name);
704  }
705 
706  // Start parsing the file line by line
707  ifstream input_file(log_file_name_full);
708  while (input_file) {
709  tools::getline(input_file, line);
710  boost::trim(line);
711  /*
712  * basis set size (is only required for overlap matrix reading, rest is
713  * in orbitals file and could be skipped
714  */
715  std::string::size_type basis_pos = line.find("number of functions");
716  if (basis_pos != std::string::npos) {
717  tools::Tokenizer tok(line, ":");
718  results = tok.ToVector();
719  std::string bf = results.back();
720  boost::trim(bf);
721  Index basis_set_size = boost::lexical_cast<Index>(bf);
722  orbitals.setBasisSetSize(basis_set_size);
723  XTP_LOG(Log::info, *_pLog)
724  << "Basis functions: " << basis_set_size << flush;
725  }
726 
727  std::string::size_type energy_pos = line.find("Total DFT energy");
728  if (energy_pos != std::string::npos) {
729 
730  tools::Tokenizer tok(line, "=");
731  results = tok.ToVector();
732  std::string energy = results.back();
733  boost::trim(energy);
734  orbitals.setQMEnergy(boost::lexical_cast<double>(energy));
735  XTP_LOG(Log::error, *_pLog) << (boost::format("QM energy[Hrt]: %4.6f ") %
736  orbitals.getDFTTotalEnergy())
737  .str()
738  << flush;
739  }
740 
741  std::string::size_type coordinates_pos = line.find("Output coordinates");
742  if (coordinates_pos != std::string::npos) {
743  XTP_LOG(Log::error, *_pLog) << "Getting the coordinates" << flush;
744  //_has_coordinates = true;
745  bool has_QMAtoms = orbitals.hasQMAtoms();
746  // three garbage lines
747  tools::getline(input_file, line);
748  tools::getline(input_file, line);
749  tools::getline(input_file, line);
750  // now starts the data in format
751  // _id type Qnuc x y z
752  std::vector<std::string> row = GetLineAndSplit(input_file, "\t ");
753  Index nfields = Index(row.size());
754 
755  while (nfields == 6) {
756  Index atom_id = boost::lexical_cast<Index>(row.at(0)) - 1;
757  std::string atom_type = row.at(1);
758  double x = boost::lexical_cast<double>(row.at(3));
759  double y = boost::lexical_cast<double>(row.at(4));
760  double z = boost::lexical_cast<double>(row.at(5));
761  Eigen::Vector3d pos(x, y, z);
762  pos *= tools::conv::ang2bohr;
763  if (has_QMAtoms == false) {
764  orbitals.QMAtoms().push_back(QMAtom(atom_id, atom_type, pos));
765  } else {
766  QMAtom& pAtom = orbitals.QMAtoms().at(atom_id);
767  pAtom.setPos(pos);
768  }
769  atom_id++;
770  row = GetLineAndSplit(input_file, "\t ");
771  nfields = Index(row.size());
772  }
773  }
774 
775  // Check for ScaHFX = factor of HF exchange included in functional
776  std::string::size_type HFX_pos = line.find("Hartree-Fock (Exact) Exchange");
777  if (HFX_pos != std::string::npos) {
778 
779  tools::Tokenizer tok(line, "\t ");
780  results = tok.ToVector();
781  double ScaHFX = boost::lexical_cast<double>(results.back());
782  orbitals.setScaHFX(ScaHFX);
783  XTP_LOG(Log::error, *_pLog)
784  << "DFT with " << ScaHFX << " of HF exchange!" << flush;
785  }
786 
787  } // end of reading the file line-by-line
788 
789  XTP_LOG(Log::error, *_pLog) << "Done parsing" << flush;
790  return true;
791 }

793 void NWChem::WriteBasisset(ofstream& nw_file, const QMMolecule& qmatoms) {
794 
795  std::vector<std::string> UniqueElements = qmatoms.FindUniqueElements();
796  BasisSet bs;
797  bs.Load(_basisset_name);
798  XTP_LOG(Log::error, *_pLog) << "Loaded Basis Set " << _basisset_name << flush;
799  nw_file << "basis spherical" << endl;
800  for (const std::string& element_name : UniqueElements) {
801  const Element& element = bs.getElement(element_name);
802  for (const Shell& shell : element) {
803  // nwchem can only use S,P,SP,D,F,G shells so we split them up if not SP
804  if (!shell.isCombined()) {
805  // shell type, number primitives, scale factor
806  nw_file << element_name << " "
807  << boost::algorithm::to_lower_copy(shell.getType()) << endl;
808  for (const GaussianPrimitive& gaussian : shell) {
809  for (Index icontr = 0; icontr < Index(gaussian.Contractions().size());
810  icontr++) {
811  if (gaussian.Contractions()[icontr] != 0.0) {
812  nw_file << FortranFormat(gaussian.decay()) << " "
813  << FortranFormat(gaussian.Contractions()[icontr]) << endl;
814  }
815  }
816  }
817  } else {
818  for (const char& subtype : shell.getType()) {
819  nw_file << element_name << " "
820  << boost::algorithm::to_lower_copy(std::string(1, subtype))
821  << endl;
822 
823  for (const GaussianPrimitive& gaussian : shell) {
824  nw_file << FortranFormat(gaussian.decay()) << " "
825  << FortranFormat(gaussian.Contractions()[FindLmax(
826  std::string(1, subtype))])
827  << endl;
828  }
829  }
830  }
831  }
832  }
833  nw_file << "end\n";
834  nw_file << endl;
835 
836  return;
