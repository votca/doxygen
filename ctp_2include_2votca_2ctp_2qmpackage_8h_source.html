109 inline bool QMPackage::WriteInputFilePBC( QMPair* pair, Orbitals* orbitals) {
110 
111  //std::cout << "IDFT writes input with PBC" << std::endl;
112 
113  Segment* seg1 = pair->Seg1();
114  Segment* seg2 = pair->Seg2();
115  Segment* ghost = NULL;
116 
117  Topology* _top = seg1->getTopology();
118 
119  vec r1 = seg1->getPos();
120  vec r2 = seg2->getPos();
121 
122  vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
123 
124  // Check whether pair formed across periodic boundary
125  if ( abs(r2 - r1 - _R) > 1e-8 ) {
126  ghost = new Segment(seg2);
127  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
128  vector<Atom*>::iterator ait;
129  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
130  (*ait)->setQMPos((*ait)->getQMPos()+r1-r2+_R);
131  }
132  }
133 
134  vector< Segment* > segments;
135  segments.push_back(seg1);
136 
137  if ( ghost ) {
138  segments.push_back(ghost);
139  } else {
140  segments.push_back(seg2);
141  }
142 
143  WriteInputFile( segments, orbitals);
144 
145  delete ghost;
146  return true;
147 }
