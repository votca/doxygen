130 inline bool QMPackage::WriteInputFilePBC( ctp::QMPair* pair, Orbitals* orbitals) {
131 
132  //std::cout << "IDFT writes input with PBC" << std::endl;
133 
134  ctp::Segment* seg1 = pair->Seg1();
135  ctp::Segment* seg2 = pair->Seg2();
136  ctp::Segment* ghost = NULL;
137 
138  ctp::Topology* _top = seg1->getTopology();
139 
140  ctp::vec r1 = seg1->getPos();
141  ctp::vec r2 = seg2->getPos();
142 
143  ctp::vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
144 
145  // Check whether pair formed across periodic boundary
146  if ( abs(r2 - r1 - _R) > 1e-8 ) {
147  ghost = new ctp::Segment(seg2);
148  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
149  std::vector<ctp::Atom*>::iterator ait;
150  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
151  (*ait)->setQMPos((*ait)->getQMPos()+r1-r2+_R);
152  }
153  }
154 
155  std::vector< ctp::Segment* > segments;
156  segments.push_back(seg1);
157 
158  if ( ghost ) {
159  segments.push_back(ghost);
160  } else {
161  segments.push_back(seg2);
162  }
163 
164  WriteInputFile( segments, orbitals);
165 
166  delete ghost;
167  return true;
168 }
169 
170 }}
