128 inline bool QMPackage::WriteInputFilePBC( ctp::QMPair* pair, Orbitals* orbitals) {
129 
130  //std::cout << "IDFT writes input with PBC" << std::endl;
131 
132  ctp::Segment* seg1 = pair->Seg1();
133  ctp::Segment* seg2 = pair->Seg2();
134  ctp::Segment* ghost = NULL;
135 
136  ctp::Topology* _top = seg1->getTopology();
137 
138  ctp::vec r1 = seg1->getPos();
139  ctp::vec r2 = seg2->getPos();
140 
141  ctp::vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
142 
143  // Check whether pair formed across periodic boundary
144  if ( abs(r2 - r1 - _R) > 1e-8 ) {
145  ghost = new ctp::Segment(seg2);
146  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
147  std::vector<ctp::Atom*>::iterator ait;
148  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
149  (*ait)->setQMPos((*ait)->getQMPos()+r1-r2+_R);
150  }
151  }
152 
153  std::vector< ctp::Segment* > segments;
154  segments.push_back(seg1);
155 
156  if ( ghost ) {
157  segments.push_back(ghost);
158  } else {
159  segments.push_back(seg2);
160  }
161 
162  WriteInputFile( segments, orbitals);
163 
164  delete ghost;
165  return true;
166 }
