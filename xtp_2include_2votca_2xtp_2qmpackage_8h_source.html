129 inline bool XQMPackage::WriteInputFilePBC( CTP::QMPair* pair, Orbitals* orbitals) {
130 
131  //std::cout << "IDFT writes input with PBC" << std::endl;
132 
133  CTP::Segment* seg1 = pair->Seg1();
134  CTP::Segment* seg2 = pair->Seg2();
135  CTP::Segment* ghost = NULL;
136 
137  CTP::Topology* _top = seg1->getTopology();
138 
139  CTP::vec r1 = seg1->getPos();
140  CTP::vec r2 = seg2->getPos();
141 
142  CTP::vec _R = _top->PbShortestConnect(r1, r2); // => _R points from 1 to 2
143 
144  // Check whether pair formed across periodic boundary
145  if ( abs(r2 - r1 - _R) > 1e-8 ) {
146  ghost = new CTP::Segment(seg2);
147  //ghost->TranslateBy(r1 - r2 + _R); // DO NOT USE THIS METHOD !
148  std::vector<CTP::Atom*>::iterator ait;
149  for (ait = ghost->Atoms().begin(); ait != ghost->Atoms().end(); ++ait) {
150  (*ait)->setQMPos((*ait)->getQMPos()+r1-r2+_R);
151  }
152  }
153 
154  std::vector< CTP::Segment* > segments;
155  segments.push_back(seg1);
156 
157  if ( ghost ) {
158  segments.push_back(ghost);
159  } else {
160  segments.push_back(seg2);
161  }
162 
163  WriteInputFile( segments, orbitals);
164 
165  delete ghost;
166  return true;
167 }
